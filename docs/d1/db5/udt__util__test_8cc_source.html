26 // Test handler used to collect the column family id and user keys contained
27 // in a WriteBatch for test verification. And verifies the value part stays
28 // the same if it's available.
29 class KeyCollector : public WriteBatch::Handler {
30 public:
31 explicit KeyCollector() = default;
32
33 ~KeyCollector() override = default;
34
35 Status PutCF(uint32_t cf, const Slice& key, const Slice& value) override {
36 if (value.compare(kValuePlaceHolder) != 0) {
37 return Status::InvalidArgument();
38 }
39 return AddKey(cf, key);
40 }
41
42 Status TimedPutCF(uint32_t cf, const Slice& key, const Slice& value,
43 uint64_t write_unix_time) override {
44 if (value.compare(kValuePlaceHolder) != 0) {
45 return Status::InvalidArgument();
46 }
47 if (write_unix_time != kWriteUnixTime) {
48 return Status::InvalidArgument();
49 }
50 return AddKey(cf, key);
51 }
52
53 Status PutEntityCF(uint32_t cf, const Slice& key,
54 const Slice& entity) override {
55 Slice entity_copy = entity;
56 WideColumns columns;
57 Status s = WideColumnSerialization::Deserialize(entity_copy, columns);
58 if (!s.ok()) {
59 return s;
60 }
61 if (columns.size() != 1) {
62 return Status::InvalidArgument();
63 }
64 if (columns[0].value().compare(kValuePlaceHolder) != 0) {
65 return Status::InvalidArgument();
66 }
67 return AddKey(cf, key);
68 }
69
70 Status DeleteCF(uint32_t cf, const Slice& key) override {
71 return AddKey(cf, key);
72 }
73
74 Status SingleDeleteCF(uint32_t cf, const Slice& key) override {
75 return AddKey(cf, key);
76 }
77
78 Status DeleteRangeCF(uint32_t cf, const Slice& begin_key,
79 const Slice& end_key) override {
80 Status status = AddKey(cf, begin_key);
81 if (!status.ok()) {
82 return status;
83 }
84 return AddKey(cf, end_key);
85 }
86
87 Status MergeCF(uint32_t cf, const Slice& key, const Slice& value) override {
88 if (value.compare(kValuePlaceHolder) != 0) {
89 return Status::InvalidArgument();
90 }
91 return AddKey(cf, key);
92 }
93
94 Status PutBlobIndexCF(uint32_t cf, const Slice& key,
95 const Slice& value) override {
96 if (value.compare(kValuePlaceHolder) != 0) {
97 return Status::InvalidArgument();
98 }
99 return AddKey(cf, key);
100 }
101
102 Status MarkBeginPrepare(bool) override { return Status::OK(); }
103
104 Status MarkEndPrepare(const Slice&) override { return Status::OK(); }
105
106 Status MarkRollback(const Slice&) override { return Status::OK(); }
107
108 Status MarkCommit(const Slice&) override { return Status::OK(); }
109
110 Status MarkCommitWithTimestamp(const Slice&, const Slice&) override {
111 return Status::OK();
112 }
113
114 Status MarkNoop(bool) override { return Status::OK(); }
115
116 const std::vector<std::pair<uint32_t, const Slice>>& GetKeys() const {
117 return keys_;
118 }
119
120 private:
121 Status AddKey(uint32_t cf, const Slice& key) {
122 keys_.emplace_back(cf, key);
123 return Status::OK();
124 }
125 std::vector<std::pair<uint32_t, const Slice>> keys_;
126 };

128 void CreateKey(std::string* key_buf, size_t ts_sz) {
129 if (ts_sz > 0) {
130 AppendKeyWithMinTimestamp(key_buf, kTestKeyWithoutTs, ts_sz);
131 } else {
132 key_buf->assign(kTestKeyWithoutTs);
133 }
134 }
135
136 void CreateWriteBatch(const UnorderedMap<uint32_t, size_t>& ts_sz_for_batch,
137 WriteBatch* batch) {
138 for (const auto& [cf_id, ts_sz] : ts_sz_for_batch) {
139 std::string key;
140 CreateKey(&key, ts_sz);
141 ASSERT_OK(WriteBatchInternal::Put(batch, cf_id, key, kValuePlaceHolder));
142 ASSERT_OK(WriteBatchInternal::Delete(batch, cf_id, key));
143 ASSERT_OK(WriteBatchInternal::SingleDelete(batch, cf_id, key));
144 ASSERT_OK(WriteBatchInternal::DeleteRange(batch, cf_id, key, key));
145 ASSERT_OK(
146 WriteBatchInternal::Merge(batch, cf_id, key, kValuePlaceHolder));
147 ASSERT_OK(WriteBatchInternal::PutBlobIndex(batch, cf_id, key,
148 kValuePlaceHolder));
149 ASSERT_OK(WriteBatchInternal::TimedPut(
150 batch, cf_id, key, kValuePlaceHolder, kWriteUnixTime));
151 WideColumns columns{{kDefaultWideColumnName, kValuePlaceHolder}};
152 ASSERT_OK(WriteBatchInternal::PutEntity(batch, cf_id, key, columns));
153 }
154 }
155
156 void CheckSequenceEqual(const WriteBatch& orig_batch,
157 const WriteBatch& new_batch) {
158 ASSERT_EQ(WriteBatchInternal::Sequence(&orig_batch),
159 WriteBatchInternal::Sequence(&new_batch));
160 }
161 void CheckCountEqual(const WriteBatch& orig_batch,
162 const WriteBatch& new_batch) {
163 ASSERT_EQ(WriteBatchInternal::Count(&orig_batch),
164 WriteBatchInternal::Count(&new_batch));
165 }
166
167 void VerifyKeys(
168 const std::vector<std::pair<uint32_t, const Slice>>& keys_with_ts,
169 const std::vector<std::pair<uint32_t, const Slice>>& keys_without_ts,
170 size_t ts_sz, std::optional<uint32_t> dropped_cf) {
171 ASSERT_EQ(keys_with_ts.size(), keys_without_ts.size());
172 const std::string kTsMin(ts_sz, static_cast<unsigned char>(0));
173 for (size_t i = 0; i < keys_with_ts.size(); i++) {
174 // TimestampRecoveryHandler ignores dropped column family and copy it over
175 // as is. Check the keys stay the same.
176 if (dropped_cf.has_value() &&
177 keys_with_ts[i].first == dropped_cf.value()) {
178 ASSERT_EQ(keys_with_ts[i].first, keys_without_ts[i].first);
179 ASSERT_EQ(keys_with_ts[i].second, keys_without_ts[i].second);
180 continue;
181 }
182 const Slice& key_with_ts = keys_with_ts[i].second;
183 const Slice& key_without_ts = keys_without_ts[i].second;
184 ASSERT_TRUE(key_with_ts.starts_with(key_without_ts));
185 ASSERT_EQ(key_with_ts.size() - key_without_ts.size(), ts_sz);
186 ASSERT_TRUE(key_with_ts.ends_with(kTsMin));
187 }
188 }
189
190 void CheckContentsWithTimestampStripping(const WriteBatch& orig_batch,
191 const WriteBatch& new_batch,
192 size_t ts_sz,
193 std::optional<uint32_t> dropped_cf) {
194 CheckSequenceEqual(orig_batch, new_batch);
195 CheckCountEqual(orig_batch, new_batch);
196 KeyCollector collector_for_orig_batch;
197 ASSERT_OK(orig_batch.Iterate(&collector_for_orig_batch));
198 KeyCollector collector_for_new_batch;
199 ASSERT_OK(new_batch.Iterate(&collector_for_new_batch));
200 VerifyKeys(collector_for_orig_batch.GetKeys(),
201 collector_for_new_batch.GetKeys(), ts_sz, dropped_cf);
202 }
203
204 void CheckContentsWithTimestampPadding(const WriteBatch& orig_batch,
205 const WriteBatch& new_batch,
206 size_t ts_sz) {
207 CheckSequenceEqual(orig_batch, new_batch);
208 CheckCountEqual(orig_batch, new_batch);
209 KeyCollector collector_for_orig_batch;
210 ASSERT_OK(orig_batch.Iterate(&collector_for_orig_batch));
211 KeyCollector collector_for_new_batch;
212 ASSERT_OK(new_batch.Iterate(&collector_for_new_batch));
213 VerifyKeys(collector_for_new_batch.GetKeys(),
214 collector_for_orig_batch.GetKeys(), ts_sz,
215 std::nullopt /* dropped_cf */);
216 }
217};
