37struct ReadaheadParams {
38 ReadaheadParams() {}

40 // The initial readahead size.
41 size_t initial_readahead_size = 0;
42
43 // The maximum readahead size.
44 // If max_readahead_size > readahead_size, then readahead size will be doubled
45 // on every IO until max_readahead_size is hit. Typically this is set as a
46 // multiple of initial_readahead_size. initial_readahead_size should be
47 // greater than equal to initial_readahead_size.
48 size_t max_readahead_size = 0;
49
50 // If true, Readahead is enabled implicitly by rocksdb
51 // after doing sequential scans for num_file_reads_for_auto_readahead.
52 bool implicit_auto_readahead = false;

55 uint64_t num_file_reads = 0;
56 uint64_t num_file_reads_for_auto_readahead = 0;

59 // > 1 then buffers will be filled asynchronously whenever they get emptied.

63struct BufferInfo {
64 void ClearBuffer() {
65 buffer_.Clear();
66 initial_end_offset_ = 0;
67 async_req_len_ = 0;
68 }
69
70 AlignedBuffer buffer_;
71
72 uint64_t offset_ = 0;
73
74 // Below parameters are used in case of async read flow.
75 // Length requested for in ReadAsync.
76 size_t async_req_len_ = 0;
77
78 // async_read_in_progress can be used as mutex. Callback can update the buffer
79 // and its size but async_read_in_progress is only set by main thread.
80 bool async_read_in_progress_ = false;
81
82 // io_handle is allocated and used by underlying file system in case of
83 // asynchronous reads.
84 void* io_handle_ = nullptr;
85
86 IOHandleDeleter del_fn_ = nullptr;
87
88 // initial_end_offset is used to keep track of the end offset of the buffer
89 // that was originally called. It's helpful in case of autotuning of readahead
90 // size when callback is made to BlockBasedTableIterator.
91 // initial end offset of this buffer which will be the starting
92 // offset of next prefetch.
93 //
94 // For example - if end offset of previous buffer was 100 and because of
95 // readahead_size optimization, end_offset was trimmed to 60. Then for next
96 // prefetch call, start_offset should be intialized to 100 i.e start_offset =
97 // buf->initial_end_offset_.
98 uint64_t initial_end_offset_ = 0;
99
100 bool IsDataBlockInBuffer(uint64_t offset, size_t length) {
101 assert(async_read_in_progress_ == false);
102 return (offset >= offset_ &&
103 offset + length <= offset_ + buffer_.CurrentSize());
104 }
105
106 bool IsOffsetInBuffer(uint64_t offset) {
107 assert(async_read_in_progress_ == false);
108 return (offset >= offset_ && offset < offset_ + buffer_.CurrentSize());
109 }
110
111 bool DoesBufferContainData() {
112 assert(async_read_in_progress_ == false);
113 return buffer_.CurrentSize() > 0;
114 }
115
116 bool IsBufferOutdated(uint64_t offset) {
117 return (!async_read_in_progress_ && DoesBufferContainData() &&
118 offset >= offset_ + buffer_.CurrentSize());
119 }
120
121 bool IsBufferOutdatedWithAsyncProgress(uint64_t offset) {
122 return (async_read_in_progress_ && io_handle_ != nullptr &&
123 offset >= offset_ + async_req_len_);
124 }
125
126 bool IsOffsetInBufferWithAsyncProgress(uint64_t offset) {
127 return (async_read_in_progress_ && offset >= offset_ &&
128 offset < offset_ + async_req_len_);
129 }
130
131 size_t CurrentSize() { return buffer_.CurrentSize(); }
132};

140// Implementation:
141// FilePrefetchBuffer maintains a dequeu of free buffers (free_bufs_) with no
142// data and bufs_ which contains the prefetched data. Whenever a buffer is
143// consumed or is outdated (w.r.t. to requested offset), that buffer is cleared
144// and returned to free_bufs_.
145//
146// If a buffer is available in free_bufs_, it's moved to bufs_ and is sent for
147// prefetching.
148// num_buffers_ defines how many buffers FilePrefetchBuffer can maintain at a
149// time that contains prefetched data with num_buffers_ == bufs_.size() +
150// free_bufs_.size().
151//
152// If num_buffers_ == 1, it's a sequential read flow. Read API will be called on
153// that one buffer whenever the data is requested and is not in the buffer.
154// When reusing the file system allocated buffer, overlap_buf_ is used if the
155// main buffer only contains part of the requested data. It is returned to
156// the caller after the remaining data is fetched.
157// If num_buffers_ > 1, then the data is prefetched asynchronosuly in the
158// buffers whenever the data is consumed from the buffers and that buffer is
159// freed.
160// If num_buffers > 1, then requested data can be overlapping between 2 buffers.
161// To return the continuous buffer, overlap_buf_ is used. The requested data is
162// copied from 2 buffers to the overlap_buf_ and overlap_buf_ is returned to
163// the caller.
164
165// FilePrefetchBuffer is a smart buffer to store and read data from a file.
166class FilePrefetchBuffer {
167 public:
168 // Constructor.
169 //
170 // All arguments are optional.
171 // ReadaheadParams : Parameters to control the readahead behavior.
172 // enable : controls whether reading from the buffer is enabled.
173 // If false, TryReadFromCache() always return false, and we
174 // only take stats for the minimum offset if
175 // track_min_offset = true.
176 // See below NOTE about mmap reads.
177 // track_min_offset : Track the minimum offset ever read and collect stats on
178 // it. Used for adaptable readahead of the file
179 // footer/metadata.
180 //
181 // A user can construct a FilePrefetchBuffer without any arguments, but use
182 // `Prefetch` to load data into the buffer.
183 // NOTE: FilePrefetchBuffer is incompatible with prefetching from
184 // RandomAccessFileReaders using mmap reads, so it is common to use
185 // `!use_mmap_reads` for the `enable` parameter.
186 FilePrefetchBuffer(
187 const ReadaheadParams& readahead_params = {}, bool enable = true,
188 bool track_min_offset = false, FileSystem* fs = nullptr,
189 SystemClock* clock = nullptr, Statistics* stats = nullptr,
190 const std::function<void(bool, uint64_t&, uint64_t&)>& cb = nullptr,
191 FilePrefetchBufferUsage usage = FilePrefetchBufferUsage::kUnknown)
192 : readahead_size_(readahead_params.initial_readahead_size),
193 initial_auto_readahead_size_(readahead_params.initial_readahead_size),
194 max_readahead_size_(readahead_params.max_readahead_size),
195 min_offset_read_(std::numeric_limits<size_t>::max()),
196 enable_(enable),
197 track_min_offset_(track_min_offset),
198 implicit_auto_readahead_(readahead_params.implicit_auto_readahead),
199 prev_offset_(0),
200 prev_len_(0),
201 num_file_reads_for_auto_readahead_(
202 readahead_params.num_file_reads_for_auto_readahead),
203 num_file_reads_(readahead_params.num_file_reads),
204 explicit_prefetch_submitted_(false),
205 fs_(fs),
206 clock_(clock),
207 stats_(stats),
208 usage_(usage),
209 readaheadsize_cb_(cb),
210 num_buffers_(readahead_params.num_buffers) {
211 assert((num_file_reads_ >= num_file_reads_for_auto_readahead_ + 1) ||
212 (num_file_reads_ == 0));
213
214 // overlap_buf_ is used whenever the main buffer only has part of the
215 // requested data. The relevant data is copied into overlap_buf_ and the
216 // remaining data is copied in later to satisfy the user's request. This is
217 // used in both the synchronous (num_buffers_ = 1) and asynchronous
218 // (num_buffers_ > 1) cases. In the asynchronous case, the requested data
219 // may be spread out over 2 buffers.
220 if (num_buffers_ > 1 ||
221 (fs_ != nullptr &&
222 CheckFSFeatureSupport(fs_, FSSupportedOps::kFSBuffer))) {
223 overlap_buf_ = new BufferInfo();
224 }
225
226 free_bufs_.resize(num_buffers_);
227 for (uint32_t i = 0; i < num_buffers_; i++) {
228 free_bufs_[i] = new BufferInfo();
229 }
230 }
231
232 ~FilePrefetchBuffer() {
233 // Abort any pending async read request before destroying the class object.
234 if (fs_ != nullptr) {
235 std::vector<void*> handles;
236 for (auto& buf : bufs_) {
237 if (buf->async_read_in_progress_ && buf->io_handle_ != nullptr) {
238 handles.emplace_back(buf->io_handle_);
239 }
240 }
241 if (!handles.empty()) {
242 StopWatch sw(clock_, stats_, ASYNC_PREFETCH_ABORT_MICROS);
243 Status s = fs_->AbortIO(handles);
244 assert(s.ok());
245 }
246
247 for (auto& buf : bufs_) {
248 if (buf->io_handle_ != nullptr) {
249 DestroyAndClearIOHandle(buf);
250 buf->ClearBuffer();
251 }
252 buf->async_read_in_progress_ = false;
253 }
254 }
255
256 // Prefetch buffer bytes discarded.
257 uint64_t bytes_discarded = 0;
258 // Iterated over buffers.
259 for (auto& buf : bufs_) {
260 if (buf->DoesBufferContainData()) {
261 // If last read was from this block and some bytes are still unconsumed.
262 if (prev_offset_ >= buf->offset_ &&
263 prev_offset_ + prev_len_ < buf->offset_ + buf->CurrentSize()) {
264 bytes_discarded +=
265 buf->CurrentSize() - (prev_offset_ + prev_len_ - buf->offset_);
266 }
267 // If last read was from previous blocks and this block is unconsumed.
268 else if (prev_offset_ < buf->offset_ &&
269 prev_offset_ + prev_len_ <= buf->offset_) {
270 bytes_discarded += buf->CurrentSize();
271 }
272 }
273 }
274
275 RecordInHistogram(stats_, PREFETCHED_BYTES_DISCARDED, bytes_discarded);
276
277 for (auto& buf : bufs_) {
278 delete buf;
279 buf = nullptr;
280 }
281
282 for (auto& buf : free_bufs_) {
283 delete buf;
284 buf = nullptr;
285 }
286
287 if (overlap_buf_ != nullptr) {
288 delete overlap_buf_;
289 overlap_buf_ = nullptr;
290 }
291 }

358 void DecreaseReadAheadIfEligible(uint64_t offset, size_t size,
359 size_t value = DEFAULT_DECREMENT) {
360 if (bufs_.empty()) {
361 return;
362 }
363
364 // Decrease the readahead_size if
365 // - its enabled internally by RocksDB (implicit_auto_readahead_) and,
366 // - readahead_size is greater than 0 and,
367 // - this block would have called prefetch API if not found in cache for
368 // which conditions are:
369 // - few/no bytes are in buffer and,
370 // - block is sequential with the previous read and,
371 // - num_file_reads_ + 1 (including this read) >
372 // num_file_reads_for_auto_readahead_
373
374 size_t curr_size = bufs_.front()->async_read_in_progress_
375 ? bufs_.front()->async_req_len_
376 : bufs_.front()->CurrentSize();
377 if (implicit_auto_readahead_ && readahead_size_ > 0) {
378 if ((offset + size > bufs_.front()->offset_ + curr_size) &&
379 IsBlockSequential(offset) &&
380 (num_file_reads_ + 1 > num_file_reads_for_auto_readahead_)) {
381 readahead_size_ =
382 std::max(initial_auto_readahead_size_,
383 (readahead_size_ >= value ? readahead_size_ - value : 0));
384 }
385 }

454 // Called in case of implicit auto prefetching.
455 bool IsEligibleForPrefetch(uint64_t offset, size_t n) {
456 // Prefetch only if this read is sequential otherwise reset readahead_size_
457 // to initial value.
458 if (!IsBlockSequential(offset)) {
459 UpdateReadPattern(offset, n, false /*decrease_readaheadsize*/);
460 ResetValues();
461 return false;
462 }
463 num_file_reads_++;
464
465 // Since async request was submitted in last call directly by calling
466 // PrefetchAsync, it skips num_file_reads_ check as this call is to poll the
467 // data submitted in previous call.
468 if (explicit_prefetch_submitted_) {
469 return true;
470 }
471 if (num_file_reads_ <= num_file_reads_for_auto_readahead_) {
472 UpdateReadPattern(offset, n, false /*decrease_readaheadsize*/);
473 return false;
474 }
475 return true;
476 }

509 // Reuses the file system allocated buffer to avoid an extra copy
510 IOStatus FSBufferDirectRead(RandomAccessFileReader* reader, BufferInfo* buf,
511 const IOOptions& opts, uint64_t offset, size_t n,
512 Slice& result) {
513 FSReadRequest read_req;
514 read_req.offset = offset;
515 read_req.len = n;
516 read_req.scratch = nullptr;
517 IOStatus s = reader->MultiRead(opts, &read_req, 1, nullptr);
518 if (!s.ok()) {
519 return s;
520 }
521 s = read_req.status;
522 if (!s.ok()) {
523 return s;
524 }
525 buf->buffer_.SetBuffer(read_req.result, std::move(read_req.fs_scratch));
526 buf->offset_ = offset;
527 buf->initial_end_offset_ = offset + read_req.result.size();
528 result = read_req.result;
529 return s;
530 }
