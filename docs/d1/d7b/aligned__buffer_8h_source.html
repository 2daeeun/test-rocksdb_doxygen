20// Truncate to a multiple of page_size, which is also a page boundary. This
21// helps to figuring out the right alignment.
22// Example:
23// TruncateToPageBoundary(4096, 5000) => 4096
24// TruncateToPageBoundary((4096, 10000) => 8192
25inline size_t TruncateToPageBoundary(size_t page_size, size_t s) {
26 s -= (s & (page_size - 1));
27 assert((s % page_size) == 0);
28 return s;
29}

31// Round up x to a multiple of y.
32// Example:
33// Roundup(13, 5) => 15
34// Roundup(201, 16) => 208
35inline size_t Roundup(size_t x, size_t y) { return ((x + y - 1) / y) * y; }
36
37// Round down x to a multiple of y.
38// Example:
39// Rounddown(13, 5) => 10
40// Rounddown(201, 16) => 192
41inline size_t Rounddown(size_t x, size_t y) { return (x / y) * y; }

43// AlignedBuffer manages a buffer by taking alignment into consideration, and
44// aligns the buffer start and end positions. It is mainly used for direct I/O,
45// though it can be used other purposes as well.
46// It also supports expanding the managed buffer, and copying whole or part of
47// the data from old buffer into the new expanded buffer. Such a copy especially
48// helps in cases avoiding an IO to re-fetch the data from disk.
49//
50// Example:
51// AlignedBuffer buf;
52// buf.Alignment(alignment);
53// buf.AllocateNewBuffer(user_requested_buf_size);
54// ...
55// buf.AllocateNewBuffer(2*user_requested_buf_size, /*copy_data*/ true,
56// copy_offset, copy_len);
57class AlignedBuffer {

59 FSAllocationPtr buf_;
60 size_t capacity_;
61 size_t cursize_;
62 char* bufstart_;

116 // Points the buffer to the result without allocating extra
117 // memory or performing any data copies. Takes ownership of the
118 // FSAllocationPtr. This method is called when we want to reuse the buffer
119 // provided by the file system
120 void SetBuffer(Slice& result, FSAllocationPtr new_buf) {
121 alignment_ = 1;
122 capacity_ = result.size();
123 cursize_ = result.size();
124 buf_ = std::move(new_buf);
125 assert(buf_.get() != nullptr);
126 // Note: bufstart_ must point to result.data() and not new_buf, which can
127 // point to any arbitrary object
128 bufstart_ = const_cast<char*>(result.data());
129 }

131 // Allocates a new buffer and sets the start position to the first aligned
132 // byte.
133 //
134 // requested_capacity: requested new buffer capacity. This capacity will be
135 // rounded up based on alignment.
136 // copy_data: Copy data from old buffer to new buffer. If copy_offset and
137 // copy_len are not passed in and the new requested capacity is bigger
138 // than the existing buffer's capacity, the data in the exising buffer is
139 // fully copied over to the new buffer.
140 // copy_offset: Copy data from this offset in old buffer.
141 // copy_len: Number of bytes to copy.
142 //
143 // The function does nothing if the new requested_capacity is smaller than
144 // the current buffer capacity and copy_data is true i.e. the old buffer is
145 // retained as is.
146 void AllocateNewBuffer(size_t requested_capacity, bool copy_data = false,
147 uint64_t copy_offset = 0, size_t copy_len = 0) {
148 assert(alignment_ > 0);
149 assert((alignment_ & (alignment_ - 1)) == 0);
150
151 copy_len = copy_len > 0 ? copy_len : cursize_;
152 if (copy_data && requested_capacity < copy_len) {
153 // If we are downsizing to a capacity that is smaller than the current
154 // data in the buffer -- Ignore the request.
155 return;
156 }
157
158 size_t new_capacity = Roundup(requested_capacity, alignment_);
159 char* new_buf = new char[new_capacity + alignment_];
160 char* new_bufstart = reinterpret_cast<char*>(
161 (reinterpret_cast<uintptr_t>(new_buf) + (alignment_ - 1)) &
162 ~static_cast<uintptr_t>(alignment_ - 1));
163
164 if (copy_data) {
165 assert(bufstart_ + copy_offset + copy_len <= bufstart_ + cursize_);
166 memcpy(new_bufstart, bufstart_ + copy_offset, copy_len);
167 cursize_ = copy_len;
168 } else {
169 cursize_ = 0;
170 }
171
172 bufstart_ = new_bufstart;
173 capacity_ = new_capacity;
174 // buf_ is a FSAllocationPtr which takes in a deleter
175 // we can just wrap the regular default delete that would have been called
176 buf_ = std::unique_ptr<void, std::function<void(void*)>>(
177 static_cast<void*>(new_buf),
178 [](void* p) { delete[] static_cast<char*>(p); });
179 }
180
181 // Append to the buffer.
182 //
183 // src : source to copy the data from.
184 // append_size : number of bytes to copy from src.
185 // Returns the number of bytes appended.
186 //
187 // If append_size is more than the remaining buffer size only the
188 // remaining-size worth of bytes are copied.
189 size_t Append(const char* src, size_t append_size) {
190 size_t buffer_remaining = capacity_ - cursize_;
191 size_t to_copy = std::min(append_size, buffer_remaining);
192
193 if (to_copy > 0) {
194 memcpy(bufstart_ + cursize_, src, to_copy);
195 cursize_ += to_copy;
196 }
197 return to_copy;
198 }
199
200 // Read from the buffer.
201 //
202 // dest : destination buffer to copy the data to.
203 // offset : the buffer offset to start reading from.
204 // read_size : the number of bytes to copy from the buffer to dest.
205 // Returns the number of bytes read/copied to dest.
206 size_t Read(char* dest, size_t offset, size_t read_size) const {
207 assert(offset < cursize_);
208
209 size_t to_read = 0;
210 if (offset < cursize_) {
211 to_read = std::min(cursize_ - offset, read_size);
212 }
213 if (to_read > 0) {
214 memcpy(dest, bufstart_ + offset, to_read);
215 }
216 return to_read;
217 }
218
219 // Pad to the end of alignment with "padding"
220 void PadToAlignmentWith(int padding) {
221 size_t total_size = Roundup(cursize_, alignment_);
222 size_t pad_size = total_size - cursize_;
223
224 if (pad_size > 0) {
225 assert((pad_size + cursize_) <= capacity_);
226 memset(bufstart_ + cursize_, padding, pad_size);
227 cursize_ += pad_size;
228 }
229 }
230
231 void PadWith(size_t pad_size, int padding) {
232 assert((pad_size + cursize_) <= capacity_);
233 memset(bufstart_ + cursize_, padding, pad_size);
234 cursize_ += pad_size;
235 }
236
237 // After a partial flush move the tail to the beginning of the buffer.
238 void RefitTail(size_t tail_offset, size_t tail_size) {
239 if (tail_size > 0) {
240 memmove(bufstart_, bufstart_ + tail_offset, tail_size);
241 }
242 cursize_ = tail_size;
243 }
244
245 // Returns a place to start appending.
246 // WARNING: Note that it is possible to write past the end of the buffer if
247 // the buffer is modified without using the write APIs or encapsulation
248 // offered by AlignedBuffer. It is up to the user to guard against such
249 // errors.
250 char* Destination() { return bufstart_ + cursize_; }
251
252 void Size(size_t cursize) { cursize_ = cursize; }
253};
