18// SharedBlobFileMetaData represents the immutable part of blob files' metadata,
19// like the blob file number, total number and size of blobs, or checksum
20// method and value. There is supposed to be one object of this class per blob
21// file (shared across all versions that include the blob file in question);
22// hence, the type is neither copyable nor movable. A blob file can be marked
23// obsolete when the corresponding SharedBlobFileMetaData object is destroyed.
24
25class SharedBlobFileMetaData {
26 public:
27 static std::shared_ptr<SharedBlobFileMetaData> Create(
28 uint64_t blob_file_number, uint64_t total_blob_count,
29 uint64_t total_blob_bytes, std::string checksum_method,
30 std::string checksum_value) {
31 return std::shared_ptr<SharedBlobFileMetaData>(new SharedBlobFileMetaData(
32 blob_file_number, total_blob_count, total_blob_bytes,
33 std::move(checksum_method), std::move(checksum_value)));
34 }
35
36 template <typename Deleter>
37 static std::shared_ptr<SharedBlobFileMetaData> Create(
38 uint64_t blob_file_number, uint64_t total_blob_count,
39 uint64_t total_blob_bytes, std::string checksum_method,
40 std::string checksum_value, Deleter deleter) {
41 return std::shared_ptr<SharedBlobFileMetaData>(
42 new SharedBlobFileMetaData(blob_file_number, total_blob_count,
43 total_blob_bytes, std::move(checksum_method),
44 std::move(checksum_value)),
45 deleter);
46 }
47
48 SharedBlobFileMetaData(const SharedBlobFileMetaData&) = delete;
49 SharedBlobFileMetaData& operator=(const SharedBlobFileMetaData&) = delete;
50
51 SharedBlobFileMetaData(SharedBlobFileMetaData&&) = delete;
52 SharedBlobFileMetaData& operator=(SharedBlobFileMetaData&&) = delete;
53
54 uint64_t GetBlobFileSize() const;
55 uint64_t GetBlobFileNumber() const { return blob_file_number_; }
56 uint64_t GetTotalBlobCount() const { return total_blob_count_; }
57 uint64_t GetTotalBlobBytes() const { return total_blob_bytes_; }
58 const std::string& GetChecksumMethod() const { return checksum_method_; }
59 const std::string& GetChecksumValue() const { return checksum_value_; }
60
61 std::string DebugString() const;
62
63 private:
64 SharedBlobFileMetaData(uint64_t blob_file_number, uint64_t total_blob_count,
65 uint64_t total_blob_bytes, std::string checksum_method,
66 std::string checksum_value)
67 : blob_file_number_(blob_file_number),
68 total_blob_count_(total_blob_count),
69 total_blob_bytes_(total_blob_bytes),
70 checksum_method_(std::move(checksum_method)),
71 checksum_value_(std::move(checksum_value)) {
72 assert(checksum_method_.empty() == checksum_value_.empty());
73 }
74
75 uint64_t blob_file_number_;
76 uint64_t total_blob_count_;
77 uint64_t total_blob_bytes_;
78 std::string checksum_method_;
79 std::string checksum_value_;
80};

85// BlobFileMetaData contains the part of the metadata for blob files that can
86// vary across versions, like the amount of garbage in the blob file. In
87// addition, BlobFileMetaData objects point to and share the ownership of the
88// SharedBlobFileMetaData object for the corresponding blob file. Similarly to
89// SharedBlobFileMetaData, BlobFileMetaData are not copyable or movable. They
90// are meant to be jointly owned by the versions in which the blob file has the
91// same (immutable *and* mutable) state.
92
93class BlobFileMetaData {
94 public:
95 using LinkedSsts = std::unordered_set<uint64_t>;
96
97 static std::shared_ptr<BlobFileMetaData> Create(
98 std::shared_ptr<SharedBlobFileMetaData> shared_meta,
99 LinkedSsts linked_ssts, uint64_t garbage_blob_count,
100 uint64_t garbage_blob_bytes) {
101 return std::shared_ptr<BlobFileMetaData>(
102 new BlobFileMetaData(std::move(shared_meta), std::move(linked_ssts),
103 garbage_blob_count, garbage_blob_bytes));
104 }
105
106 BlobFileMetaData(const BlobFileMetaData&) = delete;
107 BlobFileMetaData& operator=(const BlobFileMetaData&) = delete;
108
109 BlobFileMetaData(BlobFileMetaData&&) = delete;
110 BlobFileMetaData& operator=(BlobFileMetaData&&) = delete;
111
112 const std::shared_ptr<SharedBlobFileMetaData>& GetSharedMeta() const {
113 return shared_meta_;
114 }
115
116 uint64_t GetBlobFileSize() const {
117 assert(shared_meta_);
118 return shared_meta_->GetBlobFileSize();
119 }
120
121 uint64_t GetBlobFileNumber() const {
122 assert(shared_meta_);
123 return shared_meta_->GetBlobFileNumber();
124 }
125 uint64_t GetTotalBlobCount() const {
126 assert(shared_meta_);
127 return shared_meta_->GetTotalBlobCount();
128 }
129 uint64_t GetTotalBlobBytes() const {

132 }
133 const std::string& GetChecksumMethod() const {
134 assert(shared_meta_);
135 return shared_meta_->GetChecksumMethod();
136 }
137 const std::string& GetChecksumValue() const {
138 assert(shared_meta_);
139 return shared_meta_->GetChecksumValue();
140 }
141
142 const LinkedSsts& GetLinkedSsts() const { return linked_ssts_; }
143
144 uint64_t GetGarbageBlobCount() const { return garbage_blob_count_; }
145 uint64_t GetGarbageBlobBytes() const { return garbage_blob_bytes_; }
146
147 std::string DebugString() const;
148
149 private:
150 BlobFileMetaData(std::shared_ptr<SharedBlobFileMetaData> shared_meta,
151 LinkedSsts linked_ssts, uint64_t garbage_blob_count,
152 uint64_t garbage_blob_bytes)
153 : shared_meta_(std::move(shared_meta)),
154 linked_ssts_(std::move(linked_ssts)),
155 garbage_blob_count_(garbage_blob_count),
156 garbage_blob_bytes_(garbage_blob_bytes) {
157 assert(shared_meta_);
158 assert(garbage_blob_count_ <= shared_meta_->GetTotalBlobCount());
159 assert(garbage_blob_bytes_ <= shared_meta_->GetTotalBlobBytes());
160 }
161
162 std::shared_ptr<SharedBlobFileMetaData> shared_meta_;
163 LinkedSsts linked_ssts_;
164 uint64_t garbage_blob_count_;
165 uint64_t garbage_blob_bytes_;
166};
