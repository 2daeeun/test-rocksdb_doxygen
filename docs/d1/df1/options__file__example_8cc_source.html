40namespace {
41// A dummy compaction filter
42class DummyCompactionFilter : public CompactionFilter {
43 public:
44 virtual ~DummyCompactionFilter() {}
45 virtual bool Filter(int level, const Slice& key, const Slice& existing_value,
46 std::string* new_value, bool* value_changed) const {
47 return false;
48 }
49 virtual const char* Name() const { return "DummyCompactionFilter"; }
50};

54int main() {
55 DBOptions db_opt;
56 db_opt.create_if_missing = true;
57
58 std::vector<ColumnFamilyDescriptor> cf_descs;
59 cf_descs.push_back(
60 {ROCKSDB_NAMESPACE::kDefaultColumnFamilyName, ColumnFamilyOptions()});
61 cf_descs.push_back({"new_cf", ColumnFamilyOptions()});
62
63 // initialize BlockBasedTableOptions
64 auto cache = NewLRUCache(1 * 1024 * 1024 * 1024);
65 BlockBasedTableOptions bbt_opts;
66 bbt_opts.block_size = 32 * 1024;
67 bbt_opts.block_cache = cache;
68
69 // initialize column families options
70 std::unique_ptr<CompactionFilter> compaction_filter;
71 compaction_filter.reset(new DummyCompactionFilter());
72 cf_descs[0].options.table_factory.reset(NewBlockBasedTableFactory(bbt_opts));
73 cf_descs[0].options.compaction_filter = compaction_filter.get();
74 cf_descs[1].options.table_factory.reset(NewBlockBasedTableFactory(bbt_opts));
75
76 // destroy and open DB
77 DB* db;
78 Status s = ROCKSDB_NAMESPACE::DestroyDB(kDBPath,
79 Options(db_opt, cf_descs[0].options));
80 assert(s.ok());
81 s = DB::Open(Options(db_opt, cf_descs[0].options), kDBPath, &db);
82 assert(s.ok());
83
84 // Create column family, and rocksdb will persist the options.
85 ColumnFamilyHandle* cf;
86 s = db->CreateColumnFamily(ColumnFamilyOptions(), "new_cf", &cf);
87 assert(s.ok());
88
89 // close DB
90 delete cf;
91 delete db;
92
93 // In the following code, we will reopen the rocksdb instance using
94 // the options file stored in the db directory.
95
96 // Load the options file.
97 DBOptions loaded_db_opt;
98 std::vector<ColumnFamilyDescriptor> loaded_cf_descs;
99 ConfigOptions config_options;
100 s = LoadLatestOptions(config_options, kDBPath, &loaded_db_opt,
101 &loaded_cf_descs);
102 assert(s.ok());
103 assert(loaded_db_opt.create_if_missing == db_opt.create_if_missing);
104
105 // Initialize pointer options for each column family
106 for (size_t i = 0; i < loaded_cf_descs.size(); ++i) {
107 auto* loaded_bbt_opt =
108 loaded_cf_descs[0]
109 .options.table_factory->GetOptions<BlockBasedTableOptions>();
110 // Expect the same as BlockBasedTableOptions will be loaded form file.
111 assert(loaded_bbt_opt->block_size == bbt_opts.block_size);
112 // However, block_cache needs to be manually initialized as documented
113 // in rocksdb/utilities/options_util.h.
114 loaded_bbt_opt->block_cache = cache;
115 }
116 // In addition, as pointer options are initialized with default value,
117 // we need to properly initialized all the pointer options if non-defalut
118 // values are used before calling DB::Open().
119 assert(loaded_cf_descs[0].options.compaction_filter == nullptr);
120 loaded_cf_descs[0].options.compaction_filter = compaction_filter.get();
121
122 // reopen the db using the loaded options.
123 std::vector<ColumnFamilyHandle*> handles;
124 s = DB::Open(loaded_db_opt, kDBPath, loaded_cf_descs, &handles, &db);
125 assert(s.ok());
126
127 // close DB
128 for (auto* handle : handles) {
129 delete handle;
130 }
131 delete db;
132}
