63__attribute__((const, always_inline)) static inline intptr_t which_cache_line(
64 intptr_t addr) {
65 static const size_t assumed_cache_line_size = 64;
66 return addr / assumed_cache_line_size;
67}
68template <typename T>
69__attribute__((const, always_inline)) static inline bool crosses_boundary(
70 T *addr, size_t width) {
71 const intptr_t int_addr = reinterpret_cast<intptr_t>(addr);
72 const intptr_t last_byte = int_addr + width - 1;
73 return which_cache_line(int_addr) != which_cache_line(last_byte);
74}
