27class PartitionedFilterBlockBuilder : public FullFilterBlockBuilder {
28 public:
29 explicit PartitionedFilterBlockBuilder(
30 const SliceTransform* prefix_extractor, bool whole_key_filtering,
31 FilterBitsBuilder* filter_bits_builder, int index_block_restart_interval,
32 const bool use_value_delta_encoding,
33 PartitionedIndexBuilder* const p_index_builder,
34 const uint32_t partition_size, size_t ts_sz,
35 const bool persist_user_defined_timestamps,
36 bool decouple_from_index_partitions);
37
38 virtual ~PartitionedFilterBlockBuilder();
39
40 void Add(const Slice& key_without_ts) override;
41 void AddWithPrevKey(const Slice& key_without_ts,
42 const Slice& prev_key_without_ts) override;
43 bool IsEmpty() const override {
44 return filter_bits_builder_->EstimateEntriesAdded() == 0 &&
45 filters_.empty();
46 }
47
48 size_t EstimateEntriesAdded() override;
49
50 void PrevKeyBeforeFinish(const Slice& prev_key_without_ts) override;
51 Status Finish(const BlockHandle& last_partition_block_handle, Slice* filter,
52 std::unique_ptr<const char[]>* filter_owner = nullptr) override;
53
54 void ResetFilterBitsBuilder() override {
55 filters_.clear();
56 total_added_in_built_ = 0;
57 index_on_filter_block_builder_.Reset();
58 index_on_filter_block_builder_without_seq_.Reset();
59 FullFilterBlockBuilder::ResetFilterBitsBuilder();
60 }
61
62 // For PartitionFilter, optional post-verifing the filter is done
63 // as part of PartitionFilterBlockBuilder::Finish
64 // to avoid implementation complexity of doing it elsewhere.
65 // Therefore we are skipping it in here.
66 Status MaybePostVerifyFilter(const Slice& /* filter_content */) override {
67 return Status::OK();
68 }
69
70 private: // fns
71 // Whether to cut a filter block before the next key
72 bool DecideCutAFilterBlock();
73 void CutAFilterBlock(const Slice* next_key, const Slice* next_prefix,
74 const Slice& prev_key);
75
76 void AddImpl(const Slice& key_without_ts, const Slice& prev_key_without_ts);
77
78 private: // data
79 // Currently we keep the same number of partitions for filters and indexes.
80 // This would allow for some potentioal optimizations in future. If such
81 // optimizations did not realize we can use different number of partitions and
82 // eliminate p_index_builder_
83 PartitionedIndexBuilder* const p_index_builder_;
84 const size_t ts_sz_;
85 const bool decouple_from_index_partitions_;
86
87 // Filter data
88 struct FilterEntry {
89 std::string ikey; // internal key or separator *after* this filter
90 std::unique_ptr<const char[]> filter_owner;
91 Slice filter;
92 };
93 std::deque<FilterEntry> filters_; // list of partitioned filters and keys
94 // used in building the index
95 // The desired number of keys per partition
96 uint32_t keys_per_partition_;
97 // According to the bits builders, how many keys/prefixes added
98 // in all the filters we have fully built
99 uint64_t total_added_in_built_ = 0;
100
101 // Set to the first non-okay status if any of the filter
102 // partitions experiences construction error.
103 // If partitioned_filters_construction_status_ is non-okay,
104 // then the whole partitioned filters should not be used.
105 Status partitioned_filters_construction_status_;
106
107 // For Add without prev key
108 std::string prev_key_without_ts_;
109
110#ifndef NDEBUG
111 // For verifying accurate previous keys are provided by the caller, so that
112 // release code can be fast
113 bool DEBUG_add_with_prev_key_called_ = false;
114 std::string DEBUG_prev_key_without_ts_;
115#endif // NDEBUG
116
117 // ===== State for Finish() =====
118
119 // top-level index builder on internal keys
120 BlockBuilder index_on_filter_block_builder_;
121 // same for user keys
122 BlockBuilder index_on_filter_block_builder_without_seq_;
123 // For delta-encoding handles
124 BlockHandle last_encoded_handle_;
125 // True if we are between two calls to Finish(), because we have returned
126 // the filter at the front of filters_ but haven't yet added it to the
127 // partition index.
128 bool finishing_front_filter_ = false;
129};
