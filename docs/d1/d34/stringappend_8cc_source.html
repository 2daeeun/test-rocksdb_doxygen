38// Implementation for the merge operation (concatenates two strings)
39bool StringAppendOperator::Merge(const Slice& /*key*/,
40 const Slice* existing_value,
41 const Slice& value, std::string* new_value,
42 Logger* /*logger*/) const {
43 // Clear the *new_value for writing.
44 assert(new_value);
45 new_value->clear();
46
47 if (!existing_value) {
48 // No existing_value. Set *new_value = value
49 new_value->assign(value.data(), value.size());
50 } else {
51 // Generic append (existing_value != null).
52 // Reserve *new_value to correct size, and apply concatenation.
53 new_value->reserve(existing_value->size() + delim_.size() + value.size());
54 new_value->assign(existing_value->data(), existing_value->size());
55 new_value->append(delim_);
56 new_value->append(value.data(), value.size());
57 }
58
59 return true;
60}
