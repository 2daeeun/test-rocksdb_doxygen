15// This function reads multiple data blocks from disk using Env::MultiRead()
16// and optionally inserts them into the block cache. It uses the scratch
17// buffer provided by the caller, which is contiguous. If scratch is a nullptr
18// it allocates a separate buffer for each block. Typically, if the blocks
19// need to be uncompressed and there is no compressed block cache, callers
20// can allocate a temporary scratch buffer in order to minimize memory
21// allocations.
22// If options.fill_cache is true, it inserts the blocks into cache. If its
23// false and scratch is non-null and the blocks are uncompressed, it copies
24// the buffers to heap. In any case, the CachableEntry<Block> returned will
25// own the data bytes.
26// If compression is enabled and also there is no compressed block cache,
27// the adjacent blocks are read out in one IO (combined read)
28// batch - A MultiGetRange with only those keys with unique data blocks not
29// found in cache
30// handles - A vector of block handles. Some of them me be NULL handles
31// scratch - An optional contiguous buffer to read compressed blocks into
32DEFINE_SYNC_AND_ASYNC(void, BlockBasedTable::RetrieveMultipleBlocks)
33(const ReadOptions& options, const MultiGetRange* batch,
34 const autovector<BlockHandle, MultiGetContext::MAX_BATCH_SIZE>* handles,
35 Status* statuses, CachableEntry<Block_kData>* results, char* scratch,
36 const UncompressionDict& uncompression_dict, bool use_fs_scratch) const {
37 RandomAccessFileReader* file = rep_->file.get();
38 const Footer& footer = rep_->footer;
39 const ImmutableOptions& ioptions = rep_->ioptions;
40 size_t read_amp_bytes_per_bit = rep_->table_options.read_amp_bytes_per_bit;
41 MemoryAllocator* memory_allocator = GetMemoryAllocator(rep_->table_options);
42
43 if (ioptions.allow_mmap_reads) {
44 size_t idx_in_batch = 0;
45 for (auto mget_iter = batch->begin(); mget_iter != batch->end();
46 ++mget_iter, ++idx_in_batch) {
47 const BlockHandle& handle = (*handles)[idx_in_batch];
48 if (handle.IsNull()) {
49 continue;
50 }
51
52 // XXX: use_cache=true means double cache query?
53 statuses[idx_in_batch] = RetrieveBlock(
54 nullptr, options, handle, uncompression_dict,
55 &results[idx_in_batch].As<Block_kData>(), mget_iter->get_context,
56 /* lookup_context */ nullptr,
57 /* for_compaction */ false, /* use_cache */ true,
58 /* async_read */ false, /* use_block_cache_for_lookup */ true);
59 }
60 assert(idx_in_batch == handles->size());
61 CO_RETURN;
62 }
63
64 // In direct IO mode, blocks share the direct io buffer.
65 // Otherwise, blocks share the scratch buffer.
66 const bool use_shared_buffer = file->use_direct_io() || scratch != nullptr;
67
68 autovector<FSReadRequest, MultiGetContext::MAX_BATCH_SIZE> read_reqs;
69 size_t buf_offset = 0;
70 size_t idx_in_batch = 0;
71
72 uint64_t prev_offset = 0;
73 size_t prev_len = 0;
74 autovector<size_t, MultiGetContext::MAX_BATCH_SIZE> req_idx_for_block;
75 autovector<size_t, MultiGetContext::MAX_BATCH_SIZE> req_offset_for_block;
76 for (auto mget_iter = batch->begin(); mget_iter != batch->end();
77 ++mget_iter, ++idx_in_batch) {
78 const BlockHandle& handle = (*handles)[idx_in_batch];
79 if (handle.IsNull()) {
80 continue;
81 }
82
83 size_t prev_end = static_cast<size_t>(prev_offset) + prev_len;
84
85 // If current block is adjacent to the previous one, at the same time,
86 // compression is enabled and there is no compressed cache, we combine
87 // the two block read as one.
88 // We don't combine block reads here in direct IO mode, because when doing
89 // direct IO read, the block requests will be realigned and merged when
90 // necessary.
91 if ((use_shared_buffer || use_fs_scratch) && !file->use_direct_io() &&
92 prev_end == handle.offset()) {
93 req_offset_for_block.emplace_back(prev_len);
94 prev_len += BlockSizeWithTrailer(handle);
95 } else {
96 // No compression or current block and previous one is not adjacent:
97 // Step 1, create a new request for previous blocks
98 if (prev_len != 0) {
99 FSReadRequest req;
100 req.offset = prev_offset;
101 req.len = prev_len;
102 if (file->use_direct_io() || use_fs_scratch) {
103 req.scratch = nullptr;
104 } else if (use_shared_buffer) {
105 req.scratch = scratch + buf_offset;
106 buf_offset += req.len;
107 } else {
108 req.scratch = new char[req.len];
109 }
110 read_reqs.emplace_back(std::move(req));
111 }
112
113 // Step 2, remember the previous block info
114 prev_offset = handle.offset();
115 prev_len = BlockSizeWithTrailer(handle);
116 req_offset_for_block.emplace_back(0);
117 }
118 req_idx_for_block.emplace_back(read_reqs.size());
119
120 PERF_COUNTER_ADD(block_read_count, 1);
121 PERF_COUNTER_ADD(block_read_byte, BlockSizeWithTrailer(handle));
122 }
123 // Handle the last block and process the pending last request
124 if (prev_len != 0) {
125 FSReadRequest req;
126 req.offset = prev_offset;
127 req.len = prev_len;
128 if (file->use_direct_io() || use_fs_scratch) {
129 req.scratch = nullptr;
130 } else if (use_shared_buffer) {
131 req.scratch = scratch + buf_offset;
132 } else {
133 req.scratch = new char[req.len];
134 }
135 read_reqs.emplace_back(std::move(req));
136 }
137
138 AlignedBuf direct_io_buf;
139 {
140 IOOptions opts;
141 IOStatus s = file->PrepareIOOptions(options, opts);
142 if (s.ok()) {
143#if defined(WITH_COROUTINES)
144 if (file->use_direct_io()) {
145#endif // WITH_COROUTINES
146 s = file->MultiRead(opts, &read_reqs[0], read_reqs.size(),
147 &direct_io_buf);
148#if defined(WITH_COROUTINES)
149 } else {
150 co_await batch->context()->reader().MultiReadAsync(
151 file, opts, &read_reqs[0], read_reqs.size(), &direct_io_buf);
152 }
153#endif // WITH_COROUTINES
154 }
155 if (!s.ok()) {
156 // Discard all the results in this batch if there is any time out
157 // or overall MultiRead error
158 for (FSReadRequest& req : read_reqs) {
159 req.status = s;
160 }
161 }
162 }
163
164 idx_in_batch = 0;
165 size_t valid_batch_idx = 0;
166 for (auto mget_iter = batch->begin(); mget_iter != batch->end();
167 ++mget_iter, ++idx_in_batch) {
168 const BlockHandle& handle = (*handles)[idx_in_batch];
169
170 if (handle.IsNull()) {
171 continue;
172 }
173
174 assert(valid_batch_idx < req_idx_for_block.size());
175 assert(valid_batch_idx < req_offset_for_block.size());
176 assert(req_idx_for_block[valid_batch_idx] < read_reqs.size());
177 size_t& req_idx = req_idx_for_block[valid_batch_idx];
178 size_t& req_offset = req_offset_for_block[valid_batch_idx];
179 valid_batch_idx++;
180 FSReadRequest& req = read_reqs[req_idx];
181 Status s = req.status;
182 if (s.ok()) {
183 if ((req.result.size() != req.len) ||
184 (req_offset + BlockSizeWithTrailer(handle) > req.result.size())) {
185 s = Status::Corruption("truncated block read from " +
186 rep_->file->file_name() + " offset " +
187 std::to_string(handle.offset()) + ", expected " +
188 std::to_string(req.len) + " bytes, got " +
189 std::to_string(req.result.size()));
190 }
191 }
192
193 BlockContents serialized_block;
194 if (s.ok()) {
195 if (!use_fs_scratch && !use_shared_buffer) {
196 // We allocated a buffer for this block. Give ownership of it to
197 // BlockContents so it can free the memory
198 assert(req.result.data() == req.scratch);
199 assert(req.result.size() == BlockSizeWithTrailer(handle));
200 assert(req_offset == 0);
201 serialized_block =
202 BlockContents(std::unique_ptr<char[]>(req.scratch), handle.size());
203 } else {
204 // We used the scratch buffer or direct io buffer
205 // which are shared by the blocks.
206 // In case of use_fs_scratch, underlying file system provided buffer is
207 // used. serialized_block does not have the ownership.
208 serialized_block =
209 BlockContents(Slice(req.result.data() + req_offset, handle.size()));
210 }
211#ifndef NDEBUG
212 serialized_block.has_trailer = true;
213#endif
214
215 if (options.verify_checksums) {
216 PERF_TIMER_GUARD(block_checksum_time);
217 const char* data = serialized_block.data.data();
218 // Since the scratch might be shared, the offset of the data block in
219 // the buffer might not be 0. req.result.data() only point to the
220 // begin address of each read request, we need to add the offset
221 // in each read request. Checksum is stored in the block trailer,
222 // beyond the payload size.
223 s = VerifyBlockChecksum(footer, data, handle.size(),
224 rep_->file->file_name(), handle.offset());
225 RecordTick(ioptions.stats, BLOCK_CHECKSUM_COMPUTE_COUNT);
226 if (!s.ok()) {
227 RecordTick(ioptions.stats, BLOCK_CHECKSUM_MISMATCH_COUNT);
228 }
229 TEST_SYNC_POINT_CALLBACK("RetrieveMultipleBlocks:VerifyChecksum", &s);
230 if (!s.ok() &&
231 CheckFSFeatureSupport(ioptions.fs.get(),
232 FSSupportedOps::kVerifyAndReconstructRead)) {
233 assert(s.IsCorruption());
234 assert(!ioptions.allow_mmap_reads);
235 RecordTick(ioptions.stats, FILE_READ_CORRUPTION_RETRY_COUNT);
236
237 // Repeat the read for this particular block using the regular
238 // synchronous Read API. We can use the same chunk of memory
239 // pointed to by data, since the size is identical and we know
240 // its not a memory mapped file
241 Slice result;
242 IOOptions opts;
243 IOStatus io_s = file->PrepareIOOptions(options, opts);
244 opts.verify_and_reconstruct_read = true;
245 io_s = file->Read(opts, handle.offset(), BlockSizeWithTrailer(handle),
246 &result, const_cast<char*>(data), nullptr);
247 if (io_s.ok()) {
248 assert(result.data() == data);
249 assert(result.size() == BlockSizeWithTrailer(handle));
250 s = VerifyBlockChecksum(footer, data, handle.size(),
251 rep_->file->file_name(), handle.offset());
252 if (s.ok()) {
253 RecordTick(ioptions.stats,
254 FILE_READ_CORRUPTION_RETRY_SUCCESS_COUNT);
255 }
256 } else {
257 s = io_s;
258 }
259 }
260 }
261 } else if (!use_shared_buffer) {
262 // Free the allocated scratch buffer.
263 delete[] req.scratch;
264 }
265
266 if (s.ok()) {
267 // When the blocks share the same underlying buffer (scratch or direct io
268 // buffer), we may need to manually copy the block into heap if the
269 // serialized block has to be inserted into a cache. That falls into the
270 // following cases -
271 // 1. serialized block is not compressed, it needs to be inserted into
272 // the uncompressed block cache if there is one
273 // 2. If the serialized block is compressed, it needs to be inserted
274 // into the compressed block cache if there is one
275 //
276 // In all other cases, the serialized block is either uncompressed into a
277 // heap buffer or there is no cache at all.
278 CompressionType compression_type =
279 GetBlockCompressionType(serialized_block);
280 if ((use_fs_scratch || use_shared_buffer) &&
281 compression_type == kNoCompression) {
282 Slice serialized =
283 Slice(req.result.data() + req_offset, BlockSizeWithTrailer(handle));
284 serialized_block = BlockContents(
285 CopyBufferToHeap(GetMemoryAllocator(rep_->table_options),
286 serialized),
287 handle.size());
288#ifndef NDEBUG
289 serialized_block.has_trailer = true;
290#endif
291 }
292 }
293
294 if (s.ok()) {
295 if (options.fill_cache) {
296 CachableEntry<Block_kData>* block_entry = &results[idx_in_batch];
297 // MaybeReadBlockAndLoadToCache will insert into the block caches if
298 // necessary. Since we're passing the serialized block contents, it
299 // will avoid looking up the block cache
300 s = MaybeReadBlockAndLoadToCache(
301 nullptr, options, handle, uncompression_dict,
302 /*for_compaction=*/false, block_entry, mget_iter->get_context,
303 /*lookup_context=*/nullptr, &serialized_block,
304 /*async_read=*/false, /*use_block_cache_for_lookup=*/true);
305
306 if (!s.ok()) {
307 statuses[idx_in_batch] = s;
308 continue;
309 }
310 // block_entry value could be null if no block cache is present, i.e
311 // BlockBasedTableOptions::no_block_cache is true and no compressed
312 // block cache is configured. In that case, fall
313 // through and set up the block explicitly
314 if (block_entry->GetValue() != nullptr) {
315 continue;
316 }
317 }
318
319 CompressionType compression_type =
320 GetBlockCompressionType(serialized_block);
321 BlockContents contents;
322 if (compression_type != kNoCompression) {
323 UncompressionContext context(compression_type);
324 UncompressionInfo info(context, uncompression_dict, compression_type);
325 s = UncompressSerializedBlock(
326 info, req.result.data() + req_offset, handle.size(), &contents,
327 footer.format_version(), rep_->ioptions, memory_allocator);
328 } else {
329 // There are two cases here:
330 // 1) caller uses the shared buffer (scratch or direct io buffer);
331 // 2) we use the requst buffer.
332 // If scratch buffer or direct io buffer is used, we ensure that
333 // all serialized blocks are copyed to the heap as single blocks. If
334 // scratch buffer is not used, we also have no combined read, so the
335 // serialized block can be used directly.
336 contents = std::move(serialized_block);
337 }
338 if (s.ok()) {
339 results[idx_in_batch].SetOwnedValue(std::make_unique<Block_kData>(
340 std::move(contents), read_amp_bytes_per_bit, ioptions.stats));
341 }
342 }
343 statuses[idx_in_batch] = s;
344 }
345
346 if (use_fs_scratch) {
347 // Free the allocated scratch buffer by fs here as read requests might have
348 // been combined into one.
349 for (FSReadRequest& req : read_reqs) {
350 if (req.fs_scratch != nullptr) {
351 req.fs_scratch.reset();
352 req.fs_scratch = nullptr;
353 }
354 }
355 }
356}

358using MultiGetRange = MultiGetContext::Range;
359DEFINE_SYNC_AND_ASYNC(void, BlockBasedTable::MultiGet)
360(const ReadOptions& read_options, const MultiGetRange* mget_range,
361 const SliceTransform* prefix_extractor, bool skip_filters) {
362 if (mget_range->empty()) {
363 // Caller should ensure non-empty (performance bug)
364 assert(false);
365 CO_RETURN; // Nothing to do
366 }
367
368 FilterBlockReader* const filter =
369 !skip_filters ? rep_->filter.get() : nullptr;
370 MultiGetRange sst_file_range(*mget_range, mget_range->begin(),
371 mget_range->end());
372
373 // First check the full filter
374 // If full filter not useful, Then go into each block
375 uint64_t tracing_mget_id = BlockCacheTraceHelper::kReservedGetId;
376 if (sst_file_range.begin()->get_context) {
377 tracing_mget_id = sst_file_range.begin()->get_context->get_tracing_get_id();
378 }
379 // TODO: need more than one lookup_context here to track individual filter
380 // and index partition hits and misses.
381 BlockCacheLookupContext metadata_lookup_context{
382 TableReaderCaller::kUserMultiGet, tracing_mget_id,
383 /*_get_from_user_specified_snapshot=*/read_options.snapshot != nullptr};
384 FullFilterKeysMayMatch(filter, &sst_file_range, prefix_extractor,
385 &metadata_lookup_context, read_options);
386
387 if (!sst_file_range.empty()) {
388 IndexBlockIter iiter_on_stack;
389 // if prefix_extractor found in block differs from options, disable
390 // BlockPrefixIndex. Only do this check when index_type is kHashSearch.
391 bool need_upper_bound_check = false;
392 if (rep_->index_type == BlockBasedTableOptions::kHashSearch) {
393 need_upper_bound_check = PrefixExtractorChanged(prefix_extractor);
394 }
395 auto iiter = NewIndexIterator(
396 read_options, need_upper_bound_check, &iiter_on_stack,
397 sst_file_range.begin()->get_context, &metadata_lookup_context);
398 std::unique_ptr<InternalIteratorBase<IndexValue>> iiter_unique_ptr;
399 if (iiter != &iiter_on_stack) {
400 iiter_unique_ptr.reset(iiter);
401 }
402
403 uint64_t prev_offset = std::numeric_limits<uint64_t>::max();
404 autovector<BlockHandle, MultiGetContext::MAX_BATCH_SIZE> block_handles;
405 std::array<CachableEntry<Block_kData>, MultiGetContext::MAX_BATCH_SIZE>
406 results;
407 std::array<Status, MultiGetContext::MAX_BATCH_SIZE> statuses;
408 // Empty data_lookup_contexts means "unused," when block cache tracing is
409 // disabled. (Limited options as element type is not default contructible.)
410 std::vector<BlockCacheLookupContext> data_lookup_contexts;
411 MultiGetContext::Mask reused_mask = 0;
412 char stack_buf[kMultiGetReadStackBufSize];
413 std::unique_ptr<char[]> block_buf;
414 if (block_cache_tracer_ && block_cache_tracer_->is_tracing_enabled()) {
415 // Awkward because BlockCacheLookupContext is not CopyAssignable
416 data_lookup_contexts.reserve(MultiGetContext::MAX_BATCH_SIZE);
417 for (size_t i = 0; i < MultiGetContext::MAX_BATCH_SIZE; ++i) {
418 data_lookup_contexts.push_back(metadata_lookup_context);
419 }
420 }
421 {
422 MultiGetRange data_block_range(sst_file_range, sst_file_range.begin(),
423 sst_file_range.end());
424 CachableEntry<UncompressionDict> uncompression_dict;
425 Status uncompression_dict_status;
426 uncompression_dict_status.PermitUncheckedError();
427 bool uncompression_dict_inited = false;
428 size_t total_len = 0;
429
430 // GetContext for any key will do, as the stats will be aggregated
431 // anyway
432 GetContext* get_context = sst_file_range.begin()->get_context;
433
434 {
435 using BCI = BlockCacheInterface<Block_kData>;
436 BCI block_cache{rep_->table_options.block_cache.get()};
437 std::array<BCI::TypedAsyncLookupHandle, MultiGetContext::MAX_BATCH_SIZE>
438 async_handles;
439 BlockCreateContext create_ctx = rep_->create_context;
440 std::array<CacheKey, MultiGetContext::MAX_BATCH_SIZE> cache_keys;
441 size_t cache_lookup_count = 0;
442
443 for (auto miter = data_block_range.begin();
444 miter != data_block_range.end(); ++miter) {
445 const Slice& key = miter->ikey;
446 iiter->Seek(miter->ikey);
447
448 IndexValue v;
449 if (iiter->Valid()) {
450 v = iiter->value();
451 }
452 if (!iiter->Valid() ||
453 (!v.first_internal_key.empty() && !skip_filters &&
454 UserComparatorWrapper(
455 rep_->internal_comparator.user_comparator())
456 .CompareWithoutTimestamp(
457 ExtractUserKey(key),
458 ExtractUserKey(v.first_internal_key)) < 0)) {
459 // The requested key falls between highest key in previous block and
460 // lowest key in current block.
461 if (!iiter->status().IsNotFound()) {
462 *(miter->s) = iiter->status();
463 }
464 data_block_range.SkipKey(miter);
465 sst_file_range.SkipKey(miter);
466 continue;
467 }
468
469 if (!uncompression_dict_inited && rep_->uncompression_dict_reader) {
470 uncompression_dict_status =
471 rep_->uncompression_dict_reader
472 ->GetOrReadUncompressionDictionary(
473 nullptr /* prefetch_buffer */, read_options,
474 get_context, &metadata_lookup_context,
475 &uncompression_dict);
476 uncompression_dict_inited = true;
477 }
478
479 if (!uncompression_dict_status.ok()) {
480 assert(!uncompression_dict_status.IsNotFound());
481 *(miter->s) = uncompression_dict_status;
482 data_block_range.SkipKey(miter);
483 sst_file_range.SkipKey(miter);
484 continue;
485 }
486 create_ctx.dict = uncompression_dict.GetValue()
487 ? uncompression_dict.GetValue()
488 : &UncompressionDict::GetEmptyDict();
489
490 if (v.handle.offset() == prev_offset) {
491 // This key can reuse the previous block (later on).
492 // Mark previous as "reused"
493 reused_mask |= MultiGetContext::Mask{1}
494 << (block_handles.size() - 1);
495 // Use null handle to indicate this one reuses same block as
496 // previous.
497 block_handles.emplace_back(BlockHandle::NullBlockHandle());
498 continue;
499 }
500 prev_offset = v.handle.offset();
501 block_handles.emplace_back(v.handle);
502
503 if (block_cache) {
504 // Lookup the cache for the given data block referenced by an index
505 // iterator value (i.e BlockHandle). If it exists in the cache,
506 // initialize block to the contents of the data block.
507
508 // An async version of MaybeReadBlockAndLoadToCache /
509 // GetDataBlockFromCache
510 BCI::TypedAsyncLookupHandle& async_handle =
511 async_handles[cache_lookup_count];
512 cache_keys[cache_lookup_count] =
513 GetCacheKey(rep_->base_cache_key, v.handle);
514 async_handle.key = cache_keys[cache_lookup_count].AsSlice();
515 // NB: StartAsyncLookupFull populates async_handle.helper
516 async_handle.create_context = &create_ctx;
517 async_handle.priority = GetCachePriority<Block_kData>();
518 async_handle.stats = rep_->ioptions.statistics.get();
519
520 block_cache.StartAsyncLookupFull(
521 async_handle, rep_->ioptions.lowest_used_cache_tier);
522 ++cache_lookup_count;
523 // TODO: stats?
524 }
525 }
526
527 if (block_cache) {
528 block_cache.get()->WaitAll(&async_handles[0], cache_lookup_count);
529 }
530 size_t lookup_idx = 0;
531 for (size_t i = 0; i < block_handles.size(); ++i) {
532 // If this block was a success or failure or not needed because
533 // the corresponding key is in the same block as a prior key, skip
534 if (block_handles[i] == BlockHandle::NullBlockHandle()) {
535 continue;
536 }
537 if (!block_cache) {
538 total_len += BlockSizeWithTrailer(block_handles[i]);
539 } else {
540 BCI::TypedHandle* h = async_handles[lookup_idx].Result();
541 if (h) {
542 // Cache hit
543 results[i].SetCachedValue(block_cache.Value(h), block_cache.get(),
544 h);
545 // Don't need to fetch
546 block_handles[i] = BlockHandle::NullBlockHandle();
547 UpdateCacheHitMetrics(BlockType::kData, get_context,
548 block_cache.get()->GetUsage(h));
549 } else {
550 // Cache miss
551 total_len += BlockSizeWithTrailer(block_handles[i]);
552 UpdateCacheMissMetrics(BlockType::kData, get_context);
553 }
554 if (!data_lookup_contexts.empty()) {
555 // Populate cache key before it's discarded
556 data_lookup_contexts[i].block_key =
557 async_handles[lookup_idx].key.ToString();
558 }
559 ++lookup_idx;
560 }
561 }
562 assert(lookup_idx == cache_lookup_count);
563 }
564
565 if (total_len) {
566 char* scratch = nullptr;
567 bool use_fs_scratch = false;
568 const UncompressionDict& dict = uncompression_dict.GetValue()
569 ? *uncompression_dict.GetValue()
570 : UncompressionDict::GetEmptyDict();
571 assert(uncompression_dict_inited || !rep_->uncompression_dict_reader);
572 assert(uncompression_dict_status.ok());
573
574 if (!rep_->file->use_direct_io()) {
575 if (CheckFSFeatureSupport(rep_->ioptions.fs.get(),
576 FSSupportedOps::kFSBuffer)) {
577 use_fs_scratch = true;
578 }
579 }
580
581 // If using direct IO, then scratch is not used, so keep it nullptr.
582 // If the blocks need to be uncompressed and we don't need the
583 // compressed blocks, then we can use a contiguous block of
584 // memory to read in all the blocks as it will be temporary
585 // storage
586 // 1. If blocks are compressed and compressed block cache is there,
587 // alloc heap bufs
588 // 2. If blocks are uncompressed, alloc heap bufs
589 // 3. If blocks are compressed and no compressed block cache, use
590 // stack buf
591 if (!use_fs_scratch && !rep_->file->use_direct_io() &&
592 rep_->blocks_maybe_compressed) {
593 if (total_len <= kMultiGetReadStackBufSize) {
594 scratch = stack_buf;
595 } else {
596 scratch = new char[total_len];
597 block_buf.reset(scratch);
598 }
599 }
600 CO_AWAIT(RetrieveMultipleBlocks)
601 (read_options, &data_block_range, &block_handles, &statuses[0],
602 &results[0], scratch, dict, use_fs_scratch);
603 if (get_context) {
604 ++(get_context->get_context_stats_.num_sst_read);
605 }
606 }
607 }
608
609 DataBlockIter first_biter;
610 DataBlockIter next_biter;
611 size_t idx_in_batch = 0;
612 SharedCleanablePtr shared_cleanable;
613 for (auto miter = sst_file_range.begin(); miter != sst_file_range.end();
614 ++miter) {
615 Status s;
616 GetContext* get_context = miter->get_context;
617 const Slice& key = miter->ikey;
618 bool matched = false; // if such user key matched a key in SST
619 bool done = false;
620 bool first_block = true;
621 do {
622 DataBlockIter* biter = nullptr;
623 uint64_t referenced_data_size = 0;
624 Block_kData* parsed_block_value = nullptr;
625 bool reusing_prev_block;
626 bool later_reused;
627 bool does_referenced_key_exist = false;
628 bool handle_present = false;
629 BlockCacheLookupContext* lookup_data_block_context =
630 data_lookup_contexts.empty() ? nullptr
631 : &data_lookup_contexts[idx_in_batch];
632 if (first_block) {
633 handle_present = !block_handles[idx_in_batch].IsNull();
634 parsed_block_value = results[idx_in_batch].GetValue();
635 if (handle_present || parsed_block_value) {
636 first_biter.Invalidate(Status::OK());
637 NewDataBlockIterator<DataBlockIter>(
638 read_options, results[idx_in_batch].As<Block>(), &first_biter,
639 statuses[idx_in_batch]);
640 reusing_prev_block = false;
641 } else {
642 // If handle is null and result is empty, then the status is never
643 // set, which should be the initial value: ok().
644 assert(statuses[idx_in_batch].ok());
645 reusing_prev_block = true;
646 }
647 biter = &first_biter;
648 later_reused =
649 (reused_mask & (MultiGetContext::Mask{1} << idx_in_batch)) != 0;
650 idx_in_batch++;
651 } else {
652 IndexValue v = iiter->value();
653 if (!v.first_internal_key.empty() && !skip_filters &&
654 UserComparatorWrapper(rep_->internal_comparator.user_comparator())
655 .CompareWithoutTimestamp(
656 ExtractUserKey(key),
657 ExtractUserKey(v.first_internal_key)) < 0) {
658 // The requested key falls between highest key in previous block and
659 // lowest key in current block.
660 break;
661 }
662
663 next_biter.Invalidate(Status::OK());
664 Status tmp_s;
665 NewDataBlockIterator<DataBlockIter>(
666 read_options, iiter->value().handle, &next_biter,
667 BlockType::kData, get_context, lookup_data_block_context,
668 /* prefetch_buffer= */ nullptr, /* for_compaction = */ false,
669 /*async_read = */ false, tmp_s,
670 /* use_block_cache_for_lookup = */ true);
671 biter = &next_biter;
672 reusing_prev_block = false;
673 later_reused = false;
674 }
675
676 if (read_options.read_tier == kBlockCacheTier &&
677 biter->status().IsIncomplete()) {
678 // couldn't get block from block_cache
679 // Update Saver.state to Found because we are only looking for
680 // whether we can guarantee the key is not there with kBlockCacheTier
681 get_context->MarkKeyMayExist();
682 break;
683 }
684 if (!biter->status().ok()) {
685 s = biter->status();
686 break;
687 }
688
689 // Reusing blocks complicates pinning/Cleanable, because the cache
690 // entry referenced by biter can only be released once all returned
691 // pinned values are released. This code previously did an extra
692 // block_cache Ref for each reuse, but that unnecessarily increases
693 // block cache contention. Instead we can use a variant of shared_ptr
694 // to release in block cache only once.
695 //
696 // Although the biter loop below might SaveValue multiple times for
697 // merges, just one value_pinner suffices, as MultiGet will merge
698 // the operands before returning to the API user.
699 Cleanable* value_pinner;
700 if (biter->IsValuePinned()) {
701 if (reusing_prev_block) {
702 // Note that we don't yet know if the MultiGet results will need
703 // to pin this block, so we might wrap a block for sharing and
704 // still end up with 1 (or 0) pinning ref. Not ideal but OK.
705 //
706 // Here we avoid adding redundant cleanups if we didn't end up
707 // delegating the cleanup from last time around.
708 if (!biter->HasCleanups()) {
709 assert(shared_cleanable.get());
710 if (later_reused) {
711 shared_cleanable.RegisterCopyWith(biter);
712 } else {
713 shared_cleanable.MoveAsCleanupTo(biter);
714 }
715 }
716 } else if (later_reused) {
717 assert(biter->HasCleanups());
718 // Make the existing cleanups on `biter` sharable:
719 shared_cleanable.Allocate();
720 // Move existing `biter` cleanup(s) to `shared_cleanable`
721 biter->DelegateCleanupsTo(&*shared_cleanable);
722 // Reference `shared_cleanable` as new cleanup for `biter`
723 shared_cleanable.RegisterCopyWith(biter);
724 }
725 assert(biter->HasCleanups());
726 value_pinner = biter;
727 } else {
728 value_pinner = nullptr;
729 }
730
731 bool may_exist = biter->SeekForGet(key);
732 if (!may_exist) {
733 // HashSeek cannot find the key this block and the the iter is not
734 // the end of the block, i.e. cannot be in the following blocks
735 // either. In this case, the seek_key cannot be found, so we break
736 // from the top level for-loop.
737 break;
738 }
739
740 // Call the *saver function on each entry/block until it returns false
741 for (; biter->status().ok() && biter->Valid(); biter->Next()) {
742 ParsedInternalKey parsed_key;
743 Status pik_status = ParseInternalKey(
744 biter->key(), &parsed_key, false /* log_err_key */); // TODO
745 if (!pik_status.ok()) {
746 s = pik_status;
747 break;
748 }
749 Status read_status;
750 bool ret = get_context->SaveValue(
751 parsed_key, biter->value(), &matched, &read_status,
752 value_pinner ? value_pinner : nullptr);
753 if (!read_status.ok()) {
754 s = read_status;
755 break;
756 }
757 if (!ret) {
758 if (get_context->State() == GetContext::GetState::kFound) {
759 does_referenced_key_exist = true;
760 referenced_data_size =
761 biter->key().size() + biter->value().size();
762 }
763 done = true;
764 break;
765 }
766 }
767 // Write the block cache access.
768 // XXX: There appear to be 'break' statements above that bypass this
769 // writing of the block cache trace record
770 if (lookup_data_block_context && !reusing_prev_block && first_block) {
771 Slice referenced_key;
772 if (does_referenced_key_exist) {
773 referenced_key = biter->key();
774 } else {
775 referenced_key = key;
776 }
777
778 // block_key is self-assigned here (previously assigned from
779 // cache_keys / async_handles, now out of scope)
780 SaveLookupContextOrTraceRecord(lookup_data_block_context->block_key,
781 /*is_cache_hit=*/!handle_present,
782 read_options, parsed_block_value,
783 lookup_data_block_context);
784 FinishTraceRecord(
785 *lookup_data_block_context, lookup_data_block_context->block_key,
786 referenced_key, does_referenced_key_exist, referenced_data_size);
787 }
788 if (s.ok()) {
789 s = biter->status();
790 }
791 if (done || !s.ok()) {
792 // Avoid the extra Next which is expensive in two-level indexes
793 break;
794 }
795 if (first_block) {
796 iiter->Seek(key);
797 if (!iiter->Valid()) {
798 break;
799 }
800 }
801 first_block = false;
802 iiter->Next();
803 } while (iiter->Valid());
804
805 if (matched && filter != nullptr) {
806 if (rep_->whole_key_filtering) {
807 RecordTick(rep_->ioptions.stats, BLOOM_FILTER_FULL_TRUE_POSITIVE);
808 } else {
809 RecordTick(rep_->ioptions.stats, BLOOM_FILTER_PREFIX_TRUE_POSITIVE);
810 }
811 // Includes prefix stats
812 PERF_COUNTER_BY_LEVEL_ADD(bloom_filter_full_true_positive, 1,
813 rep_->level);
814 }
815 if (s.ok() && !iiter->status().IsNotFound()) {
816 s = iiter->status();
817 }
818 *(miter->s) = s;
819 }
820#ifdef ROCKSDB_ASSERT_STATUS_CHECKED
821 // Not sure why we need to do it. Should investigate more.
822 for (auto& st : statuses) {
823 st.PermitUncheckedError();
824 }
825#endif // ROCKSDB_ASSERT_STATUS_CHECKED
826 }
827}

829#endif
