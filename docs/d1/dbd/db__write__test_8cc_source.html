53TEST_P(DBWriteTest, WriteStallRemoveNoSlowdownWrite) {
54 Options options = GetOptions();
55 options.level0_stop_writes_trigger = options.level0_slowdown_writes_trigger =
56 4;
57 std::vector<port::Thread> threads;
58 std::atomic<int> thread_num(0);
59 port::Mutex mutex;
60 port::CondVar cv(&mutex);
61 // Guarded by mutex
62 int writers = 0;
63
64 Reopen(options);
65
66 std::function<void()> write_slowdown_func = [&]() {
67 int a = thread_num.fetch_add(1);
68 std::string key = "foo" + std::to_string(a);
69 WriteOptions wo;
70 wo.no_slowdown = false;
71 ASSERT_OK(dbfull()->Put(wo, key, "bar"));
72 };
73 std::function<void()> write_no_slowdown_func = [&]() {
74 int a = thread_num.fetch_add(1);
75 std::string key = "foo" + std::to_string(a);
76 WriteOptions wo;
77 wo.no_slowdown = true;
78 Status s = dbfull()->Put(wo, key, "bar");
79 ASSERT_TRUE(s.ok() || s.IsIncomplete());
80 };
81 std::function<void(void*)> unblock_main_thread_func = [&](void*) {
82 mutex.Lock();
83 ++writers;
84 cv.SignalAll();
85 mutex.Unlock();
86 };
87
88 // Create 3 L0 files and schedule 4th without waiting
89 ASSERT_OK(Put("foo" + std::to_string(thread_num.fetch_add(1)), "bar"));
90 ASSERT_OK(Flush());
91 ASSERT_OK(Put("foo" + std::to_string(thread_num.fetch_add(1)), "bar"));
92 ASSERT_OK(Flush());
93 ASSERT_OK(Put("foo" + std::to_string(thread_num.fetch_add(1)), "bar"));
94 ASSERT_OK(Flush());
95 ASSERT_OK(Put("foo" + std::to_string(thread_num.fetch_add(1)), "bar"));
96
97 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
98 "WriteThread::JoinBatchGroup:Start", unblock_main_thread_func);
99 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->LoadDependency(
100 {{"DBWriteTest::WriteStallRemoveNoSlowdownWrite:1",
101 "DBImpl::BackgroundCallFlush:start"},
102 {"DBWriteTest::WriteStallRemoveNoSlowdownWrite:2",
103 "DBImplWrite::PipelinedWriteImpl:AfterJoinBatchGroup"},
104 // Make compaction start wait for the write stall to be detected and
105 // implemented by a write group leader
106 {"DBWriteTest::WriteStallRemoveNoSlowdownWrite:3",
107 "BackgroundCallCompaction:0"}});
108 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
109
110 // Schedule creation of 4th L0 file without waiting. This will seal the
111 // memtable and then wait for a sync point before writing the file. We need
112 // to do it this way because SwitchMemtable() needs to enter the
113 // write_thread
114 FlushOptions fopt;
115 fopt.wait = false;
116 ASSERT_OK(dbfull()->Flush(fopt));
117
118 // Create a mix of slowdown/no_slowdown write threads
119 mutex.Lock();
120 // First leader
121 threads.emplace_back(write_slowdown_func);
122 while (writers != 1) {
123 cv.Wait();
124 }
125
126 // Second leader. Will stall writes
127 // Build a writers list with no slowdown in the middle:
128 // +-------------+
129 // | slowdown +<----+ newest
130 // +--+----------+
131 // |
132 // v
133 // +--+----------+
134 // | no slowdown |
135 // +--+----------+
136 // |
137 // v
138 // +--+----------+
139 // | slowdown +
140 // +-------------+
141 threads.emplace_back(write_slowdown_func);
142 while (writers != 2) {
143 cv.Wait();
144 }
145 threads.emplace_back(write_no_slowdown_func);
146 while (writers != 3) {
147 cv.Wait();
148 }
149 threads.emplace_back(write_slowdown_func);
150 while (writers != 4) {
151 cv.Wait();
152 }
153
154 mutex.Unlock();
155
156 TEST_SYNC_POINT("DBWriteTest::WriteStallRemoveNoSlowdownWrite:1");
157 ASSERT_OK(dbfull()->TEST_WaitForFlushMemTable(nullptr));
158 // This would have triggered a write stall. Unblock the write group leader
159 TEST_SYNC_POINT("DBWriteTest::WriteStallRemoveNoSlowdownWrite:2");
160 // The leader is going to create missing newer links. When the leader
161 // finishes, the next leader is going to delay writes and fail writers with
162 // no_slowdown
163
164 TEST_SYNC_POINT("DBWriteTest::WriteStallRemoveNoSlowdownWrite:3");
165 for (auto& t : threads) {
166 t.join();
167 }
168
169 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
170 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->ClearAllCallBacks();
171}

372TEST_F(DBWriteTestUnparameterized, PipelinedWriteRace) {
373 // This test was written to trigger a race in ExitAsBatchGroupLeader in case
374 // enable_pipelined_write_ was true.
375 // Writers for which ShouldWriteToMemtable() evaluates to false are removed
376 // from the write_group via CompleteFollower/ CompleteLeader. Writers in the
377 // middle of the group are fully unlinked, but if that writers is the
378 // last_writer, then we did not update the predecessor's link_older, i.e.,
379 // this writer was still reachable via newest_writer_.
380 //
381 // But the problem was, that CompleteFollower already wakes up the thread
382 // owning that writer before the writer has been removed. This resulted in a
383 // race - if the leader thread was fast enough, then everything was fine.
384 // However, if the woken up thread finished the current write operation and
385 // then performed yet another write, then a new writer instance was added
386 // to newest_writer_. It is possible that the new writer is located on the
387 // same address on stack, and if this happened, then we had a problem,
388 // because the old code tried to find the last_writer in the list to unlink
389 // it, which in this case produced a cycle in the list.
390 // Whether two invocations of PipelinedWriteImpl() by the same thread actually
391 // allocate the writer on the same address depends on the OS and/or compiler,
392 // so it is rather hard to create a deterministic test for this.
393
394 Options options = GetDefaultOptions();
395 options.create_if_missing = true;
396 options.enable_pipelined_write = true;
397 std::vector<port::Thread> threads;
398
399 std::atomic<int> write_counter{0};
400 std::atomic<int> active_writers{0};
401 std::atomic<bool> second_write_starting{false};
402 std::atomic<bool> second_write_in_progress{false};
403 std::atomic<WriteThread::Writer*> leader{nullptr};
404 std::atomic<bool> finished_WAL_write{false};
405
406 DestroyAndReopen(options);
407
408 auto write_one_doc = [&]() {
409 int a = write_counter.fetch_add(1);
410 std::string key = "foo" + std::to_string(a);
411 WriteOptions wo;
412 ASSERT_OK(dbfull()->Put(wo, key, "bar"));
413 --active_writers;
414 };
415
416 auto write_two_docs = [&]() {
417 write_one_doc();
418 second_write_starting = true;
419 write_one_doc();
420 };
421
422 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
423 "WriteThread::JoinBatchGroup:Wait", [&](void* arg) {
424 if (second_write_starting.load()) {
425 second_write_in_progress = true;
426 return;
427 }
428 auto* w = static_cast<WriteThread::Writer*>(arg);
429 if (w->state == WriteThread::STATE_GROUP_LEADER) {
430 active_writers++;
431 if (leader.load() == nullptr) {
432 leader.store(w);
433 while (active_writers.load() < 2) {
434 // wait for another thread to join the write_group
435 }
436 }
437 } else {
438 // we disable the memtable for all followers so that they they are
439 // removed from the write_group before enqueuing it for the memtable
440 // write
441 w->disable_memtable = true;
442 active_writers++;
443 }
444 });
445
446 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
447 "WriteThread::ExitAsBatchGroupLeader:Start", [&](void* arg) {
448 auto* wg = static_cast<WriteThread::WriteGroup*>(arg);
449 if (wg->leader == leader && !finished_WAL_write) {
450 finished_WAL_write = true;
451 while (active_writers.load() < 3) {
452 // wait for the new writer to be enqueued
453 }
454 }
455 });
456
457 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
458 "WriteThread::ExitAsBatchGroupLeader:AfterCompleteWriters",
459 [&](void* arg) {
460 auto* wg = static_cast<WriteThread::WriteGroup*>(arg);
461 if (wg->leader == leader) {
462 while (!second_write_in_progress.load()) {
463 // wait for the old follower thread to start the next write
464 }
465 }
466 });
467
468 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
469
470 // start leader + one follower
471 threads.emplace_back(write_one_doc);
472 while (leader.load() == nullptr) {
473 // wait for leader
474 }
475
476 // we perform two writes in the follower, so that for the second write
477 // the thread reinserts a Writer with the same address
478 threads.emplace_back(write_two_docs);
479
480 // wait for the leader to enter ExitAsBatchGroupLeader
481 while (!finished_WAL_write.load()) {
482 // wait for write_group to have finished the WAL writes
483 }
484
485 // start another writer thread to be enqueued before the leader can
486 // complete the writers from its write_group
487 threads.emplace_back(write_one_doc);
488
489 for (auto& t : threads) {
490 t.join();
491 }
492 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
493 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->ClearAllCallBacks();
494}

725TEST_P(DBWriteTest, LockWALConcurrentRecursive) {
726 // This is a micro-stress test of LockWAL and concurrency handling.
727 // It is considered the most convenient way to balance functional
728 // coverage and reproducibility (vs. the two extremes of (a) unit tests
729 // tailored to specific interleavings and (b) db_stress)
730 Options options = GetOptions();
731 Reopen(options);
732 ASSERT_OK(Put("k1", "k1_orig"));
733 ASSERT_OK(db_->LockWAL()); // 0 -> 1
734 auto frozen_seqno = db_->GetLatestSequenceNumber();
735
736 std::string ingest_file = dbname_ + "/external.sst";
737 {
738 SstFileWriter sst_file_writer(EnvOptions(), options);
739 ASSERT_OK(sst_file_writer.Open(ingest_file));
740 ASSERT_OK(sst_file_writer.Put("k2", "k2_val"));
741 ExternalSstFileInfo external_info;
742 ASSERT_OK(sst_file_writer.Finish(&external_info));
743 }
744 AcqRelAtomic<bool> parallel_ingest_completed{false};
745 port::Thread parallel_ingest{[&]() {
746 IngestExternalFileOptions ingest_opts;
747 ingest_opts.move_files = true; // faster than copy
748 // Shouldn't finish until WAL unlocked
749 ASSERT_OK(db_->IngestExternalFile({ingest_file}, ingest_opts));
750 parallel_ingest_completed.Store(true);
751 }};
752
753 AcqRelAtomic<bool> flush_completed{false};
754 port::Thread parallel_flush{[&]() {
755 FlushOptions flush_opts;
756 // NB: Flush with wait=false case is tested above in LockWALInEffect
757 flush_opts.wait = true;
758 // allow_write_stall = true blocks in fewer cases
759 flush_opts.allow_write_stall = true;
760 // Shouldn't finish until WAL unlocked
761 ASSERT_OK(db_->Flush(flush_opts));
762 flush_completed.Store(true);
763 }};
764
765 AcqRelAtomic<bool> parallel_put_completed{false};
766 port::Thread parallel_put{[&]() {
767 // This can make certain failure scenarios more likely:
768 // sleep(1);
769 // Shouldn't finish until WAL unlocked
770 ASSERT_OK(Put("k1", "k1_mod"));
771 parallel_put_completed.Store(true);
772 }};
773
774 ASSERT_OK(db_->LockWAL()); // 1 -> 2
775 // Read-only ops are OK
776 ASSERT_EQ(Get("k1"), "k1_orig");
777 {
778 std::vector<LiveFileStorageInfo> files;
779 LiveFilesStorageInfoOptions lf_opts;
780 // A DB flush could deadlock
781 lf_opts.wal_size_for_flush = UINT64_MAX;
782 ASSERT_OK(db_->GetLiveFilesStorageInfo({lf_opts}, &files));
783 }
784
785 port::Thread parallel_lock_wal{[&]() {
786 ASSERT_OK(db_->LockWAL()); // 2 -> 3 or 1 -> 2
787 }};
788
789 ASSERT_OK(db_->UnlockWAL()); // 2 -> 1 or 3 -> 2
790 // Give parallel_put an extra chance to jump in case of bug
791 std::this_thread::yield();
792 parallel_lock_wal.join();
793 ASSERT_FALSE(parallel_put_completed.Load());
794 ASSERT_FALSE(parallel_ingest_completed.Load());
795 ASSERT_FALSE(flush_completed.Load());
796
797 // Should now have 2 outstanding LockWAL
798 ASSERT_EQ(Get("k1"), "k1_orig");
799
800 ASSERT_OK(db_->UnlockWAL()); // 2 -> 1
801
802 ASSERT_FALSE(parallel_put_completed.Load());
803 ASSERT_FALSE(parallel_ingest_completed.Load());
804 ASSERT_FALSE(flush_completed.Load());
805
806 ASSERT_EQ(Get("k1"), "k1_orig");
807 ASSERT_EQ(Get("k2"), "NOT_FOUND");
808 ASSERT_EQ(frozen_seqno, db_->GetLatestSequenceNumber());
809
810 // Ensure final Unlock is concurrency safe and extra Unlock is safe but
811 // non-OK
812 std::atomic<int> unlock_ok{0};
813 port::Thread parallel_stuff{[&]() {
814 if (db_->UnlockWAL().ok()) {
815 unlock_ok++;
816 }
817 ASSERT_OK(db_->LockWAL());
818 if (db_->UnlockWAL().ok()) {
819 unlock_ok++;
820 }
821 }};
822
823 if (db_->UnlockWAL().ok()) {
824 unlock_ok++;
825 }
826 parallel_stuff.join();
827
828 // There was one extra unlock, so just one non-ok
829 ASSERT_EQ(unlock_ok.load(), 2);
830
831 // Write can proceed
832 parallel_put.join();
833 ASSERT_TRUE(parallel_put_completed.Load());
834 ASSERT_EQ(Get("k1"), "k1_mod");
835 parallel_ingest.join();
836 ASSERT_TRUE(parallel_ingest_completed.Load());
837 ASSERT_EQ(Get("k2"), "k2_val");
838 parallel_flush.join();
839 ASSERT_TRUE(flush_completed.Load());
840 // And new writes
841 ASSERT_OK(Put("k3", "val"));
842 ASSERT_EQ(Get("k3"), "val");
843}

889 uint64_t log_num, int record_num) {
890 std::shared_ptr<FileSystem> fs = env->GetFileSystem();
891 std::unique_ptr<SequentialFileReader> file_reader;

909 reader.reset();
910 {
911 std::unique_ptr<FSRandomRWFile> file;
912 status = fs->NewRandomRWFile(log_path, FileOptions(), &file, nullptr);
913 ASSERT_EQ(status, IOStatus::OK());
914 uint32_t bad_lognum = 0xff;
915 ASSERT_EQ(file->Write(
916 rec_start + 7,
917 Slice(reinterpret_cast<char*>(&bad_lognum), sizeof(uint32_t)),
918 IOOptions(), nullptr),
919 IOStatus::OK());
920 ASSERT_OK(file->Close(IOOptions(), nullptr));
921 file.reset();
922 }
