17const int64_t kUsPerSec = 1000000;
18const int64_t kDummyBytesPerUs = 1024;

21// From bytes to read/write, calculate service time needed by an HDD.
22// This is used to simulate latency from HDD.
23int CalculateServeTimeUs(size_t bytes) {
24 return 12200 + static_cast<int>(static_cast<double>(bytes) * 0.005215);
25}

27// There is a bug in rater limiter that would crash with small requests
28// Hack to get it around.
29void RateLimiterRequest(RateLimiter* rater_limiter, int64_t amount) {
30 int64_t left = amount * kDummyBytesPerUs;
31 const int64_t kMaxToRequest = kDummyBytesPerUs * kUsPerSec / 1024;
32 while (left > 0) {
33 int64_t to_request = std::min(kMaxToRequest, left);
34 rater_limiter->Request(to_request, Env::IOPriority::IO_LOW, nullptr);
35 left -= to_request;
36 }
37}

40// The metadata file format: each line is a full filename of a file which is
41// warm
42SimulatedHybridFileSystem::SimulatedHybridFileSystem(
43 const std::shared_ptr<FileSystem>& base,
44 const std::string& metadata_file_name, int throughput_multiplier,
45 bool is_full_fs_warm)
46 : FileSystemWrapper(base),
47 // Limit to 100 requests per second.
48 rate_limiter_(NewGenericRateLimiter(
49 int64_t{throughput_multiplier} * kDummyBytesPerUs *
50 kUsPerSec /* rate_bytes_per_sec */,
51 1000 /* refill_period_us */)),
52 metadata_file_name_(metadata_file_name),
53 name_("SimulatedHybridFileSystem: " + std::string(target()->Name())),
54 is_full_fs_warm_(is_full_fs_warm) {
55 IOStatus s = base->FileExists(metadata_file_name, IOOptions(), nullptr);
56 if (s.IsNotFound()) {
57 return;
58 }
59 std::string metadata;
60 s = ReadFileToString(base.get(), metadata_file_name, &metadata);
61 if (!s.ok()) {
62 fprintf(stderr, "Error reading from file %s: %s",
63 metadata_file_name.c_str(), s.ToString().c_str());
64 // Exit rather than assert as this file system is built to run with
65 // benchmarks, which usually run on release mode.
66 std::exit(1);
67 }
68 std::istringstream input;
69 input.str(metadata);
70 std::string line;
71 while (std::getline(input, line)) {
72 fprintf(stderr, "Warm file %s\n", line.c_str());
73 warm_file_set_.insert(line);
74 }
75}

77// Need to write out the metadata file to file. See comment of
78// SimulatedHybridFileSystem::SimulatedHybridFileSystem() for format of the
79// file.
80SimulatedHybridFileSystem::~SimulatedHybridFileSystem() {
81 if (metadata_file_name_.empty()) {
82 return;
83 }
84 std::string metadata;
85 for (const auto& f : warm_file_set_) {
86 metadata += f;
87 metadata += "\n";
88 }
89 IOOptions opts;
90 IOStatus s =
91 WriteStringToFile(target(), metadata, metadata_file_name_, true, opts);
92 if (!s.ok()) {
93 fprintf(stderr, "Error writing to file %s: %s", metadata_file_name_.c_str(),
94 s.ToString().c_str());
95 }
96}

98IOStatus SimulatedHybridFileSystem::NewRandomAccessFile(
99 const std::string& fname, const FileOptions& file_opts,
100 std::unique_ptr<FSRandomAccessFile>* result, IODebugContext* dbg) {
101 Temperature temperature = Temperature::kUnknown;
102 if (is_full_fs_warm_) {
103 temperature = Temperature::kWarm;
104 } else {
105 const std::lock_guard<std::mutex> lock(mutex_);
106 if (warm_file_set_.find(fname) != warm_file_set_.end()) {
107 temperature = Temperature::kWarm;
108 }
109 assert(temperature == file_opts.temperature);
110 }
111 IOStatus s = target()->NewRandomAccessFile(fname, file_opts, result, dbg);
112 result->reset(
113 new SimulatedHybridRaf(std::move(*result), rate_limiter_, temperature));
114 return s;
115}

117IOStatus SimulatedHybridFileSystem::NewWritableFile(
118 const std::string& fname, const FileOptions& file_opts,
119 std::unique_ptr<FSWritableFile>* result, IODebugContext* dbg) {
120 if (file_opts.temperature == Temperature::kWarm) {
121 const std::lock_guard<std::mutex> lock(mutex_);
122 warm_file_set_.insert(fname);
123 }
124
125 IOStatus s = target()->NewWritableFile(fname, file_opts, result, dbg);
126 if (file_opts.temperature == Temperature::kWarm || is_full_fs_warm_) {
127 result->reset(new SimulatedWritableFile(std::move(*result), rate_limiter_));
128 }
129 return s;
130}

171void SimulatedHybridRaf::SimulateIOWait(int64_t bytes) const {
172 int serve_time = CalculateServeTimeUs(bytes);
173 {
174 StopWatchNano stop_watch(Env::Default()->GetSystemClock().get(),
175 /*auto_start=*/true);
176 RateLimiterRequest(rate_limiter_.get(), serve_time);
177 int time_passed_us = static_cast<int>(stop_watch.ElapsedNanos() / 1000);
178 if (time_passed_us < serve_time) {
179 Env::Default()->SleepForMicroseconds(serve_time - time_passed_us);
180 }
181 }
182}

184void SimulatedWritableFile::SimulateIOWait(int64_t bytes) const {
185 int serve_time = CalculateServeTimeUs(bytes);
186 Env::Default()->SleepForMicroseconds(serve_time);
187 RateLimiterRequest(rate_limiter_.get(), serve_time);
188}
