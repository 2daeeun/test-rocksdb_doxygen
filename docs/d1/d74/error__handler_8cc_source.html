230void ErrorHandler::CancelErrorRecovery() {
231 db_mutex_->AssertHeld();
232
233 // We'll release the lock before calling sfm, so make sure no new
234 // recovery gets scheduled at that point
235 auto_recovery_ = false;
236 SstFileManagerImpl* sfm =
237 static_cast<SstFileManagerImpl*>(db_options_.sst_file_manager.get());
238 if (sfm) {
239 // This may or may not cancel a pending recovery
240 db_mutex_->Unlock();
241 bool cancelled = sfm->CancelErrorRecovery(this);
242 db_mutex_->Lock();
243 if (cancelled) {
244 recovery_in_prog_ = false;
245 }
246 }
247
248 // If auto recovery is also runing to resume from the retryable error,
249 // we should wait and end the auto recovery.
250 EndAutoRecovery();
251}

253// This is the main function for looking at an error during a background
254// operation and deciding the severity, and error recovery strategy. The high
255// level algorithm is as follows -
256// 1. Classify the severity of the error based on the ErrorSeverityMap,
257// DefaultErrorSeverityMap and DefaultReasonMap defined earlier
258// 2. Call a Status code specific override function to adjust the severity
259// if needed. The reason for this is our ability to recover may depend on
260// the exact options enabled in DBOptions
261// 3. Determine if auto recovery is possible. A listener notification callback
262// is called, which can disable the auto recovery even if we decide its
263// feasible
264// 4. For Status::NoSpace() errors, rely on SstFileManagerImpl to control
265// the actual recovery. If no sst file manager is specified in DBOptions,
266// a default one is allocated during DB::Open(), so there will always be
267// one.
268// This can also get called as part of a recovery operation. In that case, we
269// also track the error separately in recovery_error_ so we can tell in the
270// end whether recovery succeeded or not
271void ErrorHandler::HandleKnownErrors(const Status& bg_err,
272 BackgroundErrorReason reason) {
273 db_mutex_->AssertHeld();
274 if (bg_err.ok()) {
275 return;
276 }
277
278 ROCKS_LOG_INFO(db_options_.info_log,
279 "ErrorHandler: Set regular background error\n");
280
281 bool paranoid = db_options_.paranoid_checks;
282 Status::Severity sev = Status::Severity::kFatalError;
283 Status new_bg_err;
284 DBRecoverContext context;
285 bool found = false;
286
287 {
288 auto entry = ErrorSeverityMap.find(
289 std::make_tuple(reason, bg_err.code(), bg_err.subcode(), paranoid));
290 if (entry != ErrorSeverityMap.end()) {
291 sev = entry->second;
292 found = true;
293 }
294 }
295
296 if (!found) {
297 auto entry = DefaultErrorSeverityMap.find(
298 std::make_tuple(reason, bg_err.code(), paranoid));
299 if (entry != DefaultErrorSeverityMap.end()) {
300 sev = entry->second;
301 found = true;
302 }
303 }
304
305 if (!found) {
306 auto entry = DefaultReasonMap.find(std::make_tuple(reason, paranoid));
307 if (entry != DefaultReasonMap.end()) {
308 sev = entry->second;
309 }
310 }
311
312 new_bg_err = Status(bg_err, sev);
313
314 // Check if recovery is currently in progress. If it is, we will save this
315 // error so we can check it at the end to see if recovery succeeded or not
316 if (recovery_in_prog_ && recovery_error_.ok()) {
317 recovery_error_ = status_to_io_status(Status(new_bg_err));
318 }
319
320 bool auto_recovery = auto_recovery_;
321 if (new_bg_err.severity() >= Status::Severity::kFatalError && auto_recovery) {
322 auto_recovery = false;
323 }
324
325 // Allow some error specific overrides
326 if (new_bg_err.subcode() == IOStatus::SubCode::kNoSpace ||
327 new_bg_err.subcode() == IOStatus::SubCode::kSpaceLimit) {
328 new_bg_err = OverrideNoSpaceError(new_bg_err, &auto_recovery);
329 }
330
331 if (!new_bg_err.ok()) {
332 Status s = new_bg_err;
333 EventHelpers::NotifyOnBackgroundError(db_options_.listeners, reason, &s,
334 db_mutex_, &auto_recovery);
335 if (!s.ok() && (s.severity() > bg_error_.severity())) {
336 bg_error_ = s;
337 } else {
338 // This error is less severe than previously encountered error. Don't
339 // take any further action
340 return;
341 }
342 }
343
344 recover_context_ = context;
345 if (auto_recovery) {
346 recovery_in_prog_ = true;
347
348 // Kick-off error specific recovery
349 if (new_bg_err.subcode() == IOStatus::SubCode::kNoSpace ||
350 new_bg_err.subcode() == IOStatus::SubCode::kSpaceLimit) {
351 RecoverFromNoSpace();
352 }
353 }
354 if (bg_error_.severity() >= Status::Severity::kHardError) {
355 is_db_stopped_.store(true, std::memory_order_release);
356 }
357}

359// This is the main function for looking at IO related error during the
360// background operations. The main logic is:
361// File scope IO error is treated as retryable IO error in the write path. In
362// RocksDB, If a file has write IO error and it is at file scope, RocksDB never
363// write to the same file again. RocksDB will create a new file and rewrite the
364// whole content. Thus, it is retryable.
365// There are three main categories of error handling:
366// 1) if the error is caused by data loss, the error is mapped to
367// unrecoverable error. Application/user must take action to handle
368// this situation (File scope case is excluded).
369// 2) if the error is a Retryable IO error (i.e., it is a file scope IO error,
370// or its retryable flag is set and not a data loss error), auto resume (
371// DBImpl::ResumeImpl) may be called and the auto resume can be controlled
372// by resume count and resume interval options. There are three sub-cases:
373// a) if the error happens during compaction, it is mapped to a soft error.
374// the compaction thread will reschedule a new compaction. This doesn't
375// call auto resume.
376// b) if the error happens during flush and also WAL is empty, it is mapped
377// to a soft error. Note that, it includes the case that IO error happens
378// in SST or manifest write during flush. Auto resume will be called.
379// c) all other errors are mapped to hard error. Auto resume will be called.
380// 3) for other cases, HandleKnownErrors(const Status& bg_err,
381// BackgroundErrorReason reason) will be called to handle other error cases
382// such as delegating to SstFileManager to handle no space error.
383void ErrorHandler::SetBGError(const Status& bg_status,
384 BackgroundErrorReason reason, bool wal_related) {
385 db_mutex_->AssertHeld();
386 Status tmp_status = bg_status;
387 IOStatus bg_io_err = status_to_io_status(std::move(tmp_status));
388
389 if (bg_io_err.ok()) {
390 return;
391 }
392 ROCKS_LOG_WARN(db_options_.info_log, "Background IO error %s, reason %d",
393 bg_io_err.ToString().c_str(), static_cast<int>(reason));
394
395 RecordStats({ERROR_HANDLER_BG_ERROR_COUNT, ERROR_HANDLER_BG_IO_ERROR_COUNT},
396 {} /* int_histograms */);
397
398 Status new_bg_io_err = bg_io_err;
399 DBRecoverContext context;
400 if (bg_io_err.GetScope() != IOStatus::IOErrorScope::kIOErrorScopeFile &&
401 bg_io_err.GetDataLoss()) {
402 // First, data loss (non file scope) is treated as unrecoverable error. So
403 // it can directly overwrite any existing bg_error_.
404 bool auto_recovery = false;
405 Status bg_err(new_bg_io_err, Status::Severity::kUnrecoverableError);
406 CheckAndSetRecoveryAndBGError(bg_err);
407 ROCKS_LOG_INFO(
408 db_options_.info_log,
409 "ErrorHandler: Set background IO error as unrecoverable error\n");
410 EventHelpers::NotifyOnBackgroundError(db_options_.listeners, reason,
411 &bg_err, db_mutex_, &auto_recovery);
412 recover_context_ = context;
413 return;
414 }
415 if (wal_related) {
416 assert(reason == BackgroundErrorReason::kWriteCallback ||
417 reason == BackgroundErrorReason::kMemTable ||
418 reason == BackgroundErrorReason::kFlush);
419 }
420 if (db_options_.manual_wal_flush && wal_related && bg_io_err.IsIOError()) {
421 // With manual_wal_flush, a WAL write failure can drop buffered WAL writes.
422 // Memtables and WAL then become inconsistent. A successful memtable flush
423 // on one CF can cause CFs to be inconsistent upon restart. Before we fix
424 // the bug in auto recovery from WAL write failures that can flush one CF
425 // at a time, we set the error severity to fatal to disallow auto recovery.
426 // TODO: remove parameter `wal_related` once we can automatically recover
427 // from WAL write failures.
428 bool auto_recovery = false;
429 Status bg_err(new_bg_io_err, Status::Severity::kFatalError);
430 CheckAndSetRecoveryAndBGError(bg_err);
431 ROCKS_LOG_WARN(db_options_.info_log,
432 "ErrorHandler: A potentially WAL error happened, set "
433 "background IO error as fatal error\n");
434 EventHelpers::NotifyOnBackgroundError(db_options_.listeners, reason,
435 &bg_err, db_mutex_, &auto_recovery);
436 recover_context_ = context;
437 return;
438 }
439
440 if (bg_io_err.subcode() != IOStatus::SubCode::kNoSpace &&
441 (bg_io_err.GetScope() == IOStatus::IOErrorScope::kIOErrorScopeFile ||
442 bg_io_err.GetRetryable())) {
443 // Second, check if the error is a retryable IO error (file scope IO error
444 // is also treated as retryable IO error in RocksDB write path). if it is
445 // retryable error and its severity is higher than bg_error_, overwrite the
446 // bg_error_ with new error. In current stage, for retryable IO error of
447 // compaction, treat it as soft error. In other cases, treat the retryable
448 // IO error as hard error. Note that, all the NoSpace error should be
449 // handled by the SstFileManager::StartErrorRecovery(). Therefore, no matter
450 // it is retryable or file scope, this logic will be bypassed.
451
452 RecordStats({ERROR_HANDLER_BG_RETRYABLE_IO_ERROR_COUNT},
453 {} /* int_histograms */);
454 ROCKS_LOG_INFO(db_options_.info_log,
455 "ErrorHandler: Set background retryable IO error\n");
456 if (BackgroundErrorReason::kCompaction == reason) {
457 // We map the retryable IO error during compaction to soft error. Since
458 // compaction can reschedule by itself. We will not set the BG error in
459 // this case
460 // TODO: a better way to set or clean the retryable IO error which
461 // happens during compaction SST file write.
462 RecordStats({ERROR_HANDLER_AUTORESUME_COUNT}, {} /* int_histograms */);
463 ROCKS_LOG_INFO(
464 db_options_.info_log,
465 "ErrorHandler: Compaction will schedule by itself to resume\n");
466 bool auto_recovery = false;
467 EventHelpers::NotifyOnBackgroundError(db_options_.listeners, reason,
468 &new_bg_io_err, db_mutex_,
469 &auto_recovery);
470 // Not used in this code path.
471 new_bg_io_err.PermitUncheckedError();
472 return;
473 }
474
475 Status::Severity severity;
476 if (BackgroundErrorReason::kFlushNoWAL == reason ||
477 BackgroundErrorReason::kManifestWriteNoWAL == reason) {
478 // When the BG Retryable IO error reason is flush without WAL,
479 // We map it to a soft error. At the same time, all the background work
480 // should be stopped except the BG work from recovery. Therefore, we
481 // set the soft_error_no_bg_work_ to true. At the same time, since DB
482 // continues to receive writes when BG error is soft error, to avoid
483 // to many small memtable being generated during auto resume, the flush
484 // reason is set to kErrorRecoveryRetryFlush.
485 severity = Status::Severity::kSoftError;
486 soft_error_no_bg_work_ = true;
487 context.flush_reason = FlushReason::kErrorRecoveryRetryFlush;
488 } else {
489 severity = Status::Severity::kHardError;
490 }
491 Status bg_err(new_bg_io_err, severity);
492 CheckAndSetRecoveryAndBGError(bg_err);
493 recover_context_ = context;
494 bool auto_recovery = db_options_.max_bgerror_resume_count > 0;
495 EventHelpers::NotifyOnBackgroundError(db_options_.listeners, reason,
496 &new_bg_io_err, db_mutex_,
497 &auto_recovery);
498 StartRecoverFromRetryableBGIOError(bg_io_err);
499 return;
500 }
501 HandleKnownErrors(new_bg_io_err, reason);
502}

504void ErrorHandler::AddFilesToQuarantine(
505 autovector<const autovector<uint64_t>*> files_to_quarantine) {
506 db_mutex_->AssertHeld();
507 std::ostringstream quarantine_files_oss;
508 bool is_first_one = true;
509 for (const auto* files : files_to_quarantine) {
510 assert(files);
511 for (uint64_t file_number : *files) {
512 files_to_quarantine_.push_back(file_number);
513 quarantine_files_oss << (is_first_one ? "" : ", ") << file_number;
514 is_first_one = false;
515 }
516 }
517 ROCKS_LOG_INFO(db_options_.info_log,
518 "ErrorHandler: added file numbers %s to quarantine.\n",
519 quarantine_files_oss.str().c_str());
520}

529Status ErrorHandler::OverrideNoSpaceError(const Status& bg_error,
530 bool* auto_recovery) {
531 if (bg_error.severity() >= Status::Severity::kFatalError) {
532 return bg_error;
533 }
534
535 if (db_options_.sst_file_manager.get() == nullptr) {
536 // We rely on SFM to poll for enough disk space and recover
537 *auto_recovery = false;
538 return bg_error;
539 }
540
541 if (db_options_.allow_2pc &&
542 (bg_error.severity() <= Status::Severity::kSoftError)) {
543 // Don't know how to recover, as the contents of the current WAL file may
544 // be inconsistent, and it may be needed for 2PC. If 2PC is not enabled,
545 // we can just flush the memtable and discard the log
546 *auto_recovery = false;
547 return Status(bg_error, Status::Severity::kFatalError);
548 }
549
550 {
551 uint64_t free_space;
552 if (db_options_.env->GetFreeSpace(db_options_.db_paths[0].path,
553 &free_space) == Status::NotSupported()) {
554 *auto_recovery = false;
555 }
556 }
557
558 return bg_error;
559}

571Status ErrorHandler::ClearBGError() {
572 db_mutex_->AssertHeld();
573
574 // Signal that recovery succeeded
575 if (recovery_error_.ok()) {
576 assert(files_to_quarantine_.empty());
577 Status old_bg_error = bg_error_;
578 // old_bg_error is only for notifying listeners, so may not be checked
579 old_bg_error.PermitUncheckedError();
580 // Clear and check the recovery IO and BG error
581 is_db_stopped_.store(false, std::memory_order_release);
582 bg_error_ = Status::OK();
583 recovery_error_ = IOStatus::OK();
584 bg_error_.PermitUncheckedError();
585 recovery_error_.PermitUncheckedError();
586 recovery_in_prog_ = false;
587 soft_error_no_bg_work_ = false;
588 EventHelpers::NotifyOnErrorRecoveryEnd(db_options_.listeners, old_bg_error,
589 bg_error_, db_mutex_);
590 }
591 return recovery_error_;

594Status ErrorHandler::RecoverFromBGError(bool is_manual) {
595 InstrumentedMutexLock l(db_mutex_);
596 bool no_bg_work_original_flag = soft_error_no_bg_work_;
597 if (is_manual) {
598 // If its a manual recovery and there's a background recovery in progress
599 // return busy status
600 if (recovery_in_prog_) {
601 return Status::Busy("Recovery already in progress");
602 }
603 recovery_in_prog_ = true;
604
605 // In manual resume, we allow the bg work to run. If it is a auto resume,
606 // the bg work should follow this tag.
607 soft_error_no_bg_work_ = false;
608
609 // In manual resume, if the bg error is a soft error and also requires
610 // no bg work, the error must be recovered by call the flush with
611 // flush reason: kErrorRecoveryRetryFlush. In other case, the flush
612 // reason is set to kErrorRecovery.
613 if (no_bg_work_original_flag) {
614 recover_context_.flush_reason = FlushReason::kErrorRecoveryRetryFlush;
615 } else {
616 recover_context_.flush_reason = FlushReason::kErrorRecovery;
617 }
618 }
619
620 if (bg_error_.severity() == Status::Severity::kSoftError &&
621 recover_context_.flush_reason == FlushReason::kErrorRecovery) {
622 // Simply clear the background error and return
623 recovery_error_ = IOStatus::OK();
624 return ClearBGError();
625 }
626
627 // Reset recovery_error_. We will use this to record any errors that happen
628 // during the recovery process. While recovering, the only operations that
629 // can generate background errors should be the flush operations
630 recovery_error_ = IOStatus::OK();
631 recovery_error_.PermitUncheckedError();
632 Status s = db_->ResumeImpl(recover_context_);
633 if (s.ok()) {
634 soft_error_no_bg_work_ = false;
635 } else {
636 soft_error_no_bg_work_ = no_bg_work_original_flag;
637 }
638
639 // For manual recover, shutdown, and fatal error cases, set
640 // recovery_in_prog_ to false. For automatic background recovery, leave it
641 // as is regardless of success or failure as it will be retried
642 if (is_manual || s.IsShutdownInProgress() ||
643 bg_error_.severity() >= Status::Severity::kFatalError) {
644 recovery_in_prog_ = false;
645 }
646 return s;
647}

649void ErrorHandler::StartRecoverFromRetryableBGIOError(
650 const IOStatus& io_error) {
651 db_mutex_->AssertHeld();
652 if (bg_error_.ok() || io_error.ok()) {
653 return;
654 }
655 if (db_options_.max_bgerror_resume_count <= 0 || recovery_in_prog_) {
656 // Auto resume BG error is not enabled
657 return;
658 }
659 if (end_recovery_) {
660 // Can temporarily release db mutex
661 EventHelpers::NotifyOnErrorRecoveryEnd(db_options_.listeners, bg_error_,
662 Status::ShutdownInProgress(),
663 db_mutex_);
664 db_mutex_->AssertHeld();
665 return;
666 }
667 RecordStats({ERROR_HANDLER_AUTORESUME_COUNT}, {} /* int_histograms */);
668 ROCKS_LOG_INFO(
669 db_options_.info_log,
670 "ErrorHandler: Call StartRecoverFromRetryableBGIOError to resume\n");
671 // Needs to be set in the same lock hold as setting BG error, otherwise
672 // intervening writes could see a BG error without a recovery and bail out.
673 recovery_in_prog_ = true;
674
675 if (recovery_thread_) {
676 // Ensure only one thread can execute the join().
677 std::unique_ptr<port::Thread> old_recovery_thread(
678 std::move(recovery_thread_));
679 // In this case, if recovery_in_prog_ is false, current thread should
680 // wait the previous recover thread to finish and create a new thread
681 // to recover from the bg error.
682 db_mutex_->Unlock();
683 TEST_SYNC_POINT(
684 "StartRecoverFromRetryableBGIOError:BeforeWaitingForOtherThread");
685 old_recovery_thread->join();
686 TEST_SYNC_POINT(
687 "StartRecoverFromRetryableBGIOError:AfterWaitingForOtherThread");
688 db_mutex_->Lock();
689 }
690
691 recovery_thread_.reset(
692 new port::Thread(&ErrorHandler::RecoverFromRetryableBGIOError, this));
693}

695// Automatic recover from Retryable BG IO error. Must be called after db
696// mutex is released.
697void ErrorHandler::RecoverFromRetryableBGIOError() {
698 assert(recovery_in_prog_);
699 TEST_SYNC_POINT("RecoverFromRetryableBGIOError:BeforeStart");
700 TEST_SYNC_POINT("RecoverFromRetryableBGIOError:BeforeStart2");
701 InstrumentedMutexLock l(db_mutex_);
702 if (end_recovery_) {
703 EventHelpers::NotifyOnErrorRecoveryEnd(db_options_.listeners, bg_error_,
704 Status::ShutdownInProgress(),
705 db_mutex_);
706
707 recovery_in_prog_ = false;
708 return;
709 }
710 DBRecoverContext context = recover_context_;
711 context.flush_after_recovery = true;
712 int resume_count = db_options_.max_bgerror_resume_count;
713 uint64_t wait_interval = db_options_.bgerror_resume_retry_interval;
714 uint64_t retry_count = 0;
715 // Recover from the retryable error. Create a separate thread to do it.
716 while (resume_count > 0) {
717 if (end_recovery_) {
718 EventHelpers::NotifyOnErrorRecoveryEnd(db_options_.listeners, bg_error_,
719 Status::ShutdownInProgress(),
720 db_mutex_);
721 recovery_in_prog_ = false;
722 return;
723 }
724 TEST_SYNC_POINT("RecoverFromRetryableBGIOError:BeforeResume0");
725 TEST_SYNC_POINT("RecoverFromRetryableBGIOError:BeforeResume1");
726 recovery_error_ = IOStatus::OK();
727 retry_count++;
728 Status s = db_->ResumeImpl(context);
729 RecordStats({ERROR_HANDLER_AUTORESUME_RETRY_TOTAL_COUNT},
730 {} /* int_histograms */);
731 if (s.IsShutdownInProgress() ||
732 bg_error_.severity() >= Status::Severity::kFatalError) {
733 // If DB shutdown in progress or the error severity is higher than
734 // Hard Error, stop auto resume and returns.
735 recovery_in_prog_ = false;
736 RecordStats({} /* ticker_types */,
737 {{ERROR_HANDLER_AUTORESUME_RETRY_COUNT, retry_count}});
738 EventHelpers::NotifyOnErrorRecoveryEnd(db_options_.listeners, bg_error_,
739 bg_error_, db_mutex_);
740 return;
741 }
742 if (!recovery_error_.ok() &&
743 recovery_error_.severity() <= Status::Severity::kHardError &&
744 recovery_error_.GetRetryable()) {
745 // If new BG IO error happens during auto recovery and it is retryable
746 // and its severity is Hard Error or lower, the auto resmue sleep for
747 // a period of time and redo auto resume if it is allowed.
748 TEST_SYNC_POINT("RecoverFromRetryableBGIOError:BeforeWait0");
749 TEST_SYNC_POINT("RecoverFromRetryableBGIOError:BeforeWait1");
750 int64_t wait_until = db_options_.clock->NowMicros() + wait_interval;
751 cv_.TimedWait(wait_until);
752 } else {
753 // There are three possibility: 1) recovery_error_ is set during resume
754 // and the error is not retryable, 2) recover is successful, 3) other
755 // error happens during resume and cannot be resumed here.
756 if (recovery_error_.ok() && s.ok()) {
757 // recover from the retryable IO error and no other BG errors. Clean
758 // the bg_error and notify user.
759 TEST_SYNC_POINT("RecoverFromRetryableBGIOError:RecoverSuccess");
760 RecordStats({ERROR_HANDLER_AUTORESUME_SUCCESS_COUNT},
761 {{ERROR_HANDLER_AUTORESUME_RETRY_COUNT, retry_count}});
762 return;
763 } else {
764 // In this case: 1) recovery_error_ is more serious or not retryable
765 // 2) other error happens. The auto recovery stops.
766 recovery_in_prog_ = false;
767 RecordStats({} /* ticker_types */,
768 {{ERROR_HANDLER_AUTORESUME_RETRY_COUNT, retry_count}});
769 EventHelpers::NotifyOnErrorRecoveryEnd(
770 db_options_.listeners, bg_error_,
771 !recovery_error_.ok() ? recovery_error_ : s, db_mutex_);
772 return;
773 }
774 }
775 resume_count--;
776 }
777 recovery_in_prog_ = false;
778 EventHelpers::NotifyOnErrorRecoveryEnd(
779 db_options_.listeners, bg_error_,
780 Status::Aborted("Exceeded resume retry count"), db_mutex_);
781 TEST_SYNC_POINT("RecoverFromRetryableBGIOError:LoopOut");
782 RecordStats({} /* ticker_types */,
783 {{ERROR_HANDLER_AUTORESUME_RETRY_COUNT, retry_count}});
784}
