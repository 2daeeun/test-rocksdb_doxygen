28// A Cache maps keys to objects resident in memory, tracks reference counts
29// on those key-object entries, and is able to remove unreferenced entries
30// whenever it wants. All operations are fully thread safe except as noted.
31// Inserted entries have a specified "charge" which is some quantity in
32// unspecified units, typically bytes of memory used. A Cache will typically
33// have a finite capacity in units of charge, and evict entries as needed
34// to stay at or below that capacity.
35//
36// NOTE: This API is for expert use only and is intended more for customizing
37// cache behavior than for calling into outside of RocksDB. It is subject to
38// change as RocksDB evolves, especially the RocksDB block cache. Overriding
39// CacheWrapper is the preferred way of customizing some operations on an
40// existing implementation.
41//
42// INTERNAL: See typed_cache.h for convenient wrappers on top of this API.
43// New virtual functions must also be added to CacheWrapper below.
44class Cache : public Customizable {
45 public: // types hidden from API client
46 // Opaque handle to an entry stored in the cache.
47 struct Handle {};
48
49 public: // types hidden from Cache implementation
50 // Pointer to cached object of unspecified type. (This type alias is
51 // provided for clarity, not really for type checking.)
52 using ObjectPtr = void*;

54 // Opaque object providing context (settings, etc.) to create objects
55 // for primary cache from saved (serialized) secondary cache entries.

59 // Depending on implementation, cache entries with higher priority levels
60 // could be less likely to get evicted than entries with lower priority
61 // levels. The "high" priority level applies to certain SST metablocks (e.g.
62 // index and filter blocks) if the option
63 // cache_index_and_filter_blocks_with_high_priority is set. The "low" priority
64 // level is used for other kinds of SST blocks (most importantly, data
65 // blocks), as well as the above metablocks in case
66 // cache_index_and_filter_blocks_with_high_priority is
67 // not set. The "bottom" priority level is for BlobDB's blob values.
68 enum class Priority { HIGH, LOW, BOTTOM };

70 // A set of callbacks to allow objects in the primary block cache to be
71 // persisted in a secondary cache. The purpose of the secondary cache
72 // is to support other ways of caching the object, such as persistent or
73 // compressed data, that may require the object to be parsed and transformed
74 // in some way. Since the primary cache holds C++ objects and the secondary
75 // cache may only hold flat data that doesn't need relocation, these
76 // callbacks need to be provided by the user of the block
77 // cache to do the conversion.
78 // The CacheItemHelper is passed to Insert() and Lookup(). It has pointers
79 // to callback functions for size, saving and deletion of the
80 // object. The callbacks are defined in C-style in order to make them
81 // stateless and not add to the cache metadata size.
82 // Saving multiple std::function objects will take up 32 bytes per
83 // function, even if its not bound to an object and does no capture.
84 //
85 // All the callbacks are C-style function pointers in order to simplify
86 // lifecycle management. Objects in the cache can outlive the parent DB,
87 // so anything required for these operations should be contained in the
88 // object itself.
89 //
90 // The SizeCallback takes a pointer to the object and returns the size
91 // of the persistable data. It can be used by the secondary cache to allocate
92 // memory if needed.
93 //
94 // RocksDB callbacks are NOT exception-safe. A callback completing with an
95 // exception can lead to undefined behavior in RocksDB, including data loss,
96 // unreported corruption, deadlocks, and more.
97 using SizeCallback = size_t (*)(ObjectPtr obj);
98
99 // The SaveToCallback takes an object pointer and saves the persistable
100 // data into a buffer. The secondary cache may decide to not store it in a
101 // contiguous buffer, in which case this callback will be called multiple
102 // times with increasing offset
103 using SaveToCallback = Status (*)(ObjectPtr from_obj, size_t from_offset,
104 size_t length, char* out_buf);
105
106 // A function pointer type for destruction of a cache object. This will
107 // typically call the destructor for the appropriate type of the object.
108 // The Cache is responsible for copying and reclaiming space for the key,
109 // but objects are managed in part using this callback. Generally a DeleterFn
110 // can be nullptr if the ObjectPtr does not need destruction (e.g. nullptr or
111 // pointer into static data).
112 using DeleterFn = void (*)(ObjectPtr obj, MemoryAllocator* allocator);
113
114 // The CreateCallback is takes in a buffer from the secondary cache and
115 // constructs an object using it. The buffer could be compressed or
116 // uncompressed, as indicated by the type argument. If compressed,
117 // the callback is responsible for uncompressing it using information
118 // from the context, such as compression dictionary.
119 // The callback doesn't have ownership of the buffer and
120 // should copy the contents into its own buffer. The CreateContext* is
121 // provided by Lookup and may be used to follow DB- or CF-specific settings.
122 // In case of some error, non-OK is returned and the caller should ignore
123 // any result in out_obj. (The implementation must clean up after itself.)
124 using CreateCallback = Status (*)(const Slice& data, CompressionType type,
125 CacheTier source, CreateContext* context,
126 MemoryAllocator* allocator,
127 ObjectPtr* out_obj, size_t* out_charge);

129 // A struct with pointers to helper functions for spilling items from the
130 // cache into the secondary cache. May be extended in the future. An
131 // instance of this struct is expected to outlive the cache.
132 struct CacheItemHelper {
133 // Function for deleting an object on its removal from the Cache.
134 // nullptr is only for entries that require no destruction, such as
135 // "placeholder" cache entries with nullptr object.
136 DeleterFn del_cb; // (<- Most performance critical)
137 // Next three are used for persisting values as described above.
138 // If any is nullptr, then all three should be nullptr and persisting the
139 // entry to/from secondary cache is not supported.
140 SizeCallback size_cb;
141 SaveToCallback saveto_cb;
142 CreateCallback create_cb;
143 // Classification of the entry for monitoring purposes in block cache.
144 CacheEntryRole role;
145 // Another CacheItemHelper (or this one) without secondary cache support.
146 // This is provided so that items promoted from secondary cache into
147 // primary cache without removal from the secondary cache can be prevented
148 // from attempting re-insertion into secondary cache (for efficiency).
149 const CacheItemHelper* without_secondary_compat;
150
151 CacheItemHelper() : CacheItemHelper(CacheEntryRole::kMisc) {}
152
153 // For helpers without SecondaryCache support
154 explicit CacheItemHelper(CacheEntryRole _role, DeleterFn _del_cb = nullptr)
155 : CacheItemHelper(_role, _del_cb, nullptr, nullptr, nullptr, this) {}
156
157 // For helpers with SecondaryCache support
158 explicit CacheItemHelper(CacheEntryRole _role, DeleterFn _del_cb,
159 SizeCallback _size_cb, SaveToCallback _saveto_cb,
160 CreateCallback _create_cb,
161 const CacheItemHelper* _without_secondary_compat)
162 : del_cb(_del_cb),
163 size_cb(_size_cb),
164 saveto_cb(_saveto_cb),
165 create_cb(_create_cb),
166 role(_role),
167 without_secondary_compat(_without_secondary_compat) {
168 // Either all three secondary cache callbacks are non-nullptr or
169 // all three are nullptr
170 assert((size_cb != nullptr) == (saveto_cb != nullptr));
171 assert((size_cb != nullptr) == (create_cb != nullptr));
172 // without_secondary_compat points to equivalent but without
173 // secondary support
174 assert(role == without_secondary_compat->role);
175 assert(del_cb == without_secondary_compat->del_cb);
176 assert(!without_secondary_compat->IsSecondaryCacheCompatible());
177 }
178 inline bool IsSecondaryCacheCompatible() const {
179 return size_cb != nullptr;
180 }
181 };

214 // The Insert and Lookup APIs below are intended to allow cached objects
215 // to be demoted/promoted between the primary block cache and a secondary
216 // cache. The secondary cache could be a non-volatile cache, and will
217 // likely store the object in a different representation. They rely on a
218 // per object CacheItemHelper to do the conversions.
219 // The secondary cache may persist across process and system restarts,
220 // and may even be moved between hosts. Therefore, the cache key must
221 // be repeatable across restarts/reboots, and globally unique if
222 // multiple DBs share the same cache and the set of DBs can change
223 // over time.
224
225 // Insert a mapping from key->object into the cache and assign it
226 // the specified charge against the total cache capacity. If
227 // strict_capacity_limit is true and cache reaches its full capacity,
228 // return Status::MemoryLimit. `obj` must be non-nullptr if compatible
229 // with secondary cache (helper->size_cb != nullptr), because Value() ==
230 // nullptr is reserved for indicating some secondary cache failure cases.
231 // On success, returns OK and takes ownership of `obj`, eventually deleting
232 // it with helper->del_cb. On non-OK return, the caller maintains ownership
233 // of `obj` so will often need to delete it in such cases.
234 //
235 // The helper argument is saved by the cache and will be used when the
236 // inserted object is evicted or considered for promotion to the secondary
237 // cache. Promotion to secondary cache is only enabled if helper->size_cb
238 // != nullptr. The helper must outlive the cache. Callers may use
239 // &kNoopCacheItemHelper as a trivial helper (no deleter for the object,
240 // no secondary cache). `helper` must not be nullptr (efficiency).
241 //
242 // If `handle` is not nullptr and return status is OK, `handle` is set
243 // to a Handle* for the entry. The caller must call this->Release(handle)
244 // when the returned entry is no longer needed. If `handle` is nullptr, it is
245 // as if Release is called immediately after Insert.
246 //
247 // Regardless of whether the item was inserted into the cache,
248 // it will attempt to insert it into the secondary cache if one is
249 // configured, and the helper supports it.
250 // The cache implementation must support a secondary cache, otherwise
251 // the item is only inserted into the primary cache. It may
252 // defer the insertion to the secondary cache as it sees fit.
253 //
254 // Along with the object pointer, the caller may pass a Slice pointing to
255 // the compressed serialized data of the object. If compressed is
256 // non-empty, then the caller must pass the type indicating the compression
257 // algorithm used. The cache may, optionally, also insert the compressed
258 // block into one or more cache tiers.
259 //
260 // When the inserted entry is no longer needed, it will be destroyed using
261 // helper->del_cb (if non-nullptr).
262 virtual Status Insert(
263 const Slice& key, ObjectPtr obj, const CacheItemHelper* helper,
264 size_t charge, Handle** handle = nullptr,
265 Priority priority = Priority::LOW, const Slice& compressed = Slice(),
266 CompressionType type = CompressionType::kNoCompression) = 0;

268 // Similar to Insert, but used for creating cache entries that cannot
269 // be found with Lookup, such as for memory charging purposes. The
270 // key is needed for cache sharding purposes.
271 // * If allow_uncharged==true or strict_capacity_limit=false, the operation
272 // always succeeds and returns a valid Handle.
273 // * If strict_capacity_limit=true and the requested charge cannot be freed
274 // up in the cache, then
275 // * If allow_uncharged==true, it's created anyway (GetCharge() == 0).
276 // * If allow_uncharged==false, returns nullptr to indicate failure.
277 virtual Handle* CreateStandalone(const Slice& key, ObjectPtr obj,
278 const CacheItemHelper* helper, size_t charge,
279 bool allow_uncharged) = 0;

281 // Lookup the key, returning nullptr if not found. If found, returns
282 // a handle to the mapping that must eventually be passed to Release().
283 //
284 // If a non-nullptr helper argument is provided with a non-nullptr
285 // create_cb, and a secondary cache is configured, then the secondary
286 // cache is also queried if lookup in the primary cache fails. If found
287 // in secondary cache, the provided create_db and create_context are
288 // used to promote the entry to an object in the primary cache.
289 // In that case, the helper may be saved and used later when the object
290 // is evicted, so as usual, the pointed-to helper must outlive the cache.

513 // Starts a potentially asynchronous Lookup(), based on the populated
514 // "input" fields of the async_handle. The caller is responsible for
515 // keeping the AsyncLookupHandle and the key it references alive through
516 // WaitAll(), and the AsyncLookupHandle alive through
517 // AsyncLookupHandle::Result(). WaitAll() can only be skipped if
518 // AsyncLookupHandle::IsPending() is already false after StartAsyncLookup.
519 // Calling AsyncLookupHandle::Result() is essentially required so that
520 // Release() can be called on non-nullptr Handle result. Wait() is a
521 // concise version of WaitAll()+Result() on a single handle. After an
522 // AsyncLookupHandle has completed this cycle, its input fields can be
523 // updated and re-used for another StartAsyncLookup.
524 //
525 // Handle is thread-safe while AsyncLookupHandle is not thread-safe.
526 //
527 // Default implementation is appropriate for Caches without
528 // true asynchronous support: defers to synchronous Lookup().
529 // (AsyncLookupHandles will only get into the "pending" state with
530 // SecondaryCache configured.)
531 virtual void StartAsyncLookup(AsyncLookupHandle& async_handle);
532
533 // A convenient wrapper around WaitAll() and AsyncLookupHandle::Result()
534 // for a single async handle. See StartAsyncLookup().
535 Handle* Wait(AsyncLookupHandle& async_handle);
536
537 // Wait for an array of async handles to get results, so that none are left
538 // in the "pending" state. Not thread safe. See StartAsyncLookup().
539 // Default implementation is appropriate for Caches without true
540 // asynchronous support: asserts that all handles are not pending (or not
541 // expected to be handled by this cache, in case of wrapped/stacked
542 // WaitAlls()).
543 virtual void WaitAll(AsyncLookupHandle* /*async_handles*/, size_t /*count*/);
544
545 // For a function called on cache entries about to be evicted. The function
546 // returns `true` if it has taken ownership of the Value (object), or
547 // `false` if the cache should destroy it as usual. Regardless, Ref() and
548 // Release() cannot be called on this Handle that is poised for eviction.
549 using EvictionCallback =
550 std::function<bool(const Slice& key, Handle* h, bool was_hit)>;
551 // Sets an eviction callback for this Cache. Not thread safe and only
552 // supports being set once, so should only be used during initialization
553 // or destruction, guaranteed before or after any thread-shared operations.
554 void SetEvictionCallback(EvictionCallback&& fn);
555
556 protected:
557 std::shared_ptr<MemoryAllocator> memory_allocator_;
558 EvictionCallback eviction_callback_;
559};
