51// for micros < 10ms, print "XX us".
52// for micros < 10sec, print "XX ms".
53// for micros >= 10 sec, print "XX sec".
54// for micros <= 1 hour, print Y:X M:S".
55// for micros > 1 hour, print Z:Y:X H:M:S".
56int AppendHumanMicros(uint64_t micros, char* output, int len,
57 bool fixed_format) {
58 if (micros < 10000 && !fixed_format) {
59 return snprintf(output, len, "%" PRIu64 " us", micros);
60 } else if (micros < 10000000 && !fixed_format) {
61 return snprintf(output, len, "%.3lf ms",
62 static_cast<double>(micros) / 1000);
63 } else if (micros < 1000000l * 60 && !fixed_format) {
64 return snprintf(output, len, "%.3lf sec",
65 static_cast<double>(micros) / 1000000);
66 } else if (micros < 1000000ll * 60 * 60 && !fixed_format) {
67 return snprintf(output, len, "%02" PRIu64 ":%05.3f M:S",
68 micros / 1000000 / 60,
69 static_cast<double>(micros % 60000000) / 1000000);
70 } else {
71 return snprintf(output, len, "%02" PRIu64 ":%02" PRIu64 ":%05.3f H:M:S",
72 micros / 1000000 / 3600, (micros / 1000000 / 60) % 60,
73 static_cast<double>(micros % 60000000) / 1000000);
74 }
75}

77// for sizes >=10TB, print "XXTB"
78// for sizes >=10GB, print "XXGB"
79// etc.
80// append file size summary to output and return the len
81int AppendHumanBytes(uint64_t bytes, char* output, int len) {
82 const uint64_t ull10 = 10;
83 if (bytes >= ull10 << 40) {
84 return snprintf(output, len, "%" PRIu64 "TB", bytes >> 40);
85 } else if (bytes >= ull10 << 30) {
86 return snprintf(output, len, "%" PRIu64 "GB", bytes >> 30);
87 } else if (bytes >= ull10 << 20) {
88 return snprintf(output, len, "%" PRIu64 "MB", bytes >> 20);
89 } else if (bytes >= ull10 << 10) {
90 return snprintf(output, len, "%" PRIu64 "KB", bytes >> 10);
91 } else {
92 return snprintf(output, len, "%" PRIu64 "B", bytes);
93 }
94}

102void AppendEscapedStringTo(std::string* str, const Slice& value) {
103 for (size_t i = 0; i < value.size(); i++) {
104 char c = value[i];
105 if (c >= ' ' && c <= '~') {
106 str->push_back(c);
107 } else {
108 char buf[10];
109 snprintf(buf, sizeof(buf), "\\x%02x",
110 static_cast<unsigned int>(c) & 0xff);
111 str->append(buf);
112 }

116std::string NumberToHumanString(int64_t num) {
117 char buf[21];
118 int64_t absnum;
119
120 if (num < 0) {
121 // abs(INT64_MIN) is INT64_MAX+1 which overflows int64_t and become itself.
122 // So we convert it to INT64_MAX to avoid fall into <10000 slot.
123 absnum = num == INT64_MIN ? INT64_MAX : -num;
124 } else {
125 absnum = num;
126 }
127
128 if (absnum < 10000) {
129 snprintf(buf, sizeof(buf), "%" PRIi64, num);
130 } else if (absnum < 10000000) {
131 snprintf(buf, sizeof(buf), "%" PRIi64 "K", num / 1000);
132 } else if (absnum < 10000000000LL) {
133 snprintf(buf, sizeof(buf), "%" PRIi64 "M", num / 1000000);
134 } else {
135 snprintf(buf, sizeof(buf), "%" PRIi64 "G", num / 1000000000);
136 }
137 return std::string(buf);
138}

140std::string BytesToHumanString(uint64_t bytes) {
141 const char* size_name[] = {"KB", "MB", "GB", "TB"};
142 double final_size = static_cast<double>(bytes);
143 size_t size_idx;
144
145 // always start with KB
146 final_size /= 1024;
147 size_idx = 0;
148
149 while (size_idx < 3 && final_size >= 1024) {
150 final_size /= 1024;
151 size_idx++;
152 }
153
154 char buf[20];
155 snprintf(buf, sizeof(buf), "%.2f %s", final_size, size_name[size_idx]);
156 return std::string(buf);
157}

175bool ConsumeDecimalNumber(Slice* in, uint64_t* val) {
176 uint64_t v = 0;
177 int digits = 0;
178 while (!in->empty()) {
179 char c = (*in)[0];
180 if (c >= '0' && c <= '9') {
181 ++digits;
182 const unsigned int delta = (c - '0');
183 static const uint64_t kMaxUint64 = ~static_cast<uint64_t>(0);
184 if (v > kMaxUint64 / 10 ||
185 (v == kMaxUint64 / 10 && delta > kMaxUint64 % 10)) {
186 // Overflow
187 return false;
188 }
189 v = (v * 10) + delta;
190 in->remove_prefix(1);
191 } else {
192 break;
193 }
194 }
195 *val = v;
196 return (digits > 0);
197}

212
213 auto iter = std::find_if(std::begin(convert_map), std::end(convert_map),
214 [c](const CharMap& p) { return p.first == c; });
215
216 if (iter == std::end(convert_map)) {
217 return c;
218 }
219 return iter->second;

248std::string UnescapeOptionString(const std::string& escaped_string) {
249 bool escaped = false;
250 std::string output;
251
252 for (auto c : escaped_string) {
253 if (escaped) {
254 output += UnescapeChar(c);
255 escaped = false;
256 } else {
257 if (c == '\\') {
258 escaped = true;
259 continue;
260 }
261 output += c;
262 }
263 }
264 return output;

267std::string trim(const std::string& str) {
268 if (str.empty()) {
269 return std::string();
270 }
271 size_t start = 0;
272 size_t end = str.size() - 1;
273 while (isspace(str[start]) != 0 && start < end) {
274 ++start;
275 }
276 while (isspace(str[end]) != 0 && start < end) {
277 --end;
278 }
279 if (start <= end) {
280 return str.substr(start, end - start + 1);
281 }
282 return std::string();
283}

338uint64_t ParseUint64(const std::string& value) {
339 size_t endchar;
340#ifndef CYGWIN
341 uint64_t num = std::stoull(value.c_str(), &endchar);
342#else
343 char* endptr;
344 uint64_t num = std::strtoul(value.c_str(), &endptr, 0);
345 endchar = endptr - value.c_str();
346#endif
347
348 if (endchar < value.length()) {
349 char c = value[endchar];
350 if (c == 'k' || c == 'K') {
351 num <<= 10LL;
352 } else if (c == 'm' || c == 'M') {
353 num <<= 20LL;
354 } else if (c == 'g' || c == 'G') {
355 num <<= 30LL;
356 } else if (c == 't' || c == 'T') {
357 num <<= 40LL;
358 }
359 }
360
361 return num;

426std::vector<int> ParseVectorInt(const std::string& value) {
427 std::vector<int> result;
428 size_t start = 0;
429 while (start < value.size()) {
430 size_t end = value.find(':', start);
431 if (end == std::string::npos) {
432 result.push_back(ParseInt(value.substr(start)));
433 break;
434 } else {
435 result.push_back(ParseInt(value.substr(start, end - start)));
436 start = end + 1;
437 }
438 }
439 return result;

453int ParseTimeStringToSeconds(const std::string& value) {
454 int hours, minutes;
455 char colon;
456
457 std::istringstream stream(value);
458 stream >> hours >> colon >> minutes;
459
460 if (stream.fail() || !stream.eof() || colon != ':') {
461 return -1;
462 }
463
464 if (hours < 0 || hours > 23 || minutes < 0 || minutes > 59) {
465 return -1;
466 }
467 return hours * 3600 + minutes * 60;
468}

470bool TryParseTimeRangeString(const std::string& value, int& start_time,
471 int& end_time) {
472 if (value.empty()) {
473 start_time = 0;
474 end_time = 0;
475 return true;
476 }
477 auto split = StringSplit(value, '-');
478 if (split.size() != 2) {
479 return false;
480 }
481 start_time = ParseTimeStringToSeconds(split[0]);
482 if (start_time < 0) {
483 return false;
484 }
485 end_time = ParseTimeStringToSeconds(split[1]);
486 if (end_time < 0) {
487 return false;
488 }
489 return true;
490}

492// Copied from folly/string.cpp:
493// https://github.com/facebook/folly/blob/0deef031cb8aab76dc7e736f8b7c22d701d5f36b/folly/String.cpp#L457
494// There are two variants of `strerror_r` function, one returns
495// `int`, and another returns `char*`. Selecting proper version using
496// preprocessor macros portably is extremely hard.
497//
498// For example, on Android function signature depends on `__USE_GNU` and
499// `__ANDROID_API__` macros (https://git.io/fjBBE).
500//
501// So we are using C++ overloading trick: we pass a pointer of
502// `strerror_r` to `invoke_strerror_r` function, and C++ compiler
503// selects proper function.
504
505#if !(defined(_WIN32) && (defined(__MINGW32__) || defined(_MSC_VER)))
506ROCKSDB_MAYBE_UNUSED
507static std::string invoke_strerror_r(int (*strerror_r)(int, char*, size_t),
508 int err, char* buf, size_t buflen) {
509 // Using XSI-compatible strerror_r
510 int r = strerror_r(err, buf, buflen);
511
512 // OSX/FreeBSD use EINVAL and Linux uses -1 so just check for non-zero
513 if (r != 0) {
514 snprintf(buf, buflen, "Unknown error %d (strerror_r failed with error %d)",
515 err, errno);
516 }
517 return buf;
518}

521static std::string invoke_strerror_r(char* (*strerror_r)(int, char*, size_t),
522 int err, char* buf, size_t buflen) {
523 // Using GNU strerror_r
524 return strerror_r(err, buf, buflen);
525}
526#endif // !(defined(_WIN32) && (defined(__MINGW32__) || defined(_MSC_VER)))

528std::string errnoStr(int err) {
529 char buf[1024];
530 buf[0] = '\0';
531
532 std::string result;
533
534 // https://developer.apple.com/library/mac/documentation/Darwin/Reference/ManPages/man3/strerror_r.3.html
535 // http://www.kernel.org/doc/man-pages/online/pages/man3/strerror.3.html
536#if defined(_WIN32) && (defined(__MINGW32__) || defined(_MSC_VER))
537 // mingw64 has no strerror_r, but Windows has strerror_s, which C11 added
538 // as well. So maybe we should use this across all platforms (together
539 // with strerrorlen_s). Note strerror_r and _s have swapped args.
540 int r = strerror_s(buf, sizeof(buf), err);
541 if (r != 0) {
542 snprintf(buf, sizeof(buf),
543 "Unknown error %d (strerror_r failed with error %d)", err, errno);
544 }
545 result.assign(buf);
546#else
547 // Using any strerror_r
548 result.assign(invoke_strerror_r(strerror_r, err, buf, sizeof(buf)));
549#endif
550
551 return result;
552}
