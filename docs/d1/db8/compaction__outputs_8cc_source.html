21Status CompactionOutputs::Finish(
22 const Status& intput_status,
23 const SeqnoToTimeMapping& seqno_to_time_mapping) {
24 FileMetaData* meta = GetMetaData();
25 assert(meta != nullptr);
26 Status s = intput_status;
27 if (s.ok()) {
28 SeqnoToTimeMapping relevant_mapping;
29 relevant_mapping.CopyFromSeqnoRange(
30 seqno_to_time_mapping,
31 std::min(smallest_preferred_seqno_, meta->fd.smallest_seqno),
32 meta->fd.largest_seqno);
33 relevant_mapping.SetCapacity(kMaxSeqnoTimePairsPerSST);
34 builder_->SetSeqnoTimeTableProperties(relevant_mapping,
35 meta->oldest_ancester_time);
36 s = builder_->Finish();
37
38 } else {
39 builder_->Abandon();
40 }
41 Status io_s = builder_->io_status();
42 if (s.ok()) {
43 s = io_s;
44 } else {
45 io_s.PermitUncheckedError();
46 }
47 const uint64_t current_bytes = builder_->FileSize();
48 if (s.ok()) {
49 meta->fd.file_size = current_bytes;
50 meta->tail_size = builder_->GetTailSize();
51 meta->marked_for_compaction = builder_->NeedCompact();
52 meta->user_defined_timestamps_persisted = static_cast<bool>(
53 builder_->GetTableProperties().user_defined_timestamps_persisted);
54 }
55 current_output().finished = true;
56 stats_.bytes_written += current_bytes;
57 stats_.num_output_files = outputs_.size();
58
59 return s;
60}

63 SystemClock* clock,
64 Statistics* statistics,
65 bool use_fsync) {
66 IOStatus io_s;
67 IOOptions opts;
68 io_s = WritableFileWriter::PrepareIOOptions(
69 WriteOptions(Env::IOActivity::kCompaction), opts);
70 if (input_status.ok() && io_s.ok()) {

81 meta->file_checksum_func_name = file_writer_->GetFileChecksumFuncName();
82 }
83
84 file_writer_.reset();
85
86 return io_s;

89bool CompactionOutputs::UpdateFilesToCutForTTLStates(
90 const Slice& internal_key) {
91 if (!files_to_cut_for_ttl_.empty()) {
92 const InternalKeyComparator* icmp =
93 &compaction_->column_family_data()->internal_comparator();
94 if (cur_files_to_cut_for_ttl_ != -1) {
95 // Previous key is inside the range of a file
96 if (icmp->Compare(internal_key,
97 files_to_cut_for_ttl_[cur_files_to_cut_for_ttl_]
98 ->largest.Encode()) > 0) {
99 next_files_to_cut_for_ttl_ = cur_files_to_cut_for_ttl_ + 1;
100 cur_files_to_cut_for_ttl_ = -1;
101 return true;
102 }
103 } else {
104 // Look for the key position
105 while (next_files_to_cut_for_ttl_ <
106 static_cast<int>(files_to_cut_for_ttl_.size())) {
107 if (icmp->Compare(internal_key,
108 files_to_cut_for_ttl_[next_files_to_cut_for_ttl_]
109 ->smallest.Encode()) >= 0) {
110 if (icmp->Compare(internal_key,
111 files_to_cut_for_ttl_[next_files_to_cut_for_ttl_]
112 ->largest.Encode()) <= 0) {
113 // With in the current file
114 cur_files_to_cut_for_ttl_ = next_files_to_cut_for_ttl_;
115 return true;
116 }
117 // Beyond the current file
118 next_files_to_cut_for_ttl_++;
119 } else {
120 // Still fall into the gap
121 break;
122 }
123 }
124 }
125 }
126 return false;
127}

129size_t CompactionOutputs::UpdateGrandparentBoundaryInfo(
130 const Slice& internal_key) {
131 size_t curr_key_boundary_switched_num = 0;
132 const std::vector<FileMetaData*>& grandparents = compaction_->grandparents();
133
134 if (grandparents.empty()) {
135 return curr_key_boundary_switched_num;
136 }
137 const Comparator* ucmp = compaction_->column_family_data()->user_comparator();
138
139 // Move the grandparent_index_ to the file containing the current user_key.
140 // If there are multiple files containing the same user_key, make sure the
141 // index points to the last file containing the key.
142 while (grandparent_index_ < grandparents.size()) {
143 if (being_grandparent_gap_) {
144 if (sstableKeyCompare(ucmp, internal_key,
145 grandparents[grandparent_index_]->smallest) < 0) {
146 break;
147 }
148 if (seen_key_) {
149 curr_key_boundary_switched_num++;
150 grandparent_overlapped_bytes_ +=
151 grandparents[grandparent_index_]->fd.GetFileSize();
152 grandparent_boundary_switched_num_++;
153 }
154 being_grandparent_gap_ = false;
155 } else {
156 int cmp_result = sstableKeyCompare(
157 ucmp, internal_key, grandparents[grandparent_index_]->largest);
158 // If it's same key, make sure grandparent_index_ is pointing to the last
159 // one.
160 if (cmp_result < 0 ||
161 (cmp_result == 0 &&
162 (grandparent_index_ == grandparents.size() - 1 ||
163 sstableKeyCompare(ucmp, internal_key,
164 grandparents[grandparent_index_ + 1]->smallest) <
165 0))) {
166 break;
167 }
168 if (seen_key_) {
169 curr_key_boundary_switched_num++;
170 grandparent_boundary_switched_num_++;
171 }
172 being_grandparent_gap_ = true;
173 grandparent_index_++;
174 }
175 }
176
177 // If the first key is in the middle of a grandparent file, adding it to the
178 // overlap
179 if (!seen_key_ && !being_grandparent_gap_) {
180 assert(grandparent_overlapped_bytes_ == 0);
181 grandparent_overlapped_bytes_ =
182 GetCurrentKeyGrandparentOverlappedBytes(internal_key);
183 }
184
185 seen_key_ = true;
186 return curr_key_boundary_switched_num;
187}

189uint64_t CompactionOutputs::GetCurrentKeyGrandparentOverlappedBytes(
190 const Slice& internal_key) const {
191 // no overlap with any grandparent file
192 if (being_grandparent_gap_) {
193 return 0;
194 }
195 uint64_t overlapped_bytes = 0;
196
197 const std::vector<FileMetaData*>& grandparents = compaction_->grandparents();
198 const Comparator* ucmp = compaction_->column_family_data()->user_comparator();
199 InternalKey ikey;
200 ikey.DecodeFrom(internal_key);
201#ifndef NDEBUG
202 // make sure the grandparent_index_ is pointing to the last files containing
203 // the current key.
204 int cmp_result =
205 sstableKeyCompare(ucmp, ikey, grandparents[grandparent_index_]->largest);
206 assert(
207 cmp_result < 0 ||
208 (cmp_result == 0 &&
209 (grandparent_index_ == grandparents.size() - 1 ||
210 sstableKeyCompare(
211 ucmp, ikey, grandparents[grandparent_index_ + 1]->smallest) < 0)));
212 assert(sstableKeyCompare(ucmp, ikey,
213 grandparents[grandparent_index_]->smallest) >= 0);
214#endif
215 overlapped_bytes += grandparents[grandparent_index_]->fd.GetFileSize();
216
217 // go backwards to find all overlapped files, one key can overlap multiple
218 // files. In the following example, if the current output key is `c`, and one
219 // compaction file was cut before `c`, current `c` can overlap with 3 files:
220 // [a b] [c...
221 // [b, b] [c, c] [c, c] [c, d]
222 for (int64_t i = static_cast<int64_t>(grandparent_index_) - 1;
223 i >= 0 && sstableKeyCompare(ucmp, ikey, grandparents[i]->largest) == 0;
224 i--) {
225 overlapped_bytes += grandparents[i]->fd.GetFileSize();
226 }
227
228 return overlapped_bytes;
229}

231bool CompactionOutputs::ShouldStopBefore(const CompactionIterator& c_iter) {
232 assert(c_iter.Valid());
233 const Slice& internal_key = c_iter.key();
234#ifndef NDEBUG
235 bool should_stop = false;
236 std::pair<bool*, const Slice> p{&should_stop, internal_key};
237 TEST_SYNC_POINT_CALLBACK(
238 "CompactionOutputs::ShouldStopBefore::manual_decision", (void*)&p);
239 if (should_stop) {
240 return true;
241 }
242#endif // NDEBUG
243 const uint64_t previous_overlapped_bytes = grandparent_overlapped_bytes_;
244 const InternalKeyComparator* icmp =
245 &compaction_->column_family_data()->internal_comparator();
246 size_t num_grandparent_boundaries_crossed = 0;
247 bool should_stop_for_ttl = false;
248 // Always update grandparent information like overlapped file number, size
249 // etc., and TTL states.
250 // If compaction_->output_level() == 0, there is no need to update grandparent
251 // info, and that `grandparent` should be empty.
252 if (compaction_->output_level() > 0) {
253 num_grandparent_boundaries_crossed =
254 UpdateGrandparentBoundaryInfo(internal_key);
255 should_stop_for_ttl = UpdateFilesToCutForTTLStates(internal_key);
256 }
257
258 if (!HasBuilder()) {
259 return false;
260 }
261
262 if (should_stop_for_ttl) {
263 return true;
264 }
265
266 // If there's user defined partitioner, check that first
267 if (partitioner_ && partitioner_->ShouldPartition(PartitionerRequest(
268 last_key_for_partitioner_, c_iter.user_key(),
269 current_output_file_size_)) == kRequired) {
270 return true;
271 }
272
273 // files output to Level 0 won't be split
274 if (compaction_->output_level() == 0) {
275 return false;
276 }
277
278 // reach the max file size
279 if (current_output_file_size_ >= compaction_->max_output_file_size()) {
280 return true;
281 }
282
283 // Check if it needs to split for RoundRobin
284 // Invalid local_output_split_key indicates that we do not need to split
285 if (local_output_split_key_ != nullptr && !is_split_) {
286 // Split occurs when the next key is larger than/equal to the cursor
287 if (icmp->Compare(internal_key, local_output_split_key_->Encode()) >= 0) {
288 is_split_ = true;
289 return true;
290 }
291 }
292
293 // only check if the current key is going to cross the grandparents file
294 // boundary (either the file beginning or ending).
295 if (num_grandparent_boundaries_crossed > 0) {
296 // Cut the file before the current key if the size of the current output
297 // file + its overlapped grandparent files is bigger than
298 // max_compaction_bytes. Which is to prevent future bigger than
299 // max_compaction_bytes compaction from the current output level.
300 if (grandparent_overlapped_bytes_ + current_output_file_size_ >
301 compaction_->max_compaction_bytes()) {
302 return true;
303 }
304
305 // Cut the file if including the key is going to add a skippable file on
306 // the grandparent level AND its size is reasonably big (1/8 of target file
307 // size). For example, if it's compacting the files L0 + L1:
308 // L0: [1, 21]
309 // L1: [3, 23]
310 // L2: [2, 4] [11, 15] [22, 24]
311 // Without this break, it will output as:
312 // L1: [1,3, 21,23]
313 // With this break, it will output as (assuming [11, 15] at L2 is bigger
314 // than 1/8 of target size):
315 // L1: [1,3] [21,23]
316 // Then for the future compactions, [11,15] won't be included.
317 // For random datasets (either evenly distributed or skewed), it rarely
318 // triggers this condition, but if the user is adding 2 different datasets
319 // without any overlap, it may likely happen.
320 // More details, check PR #1963
321 const size_t num_skippable_boundaries_crossed =
322 being_grandparent_gap_ ? 2 : 3;
323 if (compaction_->immutable_options()->compaction_style ==
324 kCompactionStyleLevel &&
325 num_grandparent_boundaries_crossed >=
326 num_skippable_boundaries_crossed &&
327 grandparent_overlapped_bytes_ - previous_overlapped_bytes >
328 compaction_->target_output_file_size() / 8) {
329 return true;
330 }
331
332 // Pre-cut the output file if it's reaching a certain size AND it's at the
333 // boundary of a grandparent file. It can reduce the future compaction size,
334 // the cost is having smaller files.
335 // The pre-cut size threshold is based on how many grandparent boundaries
336 // it has seen before. Basically, if it has seen no boundary at all, then it
337 // will pre-cut at 50% target file size. Every boundary it has seen
338 // increases the threshold by 5%, max at 90%, which it will always cut.
339 // The idea is based on if it has seen more boundaries before, it will more
340 // likely to see another boundary (file cutting opportunity) before the
341 // target file size. The test shows it can generate larger files than a
342 // static threshold like 75% and has a similar write amplification
343 // improvement.
344 if (compaction_->immutable_options()->compaction_style ==
345 kCompactionStyleLevel &&
346 current_output_file_size_ >=
347 ((compaction_->target_output_file_size() + 99) / 100) *
348 (50 + std::min(grandparent_boundary_switched_num_ * 5,
349 size_t{40}))) {
350 return true;
351 }
352 }
353
354 return false;
355}

357Status CompactionOutputs::AddToOutput(
358 const CompactionIterator& c_iter,
359 const CompactionFileOpenFunc& open_file_func,
360 const CompactionFileCloseFunc& close_file_func) {
361 Status s;
362 bool is_range_del = c_iter.IsDeleteRangeSentinelKey();
363 if (is_range_del && compaction_->bottommost_level()) {
364 // We don't consider range tombstone for bottommost level since:
365 // 1. there is no grandparent and hence no overlap to consider
366 // 2. range tombstone may be dropped at bottommost level.
367 return s;
368 }
369 const Slice& key = c_iter.key();
370 if (ShouldStopBefore(c_iter) && HasBuilder()) {
371 s = close_file_func(*this, c_iter.InputStatus(), key);
372 if (!s.ok()) {
373 return s;
374 }
375 // reset grandparent information
376 grandparent_boundary_switched_num_ = 0;
377 grandparent_overlapped_bytes_ =
378 GetCurrentKeyGrandparentOverlappedBytes(key);
379 if (UNLIKELY(is_range_del)) {
380 // lower bound for this new output file, this is needed as the lower bound
381 // does not come from the smallest point key in this case.
382 range_tombstone_lower_bound_.DecodeFrom(key);
383 } else {
384 range_tombstone_lower_bound_.Clear();
385 }
386 }
387
388 // Open output file if necessary
389 if (!HasBuilder()) {
390 s = open_file_func(*this);
391 if (!s.ok()) {
392 return s;
393 }
394 }
395
396 // c_iter may emit range deletion keys, so update `last_key_for_partitioner_`
397 // here before returning below when `is_range_del` is true
398 if (partitioner_) {
399 last_key_for_partitioner_.assign(c_iter.user_key().data_,
400 c_iter.user_key().size_);
401 }
402
403 if (UNLIKELY(is_range_del)) {
404 return s;
405 }
406
407 assert(builder_ != nullptr);
408 const Slice& value = c_iter.value();
409 s = current_output().validator.Add(key, value);
410 if (!s.ok()) {
411 return s;
412 }
413 builder_->Add(key, value);
414
415 stats_.num_output_records++;
416 current_output_file_size_ = builder_->EstimatedFileSize();
417
418 if (blob_garbage_meter_) {
419 s = blob_garbage_meter_->ProcessOutFlow(key, value);
420 }
421
422 if (!s.ok()) {
423 return s;
424 }
425
426 const ParsedInternalKey& ikey = c_iter.ikey();
427 if (ikey.type == kTypeValuePreferredSeqno) {
428 SequenceNumber preferred_seqno = ParsePackedValueForSeqno(value);
429 smallest_preferred_seqno_ =
430 std::min(smallest_preferred_seqno_, preferred_seqno);
431 }
432 s = current_output().meta.UpdateBoundaries(key, value, ikey.sequence,
433 ikey.type);
434
435 return s;
436}

457Status CompactionOutputs::AddRangeDels(
458 const Slice* comp_start_user_key, const Slice* comp_end_user_key,
459 CompactionIterationStats& range_del_out_stats, bool bottommost_level,
460 const InternalKeyComparator& icmp, SequenceNumber earliest_snapshot,
461 const Slice& next_table_min_key, const std::string& full_history_ts_low) {
462 // The following example does not happen since
463 // CompactionOutput::ShouldStopBefore() always return false for the first
464 // point key. But we should consider removing this dependency. Suppose for the
465 // first compaction output file,
466 // - next_table_min_key.user_key == comp_start_user_key
467 // - no point key is in the output file
468 // - there is a range tombstone @seqno to be added that covers
469 // comp_start_user_key
470 // Then meta.smallest will be set to comp_start_user_key@seqno
471 // and meta.largest will be set to comp_start_user_key@kMaxSequenceNumber
472 // which violates the assumption that meta.smallest should be <= meta.largest.
473 assert(HasRangeDel());
474 FileMetaData& meta = current_output().meta;
475 const Comparator* ucmp = icmp.user_comparator();
476 InternalKey lower_bound_buf, upper_bound_buf;
477 Slice lower_bound_guard, upper_bound_guard;
478 std::string smallest_user_key;
479 const Slice *lower_bound, *upper_bound;
480
481 // We first determine the internal key lower_bound and upper_bound for
482 // this output file. All and only range tombstones that overlap with
483 // [lower_bound, upper_bound] should be added to this file. File
484 // boundaries (meta.smallest/largest) should be updated accordingly when
485 // extended by range tombstones.
486 size_t output_size = outputs_.size();
487 if (output_size == 1) {
488 // This is the first file in the subcompaction.
489 //
490 // When outputting a range tombstone that spans a subcompaction boundary,
491 // the files on either side of that boundary need to include that
492 // boundary's user key. Otherwise, the spanning range tombstone would lose
493 // coverage.
494 //
495 // To achieve this while preventing files from overlapping in internal key
496 // (an LSM invariant violation), we allow the earlier file to include the
497 // boundary user key up to `kMaxSequenceNumber,kTypeRangeDeletion`. The
498 // later file can begin at the boundary user key at the newest key version
499 // it contains. At this point that version number is unknown since we have
500 // not processed the range tombstones yet, so permit any version. Same story
501 // applies to timestamp, and a non-nullptr `comp_start_user_key` should have
502 // `kMaxTs` here, which similarly permits any timestamp.
503 if (comp_start_user_key) {
504 lower_bound_buf.Set(*comp_start_user_key, kMaxSequenceNumber,
505 kTypeRangeDeletion);
506 lower_bound_guard = lower_bound_buf.Encode();
507 lower_bound = &lower_bound_guard;
508 } else {
509 lower_bound = nullptr;
510 }
511 } else {
512 // For subsequent output tables, only include range tombstones from min
513 // key onwards since the previous file was extended to contain range
514 // tombstones falling before min key.
515 if (range_tombstone_lower_bound_.size() > 0) {
516 assert(meta.smallest.size() == 0 ||
517 icmp.Compare(range_tombstone_lower_bound_, meta.smallest) < 0);
518 lower_bound_guard = range_tombstone_lower_bound_.Encode();
519 } else {
520 assert(meta.smallest.size() > 0);
521 lower_bound_guard = meta.smallest.Encode();
522 }
523 lower_bound = &lower_bound_guard;
524 }
525
526 const size_t ts_sz = ucmp->timestamp_size();
527 if (next_table_min_key.empty()) {
528 // Last file of the subcompaction.
529 if (comp_end_user_key) {
530 upper_bound_buf.Set(*comp_end_user_key, kMaxSequenceNumber,
531 kTypeRangeDeletion);
532 upper_bound_guard = upper_bound_buf.Encode();
533 upper_bound = &upper_bound_guard;
534 } else {
535 upper_bound = nullptr;
536 }
537 } else {
538 // There is another file coming whose coverage will begin at
539 // `next_table_min_key`. The current file needs to extend range tombstone
540 // coverage through its own keys (through `meta.largest`) and through user
541 // keys preceding `next_table_min_key`'s user key.
542 ParsedInternalKey next_table_min_key_parsed;
543 ParseInternalKey(next_table_min_key, &next_table_min_key_parsed,
544 false /* log_err_key */)
545 .PermitUncheckedError();
546 assert(next_table_min_key_parsed.sequence < kMaxSequenceNumber);
547 assert(meta.largest.size() == 0 ||
548 icmp.Compare(meta.largest.Encode(), next_table_min_key) < 0);
549 assert(!lower_bound || icmp.Compare(*lower_bound, next_table_min_key) <= 0);
550 if (meta.largest.size() > 0 &&
551 ucmp->EqualWithoutTimestamp(meta.largest.user_key(),
552 next_table_min_key_parsed.user_key)) {
553 // Caution: this assumes meta.largest.Encode() lives longer than
554 // upper_bound, which is only true if meta.largest is never updated.
555 // This just happens to be the case here since meta.largest serves
556 // as the upper_bound.
557 upper_bound_guard = meta.largest.Encode();
558 } else {
559 SetMaxSeqAndTs(upper_bound_buf, next_table_min_key_parsed.user_key,
560 ts_sz);
561 upper_bound_guard = upper_bound_buf.Encode();
562 }
563 upper_bound = &upper_bound_guard;
564 }
565 if (lower_bound && upper_bound &&
566 icmp.Compare(*lower_bound, *upper_bound) > 0) {
567 assert(meta.smallest.size() == 0 &&
568 ucmp->EqualWithoutTimestamp(ExtractUserKey(*lower_bound),
569 ExtractUserKey(*upper_bound)));
570 // This can only happen when lower_bound have the same user key as
571 // next_table_min_key and that there is no point key in the current
572 // compaction output file.
573 return Status::OK();
574 }
575 // The end key of the subcompaction must be bigger or equal to the upper
576 // bound. If the end of subcompaction is null or the upper bound is null,
577 // it means that this file is the last file in the compaction. So there
578 // will be no overlapping between this file and others.
579 assert(comp_end_user_key == nullptr || upper_bound == nullptr ||
580 ucmp->CompareWithoutTimestamp(ExtractUserKey(*upper_bound),
581 *comp_end_user_key) <= 0);
582 auto it = range_del_agg_->NewIterator(lower_bound, upper_bound);
583 Slice last_tombstone_start_user_key{};
584 bool reached_lower_bound = false;
585 const ReadOptions read_options(Env::IOActivity::kCompaction);
586 for (it->SeekToFirst(); it->Valid(); it->Next()) {
587 auto tombstone = it->Tombstone();
588 auto kv = tombstone.Serialize();
589 InternalKey tombstone_end = tombstone.SerializeEndKey();
590 // TODO: the underlying iterator should support clamping the bounds.
591 // tombstone_end.Encode is of form user_key@kMaxSeqno
592 // if it is equal to lower_bound, there is no need to include
593 // such range tombstone.
594 if (!reached_lower_bound && lower_bound &&
595 icmp.Compare(tombstone_end.Encode(), *lower_bound) <= 0) {
596 continue;
597 }
598 assert(!lower_bound ||
599 icmp.Compare(*lower_bound, tombstone_end.Encode()) <= 0);
600 reached_lower_bound = true;
601
602 // Garbage collection for range tombstones.
603 // If user-defined timestamp is enabled, range tombstones are dropped if
604 // they are at bottommost_level, below full_history_ts_low and not visible
605 // in any snapshot. trim_ts_ is passed to the constructor for
606 // range_del_agg_, and range_del_agg_ internally drops tombstones above
607 // trim_ts_.
608 bool consider_drop =
609 tombstone.seq_ <= earliest_snapshot &&
610 (ts_sz == 0 ||
611 (!full_history_ts_low.empty() &&
612 ucmp->CompareTimestamp(tombstone.ts_, full_history_ts_low) < 0));
613 if (consider_drop && bottommost_level) {
614 // TODO(andrewkr): tombstones that span multiple output files are
615 // counted for each compaction output file, so lots of double
616 // counting.
617 range_del_out_stats.num_range_del_drop_obsolete++;
618 range_del_out_stats.num_record_drop_obsolete++;
619 continue;
620 }
621
622 assert(lower_bound == nullptr ||
623 ucmp->CompareWithoutTimestamp(ExtractUserKey(*lower_bound),
624 kv.second) < 0);
625 InternalKey tombstone_start = kv.first;
626 if (lower_bound &&
627 ucmp->CompareWithoutTimestamp(tombstone_start.user_key(),
628 ExtractUserKey(*lower_bound)) < 0) {
629 // This just updates the non-timestamp portion of `tombstone_start`'s user
630 // key. Ideally there would be a simpler API usage
631 ParsedInternalKey tombstone_start_parsed;
632 ParseInternalKey(tombstone_start.Encode(), &tombstone_start_parsed,
633 false /* log_err_key */)
634 .PermitUncheckedError();
635 // timestamp should be from where sequence number is from, which is from
636 // tombstone in this case
637 std::string ts =
638 tombstone_start_parsed.GetTimestamp(ucmp->timestamp_size())
639 .ToString();
640 tombstone_start_parsed.user_key = ExtractUserKey(*lower_bound);
641 tombstone_start.SetFrom(tombstone_start_parsed, ts);
642 }
643 if (upper_bound != nullptr &&
644 icmp.Compare(*upper_bound, tombstone_start.Encode()) < 0) {
645 break;
646 }
647 if (lower_bound &&
648 icmp.Compare(tombstone_start.Encode(), *lower_bound) < 0) {
649 tombstone_start.DecodeFrom(*lower_bound);
650 }
651 if (upper_bound && icmp.Compare(*upper_bound, tombstone_end.Encode()) < 0) {
652 tombstone_end.DecodeFrom(*upper_bound);
653 }
654 if (consider_drop && compaction_->KeyRangeNotExistsBeyondOutputLevel(
655 tombstone_start.user_key(),
656 tombstone_end.user_key(), &level_ptrs_)) {
657 range_del_out_stats.num_range_del_drop_obsolete++;
658 range_del_out_stats.num_record_drop_obsolete++;
659 continue;
660 }
661 // Here we show that *only* range tombstones that overlap with
662 // [lower_bound, upper_bound] are added to the current file, and
663 // sanity checking invariants that should hold:
664 // - [tombstone_start, tombstone_end] overlaps with [lower_bound,
665 // upper_bound]
666 // - meta.smallest <= meta.largest
667 // Corresponding assertions are made, the proof is broken is any of them
668 // fails.
669 // TODO: show that *all* range tombstones that overlap with
670 // [lower_bound, upper_bound] are added.
671 // TODO: some invariant about boundaries are correctly updated.
672 //
673 // Note that `tombstone_start` is updated in the if condition above, we use
674 // tombstone_start to refer to its initial value, i.e.,
675 // it->Tombstone().first, and use tombstone_start* to refer to its value
676 // after the update.
677 //
678 // To show [lower_bound, upper_bound] overlaps with [tombstone_start,
679 // tombstone_end]:
680 // lower_bound <= upper_bound from the if condition right after all
681 // bounds are initialized. We assume each tombstone fragment has
682 // start_key.user_key < end_key.user_key, so
683 // tombstone_start < tombstone_end by
684 // FragmentedTombstoneIterator::Tombstone(). So these two ranges are both
685 // non-emtpy. The flag `reached_lower_bound` and the if logic before it
686 // ensures lower_bound <= tombstone_end. tombstone_start is only updated
687 // if it has a smaller user_key than lower_bound user_key, so
688 // tombstone_start <= tombstone_start*. The above if condition implies
689 // tombstone_start* <= upper_bound. So we have
690 // tombstone_start <= upper_bound and lower_bound <= tombstone_end
691 // and the two ranges overlap.
692 //
693 // To show meta.smallest <= meta.largest:
694 // From the implementation of UpdateBoundariesForRange(), it suffices to
695 // prove that when it is first called in this function, its parameters
696 // satisfy `start <= end`, where start = max(tombstone_start*, lower_bound)
697 // and end = min(tombstone_end, upper_bound). From the above proof we have
698 // lower_bound <= tombstone_end and lower_bound <= upper_bound. We only need
699 // to show that tombstone_start* <= min(tombstone_end, upper_bound).
700 // Note that tombstone_start*.user_key = max(tombstone_start.user_key,
701 // lower_bound.user_key). Assuming tombstone_end always has
702 // kMaxSequenceNumber and lower_bound.seqno < kMaxSequenceNumber.
703 // Since lower_bound <= tombstone_end and lower_bound.seqno <
704 // tombstone_end.seqno (in absolute number order, not internal key order),
705 // lower_bound.user_key < tombstone_end.user_key.
706 // Since lower_bound.user_key < tombstone_end.user_key and
707 // tombstone_start.user_key < tombstone_end.user_key, tombstone_start* <
708 // tombstone_end. Since tombstone_start* <= upper_bound from the above proof
709 // and tombstone_start* < tombstone_end, tombstone_start* <=
710 // min(tombstone_end, upper_bound), so the two ranges overlap.
711
712 // Range tombstone is not supported by output validator yet.
713 builder_->Add(kv.first.Encode(), kv.second);
714 assert(icmp.Compare(tombstone_start, tombstone_end) <= 0);
715 meta.UpdateBoundariesForRange(tombstone_start, tombstone_end,
716 tombstone.seq_, icmp);
717 if (!bottommost_level) {
718 bool start_user_key_changed =
719 last_tombstone_start_user_key.empty() ||
720 ucmp->CompareWithoutTimestamp(last_tombstone_start_user_key,
721 it->start_key()) < 0;
722 last_tombstone_start_user_key = it->start_key();
723 if (start_user_key_changed) {
724 // If tombstone_start >= tombstone_end, then either no key range is
725 // covered, or that they have the same user key. If they have the same
726 // user key, then the internal key range should only be within this
727 // level, and no keys from older levels is covered.
728 if (ucmp->CompareWithoutTimestamp(tombstone_start.user_key(),
729 tombstone_end.user_key()) < 0) {
730 SizeApproximationOptions approx_opts;
731 approx_opts.files_size_error_margin = 0.1;
732 auto approximate_covered_size =
733 compaction_->input_version()->version_set()->ApproximateSize(
734 approx_opts, read_options, compaction_->input_version(),
735 tombstone_start.Encode(), tombstone_end.Encode(),
736 compaction_->output_level() + 1 /* start_level */,
737 -1 /* end_level */, kCompaction);
738 meta.compensated_range_deletion_size += approximate_covered_size;
739 }
740 }
741 }
742 }
743 return Status::OK();
744}

746void CompactionOutputs::FillFilesToCutForTtl() {
747 if (compaction_->immutable_options()->compaction_style !=
748 kCompactionStyleLevel ||
749 compaction_->immutable_options()->compaction_pri !=
750 kMinOverlappingRatio ||
751 compaction_->mutable_cf_options()->ttl == 0 ||
752 compaction_->num_input_levels() < 2 || compaction_->bottommost_level()) {
753 return;
754 }
755
756 // We define new file with the oldest ancestor time to be younger than 1/4
757 // TTL, and an old one to be older than 1/2 TTL time.
758 int64_t temp_current_time;
759 auto get_time_status =
760 compaction_->immutable_options()->clock->GetCurrentTime(
761 &temp_current_time);
762 if (!get_time_status.ok()) {
763 return;
764 }
765
766 auto current_time = static_cast<uint64_t>(temp_current_time);
767 if (current_time < compaction_->mutable_cf_options()->ttl) {
768 return;
769 }
770
771 uint64_t old_age_thres =
772 current_time - compaction_->mutable_cf_options()->ttl / 2;
773 const std::vector<FileMetaData*>& olevel =
774 *(compaction_->inputs(compaction_->num_input_levels() - 1));
775 for (FileMetaData* file : olevel) {
776 // Worth filtering out by start and end?
777 uint64_t oldest_ancester_time = file->TryGetOldestAncesterTime();
778 // We put old files if they are not too small to prevent a flood
779 // of small files.
780 if (oldest_ancester_time < old_age_thres &&
781 file->fd.GetFileSize() >
782 compaction_->mutable_cf_options()->target_file_size_base / 2) {
783 files_to_cut_for_ttl_.push_back(file);
784 }
785 }
786}
