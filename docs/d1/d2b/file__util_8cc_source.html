20// Utility function to copy a file up to a specified length
21IOStatus CopyFile(FileSystem* fs, const std::string& source,
22 Temperature src_temp_hint,
23 std::unique_ptr<WritableFileWriter>& dest_writer,
24 uint64_t size, bool use_fsync,
25 const std::shared_ptr<IOTracer>& io_tracer) {
26 FileOptions soptions;
27 IOStatus io_s;
28 std::unique_ptr<SequentialFileReader> src_reader;
29 const IOOptions opts;
30
31 {
32 soptions.temperature = src_temp_hint;
33 std::unique_ptr<FSSequentialFile> srcfile;
34 io_s = fs->NewSequentialFile(source, soptions, &srcfile, nullptr);
35 if (!io_s.ok()) {
36 return io_s;
37 }
38
39 if (size == 0) {
40 // default argument means copy everything
41 io_s = fs->GetFileSize(source, opts, &size, nullptr);
42 if (!io_s.ok()) {
43 return io_s;
44 }
45 }
46 src_reader.reset(
47 new SequentialFileReader(std::move(srcfile), source, io_tracer));
48 }
49
50 char buffer[4096];
51 Slice slice;
52 while (size > 0) {
53 size_t bytes_to_read = std::min(sizeof(buffer), static_cast<size_t>(size));
54 // TODO: rate limit copy file
55 io_s = status_to_io_status(
56 src_reader->Read(bytes_to_read, &slice, buffer,
57 Env::IO_TOTAL /* rate_limiter_priority */));
58 if (!io_s.ok()) {
59 return io_s;
60 }
61 if (slice.size() == 0) {
62 return IOStatus::Corruption(
63 "File smaller than expected for copy: " + source + " expecting " +
64 std::to_string(size) + " more bytes after " +
65 std::to_string(dest_writer->GetFileSize()));
66 }
67
68 io_s = dest_writer->Append(opts, slice);
69 if (!io_s.ok()) {
70 return io_s;
71 }
72 size -= slice.size();
73 }
74 return dest_writer->Sync(opts, use_fsync);
75}

77IOStatus CopyFile(FileSystem* fs, const std::string& source,
78 Temperature src_temp_hint, const std::string& destination,
79 Temperature dst_temp, uint64_t size, bool use_fsync,
80 const std::shared_ptr<IOTracer>& io_tracer) {
81 FileOptions options;
82 IOStatus io_s;
83 std::unique_ptr<WritableFileWriter> dest_writer;
84
85 {
86 options.temperature = dst_temp;
87 std::unique_ptr<FSWritableFile> destfile;
88 io_s = fs->NewWritableFile(destination, options, &destfile, nullptr);
89 if (!io_s.ok()) {
90 return io_s;
91 }
92
93 // TODO: pass in Histograms if the destination file is sst or blob
94 dest_writer.reset(
95 new WritableFileWriter(std::move(destfile), destination, options));
96 }
97
98 return CopyFile(fs, source, src_temp_hint, dest_writer, size, use_fsync,
99 io_tracer);
100}

102// Utility function to create a file with the provided contents
103IOStatus CreateFile(FileSystem* fs, const std::string& destination,
104 const std::string& contents, bool use_fsync) {
105 const EnvOptions soptions;
106 IOStatus io_s;
107 std::unique_ptr<WritableFileWriter> dest_writer;
108 const IOOptions opts;
109
110 std::unique_ptr<FSWritableFile> destfile;
111 io_s = fs->NewWritableFile(destination, soptions, &destfile, nullptr);
112 if (!io_s.ok()) {
113 return io_s;
114 }
115 // TODO: pass in Histograms if the destination file is sst or blob
116 dest_writer.reset(
117 new WritableFileWriter(std::move(destfile), destination, soptions));
118 io_s = dest_writer->Append(opts, Slice(contents));
119 if (!io_s.ok()) {
120 return io_s;
121 }
122 return dest_writer->Sync(opts, use_fsync);
123}

152// requested_checksum_func_name brings the function name of the checksum
153// generator in checksum_factory. Empty string is permitted, in which case the
154// name of the generator created by the factory is unchecked. When
155// `requested_checksum_func_name` is non-empty, however, the created generator's
156// name must match it, otherwise an `InvalidArgument` error is returned.
157IOStatus GenerateOneFileChecksum(
158 FileSystem* fs, const std::string& file_path,
159 FileChecksumGenFactory* checksum_factory,
160 const std::string& requested_checksum_func_name, std::string* file_checksum,
161 std::string* file_checksum_func_name,
162 size_t verify_checksums_readahead_size, bool /*allow_mmap_reads*/,
163 std::shared_ptr<IOTracer>& io_tracer, RateLimiter* rate_limiter,
164 const ReadOptions& read_options, Statistics* stats, SystemClock* clock) {
165 if (checksum_factory == nullptr) {
166 return IOStatus::InvalidArgument("Checksum factory is invalid");
167 }
168 assert(file_checksum != nullptr);
169 assert(file_checksum_func_name != nullptr);
170
171 FileChecksumGenContext gen_context;
172 gen_context.requested_checksum_func_name = requested_checksum_func_name;
173 gen_context.file_name = file_path;
174 std::unique_ptr<FileChecksumGenerator> checksum_generator =
175 checksum_factory->CreateFileChecksumGenerator(gen_context);
176 if (checksum_generator == nullptr) {
177 std::string msg =
178 "Cannot get the file checksum generator based on the requested "
179 "checksum function name: " +
180 requested_checksum_func_name +
181 " from checksum factory: " + checksum_factory->Name();
182 return IOStatus::InvalidArgument(msg);
183 } else {
184 // For backward compatibility and use in file ingestion clients where there
185 // is no stored checksum function name, `requested_checksum_func_name` can
186 // be empty. If we give the requested checksum function name, we expect it
187 // is the same name of the checksum generator.
188 if (!requested_checksum_func_name.empty() &&
189 checksum_generator->Name() != requested_checksum_func_name) {
190 std::string msg = "Expected file checksum generator named '" +
191 requested_checksum_func_name +
192 "', while the factory created one "
193 "named '" +
194 checksum_generator->Name() + "'";
195 return IOStatus::InvalidArgument(msg);
196 }
197 }
198
199 uint64_t size;
200 IOStatus io_s;
201 std::unique_ptr<RandomAccessFileReader> reader;
202 {
203 std::unique_ptr<FSRandomAccessFile> r_file;
204 io_s = fs->NewRandomAccessFile(file_path, FileOptions(), &r_file, nullptr);
205 if (!io_s.ok()) {
206 return io_s;
207 }
208 io_s = fs->GetFileSize(file_path, IOOptions(), &size, nullptr);
209 if (!io_s.ok()) {
210 return io_s;
211 }
212 reader.reset(new RandomAccessFileReader(
213 std::move(r_file), file_path, clock, io_tracer, stats,
214 Histograms::SST_READ_MICROS, nullptr, rate_limiter));
215 }
216
217 // Found that 256 KB readahead size provides the best performance, based on
218 // experiments, for auto readahead. Experiment data is in PR #3282.
219 size_t default_max_read_ahead_size = 256 * 1024;
220 size_t readahead_size = (verify_checksums_readahead_size != 0)
221 ? verify_checksums_readahead_size
222 : default_max_read_ahead_size;
223 std::unique_ptr<char[]> buf;
224 if (reader->use_direct_io()) {
225 size_t alignment = reader->file()->GetRequiredBufferAlignment();
226 readahead_size = (readahead_size + alignment - 1) & ~(alignment - 1);
227 }
228 buf.reset(new char[readahead_size]);
229
230 Slice slice;
231 uint64_t offset = 0;
232 IOOptions opts;
233 io_s = reader->PrepareIOOptions(read_options, opts);
234 if (!io_s.ok()) {
235 return io_s;
236 }
237 while (size > 0) {
238 size_t bytes_to_read =
239 static_cast<size_t>(std::min(uint64_t{readahead_size}, size));
240 io_s =
241 reader->Read(opts, offset, bytes_to_read, &slice, buf.get(), nullptr);
242 if (!io_s.ok()) {
243 return IOStatus::Corruption("file read failed with error: " +
244 io_s.ToString());
245 }
246 if (slice.size() == 0) {
247 return IOStatus::Corruption(
248 "File smaller than expected for checksum: " + file_path +
249 " expecting " + std::to_string(size) + " more bytes after " +
250 std::to_string(offset));
251 }
252 checksum_generator->Update(slice.data(), slice.size());
253 size -= slice.size();
254 offset += slice.size();
255
256 TEST_SYNC_POINT("GenerateOneFileChecksum::Chunk:0");
257 }
258 checksum_generator->Finalize();
259 *file_checksum = checksum_generator->GetChecksum();
260 *file_checksum_func_name = checksum_generator->Name();
261 return IOStatus::OK();
262}

264Status DestroyDir(Env* env, const std::string& dir) {
265 Status s;
266 if (env->FileExists(dir).IsNotFound()) {
267 return s;
268 }
269 std::vector<std::string> files_in_dir;
270 s = env->GetChildren(dir, &files_in_dir);
271 if (s.ok()) {
272 for (auto& file_in_dir : files_in_dir) {
273 std::string path = dir + "/" + file_in_dir;
274 bool is_dir = false;
275 s = env->IsDirectory(path, &is_dir);
276 if (s.ok()) {
277 if (is_dir) {
278 s = DestroyDir(env, path);
279 } else {
280 s = env->DeleteFile(path);
281 }
282 } else if (s.IsNotSupported()) {
283 s = Status::OK();
284 }
285 if (!s.ok()) {
286 // IsDirectory, etc. might not report NotFound
287 if (s.IsNotFound() || env->FileExists(path).IsNotFound()) {
288 // Allow files to be deleted externally
289 s = Status::OK();
290 } else {
291 break;
292 }
293 }
294 }
295 }
296
297 if (s.ok()) {
298 s = env->DeleteDir(dir);
299 // DeleteDir might or might not report NotFound
300 if (!s.ok() && (s.IsNotFound() || env->FileExists(dir).IsNotFound())) {
301 // Allow to be deleted externally
302 s = Status::OK();
303 }
304 }
305 return s;
306}
