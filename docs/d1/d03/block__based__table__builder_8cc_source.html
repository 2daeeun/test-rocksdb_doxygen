67// Create a filter block builder based on its type.
68FilterBlockBuilder* CreateFilterBlockBuilder(
69 const ImmutableCFOptions& /*opt*/, const MutableCFOptions& mopt,
70 const FilterBuildingContext& context,
71 const bool use_delta_encoding_for_index_values,
72 PartitionedIndexBuilder* const p_index_builder, size_t ts_sz,
73 const bool persist_user_defined_timestamps) {
74 const BlockBasedTableOptions& table_opt = context.table_options;
75 assert(table_opt.filter_policy); // precondition
76
77 FilterBitsBuilder* filter_bits_builder =
78 BloomFilterPolicy::GetBuilderFromContext(context);
79 if (filter_bits_builder == nullptr) {
80 return nullptr;
81 } else {
82 if (table_opt.partition_filters) {
83 assert(p_index_builder != nullptr);
84 // Since after partition cut request from filter builder it takes time
85 // until index builder actully cuts the partition, until the end of a
86 // data block potentially with many keys, we take the lower bound as
87 // partition size.
88 assert(table_opt.block_size_deviation <= 100);
89 auto partition_size =
90 static_cast<uint32_t>(((table_opt.metadata_block_size *
91 (100 - table_opt.block_size_deviation)) +
92 99) /
93 100);
94 partition_size = std::max(partition_size, static_cast<uint32_t>(1));
95 return new PartitionedFilterBlockBuilder(
96 mopt.prefix_extractor.get(), table_opt.whole_key_filtering,
97 filter_bits_builder, table_opt.index_block_restart_interval,
98 use_delta_encoding_for_index_values, p_index_builder, partition_size,
99 ts_sz, persist_user_defined_timestamps,
100 table_opt.decouple_partitioned_filters);
101 } else {
102 return new FullFilterBlockBuilder(mopt.prefix_extractor.get(),
103 table_opt.whole_key_filtering,
104 filter_bits_builder);
105 }
106 }

109bool GoodCompressionRatio(size_t compressed_size, size_t uncomp_size,
110 int max_compressed_bytes_per_kb) {
111 // For efficiency, avoid floating point and division
112 return compressed_size <=
113 (static_cast<uint64_t>(max_compressed_bytes_per_kb) * uncomp_size) >>
114 10;

119// format_version is the block format as defined in include/rocksdb/table.h
120Slice CompressBlock(const Slice& uncompressed_data, const CompressionInfo& info,
121 CompressionType* type, uint32_t format_version,
122 bool allow_sample, std::string* compressed_output,
123 std::string* sampled_output_fast,
124 std::string* sampled_output_slow) {
125 assert(type);
126 assert(compressed_output);
127 assert(compressed_output->empty());
128
129 // If requested, we sample one in every N block with a
130 // fast and slow compression algorithm and report the stats.
131 // The users can use these stats to decide if it is worthwhile
132 // enabling compression and they also get a hint about which
133 // compression algorithm wil be beneficial.
134 if (allow_sample && info.SampleForCompression() &&
135 Random::GetTLSInstance()->OneIn(
136 static_cast<int>(info.SampleForCompression()))) {
137 // Sampling with a fast compression algorithm
138 if (sampled_output_fast && (LZ4_Supported() || Snappy_Supported())) {
139 CompressionType c =
140 LZ4_Supported() ? kLZ4Compression : kSnappyCompression;
141 CompressionOptions options;
142 CompressionContext context(c, options);
143 CompressionInfo info_tmp(options, context,
144 CompressionDict::GetEmptyDict(), c,
145 info.SampleForCompression());
146
147 CompressData(uncompressed_data, info_tmp,
148 GetCompressFormatForVersion(format_version),
149 sampled_output_fast);
150 }
151
152 // Sampling with a slow but high-compression algorithm
153 if (sampled_output_slow && (ZSTD_Supported() || Zlib_Supported())) {
154 CompressionType c = ZSTD_Supported() ? kZSTD : kZlibCompression;
155 CompressionOptions options;
156 CompressionContext context(c, options);
157 CompressionInfo info_tmp(options, context,
158 CompressionDict::GetEmptyDict(), c,
159 info.SampleForCompression());
160
161 CompressData(uncompressed_data, info_tmp,
162 GetCompressFormatForVersion(format_version),
163 sampled_output_slow);
164 }
165 }
166
167 int max_compressed_bytes_per_kb = info.options().max_compressed_bytes_per_kb;
168 if (info.type() == kNoCompression || max_compressed_bytes_per_kb <= 0) {
169 *type = kNoCompression;
170 return uncompressed_data;
171 }
172
173 // Actually compress the data; if the compression method is not supported,
174 // or the compression fails etc., just fall back to uncompressed
175 if (!CompressData(uncompressed_data, info,
176 GetCompressFormatForVersion(format_version),
177 compressed_output)) {
178 *type = kNoCompression;
179 return uncompressed_data;
180 }
181
182 // Check the compression ratio; if it's not good enough, just fall back to
183 // uncompressed
184 if (!GoodCompressionRatio(compressed_output->size(), uncompressed_data.size(),
185 max_compressed_bytes_per_kb)) {
186 *type = kNoCompression;
187 return uncompressed_data;
188 }
189
190 *type = info.type();
191 return *compressed_output;
192}

326 // States of the builder.
327 //
328 // - `kBuffered`: This is the initial state where zero or more data blocks are
329 // accumulated uncompressed in-memory. From this state, call
330 // `EnterUnbuffered()` to finalize the compression dictionary if enabled,
331 // compress/write out any buffered blocks, and proceed to the `kUnbuffered`
332 // state.
333 //
334 // - `kUnbuffered`: This is the state when compression dictionary is finalized
335 // either because it wasn't enabled in the first place or it's been created
336 // from sampling previously buffered data. In this state, blocks are simply
337 // compressed/written out as they fill up. From this state, call `Finish()`
338 // to complete the file (write meta-blocks, etc.), or `Abandon()` to delete
339 // the partially created file.
340 //
341 // - `kClosed`: This indicates either `Finish()` or `Abandon()` has been
342 // called, so the table builder is no longer usable. We must be in this

673 // releasing string memories during vector clear()

676 public:
677 Keys() : keys_(kKeysInitSize), size_(0) {}

682 keys_[size_].assign(key.data(), key.size());

684 size_++;
685 }

687 size_ = keys.size();

689 }
690 void Clear() { size_ = 0; }

692 std::string& Back() { return keys_[size_ - 1]; }
693 std::string& operator[](size_t idx) {
694 assert(idx < size_);

763 // Estimate output file size when parallel compression is enabled. This is
764 // necessary because compression & flush are no longer synchronized,
765 // and BlockBasedTableBuilder::FileSize() is no longer accurate.
766 // memory_order_relaxed suffices because accurate statistics is not required.
767 class FileSizeEstimator {
768 public:
769 explicit FileSizeEstimator()
770 : uncomp_bytes_compressed(0),
771 uncomp_bytes_curr_block(0),
772 uncomp_bytes_curr_block_set(false),
773 uncomp_bytes_inflight(0),
774 blocks_inflight(0),
775 curr_compression_ratio(0),
776 estimated_file_size(0) {}

778 // Estimate file size when a block is about to be emitted to
779 // compression thread
780 void EmitBlock(uint64_t uncomp_block_size, uint64_t curr_file_size) {
781 uint64_t new_uncomp_bytes_inflight =
782 uncomp_bytes_inflight.fetch_add(uncomp_block_size,
783 std::memory_order_relaxed) +
784 uncomp_block_size;
785
786 uint64_t new_blocks_inflight =
787 blocks_inflight.fetch_add(1, std::memory_order_relaxed) + 1;
788
789 estimated_file_size.store(
790 curr_file_size +
791 static_cast<uint64_t>(
792 static_cast<double>(new_uncomp_bytes_inflight) *
793 curr_compression_ratio.load(std::memory_order_relaxed)) +
794 new_blocks_inflight * kBlockTrailerSize,
795 std::memory_order_relaxed);
796 }
797
798 // Estimate file size when a block is already reaped from
799 // compression thread
800 void ReapBlock(uint64_t compressed_block_size, uint64_t curr_file_size) {
801 assert(uncomp_bytes_curr_block_set);
802
803 uint64_t new_uncomp_bytes_compressed =
804 uncomp_bytes_compressed + uncomp_bytes_curr_block;
805 assert(new_uncomp_bytes_compressed > 0);
806
807 curr_compression_ratio.store(
808 (curr_compression_ratio.load(std::memory_order_relaxed) *
809 uncomp_bytes_compressed +
810 compressed_block_size) /
811 static_cast<double>(new_uncomp_bytes_compressed),
812 std::memory_order_relaxed);
813 uncomp_bytes_compressed = new_uncomp_bytes_compressed;
814
815 uint64_t new_uncomp_bytes_inflight =
816 uncomp_bytes_inflight.fetch_sub(uncomp_bytes_curr_block,
817 std::memory_order_relaxed) -
818 uncomp_bytes_curr_block;
819
820 uint64_t new_blocks_inflight =
821 blocks_inflight.fetch_sub(1, std::memory_order_relaxed) - 1;
822
823 estimated_file_size.store(
824 curr_file_size +
825 static_cast<uint64_t>(
826 static_cast<double>(new_uncomp_bytes_inflight) *
827 curr_compression_ratio.load(std::memory_order_relaxed)) +
828 new_blocks_inflight * kBlockTrailerSize,
829 std::memory_order_relaxed);
830
831 uncomp_bytes_curr_block_set = false;
832 }
833
834 void SetEstimatedFileSize(uint64_t size) {
835 estimated_file_size.store(size, std::memory_order_relaxed);
836 }
837
838 uint64_t GetEstimatedFileSize() {
839 return estimated_file_size.load(std::memory_order_relaxed);
840 }
841
842 void SetCurrBlockUncompSize(uint64_t size) {
843 uncomp_bytes_curr_block = size;
844 uncomp_bytes_curr_block_set = true;
845 }
846
847 private:
848 // Input bytes compressed so far.
849 uint64_t uncomp_bytes_compressed;
850 // Size of current block being appended.
851 uint64_t uncomp_bytes_curr_block;
852 // Whether uncomp_bytes_curr_block has been set for next
853 // ReapBlock call.
854 bool uncomp_bytes_curr_block_set;
855 // Input bytes under compression and not appended yet.
856 std::atomic<uint64_t> uncomp_bytes_inflight;
857 // Number of blocks under compression and not appended yet.
858 std::atomic<uint64_t> blocks_inflight;
859 // Current compression ratio, maintained by BGWorkWriteMaybeCompressedBlock.
860 std::atomic<double> curr_compression_ratio;
861 // Estimated SST file size.
862 std::atomic<uint64_t> estimated_file_size;
863 };

896 // Make a block prepared to be emitted to compression thread
897 // Used in non-buffered mode
898 BlockRep* PrepareBlock(CompressionType compression_type,
899 const Slice* first_key_in_next_block,
900 BlockBuilder* data_block) {
901 BlockRep* block_rep =
902 PrepareBlockInternal(compression_type, first_key_in_next_block);
903 assert(block_rep != nullptr);
904 data_block->SwapAndReset(*(block_rep->data));
905 block_rep->contents = *(block_rep->data);
906 std::swap(block_rep->keys, curr_block_keys);
907 curr_block_keys->Clear();
908 return block_rep;
909 }
910
911 // Used in EnterUnbuffered
912 BlockRep* PrepareBlock(CompressionType compression_type,
913 const Slice* first_key_in_next_block,
914 std::string* data_block,
915 std::vector<std::string>* keys) {
916 BlockRep* block_rep =
917 PrepareBlockInternal(compression_type, first_key_in_next_block);
918 assert(block_rep != nullptr);
919 std::swap(*(block_rep->data), *data_block);
920 block_rep->contents = *(block_rep->data);
921 block_rep->keys->SwapAssign(*keys);
922 return block_rep;
923 }
924
925 // Emit a block to compression thread
926 void EmitBlock(BlockRep* block_rep) {
927 assert(block_rep != nullptr);
928 assert(block_rep->status.ok());
929 if (!write_queue.push(block_rep->slot.get())) {
930 return;
931 }
932 if (!compress_queue.push(block_rep)) {
933 return;
934 }
935
936 if (!first_block_processed.load(std::memory_order_relaxed)) {
937 std::unique_lock<std::mutex> lock(first_block_mutex);
938 first_block_cond.wait(lock, [this] {
939 return first_block_processed.load(std::memory_order_relaxed);
940 });
941 }
942 }
943
944 // Reap a block from compression thread
945 void ReapBlock(BlockRep* block_rep) {
946 assert(block_rep != nullptr);
947 block_rep->compressed_data->clear();
948 block_rep_pool.push(block_rep);
949
950 if (!first_block_processed.load(std::memory_order_relaxed)) {
951 std::lock_guard<std::mutex> lock(first_block_mutex);
952 first_block_processed.store(true, std::memory_order_relaxed);
953 first_block_cond.notify_one();
954 }
955 }
956
957 private:
958 BlockRep* PrepareBlockInternal(CompressionType compression_type,
959 const Slice* first_key_in_next_block) {
960 BlockRep* block_rep = nullptr;
961 block_rep_pool.pop(block_rep);
962 assert(block_rep != nullptr);
963
964 assert(block_rep->data);
965
966 block_rep->compression_type = compression_type;
967
968 if (first_key_in_next_block == nullptr) {
969 block_rep->first_key_in_next_block.reset(nullptr);
970 } else {
971 block_rep->first_key_in_next_block->assign(
972 first_key_in_next_block->data(), first_key_in_next_block->size());
973 }
974
975 return block_rep;
976 }

1016void BlockBasedTableBuilder::Add(const Slice& ikey, const Slice& value) {
1017 Rep* r = rep_;
1018 assert(rep_->state != Rep::State::kClosed);
1019 if (!ok()) {
1020 return;
1021 }
1022 ValueType value_type;
1023 SequenceNumber seq;
1024 UnPackSequenceAndType(ExtractInternalKeyFooter(ikey), &seq, &value_type);
1025 r->props.key_largest_seqno = std::max(r->props.key_largest_seqno, seq);
1026 if (IsValueType(value_type)) {
1027#ifndef NDEBUG
1028 if (r->props.num_entries > r->props.num_range_deletions) {
1029 assert(r->internal_comparator.Compare(ikey, Slice(r->last_ikey)) > 0);
1030 }
1031#endif // !NDEBUG
1032
1033 auto should_flush = r->flush_block_policy->Update(ikey, value);
1034 if (should_flush) {
1035 assert(!r->data_block.empty());
1036 r->first_key_in_next_block = &ikey;
1037 Flush();
1038 if (r->state == Rep::State::kBuffered) {
1039 bool exceeds_buffer_limit =
1040 (r->buffer_limit != 0 && r->data_begin_offset > r->buffer_limit);
1041 bool exceeds_global_block_cache_limit = false;
1042
1043 // Increase cache charging for the last buffered data block
1044 // only if the block is not going to be unbuffered immediately
1045 // and there exists a cache reservation manager
1046 if (!exceeds_buffer_limit &&
1047 r->compression_dict_buffer_cache_res_mgr != nullptr) {
1048 Status s =
1049 r->compression_dict_buffer_cache_res_mgr->UpdateCacheReservation(
1050 r->data_begin_offset);
1051 exceeds_global_block_cache_limit = s.IsMemoryLimit();
1052 }
1053
1054 if (exceeds_buffer_limit || exceeds_global_block_cache_limit) {
1055 EnterUnbuffered();
1056 }
1057 }
1058
1059 // Add item to index block.
1060 // We do not emit the index entry for a block until we have seen the
1061 // first key for the next data block. This allows us to use shorter
1062 // keys in the index block. For example, consider a block boundary
1063 // between the keys "the quick brown fox" and "the who". We can use
1064 // "the r" as the key for the index block entry since it is >= all
1065 // entries in the first block and < all entries in subsequent
1066 // blocks.
1067 if (ok() && r->state == Rep::State::kUnbuffered) {
1068 if (r->IsParallelCompressionEnabled()) {
1069 r->pc_rep->curr_block_keys->Clear();
1070 } else {
1071 r->index_builder->AddIndexEntry(r->last_ikey, &ikey,
1072 r->pending_handle,
1073 &r->index_separator_scratch);
1074 }
1075 }
1076 }
1077
1078 // Note: PartitionedFilterBlockBuilder requires key being added to filter
1079 // builder after being added to index builder.
1080 if (r->state == Rep::State::kUnbuffered) {
1081 if (r->IsParallelCompressionEnabled()) {
1082 r->pc_rep->curr_block_keys->PushBack(ikey);
1083 } else {
1084 if (r->filter_builder != nullptr) {
1085 r->filter_builder->AddWithPrevKey(
1086 ExtractUserKeyAndStripTimestamp(ikey, r->ts_sz),
1087 r->last_ikey.empty()
1088 ? Slice{}
1089 : ExtractUserKeyAndStripTimestamp(r->last_ikey, r->ts_sz));
1090 }
1091 }
1092 }
1093
1094 r->data_block.AddWithLastKey(ikey, value, r->last_ikey);
1095 r->last_ikey.assign(ikey.data(), ikey.size());
1096 assert(!r->last_ikey.empty());
1097 if (r->state == Rep::State::kBuffered) {
1098 // Buffered keys will be replayed from data_block_buffers during
1099 // `Finish()` once compression dictionary has been finalized.
1100 } else {
1101 if (!r->IsParallelCompressionEnabled()) {
1102 r->index_builder->OnKeyAdded(ikey);
1103 }
1104 }
1105 // TODO offset passed in is not accurate for parallel compression case
1106 NotifyCollectTableCollectorsOnAdd(ikey, value, r->get_offset(),
1107 r->table_properties_collectors,
1108 r->ioptions.logger);
1109
1110 } else if (value_type == kTypeRangeDeletion) {
1111 Slice persisted_end = value;
1112 // When timestamps should not be persisted, we physically strip away range
1113 // tombstone end key's user timestamp before passing it along to block
1114 // builder. Physically stripping away start key's user timestamp is
1115 // handled at the block builder level in the same way as the other data
1116 // blocks.
1117 if (r->ts_sz > 0 && !r->persist_user_defined_timestamps) {
1118 persisted_end = StripTimestampFromUserKey(value, r->ts_sz);
1119 }
1120 r->range_del_block.Add(ikey, persisted_end);
1121 // TODO offset passed in is not accurate for parallel compression case
1122 NotifyCollectTableCollectorsOnAdd(ikey, value, r->get_offset(),
1123 r->table_properties_collectors,
1124 r->ioptions.logger);
1125 } else {
1126 assert(false);
1127 r->SetStatus(Status::InvalidArgument(
1128 "BlockBasedBuilder::Add() received a key with invalid value type " +
1129 std::to_string(static_cast<unsigned int>(value_type))));
1130 return;
1131 }
1132
1133 r->props.num_entries++;
1134 r->props.raw_key_size += ikey.size();
1135 if (!r->persist_user_defined_timestamps) {
1136 r->props.raw_key_size -= r->ts_sz;
1137 }
1138 r->props.raw_value_size += value.size();
1139 if (value_type == kTypeDeletion || value_type == kTypeSingleDeletion ||
1140 value_type == kTypeDeletionWithTimestamp) {
1141 r->props.num_deletions++;
1142 } else if (value_type == kTypeRangeDeletion) {
1143 r->props.num_deletions++;
1144 r->props.num_range_deletions++;
1145 } else if (value_type == kTypeMerge) {
1146 r->props.num_merge_operands++;
1147 }
1148}

1150void BlockBasedTableBuilder::Flush() {
1151 Rep* r = rep_;
1152 assert(rep_->state != Rep::State::kClosed);
1153 if (!ok()) {
1154 return;
1155 }
1156 if (r->data_block.empty()) {
1157 return;
1158 }
1159 if (r->IsParallelCompressionEnabled() &&
1160 r->state == Rep::State::kUnbuffered) {
1161 r->data_block.Finish();
1162 ParallelCompressionRep::BlockRep* block_rep = r->pc_rep->PrepareBlock(
1163 r->compression_type, r->first_key_in_next_block, &(r->data_block));
1164 assert(block_rep != nullptr);
1165 r->pc_rep->file_size_estimator.EmitBlock(block_rep->data->size(),
1166 r->get_offset());
1167 r->pc_rep->EmitBlock(block_rep);
1168 } else {
1169 WriteBlock(&r->data_block, &r->pending_handle, BlockType::kData);
1170 }

1235 const Slice& uncompressed_block_data, bool is_data_block,
1236 const CompressionContext& compression_ctx, UncompressionContext* verify_ctx,
1237 std::string* compressed_output, Slice* block_contents,
1238 CompressionType* type, Status* out_status) {
1239 Rep* r = rep_;
1240 bool is_status_ok = ok();
1241 if (!r->IsParallelCompressionEnabled()) {
1242 assert(is_status_ok);
1243 }
1244
1245 if (is_status_ok && uncompressed_block_data.size() < kCompressionSizeLimit) {
1246 StopWatchNano timer(
1247 r->ioptions.clock,
1248 ShouldReportDetailedTime(r->ioptions.env, r->ioptions.stats));
1249
1250 if (is_data_block) {
1251 r->compressible_input_data_bytes.fetch_add(uncompressed_block_data.size(),
1252 std::memory_order_relaxed);
1253 }
1254 const CompressionDict* compression_dict;
1255 if (!is_data_block || r->compression_dict == nullptr) {
1256 compression_dict = &CompressionDict::GetEmptyDict();
1257 } else {
1258 compression_dict = r->compression_dict.get();
1259 }
1260 assert(compression_dict != nullptr);
1261 CompressionInfo compression_info(r->compression_opts, compression_ctx,
1262 *compression_dict, r->compression_type,
1263 r->sample_for_compression);
1264
1265 std::string sampled_output_fast;
1266 std::string sampled_output_slow;
1267 *block_contents = CompressBlock(
1268 uncompressed_block_data, compression_info, type,
1269 r->table_options.format_version, is_data_block /* allow_sample */,
1270 compressed_output, &sampled_output_fast, &sampled_output_slow);
1271
1272 if (sampled_output_slow.size() > 0 || sampled_output_fast.size() > 0) {
1273 // Currently compression sampling is only enabled for data block.
1274 assert(is_data_block);
1275 r->sampled_input_data_bytes.fetch_add(uncompressed_block_data.size(),
1276 std::memory_order_relaxed);
1277 r->sampled_output_slow_data_bytes.fetch_add(sampled_output_slow.size(),
1278 std::memory_order_relaxed);
1279 r->sampled_output_fast_data_bytes.fetch_add(sampled_output_fast.size(),
1280 std::memory_order_relaxed);
1281 }
1282 // notify collectors on block add
1283 NotifyCollectTableCollectorsOnBlockAdd(
1284 r->table_properties_collectors, uncompressed_block_data.size(),
1285 sampled_output_fast.size(), sampled_output_slow.size());
1286
1287 // Some of the compression algorithms are known to be unreliable. If
1288 // the verify_compression flag is set then try to de-compress the
1289 // compressed data and compare to the input.
1290 if (*type != kNoCompression && r->table_options.verify_compression) {
1291 // Retrieve the uncompressed contents into a new buffer
1292 const UncompressionDict* verify_dict;
1293 if (!is_data_block || r->verify_dict == nullptr) {
1294 verify_dict = &UncompressionDict::GetEmptyDict();
1295 } else {
1296 verify_dict = r->verify_dict.get();
1297 }
1298 assert(verify_dict != nullptr);
1299 BlockContents contents;
1300 UncompressionInfo uncompression_info(*verify_ctx, *verify_dict,
1301 r->compression_type);
1302 Status uncompress_status = UncompressBlockData(
1303 uncompression_info, block_contents->data(), block_contents->size(),
1304 &contents, r->table_options.format_version, r->ioptions);
1305
1306 if (uncompress_status.ok()) {
1307 bool data_match = contents.data.compare(uncompressed_block_data) == 0;
1308 if (!data_match) {
1309 // The result of the compression was invalid. abort.
1310 const char* const msg =
1311 "Decompressed block did not match pre-compression block";
1312 ROCKS_LOG_ERROR(r->ioptions.logger, "%s", msg);
1313 *out_status = Status::Corruption(msg);
1314 *type = kNoCompression;
1315 }
1316 } else {
1317 // Decompression reported an error. abort.
1318 *out_status = Status::Corruption(std::string("Could not decompress: ") +
1319 uncompress_status.getState());
1320 *type = kNoCompression;
1321 }
1322 }
1323 if (timer.IsStarted()) {
1324 RecordTimeToHistogram(r->ioptions.stats, COMPRESSION_TIMES_NANOS,
1325 timer.ElapsedNanos());
1326 }
1327 } else {
1328 // Status is not OK, or block is too big to be compressed.
1329 if (is_data_block) {
1330 r->uncompressible_input_data_bytes.fetch_add(
1331 uncompressed_block_data.size(), std::memory_order_relaxed);
1332 }
1333 *type = kNoCompression;
1334 }
1335 if (is_data_block) {
1336 r->uncompressible_input_data_bytes.fetch_add(kBlockTrailerSize,
1337 std::memory_order_relaxed);
1338 }
1339
1340 // Abort compression if the block is too big, or did not pass
1341 // verification.
1342 if (*type == kNoCompression) {
1343 *block_contents = uncompressed_block_data;
1344 bool compression_attempted = !compressed_output->empty();
1345 RecordTick(r->ioptions.stats, compression_attempted
1346 ? NUMBER_BLOCK_COMPRESSION_REJECTED
1347 : NUMBER_BLOCK_COMPRESSION_BYPASSED);
1348 RecordTick(r->ioptions.stats,
1349 compression_attempted ? BYTES_COMPRESSION_REJECTED
1350 : BYTES_COMPRESSION_BYPASSED,
1351 uncompressed_block_data.size());
1352 } else {
1353 RecordTick(r->ioptions.stats, NUMBER_BLOCK_COMPRESSED);
1354 RecordTick(r->ioptions.stats, BYTES_COMPRESSED_FROM,
1355 uncompressed_block_data.size());
1356 RecordTick(r->ioptions.stats, BYTES_COMPRESSED_TO,
1357 compressed_output->size());
1358 }
1359}

1361void BlockBasedTableBuilder::WriteMaybeCompressedBlock(
1362 const Slice& block_contents, CompressionType comp_type, BlockHandle* handle,
1363 BlockType block_type, const Slice* uncompressed_block_data) {
1364 // File format contains a sequence of blocks where each block has:
1365 // block_data: uint8[n]
1366 // compression_type: uint8
1367 // checksum: uint32
1368 Rep* r = rep_;
1369 bool is_data_block = block_type == BlockType::kData;
1370 IOOptions io_options;
1371 IOStatus io_s =
1372 WritableFileWriter::PrepareIOOptions(r->write_options, io_options);
1373 if (!io_s.ok()) {
1374 r->SetIOStatus(io_s);
1375 return;
1376 }
1377 // Old, misleading name of this function: WriteRawBlock
1378 StopWatch sw(r->ioptions.clock, r->ioptions.stats, WRITE_RAW_BLOCK_MICROS);
1379 const uint64_t offset = r->get_offset();
1380 handle->set_offset(offset);
1381 handle->set_size(block_contents.size());
1382 assert(status().ok());
1383 assert(io_status().ok());
1384 if (uncompressed_block_data == nullptr) {
1385 uncompressed_block_data = &block_contents;
1386 assert(comp_type == kNoCompression);
1387 }
1388
1389 {
1390 io_s = r->file->Append(io_options, block_contents);
1391 if (!io_s.ok()) {
1392 r->SetIOStatus(io_s);
1393 return;
1394 }
1395 }
1396
1397 std::array<char, kBlockTrailerSize> trailer;
1398 trailer[0] = comp_type;
1399 uint32_t checksum = ComputeBuiltinChecksumWithLastByte(
1400 r->table_options.checksum, block_contents.data(), block_contents.size(),
1401 /*last_byte*/ comp_type);
1402 checksum += ChecksumModifierForContext(r->base_context_checksum, offset);
1403
1404 if (block_type == BlockType::kFilter) {
1405 Status s = r->filter_builder->MaybePostVerifyFilter(block_contents);
1406 if (!s.ok()) {
1407 r->SetStatus(s);
1408 return;
1409 }
1410 }
1411
1412 EncodeFixed32(trailer.data() + 1, checksum);
1413 TEST_SYNC_POINT_CALLBACK(
1414 "BlockBasedTableBuilder::WriteMaybeCompressedBlock:TamperWithChecksum",
1415 trailer.data());
1416 {
1417 io_s = r->file->Append(io_options, Slice(trailer.data(), trailer.size()));
1418 if (!io_s.ok()) {
1419 r->SetIOStatus(io_s);
1420 return;
1421 }
1422 }
1423
1424 {
1425 bool warm_cache;
1426 switch (r->table_options.prepopulate_block_cache) {
1427 case BlockBasedTableOptions::PrepopulateBlockCache::kFlushOnly:
1428 warm_cache = (r->reason == TableFileCreationReason::kFlush);
1429 break;
1430 case BlockBasedTableOptions::PrepopulateBlockCache::kDisable:
1431 warm_cache = false;
1432 break;
1433 default:
1434 // missing case
1435 assert(false);
1436 warm_cache = false;
1437 }
1438 if (warm_cache) {
1439 Status s = InsertBlockInCacheHelper(*uncompressed_block_data, handle,
1440 block_type);
1441 if (!s.ok()) {
1442 r->SetStatus(s);
1443 return;
1444 }
1445 }
1446 }
1447
1448 r->set_offset(r->get_offset() + block_contents.size() + kBlockTrailerSize);
1449 if (r->table_options.block_align && is_data_block) {
1450 size_t pad_bytes =
1451 (r->alignment -
1452 ((block_contents.size() + kBlockTrailerSize) & (r->alignment - 1))) &
1453 (r->alignment - 1);
1454
1455 io_s = r->file->Pad(io_options, pad_bytes);
1456 if (io_s.ok()) {
1457 r->set_offset(r->get_offset() + pad_bytes);
1458 } else {
1459 r->SetIOStatus(io_s);
1460 return;
1461 }
1462 }
1463
1464 if (r->IsParallelCompressionEnabled()) {
1465 if (is_data_block) {
1466 r->pc_rep->file_size_estimator.ReapBlock(block_contents.size(),
1467 r->get_offset());
1468 } else {
1469 r->pc_rep->file_size_estimator.SetEstimatedFileSize(r->get_offset());
1470 }
1471 }
1472}

1474void BlockBasedTableBuilder::BGWorkWriteMaybeCompressedBlock() {
1475 Rep* r = rep_;
1476 ParallelCompressionRep::BlockRepSlot* slot = nullptr;
1477 ParallelCompressionRep::BlockRep* block_rep = nullptr;
1478 // Starts empty; see FilterBlockBuilder::AddWithPrevKey
1479 std::string prev_block_last_key_no_ts;
1480 while (r->pc_rep->write_queue.pop(slot)) {
1481 assert(slot != nullptr);
1482 slot->Take(block_rep);
1483 assert(block_rep != nullptr);
1484 if (!block_rep->status.ok()) {
1485 r->SetStatus(block_rep->status);
1486 // Reap block so that blocked Flush() can finish
1487 // if there is one, and Flush() will notice !ok() next time.
1488 block_rep->status = Status::OK();
1489 r->pc_rep->ReapBlock(block_rep);
1490 continue;
1491 }
1492
1493 Slice prev_key_no_ts = prev_block_last_key_no_ts;
1494 for (size_t i = 0; i < block_rep->keys->Size(); i++) {
1495 auto& key = (*block_rep->keys)[i];
1496 if (r->filter_builder != nullptr) {
1497 Slice key_no_ts = ExtractUserKeyAndStripTimestamp(key, r->ts_sz);
1498 r->filter_builder->AddWithPrevKey(key_no_ts, prev_key_no_ts);
1499 prev_key_no_ts = key_no_ts;
1500 }
1501 r->index_builder->OnKeyAdded(key);
1502 }
1503 if (r->filter_builder != nullptr) {
1504 prev_block_last_key_no_ts.assign(prev_key_no_ts.data(),
1505 prev_key_no_ts.size());
1506 }
1507
1508 r->pc_rep->file_size_estimator.SetCurrBlockUncompSize(
1509 block_rep->data->size());
1510 WriteMaybeCompressedBlock(block_rep->compressed_contents,
1511 block_rep->compression_type, &r->pending_handle,
1512 BlockType::kData, &block_rep->contents);
1513 if (!ok()) {
1514 break;
1515 }
1516
1517 r->props.data_size = r->get_offset();
1518 ++r->props.num_data_blocks;
1519
1520 if (block_rep->first_key_in_next_block == nullptr) {
1521 r->index_builder->AddIndexEntry(block_rep->keys->Back(), nullptr,
1522 r->pending_handle,
1523 &r->index_separator_scratch);
1524 } else {
1525 Slice first_key_in_next_block =
1526 Slice(*block_rep->first_key_in_next_block);
1527 r->index_builder->AddIndexEntry(
1528 block_rep->keys->Back(), &first_key_in_next_block, r->pending_handle,
1529 &r->index_separator_scratch);
1530 }
1531
1532 r->pc_rep->ReapBlock(block_rep);
1533 }
1534}

1536void BlockBasedTableBuilder::StartParallelCompression() {
1537 rep_->pc_rep.reset(
1538 new ParallelCompressionRep(rep_->compression_opts.parallel_threads));
1539 rep_->pc_rep->compress_thread_pool.reserve(
1540 rep_->compression_opts.parallel_threads);
1541 for (uint32_t i = 0; i < rep_->compression_opts.parallel_threads; i++) {
1542 rep_->pc_rep->compress_thread_pool.emplace_back([this, i] {
1543 BGWorkCompression(*(rep_->compression_ctxs[i]),
1544 rep_->verify_ctxs[i].get());
1545 });
1546 }
1547 rep_->pc_rep->write_thread.reset(
1548 new port::Thread([this] { BGWorkWriteMaybeCompressedBlock(); }));
1549}

1592void BlockBasedTableBuilder::WriteFilterBlock(
1593 MetaIndexBuilder* meta_index_builder) {
1594 if (rep_->filter_builder == nullptr || rep_->filter_builder->IsEmpty()) {
1595 // No filter block needed
1596 return;
1597 }
1598 if (!rep_->last_ikey.empty()) {
1599 // We might have been using AddWithPrevKey, so need PrevKeyBeforeFinish
1600 // to be safe. And because we are re-synchronized after buffered/parallel
1601 // operation, rep_->last_ikey is accurate.
1602 rep_->filter_builder->PrevKeyBeforeFinish(
1603 ExtractUserKeyAndStripTimestamp(rep_->last_ikey, rep_->ts_sz));
1604 }
1605 BlockHandle filter_block_handle;
1606 bool is_partitioned_filter = rep_->table_options.partition_filters;
1607 if (ok()) {
1608 rep_->props.num_filter_entries +=
1609 rep_->filter_builder->EstimateEntriesAdded();
1610 Status s = Status::Incomplete();
1611 while (ok() && s.IsIncomplete()) {
1612 // filter_data is used to store the transferred filter data payload from
1613 // FilterBlockBuilder and deallocate the payload by going out of scope.
1614 // Otherwise, the payload will unnecessarily remain until
1615 // BlockBasedTableBuilder is deallocated.
1616 //
1617 // See FilterBlockBuilder::Finish() for more on the difference in
1618 // transferred filter data payload among different FilterBlockBuilder
1619 // subtypes.
1620 std::unique_ptr<const char[]> filter_owner;
1621 Slice filter_content;
1622 s = rep_->filter_builder->Finish(filter_block_handle, &filter_content,
1623 &filter_owner);
1624
1625 assert(s.ok() || s.IsIncomplete() || s.IsCorruption());
1626 if (s.IsCorruption()) {
1627 rep_->SetStatus(s);
1628 break;
1629 }
1630
1631 rep_->props.filter_size += filter_content.size();
1632
1633 BlockType btype = is_partitioned_filter && /* last */ s.ok()
1634 ? BlockType::kFilterPartitionIndex
1635 : BlockType::kFilter;
1636 WriteMaybeCompressedBlock(filter_content, kNoCompression,
1637 &filter_block_handle, btype);
1638 }
1639 rep_->filter_builder->ResetFilterBitsBuilder();
1640 }
1641 if (ok()) {
1642 // Add mapping from "<filter_block_prefix>.Name" to location
1643 // of filter data.
1644 std::string key;
1645 key = is_partitioned_filter ? BlockBasedTable::kPartitionedFilterBlockPrefix
1646 : BlockBasedTable::kFullFilterBlockPrefix;
1647 key.append(rep_->table_options.filter_policy->CompatibilityName());
1648 meta_index_builder->Add(key, filter_block_handle);
1649 }

1652void BlockBasedTableBuilder::WriteIndexBlock(
1653 MetaIndexBuilder* meta_index_builder, BlockHandle* index_block_handle) {
1654 if (!ok()) {
1655 return;
1656 }
1657 IndexBuilder::IndexBlocks index_blocks;
1658 auto index_builder_status = rep_->index_builder->Finish(&index_blocks);
1659 if (index_builder_status.IsIncomplete()) {
1660 // We we have more than one index partition then meta_blocks are not
1661 // supported for the index. Currently meta_blocks are used only by
1662 // HashIndexBuilder which is not multi-partition.
1663 assert(index_blocks.meta_blocks.empty());
1664 } else if (ok() && !index_builder_status.ok()) {
1665 rep_->SetStatus(index_builder_status);
1666 }
1667 if (ok()) {
1668 for (const auto& item : index_blocks.meta_blocks) {
1669 BlockHandle block_handle;
1670 WriteBlock(item.second, &block_handle, BlockType::kIndex);
1671 if (!ok()) {
1672 break;
1673 }
1674 meta_index_builder->Add(item.first, block_handle);
1675 }
1676 }
1677 if (ok()) {
1678 if (rep_->table_options.enable_index_compression) {
1679 WriteBlock(index_blocks.index_block_contents, index_block_handle,
1680 BlockType::kIndex);
1681 } else {
1682 WriteMaybeCompressedBlock(index_blocks.index_block_contents,
1683 kNoCompression, index_block_handle,
1684 BlockType::kIndex);
1685 }
1686 }
1687 // If there are more index partitions, finish them and write them out
1688 if (index_builder_status.IsIncomplete()) {
1689 bool index_building_finished = false;
1690 while (ok() && !index_building_finished) {
1691 Status s =
1692 rep_->index_builder->Finish(&index_blocks, *index_block_handle);
1693 if (s.ok()) {
1694 index_building_finished = true;
1695 } else if (s.IsIncomplete()) {
1696 // More partitioned index after this one
1697 assert(!index_building_finished);
1698 } else {
1699 // Error
1700 rep_->SetStatus(s);
1701 return;
1702 }
1703
1704 if (rep_->table_options.enable_index_compression) {
1705 WriteBlock(index_blocks.index_block_contents, index_block_handle,
1706 BlockType::kIndex);
1707 } else {
1708 WriteMaybeCompressedBlock(index_blocks.index_block_contents,
1709 kNoCompression, index_block_handle,
1710 BlockType::kIndex);
1711 }
1712 // The last index_block_handle will be for the partition index block
1713 }
1714 }
1715 // If success and need to record in metaindex rather than footer...
1716 if (!FormatVersionUsesIndexHandleInFooter(
1717 rep_->table_options.format_version)) {
1718 meta_index_builder->Add(kIndexBlockName, *index_block_handle);
1719 }

1895void BlockBasedTableBuilder::EnterUnbuffered() {
1896 Rep* r = rep_;
1897 assert(r->state == Rep::State::kBuffered);
1898 r->state = Rep::State::kUnbuffered;
1899 const size_t kSampleBytes = r->compression_opts.zstd_max_train_bytes > 0
1900 ? r->compression_opts.zstd_max_train_bytes
1901 : r->compression_opts.max_dict_bytes;
1902 const size_t kNumBlocksBuffered = r->data_block_buffers.size();
1903 if (kNumBlocksBuffered == 0) {
1904 // The below code is neither safe nor necessary for handling zero data
1905 // blocks.
1906 return;
1907 }
1908
1909 // Abstract algebra teaches us that a finite cyclic group (such as the
1910 // additive group of integers modulo N) can be generated by a number that is
1911 // coprime with N. Since N is variable (number of buffered data blocks), we
1912 // must then pick a prime number in order to guarantee coprimeness with any N.
1913 //
1914 // One downside of this approach is the spread will be poor when
1915 // `kPrimeGeneratorRemainder` is close to zero or close to
1916 // `kNumBlocksBuffered`.
1917 //
1918 // Picked a random number between one and one trillion and then chose the
1919 // next prime number greater than or equal to it.
1920 const uint64_t kPrimeGenerator = 545055921143ull;
1921 // Can avoid repeated division by just adding the remainder repeatedly.
1922 const size_t kPrimeGeneratorRemainder = static_cast<size_t>(
1923 kPrimeGenerator % static_cast<uint64_t>(kNumBlocksBuffered));
1924 const size_t kInitSampleIdx = kNumBlocksBuffered / 2;
1925
1926 std::string compression_dict_samples;
1927 std::vector<size_t> compression_dict_sample_lens;
1928 size_t buffer_idx = kInitSampleIdx;
1929 for (size_t i = 0;
1930 i < kNumBlocksBuffered && compression_dict_samples.size() < kSampleBytes;
1931 ++i) {
1932 size_t copy_len = std::min(kSampleBytes - compression_dict_samples.size(),
1933 r->data_block_buffers[buffer_idx].size());
1934 compression_dict_samples.append(r->data_block_buffers[buffer_idx], 0,
1935 copy_len);
1936 compression_dict_sample_lens.emplace_back(copy_len);
1937
1938 buffer_idx += kPrimeGeneratorRemainder;
1939 if (buffer_idx >= kNumBlocksBuffered) {
1940 buffer_idx -= kNumBlocksBuffered;
1941 }
1942 }
1943
1944 // final data block flushed, now we can generate dictionary from the samples.
1945 // OK if compression_dict_samples is empty, we'll just get empty dictionary.
1946 std::string dict;
1947 if (r->compression_opts.zstd_max_train_bytes > 0) {
1948 if (r->compression_opts.use_zstd_dict_trainer) {
1949 dict = ZSTD_TrainDictionary(compression_dict_samples,
1950 compression_dict_sample_lens,
1951 r->compression_opts.max_dict_bytes);
1952 } else {
1953 dict = ZSTD_FinalizeDictionary(
1954 compression_dict_samples, compression_dict_sample_lens,
1955 r->compression_opts.max_dict_bytes, r->compression_opts.level);
1956 }
1957 } else {
1958 dict = std::move(compression_dict_samples);
1959 }
1960 r->compression_dict.reset(new CompressionDict(dict, r->compression_type,
1961 r->compression_opts.level));
1962 r->verify_dict.reset(new UncompressionDict(
1963 dict, r->compression_type == kZSTD ||
1964 r->compression_type == kZSTDNotFinalCompression));
1965
1966 auto get_iterator_for_block = [&r](size_t i) {
1967 auto& data_block = r->data_block_buffers[i];
1968 assert(!data_block.empty());
1969
1970 Block reader{BlockContents{data_block}};
1971 DataBlockIter* iter = reader.NewDataIterator(
1972 r->internal_comparator.user_comparator(), kDisableGlobalSequenceNumber,
1973 nullptr /* iter */, nullptr /* stats */,
1974 false /* block_contents_pinned */, r->persist_user_defined_timestamps);
1975
1976 iter->SeekToFirst();
1977 assert(iter->Valid());
1978 return std::unique_ptr<DataBlockIter>(iter);
1979 };
1980
1981 std::unique_ptr<DataBlockIter> iter = nullptr, next_block_iter = nullptr;
1982
1983 for (size_t i = 0; ok() && i < r->data_block_buffers.size(); ++i) {
1984 if (iter == nullptr) {
1985 iter = get_iterator_for_block(i);
1986 assert(iter != nullptr);
1987 };
1988
1989 if (i + 1 < r->data_block_buffers.size()) {
1990 next_block_iter = get_iterator_for_block(i + 1);
1991 }
1992
1993 auto& data_block = r->data_block_buffers[i];
1994 if (r->IsParallelCompressionEnabled()) {
1995 Slice first_key_in_next_block;
1996 const Slice* first_key_in_next_block_ptr = &first_key_in_next_block;
1997 if (i + 1 < r->data_block_buffers.size()) {
1998 assert(next_block_iter != nullptr);
1999 first_key_in_next_block = next_block_iter->key();
2000 } else {
2001 first_key_in_next_block_ptr = r->first_key_in_next_block;
2002 }
2003
2004 std::vector<std::string> keys;
2005 for (; iter->Valid(); iter->Next()) {
2006 keys.emplace_back(iter->key().ToString());
2007 }
2008
2009 ParallelCompressionRep::BlockRep* block_rep = r->pc_rep->PrepareBlock(
2010 r->compression_type, first_key_in_next_block_ptr, &data_block, &keys);
2011
2012 assert(block_rep != nullptr);
2013 r->pc_rep->file_size_estimator.EmitBlock(block_rep->data->size(),
2014 r->get_offset());
2015 r->pc_rep->EmitBlock(block_rep);
2016 } else {
2017 for (; iter->Valid(); iter->Next()) {
2018 Slice key = iter->key();
2019 if (r->filter_builder != nullptr) {
2020 // NOTE: AddWithPrevKey here would only save key copying if prev is
2021 // pinned (iter->IsKeyPinned()), which is probably rare with delta
2022 // encoding. OK to go from Add() here to AddWithPrevKey() in
2023 // unbuffered operation.
2024 r->filter_builder->Add(
2025 ExtractUserKeyAndStripTimestamp(key, r->ts_sz));
2026 }
2027 r->index_builder->OnKeyAdded(key);
2028 }
2029 WriteBlock(Slice(data_block), &r->pending_handle, BlockType::kData);
2030 if (ok() && i + 1 < r->data_block_buffers.size()) {
2031 assert(next_block_iter != nullptr);
2032 Slice first_key_in_next_block = next_block_iter->key();
2033
2034 Slice* first_key_in_next_block_ptr = &first_key_in_next_block;
2035
2036 iter->SeekToLast();
2037 assert(iter->Valid());
2038 r->index_builder->AddIndexEntry(
2039 iter->key(), first_key_in_next_block_ptr, r->pending_handle,
2040 &r->index_separator_scratch);
2041 }
2042 }
2043 std::swap(iter, next_block_iter);
2044 }
2045 r->data_block_buffers.clear();
2046 r->data_begin_offset = 0;
2047 // Release all reserved cache for data block buffers
2048 if (r->compression_dict_buffer_cache_res_mgr != nullptr) {
2049 Status s = r->compression_dict_buffer_cache_res_mgr->UpdateCacheReservation(
2050 r->data_begin_offset);
2051 s.PermitUncheckedError();
2052 }
2053}

2055Status BlockBasedTableBuilder::Finish() {
2056 Rep* r = rep_;
2057 assert(r->state != Rep::State::kClosed);
2058 bool empty_data_block = r->data_block.empty();
2059 r->first_key_in_next_block = nullptr;
2060 Flush();
2061 if (r->state == Rep::State::kBuffered) {
2062 EnterUnbuffered();
2063 }
2064 if (r->IsParallelCompressionEnabled()) {
2065 StopParallelCompression();
2066#ifndef NDEBUG
2067 for (const auto& br : r->pc_rep->block_rep_buf) {
2068 assert(br.status.ok());
2069 }
2070#endif // !NDEBUG
2071 } else {
2072 // To make sure properties block is able to keep the accurate size of index
2073 // block, we will finish writing all index entries first.
2074 if (ok() && !empty_data_block) {
2075 r->index_builder->AddIndexEntry(
2076 r->last_ikey, nullptr /* no next data block */, r->pending_handle,
2077 &r->index_separator_scratch);
2078 }
2079 }
2080
2081 r->props.tail_start_offset = r->offset;
2082
2083 // Write meta blocks, metaindex block and footer in the following order.
2084 // 1. [meta block: filter]
2085 // 2. [meta block: index]
2086 // 3. [meta block: compression dictionary]
2087 // 4. [meta block: range deletion tombstone]
2088 // 5. [meta block: properties]
2089 // 6. [metaindex block]
2090 // 7. Footer
2091 BlockHandle metaindex_block_handle, index_block_handle;
2092 MetaIndexBuilder meta_index_builder;
2093 WriteFilterBlock(&meta_index_builder);
2094 WriteIndexBlock(&meta_index_builder, &index_block_handle);
2095 WriteCompressionDictBlock(&meta_index_builder);
2096 WriteRangeDelBlock(&meta_index_builder);
2097 WritePropertiesBlock(&meta_index_builder);
2098 if (ok()) {
2099 // flush the meta index block
2100 WriteMaybeCompressedBlock(meta_index_builder.Finish(), kNoCompression,
2101 &metaindex_block_handle, BlockType::kMetaIndex);
2102 }
2103 if (ok()) {
2104 WriteFooter(metaindex_block_handle, index_block_handle);
2105 }
2106 r->state = Rep::State::kClosed;
2107 r->tail_size = r->offset - r->props.tail_start_offset;
2108
2109 Status ret_status = r->CopyStatus();
2110 IOStatus ios = r->GetIOStatus();
2111 if (!ios.ok() && ret_status.ok()) {
2112 // Let io_status supersede ok status (otherwise status takes precedennce)
2113 ret_status = ios;
2114 }
2115 return ret_status;
2116}
