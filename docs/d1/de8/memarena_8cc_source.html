63void memarena::create(size_t initial_size) {
64 _current_chunk = arena_chunk();
65 _other_chunks = nullptr;
66 _size_of_other_chunks = 0;
67 _footprint_of_other_chunks = 0;
68 _n_other_chunks = 0;

76void memarena::destroy(void) {
77 if (_current_chunk.buf) {
78 toku_free(_current_chunk.buf);
79 }
80 for (int i = 0; i < _n_other_chunks; i++) {
81 toku_free(_other_chunks[i].buf);
82 }
83 if (_other_chunks) {
84 toku_free(_other_chunks);
85 }
86 _current_chunk = arena_chunk();
87 _other_chunks = nullptr;
88 _n_other_chunks = 0;
89}

91static size_t round_to_page(size_t size) {
92 const size_t page_size = 4096;
93 const size_t r = page_size + ((size - 1) & ~(page_size - 1));
94 assert((r & (page_size - 1)) == 0); // make sure it's aligned
95 assert(r >= size); // make sure it's not too small
96 assert(r <
97 size + page_size); // make sure we didn't grow by more than a page.
98 return r;
99}

103void *memarena::malloc_from_arena(size_t size) {
104 if (_current_chunk.buf == nullptr ||
105 _current_chunk.size < _current_chunk.used + size) {
106 // The existing block isn't big enough.
107 // Add the block to the vector of blocks.
108 if (_current_chunk.buf) {
109 invariant(_current_chunk.size > 0);
110 int old_n = _n_other_chunks;
111 XREALLOC_N(old_n + 1, _other_chunks);
112 _other_chunks[old_n] = _current_chunk;
113 _n_other_chunks = old_n + 1;
114 _size_of_other_chunks += _current_chunk.size;
115 _footprint_of_other_chunks +=
116 toku_memory_footprint(_current_chunk.buf, _current_chunk.used);
117 }
118
119 // Make a new one. Grow the buffer size exponentially until we hit
120 // the max chunk size, but make it at least `size' bytes so the
121 // current allocation always fit.
122 size_t new_size =
123 std::min(MEMARENA_MAX_CHUNK_SIZE, 2 * _current_chunk.size);
124 if (new_size < size) {
125 new_size = size;
126 }
127 new_size = round_to_page(
128 new_size); // at least size, but round to the next page size
129 XMALLOC_N(new_size, _current_chunk.buf);
130 _current_chunk.used = 0;
131 _current_chunk.size = new_size;
132 }
133 invariant(_current_chunk.buf != nullptr);
134
135 // allocate in the existing block.
136 char *p = _current_chunk.buf + _current_chunk.used;
137 _current_chunk.used += size;
138 return p;
139}

141void memarena::move_memory(memarena *dest) {
142 // Move memory to dest
143 XREALLOC_N(dest->_n_other_chunks + _n_other_chunks + 1, dest->_other_chunks);
144 dest->_size_of_other_chunks += _size_of_other_chunks + _current_chunk.size;
145 dest->_footprint_of_other_chunks +=
146 _footprint_of_other_chunks +
147 toku_memory_footprint(_current_chunk.buf, _current_chunk.used);
148 for (int i = 0; i < _n_other_chunks; i++) {
149 dest->_other_chunks[dest->_n_other_chunks++] = _other_chunks[i];
150 }
151 dest->_other_chunks[dest->_n_other_chunks++] = _current_chunk;
152
153 // Clear out this memarena's memory
154 toku_free(_other_chunks);
155 _current_chunk = arena_chunk();
156 _other_chunks = nullptr;
157 _size_of_other_chunks = 0;
158 _footprint_of_other_chunks = 0;
159 _n_other_chunks = 0;
160}

179const void *memarena::chunk_iterator::current(size_t *used) const {
180 if (_chunk_idx < 0) {
181 *used = _ma->_current_chunk.used;
182 return _ma->_current_chunk.buf;
183 } else if (_chunk_idx < _ma->_n_other_chunks) {
184 *used = _ma->_other_chunks[_chunk_idx].used;
185 return _ma->_other_chunks[_chunk_idx].buf;
186 }
187 *used = 0;
188 return nullptr;
189}
