36// Value types encoded as the last component of internal keys.
37// DO NOT CHANGE THESE ENUM VALUES: they are embedded in the on-disk
38// data structures.
39// The highest bit of the value type needs to be reserved to SST tables
40// for them to do more flexible encoding.
41enum ValueType : unsigned char {
42 kTypeDeletion = 0x0,
43 kTypeValue = 0x1,
44 kTypeMerge = 0x2,
45 kTypeLogData = 0x3, // WAL only.
46 kTypeColumnFamilyDeletion = 0x4, // WAL only.
47 kTypeColumnFamilyValue = 0x5, // WAL only.
48 kTypeColumnFamilyMerge = 0x6, // WAL only.
49 kTypeSingleDeletion = 0x7,
50 kTypeColumnFamilySingleDeletion = 0x8, // WAL only.
51 kTypeBeginPrepareXID = 0x9, // WAL only.
52 kTypeEndPrepareXID = 0xA, // WAL only.
53 kTypeCommitXID = 0xB, // WAL only.
54 kTypeRollbackXID = 0xC, // WAL only.
55 kTypeNoop = 0xD, // WAL only.
56 kTypeColumnFamilyRangeDeletion = 0xE, // WAL only.
57 kTypeRangeDeletion = 0xF, // meta block
58 kTypeColumnFamilyBlobIndex = 0x10, // Blob DB only
59 kTypeBlobIndex = 0x11, // Blob DB only
60 // When the prepared record is also persisted in db, we use a different
61 // record. This is to ensure that the WAL that is generated by a WritePolicy
62 // is not mistakenly read by another, which would result into data
63 // inconsistency.
64 kTypeBeginPersistedPrepareXID = 0x12, // WAL only.
65 // Similar to kTypeBeginPersistedPrepareXID, this is to ensure that WAL
66 // generated by WriteUnprepared write policy is not mistakenly read by
67 // another.
68 kTypeBeginUnprepareXID = 0x13, // WAL only.
69 kTypeDeletionWithTimestamp = 0x14,
70 kTypeCommitXIDAndTimestamp = 0x15, // WAL only
71 kTypeWideColumnEntity = 0x16,
72 kTypeColumnFamilyWideColumnEntity = 0x17, // WAL only
73 kTypeValuePreferredSeqno = 0x18, // Value with a unix write time
74 kTypeColumnFamilyValuePreferredSeqno = 0x19, // WAL only
75 kTypeMaxValid, // Should be after the last valid type, only used for
76 // validation
77 kMaxValue = 0x7F // Not used for storing records.

110// Checks whether a type is an inline value type

515inline Status ParseInternalKey(const Slice& internal_key,
516 ParsedInternalKey* result, bool log_err_key) {
517 const size_t n = internal_key.size();
518
519 if (n < kNumInternalBytes) {
520 return Status::Corruption("Corrupted Key: Internal Key too small. Size=" +
521 std::to_string(n) + ". ");
522 }
523
524 uint64_t num = DecodeFixed64(internal_key.data() + n - kNumInternalBytes);
525 unsigned char c = num & 0xff;
526 result->sequence = num >> 8;
527 result->type = static_cast<ValueType>(c);
528 assert(result->type <= ValueType::kMaxValue);
529 result->user_key = Slice(internal_key.data(), n - kNumInternalBytes);
530
531 if (IsExtendedValueType(result->type)) {
532 return Status::OK();
533 } else {
534 return Status::Corruption("Corrupted Key",
535 result->DebugString(log_err_key, true));
536 }
537}

539// Update the sequence number in the internal key.
540// Guarantees not to invalidate ikey.data().
541inline void UpdateInternalKey(std::string* ikey, uint64_t seq, ValueType t) {
542 size_t ikey_sz = ikey->size();
543 assert(ikey_sz >= kNumInternalBytes);
544 uint64_t newval = (seq << 8) | t;
545
546 // Note: Since C++11, strings are guaranteed to be stored contiguously and
547 // string::operator[]() is guaranteed not to change ikey.data().
548 EncodeFixed64(&(*ikey)[ikey_sz - kNumInternalBytes], newval);
549}

617 // Append "non_shared_data" to its back, from "shared_len"

648 // A version of `TrimAppend` assuming the last bytes of length `ts_sz` in the
649 // user key part of `key_` is not counted towards shared bytes. And the
650 // decoded key needed a min timestamp of length `ts_sz` pad to the user key.
651 void TrimAppendWithTimestamp(const size_t shared_len,
652 const char* non_shared_data,
653 const size_t non_shared_len,
654 const size_t ts_sz) {
655 // This function is only used by the UDT in memtable feature, which only
656 // support built in comparators with uint64 timestamps.
657 assert(ts_sz == sizeof(uint64_t));
658 size_t next_key_slice_index = 0;
659 if (IsUserKey()) {
660 key_slices_[next_key_slice_index++] = Slice(key_, shared_len);

663 key_slices_[next_key_slice_index++] = Slice(kTsMin, ts_sz);
664 } else {
665 assert(shared_len + non_shared_len >= kNumInternalBytes);
666 // Invaraint: shared_user_key_len + shared_internal_bytes_len = shared_len
667 // In naming below `*_len` variables, keyword `user_key` refers to the
668 // user key part of the existing key in `key_` as apposed to the new key.
669 // Similary, `internal_bytes` refers to the footer part of the existing
670 // key. These bytes potentially will move between user key part and the
671 // footer part in the new key.
672 const size_t user_key_len = key_size_ - kNumInternalBytes;
673 const size_t sharable_user_key_len = user_key_len - ts_sz;
674 const size_t shared_user_key_len =
675 std::min(shared_len, sharable_user_key_len);
676 const size_t shared_internal_bytes_len = shared_len - shared_user_key_len;

678 // One Slice among the three Slices will get split into two Slices, plus
679 // a timestamp slice.
680 bool ts_added = false;
681 // Add slice parts and find the right location to add the min timestamp.
682 MaybeAddKeyPartsWithTimestamp(
683 key_, shared_user_key_len,
684 shared_internal_bytes_len + non_shared_len < kNumInternalBytes,
685 shared_len + non_shared_len - kNumInternalBytes, ts_sz,
686 &next_key_slice_index, &ts_added);
687 MaybeAddKeyPartsWithTimestamp(
688 key_ + user_key_len, shared_internal_bytes_len,
689 non_shared_len < kNumInternalBytes,
690 shared_internal_bytes_len + non_shared_len - kNumInternalBytes, ts_sz,
691 &next_key_slice_index, &ts_added);
692 MaybeAddKeyPartsWithTimestamp(non_shared_data, non_shared_len,
693 non_shared_len >= kNumInternalBytes,
694 non_shared_len - kNumInternalBytes, ts_sz,
695 &next_key_slice_index, &ts_added);
696 assert(ts_added);
697 }
698 SetKeyImpl(next_key_slice_index,
699 /* total_bytes= */ shared_len + non_shared_len + ts_sz);
700 }

892#endif // NDEBUG
