27uint64_t GetTotalFilesSize(const std::vector<FileMetaData*>& files) {
28 uint64_t total_size = 0;
29 for (const auto& f : files) {
30 total_size += f->fd.file_size;
31 }
32 return total_size;
33}

42Compaction* FIFOCompactionPicker::PickTTLCompaction(
43 const std::string& cf_name, const MutableCFOptions& mutable_cf_options,
44 const MutableDBOptions& mutable_db_options, VersionStorageInfo* vstorage,
45 LogBuffer* log_buffer) {
46 assert(mutable_cf_options.ttl > 0);
47
48 const int kLevel0 = 0;
49 const std::vector<FileMetaData*>& level_files = vstorage->LevelFiles(kLevel0);
50 uint64_t total_size = GetTotalFilesSize(level_files);
51
52 int64_t _current_time;
53 auto status = ioptions_.clock->GetCurrentTime(&_current_time);
54 if (!status.ok()) {
55 ROCKS_LOG_BUFFER(log_buffer,
56 "[%s] FIFO compaction: Couldn't get current time: %s. "
57 "Not doing compactions based on TTL. ",
58 cf_name.c_str(), status.ToString().c_str());
59 return nullptr;
60 }
61 const uint64_t current_time = static_cast<uint64_t>(_current_time);
62
63 if (!level0_compactions_in_progress_.empty()) {
64 ROCKS_LOG_BUFFER(
65 log_buffer,
66 "[%s] FIFO compaction: Already executing compaction. No need "
67 "to run parallel compactions since compactions are very fast",
68 cf_name.c_str());
69 return nullptr;
70 }
71
72 std::vector<CompactionInputFiles> inputs;
73 inputs.emplace_back();
74 inputs[0].level = 0;
75
76 // avoid underflow
77 if (current_time > mutable_cf_options.ttl) {
78 for (auto ritr = level_files.rbegin(); ritr != level_files.rend(); ++ritr) {
79 FileMetaData* f = *ritr;
80 assert(f);
81 if (f->fd.table_reader && f->fd.table_reader->GetTableProperties()) {
82 uint64_t newest_key_time = f->TryGetNewestKeyTime();
83 uint64_t creation_time =
84 f->fd.table_reader->GetTableProperties()->creation_time;
85 uint64_t est_newest_key_time = newest_key_time == kUnknownNewestKeyTime
86 ? creation_time
87 : newest_key_time;
88 if (est_newest_key_time == kUnknownNewestKeyTime ||
89 est_newest_key_time >= (current_time - mutable_cf_options.ttl)) {
90 break;
91 }
92 }
93 total_size -= f->fd.file_size;
94 inputs[0].files.push_back(f);
95 }
96 }
97
98 // Return a nullptr and proceed to size-based FIFO compaction if:
99 // 1. there are no files older than ttl OR
100 // 2. there are a few files older than ttl, but deleting them will not bring
101 // the total size to be less than max_table_files_size threshold.
102 if (inputs[0].files.empty() ||
103 total_size >
104 mutable_cf_options.compaction_options_fifo.max_table_files_size) {
105 return nullptr;
106 }
107
108 for (const auto& f : inputs[0].files) {
109 assert(f);
110 uint64_t newest_key_time = f->TryGetNewestKeyTime();
111 uint64_t creation_time = 0;
112 if (f->fd.table_reader && f->fd.table_reader->GetTableProperties()) {
113 creation_time = f->fd.table_reader->GetTableProperties()->creation_time;
114 }
115 uint64_t est_newest_key_time = newest_key_time == kUnknownNewestKeyTime
116 ? creation_time
117 : newest_key_time;
118 ROCKS_LOG_BUFFER(log_buffer,
119 "[%s] FIFO compaction: picking file %" PRIu64
120 " with estimated newest key time %" PRIu64 " for deletion",
121 cf_name.c_str(), f->fd.GetNumber(), est_newest_key_time);
122 }
123
124 Compaction* c = new Compaction(
125 vstorage, ioptions_, mutable_cf_options, mutable_db_options,
126 std::move(inputs), 0, 0, 0, 0, kNoCompression,
127 mutable_cf_options.compression_opts,
128 mutable_cf_options.default_write_temperature,
129 /* max_subcompactions */ 0, {}, /* earliest_snapshot */ std::nullopt,
130 /* snapshot_checker */ nullptr,
131 /* is manual */ false,
132 /* trim_ts */ "", vstorage->CompactionScore(0),
133 /* is deletion compaction */ true, /* l0_files_might_overlap */ true,
134 CompactionReason::kFIFOTtl);
135 return c;
136}

138// The size-based compaction picker for FIFO.
139//
140// When the entire column family size exceeds max_table_files_size, FIFO will
141// try to delete the oldest sst file(s) until the resulting column family size
142// is smaller than max_table_files_size.
143//
144// This function also takes care the case where a DB is migrating from level /
145// universal compaction to FIFO compaction. During the migration, the column
146// family will also have non-L0 files while FIFO can only create L0 files.
147// In this case, this function will first purge the sst files in the bottom-
148// most non-empty level first, and the DB will eventually converge to the
149// regular FIFO case where there're only L0 files. Note that during the
150// migration case, the purge order will only be an approximation of "FIFO"
151// as entries inside lower-level files might sometimes be newer than some
152// entries inside upper-level files.
153Compaction* FIFOCompactionPicker::PickSizeCompaction(
154 const std::string& cf_name, const MutableCFOptions& mutable_cf_options,
155 const MutableDBOptions& mutable_db_options, VersionStorageInfo* vstorage,
156 LogBuffer* log_buffer) {
157 // compute the total size and identify the last non-empty level
158 int last_level = 0;
159 uint64_t total_size = 0;
160 for (int level = 0; level < vstorage->num_levels(); ++level) {
161 auto level_size = GetTotalFilesSize(vstorage->LevelFiles(level));
162 total_size += level_size;
163 if (level_size > 0) {
164 last_level = level;
165 }
166 }
167 const std::vector<FileMetaData*>& last_level_files =
168 vstorage->LevelFiles(last_level);
169
170 if (last_level == 0 &&
171 total_size <=
172 mutable_cf_options.compaction_options_fifo.max_table_files_size) {
173 // total size not exceeded, try to find intra level 0 compaction if enabled
174 const std::vector<FileMetaData*>& level0_files = vstorage->LevelFiles(0);
175 if (mutable_cf_options.compaction_options_fifo.allow_compaction &&
176 level0_files.size() > 0) {
177 CompactionInputFiles comp_inputs;
178 // try to prevent same files from being compacted multiple times, which
179 // could produce large files that may never TTL-expire. Achieve this by
180 // disallowing compactions with files larger than memtable (inflate its
181 // size by 10% to account for uncompressed L0 files that may have size
182 // slightly greater than memtable size limit).
183 size_t max_compact_bytes_per_del_file =
184 static_cast<size_t>(MultiplyCheckOverflow(
185 static_cast<uint64_t>(mutable_cf_options.write_buffer_size),
186 1.1));
187 if (FindIntraL0Compaction(
188 level0_files,
189 mutable_cf_options
190 .level0_file_num_compaction_trigger /* min_files_to_compact */
191 ,
192 max_compact_bytes_per_del_file,
193 mutable_cf_options.max_compaction_bytes, &comp_inputs)) {
194 Compaction* c = new Compaction(
195 vstorage, ioptions_, mutable_cf_options, mutable_db_options,
196 {comp_inputs}, 0, 16 * 1024 * 1024 /* output file size limit */,
197 0 /* max compaction bytes, not applicable */,
198 0 /* output path ID */, mutable_cf_options.compression,
199 mutable_cf_options.compression_opts,
200 mutable_cf_options.default_write_temperature,
201 0 /* max_subcompactions */, {},
202 /* earliest_snapshot */ std::nullopt,
203 /* snapshot_checker */ nullptr, /* is manual */ false,
204 /* trim_ts */ "", vstorage->CompactionScore(0),
205 /* is deletion compaction */ false,
206 /* l0_files_might_overlap */ true,
207 CompactionReason::kFIFOReduceNumFiles);
208 return c;
209 }
210 }
211
212 ROCKS_LOG_BUFFER(
213 log_buffer,
214 "[%s] FIFO compaction: nothing to do. Total size %" PRIu64
215 ", max size %" PRIu64 "\n",
216 cf_name.c_str(), total_size,
217 mutable_cf_options.compaction_options_fifo.max_table_files_size);
218 return nullptr;
219 }
220
221 if (!level0_compactions_in_progress_.empty()) {
222 ROCKS_LOG_BUFFER(
223 log_buffer,
224 "[%s] FIFO compaction: Already executing compaction. No need "
225 "to run parallel compactions since compactions are very fast",
226 cf_name.c_str());
227 return nullptr;
228 }
229
230 std::vector<CompactionInputFiles> inputs;
231 inputs.emplace_back();
232 inputs[0].level = last_level;
233
234 if (last_level == 0) {
235 // In L0, right-most files are the oldest files.
236 for (auto ritr = last_level_files.rbegin(); ritr != last_level_files.rend();
237 ++ritr) {
238 auto f = *ritr;
239 total_size -= f->fd.file_size;
240 inputs[0].files.push_back(f);
241 char tmp_fsize[16];
242 AppendHumanBytes(f->fd.GetFileSize(), tmp_fsize, sizeof(tmp_fsize));
243 ROCKS_LOG_BUFFER(log_buffer,
244 "[%s] FIFO compaction: picking file %" PRIu64
245 " with size %s for deletion",
246 cf_name.c_str(), f->fd.GetNumber(), tmp_fsize);
247 if (total_size <=
248 mutable_cf_options.compaction_options_fifo.max_table_files_size) {
249 break;
250 }
251 }
252 } else if (total_size >
253 mutable_cf_options.compaction_options_fifo.max_table_files_size) {
254 // If the last level is non-L0, we actually don't know which file is
255 // logically the oldest since the file creation time only represents
256 // when this file was compacted to this level, which is independent
257 // to when the entries in this file were first inserted.
258 //
259 // As a result, we delete files from the left instead. This means the sst
260 // file with the smallest key will be deleted first. This design decision
261 // better serves a major type of FIFO use cases where smaller keys are
262 // associated with older data.
263 for (const auto& f : last_level_files) {
264 total_size -= f->fd.file_size;
265 inputs[0].files.push_back(f);
266 char tmp_fsize[16];
267 AppendHumanBytes(f->fd.GetFileSize(), tmp_fsize, sizeof(tmp_fsize));
268 ROCKS_LOG_BUFFER(
269 log_buffer,
270 "[%s] FIFO compaction: picking file %" PRIu64
271 " with size %s for deletion under total size %" PRIu64
272 " vs max table files size %" PRIu64,
273 cf_name.c_str(), f->fd.GetNumber(), tmp_fsize, total_size,
274 mutable_cf_options.compaction_options_fifo.max_table_files_size);
275
276 if (total_size <=
277 mutable_cf_options.compaction_options_fifo.max_table_files_size) {
278 break;
279 }
280 }
281 } else {
282 ROCKS_LOG_BUFFER(
283 log_buffer,
284 "[%s] FIFO compaction: nothing to do. Total size %" PRIu64
285 ", max size %" PRIu64 "\n",
286 cf_name.c_str(), total_size,
287 mutable_cf_options.compaction_options_fifo.max_table_files_size);
288 return nullptr;
289 }
290
291 Compaction* c = new Compaction(
292 vstorage, ioptions_, mutable_cf_options, mutable_db_options,
293 std::move(inputs), last_level,
294 /* target_file_size */ 0,
295 /* max_compaction_bytes */ 0,
296 /* output_path_id */ 0, kNoCompression,
297 mutable_cf_options.compression_opts,
298 mutable_cf_options.default_write_temperature,
299 /* max_subcompactions */ 0, {}, /* earliest_snapshot */ std::nullopt,
300 /* snapshot_checker */ nullptr,
301 /* is manual */ false,
302 /* trim_ts */ "", vstorage->CompactionScore(0),
303 /* is deletion compaction */ true,
304 /* l0_files_might_overlap */ true, CompactionReason::kFIFOMaxSize);
305 return c;
306}

308Compaction* FIFOCompactionPicker::PickTemperatureChangeCompaction(
309 const std::string& cf_name, const MutableCFOptions& mutable_cf_options,
310 const MutableDBOptions& mutable_db_options, VersionStorageInfo* vstorage,
311 LogBuffer* log_buffer) const {
312 const std::vector<FileTemperatureAge>& ages =
313 mutable_cf_options.compaction_options_fifo
314 .file_temperature_age_thresholds;
315 if (ages.empty()) {
316 return nullptr;
317 }
318
319 // Does not apply to multi-level FIFO.
320 if (vstorage->num_levels() > 1) {
321 return nullptr;
322 }
323
324 const int kLevel0 = 0;
325 const std::vector<FileMetaData*>& level_files = vstorage->LevelFiles(kLevel0);
326 if (level_files.empty()) {
327 return nullptr;
328 }
329
330 int64_t _current_time;
331 auto status = ioptions_.clock->GetCurrentTime(&_current_time);
332 if (!status.ok()) {
333 ROCKS_LOG_BUFFER(
334 log_buffer,
335 "[%s] FIFO compaction: Couldn't get current time: %s. "
336 "Not doing compactions based on file temperature-age threshold. ",
337 cf_name.c_str(), status.ToString().c_str());
338 return nullptr;
339 }
340 const uint64_t current_time = static_cast<uint64_t>(_current_time);
341
342 if (!level0_compactions_in_progress_.empty()) {
343 ROCKS_LOG_BUFFER(
344 log_buffer,
345 "[%s] FIFO compaction: Already executing compaction. Parallel "
346 "compactions are not supported",
347 cf_name.c_str());
348 return nullptr;
349 }
350
351 std::vector<CompactionInputFiles> inputs;
352 inputs.emplace_back();
353 inputs[0].level = 0;
354
355 // avoid underflow
356 uint64_t min_age = ages[0].age;
357 // kLastTemperature means target temperature is to be determined.
358 Temperature compaction_target_temp = Temperature::kLastTemperature;
359 if (current_time > min_age) {
360 uint64_t create_time_threshold = current_time - min_age;
361 assert(level_files.size() >= 1);
362 for (size_t index = level_files.size(); index >= 1; --index) {
363 // Try to add cur_file to compaction inputs.
364 FileMetaData* cur_file = level_files[index - 1];
365 FileMetaData* prev_file = index < 2 ? nullptr : level_files[index - 2];
366 if (cur_file->being_compacted) {
367 // Should not happen since we check for
368 // `level0_compactions_in_progress_` above. Here we simply just don't
369 // schedule anything.
370 return nullptr;
371 }
372 uint64_t est_newest_key_time = cur_file->TryGetNewestKeyTime(prev_file);
373 // Newer file could have newest_key_time populated
374 if (est_newest_key_time == kUnknownNewestKeyTime) {
375 continue;
376 }
377 if (est_newest_key_time > create_time_threshold) {
378 break;
379 }
380 Temperature cur_target_temp = ages[0].temperature;
381 for (size_t i = 1; i < ages.size(); ++i) {
382 if (current_time >= ages[i].age &&
383 est_newest_key_time <= current_time - ages[i].age) {
384 cur_target_temp = ages[i].temperature;
385 }
386 }
387 if (cur_file->temperature == cur_target_temp) {
388 continue;
389 }
390
391 // cur_file needs to change temperature
392 assert(compaction_target_temp == Temperature::kLastTemperature);
393 compaction_target_temp = cur_target_temp;
394 inputs[0].files.push_back(cur_file);
395 ROCKS_LOG_BUFFER(
396 log_buffer,
397 "[%s] FIFO compaction: picking file %" PRIu64
398 " with estimated newest key time %" PRIu64 " for temperature %s.",
399 cf_name.c_str(), cur_file->fd.GetNumber(), est_newest_key_time,
400 temperature_to_string[cur_target_temp].c_str());
401 break;
402 }
403 }
404
405 if (inputs[0].files.empty()) {
406 return nullptr;
407 }
408 assert(compaction_target_temp != Temperature::kLastTemperature);
409 // Only compact one file at a time.
410 assert(inputs.size() == 1);
411 assert(inputs[0].size() == 1);
412 Compaction* c = new Compaction(
413 vstorage, ioptions_, mutable_cf_options, mutable_db_options,
414 std::move(inputs), 0, 0 /* output file size limit */,
415 0 /* max compaction bytes, not applicable */, 0 /* output path ID */,
416 mutable_cf_options.compression, mutable_cf_options.compression_opts,
417 compaction_target_temp,
418 /* max_subcompactions */ 0, {}, /* earliest_snapshot */ std::nullopt,
419 /* snapshot_checker */ nullptr,
420 /* is manual */ false, /* trim_ts */ "", vstorage->CompactionScore(0),
421 /* is deletion compaction */ false, /* l0_files_might_overlap */ true,
422 CompactionReason::kChangeTemperature);
423 return c;
424}

426Compaction* FIFOCompactionPicker::PickCompaction(
427 const std::string& cf_name, const MutableCFOptions& mutable_cf_options,
428 const MutableDBOptions& mutable_db_options,
429 const std::vector<SequenceNumber>& /* existing_snapshots */,
430 const SnapshotChecker* /* snapshot_checker */, VersionStorageInfo* vstorage,
431 LogBuffer* log_buffer) {
432 Compaction* c = nullptr;
433 if (mutable_cf_options.ttl > 0) {
434 c = PickTTLCompaction(cf_name, mutable_cf_options, mutable_db_options,
435 vstorage, log_buffer);
436 }
437 if (c == nullptr) {
438 c = PickSizeCompaction(cf_name, mutable_cf_options, mutable_db_options,
439 vstorage, log_buffer);
440 }
441 if (c == nullptr) {
442 c = PickTemperatureChangeCompaction(
443 cf_name, mutable_cf_options, mutable_db_options, vstorage, log_buffer);
444 }
445 RegisterCompaction(c);
446 return c;
447}
