219 // remember the info for later checking the FlushJobInfo.
220 prev_fc_info_ = info;
221 ASSERT_GT(info.db_name.size(), 0U);
222 ASSERT_GT(info.cf_name.size(), 0U);
223 ASSERT_GT(info.file_path.size(), 0U);
224 ASSERT_GT(info.job_id, 0);

231 ASSERT_EQ(info.file_checksum_func_name, kUnknownFileChecksumFuncName);
232
233#ifdef ROCKSDB_USING_THREAD_STATUS
234 // Verify the id of the current thread that created this table
235 // file matches the id of any active flush or compaction thread.
236 uint64_t thread_id = env_->GetThreadID();
237 std::vector<ThreadStatus> thread_list;
238 ASSERT_OK(env_->GetThreadList(&thread_list));
239 bool found_match = false;
240 for (const auto& thread_status : thread_list) {
241 if (thread_status.operation_type == ThreadStatus::OP_FLUSH ||
242 thread_status.operation_type == ThreadStatus::OP_COMPACTION) {
243 if (thread_id == thread_status.thread_id) {
244 found_match = true;
245 break;
246 }
247 }
248 }
249 ASSERT_TRUE(found_match);
250#endif // ROCKSDB_USING_THREAD_STATUS

253 void OnFlushCompleted(DB* db, const FlushJobInfo& info) override {
254 flushed_dbs_.push_back(db);
255 flushed_column_family_names_.push_back(info.cf_name);
256 if (info.triggered_writes_slowdown) {
257 slowdown_count++;
258 }
259 if (info.triggered_writes_stop) {
260 stop_count++;
261 }
262 // verify whether the previously created file matches the flushed file.
263 ASSERT_EQ(prev_fc_info_.db_name, db->GetName());
264 ASSERT_EQ(prev_fc_info_.cf_name, info.cf_name);
265 ASSERT_EQ(prev_fc_info_.job_id, info.job_id);
266 ASSERT_EQ(prev_fc_info_.file_path, info.file_path);
267 ASSERT_EQ(TableFileNameToNumber(info.file_path), info.file_number);

269 // Note: the following chunk relies on the notification pertaining to the
270 // database pointed to by DBTestBase::db_, and is thus bypassed when
271 // that assumption does not hold (see the test case MultiDBMultiListeners
272 // below).
273 ASSERT_TRUE(test_);
274 if (db == test_->db_) {
275 std::vector<std::vector<FileMetaData>> files_by_level;
276 ASSERT_LT(info.cf_id, test_->handles_.size());
277 ASSERT_GE(info.cf_id, 0u);
278 ASSERT_NE(test_->handles_[info.cf_id], nullptr);
279 test_->dbfull()->TEST_GetFilesMetaData(test_->handles_[info.cf_id],
280 &files_by_level);

282 ASSERT_FALSE(files_by_level.empty());
283 auto it = std::find_if(files_by_level[0].begin(), files_by_level[0].end(),

285 return meta.fd.GetNumber() == info.file_number;
286 });
287 ASSERT_NE(it, files_by_level[0].end());
288 ASSERT_EQ(info.oldest_blob_file_number, it->oldest_blob_file_number);
289 }
290
291 ASSERT_EQ(db->GetEnv()->GetThreadID(), info.thread_id);
292 ASSERT_GT(info.thread_id, 0U);
293 ASSERT_EQ(info.table_properties.user_collected_properties.find("0")->second,
294 "1");
295 }
