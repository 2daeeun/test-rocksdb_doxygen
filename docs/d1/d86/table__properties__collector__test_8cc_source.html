66// Collects keys that starts with "A" in a table.
67class RegularKeysStartWithA : public TablePropertiesCollector {
68 public:
69 const char* Name() const override { return "RegularKeysStartWithA"; }
70
71 Status Finish(UserCollectedProperties* properties) override {
72 std::string encoded;
73 std::string encoded_num_puts;
74 std::string encoded_num_deletes;
75 std::string encoded_num_single_deletes;
76 std::string encoded_num_size_changes;
77 PutVarint32(&encoded, count_);
78 PutVarint32(&encoded_num_puts, num_puts_);
79 PutVarint32(&encoded_num_deletes, num_deletes_);
80 PutVarint32(&encoded_num_single_deletes, num_single_deletes_);
81 PutVarint32(&encoded_num_size_changes, num_size_changes_);
82 *properties = UserCollectedProperties{
83 {"TablePropertiesTest", message_},
84 {"Count", encoded},
85 {"NumPuts", encoded_num_puts},
86 {"NumDeletes", encoded_num_deletes},
87 {"NumSingleDeletes", encoded_num_single_deletes},
88 {"NumSizeChanges", encoded_num_size_changes},
89 };
90 return Status::OK();
91 }
92
93 Status AddUserKey(const Slice& user_key, const Slice& /*value*/,
94 EntryType type, SequenceNumber /*seq*/,
95 uint64_t file_size) override {
96 // simply asssume all user keys are not empty.
97 if (user_key.data()[0] == 'A') {
98 ++count_;
99 }
100 if (type == kEntryPut) {
101 num_puts_++;
102 } else if (type == kEntryDelete) {
103 num_deletes_++;
104 } else if (type == kEntrySingleDelete) {
105 num_single_deletes_++;
106 }
107 if (file_size < file_size_) {
108 message_ = "File size should not decrease.";
109 } else if (file_size != file_size_) {
110 num_size_changes_++;
111 }
112
113 return Status::OK();
114 }
115
116 UserCollectedProperties GetReadableProperties() const override {
117 return UserCollectedProperties{};
118 }
119
120 private:
121 std::string message_ = "Rocksdb";
122 uint32_t count_ = 0;
123 uint32_t num_puts_ = 0;
124 uint32_t num_deletes_ = 0;
125 uint32_t num_single_deletes_ = 0;
126 uint32_t num_size_changes_ = 0;
127 uint64_t file_size_ = 0;
128};

248void TestCustomizedTablePropertiesCollector(
249 bool backward_mode, uint64_t magic_number, bool test_internal_tbl_prop_coll,
250 const Options& options, const InternalKeyComparator& internal_comparator) {
251 // make sure the entries will be inserted with order.
252 std::map<std::pair<std::string, ValueType>, std::string> kvs = {
253 {{"About ", kTypeValue}, "val5"}, // starts with 'A'
254 {{"Abstract", kTypeValue}, "val2"}, // starts with 'A'
255 {{"Around ", kTypeValue}, "val7"}, // starts with 'A'
256 {{"Beyond ", kTypeValue}, "val3"},
257 {{"Builder ", kTypeValue}, "val1"},
258 {{"Love ", kTypeDeletion}, ""},
259 {{"Cancel ", kTypeValue}, "val4"},
260 {{"Find ", kTypeValue}, "val6"},
261 {{"Rocks ", kTypeDeletion}, ""},
262 {{"Foo ", kTypeSingleDeletion}, ""},
263 };
264
265 // -- Step 1: build table
266 std::unique_ptr<TableBuilder> builder;
267 std::unique_ptr<WritableFileWriter> writer;
268 const ImmutableOptions ioptions(options);
269 const MutableCFOptions moptions(options);
270 InternalTblPropCollFactories internal_tbl_prop_coll_factories;
271 if (test_internal_tbl_prop_coll) {
272 internal_tbl_prop_coll_factories.emplace_back(
273 new RegularKeysStartWithAFactory(backward_mode));
274 } else {
275 GetInternalTblPropCollFactory(ioptions, &internal_tbl_prop_coll_factories);
276 }
277 MakeBuilder(options, ioptions, moptions, internal_comparator,
278 &internal_tbl_prop_coll_factories, &writer, &builder);
279
280 SequenceNumber seqNum = 0U;
281 for (const auto& kv : kvs) {
282 InternalKey ikey(kv.first.first, seqNum++, kv.first.second);
283 builder->Add(ikey.Encode(), kv.second);
284 }
285 ASSERT_OK(builder->Finish());
286 ASSERT_OK(writer->Flush(IOOptions()));
287
288 // -- Step 2: Read properties
289 test::StringSink* fwf =
290 static_cast<test::StringSink*>(writer->writable_file());
291 std::unique_ptr<FSRandomAccessFile> source(
292 new test::StringSource(fwf->contents()));
293 std::unique_ptr<RandomAccessFileReader> fake_file_reader(
294 new RandomAccessFileReader(std::move(source), "test"));
295
296 std::unique_ptr<TableProperties> props;
297 const ReadOptions read_options;
298 Status s = ReadTableProperties(fake_file_reader.get(), fwf->contents().size(),
299 magic_number, ioptions, read_options, &props);
300 ASSERT_OK(s);
301
302 auto user_collected = props->user_collected_properties;
303
304 ASSERT_NE(user_collected.find("TablePropertiesTest"), user_collected.end());
305 ASSERT_EQ("Rocksdb", user_collected.at("TablePropertiesTest"));
306
307 uint32_t starts_with_A = 0;
308 ASSERT_NE(user_collected.find("Count"), user_collected.end());
309 Slice key(user_collected.at("Count"));
310 ASSERT_TRUE(GetVarint32(&key, &starts_with_A));
311 ASSERT_EQ(3u, starts_with_A);
312
313 if (!backward_mode && !test_internal_tbl_prop_coll) {
314 uint32_t num_puts;
315 ASSERT_NE(user_collected.find("NumPuts"), user_collected.end());
316 Slice key_puts(user_collected.at("NumPuts"));
317 ASSERT_TRUE(GetVarint32(&key_puts, &num_puts));
318 ASSERT_EQ(7u, num_puts);
319
320 uint32_t num_deletes;
321 ASSERT_NE(user_collected.find("NumDeletes"), user_collected.end());
322 Slice key_deletes(user_collected.at("NumDeletes"));
323 ASSERT_TRUE(GetVarint32(&key_deletes, &num_deletes));
324 ASSERT_EQ(2u, num_deletes);
325
326 uint32_t num_single_deletes;
327 ASSERT_NE(user_collected.find("NumSingleDeletes"), user_collected.end());
328 Slice key_single_deletes(user_collected.at("NumSingleDeletes"));
329 ASSERT_TRUE(GetVarint32(&key_single_deletes, &num_single_deletes));
330 ASSERT_EQ(1u, num_single_deletes);
331
332 uint32_t num_size_changes;
333 ASSERT_NE(user_collected.find("NumSizeChanges"), user_collected.end());
334 Slice key_size_changes(user_collected.at("NumSizeChanges"));
335 ASSERT_TRUE(GetVarint32(&key_size_changes, &num_size_changes));
336 ASSERT_GE(num_size_changes, 2u);
337 }
338}

376void TestInternalKeyPropertiesCollector(
377 bool backward_mode, uint64_t magic_number, bool sanitized,
378 std::shared_ptr<TableFactory> table_factory) {
379 InternalKey keys[] = {
380 InternalKey("A ", 0, ValueType::kTypeValue),
381 InternalKey("B ", 1, ValueType::kTypeValue),
382 InternalKey("C ", 2, ValueType::kTypeValue),
383 InternalKey("W ", 3, ValueType::kTypeDeletion),
384 InternalKey("X ", 4, ValueType::kTypeDeletion),
385 InternalKey("Y ", 5, ValueType::kTypeDeletion),
386 InternalKey("Z ", 6, ValueType::kTypeDeletion),
387 InternalKey("a ", 7, ValueType::kTypeSingleDeletion),
388 InternalKey("b ", 8, ValueType::kTypeMerge),
389 InternalKey("c ", 9, ValueType::kTypeMerge),
390 };
391
392 std::unique_ptr<TableBuilder> builder;
393 std::unique_ptr<WritableFileWriter> writable;
394 Options options;
395 test::PlainInternalKeyComparator pikc(options.comparator);
396
397 InternalTblPropCollFactories internal_tbl_prop_coll_factories;
398 options.table_factory = table_factory;
399 if (sanitized) {
400 options.table_properties_collector_factories.emplace_back(
401 new RegularKeysStartWithAFactory(backward_mode));
402 // with sanitization, even regular properties collector will be able to
403 // handle internal keys.
404 auto comparator = options.comparator;
405 // HACK: Set options.info_log to avoid writing log in
406 // SanitizeOptions().
407 options.info_log = std::make_shared<test::NullLogger>();
408 options = SanitizeOptions("db", // just a place holder
409 options);
410 ImmutableOptions ioptions(options);
411 GetInternalTblPropCollFactory(ioptions, &internal_tbl_prop_coll_factories);
412 options.comparator = comparator;
413 }
414 const ImmutableOptions ioptions(options);
415 MutableCFOptions moptions(options);
416
417 for (int iter = 0; iter < 2; ++iter) {
418 MakeBuilder(options, ioptions, moptions, pikc,
419 &internal_tbl_prop_coll_factories, &writable, &builder);
420 for (const auto& k : keys) {
421 builder->Add(k.Encode(), "val");
422 }
423
424 ASSERT_OK(builder->Finish());
425 ASSERT_OK(writable->Flush(IOOptions()));
426
427 test::StringSink* fwf =
428 static_cast<test::StringSink*>(writable->writable_file());
429 std::unique_ptr<FSRandomAccessFile> source(
430 new test::StringSource(fwf->contents()));
431 std::unique_ptr<RandomAccessFileReader> reader(
432 new RandomAccessFileReader(std::move(source), "test"));
433
434 std::unique_ptr<TableProperties> props;
435 const ReadOptions read_options;
436 Status s =
437 ReadTableProperties(reader.get(), fwf->contents().size(), magic_number,
438 ioptions, read_options, &props);
439 ASSERT_OK(s);
440
441 auto user_collected = props->user_collected_properties;
442 uint64_t deleted = GetDeletedKeys(user_collected);
443 ASSERT_EQ(5u, deleted); // deletes + single-deletes
444
445 bool property_present;
446 uint64_t merges = GetMergeOperands(user_collected, &property_present);
447 ASSERT_TRUE(property_present);
448 ASSERT_EQ(2u, merges);
449
450 if (sanitized) {
451 uint32_t starts_with_A = 0;
452 ASSERT_NE(user_collected.find("Count"), user_collected.end());
453 Slice key(user_collected.at("Count"));
454 ASSERT_TRUE(GetVarint32(&key, &starts_with_A));
455 ASSERT_EQ(1u, starts_with_A);
456
457 if (!backward_mode) {
458 uint32_t num_puts;
459 ASSERT_NE(user_collected.find("NumPuts"), user_collected.end());
460 Slice key_puts(user_collected.at("NumPuts"));
461 ASSERT_TRUE(GetVarint32(&key_puts, &num_puts));
462 ASSERT_EQ(3u, num_puts);
463
464 uint32_t num_deletes;
465 ASSERT_NE(user_collected.find("NumDeletes"), user_collected.end());
466 Slice key_deletes(user_collected.at("NumDeletes"));
467 ASSERT_TRUE(GetVarint32(&key_deletes, &num_deletes));
468 ASSERT_EQ(4u, num_deletes);
469
470 uint32_t num_single_deletes;
471 ASSERT_NE(user_collected.find("NumSingleDeletes"),
472 user_collected.end());
473 Slice key_single_deletes(user_collected.at("NumSingleDeletes"));
474 ASSERT_TRUE(GetVarint32(&key_single_deletes, &num_single_deletes));
