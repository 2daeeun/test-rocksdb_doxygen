85// bool: whether to pass read_callback to NewIterator().
86class DBIteratorTest : public DBIteratorBaseTest,
87 public testing::WithParamInterface<bool> {
88 public:
89 DBIteratorTest() = default;
90
91 Iterator* NewIterator(const ReadOptions& read_options,
92 ColumnFamilyHandle* column_family = nullptr) {
93 if (column_family == nullptr) {
94 column_family = db_->DefaultColumnFamily();
95 }
96 auto* cfh = static_cast_with_check<ColumnFamilyHandleImpl>(column_family);
97 auto* cfd = cfh->cfd();
98 SequenceNumber seq = read_options.snapshot != nullptr
99 ? read_options.snapshot->GetSequenceNumber()
100 : db_->GetLatestSequenceNumber();
101 bool use_read_callback = GetParam();
102 DummyReadCallback* read_callback = nullptr;
103 if (use_read_callback) {
104 read_callback = new DummyReadCallback();
105 read_callback->SetSnapshot(seq);
106 InstrumentedMutexLock lock(&mutex_);
107 read_callbacks_.push_back(
108 std::unique_ptr<DummyReadCallback>(read_callback));
109 }
110 DBImpl* db_impl = dbfull();
111 SuperVersion* super_version = cfd->GetReferencedSuperVersion(db_impl);
112 return db_impl->NewIteratorImpl(read_options, cfh, super_version, seq,
113 read_callback);
114 }
115
116 private:
117 InstrumentedMutex mutex_;
118 std::vector<std::unique_ptr<DummyReadCallback>> read_callbacks_;
119};

182 options_override.full_block_cache = true;

190

192 // it is in memtable.
193 Iterator* iter = NewIterator(non_blocking_opts, handles_[1]);
194 int count = 0;
195 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
196 ASSERT_OK(iter->status());
197 count++;
198 }
199 ASSERT_OK(iter->status());
200 ASSERT_EQ(count, 1);
201 delete iter;
202
203 // flush memtable to storage. Now, the key should not be in the
204 // memtable neither in the block cache.
205 ASSERT_OK(Flush(1));
206
207 // verify that a non-blocking iterator does not find any
208 // kvs. Neither does it do any IOs to storage.
209 uint64_t numopen = TestGetTickerCount(options, NO_FILE_OPENS);
210 uint64_t cache_added = TestGetTickerCount(options, BLOCK_CACHE_ADD);
211 iter = NewIterator(non_blocking_opts, handles_[1]);
212 count = 0;
213 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
214 count++;
215 }
216 ASSERT_EQ(count, 0);
217 ASSERT_TRUE(iter->status().IsIncomplete());
218 ASSERT_EQ(numopen, TestGetTickerCount(options, NO_FILE_OPENS));
219 ASSERT_EQ(cache_added, TestGetTickerCount(options, BLOCK_CACHE_ADD));
220 delete iter;
221
222 // read in the specified block via a regular get
223 ASSERT_EQ(Get(1, "a"), "b");
224
225 // verify that we can find it via a non-blocking scan
226 numopen = TestGetTickerCount(options, NO_FILE_OPENS);
227 cache_added = TestGetTickerCount(options, BLOCK_CACHE_ADD);
228 iter = NewIterator(non_blocking_opts, handles_[1]);
229 count = 0;
230 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
231 ASSERT_OK(iter->status());
232 count++;
233 }
234 ASSERT_OK(iter->status());
235 ASSERT_EQ(count, 1);
236 ASSERT_EQ(numopen, TestGetTickerCount(options, NO_FILE_OPENS));
237 ASSERT_EQ(cache_added, TestGetTickerCount(options, BLOCK_CACHE_ADD));
238 delete iter;
239
240 // This test verifies block cache behaviors, which is not used by plain
241 // table format.
242 } while (ChangeOptions(kSkipPlainTable | kSkipNoSeekToLast | kSkipMmapReads));
