23namespace ROCKSDB_NAMESPACE {
24class CacheReservationManager;
25
26// Interface to block and signal DB instances, intended for RocksDB
27// internal use only. Each DB instance contains ptr to StallInterface.
28class StallInterface {
29 public:
30 virtual ~StallInterface() {}
31
32 virtual void Block() = 0;
33
34 virtual void Signal() = 0;
35};
36
37class WriteBufferManager final {
38 public:
39 // Parameters:
40 // _buffer_size: _buffer_size = 0 indicates no limit. Memory won't be capped.
41 // memory_usage() won't be valid and ShouldFlush() will always return true.
42 //
43 // cache_: if `cache` is provided, we'll put dummy entries in the cache and
44 // cost the memory allocated to the cache. It can be used even if _buffer_size
45 // = 0.
46 //
47 // allow_stall: if set true, it will enable stalling of writes when
48 // memory_usage() exceeds buffer_size. It will wait for flush to complete and
49 // memory usage to drop down.
50 explicit WriteBufferManager(size_t _buffer_size,
51 std::shared_ptr<Cache> cache = {},
52 bool allow_stall = false);
53 // No copying allowed
54 WriteBufferManager(const WriteBufferManager&) = delete;
55 WriteBufferManager& operator=(const WriteBufferManager&) = delete;
56
57 ~WriteBufferManager();
58
59 // Returns true if buffer_limit is passed to limit the total memory usage and
60 // is greater than 0.
61 bool enabled() const { return buffer_size() > 0; }
62
63 // Returns true if pointer to cache is passed.
64 bool cost_to_cache() const { return cache_res_mgr_ != nullptr; }
65
66 // Returns the total memory used by memtables.
67 // Only valid if enabled()
68 size_t memory_usage() const {
69 return memory_used_.load(std::memory_order_relaxed);
70 }
71
72 // Returns the total memory used by active memtables.
73 size_t mutable_memtable_memory_usage() const {
74 return memory_active_.load(std::memory_order_relaxed);
75 }
76
77 size_t dummy_entries_in_cache_usage() const;
78
79 // Returns the buffer_size.
80 size_t buffer_size() const {
81 return buffer_size_.load(std::memory_order_relaxed);
82 }
83
84 // REQUIRED: `new_size` > 0
85 void SetBufferSize(size_t new_size) {
86 assert(new_size > 0);
87 buffer_size_.store(new_size, std::memory_order_relaxed);
88 mutable_limit_.store(new_size * 7 / 8, std::memory_order_relaxed);
89 // Check if stall is active and can be ended.
90 MaybeEndWriteStall();
91 }
92
93 void SetAllowStall(bool new_allow_stall) {
94 allow_stall_.store(new_allow_stall, std::memory_order_relaxed);
95 MaybeEndWriteStall();
96 }
97
98 // Below functions should be called by RocksDB internally.
99
100 // Should only be called from write thread
101 bool ShouldFlush() const {
102 if (enabled()) {
103 if (mutable_memtable_memory_usage() >
104 mutable_limit_.load(std::memory_order_relaxed)) {
105 return true;
106 }
107 size_t local_size = buffer_size();
108 if (memory_usage() >= local_size &&
109 mutable_memtable_memory_usage() >= local_size / 2) {
110 // If the memory exceeds the buffer size, we trigger more aggressive
111 // flush. But if already more than half memory is being flushed,
112 // triggering more flush may not help. We will hold it instead.
113 return true;
114 }
115 }
116 return false;
117 }
118
119 // Returns true if total memory usage exceeded buffer_size.
120 // We stall the writes untill memory_usage drops below buffer_size. When the
121 // function returns true, all writer threads (including one checking this
122 // condition) across all DBs will be stalled. Stall is allowed only if user
123 // pass allow_stall = true during WriteBufferManager instance creation.
124 //
125 // Should only be called by RocksDB internally .
126 bool ShouldStall() const {
127 if (!allow_stall_.load(std::memory_order_relaxed) || !enabled()) {
128 return false;
129 }
130
131 return IsStallActive() || IsStallThresholdExceeded();
132 }
133
134 // Returns true if stall is active.
135 bool IsStallActive() const {
136 return stall_active_.load(std::memory_order_relaxed);
137 }
138
139 // Returns true if stalling condition is met.
140 bool IsStallThresholdExceeded() const {
141 return memory_usage() >= buffer_size_;
142 }
143
144 void ReserveMem(size_t mem);
145
146 // We are in the process of freeing `mem` bytes, so it is not considered
147 // when checking the soft limit.
148 void ScheduleFreeMem(size_t mem);
149
150 void FreeMem(size_t mem);
151
152 // Add the DB instance to the queue and block the DB.
153 // Should only be called by RocksDB internally.
154 void BeginWriteStall(StallInterface* wbm_stall);
155
156 // If stall conditions have resolved, remove DB instances from queue and
157 // signal them to continue.
158 void MaybeEndWriteStall();
159
160 void RemoveDBFromQueue(StallInterface* wbm_stall);
161
162 private:
163 std::atomic<size_t> buffer_size_;
164 std::atomic<size_t> mutable_limit_;
165 std::atomic<size_t> memory_used_;
166 // Memory that hasn't been scheduled to free.
167 std::atomic<size_t> memory_active_;
168 std::shared_ptr<CacheReservationManager> cache_res_mgr_;
169 // Protects cache_res_mgr_
170 std::mutex cache_res_mgr_mu_;
171
172 std::list<StallInterface*> queue_;
173 // Protects the queue_ and stall_active_.
174 std::mutex mu_;
175 std::atomic<bool> allow_stall_;
176 // Value should only be changed by BeginWriteStall() and MaybeEndWriteStall()
177 // while holding mu_, but it can be read without a lock.
178 std::atomic<bool> stall_active_;
179
180 void ReserveMemWithCache(size_t mem);
181 void FreeMemWithCache(size_t mem);
182};
183} // namespace ROCKSDB_NAMESPACE
