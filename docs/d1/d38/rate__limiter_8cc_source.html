20size_t RateLimiter::RequestToken(size_t bytes, size_t alignment,
21 Env::IOPriority io_priority, Statistics* stats,
22 RateLimiter::OpType op_type) {
23 if (io_priority < Env::IO_TOTAL && IsRateLimited(op_type)) {
24 bytes = std::min(bytes, static_cast<size_t>(GetSingleBurstBytes()));
25
26 if (alignment > 0) {
27 // Here we may actually require more than burst and block
28 // as we can not write/read less than one page at a time on direct I/O
29 // thus we do not want to be strictly constrained by burst
30 bytes = std::max(alignment, TruncateToPageBoundary(alignment, bytes));
31 }
32 Request(bytes, io_priority, stats, op_type);
33 }
34 return bytes;
35}

76GenericRateLimiter::~GenericRateLimiter() {
77 MutexLock g(&request_mutex_);
78 stop_ = true;
79 std::deque<Req*>::size_type queues_size_sum = 0;
80 for (int i = Env::IO_LOW; i < Env::IO_TOTAL; ++i) {
81 queues_size_sum += queue_[i].size();
82 }
83 requests_to_wait_ = static_cast<int32_t>(queues_size_sum);
84
85 for (int i = Env::IO_TOTAL - 1; i >= Env::IO_LOW; --i) {
86 std::deque<Req*> queue = queue_[i];
87 for (auto& r : queue) {
88 r->cv.Signal();
89 }
90 }
91
92 while (requests_to_wait_ > 0) {
93 exit_cv_.Wait();
94 }
95}

122void GenericRateLimiter::Request(int64_t bytes, const Env::IOPriority pri,
123 Statistics* stats) {
124 assert(bytes <= GetSingleBurstBytes());
125 bytes = std::max(static_cast<int64_t>(0), bytes);
126 TEST_SYNC_POINT("GenericRateLimiter::Request");
127 TEST_SYNC_POINT_CALLBACK("GenericRateLimiter::Request:1",
128 &rate_bytes_per_sec_);
129 MutexLock g(&request_mutex_);
130
131 if (auto_tuned_) {
132 static const int kRefillsPerTune = 100;
133 std::chrono::microseconds now(NowMicrosMonotonicLocked());
134 if (now - tuned_time_ >=
135 kRefillsPerTune * std::chrono::microseconds(refill_period_us_)) {
136 Status s = TuneLocked();
137 s.PermitUncheckedError(); //**TODO: What to do on error?
138 }
139 }
140
141 if (stop_) {
142 // It is now in the clean-up of ~GenericRateLimiter().
143 // Therefore any new incoming request will exit from here
144 // and not get satiesfied.
145 return;
146 }
147
148 ++total_requests_[pri];
149
150 if (available_bytes_ > 0) {
151 int64_t bytes_through = std::min(available_bytes_, bytes);
152 total_bytes_through_[pri] += bytes_through;
153 available_bytes_ -= bytes_through;
154 bytes -= bytes_through;
155 }
156
157 if (bytes == 0) {
158 return;
159 }
160
161 // Request cannot be satisfied at this moment, enqueue
162 Req r(bytes, &request_mutex_);
163 queue_[pri].push_back(&r);
164 TEST_SYNC_POINT_CALLBACK("GenericRateLimiter::Request:PostEnqueueRequest",
165 &request_mutex_);
166 // A thread representing a queued request coordinates with other such threads.
167 // There are two main duties.
168 //
169 // (1) Waiting for the next refill time.
170 // (2) Refilling the bytes and granting requests.
171 do {
172 int64_t time_until_refill_us = next_refill_us_ - NowMicrosMonotonicLocked();
173 if (time_until_refill_us > 0) {
174 if (wait_until_refill_pending_) {
175 // Somebody is performing (1). Trust we'll be woken up when our request
176 // is granted or we are needed for future duties.
177 r.cv.Wait();
178 } else {
179 // Whichever thread reaches here first performs duty (1) as described
180 // above.
181 int64_t wait_until = clock_->NowMicros() + time_until_refill_us;
182 RecordTick(stats, NUMBER_RATE_LIMITER_DRAINS);
183 ++num_drains_;
184 wait_until_refill_pending_ = true;
185 clock_->TimedWait(&r.cv, std::chrono::microseconds(wait_until));
186 TEST_SYNC_POINT_CALLBACK("GenericRateLimiter::Request:PostTimedWait",
187 &time_until_refill_us);
188 wait_until_refill_pending_ = false;
189 }
190 } else {
191 // Whichever thread reaches here first performs duty (2) as described
192 // above.
193 RefillBytesAndGrantRequestsLocked();
194 }
195 if (r.request_bytes == 0) {
196 // If there is any remaining requests, make sure there exists at least
197 // one candidate is awake for future duties by signaling a front request
198 // of a queue.
199 for (int i = Env::IO_TOTAL - 1; i >= Env::IO_LOW; --i) {
200 auto& queue = queue_[i];
201 if (!queue.empty()) {
202 queue.front()->cv.Signal();
203 break;
204 }
205 }
206 }
207 // Invariant: non-granted request is always in one queue, and granted
208 // request is always in zero queues.
209#ifndef NDEBUG
210 int num_found = 0;
211 for (int i = Env::IO_LOW; i < Env::IO_TOTAL; ++i) {
212 if (std::find(queue_[i].begin(), queue_[i].end(), &r) !=
213 queue_[i].end()) {
214 ++num_found;
215 }
216 }
217 if (r.request_bytes == 0) {
218 assert(num_found == 0);
219 } else {
220 assert(num_found == 1);
221 }
222#endif // NDEBUG
223 } while (!stop_ && r.request_bytes > 0);
224
225 if (stop_) {
226 // It is now in the clean-up of ~GenericRateLimiter().
227 // Therefore any woken-up request will have come out of the loop and then
228 // exit here. It might or might not have been satisfied.
229 --requests_to_wait_;
230 exit_cv_.Signal();
231 }
232}

234std::vector<Env::IOPriority>
235GenericRateLimiter::GeneratePriorityIterationOrderLocked() {
236 std::vector<Env::IOPriority> pri_iteration_order(Env::IO_TOTAL /* 4 */);
237 // We make Env::IO_USER a superior priority by always iterating its queue
238 // first
239 pri_iteration_order[0] = Env::IO_USER;
240
241 bool high_pri_iterated_after_mid_low_pri = rnd_.OneIn(fairness_);
242 TEST_SYNC_POINT_CALLBACK(
243 "GenericRateLimiter::GeneratePriorityIterationOrderLocked::"
244 "PostRandomOneInFairnessForHighPri",
245 &high_pri_iterated_after_mid_low_pri);
246 bool mid_pri_itereated_after_low_pri = rnd_.OneIn(fairness_);
247 TEST_SYNC_POINT_CALLBACK(
248 "GenericRateLimiter::GeneratePriorityIterationOrderLocked::"
249 "PostRandomOneInFairnessForMidPri",
250 &mid_pri_itereated_after_low_pri);
251
252 if (high_pri_iterated_after_mid_low_pri) {
253 pri_iteration_order[3] = Env::IO_HIGH;
254 pri_iteration_order[2] =
255 mid_pri_itereated_after_low_pri ? Env::IO_MID : Env::IO_LOW;
256 pri_iteration_order[1] =
257 (pri_iteration_order[2] == Env::IO_MID) ? Env::IO_LOW : Env::IO_MID;
258 } else {
259 pri_iteration_order[1] = Env::IO_HIGH;
260 pri_iteration_order[3] =
261 mid_pri_itereated_after_low_pri ? Env::IO_MID : Env::IO_LOW;
262 pri_iteration_order[2] =
263 (pri_iteration_order[3] == Env::IO_MID) ? Env::IO_LOW : Env::IO_MID;
264 }
265
266 TEST_SYNC_POINT_CALLBACK(
267 "GenericRateLimiter::GeneratePriorityIterationOrderLocked::"
268 "PreReturnPriIterationOrder",
269 &pri_iteration_order);
270 return pri_iteration_order;
271}

273void GenericRateLimiter::RefillBytesAndGrantRequestsLocked() {
274 TEST_SYNC_POINT_CALLBACK(
275 "GenericRateLimiter::RefillBytesAndGrantRequestsLocked", &request_mutex_);
276 next_refill_us_ = NowMicrosMonotonicLocked() + refill_period_us_;
277 // Carry over the left over quota from the last period
278 auto refill_bytes_per_period =
279 refill_bytes_per_period_.load(std::memory_order_relaxed);
280 assert(available_bytes_ == 0);
281 available_bytes_ = refill_bytes_per_period;
282
283 std::vector<Env::IOPriority> pri_iteration_order =
284 GeneratePriorityIterationOrderLocked();
285
286 for (int i = Env::IO_LOW; i < Env::IO_TOTAL; ++i) {
287 assert(!pri_iteration_order.empty());
288 Env::IOPriority current_pri = pri_iteration_order[i];
289 auto* queue = &queue_[current_pri];
290 while (!queue->empty()) {
291 auto* next_req = queue->front();
292 if (available_bytes_ < next_req->request_bytes) {
293 // Grant partial request_bytes even if request is for more than
294 // `available_bytes_`, which can happen in a few situations:
295 //
296 // - The available bytes were partially consumed by other request(s)
297 // - The rate was dynamically reduced while requests were already
298 // enqueued
299 // - The burst size was explicitly set to be larger than the refill size
300 next_req->request_bytes -= available_bytes_;
301 available_bytes_ = 0;
302 break;
303 }
304 available_bytes_ -= next_req->request_bytes;
305 next_req->request_bytes = 0;
306 total_bytes_through_[current_pri] += next_req->bytes;
307 queue->pop_front();
308
309 // Quota granted, signal the thread to exit
310 next_req->cv.Signal();
311 }
312 }
313}

315int64_t GenericRateLimiter::CalculateRefillBytesPerPeriodLocked(
316 int64_t rate_bytes_per_sec) {
317 if (std::numeric_limits<int64_t>::max() / rate_bytes_per_sec <
318 refill_period_us_) {
319 // Avoid unexpected result in the overflow case. The result now is still
320 // inaccurate but is a number that is large enough.
321 return std::numeric_limits<int64_t>::max() / kMicrosecondsPerSecond;
322 } else {
323 return rate_bytes_per_sec * refill_period_us_ / kMicrosecondsPerSecond;
324 }
325}

327Status GenericRateLimiter::TuneLocked() {
328 const int kLowWatermarkPct = 50;
329 const int kHighWatermarkPct = 90;
330 const int kAdjustFactorPct = 5;
331 // computed rate limit will be in
332 // `[max_bytes_per_sec_ / kAllowedRangeFactor, max_bytes_per_sec_]`.
333 const int kAllowedRangeFactor = 20;
334
335 std::chrono::microseconds prev_tuned_time = tuned_time_;
336 tuned_time_ = std::chrono::microseconds(NowMicrosMonotonicLocked());
337
338 int64_t elapsed_intervals = (tuned_time_ - prev_tuned_time +
339 std::chrono::microseconds(refill_period_us_) -
340 std::chrono::microseconds(1)) /
341 std::chrono::microseconds(refill_period_us_);
342 // We tune every kRefillsPerTune intervals, so the overflow and division-by-
343 // zero conditions should never happen.
344 assert(num_drains_ <= std::numeric_limits<int64_t>::max() / 100);
345 assert(elapsed_intervals > 0);
346 int64_t drained_pct = num_drains_ * 100 / elapsed_intervals;
347
348 int64_t prev_bytes_per_sec = GetBytesPerSecond();
349 int64_t new_bytes_per_sec;
350 if (drained_pct == 0) {
351 new_bytes_per_sec = max_bytes_per_sec_ / kAllowedRangeFactor;
352 } else if (drained_pct < kLowWatermarkPct) {
353 // sanitize to prevent overflow
354 int64_t sanitized_prev_bytes_per_sec =
355 std::min(prev_bytes_per_sec, std::numeric_limits<int64_t>::max() / 100);
356 new_bytes_per_sec =
357 std::max(max_bytes_per_sec_ / kAllowedRangeFactor,
358 sanitized_prev_bytes_per_sec * 100 / (100 + kAdjustFactorPct));
359 } else if (drained_pct > kHighWatermarkPct) {
360 // sanitize to prevent overflow
361 int64_t sanitized_prev_bytes_per_sec =
362 std::min(prev_bytes_per_sec, std::numeric_limits<int64_t>::max() /
363 (100 + kAdjustFactorPct));
364 new_bytes_per_sec =
365 std::min(max_bytes_per_sec_,
366 sanitized_prev_bytes_per_sec * (100 + kAdjustFactorPct) / 100);
367 } else {
368 new_bytes_per_sec = prev_bytes_per_sec;
369 }
370 if (new_bytes_per_sec != prev_bytes_per_sec) {
371 SetBytesPerSecondLocked(new_bytes_per_sec);
372 }
373 num_drains_ = 0;
374 return Status::OK();
375}

377RateLimiter* NewGenericRateLimiter(
378 int64_t rate_bytes_per_sec, int64_t refill_period_us /* = 100 * 1000 */,
379 int32_t fairness /* = 10 */,
380 RateLimiter::Mode mode /* = RateLimiter::Mode::kWritesOnly */,
381 bool auto_tuned /* = false */, int64_t single_burst_bytes /* = 0 */) {

386 rate_bytes_per_sec, refill_period_us, fairness, mode,
387 SystemClock::Default(), auto_tuned, single_burst_bytes));
388 return limiter.release();
389}
