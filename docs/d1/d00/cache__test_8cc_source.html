37// Conversions between numeric keys/values and the types expected by Cache.
38std::string EncodeKey16Bytes(int k) {
39 std::string result;
40 PutFixed32(&result, k);
41 result.append(std::string(12, 'a')); // Because we need a 16B output, we
42 // add a 12-byte padding.
43 return result;
44}
45
46int DecodeKey16Bytes(const Slice& k) {
47 assert(k.size() == 16);
48 return DecodeFixed32(k.data()); // Decodes only the first 4 bytes of k.
49}
50
51std::string EncodeKey32Bits(int k) {
52 std::string result;
53 PutFixed32(&result, k);
54 return result;
55}
56
57int DecodeKey32Bits(const Slice& k) {
58 assert(k.size() == 4);
59 return DecodeFixed32(k.data());
60}

62Cache::ObjectPtr EncodeValue(uintptr_t v) {
63 return reinterpret_cast<Cache::ObjectPtr>(v);

66int DecodeValue(void* v) {
67 return static_cast<int>(reinterpret_cast<uintptr_t>(v));

112 // These functions encode/decode keys in tests cases that use
113 // int keys.
114 // Currently, HyperClockCache requires keys to be 16B long, whereas
115 // LRUCache doesn't, so the encoding depends on the cache type.
116 std::string EncodeKey(int k) {
117 if (IsHyperClock()) {
118 return EncodeKey16Bytes(k);
119 } else {
120 return EncodeKey32Bits(k);
121 }
122 }
123
124 int DecodeKey(const Slice& k) {
125 if (IsHyperClock()) {
126 return DecodeKey16Bytes(k);
127 } else {
128 return DecodeKey32Bits(k);
129 }
130 }
131
132 int Lookup(std::shared_ptr<Cache> cache, int key) {
133 Cache::Handle* handle = cache->Lookup(EncodeKey(key));
134 const int r = (handle == nullptr) ? -1 : DecodeValue(cache->Value(handle));
135 if (handle != nullptr) {
136 cache->Release(handle);
137 }
138 return r;
139 }
140
141 void Insert(std::shared_ptr<Cache> cache, int key, int value,
142 int charge = 1) {
143 EXPECT_OK(cache->Insert(EncodeKey(key), EncodeValue(value), &kHelper,
144 charge, /*handle*/ nullptr, Cache::Priority::HIGH));
145 }
146
147 void Erase(std::shared_ptr<Cache> cache, int key) {
148 cache->Erase(EncodeKey(key));
149 }
150
151 int Lookup(int key) { return Lookup(cache_, key); }
152
153 void Insert(int key, int value, int charge = 1) {
154 Insert(cache_, key, value, charge);
155 }
156
157 void Erase(int key) { Erase(cache_, key); }
158
159 int Lookup2(int key) { return Lookup(cache2_, key); }
160
161 void Insert2(int key, int value, int charge = 1) {
162 Insert(cache2_, key, value, charge);
163 }
164
165 void Erase2(int key) { Erase(cache2_, key); }
166};

176TEST_P(CacheTest, UsageTest) {
177 // cache is std::shared_ptr and will be automatically cleaned up.
178 const size_t kCapacity = 100000;
179 auto cache = NewCache(kCapacity, 6, false, kDontChargeCacheMetadata);
180 auto precise_cache = NewCache(kCapacity, 0, false, kFullChargeCacheMetadata);
181 ASSERT_EQ(0, cache->GetUsage());
182 size_t baseline_meta_usage = precise_cache->GetUsage();
183 if (!IsHyperClock()) {
184 ASSERT_EQ(0, baseline_meta_usage);
185 }
186
187 size_t usage = 0;
188 char value[10] = "abcdef";
189 // make sure everything will be cached
190 for (int i = 1; i < 100; ++i) {
191 std::string key = EncodeKey(i);
192 auto kv_size = key.size() + 5;
193 ASSERT_OK(cache->Insert(key, value, &kDumbHelper, kv_size));
194 ASSERT_OK(precise_cache->Insert(key, value, &kDumbHelper, kv_size));
195 usage += kv_size;
196 ASSERT_EQ(usage, cache->GetUsage());
197 if (GetParam() == kFixedHyperClock) {
198 ASSERT_EQ(baseline_meta_usage + usage, precise_cache->GetUsage());
199 } else {
200 // AutoHyperClockCache meta usage grows in proportion to lifetime
201 // max number of entries. LRUCache in proportion to resident number of
202 // entries, though there is an untracked component proportional to
203 // lifetime max number of entries.
204 ASSERT_LT(usage, precise_cache->GetUsage());
205 }
206 }
207
208 cache->EraseUnRefEntries();
209 precise_cache->EraseUnRefEntries();
210 ASSERT_EQ(0, cache->GetUsage());
211 if (GetParam() != kAutoHyperClock) {
212 // NOTE: AutoHyperClockCache meta usage grows in proportion to lifetime
213 // max number of entries.
214 ASSERT_EQ(baseline_meta_usage, precise_cache->GetUsage());
215 }
216
217 // make sure the cache will be overloaded
218 for (size_t i = 1; i < kCapacity; ++i) {
219 std::string key = EncodeKey(static_cast<int>(1000 + i));
220 ASSERT_OK(cache->Insert(key, value, &kDumbHelper, key.size() + 5));
221 ASSERT_OK(precise_cache->Insert(key, value, &kDumbHelper, key.size() + 5));
222 }
223
224 // the usage should be close to the capacity
225 ASSERT_GT(kCapacity, cache->GetUsage());
226 ASSERT_GT(kCapacity, precise_cache->GetUsage());
227 ASSERT_LT(kCapacity * 0.95, cache->GetUsage());
228 if (!IsHyperClock()) {
229 ASSERT_LT(kCapacity * 0.95, precise_cache->GetUsage());
230 } else {
231 // estimated value size of 1 is weird for clock cache, because
232 // almost all of the capacity will be used for metadata, and due to only
233 // using power of 2 table sizes, we might hit strict occupancy limit
234 // before hitting capacity limit.
235 ASSERT_LT(kCapacity * 0.80, precise_cache->GetUsage());
236 }
237}

246TEST_P(CacheTest, PinnedUsageTest) {
247 // cache is std::shared_ptr and will be automatically cleaned up.
248 const size_t kCapacity = 200000;
249 auto cache = NewCache(kCapacity, 8, false, kDontChargeCacheMetadata);
250 auto precise_cache = NewCache(kCapacity, 8, false, kFullChargeCacheMetadata);
251 size_t baseline_meta_usage = precise_cache->GetUsage();
252 if (!IsHyperClock()) {
253 ASSERT_EQ(0, baseline_meta_usage);
254 }
255
256 size_t pinned_usage = 0;
257 char value[10] = "abcdef";
258
259 std::forward_list<Cache::Handle*> unreleased_handles;
260 std::forward_list<Cache::Handle*> unreleased_handles_in_precise_cache;
261
262 // Add entries. Unpin some of them after insertion. Then, pin some of them
263 // again. Check GetPinnedUsage().
264 for (int i = 1; i < 100; ++i) {
265 std::string key = EncodeKey(i);
266 auto kv_size = key.size() + 5;
267 Cache::Handle* handle;
268 Cache::Handle* handle_in_precise_cache;
269 ASSERT_OK(cache->Insert(key, value, &kDumbHelper, kv_size, &handle));
270 assert(handle);
271 ASSERT_OK(precise_cache->Insert(key, value, &kDumbHelper, kv_size,
272 &handle_in_precise_cache));
273 assert(handle_in_precise_cache);
274 pinned_usage += kv_size;
275 ASSERT_EQ(pinned_usage, cache->GetPinnedUsage());
276 ASSERT_LT(pinned_usage, precise_cache->GetPinnedUsage());
277 if (i % 2 == 0) {
278 cache->Release(handle);
279 precise_cache->Release(handle_in_precise_cache);
280 pinned_usage -= kv_size;
281 ASSERT_EQ(pinned_usage, cache->GetPinnedUsage());
282 ASSERT_LT(pinned_usage, precise_cache->GetPinnedUsage());
283 } else {
284 unreleased_handles.push_front(handle);
285 unreleased_handles_in_precise_cache.push_front(handle_in_precise_cache);
286 }
287 if (i % 3 == 0) {
288 unreleased_handles.push_front(cache->Lookup(key));
289 auto x = precise_cache->Lookup(key);
290 assert(x);
291 unreleased_handles_in_precise_cache.push_front(x);
292 // If i % 2 == 0, then the entry was unpinned before Lookup, so pinned
293 // usage increased
294 if (i % 2 == 0) {
295 pinned_usage += kv_size;
296 }
297 ASSERT_EQ(pinned_usage, cache->GetPinnedUsage());
298 ASSERT_LT(pinned_usage, precise_cache->GetPinnedUsage());
299 }
300 }
301 auto precise_cache_pinned_usage = precise_cache->GetPinnedUsage();
302 ASSERT_LT(pinned_usage, precise_cache_pinned_usage);
303
304 // check that overloading the cache does not change the pinned usage
305 for (size_t i = 1; i < 2 * kCapacity; ++i) {
306 std::string key = EncodeKey(static_cast<int>(1000 + i));
307 ASSERT_OK(cache->Insert(key, value, &kDumbHelper, key.size() + 5));
308 ASSERT_OK(precise_cache->Insert(key, value, &kDumbHelper, key.size() + 5));
309 }
310 ASSERT_EQ(pinned_usage, cache->GetPinnedUsage());
311 ASSERT_EQ(precise_cache_pinned_usage, precise_cache->GetPinnedUsage());
312
313 cache->EraseUnRefEntries();
314 precise_cache->EraseUnRefEntries();
315 ASSERT_EQ(pinned_usage, cache->GetPinnedUsage());
316 ASSERT_EQ(precise_cache_pinned_usage, precise_cache->GetPinnedUsage());
317
318 // release handles for pinned entries to prevent memory leaks
319 for (auto handle : unreleased_handles) {
320 cache->Release(handle);
321 }
322 for (auto handle : unreleased_handles_in_precise_cache) {
323 precise_cache->Release(handle);
324 }
325 ASSERT_EQ(0, cache->GetPinnedUsage());
326 ASSERT_EQ(0, precise_cache->GetPinnedUsage());
327 cache->EraseUnRefEntries();
328 precise_cache->EraseUnRefEntries();
329 ASSERT_EQ(0, cache->GetUsage());
330 if (GetParam() != kAutoHyperClock) {
331 // NOTE: AutoHyperClockCache meta usage grows in proportion to lifetime
332 // max number of entries.
333 ASSERT_EQ(baseline_meta_usage, precise_cache->GetUsage());
334 }
335}

442TEST_P(CacheTest, ExternalRefPinsEntries) {
443 Insert(100, 101);
444 Cache::Handle* h = cache_->Lookup(EncodeKey(100));
445 ASSERT_TRUE(cache_->Ref(h));
446 ASSERT_EQ(101, DecodeValue(cache_->Value(h)));
447 ASSERT_EQ(1U, cache_->GetUsage());
448
449 for (int i = 0; i < 3; ++i) {
450 if (i > 0) {
451 // First release (i == 1) corresponds to Ref(), second release (i == 2)
452 // corresponds to Lookup(). Then, since all external refs are released,
453 // the below insertions should push out the cache entry.
454 cache_->Release(h);
455 }
456 // double cache size because the usage bit in block cache prevents 100 from
457 // being evicted in the first kCacheSize iterations
458 for (int j = 0; j < 2 * kCacheSize + 100; j++) {
459 Insert(1000 + j, 2000 + j);
460 }
461 // Clock cache is even more stateful and needs more churn to evict
462 if (IsHyperClock()) {
463 for (int j = 0; j < kCacheSize; j++) {
464 Insert(11000 + j, 11000 + j);
465 }
466 }
467 if (i < 2) {
468 ASSERT_EQ(101, Lookup(100));
469 }
470 }
471 ASSERT_EQ(-1, Lookup(100));
472}

567TEST_P(CacheTest, HeavyEntries) {
568 // Add a bunch of light and heavy entries and then count the combined
569 // size of items still in the cache, which must be approximately the
570 // same as the total capacity.
571 const int kLight = 1;
572 const int kHeavy = 10;
573 int added = 0;
574 int index = 0;
575 while (added < 2 * kCacheSize) {
576 const int weight = (index & 1) ? kLight : kHeavy;
577 Insert(index, 1000 + index, weight);
578 added += weight;
579 index++;
580 }
581
582 int cached_weight = 0;
583 for (int i = 0; i < index; i++) {
584 const int weight = (i & 1 ? kLight : kHeavy);
585 int r = Lookup(i);
586 if (r >= 0) {
587 cached_weight += weight;
588 ASSERT_EQ(1000 + i, r);
589 }
590 }
591 ASSERT_LE(cached_weight, kCacheSize + kCacheSize / 10);
592}

673 // then decrease capacity to 7, final capacity should be 7
674 // and usage should be 7
675 for (int i = 5; i < 10; i++) {

677 Status s = cache.Insert(key, new Value(i + 1), 1, &handles[i]);

751TEST_P(CacheTest, OverCapacity) {
752 size_t n = 10;
753
754 // a LRUCache with n entries and one shard only
755 SharedCache cache{NewCache(n, 0, false)};
756 std::vector<TypedHandle*> handles(n + 1);
757
758 // Insert n+1 entries, but not releasing.
759 for (int i = 0; i < static_cast<int>(n + 1); i++) {
760 std::string key = EncodeKey(i + 1);
761 Status s = cache.Insert(key, new Value(i + 1), 1, &handles[i]);
762 ASSERT_TRUE(s.ok());
763 }
764
765 // Guess what's in the cache now?
766 for (int i = 0; i < static_cast<int>(n + 1); i++) {
767 std::string key = EncodeKey(i + 1);
768 auto h = cache.Lookup(key);
769 ASSERT_TRUE(h != nullptr);
770 if (h) {
771 cache.Release(h);
772 }
773 }
774
775 // the cache is over capacity since nothing could be evicted
776 ASSERT_EQ(n + 1U, cache.get()->GetUsage());
777 for (int i = 0; i < static_cast<int>(n + 1); i++) {
778 cache.Release(handles[i]);
779 }
780
781 if (IsHyperClock()) {
782 // Make sure eviction is triggered.
783 ASSERT_OK(cache.Insert(EncodeKey(-1), nullptr, 1, handles.data()));
784
785 // cache is under capacity now since elements were released
786 ASSERT_GE(n, cache.get()->GetUsage());
787
788 // clean up
789 cache.Release(handles[0]);
790 } else {
791 // LRUCache checks for over-capacity in Release.
792
793 // cache is exactly at capacity now with minimal eviction
794 ASSERT_EQ(n, cache.get()->GetUsage());
795
796 // element 0 is evicted and the rest is there
797 // This is consistent with the LRU policy since the element 0
798 // was released first
799 for (int i = 0; i < static_cast<int>(n + 1); i++) {
800 std::string key = EncodeKey(i + 1);
801 auto h = cache.Lookup(key);
802 if (h) {
803 ASSERT_NE(static_cast<size_t>(i), 0U);
804 cache.Release(h);
805 } else {
806 ASSERT_EQ(static_cast<size_t>(i), 0U);
807 }
808 }
809 }

841TEST_P(CacheTest, ApplyToAllEntriesDuringResize) {
842 // This is a mini-stress test of ApplyToAllEntries, to ensure
843 // items in the cache that are neither added nor removed
844 // during ApplyToAllEntries are counted exactly once.
845
846 // Insert some entries that we expect to be seen exactly once
847 // during iteration.
848 constexpr int kSpecialCharge = 2;
849 constexpr int kNotSpecialCharge = 1;
850 constexpr int kSpecialCount = 100;
851 size_t expected_usage = 0;
852 for (int i = 0; i < kSpecialCount; ++i) {
853 Insert(i, i * 2, kSpecialCharge);
854 expected_usage += kSpecialCharge;
855 }
856
857 // For callback
858 int special_count = 0;
859 const auto callback = [&](const Slice&, Cache::ObjectPtr, size_t charge,
860 const Cache::CacheItemHelper*) {
861 if (charge == static_cast<size_t>(kSpecialCharge)) {
862 ++special_count;
863 }
864 };
865
866 // Start counting
867 std::thread apply_thread([&]() {
868 // Use small average_entries_per_lock to make the problem difficult
869 Cache::ApplyToAllEntriesOptions opts;
870 opts.average_entries_per_lock = 2;
871 cache_->ApplyToAllEntries(callback, opts);
872 });
873
874 // In parallel, add more entries, enough to cause resize but not enough
875 // to cause ejections. (Note: if any cache shard is over capacity, there
876 // will be ejections)
877 for (int i = kSpecialCount * 1; i < kSpecialCount * 5; ++i) {
878 Insert(i, i * 2, kNotSpecialCharge);
879 expected_usage += kNotSpecialCharge;
880 }
881
882 apply_thread.join();
883 // verify no evictions
884 ASSERT_EQ(cache_->GetUsage(), expected_usage);
885 // verify everything seen in ApplyToAllEntries
886 ASSERT_EQ(special_count, kSpecialCount);
887}

967 // kQuasiRandomHashSeed should generate unique seeds (up to 2 billion before
968 // repeating)
969 UnorderedSet<uint32_t> seeds_seen;
970 // Roughly sqrt(number of possible values) for a decent chance at detecting
971 // a random collision if it's possible (shouldn't be)
972 uint16_t kSamples = 20000;
973 seeds_seen.reserve(kSamples);
974
975 // Hash seed should affect ordering of entries in the table, so we should
976 // have extremely high chance of seeing two entries ordered both ways.
977 bool seen_forward_order = false;
978 bool seen_reverse_order = false;

981 auto cache = NewCache(2, [=](ShardedCacheOptions& opts) {
982 opts.hash_seed = LRUCacheOptions::kQuasiRandomHashSeed;
983 opts.num_shard_bits = 0;
984 opts.metadata_charge_policy = kDontChargeCacheMetadata;
985 });
986 auto val = cache->GetHashSeed();
987 ASSERT_TRUE(seeds_seen.insert(val).second);
988
989 ASSERT_OK(cache->Insert(EncodeKey(1), nullptr, &kHelper, /*charge*/ 1));
990 ASSERT_OK(cache->Insert(EncodeKey(2), nullptr, &kHelper, /*charge*/ 1));
991
992 if (AreTwoCacheKeysOrdered(cache.get())) {
993 seen_forward_order = true;
994 } else {
995 seen_reverse_order = true;
996 }
997 }
998
999 ASSERT_TRUE(seen_forward_order);
