55def read_version(config):
56 majorRegex = re.compile(r"#define ROCKSDB_MAJOR\s([0-9]+)")
57 minorRegex = re.compile(r"#define ROCKSDB_MINOR\s([0-9]+)")
58 patchRegex = re.compile(r"#define ROCKSDB_PATCH\s([0-9]+)")
59 with open(config.version_file) as reader:
60 major = None
61 minor = None
62 patch = None
63 for line in reader:
64 if major is None:
65 major = majorRegex.match(line)
66 elif minor is None:
67 minor = minorRegex.match(line)
68 elif patch is None:
69 patch = patchRegex.match(line)
70
71 if patch is not None:
72 break
73
74 if patch is not None:
75 return (major.group(1), minor.group(1), patch.group(1))
76
77 # Didn't complete a match
78 return None

81def prepare(version_str, config):

121def main():
122 """Tool for running benchmark_compare.sh on the most recent build, for CI
123 This tool will
124
125 (1) Work out the current version of RocksDB
126 (2) Run benchmark_compare with that version alone
127 """
128
129 parser = argparse.ArgumentParser(
130 description="benchmark_compare.sh Python wrapper for CI."
131 )
132
133 # --tsvfile is the name of the file to read results from
134 # --esdocument is the ElasticSearch document to push these results into
135 #
136 parser.add_argument(
137 "--db_dir",
138 default="~/tmp/rocksdb-benchmark-datadir",
139 help="Database directory hierarchy to use",
140 )
141 parser.add_argument(
142 "--output_dir",
143 default="~/tmp/benchmark-results",
144 help="Benchmark output goes here",
145 )
146 parser.add_argument(
147 "--num_keys",
148 default="10000",
149 help="Number of database keys to use in benchmark test(s) (determines size of test job)",
150 )
151 args = parser.parse_args()
152 config = Config(args)
153
154 version = read_version(config)
155 if version is None:
156 raise Exception(f"Could not read RocksDB version from {config.version_file}")
157 version_str = f"{version[0]}.{version[1]}.{version[2]}"
158 logging.info(f"Run benchmark_ci with RocksDB version {version_str}")
159
160 prepare(version_str, config)
161
162 try:
163 env = get_benchmark_env()
164 env.append(("NUM_KEYS", args.num_keys))
165 cmd = [
166 config.benchmark_script,
167 config.data_dir,
168 config.results_dir,
169 version_str,
170 ]
171 logging.info(f"Run {cmd} env={env} cwd={config.benchmark_cwd}")
172 subprocess.run(cmd, env=dict(env), cwd=config.benchmark_cwd)
173
174 results(version_str, config)
175 finally:
176 cleanup(version_str, config)
177
178 return 0

182 sys.exit(main())
