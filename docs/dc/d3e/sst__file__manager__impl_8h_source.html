22// SstFileManager is used to track SST and blob files in the DB and control
23// their deletion rate. All SstFileManager public functions are thread-safe.
24class SstFileManagerImpl : public SstFileManager {
25 public:
26 explicit SstFileManagerImpl(const std::shared_ptr<SystemClock>& clock,
27 const std::shared_ptr<FileSystem>& fs,
28 const std::shared_ptr<Logger>& logger,
29 int64_t rate_bytes_per_sec,
30 double max_trash_db_ratio,
31 uint64_t bytes_max_delete_chunk);
32
33 // No copy
34 SstFileManagerImpl(const SstFileManagerImpl& sfm) = delete;
35 SstFileManagerImpl& operator=(const SstFileManagerImpl& sfm) = delete;
36
37 ~SstFileManagerImpl();
38
39 // DB will call OnAddFile whenever a new sst/blob file is added.
40 Status OnAddFile(const std::string& file_path);
41
42 // Overload where size of the file is provided by the caller rather than
43 // queried from the filesystem. This is an optimization.
44 Status OnAddFile(const std::string& file_path, uint64_t file_size);
45
46 // DB will call OnDeleteFile whenever a sst/blob file is deleted.
47 Status OnDeleteFile(const std::string& file_path);
48
49 // DB will call OnMoveFile whenever a sst/blob file is move to a new path.
50 Status OnMoveFile(const std::string& old_path, const std::string& new_path,
51 uint64_t* file_size = nullptr);
52
53 // DB will call OnUntrackFile when closing with an unowned SstFileManager.
54 Status OnUntrackFile(const std::string& file_path);
55
56 // Update the maximum allowed space that should be used by RocksDB, if
57 // the total size of the SST and blob files exceeds max_allowed_space, writes
58 // to RocksDB will fail.
59 //
60 // Setting max_allowed_space to 0 will disable this feature, maximum allowed
61 // space will be infinite (Default value).
62 //
63 // thread-safe.
64 void SetMaxAllowedSpaceUsage(uint64_t max_allowed_space) override;
65
66 void SetCompactionBufferSize(uint64_t compaction_buffer_size) override;
67
68 // Return true if the total size of SST and blob files exceeded the maximum
69 // allowed space usage.
70 //
71 // thread-safe.
72 bool IsMaxAllowedSpaceReached() override;
73
74 bool IsMaxAllowedSpaceReachedIncludingCompactions() override;
75
76 // Returns true is there is enough (approximate) space for the specified
77 // compaction. Space is approximate because this function conservatively
78 // estimates how much space is currently being used by compactions (i.e.
79 // if a compaction has started, this function bumps the used space by
80 // the full compaction size).
81 bool EnoughRoomForCompaction(ColumnFamilyData* cfd,
82 const std::vector<CompactionInputFiles>& inputs,
83 const Status& bg_error);
84
85 // Bookkeeping so total_file_sizes_ goes back to normal after compaction
86 // finishes
87 void OnCompactionCompletion(Compaction* c);
88
89 uint64_t GetCompactionsReservedSize();
90
91 // Return the total size of all tracked files.
92 uint64_t GetTotalSize() override;
93
94 // Return a map containing all tracked files and there corresponding sizes.
95 std::unordered_map<std::string, uint64_t> GetTrackedFiles() override;
96
97 // Return delete rate limit in bytes per second.
98 int64_t GetDeleteRateBytesPerSecond() override;
99
100 // Update the delete rate limit in bytes per second.
101 void SetDeleteRateBytesPerSecond(int64_t delete_rate) override;
102
103 // Return trash/DB size ratio where new files will be deleted immediately
104 double GetMaxTrashDBRatio() override;
105
106 // Update trash/DB size ratio where new files will be deleted immediately
107 void SetMaxTrashDBRatio(double ratio) override;
108
109 // Return the total size of trash files
110 uint64_t GetTotalTrashSize() override;
111
112 // Called by each DB instance using this sst file manager to reserve
113 // disk buffer space for recovery from out of space errors
114 void ReserveDiskBuffer(uint64_t buffer, const std::string& path);
115
116 // Set a flag upon encountering disk full. May enqueue the ErrorHandler
117 // instance for background polling and recovery
118 void StartErrorRecovery(ErrorHandler* db, Status bg_error);
119
120 // Remove the given Errorhandler instance from the recovery queue. Its
121 // not guaranteed
122 bool CancelErrorRecovery(ErrorHandler* db);
123
124 // Mark a file as trash and schedule its deletion. If force_bg is set, it
125 // forces the file to be deleting in the background regardless of DB size,
126 // except when rate limited delete is disabled.
127 virtual Status ScheduleFileDeletion(const std::string& file_path,
128 const std::string& dir_to_sync,
129 const bool force_bg = false);
130
131 // Delete an unaccounted file. The file is deleted immediately if slow
132 // deletion is disabled. A file with more than 1 hard links will be deleted
133 // immediately unless force_bg is set. In other cases, files will be scheduled
134 // for slow deletion, and assigned to the specified bucket if a legitimate one
135 // is provided. A legitimate bucket is one that is created with the
136 // `NewTrashBucket` API, and for which `WaitForEmptyTrashBucket` hasn't been
137 // called yet.
138 virtual Status ScheduleUnaccountedFileDeletion(
139 const std::string& file_path, const std::string& dir_to_sync,
140 const bool force_bg = false,
141 std::optional<int32_t> bucket = std::nullopt);
142
143 // Wait for all files being deleted in the background to finish or for
144 // destructor to be called.
145 virtual void WaitForEmptyTrash();
146
147 // Creates a new trash bucket. A legitimate bucket is only created and
148 // returned when slow deletion is enabled.
149 // For each bucket that is created and used, the user should also call
150 // `WaitForEmptyTrashBucket` after scheduling file deletions to make sure all
151 // the trash files are cleared.
152 std::optional<int32_t> NewTrashBucket();
153
154 // Wait for all the files in the specified bucket to be deleted in the
155 // background or for destructor to be called.
156 virtual void WaitForEmptyTrashBucket(int32_t bucket);
157
158 DeleteScheduler* delete_scheduler() { return &delete_scheduler_; }
159
160 // Stop the error recovery background thread. This should be called only
161 // once in the object's lifetime, and before the destructor
162 void Close();
163
164 void SetStatisticsPtr(const std::shared_ptr<Statistics>& stats) override {
165 stats_ = stats;
166 delete_scheduler_.SetStatisticsPtr(stats);
167 }
168
169 private:
170 // REQUIRES: mutex locked
171 void OnAddFileImpl(const std::string& file_path, uint64_t file_size);
172 // REQUIRES: mutex locked
173 void OnDeleteFileImpl(const std::string& file_path);
174
175 void ClearError();
176 bool CheckFreeSpace() {
177 return bg_err_.severity() == Status::Severity::kSoftError;
178 }
179
180 std::shared_ptr<SystemClock> clock_;
181 std::shared_ptr<FileSystem> fs_;
182 std::shared_ptr<Logger> logger_;
183 // Mutex to protect tracked_files_, total_files_size_
184 port::Mutex mu_;
185 // The summation of the sizes of all files in tracked_files_ map
186 uint64_t total_files_size_;
187 // Compactions should only execute if they can leave at least
188 // this amount of buffer space for logs and flushes
189 uint64_t compaction_buffer_size_;
190 // Estimated size of the current ongoing compactions
191 uint64_t cur_compactions_reserved_size_;
192 // A map containing all tracked files and there sizes
193 // file_path => file_size
194 std::unordered_map<std::string, uint64_t> tracked_files_;
195 // The maximum allowed space (in bytes) for sst and blob files.
196 uint64_t max_allowed_space_;
197 // DeleteScheduler used to throttle file deletion.
198 DeleteScheduler delete_scheduler_;
199 port::CondVar cv_;
200 // Flag to force error recovery thread to exit
201 bool closing_;
202 // Background error recovery thread
203 std::unique_ptr<port::Thread> bg_thread_;
204 // A path in the filesystem corresponding to this SFM. This is used for
205 // calling Env::GetFreeSpace. Posix requires a path in the filesystem
206 std::string path_;
207 // Save the current background error
208 Status bg_err_;
209 // Amount of free disk headroom before allowing recovery from hard errors
210 uint64_t reserved_disk_buffer_;
211 // For soft errors, amount of free disk space before we can allow
212 // compactions to run full throttle. If disk space is below this trigger,
213 // compactions will be gated by free disk space > input size
214 uint64_t free_space_trigger_;
215 // List of database error handler instances tracked by this SstFileManager.
216 std::list<ErrorHandler*> error_handler_list_;
217 // Pointer to ErrorHandler instance that is currently processing recovery
218 ErrorHandler* cur_instance_;
219 std::shared_ptr<Statistics> stats_;
220};
