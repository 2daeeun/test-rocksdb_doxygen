18enum PlainTableEntryType : unsigned char {
19 kFullKey = 0,
20 kPrefixFromPreviousKey = 1,
21 kKeySuffix = 2,
22};

26// Control byte:
27// First two bits indicate type of entry
28// Other bytes are inlined sizes. If all bits are 1 (0x03F), overflow bytes
29// are used. key_size-0x3F will be encoded as a variint32 after this bytes.
30
31const unsigned char kSizeInlineLimit = 0x3F;
32
33// Return 0 for error
34size_t EncodeSize(PlainTableEntryType type, uint32_t key_size,
35 char* out_buffer) {
36 out_buffer[0] = type << 6;
37
38 if (key_size < static_cast<uint32_t>(kSizeInlineLimit)) {
39 // size inlined
40 out_buffer[0] |= static_cast<char>(key_size);
41 return 1;
42 } else {
43 out_buffer[0] |= kSizeInlineLimit;
44 char* ptr = EncodeVarint32(out_buffer + 1, key_size - kSizeInlineLimit);
45 return ptr - out_buffer;
46 }
47}
48} // namespace

50// Fill bytes_read with number of bytes read.
51inline Status PlainTableKeyDecoder::DecodeSize(uint32_t start_offset,
52 PlainTableEntryType* entry_type,
53 uint32_t* key_size,
54 uint32_t* bytes_read) {
55 Slice next_byte_slice;
56 bool success = file_reader_.Read(start_offset, 1, &next_byte_slice);
57 if (!success) {
58 return file_reader_.status();
59 }
60 *entry_type = static_cast<PlainTableEntryType>(
61 (static_cast<unsigned char>(next_byte_slice[0]) & ~kSizeInlineLimit) >>
62 6);
63 char inline_key_size = next_byte_slice[0] & kSizeInlineLimit;
64 if (inline_key_size < kSizeInlineLimit) {
65 *key_size = inline_key_size;
66 *bytes_read = 1;
67 return Status::OK();
68 } else {
69 uint32_t extra_size;
70 uint32_t tmp_bytes_read;
71 success = file_reader_.ReadVarint32(start_offset + 1, &extra_size,
72 &tmp_bytes_read);
73 if (!success) {
74 return file_reader_.status();
75 }
76 assert(tmp_bytes_read > 0);
77 *key_size = kSizeInlineLimit + extra_size;
78 *bytes_read = tmp_bytes_read + 1;
79 return Status::OK();
80 }
81}

83IOStatus PlainTableKeyEncoder::AppendKey(const Slice& key,
84 WritableFileWriter* file,
85 uint64_t* offset, char* meta_bytes_buf,
86 size_t* meta_bytes_buf_size) {
87 ParsedInternalKey parsed_key;
88 Status pik_status =
89 ParseInternalKey(key, &parsed_key, false /* log_err_key */); // TODO
90 if (!pik_status.ok()) {
91 return IOStatus::Corruption(pik_status.getState());
92 }
93
94 Slice key_to_write = key; // Portion of internal key to write out.
95
96 uint32_t user_key_size = static_cast<uint32_t>(key.size() - 8);
97 const IOOptions opts;
98
99 if (encoding_type_ == kPlain) {
100 if (fixed_user_key_len_ == kPlainTableVariableLength) {
101 // Write key length
102 char key_size_buf[5]; // tmp buffer for key size as varint32
103 char* ptr = EncodeVarint32(key_size_buf, user_key_size);
104 assert(ptr <= key_size_buf + sizeof(key_size_buf));
105 auto len = ptr - key_size_buf;
106 IOStatus io_s = file->Append(opts, Slice(key_size_buf, len));
107 if (!io_s.ok()) {
108 return io_s;
109 }
110 *offset += len;
111 }
112 } else {
113 assert(encoding_type_ == kPrefix);
114 char size_bytes[12];
115 size_t size_bytes_pos = 0;
116
117 Slice prefix =
118 prefix_extractor_->Transform(Slice(key.data(), user_key_size));
119 if (key_count_for_prefix_ == 0 || prefix != pre_prefix_.GetUserKey() ||
120 key_count_for_prefix_ % index_sparseness_ == 0) {
121 key_count_for_prefix_ = 1;
122 pre_prefix_.SetUserKey(prefix);
123 size_bytes_pos += EncodeSize(kFullKey, user_key_size, size_bytes);
124 IOStatus io_s = file->Append(opts, Slice(size_bytes, size_bytes_pos));
125 if (!io_s.ok()) {
126 return io_s;
127 }
128 *offset += size_bytes_pos;
129 } else {
130 key_count_for_prefix_++;
131 if (key_count_for_prefix_ == 2) {
132 // For second key within a prefix, need to encode prefix length
133 size_bytes_pos +=
134 EncodeSize(kPrefixFromPreviousKey,
135 static_cast<uint32_t>(pre_prefix_.GetUserKey().size()),
136 size_bytes + size_bytes_pos);
137 }
138 uint32_t prefix_len =
139 static_cast<uint32_t>(pre_prefix_.GetUserKey().size());
140 size_bytes_pos += EncodeSize(kKeySuffix, user_key_size - prefix_len,
141 size_bytes + size_bytes_pos);
142 IOStatus io_s = file->Append(opts, Slice(size_bytes, size_bytes_pos));
143 if (!io_s.ok()) {
144 return io_s;
145 }
146 *offset += size_bytes_pos;
147 key_to_write = Slice(key.data() + prefix_len, key.size() - prefix_len);
148 }
149 }
150
151 // Encode full key
152 // For value size as varint32 (up to 5 bytes).
153 // If the row is of value type with seqId 0, flush the special flag together
154 // in this buffer to safe one file append call, which takes 1 byte.
155 if (parsed_key.sequence == 0 && parsed_key.type == kTypeValue) {
156 IOStatus io_s =
157 file->Append(opts, Slice(key_to_write.data(), key_to_write.size() - 8));
158 if (!io_s.ok()) {
159 return io_s;
160 }
161 *offset += key_to_write.size() - 8;
162 meta_bytes_buf[*meta_bytes_buf_size] = PlainTableFactory::kValueTypeSeqId0;
163 *meta_bytes_buf_size += 1;
164 } else {
165 IOStatus io_s = file->Append(opts, key_to_write);
166 if (!io_s.ok()) {
167 return io_s;
168 }
169 *offset += key_to_write.size();
170 }
171
172 return IOStatus::OK();
173}

182bool PlainTableFileReader::ReadNonMmap(uint32_t file_offset, uint32_t len,
183 Slice* out) {
184 const uint32_t kPrefetchSize = 256u;
185
186 // Try to read from buffers.
187 for (uint32_t i = 0; i < num_buf_; i++) {
188 Buffer* buffer = buffers_[num_buf_ - 1 - i].get();
189 if (file_offset >= buffer->buf_start_offset &&
190 file_offset + len <= buffer->buf_start_offset + buffer->buf_len) {
191 *out = GetFromBuffer(buffer, file_offset, len);
192 return true;
193 }
194 }
195
196 Buffer* new_buffer;
197 // Data needed is not in any of the buffer. Allocate a new buffer.
198 if (num_buf_ < buffers_.size()) {
199 // Add a new buffer
200 new_buffer = new Buffer();
201 buffers_[num_buf_++].reset(new_buffer);
202 } else {
203 // Now simply replace the last buffer. Can improve the placement policy
204 // if needed.
205 new_buffer = buffers_[num_buf_ - 1].get();
206 }
207
208 assert(file_offset + len <= file_info_->data_end_offset);
209 uint32_t size_to_read = std::min(file_info_->data_end_offset - file_offset,
210 std::max(kPrefetchSize, len));
211 if (size_to_read > new_buffer->buf_capacity) {
212 new_buffer->buf.reset(new char[size_to_read]);
213 new_buffer->buf_capacity = size_to_read;
214 new_buffer->buf_len = 0;
215 }
216 Slice read_result;
217 // TODO: rate limit plain table reads.
218 Status s =
219 file_info_->file->Read(IOOptions(), file_offset, size_to_read,
220 &read_result, new_buffer->buf.get(), nullptr);
221 if (!s.ok()) {
222 status_ = s;
223 return false;
224 }
225 new_buffer->buf_start_offset = file_offset;
226 new_buffer->buf_len = size_to_read;
227 *out = GetFromBuffer(new_buffer, file_offset, len);
228 return true;
229}

231inline bool PlainTableFileReader::ReadVarint32(uint32_t offset, uint32_t* out,
232 uint32_t* bytes_read) {
233 if (file_info_->is_mmap_mode) {
234 const char* start = file_info_->file_data.data() + offset;
235 const char* limit =
236 file_info_->file_data.data() + file_info_->data_end_offset;
237 const char* key_ptr = GetVarint32Ptr(start, limit, out);
238 assert(key_ptr != nullptr);
239 *bytes_read = static_cast<uint32_t>(key_ptr - start);
240 return true;
241 } else {
242 return ReadVarint32NonMmap(offset, out, bytes_read);
243 }
244}

246bool PlainTableFileReader::ReadVarint32NonMmap(uint32_t offset, uint32_t* out,
247 uint32_t* bytes_read) {
248 const char* start;
249 const char* limit;
250 const uint32_t kMaxVarInt32Size = 6u;
251 uint32_t bytes_to_read =
252 std::min(file_info_->data_end_offset - offset, kMaxVarInt32Size);
253 Slice bytes;
254 if (!Read(offset, bytes_to_read, &bytes)) {
255 return false;
256 }
257 start = bytes.data();
258 limit = bytes.data() + bytes.size();
259
260 const char* key_ptr = GetVarint32Ptr(start, limit, out);
261 *bytes_read =
262 (key_ptr != nullptr) ? static_cast<uint32_t>(key_ptr - start) : 0;
263 return true;
264}

266Status PlainTableKeyDecoder::ReadInternalKey(
267 uint32_t file_offset, uint32_t user_key_size, ParsedInternalKey* parsed_key,
268 uint32_t* bytes_read, bool* internal_key_valid, Slice* internal_key) {
269 Slice tmp_slice;
270 bool success = file_reader_.Read(file_offset, user_key_size + 1, &tmp_slice);
271 if (!success) {
272 return file_reader_.status();
273 }
274 if (tmp_slice[user_key_size] == PlainTableFactory::kValueTypeSeqId0) {
275 // Special encoding for the row with seqID=0
276 parsed_key->user_key = Slice(tmp_slice.data(), user_key_size);
277 parsed_key->sequence = 0;
278 parsed_key->type = kTypeValue;
279 *bytes_read += user_key_size + 1;
280 *internal_key_valid = false;
281 } else {
282 success = file_reader_.Read(file_offset, user_key_size + 8, internal_key);
283 if (!success) {
284 return file_reader_.status();
285 }
286 *internal_key_valid = true;
287 Status pik_status = ParseInternalKey(*internal_key, parsed_key,
288 false /* log_err_key */); // TODO
289 if (!pik_status.ok()) {
290 return Status::Corruption(
291 Slice("Corrupted key found during next key read. "),
292 pik_status.getState());
293 }
294 *bytes_read += user_key_size + 8;
295 }
296 return Status::OK();
297}

299Status PlainTableKeyDecoder::NextPlainEncodingKey(uint32_t start_offset,
300 ParsedInternalKey* parsed_key,
301 Slice* internal_key,
302 uint32_t* bytes_read,
303 bool* /*seekable*/) {
304 uint32_t user_key_size = 0;
305 Status s;
306 if (fixed_user_key_len_ != kPlainTableVariableLength) {
307 user_key_size = fixed_user_key_len_;
308 } else {
309 uint32_t tmp_size = 0;
310 uint32_t tmp_read;
311 bool success =
312 file_reader_.ReadVarint32(start_offset, &tmp_size, &tmp_read);
313 if (!success) {
314 return file_reader_.status();
315 }
316 assert(tmp_read > 0);
317 user_key_size = tmp_size;
318 *bytes_read = tmp_read;
319 }
320 // dummy initial value to avoid compiler complain
321 bool decoded_internal_key_valid = true;
322 Slice decoded_internal_key;
323 s = ReadInternalKey(start_offset + *bytes_read, user_key_size, parsed_key,
324 bytes_read, &decoded_internal_key_valid,
325 &decoded_internal_key);
326 if (!s.ok()) {
327 return s;
328 }
329 if (!file_reader_.file_info()->is_mmap_mode) {
330 cur_key_.SetInternalKey(*parsed_key);
331 parsed_key->user_key =
332 Slice(cur_key_.GetInternalKey().data(), user_key_size);
333 if (internal_key != nullptr) {
334 *internal_key = cur_key_.GetInternalKey();
335 }
336 } else if (internal_key != nullptr) {
337 if (decoded_internal_key_valid) {
338 *internal_key = decoded_internal_key;
339 } else {
340 // Need to copy out the internal key
341 cur_key_.SetInternalKey(*parsed_key);
342 *internal_key = cur_key_.GetInternalKey();
343 }
344 }
345 return Status::OK();
346}

348Status PlainTableKeyDecoder::NextPrefixEncodingKey(
349 uint32_t start_offset, ParsedInternalKey* parsed_key, Slice* internal_key,
350 uint32_t* bytes_read, bool* seekable) {
351 PlainTableEntryType entry_type;
352
353 bool expect_suffix = false;
354 Status s;
355 do {
356 uint32_t size = 0;
357 // dummy initial value to avoid compiler complain
358 bool decoded_internal_key_valid = true;
359 uint32_t my_bytes_read = 0;
360 s = DecodeSize(start_offset + *bytes_read, &entry_type, &size,
361 &my_bytes_read);
362 if (!s.ok()) {
363 return s;
364 }
365 if (my_bytes_read == 0) {
366 return Status::Corruption("Unexpected EOF when reading size of the key");
367 }
368 *bytes_read += my_bytes_read;
369
370 switch (entry_type) {
371 case kFullKey: {
372 expect_suffix = false;
373 Slice decoded_internal_key;
374 s = ReadInternalKey(start_offset + *bytes_read, size, parsed_key,
375 bytes_read, &decoded_internal_key_valid,
376 &decoded_internal_key);
377 if (!s.ok()) {
378 return s;
379 }
380 if (!file_reader_.file_info()->is_mmap_mode ||
381 (internal_key != nullptr && !decoded_internal_key_valid)) {
382 // In non-mmap mode, always need to make a copy of keys returned to
383 // users, because after reading value for the key, the key might
384 // be invalid.
385 cur_key_.SetInternalKey(*parsed_key);
386 saved_user_key_ = cur_key_.GetUserKey();
387 if (!file_reader_.file_info()->is_mmap_mode) {
388 parsed_key->user_key =
389 Slice(cur_key_.GetInternalKey().data(), size);
390 }
391 if (internal_key != nullptr) {
392 *internal_key = cur_key_.GetInternalKey();
393 }
394 } else {
395 if (internal_key != nullptr) {
396 *internal_key = decoded_internal_key;
397 }
398 saved_user_key_ = parsed_key->user_key;
399 }
400 break;
401 }
402 case kPrefixFromPreviousKey: {
403 if (seekable != nullptr) {
404 *seekable = false;
405 }
406 prefix_len_ = size;
407 assert(prefix_extractor_ == nullptr ||
408 prefix_extractor_->Transform(saved_user_key_).size() ==
409 prefix_len_);
410 // Need read another size flag for suffix
411 expect_suffix = true;
412 break;
413 }
414 case kKeySuffix: {
415 expect_suffix = false;
416 if (seekable != nullptr) {
417 *seekable = false;
418 }
419
420 Slice tmp_slice;
421 s = ReadInternalKey(start_offset + *bytes_read, size, parsed_key,
422 bytes_read, &decoded_internal_key_valid,
423 &tmp_slice);
424 if (!s.ok()) {
425 return s;
426 }
427 if (!file_reader_.file_info()->is_mmap_mode) {
428 // In non-mmap mode, we need to make a copy of keys returned to
429 // users, because after reading value for the key, the key might
430 // be invalid.
431 // saved_user_key_ points to cur_key_. We are making a copy of
432 // the prefix part to another string, and construct the current
433 // key from the prefix part and the suffix part back to cur_key_.
434 std::string tmp =
435 Slice(saved_user_key_.data(), prefix_len_).ToString();
436 cur_key_.Reserve(prefix_len_ + size);
437 cur_key_.SetInternalKey(tmp, *parsed_key);
438 parsed_key->user_key =
439 Slice(cur_key_.GetInternalKey().data(), prefix_len_ + size);
440 saved_user_key_ = cur_key_.GetUserKey();
441 } else {
442 cur_key_.Reserve(prefix_len_ + size);
443 cur_key_.SetInternalKey(Slice(saved_user_key_.data(), prefix_len_),
444 *parsed_key);
445 }
446 parsed_key->user_key = cur_key_.GetUserKey();
447 if (internal_key != nullptr) {
448 *internal_key = cur_key_.GetInternalKey();
449 }
450 break;
451 }
452 default:
453 return Status::Corruption("Un-identified size flag.");
454 }
455 } while (expect_suffix); // Another round if suffix is expected.
456 return Status::OK();
457}

459Status PlainTableKeyDecoder::NextKey(uint32_t start_offset,
460 ParsedInternalKey* parsed_key,
461 Slice* internal_key, Slice* value,
462 uint32_t* bytes_read, bool* seekable) {
463 assert(value != nullptr);
464 Status s = NextKeyNoValue(start_offset, parsed_key, internal_key, bytes_read,
465 seekable);
466 if (s.ok()) {
467 assert(bytes_read != nullptr);
468 uint32_t value_size;
469 uint32_t value_size_bytes;
470 bool success = file_reader_.ReadVarint32(start_offset + *bytes_read,
471 &value_size, &value_size_bytes);
472 if (!success) {
473 return file_reader_.status();
474 }
475 if (value_size_bytes == 0) {
476 return Status::Corruption(
477 "Unexpected EOF when reading the next value's size.");
478 }
479 *bytes_read += value_size_bytes;
480 success = file_reader_.Read(start_offset + *bytes_read, value_size, value);
481 if (!success) {
482 return file_reader_.status();
483 }
484 *bytes_read += value_size;
485 }
486 return s;
487}

489Status PlainTableKeyDecoder::NextKeyNoValue(uint32_t start_offset,
490 ParsedInternalKey* parsed_key,
491 Slice* internal_key,
492 uint32_t* bytes_read,
493 bool* seekable) {
494 *bytes_read = 0;
495 if (seekable != nullptr) {
496 *seekable = true;
497 }
498 if (encoding_type_ == kPlain) {
499 return NextPlainEncodingKey(start_offset, parsed_key, internal_key,
500 bytes_read, seekable);
501 } else {
502 assert(encoding_type_ == kPrefix);
503 return NextPrefixEncodingKey(start_offset, parsed_key, internal_key,
504 bytes_read, seekable);
505 }
506}
