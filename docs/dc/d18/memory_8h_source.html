96/* MALLOC is a macro that helps avoid a common error:
97 * Suppose I write
98 * struct foo *x = malloc(sizeof(struct foo));
99 * That works fine. But if I change it to this, I've probably made an mistake:
100 * struct foo *x = malloc(sizeof(struct bar));
101 * It can get worse, since one might have something like
102 * struct foo *x = malloc(sizeof(struct foo *))
103 * which looks reasonable, but it allocoates enough to hold a pointer instead of
104 * the amount needed for the struct. So instead, write struct foo *MALLOC(x);
105 * and you cannot go wrong.
106 */
107#define MALLOC(v) CAST_FROM_VOIDP(v, toku_malloc(sizeof(*v)))
108/* MALLOC_N is like calloc(Except no 0ing of data): It makes an array. Write
109 * int *MALLOC_N(5,x);
110 * to make an array of 5 integers.
111 */
112#define MALLOC_N(n, v) CAST_FROM_VOIDP(v, toku_malloc((n) * sizeof(*v)))
113#define MALLOC_N_ALIGNED(align, n, v) \
114 CAST_FROM_VOIDP(v, toku_malloc_aligned((align), (n) * sizeof(*v)))
115
116// CALLOC_N is like calloc with auto-figuring out size of members
117#define CALLOC_N(n, v) CAST_FROM_VOIDP(v, toku_calloc((n), sizeof(*v)))
118
119#define CALLOC(v) CALLOC_N(1, v)
120
121// XMALLOC macros are like MALLOC except they abort if the operation fails
122#define XMALLOC(v) CAST_FROM_VOIDP(v, toku_xmalloc(sizeof(*v)))
123#define XMALLOC_N(n, v) CAST_FROM_VOIDP(v, toku_xmalloc((n) * sizeof(*v)))
124#define XCALLOC_N(n, v) CAST_FROM_VOIDP(v, toku_xcalloc((n), (sizeof(*v))))
125#define XCALLOC(v) XCALLOC_N(1, v)
126#define XREALLOC(v, s) CAST_FROM_VOIDP(v, toku_xrealloc(v, s))
127#define XREALLOC_N(n, v) CAST_FROM_VOIDP(v, toku_xrealloc(v, (n) * sizeof(*v)))
128
129#define XMALLOC_N_ALIGNED(align, n, v) \
130 CAST_FROM_VOIDP(v, toku_xmalloc_aligned((align), (n) * sizeof(*v)))
131
132#define XMEMDUP(dst, src) CAST_FROM_VOIDP(dst, toku_xmemdup(src, sizeof(*src)))
133#define XMEMDUP_N(dst, src, len) CAST_FROM_VOIDP(dst, toku_xmemdup(src, len))
