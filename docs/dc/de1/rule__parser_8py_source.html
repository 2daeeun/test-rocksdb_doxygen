28class Rule(Section):
29 def __init__(self, name):
30 super().__init__(name)
31 self.conditions = None
32 self.suggestions = None
33 self.overlap_time_seconds = None
34 self.trigger_entities = None
35 self.trigger_column_families = None

37 def set_parameter(self, key, value):
38 # If the Rule is associated with a single suggestion/condition, then
39 # value will be a string and not a list. Hence, convert it to a single
40 # element list before storing it in self.suggestions or
41 # self.conditions.
42 if key == "conditions":
43 if isinstance(value, str):
44 self.conditions = [value]
45 else:
46 self.conditions = value
47 elif key == "suggestions":
48 if isinstance(value, str):
49 self.suggestions = [value]
50 else:
51 self.suggestions = value
52 elif key == "overlap_time_period":
53 self.overlap_time_seconds = value
54
55 def get_suggestions(self):
56 return self.suggestions
57
58 def perform_checks(self):
59 if not self.conditions or len(self.conditions) < 1:
60 raise ValueError(self.name + ": rule must have at least one condition")
61 if not self.suggestions or len(self.suggestions) < 1:
62 raise ValueError(self.name + ": rule must have at least one suggestion")
63 if self.overlap_time_seconds:
64 if len(self.conditions) != 2:
65 raise ValueError(
66 self.name
67 + ": rule must be associated with 2 conditions\
68 in order to check for a time dependency between them"
69 )
70 time_format = r"^\d+[s|m|h|d]$" # noqa
71 if not re.match(time_format, self.overlap_time_seconds, re.IGNORECASE):
72 raise ValueError(
73 self.name + r": overlap_time_seconds format: \d+[s|m|h|d]"
74 )
75 else: # convert to seconds
76 in_seconds = int(self.overlap_time_seconds[:-1])
77 if self.overlap_time_seconds[-1] == "m":
78 in_seconds *= 60
79 elif self.overlap_time_seconds[-1] == "h":
80 in_seconds *= 60 * 60
81 elif self.overlap_time_seconds[-1] == "d":
82 in_seconds *= 24 * 60 * 60
83 self.overlap_time_seconds = in_seconds

85 def get_overlap_timestamps(self, key1_trigger_epochs, key2_trigger_epochs):
86 # this method takes in 2 timeseries i.e. timestamps at which the
87 # rule's 2 TIME_SERIES conditions were triggered and it finds
88 # (if present) the first pair of timestamps at which the 2 conditions
89 # were triggered within 'overlap_time_seconds' of each other
90 key1_lower_bounds = [
91 epoch - self.overlap_time_seconds for epoch in key1_trigger_epochs
92 ]
93 key1_lower_bounds.sort()
94 key2_trigger_epochs.sort()
95 trigger_ix = 0
96 overlap_pair = None
97 for key1_lb in key1_lower_bounds:
98 while key2_trigger_epochs[trigger_ix] < key1_lb and trigger_ix < len(
99 key2_trigger_epochs
100 ):
101 trigger_ix += 1
102 if trigger_ix >= len(key2_trigger_epochs):
103 break
104 if key2_trigger_epochs[trigger_ix] <= key1_lb + (
105 2 * self.overlap_time_seconds
106 ):
107 overlap_pair = (
108 key2_trigger_epochs[trigger_ix],
109 key1_lb + self.overlap_time_seconds,
110 )
111 break
112 return overlap_pair

120 def is_triggered(self, conditions_dict, column_families):
121 if self.overlap_time_seconds:
122 condition1 = conditions_dict[self.conditions[0]]
123 condition2 = conditions_dict[self.conditions[1]]
124 if not (
125 condition1.get_data_source() is DataSource.Type.TIME_SERIES
126 and condition2.get_data_source() is DataSource.Type.TIME_SERIES
127 ):
128 raise ValueError(self.name + ": need 2 timeseries conditions")
129
130 map1 = condition1.get_trigger()
131 map2 = condition2.get_trigger()
132 if not (map1 and map2):
133 return False
134
135 self.trigger_entities = {}
136 is_triggered = False
137 entity_intersection = set(map1.keys()).intersection(set(map2.keys()))
138 for entity in entity_intersection:
139 overlap_timestamps_pair = self.get_overlap_timestamps(
140 list(map1[entity].keys()), list(map2[entity].keys())
141 )
142 if overlap_timestamps_pair:
143 self.trigger_entities[entity] = overlap_timestamps_pair
144 is_triggered = True
145 if is_triggered:
146 self.trigger_column_families = set(column_families)
147 return is_triggered
148 else:
149 all_conditions_triggered = True
150 self.trigger_column_families = set(column_families)
151 for cond_name in self.conditions:
152 cond = conditions_dict[cond_name]
153 if not cond.get_trigger():
154 all_conditions_triggered = False
155 break
156 if (
157 cond.get_data_source() is DataSource.Type.LOG
158 or cond.get_data_source() is DataSource.Type.DB_OPTIONS
159 ):
160 cond_col_fam = set(cond.get_trigger().keys())
161 if NO_COL_FAMILY in cond_col_fam:
162 cond_col_fam = set(column_families)
163 self.trigger_column_families = (
164 self.trigger_column_families.intersection(cond_col_fam)
165 )
166 elif cond.get_data_source() is DataSource.Type.TIME_SERIES:
167 cond_entities = set(cond.get_trigger().keys())
168 if self.trigger_entities is None:
169 self.trigger_entities = cond_entities
170 else:
171 self.trigger_entities = self.trigger_entities.intersection(
172 cond_entities
173 )
174 if not (self.trigger_entities or self.trigger_column_families):
175 all_conditions_triggered = False
176 break
177 if not all_conditions_triggered: # clean up if rule not triggered
178 self.trigger_column_families = None
179 self.trigger_entities = None
180 return all_conditions_triggered

182 def __repr__(self):
183 # Append conditions
184 rule_string = "Rule: " + self.name + " has conditions:: "
185 is_first = True
186 for cond in self.conditions:
187 if is_first:
188 rule_string += cond
189 is_first = False
190 else:
191 rule_string += " AND " + cond
192 # Append suggestions
193 rule_string += "\nsuggestions:: "
194 is_first = True
195 for sugg in self.suggestions:
196 if is_first:
197 rule_string += sugg
198 is_first = False
199 else:
200 rule_string += ", " + sugg
201 if self.trigger_entities:
202 rule_string += ", entities:: " + str(self.trigger_entities)
203 if self.trigger_column_families:
204 rule_string += ", col_fam:: " + str(self.trigger_column_families)
205 # Return constructed string
206 return rule_string

209class Suggestion(Section):
210 class Action(Enum):
211 set = 1
212 increase = 2
213 decrease = 3
214
215 def __init__(self, name):
216 super().__init__(name)
217 self.option = None
218 self.action = None
219 self.suggested_values = None
220 self.description = None
221
222 def set_parameter(self, key, value):
223 if key == "option":
224 # Note:
225 # case 1: 'option' is supported by Rocksdb OPTIONS file; in this
226 # case the option belongs to one of the sections in the config
227 # file and it's name is prefixed by "<section_type>."
228 # case 2: 'option' is not supported by Rocksdb OPTIONS file; the
229 # option is not expected to have the character '.' in its name
230 self.option = value
231 elif key == "action":
232 if self.option and not value:
233 raise ValueError(self.name + ": provide action for option")
234 self.action = self.Action[value]
235 elif key == "suggested_values":
236 if isinstance(value, str):
237 self.suggested_values = [value]
238 else:
239 self.suggested_values = value
240 elif key == "description":
241 self.description = value
242
243 def perform_checks(self):
244 if not self.description:
245 if not self.option:
246 raise ValueError(self.name + ": provide option or description")
247 if not self.action:
248 raise ValueError(self.name + ": provide action for option")
249 if self.action is self.Action.set and not self.suggested_values:
250 raise ValueError(self.name + ": provide suggested value for option")
251
252 def __repr__(self):
253 sugg_string = "Suggestion: " + self.name
254 if self.description:
255 sugg_string += " description : " + self.description
256 else:
257 sugg_string += " option : " + self.option + " action : " + self.action.name
258 if self.suggested_values:
259 sugg_string += " suggested_values : " + str(self.suggested_values)
260 return sugg_string

354class TimeSeriesCondition(Condition):
355 @classmethod
356 def create(cls, base_condition):
357 base_condition.set_data_source(DataSource.Type["TIME_SERIES"])
358 base_condition.__class__ = cls
359 return base_condition
360
361 def set_parameter(self, key, value):
362 if key == "keys":
363 if isinstance(value, str):
364 self.keys = [value]
365 else:
366 self.keys = value
367 elif key == "behavior":
368 self.behavior = TimeSeriesData.Behavior[value]
369 elif key == "rate_threshold":
370 self.rate_threshold = float(value)
371 elif key == "window_sec":
372 self.window_sec = int(value)
373 elif key == "evaluate":
374 self.expression = value
375 elif key == "aggregation_op":
376 self.aggregation_op = TimeSeriesData.AggregationOperator[value]
377
378 def perform_checks(self):
379 if not self.keys:
380 raise ValueError(self.name + ": specify timeseries key")
381 if not self.behavior:
382 raise ValueError(self.name + ": specify triggering behavior")
383 if self.behavior is TimeSeriesData.Behavior.bursty:
384 if not self.rate_threshold:
385 raise ValueError(self.name + ": specify rate burst threshold")
386 if not self.window_sec:
387 self.window_sec = 300 # default window length is 5 minutes
388 if len(self.keys) > 1:
389 raise ValueError(self.name + ": specify only one key")
390 elif self.behavior is TimeSeriesData.Behavior.evaluate_expression:
391 if not (self.expression):
392 raise ValueError(self.name + ": specify evaluation expression")
393 else:
394 raise ValueError(self.name + ": trigger behavior not supported")
395
396 def __repr__(self):
397 ts_cond_str = "TimeSeriesCondition: " + self.name
398 ts_cond_str += " statistics: " + str(self.keys)
399 ts_cond_str += " behavior: " + self.behavior.name
400 if self.behavior is TimeSeriesData.Behavior.bursty:
401 ts_cond_str += " rate_threshold: " + str(self.rate_threshold)
402 ts_cond_str += " window_sec: " + str(self.window_sec)
403 if self.behavior is TimeSeriesData.Behavior.evaluate_expression:
404 ts_cond_str += " expression: " + self.expression
405 if hasattr(self, "aggregation_op"):
406 ts_cond_str += " aggregation_op: " + self.aggregation_op.name
407 if self.trigger:
408 ts_cond_str += " trigger: " + str(self.trigger)
409 return ts_cond_str

429 def load_rules_from_spec(self):
430 self.initialise_fields()
431 with open(self.file_path) as db_rules:
432 curr_section = None
433 for line in db_rules:
434 line = IniParser.remove_trailing_comment(line)
435 if not line:
436 continue
437 element = IniParser.get_element(line)
438 if element is IniParser.Element.comment:
439 continue
440 elif element is not IniParser.Element.key_val:
441 curr_section = element # it's a new IniParser header
442 section_name = IniParser.get_section_name(line)
443 if element is IniParser.Element.rule:
444 new_rule = Rule(section_name)
445 self.rules_dict[section_name] = new_rule
446 elif element is IniParser.Element.cond:
447 new_cond = Condition(section_name)
448 self.conditions_dict[section_name] = new_cond
449 elif element is IniParser.Element.sugg:
450 new_suggestion = Suggestion(section_name)
451 self.suggestions_dict[section_name] = new_suggestion
452 elif element is IniParser.Element.key_val:
453 key, value = IniParser.get_key_value_pair(line)
454 if curr_section is IniParser.Element.rule:
455 new_rule.set_parameter(key, value)
456 elif curr_section is IniParser.Element.cond:
457 if key == "source":
458 if value == "LOG":
459 new_cond = LogCondition.create(new_cond)
460 elif value == "OPTIONS":
461 new_cond = OptionCondition.create(new_cond)
462 elif value == "TIME_SERIES":
463 new_cond = TimeSeriesCondition.create(new_cond)
464 else:
465 new_cond.set_parameter(key, value)
466 elif curr_section is IniParser.Element.sugg:
467 new_suggestion.set_parameter(key, value)
