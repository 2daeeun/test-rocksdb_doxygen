16Status UncompressionDictReader::Create(
17 const BlockBasedTable* table, const ReadOptions& ro,
18 FilePrefetchBuffer* prefetch_buffer, bool use_cache, bool prefetch,
19 bool pin, BlockCacheLookupContext* lookup_context,
20 std::unique_ptr<UncompressionDictReader>* uncompression_dict_reader) {
21 assert(table);
22 assert(table->get_rep());
23 assert(!pin || prefetch);
24 assert(uncompression_dict_reader);
25
26 CachableEntry<UncompressionDict> uncompression_dict;
27 if (prefetch || !use_cache) {
28 const Status s = ReadUncompressionDictionary(
29 table, prefetch_buffer, ro, use_cache, nullptr /* get_context */,
30 lookup_context, &uncompression_dict);
31 if (!s.ok()) {
32 return s;
33 }
34
35 if (use_cache && !pin) {
36 uncompression_dict.Reset();
37 }
38 }
39
40 uncompression_dict_reader->reset(
41 new UncompressionDictReader(table, std::move(uncompression_dict)));
42
43 return Status::OK();
44}

46Status UncompressionDictReader::ReadUncompressionDictionary(
47 const BlockBasedTable* table, FilePrefetchBuffer* prefetch_buffer,
48 const ReadOptions& read_options, bool use_cache, GetContext* get_context,
49 BlockCacheLookupContext* lookup_context,
50 CachableEntry<UncompressionDict>* uncompression_dict) {
51 // TODO: add perf counter for compression dictionary read time
52
53 assert(table);
54 assert(uncompression_dict);
55 assert(uncompression_dict->IsEmpty());
56
57 const BlockBasedTable::Rep* const rep = table->get_rep();
58 assert(rep);
59 assert(!rep->compression_dict_handle.IsNull());
60
61 const Status s = table->RetrieveBlock(
62 prefetch_buffer, read_options, rep->compression_dict_handle,
63 UncompressionDict::GetEmptyDict(), uncompression_dict, get_context,
64 lookup_context,
65 /* for_compaction */ false, use_cache,
66 /* async_read */ false, /* use_block_cache_for_lookup */ true);
67
68 if (!s.ok()) {
69 ROCKS_LOG_WARN(
70 rep->ioptions.logger,
71 "Encountered error while reading data from compression dictionary "
72 "block %s",
73 s.ToString().c_str());
74 }
75
76 return s;
77}

79Status UncompressionDictReader::GetOrReadUncompressionDictionary(
80 FilePrefetchBuffer* prefetch_buffer, const ReadOptions& ro,
81 GetContext* get_context, BlockCacheLookupContext* lookup_context,
82 CachableEntry<UncompressionDict>* uncompression_dict) const {
83 assert(uncompression_dict);
84
85 if (!uncompression_dict_.IsEmpty()) {
86 uncompression_dict->SetUnownedValue(uncompression_dict_.GetValue());
87 return Status::OK();
88 }
89
90 return ReadUncompressionDictionary(table_, prefetch_buffer, ro,
91 cache_dictionary_blocks(), get_context,
92 lookup_context, uncompression_dict);
93}

95size_t UncompressionDictReader::ApproximateMemoryUsage() const {
96 assert(!uncompression_dict_.GetOwnValue() ||
97 uncompression_dict_.GetValue() != nullptr);
98 size_t usage = uncompression_dict_.GetOwnValue()
99 ? uncompression_dict_.GetValue()->ApproximateMemoryUsage()
100 : 0;
101
102#ifdef ROCKSDB_MALLOC_USABLE_SIZE
103 usage += malloc_usable_size(const_cast<UncompressionDictReader*>(this));
104#else
105 usage += sizeof(*this);
106#endif // ROCKSDB_MALLOC_USABLE_SIZE
107
108 return usage;
109}
