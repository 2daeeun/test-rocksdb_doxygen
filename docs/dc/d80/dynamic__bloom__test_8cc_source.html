39struct KeyMaker {
40 uint64_t a;
41 uint64_t b;
42
43 // Sequential, within a hash function block
44 inline Slice Seq(uint64_t i) {
45 a = i;
46 return Slice(reinterpret_cast<char *>(&a), sizeof(a));
47 }
48 // Not quite sequential, varies across hash function blocks
49 inline Slice Nonseq(uint64_t i) {
50 a = i;
51 b = i * 123;
52 return Slice(reinterpret_cast<char *>(this), sizeof(*this));
53 }
54 inline Slice Key(uint64_t i, bool nonseq) {
55 return nonseq ? Nonseq(i) : Seq(i);
56 }
57};

110static uint32_t NextNum(uint32_t num) {
111 if (num < 10) {
112 num += 1;
113 } else if (num < 100) {
114 num += 10;
115 } else if (num < 1000) {
116 num += 100;

123TEST_F(DynamicBloomTest, VaryingLengths) {
124 KeyMaker km;
125
126 // Count number of filters that significantly exceed the false positive rate
127 int mediocre_filters = 0;
128 int good_filters = 0;
129 uint32_t num_probes = static_cast<uint32_t>(FLAGS_num_probes);
130
131 fprintf(stderr, "bits_per_key: %d num_probes: %d\n", FLAGS_bits_per_key,
132 num_probes);
133
134 // NB: FP rate impact of 32-bit hash is noticeable starting around 10M keys.
135 // But that effect is hidden if using sequential keys (unique hashes).
136 for (bool nonseq : {false, true}) {
137 const uint32_t max_num = FLAGS_enable_perf ? 40000000 : 400000;
138 for (uint32_t num = 1; num <= max_num; num = NextNum(num)) {
139 uint32_t bloom_bits = 0;
140 Arena arena;
141 bloom_bits = num * FLAGS_bits_per_key;
142 DynamicBloom bloom(&arena, bloom_bits, num_probes);
143 for (uint64_t i = 0; i < num; i++) {
144 bloom.Add(km.Key(i, nonseq));
145 ASSERT_TRUE(bloom.MayContain(km.Key(i, nonseq)));
146 }
147
148 // All added keys must match
149 for (uint64_t i = 0; i < num; i++) {
150 ASSERT_TRUE(bloom.MayContain(km.Key(i, nonseq)));
151 }
152
153 // Check false positive rate
154 int result = 0;
155 for (uint64_t i = 0; i < 30000; i++) {
156 if (bloom.MayContain(km.Key(i + 1000000000, nonseq))) {
157 result++;
158 }
159 }
160 double rate = result / 30000.0;
161
162 fprintf(stderr,
163 "False positives (%s keys): "
164 "%5.2f%% @ num = %6u, bloom_bits = %6u\n",
165 nonseq ? "nonseq" : "seq", rate * 100.0, num, bloom_bits);
166
167 if (rate > 0.0125) {
168 mediocre_filters++; // Allowed, but not too often
169 } else {
170 good_filters++;
171 }
172 }
173 }
174
175 fprintf(stderr, "Filters: %d good, %d mediocre\n", good_filters,
176 mediocre_filters);
177 ASSERT_LE(mediocre_filters, good_filters / 25);
178}

180TEST_F(DynamicBloomTest, perf) {
181 KeyMaker km;
182 StopWatchNano timer(SystemClock::Default().get());
183 uint32_t num_probes = static_cast<uint32_t>(FLAGS_num_probes);
184
185 if (!FLAGS_enable_perf) {
186 return;
187 }
188
189 for (uint32_t m = 1; m <= 8; ++m) {
190 Arena arena;
191 const uint32_t num_keys = m * 8 * 1024 * 1024;
192 fprintf(stderr, "testing %" PRIu32 "M keys\n", m * 8);
193
194 DynamicBloom std_bloom(&arena, num_keys * 10, num_probes);
195
196 timer.Start();
197 for (uint64_t i = 1; i <= num_keys; ++i) {
198 std_bloom.Add(km.Seq(i));
199 }
200
201 uint64_t elapsed = timer.ElapsedNanos();
202 fprintf(stderr, "dynamic bloom, avg add latency %3g\n",
203 static_cast<double>(elapsed) / num_keys);
204
205 uint32_t count = 0;
206 timer.Start();
207 for (uint64_t i = 1; i <= num_keys; ++i) {
208 if (std_bloom.MayContain(km.Seq(i))) {
209 ++count;
210 }
211 }
212 ASSERT_EQ(count, num_keys);
213 elapsed = timer.ElapsedNanos();
214 assert(count > 0);
215 fprintf(stderr, "dynamic bloom, avg query latency %3g\n",
216 static_cast<double>(elapsed) / count);
217 }

220TEST_F(DynamicBloomTest, concurrent_with_perf) {
221 uint32_t num_probes = static_cast<uint32_t>(FLAGS_num_probes);
222
223 uint32_t m_limit = FLAGS_enable_perf ? 8 : 1;
224
225 uint32_t num_threads = 4;
226 std::vector<port::Thread> threads;
227
228 // NB: Uses sequential keys for speed, but that hides the FP rate
229 // impact of 32-bit hash, which is noticeable starting around 10M keys
230 // when they vary across hashing blocks.
231 for (uint32_t m = 1; m <= m_limit; ++m) {
232 Arena arena;
233 const uint32_t num_keys = m * 8 * 1024 * 1024;
234 fprintf(stderr, "testing %" PRIu32 "M keys\n", m * 8);
235
236 DynamicBloom std_bloom(&arena, num_keys * 10, num_probes);
237
238 std::atomic<uint64_t> elapsed(0);
239
240 std::function<void(size_t)> adder([&](size_t t) {
241 KeyMaker km;
242 StopWatchNano timer(SystemClock::Default().get());
243 timer.Start();
244 for (uint64_t i = 1 + t; i <= num_keys; i += num_threads) {
245 std_bloom.AddConcurrently(km.Seq(i));
246 }
247 elapsed += timer.ElapsedNanos();
248 });
249 for (size_t t = 0; t < num_threads; ++t) {
250 threads.emplace_back(adder, t);
251 }
252 while (threads.size() > 0) {
253 threads.back().join();
254 threads.pop_back();
255 }
256
257 fprintf(stderr,
258 "dynamic bloom, avg parallel add latency %3g"
259 " nanos/key\n",
260 static_cast<double>(elapsed) / num_threads / num_keys);
261
262 elapsed = 0;
263 std::function<void(size_t)> hitter([&](size_t t) {
264 KeyMaker km;
265 StopWatchNano timer(SystemClock::Default().get());
266 timer.Start();
267 for (uint64_t i = 1 + t; i <= num_keys; i += num_threads) {
268 bool f = std_bloom.MayContain(km.Seq(i));
269 ASSERT_TRUE(f);
270 }
271 elapsed += timer.ElapsedNanos();
272 });
273 for (size_t t = 0; t < num_threads; ++t) {
274 threads.emplace_back(hitter, t);
275 }
276 while (threads.size() > 0) {
277 threads.back().join();
278 threads.pop_back();
279 }
280
281 fprintf(stderr,
282 "dynamic bloom, avg parallel hit latency %3g"
283 " nanos/key\n",
284 static_cast<double>(elapsed) / num_threads / num_keys);
285
286 elapsed = 0;
287 std::atomic<uint32_t> false_positives(0);
288 std::function<void(size_t)> misser([&](size_t t) {
289 KeyMaker km;
290 StopWatchNano timer(SystemClock::Default().get());
291 timer.Start();
292 for (uint64_t i = num_keys + 1 + t; i <= 2 * num_keys; i += num_threads) {
293 bool f = std_bloom.MayContain(km.Seq(i));
294 if (f) {
295 ++false_positives;
296 }
297 }
298 elapsed += timer.ElapsedNanos();
299 });
300 for (size_t t = 0; t < num_threads; ++t) {
301 threads.emplace_back(misser, t);
302 }
303 while (threads.size() > 0) {
304 threads.back().join();
305 threads.pop_back();
306 }
307
308 fprintf(stderr,
309 "dynamic bloom, avg parallel miss latency %3g"
310 " nanos/key, %f%% false positive rate\n",
311 static_cast<double>(elapsed) / num_threads / num_keys,
312 false_positives.load() * 100.0 / num_keys);
313 }
314}
