49 // Sanity check that all fields have been set
50 assert(offset_ != ~uint64_t{0});
51 assert(size_ != ~uint64_t{0});
52 PutVarint64Varint64(dst, offset_, size_);
53}

55char* BlockHandle::EncodeTo(char* dst) const {
56 // Sanity check that all fields have been set
57 assert(offset_ != ~uint64_t{0});
58 assert(size_ != ~uint64_t{0});
59 char* cur = EncodeVarint64(dst, offset_);
60 cur = EncodeVarint64(cur, size_);
61 return cur;

100void IndexValue::EncodeTo(std::string* dst, bool have_first_key,
101 const BlockHandle* previous_handle) const {
102 if (previous_handle) {
103 // WART: this is specific to Block-based table
104 assert(handle.offset() == previous_handle->offset() +
105 previous_handle->size() +
106 BlockBasedTable::kBlockTrailerSize);
107 PutVarsignedint64(dst, handle.size() - previous_handle->size());
108 } else {
109 handle.EncodeTo(dst);
110 }
111 assert(dst->size() != 0);
112
113 if (have_first_key) {
114 PutLengthPrefixedSlice(dst, first_internal_key);
115 }

118Status IndexValue::DecodeFrom(Slice* input, bool have_first_key,
119 const BlockHandle* previous_handle) {
120 if (previous_handle) {
121 int64_t delta;
122 if (!GetVarsignedint64(input, &delta)) {
123 return Status::Corruption("bad delta-encoded index value");
124 }
125 // WART: this is specific to Block-based table
126 handle = BlockHandle(previous_handle->offset() + previous_handle->size() +
127 BlockBasedTable::kBlockTrailerSize,
128 previous_handle->size() + delta);
129 } else {
130 Status s = handle.DecodeFrom(input);
131 if (!s.ok()) {
132 return s;
133 }
134 }
135
136 if (!have_first_key) {
137 first_internal_key = Slice();
138 } else if (!GetLengthPrefixedSlice(input, &first_internal_key)) {
139 return Status::Corruption("bad first key in block info");
140 }
141
142 return Status::OK();
143}

156inline bool IsLegacyFooterFormat(uint64_t magic_number) {
157 return magic_number == kLegacyBlockBasedTableMagicNumber ||
158 magic_number == kLegacyPlainTableMagicNumber;
159}
160inline uint64_t UpconvertLegacyFooterFormat(uint64_t magic_number) {
161 if (magic_number == kLegacyBlockBasedTableMagicNumber) {
162 return kBlockBasedTableMagicNumber;
163 }
164 if (magic_number == kLegacyPlainTableMagicNumber) {
165 return kPlainTableMagicNumber;
166 }
167 assert(false);
168 return magic_number;
169}
170inline uint64_t DownconvertToLegacyFooterFormat(uint64_t magic_number) {
171 if (magic_number == kBlockBasedTableMagicNumber) {
172 return kLegacyBlockBasedTableMagicNumber;
173 }
174 if (magic_number == kPlainTableMagicNumber) {
175 return kLegacyPlainTableMagicNumber;
176 }
177 assert(false);
178 return magic_number;
179}
180inline uint8_t BlockTrailerSizeForMagicNumber(uint64_t magic_number) {
181 if (magic_number == kBlockBasedTableMagicNumber ||
182 magic_number == kLegacyBlockBasedTableMagicNumber) {
183 return static_cast<uint8_t>(BlockBasedTable::kBlockTrailerSize);
184 } else {
185 return 0;
186 }
187}

189// Footer format, in three parts:
190// * Part1
191// -> format_version == 0 (inferred from legacy magic number)
192// <empty> (0 bytes)
193// -> format_version >= 1
194// checksum type (char, 1 byte)
195// * Part2
196// -> format_version <= 5
197// metaindex handle (varint64 offset, varint64 size)
198// index handle (varint64 offset, varint64 size)
199// <zero padding> for part2 size = 2 * BlockHandle::kMaxEncodedLength = 40
200// - This padding is unchecked/ignored
201// -> format_version >= 6
202// extended magic number (4 bytes) = 0x3e 0x00 0x7a 0x00
203// - Also surely invalid (size 0) handles if interpreted as older version
204// - (Helps ensure a corrupted format_version doesn't get us far with no
205// footer checksum.)
206// footer_checksum (uint32LE, 4 bytes)
207// - Checksum of above checksum type of whole footer, with this field
208// set to all zeros.
209// base_context_checksum (uint32LE, 4 bytes)
210// metaindex block size (uint32LE, 4 bytes)
211// - Assumed to be immediately before footer, < 4GB
212// <zero padding> (24 bytes, reserved for future use)
213// - Brings part2 size also to 40 bytes
214// - Checked that last eight bytes == 0, so reserved for a future
215// incompatible feature (but under format_version=6)
216// * Part3
217// -> format_version == 0 (inferred from legacy magic number)
218// legacy magic number (8 bytes)
219// -> format_version >= 1 (inferred from NOT legacy magic number)
220// format_version (uint32LE, 4 bytes), also called "footer version"
221// newer magic number (8 bytes)
222const std::array<char, 4> kExtendedMagic{{0x3e, 0x00, 0x7a, 0x00}};
223constexpr size_t kFooterPart2Size = 2 * BlockHandle::kMaxEncodedLength;
224} // namespace

226Status FooterBuilder::Build(uint64_t magic_number, uint32_t format_version,
227 uint64_t footer_offset, ChecksumType checksum_type,
228 const BlockHandle& metaindex_handle,
229 const BlockHandle& index_handle,
230 uint32_t base_context_checksum) {
231 assert(magic_number != Footer::kNullTableMagicNumber);
232 assert(IsSupportedFormatVersion(format_version));
233
234 char* part2;
235 char* part3;
236 if (format_version > 0) {
237 slice_ = Slice(data_.data(), Footer::kNewVersionsEncodedLength);
238 // Generate parts 1 and 3
239 char* cur = data_.data();
240 // Part 1
241 *(cur++) = checksum_type;
242 // Part 2
243 part2 = cur;
244 // Skip over part 2 for now
245 cur += kFooterPart2Size;
246 // Part 3
247 part3 = cur;
248 EncodeFixed32(cur, format_version);
249 cur += 4;
250 EncodeFixed64(cur, magic_number);
251 assert(cur + 8 == slice_.data() + slice_.size());
252 } else {
253 slice_ = Slice(data_.data(), Footer::kVersion0EncodedLength);
254 // Legacy SST files use kCRC32c checksum but it's not stored in footer.
255 assert(checksum_type == kNoChecksum || checksum_type == kCRC32c);
256 // Generate part 3 (part 1 empty, skip part 2 for now)
257 part2 = data_.data();
258 part3 = part2 + kFooterPart2Size;
259 char* cur = part3;
260 // Use legacy magic numbers to indicate format_version=0, for
261 // compatibility. No other cases should use format_version=0.
262 EncodeFixed64(cur, DownconvertToLegacyFooterFormat(magic_number));
263 assert(cur + 8 == slice_.data() + slice_.size());
264 }
265
266 if (format_version >= 6) {
267 if (BlockTrailerSizeForMagicNumber(magic_number) != 0) {
268 // base context checksum required for table formats with block checksums
269 assert(base_context_checksum != 0);
270 assert(ChecksumModifierForContext(base_context_checksum, 0) != 0);
271 } else {
272 // base context checksum not used
273 assert(base_context_checksum == 0);
274 assert(ChecksumModifierForContext(base_context_checksum, 0) == 0);
275 }
276
277 // Start populating Part 2
278 char* cur = data_.data() + /* part 1 size */ 1;
279 // Set extended magic of part2
280 std::copy(kExtendedMagic.begin(), kExtendedMagic.end(), cur);
281 cur += kExtendedMagic.size();
282 // Fill checksum data with zeros (for later computing checksum)
283 char* checksum_data = cur;
284 EncodeFixed32(cur, 0);
285 cur += 4;
286 // Save base context checksum
287 EncodeFixed32(cur, base_context_checksum);
288 cur += 4;
289 // Compute and save metaindex size
290 uint32_t metaindex_size = static_cast<uint32_t>(metaindex_handle.size());
291 if (metaindex_size != metaindex_handle.size()) {
292 return Status::NotSupported("Metaindex block size > 4GB");
293 }
294 // Metaindex must be adjacent to footer
295 assert(metaindex_size == 0 ||
296 metaindex_handle.offset() + metaindex_handle.size() ==
297 footer_offset - BlockTrailerSizeForMagicNumber(magic_number));
298 EncodeFixed32(cur, metaindex_size);
299 cur += 4;
300
301 // Zero pad remainder (for future use)
302 std::fill_n(cur, 24U, char{0});
303 assert(cur + 24 == part3);
304
305 // Compute checksum, add context
306 uint32_t checksum = ComputeBuiltinChecksum(
307 checksum_type, data_.data(), Footer::kNewVersionsEncodedLength);
308 checksum +=
309 ChecksumModifierForContext(base_context_checksum, footer_offset);
310 // Store it
311 EncodeFixed32(checksum_data, checksum);
312 } else {
313 // Base context checksum not used
314 assert(!FormatVersionUsesContextChecksum(format_version));
315 // Should be left empty
316 assert(base_context_checksum == 0);
317 assert(ChecksumModifierForContext(base_context_checksum, 0) == 0);
318
319 // Populate all of part 2
320 char* cur = part2;
321 cur = metaindex_handle.EncodeTo(cur);
322 cur = index_handle.EncodeTo(cur);
323 // Zero pad remainder
324 std::fill(cur, part3, char{0});
325 }
326 return Status::OK();
327}

329Status Footer::DecodeFrom(Slice input, uint64_t input_offset,
330 uint64_t enforce_table_magic_number) {
331 // Only decode to unused Footer
332 assert(table_magic_number_ == kNullTableMagicNumber);
333 assert(input != nullptr);
334 assert(input.size() >= kMinEncodedLength);
335
336 const char* magic_ptr = input.data() + input.size() - kMagicNumberLengthByte;
337 uint64_t magic = DecodeFixed64(magic_ptr);
338
339 // We check for legacy formats here and silently upconvert them
340 bool legacy = IsLegacyFooterFormat(magic);
341 if (legacy) {
342 magic = UpconvertLegacyFooterFormat(magic);
343 }
344 if (enforce_table_magic_number != 0 && enforce_table_magic_number != magic) {
345 return Status::Corruption("Bad table magic number: expected " +
346 std::to_string(enforce_table_magic_number) +
347 ", found " + std::to_string(magic));
348 }
349 table_magic_number_ = magic;
350 block_trailer_size_ = BlockTrailerSizeForMagicNumber(magic);
351
352 // Parse Part3
353 const char* part3_ptr = magic_ptr;
354 uint32_t computed_checksum = 0;
355 uint64_t footer_offset = 0;
356 if (legacy) {
357 // The size is already asserted to be at least kMinEncodedLength
358 // at the beginning of the function
359 input.remove_prefix(input.size() - kVersion0EncodedLength);
360 format_version_ = 0 /* legacy */;
361 checksum_type_ = kCRC32c;
362 } else {
363 part3_ptr = magic_ptr - 4;
364 format_version_ = DecodeFixed32(part3_ptr);
365 if (UNLIKELY(!IsSupportedFormatVersion(format_version_))) {
366 return Status::Corruption("Corrupt or unsupported format_version: " +
367 std::to_string(format_version_));
368 }
369 // All known format versions >= 1 occupy exactly this many bytes.
370 if (UNLIKELY(input.size() < kNewVersionsEncodedLength)) {
371 return Status::Corruption("Input is too short to be an SST file");
372 }
373 uint64_t adjustment = input.size() - kNewVersionsEncodedLength;
374 input.remove_prefix(adjustment);
375 footer_offset = input_offset + adjustment;
376
377 // Parse Part1
378 char chksum = input.data()[0];
379 checksum_type_ = lossless_cast<ChecksumType>(chksum);
380 if (UNLIKELY(!IsSupportedChecksumType(checksum_type()))) {
381 return Status::Corruption("Corrupt or unsupported checksum type: " +
382 std::to_string(lossless_cast<uint8_t>(chksum)));
383 }
384 // This is the most convenient place to compute the checksum
385 if (checksum_type_ != kNoChecksum && format_version_ >= 6) {
386 std::array<char, kNewVersionsEncodedLength> copy_without_checksum;
387 std::copy_n(input.data(), kNewVersionsEncodedLength,
388 copy_without_checksum.data());
389 EncodeFixed32(&copy_without_checksum[5], 0); // Clear embedded checksum
390 computed_checksum =
391 ComputeBuiltinChecksum(checksum_type(), copy_without_checksum.data(),
392 kNewVersionsEncodedLength);
393 }
394 // Consume checksum type field
395 input.remove_prefix(1);
396 }
397
398 // Parse Part2
399 if (format_version_ >= 6) {
400 Slice ext_magic(input.data(), 4);
401 if (UNLIKELY(ext_magic.compare(Slice(kExtendedMagic.data(),
402 kExtendedMagic.size())) != 0)) {
403 return Status::Corruption("Bad extended magic number: 0x" +
404 ext_magic.ToString(/*hex*/ true));
405 }
406 input.remove_prefix(4);
407 uint32_t stored_checksum = 0, metaindex_size = 0;
408 bool success;
409 success = GetFixed32(&input, &stored_checksum);
410 assert(success);
411 success = GetFixed32(&input, &base_context_checksum_);
412 assert(success);
413 if (UNLIKELY(ChecksumModifierForContext(base_context_checksum_, 0) == 0)) {
414 return Status::Corruption("Invalid base context checksum");
415 }
416 computed_checksum +=
417 ChecksumModifierForContext(base_context_checksum_, footer_offset);
418 if (UNLIKELY(computed_checksum != stored_checksum)) {
419 return Status::Corruption("Footer at " + std::to_string(footer_offset) +
420 " checksum mismatch");
421 }
422 success = GetFixed32(&input, &metaindex_size);
423 assert(success);
424 (void)success;
425 uint64_t metaindex_end = footer_offset - GetBlockTrailerSize();
426 metaindex_handle_ =
427 BlockHandle(metaindex_end - metaindex_size, metaindex_size);
428
429 // Mark unpopulated
430 index_handle_ = BlockHandle::NullBlockHandle();
431
432 // 16 bytes of unchecked reserved padding
433 input.remove_prefix(16U);
434
435 // 8 bytes of checked reserved padding (expected to be zero unless using a
436 // future feature).
437 uint64_t reserved = 0;
438 success = GetFixed64(&input, &reserved);
439 assert(success);
440 if (UNLIKELY(reserved != 0)) {
441 return Status::NotSupported(
442 "File uses a future feature not supported in this version");
443 }
444 // End of part 2
445 assert(input.data() == part3_ptr);
446 } else {
447 // format_version_ < 6
448 Status result = metaindex_handle_.DecodeFrom(&input);
449 if (result.ok()) {
450 result = index_handle_.DecodeFrom(&input);
451 }
452 if (!result.ok()) {
453 return result;
454 }
455 // Padding in part2 is ignored
456 }
457 return Status::OK();
458}

460std::string Footer::ToString() const {
461 std::string result;
462 result.reserve(1024);
463
464 result.append("metaindex handle: " + metaindex_handle_.ToString() +
465 " offset: " + std::to_string(metaindex_handle_.offset()) +
466 " size: " + std::to_string(metaindex_handle_.size()) + "\n ");
467 result.append("index handle: " + index_handle_.ToString() +
468 " offset: " + std::to_string(index_handle_.offset()) +
469 " size: " + std::to_string(index_handle_.size()) + "\n ");
470 result.append("table_magic_number: " + std::to_string(table_magic_number_) +
471 "\n ");
472 if (!IsLegacyFooterFormat(table_magic_number_)) {
473 result.append("format version: " + std::to_string(format_version_) + "\n");
474 }
475 return result;
476}

478static Status ReadFooterFromFileInternal(const IOOptions& opts,
479 RandomAccessFileReader* file,
480 FileSystem& fs,
481 FilePrefetchBuffer* prefetch_buffer,
482 uint64_t file_size, Footer* footer,
483 uint64_t enforce_table_magic_number) {
484 if (file_size < Footer::kMinEncodedLength) {
485 return Status::Corruption("file is too short (" +
486 std::to_string(file_size) +
487 " bytes) to be an "
488 "sstable: " +
489 file->file_name());
490 }
491
492 std::array<char, Footer::kMaxEncodedLength + 1> footer_buf;
493 AlignedBuf internal_buf;
494 Slice footer_input;
495 uint64_t read_offset = (file_size > Footer::kMaxEncodedLength)
496 ? file_size - Footer::kMaxEncodedLength
497 : 0;
498 Status s;
499 // TODO: Need to pass appropriate deadline to TryReadFromCache(). Right now,
500 // there is no readahead for point lookups, so TryReadFromCache will fail if
501 // the required data is not in the prefetch buffer. Once deadline is enabled
502 // for iterator, TryReadFromCache might do a readahead. Revisit to see if we
503 // need to pass a timeout at that point
504 // TODO: rate limit footer reads.
505 if (prefetch_buffer == nullptr ||
506 !prefetch_buffer->TryReadFromCache(opts, file, read_offset,
507 Footer::kMaxEncodedLength,
508 &footer_input, nullptr)) {
509 if (file->use_direct_io()) {
510 s = file->Read(opts, read_offset, Footer::kMaxEncodedLength,
511 &footer_input, nullptr, &internal_buf);
512 } else {
513 s = file->Read(opts, read_offset, Footer::kMaxEncodedLength,
514 &footer_input, footer_buf.data(), nullptr);
515 }
516 if (!s.ok()) {
517 return s;
518 }
519 }
520
521 TEST_SYNC_POINT_CALLBACK("ReadFooterFromFileInternal:0", &footer_input);
522
523 // Check that we actually read the whole footer from the file. It may be
524 // that size isn't correct.
525 if (footer_input.size() < Footer::kMinEncodedLength) {
526 uint64_t size_on_disk = 0;
527 if (fs.GetFileSize(file->file_name(), IOOptions(), &size_on_disk, nullptr)
528 .ok()) {
529 // Similar to CheckConsistency message, but not completely sure the
530 // expected size always came from manifest.
531 return Status::Corruption("Sst file size mismatch: " + file->file_name() +
532 ". Expected " + std::to_string(file_size) +
533 ", actual size " +
534 std::to_string(size_on_disk) + "\n");
535 } else {
536 return Status::Corruption(
537 "Missing SST footer data in file " + file->file_name() +
538 " File too short? Expected size: " + std::to_string(file_size));
539 }
540 }
541
542 s = footer->DecodeFrom(footer_input, read_offset, enforce_table_magic_number);
543 if (!s.ok()) {
544 s = Status::CopyAppendMessage(s, " in ", file->file_name());
545 return s;
546 }
547 return Status::OK();
548}
549
550Status ReadFooterFromFile(const IOOptions& opts, RandomAccessFileReader* file,
551 FileSystem& fs, FilePrefetchBuffer* prefetch_buffer,
552 uint64_t file_size, Footer* footer,
553 uint64_t enforce_table_magic_number,
554 Statistics* stats) {
555 Status s =
556 ReadFooterFromFileInternal(opts, file, fs, prefetch_buffer, file_size,
557 footer, enforce_table_magic_number);
558 if (s.IsCorruption() &&
559 CheckFSFeatureSupport(&fs, FSSupportedOps::kVerifyAndReconstructRead)) {
560 IOOptions new_opts = opts;
561 new_opts.verify_and_reconstruct_read = true;
562 footer->Reset();
563 s = ReadFooterFromFileInternal(new_opts, file, fs,
564 /*prefetch_buffer=*/nullptr, file_size,
565 footer, enforce_table_magic_number);
566 RecordTick(stats, FILE_READ_CORRUPTION_RETRY_COUNT);
567 if (s.ok()) {
568 RecordTick(stats, FILE_READ_CORRUPTION_RETRY_SUCCESS_COUNT);
569 }
570 }
571 return s;
572}

574namespace {
575// Custom handling for the last byte of a block, to avoid invoking streaming
576// API to get an effective block checksum. This function is its own inverse
577// because it uses xor.
578inline uint32_t ModifyChecksumForLastByte(uint32_t checksum, char last_byte) {
579 // This strategy bears some resemblance to extending a CRC checksum by one
580 // more byte, except we don't need to re-mix the input checksum as long as
581 // we do this step only once (per checksum).
582 const uint32_t kRandomPrime = 0x6b9083d9;
583 return checksum ^ lossless_cast<uint8_t>(last_byte) * kRandomPrime;
584}
585} // namespace

587uint32_t ComputeBuiltinChecksum(ChecksumType type, const char* data,
588 size_t data_size) {
589 switch (type) {
590 case kCRC32c:
591 return crc32c::Mask(crc32c::Value(data, data_size));
592 case kxxHash:
593 return XXH32(data, data_size, /*seed*/ 0);
594 case kxxHash64:
595 return Lower32of64(XXH64(data, data_size, /*seed*/ 0));
596 case kXXH3: {
597 if (data_size == 0) {
598 // Special case because of special handling for last byte, not
599 // present in this case. Can be any value different from other
600 // small input size checksums.
601 return 0;
602 } else {
603 // See corresponding code in ComputeBuiltinChecksumWithLastByte
604 uint32_t v = Lower32of64(XXH3_64bits(data, data_size - 1));
605 return ModifyChecksumForLastByte(v, data[data_size - 1]);
606 }
607 }
608 default: // including kNoChecksum
609 return 0;
610 }
611}

613uint32_t ComputeBuiltinChecksumWithLastByte(ChecksumType type, const char* data,
614 size_t data_size, char last_byte) {
615 switch (type) {
616 case kCRC32c: {
617 uint32_t crc = crc32c::Value(data, data_size);
618 // Extend to cover last byte (compression type)
619 crc = crc32c::Extend(crc, &last_byte, 1);
620 return crc32c::Mask(crc);
621 }
622 case kxxHash: {
623 XXH32_state_t* const state = XXH32_createState();
624 XXH32_reset(state, 0);
625 XXH32_update(state, data, data_size);
626 // Extend to cover last byte (compression type)
627 XXH32_update(state, &last_byte, 1);
628 uint32_t v = XXH32_digest(state);
629 XXH32_freeState(state);
630 return v;
631 }
632 case kxxHash64: {
633 XXH64_state_t* const state = XXH64_createState();
634 XXH64_reset(state, 0);
635 XXH64_update(state, data, data_size);
636 // Extend to cover last byte (compression type)
637 XXH64_update(state, &last_byte, 1);
638 uint32_t v = Lower32of64(XXH64_digest(state));
639 XXH64_freeState(state);
640 return v;
641 }
642 case kXXH3: {
643 // XXH3 is a complicated hash function that is extremely fast on
644 // contiguous input, but that makes its streaming support rather
645 // complex. It is worth custom handling of the last byte (`type`)
646 // in order to avoid allocating a large state object and bringing
647 // that code complexity into CPU working set.
648 uint32_t v = Lower32of64(XXH3_64bits(data, data_size));
649 return ModifyChecksumForLastByte(v, last_byte);
650 }
651 default: // including kNoChecksum
652 return 0;
653 }
654}

656Status UncompressBlockData(const UncompressionInfo& uncompression_info,
657 const char* data, size_t size,
658 BlockContents* out_contents, uint32_t format_version,
659 const ImmutableOptions& ioptions,
660 MemoryAllocator* allocator) {
661 Status ret = Status::OK();
662
663 assert(uncompression_info.type() != kNoCompression &&
664 "Invalid compression type");
665
666 StopWatchNano timer(ioptions.clock,
667 ShouldReportDetailedTime(ioptions.env, ioptions.stats));
668 size_t uncompressed_size = 0;
669 const char* error_msg = nullptr;
670 CacheAllocationPtr ubuf = UncompressData(
671 uncompression_info, data, size, &uncompressed_size,
672 GetCompressFormatForVersion(format_version), allocator, &error_msg);
673 if (!ubuf) {
674 if (!CompressionTypeSupported(uncompression_info.type())) {
675 ret = Status::NotSupported(
676 "Unsupported compression method for this build",
677 CompressionTypeToString(uncompression_info.type()));
678 } else {
679 std::ostringstream oss;
680 oss << "Corrupted compressed block contents";
681 if (error_msg) {
682 oss << ": " << error_msg;
683 }
684 ret = Status::Corruption(
685 oss.str(), CompressionTypeToString(uncompression_info.type()));
686 }
687 return ret;
688 }
689
690 *out_contents = BlockContents(std::move(ubuf), uncompressed_size);
691
692 if (ShouldReportDetailedTime(ioptions.env, ioptions.stats)) {
693 RecordTimeToHistogram(ioptions.stats, DECOMPRESSION_TIMES_NANOS,
694 timer.ElapsedNanos());
695 }
696 RecordTick(ioptions.stats, BYTES_DECOMPRESSED_FROM, size);
697 RecordTick(ioptions.stats, BYTES_DECOMPRESSED_TO, out_contents->data.size());
698 RecordTick(ioptions.stats, NUMBER_BLOCK_DECOMPRESSED);
699
700 TEST_SYNC_POINT_CALLBACK("UncompressBlockData:TamperWithReturnValue",
701 static_cast<void*>(&ret));
702 TEST_SYNC_POINT_CALLBACK(
703 "UncompressBlockData:"
704 "TamperWithDecompressionOutput",
705 static_cast<void*>(out_contents));
706
707 return ret;
708}

710Status UncompressSerializedBlock(const UncompressionInfo& uncompression_info,
711 const char* data, size_t size,
712 BlockContents* out_contents,
713 uint32_t format_version,
714 const ImmutableOptions& ioptions,
715 MemoryAllocator* allocator) {
716 assert(data[size] != kNoCompression);
717 assert(data[size] == static_cast<char>(uncompression_info.type()));
