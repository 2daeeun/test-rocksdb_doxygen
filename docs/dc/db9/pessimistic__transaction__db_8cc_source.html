74Status PessimisticTransactionDB::VerifyCFOptions(
75 const ColumnFamilyOptions& cf_options) {
76 const Comparator* const ucmp = cf_options.comparator;
77 assert(ucmp);
78 size_t ts_sz = ucmp->timestamp_size();
79 if (0 == ts_sz) {
80 return Status::OK();
81 }
82 if (ts_sz != sizeof(TxnTimestamp)) {
83 std::ostringstream oss;
84 oss << "Timestamp of transaction must have " << sizeof(TxnTimestamp)
85 << " bytes. CF comparator " << std::string(ucmp->Name())
86 << " timestamp size is " << ts_sz << " bytes";
87 return Status::InvalidArgument(oss.str());
88 }
89 if (txn_db_options_.write_policy != WRITE_COMMITTED) {
90 return Status::NotSupported("Only WriteCommittedTxn supports timestamp");
91 }
92 return Status::OK();
93}

95Status PessimisticTransactionDB::Initialize(
96 const std::vector<size_t>& compaction_enabled_cf_indices,
97 const std::vector<ColumnFamilyHandle*>& handles) {
98 for (auto cf_ptr : handles) {
99 AddColumnFamily(cf_ptr);
100 }
101 // Verify cf options
102 for (auto handle : handles) {
103 ColumnFamilyDescriptor cfd;
104 Status s = handle->GetDescriptor(&cfd);
105 if (!s.ok()) {
106 return s;
107 }
108 s = VerifyCFOptions(cfd.options);
109 if (!s.ok()) {
110 return s;
111 }
112 }
113
114 // Re-enable compaction for the column families that initially had
115 // compaction enabled.
116 std::vector<ColumnFamilyHandle*> compaction_enabled_cf_handles;
117 compaction_enabled_cf_handles.reserve(compaction_enabled_cf_indices.size());
118 for (auto index : compaction_enabled_cf_indices) {
119 compaction_enabled_cf_handles.push_back(handles[index]);
120 }
121
122 Status s = EnableAutoCompaction(compaction_enabled_cf_handles);
123
124 // create 'real' transactions from recovered shell transactions
125 auto dbimpl = static_cast_with_check<DBImpl>(GetRootDB());
126 assert(dbimpl != nullptr);
127 auto rtrxs = dbimpl->recovered_transactions();
128
129 for (auto it = rtrxs.begin(); it != rtrxs.end(); ++it) {
130 auto recovered_trx = it->second;
131 assert(recovered_trx);
132 assert(recovered_trx->batches_.size() == 1);
133 const auto& seq = recovered_trx->batches_.begin()->first;
134 const auto& batch_info = recovered_trx->batches_.begin()->second;
135 assert(batch_info.log_number_);
136 assert(recovered_trx->name_.length());
137
138 // TODO: plumb Env::IOActivity, Env::IOPriority
139 WriteOptions w_options;
140 w_options.sync = true;
141 TransactionOptions t_options;
142 // This would help avoiding deadlock for keys that although exist in the WAL
143 // did not go through concurrency control. This includes the merge that
144 // MyRocks uses for auto-inc columns. It is safe to do so, since (i) if
145 // there is a conflict between the keys of two transactions that must be
146 // avoided, it is already avoided by the application, MyRocks, before the
147 // restart (ii) application, MyRocks, guarntees to rollback/commit the
148 // recovered transactions before new transactions start.
149 t_options.skip_concurrency_control = true;
150
151 Transaction* real_trx = BeginTransaction(w_options, t_options, nullptr);
152 assert(real_trx);
153 real_trx->SetLogNumber(batch_info.log_number_);
154 assert(seq != kMaxSequenceNumber);
155 if (GetTxnDBOptions().write_policy != WRITE_COMMITTED) {
156 real_trx->SetId(seq);
157 }
158
159 s = real_trx->SetName(recovered_trx->name_);
160 if (!s.ok()) {
161 break;
162 }
163
164 s = real_trx->RebuildFromWriteBatch(batch_info.batch_);
165 // WriteCommitted set this to to disable this check that is specific to
166 // WritePrepared txns
167 assert(batch_info.batch_cnt_ == 0 ||
168 real_trx->GetWriteBatch()->SubBatchCnt() == batch_info.batch_cnt_);
169 real_trx->SetState(Transaction::PREPARED);
170 if (!s.ok()) {
171 break;
172 }
173 }
174 if (s.ok()) {
175 dbimpl->DeleteAllRecoveredTransactions();
176 }
177 return s;
178}

180Transaction* WriteCommittedTxnDB::BeginTransaction(
181 const WriteOptions& write_options, const TransactionOptions& txn_options,
182 Transaction* old_txn) {
183 if (old_txn != nullptr) {

190 } else {

192 }

227Status TransactionDB::Open(
228 const DBOptions& db_options, const TransactionDBOptions& txn_db_options,
229 const std::string& dbname,
230 const std::vector<ColumnFamilyDescriptor>& column_families,
231 std::vector<ColumnFamilyHandle*>* handles, TransactionDB** dbptr) {
232 Status s;
233 DB* db = nullptr;
234 if (txn_db_options.write_policy == WRITE_COMMITTED &&
235 db_options.unordered_write) {
236 return Status::NotSupported(
237 "WRITE_COMMITTED is incompatible with unordered_writes");
238 }
239 if (txn_db_options.write_policy == WRITE_UNPREPARED &&
240 db_options.unordered_write) {
241 // TODO(lth): support it
242 return Status::NotSupported(
243 "WRITE_UNPREPARED is currently incompatible with unordered_writes");
244 }
245 if (txn_db_options.write_policy == WRITE_PREPARED &&
246 db_options.unordered_write && !db_options.two_write_queues) {
247 return Status::NotSupported(
248 "WRITE_PREPARED is incompatible with unordered_writes if "
249 "two_write_queues is not enabled.");
250 }
251
252 std::vector<ColumnFamilyDescriptor> column_families_copy = column_families;
253 std::vector<size_t> compaction_enabled_cf_indices;
254 DBOptions db_options_2pc = db_options;
255 PrepareWrap(&db_options_2pc, &column_families_copy,
256 &compaction_enabled_cf_indices);
257 const bool use_seq_per_batch =
258 txn_db_options.write_policy == WRITE_PREPARED ||
259 txn_db_options.write_policy == WRITE_UNPREPARED;
260 const bool use_batch_per_txn =
261 txn_db_options.write_policy == WRITE_COMMITTED ||
262 txn_db_options.write_policy == WRITE_PREPARED;
263 s = DBImpl::Open(db_options_2pc, dbname, column_families_copy, handles, &db,
264 use_seq_per_batch, use_batch_per_txn,
265 /*is_retry=*/false, /*can_retry=*/nullptr);
266 if (s.ok()) {
267 ROCKS_LOG_WARN(db->GetDBOptions().info_log,
268 "Transaction write_policy is %" PRId32,
269 static_cast<int>(txn_db_options.write_policy));
270 // if WrapDB return non-ok, db will be deleted in WrapDB() via
271 // ~StackableDB().
272 s = WrapDB(db, txn_db_options, compaction_enabled_cf_indices, *handles,
273 dbptr);
274 }
275 return s;
276}

278void TransactionDB::PrepareWrap(
279 DBOptions* db_options, std::vector<ColumnFamilyDescriptor>* column_families,
280 std::vector<size_t>* compaction_enabled_cf_indices) {
281 compaction_enabled_cf_indices->clear();
282
283 // Enable MemTable History if not already enabled
284 for (size_t i = 0; i < column_families->size(); i++) {
285 ColumnFamilyOptions* cf_options = &(*column_families)[i].options;
286
287 if (cf_options->max_write_buffer_size_to_maintain == 0 &&
288 cf_options->max_write_buffer_number_to_maintain == 0) {
289 // Setting to -1 will set the History size to
290 // max_write_buffer_number * write_buffer_size.
291 cf_options->max_write_buffer_size_to_maintain = -1;
292 }
293 if (!cf_options->disable_auto_compactions) {
294 // Disable compactions momentarily to prevent race with DB::Open
295 cf_options->disable_auto_compactions = true;
296 compaction_enabled_cf_indices->push_back(i);
297 }
298 }
299 db_options->allow_2pc = true;
300}

302namespace {
303template <typename DBType>
304Status WrapAnotherDBInternal(
305 DBType* db, const TransactionDBOptions& txn_db_options,
306 const std::vector<size_t>& compaction_enabled_cf_indices,
307 const std::vector<ColumnFamilyHandle*>& handles, TransactionDB** dbptr) {
308 assert(db != nullptr);
309 assert(dbptr != nullptr);
310 *dbptr = nullptr;
311 std::unique_ptr<PessimisticTransactionDB> txn_db;
312 // txn_db owns object pointed to by the raw db pointer.
313 switch (txn_db_options.write_policy) {
314 case WRITE_UNPREPARED:
315 txn_db.reset(new WriteUnpreparedTxnDB(
316 db, PessimisticTransactionDB::ValidateTxnDBOptions(txn_db_options)));
317 break;
318 case WRITE_PREPARED:
319 txn_db.reset(new WritePreparedTxnDB(
320 db, PessimisticTransactionDB::ValidateTxnDBOptions(txn_db_options)));
321 break;
322 case WRITE_COMMITTED:
323 default:
324 txn_db.reset(new WriteCommittedTxnDB(
325 db, PessimisticTransactionDB::ValidateTxnDBOptions(txn_db_options)));
326 }
327 txn_db->UpdateCFComparatorMap(handles);
328 Status s = txn_db->Initialize(compaction_enabled_cf_indices, handles);
329 // In case of a failure at this point, db is deleted via the txn_db destructor
330 // and set to nullptr.
331 if (s.ok()) {
332 *dbptr = txn_db.release();
333 } else {
334 for (auto* h : handles) {
335 delete h;
336 }
337 // txn_db still owns db, and ~StackableDB() will be called when txn_db goes
338 // out of scope, deleting the input db pointer.
339 ROCKS_LOG_FATAL(db->GetDBOptions().info_log,
340 "Failed to initialize txn_db: %s", s.ToString().c_str());
341 }
342 return s;
343}

510// Used when wrapping DB write operations in a transaction
511Transaction* PessimisticTransactionDB::BeginInternalTransaction(
512 const WriteOptions& options) {
513 TransactionOptions txn_options;
514 Transaction* txn = BeginTransaction(options, txn_options, nullptr);
515
516 // Use default timeout for non-transactional writes
517 txn->SetLockTimeout(txn_db_options_.default_lock_timeout);
518 return txn;
519}

521// All user Put, PutEntity, Merge, Delete, and Write requests must be
522// intercepted to make sure that they lock all keys that they are writing to
523// avoid causing conflicts with any concurrent transactions. The easiest way to
524// do this is to wrap all write operations in a transaction.
525//
526// Put(), PutEntity(), Merge(), and Delete() only lock a single key per call.
527// Write() will sort its keys before locking them. This guarantees that
528// TransactionDB write methods cannot deadlock with each other (but still could
529// deadlock with a Transaction).
530Status PessimisticTransactionDB::Put(const WriteOptions& options,
531 ColumnFamilyHandle* column_family,
532 const Slice& key, const Slice& val) {
533 Status s = FailIfCfEnablesTs(this, column_family);
534 if (!s.ok()) {
535 return s;
536 }
537
538 Transaction* txn = BeginInternalTransaction(options);
539 txn->DisableIndexing();
540
541 // Since the client didn't create a transaction, they don't care about
542 // conflict checking for this write. So we just need to do PutUntracked().
543 s = txn->PutUntracked(column_family, key, val);
544
545 if (s.ok()) {
546 s = txn->Commit();
547 }
548
549 delete txn;
550
551 return s;
552}
553
554Status PessimisticTransactionDB::PutEntity(const WriteOptions& options,
555 ColumnFamilyHandle* column_family,
556 const Slice& key,
557 const WideColumns& columns) {
558 {
559 const Status s = FailIfCfEnablesTs(this, column_family);
560 if (!s.ok()) {
561 return s;
562 }
563 }
564
565 {
566 std::unique_ptr<Transaction> txn(BeginInternalTransaction(options));
567 txn->DisableIndexing();
568
569 // Since the client didn't create a transaction, they don't care about
570 // conflict checking for this write. So we just need to do
571 // PutEntityUntracked().
572 {
573 const Status s = txn->PutEntityUntracked(column_family, key, columns);
574 if (!s.ok()) {
575 return s;
576 }
577 }
578
579 {
580 const Status s = txn->Commit();
581 if (!s.ok()) {
582 return s;
583 }
584 }
585 }
586
587 return Status::OK();
588}

697void PessimisticTransactionDB::InsertExpirableTransaction(
698 TransactionID tx_id, PessimisticTransaction* tx) {
699 assert(tx->GetExpirationTime() > 0);

816Status SnapshotCreationCallback::operator()(SequenceNumber seq,
817 bool disable_memtable) {
818 assert(db_impl_);
819 assert(commit_ts_ != kMaxTxnTimestamp);
820
821 const bool two_write_queues =
822 db_impl_->immutable_db_options().two_write_queues;
823 assert(!two_write_queues || !disable_memtable);
824#ifdef NDEBUG
825 (void)two_write_queues;
826 (void)disable_memtable;
827#endif
828
829 const bool seq_per_batch = db_impl_->seq_per_batch();
830 if (!seq_per_batch) {
831 assert(db_impl_->GetLastPublishedSequence() <= seq);
832 } else {
833 assert(db_impl_->GetLastPublishedSequence() < seq);
834 }
835
836 // Create a snapshot which can also be used for write conflict checking.
837 auto ret = db_impl_->CreateTimestampedSnapshot(seq, commit_ts_);
838 snapshot_creation_status_ = ret.first;
839 snapshot_ = ret.second;
840 if (snapshot_creation_status_.ok()) {
841 assert(snapshot_);
842 } else {
843 assert(!snapshot_);
844 }
845 if (snapshot_ && snapshot_notifier_) {
846 snapshot_notifier_->SnapshotCreated(snapshot_.get());
847 }
848 return Status::OK();
849}
