20Status ChargedCache::Insert(const Slice& key, ObjectPtr obj,
21 const CacheItemHelper* helper, size_t charge,
22 Handle** handle, Priority priority,
23 const Slice& compressed_val, CompressionType type) {
24 Status s = target_->Insert(key, obj, helper, charge, handle, priority,
25 compressed_val, type);
26 if (s.ok()) {
27 // Insert may cause the cache entry eviction if the cache is full. So we
28 // directly call the reservation manager to update the total memory used
29 // in the cache.
30 assert(cache_res_mgr_);
31 cache_res_mgr_->UpdateCacheReservation(target_->GetUsage())
32 .PermitUncheckedError();
33 }
34 return s;
35}

37Cache::Handle* ChargedCache::Lookup(const Slice& key,
38 const CacheItemHelper* helper,
39 CreateContext* create_context,
40 Priority priority, Statistics* stats) {
41 auto handle = target_->Lookup(key, helper, create_context, priority, stats);
42 // Lookup may promote the KV pair from the secondary cache to the primary
43 // cache. So we directly call the reservation manager to update the total
44 // memory used in the cache.
45 if (helper && helper->create_cb) {
46 assert(cache_res_mgr_);
47 cache_res_mgr_->UpdateCacheReservation(target_->GetUsage())
48 .PermitUncheckedError();
49 }
50 return handle;
51}

53void ChargedCache::WaitAll(AsyncLookupHandle* async_handles, size_t count) {
54 target_->WaitAll(async_handles, count);
55 // In case of any promotions. Although some could finish by return of
56 // StartAsyncLookup, Wait/WaitAll will generally be used, so simpler to
57 // update here.
58 assert(cache_res_mgr_);
59 cache_res_mgr_->UpdateCacheReservation(target_->GetUsage())
60 .PermitUncheckedError();
61}

63bool ChargedCache::Release(Cache::Handle* handle, bool useful,
64 bool erase_if_last_ref) {
65 size_t memory_used_delta = target_->GetUsage(handle);
66 bool erased = target_->Release(handle, useful, erase_if_last_ref);
67 if (erased) {
68 assert(cache_res_mgr_);
69 cache_res_mgr_
70 ->UpdateCacheReservation(memory_used_delta, /* increase */ false)
71 .PermitUncheckedError();
72 }
73 return erased;
74}
