85 void ValidateLRUList(std::vector<std::string> keys,
86 size_t num_high_pri_pool_keys = 0,
87 size_t num_low_pri_pool_keys = 0,
88 size_t num_bottom_pri_pool_keys = 0) {
89 LRUHandle* lru;
90 LRUHandle* lru_low_pri;
91 LRUHandle* lru_bottom_pri;
92 cache_->TEST_GetLRUList(&lru, &lru_low_pri, &lru_bottom_pri);
93
94 LRUHandle* iter = lru;
95
96 bool in_low_pri_pool = false;
97 bool in_high_pri_pool = false;
98
99 size_t high_pri_pool_keys = 0;
100 size_t low_pri_pool_keys = 0;
101 size_t bottom_pri_pool_keys = 0;
102
103 if (iter == lru_bottom_pri) {
104 in_low_pri_pool = true;
105 in_high_pri_pool = false;
106 }
107 if (iter == lru_low_pri) {
108 in_low_pri_pool = false;
109 in_high_pri_pool = true;
110 }
111
112 for (const auto& key : keys) {
113 iter = iter->next;
114 ASSERT_NE(lru, iter);
115 ASSERT_EQ(key, iter->key().ToString());
116 ASSERT_EQ(in_high_pri_pool, iter->InHighPriPool());
117 ASSERT_EQ(in_low_pri_pool, iter->InLowPriPool());
118 if (in_high_pri_pool) {
119 ASSERT_FALSE(iter->InLowPriPool());
120 high_pri_pool_keys++;
121 } else if (in_low_pri_pool) {
122 ASSERT_FALSE(iter->InHighPriPool());
123 low_pri_pool_keys++;
124 } else {
125 bottom_pri_pool_keys++;
126 }
127 if (iter == lru_bottom_pri) {
128 ASSERT_FALSE(in_low_pri_pool);
129 ASSERT_FALSE(in_high_pri_pool);
130 in_low_pri_pool = true;
131 in_high_pri_pool = false;
132 }
133 if (iter == lru_low_pri) {
134 ASSERT_TRUE(in_low_pri_pool);
135 ASSERT_FALSE(in_high_pri_pool);
136 in_low_pri_pool = false;
137 in_high_pri_pool = true;
138 }
139 }
140 ASSERT_EQ(lru, iter->next);
141 ASSERT_FALSE(in_low_pri_pool);
142 ASSERT_TRUE(in_high_pri_pool);
143 ASSERT_EQ(num_high_pri_pool_keys, high_pri_pool_keys);
144 ASSERT_EQ(num_low_pri_pool_keys, low_pri_pool_keys);
145 ASSERT_EQ(num_bottom_pri_pool_keys, bottom_pri_pool_keys);
146 }

499TYPED_TEST(ClockCacheTest, Limits) {
500 constexpr size_t kCapacity = 64;
501 this->NewShard(kCapacity, false /*strict_capacity_limit*/);
502 auto& shard = *this->shard_;
503 using HandleImpl = typename ClockCacheTest<TypeParam>::Shard::HandleImpl;
504
505 for (bool strict_capacity_limit : {false, true, false}) {
506 SCOPED_TRACE("strict_capacity_limit = " +
507 std::to_string(strict_capacity_limit));
508
509 // Also tests switching between strict limit and not
510 shard.SetStrictCapacityLimit(strict_capacity_limit);
511
512 UniqueId64x2 hkey = this->TestHashedKey('x');
513
514 // Single entry charge beyond capacity
515 {
516 Status s = shard.Insert(this->TestKey(hkey), hkey, nullptr /*value*/,
517 &kNoopCacheItemHelper, kCapacity + 2 /*charge*/,
518 nullptr /*handle*/, Cache::Priority::LOW);
519 if (strict_capacity_limit) {
520 EXPECT_TRUE(s.IsMemoryLimit());
521 } else {
522 EXPECT_OK(s);
523 }
524 }
525
526 // Single entry fills capacity
527 {
528 HandleImpl* h;
529 ASSERT_OK(shard.Insert(this->TestKey(hkey), hkey, nullptr /*value*/,
530 &kNoopCacheItemHelper, kCapacity /*charge*/, &h,
531 Cache::Priority::LOW));
532 // Try to insert more
533 Status s = this->Insert('a');
534 if (strict_capacity_limit) {
535 EXPECT_TRUE(s.IsMemoryLimit());
536 } else {
537 EXPECT_OK(s);
538 }
539 // Release entry filling capacity.
540 // Cover useful = false case.
541 shard.Release(h, false /*useful*/, false /*erase_if_last_ref*/);
542 }
543
544 // Insert more than table size can handle to exceed occupancy limit.
545 // (Cleverly using mostly zero-charge entries, but some non-zero to
546 // verify usage tracking on detached entries.)
547 {
548 size_t n = kCapacity * 5 + 1;
549 std::unique_ptr<HandleImpl* []> ha { new HandleImpl* [n] {} };
550 Status s;
551 for (size_t i = 0; i < n && s.ok(); ++i) {
552 hkey[1] = i;
553 s = shard.Insert(this->TestKey(hkey), hkey, nullptr /*value*/,
554 &kNoopCacheItemHelper,
555 (i + kCapacity < n) ? 0 : 1 /*charge*/, &ha[i],
556 Cache::Priority::LOW);
557 if (i == 0) {
558 EXPECT_OK(s);
559 }
560 }
561 if (strict_capacity_limit) {
562 EXPECT_TRUE(s.IsMemoryLimit());
563 } else {
564 EXPECT_OK(s);
565 }
566 // Same result if not keeping a reference
567 s = this->Insert('a');
568 if (strict_capacity_limit) {
569 EXPECT_TRUE(s.IsMemoryLimit());
570 } else {
571 EXPECT_OK(s);
572 }
573
574 EXPECT_EQ(shard.GetOccupancyCount(), shard.GetOccupancyLimit());
