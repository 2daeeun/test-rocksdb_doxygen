353 // This function behaves with the implicit understanding that two
354 // rounds of keys are inserted into the database, as per the behavior
355 // of the DeletionStatsTest.
356 void SelectivelyDeleteKeys(uint64_t smallest, uint64_t largest,
357 uint64_t interval, int deletion_interval,
358 int key_size, uint64_t cutoff_key_num,
359 CompactionJobStats* stats, int cf = 0) {
360 // interval needs to be >= 2 so that deletion entries can be inserted
361 // that are intended to not result in an actual key deletion by using
362 // an offset of 1 from another existing key
363 ASSERT_GE(interval, 2);
364
365 uint64_t ctr = 1;
366 uint32_t deletions_made = 0;
367 uint32_t num_deleted = 0;
368 uint32_t num_expired = 0;
369 for (auto key = smallest; key <= largest; key += interval, ctr++) {
370 if (ctr % deletion_interval == 0) {
371 ASSERT_OK(Delete(cf, Key(key, key_size)));
372 deletions_made++;
373 num_deleted++;
374
375 if (key > cutoff_key_num) {
376 num_expired++;
377 }
378 }
379 }
380
381 // Insert some deletions for keys that don't exist that
382 // are both in and out of the key range
383 ASSERT_OK(Delete(cf, Key(smallest + 1, key_size)));
384 deletions_made++;
385
386 ASSERT_OK(Delete(cf, Key(smallest - 1, key_size)));
387 deletions_made++;
388 num_expired++;
389
390 ASSERT_OK(Delete(cf, Key(smallest - 9, key_size)));
391 deletions_made++;
392 num_expired++;
393
394 ASSERT_OK(Flush(cf));
395 SetDeletionCompactionStats(stats, deletions_made, num_expired, num_deleted);
396 }

399// An EventListener which helps verify the compaction results in
400// test CompactionJobStatsTest.
401class CompactionJobStatsChecker : public EventListener {
402 public:
403 CompactionJobStatsChecker()
404 : compression_enabled_(false), verify_next_comp_io_stats_(false) {}
405
406 size_t NumberOfUnverifiedStats() { return expected_stats_.size(); }
407
408 void set_verify_next_comp_io_stats(bool v) { verify_next_comp_io_stats_ = v; }
409
410 // Once a compaction completed, this function will verify the returned
411 // CompactionJobInfo with the oldest CompactionJobInfo added earlier
412 // in "expected_stats_" which has not yet being used for verification.
413 void OnCompactionCompleted(DB* /*db*/, const CompactionJobInfo& ci) override {
414 if (verify_next_comp_io_stats_) {
415 ASSERT_GT(ci.stats.file_write_nanos, 0);
416 ASSERT_GT(ci.stats.file_range_sync_nanos, 0);
417 ASSERT_GT(ci.stats.file_fsync_nanos, 0);
418 ASSERT_GT(ci.stats.file_prepare_write_nanos, 0);
419 verify_next_comp_io_stats_ = false;
420 }
421
422 std::lock_guard<std::mutex> lock(mutex_);
423 if (expected_stats_.size()) {
424 Verify(ci.stats, expected_stats_.front());
425 expected_stats_.pop();
426 }
427 }
428
429 // A helper function which verifies whether two CompactionJobStats
430 // match. The verification of all compaction stats are done by
431 // ASSERT_EQ except for the total input / output bytes, which we
432 // use ASSERT_GE and ASSERT_LE with a reasonable bias ---
433 // 10% in uncompressed case and 20% when compression is used.
434 virtual void Verify(const CompactionJobStats& current_stats,
435 const CompactionJobStats& stats) {
436 // time
437 ASSERT_GT(current_stats.elapsed_micros, 0U);
438
439 ASSERT_EQ(current_stats.num_input_records, stats.num_input_records);
440 ASSERT_EQ(current_stats.num_input_files, stats.num_input_files);
441 ASSERT_EQ(current_stats.num_input_files_at_output_level,
442 stats.num_input_files_at_output_level);
443
444 ASSERT_EQ(current_stats.num_output_records, stats.num_output_records);
445 ASSERT_EQ(current_stats.num_output_files, stats.num_output_files);
446
447 ASSERT_EQ(current_stats.is_full_compaction, stats.is_full_compaction);
448 ASSERT_EQ(current_stats.is_manual_compaction, stats.is_manual_compaction);
449
450 // file size
451 double kFileSizeBias = compression_enabled_ ? 0.20 : 0.10;
452 ASSERT_GE(current_stats.total_input_bytes * (1.00 + kFileSizeBias),
453 stats.total_input_bytes);
454 ASSERT_LE(current_stats.total_input_bytes,
455 stats.total_input_bytes * (1.00 + kFileSizeBias));
456 ASSERT_GE(current_stats.total_output_bytes * (1.00 + kFileSizeBias),
457 stats.total_output_bytes);
458 ASSERT_LE(current_stats.total_output_bytes,
459 stats.total_output_bytes * (1.00 + kFileSizeBias));
460 ASSERT_EQ(current_stats.total_input_raw_key_bytes,
461 stats.total_input_raw_key_bytes);
462 ASSERT_EQ(current_stats.total_input_raw_value_bytes,
463 stats.total_input_raw_value_bytes);
464
465 ASSERT_EQ(current_stats.num_records_replaced, stats.num_records_replaced);
466
467 ASSERT_EQ(current_stats.num_corrupt_keys, stats.num_corrupt_keys);
468
469 ASSERT_EQ(std::string(current_stats.smallest_output_key_prefix),
470 std::string(stats.smallest_output_key_prefix));
471 ASSERT_EQ(std::string(current_stats.largest_output_key_prefix),
472 std::string(stats.largest_output_key_prefix));
473 }
474
475 // Add an expected compaction stats, which will be used to
476 // verify the CompactionJobStats returned by the OnCompactionCompleted()
477 // callback.
478 void AddExpectedStats(const CompactionJobStats& stats) {
479 std::lock_guard<std::mutex> lock(mutex_);
480 expected_stats_.push(stats);
481 }
482
483 void EnableCompression(bool flag) { compression_enabled_ = flag; }
484
485 bool verify_next_comp_io_stats() const { return verify_next_comp_io_stats_; }
486
487 private:
488 std::mutex mutex_;
489 std::queue<CompactionJobStats> expected_stats_;
490 bool compression_enabled_;
491 bool verify_next_comp_io_stats_;
492};

513uint64_t EstimatedFileSize(uint64_t num_records, size_t key_size,
514 size_t value_size, double compression_ratio = 1.0,
515 size_t block_size = 4096,
516 int bloom_bits_per_key = 10) {
517 const size_t kPerKeyOverhead = 8;
518 const size_t kFooterSize = 512;
519
520 uint64_t data_size = static_cast<uint64_t>(
521 num_records *
522 (key_size + value_size * compression_ratio + kPerKeyOverhead));
523
524 return data_size + kFooterSize +
525 num_records * bloom_bits_per_key / 8 // filter block
526 + data_size * (key_size + 8) / block_size; // index block
527}

539CompactionJobStats NewManualCompactionJobStats(
540 const std::string& smallest_key, const std::string& largest_key,
541 size_t num_input_files, size_t num_input_files_at_output_level,
542 uint64_t num_input_records, size_t key_size, size_t value_size,
543 size_t num_output_files, uint64_t num_output_records,
544 double compression_ratio, uint64_t num_records_replaced,
545 bool is_full = false, bool is_manual = true) {
546 CompactionJobStats stats;
547 stats.Reset();
548
549 stats.num_input_records = num_input_records;
550 stats.num_input_files = num_input_files;
551 stats.num_input_files_at_output_level = num_input_files_at_output_level;
552
553 stats.num_output_records = num_output_records;
554 stats.num_output_files = num_output_files;
555
556 stats.total_input_bytes =
557 EstimatedFileSize(num_input_records / num_input_files, key_size,
558 value_size, compression_ratio) *
559 num_input_files;
560 stats.total_output_bytes =
561 EstimatedFileSize(num_output_records / num_output_files, key_size,
562 value_size, compression_ratio) *
563 num_output_files;
564 stats.total_input_raw_key_bytes = num_input_records * (key_size + 8);
565 stats.total_input_raw_value_bytes = num_input_records * value_size;
566
567 stats.is_full_compaction = is_full;
568 stats.is_manual_compaction = is_manual;
569
570 stats.num_records_replaced = num_records_replaced;
571
572 CopyPrefix(smallest_key, CompactionJobStats::kMaxPrefixLength,
573 &stats.smallest_output_key_prefix);
574 CopyPrefix(largest_key, CompactionJobStats::kMaxPrefixLength,
575 &stats.largest_output_key_prefix);
576
577 return stats;

598TEST_P(CompactionJobStatsTest, CompactionJobStatsTest) {
599 Random rnd(301);
600 const int kBufSize = 100;
601 char buf[kBufSize];
602 uint64_t key_base = 100000000l;
603 // Note: key_base must be multiple of num_keys_per_L0_file
604 int num_keys_per_L0_file = 100;
605 const int kTestScale = 8;
606 const int kKeySize = 10;
607 const int kValueSize = 1000;
608 const double kCompressionRatio = 0.5;
609 double compression_ratio = 1.0;
610 uint64_t key_interval = key_base / num_keys_per_L0_file;
611
612 // Whenever a compaction completes, this listener will try to
613 // verify whether the returned CompactionJobStats matches
614 // what we expect. The expected CompactionJobStats is added
615 // via AddExpectedStats().
616 auto* stats_checker = new CompactionJobStatsChecker();
617 Options options;
618 options.level_compaction_dynamic_level_bytes = false;
619 options.listeners.emplace_back(stats_checker);
620 options.create_if_missing = true;
621 // just enough setting to hold off auto-compaction.
622 options.level0_file_num_compaction_trigger = kTestScale + 1;
623 options.num_levels = 3;
624 options.compression = kNoCompression;
625 options.max_subcompactions = max_subcompactions_;
626 options.bytes_per_sync = 512 * 1024;
627
628 options.report_bg_io_stats = true;
629 for (int test = 0; test < 2; ++test) {
630 DestroyAndReopen(options);
631 CreateAndReopenWithCF({"pikachu"}, options);
632
633 // 1st Phase: generate "num_L0_files" L0 files.
634 int num_L0_files = 0;
635 for (uint64_t start_key = key_base; start_key <= key_base * kTestScale;
636 start_key += key_base) {
637 MakeTableWithKeyValues(&rnd, start_key, start_key + key_base - 1,
638 kKeySize, kValueSize, key_interval,
639 compression_ratio, 1);
640 snprintf(buf, kBufSize, "%d", ++num_L0_files);
