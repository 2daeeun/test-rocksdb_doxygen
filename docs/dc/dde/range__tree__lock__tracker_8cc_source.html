61void RangeLockList::Append(ColumnFamilyId cf_id, const DBT *left_key,
62 const DBT *right_key) {
63 MutexLock l(&mutex_);
64 // Only the transaction owner thread calls this function.
65 // The same thread does the lock release, so we can be certain nobody is
66 // releasing the locks concurrently.
67 assert(!releasing_locks_.load());
68 auto it = buffers_.find(cf_id);
69 if (it == buffers_.end()) {
70 // create a new one
71 it = buffers_.emplace(cf_id, std::make_shared<toku::range_buffer>()).first;
72 it->second->create();
73 }
74 it->second->append(left_key, right_key);
75}

77void RangeLockList::ReleaseLocks(RangeTreeLockManager *mgr,
78 PessimisticTransaction *txn,
79 bool all_trx_locks) {
80 {
81 MutexLock l(&mutex_);
82 // The lt->release_locks() call below will walk range_list->buffer_. We
83 // need to prevent lock escalation callback from replacing
84 // range_list->buffer_ while we are doing that.
85 //
86 // Additional complication here is internal mutex(es) in the locktree
87 // (let's call them latches):
88 // - Lock escalation first obtains latches on the lock tree
89 // - Then, it calls RangeTreeLockManager::on_escalate to replace
90 // transaction's range_list->buffer_. = Access to that buffer must be
91 // synchronized, so it will want to acquire the range_list->mutex_.
92 //
93 // While in this function we would want to do the reverse:
94 // - Acquire range_list->mutex_ to prevent access to the range_list.
95 // - Then, lt->release_locks() call will walk through the range_list
96 // - and acquire latches on parts of the lock tree to remove locks from
97 // it.
98 //
99 // How do we avoid the deadlock? The idea is that here we set
100 // releasing_locks_=true, and release the mutex.
101 // All other users of the range_list must:
102 // - Acquire the mutex, then check that releasing_locks_=false.
103 // (the code in this function doesnt do that as there's only one thread
104 // that releases transaction's locks)
105 releasing_locks_.store(true);
106 }
107
108 for (const auto &it : buffers_) {
109 // Don't try to call release_locks() if the buffer is empty! if we are
110 // not holding any locks, the lock tree might be in the STO-mode with
111 // another transaction, and our attempt to release an empty set of locks
112 // will cause an assertion failure.
113 if (it.second->get_num_ranges()) {
114 auto lt_ptr = mgr->GetLockTreeForCF(it.first);
115 toku::locktree *lt = lt_ptr.get();
116
117 lt->release_locks((TXNID)txn, it.second.get(), all_trx_locks);
118
119 it.second->destroy();
120 it.second->create();
121
122 toku::lock_request::retry_all_lock_requests(lt,
123 wait_callback_for_locktree);
124 }
125 }
126
127 Clear();
128 releasing_locks_.store(false);
129}

131void RangeLockList::ReplaceLocks(const toku::locktree *lt,
132 const toku::range_buffer &buffer) {
133 MutexLock l(&mutex_);
134 if (releasing_locks_.load()) {
135 // Do nothing. The transaction is releasing its locks, so it will not care
136 // about having a correct list of ranges. (In TokuDB,
137 // toku_db_txn_escalate_callback() makes use of this property, too)
138 return;
139 }
140
141 ColumnFamilyId cf_id = (ColumnFamilyId)lt->get_dict_id().dictid;
142
143 auto it = buffers_.find(cf_id);
144 it->second->destroy();
145 it->second->create();
146
147 toku::range_buffer::iterator iter(&buffer);
148 toku::range_buffer::iterator::record rec;
149 while (iter.current(&rec)) {
150 it->second->append(rec.get_left_key(), rec.get_right_key());
151 iter.next();
152 }
153}
