38 void AddKeys(int numkeys, int startkey) {
39 WriteOptions options;
40 options.sync = false;
41 for (int i = startkey; i < (numkeys + startkey); i++) {
42 std::string temp = std::to_string(i);
43 Slice key(temp);
44 Slice value(temp);
45 ASSERT_OK(db_->Put(options, key, value));
46 }
47 }

49 void createLevel0Files(int numFiles, int numKeysPerFile) {
50 int startKey = 0;
51 for (int i = 0; i < numFiles; i++) {
52 AddKeys(numKeysPerFile, startKey);
53 startKey += numKeysPerFile;
54 ASSERT_OK(dbfull()->TEST_FlushMemTable());
55 ASSERT_OK(
56 dbfull()->TEST_WaitForCompact()); // wait for background flush (flush
57 // is also a kind of compaction).
58 }
59 }

61 void CheckFileTypeCounts(const std::string& dir, int required_log,
62 int required_sst, int required_manifest) {
63 std::vector<std::string> filenames;
64 ASSERT_OK(env_->GetChildren(dir, &filenames));
65
66 int log_cnt = 0;
67 int sst_cnt = 0;
68 int manifest_cnt = 0;
69 for (const auto& file : filenames) {
70 uint64_t number;
71 FileType type;
72 if (ParseFileName(file, &number, &type)) {
73 log_cnt += (type == kWalFile);
74 sst_cnt += (type == kTableFile);
75 manifest_cnt += (type == kDescriptorFile);
76 }
77 }
78 ASSERT_EQ(required_log, log_cnt);
79 ASSERT_EQ(required_sst, sst_cnt);
80 ASSERT_EQ(required_manifest, manifest_cnt);
81 }

83 void ReopenDB() {
84 Options options = CurrentOptions();
85 // Trigger compaction when the number of level 0 files reaches 2.
86 options.create_if_missing = true;
87 options.level0_file_num_compaction_trigger = 2;
88 options.disable_auto_compactions = false;
89 options.delete_obsolete_files_period_micros = 0; // always do full purge
90 options.enable_thread_tracking = true;
91 options.write_buffer_size = 1024 * 1024 * 1000;
92 options.target_file_size_base = 1024 * 1024 * 1000;
93 options.max_bytes_for_level_base = 1024 * 1024 * 1000;
94 options.WAL_ttl_seconds = 300; // Used to test log files
95 options.WAL_size_limit_MB = 1024; // Used to test log files
96 options.wal_dir = wal_dir_;
97
98 // Note: the following prevents an otherwise harmless data race between the
99 // test setup code (AddBlobFile) in ObsoleteFilesTest.BlobFiles and the
100 // periodic stat dumping thread.
101 options.stats_dump_period_sec = 0;
102
103 Destroy(options);
104 Reopen(options);
105 }

107 const std::string wal_dir_;
108};

110TEST_F(ObsoleteFilesTest, RaceForObsoleteFileDeletion) {
111 ReopenDB();
112 SyncPoint::GetInstance()->DisableProcessing();
113 SyncPoint::GetInstance()->LoadDependency({
114 {"DBImpl::BackgroundCallCompaction:FoundObsoleteFiles",
115 "ObsoleteFilesTest::RaceForObsoleteFileDeletion:1"},
116 {"DBImpl::BackgroundCallCompaction:PurgedObsoleteFiles",
117 "ObsoleteFilesTest::RaceForObsoleteFileDeletion:2"},
118 });
119 SyncPoint::GetInstance()->SetCallBack(
120 "DBImpl::DeleteObsoleteFileImpl:AfterDeletion", [&](void* arg) {
121 Status* p_status = static_cast<Status*>(arg);
122 ASSERT_OK(*p_status);
123 });
124 SyncPoint::GetInstance()->SetCallBack(
125 "DBImpl::CloseHelper:PendingPurgeFinished", [&](void* arg) {
126 std::unordered_set<uint64_t>* files_grabbed_for_purge_ptr =
127 reinterpret_cast<std::unordered_set<uint64_t>*>(arg);
128 ASSERT_TRUE(files_grabbed_for_purge_ptr->empty());
129 });
130 SyncPoint::GetInstance()->EnableProcessing();
131
132 createLevel0Files(2, 50000);
133 CheckFileTypeCounts(wal_dir_, 1, 0, 0);
134
135 port::Thread user_thread([this]() {
136 JobContext jobCxt(0);
137 TEST_SYNC_POINT("ObsoleteFilesTest::RaceForObsoleteFileDeletion:1");
138 dbfull()->TEST_LockMutex();
139 dbfull()->FindObsoleteFiles(&jobCxt, true /* force=true */,
140 false /* no_full_scan=false */);
141 dbfull()->TEST_UnlockMutex();
142 TEST_SYNC_POINT("ObsoleteFilesTest::RaceForObsoleteFileDeletion:2");
143 dbfull()->PurgeObsoleteFiles(jobCxt);
144 jobCxt.Clean();
145 });
146
147 user_thread.join();
148}

150TEST_F(ObsoleteFilesTest, DeleteObsoleteOptionsFile) {
151 ReopenDB();
152
153 createLevel0Files(2, 50000);
154 CheckFileTypeCounts(wal_dir_, 1, 0, 0);
155
156 ASSERT_OK(dbfull()->DisableFileDeletions());
157 for (int i = 0; i != 4; ++i) {
158 if (i % 2) {
159 ASSERT_OK(dbfull()->SetOptions(dbfull()->DefaultColumnFamily(),
160 {{"paranoid_file_checks", "false"}}));
161 } else {
162 ASSERT_OK(dbfull()->SetOptions(dbfull()->DefaultColumnFamily(),
163 {{"paranoid_file_checks", "true"}}));
164 }
165 }
166 ASSERT_OK(dbfull()->EnableFileDeletions());
167
168 Close();
169
170 std::vector<std::string> files;
171 int opts_file_count = 0;
172 ASSERT_OK(env_->GetChildren(dbname_, &files));
173 for (const auto& file : files) {
174 uint64_t file_num;
175 Slice dummy_info_log_name_prefix;
176 FileType type;
177 WalFileType log_type;
178 if (ParseFileName(file, &file_num, dummy_info_log_name_prefix, &type,
179 &log_type) &&
180 type == kOptionsFile) {
181 opts_file_count++;
182 }
183 }
184 ASSERT_EQ(2, opts_file_count);
185}

187TEST_F(ObsoleteFilesTest, BlobFiles) {
188 ReopenDB();
189
190 VersionSet* const versions = dbfull()->GetVersionSet();
191 assert(versions);
192 assert(versions->GetColumnFamilySet());
193
194 ColumnFamilyData* const cfd = versions->GetColumnFamilySet()->GetDefault();
195 assert(cfd);
196
197 const ImmutableCFOptions* const ioptions = cfd->ioptions();
198 assert(ioptions);
199 assert(!ioptions->cf_paths.empty());
200
201 const std::string& path = ioptions->cf_paths.front().path;
202
203 // Add an obsolete blob file.
204 constexpr uint64_t first_blob_file_number = 234;
205 versions->AddObsoleteBlobFile(first_blob_file_number, path);
206
207 // Add a live blob file.
208 Version* const version = cfd->current();
209 assert(version);
210
211 VersionStorageInfo* const storage_info = version->storage_info();
212 assert(storage_info);
213
214 constexpr uint64_t second_blob_file_number = 456;
215 constexpr uint64_t second_total_blob_count = 100;
216 constexpr uint64_t second_total_blob_bytes = 2000000;
217 constexpr char second_checksum_method[] = "CRC32B";
218 constexpr char second_checksum_value[] = "\x6d\xbd\xf2\x3a";
219
220 auto shared_meta = SharedBlobFileMetaData::Create(
221 second_blob_file_number, second_total_blob_count, second_total_blob_bytes,
222 second_checksum_method, second_checksum_value);
223
224 constexpr uint64_t second_garbage_blob_count = 0;
225 constexpr uint64_t second_garbage_blob_bytes = 0;
226
227 auto meta = BlobFileMetaData::Create(
228 std::move(shared_meta), BlobFileMetaData::LinkedSsts(),
229 second_garbage_blob_count, second_garbage_blob_bytes);
230
231 storage_info->AddBlobFile(std::move(meta));
232
233 // Check for obsolete files and make sure the first blob file is picked up
234 // and grabbed for purge. The second blob file should be on the live list.
235 constexpr int job_id = 0;
236 JobContext job_context{job_id};
237
238 dbfull()->TEST_LockMutex();
239 constexpr bool force_full_scan = false;
240 dbfull()->FindObsoleteFiles(&job_context, force_full_scan);
241 dbfull()->TEST_UnlockMutex();
242
243 ASSERT_TRUE(job_context.HaveSomethingToDelete());
244 ASSERT_EQ(job_context.blob_delete_files.size(), 1);
245 ASSERT_EQ(job_context.blob_delete_files[0].GetBlobFileNumber(),
246 first_blob_file_number);
247
248 const auto& files_grabbed_for_purge =
249 dbfull()->TEST_GetFilesGrabbedForPurge();
250 ASSERT_NE(files_grabbed_for_purge.find(first_blob_file_number),
251 files_grabbed_for_purge.end());
252
253 ASSERT_EQ(job_context.blob_live.size(), 1);
254 ASSERT_EQ(job_context.blob_live[0], second_blob_file_number);
255
256 // Hack the job context a bit by adding a few files to the full scan
257 // list and adjusting the pending file number. We add the two files
258 // above as well as two additional ones, where one is old
259 // and should be cleaned up, and the other is still pending.
260 constexpr uint64_t old_blob_file_number = 123;
261 constexpr uint64_t pending_blob_file_number = 567;
262
263 job_context.full_scan_candidate_files.emplace_back(
264 BlobFileName(old_blob_file_number), path);
265 job_context.full_scan_candidate_files.emplace_back(
266 BlobFileName(first_blob_file_number), path);
267 job_context.full_scan_candidate_files.emplace_back(
268 BlobFileName(second_blob_file_number), path);
269 job_context.full_scan_candidate_files.emplace_back(
270 BlobFileName(pending_blob_file_number), path);
271
272 job_context.min_pending_output = pending_blob_file_number;
273
274 // Purge obsolete files and make sure we purge the old file and the first file
275 // (and keep the second file and the pending file).
276 std::vector<std::string> deleted_files;
277 SyncPoint::GetInstance()->SetCallBack(
278 "DBImpl::DeleteObsoleteFileImpl::BeforeDeletion", [&](void* arg) {
279 const std::string* file = static_cast<std::string*>(arg);
280 assert(file);
281
282 constexpr char blob_extension[] = ".blob";
283
284 if (file->find(blob_extension) != std::string::npos) {
285 deleted_files.emplace_back(*file);
286 }
287 });
288 SyncPoint::GetInstance()->EnableProcessing();
289
290 dbfull()->PurgeObsoleteFiles(job_context);
291 job_context.Clean();
292
293 SyncPoint::GetInstance()->DisableProcessing();
294 SyncPoint::GetInstance()->ClearAllCallBacks();
295
296 ASSERT_EQ(files_grabbed_for_purge.find(first_blob_file_number),
297 files_grabbed_for_purge.end());
298
299 std::sort(deleted_files.begin(), deleted_files.end());
300 const std::vector<std::string> expected_deleted_files{
301 BlobFileName(path, old_blob_file_number),
302 BlobFileName(path, first_blob_file_number)};
303
304 ASSERT_EQ(deleted_files, expected_deleted_files);
305}
