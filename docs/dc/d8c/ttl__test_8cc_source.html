110 // Populates and returns a kv-map
111 void MakeKVMap(int64_t num_entries) {
112 kvmap_.clear();
113 int digits = 1;
114 for (int64_t dummy = num_entries; dummy /= 10; ++digits) {
115 }
116 int digits_in_i = 1;
117 for (int64_t i = 0; i < num_entries; i++) {
118 std::string key = "key";
119 std::string value = "value";
120 if (i % 10 == 0) {
121 digits_in_i++;
122 }
123 for (int j = digits_in_i; j < digits; j++) {
124 key.append("0");
125 value.append("0");
126 }
127 AppendNumberTo(&key, i);
128 AppendNumberTo(&value, i);
129 kvmap_[key] = value;
130 }
131 ASSERT_EQ(static_cast<int64_t>(kvmap_.size()),
132 num_entries); // check all insertions done
133 }

135 // Makes a write-batch with key-vals from kvmap_ and 'Write''s it
136 void MakePutWriteBatch(const BatchOperation* batch_ops, int64_t num_ops) {
137 ASSERT_LE(num_ops, static_cast<int64_t>(kvmap_.size()));
138 static WriteOptions wopts;
139 static FlushOptions flush_opts;
140 WriteBatch batch;
141 kv_it_ = kvmap_.begin();
142 for (int64_t i = 0; i < num_ops && kv_it_ != kvmap_.end(); i++, ++kv_it_) {
143 switch (batch_ops[i]) {
144 case OP_PUT:
145 ASSERT_OK(batch.Put(kv_it_->first, kv_it_->second));
146 break;
147 case OP_DELETE:
148 ASSERT_OK(batch.Delete(kv_it_->first));
149 break;
150 default:
151 FAIL();
152 }
153 }
154 ASSERT_OK(db_ttl_->Write(wopts, &batch));
155 ASSERT_OK(db_ttl_->Flush(flush_opts));
156 }

158 // Puts num_entries starting from start_pos_map from kvmap_ into the database
159 void PutValues(int64_t start_pos_map, int64_t num_entries, bool flush = true,
160 ColumnFamilyHandle* cf = nullptr) {
161 ASSERT_TRUE(db_ttl_);
162 ASSERT_LE(start_pos_map + num_entries, static_cast<int64_t>(kvmap_.size()));
163 static WriteOptions wopts;
164 static FlushOptions flush_opts;
165 kv_it_ = kvmap_.begin();
166 advance(kv_it_, start_pos_map);
167 for (int64_t i = 0; kv_it_ != kvmap_.end() && i < num_entries;
168 i++, ++kv_it_) {
169 ASSERT_OK(cf == nullptr
170 ? db_ttl_->Put(wopts, kv_it_->first, kv_it_->second)
171 : db_ttl_->Put(wopts, cf, kv_it_->first, kv_it_->second));
172 }
173 // Put a mock kv at the end because CompactionFilter doesn't delete last key
174 ASSERT_OK(cf == nullptr ? db_ttl_->Put(wopts, "keymock", "valuemock")
175 : db_ttl_->Put(wopts, cf, "keymock", "valuemock"));
176 if (flush) {
177 if (cf == nullptr) {
178 ASSERT_OK(db_ttl_->Flush(flush_opts));
179 } else {
180 ASSERT_OK(db_ttl_->Flush(flush_opts, cf));
181 }
182 }
183 }

249 void CompactCheck(int64_t st_pos, int64_t span, bool check = true,
250 bool test_compaction_change = false,
251 ColumnFamilyHandle* cf = nullptr) {
252 static ReadOptions ropts;
253 kv_it_ = kvmap_.begin();
254 advance(kv_it_, st_pos);
255 std::string v;
256 for (int64_t i = 0; kv_it_ != kvmap_.end() && i < span; i++, ++kv_it_) {
257 Status s = (cf == nullptr) ? db_ttl_->Get(ropts, kv_it_->first, &v)
258 : db_ttl_->Get(ropts, cf, kv_it_->first, &v);
259 if (s.ok() != check) {
260 fprintf(stderr, "key=%s ", kv_it_->first.c_str());
261 if (!s.ok()) {
262 fprintf(stderr, "is absent from db but was expected to be present\n");
263 } else {
264 fprintf(stderr, "is present in db but was expected to be absent\n");
265 }
266 FAIL();
267 } else if (s.ok()) {
268 if (test_compaction_change && v.compare(kNewValue_) != 0) {
269 fprintf(stderr,
270 " value for key=%s present in database is %s but "
271 " should be %s\n",
272 kv_it_->first.c_str(), v.c_str(), kNewValue_.c_str());
273 FAIL();
274 } else if (!test_compaction_change && v.compare(kv_it_->second) != 0) {
275 fprintf(stderr,
276 " value for key=%s present in database is %s but "
277 " should be %s\n",
278 kv_it_->first.c_str(), v.c_str(), kv_it_->second.c_str());
279 FAIL();
280 }
281 }
282 }

284 // Sleeps for slp_tim then runs a manual compaction
285 // Checks span starting from st_pos from kvmap_ in the db and
286 // Gets should return true if check is true and false otherwise
287 // Also checks that value that we got is the same as inserted; and =kNewValue
288 // if test_compaction_change is true
289 void SleepCompactCheck(int slp_tim, int64_t st_pos, int64_t span,
290 bool check = true, bool test_compaction_change = false,
291 ColumnFamilyHandle* cf = nullptr) {
292 ASSERT_TRUE(db_ttl_);
293
294 env_->Sleep(slp_tim);
295 ASSERT_OK(ManualCompact(cf));
296 CompactCheck(st_pos, span, check, test_compaction_change, cf);
297 }

299 // Similar as SleepCompactCheck but uses TtlIterator to read from db
300 void SleepCompactCheckIter(int slp, int st_pos, int64_t span,
301 bool check = true) {
302 ASSERT_TRUE(db_ttl_);
303 env_->Sleep(slp);
304 ASSERT_OK(ManualCompact());
305 static ReadOptions ropts;
306 Iterator* dbiter = db_ttl_->NewIterator(ropts);
307 kv_it_ = kvmap_.begin();
308 advance(kv_it_, st_pos);
309
310 dbiter->Seek(kv_it_->first);
311 if (!check) {
312 if (dbiter->Valid()) {
313 ASSERT_NE(dbiter->value().compare(kv_it_->second), 0);
314 }
315 } else { // dbiter should have found out kvmap_[st_pos]
316 for (int64_t i = st_pos; kv_it_ != kvmap_.end() && i < st_pos + span;
317 i++, ++kv_it_) {
318 ASSERT_TRUE(dbiter->Valid());
319 ASSERT_EQ(dbiter->value().compare(kv_it_->second), 0);
320 dbiter->Next();
321 }
322 }
323 ASSERT_OK(dbiter->status());
324 delete dbiter;
325 }

333 class TestFilter : public CompactionFilter {
334 public:
335 TestFilter(const int64_t kSampleSize, const std::string& kNewValue)
336 : kSampleSize_(kSampleSize), kNewValue_(kNewValue) {}
337
338 // Works on keys of the form "key<number>"
339 // Drops key if number at the end of key is in [0, kSampleSize_/3),
340 // Keeps key if it is in [kSampleSize_/3, 2*kSampleSize_/3),
341 // Change value if it is in [2*kSampleSize_/3, kSampleSize_)
342 // Eg. kSampleSize_=6. Drop:key0-1...Keep:key2-3...Change:key4-5...
343 bool Filter(int /*level*/, const Slice& key, const Slice& /*value*/,
344 std::string* new_value, bool* value_changed) const override {
345 assert(new_value != nullptr);
346
347 std::string search_str = "0123456789";
348 std::string key_string = key.ToString();
349 size_t pos = key_string.find_first_of(search_str);
350 int num_key_end;
351 if (pos != std::string::npos) {
352 auto key_substr = key_string.substr(pos, key.size() - pos);
353#ifndef CYGWIN
354 num_key_end = std::stoi(key_substr);
355#else
356 num_key_end = std::strtol(key_substr.c_str(), 0, 10);
357#endif
358
359 } else {
360 return false; // Keep keys not matching the format "key<NUMBER>"
361 }
362
363 int64_t partition = kSampleSize_ / 3;
364 if (num_key_end < partition) {
365 return true;
366 } else if (num_key_end < partition * 2) {
367 return false;
368 } else {
369 *new_value = kNewValue_;
370 *value_changed = true;
371 return false;
372 }
373 }
374
375 const char* Name() const override { return "TestFilter"; }
376
377 private:
378 const int64_t kSampleSize_;
379 const std::string kNewValue_;
380 };
