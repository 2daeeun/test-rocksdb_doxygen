26Status VerifyBlockChecksum(const Footer& footer, const char* data,
27 size_t block_size, const std::string& file_name,
28 uint64_t offset) {
29 PERF_TIMER_GUARD(block_checksum_time);
30
31 assert(footer.GetBlockTrailerSize() == 5);
32 ChecksumType type = footer.checksum_type();
33
34 // After block_size bytes is compression type (1 byte), which is part of
35 // the checksummed section.
36 size_t len = block_size + 1;
37 // And then the stored checksum value (4 bytes).
38 uint32_t stored = DecodeFixed32(data + len);
39
40 uint32_t computed = ComputeBuiltinChecksum(type, data, len);
41
42 // Unapply context to 'stored' rather than apply to 'computed, for people
43 // who might look for reference crc value in error message
44 uint32_t modifier =
45 ChecksumModifierForContext(footer.base_context_checksum(), offset);
46 stored -= modifier;
47
48 if (stored == computed) {
49 return Status::OK();
50 } else {
51 // Unmask for people who might look for reference crc value
52 if (type == kCRC32c) {
53 stored = crc32c::Unmask(stored);
54 computed = crc32c::Unmask(computed);
55 }
56 return Status::Corruption(
57 "block checksum mismatch: stored" +
58 std::string(modifier ? "(context removed)" : "") + " = " +
59 std::to_string(stored) + ", computed = " + std::to_string(computed) +
60 ", type = " + std::to_string(type) + " in " + file_name + " offset " +
61 std::to_string(offset) + " size " + std::to_string(block_size));
62 }
63}
