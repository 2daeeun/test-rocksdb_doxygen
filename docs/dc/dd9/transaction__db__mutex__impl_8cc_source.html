68Status TransactionDBMutexImpl::TryLockFor(int64_t timeout_time) {
69 bool locked = true;
70
71 if (timeout_time == 0) {
72 locked = mutex_.try_lock();
73 } else {
74 // Previously, this code used a std::timed_mutex. However, this was changed
75 // due to known bugs in gcc versions < 4.9.
76 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=54562
77 //
78 // Since this mutex isn't held for long and only a single mutex is ever
79 // held at a time, it is reasonable to ignore the lock timeout_time here
80 // and only check it when waiting on the condition_variable.
81 mutex_.lock();
82 }
83
84 if (!locked) {
85 // timeout acquiring mutex
86 return Status::TimedOut(Status::SubCode::kMutexTimeout);
87 }
88
89 return Status::OK();
90}

92Status TransactionDBCondVarImpl::Wait(
93 std::shared_ptr<TransactionDBMutex> mutex) {
94 auto mutex_impl = static_cast<TransactionDBMutexImpl*>(mutex.get());
95
96 std::unique_lock<std::mutex> lock(mutex_impl->mutex_, std::adopt_lock);
97 cv_.wait(lock);
98
99 // Make sure unique_lock doesn't unlock mutex when it destructs
100 lock.release();
101
102 return Status::OK();
103}

105Status TransactionDBCondVarImpl::WaitFor(
106 std::shared_ptr<TransactionDBMutex> mutex, int64_t timeout_time) {
107 Status s;
108
109 auto mutex_impl = static_cast<TransactionDBMutexImpl*>(mutex.get());
110 std::unique_lock<std::mutex> lock(mutex_impl->mutex_, std::adopt_lock);
111
112 if (timeout_time < 0) {
113 // If timeout is negative, do not use a timeout
114 cv_.wait(lock);
115 } else {
116 auto duration = std::chrono::microseconds(timeout_time);
117 auto cv_status = cv_.wait_for(lock, duration);
118
119 // Check if the wait stopped due to timing out.
120 if (cv_status == std::cv_status::timeout) {
121 s = Status::TimedOut(Status::SubCode::kMutexTimeout);
122 }
123 }
124
125 // Make sure unique_lock doesn't unlock mutex when it destructs
126 lock.release();
127
128 // CV was signaled, or we spuriously woke up (but didn't time out)
129 return s;
130}
