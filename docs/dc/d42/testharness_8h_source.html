18// A "skipped" test has a specific meaning in Facebook infrastructure: the
19// test is in good shape and should be run, but something about the
20// compilation or execution environment means the test cannot be run.
21// Specifically, there is a hole in intended testing if any
22// parameterization of a test (e.g. Foo/FooTest.Bar/42) is skipped for all
23// tested build configurations/platforms/etc.
24//
25// If GTEST_SKIP is available, use it. Otherwise, define skip as success.
26//
27// The GTEST macros do not seem to print the message, even with -verbose,
28// so these print to stderr. Note that these do not exit the test themselves;
29// calling code should 'return' or similar from the test.
30#ifdef GTEST_SKIP_
31#define ROCKSDB_GTEST_SKIP(m) \
32 do { \
33 fputs("SKIPPED: " m "\n", stderr); \
34 GTEST_SKIP_(m); \
35 } while (false) /* user ; */
36#else
37#define ROCKSDB_GTEST_SKIP(m) \
38 do { \
39 fputs("SKIPPED: " m "\n", stderr); \
40 GTEST_SUCCESS_("SKIPPED: " m); \
41 } while (false) /* user ; */
42#endif
43
44// We add "bypass" as an alternative to ROCKSDB_GTEST_SKIP that is allowed to
45// be a permanent condition, e.g. for intentionally omitting or disabling some
46// parameterizations for some tests. (Use _DISABLED at the end of the test
47// name to disable an entire test.)
48#define ROCKSDB_GTEST_BYPASS(m) \
49 do { \
50 fputs("BYPASSED: " m "\n", stderr); \
51 GTEST_SUCCESS_("BYPASSED: " m); \
52 } while (false) /* user ; */
