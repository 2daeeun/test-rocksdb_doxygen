90struct Reader {
91 public:
92 explicit Reader(std::vector<ShardState>* shard_states,
93 ROCKSDB_NAMESPACE::DB* db)
94 : shard_states_(shard_states), db_(db) {
95 sem_init(&sem_, 0, 0);
96 thread_ = port::Thread(&Reader::run, this);
97 }
98
99 void run() {
100 while (1) {
101 sem_wait(&sem_);
102 if (done_.load()) {
103 break;
104 }
105
106 uint64_t shard;
107 {
108 std::lock_guard<std::mutex> guard(queue_mutex_);
109 assert(!shards_pending_queue_.empty());
110 shard = shards_pending_queue_.front();
111 shards_pending_queue_.pop();
112 shards_pending_set_.reset(shard);
113 }
114 readOnceFromShard(shard);
115 }
116 }
117
118 void readOnceFromShard(uint64_t shard) {
119 ShardState& state = (*shard_states_)[shard];
120 if (!state.it) {
121 // Initialize iterators
122 ROCKSDB_NAMESPACE::ReadOptions options;
123 options.tailing = true;
124 if (FLAGS_iterate_upper_bound) {
125 state.upper_bound = Key(shard, std::numeric_limits<uint64_t>::max());
126 state.upper_bound_slice = ROCKSDB_NAMESPACE::Slice(
127 (const char*)&state.upper_bound, sizeof(state.upper_bound));
128 options.iterate_upper_bound = &state.upper_bound_slice;
129 }
130
131 state.it.reset(db_->NewIterator(options));
132
133 if (FLAGS_cache_only_first) {
134 options.read_tier = ROCKSDB_NAMESPACE::ReadTier::kBlockCacheTier;
135 state.it_cacheonly.reset(db_->NewIterator(options));
136 }
137 }
138
139 const uint64_t upto = state.last_written.load();
140 for (ROCKSDB_NAMESPACE::Iterator* it :
141 {state.it_cacheonly.get(), state.it.get()}) {
142 if (it == nullptr) {
143 continue;
144 }
145 if (state.last_read.load() >= upto) {
146 break;
147 }
148 bool need_seek = true;
149 for (uint64_t seq = state.last_read.load() + 1; seq <= upto; ++seq) {
150 if (need_seek) {
151 Key from(shard, state.last_read.load() + 1);
152 it->Seek(ROCKSDB_NAMESPACE::Slice((const char*)&from, sizeof(from)));
153 need_seek = false;
154 } else {
155 it->Next();
156 }
157 if (it->status().IsIncomplete()) {
158 ++::stats.cache_misses;
159 break;
160 }
161 assert(it->Valid());
162 assert(it->key().size() == sizeof(Key));
163 Key key;
164 memcpy(&key, it->key().data(), it->key().size());
165 // fprintf(stderr, "Expecting (%ld, %ld) read (%ld, %ld)\n",
166 // shard, seq, key.shard(), key.seqno());
167 assert(key.shard() == shard);
168 assert(key.seqno() == seq);
169 state.last_read.store(seq);
170 ++::stats.read;
171 }
172 }
173 }

175 void onWrite(uint64_t shard) {
176 {
177 std::lock_guard<std::mutex> guard(queue_mutex_);
178 if (!shards_pending_set_.test(shard)) {
179 shards_pending_queue_.push(shard);
180 shards_pending_set_.set(shard);
181 sem_post(&sem_);
182 }
183 }
184 }

205struct Writer {
206 explicit Writer(std::vector<ShardState>* shard_states,
207 ROCKSDB_NAMESPACE::DB* db)
208 : shard_states_(shard_states), db_(db) {}
209
210 void start() { thread_ = port::Thread(&Writer::run, this); }
211
212 void run() {
213 std::queue<std::chrono::steady_clock::time_point> workq;
214 std::chrono::steady_clock::time_point deadline(
215 std::chrono::steady_clock::now() +
216 std::chrono::nanoseconds((uint64_t)(1000000000 * FLAGS_runtime)));
217 std::vector<uint64_t> my_shards;
218 for (int i = 1; i <= FLAGS_shards; ++i) {
219 if ((*shard_states_)[i].writer == this) {
220 my_shards.push_back(i);
221 }
222 }
223
224 std::mt19937 rng{std::random_device()()};
225 std::uniform_int_distribution<int> shard_dist(
226 0, static_cast<int>(my_shards.size()) - 1);
227 std::string value(FLAGS_value_size, '*');
228
229 while (1) {
230 auto now = std::chrono::steady_clock::now();
231 if (FLAGS_runtime >= 0 && now >= deadline) {
232 break;
233 }
234 if (workq.empty()) {
235 for (int i = 0; i < FLAGS_rate; i += FLAGS_writers) {
236 std::chrono::nanoseconds offset(1000000000LL * i / FLAGS_rate);
237 workq.push(now + offset);
238 }
239 }
240 while (!workq.empty() && workq.front() < now) {
241 workq.pop();
242 uint64_t shard = my_shards[shard_dist(rng)];
243 ShardState& state = (*shard_states_)[shard];
244 uint64_t seqno = state.last_written.load() + 1;
245 Key key(shard, seqno);
246 // fprintf(stderr, "Writing (%ld, %ld)\n", shard, seqno);
247 ROCKSDB_NAMESPACE::Status status =
248 db_->Put(ROCKSDB_NAMESPACE::WriteOptions(),
249 ROCKSDB_NAMESPACE::Slice((const char*)&key, sizeof(key)),
250 ROCKSDB_NAMESPACE::Slice(value));
251 assert(status.ok());
252 state.last_written.store(seqno);
253 state.reader->onWrite(shard);
254 ++::stats.written;
255 }
256 std::this_thread::sleep_for(std::chrono::milliseconds(1));
257 }
258 // fprintf(stderr, "Writer done\n");
259 }
260
261 ~Writer() { thread_.join(); }
262
263 private:
264 char pad1[128] __attribute__((__unused__));
265 std::vector<ShardState>* shard_states_;
266 ROCKSDB_NAMESPACE::DB* db_;
267 ROCKSDB_NAMESPACE::port::Thread thread_;
268 char pad2[128] __attribute__((__unused__));
269};

271struct StatsThread {
272 explicit StatsThread(ROCKSDB_NAMESPACE::DB* db)
273 : db_(db), thread_(&StatsThread::run, this) {}
274
275 void run() {
276 auto tstart = std::chrono::steady_clock::now(), tlast = tstart;
277 uint64_t wlast = 0, rlast = 0;
278 while (!done_.load()) {
279 {
280 std::unique_lock<std::mutex> lock(cvm_);
281 cv_.wait_for(lock, std::chrono::seconds(1));
282 }
283 auto now = std::chrono::steady_clock::now();
284 double elapsed =
285 std::chrono::duration_cast<std::chrono::duration<double> >(now -
286 tlast)
287 .count();
288 uint64_t w = ::stats.written.load();
289 uint64_t r = ::stats.read.load();
290 fprintf(stderr,
291 "%s elapsed %4lds | written %10ld | w/s %10.0f | read %10ld | "
292 "r/s %10.0f | cache misses %10ld\n",
293 db_->GetEnv()->TimeToString(time(nullptr)).c_str(),
294 std::chrono::duration_cast<std::chrono::seconds>(now - tstart)
295 .count(),
296 w, (w - wlast) / elapsed, r, (r - rlast) / elapsed,
297 ::stats.cache_misses.load());
298 wlast = w;
299 rlast = r;
300 tlast = now;
301 }
302 }
303
304 ~StatsThread() {
305 {
306 std::lock_guard<std::mutex> guard(cvm_);
307 done_.store(true);
308 }
309 cv_.notify_all();
310 thread_.join();
311 }
312
313 private:
314 ROCKSDB_NAMESPACE::DB* db_;
315 std::mutex cvm_;
316 std::condition_variable cv_;
317 ROCKSDB_NAMESPACE::port::Thread thread_;
318 std::atomic<bool> done_{false};
319};

321int main(int argc, char** argv) {
322 GFLAGS_NAMESPACE::ParseCommandLineFlags(&argc, &argv, true);
323
324 std::mt19937 rng{std::random_device()()};
325 ROCKSDB_NAMESPACE::Status status;
326 std::string path =
327 ROCKSDB_NAMESPACE::test::PerThreadDBPath("forward_iterator_test");
328 fprintf(stderr, "db path is %s\n", path.c_str());
329 ROCKSDB_NAMESPACE::Options options;
330 options.create_if_missing = true;
331 options.compression = ROCKSDB_NAMESPACE::CompressionType::kNoCompression;
332 options.compaction_style =
333 ROCKSDB_NAMESPACE::CompactionStyle::kCompactionStyleNone;
334 options.level0_slowdown_writes_trigger = 99999;
335 options.level0_stop_writes_trigger = 99999;
336 options.use_direct_io_for_flush_and_compaction = true;
337 options.write_buffer_size = FLAGS_memtable_size;
338 ROCKSDB_NAMESPACE::BlockBasedTableOptions table_options;
339 table_options.block_cache =
340 ROCKSDB_NAMESPACE::NewLRUCache(FLAGS_block_cache_size);
341 table_options.block_size = FLAGS_block_size;
342 options.table_factory.reset(
343 ROCKSDB_NAMESPACE::NewBlockBasedTableFactory(table_options));
344
345 status = ROCKSDB_NAMESPACE::DestroyDB(path, options);
346 assert(status.ok());
347 ROCKSDB_NAMESPACE::DB* db_raw;
348 status = ROCKSDB_NAMESPACE::DB::Open(options, path, &db_raw);
349 assert(status.ok());
350 std::unique_ptr<ROCKSDB_NAMESPACE::DB> db(db_raw);
351
352 std::vector<ShardState> shard_states(FLAGS_shards + 1);
353 std::deque<Reader> readers;
354 while (static_cast<int>(readers.size()) < FLAGS_readers) {
355 readers.emplace_back(&shard_states, db_raw);
356 }
357 std::deque<Writer> writers;
358 while (static_cast<int>(writers.size()) < FLAGS_writers) {
359 writers.emplace_back(&shard_states, db_raw);
360 }
361
362 // Each shard gets a random reader and random writer assigned to it
363 for (int i = 1; i <= FLAGS_shards; ++i) {
364 std::uniform_int_distribution<int> reader_dist(0, FLAGS_readers - 1);
365 std::uniform_int_distribution<int> writer_dist(0, FLAGS_writers - 1);
366 shard_states[i].reader = &readers[reader_dist(rng)];
367 shard_states[i].writer = &writers[writer_dist(rng)];
368 }
369
370 StatsThread stats_thread(db_raw);
371 for (Writer& w : writers) {
372 w.start();
373 }
374
375 writers.clear();
376 readers.clear();
377}
