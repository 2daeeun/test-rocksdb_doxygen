18// DbStressCompactionFilter is safe to use with db_stress as it does not perform
19// any mutation. It only makes `kRemove` decisions for keys that are already
20// non-existent according to the `SharedState`.
21class DbStressCompactionFilter : public CompactionFilter {
22 public:
23 DbStressCompactionFilter(SharedState* state, int cf_id)
24 : state_(state), cf_id_(cf_id) {}
25
26 Decision FilterV2(int /*level*/, const Slice& key, ValueType /*value_type*/,
27 const Slice& /*existing_value*/, std::string* /*new_value*/,
28 std::string* /*skip_until*/) const override {
29 if (state_ == nullptr) {
30 return Decision::kKeep;
31 }
32 if (key.empty() || ('0' <= key[0] && key[0] <= '9')) {
33 // It is likely leftover from a test_batches_snapshots run. Below this
34 // conditional, the test_batches_snapshots key format is not handled
35 // properly. Just keep it to be safe.
36 return Decision::kKeep;
37 }
38 uint64_t key_num = 0;
39 {
40 Slice ukey_without_ts = key;
41 assert(ukey_without_ts.size() >= FLAGS_user_timestamp_size);
42 ukey_without_ts.remove_suffix(FLAGS_user_timestamp_size);
43 [[maybe_unused]] bool ok =
44 GetIntVal(ukey_without_ts.ToString(), &key_num);
45 assert(ok);
46 }
47 port::Mutex* key_mutex = state_->GetMutexForKey(cf_id_, key_num);
48 if (!key_mutex->TryLock()) {
49 return Decision::kKeep;
50 }
51 // Reaching here means we acquired the lock.
52 key_mutex->AssertHeld();
53 bool key_exists = state_->Exists(cf_id_, key_num);
54 const bool allow_overwrite = state_->AllowsOverwrite(key_num);
55
56 key_mutex->Unlock();
57
58 if (!key_exists) {
59 return allow_overwrite ? Decision::kRemove : Decision::kPurge;
60 }
61 return Decision::kKeep;
62 }
63
64 const char* Name() const override { return "DbStressCompactionFilter"; }
65
66 private:
67 SharedState* const state_;
68 const int cf_id_;
69};
