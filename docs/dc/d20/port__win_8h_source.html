80class Mutex {
81 public:
82 static const char* kName() { return "std::mutex"; }
83
84 explicit Mutex(bool IGNORED_adaptive = kDefaultToAdaptiveMutex)
85#ifndef NDEBUG
86 : locked_(false)
87#endif
88 {
89 (void)IGNORED_adaptive;
90 }
91
92 ~Mutex();
93
94 void Lock() {
95 mutex_.lock();
96#ifndef NDEBUG
97 locked_ = true;
98#endif
99 }
100
101 void Unlock() {
102#ifndef NDEBUG
103 locked_ = false;
104#endif
105 mutex_.unlock();
106 }
107
108 bool TryLock() {
109 bool ret = mutex_.try_lock();
110#ifndef NDEBUG
111 if (ret) {
112 locked_ = true;
113 }
114#endif
115 return ret;
116 }
117
118 // this will assert if the mutex is not locked
119 // it does NOT verify that mutex is held by a calling thread
120 void AssertHeld() const {
121#ifndef NDEBUG
122 assert(locked_);
123#endif
124 }
125
126 // Also implement std Lockable
127 inline void lock() { Lock(); }
128 inline void unlock() { Unlock(); }
129 inline bool try_lock() { return TryLock(); }
130
131 // Mutex is move only with lock ownership transfer
132 Mutex(const Mutex&) = delete;
133 void operator=(const Mutex&) = delete;
134
135 private:
136 friend class CondVar;
137
138 std::mutex& getLock() { return mutex_; }
139
140 std::mutex mutex_;
141#ifndef NDEBUG
142 bool locked_;
143#endif
144};

261inline int pthread_key_create(pthread_key_t* key, void (*destructor)(void*)) {
262 // Not used
263 (void)destructor;
264
265 pthread_key_t k = TlsAlloc();
266 if (TLS_OUT_OF_INDEXES == k) {
267 return ENOMEM;
268 }
269
270 *key = k;
271 return 0;
272}
273
274inline int pthread_key_delete(pthread_key_t key) {
275 if (!TlsFree(key)) {
276 return EINVAL;
277 }
278 return 0;
279}
280
281inline int pthread_setspecific(pthread_key_t key, const void* value) {
282 if (!TlsSetValue(key, const_cast<void*>(value))) {
283 return ENOMEM;
284 }
285 return 0;
286}
287
288inline void* pthread_getspecific(pthread_key_t key) {
289 void* result = TlsGetValue(key);
290 if (!result) {
291 if (GetLastError() != ERROR_SUCCESS) {
292 errno = EINVAL;
293 } else {
294 errno = NOERROR;
295 }
296 }
297 return result;
298}
