25uint64_t PackFileNumberAndPathId(uint64_t number, uint64_t path_id) {
26 assert(number <= kFileNumberMask);
27 return number | (path_id * (kFileNumberMask + 1));

30Status FileMetaData::UpdateBoundaries(const Slice& key, const Slice& value,
31 SequenceNumber seqno,
32 ValueType value_type) {
33 if (value_type == kTypeBlobIndex) {
34 BlobIndex blob_index;
35 const Status s = blob_index.DecodeFrom(value);
36 if (!s.ok()) {
37 return s;
38 }
39
40 if (!blob_index.IsInlined() && !blob_index.HasTTL()) {
41 if (blob_index.file_number() == kInvalidBlobFileNumber) {
42 return Status::Corruption("Invalid blob file number");
43 }
44
45 if (oldest_blob_file_number == kInvalidBlobFileNumber ||
46 oldest_blob_file_number > blob_index.file_number()) {
47 oldest_blob_file_number = blob_index.file_number();
48 }
49 }
50 }
51
52 if (smallest.size() == 0) {
53 smallest.DecodeFrom(key);
54 }
55 largest.DecodeFrom(key);
56 fd.smallest_seqno = std::min(fd.smallest_seqno, seqno);
57 fd.largest_seqno = std::max(fd.largest_seqno, seqno);
58
59 return Status::OK();
60}

96bool VersionEdit::EncodeTo(std::string* dst,
97 std::optional<size_t> ts_sz) const {
98 if (has_db_id_) {
99 PutVarint32(dst, kDbId);
100 PutLengthPrefixedSlice(dst, db_id_);
101 }
102 if (has_comparator_) {
103 assert(has_persist_user_defined_timestamps_);
104 PutVarint32(dst, kComparator);
105 PutLengthPrefixedSlice(dst, comparator_);
106 }
107 if (has_log_number_) {
108 PutVarint32Varint64(dst, kLogNumber, log_number_);
109 }
110 if (has_prev_log_number_) {
111 PutVarint32Varint64(dst, kPrevLogNumber, prev_log_number_);
112 }
113 if (has_next_file_number_) {
114 PutVarint32Varint64(dst, kNextFileNumber, next_file_number_);
115 }
116 if (has_max_column_family_) {
117 PutVarint32Varint32(dst, kMaxColumnFamily, max_column_family_);
118 }
119 if (has_min_log_number_to_keep_) {
120 PutVarint32Varint64(dst, kMinLogNumberToKeep, min_log_number_to_keep_);
121 }
122 if (has_last_sequence_) {
123 PutVarint32Varint64(dst, kLastSequence, last_sequence_);
124 }
125 for (size_t i = 0; i < compact_cursors_.size(); i++) {
126 if (compact_cursors_[i].second.Valid()) {
127 PutVarint32(dst, kCompactCursor);
128 PutVarint32(dst, compact_cursors_[i].first); // level
129 PutLengthPrefixedSlice(dst, compact_cursors_[i].second.Encode());
130 }
131 }
132 for (const auto& deleted : deleted_files_) {
133 PutVarint32Varint32Varint64(dst, kDeletedFile, deleted.first /* level */,
134 deleted.second /* file number */);
135 }
136
137 bool min_log_num_written = false;
138
139 assert(new_files_.empty() || ts_sz.has_value());
140 for (size_t i = 0; i < new_files_.size(); i++) {
141 const FileMetaData& f = new_files_[i].second;
142 if (!f.smallest.Valid() || !f.largest.Valid() ||
143 f.epoch_number == kUnknownEpochNumber) {
144 return false;
145 }
146 PutVarint32(dst, kNewFile4);
147 PutVarint32Varint64(dst, new_files_[i].first /* level */, f.fd.GetNumber());
148 PutVarint64(dst, f.fd.GetFileSize());
149 EncodeFileBoundaries(dst, f, ts_sz.value());
150 PutVarint64Varint64(dst, f.fd.smallest_seqno, f.fd.largest_seqno);
151 // Customized fields' format:
152 // +-----------------------------+
153 // | 1st field's tag (varint32) |
154 // +-----------------------------+
155 // | 1st field's size (varint32) |
156 // +-----------------------------+
157 // | bytes for 1st field |
158 // | (based on size decoded) |
159 // +-----------------------------+
160 // | |
161 // | ...... |
162 // | |
163 // +-----------------------------+
164 // | last field's size (varint32)|
165 // +-----------------------------+
166 // | bytes for last field |
167 // | (based on size decoded) |
168 // +-----------------------------+
169 // | terminating tag (varint32) |
170 // +-----------------------------+
171 //
172 // Customized encoding for fields:
173 // tag kPathId: 1 byte as path_id
174 // tag kNeedCompaction:
175 // now only can take one char value 1 indicating need-compaction
176 //
177 PutVarint32(dst, NewFileCustomTag::kOldestAncesterTime);
178 std::string varint_oldest_ancester_time;
179 PutVarint64(&varint_oldest_ancester_time, f.oldest_ancester_time);
180 TEST_SYNC_POINT_CALLBACK("VersionEdit::EncodeTo:VarintOldestAncesterTime",
181 &varint_oldest_ancester_time);
182 PutLengthPrefixedSlice(dst, Slice(varint_oldest_ancester_time));
183
184 PutVarint32(dst, NewFileCustomTag::kFileCreationTime);
185 std::string varint_file_creation_time;
186 PutVarint64(&varint_file_creation_time, f.file_creation_time);
187 TEST_SYNC_POINT_CALLBACK("VersionEdit::EncodeTo:VarintFileCreationTime",
188 &varint_file_creation_time);
189 PutLengthPrefixedSlice(dst, Slice(varint_file_creation_time));
190
191 PutVarint32(dst, NewFileCustomTag::kEpochNumber);
192 std::string varint_epoch_number;
193 PutVarint64(&varint_epoch_number, f.epoch_number);
194 PutLengthPrefixedSlice(dst, Slice(varint_epoch_number));
195
196 PutVarint32(dst, NewFileCustomTag::kFileChecksum);
197 PutLengthPrefixedSlice(dst, Slice(f.file_checksum));
198
199 PutVarint32(dst, NewFileCustomTag::kFileChecksumFuncName);
200 PutLengthPrefixedSlice(dst, Slice(f.file_checksum_func_name));
201
202 if (f.fd.GetPathId() != 0) {
203 PutVarint32(dst, NewFileCustomTag::kPathId);
204 char p = static_cast<char>(f.fd.GetPathId());
205 PutLengthPrefixedSlice(dst, Slice(&p, 1));
206 }
207 if (f.temperature != Temperature::kUnknown) {
208 PutVarint32(dst, NewFileCustomTag::kTemperature);
209 char p = static_cast<char>(f.temperature);
210 PutLengthPrefixedSlice(dst, Slice(&p, 1));
211 }
212 if (f.marked_for_compaction) {
213 PutVarint32(dst, NewFileCustomTag::kNeedCompaction);
214 char p = static_cast<char>(1);
215 PutLengthPrefixedSlice(dst, Slice(&p, 1));
216 }
217 if (has_min_log_number_to_keep_ && !min_log_num_written) {
218 PutVarint32(dst, NewFileCustomTag::kMinLogNumberToKeepHack);
219 std::string varint_log_number;
220 PutFixed64(&varint_log_number, min_log_number_to_keep_);
221 PutLengthPrefixedSlice(dst, Slice(varint_log_number));
222 min_log_num_written = true;
223 }
224 if (f.oldest_blob_file_number != kInvalidBlobFileNumber) {
225 PutVarint32(dst, NewFileCustomTag::kOldestBlobFileNumber);
226 std::string oldest_blob_file_number;
227 PutVarint64(&oldest_blob_file_number, f.oldest_blob_file_number);
228 PutLengthPrefixedSlice(dst, Slice(oldest_blob_file_number));
229 }
230 UniqueId64x2 unique_id = f.unique_id;
231 TEST_SYNC_POINT_CALLBACK("VersionEdit::EncodeTo:UniqueId", &unique_id);
232 if (unique_id != kNullUniqueId64x2) {
233 PutVarint32(dst, NewFileCustomTag::kUniqueId);
234 std::string unique_id_str = EncodeUniqueIdBytes(&unique_id);
235 PutLengthPrefixedSlice(dst, Slice(unique_id_str));
236 }
237 if (f.compensated_range_deletion_size) {
238 PutVarint32(dst, kCompensatedRangeDeletionSize);
239 std::string compensated_range_deletion_size;
240 PutVarint64(&compensated_range_deletion_size,
241 f.compensated_range_deletion_size);
242 PutLengthPrefixedSlice(dst, Slice(compensated_range_deletion_size));
243 }
244 if (f.tail_size) {
245 PutVarint32(dst, NewFileCustomTag::kTailSize);
246 std::string varint_tail_size;
247 PutVarint64(&varint_tail_size, f.tail_size);
248 PutLengthPrefixedSlice(dst, Slice(varint_tail_size));
249 }
250 if (!f.user_defined_timestamps_persisted) {
251 // The default value for the flag is true, it's only explicitly persisted
252 // when it's false. We are putting 0 as the value here to signal false
253 // (i.e. UDTS not persisted).
254 PutVarint32(dst, NewFileCustomTag::kUserDefinedTimestampsPersisted);
255 char p = static_cast<char>(0);
256 PutLengthPrefixedSlice(dst, Slice(&p, 1));
257 }
258 TEST_SYNC_POINT_CALLBACK("VersionEdit::EncodeTo:NewFile4:CustomizeFields",
259 dst);
260
261 PutVarint32(dst, NewFileCustomTag::kTerminate);
262 }
263
264 for (const auto& blob_file_addition : blob_file_additions_) {
265 PutVarint32(dst, kBlobFileAddition);
266 blob_file_addition.EncodeTo(dst);
267 }
268
269 for (const auto& blob_file_garbage : blob_file_garbages_) {
270 PutVarint32(dst, kBlobFileGarbage);
271 blob_file_garbage.EncodeTo(dst);
272 }
273
274 for (const auto& wal_addition : wal_additions_) {
275 PutVarint32(dst, kWalAddition2);
276 std::string encoded;
277 wal_addition.EncodeTo(&encoded);
278 PutLengthPrefixedSlice(dst, encoded);
279 }
280
281 if (!wal_deletion_.IsEmpty()) {
282 PutVarint32(dst, kWalDeletion2);
283 std::string encoded;
284 wal_deletion_.EncodeTo(&encoded);
285 PutLengthPrefixedSlice(dst, encoded);
286 }
287
288 // 0 is default and does not need to be explicitly written
289 if (column_family_ != 0) {
290 PutVarint32Varint32(dst, kColumnFamily, column_family_);
291 }
292
293 if (is_column_family_add_) {
294 PutVarint32(dst, kColumnFamilyAdd);
295 PutLengthPrefixedSlice(dst, Slice(column_family_name_));
296 }
297
298 if (is_column_family_drop_) {
299 PutVarint32(dst, kColumnFamilyDrop);
300 }
301
302 if (is_in_atomic_group_) {
303 PutVarint32(dst, kInAtomicGroup);
304 PutVarint32(dst, remaining_entries_);
305 }
306
307 if (HasFullHistoryTsLow()) {
308 PutVarint32(dst, kFullHistoryTsLow);
309 PutLengthPrefixedSlice(dst, full_history_ts_low_);
310 }
311
312 if (HasPersistUserDefinedTimestamps()) {
313 // persist_user_defined_timestamps flag should be logged in the same
314 // VersionEdit as the user comparator name.
315 assert(has_comparator_);
316 PutVarint32(dst, kPersistUserDefinedTimestamps);
317 char p = static_cast<char>(persist_user_defined_timestamps_);
318 PutLengthPrefixedSlice(dst, Slice(&p, 1));
319 }
320 return true;
321}

323static bool GetInternalKey(Slice* input, InternalKey* dst) {
324 Slice str;
325 if (GetLengthPrefixedSlice(input, &str)) {
326 dst->DecodeFrom(str);
327 return dst->Valid();
328 } else {
329 return false;

333bool VersionEdit::GetLevel(Slice* input, int* level, const char** /*msg*/) {
334 uint32_t v = 0;
335 if (GetVarint32(input, &v)) {
336 *level = v;
337 if (max_level_ < *level) {
338 max_level_ = *level;
339 }
340 return true;
341 } else {
342 return false;
343 }

346const char* VersionEdit::DecodeNewFile4From(Slice* input) {
347 const char* msg = nullptr;
348 int level = 0;
349 FileMetaData f;
350 uint64_t number = 0;
351 uint32_t path_id = 0;
352 uint64_t file_size = 0;
353 SequenceNumber smallest_seqno = 0;
354 SequenceNumber largest_seqno = kMaxSequenceNumber;
355 if (GetLevel(input, &level, &msg) && GetVarint64(input, &number) &&
356 GetVarint64(input, &file_size) && GetInternalKey(input, &f.smallest) &&
357 GetInternalKey(input, &f.largest) &&
358 GetVarint64(input, &smallest_seqno) &&
359 GetVarint64(input, &largest_seqno)) {
360 // See comments in VersionEdit::EncodeTo() for format of customized fields
361 while (true) {
362 uint32_t custom_tag = 0;
363 Slice field;
364 if (!GetVarint32(input, &custom_tag)) {
365 return "new-file4 custom field";
366 }
367 if (custom_tag == kTerminate) {
368 break;
369 }
370 if (!GetLengthPrefixedSlice(input, &field)) {
371 return "new-file4 custom field length prefixed slice error";
372 }
373 switch (custom_tag) {
374 case kPathId:
375 if (field.size() != 1) {
376 return "path_id field wrong size";
377 }
378 path_id = field[0];
379 if (path_id > 3) {
380 return "path_id wrong vaue";
381 }
382 break;
383 case kOldestAncesterTime:
384 if (!GetVarint64(&field, &f.oldest_ancester_time)) {
385 return "invalid oldest ancester time";
386 }
387 break;
388 case kFileCreationTime:
389 if (!GetVarint64(&field, &f.file_creation_time)) {
390 return "invalid file creation time";
391 }
392 break;
393 case kEpochNumber:
394 if (!GetVarint64(&field, &f.epoch_number)) {
395 return "invalid epoch number";
396 }
397 break;
398 case kFileChecksum:
399 f.file_checksum = field.ToString();
400 break;
401 case kFileChecksumFuncName:
402 f.file_checksum_func_name = field.ToString();
403 break;
404 case kNeedCompaction:
405 if (field.size() != 1) {
406 return "need_compaction field wrong size";
407 }
408 f.marked_for_compaction = (field[0] == 1);
409 break;
410 case kMinLogNumberToKeepHack:
411 // This is a hack to encode kMinLogNumberToKeep in a
412 // forward-compatible fashion.
413 if (!GetFixed64(&field, &min_log_number_to_keep_)) {
414 return "deleted log number malformatted";
415 }
416 has_min_log_number_to_keep_ = true;
417 break;
418 case kOldestBlobFileNumber:
419 if (!GetVarint64(&field, &f.oldest_blob_file_number)) {
420 return "invalid oldest blob file number";
421 }
422 break;
423 case kTemperature:
424 if (field.size() != 1) {
425 return "temperature field wrong size";
426 } else {
427 Temperature casted_field = static_cast<Temperature>(field[0]);
428 if (casted_field <= Temperature::kCold) {
429 f.temperature = casted_field;
430 }
431 }
432 break;
433 case kUniqueId:
434 if (!DecodeUniqueIdBytes(field.ToString(), &f.unique_id).ok()) {
435 f.unique_id = kNullUniqueId64x2;
436 return "invalid unique id";
437 }
438 break;
439 case kCompensatedRangeDeletionSize:
440 if (!GetVarint64(&field, &f.compensated_range_deletion_size)) {
441 return "Invalid compensated range deletion size";
442 }
443 break;
444 case kTailSize:
445 if (!GetVarint64(&field, &f.tail_size)) {
446 return "invalid tail start offset";
447 }
448 break;
449 case kUserDefinedTimestampsPersisted:
450 if (field.size() != 1) {
451 return "user-defined timestamps persisted field wrong size";
452 }
453 f.user_defined_timestamps_persisted = (field[0] == 1);
454 break;
455 default:
456 if ((custom_tag & kCustomTagNonSafeIgnoreMask) != 0) {
457 // Should not proceed if cannot understand it
458 return "new-file4 custom field not supported";
459 }
460 break;
461 }
462 }
463 } else {
464 return "new-file4 entry";
465 }
466 f.fd =
467 FileDescriptor(number, path_id, file_size, smallest_seqno, largest_seqno);
468 new_files_.push_back(std::make_pair(level, f));
469 return nullptr;
470}

472void VersionEdit::EncodeFileBoundaries(std::string* dst,
473 const FileMetaData& meta,
474 size_t ts_sz) const {
475 if (ts_sz == 0 || meta.user_defined_timestamps_persisted) {
476 PutLengthPrefixedSlice(dst, meta.smallest.Encode());
477 PutLengthPrefixedSlice(dst, meta.largest.Encode());
478 return;
479 }
480 std::string smallest_buf;
481 std::string largest_buf;
482 StripTimestampFromInternalKey(&smallest_buf, meta.smallest.Encode(), ts_sz);
483 StripTimestampFromInternalKey(&largest_buf, meta.largest.Encode(), ts_sz);
484 PutLengthPrefixedSlice(dst, smallest_buf);
485 PutLengthPrefixedSlice(dst, largest_buf);
486}

488Status VersionEdit::DecodeFrom(const Slice& src) {
489 Clear();
490#ifndef NDEBUG
491 bool ignore_ignorable_tags = false;
492 TEST_SYNC_POINT_CALLBACK("VersionEdit::EncodeTo:IgnoreIgnorableTags",
493 &ignore_ignorable_tags);
494#endif
495 Slice input = src;
496 const char* msg = nullptr;
497 uint32_t tag = 0;
498
499 // Temporary storage for parsing
500 int level = 0;
501 FileMetaData f;
502 Slice str;
503 InternalKey key;
504 while (msg == nullptr && GetVarint32(&input, &tag)) {
505#ifndef NDEBUG
506 if (ignore_ignorable_tags && tag > kTagSafeIgnoreMask) {
507 tag = kTagSafeIgnoreMask;
508 }
509#endif
510 switch (tag) {
511 case kDbId:
512 if (GetLengthPrefixedSlice(&input, &str)) {
513 db_id_ = str.ToString();
514 has_db_id_ = true;
515 } else {
516 msg = "db id";
517 }
518 break;
519 case kComparator:
520 if (GetLengthPrefixedSlice(&input, &str)) {
521 comparator_ = str.ToString();
522 has_comparator_ = true;
523 } else {
524 msg = "comparator name";
525 }
526 break;
527
528 case kLogNumber:
529 if (GetVarint64(&input, &log_number_)) {
530 has_log_number_ = true;
531 } else {
532 msg = "log number";
533 }
534 break;
535
536 case kPrevLogNumber:
537 if (GetVarint64(&input, &prev_log_number_)) {
538 has_prev_log_number_ = true;
539 } else {
540 msg = "previous log number";
541 }
542 break;
543
544 case kNextFileNumber:
545 if (GetVarint64(&input, &next_file_number_)) {
546 has_next_file_number_ = true;
547 } else {
548 msg = "next file number";
549 }
550 break;
551
552 case kMaxColumnFamily:
553 if (GetVarint32(&input, &max_column_family_)) {
554 has_max_column_family_ = true;
555 } else {
556 msg = "max column family";
557 }
558 break;
559
560 case kMinLogNumberToKeep:
561 if (GetVarint64(&input, &min_log_number_to_keep_)) {
562 has_min_log_number_to_keep_ = true;
563 } else {
564 msg = "min log number to kee";
565 }
566 break;
567
568 case kLastSequence:
569 if (GetVarint64(&input, &last_sequence_)) {
570 has_last_sequence_ = true;
571 } else {
572 msg = "last sequence number";
573 }
574 break;
575
576 case kCompactCursor:
577 if (GetLevel(&input, &level, &msg) && GetInternalKey(&input, &key)) {
578 // Here we re-use the output format of compact pointer in LevelDB
579 // to persist compact_cursors_
580 compact_cursors_.push_back(std::make_pair(level, key));
581 } else {
582 if (!msg) {
583 msg = "compaction cursor";
584 }
585 }
586 break;
587
588 case kDeletedFile: {
589 uint64_t number = 0;
590 if (GetLevel(&input, &level, &msg) && GetVarint64(&input, &number)) {
591 deleted_files_.insert(std::make_pair(level, number));
592 } else {
593 if (!msg) {
594 msg = "deleted file";
595 }
596 }
597 break;
598 }
599
600 case kNewFile: {
601 uint64_t number = 0;
602 uint64_t file_size = 0;
603 if (GetLevel(&input, &level, &msg) && GetVarint64(&input, &number) &&
604 GetVarint64(&input, &file_size) &&
605 GetInternalKey(&input, &f.smallest) &&
606 GetInternalKey(&input, &f.largest)) {
607 f.fd = FileDescriptor(number, 0, file_size);
608 new_files_.push_back(std::make_pair(level, f));
609 } else {
610 if (!msg) {
611 msg = "new-file entry";
612 }
613 }
614 break;
615 }
616 case kNewFile2: {
617 uint64_t number = 0;
618 uint64_t file_size = 0;
619 SequenceNumber smallest_seqno = 0;
620 SequenceNumber largest_seqno = kMaxSequenceNumber;
621 if (GetLevel(&input, &level, &msg) && GetVarint64(&input, &number) &&
622 GetVarint64(&input, &file_size) &&
623 GetInternalKey(&input, &f.smallest) &&
624 GetInternalKey(&input, &f.largest) &&
625 GetVarint64(&input, &smallest_seqno) &&
626 GetVarint64(&input, &largest_seqno)) {
627 f.fd = FileDescriptor(number, 0, file_size, smallest_seqno,
628 largest_seqno);
629 new_files_.push_back(std::make_pair(level, f));
630 } else {
631 if (!msg) {
632 msg = "new-file2 entry";
633 }
634 }
635 break;
636 }
637
638 case kNewFile3: {
639 uint64_t number = 0;
640 uint32_t path_id = 0;
641 uint64_t file_size = 0;
642 SequenceNumber smallest_seqno = 0;
643 SequenceNumber largest_seqno = kMaxSequenceNumber;
644 if (GetLevel(&input, &level, &msg) && GetVarint64(&input, &number) &&
645 GetVarint32(&input, &path_id) && GetVarint64(&input, &file_size) &&
646 GetInternalKey(&input, &f.smallest) &&
647 GetInternalKey(&input, &f.largest) &&
648 GetVarint64(&input, &smallest_seqno) &&
649 GetVarint64(&input, &largest_seqno)) {
650 f.fd = FileDescriptor(number, path_id, file_size, smallest_seqno,
651 largest_seqno);
652 new_files_.push_back(std::make_pair(level, f));
653 } else {
654 if (!msg) {
655 msg = "new-file3 entry";
656 }
657 }
658 break;
659 }
660
661 case kNewFile4: {
662 msg = DecodeNewFile4From(&input);
663 break;
664 }
665
666 case kBlobFileAddition:
667 case kBlobFileAddition_DEPRECATED: {
668 BlobFileAddition blob_file_addition;
669 const Status s = blob_file_addition.DecodeFrom(&input);
670 if (!s.ok()) {
671 return s;
672 }
673
674 AddBlobFile(std::move(blob_file_addition));
675 break;
676 }
677
678 case kBlobFileGarbage:
679 case kBlobFileGarbage_DEPRECATED: {
680 BlobFileGarbage blob_file_garbage;
681 const Status s = blob_file_garbage.DecodeFrom(&input);
682 if (!s.ok()) {
683 return s;
684 }
685
686 AddBlobFileGarbage(std::move(blob_file_garbage));
687 break;
688 }
689
690 case kWalAddition: {
691 WalAddition wal_addition;
692 const Status s = wal_addition.DecodeFrom(&input);
693 if (!s.ok()) {
694 return s;
695 }
696
697 wal_additions_.emplace_back(std::move(wal_addition));
698 break;
699 }
700
701 case kWalAddition2: {
702 Slice encoded;
703 if (!GetLengthPrefixedSlice(&input, &encoded)) {
704 msg = "WalAddition not prefixed by length";
705 break;
706 }
707
708 WalAddition wal_addition;
709 const Status s = wal_addition.DecodeFrom(&encoded);
710 if (!s.ok()) {
711 return s;
712 }
713
714 wal_additions_.emplace_back(std::move(wal_addition));
715 break;
716 }
717
718 case kWalDeletion: {
719 WalDeletion wal_deletion;
720 const Status s = wal_deletion.DecodeFrom(&input);
721 if (!s.ok()) {
722 return s;
723 }
724
725 wal_deletion_ = std::move(wal_deletion);
726 break;
727 }
728
729 case kWalDeletion2: {
730 Slice encoded;
731 if (!GetLengthPrefixedSlice(&input, &encoded)) {
732 msg = "WalDeletion not prefixed by length";
733 break;
734 }
735
736 WalDeletion wal_deletion;
737 const Status s = wal_deletion.DecodeFrom(&encoded);
738 if (!s.ok()) {
739 return s;
740 }
741
742 wal_deletion_ = std::move(wal_deletion);
743 break;
744 }
745
746 case kColumnFamily:
747 if (!GetVarint32(&input, &column_family_)) {
748 if (!msg) {
749 msg = "set column family id";
750 }
751 }
752 break;
753
754 case kColumnFamilyAdd:
755 if (GetLengthPrefixedSlice(&input, &str)) {
756 is_column_family_add_ = true;
757 column_family_name_ = str.ToString();
758 } else {
759 if (!msg) {
760 msg = "column family add";
761 }
762 }
763 break;
764
765 case kColumnFamilyDrop:
766 is_column_family_drop_ = true;
767 break;
768
769 case kInAtomicGroup:
770 is_in_atomic_group_ = true;
771 if (!GetVarint32(&input, &remaining_entries_)) {
772 if (!msg) {
773 msg = "remaining entries";
774 }
775 }
776 break;
777
778 case kFullHistoryTsLow:
779 if (!GetLengthPrefixedSlice(&input, &str)) {
780 msg = "full_history_ts_low";
781 } else if (str.empty()) {
782 msg = "full_history_ts_low: empty";
783 } else {
784 full_history_ts_low_.assign(str.data(), str.size());
785 }
786 break;
787
788 case kPersistUserDefinedTimestamps:
789 if (!GetLengthPrefixedSlice(&input, &str)) {
790 msg = "persist_user_defined_timestamps";
791 } else if (str.size() != 1) {
792 msg = "persist_user_defined_timestamps field wrong size";
793 } else {
794 persist_user_defined_timestamps_ = (str[0] == 1);
795 has_persist_user_defined_timestamps_ = true;
796 }
797 break;
798
799 default:
800 if (tag & kTagSafeIgnoreMask) {
801 // Tag from future which can be safely ignored.
802 // The next field must be the length of the entry.
803 uint32_t field_len;
804 if (!GetVarint32(&input, &field_len) ||
805 static_cast<size_t>(field_len) > input.size()) {
806 if (!msg) {
807 msg = "safely ignoreable tag length error";
808 }
809 } else {
810 input.remove_prefix(static_cast<size_t>(field_len));
811 }
812 } else {
813 msg = "unknown tag";
814 }
815 break;
816 }
817 }
818
819 if (msg == nullptr && !input.empty()) {
820 msg = "invalid tag";
821 }
822
823 Status result;
824 if (msg != nullptr) {
825 result = Status::Corruption("VersionEdit", msg);
826 }
827 return result;
828}
