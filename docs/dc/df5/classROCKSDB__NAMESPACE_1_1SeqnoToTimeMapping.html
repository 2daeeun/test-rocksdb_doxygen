135 }
136 // Treat cap of 1 as 2 to work with the below algorithm (etc.)
137 if (strict_cap == 1) {
138 strict_cap = 2;
139 }
140 // When !strict, allow being over nominal capacity by a modest fraction.
141 uint64_t effective_cap = strict_cap + (strict ? 0 : strict_cap / 8);
142 if (effective_cap < strict_cap) {
143 // Correct overflow
144 effective_cap = UINT64_MAX;
145 }
146 if (pairs_.size() <= effective_cap) {
147 return;
148 }
149 // The below algorithm expects at least one removal candidate between first
150 // and last.
151 assert(pairs_.size() >= 3);
152 size_t to_remove_count = pairs_.size() - strict_cap;
153
154 struct RemovalCandidate {
155 uint64_t new_time_gap;
156 std::deque<SeqnoTimePair>::iterator it;
157 RemovalCandidate(uint64_t _new_time_gap,
158 std::deque<SeqnoTimePair>::iterator _it)
159 : new_time_gap(_new_time_gap), it(_it) {}
160 bool operator>(const RemovalCandidate& other) const {
161 if (new_time_gap == other.new_time_gap) {
162 // If same gap, treat the newer entry as less attractive
163 // for removal (like larger gap)
164 return it->seqno > other.it->seqno;
165 }
166 return new_time_gap > other.new_time_gap;
167 }
168 };
169
170 // A priority queue of best removal candidates (smallest time gap remaining
171 // after removal)
172 using RC = RemovalCandidate;
173 using PQ = std::priority_queue<RC, std::vector<RC>, std::greater<RC>>;
174 PQ pq;
175
176 // Add all the candidates (not including first and last)
177 {
178 auto it = pairs_.begin();
179 assert(it->time != kUnknownTimeBeforeAll);
180 uint64_t prev_prev_time = it->time;
181 ++it;
182 assert(it->time != kUnknownTimeBeforeAll);
183 auto prev_it = it;
184 ++it;
185 while (it != pairs_.end()) {
186 assert(it->time != kUnknownTimeBeforeAll);
187 uint64_t gap = it->time - prev_prev_time;
188 pq.emplace(gap, prev_it);
189 prev_prev_time = prev_it->time;
190 prev_it = it;
191 ++it;
192 }
193 }
194
195 // Greedily remove the best candidate, iteratively
196 while (to_remove_count > 0) {
197 assert(!pq.empty());
198 // Remove the candidate with smallest gap
199 auto rc = pq.top();
200 pq.pop();
201
202 // NOTE: priority_queue does not support updating an existing element,
203 // but we can work around that because the gap tracked in pq is only
204 // going to be better than actuality, and we can detect and adjust
205 // when a better-than-actual gap is found.
206
207 // Determine actual time gap if this entry is removed (zero entries are
208 // marked for deletion)
209 auto it = rc.it + 1;
210 uint64_t after_time = it->time;
211 while (after_time == kUnknownTimeBeforeAll) {
212 assert(it != pairs_.end());
213 ++it;
214 after_time = it->time;
215 }
216 it = rc.it - 1;
217 uint64_t before_time = it->time;
218 while (before_time == kUnknownTimeBeforeAll) {
219 assert(it != pairs_.begin());
220 --it;
221 before_time = it->time;
222 }
223 // Check whether the gap is still valid (or needs to be recomputed)
224 if (rc.new_time_gap == after_time - before_time) {
225 // Mark the entry as removed
226 rc.it->time = kUnknownTimeBeforeAll;
227 --to_remove_count;
228 } else {
229 // Insert a replacement up-to-date removal candidate
230 pq.emplace(after_time - before_time, rc.it);
231 }
232 }
233
234 // Collapse away entries marked for deletion
235 auto from_it = pairs_.begin();
236 auto to_it = from_it;
237
238 for (; from_it != pairs_.end(); ++from_it) {
239 if (from_it->time != kUnknownTimeBeforeAll) {
240 if (from_it != to_it) {
241 *to_it = *from_it;
242 }
243 ++to_it;
244 }
245 }
246
247 // Erase slots freed up
248 pairs_.erase(to_it, pairs_.end());
