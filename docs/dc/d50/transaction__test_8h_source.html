162 Status OpenWithStackableDB(std::vector<ColumnFamilyDescriptor>& cfs,
163 std::vector<ColumnFamilyHandle*>* handles) {
164 std::vector<size_t> compaction_enabled_cf_indices;
165 TransactionDB::PrepareWrap(&options, &cfs, &compaction_enabled_cf_indices);
166 DB* root_db = nullptr;
167 Options options_copy(options);
168 const bool use_seq_per_batch =
169 txn_db_options.write_policy == WRITE_PREPARED ||
170 txn_db_options.write_policy == WRITE_UNPREPARED;
171 const bool use_batch_per_txn =
172 txn_db_options.write_policy == WRITE_COMMITTED ||
173 txn_db_options.write_policy == WRITE_PREPARED;
174 Status s = DBImpl::Open(options_copy, dbname, cfs, handles, &root_db,
175 use_seq_per_batch, use_batch_per_txn,
176 /*is_retry=*/false, /*can_retry=*/nullptr);
177 auto stackable_db = std::make_unique<StackableDB>(root_db);
178 if (s.ok()) {
179 assert(root_db != nullptr);
180 // If WrapStackableDB() returns non-ok, then stackable_db is already
181 // deleted within WrapStackableDB().
182 s = TransactionDB::WrapStackableDB(stackable_db.release(), txn_db_options,
183 compaction_enabled_cf_indices,
184 *handles, &db);
185 }
186 return s;
187 }

189 Status OpenWithStackableDB() {
190 std::vector<size_t> compaction_enabled_cf_indices;
191 std::vector<ColumnFamilyDescriptor> column_families{ColumnFamilyDescriptor(
192 kDefaultColumnFamilyName, ColumnFamilyOptions(options))};
193
194 TransactionDB::PrepareWrap(&options, &column_families,
195 &compaction_enabled_cf_indices);
196 std::vector<ColumnFamilyHandle*> handles;
197 DB* root_db = nullptr;
198 Options options_copy(options);
199 const bool use_seq_per_batch =
200 txn_db_options.write_policy == WRITE_PREPARED ||
201 txn_db_options.write_policy == WRITE_UNPREPARED;
202 const bool use_batch_per_txn =
203 txn_db_options.write_policy == WRITE_COMMITTED ||
204 txn_db_options.write_policy == WRITE_PREPARED;
205 Status s = DBImpl::Open(options_copy, dbname, column_families, &handles,
206 &root_db, use_seq_per_batch, use_batch_per_txn,
207 /*is_retry=*/false, /*can_retry=*/nullptr);
208 if (!s.ok()) {
209 delete root_db;
210 return s;
211 }
212 StackableDB* stackable_db = new StackableDB(root_db);
213 assert(root_db != nullptr);
214 assert(handles.size() == 1);
215 s = TransactionDB::WrapStackableDB(stackable_db, txn_db_options,
216 compaction_enabled_cf_indices, handles,
217 &db);
218 delete handles[0];
219 if (!s.ok()) {
220 delete stackable_db;
221 }
222 return s;
223 }

273 void TestTxn2(size_t index) {
274 // Commit without prepare. It should write to DB without a commit marker.

281 ASSERT_OK(txn->Put(Slice("foo3" + istr), Slice("bar3")));
282 ASSERT_OK(txn->Put(Slice("foo4" + istr), Slice("bar4")));
283 ASSERT_OK(txn->Commit());

285 // Consume one seq per key
286 exp_seq += 4;
287 } else if (txn_db_options.write_policy ==
288 TxnDBWritePolicy::WRITE_PREPARED) {
289 // Consume one seq per batch
290 exp_seq++;
291 if (options.two_write_queues) {
292 // Consume one seq for commit
293 exp_seq++;
294 }
295 } else {
296 // Flushed after each key, consume one seq per flushed batch
297 exp_seq += 4;
298 // WriteUnprepared implements CommitWithoutPrepareInternal by simply
299 // calling Prepare then Commit. Consume one seq for the prepare.
300 exp_seq++;
301 }
302 delete txn;

380 // Test that we can change write policy after a clean shutdown (which would
381 // empty the WAL)
382 void CrossCompatibilityTest(TxnDBWritePolicy from_policy,
383 TxnDBWritePolicy to_policy, bool empty_wal) {
384 TransactionOptions txn_options;
385 ReadOptions read_options;
386 WriteOptions write_options;
387 uint32_t index = 0;
388 Random rnd(1103);
389 options.write_buffer_size = 1024; // To create more sst files
390 std::unordered_map<std::string, std::string> committed_kvs;
391 Transaction* txn;
392
393 txn_db_options.write_policy = from_policy;
394 if (txn_db_options.write_policy == WRITE_COMMITTED) {
395 options.unordered_write = false;
396 }
397 ASSERT_OK(ReOpen());
398
399 for (int i = 0; i < 1024; i++) {
400 auto istr = std::to_string(index);
401 auto k = Slice("foo-" + istr).ToString();
402 auto v = Slice("bar-" + istr).ToString();
403 // For test the duplicate keys
404 auto v2 = Slice("bar2-" + istr).ToString();
405 auto type = rnd.Uniform(4);
406 switch (type) {
407 case 0:
408 committed_kvs[k] = v;
409 ASSERT_OK(db->Put(write_options, k, v));
410 committed_kvs[k] = v2;
411 ASSERT_OK(db->Put(write_options, k, v2));
412 break;
413 case 1: {
414 WriteBatch wb;
415 committed_kvs[k] = v;
416 ASSERT_OK(wb.Put(k, v));
417 committed_kvs[k] = v2;
418 ASSERT_OK(wb.Put(k, v2));
419 ASSERT_OK(db->Write(write_options, &wb));
420
421 } break;
422 case 2:
423 case 3:
424 txn = db->BeginTransaction(write_options, txn_options);
425 ASSERT_OK(txn->SetName("xid" + istr));
426 committed_kvs[k] = v;
427 ASSERT_OK(txn->Put(k, v));
428 committed_kvs[k] = v2;
429 ASSERT_OK(txn->Put(k, v2));
430
431 if (type == 3) {
432 ASSERT_OK(txn->Prepare());
433 }
434 ASSERT_OK(txn->Commit());
435 delete txn;
436 break;
437 default:
438 FAIL();
439 }
440
441 index++;
442 } // for i
443
444 txn_db_options.write_policy = to_policy;
445 if (txn_db_options.write_policy == WRITE_COMMITTED) {
446 options.unordered_write = false;
447 }
448 auto db_impl = static_cast_with_check<DBImpl>(db->GetRootDB());
449 // Before upgrade/downgrade the WAL must be emptied
450 if (empty_wal) {
451 ASSERT_OK(db_impl->TEST_FlushMemTable());
452 } else {
453 ASSERT_OK(db_impl->FlushWAL(true));
454 }
455 auto s = ReOpenNoDelete();
456 if (empty_wal) {
457 ASSERT_OK(s);
458 } else {
459 // Test that we can detect the WAL that is produced by an incompatible
460 // WritePolicy and fail fast before mis-interpreting the WAL.
461 ASSERT_TRUE(s.IsNotSupported());
462 return;
463 }
464 db_impl = static_cast_with_check<DBImpl>(db->GetRootDB());
465 // Check that WAL is empty
466 VectorWalPtr log_files;
467 ASSERT_OK(db_impl->GetSortedWalFiles(log_files));
468 ASSERT_EQ(0, log_files.size());
469
470 for (auto& kv : committed_kvs) {
471 std::string value;
472 s = db->Get(read_options, kv.first, &value);
473 if (s.IsNotFound()) {
474 printf("key = %s\n", kv.first.c_str());
475 }
476 ASSERT_OK(s);
477 if (kv.second != value) {
478 printf("key = %s\n", kv.first.c_str());
479 }
480 ASSERT_EQ(kv.second, value);
481 }
482 }
