26size_t CompactedDBImpl::FindFile(const Slice& key) {
27 size_t right = files_.num_files - 1;
28 auto cmp = [&](const FdWithKeyRange& f, const Slice& k) -> bool {
29 return user_comparator_->Compare(ExtractUserKey(f.largest_key), k) < 0;
30 };
31 return static_cast<size_t>(
32 std::lower_bound(files_.files, files_.files + right, key, cmp) -
33 files_.files);
34}

36Status CompactedDBImpl::Get(const ReadOptions& _read_options,
37 ColumnFamilyHandle*, const Slice& key,
38 PinnableSlice* value, std::string* timestamp) {
39 if (_read_options.io_activity != Env::IOActivity::kUnknown &&
40 _read_options.io_activity != Env::IOActivity::kGet) {
41 return Status::InvalidArgument(
42 "Can only call Get with `ReadOptions::io_activity` is "
43 "`Env::IOActivity::kUnknown` or `Env::IOActivity::kGet`");
44 }
45 ReadOptions read_options(_read_options);
46 if (read_options.io_activity == Env::IOActivity::kUnknown) {
47 read_options.io_activity = Env::IOActivity::kGet;
48 }
49
50 assert(user_comparator_);
51 if (read_options.timestamp) {
52 Status s =
53 FailIfTsMismatchCf(DefaultColumnFamily(), *(read_options.timestamp));
54 if (!s.ok()) {
55 return s;
56 }
57 if (read_options.timestamp->size() > 0) {
58 s = FailIfReadCollapsedHistory(cfd_, cfd_->GetSuperVersion(),
59 *(read_options.timestamp));
60 if (!s.ok()) {
61 return s;
62 }
63 }
64 } else {
65 const Status s = FailIfCfHasTs(DefaultColumnFamily());
66 if (!s.ok()) {
67 return s;
68 }
69 }
70
71 // Clear the timestamps for returning results so that we can distinguish
72 // between tombstone or key that has never been written
73 if (timestamp) {
74 timestamp->clear();
75 }
76
77 GetWithTimestampReadCallback read_cb(kMaxSequenceNumber);
78 std::string* ts =
79 user_comparator_->timestamp_size() > 0 ? timestamp : nullptr;
80 LookupKey lkey(key, kMaxSequenceNumber, read_options.timestamp);
81 GetContext get_context(user_comparator_, nullptr, nullptr, nullptr,
82 GetContext::kNotFound, lkey.user_key(), value,
83 /*columns=*/nullptr, ts, nullptr, nullptr, true,
84 nullptr, nullptr, nullptr, nullptr, &read_cb);
85
86 const FdWithKeyRange& f = files_.files[FindFile(lkey.user_key())];
87 if (user_comparator_->CompareWithoutTimestamp(
88 key, /*a_has_ts=*/false,
89 ExtractUserKeyAndStripTimestamp(f.smallest_key,
90 user_comparator_->timestamp_size()),
91 /*b_has_ts=*/false) < 0) {
92 return Status::NotFound();
93 }
94 Status s = f.fd.table_reader->Get(read_options, lkey.internal_key(),
95 &get_context, nullptr);
96 if (!s.ok() && !s.IsNotFound()) {
97 return s;
98 }
99 if (get_context.State() == GetContext::kFound) {
100 return Status::OK();
101 }
102 return Status::NotFound();
103}

105void CompactedDBImpl::MultiGet(const ReadOptions& _read_options,
106 size_t num_keys,
107 ColumnFamilyHandle** /*column_families*/,
108 const Slice* keys, PinnableSlice* values,
109 std::string* timestamps, Status* statuses,
110 const bool /*sorted_input*/) {
111 assert(user_comparator_);
112 Status s;
113 if (_read_options.io_activity != Env::IOActivity::kUnknown &&
114 _read_options.io_activity != Env::IOActivity::kMultiGet) {
115 s = Status::InvalidArgument(
116 "Can only call MultiGet with `ReadOptions::io_activity` is "
117 "`Env::IOActivity::kUnknown` or `Env::IOActivity::kMultiGet`");
118 }
119
120 ReadOptions read_options(_read_options);
121 if (s.ok()) {
122 if (read_options.io_activity == Env::IOActivity::kUnknown) {
123 read_options.io_activity = Env::IOActivity::kMultiGet;
124 }
125
126 if (read_options.timestamp) {
127 s = FailIfTsMismatchCf(DefaultColumnFamily(), *(read_options.timestamp));
128 if (s.ok()) {
129 if (read_options.timestamp->size() > 0) {
130 s = FailIfReadCollapsedHistory(cfd_, cfd_->GetSuperVersion(),
131 *(read_options.timestamp));
132 }
133 }
134 } else {
135 s = FailIfCfHasTs(DefaultColumnFamily());
136 }
137 }
138
139 if (!s.ok()) {
140 for (size_t i = 0; i < num_keys; ++i) {
141 statuses[i] = s;
142 }
143 return;
144 }
145
146 // Clear the timestamps for returning results so that we can distinguish
147 // between tombstone or key that has never been written
148 if (timestamps) {
149 for (size_t i = 0; i < num_keys; ++i) {
150 timestamps[i].clear();
151 }
152 }
153
154 GetWithTimestampReadCallback read_cb(kMaxSequenceNumber);
155 autovector<TableReader*, 16> reader_list;
156 for (size_t i = 0; i < num_keys; ++i) {
157 const Slice& key = keys[i];
158 LookupKey lkey(key, kMaxSequenceNumber, read_options.timestamp);
159 const FdWithKeyRange& f = files_.files[FindFile(lkey.user_key())];
160 if (user_comparator_->CompareWithoutTimestamp(
161 key, /*a_has_ts=*/false,
162 ExtractUserKeyAndStripTimestamp(f.smallest_key,
163 user_comparator_->timestamp_size()),
164 /*b_has_ts=*/false) < 0) {
165 reader_list.push_back(nullptr);
166 } else {
167 f.fd.table_reader->Prepare(lkey.internal_key());
168 reader_list.push_back(f.fd.table_reader);
169 }
170 }
171 for (size_t i = 0; i < num_keys; ++i) {
172 statuses[i] = Status::NotFound();
173 }
174 int idx = 0;
175 for (auto* r : reader_list) {
176 if (r != nullptr) {
177 PinnableSlice& pinnable_val = values[idx];
178 LookupKey lkey(keys[idx], kMaxSequenceNumber, read_options.timestamp);
179 std::string* timestamp = timestamps ? &timestamps[idx] : nullptr;
180 GetContext get_context(
181 user_comparator_, nullptr, nullptr, nullptr, GetContext::kNotFound,
182 lkey.user_key(), &pinnable_val, /*columns=*/nullptr,
183 user_comparator_->timestamp_size() > 0 ? timestamp : nullptr, nullptr,
184 nullptr, true, nullptr, nullptr, nullptr, nullptr, &read_cb);
185 Status status =
186 r->Get(read_options, lkey.internal_key(), &get_context, nullptr);
187 assert(static_cast<size_t>(idx) < num_keys);
188 if (!status.ok() && !status.IsNotFound()) {
189 statuses[idx] = status;
190 } else {
191 if (get_context.State() == GetContext::kFound) {
192 statuses[idx] = Status::OK();
193 }
194 }
195 }
196 ++idx;
197 }
198}

200Status CompactedDBImpl::Init(const Options& options) {
201 SuperVersionContext sv_context(/* create_superversion */ true);
202 mutex_.Lock();
203 ColumnFamilyDescriptor cf(kDefaultColumnFamilyName,
204 ColumnFamilyOptions(options));
205 Status s = Recover({cf}, true /* read only */, false, true);
206 if (s.ok()) {
207 cfd_ = static_cast_with_check<ColumnFamilyHandleImpl>(DefaultColumnFamily())
208 ->cfd();
209 cfd_->InstallSuperVersion(&sv_context, &mutex_);
210 }
211 mutex_.Unlock();
212 sv_context.Clean();
213 if (!s.ok()) {
214 return s;
215 }
216 NewThreadStatusCfInfo(cfd_);
217 version_ = cfd_->GetSuperVersion()->current;
218 user_comparator_ = cfd_->user_comparator();
219 auto* vstorage = version_->storage_info();
220 if (vstorage->num_non_empty_levels() == 0) {
221 return Status::NotSupported("no file exists");
222 }
223 const LevelFilesBrief& l0 = vstorage->LevelFilesBrief(0);
224 // L0 should not have files
225 if (l0.num_files > 1) {
226 return Status::NotSupported("L0 contain more than 1 file");
227 }
228 if (l0.num_files == 1) {
229 if (vstorage->num_non_empty_levels() > 1) {
230 return Status::NotSupported("Both L0 and other level contain files");
231 }
232 files_ = l0;
233 return Status::OK();
234 }
235
236 for (int i = 1; i < vstorage->num_non_empty_levels() - 1; ++i) {
237 if (vstorage->LevelFilesBrief(i).num_files > 0) {
238 return Status::NotSupported("Other levels also contain files");
239 }
240 }
241
242 int level = vstorage->num_non_empty_levels() - 1;
243 if (vstorage->LevelFilesBrief(level).num_files > 0) {
244 files_ = vstorage->LevelFilesBrief(level);
245 return Status::OK();
246 }
247 return Status::NotSupported("no file exists");
248}

250Status CompactedDBImpl::Open(const Options& options, const std::string& dbname,
251 DB** dbptr) {
252 *dbptr = nullptr;
253
254 if (options.max_open_files != -1) {
255 return Status::InvalidArgument("require max_open_files = -1");
256 }
257 if (options.merge_operator.get() != nullptr) {
258 return Status::InvalidArgument("merge operator is not supported");
259 }
260 DBOptions db_options(options);
261 std::unique_ptr<CompactedDBImpl> db(new CompactedDBImpl(db_options, dbname));
262 Status s = db->Init(options);
263 if (s.ok()) {
264 s = db->StartPeriodicTaskScheduler();
265 }
266 if (s.ok()) {
267 ROCKS_LOG_INFO(db->immutable_db_options_.info_log,
268 "Opened the db as fully compacted mode");
269 LogFlush(db->immutable_db_options_.info_log);
270 *dbptr = db.release();
271 }
272 return s;
273}
