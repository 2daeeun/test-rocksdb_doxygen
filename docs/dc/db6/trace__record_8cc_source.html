167// MultiGetQueryTraceRecord
168MultiGetQueryTraceRecord::MultiGetQueryTraceRecord(
169 std::vector<uint32_t> column_family_ids, std::vector<PinnableSlice>&& keys,
170 uint64_t timestamp)
171 : QueryTraceRecord(timestamp),
172 cf_ids_(column_family_ids),
173 keys_(std::move(keys)) {}
174
175MultiGetQueryTraceRecord::MultiGetQueryTraceRecord(
176 std::vector<uint32_t> column_family_ids,
177 const std::vector<std::string>& keys, uint64_t timestamp)
178 : QueryTraceRecord(timestamp), cf_ids_(column_family_ids) {
179 keys_.reserve(keys.size());
180 for (const std::string& key : keys) {
181 PinnableSlice ps;
182 ps.PinSelf(key);
183 keys_.push_back(std::move(ps));
184 }
185}
186
187MultiGetQueryTraceRecord::~MultiGetQueryTraceRecord() {
188 cf_ids_.clear();
189 keys_.clear();
190}

193 return cf_ids_;
194}
