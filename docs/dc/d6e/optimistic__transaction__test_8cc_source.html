398// Trigger the condition where some old memtables are skipped when doing
399// TransactionUtil::CheckKey(), and make sure the result is still correct.
400TEST_P(OptimisticTransactionTest, CheckKeySkipOldMemtable) {
401 const int kAttemptHistoryMemtable = 0;
402 const int kAttemptImmMemTable = 1;
403 for (int attempt = kAttemptHistoryMemtable; attempt <= kAttemptImmMemTable;
404 attempt++) {
405 Reopen();
406
407 WriteOptions write_options;
408 ReadOptions read_options;
409 ReadOptions snapshot_read_options;
410 ReadOptions snapshot_read_options2;
411 std::string value;
412
413 ASSERT_OK(txn_db->Put(write_options, Slice("foo"), Slice("bar")));
414 ASSERT_OK(txn_db->Put(write_options, Slice("foo2"), Slice("bar")));
415
416 Transaction* txn = txn_db->BeginTransaction(write_options);
417 ASSERT_TRUE(txn != nullptr);
418
419 Transaction* txn2 = txn_db->BeginTransaction(write_options);
420 ASSERT_TRUE(txn2 != nullptr);
421
422 snapshot_read_options.snapshot = txn->GetSnapshot();
423 ASSERT_OK(txn->GetForUpdate(snapshot_read_options, "foo", &value));
424 ASSERT_EQ(value, "bar");
425 ASSERT_OK(txn->Put(Slice("foo"), Slice("bar2")));
426
427 snapshot_read_options2.snapshot = txn2->GetSnapshot();
428 ASSERT_OK(txn2->GetForUpdate(snapshot_read_options2, "foo2", &value));
429 ASSERT_EQ(value, "bar");
430 ASSERT_OK(txn2->Put(Slice("foo2"), Slice("bar2")));
431
432 // txn updates "foo" and txn2 updates "foo2", and now a write is
433 // issued for "foo", which conflicts with txn but not txn2
434 ASSERT_OK(txn_db->Put(write_options, "foo", "bar"));
435
436 if (attempt == kAttemptImmMemTable) {
437 // For the second attempt, hold flush from beginning. The memtable
438 // will be switched to immutable after calling TEST_SwitchMemtable()
439 // while CheckKey() is called.
440 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->LoadDependency(
441 {{"OptimisticTransactionTest.CheckKeySkipOldMemtable",
442 "FlushJob::Start"}});
443 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
444 }
445
446 // force a memtable flush. The memtable should still be kept
447 FlushOptions flush_ops;
448 if (attempt == kAttemptHistoryMemtable) {
449 ASSERT_OK(txn_db->Flush(flush_ops));
450 } else {
451 ASSERT_EQ(attempt, kAttemptImmMemTable);
452 DBImpl* db_impl = static_cast<DBImpl*>(txn_db->GetRootDB());
453 ASSERT_OK(db_impl->TEST_SwitchMemtable());
454 }
455 uint64_t num_imm_mems;
456 ASSERT_TRUE(txn_db->GetIntProperty(DB::Properties::kNumImmutableMemTable,
457 &num_imm_mems));
458 if (attempt == kAttemptHistoryMemtable) {
459 ASSERT_EQ(0, num_imm_mems);
460 } else {
461 ASSERT_EQ(attempt, kAttemptImmMemTable);
462 ASSERT_EQ(1, num_imm_mems);
463 }
464
465 // Put something in active memtable
466 ASSERT_OK(txn_db->Put(write_options, Slice("foo3"), Slice("bar")));
467
468 // Create txn3 after flushing, when this transaction is commited,
469 // only need to check the active memtable
470 Transaction* txn3 = txn_db->BeginTransaction(write_options);
471 ASSERT_TRUE(txn3 != nullptr);
472
473 // Commit both of txn and txn2. txn will conflict but txn2 will
474 // pass. In both ways, both memtables are queried.
475 SetPerfLevel(PerfLevel::kEnableCount);
476
477 get_perf_context()->Reset();
478 Status s = txn->Commit();
479 // We should have checked two memtables
480 ASSERT_EQ(2, get_perf_context()->get_from_memtable_count);
481 // txn should fail because of conflict, even if the memtable
482 // has flushed, because it is still preserved in history.
483 ASSERT_TRUE(s.IsBusy());
484
485 get_perf_context()->Reset();
486 s = txn2->Commit();
487 // We should have checked two memtables
488 ASSERT_EQ(2, get_perf_context()->get_from_memtable_count);
489 ASSERT_TRUE(s.ok());
490
491 ASSERT_OK(txn3->Put(Slice("foo2"), Slice("bar2")));
492 get_perf_context()->Reset();
493 s = txn3->Commit();
494 // txn3 is created after the active memtable is created, so that is the only
495 // memtable to check.
496 ASSERT_EQ(1, get_perf_context()->get_from_memtable_count);
497 ASSERT_TRUE(s.ok());
498
499 TEST_SYNC_POINT("OptimisticTransactionTest.CheckKeySkipOldMemtable");
500 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
501
502 SetPerfLevel(PerfLevel::kDisable);
503
504 delete txn;
505 delete txn2;
506 delete txn3;
507 }

1546Status OptimisticTransactionStressTestInserter(OptimisticTransactionDB* db,
1547 const size_t num_transactions,
1548 const size_t num_sets,
1549 const size_t num_keys_per_set) {
1550 size_t seed = std::hash<std::thread::id>()(std::this_thread::get_id());
1551 Random64 _rand(seed);
1552 WriteOptions write_options;
1553 ReadOptions read_options;
1554 OptimisticTransactionOptions txn_options;
1555 txn_options.set_snapshot = true;
1556
1557 RandomTransactionInserter inserter(&_rand, write_options, read_options,
1558 num_keys_per_set,
1559 static_cast<uint16_t>(num_sets));
1560
1561 for (size_t t = 0; t < num_transactions; t++) {
1562 bool success = inserter.OptimisticTransactionDBInsert(db, txn_options);
1563 if (!success) {
1564 // unexpected failure
1565 return inserter.GetLastStatus();
1566 }
1567 }
1568
1569 inserter.GetLastStatus().PermitUncheckedError();
1570
1571 // Make sure at least some of the transactions succeeded. It's ok if
1572 // some failed due to write-conflicts.
1573 if (inserter.GetFailureCount() > num_transactions / 2) {
1574 return Status::TryAgain("Too many transactions failed! " +
1575 std::to_string(inserter.GetFailureCount()) + " / " +
1576 std::to_string(num_transactions));
1577 }
1578
1579 return Status::OK();
1580}
