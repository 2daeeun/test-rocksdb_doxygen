118 """Parse a line into items
119 Being clever about separators
120 """
121 line = line_in
122 row = []
123 while line != "":
124 match_item = self.field.match(line)
125 if match_item:
126 item = match_item.group(0)
127 row.append(item)
128 line = line[len(item) :]
129 else:
130 match_intra = self.intra.match(line)
131 if match_intra:
132 intra = match_intra.group(0)
133 # Count the separators
134 # If there are >1 then generate extra blank fields
135 # White space with no true separators fakes up a single separator
136 tabbed = self.sep.split(intra)
137 sep_count = len(tabbed) - 1

139 sep_count = 1

141 row.append("")

143 else:
