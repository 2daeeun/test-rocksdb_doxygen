31class Log:
32 @staticmethod
33 def is_new_log(log_line):
34 # The assumption is that a new log will start with a date printed in
35 # the below regex format.
36 date_regex = r"\d{4}/\d{2}/\d{2}-\d{2}:\d{2}:\d{2}\.\d{6}" # noqa
37 return re.match(date_regex, log_line)

39 def __init__(self, log_line, column_families):
40 token_list = log_line.strip().split()
41 self.time = token_list[0]
42 self.context = token_list[1]
43 self.message = " ".join(token_list[2:])
44 self.column_family = None
45 # example log for 'default' column family:
46 # "2018/07/25-17:29:05.176080 7f969de68700 [db/compaction_job.cc:1634]
47 # [default] [JOB 3] Compacting 24@0 + 16@1 files to L1, score 6.00\n"
48 for col_fam in column_families:
49 search_for_str = r"\[" + col_fam + r"\]" # noqa
50 if re.search(search_for_str, self.message):
51 self.column_family = col_fam
52 break
53 if not self.column_family:
54 self.column_family = NO_COL_FAMILY

72 def get_timestamp(self):
73 # example: '2018/07/25-11:25:45.782710' will be converted to the GMT
74 # Unix timestamp 1532517945 (note: this method assumes that self.time
75 # is in GMT)
76 hr_time = self.time + "GMT"
77 timestamp = timegm(time.strptime(hr_time, "%Y/%m/%d-%H:%M:%S.%f%Z"))
78 return timestamp

94 def __init__(self, logs_path_prefix, column_families):
95 super().__init__(DataSource.Type.LOG)

99 def trigger_conditions_for_log(self, conditions, log):
100 # For a LogCondition object, trigger is:
101 # Dict[column_family_name, List[Log]]. This explains why the condition
102 # was triggered and for which column families.
103 for cond in conditions:
104 if re.search(cond.regex, log.get_message(), re.IGNORECASE):
105 trigger = cond.get_trigger()
106 if not trigger:
107 trigger = {}
108 if log.get_column_family() not in trigger:
109 trigger[log.get_column_family()] = []
110 trigger[log.get_column_family()].append(log)
111 cond.set_trigger(trigger)
112
113 def check_and_trigger_conditions(self, conditions):
114 for file_name in glob.glob(self.logs_path_prefix + "*"):
115 # TODO(poojam23): find a way to distinguish between log files
116 # - generated in the current experiment but are labeled 'old'
117 # because they LOGs exceeded the file size limit AND
118 # - generated in some previous experiment that are also labeled
119 # 'old' and were not deleted for some reason
120 if re.search("old", file_name, re.IGNORECASE):
121 continue
122 with open(file_name) as db_logs:
123 new_log = None
124 for line in db_logs:
125 if Log.is_new_log(line):
126 if new_log:
127 self.trigger_conditions_for_log(conditions, new_log)
128 new_log = Log(line, self.column_families)
129 else:
130 # To account for logs split into multiple lines
131 new_log.append_message(line)
132 # Check for the last log in the file.
133 if new_log:
134 self.trigger_conditions_for_log(conditions, new_log)
