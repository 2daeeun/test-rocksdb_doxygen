40std::unique_ptr<TaskLimiterToken> ConcurrentTaskLimiterImpl::GetToken(
41 bool force) {
42 int32_t limit = max_outstanding_tasks_.load(std::memory_order_relaxed);
43 int32_t tasks = outstanding_tasks_.load(std::memory_order_relaxed);
44 // force = true, bypass the throttle.
45 // limit < 0 means unlimited tasks.
46 while (force || limit < 0 || tasks < limit) {
47 if (outstanding_tasks_.compare_exchange_weak(tasks, tasks + 1)) {
48 return std::unique_ptr<TaskLimiterToken>(new TaskLimiterToken(this));
49 }
50 }
51 return nullptr;
52}
