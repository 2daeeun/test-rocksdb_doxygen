76 const ImmutableOptions& ioptions,

92 comparator_, &arena_, mutable_cf_options.prefix_extractor.get(),
93 ioptions.logger, column_family_id)),

95 comparator_, &arena_, nullptr /* transform */, ioptions.logger,
96 column_family_id)),
97 is_range_del_table_empty_(true),
98 data_size_(0),
99 num_entries_(0),
100 num_deletes_(0),
101 num_range_deletes_(0),
102 write_buffer_size_(mutable_cf_options.write_buffer_size),
103 first_seqno_(0),
104 earliest_seqno_(latest_seq),
105 creation_seq_(latest_seq),

117 memtable_max_range_deletions_(
118 mutable_cf_options.memtable_max_range_deletions) {
119 UpdateFlushState();
120 // something went wrong if we need to flush before inserting anything
121 assert(!ShouldScheduleFlush());
122
123 // use bloom_filter_ for both whole key and prefix bloom filter
124 if ((prefix_extractor_ || moptions_.memtable_whole_key_filtering) &&
125 moptions_.memtable_prefix_bloom_bits > 0) {
126 bloom_filter_.reset(
127 new DynamicBloom(&arena_, moptions_.memtable_prefix_bloom_bits,
128 6 /* hard coded 6 probes */,
129 moptions_.memtable_huge_page_size, ioptions.logger));
130 }
131 // Initialize cached_range_tombstone_ here since it could
132 // be read before it is constructed in MemTable::Add(), which could also lead
133 // to a data race on the global mutex table backing atomic shared_ptr.
134 auto new_cache = std::make_shared<FragmentedRangeTombstoneListCache>();
135 size_t size = cached_range_tombstone_.Size();
136 for (size_t i = 0; i < size; ++i) {
137 std::shared_ptr<FragmentedRangeTombstoneListCache>* local_cache_ref_ptr =
138 cached_range_tombstone_.AccessAtCore(i);
139 auto new_local_cache_ref = std::make_shared<
140 const std::shared_ptr<FragmentedRangeTombstoneListCache>>(new_cache);
141 std::atomic_store_explicit(
142 local_cache_ref_ptr,
143 std::shared_ptr<FragmentedRangeTombstoneListCache>(new_local_cache_ref,
144 new_cache.get()),
145 std::memory_order_relaxed);
146 }
147 const Comparator* ucmp = cmp.user_comparator();
148 assert(ucmp);
149 ts_sz_ = ucmp->timestamp_size();
150 persist_user_defined_timestamps_ = ioptions.persist_user_defined_timestamps;
151}

158size_t MemTable::ApproximateMemoryUsage() {
159 autovector<size_t> usages = {
160 arena_.ApproximateMemoryUsage(), table_->ApproximateMemoryUsage(),
161 range_del_table_->ApproximateMemoryUsage(),
162 ROCKSDB_NAMESPACE::ApproximateMemoryUsage(insert_hints_)};
163 size_t total_usage = 0;
164 for (size_t usage : usages) {
165 // If usage + total_usage >= kMaxSizet, return kMaxSizet.
166 // the following variation is to avoid numeric overflow.
167 if (usage >= std::numeric_limits<size_t>::max() - total_usage) {
168 return std::numeric_limits<size_t>::max();
169 }
170 total_usage += usage;
171 }
172 approximate_memory_usage_.store(total_usage, std::memory_order_relaxed);
173 // otherwise, return the actual usage
174 return total_usage;
175}

177bool MemTable::ShouldFlushNow() {
178 // This is set if memtable_max_range_deletions is > 0,
179 // and that many range deletions are done
180 if (memtable_max_range_deletions_ > 0 &&
181 num_range_deletes_.load(std::memory_order_relaxed) >=
182 static_cast<uint64_t>(memtable_max_range_deletions_)) {
183 return true;
184 }
185
186 size_t write_buffer_size = write_buffer_size_.load(std::memory_order_relaxed);
187 // In a lot of times, we cannot allocate arena blocks that exactly matches the
188 // buffer size. Thus we have to decide if we should over-allocate or
189 // under-allocate.
190 // This constant variable can be interpreted as: if we still have more than
191 // "kAllowOverAllocationRatio * kArenaBlockSize" space left, we'd try to over
192 // allocate one more block.
193 const double kAllowOverAllocationRatio = 0.6;
194
195 // If arena still have room for new block allocation, we can safely say it
196 // shouldn't flush.
197 auto allocated_memory = table_->ApproximateMemoryUsage() +
198 range_del_table_->ApproximateMemoryUsage() +
199 arena_.MemoryAllocatedBytes();
200
201 approximate_memory_usage_.store(allocated_memory, std::memory_order_relaxed);
202
203 // if we can still allocate one more block without exceeding the
204 // over-allocation ratio, then we should not flush.
205 if (allocated_memory + kArenaBlockSize <
206 write_buffer_size + kArenaBlockSize * kAllowOverAllocationRatio) {
207 return false;
208 }
209
210 // if user keeps adding entries that exceeds write_buffer_size, we need to
211 // flush earlier even though we still have much available memory left.
212 if (allocated_memory >
213 write_buffer_size + kArenaBlockSize * kAllowOverAllocationRatio) {
214 return true;
215 }
216
217 // In this code path, Arena has already allocated its "last block", which
218 // means the total allocatedmemory size is either:
219 // (1) "moderately" over allocated the memory (no more than `0.6 * arena
220 // block size`. Or,
221 // (2) the allocated memory is less than write buffer size, but we'll stop
222 // here since if we allocate a new arena block, we'll over allocate too much
223 // more (half of the arena block size) memory.
224 //
225 // In either case, to avoid over-allocate, the last block will stop allocation
226 // when its usage reaches a certain ratio, which we carefully choose "0.75
227 // full" as the stop condition because it addresses the following issue with
228 // great simplicity: What if the next inserted entry's size is
229 // bigger than AllocatedAndUnused()?
230 //
231 // The answer is: if the entry size is also bigger than 0.25 *
232 // kArenaBlockSize, a dedicated block will be allocated for it; otherwise
233 // arena will anyway skip the AllocatedAndUnused() and allocate a new, empty
234 // and regular block. In either case, we *overly* over-allocated.
235 //
236 // Therefore, setting the last block to be at most "0.75 full" avoids both
237 // cases.
238 //
239 // NOTE: the average percentage of waste space of this approach can be counted
240 // as: "arena block size * 0.25 / write buffer size". User who specify a small
241 // write buffer size and/or big arena block size may suffer.
242 return arena_.AllocatedAndUnused() < kArenaBlockSize / 4;
243}

245void MemTable::UpdateFlushState() {
246 auto state = flush_state_.load(std::memory_order_relaxed);
247 if (state == FLUSH_NOT_REQUESTED && ShouldFlushNow()) {
248 // ignore CAS failure, because that means somebody else requested
249 // a flush
250 flush_state_.compare_exchange_strong(state, FLUSH_REQUESTED,
251 std::memory_order_relaxed,
252 std::memory_order_relaxed);
253 }
254}

256void MemTable::UpdateOldestKeyTime() {
257 uint64_t oldest_key_time = oldest_key_time_.load(std::memory_order_relaxed);
258 if (oldest_key_time == std::numeric_limits<uint64_t>::max()) {
259 int64_t current_time = 0;
260 auto s = clock_->GetCurrentTime(&current_time);
261 if (s.ok()) {
262 assert(current_time >= 0);
263 // If fail, the timestamp is already set.
264 oldest_key_time_.compare_exchange_strong(
265 oldest_key_time, static_cast<uint64_t>(current_time),
266 std::memory_order_relaxed, std::memory_order_relaxed);
267 }
268 }

271Status MemTable::VerifyEntryChecksum(const char* entry,
272 uint32_t protection_bytes_per_key,
273 bool allow_data_in_errors) {
274 if (protection_bytes_per_key == 0) {
275 return Status::OK();
276 }
277 uint32_t key_length;
278 const char* key_ptr = GetVarint32Ptr(entry, entry + 5, &key_length);
279 if (key_ptr == nullptr) {
280 return Status::Corruption("Unable to parse internal key length");
281 }
282 if (key_length < 8) {
283 return Status::Corruption("Memtable entry internal key length too short.");
284 }
285 Slice user_key = Slice(key_ptr, key_length - 8);
286
287 const uint64_t tag = DecodeFixed64(key_ptr + key_length - 8);
288 ValueType type;
289 SequenceNumber seq;
290 UnPackSequenceAndType(tag, &seq, &type);
291
292 uint32_t value_length = 0;
293 const char* value_ptr = GetVarint32Ptr(
294 key_ptr + key_length, key_ptr + key_length + 5, &value_length);
295 if (value_ptr == nullptr) {
296 return Status::Corruption("Unable to parse internal key value");
297 }
298 Slice value = Slice(value_ptr, value_length);
299
300 const char* checksum_ptr = value_ptr + value_length;
301 bool match =
302 ProtectionInfo64()
303 .ProtectKVO(user_key, value, type)
304 .ProtectS(seq)
305 .Verify(static_cast<uint8_t>(protection_bytes_per_key), checksum_ptr);
306 if (!match) {
307 std::string msg(
308 "Corrupted memtable entry, per key-value checksum verification "
309 "failed.");
310 if (allow_data_in_errors) {
311 msg.append("Unrecognized value type: " +
312 std::to_string(static_cast<int>(type)) + ". ");
313 msg.append("User key: " + user_key.ToString(/*hex=*/true) + ". ");
314 msg.append("seq: " + std::to_string(seq) + ".");
315 }
316 return Status::Corruption(msg.c_str());
317 }
318 return Status::OK();
319}

360class MemTableIterator : public InternalIterator {
361 public:
362 enum Kind { kPointEntries, kRangeDelEntries };
363 MemTableIterator(
364 Kind kind, const MemTable& mem, const ReadOptions& read_options,
365 UnownedPtr<const SeqnoToTimeMapping> seqno_to_time_mapping = nullptr,
366 Arena* arena = nullptr,
367 const SliceTransform* cf_prefix_extractor = nullptr)
368 : bloom_(nullptr),
369 prefix_extractor_(mem.prefix_extractor_),
370 comparator_(mem.comparator_),
371 seqno_to_time_mapping_(seqno_to_time_mapping),
372 status_(Status::OK()),
373 logger_(mem.moptions_.info_log),
374 ts_sz_(mem.ts_sz_),
375 protection_bytes_per_key_(mem.moptions_.protection_bytes_per_key),
376 valid_(false),
377 value_pinned_(
378 !mem.GetImmutableMemTableOptions()->inplace_update_support),
379 arena_mode_(arena != nullptr),
380 paranoid_memory_checks_(mem.moptions_.paranoid_memory_checks),
381 allow_data_in_error(mem.moptions_.allow_data_in_errors) {
382 if (kind == kRangeDelEntries) {
383 iter_ = mem.range_del_table_->GetIterator(arena);
384 } else if (prefix_extractor_ != nullptr &&
385 // NOTE: checking extractor equivalence when not pointer
386 // equivalent is arguably too expensive for memtable
387 prefix_extractor_ == cf_prefix_extractor &&
388 (read_options.prefix_same_as_start ||
389 (!read_options.total_order_seek &&
390 !read_options.auto_prefix_mode))) {
391 // Auto prefix mode is not implemented in memtable yet.
392 assert(kind == kPointEntries);
393 bloom_ = mem.bloom_filter_.get();
394 iter_ = mem.table_->GetDynamicPrefixIterator(arena);
395 } else {
396 assert(kind == kPointEntries);
397 iter_ = mem.table_->GetIterator(arena);
398 }
399 status_.PermitUncheckedError();
400 }
401 // No copying allowed
402 MemTableIterator(const MemTableIterator&) = delete;
403 void operator=(const MemTableIterator&) = delete;
404
405 ~MemTableIterator() override {
406#ifndef NDEBUG
407 // Assert that the MemTableIterator is never deleted while
408 // Pinning is Enabled.
409 assert(!pinned_iters_mgr_ || !pinned_iters_mgr_->PinningEnabled());
410#endif
411 if (arena_mode_) {
412 iter_->~Iterator();
413 } else {
414 delete iter_;
415 }
416 status_.PermitUncheckedError();
417 }
418
419#ifndef NDEBUG
420 void SetPinnedItersMgr(PinnedIteratorsManager* pinned_iters_mgr) override {
421 pinned_iters_mgr_ = pinned_iters_mgr;
422 }
423 PinnedIteratorsManager* pinned_iters_mgr_ = nullptr;
424#endif
425
426 bool Valid() const override {
427 // If inner iter_ is not valid, then this iter should also not be valid.
428 assert(iter_->Valid() || !(valid_ && status_.ok()));
429 return valid_ && status_.ok();
430 }
431
432 void Seek(const Slice& k) override {
433 PERF_TIMER_GUARD(seek_on_memtable_time);
434 PERF_COUNTER_ADD(seek_on_memtable_count, 1);
435 status_ = Status::OK();
436 if (bloom_) {
437 // iterator should only use prefix bloom filter
438 Slice user_k_without_ts(ExtractUserKeyAndStripTimestamp(k, ts_sz_));
439 if (prefix_extractor_->InDomain(user_k_without_ts)) {
440 Slice prefix = prefix_extractor_->Transform(user_k_without_ts);
441 if (!bloom_->MayContain(prefix)) {
442 PERF_COUNTER_ADD(bloom_memtable_miss_count, 1);
443 valid_ = false;
444 return;
445 } else {
446 PERF_COUNTER_ADD(bloom_memtable_hit_count, 1);
447 }
448 }
449 }
450 if (paranoid_memory_checks_) {
451 status_ = iter_->SeekAndValidate(k, nullptr, allow_data_in_error);
452 } else {
453 iter_->Seek(k, nullptr);
454 }
455 valid_ = iter_->Valid();
456 VerifyEntryChecksum();
457 }
458 void SeekForPrev(const Slice& k) override {
459 PERF_TIMER_GUARD(seek_on_memtable_time);
460 PERF_COUNTER_ADD(seek_on_memtable_count, 1);
461 status_ = Status::OK();
462 if (bloom_) {
463 Slice user_k_without_ts(ExtractUserKeyAndStripTimestamp(k, ts_sz_));
464 if (prefix_extractor_->InDomain(user_k_without_ts)) {
465 if (!bloom_->MayContain(
466 prefix_extractor_->Transform(user_k_without_ts))) {
467 PERF_COUNTER_ADD(bloom_memtable_miss_count, 1);
468 valid_ = false;
469 return;
470 } else {
471 PERF_COUNTER_ADD(bloom_memtable_hit_count, 1);
472 }
473 }
474 }
475 if (paranoid_memory_checks_) {
476 status_ = iter_->SeekAndValidate(k, nullptr, allow_data_in_error);
477 } else {
478 iter_->Seek(k, nullptr);
479 }
480 valid_ = iter_->Valid();
481 VerifyEntryChecksum();
482 if (!Valid() && status().ok()) {
483 SeekToLast();
484 }
485 while (Valid() && comparator_.comparator.Compare(k, key()) < 0) {
486 Prev();
487 }
488 }
489 void SeekToFirst() override {
490 status_ = Status::OK();
491 iter_->SeekToFirst();
492 valid_ = iter_->Valid();
493 VerifyEntryChecksum();
494 }
495 void SeekToLast() override {
496 status_ = Status::OK();
497 iter_->SeekToLast();
498 valid_ = iter_->Valid();
499 VerifyEntryChecksum();
500 }
501 void Next() override {
502 PERF_COUNTER_ADD(next_on_memtable_count, 1);
503 assert(Valid());
504 if (paranoid_memory_checks_) {
505 status_ = iter_->NextAndValidate(allow_data_in_error);
506 } else {
507 iter_->Next();
508 TEST_SYNC_POINT_CALLBACK("MemTableIterator::Next:0", iter_);
509 }
510 valid_ = iter_->Valid();
511 VerifyEntryChecksum();
512 }
513 bool NextAndGetResult(IterateResult* result) override {
514 Next();
515 bool is_valid = Valid();
516 if (is_valid) {
517 result->key = key();
518 result->bound_check_result = IterBoundCheck::kUnknown;
519 result->value_prepared = true;
520 }
521 return is_valid;
522 }
523 void Prev() override {
524 PERF_COUNTER_ADD(prev_on_memtable_count, 1);
525 assert(Valid());
526 if (paranoid_memory_checks_) {
527 status_ = iter_->PrevAndValidate(allow_data_in_error);
528 } else {
529 iter_->Prev();
530 }
531 valid_ = iter_->Valid();
532 VerifyEntryChecksum();
533 }
534 Slice key() const override {
535 assert(Valid());
536 return GetLengthPrefixedSlice(iter_->key());
537 }
538
539 uint64_t write_unix_time() const override {
540 assert(Valid());
541 ParsedInternalKey pikey;
542 Status s = ParseInternalKey(key(), &pikey, /*log_err_key=*/false);
543 if (!s.ok()) {
544 return std::numeric_limits<uint64_t>::max();
545 } else if (kTypeValuePreferredSeqno == pikey.type) {
546 return ParsePackedValueForWriteTime(value());
547 } else if (!seqno_to_time_mapping_ || seqno_to_time_mapping_->Empty()) {
548 return std::numeric_limits<uint64_t>::max();
549 }
550 return seqno_to_time_mapping_->GetProximalTimeBeforeSeqno(pikey.sequence);
551 }
552
553 Slice value() const override {
554 assert(Valid());
555 Slice key_slice = GetLengthPrefixedSlice(iter_->key());
556 return GetLengthPrefixedSlice(key_slice.data() + key_slice.size());
557 }
558
559 Status status() const override { return status_; }
560
561 bool IsKeyPinned() const override {
562 // memtable data is always pinned
563 return true;
564 }
565
566 bool IsValuePinned() const override {
567 // memtable value is always pinned, except if we allow inplace update.
568 return value_pinned_;
569 }
570
571 private:
572 DynamicBloom* bloom_;
573 const SliceTransform* const prefix_extractor_;
574 const MemTable::KeyComparator comparator_;
575 MemTableRep::Iterator* iter_;
576 // The seqno to time mapping is owned by the SuperVersion.
577 UnownedPtr<const SeqnoToTimeMapping> seqno_to_time_mapping_;
578 Status status_;
579 Logger* logger_;
580 size_t ts_sz_;
581 uint32_t protection_bytes_per_key_;
582 bool valid_;
583 bool value_pinned_;
584 bool arena_mode_;
585 const bool paranoid_memory_checks_;
586 const bool allow_data_in_error;
587
588 void VerifyEntryChecksum() {
589 if (protection_bytes_per_key_ > 0 && Valid()) {
590 status_ = MemTable::VerifyEntryChecksum(iter_->key(),
591 protection_bytes_per_key_);
592 if (!status_.ok()) {
593 ROCKS_LOG_ERROR(logger_, "In MemtableIterator: %s", status_.getState());
594 }
595 }
596 }
597};

614 TimestampStrippingIterator(

850Status MemTable::VerifyEncodedEntry(Slice encoded,
851 const ProtectionInfoKVOS64& kv_prot_info) {
852 uint32_t ikey_len = 0;
853 if (!GetVarint32(&encoded, &ikey_len)) {
854 return Status::Corruption("Unable to parse internal key length");
855 }
856 if (ikey_len < 8 + ts_sz_) {
857 return Status::Corruption("Internal key length too short");
858 }
859 if (ikey_len > encoded.size()) {
860 return Status::Corruption("Internal key length too long");
861 }
862 uint32_t value_len = 0;
863 const size_t user_key_len = ikey_len - 8;
864 Slice key(encoded.data(), user_key_len);
865 encoded.remove_prefix(user_key_len);
866
867 uint64_t packed = DecodeFixed64(encoded.data());
868 ValueType value_type = kMaxValue;
869 SequenceNumber sequence_number = kMaxSequenceNumber;
870 UnPackSequenceAndType(packed, &sequence_number, &value_type);
871 encoded.remove_prefix(8);
872
873 if (!GetVarint32(&encoded, &value_len)) {
874 return Status::Corruption("Unable to parse value length");
875 }
876 if (value_len < encoded.size()) {
877 return Status::Corruption("Value length too short");
878 }
879 if (value_len > encoded.size()) {
880 return Status::Corruption("Value length too long");
881 }
882 Slice value(encoded.data(), value_len);
883
884 return kv_prot_info.StripS(sequence_number)
885 .StripKVO(key, value, value_type)
886 .GetStatus();
887}

909Status MemTable::Add(SequenceNumber s, ValueType type,
910 const Slice& key, /* user key */
911 const Slice& value,
912 const ProtectionInfoKVOS64* kv_prot_info,
913 bool allow_concurrent,
914 MemTablePostProcessInfo* post_process_info, void** hint) {
915 // Format of an entry is concatenation of:
916 // key_size : varint32 of internal_key.size()
917 // key bytes : char[internal_key.size()]
918 // value_size : varint32 of value.size()
919 // value bytes : char[value.size()]
920 // checksum : char[moptions_.protection_bytes_per_key]
921 uint32_t key_size = static_cast<uint32_t>(key.size());
922 uint32_t val_size = static_cast<uint32_t>(value.size());
923 uint32_t internal_key_size = key_size + 8;
924 const uint32_t encoded_len = VarintLength(internal_key_size) +
925 internal_key_size + VarintLength(val_size) +
926 val_size + moptions_.protection_bytes_per_key;
927 char* buf = nullptr;
928 std::unique_ptr<MemTableRep>& table =
929 type == kTypeRangeDeletion ? range_del_table_ : table_;
930 KeyHandle handle = table->Allocate(encoded_len, &buf);
931
932 char* p = EncodeVarint32(buf, internal_key_size);
933 memcpy(p, key.data(), key_size);
934 Slice key_slice(p, key_size);
935 p += key_size;
936 uint64_t packed = PackSequenceAndType(s, type);
937 EncodeFixed64(p, packed);
938 p += 8;
939 p = EncodeVarint32(p, val_size);
940 memcpy(p, value.data(), val_size);
941 assert((unsigned)(p + val_size - buf + moptions_.protection_bytes_per_key) ==
942 (unsigned)encoded_len);
943
944 UpdateEntryChecksum(kv_prot_info, key, value, type, s,
945 buf + encoded_len - moptions_.protection_bytes_per_key);
946 Slice encoded(buf, encoded_len - moptions_.protection_bytes_per_key);
947 if (kv_prot_info != nullptr) {
948 TEST_SYNC_POINT_CALLBACK("MemTable::Add:Encoded", &encoded);
949 Status status = VerifyEncodedEntry(encoded, *kv_prot_info);
950 if (!status.ok()) {
951 return status;
952 }
953 }
954
955 Slice key_without_ts = StripTimestampFromUserKey(key, ts_sz_);
956
957 if (!allow_concurrent) {
958 // Extract prefix for insert with hint. Hints are for point key table
959 // (`table_`) only, not `range_del_table_`.
960 if (table == table_ && insert_with_hint_prefix_extractor_ != nullptr &&
961 insert_with_hint_prefix_extractor_->InDomain(key_slice)) {
962 Slice prefix = insert_with_hint_prefix_extractor_->Transform(key_slice);
963 bool res = table->InsertKeyWithHint(handle, &insert_hints_[prefix]);
964 if (UNLIKELY(!res)) {
965 return Status::TryAgain("key+seq exists");
966 }
967 } else {
968 bool res = table->InsertKey(handle);
969 if (UNLIKELY(!res)) {
970 return Status::TryAgain("key+seq exists");
971 }
972 }
973
974 // this is a bit ugly, but is the way to avoid locked instructions
975 // when incrementing an atomic
976 num_entries_.store(num_entries_.load(std::memory_order_relaxed) + 1,
977 std::memory_order_relaxed);
978 data_size_.store(data_size_.load(std::memory_order_relaxed) + encoded_len,
979 std::memory_order_relaxed);
980 if (type == kTypeDeletion || type == kTypeSingleDeletion ||
981 type == kTypeDeletionWithTimestamp) {
982 num_deletes_.store(num_deletes_.load(std::memory_order_relaxed) + 1,
983 std::memory_order_relaxed);
984 } else if (type == kTypeRangeDeletion) {
985 uint64_t val = num_range_deletes_.load(std::memory_order_relaxed) + 1;
986 num_range_deletes_.store(val, std::memory_order_relaxed);
987 }
988
989 if (bloom_filter_ && prefix_extractor_ &&
990 prefix_extractor_->InDomain(key_without_ts)) {
991 bloom_filter_->Add(prefix_extractor_->Transform(key_without_ts));
992 }
993 if (bloom_filter_ && moptions_.memtable_whole_key_filtering) {
994 bloom_filter_->Add(key_without_ts);
995 }
996
997 // The first sequence number inserted into the memtable
998 assert(first_seqno_ == 0 || s >= first_seqno_);
999 if (first_seqno_ == 0) {
1000 first_seqno_.store(s, std::memory_order_relaxed);
1001
1002 if (earliest_seqno_ == kMaxSequenceNumber) {
1003 earliest_seqno_.store(GetFirstSequenceNumber(),
1004 std::memory_order_relaxed);
1005 }
1006 assert(first_seqno_.load() >= earliest_seqno_.load());
1007 }
1008 assert(post_process_info == nullptr);
1009 // TODO(yuzhangyu): support updating newest UDT for when `allow_concurrent`
1010 // is true.
1011 MaybeUpdateNewestUDT(key_slice);
1012 UpdateFlushState();
1013 } else {
1014 bool res = (hint == nullptr)
1015 ? table->InsertKeyConcurrently(handle)
1016 : table->InsertKeyWithHintConcurrently(handle, hint);
1017 if (UNLIKELY(!res)) {
1018 return Status::TryAgain("key+seq exists");
1019 }
1020
1021 assert(post_process_info != nullptr);
1022 post_process_info->num_entries++;
1023 post_process_info->data_size += encoded_len;
1024 if (type == kTypeDeletion) {
1025 post_process_info->num_deletes++;
1026 }
1027
1028 if (bloom_filter_ && prefix_extractor_ &&
1029 prefix_extractor_->InDomain(key_without_ts)) {
1030 bloom_filter_->AddConcurrently(
1031 prefix_extractor_->Transform(key_without_ts));
1032 }
1033 if (bloom_filter_ && moptions_.memtable_whole_key_filtering) {
1034 bloom_filter_->AddConcurrently(key_without_ts);
1035 }
1036
1037 // atomically update first_seqno_ and earliest_seqno_.
1038 uint64_t cur_seq_num = first_seqno_.load(std::memory_order_relaxed);
1039 while ((cur_seq_num == 0 || s < cur_seq_num) &&
1040 !first_seqno_.compare_exchange_weak(cur_seq_num, s)) {
1041 }
1042 uint64_t cur_earliest_seqno =
1043 earliest_seqno_.load(std::memory_order_relaxed);
1044 while (
1045 (cur_earliest_seqno == kMaxSequenceNumber || s < cur_earliest_seqno) &&
1046 !earliest_seqno_.compare_exchange_weak(cur_earliest_seqno, s)) {
1047 }
1048 }
1049 if (type == kTypeRangeDeletion) {
1050 auto new_cache = std::make_shared<FragmentedRangeTombstoneListCache>();
1051 size_t size = cached_range_tombstone_.Size();
1052 if (allow_concurrent) {
1053 post_process_info->num_range_deletes++;
1054 range_del_mutex_.lock();
1055 }
1056 for (size_t i = 0; i < size; ++i) {
1057 std::shared_ptr<FragmentedRangeTombstoneListCache>* local_cache_ref_ptr =
1058 cached_range_tombstone_.AccessAtCore(i);
1059 auto new_local_cache_ref = std::make_shared<
1060 const std::shared_ptr<FragmentedRangeTombstoneListCache>>(new_cache);
1061 // It is okay for some reader to load old cache during invalidation as
1062 // the new sequence number is not published yet.
1063 // Each core will have a shared_ptr to a shared_ptr to the cached
1064 // fragmented range tombstones, so that ref count is maintianed locally
1065 // per-core using the per-core shared_ptr.
1066 std::atomic_store_explicit(
1067 local_cache_ref_ptr,
1068 std::shared_ptr<FragmentedRangeTombstoneListCache>(
1069 new_local_cache_ref, new_cache.get()),
1070 std::memory_order_relaxed);
1071 }
1072
1073 if (allow_concurrent) {
1074 range_del_mutex_.unlock();
1075 }
1076 is_range_del_table_empty_.store(false, std::memory_order_relaxed);
1077 }
1078 UpdateOldestKeyTime();
1079
1080 TEST_SYNC_POINT_CALLBACK("MemTable::Add:BeforeReturn:Encoded", &encoded);
1081 return Status::OK();
1082}

1120static bool SaveValue(void* arg, const char* entry) {
1121 Saver* s = static_cast<Saver*>(arg);
1122 assert(s != nullptr);
1123 assert(!s->value || !s->columns);
1124 assert(!*(s->found_final_value));
1125 assert(s->status->ok() || s->status->IsMergeInProgress());
1126
1127 MergeContext* merge_context = s->merge_context;
1128 SequenceNumber max_covering_tombstone_seq = s->max_covering_tombstone_seq;
1129 const MergeOperator* merge_operator = s->merge_operator;
1130
1131 assert(merge_context != nullptr);
1132
1133 // Refer to comments under MemTable::Add() for entry format.
1134 // Check that it belongs to same user key.
1135 uint32_t key_length = 0;
1136 const char* key_ptr = GetVarint32Ptr(entry, entry + 5, &key_length);
1137 assert(key_length >= 8);
1138 Slice user_key_slice = Slice(key_ptr, key_length - 8);
1139 const Comparator* user_comparator =
1140 s->mem->GetInternalKeyComparator().user_comparator();
1141 size_t ts_sz = user_comparator->timestamp_size();
1142 if (ts_sz && s->timestamp && max_covering_tombstone_seq > 0) {
1143 // timestamp should already be set to range tombstone timestamp
1144 assert(s->timestamp->size() == ts_sz);
1145 }
1146 if (user_comparator->EqualWithoutTimestamp(user_key_slice,
1147 s->key->user_key())) {
1148 // Correct user key
1149 TEST_SYNC_POINT_CALLBACK("Memtable::SaveValue:Found:entry", &entry);
1150 std::optional<ReadLock> read_lock;
1151 if (s->inplace_update_support) {
1152 read_lock.emplace(s->mem->GetLock(s->key->user_key()));
1153 }
1154
1155 if (s->protection_bytes_per_key > 0) {
1156 *(s->status) = MemTable::VerifyEntryChecksum(
1157 entry, s->protection_bytes_per_key, s->allow_data_in_errors);
1158 if (!s->status->ok()) {
1159 *(s->found_final_value) = true;
1160 ROCKS_LOG_ERROR(s->logger, "In SaveValue: %s", s->status->getState());
1161 // Memtable entry corrupted
1162 return false;
1163 }
1164 }
1165
1166 const uint64_t tag = DecodeFixed64(key_ptr + key_length - 8);
1167 ValueType type;
1168 SequenceNumber seq;
1169 UnPackSequenceAndType(tag, &seq, &type);
1170 // If the value is not in the snapshot, skip it
1171 if (!s->CheckCallback(seq)) {
1172 return true; // to continue to the next seq
1173 }
1174
1175 if (s->seq == kMaxSequenceNumber) {
1176 s->seq = seq;
1177 if (s->seq > max_covering_tombstone_seq) {
1178 if (ts_sz && s->timestamp != nullptr) {
1179 // `timestamp` was set to range tombstone's timestamp before
1180 // `SaveValue` is ever called. This key has a higher sequence number
1181 // than range tombstone, and is the key with the highest seqno across
1182 // all keys with this user_key, so we update timestamp here.
1183 Slice ts = ExtractTimestampFromUserKey(user_key_slice, ts_sz);
1184 s->timestamp->assign(ts.data(), ts_sz);
1185 }
1186 } else {
1187 s->seq = max_covering_tombstone_seq;
1188 }
1189 }
1190
1191 if (ts_sz > 0 && s->timestamp != nullptr) {
1192 if (!s->timestamp->empty()) {
1193 assert(ts_sz == s->timestamp->size());
1194 }
1195 // TODO optimize for smaller size ts
1196 const std::string kMaxTs(ts_sz, '\xff');
1197 if (s->timestamp->empty() ||
1198 user_comparator->CompareTimestamp(*(s->timestamp), kMaxTs) == 0) {
1199 Slice ts = ExtractTimestampFromUserKey(user_key_slice, ts_sz);
1200 s->timestamp->assign(ts.data(), ts_sz);
1201 }
1202 }
1203
1204 if ((type == kTypeValue || type == kTypeMerge || type == kTypeBlobIndex ||
1205 type == kTypeWideColumnEntity || type == kTypeDeletion ||
1206 type == kTypeSingleDeletion || type == kTypeDeletionWithTimestamp ||
1207 type == kTypeValuePreferredSeqno) &&
1208 max_covering_tombstone_seq > seq) {
1209 type = kTypeRangeDeletion;
1210 }
1211 switch (type) {
1212 case kTypeBlobIndex: {
1213 if (!s->do_merge) {
1214 *(s->status) = Status::NotSupported(
1215 "GetMergeOperands not supported by stacked BlobDB");
1216 *(s->found_final_value) = true;
1217 return false;
1218 }
1219
1220 if (*(s->merge_in_progress)) {
1221 *(s->status) = Status::NotSupported(
1222 "Merge operator not supported by stacked BlobDB");
1223 *(s->found_final_value) = true;
1224 return false;
1225 }
1226
1227 if (s->is_blob_index == nullptr) {
1228 ROCKS_LOG_ERROR(s->logger, "Encountered unexpected blob index.");
1229 *(s->status) = Status::NotSupported(
1230 "Encountered unexpected blob index. Please open DB with "
1231 "ROCKSDB_NAMESPACE::blob_db::BlobDB.");
1232 *(s->found_final_value) = true;
1233 return false;
1234 }
1235
1236 Slice v = GetLengthPrefixedSlice(key_ptr + key_length);
1237
1238 *(s->status) = Status::OK();
1239
1240 if (s->value) {
1241 s->value->assign(v.data(), v.size());
1242 } else if (s->columns) {
1243 s->columns->SetPlainValue(v);
1244 }
1245
1246 *(s->found_final_value) = true;
1247 *(s->is_blob_index) = true;
1248
1249 return false;
1250 }
1251 case kTypeValue:
1252 case kTypeValuePreferredSeqno: {
1253 Slice v = GetLengthPrefixedSlice(key_ptr + key_length);
1254 if (type == kTypeValuePreferredSeqno) {
1255 v = ParsePackedValueForValue(v);
1256 }
1257
1258 ReadOnlyMemTable::HandleTypeValue(
1259 s->key->user_key(), v, s->inplace_update_support == false,
1260 s->do_merge, *(s->merge_in_progress), merge_context,
1261 s->merge_operator, s->clock, s->statistics, s->logger, s->status,
1262 s->value, s->columns, s->is_blob_index);
1263 *(s->found_final_value) = true;
1264 return false;
1265 }
1266 case kTypeWideColumnEntity: {
1267 Slice v = GetLengthPrefixedSlice(key_ptr + key_length);
1268
1269 *(s->status) = Status::OK();
1270
1271 if (!s->do_merge) {
1272 // Preserve the value with the goal of returning it as part of
1273 // raw merge operands to the user
1274
1275 Slice value_of_default;
1276 *(s->status) = WideColumnSerialization::GetValueOfDefaultColumn(
1277 v, value_of_default);
1278
1279 if (s->status->ok()) {
1280 merge_context->PushOperand(
1281 value_of_default,
1282 s->inplace_update_support == false /* operand_pinned */);
1283 }
1284 } else if (*(s->merge_in_progress)) {
1285 assert(s->do_merge);
1286
1287 if (s->value || s->columns) {
1288 // `op_failure_scope` (an output parameter) is not provided (set
1289 // to nullptr) since a failure must be propagated regardless of
1290 // its value.
1291 *(s->status) = MergeHelper::TimedFullMerge(
1292 merge_operator, s->key->user_key(), MergeHelper::kWideBaseValue,
1293 v, merge_context->GetOperands(), s->logger, s->statistics,
1294 s->clock, /* update_num_ops_stats */ true,
1295 /* op_failure_scope */ nullptr, s->value, s->columns);
1296 }
1297 } else if (s->value) {
1298 Slice value_of_default;
1299 *(s->status) = WideColumnSerialization::GetValueOfDefaultColumn(
1300 v, value_of_default);
1301 if (s->status->ok()) {
1302 s->value->assign(value_of_default.data(), value_of_default.size());
1303 }
1304 } else if (s->columns) {
1305 *(s->status) = s->columns->SetWideColumnValue(v);
1306 }
1307
1308 *(s->found_final_value) = true;
1309
1310 if (s->is_blob_index != nullptr) {
1311 *(s->is_blob_index) = false;
1312 }
1313
1314 return false;
1315 }
1316 case kTypeDeletion:
1317 case kTypeDeletionWithTimestamp:
1318 case kTypeSingleDeletion:
1319 case kTypeRangeDeletion: {
1320 ReadOnlyMemTable::HandleTypeDeletion(
1321 s->key->user_key(), *(s->merge_in_progress), s->merge_context,
1322 s->merge_operator, s->clock, s->statistics, s->logger, s->status,
1323 s->value, s->columns);
1324 *(s->found_final_value) = true;
1325 return false;
1326 }
1327 case kTypeMerge: {
1328 if (!merge_operator) {
1329 *(s->status) = Status::InvalidArgument(
1330 "merge_operator is not properly initialized.");
1331 // Normally we continue the loop (return true) when we see a merge
1332 // operand. But in case of an error, we should stop the loop
1333 // immediately and pretend we have found the value to stop further
1334 // seek. Otherwise, the later call will override this error status.
1335 *(s->found_final_value) = true;
1336 return false;
1337 }
1338 Slice v = GetLengthPrefixedSlice(key_ptr + key_length);
1339 *(s->merge_in_progress) = true;
1340 merge_context->PushOperand(
1341 v, s->inplace_update_support == false /* operand_pinned */);
1342 PERF_COUNTER_ADD(internal_merge_point_lookup_count, 1);
1343
1344 if (s->do_merge && merge_operator->ShouldMerge(
1345 merge_context->GetOperandsDirectionBackward())) {
1346 if (s->value || s->columns) {
1347 // `op_failure_scope` (an output parameter) is not provided (set to
1348 // nullptr) since a failure must be propagated regardless of its
1349 // value.
1350 *(s->status) = MergeHelper::TimedFullMerge(
1351 merge_operator, s->key->user_key(), MergeHelper::kNoBaseValue,
1352 merge_context->GetOperands(), s->logger, s->statistics,
1353 s->clock, /* update_num_ops_stats */ true,
1354 /* op_failure_scope */ nullptr, s->value, s->columns);
1355 }
1356
1357 *(s->found_final_value) = true;
1358 return false;
1359 }
1360 if (merge_context->get_merge_operands_options != nullptr &&
1361 merge_context->get_merge_operands_options->continue_cb != nullptr &&
1362 !merge_context->get_merge_operands_options->continue_cb(v)) {
1363 // We were told not to continue.
1364 *(s->found_final_value) = true;
1365 return false;
1366 }
1367
1368 return true;
1369 }
1370 default: {
1371 std::string msg("Corrupted value not expected.");
1372 if (s->allow_data_in_errors) {
1373 msg.append("Unrecognized value type: " +
1374 std::to_string(static_cast<int>(type)) + ". ");
1375 msg.append("User key: " + user_key_slice.ToString(/*hex=*/true) +
1376 ". ");
1377 msg.append("seq: " + std::to_string(seq) + ".");
1378 }
1379 *(s->found_final_value) = true;
1380 *(s->status) = Status::Corruption(msg.c_str());
1381 return false;
1382 }
1383 }
1384 }
1385
1386 // s->state could be Corrupt, merge or notfound
1387 return false;
1388}

1390bool MemTable::Get(const LookupKey& key, std::string* value,
1391 PinnableWideColumns* columns, std::string* timestamp,
1392 Status* s, MergeContext* merge_context,
1393 SequenceNumber* max_covering_tombstone_seq,
1394 SequenceNumber* seq, const ReadOptions& read_opts,
1395 bool immutable_memtable, ReadCallback* callback,
1396 bool* is_blob_index, bool do_merge) {
1397 // The sequence number is updated synchronously in version_set.h
1398 if (IsEmpty()) {
1399 // Avoiding recording stats for speed.
1400 return false;
1401 }
1402
1403 PERF_TIMER_GUARD(get_from_memtable_time);
1404
1405 std::unique_ptr<FragmentedRangeTombstoneIterator> range_del_iter(
1406 NewRangeTombstoneIterator(read_opts,
1407 GetInternalKeySeqno(key.internal_key()),
1408 immutable_memtable));
1409 if (range_del_iter != nullptr) {
1410 SequenceNumber covering_seq =
1411 range_del_iter->MaxCoveringTombstoneSeqnum(key.user_key());
1412 if (covering_seq > *max_covering_tombstone_seq) {
1413 *max_covering_tombstone_seq = covering_seq;
1414 if (timestamp) {
1415 // Will be overwritten in SaveValue() if there is a point key with
1416 // a higher seqno.
1417 timestamp->assign(range_del_iter->timestamp().data(),
1418 range_del_iter->timestamp().size());
1419 }
1420 }
1421 }
1422
1423 bool found_final_value = false;
1424 bool merge_in_progress = s->IsMergeInProgress();
1425 bool may_contain = true;
1426 Slice user_key_without_ts = StripTimestampFromUserKey(key.user_key(), ts_sz_);
1427 bool bloom_checked = false;
1428 if (bloom_filter_) {
1429 // when both memtable_whole_key_filtering and prefix_extractor_ are set,
1430 // only do whole key filtering for Get() to save CPU
1431 if (moptions_.memtable_whole_key_filtering) {
1432 may_contain = bloom_filter_->MayContain(user_key_without_ts);
1433 bloom_checked = true;
1434 } else {
1435 assert(prefix_extractor_);
1436 if (prefix_extractor_->InDomain(user_key_without_ts)) {
1437 may_contain = bloom_filter_->MayContain(
1438 prefix_extractor_->Transform(user_key_without_ts));
1439 bloom_checked = true;
1440 }
1441 }
1442 }
1443
1444 if (bloom_filter_ && !may_contain) {
1445 // iter is null if prefix bloom says the key does not exist
1446 PERF_COUNTER_ADD(bloom_memtable_miss_count, 1);
1447 *seq = kMaxSequenceNumber;
1448 } else {
1449 if (bloom_checked) {
1450 PERF_COUNTER_ADD(bloom_memtable_hit_count, 1);
1451 }
1452 GetFromTable(key, *max_covering_tombstone_seq, do_merge, callback,
1453 is_blob_index, value, columns, timestamp, s, merge_context,
1454 seq, &found_final_value, &merge_in_progress);
1455 }
1456
1457 // No change to value, since we have not yet found a Put/Delete
1458 // Propagate corruption error
1459 if (!found_final_value && merge_in_progress) {
1460 if (s->ok()) {
1461 *s = Status::MergeInProgress();
1462 } else {
1463 assert(s->IsMergeInProgress());
1464 }
1465 }
1466 PERF_COUNTER_ADD(get_from_memtable_count, 1);
1467 return found_final_value;
1468}

1516void MemTable::MultiGet(const ReadOptions& read_options, MultiGetRange* range,
1517 ReadCallback* callback, bool immutable_memtable) {
1518 // The sequence number is updated synchronously in version_set.h
1519 if (IsEmpty()) {
1520 // Avoiding recording stats for speed.
1521 return;
1522 }
1523 PERF_TIMER_GUARD(get_from_memtable_time);
1524
1525 // For now, memtable Bloom filter is effectively disabled if there are any
1526 // range tombstones. This is the simplest way to ensure range tombstones are
1527 // handled. TODO: allow Bloom checks where max_covering_tombstone_seq==0
1528 bool no_range_del = read_options.ignore_range_deletions ||
1529 is_range_del_table_empty_.load(std::memory_order_relaxed);
1530 MultiGetRange temp_range(*range, range->begin(), range->end());
1531 if (bloom_filter_ && no_range_del) {
1532 bool whole_key =
1533 !prefix_extractor_ || moptions_.memtable_whole_key_filtering;
1534 std::array<Slice, MultiGetContext::MAX_BATCH_SIZE> bloom_keys;
1535 std::array<bool, MultiGetContext::MAX_BATCH_SIZE> may_match;
1536 std::array<size_t, MultiGetContext::MAX_BATCH_SIZE> range_indexes;
1537 int num_keys = 0;
1538 for (auto iter = temp_range.begin(); iter != temp_range.end(); ++iter) {
1539 if (whole_key) {
1540 bloom_keys[num_keys] = iter->ukey_without_ts;
1541 range_indexes[num_keys++] = iter.index();
1542 } else if (prefix_extractor_->InDomain(iter->ukey_without_ts)) {
1543 bloom_keys[num_keys] =
1544 prefix_extractor_->Transform(iter->ukey_without_ts);
1545 range_indexes[num_keys++] = iter.index();
1546 }
1547 }
1548 bloom_filter_->MayContain(num_keys, bloom_keys.data(), may_match.data());
1549 for (int i = 0; i < num_keys; ++i) {
1550 if (!may_match[i]) {
1551 temp_range.SkipIndex(range_indexes[i]);
1552 PERF_COUNTER_ADD(bloom_memtable_miss_count, 1);
1553 } else {
1554 PERF_COUNTER_ADD(bloom_memtable_hit_count, 1);
1555 }
1556 }
1557 }
1558 for (auto iter = temp_range.begin(); iter != temp_range.end(); ++iter) {
1559 bool found_final_value{false};
1560 bool merge_in_progress = iter->s->IsMergeInProgress();
1561 if (!no_range_del) {
1562 std::unique_ptr<FragmentedRangeTombstoneIterator> range_del_iter(
1563 NewRangeTombstoneIteratorInternal(
1564 read_options, GetInternalKeySeqno(iter->lkey->internal_key()),
1565 immutable_memtable));
1566 SequenceNumber covering_seq =
1567 range_del_iter->MaxCoveringTombstoneSeqnum(iter->lkey->user_key());
1568 if (covering_seq > iter->max_covering_tombstone_seq) {
1569 iter->max_covering_tombstone_seq = covering_seq;
1570 if (iter->timestamp) {
1571 // Will be overwritten in SaveValue() if there is a point key with
1572 // a higher seqno.
1573 iter->timestamp->assign(range_del_iter->timestamp().data(),
1574 range_del_iter->timestamp().size());
1575 }
1576 }
1577 }
1578 SequenceNumber dummy_seq;
1579 GetFromTable(*(iter->lkey), iter->max_covering_tombstone_seq, true,
1580 callback, &iter->is_blob_index,
1581 iter->value ? iter->value->GetSelf() : nullptr, iter->columns,
1582 iter->timestamp, iter->s, &(iter->merge_context), &dummy_seq,
1583 &found_final_value, &merge_in_progress);
1584
1585 if (!found_final_value && merge_in_progress) {
1586 if (iter->s->ok()) {
1587 *(iter->s) = Status::MergeInProgress();
1588 } else {
1589 assert(iter->s->IsMergeInProgress());
1590 }
1591 }
1592
1593 if (found_final_value ||
1594 (!iter->s->ok() && !iter->s->IsMergeInProgress())) {
1595 // `found_final_value` should be set if an error/corruption occurs.
1596 // The check on iter->s is just there in case GetFromTable() did not
1597 // set `found_final_value` properly.
1598 assert(found_final_value);
1599 if (iter->value) {
1600 iter->value->PinSelf();
1601 range->AddValueSize(iter->value->size());
1602 } else {
1603 assert(iter->columns);
1604 range->AddValueSize(iter->columns->serialized_size());
1605 }
1606
1607 range->MarkKeyDone(iter);
1608 RecordTick(moptions_.statistics, MEMTABLE_HIT);
1609 if (range->GetValueSize() > read_options.value_size_soft_limit) {
1610 // Set all remaining keys in range to Abort
1611 for (auto range_iter = range->begin(); range_iter != range->end();
1612 ++range_iter) {
1613 range->MarkKeyDone(range_iter);
1614 *(range_iter->s) = Status::Aborted();
1615 }
1616 break;
1617 }
1618 }
1619 }
1620 PERF_COUNTER_ADD(get_from_memtable_count, 1);
1621}

1623Status MemTable::Update(SequenceNumber seq, ValueType value_type,
1624 const Slice& key, const Slice& value,
1625 const ProtectionInfoKVOS64* kv_prot_info) {
1626 LookupKey lkey(key, seq);
1627 Slice mem_key = lkey.memtable_key();
1628
1629 std::unique_ptr<MemTableRep::Iterator> iter(
1630 table_->GetDynamicPrefixIterator());
1631 iter->Seek(lkey.internal_key(), mem_key.data());
1632
1633 if (iter->Valid()) {
1634 // Refer to comments under MemTable::Add() for entry format.
1635 // Check that it belongs to same user key. We do not check the
1636 // sequence number since the Seek() call above should have skipped
1637 // all entries with overly large sequence numbers.
1638 const char* entry = iter->key();
1639 uint32_t key_length = 0;
1640 const char* key_ptr = GetVarint32Ptr(entry, entry + 5, &key_length);
1641 if (comparator_.comparator.user_comparator()->Equal(
1642 Slice(key_ptr, key_length - 8), lkey.user_key())) {
1643 // Correct user key
1644 const uint64_t tag = DecodeFixed64(key_ptr + key_length - 8);
1645 ValueType type;
1646 SequenceNumber existing_seq;
1647 UnPackSequenceAndType(tag, &existing_seq, &type);
1648 assert(existing_seq != seq);
1649 if (type == value_type) {
1650 Slice prev_value = GetLengthPrefixedSlice(key_ptr + key_length);
1651 uint32_t prev_size = static_cast<uint32_t>(prev_value.size());
1652 uint32_t new_size = static_cast<uint32_t>(value.size());
1653
1654 // Update value, if new value size <= previous value size
1655 if (new_size <= prev_size) {
1656 WriteLock wl(GetLock(lkey.user_key()));
1657 char* p =
1658 EncodeVarint32(const_cast<char*>(key_ptr) + key_length, new_size);
1659 memcpy(p, value.data(), value.size());
1660 assert((unsigned)((p + value.size()) - entry) ==
1661 (unsigned)(VarintLength(key_length) + key_length +
1662 VarintLength(value.size()) + value.size()));
1663 RecordTick(moptions_.statistics, NUMBER_KEYS_UPDATED);
1664 if (kv_prot_info != nullptr) {
1665 ProtectionInfoKVOS64 updated_kv_prot_info(*kv_prot_info);
1666 // `seq` is swallowed and `existing_seq` prevails.
1667 updated_kv_prot_info.UpdateS(seq, existing_seq);
1668 UpdateEntryChecksum(&updated_kv_prot_info, key, value, type,
1669 existing_seq, p + value.size());
1670 Slice encoded(entry, p + value.size() - entry);
1671 return VerifyEncodedEntry(encoded, updated_kv_prot_info);
1672 } else {
1673 UpdateEntryChecksum(nullptr, key, value, type, existing_seq,
1674 p + value.size());
1675 }
1676 return Status::OK();
1677 }
1678 }
1679 }
1680 }
1681
1682 // The latest value is not value_type or key doesn't exist
1683 return Add(seq, value_type, key, value, kv_prot_info);
1684}

1686Status MemTable::UpdateCallback(SequenceNumber seq, const Slice& key,
1687 const Slice& delta,
1688 const ProtectionInfoKVOS64* kv_prot_info) {
1689 LookupKey lkey(key, seq);
1690 Slice memkey = lkey.memtable_key();
1691
1692 std::unique_ptr<MemTableRep::Iterator> iter(
1693 table_->GetDynamicPrefixIterator());
1694 iter->Seek(lkey.internal_key(), memkey.data());
1695
1696 if (iter->Valid()) {
1697 // Refer to comments under MemTable::Add() for entry format.
1698 // Check that it belongs to same user key. We do not check the
1699 // sequence number since the Seek() call above should have skipped
1700 // all entries with overly large sequence numbers.
1701 const char* entry = iter->key();
1702 uint32_t key_length = 0;
1703 const char* key_ptr = GetVarint32Ptr(entry, entry + 5, &key_length);
1704 if (comparator_.comparator.user_comparator()->Equal(
1705 Slice(key_ptr, key_length - 8), lkey.user_key())) {
1706 // Correct user key
1707 const uint64_t tag = DecodeFixed64(key_ptr + key_length - 8);
1708 ValueType type;
1709 uint64_t existing_seq;
1710 UnPackSequenceAndType(tag, &existing_seq, &type);
1711 if (type == kTypeValue) {
1712 Slice prev_value = GetLengthPrefixedSlice(key_ptr + key_length);
1713 uint32_t prev_size = static_cast<uint32_t>(prev_value.size());
1714
1715 char* prev_buffer = const_cast<char*>(prev_value.data());
1716 uint32_t new_prev_size = prev_size;
1717
1718 std::string str_value;
1719 WriteLock wl(GetLock(lkey.user_key()));
1720 auto status = moptions_.inplace_callback(prev_buffer, &new_prev_size,
1721 delta, &str_value);
1722 if (status == UpdateStatus::UPDATED_INPLACE) {
1723 // Value already updated by callback.
1724 assert(new_prev_size <= prev_size);
1725 if (new_prev_size < prev_size) {
1726 // overwrite the new prev_size
1727 char* p = EncodeVarint32(const_cast<char*>(key_ptr) + key_length,
1728 new_prev_size);
1729 if (VarintLength(new_prev_size) < VarintLength(prev_size)) {
1730 // shift the value buffer as well.
1731 memcpy(p, prev_buffer, new_prev_size);
1732 prev_buffer = p;
1733 }
1734 }
1735 RecordTick(moptions_.statistics, NUMBER_KEYS_UPDATED);
1736 UpdateFlushState();
1737 Slice new_value(prev_buffer, new_prev_size);
1738 if (kv_prot_info != nullptr) {
1739 ProtectionInfoKVOS64 updated_kv_prot_info(*kv_prot_info);
1740 // `seq` is swallowed and `existing_seq` prevails.
1741 updated_kv_prot_info.UpdateS(seq, existing_seq);
1742 updated_kv_prot_info.UpdateV(delta, new_value);
1743 Slice encoded(entry, prev_buffer + new_prev_size - entry);
1744 UpdateEntryChecksum(&updated_kv_prot_info, key, new_value, type,
1745 existing_seq, prev_buffer + new_prev_size);
1746 return VerifyEncodedEntry(encoded, updated_kv_prot_info);
1747 } else {
1748 UpdateEntryChecksum(nullptr, key, new_value, type, existing_seq,
1749 prev_buffer + new_prev_size);
1750 }
1751 return Status::OK();
1752 } else if (status == UpdateStatus::UPDATED) {
1753 Status s;
1754 if (kv_prot_info != nullptr) {
1755 ProtectionInfoKVOS64 updated_kv_prot_info(*kv_prot_info);
1756 updated_kv_prot_info.UpdateV(delta, str_value);
1757 s = Add(seq, kTypeValue, key, Slice(str_value),
1758 &updated_kv_prot_info);
1759 } else {
1760 s = Add(seq, kTypeValue, key, Slice(str_value),
1761 nullptr /* kv_prot_info */);
1762 }
1763 RecordTick(moptions_.statistics, NUMBER_KEYS_WRITTEN);
1764 UpdateFlushState();
1765 return s;
1766 } else if (status == UpdateStatus::UPDATE_FAILED) {
1767 // `UPDATE_FAILED` is named incorrectly. It indicates no update
1768 // happened. It does not indicate a failure happened.
1769 UpdateFlushState();
1770 return Status::OK();
1771 }
1772 }
1773 }
1774 }
1775 // The latest value is not `kTypeValue` or key doesn't exist
1776 return Status::NotFound();
1777}

1779size_t MemTable::CountSuccessiveMergeEntries(const LookupKey& key,
1780 size_t limit) {
1781 Slice memkey = key.memtable_key();
1782
1783 // A total ordered iterator is costly for some memtablerep (prefix aware
1784 // reps). By passing in the user key, we allow efficient iterator creation.
1785 // The iterator only needs to be ordered within the same user key.
1786 std::unique_ptr<MemTableRep::Iterator> iter(
1787 table_->GetDynamicPrefixIterator());
1788 iter->Seek(key.internal_key(), memkey.data());
1789
1790 size_t num_successive_merges = 0;
1791
1792 for (; iter->Valid() && num_successive_merges < limit; iter->Next()) {
1793 const char* entry = iter->key();
1794 uint32_t key_length = 0;
1795 const char* iter_key_ptr = GetVarint32Ptr(entry, entry + 5, &key_length);
1796 if (!comparator_.comparator.user_comparator()->Equal(
1797 Slice(iter_key_ptr, key_length - 8), key.user_key())) {
1798 break;
1799 }
1800
1801 const uint64_t tag = DecodeFixed64(iter_key_ptr + key_length - 8);
1802 ValueType type;
1803 uint64_t unused;
1804 UnPackSequenceAndType(tag, &unused, &type);
1805 if (type != kTypeMerge) {
1806 break;
