27 Status GetFirstSstPath(std::string* first_sst_path) {
28 assert(first_sst_path != nullptr);
29 first_sst_path->clear();
30 uint64_t manifest_size;
31 std::vector<std::string> files;
32 Status s = db_->GetLiveFiles(files, &manifest_size);
33 if (s.ok()) {
34 auto sst_iter =
35 std::find_if(files.begin(), files.end(), [](const std::string& file) {
36 uint64_t number;
37 FileType type;
38 bool ok = ParseFileName(file, &number, &type);
39 return ok && type == kTableFile;
40 });
41 *first_sst_path = sst_iter == files.end() ? "" : dbname_ + *sst_iter;
42 }
43 return s;
44 }
45
46 void ReopenWithSstIdVerify() {
47 std::atomic_int verify_passed{0};
48 SyncPoint::GetInstance()->SetCallBack(
49 "BlockBasedTable::Open::PassedVerifyUniqueId", [&](void* arg) {
50 // override job status
51 auto id = static_cast<UniqueId64x2*>(arg);
52 assert(*id != kNullUniqueId64x2);
53 verify_passed++;
54 });
55 SyncPoint::GetInstance()->EnableProcessing();
56 auto options = CurrentOptions();
57 options.verify_sst_unique_id_in_manifest = true;
58 Reopen(options);
59
60 ASSERT_GT(verify_passed, 0);
61 SyncPoint::GetInstance()->DisableProcessing();
62 }

345TEST_P(RepairTestWithTimestamp, UnflushedSst) {
346 Destroy(last_options_);
347
348 bool paranoid_file_checks = std::get<0>(GetParam());
349 bool persist_udt = test::ShouldPersistUDT(std::get<1>(GetParam()));
350 std::string smallest_ukey_without_ts = "bar";
351 std::string largest_ukey_without_ts = "foo";
352 Options options = CurrentOptions();
353 options.env = env_;
354 options.create_if_missing = true;
355 std::string min_ts;
356 std::string write_ts;
357 PutFixed64(&min_ts, 0);
358 PutFixed64(&write_ts, 1);
359 options.comparator = test::BytewiseComparatorWithU64TsWrapper();
360 options.persist_user_defined_timestamps = persist_udt;
361 if (!persist_udt) {
362 options.allow_concurrent_memtable_write = false;
363 }
364 options.paranoid_file_checks = paranoid_file_checks;
365
366 ColumnFamilyOptions cf_options(options);
367 std::vector<ColumnFamilyDescriptor> column_families;
368 column_families.emplace_back(kDefaultColumnFamilyName, cf_options);
369
370 ASSERT_OK(DB::Open(options, dbname_, column_families, &handles_, &db_));
371
372 ASSERT_OK(Put(smallest_ukey_without_ts, write_ts,
373 smallest_ukey_without_ts + ":val"));
374 ASSERT_OK(
375 Put(largest_ukey_without_ts, write_ts, largest_ukey_without_ts + ":val"));
376 VectorLogPtr wal_files;
377 ASSERT_OK(dbfull()->GetSortedWalFiles(wal_files));
378 ASSERT_EQ(wal_files.size(), 1);
379 {
380 uint64_t total_ssts_size;
381 std::unordered_map<std::string, uint64_t> sst_files;
382 ASSERT_OK(GetAllDataFiles(kTableFile, &sst_files, &total_ssts_size));
383 ASSERT_EQ(total_ssts_size, 0);
384 }
385 // Need to get path before Close() deletes db_, but delete it after Close() to
386 // ensure Close() didn't change the manifest.
387 std::string manifest_path =
388 DescriptorFileName(dbname_, dbfull()->TEST_Current_Manifest_FileNo());
389
390 Close();
391 ASSERT_OK(env_->FileExists(manifest_path));
392 ASSERT_OK(env_->DeleteFile(manifest_path));
393 ASSERT_OK(RepairDB(dbname_, options));
394 ASSERT_OK(DB::Open(options, dbname_, column_families, &handles_, &db_));
395
396 ASSERT_OK(dbfull()->GetSortedWalFiles(wal_files));
397 ASSERT_EQ(wal_files.size(), 0);
398 {
399 uint64_t total_ssts_size;
400 std::unordered_map<std::string, uint64_t> sst_files;
401 ASSERT_OK(GetAllDataFiles(kTableFile, &sst_files, &total_ssts_size));
402 ASSERT_GT(total_ssts_size, 0);
403 }
404
405 // Check file boundaries are correct for different
406 // `persist_user_defined_timestamps` option values.
407 if (persist_udt) {
408 CheckFileBoundaries(smallest_ukey_without_ts + write_ts,
409 largest_ukey_without_ts + write_ts);
410 } else {
411 CheckFileBoundaries(smallest_ukey_without_ts + min_ts,
412 largest_ukey_without_ts + min_ts);
413 }
414
415 ReadOptions read_opts;
416 Slice read_ts_slice = write_ts;
417 read_opts.timestamp = &read_ts_slice;
418 if (persist_udt) {
419 CheckGet(read_opts, smallest_ukey_without_ts,
420 smallest_ukey_without_ts + ":val", write_ts);
421 CheckGet(read_opts, largest_ukey_without_ts,
422 largest_ukey_without_ts + ":val", write_ts);
423 } else {
424 // TODO(yuzhangyu): currently when `persist_user_defined_timestamps` is
425 // false, ts is unconditionally stripped during flush.
426 // When `full_history_ts_low` is set and respected during flush.
427 // We should prohibit reading below `full_history_ts_low` all together.
428 CheckGet(read_opts, smallest_ukey_without_ts,
429 smallest_ukey_without_ts + ":val", min_ts);
430 CheckGet(read_opts, largest_ukey_without_ts,
431 largest_ukey_without_ts + ":val", min_ts);
432 }
433}

524TEST_F(RepairTest, RepairColumnFamilyOptions) {
525 // Verify repair logic uses correct ColumnFamilyOptions when repairing a
526 // database with different options for column families.
527 const int kNumCfs = 2;
528 const int kEntriesPerCf = 2;
529
530 Options opts(CurrentOptions()), rev_opts(CurrentOptions());
531 opts.comparator = BytewiseComparator();
532 rev_opts.comparator = ReverseBytewiseComparator();
533
534 DestroyAndReopen(opts);
535 CreateColumnFamilies({"reverse"}, rev_opts);
536 ReopenWithColumnFamilies({"default", "reverse"},
537 std::vector<Options>{opts, rev_opts});
538 for (int i = 0; i < kNumCfs; ++i) {
539 for (int j = 0; j < kEntriesPerCf; ++j) {
540 ASSERT_OK(Put(i, "key" + std::to_string(j), "val" + std::to_string(j)));
541 if (i == kNumCfs - 1 && j == kEntriesPerCf - 1) {
542 // Leave one unflushed so we can verify RepairDB's flush logic
543 continue;
544 }
545 ASSERT_OK(Flush(i));
546 }
547 }
548 Close();
549
550 // RepairDB() records the comparator in the manifest, and DB::Open would fail
551 // if a different comparator were used.
552 ASSERT_OK(RepairDB(dbname_, opts, {{"default", opts}, {"reverse", rev_opts}},
553 opts /* unknown_cf_opts */));
554 ASSERT_OK(TryReopenWithColumnFamilies({"default", "reverse"},
555 std::vector<Options>{opts, rev_opts}));
556 for (int i = 0; i < kNumCfs; ++i) {
557 for (int j = 0; j < kEntriesPerCf; ++j) {
558 ASSERT_EQ(Get(i, "key" + std::to_string(j)), "val" + std::to_string(j));
559 }
560 }
561
562 // Examine table properties to verify RepairDB() used the right options when
563 // converting WAL->SST
564 TablePropertiesCollection fname_to_props;
565 ASSERT_OK(db_->GetPropertiesOfAllTables(handles_[1], &fname_to_props));
566 ASSERT_EQ(fname_to_props.size(), 2U);
567 for (const auto& fname_and_props : fname_to_props) {
568 std::string comparator_name(rev_opts.comparator->Name());
569 ASSERT_EQ(comparator_name, fname_and_props.second->comparator_name);
570 }
571 Close();
572
573 // Also check comparator when it's provided via "unknown" CF options
574 ASSERT_OK(RepairDB(dbname_, opts, {{"default", opts}},
575 rev_opts /* unknown_cf_opts */));
576 ASSERT_OK(TryReopenWithColumnFamilies({"default", "reverse"},
577 std::vector<Options>{opts, rev_opts}));
578 for (int i = 0; i < kNumCfs; ++i) {
579 for (int j = 0; j < kEntriesPerCf; ++j) {
580 ASSERT_EQ(Get(i, "key" + std::to_string(j)), "val" + std::to_string(j));
581 }
582 }
583}
