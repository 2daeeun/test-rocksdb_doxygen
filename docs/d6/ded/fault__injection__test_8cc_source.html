196 void Build(const WriteOptions& write_options, int start_idx, int num_vals) {
197 std::string key_space, value_space;
198 WriteBatch batch;
199 for (int i = start_idx; i < start_idx + num_vals; i++) {
200 Slice key = Key(i, &key_space);
201 batch.Clear();
202 ASSERT_OK(batch.Put(key, Value(i, &value_space)));
203 ASSERT_OK(db_->Write(write_options, &batch));
204 }
205 }
206
207 Status ReadValue(int i, std::string* val) const {
208 std::string key_space, value_space;
209 Slice key = Key(i, &key_space);
210 Value(i, &value_space);
211 ReadOptions options;
212 return db_->Get(options, key, val);
213 }
214
215 Status Verify(int start_idx, int num_vals,
216 ExpectedVerifResult expected) const {
217 std::string val;
218 std::string value_space;
219 Status s;
220 for (int i = start_idx; i < start_idx + num_vals && s.ok(); i++) {
221 Value(i, &value_space);
222 s = ReadValue(i, &val);
223 if (s.ok()) {
224 EXPECT_EQ(value_space, val);
225 }
226 if (expected == kValExpectFound) {
227 if (!s.ok()) {
228 fprintf(stderr, "Error when read %dth record (expect found): %s\n", i,
229 s.ToString().c_str());
230 return s;
231 }
232 } else if (!s.ok() && !s.IsNotFound()) {
233 fprintf(stderr, "Error when read %dth record: %s\n", i,
234 s.ToString().c_str());
235 return s;
236 }
237 }
238 return Status::OK();
239 }

241 // Return the ith key
242 Slice Key(int i, std::string* storage) const {
243 unsigned long long num = i;
244 if (!sequential_order_) {
245 // random transfer
246 const int m = 0x5bd1e995;
247 num *= m;
248 num ^= num << 24;
249 }
250 char buf[100];
251 snprintf(buf, sizeof(buf), "%016d", static_cast<int>(num));
252 storage->assign(buf, strlen(buf));
253 return Slice(*storage);
254 }
255
256 // Return the value to associate with the specified key
257 Slice Value(int k, std::string* storage) const {
258 Random r(k);
259 *storage = r.RandomString(kValueSize);
260 return Slice(*storage);
261 }

290 // rnd cannot be null for kResetDropRandomUnsyncedData
291 void ResetDBState(ResetMethod reset_method, Random* rnd = nullptr) {
292 env_->AssertNoOpenFile();
293 switch (reset_method) {
294 case kResetDropUnsyncedData:
295 ASSERT_OK(env_->DropUnsyncedFileData());
296 break;
297 case kResetDropRandomUnsyncedData:
298 ASSERT_OK(env_->DropRandomUnsyncedFileData(rnd));
299 break;
300 case kResetDeleteUnsyncedFiles:
301 ASSERT_OK(env_->DeleteFilesCreatedAfterLastDirSync());
302 break;
303 case kResetDropAndDeleteUnsynced:
304 ASSERT_OK(env_->DropUnsyncedFileData());
305 ASSERT_OK(env_->DeleteFilesCreatedAfterLastDirSync());
306 break;
307 default:
308 assert(false);
309 }
310 }

362TEST_P(FaultInjectionTestSplitted, FaultTest) {
363 do {
364 Random rnd(301);
365
366 for (size_t idx = 0; idx < kNumIterations; idx++) {
367 int num_pre_sync = rnd.Uniform(kMaxNumValues);
368 int num_post_sync = rnd.Uniform(kMaxNumValues);
369
370 PartialCompactTestPreFault(num_pre_sync, num_post_sync);
371 PartialCompactTestReopenWithFault(kResetDropUnsyncedData, num_pre_sync,
372 num_post_sync);
373 NoWriteTestPreFault();
374 NoWriteTestReopenWithFault(kResetDropUnsyncedData);
375
376 PartialCompactTestPreFault(num_pre_sync, num_post_sync);
377 PartialCompactTestReopenWithFault(kResetDropRandomUnsyncedData,
378 num_pre_sync, num_post_sync, &rnd);
379 NoWriteTestPreFault();
380 NoWriteTestReopenWithFault(kResetDropUnsyncedData);
381
382 // Setting a separate data path won't pass the test as we don't sync
383 // it after creating new files,
384 PartialCompactTestPreFault(num_pre_sync, num_post_sync);
385 PartialCompactTestReopenWithFault(kResetDropAndDeleteUnsynced,
386 num_pre_sync, num_post_sync);
387 NoWriteTestPreFault();
388 NoWriteTestReopenWithFault(kResetDropAndDeleteUnsynced);
389
390 PartialCompactTestPreFault(num_pre_sync, num_post_sync);
391 // No new files created so we expect all values since no files will be
392 // dropped.
393 PartialCompactTestReopenWithFault(kResetDeleteUnsyncedFiles, num_pre_sync,
394 num_post_sync);
395 NoWriteTestPreFault();
396 NoWriteTestReopenWithFault(kResetDeleteUnsyncedFiles);
397 }
398 } while (ChangeOptions());
399}

567TEST(FaultInjectionFSTest, ReadUnsyncedData) {
568 std::shared_ptr<FaultInjectionTestFS> fault_fs =
569 std::make_shared<FaultInjectionTestFS>(FileSystem::Default());
570 fault_fs->SetInjectUnsyncedDataLoss(true);
571 ASSERT_TRUE(fault_fs->ReadUnsyncedData());
572 ASSERT_TRUE(fault_fs->InjectUnsyncedDataLoss());
573
574 // This is a randomized mini-stress test, to reduce the chances of bugs in
575 // FaultInjectionTestFS being caught only in db_stress, where they are
576 // difficult to debug. ~1000 iterations might be needed to debug relevant
577 // code changes. Limiting to 10 for each regular unit test run.
578 auto seed = Random::GetTLSInstance()->Next();
579 for (int i = 0; i < 10; i++, seed++) {
580 Random rnd(seed);
581 uint32_t len = rnd.Uniform(10000) + 1;
582
583 std::string f =
584 test::PerThreadDBPath("read_unsynced." + std::to_string(seed));
585 std::string data = rnd.RandomString(len);
586
587 // Create partially synced file
588 std::unique_ptr<FSWritableFile> w;
589 ASSERT_OK(fault_fs->NewWritableFile(f, {}, &w, nullptr));
590 uint32_t synced_len = rnd.Uniform(len + 1);
591 ASSERT_OK(w->Append(Slice(data.data(), synced_len), {}, nullptr));
592 if (synced_len > 0) {
593 ASSERT_OK(w->Sync({}, nullptr));
594 }
595 ASSERT_OK(w->Append(Slice(data.data() + synced_len, len - synced_len), {},
596 nullptr));
597
598 // Test file size includes unsynced data
599 {
600 uint64_t file_size;
601 ASSERT_OK(fault_fs->GetFileSize(f, {}, &file_size, nullptr));
602 ASSERT_EQ(len, file_size);
603 }
604
605 // Test read file contents, with two reads that probably don't
606 // align with the unsynced split. And maybe a sync or write between
607 // the two reads.
608 std::unique_ptr<FSSequentialFile> r;
609 ASSERT_OK(fault_fs->NewSequentialFile(f, {}, &r, nullptr));
610 uint32_t first_read_len = rnd.Uniform(len + 1);
611 Slice sl;
612 std::unique_ptr<char[]> scratch(new char[first_read_len]);
613 ASSERT_OK(r->Read(first_read_len, {}, &sl, scratch.get(), nullptr));
614 ASSERT_EQ(first_read_len, sl.size());
615 ASSERT_EQ(0, sl.compare(Slice(data.data(), first_read_len)));
616
617 // Maybe a sync and/or write and/or close between the two reads.
618 if (rnd.OneIn(2)) {
619 ASSERT_OK(w->Sync({}, nullptr));
620 }
621 if (rnd.OneIn(2)) {
622 uint32_t more_len = rnd.Uniform(1000) + 1;
623 std::string more_data = rnd.RandomString(more_len);
624 ASSERT_OK(w->Append(more_data, {}, nullptr));
625 data += more_data;
626 len += more_len;
627 }
628 if (rnd.OneIn(2)) {
629 ASSERT_OK(w->Sync({}, nullptr));
630 }
631 if (rnd.OneIn(2)) {
632 ASSERT_OK(w->Close({}, nullptr));
633 w.reset();
634 }
635
636 // Second read some of, all of, or more than rest of file
637 uint32_t second_read_len = rnd.Uniform(len + 1);
638 scratch.reset(new char[second_read_len]);
639 ASSERT_OK(r->Read(second_read_len, {}, &sl, scratch.get(), nullptr));
640 if (len - first_read_len < second_read_len) {
641 ASSERT_EQ(len - first_read_len, sl.size());
642 } else {
643 ASSERT_EQ(second_read_len, sl.size());
644 }
645 ASSERT_EQ(0, sl.compare(Slice(data.data() + first_read_len, sl.size())));
646 }
