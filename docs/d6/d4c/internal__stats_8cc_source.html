92void PrintLevelStatsHeader(char* buf, size_t len, const std::string& cf_name,
93 const std::string& group_by) {
94 int written_size =
95 snprintf(buf, len, "\n** Compaction Stats [%s] **\n", cf_name.c_str());
96 written_size = std::min(written_size, static_cast<int>(len));
97 auto hdr = [](LevelStatType t) {
98 return InternalStats::compaction_level_stats.at(t).header_name.c_str();
99 };
100 int line_size = snprintf(
101 buf + written_size, len - written_size,
102 "%s %s %s %s %s %s %s %s %s %s %s %s %s %s %s %s %s %s %s %s "
103 "%s\n",
104 // Note that we skip COMPACTED_FILES and merge it with Files column
105 group_by.c_str(), hdr(LevelStatType::NUM_FILES),
106 hdr(LevelStatType::SIZE_BYTES), hdr(LevelStatType::SCORE),
107 hdr(LevelStatType::READ_GB), hdr(LevelStatType::RN_GB),
108 hdr(LevelStatType::RNP1_GB), hdr(LevelStatType::WRITE_GB),
109 hdr(LevelStatType::W_NEW_GB), hdr(LevelStatType::MOVED_GB),
110 hdr(LevelStatType::WRITE_AMP), hdr(LevelStatType::READ_MBPS),
111 hdr(LevelStatType::WRITE_MBPS), hdr(LevelStatType::COMP_SEC),
112 hdr(LevelStatType::COMP_CPU_SEC), hdr(LevelStatType::COMP_COUNT),
113 hdr(LevelStatType::AVG_SEC), hdr(LevelStatType::KEY_IN),
114 hdr(LevelStatType::KEY_DROP), hdr(LevelStatType::R_BLOB_GB),
115 hdr(LevelStatType::W_BLOB_GB));
116
117 written_size += line_size;
118 written_size = std::min(written_size, static_cast<int>(len));
119 snprintf(buf + written_size, len - written_size, "%s\n",
120 std::string(line_size, '-').c_str());
121}

123void PrepareLevelStats(std::map<LevelStatType, double>* level_stats,
124 int num_files, int being_compacted,
125 double total_file_size, double score, double w_amp,
126 const InternalStats::CompactionStats& stats) {
127 const uint64_t bytes_read = stats.bytes_read_non_output_levels +
128 stats.bytes_read_output_level +
129 stats.bytes_read_blob;
130 const uint64_t bytes_written = stats.bytes_written + stats.bytes_written_blob;
131 const int64_t bytes_new = stats.bytes_written - stats.bytes_read_output_level;
132 const double elapsed = (stats.micros + 1) / kMicrosInSec;
133
134 (*level_stats)[LevelStatType::NUM_FILES] = num_files;
135 (*level_stats)[LevelStatType::COMPACTED_FILES] = being_compacted;
136 (*level_stats)[LevelStatType::SIZE_BYTES] = total_file_size;
137 (*level_stats)[LevelStatType::SCORE] = score;
138 (*level_stats)[LevelStatType::READ_GB] = bytes_read / kGB;
139 (*level_stats)[LevelStatType::RN_GB] =
140 stats.bytes_read_non_output_levels / kGB;
141 (*level_stats)[LevelStatType::RNP1_GB] = stats.bytes_read_output_level / kGB;
142 (*level_stats)[LevelStatType::WRITE_GB] = stats.bytes_written / kGB;
143 (*level_stats)[LevelStatType::W_NEW_GB] = bytes_new / kGB;
144 (*level_stats)[LevelStatType::MOVED_GB] = stats.bytes_moved / kGB;
145 (*level_stats)[LevelStatType::WRITE_AMP] = w_amp;
146 (*level_stats)[LevelStatType::READ_MBPS] = bytes_read / kMB / elapsed;
147 (*level_stats)[LevelStatType::WRITE_MBPS] = bytes_written / kMB / elapsed;
148 (*level_stats)[LevelStatType::COMP_SEC] = stats.micros / kMicrosInSec;
149 (*level_stats)[LevelStatType::COMP_CPU_SEC] = stats.cpu_micros / kMicrosInSec;
150 (*level_stats)[LevelStatType::COMP_COUNT] = stats.count;
151 (*level_stats)[LevelStatType::AVG_SEC] =
152 stats.count == 0 ? 0 : stats.micros / kMicrosInSec / stats.count;
153 (*level_stats)[LevelStatType::KEY_IN] =
154 static_cast<double>(stats.num_input_records);
155 (*level_stats)[LevelStatType::KEY_DROP] =
156 static_cast<double>(stats.num_dropped_records);
157 (*level_stats)[LevelStatType::R_BLOB_GB] = stats.bytes_read_blob / kGB;
158 (*level_stats)[LevelStatType::W_BLOB_GB] = stats.bytes_written_blob / kGB;
159}

161void PrintLevelStats(char* buf, size_t len, const std::string& name,
162 const std::map<LevelStatType, double>& stat_value) {
163 snprintf(
164 buf, len,
165 "%4s " /* Level */
166 "%6d/%-3d " /* Files */
167 "%8s " /* Size */
168 "%5.1f " /* Score */
169 "%8.1f " /* Read(GB) */
170 "%7.1f " /* Rn(GB) */
171 "%8.1f " /* Rnp1(GB) */
172 "%9.1f " /* Write(GB) */
173 "%8.1f " /* Wnew(GB) */
174 "%9.1f " /* Moved(GB) */
175 "%5.1f " /* W-Amp */
176 "%8.1f " /* Rd(MB/s) */
177 "%8.1f " /* Wr(MB/s) */
178 "%9.2f " /* Comp(sec) */
179 "%17.2f " /* CompMergeCPU(sec) */
180 "%9d " /* Comp(cnt) */
181 "%8.3f " /* Avg(sec) */
182 "%7s " /* KeyIn */
183 "%6s " /* KeyDrop */
184 "%9.1f " /* Rblob(GB) */
185 "%9.1f\n", /* Wblob(GB) */
186 name.c_str(), static_cast<int>(stat_value.at(LevelStatType::NUM_FILES)),
187 static_cast<int>(stat_value.at(LevelStatType::COMPACTED_FILES)),
188 BytesToHumanString(
189 static_cast<uint64_t>(stat_value.at(LevelStatType::SIZE_BYTES)))
190 .c_str(),
191 stat_value.at(LevelStatType::SCORE),
192 stat_value.at(LevelStatType::READ_GB),
193 stat_value.at(LevelStatType::RN_GB),
194 stat_value.at(LevelStatType::RNP1_GB),
195 stat_value.at(LevelStatType::WRITE_GB),
196 stat_value.at(LevelStatType::W_NEW_GB),
197 stat_value.at(LevelStatType::MOVED_GB),
198 stat_value.at(LevelStatType::WRITE_AMP),
199 stat_value.at(LevelStatType::READ_MBPS),
200 stat_value.at(LevelStatType::WRITE_MBPS),
201 stat_value.at(LevelStatType::COMP_SEC),
202 stat_value.at(LevelStatType::COMP_CPU_SEC),
203 static_cast<int>(stat_value.at(LevelStatType::COMP_COUNT)),
204 stat_value.at(LevelStatType::AVG_SEC),
205 NumberToHumanString(
206 static_cast<std::int64_t>(stat_value.at(LevelStatType::KEY_IN)))
207 .c_str(),
208 NumberToHumanString(
209 static_cast<std::int64_t>(stat_value.at(LevelStatType::KEY_DROP)))
210 .c_str(),
211 stat_value.at(LevelStatType::R_BLOB_GB),
212 stat_value.at(LevelStatType::W_BLOB_GB));
213}

225// Assumes that trailing numbers represent an optional argument. This requires
226// property names to not end with numbers.
227std::pair<Slice, Slice> GetPropertyNameAndArg(const Slice& property) {
228 Slice name = property, arg = property;
229 size_t sfx_len = 0;
230 while (sfx_len < property.size() &&
231 isdigit(property[property.size() - sfx_len - 1])) {
232 ++sfx_len;
233 }
234 name.remove_suffix(sfx_len);
235 arg.remove_prefix(property.size() - sfx_len);
236 return {name, arg};
237}

664void InternalStats::CollectCacheEntryStats(bool foreground) {
665 // This function is safe to call from any thread because
666 // cache_entry_stats_collector_ field is const after constructor
667 // and ->GetStats does its own synchronization, which also suffices for
668 // cache_entry_stats_.
669
670 if (!cache_entry_stats_collector_) {
671 return; // nothing to do (e.g. no block cache)
672 }
673
674 // For "background" collections, strictly cap the collection time by
675 // expanding effective cache TTL. For foreground, be more aggressive about
676 // getting latest data.
677 int min_interval_seconds = foreground ? 10 : 180;
678 // 1/500 = max of 0.2% of one CPU thread
679 int min_interval_factor = foreground ? 10 : 500;
680 cache_entry_stats_collector_->CollectStats(min_interval_seconds,
681 min_interval_factor);
682}

684std::function<void(const Slice& key, Cache::ObjectPtr value, size_t charge,
685 const Cache::CacheItemHelper* helper)>
686InternalStats::CacheEntryRoleStats::GetEntryCallback() {
687 return [&](const Slice& /*key*/, Cache::ObjectPtr /*value*/, size_t charge,
688 const Cache::CacheItemHelper* helper) -> void {
689 size_t role_idx =

729std::string InternalStats::CacheEntryRoleStats::ToString(
730 SystemClock* clock) const {
731 std::ostringstream str;
732 str << "Block cache " << cache_id
733 << " capacity: " << BytesToHumanString(cache_capacity)
734 << " seed: " << hash_seed << " usage: " << BytesToHumanString(cache_usage)
735 << " table_size: " << table_size << " occupancy: " << occupancy
736 << " collections: " << collection_count
737 << " last_copies: " << copies_of_last_collection
738 << " last_secs: " << (GetLastDurationMicros() / 1000000.0)
739 << " secs_since: "
740 << ((clock->NowMicros() - last_end_time_micros_) / 1000000U) << "\n";
741 str << "Block cache entry stats(count,size,portion):";
742 for (size_t i = 0; i < kNumCacheEntryRoles; ++i) {
743 if (entry_counts[i] > 0) {
744 str << " " << kCacheEntryRoleToCamelString[i] << "(" << entry_counts[i]
745 << "," << BytesToHumanString(total_charges[i]) << ","
746 << (100.0 * total_charges[i] / cache_capacity) << "%)";
747 }
748 }
749 str << "\n";
750 return str.str();
751}

759 std::to_string(cache_capacity);

762 v[BlockCacheEntryStatsMapKeys::LastCollectionAgeSeconds()] =

769 std::to_string(total_charges[i]);

772 }

1070bool InternalStats::HandleCFStatsPeriodic(std::string* value,
1071 Slice /*suffix*/) {
1072 bool has_change = has_cf_change_since_dump_;
1073 if (!has_change) {
1074 // If file histogram changes, there is activity in this period too.
1075 uint64_t new_histogram_num = 0;
1076 for (int level = 0; level < number_levels_; level++) {
1077 new_histogram_num += file_read_latency_[level].num();
1078 }
1079 new_histogram_num += blob_file_read_latency_.num();
1080 if (new_histogram_num != last_histogram_num) {
1081 has_change = true;
1082 last_histogram_num = new_histogram_num;
1083 }
1084 }
1085 if (has_change) {
1086 no_cf_change_period_since_dump_ = 0;
1087 has_cf_change_since_dump_ = false;
1088 } else if (no_cf_change_period_since_dump_++ > 0) {
1089 // Not ready to sync
1090 if (no_cf_change_period_since_dump_ == kMaxNoChangePeriodSinceDump) {
1091 // Next periodic, we need to dump stats even if there is no change.
1092 no_cf_change_period_since_dump_ = 0;
1093 }
1094 return true;
1095 }
1096
1097 DumpCFStatsNoFileHistogram(/*is_periodic=*/true, value);
1098 DumpCFFileHistogram(value);
1099 return true;
1100}

1546void InternalStats::DumpDBStats(std::string* value) {
1547 char buf[1000];
1548 // DB-level stats, only available from default column family
1549 double seconds_up = (clock_->NowMicros() - started_at_) / kMicrosInSec;
1550 double interval_seconds_up = seconds_up - db_stats_snapshot_.seconds_up;
1551 snprintf(buf, sizeof(buf),
1552 "\n** DB Stats **\nUptime(secs): %.1f total, %.1f interval\n",
1553 seconds_up, interval_seconds_up);
1554 value->append(buf);
1555 // Cumulative
1556 uint64_t user_bytes_written =
1557 GetDBStats(InternalStats::kIntStatsBytesWritten);
1558 uint64_t num_keys_written =
1559 GetDBStats(InternalStats::kIntStatsNumKeysWritten);
1560 uint64_t write_other = GetDBStats(InternalStats::kIntStatsWriteDoneByOther);
1561 uint64_t write_self = GetDBStats(InternalStats::kIntStatsWriteDoneBySelf);
1562 uint64_t wal_bytes = GetDBStats(InternalStats::kIntStatsWalFileBytes);
1563 uint64_t wal_synced = GetDBStats(InternalStats::kIntStatsWalFileSynced);
1564 uint64_t write_with_wal = GetDBStats(InternalStats::kIntStatsWriteWithWal);
1565 uint64_t write_stall_micros =
1566 GetDBStats(InternalStats::kIntStatsWriteStallMicros);
1567
1568 const int kHumanMicrosLen = 32;
1569 char human_micros[kHumanMicrosLen];
1570
1571 // Data
1572 // writes: total number of write requests.
1573 // keys: total number of key updates issued by all the write requests
1574 // commit groups: number of group commits issued to the DB. Each group can
1575 // contain one or more writes.
1576 // so writes/keys is the average number of put in multi-put or put
1577 // writes/groups is the average group commit size.
1578 //
1579 // The format is the same for interval stats.
1580 snprintf(buf, sizeof(buf),
1581 "Cumulative writes: %s writes, %s keys, %s commit groups, "
1582 "%.1f writes per commit group, ingest: %.2f GB, %.2f MB/s\n",
1583 NumberToHumanString(write_other + write_self).c_str(),
1584 NumberToHumanString(num_keys_written).c_str(),
1585 NumberToHumanString(write_self).c_str(),
1586 (write_other + write_self) /
1587 std::max(1.0, static_cast<double>(write_self)),
1588 user_bytes_written / kGB,
1589 user_bytes_written / kMB / std::max(seconds_up, 0.001));
1590 value->append(buf);
1591 // WAL
1592 snprintf(buf, sizeof(buf),
1593 "Cumulative WAL: %s writes, %s syncs, "
1594 "%.2f writes per sync, written: %.2f GB, %.2f MB/s\n",
1595 NumberToHumanString(write_with_wal).c_str(),
1596 NumberToHumanString(wal_synced).c_str(),
1597 write_with_wal / std::max(1.0, static_cast<double>(wal_synced)),
1598 wal_bytes / kGB, wal_bytes / kMB / std::max(seconds_up, 0.001));
1599 value->append(buf);
1600 // Stall
1601 AppendHumanMicros(write_stall_micros, human_micros, kHumanMicrosLen, true);
1602 snprintf(buf, sizeof(buf), "Cumulative stall: %s, %.1f percent\n",
1603 human_micros,
1604 // 10000 = divide by 1M to get secs, then multiply by 100 for pct
1605 write_stall_micros / 10000.0 / std::max(seconds_up, 0.001));
1606 value->append(buf);
1607
1608 // Interval
1609 uint64_t interval_write_other = write_other - db_stats_snapshot_.write_other;
1610 uint64_t interval_write_self = write_self - db_stats_snapshot_.write_self;
1611 uint64_t interval_num_keys_written =
1612 num_keys_written - db_stats_snapshot_.num_keys_written;
1613 snprintf(
1614 buf, sizeof(buf),
1615 "Interval writes: %s writes, %s keys, %s commit groups, "
1616 "%.1f writes per commit group, ingest: %.2f MB, %.2f MB/s\n",
1617 NumberToHumanString(interval_write_other + interval_write_self).c_str(),
1618 NumberToHumanString(interval_num_keys_written).c_str(),
1619 NumberToHumanString(interval_write_self).c_str(),
1620 static_cast<double>(interval_write_other + interval_write_self) /
1621 std::max(1.0, static_cast<double>(interval_write_self)),
1622 (user_bytes_written - db_stats_snapshot_.ingest_bytes) / kMB,
1623 (user_bytes_written - db_stats_snapshot_.ingest_bytes) / kMB /
1624 std::max(interval_seconds_up, 0.001)),
1625 value->append(buf);
1626
1627 uint64_t interval_write_with_wal =
1628 write_with_wal - db_stats_snapshot_.write_with_wal;
1629 uint64_t interval_wal_synced = wal_synced - db_stats_snapshot_.wal_synced;
1630 uint64_t interval_wal_bytes = wal_bytes - db_stats_snapshot_.wal_bytes;
1631
1632 snprintf(buf, sizeof(buf),
1633 "Interval WAL: %s writes, %s syncs, "
1634 "%.2f writes per sync, written: %.2f GB, %.2f MB/s\n",
1635 NumberToHumanString(interval_write_with_wal).c_str(),
1636 NumberToHumanString(interval_wal_synced).c_str(),
1637 interval_write_with_wal /
1638 std::max(1.0, static_cast<double>(interval_wal_synced)),
1639 interval_wal_bytes / kGB,
1640 interval_wal_bytes / kMB / std::max(interval_seconds_up, 0.001));
1641 value->append(buf);
1642
1643 // Stall
1644 AppendHumanMicros(write_stall_micros - db_stats_snapshot_.write_stall_micros,
1645 human_micros, kHumanMicrosLen, true);
1646 snprintf(buf, sizeof(buf), "Interval stall: %s, %.1f percent\n", human_micros,
1647 // 10000 = divide by 1M to get secs, then multiply by 100 for pct
1648 (write_stall_micros - db_stats_snapshot_.write_stall_micros) /
1649 10000.0 / std::max(interval_seconds_up, 0.001));
1650 value->append(buf);
1651
1652 std::string write_stall_stats;
1653 DumpDBStatsWriteStall(&write_stall_stats);
1654 value->append(write_stall_stats);
1655
1656 db_stats_snapshot_.seconds_up = seconds_up;
1657 db_stats_snapshot_.ingest_bytes = user_bytes_written;
1658 db_stats_snapshot_.write_other = write_other;
1659 db_stats_snapshot_.write_self = write_self;
1660 db_stats_snapshot_.num_keys_written = num_keys_written;
1661 db_stats_snapshot_.wal_bytes = wal_bytes;
1662 db_stats_snapshot_.wal_synced = wal_synced;
1663 db_stats_snapshot_.write_with_wal = write_with_wal;
1664 db_stats_snapshot_.write_stall_micros = write_stall_micros;
1665}

1748void InternalStats::DumpCFMapStats(
1749 const VersionStorageInfo* vstorage,
1750 std::map<int, std::map<LevelStatType, double>>* levels_stats,
1751 CompactionStats* compaction_stats_sum) {
1752 assert(vstorage);
1753
1754 int num_levels_to_check =
1755 (cfd_->ioptions()->compaction_style == kCompactionStyleLevel)
1756 ? vstorage->num_levels() - 1
1757 : 1;
1758
1759 // Compaction scores are sorted based on its value. Restore them to the
1760 // level order
1761 std::vector<double> compaction_score(number_levels_, 0);
1762 for (int i = 0; i < num_levels_to_check; ++i) {
1763 compaction_score[vstorage->CompactionScoreLevel(i)] =
1764 vstorage->CompactionScore(i);
1765 }
1766 // Count # of files being compacted for each level
1767 std::vector<int> files_being_compacted(number_levels_, 0);
1768 for (int level = 0; level < number_levels_; ++level) {
1769 for (auto* f : vstorage->LevelFiles(level)) {
1770 if (f->being_compacted) {
1771 ++files_being_compacted[level];
1772 }
1773 }
1774 }
1775
1776 int total_files = 0;
1777 int total_files_being_compacted = 0;
1778 double total_file_size = 0;
1779 uint64_t flush_ingest = cf_stats_value_[BYTES_FLUSHED];
1780 uint64_t add_file_ingest = cf_stats_value_[BYTES_INGESTED_ADD_FILE];
1781 uint64_t curr_ingest = flush_ingest + add_file_ingest;
1782 for (int level = 0; level < number_levels_; level++) {
1783 int files = vstorage->NumLevelFiles(level);
1784 total_files += files;
1785 total_files_being_compacted += files_being_compacted[level];
1786 if (comp_stats_[level].micros > 0 || comp_stats_[level].cpu_micros > 0 ||
1787 files > 0) {
1788 compaction_stats_sum->Add(comp_stats_[level]);
1789 total_file_size += vstorage->NumLevelBytes(level);
1790 uint64_t input_bytes;
1791 if (level == 0) {
1792 input_bytes = curr_ingest;
1793 } else {
1794 input_bytes = comp_stats_[level].bytes_read_non_output_levels +
1795 comp_stats_[level].bytes_read_blob;
1796 }
1797 double w_amp =
1798 (input_bytes == 0)
1799 ? 0.0
1800 : static_cast<double>(comp_stats_[level].bytes_written +
1801 comp_stats_[level].bytes_written_blob) /
1802 input_bytes;
1803 std::map<LevelStatType, double> level_stats;
1804 PrepareLevelStats(&level_stats, files, files_being_compacted[level],
1805 static_cast<double>(vstorage->NumLevelBytes(level)),
1806 compaction_score[level], w_amp, comp_stats_[level]);
1807 (*levels_stats)[level] = level_stats;
1808 }
1809 }
1810 // Cumulative summary
1811 double w_amp = (0 == curr_ingest)
1812 ? 0.0
1813 : (compaction_stats_sum->bytes_written +
1814 compaction_stats_sum->bytes_written_blob) /
1815 static_cast<double>(curr_ingest);
1816 // Stats summary across levels
1817 std::map<LevelStatType, double> sum_stats;
1818 PrepareLevelStats(&sum_stats, total_files, total_files_being_compacted,
1819 total_file_size, 0, w_amp, *compaction_stats_sum);
1820 (*levels_stats)[-1] = sum_stats; // -1 is for the Sum level
1821}
