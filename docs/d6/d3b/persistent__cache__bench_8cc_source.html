70// create a new cache tier
71// construct a tiered RAM+Block cache
72std::unique_ptr<PersistentTieredCache> NewTieredCache(
73 const size_t mem_size, const PersistentCacheConfig& opt) {
74 std::unique_ptr<PersistentTieredCache> tcache(new PersistentTieredCache());
75 // create primary tier
76 assert(mem_size);
77 auto pcache =
78 std::shared_ptr<PersistentCacheTier>(new VolatileCacheTier(mem_size));
79 tcache->AddTier(pcache);
80 // create secondary tier
81 auto scache = std::shared_ptr<PersistentCacheTier>(new BlockCacheTier(opt));
82 tcache->AddTier(scache);
83
84 Status s = tcache->Open();
85 assert(s.ok());
86 return tcache;
87}

89std::unique_ptr<PersistentTieredCache> NewTieredCache() {

91 if (!Env::Default()->NewLogger(FLAGS_log_path, &log).ok()) {
92 fprintf(stderr, "Error creating log %s \n", FLAGS_log_path.c_str());

102 opt.max_write_pipeline_backlog_size = std::numeric_limits<uint64_t>::max();
103 return NewTieredCache(FLAGS_cache_size * pct, opt);
104}

107// Benchmark driver

109class CacheTierBenchmark {
110 public:
111 explicit CacheTierBenchmark(std::shared_ptr<PersistentCacheTier>&& cache)
112 : cache_(cache) {
113 if (FLAGS_nthread_read) {
114 fprintf(stdout, "Pre-populating\n");
115 Prepop();
116 fprintf(stdout, "Pre-population completed\n");
117 }
118
119 stats_.Clear();
120
121 // Start IO threads
122 std::list<port::Thread> threads;
123 Spawn(FLAGS_nthread_write, &threads,
124 std::bind(&CacheTierBenchmark::Write, this));
125 Spawn(FLAGS_nthread_read, &threads,
126 std::bind(&CacheTierBenchmark::Read, this));
127
128 // Wait till FLAGS_nsec and then signal to quit
129 StopWatchNano t(SystemClock::Default().get(), /*auto_start=*/true);
130 size_t sec = t.ElapsedNanos() / 1000000000ULL;
131 while (!quit_) {
132 sec = t.ElapsedNanos() / 1000000000ULL;
133 quit_ = sec > size_t(FLAGS_nsec);
134 /* sleep override */ sleep(1);
135 }
136
137 // Wait for threads to exit
138 Join(&threads);
139 // Print stats
140 PrintStats(sec);
141 // Close the cache
142 cache_->TEST_Flush();
143 cache_->Close();
144 }
145
146 private:
147 void PrintStats(const size_t sec) {
148 std::ostringstream msg;
149 msg << "Test stats" << std::endl
150 << "* Elapsed: " << sec << " s" << std::endl
151 << "* Write Latency:" << std::endl
152 << stats_.write_latency_.ToString() << std::endl
153 << "* Read Latency:" << std::endl
154 << stats_.read_latency_.ToString() << std::endl
155 << "* Bytes written:" << std::endl
156 << stats_.bytes_written_.ToString() << std::endl
157 << "* Bytes read:" << std::endl
158 << stats_.bytes_read_.ToString() << std::endl
159 << "Cache stats:" << std::endl
160 << cache_->PrintStats() << std::endl;
161 fprintf(stderr, "%s\n", msg.str().c_str());
162 }

164 //
165 // Insert implementation and corresponding helper functions
166 //
167 void Prepop() {
168 for (uint64_t i = 0; i < 1024 * 1024; ++i) {
169 InsertKey(i);
170 insert_key_limit_++;
171 read_key_limit_++;
172 }
173
174 // Wait until data is flushed
175 cache_->TEST_Flush();
176 // warmup the cache
177 for (uint64_t i = 0; i < 1024 * 1024; ReadKey(i++)) {
178 }
179 }
180
181 void Write() {
182 while (!quit_) {
183 InsertKey(insert_key_limit_++);
184 }
185 }
186
187 void InsertKey(const uint64_t key) {
188 // construct key
189 uint64_t k[3];
190 Slice block_key = FillKey(k, key);
191
192 // construct value
193 auto block = NewBlock(key);
194
195 // insert
196 StopWatchNano timer(SystemClock::Default().get(), /*auto_start=*/true);
197 while (true) {
198 Status status = cache_->Insert(block_key, block.get(), FLAGS_iosize);
199 if (status.ok()) {
200 break;
201 }
202
203 // transient error is possible if we run without pipelining
204 assert(!FLAGS_enable_pipelined_writes);
205 }
206
207 // adjust stats
208 const size_t elapsed_micro = timer.ElapsedNanos() / 1000;
209 stats_.write_latency_.Add(elapsed_micro);
210 stats_.bytes_written_.Add(FLAGS_iosize);
211 }
212
213 //
214 // Read implementation
215 //
216 void Read() {
217 while (!quit_) {
218 ReadKey(random() % read_key_limit_);
219 }
220 }
221
222 void ReadKey(const uint64_t val) {
223 // construct key
224 uint64_t k[3];
225 Slice key = FillKey(k, val);
226
227 // Lookup in cache
228 StopWatchNano timer(SystemClock::Default().get(), /*auto_start=*/true);
229 std::unique_ptr<char[]> block;
230 size_t size;
231 Status status = cache_->Lookup(key, &block, &size);
232 if (!status.ok()) {
233 fprintf(stderr, "%s\n", status.ToString().c_str());
234 }
235 assert(status.ok());
236 assert(size == (size_t)FLAGS_iosize);
237
238 // adjust stats
239 const size_t elapsed_micro = timer.ElapsedNanos() / 1000;
240 stats_.read_latency_.Add(elapsed_micro);
241 stats_.bytes_read_.Add(FLAGS_iosize);
242
243 // verify content
244 if (!FLAGS_benchmark) {
245 auto expected_block = NewBlock(val);
246 assert(memcmp(block.get(), expected_block.get(), FLAGS_iosize) == 0);
247 }
248 }
249
250 // create data for a key by filling with a certain pattern
251 std::unique_ptr<char[]> NewBlock(const uint64_t val) {
252 std::unique_ptr<char[]> data(new char[FLAGS_iosize]);
253 memset(data.get(), val % 255, FLAGS_iosize);
254 return data;
255 }
256
257 // spawn threads
258 void Spawn(const size_t n, std::list<port::Thread>* threads,
259 const std::function<void()>& fn) {
260 for (size_t i = 0; i < n; ++i) {
261 threads->emplace_back(fn);
262 }
263 }
264
265 // join threads
266 void Join(std::list<port::Thread>* threads) {
267 for (auto& th : *threads) {
268 th.join();
269 }
270 }
271
272 // construct key
273 Slice FillKey(uint64_t (&k)[3], const uint64_t val) {
274 k[0] = k[1] = 0;
275 k[2] = val;
276 void* p = static_cast<void*>(&k);
277 return Slice(static_cast<char*>(p), sizeof(k));
278 }

308 GFLAGS_NAMESPACE::SetUsageMessage(std::string("\nUSAGE:\n") +
309 std::string(argv[0]) + " [OPTIONS]...");
310 GFLAGS_NAMESPACE::ParseCommandLineFlags(&argc, &argv, false);

314 << "======" << std::endl
315 << "* nsec=" << FLAGS_nsec << std::endl
316 << "* nthread_write=" << FLAGS_nthread_write << std::endl
317 << "* path=" << FLAGS_path << std::endl
318 << "* cache_size=" << FLAGS_cache_size << std::endl
319 << "* iosize=" << FLAGS_iosize << std::endl
320 << "* writer_iosize=" << FLAGS_writer_iosize << std::endl
321 << "* writer_qdepth=" << FLAGS_writer_qdepth << std::endl
322 << "* enable_pipelined_writes=" << FLAGS_enable_pipelined_writes
323 << std::endl
324 << "* cache_type=" << FLAGS_cache_type << std::endl
325 << "* benchmark=" << FLAGS_benchmark << std::endl
326 << "* volatile_cache_pct=" << FLAGS_volatile_cache_pct << std::endl;
327
328 fprintf(stderr, "%s\n", msg.str().c_str());
329
330 std::shared_ptr<ROCKSDB_NAMESPACE::PersistentCacheTier> cache;
331 if (FLAGS_cache_type == "block_cache") {
332 fprintf(stderr, "Using block cache implementation\n");
333 cache = ROCKSDB_NAMESPACE::NewBlockCache();
334 } else if (FLAGS_cache_type == "volatile") {
335 fprintf(stderr, "Using volatile cache implementation\n");
336 cache = ROCKSDB_NAMESPACE::NewVolatileCache();
337 } else if (FLAGS_cache_type == "tiered") {
338 fprintf(stderr, "Using tiered cache implementation\n");
339 cache = ROCKSDB_NAMESPACE::NewTieredCache();
340 } else {
341 fprintf(stderr, "Unknown option for cache\n");
342 }
343
344 assert(cache);
345 if (!cache) {
346 fprintf(stderr, "Error creating cache\n");
347 abort();
348 }
349
350 std::unique_ptr<ROCKSDB_NAMESPACE::CacheTierBenchmark> benchmark(
351 new ROCKSDB_NAMESPACE::CacheTierBenchmark(std::move(cache)));
352
353 return 0;
354}
