59Status BuildTable(
60 const std::string& dbname, VersionSet* versions,
61 const ImmutableDBOptions& db_options, const TableBuilderOptions& tboptions,
62 const FileOptions& file_options, TableCache* table_cache,
63 InternalIterator* iter,
64 std::vector<std::unique_ptr<FragmentedRangeTombstoneIterator>>
65 range_del_iters,
66 FileMetaData* meta, std::vector<BlobFileAddition>* blob_file_additions,
67 std::vector<SequenceNumber> snapshots, SequenceNumber earliest_snapshot,
68 SequenceNumber earliest_write_conflict_snapshot,
69 SequenceNumber job_snapshot, SnapshotChecker* snapshot_checker,
70 bool paranoid_file_checks, InternalStats* internal_stats,
71 IOStatus* io_status, const std::shared_ptr<IOTracer>& io_tracer,
72 BlobFileCreationReason blob_creation_reason,
73 UnownedPtr<const SeqnoToTimeMapping> seqno_to_time_mapping,
74 EventLogger* event_logger, int job_id, TableProperties* table_properties,
75 Env::WriteLifeTimeHint write_hint, const std::string* full_history_ts_low,
76 BlobFileCompletionCallback* blob_callback, Version* version,
77 uint64_t* num_input_entries, uint64_t* memtable_payload_bytes,
78 uint64_t* memtable_garbage_bytes) {
79 assert((tboptions.column_family_id ==
80 TablePropertiesCollectorFactory::Context::kUnknownColumnFamily) ==
81 tboptions.column_family_name.empty());
82 auto& mutable_cf_options = tboptions.moptions;
83 auto& ioptions = tboptions.ioptions;
84 // Reports the IOStats for flush for every following bytes.
85 const size_t kReportFlushIOStatsEvery = 1048576;
86 OutputValidator output_validator(tboptions.internal_comparator,
87 /*enable_hash=*/paranoid_file_checks);
88 Status s;
89 meta->fd.file_size = 0;
90 iter->SeekToFirst();
91 std::unique_ptr<CompactionRangeDelAggregator> range_del_agg(
92 new CompactionRangeDelAggregator(&tboptions.internal_comparator,
93 snapshots, full_history_ts_low));
94 uint64_t num_unfragmented_tombstones = 0;
95 uint64_t total_tombstone_payload_bytes = 0;
96 for (auto& range_del_iter : range_del_iters) {
97 num_unfragmented_tombstones +=
98 range_del_iter->num_unfragmented_tombstones();
99 total_tombstone_payload_bytes +=
100 range_del_iter->total_tombstone_payload_bytes();
101 range_del_agg->AddTombstones(std::move(range_del_iter));
102 }
103
104 std::string fname = TableFileName(ioptions.cf_paths, meta->fd.GetNumber(),
105 meta->fd.GetPathId());
106 std::vector<std::string> blob_file_paths;
107 std::string file_checksum = kUnknownFileChecksum;
108 std::string file_checksum_func_name = kUnknownFileChecksumFuncName;
109 EventHelpers::NotifyTableFileCreationStarted(ioptions.listeners, dbname,
110 tboptions.column_family_name,
111 fname, job_id, tboptions.reason);
112 Env* env = db_options.env;
113 assert(env);
114 FileSystem* fs = db_options.fs.get();
115 assert(fs);
116
117 TableProperties tp;
118 bool table_file_created = false;
119 if (iter->Valid() || !range_del_agg->IsEmpty()) {
120 std::unique_ptr<CompactionFilter> compaction_filter;
121 if (ioptions.compaction_filter_factory != nullptr &&
122 ioptions.compaction_filter_factory->ShouldFilterTableFileCreation(
123 tboptions.reason)) {
124 CompactionFilter::Context context;
125 context.is_full_compaction = false;
126 context.is_manual_compaction = false;
127 context.column_family_id = tboptions.column_family_id;
128 context.reason = tboptions.reason;
129 compaction_filter =
130 ioptions.compaction_filter_factory->CreateCompactionFilter(context);
131 if (compaction_filter != nullptr &&
132 !compaction_filter->IgnoreSnapshots()) {
133 s.PermitUncheckedError();
134 return Status::NotSupported(
135 "CompactionFilter::IgnoreSnapshots() = false is not supported "
136 "anymore.");
137 }
138 }
139
140 TableBuilder* builder;
141 std::unique_ptr<WritableFileWriter> file_writer;
142 {
143 std::unique_ptr<FSWritableFile> file;
144#ifndef NDEBUG
145 bool use_direct_writes = file_options.use_direct_writes;
146 TEST_SYNC_POINT_CALLBACK("BuildTable:create_file", &use_direct_writes);
147#endif // !NDEBUG
148 IOStatus io_s = NewWritableFile(fs, fname, &file, file_options);
149 assert(s.ok());
150 s = io_s;
151 if (io_status->ok()) {
152 *io_status = io_s;
153 }
154 if (!s.ok()) {
155 EventHelpers::LogAndNotifyTableFileCreationFinished(
156 event_logger, ioptions.listeners, dbname,
157 tboptions.column_family_name, fname, job_id, meta->fd,
158 kInvalidBlobFileNumber, tp, tboptions.reason, s, file_checksum,
159 file_checksum_func_name);
160 return s;
161 }
162
163 table_file_created = true;
164 FileTypeSet tmp_set = ioptions.checksum_handoff_file_types;
165 file->SetIOPriority(tboptions.write_options.rate_limiter_priority);
166 file->SetWriteLifeTimeHint(write_hint);
167 file_writer.reset(new WritableFileWriter(
168 std::move(file), fname, file_options, ioptions.clock, io_tracer,
169 ioptions.stats, Histograms::SST_WRITE_MICROS, ioptions.listeners,
170 ioptions.file_checksum_gen_factory.get(),
171 tmp_set.Contains(FileType::kTableFile), false));
172
173 builder = NewTableBuilder(tboptions, file_writer.get());
174 }
175
176 auto ucmp = tboptions.internal_comparator.user_comparator();
177 MergeHelper merge(
178 env, ucmp, ioptions.merge_operator.get(), compaction_filter.get(),
179 ioptions.logger, true /* internal key corruption is not ok */,
180 snapshots.empty() ? 0 : snapshots.back(), snapshot_checker);
181
182 std::unique_ptr<BlobFileBuilder> blob_file_builder(
183 (mutable_cf_options.enable_blob_files &&
184 tboptions.level_at_creation >=
185 mutable_cf_options.blob_file_starting_level &&
186 blob_file_additions)
187 ? new BlobFileBuilder(
188 versions, fs, &ioptions, &mutable_cf_options, &file_options,
189 &(tboptions.write_options), tboptions.db_id,
190 tboptions.db_session_id, job_id, tboptions.column_family_id,
191 tboptions.column_family_name, write_hint, io_tracer,
192 blob_callback, blob_creation_reason, &blob_file_paths,
193 blob_file_additions)
194 : nullptr);
195
196 const std::atomic<bool> kManualCompactionCanceledFalse{false};
197 CompactionIterator c_iter(
198 iter, ucmp, &merge, kMaxSequenceNumber, &snapshots, earliest_snapshot,
199 earliest_write_conflict_snapshot, job_snapshot, snapshot_checker, env,
200 ShouldReportDetailedTime(env, ioptions.stats),
201 true /* internal key corruption is not ok */, range_del_agg.get(),
202 blob_file_builder.get(), ioptions.allow_data_in_errors,
203 ioptions.enforce_single_del_contracts,
204 /*manual_compaction_canceled=*/kManualCompactionCanceledFalse,
205 true /* must_count_input_entries */,
206 /*compaction=*/nullptr, compaction_filter.get(),
207 /*shutting_down=*/nullptr, db_options.info_log, full_history_ts_low);
208
209 SequenceNumber smallest_preferred_seqno = kMaxSequenceNumber;
210 std::string key_after_flush_buf;
211 std::string value_buf;
212 c_iter.SeekToFirst();
213 for (; c_iter.Valid(); c_iter.Next()) {
214 const Slice& key = c_iter.key();
215 const Slice& value = c_iter.value();
216 ParsedInternalKey ikey = c_iter.ikey();
217 key_after_flush_buf.assign(key.data(), key.size());
218 Slice key_after_flush = key_after_flush_buf;
219 Slice value_after_flush = value;
220
221 if (ikey.type == kTypeValuePreferredSeqno) {
222 auto [unpacked_value, unix_write_time] =
223 ParsePackedValueWithWriteTime(value);
224 SequenceNumber preferred_seqno =
225 seqno_to_time_mapping
226 ? seqno_to_time_mapping->GetProximalSeqnoBeforeTime(
227 unix_write_time)
228 : kMaxSequenceNumber;
229 if (preferred_seqno < ikey.sequence) {
230 value_after_flush =
231 PackValueAndSeqno(unpacked_value, preferred_seqno, &value_buf);
232 smallest_preferred_seqno =
233 std::min(smallest_preferred_seqno, preferred_seqno);
234 } else {
235 // Cannot get a useful preferred seqno, convert it to a kTypeValue.
236 UpdateInternalKey(&key_after_flush_buf, ikey.sequence, kTypeValue);
237 ikey = ParsedInternalKey(ikey.user_key, ikey.sequence, kTypeValue);
238 key_after_flush = key_after_flush_buf;
239 value_after_flush = ParsePackedValueForValue(value);
240 }
241 }
242
243 // Generate a rolling 64-bit hash of the key and values
244 // Note :
245 // Here "key" integrates 'sequence_number'+'kType'+'user key'.
246 s = output_validator.Add(key_after_flush, value_after_flush);
247 if (!s.ok()) {
248 break;
249 }
250 builder->Add(key_after_flush, value_after_flush);
251
252 s = meta->UpdateBoundaries(key_after_flush, value_after_flush,
253 ikey.sequence, ikey.type);
254 if (!s.ok()) {
255 break;
256 }
257
258 // TODO(noetzli): Update stats after flush, too.
259 // TODO(hx235): Replace `rate_limiter_priority` with `io_activity` for
260 // flush IO in repair when we have an `Env::IOActivity` enum for it
261 if ((tboptions.write_options.io_activity == Env::IOActivity::kFlush ||
262 tboptions.write_options.io_activity == Env::IOActivity::kDBOpen ||
263 tboptions.write_options.rate_limiter_priority == Env::IO_HIGH) &&
264 IOSTATS(bytes_written) >= kReportFlushIOStatsEvery) {
265 ThreadStatusUtil::SetThreadOperationProperty(
266 ThreadStatus::FLUSH_BYTES_WRITTEN, IOSTATS(bytes_written));
267 }
268 }
269 if (!s.ok()) {
270 c_iter.status().PermitUncheckedError();
271 } else if (!c_iter.status().ok()) {
272 s = c_iter.status();
273 }
274
275 if (s.ok()) {
276 auto range_del_it = range_del_agg->NewIterator();
277 Slice last_tombstone_start_user_key{};
278 for (range_del_it->SeekToFirst(); range_del_it->Valid();
279 range_del_it->Next()) {
280 auto tombstone = range_del_it->Tombstone();
281 std::pair<InternalKey, Slice> kv = tombstone.Serialize();
282 builder->Add(kv.first.Encode(), kv.second);
283 InternalKey tombstone_end = tombstone.SerializeEndKey();
284 meta->UpdateBoundariesForRange(kv.first, tombstone_end, tombstone.seq_,
285 tboptions.internal_comparator);
286 if (version) {
287 if (last_tombstone_start_user_key.empty() ||
288 ucmp->CompareWithoutTimestamp(last_tombstone_start_user_key,
289 range_del_it->start_key()) < 0) {
290 SizeApproximationOptions approx_opts;
291 approx_opts.files_size_error_margin = 0.1;
292 meta->compensated_range_deletion_size += versions->ApproximateSize(
293 approx_opts, tboptions.read_options, version, kv.first.Encode(),
294 tombstone_end.Encode(), 0 /* start_level */, -1 /* end_level */,
295 TableReaderCaller::kFlush);
296 }
297 last_tombstone_start_user_key = range_del_it->start_key();
298 }
299 }
300 }
301
302 TEST_SYNC_POINT("BuildTable:BeforeFinishBuildTable");
303 const bool empty = builder->IsEmpty();
304 if (num_input_entries != nullptr) {
305 assert(c_iter.HasNumInputEntryScanned());
306 *num_input_entries =
307 c_iter.NumInputEntryScanned() + num_unfragmented_tombstones;
308 }
309 if (!s.ok() || empty) {
310 builder->Abandon();
311 } else {
312 SeqnoToTimeMapping relevant_mapping;
313 if (seqno_to_time_mapping) {
314 relevant_mapping.CopyFromSeqnoRange(
315 *seqno_to_time_mapping,
316 std::min(meta->fd.smallest_seqno, smallest_preferred_seqno),
317 meta->fd.largest_seqno);
318 relevant_mapping.SetCapacity(kMaxSeqnoTimePairsPerSST);
319 relevant_mapping.Enforce(tboptions.file_creation_time);
320 }
321 builder->SetSeqnoTimeTableProperties(
322 relevant_mapping,
323 ioptions.compaction_style == CompactionStyle::kCompactionStyleFIFO
324 ? meta->file_creation_time
325 : meta->oldest_ancester_time);
326 s = builder->Finish();
327 }
328 if (io_status->ok()) {
329 *io_status = builder->io_status();
330 }
331
332 if (s.ok() && !empty) {
333 uint64_t file_size = builder->FileSize();
334 meta->fd.file_size = file_size;
335 meta->tail_size = builder->GetTailSize();
336 meta->marked_for_compaction = builder->NeedCompact();
337 meta->user_defined_timestamps_persisted =
338 ioptions.persist_user_defined_timestamps;
339 assert(meta->fd.GetFileSize() > 0);
340 tp = builder
341 ->GetTableProperties(); // refresh now that builder is finished
342 if (memtable_payload_bytes != nullptr &&
343 memtable_garbage_bytes != nullptr) {
344 const CompactionIterationStats& ci_stats = c_iter.iter_stats();
345 uint64_t total_payload_bytes = ci_stats.total_input_raw_key_bytes +
346 ci_stats.total_input_raw_value_bytes +
347 total_tombstone_payload_bytes;
348 uint64_t total_payload_bytes_written =
349 (tp.raw_key_size + tp.raw_value_size);
350 // Prevent underflow, which may still happen at this point
351 // since we only support inserts, deletes, and deleteRanges.
352 if (total_payload_bytes_written <= total_payload_bytes) {
353 *memtable_payload_bytes = total_payload_bytes;
354 *memtable_garbage_bytes =
355 total_payload_bytes - total_payload_bytes_written;
356 } else {
357 *memtable_payload_bytes = 0;
358 *memtable_garbage_bytes = 0;
359 }
360 }
361 if (table_properties) {
362 *table_properties = tp;
363 }
364 }
365 delete builder;
366
367 // Finish and check for file errors
368 TEST_SYNC_POINT("BuildTable:BeforeSyncTable");
369 IOOptions opts;
370 *io_status =
371 WritableFileWriter::PrepareIOOptions(tboptions.write_options, opts);
372 if (s.ok() && io_status->ok() && !empty) {
373 StopWatch sw(ioptions.clock, ioptions.stats, TABLE_SYNC_MICROS);
374 *io_status = file_writer->Sync(opts, ioptions.use_fsync);
375 }
376 TEST_SYNC_POINT("BuildTable:BeforeCloseTableFile");
377 if (s.ok() && io_status->ok() && !empty) {
378 *io_status = file_writer->Close(opts);
379 }
380 if (s.ok() && io_status->ok() && !empty) {
381 // Add the checksum information to file metadata.
382 meta->file_checksum = file_writer->GetFileChecksum();
383 meta->file_checksum_func_name = file_writer->GetFileChecksumFuncName();
384 file_checksum = meta->file_checksum;
385 file_checksum_func_name = meta->file_checksum_func_name;
386 // Set unique_id only if db_id and db_session_id exist
387 if (!tboptions.db_id.empty() && !tboptions.db_session_id.empty()) {
388 if (!GetSstInternalUniqueId(tboptions.db_id, tboptions.db_session_id,
389 meta->fd.GetNumber(), &(meta->unique_id))
390 .ok()) {
391 // if failed to get unique id, just set it Null
392 meta->unique_id = kNullUniqueId64x2;
393 }
394 }
395 }
396
397 if (s.ok()) {
398 s = *io_status;
399 }
400
401 // TODO(yuzhangyu): handle the key copy in the blob when ts should be
402 // stripped.
403 if (blob_file_builder) {
404 if (s.ok()) {
405 s = blob_file_builder->Finish();
406 } else {
407 blob_file_builder->Abandon(s);
408 }
409 blob_file_builder.reset();
410 }
411
412 // TODO Also check the IO status when create the Iterator.
413
414 TEST_SYNC_POINT("BuildTable:BeforeOutputValidation");
415 if (s.ok() && !empty) {
416 // Verify that the table is usable
417 // We set for_compaction to false and don't OptimizeForCompactionTableRead
418 // here because this is a special case after we finish the table building.
419 // No matter whether use_direct_io_for_flush_and_compaction is true,
420 // the goal is to cache it here for further user reads.
421 std::unique_ptr<InternalIterator> it(table_cache->NewIterator(
422 tboptions.read_options, file_options, tboptions.internal_comparator,
423 *meta, nullptr /* range_del_agg */, mutable_cf_options, nullptr,
424 (internal_stats == nullptr) ? nullptr
425 : internal_stats->GetFileReadHist(0),
426 TableReaderCaller::kFlush, /*arena=*/nullptr,
427 /*skip_filter=*/false, tboptions.level_at_creation,
428 MaxFileSizeForL0MetaPin(mutable_cf_options),
429 /*smallest_compaction_key=*/nullptr,
430 /*largest_compaction_key*/ nullptr,
431 /*allow_unprepared_value*/ false));
432 s = it->status();
433 if (s.ok() && paranoid_file_checks) {
434 OutputValidator file_validator(tboptions.internal_comparator,
435 /*enable_hash=*/true);
436 for (it->SeekToFirst(); it->Valid(); it->Next()) {
437 // Generate a rolling 64-bit hash of the key and values
438 file_validator.Add(it->key(), it->value()).PermitUncheckedError();
439 }
440 s = it->status();
441 if (s.ok() && !output_validator.CompareValidator(file_validator)) {
442 s = Status::Corruption("Paranoid checksums do not match");
443 }
444 }
445 }
446 }
447
448 // Check for input iterator errors
449 if (!iter->status().ok()) {
450 s = iter->status();
451 }
452
453 if (!s.ok() || meta->fd.GetFileSize() == 0) {
454 TEST_SYNC_POINT("BuildTable:BeforeDeleteFile");
455
456 constexpr IODebugContext* dbg = nullptr;
457
458 if (table_file_created) {
459 IOOptions opts;
460 Status prepare =
461 WritableFileWriter::PrepareIOOptions(tboptions.write_options, opts);
462 if (prepare.ok()) {
463 // FIXME: track file for "slow" deletion, e.g. into the
464 // VersionSet::obsolete_files_ pipeline
465 Status ignored = fs->DeleteFile(fname, opts, dbg);
466 ignored.PermitUncheckedError();
467 }
468 // Ensure we don't leak table cache entries when throwing away output
469 // files. (The usual logic in PurgeObsoleteFiles is not applicable because
470 // this function deletes the obsolete file itself, while they should
471 // probably go into the VersionSet::obsolete_files_ pipeline.)
472 TableCache::ReleaseObsolete(table_cache->get_cache().get(),
473 meta->fd.GetNumber(), nullptr /*handle*/,
474 mutable_cf_options.uncache_aggressiveness);
475 }
476
477 assert(blob_file_additions || blob_file_paths.empty());
478
479 if (blob_file_additions) {
480 for (const std::string& blob_file_path : blob_file_paths) {
481 Status ignored = DeleteDBFile(&db_options, blob_file_path, dbname,
482 /*force_bg=*/false, /*force_fg=*/false);
483 ignored.PermitUncheckedError();
484 TEST_SYNC_POINT("BuildTable::AfterDeleteFile");
485 }
486 }
487 }
488
489 Status status_for_listener = s;
490 if (meta->fd.GetFileSize() == 0) {
491 fname = "(nil)";
492 if (s.ok()) {
493 status_for_listener = Status::Aborted("Empty SST file not kept");
494 }
495 }
496 // Output to event logger and fire events.
497 EventHelpers::LogAndNotifyTableFileCreationFinished(
498 event_logger, ioptions.listeners, dbname, tboptions.column_family_name,
499 fname, job_id, meta->fd, meta->oldest_blob_file_number, tp,
500 tboptions.reason, status_for_listener, file_checksum,
501 file_checksum_func_name);
502
503 return s;
504}
