20// This structure is used to store the DB recovery context. The context is
21// the information that related to the recover actions. For example, it contains
22// FlushReason, which tells the flush job why this flush is called.
23struct DBRecoverContext {
24 FlushReason flush_reason;
25 bool flush_after_recovery;
26
27 DBRecoverContext()
28 : flush_reason(FlushReason::kErrorRecovery),
29 flush_after_recovery(false) {}
30 DBRecoverContext(FlushReason reason)
31 : flush_reason(reason), flush_after_recovery(false) {}
32};
