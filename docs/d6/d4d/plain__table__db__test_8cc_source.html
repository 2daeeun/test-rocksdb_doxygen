43TEST_F(PlainTableKeyDecoderTest, ReadNonMmap) {
44 Random rnd(301);
45 const uint32_t kLength = 2222;
46 std::string tmp = rnd.RandomString(kLength);
47 Slice contents(tmp);
48 test::StringSource* string_source =
49 new test::StringSource(contents, 0, false);
50 std::unique_ptr<FSRandomAccessFile> holder(string_source);
51 std::unique_ptr<RandomAccessFileReader> file_reader(
52 new RandomAccessFileReader(std::move(holder), "test"));
53 std::unique_ptr<PlainTableReaderFileInfo> file_info(
54 new PlainTableReaderFileInfo(std::move(file_reader), EnvOptions(),
55 kLength));
56
57 {
58 PlainTableFileReader reader(file_info.get());
59
60 const uint32_t kReadSize = 77;
61 for (uint32_t pos = 0; pos < kLength; pos += kReadSize) {
62 uint32_t read_size = std::min(kLength - pos, kReadSize);
63 Slice out;
64 ASSERT_TRUE(reader.Read(pos, read_size, &out));
65 ASSERT_EQ(0, out.compare(tmp.substr(pos, read_size)));
66 }
67
68 ASSERT_LT(uint32_t(string_source->total_reads()), kLength / kReadSize / 2);
69 }
70
71 std::vector<std::vector<std::pair<uint32_t, uint32_t>>> reads = {
72 {{600, 30}, {590, 30}, {600, 20}, {600, 40}},
73 {{800, 20}, {100, 20}, {500, 20}, {1500, 20}, {100, 20}, {80, 20}},
74 {{1000, 20}, {500, 20}, {1000, 50}},
75 {{1000, 20}, {500, 20}, {500, 20}},
76 {{1000, 20}, {500, 20}, {200, 20}, {500, 20}},
77 {{1000, 20}, {500, 20}, {200, 20}, {1000, 50}},
78 {{600, 500}, {610, 20}, {100, 20}},
79 {{500, 100}, {490, 100}, {550, 50}},
80 };
81
82 std::vector<int> num_file_reads = {2, 6, 2, 2, 4, 3, 2, 2};
83
84 for (size_t i = 0; i < reads.size(); i++) {
85 string_source->set_total_reads(0);
86 PlainTableFileReader reader(file_info.get());
87 for (auto p : reads[i]) {
88 Slice out;
89 ASSERT_TRUE(reader.Read(p.first, p.second, &out));
90 ASSERT_EQ(0, out.compare(tmp.substr(p.first, p.second)));
91 }
92 ASSERT_EQ(num_file_reads[i], string_source->total_reads());
93 }
