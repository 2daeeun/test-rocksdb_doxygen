23HistogramBucketMapper::HistogramBucketMapper() {
24 // If you change this, you also need to change
25 // size of array buckets_ in HistogramImpl
26 bucketValues_ = {1, 2};
27 double bucket_val = static_cast<double>(bucketValues_.back());
28 while ((bucket_val = 1.5 * bucket_val) <=
29 static_cast<double>(std::numeric_limits<uint64_t>::max())) {
30 bucketValues_.push_back(static_cast<uint64_t>(bucket_val));
31 // Extracts two most significant digits to make histogram buckets more
32 // human-readable. E.g., 172 becomes 170.
33 uint64_t pow_of_ten = 1;
34 while (bucketValues_.back() / 10 > 10) {
35 bucketValues_.back() /= 10;
36 pow_of_ten *= 10;
37 }
38 bucketValues_.back() *= pow_of_ten;
39 }
40 maxBucketValue_ = bucketValues_.back();
41 minBucketValue_ = bucketValues_.front();
42}

44size_t HistogramBucketMapper::IndexForValue(const uint64_t value) const {
45 auto beg = bucketValues_.begin();
46 auto end = bucketValues_.end();
47 if (value >= maxBucketValue_) {
48 return end - beg - 1; // bucketValues_.size() - 1
49 } else {
50 return std::lower_bound(beg, end, value) - beg;
51 }
52}

63void HistogramStat::Clear() {
64 min_.store(bucketMapper.LastValue(), std::memory_order_relaxed);
65 max_.store(0, std::memory_order_relaxed);
66 num_.store(0, std::memory_order_relaxed);
67 sum_.store(0, std::memory_order_relaxed);
68 sum_squares_.store(0, std::memory_order_relaxed);
69 for (unsigned int b = 0; b < num_buckets_; b++) {
70 buckets_[b].store(0, std::memory_order_relaxed);
71 }
72}

76void HistogramStat::Add(uint64_t value) {
77 // This function is designed to be lock free, as it's in the critical path
78 // of any operation. Each individual value is atomic and the order of updates
79 // by concurrent threads is tolerable.
80 const size_t index = bucketMapper.IndexForValue(value);
81 assert(index < num_buckets_);
82 buckets_[index].store(buckets_[index].load(std::memory_order_relaxed) + 1,
83 std::memory_order_relaxed);
84
85 uint64_t old_min = min();
86 if (value < old_min) {
87 min_.store(value, std::memory_order_relaxed);
88 }
89
90 uint64_t old_max = max();
91 if (value > old_max) {
92 max_.store(value, std::memory_order_relaxed);
93 }
94
95 num_.store(num_.load(std::memory_order_relaxed) + 1,
96 std::memory_order_relaxed);
97 sum_.store(sum_.load(std::memory_order_relaxed) + value,
98 std::memory_order_relaxed);
99 sum_squares_.store(
100 sum_squares_.load(std::memory_order_relaxed) + value * value,
101 std::memory_order_relaxed);
102}

104void HistogramStat::Merge(const HistogramStat& other) {
105 // This function needs to be performned with the outer lock acquired
106 // However, atomic operation on every member is still need, since Add()
107 // requires no lock and value update can still happen concurrently
108 uint64_t old_min = min();
109 uint64_t other_min = other.min();
110 while (other_min < old_min &&
111 !min_.compare_exchange_weak(old_min, other_min)) {
112 }
113
114 uint64_t old_max = max();
115 uint64_t other_max = other.max();
116 while (other_max > old_max &&
117 !max_.compare_exchange_weak(old_max, other_max)) {
118 }
119
120 num_.fetch_add(other.num(), std::memory_order_relaxed);
121 sum_.fetch_add(other.sum(), std::memory_order_relaxed);
122 sum_squares_.fetch_add(other.sum_squares(), std::memory_order_relaxed);
123 for (unsigned int b = 0; b < num_buckets_; b++) {
124 buckets_[b].fetch_add(other.bucket_at(b), std::memory_order_relaxed);
125 }
126}

130double HistogramStat::Percentile(double p) const {
131 double threshold = num() * (p / 100.0);
132 uint64_t cumulative_sum = 0;
133 for (unsigned int b = 0; b < num_buckets_; b++) {
134 uint64_t bucket_value = bucket_at(b);
135 cumulative_sum += bucket_value;
136 if (cumulative_sum >= threshold) {
137 // Scale linearly within this bucket
138 uint64_t left_point = (b == 0) ? 0 : bucketMapper.BucketLimit(b - 1);
139 uint64_t right_point = bucketMapper.BucketLimit(b);
140 uint64_t left_sum = cumulative_sum - bucket_value;
141 uint64_t right_sum = cumulative_sum;
142 double pos = 0;
143 uint64_t right_left_diff = right_sum - left_sum;
144 if (right_left_diff != 0) {
145 pos = (threshold - left_sum) / right_left_diff;
146 }
147 double r = left_point + (right_point - left_point) * pos;
148 uint64_t cur_min = min();
149 uint64_t cur_max = max();
150 if (r < cur_min) {
151 r = static_cast<double>(cur_min);
152 }
153 if (r > cur_max) {
154 r = static_cast<double>(cur_max);
155 }
156 return r;
157 }
158 }
159 return static_cast<double>(max());
160}

162double HistogramStat::Average() const {
163 uint64_t cur_num = num();
164 uint64_t cur_sum = sum();
165 if (cur_num == 0) {
166 return 0;
167 }
168 return static_cast<double>(cur_sum) / static_cast<double>(cur_num);
169}

171double HistogramStat::StandardDeviation() const {
172 double cur_num =
173 static_cast<double>(num()); // Use double to avoid integer overflow
174 double cur_sum = static_cast<double>(sum());
175 double cur_sum_squares = static_cast<double>(sum_squares());
176 if (cur_num == 0.0) {
177 return 0.0;
178 }
179 double variance =
180 (cur_sum_squares * cur_num - cur_sum * cur_sum) / (cur_num * cur_num);
181 return std::sqrt(std::max(variance, 0.0));
182}

184std::string HistogramStat::ToString() const {
185 uint64_t cur_num = num();
186 std::string r;
187 char buf[1650];
188 snprintf(buf, sizeof(buf), "Count: %" PRIu64 " Average: %.4f StdDev: %.2f\n",
189 cur_num, Average(), StandardDeviation());
190 r.append(buf);
191 snprintf(buf, sizeof(buf),
192 "Min: %" PRIu64 " Median: %.4f Max: %" PRIu64 "\n",
193 (cur_num == 0 ? 0 : min()), Median(), (cur_num == 0 ? 0 : max()));
194 r.append(buf);
195 snprintf(buf, sizeof(buf),
196 "Percentiles: "
197 "P50: %.2f P75: %.2f P99: %.2f P99.9: %.2f P99.99: %.2f\n",
198 Percentile(50), Percentile(75), Percentile(99), Percentile(99.9),
199 Percentile(99.99));
200 r.append(buf);
201 r.append("------------------------------------------------------\n");
202 if (cur_num == 0) {
203 return r; // all buckets are empty
204 }
205 const double mult = 100.0 / cur_num;
206 uint64_t cumulative_sum = 0;
207 for (unsigned int b = 0; b < num_buckets_; b++) {
208 uint64_t bucket_value = bucket_at(b);
209 if (bucket_value <= 0.0) {
210 continue;
211 }
212 cumulative_sum += bucket_value;
213 snprintf(buf, sizeof(buf),
214 "%c %7" PRIu64 ", %7" PRIu64 " ] %8" PRIu64 " %7.3f%% %7.3f%% ",
215 (b == 0) ? '[' : '(',
216 (b == 0) ? 0 : bucketMapper.BucketLimit(b - 1), // left
217 bucketMapper.BucketLimit(b), // right
218 bucket_value, // count
219 (mult * bucket_value), // percentage
220 (mult * cumulative_sum)); // cumulative percentage
221 r.append(buf);
222
223 // Add hash marks based on percentage; 20 marks for 100%.
224 size_t marks = static_cast<size_t>(mult * bucket_value / 5 + 0.5);
225 r.append(marks, '#');
226 r.push_back('\n');
227 }
228 return r;
229}
