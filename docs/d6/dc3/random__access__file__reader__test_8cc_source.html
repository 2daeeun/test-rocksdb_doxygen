33 void Write(const std::string& fname, const std::string& content) {
34 std::unique_ptr<FSWritableFile> f;
35 ASSERT_OK(fs_->NewWritableFile(Path(fname), FileOptions(), &f, nullptr));
36 ASSERT_OK(f->Append(content, IOOptions(), nullptr));
37 ASSERT_OK(f->Close(IOOptions(), nullptr));
38 }

40 void Read(const std::string& fname, const FileOptions& opts,
41 std::unique_ptr<RandomAccessFileReader>* reader) {
42 std::string fpath = Path(fname);
43 std::unique_ptr<FSRandomAccessFile> f;
44 ASSERT_OK(fs_->NewRandomAccessFile(fpath, opts, &f, nullptr));
45 reader->reset(new RandomAccessFileReader(std::move(f), fpath,
46 env_->GetSystemClock().get()));
47 }
48
49 void AssertResult(const std::string& content,
50 const std::vector<FSReadRequest>& reqs) {
51 for (const auto& r : reqs) {
52 ASSERT_OK(r.status);
53 ASSERT_EQ(r.len, r.result.size());
54 ASSERT_EQ(content.substr(r.offset, r.len), r.result.ToString());
55 }
56 }

68TEST_F(RandomAccessFileReaderTest, ReadDirectIO) {
69 std::string fname = "read-direct-io";
70 Random rand(0);
71 std::string content = rand.RandomString(kDefaultPageSize);
72 Write(fname, content);
73
74 FileOptions opts;
75 opts.use_direct_reads = true;
76 std::unique_ptr<RandomAccessFileReader> r;
77 Read(fname, opts, &r);
78 ASSERT_TRUE(r->use_direct_io());
79
80 const size_t page_size = r->file()->GetRequiredBufferAlignment();
81 size_t offset = page_size / 2;
82 size_t len = page_size / 3;
83 Slice result;
84 AlignedBuf buf;
85 for (Env::IOPriority rate_limiter_priority : {Env::IO_LOW, Env::IO_TOTAL}) {
86 IOOptions io_opts;
87 io_opts.rate_limiter_priority = rate_limiter_priority;
88 ASSERT_OK(r->Read(io_opts, offset, len, &result, nullptr, &buf));
89 ASSERT_EQ(result.ToString(), content.substr(offset, len));
90 }

93TEST_F(RandomAccessFileReaderTest, MultiReadDirectIO) {
94 std::vector<FSReadRequest> aligned_reqs;
95 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
96 "RandomAccessFileReader::MultiRead:AlignedReqs", [&](void* reqs) {
97 // Copy reqs, since it's allocated on stack inside MultiRead, which will
98 // be deallocated after MultiRead returns.
99 size_t i = 0;
100 aligned_reqs.resize(
101 (*reinterpret_cast<std::vector<FSReadRequest>*>(reqs)).size());
102 for (auto& req :
103 (*reinterpret_cast<std::vector<FSReadRequest>*>(reqs))) {
104 aligned_reqs[i].offset = req.offset;
105 aligned_reqs[i].len = req.len;
106 aligned_reqs[i].result = req.result;
107 aligned_reqs[i].status = req.status;
108 aligned_reqs[i].scratch = req.scratch;
109 i++;
110 }
111 });
112 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
113
114 // Creates a file with 3 pages.
115 std::string fname = "multi-read-direct-io";
116 Random rand(0);
117 std::string content = rand.RandomString(3 * kDefaultPageSize);
118 Write(fname, content);
119
120 FileOptions opts;
121 opts.use_direct_reads = true;
122 std::unique_ptr<RandomAccessFileReader> r;
123 Read(fname, opts, &r);
124 ASSERT_TRUE(r->use_direct_io());
125
126 const size_t page_size = r->file()->GetRequiredBufferAlignment();
127
128 {
129 // Reads 2 blocks in the 1st page.
130 // The results should be SharedSlices of the same underlying buffer.
131 //
132 // Illustration (each x is a 1/4 page)
133 // First page: xxxx
134 // 1st block: x
135 // 2nd block: xx
136 FSReadRequest r0;
137 r0.offset = 0;
138 r0.len = page_size / 4;
139 r0.scratch = nullptr;
140
141 FSReadRequest r1;
142 r1.offset = page_size / 2;
143 r1.len = page_size / 2;
144 r1.scratch = nullptr;
145
146 std::vector<FSReadRequest> reqs;
147 reqs.push_back(std::move(r0));
148 reqs.push_back(std::move(r1));
149 AlignedBuf aligned_buf;
150 ASSERT_OK(
151 r->MultiRead(IOOptions(), reqs.data(), reqs.size(), &aligned_buf));
152
153 AssertResult(content, reqs);
154
155 // Reads the first page internally.
156 ASSERT_EQ(aligned_reqs.size(), 1);
157 const FSReadRequest& aligned_r = aligned_reqs[0];
158 ASSERT_OK(aligned_r.status);
159 ASSERT_EQ(aligned_r.offset, 0);
160 ASSERT_EQ(aligned_r.len, page_size);
161 }
162
163 {
164 // Reads 3 blocks:
165 // 1st block in the 1st page;
166 // 2nd block from the middle of the 1st page to the middle of the 2nd page;
167 // 3rd block in the 2nd page.
168 // The results should be SharedSlices of the same underlying buffer.
169 //
170 // Illustration (each x is a 1/4 page)
171 // 2 pages: xxxxxxxx
172 // 1st block: x
173 // 2nd block: xxxx
174 // 3rd block: x
175 FSReadRequest r0;
176 r0.offset = 0;
177 r0.len = page_size / 4;
178 r0.scratch = nullptr;
179
180 FSReadRequest r1;
181 r1.offset = page_size / 2;
182 r1.len = page_size;
183 r1.scratch = nullptr;
184
185 FSReadRequest r2;
186 r2.offset = 2 * page_size - page_size / 4;
187 r2.len = page_size / 4;
188 r2.scratch = nullptr;
189
190 std::vector<FSReadRequest> reqs;
191 reqs.push_back(std::move(r0));
192 reqs.push_back(std::move(r1));
193 reqs.push_back(std::move(r2));
194 AlignedBuf aligned_buf;
195 ASSERT_OK(
196 r->MultiRead(IOOptions(), reqs.data(), reqs.size(), &aligned_buf));
197
198 AssertResult(content, reqs);
199
200 // Reads the first two pages in one request internally.
201 ASSERT_EQ(aligned_reqs.size(), 1);
202 const FSReadRequest& aligned_r = aligned_reqs[0];
203 ASSERT_OK(aligned_r.status);
204 ASSERT_EQ(aligned_r.offset, 0);
205 ASSERT_EQ(aligned_r.len, 2 * page_size);
206 }
207
208 {
209 // Reads 3 blocks:
210 // 1st block in the middle of the 1st page;
211 // 2nd block in the middle of the 2nd page;
212 // 3rd block in the middle of the 3rd page.
213 // The results should be SharedSlices of the same underlying buffer.
214 //
215 // Illustration (each x is a 1/4 page)
216 // 3 pages: xxxxxxxxxxxx
217 // 1st block: xx
218 // 2nd block: xx
219 // 3rd block: xx
220 FSReadRequest r0;
221 r0.offset = page_size / 4;
222 r0.len = page_size / 2;
223 r0.scratch = nullptr;
224
225 FSReadRequest r1;
226 r1.offset = page_size + page_size / 4;
227 r1.len = page_size / 2;
228 r1.scratch = nullptr;
229
230 FSReadRequest r2;
231 r2.offset = 2 * page_size + page_size / 4;
232 r2.len = page_size / 2;
233 r2.scratch = nullptr;
234
235 std::vector<FSReadRequest> reqs;
236 reqs.push_back(std::move(r0));
237 reqs.push_back(std::move(r1));
238 reqs.push_back(std::move(r2));
239 AlignedBuf aligned_buf;
240 ASSERT_OK(
241 r->MultiRead(IOOptions(), reqs.data(), reqs.size(), &aligned_buf));
242
243 AssertResult(content, reqs);
244

251 }

255 // 1st block in the middle of the 1st page;
256 // 2nd block in the middle of the 3rd page.

259 // Illustration (each x is a 1/4 page)

285 const FSReadRequest& aligned_r1 = aligned_reqs[1];
286 ASSERT_OK(aligned_r0.status);
287 ASSERT_EQ(aligned_r0.offset, 0);
288 ASSERT_EQ(aligned_r0.len, page_size);
