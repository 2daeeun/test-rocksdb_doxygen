20TEST_F(FileNameTest, Parse) {
21 Slice db;
22 FileType type;
23 uint64_t number;
24
25 char kDefautInfoLogDir = 1;
26 char kDifferentInfoLogDir = 2;
27 char kNoCheckLogDir = 4;
28 char kAllMode = kDefautInfoLogDir | kDifferentInfoLogDir | kNoCheckLogDir;
29
30 // Successful parses
31 static struct {
32 const char* fname;
33 uint64_t number;
34 FileType type;
35 char mode;
36 } cases[] = {
37 {"100.log", 100, kWalFile, kAllMode},
38 {"0.log", 0, kWalFile, kAllMode},
39 {"0.sst", 0, kTableFile, kAllMode},
40 {"CURRENT", 0, kCurrentFile, kAllMode},
41 {"LOCK", 0, kDBLockFile, kAllMode},
42 {"MANIFEST-2", 2, kDescriptorFile, kAllMode},
43 {"MANIFEST-7", 7, kDescriptorFile, kAllMode},
44 {"METADB-2", 2, kMetaDatabase, kAllMode},
45 {"METADB-7", 7, kMetaDatabase, kAllMode},
46 {"LOG", 0, kInfoLogFile, kDefautInfoLogDir},
47 {"LOG.old", 0, kInfoLogFile, kDefautInfoLogDir},
48 {"LOG.old.6688", 6688, kInfoLogFile, kDefautInfoLogDir},
49 {"rocksdb_dir_LOG", 0, kInfoLogFile, kDifferentInfoLogDir},
50 {"rocksdb_dir_LOG.old", 0, kInfoLogFile, kDifferentInfoLogDir},
51 {"rocksdb_dir_LOG.old.6688", 6688, kInfoLogFile, kDifferentInfoLogDir},
52 {"18446744073709551615.log", 18446744073709551615ull, kWalFile, kAllMode},
53 };
54 for (char mode : {kDifferentInfoLogDir, kDefautInfoLogDir, kNoCheckLogDir}) {
55 for (unsigned int i = 0; i < sizeof(cases) / sizeof(cases[0]); i++) {
56 InfoLogPrefix info_log_prefix(mode != kDefautInfoLogDir, "/rocksdb/dir");
57 if (cases[i].mode & mode) {
58 std::string f = cases[i].fname;
59 if (mode == kNoCheckLogDir) {
60 ASSERT_TRUE(ParseFileName(f, &number, &type)) << f;
61 } else {
62 ASSERT_TRUE(ParseFileName(f, &number, info_log_prefix.prefix, &type))
63 << f;
64 }
65 ASSERT_EQ(cases[i].type, type) << f;
66 ASSERT_EQ(cases[i].number, number) << f;
67 }
68 }
69 }

121TEST_F(FileNameTest, Construction) {
122 uint64_t number;
123 FileType type;
124 std::string fname;
125
126 fname = CurrentFileName("foo");
127 ASSERT_EQ("foo/", std::string(fname.data(), 4));
128 ASSERT_TRUE(ParseFileName(fname.c_str() + 4, &number, &type));
129 ASSERT_EQ(0U, number);
130 ASSERT_EQ(kCurrentFile, type);
131
132 fname = LockFileName("foo");
133 ASSERT_EQ("foo/", std::string(fname.data(), 4));
134 ASSERT_TRUE(ParseFileName(fname.c_str() + 4, &number, &type));
135 ASSERT_EQ(0U, number);
136 ASSERT_EQ(kDBLockFile, type);
137
138 fname = LogFileName("foo", 192);
139 ASSERT_EQ("foo/", std::string(fname.data(), 4));
140 ASSERT_TRUE(ParseFileName(fname.c_str() + 4, &number, &type));
141 ASSERT_EQ(192U, number);
142 ASSERT_EQ(kWalFile, type);
143
144 fname = TableFileName({DbPath("bar", 0)}, 200, 0);
145 std::string fname1 =
146 TableFileName({DbPath("foo", 0), DbPath("bar", 0)}, 200, 1);
147 ASSERT_EQ(fname, fname1);
148 ASSERT_EQ("bar/", std::string(fname.data(), 4));
149 ASSERT_TRUE(ParseFileName(fname.c_str() + 4, &number, &type));
150 ASSERT_EQ(200U, number);
151 ASSERT_EQ(kTableFile, type);
152
153 fname = DescriptorFileName("bar", 100);
154 ASSERT_EQ("bar/", std::string(fname.data(), 4));
155 ASSERT_TRUE(ParseFileName(fname.c_str() + 4, &number, &type));
156 ASSERT_EQ(100U, number);
157 ASSERT_EQ(kDescriptorFile, type);
158
159 fname = TempFileName("tmp", 999);
160 ASSERT_EQ("tmp/", std::string(fname.data(), 4));
161 ASSERT_TRUE(ParseFileName(fname.c_str() + 4, &number, &type));
162 ASSERT_EQ(999U, number);
163 ASSERT_EQ(kTempFile, type);
164
165 fname = MetaDatabaseName("met", 100);
166 ASSERT_EQ("met/", std::string(fname.data(), 4));
167 ASSERT_TRUE(ParseFileName(fname.c_str() + 4, &number, &type));
168 ASSERT_EQ(100U, number);
169 ASSERT_EQ(kMetaDatabase, type);
170}

172TEST_F(FileNameTest, NormalizePath) {
173 // No leading slash
174 const std::string sep = std::string(1, kFilePathSeparator);
175
176 std::string expected = "FOLDER" + sep + "filename.ext";
177 std::string given = "FOLDER" + sep + "filename.ext";
178
179 ASSERT_EQ(expected, NormalizePath(given));
180
181 // Two chars /a
182
183 expected = sep + "a";
184 given = expected;
185 ASSERT_EQ(expected, NormalizePath(given));
186
187 // Two chars a/
188 expected = "a" + sep;
189 given = expected;
190 ASSERT_EQ(expected, NormalizePath(given));
191
192 // Server only
193 expected = sep + sep + "a";
194 given = expected;
195 ASSERT_EQ(expected, NormalizePath(given));
196
197 // Two slashes after character
198 expected = "a" + sep;
199 given = "a" + sep + sep;
200
201 ASSERT_EQ(expected, NormalizePath(given));
202
203 // slash only /
204 expected = sep;
205 given = expected;
206 ASSERT_EQ(expected, NormalizePath(given));
207
208 // UNC only //
209 expected = sep;
210 given = sep + sep;
211
212 ASSERT_EQ(expected, NormalizePath(given));
213
214 // 3 slashesy //
215 expected = sep + sep;
216 given = sep + sep + sep;
217 ASSERT_EQ(expected, NormalizePath(given));
218
219 // 3 slashes //
220 expected = sep + sep + "a" + sep;
221 given = sep + sep + sep + "a" + sep;
222 ASSERT_EQ(expected, NormalizePath(given));
223
224 // 2 separators in the middle
225 expected = "a" + sep + "b";
226 given = "a" + sep + sep + "b";
227 ASSERT_EQ(expected, NormalizePath(given));
228
229 // UNC with duplicate slashes
230 expected = sep + sep + "SERVER" + sep + "a" + sep + "b" + sep + "c";
231 given = sep + sep + "SERVER" + sep + "a" + sep + sep + "b" + sep + "c";
232 ASSERT_EQ(expected, NormalizePath(given));
233}
