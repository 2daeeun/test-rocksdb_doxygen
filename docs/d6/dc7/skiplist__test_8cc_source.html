141// We want to make sure that with a single writer and multiple
142// concurrent readers (with no synchronization other than when a
143// reader's iterator is created), the reader always observes all the
144// data that was present in the skip list when the iterator was
145// constructor. Because insertions are happening concurrently, we may
146// also observe new values that were inserted since the iterator was
147// constructed, but we should never miss any values that were present
148// at iterator construction time.
149//
150// We generate multi-part keys:
151// <key,gen,hash>
152// where:
153// key is in range [0..K-1]
154// gen is a generation number for key
155// hash is hash(key,gen)
156//
157// The insertion code picks a random key, sets gen to be 1 + the last
158// generation number inserted for that key, and sets hash to Hash(key,gen).
159//
160// At the beginning of a read, we snapshot the last inserted
161// generation number for each key. We then iterate, including random
162// calls to Next() and Seek(). For every key we encounter, we
163// check that it is either expected given the initial snapshot or has
164// been concurrently added since the iterator started.
165class ConcurrentTest {
166 private:
167 static const uint32_t K = 4;
168
169 static uint64_t key(Key key) { return (key >> 40); }
170 static uint64_t gen(Key key) { return (key >> 8) & 0xffffffffu; }
171 static uint64_t hash(Key key) { return key & 0xff; }
172
173 static uint64_t HashNumbers(uint64_t k, uint64_t g) {
174 uint64_t data[2] = {k, g};
175 return Hash(reinterpret_cast<char*>(data), sizeof(data), 0);
176 }
177
178 static Key MakeKey(uint64_t k, uint64_t g) {
179 assert(sizeof(Key) == sizeof(uint64_t));
180 assert(k <= K); // We sometimes pass K to seek to the end of the skiplist
181 assert(g <= 0xffffffffu);
182 return ((k << 40) | (g << 8) | (HashNumbers(k, g) & 0xff));
183 }
184
185 static bool IsValidKey(Key k) {
186 return hash(k) == (HashNumbers(key(k), gen(k)) & 0xff);
187 }
188
189 static Key RandomTarget(Random* rnd) {
190 switch (rnd->Next() % 10) {
191 case 0:
192 // Seek to beginning
193 return MakeKey(0, 0);
194 case 1:
195 // Seek to end
196 return MakeKey(K, 0);
197 default:
198 // Seek to middle
199 return MakeKey(rnd->Next() % K, 0);
200 }
201 }
202
203 // Per-key generation
204 struct State {
205 std::atomic<int> generation[K];
206 void Set(int k, int v) {
207 generation[k].store(v, std::memory_order_release);
208 }
209 int Get(int k) { return generation[k].load(std::memory_order_acquire); }
210
211 State() {
212 for (unsigned int k = 0; k < K; k++) {
213 Set(k, 0);
214 }
215 }
216 };
217
218 // Current state of the test
219 State current_;
220
221 Arena arena_;
222
223 // SkipList is not protected by mu_. We just use a single writer
224 // thread to modify it.
225 SkipList<Key, TestComparator> list_;
226
227 public:
228 ConcurrentTest() : list_(TestComparator(), &arena_) {}
229
230 // REQUIRES: External synchronization
231 void WriteStep(Random* rnd) {
232 const uint32_t k = rnd->Next() % K;
233 const int g = current_.Get(k) + 1;
234 const Key new_key = MakeKey(k, g);
235 list_.Insert(new_key);
236 current_.Set(k, g);
237 }
238
239 void ReadStep(Random* rnd) {
240 // Remember the initial committed state of the skiplist.
241 State initial_state;
242 for (unsigned int k = 0; k < K; k++) {
243 initial_state.Set(k, current_.Get(k));
244 }
245
246 Key pos = RandomTarget(rnd);
247 SkipList<Key, TestComparator>::Iterator iter(&list_);
248 iter.Seek(pos);
249 while (true) {
250 Key current;
251 if (!iter.Valid()) {
252 current = MakeKey(K, 0);
253 } else {
254 current = iter.key();
255 ASSERT_TRUE(IsValidKey(current)) << current;
256 }
257 ASSERT_LE(pos, current) << "should not go backwards";
258
259 // Verify that everything in [pos,current) was not present in
260 // initial_state.
261 while (pos < current) {
262 ASSERT_LT(key(pos), K) << pos;
263
264 // Note that generation 0 is never inserted, so it is ok if
265 // <*,0,*> is missing.
266 ASSERT_TRUE((gen(pos) == 0U) ||
267 (gen(pos) > static_cast<uint64_t>(initial_state.Get(
268 static_cast<int>(key(pos))))))
269 << "key: " << key(pos) << "; gen: " << gen(pos)
270 << "; initgen: " << initial_state.Get(static_cast<int>(key(pos)));
271
272 // Advance to next key in the valid key space
273 if (key(pos) < key(current)) {
274 pos = MakeKey(key(pos) + 1, 0);
275 } else {
276 pos = MakeKey(key(pos), gen(pos) + 1);
277 }
278 }
279
280 if (!iter.Valid()) {
281 break;
282 }
283
284 if (rnd->Next() % 2) {
285 iter.Next();
286 pos = MakeKey(key(pos), gen(pos) + 1);
287 } else {
288 Key new_target = RandomTarget(rnd);
289 if (new_target > pos) {
290 pos = new_target;
291 iter.Seek(new_target);
292 }
293 }
294 }
295 }
296};

310class TestState {
311 public:
312 ConcurrentTest t_;
313 int seed_;
314 std::atomic<bool> quit_flag_;
315
316 enum ReaderState { STARTING, RUNNING, DONE };
317
318 explicit TestState(int s)
319 : seed_(s), quit_flag_(false), state_(STARTING), state_cv_(&mu_) {}
320
321 void Wait(ReaderState s) {
322 mu_.Lock();
323 while (state_ != s) {
324 state_cv_.Wait();
325 }
326 mu_.Unlock();
327 }
328
329 void Change(ReaderState s) {
330 mu_.Lock();
331 state_ = s;
332 state_cv_.Signal();
333 mu_.Unlock();
334 }
335
336 private:
337 port::Mutex mu_;
338 ReaderState state_;
339 port::CondVar state_cv_;
340};
341
342static void ConcurrentReader(void* arg) {
343 TestState* state = static_cast<TestState*>(arg);
344 Random rnd(state->seed_);
345 int64_t reads = 0;
346 state->Change(TestState::RUNNING);
347 while (!state->quit_flag_.load(std::memory_order_acquire)) {
348 state->t_.ReadStep(&rnd);
349 ++reads;
350 }
351 (void)reads;
352 state->Change(TestState::DONE);
353}
354
355static void RunConcurrent(int run) {
356 const int seed = test::RandomSeed() + (run * 100);
357 Random rnd(seed);
358 const int N = 1000;
359 const int kSize = 1000;
360 for (int i = 0; i < N; i++) {
361 if ((i % 100) == 0) {
362 fprintf(stderr, "Run %d of %d\n", i, N);
363 }
364 TestState state(seed + 1);
365 Env::Default()->SetBackgroundThreads(1);
366 Env::Default()->Schedule(ConcurrentReader, &state);
367 state.Wait(TestState::RUNNING);
368 for (int k = 0; k < kSize; k++) {
369 state.t_.WriteStep(&rnd);
370 }
371 state.quit_flag_.store(true, std::memory_order_release);
372 state.Wait(TestState::DONE);
373 }
374}
