196uint64_t MultiplyCheckOverflow(uint64_t op1, uint64_t op2) {
197 if (op1 == 0 || op2 == 0) {
198 return 0;
199 }
200 if (std::numeric_limits<uint64_t>::max() / op1 < op2) {
201 return op1;
202 }
203 return (op1 * op2);
204}

214void AnalyzerOptions::SparseCorrelationInput(const std::string& in_str) {
215 std::string cur = in_str;
216 if (cur.size() == 0) {
217 return;
218 }
219 while (!cur.empty()) {
220 if (cur.compare(0, 1, "[") != 0) {
221 fprintf(stderr, "Invalid correlation input: %s\n", in_str.c_str());
222 exit(1);
223 }
224 std::string opt1, opt2;
225 std::size_t split = cur.find_first_of(',');
226 if (split != std::string::npos) {
227 opt1 = cur.substr(1, split - 1);
228 } else {
229 fprintf(stderr, "Invalid correlation input: %s\n", in_str.c_str());
230 exit(1);
231 }
232 std::size_t end = cur.find_first_of(']');
233 if (end != std::string::npos) {
234 opt2 = cur.substr(split + 1, end - split - 1);
235 } else {
236 fprintf(stderr, "Invalid correlation input: %s\n", in_str.c_str());
237 exit(1);
238 }
239 cur = cur.substr(end + 1);
240
241 if (taOptToIndex.find(opt1) != taOptToIndex.end() &&
242 taOptToIndex.find(opt2) != taOptToIndex.end()) {
243 correlation_list.emplace_back(taOptToIndex[opt1], taOptToIndex[opt2]);
244 } else {
245 fprintf(stderr, "Invalid correlation input: %s\n", in_str.c_str());
246 exit(1);
247 }
248 }
249
250 int sequence = 0;
251 for (auto& it : correlation_list) {
252 correlation_map[it.first][it.second] = sequence;
253 sequence++;
254 }
255}

412 assert(header != nullptr);
413 std::string encoded_trace;
414 // Read the trace head

424 }

426 return Status::Corruption("Corrupted trace file. Incorrect magic.");

428
429 return s;

454// process the trace itself and redirect the trace content
455// to different operation type handler. With different race
456// format, this function can be changed
457Status TraceAnalyzer::StartProcessing() {
458 Status s;
459 Trace header;
460 s = ReadTraceHeader(&header);
461 if (!s.ok()) {
462 fprintf(stderr, "Cannot read the header\n");
463 return s;
464 }
465 // Set the default trace file version as version 0.2
466 int trace_file_version = 2;
467 s = TracerHelper::ParseTraceHeader(header, &trace_file_version, &db_version_);
468 if (!s.ok()) {
469 return s;
470 }
471 trace_create_time_ = header.ts;
472 if (FLAGS_output_time_series) {
473 time_series_start_ = header.ts;
474 }
475
476 Trace trace;
477 std::unique_ptr<TraceRecord> record;
478 while (s.ok()) {
479 trace.reset();
480 s = ReadTraceRecord(&trace);
481 if (!s.ok()) {
482 break;
483 }
484
485 end_time_ = trace.ts;
486 if (trace.type == kTraceEnd) {
487 break;
488 }
489 // Do not count TraceEnd (if there is one)
490 total_requests_++;
491
492 s = TracerHelper::DecodeTraceRecord(&trace, trace_file_version, &record);
493 if (s.IsNotSupported()) {
494 continue;
495 }
496 if (!s.ok()) {
497 return s;
498 }
499 s = record->Accept(this, nullptr);
500 if (!s.ok()) {
501 fprintf(stderr, "Cannot process the TraceRecord\n");
502 return s;
503 }
504 }
505 if (s.IsIncomplete()) {
506 // Fix it: Reaching eof returns Incomplete status at the moment.
507 return Status::OK();
508 }
509 return s;
510}

512// After the trace is processed by StartProcessing, the statistic data
513// is stored in the map or other in memory data structures. To get the
514// other statistic result such as key size distribution, value size
515// distribution, these data structures are re-processed here.
516Status TraceAnalyzer::MakeStatistics() {
517 int ret;
518 Status s;
519 for (int type = 0; type < kTaTypeNum; type++) {
520 if (!ta_[type].enabled) {
521 continue;
522 }
523 for (auto& stat : ta_[type].stats) {
524 stat.second.a_key_id = 0;
525 for (auto& record : stat.second.a_key_stats) {
526 record.second.key_id = stat.second.a_key_id;
527 stat.second.a_key_id++;
528 if (record.second.access_count <=
529 static_cast<uint64_t>(FLAGS_output_ignore_count)) {
530 continue;
531 }
532
533 // Generate the key access count distribution data
534 if (FLAGS_output_access_count_stats) {
535 if (stat.second.a_count_stats.find(record.second.access_count) ==
536 stat.second.a_count_stats.end()) {
537 stat.second.a_count_stats[record.second.access_count] = 1;
538 } else {
539 stat.second.a_count_stats[record.second.access_count]++;
540 }
541 }
542
543 // Generate the key size distribution data
544 if (FLAGS_output_key_distribution) {
545 if (stat.second.a_key_size_stats.find(record.first.size()) ==
546 stat.second.a_key_size_stats.end()) {
547 stat.second.a_key_size_stats[record.first.size()] = 1;
548 } else {
549 stat.second.a_key_size_stats[record.first.size()]++;
550 }
551 }
552
553 if (!FLAGS_print_correlation.empty()) {
554 s = MakeStatisticCorrelation(stat.second, record.second);
555 if (!s.ok()) {
556 return s;
557 }
558 }
559 }
560
561 // Output the prefix cut or the whole content of the accessed key space
562 if (FLAGS_output_key_stats || FLAGS_output_prefix_cut > 0) {
563 s = MakeStatisticKeyStatsOrPrefix(stat.second);
564 if (!s.ok()) {
565 return s;
566 }
567 }
568
569 // output the access count distribution
570 if (FLAGS_output_access_count_stats && stat.second.a_count_dist_f) {
571 for (auto& record : stat.second.a_count_stats) {
572 ret = snprintf(buffer_, sizeof(buffer_),
573 "access_count: %" PRIu64 " num: %" PRIu64 "\n",
574 record.first, record.second);
575 if (ret < 0) {
576 return Status::IOError("Format the output failed");
577 }
578 std::string printout(buffer_);
579 s = stat.second.a_count_dist_f->Append(printout);
580 if (!s.ok()) {
581 fprintf(stderr, "Write access count distribution file failed\n");
582 return s;
583 }
584 }
585 }
586
587 // find the medium of the key size
588 uint64_t k_count = 0;
589 bool get_mid = false;
590 for (auto& record : stat.second.a_key_size_stats) {
591 k_count += record.second;
592 if (!get_mid && k_count >= stat.second.a_key_mid) {
593 stat.second.a_key_mid = record.first;
594 get_mid = true;
595 }
596 if (FLAGS_output_key_distribution && stat.second.a_key_size_f) {
597 ret = snprintf(buffer_, sizeof(buffer_), "%" PRIu64 " %" PRIu64 "\n",
598 record.first, record.second);
599 if (ret < 0) {
600 return Status::IOError("Format output failed");
601 }
602 std::string printout(buffer_);
603 s = stat.second.a_key_size_f->Append(printout);
604 if (!s.ok()) {
605 fprintf(stderr, "Write key size distribution file failed\n");
606 return s;
607 }
608 }
609 }
610
611 // output the value size distribution
612 uint64_t v_begin = 0, v_end = 0, v_count = 0;
613 get_mid = false;
614 for (auto& record : stat.second.a_value_size_stats) {
615 v_begin = v_end;
616 v_end = (record.first + 1) * FLAGS_value_interval;
617 v_count += record.second;
618 if (!get_mid && v_count >= stat.second.a_count / 2) {
619 stat.second.a_value_mid = (v_begin + v_end) / 2;
620 get_mid = true;
621 }
622 if (FLAGS_output_value_distribution && stat.second.a_value_size_f &&
623 (type == TraceOperationType::kPut ||
624 type == TraceOperationType::kMerge)) {
625 ret = snprintf(buffer_, sizeof(buffer_),
626 "Number_of_value_size_between %" PRIu64 " and %" PRIu64
627 " is: %" PRIu64 "\n",
628 v_begin, v_end, record.second);
629 if (ret < 0) {
630 return Status::IOError("Format output failed");
631 }
632 std::string printout(buffer_);
633 s = stat.second.a_value_size_f->Append(printout);
634 if (!s.ok()) {
635 fprintf(stderr, "Write value size distribution file failed\n");
636 return s;
637 }
638 }
639 }
640 }
641 }
642
643 // Make the QPS statistics
644 if (FLAGS_output_qps_stats) {
645 s = MakeStatisticQPS();
646 if (!s.ok()) {
647 return s;
648 }
649 }
650
651 return Status::OK();
652}

656Status TraceAnalyzer::MakeStatisticKeyStatsOrPrefix(TraceStats& stats) {
657 int ret;
658 Status s;
659 std::string prefix = "0";
660 uint64_t prefix_access = 0;
661 uint64_t prefix_count = 0;
662 uint64_t prefix_succ_access = 0;
663 double prefix_ave_access = 0.0;
664 stats.a_succ_count = 0;
665 for (auto& record : stats.a_key_stats) {
666 // write the key access statistic file
667 if (!stats.a_key_f) {
668 return Status::IOError("Failed to open accessed_key_stats file.");
669 }
670 stats.a_succ_count += record.second.succ_count;
671 double succ_ratio = 0.0;
672 if (record.second.access_count > 0) {
673 succ_ratio = (static_cast<double>(record.second.succ_count)) /
674 record.second.access_count;
675 }
676 ret = snprintf(buffer_, sizeof(buffer_),
677 "%u %zu %" PRIu64 " %" PRIu64 " %f\n", record.second.cf_id,
678 record.second.value_size, record.second.key_id,
679 record.second.access_count, succ_ratio);
680 if (ret < 0) {
681 return Status::IOError("Format output failed");
682 }
683 std::string printout(buffer_);
684 s = stats.a_key_f->Append(printout);
685 if (!s.ok()) {
686 fprintf(stderr, "Write key access file failed\n");
687 return s;
688 }
689
690 // write the prefix cut of the accessed keys
691 if (FLAGS_output_prefix_cut > 0 && stats.a_prefix_cut_f) {
692 if (record.first.compare(0, FLAGS_output_prefix_cut, prefix) != 0) {
693 std::string prefix_out =
694 ROCKSDB_NAMESPACE::LDBCommand::StringToHex(prefix);
695 if (prefix_count == 0) {
696 prefix_ave_access = 0.0;
697 } else {
698 prefix_ave_access =
699 (static_cast<double>(prefix_access)) / prefix_count;
700 }
701 double prefix_succ_ratio = 0.0;
702 if (prefix_access > 0) {
703 prefix_succ_ratio =
704 (static_cast<double>(prefix_succ_access)) / prefix_access;
705 }
706 ret =
707 snprintf(buffer_, sizeof(buffer_),
708 "%" PRIu64 " %" PRIu64 " %" PRIu64 " %f %f %s\n",
709 record.second.key_id, prefix_access, prefix_count,
710 prefix_ave_access, prefix_succ_ratio, prefix_out.c_str());
711 if (ret < 0) {
712 return Status::IOError("Format output failed");
713 }
714 std::string pout(buffer_);
715 s = stats.a_prefix_cut_f->Append(pout);
716 if (!s.ok()) {
717 fprintf(stderr, "Write accessed key prefix file failed\n");
718 return s;
719 }
720
721 // make the top k statistic for the prefix
722 if (static_cast<int32_t>(stats.top_k_prefix_access.size()) <
723 FLAGS_print_top_k_access) {
724 stats.top_k_prefix_access.push(
725 std::make_pair(prefix_access, prefix_out));
726 } else {
727 if (prefix_access > stats.top_k_prefix_access.top().first) {
728 stats.top_k_prefix_access.pop();
729 stats.top_k_prefix_access.push(
730 std::make_pair(prefix_access, prefix_out));
731 }
732 }
733
734 if (static_cast<int32_t>(stats.top_k_prefix_ave.size()) <
735 FLAGS_print_top_k_access) {
736 stats.top_k_prefix_ave.push(
737 std::make_pair(prefix_ave_access, prefix_out));
738 } else {
739 if (prefix_ave_access > stats.top_k_prefix_ave.top().first) {
740 stats.top_k_prefix_ave.pop();
741 stats.top_k_prefix_ave.push(
742 std::make_pair(prefix_ave_access, prefix_out));
743 }
744 }
745
746 prefix = record.first.substr(0, FLAGS_output_prefix_cut);
747 prefix_access = 0;
748 prefix_count = 0;
749 prefix_succ_access = 0;
750 }
751 prefix_access += record.second.access_count;
752 prefix_count += 1;
753 prefix_succ_access += record.second.succ_count;
754 }
755 }
756 return Status::OK();
757}

780// Process the statistics of QPS
781Status TraceAnalyzer::MakeStatisticQPS() {
782 if (begin_time_ == 0) {
783 begin_time_ = trace_create_time_;
784 }
785 uint32_t duration =
786 static_cast<uint32_t>((end_time_ - begin_time_) / 1000000);
787 int ret;
788 Status s;
789 std::vector<std::vector<uint32_t>> type_qps(
790 duration, std::vector<uint32_t>(kTaTypeNum + 1, 0));
791 std::vector<uint64_t> qps_sum(kTaTypeNum + 1, 0);
792 std::vector<uint32_t> qps_peak(kTaTypeNum + 1, 0);
793 qps_ave_.resize(kTaTypeNum + 1);
794
795 for (int type = 0; type < kTaTypeNum; type++) {
796 if (!ta_[type].enabled) {
797 continue;
798 }
799 for (auto& stat : ta_[type].stats) {
800 uint32_t time_line = 0;
801 uint64_t cf_qps_sum = 0;
802 for (auto& time_it : stat.second.a_qps_stats) {
803 if (time_it.first >= duration) {
804 continue;
805 }
806 type_qps[time_it.first][kTaTypeNum] += time_it.second;
807 type_qps[time_it.first][type] += time_it.second;
808 cf_qps_sum += time_it.second;
809 if (time_it.second > stat.second.a_peak_qps) {
810 stat.second.a_peak_qps = time_it.second;
811 }
812 if (stat.second.a_qps_f) {
813 while (time_line < time_it.first) {
814 ret = snprintf(buffer_, sizeof(buffer_), "%u\n", 0);
815 if (ret < 0) {
816 return Status::IOError("Format the output failed");
817 }
818 std::string printout(buffer_);
819 s = stat.second.a_qps_f->Append(printout);
820 if (!s.ok()) {
821 fprintf(stderr, "Write QPS file failed\n");
822 return s;
823 }
824 time_line++;
825 }
826 ret = snprintf(buffer_, sizeof(buffer_), "%u\n", time_it.second);
827 if (ret < 0) {
828 return Status::IOError("Format the output failed");
829 }
830 std::string printout(buffer_);
831 s = stat.second.a_qps_f->Append(printout);
832 if (!s.ok()) {
833 fprintf(stderr, "Write QPS file failed\n");
834 return s;
835 }
836 if (time_line == time_it.first) {
837 time_line++;
838 }
839 }
840
841 // Process the top k QPS peaks
842 if (FLAGS_output_prefix_cut > 0) {
843 if (static_cast<int32_t>(stat.second.top_k_qps_sec.size()) <
844 FLAGS_print_top_k_access) {
845 stat.second.top_k_qps_sec.push(
846 std::make_pair(time_it.second, time_it.first));
847 } else {
848 if (stat.second.top_k_qps_sec.size() > 0 &&
849 stat.second.top_k_qps_sec.top().first < time_it.second) {
850 stat.second.top_k_qps_sec.pop();
851 stat.second.top_k_qps_sec.push(
852 std::make_pair(time_it.second, time_it.first));
853 }
854 }
855 }
856 }
857 if (duration == 0) {
858 stat.second.a_ave_qps = 0;
859 } else {
860 stat.second.a_ave_qps = (static_cast<double>(cf_qps_sum)) / duration;
861 }
862
863 // Output the accessed unique key number change overtime
864 if (stat.second.a_key_num_f) {
865 uint64_t cur_uni_key =
866 static_cast<uint64_t>(stat.second.a_key_stats.size());
867 double cur_ratio = 0.0;
868 uint64_t cur_num = 0;
869 for (uint32_t i = 0; i < duration; i++) {
870 auto find_time = stat.second.uni_key_num.find(i);
871 if (find_time != stat.second.uni_key_num.end()) {
872 cur_ratio = (static_cast<double>(find_time->second)) / cur_uni_key;
873 cur_num = find_time->second;
874 }
875 ret = snprintf(buffer_, sizeof(buffer_), "%" PRIu64 " %.12f\n",
876 cur_num, cur_ratio);
877 if (ret < 0) {
878 return Status::IOError("Format the output failed");
879 }
880 std::string printout(buffer_);
881 s = stat.second.a_key_num_f->Append(printout);
882 if (!s.ok()) {
883 fprintf(stderr,
884 "Write accessed unique key number change file failed\n");
885 return s;
886 }
887 }
888 }
889
890 // output the prefix of top k access peak
891 if (FLAGS_output_prefix_cut > 0 && stat.second.a_top_qps_prefix_f) {
892 while (!stat.second.top_k_qps_sec.empty()) {
893 ret = snprintf(buffer_, sizeof(buffer_), "At time: %u with QPS: %u\n",
894 stat.second.top_k_qps_sec.top().second,
895 stat.second.top_k_qps_sec.top().first);
896 if (ret < 0) {
897 return Status::IOError("Format the output failed");
898 }
899 std::string printout(buffer_);
900 s = stat.second.a_top_qps_prefix_f->Append(printout);
901 if (!s.ok()) {
902 fprintf(stderr, "Write prefix QPS top K file failed\n");
903 return s;
904 }
905 uint32_t qps_time = stat.second.top_k_qps_sec.top().second;
906 stat.second.top_k_qps_sec.pop();
907 if (stat.second.a_qps_prefix_stats.find(qps_time) !=
908 stat.second.a_qps_prefix_stats.end()) {
909 for (auto& qps_prefix : stat.second.a_qps_prefix_stats[qps_time]) {
910 std::string qps_prefix_out =
911 ROCKSDB_NAMESPACE::LDBCommand::StringToHex(qps_prefix.first);
912 ret = snprintf(buffer_, sizeof(buffer_),
913 "The prefix: %s Access count: %u\n",
914 qps_prefix_out.c_str(), qps_prefix.second);
915 if (ret < 0) {
916 return Status::IOError("Format the output failed");
917 }
918 std::string pout(buffer_);
919 s = stat.second.a_top_qps_prefix_f->Append(pout);
920 if (!s.ok()) {
921 fprintf(stderr, "Write prefix QPS top K file failed\n");
922 return s;
923 }
924 }
925 }
926 }
927 }
928 }
929 }
930
931 if (qps_f_) {
932 for (uint32_t i = 0; i < duration; i++) {
933 for (int type = 0; type <= kTaTypeNum; type++) {
934 if (type < kTaTypeNum) {
935 ret = snprintf(buffer_, sizeof(buffer_), "%u ", type_qps[i][type]);
936 } else {
937 ret = snprintf(buffer_, sizeof(buffer_), "%u\n", type_qps[i][type]);
938 }
939 if (ret < 0) {
940 return Status::IOError("Format the output failed");
941 }
942 std::string printout(buffer_);
943 s = qps_f_->Append(printout);
944 if (!s.ok()) {
945 return s;
946 }
947 qps_sum[type] += type_qps[i][type];
948 if (type_qps[i][type] > qps_peak[type]) {
949 qps_peak[type] = type_qps[i][type];
950 }
951 }
952 }
953 }
954
955 if (cf_qps_f_) {
956 int cfs_size = static_cast<uint32_t>(cfs_.size());
957 uint32_t v;
958 for (uint32_t i = 0; i < duration; i++) {
959 for (int cf = 0; cf < cfs_size; cf++) {
960 if (cfs_[cf].cf_qps.find(i) != cfs_[cf].cf_qps.end()) {
961 v = cfs_[cf].cf_qps[i];
962 } else {
963 v = 0;
964 }
965 if (cf < cfs_size - 1) {
966 ret = snprintf(buffer_, sizeof(buffer_), "%u ", v);
967 } else {
968 ret = snprintf(buffer_, sizeof(buffer_), "%u\n", v);
969 }
970 if (ret < 0) {
971 return Status::IOError("Format the output failed");
972 }
973 std::string printout(buffer_);
974 s = cf_qps_f_->Append(printout);
975 if (!s.ok()) {
976 return s;
977 }
978 }
979 }
980 }
981
982 qps_peak_ = qps_peak;
983 for (int type = 0; type <= kTaTypeNum; type++) {
984 if (duration == 0) {
985 qps_ave_[type] = 0;
986 } else {
987 qps_ave_[type] = (static_cast<double>(qps_sum[type])) / duration;
988 }
989 }
990
991 return Status::OK();
992}

995// we can output the access count of all keys in a cf
996// we can make some statistics of the whole key space
997// also, we output the top k accessed keys here
998Status TraceAnalyzer::ReProcessing() {
999 int ret;
1000 Status s;
1001 for (auto& cf_it : cfs_) {
1002 uint32_t cf_id = cf_it.first;
1003
1004 // output the time series;
1005 if (FLAGS_output_time_series) {

1013 fprintf(stderr, "Cannot write time_series of '%s' in '%u'\n",
1014 ta_[type].type_name.c_str(), cf_id);

1031 s = stat.time_series_f->Append(printout);
1032 if (!s.ok()) {
1033 fprintf(stderr, "Write time series file failed\n");
1034 return s;
1035 }
1036 stat.time_series.pop_front();
1037 }
1038 }
1039 }
1040
1041 // process the whole key space if needed
1042 if (!FLAGS_key_space_dir.empty()) {

1044 FLAGS_key_space_dir + "/" + std::to_string(cf_id) + ".txt";
1045 std::string input_key, get_key;
1046 std::vector<std::string> prefix(kTaTypeNum);
1047 std::unique_ptr<FSSequentialFile> file;
1048
1049 s = env_->GetFileSystem()->NewSequentialFile(
1050 whole_key_path, FileOptions(env_options_), &file, nullptr);
1051 if (!s.ok()) {
1052 fprintf(stderr, "Cannot open the whole key space file of CF: %u\n",
1053 cf_id);
1054 file.reset();
1055 }
1056
1057 if (file) {
1058 size_t kTraceFileReadaheadSize = 2 * 1024 * 1024;
1059 LineFileReader lf_reader(
1060 std::move(file), whole_key_path,
1061 kTraceFileReadaheadSize /* filereadahead_size */);
1062 for (cfs_[cf_id].w_count = 0; lf_reader.ReadLine(
1063 &get_key, Env::IO_TOTAL /* rate_limiter_priority */);
1064 ++cfs_[cf_id].w_count) {
1065 input_key = ROCKSDB_NAMESPACE::LDBCommand::HexToString(get_key);
1066 for (int type = 0; type < kTaTypeNum; type++) {
1067 if (!ta_[type].enabled) {
1068 continue;
1069 }
1070 TraceStats& stat = ta_[type].stats[cf_id];
1071 if (stat.w_key_f) {
1072 if (stat.a_key_stats.find(input_key) != stat.a_key_stats.end()) {
1073 ret = snprintf(buffer_, sizeof(buffer_),
1074 "%" PRIu64 " %" PRIu64 "\n", cfs_[cf_id].w_count,
1075 stat.a_key_stats[input_key].access_count);
1076 if (ret < 0) {
1077 return Status::IOError("Format the output failed");
1078 }
1079 std::string printout(buffer_);
1080 s = stat.w_key_f->Append(printout);
1081 if (!s.ok()) {
1082 fprintf(stderr, "Write whole key space access file failed\n");
1083 return s;
1084 }
1085 }
1086 }
1087
1088 // Output the prefix cut file of the whole key space
1089 if (FLAGS_output_prefix_cut > 0 && stat.w_prefix_cut_f) {
1090 if (input_key.compare(0, FLAGS_output_prefix_cut, prefix[type]) !=
1091 0) {
1092 prefix[type] = input_key.substr(0, FLAGS_output_prefix_cut);
1093 std::string prefix_out =
1094 ROCKSDB_NAMESPACE::LDBCommand::StringToHex(prefix[type]);
1095 ret = snprintf(buffer_, sizeof(buffer_), "%" PRIu64 " %s\n",
1096 cfs_[cf_id].w_count, prefix_out.c_str());
1097 if (ret < 0) {
1098 return Status::IOError("Format the output failed");
1099 }
1100 std::string printout(buffer_);
1101 s = stat.w_prefix_cut_f->Append(printout);
1102 if (!s.ok()) {
1103 fprintf(stderr,
1104 "Write whole key space prefix cut file failed\n");
1105 return s;
1106 }
1107 }
1108 }
1109 }
1110
1111 // Make the statistics fo the key size distribution
1112 if (FLAGS_output_key_distribution) {
1113 if (cfs_[cf_id].w_key_size_stats.find(input_key.size()) ==
1114 cfs_[cf_id].w_key_size_stats.end()) {
1115 cfs_[cf_id].w_key_size_stats[input_key.size()] = 1;
1116 } else {
1117 cfs_[cf_id].w_key_size_stats[input_key.size()]++;
1118 }
1119 }
1120 }
1121 s = lf_reader.GetStatus();
1122 if (!s.ok()) {
1123 fprintf(stderr, "Read whole key space file failed\n");
1124 return s;
1125 }
1126 }
1127 }
1128
1129 // process the top k accessed keys
1130 if (FLAGS_print_top_k_access > 0) {
1131 for (int type = 0; type < kTaTypeNum; type++) {
1132 if (!ta_[type].enabled ||
1133 ta_[type].stats.find(cf_id) == ta_[type].stats.end()) {
1134 continue;
1135 }
1136 TraceStats& stat = ta_[type].stats[cf_id];
1137 for (auto& record : stat.a_key_stats) {
1138 if (static_cast<int32_t>(stat.top_k_queue.size()) <
1139 FLAGS_print_top_k_access) {
1140 stat.top_k_queue.push(
1141 std::make_pair(record.second.access_count, record.first));
1142 } else {
1143 if (record.second.access_count > stat.top_k_queue.top().first) {
1144 stat.top_k_queue.pop();
1145 stat.top_k_queue.push(
1146 std::make_pair(record.second.access_count, record.first));
1147 }
1148 }
1149 }
1150 }
1151 }
1152 }
1153 return Status::OK();
1154}

1172// Insert the corresponding key statistics to the correct type
1173// and correct CF, output the time-series file if needed
1174Status TraceAnalyzer::KeyStatsInsertion(const uint32_t& type,
1175 const uint32_t& cf_id,
1176 const std::string& key,
1177 const size_t value_size,
1178 const uint64_t ts) {
1179 Status s;
1180 StatsUnit unit;
1181 unit.key_id = 0;
1182 unit.cf_id = cf_id;
1183 unit.value_size = value_size;
1184 unit.access_count = 1;
1185 unit.latest_ts = ts;
1186 if ((type != TraceOperationType::kGet &&
1187 type != TraceOperationType::kMultiGet) ||
1188 value_size > 0) {
1189 unit.succ_count = 1;
1190 } else {
1191 unit.succ_count = 0;
1192 }
1193 unit.v_correlation.resize(analyzer_opts_.correlation_list.size());
1194 for (int i = 0;
1195 i < (static_cast<int>(analyzer_opts_.correlation_list.size())); i++) {
1196 unit.v_correlation[i].count = 0;
1197 unit.v_correlation[i].total_ts = 0;
1198 }
1199 std::string prefix;
1200 if (FLAGS_output_prefix_cut > 0) {
1201 prefix = key.substr(0, FLAGS_output_prefix_cut);
1202 }
1203
1204 if (begin_time_ == 0) {
1205 begin_time_ = ts;
1206 }
1207 uint32_t time_in_sec;
1208 if (ts < begin_time_) {
1209 time_in_sec = 0;
1210 } else {
1211 time_in_sec = static_cast<uint32_t>((ts - begin_time_) / 1000000);
1212 }
1213
1214 uint64_t dist_value_size = value_size / FLAGS_value_interval;
1215 auto found_stats = ta_[type].stats.find(cf_id);
1216 if (found_stats == ta_[type].stats.end()) {
1217 ta_[type].stats[cf_id].cf_id = cf_id;
1218 ta_[type].stats[cf_id].cf_name = std::to_string(cf_id);
1219 ta_[type].stats[cf_id].a_count = 1;
1220 ta_[type].stats[cf_id].a_key_id = 0;
1221 ta_[type].stats[cf_id].a_key_size_sqsum = MultiplyCheckOverflow(
1222 static_cast<uint64_t>(key.size()), static_cast<uint64_t>(key.size()));
1223 ta_[type].stats[cf_id].a_key_size_sum = key.size();
1224 ta_[type].stats[cf_id].a_value_size_sqsum = MultiplyCheckOverflow(
1225 static_cast<uint64_t>(value_size), static_cast<uint64_t>(value_size));
1226 ta_[type].stats[cf_id].a_value_size_sum = value_size;
1227 s = OpenStatsOutputFiles(ta_[type].type_name, ta_[type].stats[cf_id]);
1228 if (!FLAGS_print_correlation.empty()) {
1229 s = StatsUnitCorrelationUpdate(unit, type, ts, key);
1230 }
1231 ta_[type].stats[cf_id].a_key_stats[key] = unit;
1232 ta_[type].stats[cf_id].a_value_size_stats[dist_value_size] = 1;
1233 ta_[type].stats[cf_id].a_qps_stats[time_in_sec] = 1;
1234 ta_[type].stats[cf_id].correlation_output.resize(
1235 analyzer_opts_.correlation_list.size());
1236 if (FLAGS_output_prefix_cut > 0) {
1237 std::map<std::string, uint32_t> tmp_qps_map;
1238 tmp_qps_map[prefix] = 1;
1239 ta_[type].stats[cf_id].a_qps_prefix_stats[time_in_sec] = tmp_qps_map;
1240 }
1241 if (time_in_sec != cur_time_sec_) {
1242 ta_[type].stats[cf_id].uni_key_num[cur_time_sec_] =
1243 static_cast<uint64_t>(ta_[type].stats[cf_id].a_key_stats.size());
1244 cur_time_sec_ = time_in_sec;
1245 }
1246 } else {
1247 found_stats->second.a_count++;
1248 found_stats->second.a_key_size_sqsum += MultiplyCheckOverflow(
1249 static_cast<uint64_t>(key.size()), static_cast<uint64_t>(key.size()));
1250 found_stats->second.a_key_size_sum += key.size();
1251 found_stats->second.a_value_size_sqsum += MultiplyCheckOverflow(
1252 static_cast<uint64_t>(value_size), static_cast<uint64_t>(value_size));
1253 found_stats->second.a_value_size_sum += value_size;
1254 auto found_key = found_stats->second.a_key_stats.find(key);
1255 if (found_key == found_stats->second.a_key_stats.end()) {
1256 found_stats->second.a_key_stats[key] = unit;
1257 } else {
1258 found_key->second.access_count++;
1259 if (type != TraceOperationType::kGet || value_size > 0) {
1260 found_key->second.succ_count++;
1261 }
1262 if (!FLAGS_print_correlation.empty()) {
1263 s = StatsUnitCorrelationUpdate(found_key->second, type, ts, key);
1264 }
1265 }
1266 if (time_in_sec != cur_time_sec_) {
1267 found_stats->second.uni_key_num[cur_time_sec_] =
1268 static_cast<uint64_t>(found_stats->second.a_key_stats.size());
1269 cur_time_sec_ = time_in_sec;
1270 }
1271
1272 auto found_value =
1273 found_stats->second.a_value_size_stats.find(dist_value_size);
1274 if (found_value == found_stats->second.a_value_size_stats.end()) {
1275 found_stats->second.a_value_size_stats[dist_value_size] = 1;
1276 } else {
1277 found_value->second++;
1278 }
1279
1280 auto found_qps = found_stats->second.a_qps_stats.find(time_in_sec);
1281 if (found_qps == found_stats->second.a_qps_stats.end()) {
1282 found_stats->second.a_qps_stats[time_in_sec] = 1;
1283 } else {
1284 found_qps->second++;
1285 }
1286
1287 if (FLAGS_output_prefix_cut > 0) {
1288 auto found_qps_prefix =
1289 found_stats->second.a_qps_prefix_stats.find(time_in_sec);
1290 if (found_qps_prefix == found_stats->second.a_qps_prefix_stats.end()) {
1291 std::map<std::string, uint32_t> tmp_qps_map;
1292 found_stats->second.a_qps_prefix_stats[time_in_sec] = tmp_qps_map;
1293 }
1294 if (found_stats->second.a_qps_prefix_stats[time_in_sec].find(prefix) ==
1295 found_stats->second.a_qps_prefix_stats[time_in_sec].end()) {
1296 found_stats->second.a_qps_prefix_stats[time_in_sec][prefix] = 1;
1297 } else {
1298 found_stats->second.a_qps_prefix_stats[time_in_sec][prefix]++;
1299 }
1300 }
1301 }
1302
1303 if (cfs_.find(cf_id) == cfs_.end()) {
1304 CfUnit cf_unit;
1305 cf_unit.cf_id = cf_id;
1306 cf_unit.w_count = 0;
1307 cf_unit.a_count = 0;
1308 cfs_[cf_id] = cf_unit;
1309 }
1310
1311 if (FLAGS_output_qps_stats) {
1312 cfs_[cf_id].cf_qps[time_in_sec]++;
1313 }
1314
1315 if (FLAGS_output_time_series) {
1316 TraceUnit trace_u;
1317 trace_u.type = type;
1318 trace_u.key = key;
1319 trace_u.value_size = value_size;
1320 trace_u.ts = (ts - time_series_start_) / 1000000;
1321 trace_u.cf_id = cf_id;
1322 ta_[type].stats[cf_id].time_series.push_back(trace_u);
1323 }
1324
1325 return s;
1326}

1328// Update the correlation unit of each key if enabled
1329Status TraceAnalyzer::StatsUnitCorrelationUpdate(StatsUnit& unit,
1330 const uint32_t& type_second,
1331 const uint64_t& ts,
1332 const std::string& key) {
1333 if (type_second >= kTaTypeNum) {
1334 fprintf(stderr, "Unknown Type Id: %u\n", type_second);
1335 return Status::NotFound();
1336 }
1337
1338 for (int type_first = 0; type_first < kTaTypeNum; type_first++) {
1339 if (type_first >= static_cast<int>(ta_.size()) ||
1340 type_first >= static_cast<int>(analyzer_opts_.correlation_map.size())) {
1341 break;
1342 }
1343 if (analyzer_opts_.correlation_map[type_first][type_second] < 0 ||
1344 ta_[type_first].stats.find(unit.cf_id) == ta_[type_first].stats.end() ||
1345 ta_[type_first].stats[unit.cf_id].a_key_stats.find(key) ==
1346 ta_[type_first].stats[unit.cf_id].a_key_stats.end() ||
1347 ta_[type_first].stats[unit.cf_id].a_key_stats[key].latest_ts == ts) {
1348 continue;
1349 }
1350
1351 int correlation_id =
1352 analyzer_opts_.correlation_map[type_first][type_second];
1353
1354 // after get the x-y operation time or x, update;
1355 if (correlation_id < 0 ||
1356 correlation_id >= static_cast<int>(unit.v_correlation.size())) {
1357 continue;
1358 }
1359 unit.v_correlation[correlation_id].count++;
1360 unit.v_correlation[correlation_id].total_ts +=
1361 (ts - ta_[type_first].stats[unit.cf_id].a_key_stats[key].latest_ts);
1362 }
1363
1364 unit.latest_ts = ts;
1365 return Status::OK();
1366}

1504Status TraceAnalyzer::Handle(const WriteQueryTraceRecord& record,
1505 std::unique_ptr<TraceRecordResult>* /*result*/) {
1506 total_writes_++;
1507 // Note that, if the write happens in a transaction,
1508 // 'Write' will be called twice, one for Prepare, one for
1509 // Commit. Thus, in the trace, for the same WriteBatch, there
1510 // will be two records if it is in a transaction. Here, we only
1511 // process the reord that is committed. If write is non-transaction,
1512 // HasBeginPrepare()==false, so we process it normally.
1513 WriteBatch batch(record.GetWriteBatchRep().ToString());
1514 if (batch.Count() == 0 || (batch.HasBeginPrepare() && !batch.HasCommit())) {
1515 return Status::OK();
1516 }
1517 write_batch_ts_ = record.GetTimestamp();
1518
1519 // write_result_ will be updated in batch's handler during iteration.
1520 Status s = batch.Iterate(this);
1521 write_batch_ts_ = 0;
1522 if (!s.ok()) {
1523 fprintf(stderr, "Cannot process the write batch in the trace\n");
1524 return s;
1525 }
1526
1527 return Status::OK();
1528}
