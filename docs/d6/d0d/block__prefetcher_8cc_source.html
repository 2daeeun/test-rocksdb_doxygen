15void BlockPrefetcher::PrefetchIfNeeded(
16 const BlockBasedTable::Rep* rep, const BlockHandle& handle,
17 const size_t readahead_size, bool is_for_compaction,
18 const bool no_sequential_checking, const ReadOptions& read_options,
19 const std::function<void(bool, uint64_t&, uint64_t&)>& readaheadsize_cb,
20 bool is_async_io_prefetch) {
21 if (read_options.read_tier == ReadTier::kBlockCacheTier) {
22 // Disable prefetching when IO disallowed. (Note that we haven't allocated
23 // any buffers yet despite the various tracked settings.)
24 return;
25 }
26
27 ReadaheadParams readahead_params;
28 readahead_params.initial_readahead_size = readahead_size;
29 readahead_params.max_readahead_size = readahead_size;
30 readahead_params.num_buffers = is_async_io_prefetch ? 2 : 1;
31
32 const size_t len = BlockBasedTable::BlockSizeWithTrailer(handle);
33 const size_t offset = handle.offset();
34 if (is_for_compaction) {
35 if (!rep->file->use_direct_io() && compaction_readahead_size_ > 0) {
36 // If FS supports prefetching (readahead_limit_ will be non zero in that
37 // case) and current block exists in prefetch buffer then return.
38 if (offset + len <= readahead_limit_) {
39 return;
40 }
41 IOOptions opts;
42 Status s = rep->file->PrepareIOOptions(read_options, opts);
43 if (!s.ok()) {
44 return;
45 }
46 s = rep->file->Prefetch(opts, offset, len + compaction_readahead_size_);
47 if (s.ok()) {
48 readahead_limit_ = offset + len + compaction_readahead_size_;
49 return;
50 } else if (!s.IsNotSupported()) {
51 return;
52 }
53 }
54 // If FS prefetch is not supported, fall back to use internal prefetch
55 // buffer.
56 //
57 // num_file_reads is used by FilePrefetchBuffer only when
58 // implicit_auto_readahead is set.
59 readahead_params.initial_readahead_size = compaction_readahead_size_;
60 readahead_params.max_readahead_size = compaction_readahead_size_;
61 rep->CreateFilePrefetchBufferIfNotExists(readahead_params,
62 &prefetch_buffer_,
63 /*readaheadsize_cb=*/nullptr);
64 return;
65 }
66
67 // Explicit user requested readahead.
68 if (readahead_size > 0) {
69 rep->CreateFilePrefetchBufferIfNotExists(
70 readahead_params, &prefetch_buffer_, readaheadsize_cb,
71 /*usage=*/FilePrefetchBufferUsage::kUserScanPrefetch);
72 return;
73 }
74
75 // Implicit readahead.
76
77 // If max_auto_readahead_size is set to be 0 by user, no data will be
78 // prefetched.
79 size_t max_auto_readahead_size = rep->table_options.max_auto_readahead_size;
80 if (max_auto_readahead_size == 0 || initial_auto_readahead_size_ == 0) {
81 return;
82 }
83
84 if (initial_auto_readahead_size_ > max_auto_readahead_size) {
85 initial_auto_readahead_size_ = max_auto_readahead_size;
86 }
87
88 readahead_params.initial_readahead_size = initial_auto_readahead_size_;
89 readahead_params.max_readahead_size = max_auto_readahead_size;
90 readahead_params.implicit_auto_readahead = true;
91 readahead_params.num_file_reads_for_auto_readahead =
92 rep->table_options.num_file_reads_for_auto_readahead;
93
94 // In case of no_sequential_checking, it will skip the num_file_reads_ and
95 // will always creates the FilePrefetchBuffer.
96 if (no_sequential_checking) {
97 rep->CreateFilePrefetchBufferIfNotExists(
98 readahead_params, &prefetch_buffer_, readaheadsize_cb,
99 /*usage=*/FilePrefetchBufferUsage::kUserScanPrefetch);
100 return;
101 }
102
103 // If FS supports prefetching (readahead_limit_ will be non zero in that case)
104 // and current block exists in prefetch buffer then return.
105 if (offset + len <= readahead_limit_) {
106 UpdateReadPattern(offset, len);
107 return;
108 }
109
110 if (!IsBlockSequential(offset)) {
111 UpdateReadPattern(offset, len);
112 ResetValues(rep->table_options.initial_auto_readahead_size);
113 return;
114 }
115 UpdateReadPattern(offset, len);
116
117 // Implicit auto readahead, which will be enabled if the number of reads
118 // reached `table_options.num_file_reads_for_auto_readahead` (default: 2) and
119 // scans are sequential.
120 num_file_reads_++;
121 if (num_file_reads_ <= rep->table_options.num_file_reads_for_auto_readahead) {
122 return;
123 }
124
125 readahead_params.num_file_reads = num_file_reads_;
126 if (rep->file->use_direct_io()) {
127 rep->CreateFilePrefetchBufferIfNotExists(
128 readahead_params, &prefetch_buffer_, readaheadsize_cb,
129 /*usage=*/FilePrefetchBufferUsage::kUserScanPrefetch);
130 return;
131 }
132
133 if (readahead_size_ > max_auto_readahead_size) {
134 readahead_size_ = max_auto_readahead_size;
135 }
136
137 // If prefetch is not supported, fall back to use internal prefetch buffer.
138 IOOptions opts;
139 Status s = rep->file->PrepareIOOptions(read_options, opts);
140 if (!s.ok()) {
141 return;
142 }
143 s = rep->file->Prefetch(
144 opts, handle.offset(),
145 BlockBasedTable::BlockSizeWithTrailer(handle) + readahead_size_);
146 if (s.IsNotSupported()) {
147 rep->CreateFilePrefetchBufferIfNotExists(
148 readahead_params, &prefetch_buffer_, readaheadsize_cb,
149 /*usage=*/FilePrefetchBufferUsage::kUserScanPrefetch);
150 return;
151 }
152
153 readahead_limit_ = offset + len + readahead_size_;
154 // Keep exponentially increasing readahead size until
155 // max_auto_readahead_size.
156 readahead_size_ = std::min(max_auto_readahead_size, readahead_size_ * 2);
157}
158} // namespace ROCKSDB_NAMESPACE
