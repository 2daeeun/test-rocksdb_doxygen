64DBTestBase::DBTestBase(const std::string path, bool env_do_fsync)
65 : mem_env_(nullptr), encrypted_env_(nullptr), option_config_(kDefault) {
66 Env* base_env = Env::Default();
67 ConfigOptions config_options;
68 EXPECT_OK(test::CreateEnvFromSystem(config_options, &base_env, &env_guard_));
69 EXPECT_NE(nullptr, base_env);
70 if (getenv("MEM_ENV")) {
71 mem_env_ = MockEnv::Create(base_env, base_env->GetSystemClock());
72 }
73 if (getenv("ENCRYPTED_ENV")) {
74 std::shared_ptr<EncryptionProvider> provider;
75 std::string provider_id = getenv("ENCRYPTED_ENV");

85 : (mem_env_ ? mem_env_ : base_env));

87 env_->SetBackgroundThreads(1, Env::HIGH);
88 env_->skip_fsync_ = !env_do_fsync;

101 Random::GetTLSInstance()->Reset(0xdeadbeef);
102}

124bool DBTestBase::ShouldSkipOptions(int option_config, int skip_mask) {
125
126 if ((skip_mask & kSkipUniversalCompaction) &&
127 (option_config == kUniversalCompaction ||
128 option_config == kUniversalCompactionMultiLevel ||
129 option_config == kUniversalSubcompactions)) {
130 return true;
131 }
132 if ((skip_mask & kSkipMergePut) && option_config == kMergePut) {
133 return true;
134 }
135 if ((skip_mask & kSkipNoSeekToLast) &&
136 (option_config == kHashLinkList || option_config == kHashSkipList)) {
137 return true;
138 }
139 if ((skip_mask & kSkipPlainTable) &&
140 (option_config == kPlainTableAllBytesPrefix ||
141 option_config == kPlainTableFirstBytePrefix ||
142 option_config == kPlainTableCappedPrefix ||
143 option_config == kPlainTableCappedPrefixNonMmap)) {
144 return true;
145 }
146 if ((skip_mask & kSkipHashIndex) &&
147 (option_config == kBlockBasedTableWithPrefixHashIndex ||
148 option_config == kBlockBasedTableWithWholeKeyHashIndex)) {
149 return true;
150 }
151 if ((skip_mask & kSkipFIFOCompaction) && option_config == kFIFOCompaction) {
152 return true;
153 }
154 if ((skip_mask & kSkipMmapReads) && option_config == kWalDirAndMmapReads) {
155 return true;
156 }
157 if ((skip_mask & kSkipRowCache) && option_config == kRowCache) {
158 return true;
159 }
160 return false;
161}

163// Switch to a fresh database with the next option configuration to
164// test. Return false if there are no more configurations to test.
165bool DBTestBase::ChangeOptions(int skip_mask) {
166 for (option_config_++; option_config_ < kEnd; option_config_++) {
167 if (ShouldSkipOptions(option_config_, skip_mask)) {
168 continue;
169 }
170 break;
171 }
172
173 if (option_config_ >= kEnd) {
174 Destroy(last_options_);
175 return false;
176 } else {
177 auto options = CurrentOptions();
178 options.create_if_missing = true;
179 DestroyAndReopen(options);
180 return true;
181 }

343Options DBTestBase::GetOptions(
344 int option_config, const Options& default_options,
345 const anon::OptionsOverride& options_override) const {
346 // this redundant copy is to minimize code change w/o having lint error.
347 Options options = default_options;
348 BlockBasedTableOptions table_options;
349 bool set_block_based_table_factory = true;
350#if !defined(OS_MACOSX) && !defined(OS_WIN) && !defined(OS_SOLARIS) && \
351 !defined(OS_AIX)
352 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->ClearCallBack(
353 "NewRandomAccessFile:O_DIRECT");
354 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->ClearCallBack(
355 "NewWritableFile:O_DIRECT");
356#endif
357 // kMustFreeHeapAllocations -> indicates ASAN build
358 if (kMustFreeHeapAllocations && !options_override.full_block_cache) {
359 // Detecting block cache use-after-free is normally difficult in unit
360 // tests, because as a cache, it tends to keep unreferenced entries in
361 // memory, and we normally want unit tests to take advantage of block
362 // cache for speed. However, we also want a strong chance of detecting
363 // block cache use-after-free in unit tests in ASAN builds, so for ASAN
364 // builds we use a trivially small block cache to which entries can be
365 // added but are immediately freed on no more references.
366 table_options.block_cache = NewLRUCache(/* too small */ 1);
367 }
368
369 // Test anticipated new default as much as reasonably possible (and remove
370 // this code when obsolete)
371 assert(!table_options.decouple_partitioned_filters);
372 table_options.decouple_partitioned_filters = true;
373
374 bool can_allow_mmap = IsMemoryMappedAccessSupported();
375 switch (option_config) {
376 case kHashSkipList:
377 options.prefix_extractor.reset(NewFixedPrefixTransform(1));
378 options.memtable_factory.reset(NewHashSkipListRepFactory(16));
379 options.allow_concurrent_memtable_write = false;
380 options.unordered_write = false;
381 break;
382 case kPlainTableFirstBytePrefix:
383 options.table_factory.reset(NewPlainTableFactory());
384 options.prefix_extractor.reset(NewFixedPrefixTransform(1));
385 options.allow_mmap_reads = can_allow_mmap;
386 options.max_sequential_skip_in_iterations = 999999;
387 set_block_based_table_factory = false;
388 break;
389 case kPlainTableCappedPrefix:
390 options.table_factory.reset(NewPlainTableFactory());
391 options.prefix_extractor.reset(NewCappedPrefixTransform(8));
392 options.allow_mmap_reads = can_allow_mmap;
393 options.max_sequential_skip_in_iterations = 999999;
394 set_block_based_table_factory = false;
395 break;
396 case kPlainTableCappedPrefixNonMmap:
397 options.table_factory.reset(NewPlainTableFactory());
398 options.prefix_extractor.reset(NewCappedPrefixTransform(8));
399 options.allow_mmap_reads = false;
400 options.max_sequential_skip_in_iterations = 999999;
401 set_block_based_table_factory = false;
402 break;
403 case kPlainTableAllBytesPrefix:
404 options.table_factory.reset(NewPlainTableFactory());
405 options.prefix_extractor.reset(NewNoopTransform());
406 options.allow_mmap_reads = can_allow_mmap;
407 options.max_sequential_skip_in_iterations = 999999;
408 set_block_based_table_factory = false;
409 break;
410 case kVectorRep:
411 options.memtable_factory.reset(new VectorRepFactory(100));
412 options.allow_concurrent_memtable_write = false;
413 options.unordered_write = false;
414 break;
415 case kHashLinkList:
416 options.prefix_extractor.reset(NewFixedPrefixTransform(1));
417 options.memtable_factory.reset(
418 NewHashLinkListRepFactory(4, 0, 3, true, 4));
419 options.allow_concurrent_memtable_write = false;
420 options.unordered_write = false;
421 break;
422 case kDirectIO: {
423 options.use_direct_reads = true;
424 options.use_direct_io_for_flush_and_compaction = true;
425 options.compaction_readahead_size = 2 * 1024 * 1024;
426 SetupSyncPointsToMockDirectIO();
427 break;
428 }
429 case kMergePut:
430 options.merge_operator = MergeOperators::CreatePutOperator();
431 break;
432 case kFilter:
433 table_options.filter_policy.reset(NewBloomFilterPolicy(10, true));
434 break;
435 case kFullFilterWithNewTableReaderForCompactions:
436 table_options.filter_policy.reset(NewBloomFilterPolicy(10, false));
437 options.compaction_readahead_size = 10 * 1024 * 1024;
438 break;
439 case kPartitionedFilterWithNewTableReaderForCompactions:
440 table_options.filter_policy.reset(NewBloomFilterPolicy(10, false));
441 table_options.partition_filters = true;
442 table_options.index_type =
443 BlockBasedTableOptions::IndexType::kTwoLevelIndexSearch;
444 options.compaction_readahead_size = 10 * 1024 * 1024;
445 break;
446 case kUncompressed:
447 options.compression = kNoCompression;
448 break;
449 case kNumLevel_3:
450 options.num_levels = 3;
451 break;
452 case kDBLogDir:
453 options.db_log_dir = alternative_db_log_dir_;
454 break;
455 case kWalDirAndMmapReads:
456 options.wal_dir = alternative_wal_dir_;
457 // mmap reads should be orthogonal to WalDir setting, so we piggyback to
458 // this option config to test mmap reads as well
459 options.allow_mmap_reads = can_allow_mmap;
460 break;
461 case kManifestFileSize:
462 options.max_manifest_file_size = 50; // 50 bytes
463 break;
464 case kPerfOptions:
465 options.delayed_write_rate = 8 * 1024 * 1024;
466 options.report_bg_io_stats = true;
467 // TODO(3.13) -- test more options
468 break;
469 case kUniversalCompaction:
470 options.compaction_style = kCompactionStyleUniversal;
471 options.num_levels = 1;
472 break;
473 case kUniversalCompactionMultiLevel:
474 options.compaction_style = kCompactionStyleUniversal;
475 options.num_levels = 8;
476 break;
477 case kInfiniteMaxOpenFiles:
478 options.max_open_files = -1;
479 break;
480 case kCRC32cChecksum: {
481 // Old default was CRC32c, but XXH3 (new default) is faster on common
482 // hardware
483 table_options.checksum = kCRC32c;
484 // Thrown in here for basic coverage:
485 options.DisableExtraChecks();
486 break;
487 }
488 case kFIFOCompaction: {
489 options.compaction_style = kCompactionStyleFIFO;
490 options.max_open_files = -1;
491 break;
492 }
493 case kBlockBasedTableWithPrefixHashIndex: {
494 table_options.index_type = BlockBasedTableOptions::kHashSearch;
495 options.prefix_extractor.reset(NewFixedPrefixTransform(1));
496 break;
497 }
498 case kBlockBasedTableWithWholeKeyHashIndex: {
499 table_options.index_type = BlockBasedTableOptions::kHashSearch;
500 options.prefix_extractor.reset(NewNoopTransform());
501 break;
502 }
503 case kBlockBasedTableWithPartitionedIndex: {
504 table_options.format_version = 3;
505 table_options.index_type = BlockBasedTableOptions::kTwoLevelIndexSearch;
506 options.prefix_extractor.reset(NewNoopTransform());
507 break;
508 }
509 case kBlockBasedTableWithPartitionedIndexFormat4: {
510 table_options.format_version = 4;
511 // Format 4 changes the binary index format. Since partitioned index is a
512 // super-set of simple indexes, we are also using kTwoLevelIndexSearch to
513 // test this format.
514 table_options.index_type = BlockBasedTableOptions::kTwoLevelIndexSearch;
515 // The top-level index in partition filters are also affected by format 4.
516 table_options.filter_policy.reset(NewBloomFilterPolicy(10, false));
517 table_options.partition_filters = true;
518 table_options.index_block_restart_interval = 8;
519 break;
520 }
521 case kBlockBasedTableWithIndexRestartInterval: {
522 table_options.index_block_restart_interval = 8;
523 break;
524 }
525 case kBlockBasedTableWithLatestFormat: {
526 // In case different from default
527 table_options.format_version = kLatestFormatVersion;
528 break;
529 }
530 case kOptimizeFiltersForHits: {
531 options.optimize_filters_for_hits = true;
532 set_block_based_table_factory = true;
533 break;
534 }
535 case kRowCache: {
536 options.row_cache = NewLRUCache(1024 * 1024);
537 break;
538 }
539 case kRecycleLogFiles: {
540 options.recycle_log_file_num = 2;
541 break;
542 }
543 case kLevelSubcompactions: {
544 options.max_subcompactions = 4;
545 break;
546 }
547 case kUniversalSubcompactions: {
548 options.compaction_style = kCompactionStyleUniversal;
549 options.num_levels = 8;
550 options.max_subcompactions = 4;
551 break;
552 }
553 case kConcurrentSkipList: {
554 options.allow_concurrent_memtable_write = true;
555 options.enable_write_thread_adaptive_yield = true;
556 break;
557 }
558 case kPipelinedWrite: {
559 options.enable_pipelined_write = true;
560 break;
561 }
562 case kConcurrentWALWrites: {
563 // This options optimize 2PC commit path
564 options.two_write_queues = true;
565 options.manual_wal_flush = true;
566 break;
567 }
568 case kUnorderedWrite: {
569 options.allow_concurrent_memtable_write = false;
570 options.unordered_write = false;
571 break;
572 }
573 case kBlockBasedTableWithBinarySearchWithFirstKeyIndex: {
574 table_options.index_type =
575 BlockBasedTableOptions::kBinarySearchWithFirstKey;
576 break;
577 }
578
579 default:
580 break;
581 }
582
583 if (options_override.filter_policy) {
584 table_options.filter_policy = options_override.filter_policy;
585 table_options.partition_filters = options_override.partition_filters;
586 table_options.metadata_block_size = options_override.metadata_block_size;
587 }
588 if (set_block_based_table_factory) {
589 options.table_factory.reset(NewBlockBasedTableFactory(table_options));
590 }
591 options.level_compaction_dynamic_level_bytes =
592 options_override.level_compaction_dynamic_level_bytes;
593 options.env = env_;
594 options.create_if_missing = true;
595 options.fail_if_options_file_error = true;
596 return options;
597}

628void DBTestBase::SetTimeElapseOnlySleepOnReopen(DBOptions* options) {
629 time_elapse_only_sleep_on_reopen_ = true;
630
631 // Need to disable stats dumping and persisting which also use
632 // RepeatableThread, which uses InstrumentedCondVar::TimedWaitInternal.
633 // With time_elapse_only_sleep_, this can hang on some platforms (MacOS)
634 // because (a) on some platforms, pthread_cond_timedwait does not appear
635 // to release the lock for other threads to operate if the deadline time
636 // is already passed, and (b) TimedWait calls are currently a bad abstraction
637 // because the deadline parameter is usually computed from Env time,
638 // but is interpreted in real clock time.
639 options->stats_dump_period_sec = 0;
640 options->stats_persist_period_sec = 0;
641}

643void DBTestBase::MaybeInstallTimeElapseOnlySleep(const DBOptions& options) {
644 if (time_elapse_only_sleep_on_reopen_) {
645 assert(options.env == env_ ||
646 static_cast_with_check<CompositeEnvWrapper>(options.env)
647 ->env_target() == env_);
648 assert(options.stats_dump_period_sec == 0);
649 assert(options.stats_persist_period_sec == 0);

964// Return a string that contains all key,value pairs in order,
965// formatted like "(k1->v1)(k2->v2)".
966std::string DBTestBase::Contents(int cf) {
967 std::vector<std::string> forward;
968 std::string result;
969 Iterator* iter = (cf == 0) ? db_->NewIterator(ReadOptions())
970 : db_->NewIterator(ReadOptions(), handles_[cf]);
971 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
972 std::string s = IterStatus(iter);
973 result.push_back('(');
974 result.append(s);
975 result.push_back(')');
976 forward.push_back(s);
977 }
978
979 // Check reverse iteration results are the reverse of forward results
980 unsigned int matched = 0;
981 for (iter->SeekToLast(); iter->Valid(); iter->Prev()) {
982 EXPECT_LT(matched, forward.size());
983 EXPECT_EQ(IterStatus(iter), forward[forward.size() - matched - 1]);
984 matched++;
985 }
986 EXPECT_OK(iter->status());
987 EXPECT_EQ(matched, forward.size());
988
989 delete iter;
990 return result;
991}

1012std::string DBTestBase::AllEntriesFor(const Slice& user_key, int cf) {
1013 Arena arena;
1014 auto options = CurrentOptions();
1015 InternalKeyComparator icmp(options.comparator);
1016 ReadOptions read_options;
1017 ScopedArenaPtr<InternalIterator> iter;
1018 if (cf == 0) {
1019 iter.reset(dbfull()->NewInternalIterator(read_options, &arena,
1020 kMaxSequenceNumber));
1021 } else {
1022 iter.reset(dbfull()->NewInternalIterator(read_options, &arena,
1023 kMaxSequenceNumber, handles_[cf]));
1024 }
1025 InternalKey target(user_key, kMaxSequenceNumber, kTypeValue);
1026 iter->Seek(target.Encode());
1027 std::string result;
1028 if (!iter->status().ok()) {
1029 result = iter->status().ToString();
1030 } else {
1031 result = "[ ";
1032 bool first = true;
1033 while (iter->Valid()) {
1034 ParsedInternalKey ikey(Slice(), 0, kTypeValue);
1035 if (ParseInternalKey(iter->key(), &ikey, true /* log_err_key */) !=
1036 Status::OK()) {
1037 result += "CORRUPTED";
1038 } else {
1039 if (!last_options_.comparator->Equal(ikey.user_key, user_key)) {
1040 break;
1041 }
1042 if (!first) {
1043 result += ", ";
1044 }
1045 first = false;
1046 switch (ikey.type) {
1047 case kTypeValue:
1048 result += iter->value().ToString();
1049 break;
1050 case kTypeMerge:
1051 // keep it the same as kTypeValue for testing kMergePut
1052 result += iter->value().ToString();
1053 break;
1054 case kTypeDeletion:
1055 result += "DEL";
1056 break;
1057 case kTypeSingleDeletion:
1058 result += "SDEL";
1059 break;
1060 default:
1061 assert(false);
1062 break;
1063 }
1064 }
1065 iter->Next();
1066 }
1067 if (!first) {
1068 result += " ";
1069 }
1070 result += "]";
1071 }
1072 return result;
