30// BlockAccessCipherStream is the base class for any cipher stream that
31// supports random access at block level (without requiring data from other
32// blocks). E.g. CTR (Counter operation mode) supports this requirement.
33class BlockAccessCipherStream {
34 public:
35 virtual ~BlockAccessCipherStream(){}
36
37 // BlockSize returns the size of each block supported by this cipher stream.
38 virtual size_t BlockSize() = 0;
39
40 // Encrypt one or more (partial) blocks of data at the file offset.
41 // Length of data is given in dataSize.
42 virtual Status Encrypt(uint64_t fileOffset, char* data, size_t dataSize);
43
44 // Decrypt one or more (partial) blocks of data at the file offset.
45 // Length of data is given in dataSize.
46 virtual Status Decrypt(uint64_t fileOffset, char* data, size_t dataSize);
47
48 protected:
49 // Allocate scratch space which is passed to EncryptBlock/DecryptBlock.
50 virtual void AllocateScratch(std::string&) = 0;
51
52 // Encrypt a block of data at the given block index.
53 // Length of data is equal to BlockSize();
54 virtual Status EncryptBlock(uint64_t blockIndex, char* data,
55 char* scratch) = 0;
56
57 // Decrypt a block of data at the given block index.
58 // Length of data is equal to BlockSize();
59 virtual Status DecryptBlock(uint64_t blockIndex, char* data,
60 char* scratch) = 0;
61};

108// The encryption provider is used to create a cipher stream for a specific
109// file. The returned cipher stream will be used for actual
110// encryption/decryption actions.
111//
112// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
113// because RocksDB is not exception-safe. This could cause undefined behavior
114// including data loss, unreported corruption, deadlocks, and more.
115class EncryptionProvider : public Customizable {
116 public:
117 virtual ~EncryptionProvider() {}
118
119 // Creates a new EncryptionProvider from the input config_options and value.
120 // The value describes the type of provider (and potentially optional
121 // configuration parameters) used to create this provider.
122 // For example, if the value is "CTR", a CTREncryptionProvider will be
123 // created. If the value is end with "://test" (e.g CTR://test"), the
124 // provider will be initialized in "TEST" mode prior to being returned.
125 //
126 // @param config_options Options to control how this provider is created
127 // and initialized.
128 // @param value The value might be:
129 // - CTR Create a CTR provider
130 // - CTR://test Create a CTR provider and initialize it for tests.
131 // @param result The new provider object
132 // @return OK if the provider was successfully created
133 // @return NotFound if an invalid name was specified in the value
134 // @return InvalidArgument if either the options were not valid
135 static Status CreateFromString(const ConfigOptions& config_options,
136 const std::string& value,
137 std::shared_ptr<EncryptionProvider>* result);
138
139 static const char* Type() { return "EncryptionProvider"; }
140
141 // Short-cut method to create a CTR-provider
142 static std::shared_ptr<EncryptionProvider> NewCTRProvider(
143 const std::shared_ptr<BlockCipher>& cipher);
144
145 // GetPrefixLength returns the length of the prefix that is added to every
146 // file and used for storing encryption options. For optimal performance, the
147 // prefix length should be a multiple of the page size.
148 virtual size_t GetPrefixLength() const = 0;
149
150 // CreateNewPrefix initialized an allocated block of prefix memory
151 // for a new file.
152 virtual Status CreateNewPrefix(const std::string& fname, char* prefix,
153 size_t prefixLength) const = 0;
154
155 // Method to add a new cipher key for use by the EncryptionProvider.
156 // @param descriptor Descriptor for this key
157 // @param cipher The cryptographic key to use
158 // @param len The length of the cipher key
159 // @param for_write If true, this cipher should be used for writing files.
160 // If false, this cipher should only be used for reading
161 // files
162 // @return OK if the cipher was successfully added to the provider, non-OK
163 // otherwise
164 virtual Status AddCipher(const std::string& descriptor, const char* cipher,
165 size_t len, bool for_write) = 0;
166
167 // CreateCipherStream creates a block access cipher stream for a file given
168 // name and options.
169 virtual Status CreateCipherStream(
170 const std::string& fname, const EnvOptions& options, Slice& prefix,
171 std::unique_ptr<BlockAccessCipherStream>* result) = 0;
172
173 // Returns a string representing an encryption marker prefix for this
174 // provider. If a marker is provided, this marker can be used to tell whether
175 // a file is encrypted by this provider. The marker will also be part of any
176 // encryption prefix for this provider.
177 virtual std::string GetMarker() const { return ""; }
178};
