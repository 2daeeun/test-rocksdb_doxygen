14Status BlockBasedTable::IndexReaderCommon::ReadIndexBlock(
15 const BlockBasedTable* table, FilePrefetchBuffer* prefetch_buffer,
16 const ReadOptions& read_options, bool use_cache, GetContext* get_context,
17 BlockCacheLookupContext* lookup_context,
18 CachableEntry<Block>* index_block) {
19 PERF_TIMER_GUARD(read_index_block_nanos);
20
21 assert(table != nullptr);
22 assert(index_block != nullptr);
23 assert(index_block->IsEmpty());
24
25 const Rep* const rep = table->get_rep();
26 assert(rep != nullptr);
27
28 const Status s = table->RetrieveBlock(
29 prefetch_buffer, read_options, rep->index_handle,
30 UncompressionDict::GetEmptyDict(), &index_block->As<Block_kIndex>(),
31 get_context, lookup_context, /* for_compaction */ false, use_cache,
32 /* async_read */ false, /* use_block_cache_for_lookup */ true);
33
34 return s;
35}

37Status BlockBasedTable::IndexReaderCommon::GetOrReadIndexBlock(
38 GetContext* get_context, BlockCacheLookupContext* lookup_context,
39 CachableEntry<Block>* index_block, const ReadOptions& ro) const {
40 assert(index_block != nullptr);
41
42 if (!index_block_.IsEmpty()) {
43 index_block->SetUnownedValue(index_block_.GetValue());
44 return Status::OK();
45 }
46
47 return ReadIndexBlock(table_, /*prefetch_buffer=*/nullptr, ro,
48 cache_index_blocks(), get_context, lookup_context,
49 index_block);
50}

52void BlockBasedTable::IndexReaderCommon::EraseFromCacheBeforeDestruction(
53 uint32_t uncache_aggressiveness) {
54 if (uncache_aggressiveness > 0) {
55 if (index_block_.IsCached()) {
56 index_block_.ResetEraseIfLastRef();
57 } else {
58 table()->EraseFromCache(table()->get_rep()->index_handle);
59 }
60 }
