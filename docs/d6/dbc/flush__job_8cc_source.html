52const char* GetFlushReasonString(FlushReason flush_reason) {
53 switch (flush_reason) {
54 case FlushReason::kOthers:
55 return "Other Reasons";
56 case FlushReason::kGetLiveFiles:
57 return "Get Live Files";
58 case FlushReason::kShutDown:
59 return "Shut down";
60 case FlushReason::kExternalFileIngestion:
61 return "External File Ingestion";
62 case FlushReason::kManualCompaction:
63 return "Manual Compaction";
64 case FlushReason::kWriteBufferManager:
65 return "Write Buffer Manager";
66 case FlushReason::kWriteBufferFull:
67 return "Write Buffer Full";
68 case FlushReason::kTest:
69 return "Test";
70 case FlushReason::kDeleteFiles:
71 return "Delete Files";
72 case FlushReason::kAutoCompaction:
73 return "Auto Compaction";
74 case FlushReason::kManualFlush:
75 return "Manual Flush";
76 case FlushReason::kErrorRecovery:
77 return "Error Recovery";
78 case FlushReason::kErrorRecoveryRetryFlush:
79 return "Error Recovery Retry Flush";
80 case FlushReason::kWalFull:
81 return "WAL Full";
82 case FlushReason::kCatchUpAfterErrorRecovery:
83 return "Catch Up After Error Recovery";
84 default:
85 return "Invalid";
86 }
87}

175void FlushJob::PickMemTable() {
176 db_mutex_->AssertHeld();
177 assert(!pick_memtable_called);
178 pick_memtable_called = true;
179
180 // Maximum "NextLogNumber" of the memtables to flush.
181 // When mempurge feature is turned off, this variable is useless
182 // because the memtables are implicitly sorted by increasing order of creation
183 // time. Therefore mems_->back()->GetNextLogNumber() is already equal to
184 // max_next_log_number. However when Mempurge is on, the memtables are no
185 // longer sorted by increasing order of creation time. Therefore this variable
186 // becomes necessary because mems_->back()->GetNextLogNumber() is no longer
187 // necessarily equal to max_next_log_number.
188 uint64_t max_next_log_number = 0;
189
190 // Save the contents of the earliest memtable as a new Table
191 cfd_->imm()->PickMemtablesToFlush(max_memtable_id_, &mems_,
192 &max_next_log_number);
193 if (mems_.empty()) {
194 return;
195 }
196
197 // Track effective cutoff user-defined timestamp during flush if
198 // user-defined timestamps can be stripped.
199 GetEffectiveCutoffUDTForPickedMemTables();
200 GetPrecludeLastLevelMinSeqno();
201
202 ReportFlushInputSize(mems_);
203
204 // entries mems are (implicitly) sorted in ascending order by their created
205 // time. We will use the first memtable's `edit` to keep the meta info for
206 // this flush.
207 ReadOnlyMemTable* m = mems_[0];
208 edit_ = m->GetEdits();
209 edit_->SetPrevLogNumber(0);
210 // SetLogNumber(log_num) indicates logs with number smaller than log_num
211 // will no longer be picked up for recovery.
212 edit_->SetLogNumber(max_next_log_number);
213 edit_->SetColumnFamily(cfd_->GetID());
214
215 // path 0 for level 0 file.
216 meta_.fd = FileDescriptor(versions_->NewFileNumber(), 0, 0);
217 meta_.epoch_number = cfd_->NewEpochNumber();
218
219 base_ = cfd_->current();
220 base_->Ref(); // it is likely that we do not need this reference
221}

223Status FlushJob::Run(LogsWithPrepTracker* prep_tracker, FileMetaData* file_meta,
224 bool* switched_to_mempurge, bool* skipped_since_bg_error,
225 ErrorHandler* error_handler) {
226 TEST_SYNC_POINT("FlushJob::Start");
227 db_mutex_->AssertHeld();
228 assert(pick_memtable_called);
229 // Mempurge threshold can be dynamically changed.
230 // For sake of consistency, mempurge_threshold is
231 // saved locally to maintain consistency in each
232 // FlushJob::Run call.
233 double mempurge_threshold =
234 mutable_cf_options_.experimental_mempurge_threshold;
235
236 AutoThreadOperationStageUpdater stage_run(ThreadStatus::STAGE_FLUSH_RUN);
237 if (mems_.empty()) {
238 ROCKS_LOG_BUFFER(log_buffer_, "[%s] No memtable to flush",
239 cfd_->GetName().c_str());
240 return Status::OK();
241 }
242
243 // I/O measurement variables
244 PerfLevel prev_perf_level = PerfLevel::kEnableTime;
245 uint64_t prev_write_nanos = 0;
246 uint64_t prev_fsync_nanos = 0;
247 uint64_t prev_range_sync_nanos = 0;
248 uint64_t prev_prepare_write_nanos = 0;
249 uint64_t prev_cpu_write_nanos = 0;
250 uint64_t prev_cpu_read_nanos = 0;
251 if (measure_io_stats_) {
252 prev_perf_level = GetPerfLevel();
253 SetPerfLevel(PerfLevel::kEnableTime);
254 prev_write_nanos = IOSTATS(write_nanos);
255 prev_fsync_nanos = IOSTATS(fsync_nanos);
256 prev_range_sync_nanos = IOSTATS(range_sync_nanos);
257 prev_prepare_write_nanos = IOSTATS(prepare_write_nanos);
258 prev_cpu_write_nanos = IOSTATS(cpu_write_nanos);
259 prev_cpu_read_nanos = IOSTATS(cpu_read_nanos);
260 }
261 Status mempurge_s = Status::NotFound("No MemPurge.");
262 if ((mempurge_threshold > 0.0) &&
263 (flush_reason_ == FlushReason::kWriteBufferFull) && (!mems_.empty()) &&
264 MemPurgeDecider(mempurge_threshold) && !(db_options_.atomic_flush)) {
265 cfd_->SetMempurgeUsed();
266 mempurge_s = MemPurge();
267 if (!mempurge_s.ok()) {
268 // Mempurge is typically aborted when the output
269 // bytes cannot be contained onto a single output memtable.
270 if (mempurge_s.IsAborted()) {
271 ROCKS_LOG_INFO(db_options_.info_log, "Mempurge process aborted: %s\n",
272 mempurge_s.ToString().c_str());
273 } else {
274 // However the mempurge process can also fail for
275 // other reasons (eg: new_mem->Add() fails).
276 ROCKS_LOG_WARN(db_options_.info_log, "Mempurge process failed: %s\n",
277 mempurge_s.ToString().c_str());
278 }
279 } else {
280 if (switched_to_mempurge) {
281 *switched_to_mempurge = true;
282 } else {
283 // The mempurge process was successful, but no switch_to_mempurge
284 // pointer provided so no way to propagate the state of flush job.
285 ROCKS_LOG_WARN(db_options_.info_log,
286 "Mempurge process succeeded"
287 "but no 'switched_to_mempurge' ptr provided.\n");
288 }
289 }
290 }
291 Status s;
292 if (mempurge_s.ok()) {
293 base_->Unref();
294 s = Status::OK();
295 } else {
296 // This will release and re-acquire the mutex.
297 s = WriteLevel0Table();
298 }
299
300 if (s.ok() && cfd_->IsDropped()) {
301 s = Status::ColumnFamilyDropped("Column family dropped during compaction");
302 }
303 if ((s.ok() || s.IsColumnFamilyDropped()) &&
304 shutting_down_->load(std::memory_order_acquire)) {
305 s = Status::ShutdownInProgress("Database shutdown");
306 }
307
308 if (s.ok()) {
309 s = MaybeIncreaseFullHistoryTsLowToAboveCutoffUDT();
310 }
311
312 if (!s.ok()) {
313 cfd_->imm()->RollbackMemtableFlush(
314 mems_, /*rollback_succeeding_memtables=*/!db_options_.atomic_flush);
315 } else if (write_manifest_) {
316 assert(!db_options_.atomic_flush);
317 if (!db_options_.atomic_flush &&
318 flush_reason_ != FlushReason::kErrorRecovery &&
319 flush_reason_ != FlushReason::kErrorRecoveryRetryFlush &&
320 error_handler && !error_handler->GetBGError().ok() &&
321 error_handler->IsBGWorkStopped()) {
322 cfd_->imm()->RollbackMemtableFlush(
323 mems_, /*rollback_succeeding_memtables=*/!db_options_.atomic_flush);
324 s = error_handler->GetBGError();
325 if (skipped_since_bg_error) {
326 *skipped_since_bg_error = true;
327 }
328 } else {
329 TEST_SYNC_POINT("FlushJob::InstallResults");
330 // Replace immutable memtable with the generated Table
331 s = cfd_->imm()->TryInstallMemtableFlushResults(
332 cfd_, mutable_cf_options_, mems_, prep_tracker, versions_, db_mutex_,
333 meta_.fd.GetNumber(), &job_context_->memtables_to_free, db_directory_,
334 log_buffer_, &committed_flush_jobs_info_,
335 !(mempurge_s.ok()) /* write_edit : true if no mempurge happened (or if aborted),
336 but 'false' if mempurge successful: no new min log number
337 or new level 0 file path to write to manifest. */);
338 }
339 }
340
341 if (s.ok() && file_meta != nullptr) {
342 *file_meta = meta_;
343 }
344 RecordFlushIOStats();
345
346 // When measure_io_stats_ is true, the default 512 bytes is not enough.
347 auto stream = event_logger_->LogToBuffer(log_buffer_, 1024);
348 stream << "job" << job_context_->job_id << "event"
349 << "flush_finished";
350 stream << "output_compression"
351 << CompressionTypeToString(output_compression_);
352 stream << "lsm_state";
353 stream.StartArray();
354 auto vstorage = cfd_->current()->storage_info();
355 for (int level = 0; level < vstorage->num_levels(); ++level) {
356 stream << vstorage->NumLevelFiles(level);
357 }
358 stream.EndArray();
359
360 const auto& blob_files = vstorage->GetBlobFiles();
361 if (!blob_files.empty()) {
362 assert(blob_files.front());
363 stream << "blob_file_head" << blob_files.front()->GetBlobFileNumber();
364
365 assert(blob_files.back());
366 stream << "blob_file_tail" << blob_files.back()->GetBlobFileNumber();
367 }
368
369 stream << "immutable_memtables" << cfd_->imm()->NumNotFlushed();
370
371 if (measure_io_stats_) {
372 if (prev_perf_level != PerfLevel::kEnableTime) {
373 SetPerfLevel(prev_perf_level);
374 }
375 stream << "file_write_nanos" << (IOSTATS(write_nanos) - prev_write_nanos);
376 stream << "file_range_sync_nanos"
377 << (IOSTATS(range_sync_nanos) - prev_range_sync_nanos);
378 stream << "file_fsync_nanos" << (IOSTATS(fsync_nanos) - prev_fsync_nanos);
379 stream << "file_prepare_write_nanos"
380 << (IOSTATS(prepare_write_nanos) - prev_prepare_write_nanos);
381 stream << "file_cpu_write_nanos"
382 << (IOSTATS(cpu_write_nanos) - prev_cpu_write_nanos);
383 stream << "file_cpu_read_nanos"
384 << (IOSTATS(cpu_read_nanos) - prev_cpu_read_nanos);
385 }
386
387 TEST_SYNC_POINT("FlushJob::End");
388 return s;
389}

397Status FlushJob::MemPurge() {
398 Status s;
399 db_mutex_->AssertHeld();
400 db_mutex_->Unlock();
401 assert(!mems_.empty());
402
403 // Measure purging time.
404 const uint64_t start_micros = clock_->NowMicros();
405 const uint64_t start_cpu_micros = clock_->CPUMicros();
406
407 MemTable* new_mem = nullptr;
408 // For performance/log investigation purposes:
409 // look at how much useful payload we harvest in the new_mem.
410 // This value is then printed to the DB log.
411 double new_mem_capacity = 0.0;
412
413 // Create two iterators, one for the memtable data (contains
414 // info from puts + deletes), and one for the memtable
415 // Range Tombstones (from DeleteRanges).
416 // TODO: plumb Env::IOActivity, Env::IOPriority
417 ReadOptions ro;
418 ro.total_order_seek = true;
419 Arena arena;
420 std::vector<InternalIterator*> memtables;
421 std::vector<std::unique_ptr<FragmentedRangeTombstoneIterator>>
422 range_del_iters;
423 for (ReadOnlyMemTable* m : mems_) {
424 memtables.push_back(m->NewIterator(ro, /*seqno_to_time_mapping=*/nullptr,
425 &arena, /*prefix_extractor=*/nullptr,
426 /*for_flush=*/true));
427 auto* range_del_iter = m->NewRangeTombstoneIterator(
428 ro, kMaxSequenceNumber, true /* immutable_memtable */);
429 if (range_del_iter != nullptr) {
430 range_del_iters.emplace_back(range_del_iter);
431 }
432 }
433
434 assert(!memtables.empty());
435 SequenceNumber first_seqno = kMaxSequenceNumber;
436 SequenceNumber earliest_seqno = kMaxSequenceNumber;
437 // Pick first and earliest seqno as min of all first_seqno
438 // and earliest_seqno of the mempurged memtables.
439 for (const auto& mem : mems_) {
440 first_seqno = mem->GetFirstSequenceNumber() < first_seqno
441 ? mem->GetFirstSequenceNumber()
442 : first_seqno;
443 earliest_seqno = mem->GetEarliestSequenceNumber() < earliest_seqno
444 ? mem->GetEarliestSequenceNumber()
445 : earliest_seqno;
446 }
447
448 ScopedArenaPtr<InternalIterator> iter(
449 NewMergingIterator(&(cfd_->internal_comparator()), memtables.data(),
450 static_cast<int>(memtables.size()), &arena));
451
452 auto* ioptions = cfd_->ioptions();
453
454 // Place iterator at the First (meaning most recent) key node.
455 iter->SeekToFirst();
456
457 const std::string* const full_history_ts_low = &(cfd_->GetFullHistoryTsLow());
458 std::unique_ptr<CompactionRangeDelAggregator> range_del_agg(
459 new CompactionRangeDelAggregator(&(cfd_->internal_comparator()),
460 existing_snapshots_,
461 full_history_ts_low));
462 for (auto& rd_iter : range_del_iters) {
463 range_del_agg->AddTombstones(std::move(rd_iter));
464 }
465
466 // If there is valid data in the memtable,
467 // or at least range tombstones, copy over the info
468 // to the new memtable.
469 if (iter->Valid() || !range_del_agg->IsEmpty()) {
470 // MaxSize is the size of a memtable.
471 size_t maxSize = mutable_cf_options_.write_buffer_size;
472 std::unique_ptr<CompactionFilter> compaction_filter;
473 if (ioptions->compaction_filter_factory != nullptr &&
474 ioptions->compaction_filter_factory->ShouldFilterTableFileCreation(
475 TableFileCreationReason::kFlush)) {
476 CompactionFilter::Context ctx;
477 ctx.is_full_compaction = false;
478 ctx.is_manual_compaction = false;
479 ctx.column_family_id = cfd_->GetID();
480 ctx.reason = TableFileCreationReason::kFlush;
481 compaction_filter =
482 ioptions->compaction_filter_factory->CreateCompactionFilter(ctx);
483 if (compaction_filter != nullptr &&
484 !compaction_filter->IgnoreSnapshots()) {
485 s = Status::NotSupported(
486 "CompactionFilter::IgnoreSnapshots() = false is not supported "
487 "anymore.");
488 return s;
489 }
490 }
491
492 new_mem = new MemTable((cfd_->internal_comparator()), *(cfd_->ioptions()),
493 mutable_cf_options_, cfd_->write_buffer_mgr(),
494 earliest_seqno, cfd_->GetID());
495 assert(new_mem != nullptr);
496
497 Env* env = db_options_.env;
498 assert(env);
499 MergeHelper merge(
500 env, (cfd_->internal_comparator()).user_comparator(),
501 (ioptions->merge_operator).get(), compaction_filter.get(),
502 ioptions->logger, true /* internal key corruption is not ok */,
503 existing_snapshots_.empty() ? 0 : existing_snapshots_.back(),
504 snapshot_checker_);
505 assert(job_context_);
506 SequenceNumber job_snapshot_seq = job_context_->GetJobSnapshotSequence();
507 const std::atomic<bool> kManualCompactionCanceledFalse{false};
508 CompactionIterator c_iter(
509 iter.get(), (cfd_->internal_comparator()).user_comparator(), &merge,
510 kMaxSequenceNumber, &existing_snapshots_, earliest_snapshot_,
511 earliest_write_conflict_snapshot_, job_snapshot_seq, snapshot_checker_,
512 env, ShouldReportDetailedTime(env, ioptions->stats),
513 true /* internal key corruption is not ok */, range_del_agg.get(),
514 nullptr, ioptions->allow_data_in_errors,
515 ioptions->enforce_single_del_contracts,
516 /*manual_compaction_canceled=*/kManualCompactionCanceledFalse,
517 false /* must_count_input_entries */,
518 /*compaction=*/nullptr, compaction_filter.get(),
519 /*shutting_down=*/nullptr, ioptions->info_log, full_history_ts_low);
520
521 // Set earliest sequence number in the new memtable
522 // to be equal to the earliest sequence number of the
523 // memtable being flushed (See later if there is a need
524 // to update this number!).
525 new_mem->SetEarliestSequenceNumber(earliest_seqno);
526 // Likewise for first seq number.
527 new_mem->SetFirstSequenceNumber(first_seqno);
528 SequenceNumber new_first_seqno = kMaxSequenceNumber;
529
530 c_iter.SeekToFirst();
531
532 // Key transfer
533 for (; c_iter.Valid(); c_iter.Next()) {
534 const ParsedInternalKey ikey = c_iter.ikey();
535 const Slice value = c_iter.value();
536 new_first_seqno =
537 ikey.sequence < new_first_seqno ? ikey.sequence : new_first_seqno;
538
539 // Should we update "OldestKeyTime" ???? -> timestamp appear
540 // to still be an "experimental" feature.
541 s = new_mem->Add(
542 ikey.sequence, ikey.type, ikey.user_key, value,
543 nullptr, // KV protection info set as nullptr since it
544 // should only be useful for the first add to
545 // the original memtable.
546 false, // : allow concurrent_memtable_writes_
547 // Not seen as necessary for now.
548 nullptr, // get_post_process_info(m) must be nullptr
549 // when concurrent_memtable_writes is switched off.
550 nullptr); // hint, only used when concurrent_memtable_writes_
551 // is switched on.
552 if (!s.ok()) {
553 break;
554 }
555
556 // If new_mem has size greater than maxSize,
557 // then rollback to regular flush operation,
558 // and destroy new_mem.
559 if (new_mem->ApproximateMemoryUsage() > maxSize) {
560 s = Status::Aborted("Mempurge filled more than one memtable.");
561 new_mem_capacity = 1.0;
562 break;
563 }
564 }
565
566 // Check status and propagate
567 // potential error status from c_iter
568 if (!s.ok()) {
569 c_iter.status().PermitUncheckedError();
570 } else if (!c_iter.status().ok()) {
571 s = c_iter.status();
572 }
573
574 // Range tombstone transfer.
575 if (s.ok()) {
576 auto range_del_it = range_del_agg->NewIterator();
577 for (range_del_it->SeekToFirst(); range_del_it->Valid();
578 range_del_it->Next()) {
579 auto tombstone = range_del_it->Tombstone();
580 new_first_seqno =
581 tombstone.seq_ < new_first_seqno ? tombstone.seq_ : new_first_seqno;
582 s = new_mem->Add(
583 tombstone.seq_, // Sequence number
584 kTypeRangeDeletion, // KV type
585 tombstone.start_key_, // Key is start key.
586 tombstone.end_key_, // Value is end key.
587 nullptr, // KV protection info set as nullptr since it
588 // should only be useful for the first add to
589 // the original memtable.
590 false, // : allow concurrent_memtable_writes_
591 // Not seen as necessary for now.
592 nullptr, // get_post_process_info(m) must be nullptr
593 // when concurrent_memtable_writes is switched off.
594 nullptr); // hint, only used when concurrent_memtable_writes_
595 // is switched on.
596
597 if (!s.ok()) {
598 break;
599 }
600
601 // If new_mem has size greater than maxSize,
602 // then rollback to regular flush operation,
603 // and destroy new_mem.
604 if (new_mem->ApproximateMemoryUsage() > maxSize) {
605 s = Status::Aborted(Slice("Mempurge filled more than one memtable."));
606 new_mem_capacity = 1.0;
607 break;
608 }
609 }
610 }
611
612 // If everything happened smoothly and new_mem contains valid data,
613 // decide if it is flushed to storage or kept in the imm()
614 // memtable list (memory).
615 if (s.ok() && (new_first_seqno != kMaxSequenceNumber)) {
616 // Rectify the first sequence number, which (unlike the earliest seq
617 // number) needs to be present in the new memtable.
618 new_mem->SetFirstSequenceNumber(new_first_seqno);
619
620 // The new_mem is added to the list of immutable memtables
621 // only if it filled at less than 100% capacity and isn't flagged
622 // as in need of being flushed.
623 if (new_mem->ApproximateMemoryUsage() < maxSize &&
624 !(new_mem->ShouldFlushNow())) {
625 // Construct fragmented memtable range tombstones without mutex
626 new_mem->ConstructFragmentedRangeTombstones();
627 db_mutex_->Lock();
628 // Take the newest id, so that memtables in MemtableList don't have
629 // out-of-order memtable ids.
630 uint64_t new_mem_id = mems_.back()->GetID();
631
632 new_mem->SetID(new_mem_id);
633 // Take the latest memtable's next log number.
634 new_mem->SetNextLogNumber(mems_.back()->GetNextLogNumber());
635
636 // This addition will not trigger another flush, because
637 // we do not call EnqueuePendingFlush().
638 cfd_->imm()->Add(new_mem, &job_context_->memtables_to_free);
639 new_mem->Ref();
640 // Piggyback FlushJobInfo on the first flushed memtable.
641 db_mutex_->AssertHeld();
642 meta_.fd.file_size = 0;
643 mems_[0]->SetFlushJobInfo(GetFlushJobInfo());
644 db_mutex_->Unlock();
645 } else {
646 s = Status::Aborted(Slice("Mempurge filled more than one memtable."));
647 new_mem_capacity = 1.0;
648 if (new_mem) {
649 job_context_->memtables_to_free.push_back(new_mem);
650 }
651 }
652 } else {
653 // In this case, the newly allocated new_mem is empty.
654 assert(new_mem != nullptr);
655 job_context_->memtables_to_free.push_back(new_mem);
656 }
657 }
658
659 // Reacquire the mutex for WriteLevel0 function.
660 db_mutex_->Lock();
661
662 // If mempurge successful, don't write input tables to level0,
663 // but write any full output table to level0.
664 if (s.ok()) {
665 TEST_SYNC_POINT("DBImpl::FlushJob:MemPurgeSuccessful");
666 } else {
667 TEST_SYNC_POINT("DBImpl::FlushJob:MemPurgeUnsuccessful");
668 }
669 const uint64_t micros = clock_->NowMicros() - start_micros;
670 const uint64_t cpu_micros = clock_->CPUMicros() - start_cpu_micros;
671 ROCKS_LOG_INFO(db_options_.info_log,
672 "[%s] [JOB %d] Mempurge lasted %" PRIu64
673 " microseconds, and %" PRIu64
674 " cpu "
675 "microseconds. Status is %s ok. Perc capacity: %f\n",
676 cfd_->GetName().c_str(), job_context_->job_id, micros,
677 cpu_micros, s.ok() ? "" : "not", new_mem_capacity);
678
679 return s;
680}

682bool FlushJob::MemPurgeDecider(double threshold) {
683 // Never trigger mempurge if threshold is not a strictly positive value.
684 if (!(threshold > 0.0)) {
685 return false;
686 }
687 if (threshold > (1.0 * mems_.size())) {
688 return true;
689 }
690 // Payload and useful_payload (in bytes).
691 // The useful payload ratio of a given MemTable
692 // is estimated to be useful_payload/payload.
693 uint64_t payload = 0, useful_payload = 0, entry_size = 0;
694
695 // Local variables used repetitively inside the for-loop
696 // when iterating over the sampled entries.
697 Slice key_slice, value_slice;
698 ParsedInternalKey res;
699 SnapshotImpl min_snapshot;
700 std::string vget;
701 Status mget_s, parse_s;
702 MergeContext merge_context;
703 SequenceNumber max_covering_tombstone_seq = 0, sqno = 0,
704 min_seqno_snapshot = 0;
705 bool get_res, can_be_useful_payload, not_in_next_mems;
706
707 // If estimated_useful_payload is > threshold,
708 // then flush to storage, else MemPurge.
709 double estimated_useful_payload = 0.0;
710 // Cochran formula for determining sample size.
711 // 95% confidence interval, 7% precision.
712 // n0 = (1.96*1.96)*0.25/(0.07*0.07) = 196.0
713 double n0 = 196.0;
714 // TODO: plumb Env::IOActivity, Env::IOPriority
715 ReadOptions ro;
716 ro.total_order_seek = true;
717
718 // Iterate over each memtable of the set.
719 for (auto mem_iter = std::begin(mems_); mem_iter != std::end(mems_);
720 ++mem_iter) {
721 ReadOnlyMemTable* mt = *mem_iter;
722
723 // Else sample from the table.
724 uint64_t nentries = mt->NumEntries();
725 // Corrected Cochran formula for small populations
726 // (converges to n0 for large populations).
727 uint64_t target_sample_size =
728 static_cast<uint64_t>(ceil(n0 / (1.0 + (n0 / nentries))));
729 std::unordered_set<const char*> sentries = {};
730 // Populate sample entries set.
731 mt->UniqueRandomSample(target_sample_size, &sentries);
732
733 // Estimate the garbage ratio by comparing if
734 // each sample corresponds to a valid entry.
735 for (const char* ss : sentries) {
736 key_slice = GetLengthPrefixedSlice(ss);
737 parse_s = ParseInternalKey(key_slice, &res, true /*log_err_key*/);
738 if (!parse_s.ok()) {
739 ROCKS_LOG_WARN(db_options_.info_log,
740 "Memtable Decider: ParseInternalKey did not parse "
741 "key_slice %s successfully.",
742 key_slice.data());
743 }
744
745 // Size of the entry is "key size (+ value size if KV entry)"
746 entry_size = key_slice.size();
747 if (res.type == kTypeValue) {
748 value_slice =
749 GetLengthPrefixedSlice(key_slice.data() + key_slice.size());
750 entry_size += value_slice.size();
751 }
752
753 // Count entry bytes as payload.
754 payload += entry_size;
755
756 LookupKey lkey(res.user_key, kMaxSequenceNumber);
757
758 // Paranoia: zero out these values just in case.
759 max_covering_tombstone_seq = 0;
760 sqno = 0;
761
762 // Pick the oldest existing snapshot that is more recent
763 // than the sequence number of the sampled entry.
764 min_seqno_snapshot = kMaxSequenceNumber;
765 for (SequenceNumber seq_num : existing_snapshots_) {
766 if (seq_num > res.sequence && seq_num < min_seqno_snapshot) {
767 min_seqno_snapshot = seq_num;
768 }
769 }
770 min_snapshot.number_ = min_seqno_snapshot;
771 ro.snapshot =
772 min_seqno_snapshot < kMaxSequenceNumber ? &min_snapshot : nullptr;
773
774 // Estimate if the sample entry is valid or not.
775 get_res = mt->Get(lkey, &vget, /*columns=*/nullptr, /*timestamp=*/nullptr,
776 &mget_s, &merge_context, &max_covering_tombstone_seq,
777 &sqno, ro, true /* immutable_memtable */);
778 if (!get_res) {
779 ROCKS_LOG_WARN(
780 db_options_.info_log,
781 "Memtable Get returned false when Get(sampled entry). "
782 "Yet each sample entry should exist somewhere in the memtable, "
783 "unrelated to whether it has been deleted or not.");
784 }
785
786 // TODO(bjlemaire): evaluate typeMerge.
787 // This is where the sampled entry is estimated to be
788 // garbage or not. Note that this is a garbage *estimation*
789 // because we do not include certain items such as
790 // CompactionFitlers triggered at flush, or if the same delete
791 // has been inserted twice or more in the memtable.
792
793 // Evaluate if the entry can be useful payload
794 // Situation #1: entry is a KV entry, was found in the memtable mt
795 // and the sequence numbers match.
796 can_be_useful_payload = (res.type == kTypeValue) && get_res &&
797 mget_s.ok() && (sqno == res.sequence);
798
799 // Situation #2: entry is a delete entry, was found in the memtable mt
800 // (because gres==true) and no valid KV entry is found.
801 // (note: duplicate delete entries are also taken into
802 // account here, because the sequence number 'sqno'
803 // in memtable->Get(&sqno) operation is set to be equal
804 // to the most recent delete entry as well).
805 can_be_useful_payload |=
806 ((res.type == kTypeDeletion) || (res.type == kTypeSingleDeletion)) &&
807 mget_s.IsNotFound() && get_res && (sqno == res.sequence);
808
809 // If there is a chance that the entry is useful payload
810 // Verify that the entry does not appear in the following memtables
811 // (memtables with greater memtable ID/larger sequence numbers).
812 if (can_be_useful_payload) {
813 not_in_next_mems = true;
814 for (auto next_mem_iter = mem_iter + 1;
815 next_mem_iter != std::end(mems_); next_mem_iter++) {
816 if ((*next_mem_iter)
817 ->Get(lkey, &vget, /*columns=*/nullptr, /*timestamp=*/nullptr,
818 &mget_s, &merge_context, &max_covering_tombstone_seq,
819 &sqno, ro, true /* immutable_memtable */)) {
820 not_in_next_mems = false;
821 break;
822 }
823 }
824 if (not_in_next_mems) {
825 useful_payload += entry_size;
826 }
827 }
828 }
829 if (payload > 0) {
830 // We use the estimated useful payload ratio to
831 // evaluate how many of the memtable bytes are useful bytes.
832 estimated_useful_payload +=
833 (mt->ApproximateMemoryUsage()) * (useful_payload * 1.0 / payload);
834
835 ROCKS_LOG_INFO(db_options_.info_log,
836 "Mempurge sampling [CF %s] - found garbage ratio from "
837 "sampling: %f. Threshold is %f\n",
838 cfd_->GetName().c_str(),
839 (payload - useful_payload) * 1.0 / payload, threshold);
840 } else {
841 ROCKS_LOG_WARN(db_options_.info_log,
842 "Mempurge sampling: null payload measured, and collected "
843 "sample size is %zu\n.",
844 sentries.size());
845 }
846 }
847 // We convert the total number of useful payload bytes
848 // into the proportion of memtable necessary to store all these bytes.
849 // We compare this proportion with the threshold value.
850 return ((estimated_useful_payload / mutable_cf_options_.write_buffer_size) <
851 threshold);
852}
