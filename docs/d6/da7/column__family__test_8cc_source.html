103 Slice Value(int k, std::string* storage) {
104 if (k == 0) {

116 void Build(int base, int n, int flush_every = 0) {
117 std::string key_space, value_space;
118 WriteBatch batch;
119
120 for (int i = 0; i < n; i++) {
121 if (flush_every != 0 && i != 0 && i % flush_every == 0) {

123 dbi->TEST_FlushMemTable();

125

129 batch.Clear();
130 batch.Put(handles_[0], key, Value(keyi, &value_space));
131 batch.Put(handles_[1], key, Value(keyi, &value_space));
132 batch.Put(handles_[2], key, Value(keyi, &value_space));

137 void CheckMissed() {
138 uint64_t next_expected = 0;
139 uint64_t missed = 0;
140 int bad_keys = 0;
141 int bad_values = 0;
142 int correct = 0;
143 std::string value_space;
144 for (int cf = 0; cf < 3; cf++) {
145 next_expected = 0;
146 Iterator* iter = db_->NewIterator(ReadOptions(false, true), handles_[cf]);
147 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
148 uint64_t key;
149 Slice in(iter->key());
150 in.remove_prefix(3);
151 if (!ConsumeDecimalNumber(&in, &key) || !in.empty() ||
152 key < next_expected) {
153 bad_keys++;
154 continue;
155 }
156 missed += (key - next_expected);
157 next_expected = key + 1;
158 if (iter->value() != Value(static_cast<int>(key), &value_space)) {
159 bad_values++;
160 } else {
161 correct++;
162 }
163 }
164 delete iter;
165 }
166
167 ASSERT_EQ(0, bad_keys);
168 ASSERT_EQ(0, bad_values);
169 ASSERT_EQ(0, missed);
170 (void)correct;
171 }

417 env_->SleepForMicroseconds(micros_wait_for_log_deletion);
418 int ret = 0;
419 VectorLogPtr wal_files;
420 Status s;
421 // GetSortedWalFiles is a flakey function -- it gets all the wal_dir
422 // children files and then later checks for their existence. if some of the
423 // log files doesn't exist anymore, it reports an error. it does all of this
424 // without DB mutex held, so if a background process deletes the log file

434 EXPECT_OK(s);
435 for (const auto& wal : wal_files) {
436 if (wal->Type() == kAliveLogFile) {
437 ++ret;
438 }

441 return 0;

457 void CopyFile(const std::string& source, const std::string& destination,
458 uint64_t size = 0) {
459 const EnvOptions soptions;
460 std::unique_ptr<SequentialFile> srcfile;
461 ASSERT_OK(env_->NewSequentialFile(source, &srcfile, soptions));
462 std::unique_ptr<WritableFile> destfile;
463 ASSERT_OK(env_->NewWritableFile(destination, &destfile, soptions));
464
465 if (size == 0) {
466 // default argument means copy everything
467 ASSERT_OK(env_->GetFileSize(source, &size));
468 }
469
470 char buffer[4096];
471 Slice slice;
472 while (size > 0) {
473 uint64_t one = std::min(uint64_t(sizeof(buffer)), size);
474 ASSERT_OK(srcfile->Read(one, &slice, buffer));
475 ASSERT_OK(destfile->Append(slice));
476 size -= slice.size();
477 }
478 ASSERT_OK(destfile->Close());
479 }
