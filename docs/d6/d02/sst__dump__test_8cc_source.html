51static std::string MakeWideColumn(int i) {
52 std::string val = MakeValue(i);

100 template <std::size_t N>
101 void PopulateCommandArgs(const std::string& file_path, const char* command,
102 char* (&usage)[N]) const {
103 for (int i = 0; i < static_cast<int>(N); ++i) {
104 usage[i] = new char[kOptLength];
105 }
106 snprintf(usage[0], kOptLength, "./sst_dump");
107 snprintf(usage[1], kOptLength, "%s", command);
108 snprintf(usage[2], kOptLength, "--file=%s", file_path.c_str());

111 void createSST(const Options& opts, const std::string& file_name,
112 uint32_t wide_column_one_in = 0, bool range_del = false) {
113 Env* test_env = opts.env;
114 FileOptions file_options(opts);
115 ReadOptions read_options;
116 const ImmutableOptions imoptions(opts);
117 const MutableCFOptions moptions(opts);
118 ROCKSDB_NAMESPACE::InternalKeyComparator ikc(opts.comparator);
119 std::unique_ptr<TableBuilder> tb;
120
121 InternalTblPropCollFactories internal_tbl_prop_coll_factories;
122 std::unique_ptr<WritableFileWriter> file_writer;
123 ASSERT_OK(WritableFileWriter::Create(test_env->GetFileSystem(), file_name,
124 file_options, &file_writer, nullptr));
125
126 std::string column_family_name;
127 int unknown_level = -1;
128 const WriteOptions write_options;
129 tb.reset(opts.table_factory->NewTableBuilder(
130 TableBuilderOptions(
131 imoptions, moptions, read_options, write_options, ikc,
132 &internal_tbl_prop_coll_factories, CompressionType::kNoCompression,
133 CompressionOptions(),
134 TablePropertiesCollectorFactory::Context::kUnknownColumnFamily,
135 column_family_name, unknown_level, kUnknownNewestKeyTime),
136 file_writer.get()));
137
138 // Populate slightly more than 1K keys
139 uint32_t num_keys = kNumKey;
140 const char* comparator_name = ikc.user_comparator()->Name();
141 if (strcmp(comparator_name, ReverseBytewiseComparator()->Name()) == 0) {
142 for (int32_t i = num_keys; i > 0; i--) {
143 if (wide_column_one_in == 0 || i % wide_column_one_in != 0) {
144 tb->Add(MakeKey(i), MakeValue(i));
145 } else {
146 tb->Add(MakeKey(i, ValueType::kTypeWideColumnEntity),
147 MakeWideColumn(i));
148 }
149 }
150 } else if (strcmp(comparator_name,
151 test::BytewiseComparatorWithU64TsWrapper()->Name()) ==
152 0) {
153 for (uint32_t i = 0; i < num_keys; i++) {
154 tb->Add(MakeKeyWithTimeStamp(i, 100 + i), MakeValue(i));
155 }
156 } else {
157 uint32_t i = 0;
158 if (range_del) {
159 tb->Add(MakeKey(i, kTypeRangeDeletion), MakeValue(i + 1));
160 i = 1;
161 }
162 for (; i < num_keys; i++) {
163 if (wide_column_one_in == 0 || i % wide_column_one_in != 0) {
164 tb->Add(MakeKey(i), MakeValue(i));
165 } else {
166 tb->Add(MakeKey(i, ValueType::kTypeWideColumnEntity),
167 MakeWideColumn(i));
168 }
169 }
170 }
171 ASSERT_OK(tb->Finish());
172 ASSERT_OK(file_writer->Close(IOOptions()));
173 }
174
175 protected:
176 constexpr static int kNumKey = 1024;
177};
