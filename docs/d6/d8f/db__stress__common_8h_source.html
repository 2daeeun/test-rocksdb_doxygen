10// The test uses an array to compare against values written to the database.
11// Keys written to the array are in 1:1 correspondence to the actual values in
12// the database according to the formula in the function GenerateValue.
13
14// Space is reserved in the array from 0 to FLAGS_max_key and values are
15// randomly written/deleted/read from those positions. During verification we
16// compare all the positions in the array. To shorten/elongate the running
17// time, you could change the settings: FLAGS_max_key, FLAGS_ops_per_thread,
18// (sometimes also FLAGS_threads).
19//
20// NOTE that if FLAGS_test_batches_snapshots is set, the test will have
21// different behavior. See comment of the flag for details.

461inline enum ROCKSDB_NAMESPACE::CompressionType StringToCompressionType(
462 const char* ctype) {
463 assert(ctype);
464
465 ROCKSDB_NAMESPACE::CompressionType ret_compression_type;
466
467 if (!strcasecmp(ctype, "disable")) {
468 ret_compression_type = ROCKSDB_NAMESPACE::kDisableCompressionOption;
469 } else if (!strcasecmp(ctype, "none")) {
470 ret_compression_type = ROCKSDB_NAMESPACE::kNoCompression;
471 } else if (!strcasecmp(ctype, "snappy")) {
472 ret_compression_type = ROCKSDB_NAMESPACE::kSnappyCompression;
473 } else if (!strcasecmp(ctype, "zlib")) {
474 ret_compression_type = ROCKSDB_NAMESPACE::kZlibCompression;
475 } else if (!strcasecmp(ctype, "bzip2")) {
476 ret_compression_type = ROCKSDB_NAMESPACE::kBZip2Compression;
477 } else if (!strcasecmp(ctype, "lz4")) {
478 ret_compression_type = ROCKSDB_NAMESPACE::kLZ4Compression;
479 } else if (!strcasecmp(ctype, "lz4hc")) {
480 ret_compression_type = ROCKSDB_NAMESPACE::kLZ4HCCompression;
481 } else if (!strcasecmp(ctype, "xpress")) {
482 ret_compression_type = ROCKSDB_NAMESPACE::kXpressCompression;
483 } else if (!strcasecmp(ctype, "zstd")) {
484 ret_compression_type = ROCKSDB_NAMESPACE::kZSTD;
485 } else {
486 fprintf(stderr, "Cannot parse compression type '%s'\n", ctype);
487 ret_compression_type =
488 ROCKSDB_NAMESPACE::kSnappyCompression; // default value
489 }
490 if (ret_compression_type != ROCKSDB_NAMESPACE::kDisableCompressionOption &&
491 !CompressionTypeSupported(ret_compression_type)) {
492 // Use no compression will be more portable but considering this is
493 // only a stress test and snappy is widely available. Use snappy here.
494 ret_compression_type = ROCKSDB_NAMESPACE::kSnappyCompression;
495 }
496 return ret_compression_type;
497}

542inline std::vector<std::string> SplitString(std::string src) {
543 std::vector<std::string> ret;
544 if (src.empty()) {
545 return ret;
546 }
547 size_t pos = 0;
548 size_t pos_comma;
549 while ((pos_comma = src.find(',', pos)) != std::string::npos) {
550 ret.push_back(src.substr(pos, pos_comma - pos));
551 pos = pos_comma + 1;
552 }
553 ret.push_back(src.substr(pos, src.length()));

582// Append `val` to `*key` in fixed-width big-endian format
583inline void AppendIntToString(uint64_t val, std::string* key) {
584 // PutFixed64 uses little endian
585 PutFixed64(key, val);
586 // Reverse to get big endian
587 char* int_data = &((*key)[key->size() - sizeof(uint64_t)]);
588 for (size_t i = 0; i < sizeof(uint64_t) / 2; ++i) {

593// A struct for maintaining the parameters for generating variable length keys
594struct KeyGenContext {
595 // Number of adjacent keys in one cycle of key lengths
596 uint64_t window;
597 // Number of keys of each possible length in a given window
598 std::vector<uint64_t> weights;
599};
600extern KeyGenContext key_gen_ctx;
601
602// Generate a variable length key string from the given int64 val. The
603// order of the keys is preserved. The key could be anywhere from 8 to
604// max_key_len * 8 bytes.
605// The algorithm picks the length based on the
606// offset of the val within a configured window and the distribution of the
607// number of keys of various lengths in that window. For example, if x, y, x are
608// the weights assigned to each possible key length, the keys generated would be
609// - {0}...{x-1}
610// {(x-1),0}..{(x-1),(y-1)},{(x-1),(y-1),0}..{(x-1),(y-1),(z-1)} and so on.
611// Additionally, a trailer of 0-7 bytes could be appended.
612inline std::string Key(int64_t val) {
613 uint64_t window = key_gen_ctx.window;
614 size_t levels = key_gen_ctx.weights.size();
615 std::string key;
616 // Over-reserve and for now do not bother `shrink_to_fit()` since the key
617 // strings are transient.
618 key.reserve(FLAGS_max_key_len * 8);
619
620 uint64_t window_idx = static_cast<uint64_t>(val) / window;
621 uint64_t offset = static_cast<uint64_t>(val) % window;
622 for (size_t level = 0; level < levels; ++level) {
623 uint64_t weight = key_gen_ctx.weights[level];
624 uint64_t pfx;
625 if (level == 0) {
626 pfx = window_idx * weight;
627 } else {
628 pfx = 0;
629 }
630 pfx += offset >= weight ? weight - 1 : offset;
631 AppendIntToString(pfx, &key);
632 if (offset < weight) {
633 // Use the bottom 3 bits of offset as the number of trailing 'x's in the
634 // key. If the next key is going to be of the next level, then skip the
635 // trailer as it would break ordering. If the key length is already at
636 // max, skip the trailer.
637 if (offset < weight - 1 && level < levels - 1) {
638 size_t trailer_len = offset & 0x7;
639 key.append(trailer_len, 'x');
640 }
641 break;
642 }
643 offset -= weight;
644 }
645
646 return key;
647}

649// Given a string key, map it to an index into the expected values buffer
650inline bool GetIntVal(std::string big_endian_key, uint64_t* key_p) {
651 size_t size_key = big_endian_key.size();
652 std::vector<uint64_t> prefixes;
653
654 assert(size_key <= key_gen_ctx.weights.size() * sizeof(uint64_t));
655
656 std::string little_endian_key;
657 little_endian_key.resize(size_key);
658 for (size_t start = 0; start + sizeof(uint64_t) <= size_key;
659 start += sizeof(uint64_t)) {
660 size_t end = start + sizeof(uint64_t);
661 for (size_t i = 0; i < sizeof(uint64_t); ++i) {
662 little_endian_key[start + i] = big_endian_key[end - 1 - i];
663 }
664 Slice little_endian_slice =
665 Slice(&little_endian_key[start], sizeof(uint64_t));
666 uint64_t pfx;
667 if (!GetFixed64(&little_endian_slice, &pfx)) {
668 return false;
669 }
670 prefixes.emplace_back(pfx);
671 }
672
673 uint64_t key = 0;
674 for (size_t i = 0; i < prefixes.size(); ++i) {
675 uint64_t pfx = prefixes[i];
676 key += (pfx / key_gen_ctx.weights[i]) * key_gen_ctx.window +
677 pfx % key_gen_ctx.weights[i];
678 if (i < prefixes.size() - 1) {
679 // The encoding writes a `key_gen_ctx.weights[i] - 1` that counts for
680 // `key_gen_ctx.weights[i]` when there are more prefixes to come. So we
681 // need to add back the one here as we're at a non-last prefix.
682 ++key;
683 }
684 }
685 *key_p = key;
686 return true;
687}

689// Given a string prefix, map it to the first corresponding index in the
690// expected values buffer.
691inline bool GetFirstIntValInPrefix(std::string big_endian_prefix,
692 uint64_t* key_p) {
693 size_t size_key = big_endian_prefix.size();
694 // Pad with zeros to make it a multiple of 8. This function may be called
695 // with a prefix, in which case we return the first index that falls
696 // inside or outside that prefix, dependeing on whether the prefix is
697 // the start of upper bound of a scan
698 unsigned int pad = sizeof(uint64_t) - (size_key % sizeof(uint64_t));
699 if (pad < sizeof(uint64_t)) {
700 big_endian_prefix.append(pad, '\0');
701 }
702 return GetIntVal(std::move(big_endian_prefix), key_p);
703}
