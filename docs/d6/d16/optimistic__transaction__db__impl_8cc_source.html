69Status OptimisticTransactionDB::Open(
70 const DBOptions& db_options,
71 const OptimisticTransactionDBOptions& occ_options,
72 const std::string& dbname,
73 const std::vector<ColumnFamilyDescriptor>& column_families,
74 std::vector<ColumnFamilyHandle*>* handles,
75 OptimisticTransactionDB** dbptr) {
76 Status s;
77 DB* db;
78
79 std::vector<ColumnFamilyDescriptor> column_families_copy = column_families;
80
81 // Enable MemTable History if not already enabled
82 for (auto& column_family : column_families_copy) {
83 ColumnFamilyOptions* options = &column_family.options;
84
85 if (options->max_write_buffer_size_to_maintain == 0 &&
86 options->max_write_buffer_number_to_maintain == 0) {
87 // Setting to -1 will set the History size to
88 // max_write_buffer_number * write_buffer_size.
89 options->max_write_buffer_size_to_maintain = -1;
90 }
91 }
92
93 s = DB::Open(db_options, dbname, column_families_copy, handles, &db);
94
95 if (s.ok()) {
96 *dbptr = new OptimisticTransactionDBImpl(db, occ_options);
97 }
98
99 return s;
100}
