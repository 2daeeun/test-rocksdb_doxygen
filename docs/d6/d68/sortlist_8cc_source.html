13bool SortList::FullMergeV2(const MergeOperationInput& merge_in,

52void SortList::MakeVector(std::vector<int>& operand, Slice slice) const {
53 do {
54 const char* begin = slice.data_;
55 while (*slice.data_ != ',' && *slice.data_) {
56 slice.data_++;
57 }
58 operand.push_back(std::stoi(std::string(begin, slice.data_)));
59 } while (0 != *slice.data_++);
60}

62std::vector<int> SortList::Merge(std::vector<int>& left,
63 std::vector<int>& right) const {
64 // Fill the resultant vector with sorted results from both vectors
65 std::vector<int> result;
66 unsigned left_it = 0, right_it = 0;
67
68 while (left_it < left.size() && right_it < right.size()) {
69 // If the left value is smaller than the right it goes next
70 // into the resultant vector
71 if (left[left_it] < right[right_it]) {
72 result.push_back(left[left_it]);
73 left_it++;
74 } else {
75 result.push_back(right[right_it]);
76 right_it++;
77 }
78 }
79
80 // Push the remaining data from both vectors onto the resultant
81 while (left_it < left.size()) {
82 result.push_back(left[left_it]);
83 left_it++;
84 }
85
86 while (right_it < right.size()) {
87 result.push_back(right[right_it]);
88 right_it++;
89 }
90
91 return result;
92}
