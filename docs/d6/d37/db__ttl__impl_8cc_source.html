32bool TtlMergeOperator::FullMergeV2(const MergeOperationInput& merge_in,
33 MergeOperationOutput* merge_out) const {
34 const uint32_t ts_len = DBWithTTLImpl::kTSLength;
35 if (merge_in.existing_value && merge_in.existing_value->size() < ts_len) {
36 ROCKS_LOG_ERROR(merge_in.logger,
37 "Error: Could not remove timestamp from existing value.");
38 return false;
39 }
40
41 // Extract time-stamp from each operand to be passed to user_merge_op_
42 std::vector<Slice> operands_without_ts;
43 for (const auto& operand : merge_in.operand_list) {
44 if (operand.size() < ts_len) {
45 ROCKS_LOG_ERROR(merge_in.logger,
46 "Error: Could not remove timestamp from operand value.");
47 return false;
48 }
49 operands_without_ts.push_back(operand);
50 operands_without_ts.back().remove_suffix(ts_len);
51 }
52
53 // Apply the user merge operator (store result in *new_value)
54 bool good = true;
55 MergeOperationOutput user_merge_out(merge_out->new_value,
56 merge_out->existing_operand);
57 if (merge_in.existing_value) {
58 Slice existing_value_without_ts(merge_in.existing_value->data(),
59 merge_in.existing_value->size() - ts_len);
60 good = user_merge_op_->FullMergeV2(
61 MergeOperationInput(merge_in.key, &existing_value_without_ts,
62 operands_without_ts, merge_in.logger),
63 &user_merge_out);
64 } else {
65 good = user_merge_op_->FullMergeV2(
66 MergeOperationInput(merge_in.key, nullptr, operands_without_ts,
67 merge_in.logger),
68 &user_merge_out);
69 }
70 merge_out->op_failure_scope = user_merge_out.op_failure_scope;
71
72 // Return false if the user merge operator returned false
73 if (!good) {
74 return false;
75 }
76
77 if (merge_out->existing_operand.data()) {
78 merge_out->new_value.assign(merge_out->existing_operand.data(),
79 merge_out->existing_operand.size());
80 merge_out->existing_operand = Slice(nullptr, 0);
81 }
82
83 // Augment the *new_value with the ttl time-stamp
84 int64_t curtime;
85 if (!clock_->GetCurrentTime(&curtime).ok()) {
86 ROCKS_LOG_ERROR(
87 merge_in.logger,
88 "Error: Could not get current time to be attached internally "
89 "to the new value.");
90 return false;
91 } else {
92 char ts_string[ts_len];
93 EncodeFixed32(ts_string, (int32_t)curtime);
94 merge_out->new_value.append(ts_string, ts_len);
95 return true;
96 }
97}

99bool TtlMergeOperator::PartialMergeMulti(const Slice& key,
100 const std::deque<Slice>& operand_list,
101 std::string* new_value,
102 Logger* logger) const {
103 const uint32_t ts_len = DBWithTTLImpl::kTSLength;
104 std::deque<Slice> operands_without_ts;
105
106 for (const auto& operand : operand_list) {
107 if (operand.size() < ts_len) {
108 ROCKS_LOG_ERROR(logger, "Error: Could not remove timestamp from value.");
109 return false;
110 }
111
112 operands_without_ts.emplace_back(operand.data(), operand.size() - ts_len);
113 }
114
115 // Apply the user partial-merge operator (store result in *new_value)
116 assert(new_value);
117 if (!user_merge_op_->PartialMergeMulti(key, operands_without_ts, new_value,
118 logger)) {
119 return false;
120 }
121
122 // Augment the *new_value with the ttl time-stamp
123 int64_t curtime;
124 if (!clock_->GetCurrentTime(&curtime).ok()) {
125 ROCKS_LOG_ERROR(
126 logger,
127 "Error: Could not get current time to be attached internally "
128 "to the new value.");
129 return false;
130 } else {
131 char ts_string[ts_len];
132 EncodeFixed32(ts_string, (int32_t)curtime);
133 new_value->append(ts_string, ts_len);
134 return true;
135 }

199bool TtlCompactionFilter::Filter(int level, const Slice& key,
200 const Slice& old_val, std::string* new_val,
201 bool* value_changed) const {
202 if (DBWithTTLImpl::IsStale(old_val, ttl_, clock_)) {
203 return true;
204 }
205 if (user_comp_filter() == nullptr) {
206 return false;
207 }
208 assert(old_val.size() >= DBWithTTLImpl::kTSLength);
209 Slice old_val_without_ts(old_val.data(),
210 old_val.size() - DBWithTTLImpl::kTSLength);
211 if (user_comp_filter()->Filter(level, key, old_val_without_ts, new_val,
212 value_changed)) {
213 return true;
214 }
215 if (*value_changed) {
216 new_val->append(old_val.data() + old_val.size() - DBWithTTLImpl::kTSLength,
217 DBWithTTLImpl::kTSLength);
218 }
219 return false;
220}

354Status DBWithTTL::Open(
355 const DBOptions& db_options, const std::string& dbname,
356 const std::vector<ColumnFamilyDescriptor>& column_families,
357 std::vector<ColumnFamilyHandle*>* handles, DBWithTTL** dbptr,
358 const std::vector<int32_t>& ttls, bool read_only) {
359 DBWithTTLImpl::RegisterTtlClasses();
360 if (ttls.size() != column_families.size()) {
361 return Status::InvalidArgument(
362 "ttls size has to be the same as number of column families");
363 }
364
365 SystemClock* clock = (db_options.env == nullptr)
366 ? SystemClock::Default().get()
367 : db_options.env->GetSystemClock().get();
368
369 std::vector<ColumnFamilyDescriptor> column_families_sanitized =
370 column_families;
371 for (size_t i = 0; i < column_families_sanitized.size(); ++i) {
372 DBWithTTLImpl::SanitizeOptions(
373 ttls[i], &column_families_sanitized[i].options, clock);
374 }
375 DB* db;
376
377 Status st;
378 if (read_only) {
379 st = DB::OpenForReadOnly(db_options, dbname, column_families_sanitized,
380 handles, &db);
381 } else {
382 st = DB::Open(db_options, dbname, column_families_sanitized, handles, &db);
383 }
384 if (st.ok()) {
385 *dbptr = new DBWithTTLImpl(db);
386 } else {
387 *dbptr = nullptr;
388 }
389 return st;

410// Appends the current timestamp to the string.
411// Returns false if could not get the current_time, true if append succeeds
412Status DBWithTTLImpl::AppendTS(const Slice& val, std::string* val_with_ts,
413 SystemClock* clock) {
414 val_with_ts->reserve(kTSLength + val.size());
415 char ts_string[kTSLength];
416 int64_t curtime;
417 Status st = clock->GetCurrentTime(&curtime);
418 if (!st.ok()) {
419 return st;
420 }
421 EncodeFixed32(ts_string, (int32_t)curtime);
422 val_with_ts->append(val.data(), val.size());
423 val_with_ts->append(ts_string, kTSLength);
424 return st;
425}

427// Returns corruption if the length of the string is lesser than timestamp, or
428// timestamp refers to a time lesser than ttl-feature release time
429Status DBWithTTLImpl::SanityCheckTimestamp(const Slice& str) {
430 if (str.size() < kTSLength) {
431 return Status::Corruption("Error: value's length less than timestamp's\n");
432 }
433 // Checks that TS is not lesser than kMinTimestamp
434 // Gaurds against corruption & normal database opened incorrectly in ttl mode
435 int32_t timestamp_value = DecodeFixed32(str.data() + str.size() - kTSLength);
436 if (timestamp_value < kMinTimestamp) {
437 return Status::Corruption("Error: Timestamp < ttl feature release time!\n");
438 }
439 return Status::OK();
440}

442// Checks if the string is stale or not according to TTl provided
443bool DBWithTTLImpl::IsStale(const Slice& value, int32_t ttl,
444 SystemClock* clock) {
445 if (ttl <= 0) { // Data is fresh if TTL is non-positive
446 return false;
447 }
448 int64_t curtime;
449 if (!clock->GetCurrentTime(&curtime).ok()) {
450 return false; // Treat the data as fresh if could not get current time
451 }
452 /* int32_t may overflow when timestamp_value + ttl
453 * for example ttl = 86400 * 365 * 15
454 * convert timestamp_value to int64_t
455 */
456 int64_t timestamp_value =
457 DecodeFixed32(value.data() + value.size() - kTSLength);
458 return (timestamp_value + ttl) < curtime;
459}

462Status DBWithTTLImpl::StripTS(PinnableSlice* pinnable_val) {
463 if (pinnable_val->size() < kTSLength) {

510void DBWithTTLImpl::MultiGet(const ReadOptions& options, const size_t num_keys,
511 ColumnFamilyHandle** column_families,
512 const Slice* keys, PinnableSlice* values,
513 std::string* timestamps, Status* statuses,
514 const bool /*sorted_input*/) {
515 if (timestamps) {
516 for (size_t i = 0; i < num_keys; ++i) {
517 statuses[i] = Status::NotSupported(
518 "MultiGet() returning timestamps not implemented.");
519 }
520 return;
521 }
522
523 db_->MultiGet(options, num_keys, column_families, keys, values, timestamps,
524 statuses);
525 for (size_t i = 0; i < num_keys; ++i) {
526 if (!statuses[i].ok()) {
527 continue;
528 }
529 PinnableSlice tmp_val = std::move(values[i]);
530 values[i].PinSelf(tmp_val);
531 assert(!values[i].IsPinned());
532 statuses[i] = SanityCheckTimestamp(values[i]);
533 if (!statuses[i].ok()) {
534 continue;
535 }
536 statuses[i] = StripTS(&values[i]);
537 }

564Status DBWithTTLImpl::Write(const WriteOptions& opts, WriteBatch* updates) {
565 class Handler : public WriteBatch::Handler {
566 public:
567 explicit Handler(SystemClock* clock) : clock_(clock) {}
568 WriteBatch updates_ttl;
569 Status PutCF(uint32_t column_family_id, const Slice& key,
570 const Slice& value) override {
571 std::string value_with_ts;
572 Status st = AppendTS(value, &value_with_ts, clock_);
573 if (!st.ok()) {
574 return st;
575 }
576 return WriteBatchInternal::Put(&updates_ttl, column_family_id, key,
577 value_with_ts);
578 }
579 Status MergeCF(uint32_t column_family_id, const Slice& key,
580 const Slice& value) override {
581 std::string value_with_ts;
582 Status st = AppendTS(value, &value_with_ts, clock_);
583 if (!st.ok()) {
584 return st;
585 }
586 return WriteBatchInternal::Merge(&updates_ttl, column_family_id, key,
587 value_with_ts);
588 }
589 Status DeleteCF(uint32_t column_family_id, const Slice& key) override {
590 return WriteBatchInternal::Delete(&updates_ttl, column_family_id, key);
591 }
592 Status DeleteRangeCF(uint32_t column_family_id, const Slice& begin_key,
593 const Slice& end_key) override {
594 return WriteBatchInternal::DeleteRange(&updates_ttl, column_family_id,
595 begin_key, end_key);
596 }
597 void LogData(const Slice& blob) override { updates_ttl.PutLogData(blob); }
598
599 private:
600 SystemClock* clock_;
601 };
602 Handler handler(GetEnv()->GetSystemClock().get());
603 Status st = updates->Iterate(&handler);
604 if (!st.ok()) {
605 return st;
606 } else {
607 return db_->Write(opts, &(handler.updates_ttl));
608 }
609}
