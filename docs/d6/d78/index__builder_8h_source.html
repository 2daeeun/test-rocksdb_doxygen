24// The interface for building index.
25// Instruction for adding a new concrete IndexBuilder:
26// 1. Create a subclass instantiated from IndexBuilder.
27// 2. Add a new entry associated with that subclass in TableOptions::IndexType.
28// 3. Add a create function for the new subclass in CreateIndexBuilder.
29// Note: we can devise more advanced design to simplify the process for adding
30// new subclass, which will, on the other hand, increase the code complexity and
31// catch unwanted attention from readers. Given that we won't add/change
32// indexes frequently, it makes sense to just embrace a more straightforward
33// design that just works.
34class IndexBuilder {
35 public:
36 static IndexBuilder* CreateIndexBuilder(
37 BlockBasedTableOptions::IndexType index_type,
38 const InternalKeyComparator* comparator,
39 const InternalKeySliceTransform* int_key_slice_transform,
40 bool use_value_delta_encoding, const BlockBasedTableOptions& table_opt,
41 size_t ts_sz, bool persist_user_defined_timestamps);
42
43 // Index builder will construct a set of blocks which contain:
44 // 1. One primary index block.
45 // 2. (Optional) a set of metablocks that contains the metadata of the
46 // primary index.
47 struct IndexBlocks {
48 Slice index_block_contents;
49 std::unordered_map<std::string, Slice> meta_blocks;
50 };
51 IndexBuilder(const InternalKeyComparator* comparator, size_t ts_sz,
52 bool persist_user_defined_timestamps)
53 : comparator_(comparator),
54 ts_sz_(ts_sz),
55 persist_user_defined_timestamps_(persist_user_defined_timestamps) {}
56
57 virtual ~IndexBuilder() = default;
58
59 // Add a new index entry to index block.
60 //
61 // To allow further optimization, we provide `last_key_in_current_block` and
62 // `first_key_in_next_block`, based on which the specific implementation can
63 // determine the best index key to be used for the index block.
64 // Called before the OnKeyAdded() call for first_key_in_next_block.
65 // @last_key_in_current_block: TODO lifetime details
66 // @first_key_in_next_block: it will be nullptr if the entry being added is
67 // the last one in the table
68 // @separator_scratch: a scratch buffer to back a computed separator between
69 // those, as needed. May be modified on each call.
70 // @return: the key or separator stored in the index, which could be
71 // last_key_in_current_block or a computed separator backed by
72 // separator_scratch or last_key_in_current_block.
73 // REQUIRES: Finish() has not yet been called.
74 virtual Slice AddIndexEntry(const Slice& last_key_in_current_block,
75 const Slice* first_key_in_next_block,
76 const BlockHandle& block_handle,
77 std::string* separator_scratch) = 0;
78
79 // This method will be called whenever a key is added. The subclasses may
80 // override OnKeyAdded() if they need to collect additional information.
81 virtual void OnKeyAdded(const Slice& /*key*/) {}
82
83 // Inform the index builder that all entries has been written. Block builder
84 // may therefore perform any operation required for block finalization.
85 //
86 // REQUIRES: Finish() has not yet been called.
87 inline Status Finish(IndexBlocks* index_blocks) {
88 // Throw away the changes to last_partition_block_handle. It has no effect
89 // on the first call to Finish anyway.
90 BlockHandle last_partition_block_handle;
91 return Finish(index_blocks, last_partition_block_handle);
92 }
93
94 // This override of Finish can be utilized to build the 2nd level index in
95 // PartitionIndexBuilder.
96 //
97 // index_blocks will be filled with the resulting index data. If the return
98 // value is Status::InComplete() then it means that the index is partitioned
99 // and the callee should keep calling Finish until Status::OK() is returned.
100 // In that case, last_partition_block_handle is pointer to the block written
101 // with the result of the last call to Finish. This can be utilized to build
102 // the second level index pointing to each block of partitioned indexes. The
103 // last call to Finish() that returns Status::OK() populates index_blocks with
104 // the 2nd level index content.
105 virtual Status Finish(IndexBlocks* index_blocks,
106 const BlockHandle& last_partition_block_handle) = 0;
107
108 // Get the size for index block. Must be called after ::Finish.
109 virtual size_t IndexSize() const = 0;
110
111 virtual bool seperator_is_key_plus_seq() { return true; }
112
113 protected:
114 // Given the last key in current block and the first key in the next block,
115 // return true if internal key should be used as separator, false if user key
116 // can be used as separator.
117 inline bool ShouldUseKeyPlusSeqAsSeparator(
118 const Slice& last_key_in_current_block,
119 const Slice& first_key_in_next_block) {
120 Slice l_user_key = ExtractUserKey(last_key_in_current_block);
121 Slice r_user_key = ExtractUserKey(first_key_in_next_block);
122 // If user defined timestamps are not persisted. All the user keys will
123 // act like they have minimal timestamp. Only having user key is not
124 // sufficient, even if they are different user keys for now, they have to be
125 // different user keys without the timestamp part.
126 return persist_user_defined_timestamps_
127 ? comparator_->user_comparator()->Compare(l_user_key,
128 r_user_key) == 0
129 : comparator_->user_comparator()->CompareWithoutTimestamp(
130 l_user_key, r_user_key) == 0;
131 }
132
133 const InternalKeyComparator* comparator_;
134 // Size of user-defined timestamp in bytes.
135 size_t ts_sz_;
136 // Whether user-defined timestamp in the user key should be persisted when
137 // creating index block. If this flag is false, user-defined timestamp will
138 // be stripped from user key for each index entry, and the
139 // `first_internal_key` in `IndexValue` if it's included.
140 bool persist_user_defined_timestamps_;
141 // Set after ::Finish is called
142 size_t index_size_ = 0;
143};

145// This index builder builds space-efficient index block.
146//
147// Optimizations:
148// 1. Made block's `block_restart_interval` to be 1, which will avoid linear
149// search when doing index lookup (can be disabled by setting
150// index_block_restart_interval).
151// 2. Shorten the key length for index block. Other than honestly using the
152// last key in the data block as the index key, we instead find a shortest
153// substitute key that serves the same function.
154class ShortenedIndexBuilder : public IndexBuilder {
155 public:
156 ShortenedIndexBuilder(
157 const InternalKeyComparator* comparator,
158 const int index_block_restart_interval, const uint32_t format_version,
159 const bool use_value_delta_encoding,
160 BlockBasedTableOptions::IndexShorteningMode shortening_mode,
161 bool include_first_key, size_t ts_sz,
162 const bool persist_user_defined_timestamps)
163 : IndexBuilder(comparator, ts_sz, persist_user_defined_timestamps),
164 index_block_builder_(
165 index_block_restart_interval, true /*use_delta_encoding*/,
166 use_value_delta_encoding,
167 BlockBasedTableOptions::kDataBlockBinarySearch /* index_type */,
168 0.75 /* data_block_hash_table_util_ratio */, ts_sz,
169 persist_user_defined_timestamps, false /* is_user_key */),
170 index_block_builder_without_seq_(
171 index_block_restart_interval, true /*use_delta_encoding*/,
172 use_value_delta_encoding,
173 BlockBasedTableOptions::kDataBlockBinarySearch /* index_type */,
174 0.75 /* data_block_hash_table_util_ratio */, ts_sz,
175 persist_user_defined_timestamps, true /* is_user_key */),
176 use_value_delta_encoding_(use_value_delta_encoding),
177 include_first_key_(include_first_key),
178 shortening_mode_(shortening_mode) {
179 // Making the default true will disable the feature for old versions
180 seperator_is_key_plus_seq_ = (format_version <= 2);
181 }
182
183 void OnKeyAdded(const Slice& key) override {
184 if (include_first_key_ && current_block_first_internal_key_.empty()) {
185 current_block_first_internal_key_.assign(key.data(), key.size());
186 }
187 }
188
189 Slice AddIndexEntry(const Slice& last_key_in_current_block,
190 const Slice* first_key_in_next_block,
191 const BlockHandle& block_handle,
192 std::string* separator_scratch) override {
193 Slice separator;
194 if (first_key_in_next_block != nullptr) {
195 if (shortening_mode_ !=
196 BlockBasedTableOptions::IndexShorteningMode::kNoShortening) {
197 separator = FindShortestInternalKeySeparator(
198 *comparator_->user_comparator(), last_key_in_current_block,
199 *first_key_in_next_block, separator_scratch);
200 } else {
201 separator = last_key_in_current_block;
202 }
203 if (!seperator_is_key_plus_seq_ &&
204 ShouldUseKeyPlusSeqAsSeparator(last_key_in_current_block,
205 *first_key_in_next_block)) {
206 seperator_is_key_plus_seq_ = true;
207 }
208 } else {
209 if (shortening_mode_ == BlockBasedTableOptions::IndexShorteningMode::
210 kShortenSeparatorsAndSuccessor) {
211 separator = FindShortInternalKeySuccessor(
212 *comparator_->user_comparator(), last_key_in_current_block,
213 separator_scratch);
214 } else {
215 separator = last_key_in_current_block;
216 }
217 }
218
219 assert(!include_first_key_ || !current_block_first_internal_key_.empty());
220 // When UDT should not be persisted, the index block builders take care of
221 // stripping UDT from the key, for the first internal key contained in the
222 // IndexValue, we need to explicitly do the stripping here before passing
223 // it to the block builders.
224 std::string first_internal_key_buf;
225 Slice first_internal_key = current_block_first_internal_key_;
226 if (!current_block_first_internal_key_.empty() && ts_sz_ > 0 &&
227 !persist_user_defined_timestamps_) {
228 StripTimestampFromInternalKey(&first_internal_key_buf,
229 current_block_first_internal_key_, ts_sz_);
230 first_internal_key = first_internal_key_buf;
231 }
232 IndexValue entry(block_handle, first_internal_key);
233 std::string encoded_entry;
234 std::string delta_encoded_entry;
235 entry.EncodeTo(&encoded_entry, include_first_key_, nullptr);
236 if (use_value_delta_encoding_ && !last_encoded_handle_.IsNull()) {
237 entry.EncodeTo(&delta_encoded_entry, include_first_key_,
238 &last_encoded_handle_);
239 } else {
240 // If it's the first block, or delta encoding is disabled,
241 // BlockBuilder::Add() below won't use delta-encoded slice.
242 }
243 last_encoded_handle_ = block_handle;
244 const Slice delta_encoded_entry_slice(delta_encoded_entry);
245
246 // TODO(yuzhangyu): fix this when "FindShortInternalKeySuccessor"
247 // optimization is available.
248 // Timestamp aware comparator currently doesn't provide override for
249 // "FindShortInternalKeySuccessor" optimization. So the actual
250 // last key in current block is used as the key for indexing the current
251 // block. As a result, when UDTs should not be persisted, it's safe to strip
252 // away the UDT from key in index block as data block does the same thing.
253 // What are the implications if a "FindShortInternalKeySuccessor"
254 // optimization is provided.
255 index_block_builder_.Add(separator, encoded_entry,
256 &delta_encoded_entry_slice);
257 if (!seperator_is_key_plus_seq_) {
258 index_block_builder_without_seq_.Add(
259 ExtractUserKey(separator), encoded_entry, &delta_encoded_entry_slice);
260 }
261
262 current_block_first_internal_key_.clear();
263 return separator;
264 }
265
266 using IndexBuilder::Finish;
267 Status Finish(IndexBlocks* index_blocks,
268 const BlockHandle& /*last_partition_block_handle*/) override {
269 if (seperator_is_key_plus_seq_) {
270 index_blocks->index_block_contents = index_block_builder_.Finish();
271 } else {
272 index_blocks->index_block_contents =
273 index_block_builder_without_seq_.Finish();
274 }
275 index_size_ = index_blocks->index_block_contents.size();
276 return Status::OK();
277 }
278
279 size_t IndexSize() const override { return index_size_; }
280
281 bool seperator_is_key_plus_seq() override {
282 return seperator_is_key_plus_seq_;
283 }
284
285 // Changes *key to a short string >= *key.
286 //
287 static Slice FindShortestInternalKeySeparator(const Comparator& comparator,
288 const Slice& start,
289 const Slice& limit,
290 std::string* scratch);
291
292 static Slice FindShortInternalKeySuccessor(const Comparator& comparator,
293 const Slice& key,
294 std::string* scratch);
295
296 friend class PartitionedIndexBuilder;
297
298 private:
299 BlockBuilder index_block_builder_;
300 BlockBuilder index_block_builder_without_seq_;
301 const bool use_value_delta_encoding_;
302 bool seperator_is_key_plus_seq_;
303 const bool include_first_key_;
304 BlockBasedTableOptions::IndexShorteningMode shortening_mode_;
305 BlockHandle last_encoded_handle_ = BlockHandle::NullBlockHandle();
306 std::string current_block_first_internal_key_;
307};

309// HashIndexBuilder contains a binary-searchable primary index and the
310// metadata for secondary hash index construction.
311// The metadata for hash index consists two parts:
312// - a metablock that compactly contains a sequence of prefixes. All prefixes
313// are stored consectively without any metadata (like, prefix sizes) being
314// stored, which is kept in the other metablock.
315// - a metablock contains the metadata of the prefixes, including prefix size,
316// restart index and number of block it spans. The format looks like:
317//
318// +-----------------+---------------------------+---------------------+
319// <=prefix 1
320// | length: 4 bytes | restart interval: 4 bytes | num-blocks: 4 bytes |
321// +-----------------+---------------------------+---------------------+
322// <=prefix 2
323// | length: 4 bytes | restart interval: 4 bytes | num-blocks: 4 bytes |
324// +-----------------+---------------------------+---------------------+
325// | |
326// | .... |
327// | |
328// +-----------------+---------------------------+---------------------+
329// <=prefix n
330// | length: 4 bytes | restart interval: 4 bytes | num-blocks: 4 bytes |
331// +-----------------+---------------------------+---------------------+
332//
333// The reason of separating these two metablocks is to enable the efficiently
334// reuse the first metablock during hash index construction without unnecessary
335// data copy or small heap allocations for prefixes.
336class HashIndexBuilder : public IndexBuilder {
337 public:
338 HashIndexBuilder(const InternalKeyComparator* comparator,
339 const SliceTransform* hash_key_extractor,
340 int index_block_restart_interval, int format_version,
341 bool use_value_delta_encoding,
342 BlockBasedTableOptions::IndexShorteningMode shortening_mode,
343 size_t ts_sz, const bool persist_user_defined_timestamps)
344 : IndexBuilder(comparator, ts_sz, persist_user_defined_timestamps),
345 primary_index_builder_(comparator, index_block_restart_interval,
346 format_version, use_value_delta_encoding,
347 shortening_mode, /* include_first_key */ false,
348 ts_sz, persist_user_defined_timestamps),
349 hash_key_extractor_(hash_key_extractor) {}
350
351 Slice AddIndexEntry(const Slice& last_key_in_current_block,
352 const Slice* first_key_in_next_block,
353 const BlockHandle& block_handle,
354 std::string* separator_scratch) override {
355 ++current_restart_index_;
356 return primary_index_builder_.AddIndexEntry(
357 last_key_in_current_block, first_key_in_next_block, block_handle,
358 separator_scratch);
359 }
360
361 void OnKeyAdded(const Slice& key) override {
362 auto key_prefix = hash_key_extractor_->Transform(key);
363 bool is_first_entry = pending_block_num_ == 0;
364
365 // Keys may share the prefix
366 if (is_first_entry || pending_entry_prefix_ != key_prefix) {
367 if (!is_first_entry) {
368 FlushPendingPrefix();
369 }
370
371 // need a hard copy otherwise the underlying data changes all the time.
372 // TODO(kailiu) std::to_string() is expensive. We may speed up can avoid
373 // data copy.
374 pending_entry_prefix_ = key_prefix.ToString();
375 pending_block_num_ = 1;
376 pending_entry_index_ = static_cast<uint32_t>(current_restart_index_);
377 } else {
378 // entry number increments when keys share the prefix reside in
379 // different data blocks.
380 auto last_restart_index = pending_entry_index_ + pending_block_num_ - 1;
381 assert(last_restart_index <= current_restart_index_);
382 if (last_restart_index != current_restart_index_) {
383 ++pending_block_num_;
384 }
385 }
386 }
387
388 Status Finish(IndexBlocks* index_blocks,
389 const BlockHandle& last_partition_block_handle) override {
390 if (pending_block_num_ != 0) {
391 FlushPendingPrefix();
392 }
393 Status s = primary_index_builder_.Finish(index_blocks,
394 last_partition_block_handle);
395 index_blocks->meta_blocks.insert(
396 {kHashIndexPrefixesBlock.c_str(), prefix_block_});
397 index_blocks->meta_blocks.insert(
398 {kHashIndexPrefixesMetadataBlock.c_str(), prefix_meta_block_});
399 return s;
400 }
401
402 size_t IndexSize() const override {
403 return primary_index_builder_.IndexSize() + prefix_block_.size() +
404 prefix_meta_block_.size();
405 }
406
407 bool seperator_is_key_plus_seq() override {
408 return primary_index_builder_.seperator_is_key_plus_seq();
409 }
410
411 private:
412 void FlushPendingPrefix() {
413 prefix_block_.append(pending_entry_prefix_.data(),
414 pending_entry_prefix_.size());
415 PutVarint32Varint32Varint32(
416 &prefix_meta_block_,
417 static_cast<uint32_t>(pending_entry_prefix_.size()),
418 pending_entry_index_, pending_block_num_);
419 }
420
421 ShortenedIndexBuilder primary_index_builder_;
422 const SliceTransform* hash_key_extractor_;
423
424 // stores a sequence of prefixes
425 std::string prefix_block_;
426 // stores the metadata of prefixes
427 std::string prefix_meta_block_;
428
429 // The following 3 variables keeps unflushed prefix and its metadata.
430 // The details of block_num and entry_index can be found in
431 // "block_hash_index.{h,cc}"
432 uint32_t pending_block_num_ = 0;
433 uint32_t pending_entry_index_ = 0;
434 std::string pending_entry_prefix_;
435
436 uint64_t current_restart_index_ = 0;
437};
