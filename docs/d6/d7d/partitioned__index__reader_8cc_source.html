17Status PartitionIndexReader::Create(
18 const BlockBasedTable* table, const ReadOptions& ro,
19 FilePrefetchBuffer* prefetch_buffer, bool use_cache, bool prefetch,
20 bool pin, BlockCacheLookupContext* lookup_context,
21 std::unique_ptr<IndexReader>* index_reader) {
22 assert(table != nullptr);
23 assert(table->get_rep());
24 assert(!pin || prefetch);
25 assert(index_reader != nullptr);
26
27 CachableEntry<Block> index_block;
28 if (prefetch || !use_cache) {
29 const Status s =
30 ReadIndexBlock(table, prefetch_buffer, ro, use_cache,
31 /*get_context=*/nullptr, lookup_context, &index_block);
32 if (!s.ok()) {
33 return s;
34 }
35
36 if (use_cache && !pin) {
37 index_block.Reset();
38 }
39 }
40
41 index_reader->reset(new PartitionIndexReader(table, std::move(index_block)));
42
43 return Status::OK();
44}

46InternalIteratorBase<IndexValue>* PartitionIndexReader::NewIterator(
47 const ReadOptions& read_options, bool /* disable_prefix_seek */,
48 IndexBlockIter* iter, GetContext* get_context,
49 BlockCacheLookupContext* lookup_context) {
50 CachableEntry<Block> index_block;
51 const Status s = GetOrReadIndexBlock(get_context, lookup_context,
52 &index_block, read_options);
53 if (!s.ok()) {
54 if (iter != nullptr) {
55 iter->Invalidate(s);
56 return iter;
57 }
58
59 return NewErrorInternalIterator<IndexValue>(s);
60 }
61
62 const BlockBasedTable::Rep* rep = table()->rep_;
63 InternalIteratorBase<IndexValue>* it = nullptr;
64
65 Statistics* kNullStats = nullptr;
66 // Filters are already checked before seeking the index
67 if (!partition_map_.empty()) {
68 // We don't return pinned data from index blocks, so no need
69 // to set `block_contents_pinned`.
70 it = NewTwoLevelIterator(
71 new BlockBasedTable::PartitionedIndexIteratorState(table(),
72 &partition_map_),
73 index_block.GetValue()->NewIndexIterator(
74 internal_comparator()->user_comparator(),
75 rep->get_global_seqno(BlockType::kIndex), nullptr, kNullStats, true,
76 index_has_first_key(), index_key_includes_seq(),
77 index_value_is_full(), false /* block_contents_pinned */,
78 user_defined_timestamps_persisted()));
79 } else {
80 ReadOptions ro{read_options};
81 // FIXME? Possible regression seen in prefetch_test if this field is
82 // propagated
83 ro.readahead_size = ReadOptions{}.readahead_size;
84
85 // We don't return pinned data from index blocks, so no need
86 // to set `block_contents_pinned`.
87 std::unique_ptr<InternalIteratorBase<IndexValue>> index_iter(
88 index_block.GetValue()->NewIndexIterator(
89 internal_comparator()->user_comparator(),
90 rep->get_global_seqno(BlockType::kIndex), nullptr, kNullStats, true,
91 index_has_first_key(), index_key_includes_seq(),
92 index_value_is_full(), false /* block_contents_pinned */,
93 user_defined_timestamps_persisted()));
94
95 it = new PartitionedIndexIterator(
96 table(), ro, *internal_comparator(), std::move(index_iter),
97 lookup_context ? lookup_context->caller
98 : TableReaderCaller::kUncategorized);
99 }
100
101 assert(it != nullptr);
102 index_block.TransferTo(it);
103
104 return it;
105
106 // TODO(myabandeh): Update TwoLevelIterator to be able to make use of
107 // on-stack BlockIter while the state is on heap. Currentlly it assumes
108 // the first level iter is always on heap and will attempt to delete it
109 // in its destructor.

111Status PartitionIndexReader::CacheDependencies(
112 const ReadOptions& ro, bool pin, FilePrefetchBuffer* tail_prefetch_buffer) {
113 if (!partition_map_.empty()) {
114 // The dependencies are already cached since `partition_map_` is filled in
115 // an all-or-nothing manner.
116 return Status::OK();
117 }
118 // Before read partitions, prefetch them to avoid lots of IOs
119 BlockCacheLookupContext lookup_context{TableReaderCaller::kPrefetch};
120 const BlockBasedTable::Rep* rep = table()->rep_;
121 IndexBlockIter biter;
122 BlockHandle handle;
123 Statistics* kNullStats = nullptr;
124
125 CachableEntry<Block> index_block;
126 {
127 Status s = GetOrReadIndexBlock(nullptr /* get_context */, &lookup_context,
128 &index_block, ro);
129 if (!s.ok()) {
130 return s;
131 }
132 }
133
134 // We don't return pinned data from index blocks, so no need
135 // to set `block_contents_pinned`.
136 index_block.GetValue()->NewIndexIterator(
137 internal_comparator()->user_comparator(),
138 rep->get_global_seqno(BlockType::kIndex), &biter, kNullStats, true,
139 index_has_first_key(), index_key_includes_seq(), index_value_is_full(),
140 false /* block_contents_pinned */, user_defined_timestamps_persisted());
141 // Index partitions are assumed to be consecuitive. Prefetch them all.
142 // Read the first block offset
143 biter.SeekToFirst();
144 if (!biter.Valid()) {
145 // Empty index.
146 return biter.status();
147 }
148 handle = biter.value().handle;
149 uint64_t prefetch_off = handle.offset();
150
151 // Read the last block's offset
152 biter.SeekToLast();
153 if (!biter.Valid()) {
154 // Empty index.
155 return biter.status();
156 }
157 handle = biter.value().handle;
158 uint64_t last_off =
159 handle.offset() + BlockBasedTable::BlockSizeWithTrailer(handle);
160 uint64_t prefetch_len = last_off - prefetch_off;
161 std::unique_ptr<FilePrefetchBuffer> prefetch_buffer;
162 if (tail_prefetch_buffer == nullptr || !tail_prefetch_buffer->Enabled() ||
163 tail_prefetch_buffer->GetPrefetchOffset() > prefetch_off) {
164 rep->CreateFilePrefetchBuffer(ReadaheadParams(), &prefetch_buffer,
165 /*readaheadsize_cb*/ nullptr,
166 /*usage=*/FilePrefetchBufferUsage::kUnknown);
167 IOOptions opts;
168 {
169 Status s = rep->file->PrepareIOOptions(ro, opts);
170 if (s.ok()) {
171 s = prefetch_buffer->Prefetch(opts, rep->file.get(), prefetch_off,
172 static_cast<size_t>(prefetch_len));
173 }
174 if (!s.ok()) {
175 return s;
176 }
177 }
178 }
179 // For saving "all or nothing" to partition_map_
180 UnorderedMap<uint64_t, CachableEntry<Block>> map_in_progress;
181
182 // After prefetch, read the partitions one by one
183 biter.SeekToFirst();
184 size_t partition_count = 0;
185 for (; biter.Valid(); biter.Next()) {
186 handle = biter.value().handle;
187 CachableEntry<Block> block;
188 ++partition_count;
189 // TODO: Support counter batch update for partitioned index and
190 // filter blocks
191 Status s = table()->MaybeReadBlockAndLoadToCache(
192 prefetch_buffer ? prefetch_buffer.get() : tail_prefetch_buffer, ro,
193 handle, UncompressionDict::GetEmptyDict(),
194 /*for_compaction=*/false, &block.As<Block_kIndex>(),
195 /*get_context=*/nullptr, &lookup_context, /*contents=*/nullptr,
196 /*async_read=*/false, /*use_block_cache_for_lookup=*/true);
197
198 if (!s.ok()) {
199 return s;
200 }
201 if (block.GetValue() != nullptr) {
202 // Might need to "pin" some mmap-read blocks (GetOwnValue) if some
203 // partitions are successfully compressed (cached) and some are not
204 // compressed (mmap eligible)
205 if (block.IsCached() || block.GetOwnValue()) {
206 if (pin) {
207 map_in_progress[handle.offset()] = std::move(block);
208 }
209 }
210 }
211 }
212 Status s = biter.status();
213 // Save (pin) them only if everything checks out
214 if (map_in_progress.size() == partition_count && s.ok()) {
215 std::swap(partition_map_, map_in_progress);
216 }
217 return s;
218}

220void PartitionIndexReader::EraseFromCacheBeforeDestruction(
221 uint32_t uncache_aggressiveness) {
222 // NOTE: essentially a copy of
223 // PartitionedFilterBlockReader::EraseFromCacheBeforeDestruction
224 if (uncache_aggressiveness > 0) {
225 CachableEntry<Block> top_level_block;
226
227 ReadOptions ro_no_io;
228 ro_no_io.read_tier = ReadTier::kBlockCacheTier;
229 GetOrReadIndexBlock(/*get_context=*/nullptr,
230 /*lookup_context=*/nullptr, &top_level_block, ro_no_io)
231 .PermitUncheckedError();
232
233 if (!partition_map_.empty()) {
234 // All partitions present if any
235 for (auto& e : partition_map_) {
236 e.second.ResetEraseIfLastRef();
237 }
238 } else if (!top_level_block.IsEmpty()) {
239 IndexBlockIter biter;
240 const InternalKeyComparator* const comparator = internal_comparator();
241 Statistics* kNullStats = nullptr;
242 top_level_block.GetValue()->NewIndexIterator(
243 comparator->user_comparator(),
244 table()->get_rep()->get_global_seqno(BlockType::kIndex), &biter,
245 kNullStats, true /* total_order_seek */, index_has_first_key(),
246 index_key_includes_seq(), index_value_is_full(),
247 false /* block_contents_pinned */,
248 user_defined_timestamps_persisted());
249
250 UncacheAggressivenessAdvisor advisor(uncache_aggressiveness);
251 for (biter.SeekToFirst(); biter.Valid() && advisor.ShouldContinue();
252 biter.Next()) {
253 bool erased = table()->EraseFromCache(biter.value().handle);
254 advisor.Report(erased);
255 }
256 biter.status().PermitUncheckedError();
257 }
258 top_level_block.ResetEraseIfLastRef();
259 }
260 // Might be needed to un-cache a pinned top-level block
261 BlockBasedTable::IndexReaderCommon::EraseFromCacheBeforeDestruction(
262 uncache_aggressiveness);
263}
