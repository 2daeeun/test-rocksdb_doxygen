100// A class used for scanning MANIFEST file.
101// VersionEditHandler reads a MANIFEST file, parses the version edits, and
102// builds the version set's in-memory state, e.g. the version storage info for
103// the versions of column families. It replays all the version edits in one
104// MANIFEST file to build the end version.
105//
106// To use this class and its subclasses,
107// 1. Create an object of VersionEditHandler or its subclasses.
108// VersionEditHandler handler(read_only, column_families, version_set,
109// track_found_and_missing_files,
110// no_error_if_files_missing);
111// 2. Status s = handler.Iterate(reader, &db_id);
112// 3. Check s and handle possible errors.
113//
114// Not thread-safe, external synchronization is necessary if an object of
115// VersionEditHandler is shared by multiple threads.

117 public:
118 explicit VersionEditHandler(
119 bool read_only,
120 const std::vector<ColumnFamilyDescriptor>& column_families,
121 VersionSet* version_set, bool track_found_and_missing_files,
122 bool no_error_if_files_missing,
123 const std::shared_ptr<IOTracer>& io_tracer,
124 const ReadOptions& read_options, bool allow_incomplete_valid_version,
125 EpochNumberRequirement epoch_number_requirement =
126 EpochNumberRequirement::kMustPresent)
127 : VersionEditHandler(read_only, column_families, version_set,
128 track_found_and_missing_files,
129 no_error_if_files_missing, io_tracer, read_options,
130 /*skip_load_table_files=*/false,
131 allow_incomplete_valid_version,
132 epoch_number_requirement) {}
133
134 ~VersionEditHandler() override {}
135
136 const VersionEditParams& GetVersionEditParams() const {
137 return version_edit_params_;
138 }

140 void GetDbId(std::string* db_id) const {
141 if (db_id && version_edit_params_.HasDbId()) {

144 }

146 virtual Status VerifyFile(ColumnFamilyData* /*cfd*/,
147 const std::string& /*fpath*/, int /*level*/,
148 const FileMetaData& /*fmeta*/) {
149 return Status::OK();

158 protected:

244// A class similar to its base class, i.e. VersionEditHandler.
245// Unlike VersionEditHandler that only aims to build the end version, this class
246// supports building the most recent point in time version. A point in time
247// version is a version for which no files are missing, or if
248// `allow_incomplete_valid_version` is true, only a suffix of L0 files (and
249// their associated blob files) are missing.
250//
251// Building a point in time version when end version is not available can
252// be useful for best efforts recovery (options.best_efforts_recovery), which
253// uses this class and sets `allow_incomplete_valid_version` to true.
254// It's also useful for secondary instances/follower instances for which end
255// version could be transiently unavailable. These two cases use subclass
256// `ManifestTailer` and sets `allow_incomplete_valid_version` to false.
257//
258// Not thread-safe, external synchronization is necessary if an object of
259// VersionEditHandlerPointInTime is shared by multiple threads.
260class VersionEditHandlerPointInTime : public VersionEditHandler {
261 public:
262 VersionEditHandlerPointInTime(
263 bool read_only, std::vector<ColumnFamilyDescriptor> column_families,
264 VersionSet* version_set, const std::shared_ptr<IOTracer>& io_tracer,
265 const ReadOptions& read_options, bool allow_incomplete_valid_version,
266 EpochNumberRequirement epoch_number_requirement =
267 EpochNumberRequirement::kMustPresent);
268 ~VersionEditHandlerPointInTime() override;
269
270 bool HasMissingFiles() const;
271
272 virtual Status VerifyFile(ColumnFamilyData* cfd, const std::string& fpath,
273 int level, const FileMetaData& fmeta) override;
274 virtual Status VerifyBlobFile(ColumnFamilyData* cfd, uint64_t blob_file_num,
275 const BlobFileAddition& blob_addition) override;
276
277 protected:
278 Status OnAtomicGroupReplayBegin() override;
279 Status OnAtomicGroupReplayEnd() override;
280 void CheckIterationResult(const log::Reader& reader, Status* s) override;
281
282 ColumnFamilyData* DestroyCfAndCleanup(const VersionEdit& edit) override;
283 // `MaybeCreateVersionBeforeApplyEdit(..., false)` creates a version upon a
284 // negative edge trigger (transition from valid to invalid).
285 //
286 // `MaybeCreateVersionBeforeApplyEdit(..., true)` creates a version on a
287 // positive level trigger (state is valid).
288 Status MaybeCreateVersionBeforeApplyEdit(const VersionEdit& edit,
289 ColumnFamilyData* cfd,
290 bool force_create_version) override;
291
292 Status LoadTables(ColumnFamilyData* cfd,
293 bool prefetch_index_and_filter_in_cache,
294 bool is_initial_load) override;
295
296 std::unordered_map<uint32_t, Version*> versions_;
297
298 // `atomic_update_versions_` is for ensuring all-or-nothing AtomicGroup
299 // recoveries. When `atomic_update_versions_` is nonempty, it serves as a
300 // barrier to updating `versions_` until all its values are populated.
301 std::unordered_map<uint32_t, Version*> atomic_update_versions_;
302 // `atomic_update_versions_missing_` counts the nullptr values in
303 // `atomic_update_versions_`.
304 size_t atomic_update_versions_missing_;
305
306 bool in_atomic_group_ = false;
307
308 private:
309 bool AtomicUpdateVersionsCompleted();
310 bool AtomicUpdateVersionsContains(uint32_t cfid);
311 void AtomicUpdateVersionsDropCf(uint32_t cfid);
312
313 // This function is called for `Version*` updates for column families in an
314 // incomplete atomic update. It buffers `Version*` updates in
315 // `atomic_update_versions_`.
316 void AtomicUpdateVersionsPut(Version* version);
317
318 // This function is called upon completion of an atomic update. It applies
319 // `Version*` updates in `atomic_update_versions_` to `versions_`.
320 void AtomicUpdateVersionsApply();
321};

345

353 }
354

400 Status ApplyVersionEdit(VersionEdit& edit, ColumnFamilyData** cfd) override {
401 // Write out each individual edit
402 if (json_) {
403 // Print out DebugStrings. Can include non-terminating null characters.

405 fwrite(edit_dump_str.data(), sizeof(char), edit_dump_str.size(), stdout);
406 fwrite("\n", sizeof(char), 1, stdout);
407 } else if (verbose_) {
408 // Print out DebugStrings. Can include non-terminating null characters.
409 std::string edit_dump_str = edit.DebugString(hex_);
410 fwrite(edit_dump_str.data(), sizeof(char), edit_dump_str.size(), stdout);
411 }
412 ++count_;
413 return VersionEditHandler::ApplyVersionEdit(edit, cfd);
414 }
415
416 void CheckIterationResult(const log::Reader& reader, Status* s) override;
417
418 private:
419 const bool verbose_;
420 const bool hex_;
421 const bool json_;
422 int count_;
423};
