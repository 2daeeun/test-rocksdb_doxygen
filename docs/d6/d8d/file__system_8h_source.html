7// storage functionality like the filesystem etc. Callers
8// may wish to provide a custom FileSystem object when opening a database to
9// get fine gain control; e.g., to rate limit file system operations.
10//
11// All FileSystem implementations are safe for concurrent access from
12// multiple threads without any external synchronization.
13//
14// WARNING: Since this is a new interface, it is expected that there will be
15// some changes as storage systems are ported over.

93// Per-request options that can be passed down to the FileSystem
94// implementation. These are hints and are not necessarily guaranteed to be
95// honored. More hints can be added here in the future to indicate things like
96// storage media (HDD/SSD) to be used, replication level etc.
97struct IOOptions {
98 // Timeout for the operation in microseconds
99 std::chrono::microseconds timeout;
100
101 // DEPRECATED
102 // Priority - high or low
103 IOPriority prio;
104
105 // Priority used to charge rate limiter configured in file system level (if
106 // any)
107 // Limitation: right now RocksDB internal does not consider this
108 // rate_limiter_priority
109 Env::IOPriority rate_limiter_priority;
110
111 // Type of data being read/written
112 IOType type;
113
114 // EXPERIMENTAL
115 // An option map that's opaque to RocksDB. It can be used to implement a
116 // custom contract between a FileSystem user and the provider. This is only
117 // useful in cases where a RocksDB user directly uses the FileSystem or file
118 // object for their own purposes, and wants to pass extra options to APIs
119 // such as NewRandomAccessFile and NewWritableFile.
120 std::unordered_map<std::string, std::string> property_bag;
121
122 // Force directory fsync, some file systems like btrfs may skip directory
123 // fsync, set this to force the fsync
124 bool force_dir_fsync;
125
126 // Can be used by underlying file systems to skip recursing through sub
127 // directories and list only files in GetChildren API.
128 bool do_not_recurse;
129
130 // Setting this flag indicates a corruption was detected by a previous read,
131 // so the caller wants to re-read the data with much stronger data integrity
132 // checking and correction, i.e requests the file system to reconstruct the
133 // data from redundant copies and verify checksums, if available, in order
134 // to have a better chance of success. It is expected that this will have a
135 // much higher overhead than a normal read.
136 // This is a hint. At a minimum, the file system should implement this flag in
137 // FSRandomAccessFile::Read and FSSequentialFile::Read
138 // NOTE: The file system must support kVerifyAndReconstructRead in
139 // FSSupportedOps, otherwise this feature will not be used.
140 bool verify_and_reconstruct_read;
141
142 // EXPERIMENTAL
143 Env::IOActivity io_activity = Env::IOActivity::kUnknown;
144
145 IOOptions() : IOOptions(false) {}
146
147 explicit IOOptions(bool force_dir_fsync_)
148 : timeout(std::chrono::microseconds::zero()),
149 prio(IOPriority::kIOLow),
150 rate_limiter_priority(Env::IO_TOTAL),
151 type(IOType::kUnknown),
152 force_dir_fsync(force_dir_fsync_),
153 do_not_recurse(false),
154 verify_and_reconstruct_read(false) {}
155};

214// A structure to pass back some debugging information from the FileSystem
215// implementation to RocksDB in case of an IO error
216struct IODebugContext {
217 // file_path to be filled in by RocksDB in case of an error
218 std::string file_path;
219
220 // A map of counter names to values - set by the FileSystem implementation
221 std::map<std::string, uint64_t> counters;
222
223 // To be set by the FileSystem implementation
224 std::string msg;
225
226 // To be set by the underlying FileSystem implementation.
227 std::string request_id;
228
229 // In order to log required information in IO tracing for different
230 // operations, Each bit in trace_data stores which corresponding info from
231 // IODebugContext will be added in the trace. Foreg, if trace_data = 1, it
232 // means bit at position 0 is set so TraceData::kRequestID (request_id) will
233 // be logged in the trace record.
234 //
235 enum TraceData : char {
236 // The value of each enum represents the bitwise position for
237 // that information in trace_data which will be used by IOTracer for
238 // tracing. Make sure to add them sequentially.
239 kRequestID = 0,
240 };
241 uint64_t trace_data = 0;
242
243 IODebugContext() {}
244
245 void AddCounter(std::string& name, uint64_t value) {
246 counters.emplace(name, value);
247 }
248
249 // Called by underlying file system to set request_id and log request_id in
250 // IOTracing.
251 void SetRequestId(const std::string& _request_id) {
252 request_id = _request_id;
253 trace_data |= (1 << TraceData::kRequestID);
254 }
255
256 std::string ToString() {
257 std::ostringstream ss;
258 ss << file_path << ", ";
259 for (auto counter : counters) {
260 ss << counter.first << " = " << counter.second << ",";
261 }
262 ss << msg;
263 return ss.str();
264 }
265};

272// The FileSystem, FSSequentialFile, FSRandomAccessFile, FSWritableFile,
273// FSRandomRWFileclass, and FSDIrectory classes define the interface between
274// RocksDB and storage systems, such as Posix filesystems,
275// remote filesystems etc.
276// The interface allows for fine grained control of individual IO operations,
277// such as setting a timeout, prioritization, hints on data placement,
278// different handling based on type of IO etc.
279// This is accomplished by passing an instance of IOOptions to every
280// API call that can potentially perform IO. Additionally, each such API is
281// passed a pointer to a IODebugContext structure that can be used by the
282// storage system to include troubleshooting information. The return values
283// of the APIs is of type IOStatus, which can indicate an error code/sub-code,
284// as well as metadata about the error such as its scope and whether its
285// retryable.
286// NewCompositeEnv can be used to create an Env with a custom FileSystem for
287// DBOptions::env.
288//
289// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
290// because RocksDB is not exception-safe. This could cause undefined behavior
291// including data loss, unreported corruption, deadlocks, and more.

470 // Store in *result the attributes of the children of the specified directory.
471 // In case the implementation lists the directory prior to iterating the files
472 // and files are concurrently deleted, the deleted files will be omitted from
473 // result.
474 // The name attributes are relative to "dir".
475 // Original contents of *results are dropped.
476 // Returns OK if "dir" exists and "*result" contains its children.
477 // NotFound if "dir" does not exist, the calling process does not have
478 // permission to access "dir", or if "dir" is invalid.
479 // IOError if an IO Error was encountered
480 virtual IOStatus GetChildrenFileAttributes(
481 const std::string& dir, const IOOptions& options,
482 std::vector<FileAttributes>* result, IODebugContext* dbg) {
483 assert(result != nullptr);
484 std::vector<std::string> child_fnames;
485 IOStatus s = GetChildren(dir, options, &child_fnames, dbg);
486 if (!s.ok()) {
487 return s;
488 }
489 result->resize(child_fnames.size());
490 size_t result_size = 0;
491 for (size_t i = 0; i < child_fnames.size(); ++i) {
492 const std::string path = dir + "/" + child_fnames[i];
493 if (!(s = GetFileSize(path, options, &(*result)[result_size].size_bytes,
494 dbg))
495 .ok()) {
496 if (FileExists(path, options, dbg).IsNotFound()) {
497 // The file may have been deleted since we listed the directory
498 continue;
499 }
500 return s;
501 }
502 (*result)[result_size].name = std::move(child_fnames[i]);
503 result_size++;
504 }
505 result->resize(result_size);
506 return IOStatus::OK();
507 }

806// A read IO request structure for use in MultiRead and asynchronous Read APIs.
807struct FSReadRequest {
808 // Input parameter that represents the file offset in bytes.
809 uint64_t offset;
810
811 // Input parameter that represents the length to read in bytes. `result` only
812 // returns fewer bytes if end of file is hit (or `status` is not OK).
813 size_t len;
814
815 // A buffer that MultiRead() can optionally place data in. It can
816 // ignore this and allocate its own buffer.
817 // The lifecycle of scratch will be until IO is completed.
818 //
819 // In case of asynchronous reads, its an output parameter and it will be
820 // maintained until callback has been called. Scratch is allocated by RocksDB
821 // and will be passed to underlying FileSystem.
822 char* scratch;
823
824 // Output parameter set by MultiRead() to point to the start of the data
825 // buffer.
826 //
827 // When FSReadRequest::scratch is provided, this should point to
828 // FSReadRequest::scratch. When FSSupportedOps::kFSBuffer is enabled and
829 // FSReadRequest::scratch is nullptr, this points to the start of the
830 // data buffer allocated by the FileSystem.
831 //
832 // WARNING: Even with the FSSupportedOps::kFSBuffer optimization, you must
833 // still use result.data() to get the start of the actual data that was read.
834 // Do NOT treat FSReadRequest::fs_scratch as a char* to the start of a valid
835 // data buffer.
836 Slice result;
837
838 // Output parameter set by underlying FileSystem that represents status of
839 // read request.
840 IOStatus status;
841
842 // fs_scratch is a unique pointer to an arbitrary object allocated by the
843 // underlying FileSystem.
844 //
845 // Instead of having the FileSystem spend CPU cycles copying data into the
846 // FSReadRequest::scratch buffer provided by RocksDB, RocksDB can directly use
847 // a buffer allocated by the FileSystem.
848 //
849 // This optimization is enabled for MultiReads (sync and async) with non
850 // direct io, when these conditions hold:
851 // 1. The FileSystem has overriden the SupportedOps() API and set
852 // FSSupportedOps::kFSBuffer.
853 // 2. FSReadRequest::scratch is set to nullptr.
854 //
855 // RocksDB will:
856 // 1. Reuse the buffer allocated by the FileSystem.
857 // 2. Take ownership of the object managed by fs_scratch.
858 // 3. Handle invoking the custom deleter function from the FSAllocationPtr.
859 //
860 // WARNING: Do NOT assume that fs_scratch points to the start of the actual
861 // char* data returned by the read. As the type signature suggests, fs_scratch
862 // is a pointer to any arbitrary data type. Use result.data() to get a valid
863 // start to the real data. See https://github.com/facebook/rocksdb/pull/13189
864 // for more context.
865 FSAllocationPtr fs_scratch;
866};
