30static std::string PrintContents(WriteBatch* b,
31 bool merge_operator_supported = true) {
32 InternalKeyComparator cmp(BytewiseComparator());
33 auto factory = std::make_shared<SkipListFactory>();
34 Options options;
35 options.memtable_factory = factory;
36 if (merge_operator_supported) {
37 options.merge_operator.reset(new TestPutOperator());
38 }
39 ImmutableOptions ioptions(options);
40 WriteBufferManager wb(options.db_write_buffer_size);
41 MemTable* mem = new MemTable(cmp, ioptions, MutableCFOptions(options), &wb,
42 kMaxSequenceNumber, 0 /* column_family_id */);
43 mem->Ref();
44 std::string state;
45 ColumnFamilyMemTablesDefault cf_mems_default(mem);
46 Status s =
47 WriteBatchInternal::InsertInto(b, &cf_mems_default, nullptr, nullptr);
48 uint32_t count = 0;
49 int put_count = 0;
50 int timed_put_count = 0;
51 int delete_count = 0;
52 int single_delete_count = 0;
53 int delete_range_count = 0;
54 int merge_count = 0;
55 for (int i = 0; i < 2; ++i) {
56 Arena arena;
57 ScopedArenaPtr<InternalIterator> arena_iter_guard;
58 std::unique_ptr<InternalIterator> iter_guard;
59 InternalIterator* iter;
60 if (i == 0) {
61 iter = mem->NewIterator(ReadOptions(), /*seqno_to_time_mapping=*/nullptr,
62 &arena, /*prefix_extractor=*/nullptr,
63 /*for_flush=*/false);
64 arena_iter_guard.reset(iter);
65 } else {
66 iter = mem->NewRangeTombstoneIterator(ReadOptions(),
67 kMaxSequenceNumber /* read_seq */,
68 false /* immutable_memtable */);
69 iter_guard.reset(iter);
70 }
71 if (iter == nullptr) {
72 continue;
73 }
74 EXPECT_OK(iter->status());
75 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
76 ParsedInternalKey ikey;
77 ikey.clear();
78 EXPECT_OK(ParseInternalKey(iter->key(), &ikey, true /* log_err_key */));
79 switch (ikey.type) {
80 case kTypeValue:
81 state.append("Put(");
82 state.append(ikey.user_key.ToString());
83 state.append(", ");
84 state.append(iter->value().ToString());
85 state.append(")");
86 count++;
87 put_count++;
88 break;
89 case kTypeDeletion:
90 state.append("Delete(");
91 state.append(ikey.user_key.ToString());
92 state.append(")");
93 count++;
94 delete_count++;
95 break;
96 case kTypeSingleDeletion:
97 state.append("SingleDelete(");
98 state.append(ikey.user_key.ToString());
99 state.append(")");
100 count++;
101 single_delete_count++;
102 break;
103 case kTypeRangeDeletion:
104 state.append("DeleteRange(");
105 state.append(ikey.user_key.ToString());
106 state.append(", ");
107 state.append(iter->value().ToString());
108 state.append(")");
109 count++;
110 delete_range_count++;
111 break;
112 case kTypeMerge:
113 state.append("Merge(");
114 state.append(ikey.user_key.ToString());
115 state.append(", ");
116 state.append(iter->value().ToString());
117 state.append(")");
118 count++;
119 merge_count++;
120 break;
121 case kTypeValuePreferredSeqno: {
122 state.append("TimedPut(");
123 state.append(ikey.user_key.ToString());
124 state.append(", ");
125 auto [unpacked_value, unix_write_time] =
126 ParsePackedValueWithWriteTime(iter->value());
127 state.append(unpacked_value.ToString());
128 state.append(", ");
129 state.append(std::to_string(unix_write_time));
130 state.append(")");
131 count++;
132 timed_put_count++;
133 break;
134 }
135 default:
136 assert(false);
137 break;
138 }
139 state.append("@");
140 state.append(std::to_string(ikey.sequence));
141 }
142 EXPECT_OK(iter->status());
143 }
144 if (s.ok()) {
145 EXPECT_EQ(b->HasPut(), put_count > 0);
146 EXPECT_EQ(b->HasTimedPut(), timed_put_count > 0);
147 EXPECT_EQ(b->HasDelete(), delete_count > 0);
148 EXPECT_EQ(b->HasSingleDelete(), single_delete_count > 0);
149 EXPECT_EQ(b->HasDeleteRange(), delete_range_count > 0);
150 EXPECT_EQ(b->HasMerge(), merge_count > 0);
151 if (count != WriteBatchInternal::Count(b)) {
152 state.append("CountMismatch()");
153 }
154 } else {
155 state.append(s.ToString());
156 }
157 delete mem->Unref();
158 return state;
159}
