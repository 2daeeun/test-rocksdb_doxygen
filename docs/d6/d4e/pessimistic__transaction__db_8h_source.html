82 inline Status WriteWithConcurrencyControl(const WriteOptions& opts,
83 WriteBatch* updates) {
84 Status s;
85 if (opts.protection_bytes_per_key > 0) {
86 s = WriteBatchInternal::UpdateProtectionInfo(
87 updates, opts.protection_bytes_per_key);
88 }
89 if (s.ok()) {
90 // Need to lock all keys in this batch to prevent write conflicts with
91 // concurrent transactions.
92 Transaction* txn = BeginInternalTransaction(opts);
93 txn->DisableIndexing();
94
95 auto txn_impl = static_cast_with_check<PessimisticTransaction>(txn);
96
97 // Since commitBatch sorts the keys before locking, concurrent Write()
98 // operations will not cause a deadlock.
99 // In order to avoid a deadlock with a concurrent Transaction,
100 // Transactions should use a lock timeout.
101 s = txn_impl->CommitBatch(updates);
102
103 delete txn;
104 }
105
106 return s;
107 }

292inline Status PessimisticTransactionDB::FailIfBatchHasTs(
293 const WriteBatch* batch) {
294 if (batch != nullptr && WriteBatchInternal::HasKeyWithTimestamp(*batch)) {
295 return Status::NotSupported(
296 "Writes with timestamp must go through transaction API instead of "
297 "TransactionDB.");
298 }
299 return Status::OK();
300}
301
302inline Status PessimisticTransactionDB::FailIfCfEnablesTs(
303 const DB* db, const ColumnFamilyHandle* column_family) {
304 assert(db);
305 column_family = column_family ? column_family : db->DefaultColumnFamily();
306 assert(column_family);
307 const Comparator* const ucmp = column_family->GetComparator();
308 assert(ucmp);
309 if (ucmp->timestamp_size() > 0) {
310 return Status::NotSupported(
311 "Write operation with user timestamp must go through the transaction "
312 "API instead of TransactionDB.");
313 }
314 return Status::OK();
315}
