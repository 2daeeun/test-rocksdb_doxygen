39class BlockBasedTableBuilder : public TableBuilder {
40 public:
41 // Create a builder that will store the contents of the table it is
42 // building in *file. Does not close the file. It is up to the
43 // caller to close the file after calling Finish().
44 BlockBasedTableBuilder(const BlockBasedTableOptions& table_options,
45 const TableBuilderOptions& table_builder_options,
46 WritableFileWriter* file);
47
48 // No copying allowed
49 BlockBasedTableBuilder(const BlockBasedTableBuilder&) = delete;
50 BlockBasedTableBuilder& operator=(const BlockBasedTableBuilder&) = delete;
51
52 // REQUIRES: Either Finish() or Abandon() has been called.
53 ~BlockBasedTableBuilder();
54
55 // Add key,value to the table being constructed.
56 // REQUIRES: Unless key has type kTypeRangeDeletion, key is after any
57 // previously added non-kTypeRangeDeletion key according to
58 // comparator.
59 // REQUIRES: Finish(), Abandon() have not been called
60 void Add(const Slice& key, const Slice& value) override;
61
62 // Return non-ok iff some error has been detected.
63 Status status() const override;
64
65 // Return non-ok iff some error happens during IO.
66 IOStatus io_status() const override;
67
68 // Finish building the table. Stops using the file passed to the
69 // constructor after this function returns.
70 // REQUIRES: Finish(), Abandon() have not been called
71 Status Finish() override;
72
73 // Indicate that the contents of this builder should be abandoned. Stops
74 // using the file passed to the constructor after this function returns.
75 // If the caller is not going to call Finish(), it must call Abandon()
76 // before destroying this builder.
77 // REQUIRES: Finish(), Abandon() have not been called
78 void Abandon() override;
79
80 // Number of calls to Add() so far.
81 uint64_t NumEntries() const override;
82
83 bool IsEmpty() const override;
84
85 // Size of the file generated so far. If invoked after a successful
86 // Finish() call, returns the size of the final generated file.
87 uint64_t FileSize() const override;
88
89 // Estimated size of the file generated so far. This is used when
90 // FileSize() cannot estimate final SST size, e.g. parallel compression
91 // is enabled.
92 uint64_t EstimatedFileSize() const override;
93
94 // Get the size of the "tail" part of a SST file. "Tail" refers to
95 // all blocks after data blocks till the end of the SST file.
96 uint64_t GetTailSize() const override;
97
98 bool NeedCompact() const override;
99
100 // Get table properties
101 TableProperties GetTableProperties() const override;
102
103 // Get file checksum
104 std::string GetFileChecksum() const override;
105
106 // Get file checksum function name
107 const char* GetFileChecksumFuncName() const override;
108
109 void SetSeqnoTimeTableProperties(const SeqnoToTimeMapping& relevant_mapping,
110 uint64_t oldest_ancestor_time) override;
111
112 private:
113 bool ok() const { return status().ok(); }
114
115 // Transition state from buffered to unbuffered. See `Rep::State` API comment
116 // for details of the states.
117 // REQUIRES: `rep_->state == kBuffered`
118 void EnterUnbuffered();
119
120 // Call block's Finish() method and then
121 // - in buffered mode, buffer the uncompressed block contents.
122 // - in unbuffered mode, write the compressed block contents to file.
123 void WriteBlock(BlockBuilder* block, BlockHandle* handle,
124 BlockType blocktype);
125
126 // Compress and write block content to the file.
127 void WriteBlock(const Slice& block_contents, BlockHandle* handle,
128 BlockType block_type);
129 // Directly write data to the file.
130 void WriteMaybeCompressedBlock(
131 const Slice& block_contents, CompressionType, BlockHandle* handle,
132 BlockType block_type, const Slice* uncompressed_block_data = nullptr);
133
134 void SetupCacheKeyPrefix(const TableBuilderOptions& tbo);
135
136 template <typename TBlocklike>
137 Status InsertBlockInCache(const Slice& block_contents,
138 const BlockHandle* handle, BlockType block_type);
139
140 Status InsertBlockInCacheHelper(const Slice& block_contents,
141 const BlockHandle* handle,
142 BlockType block_type);
143
144 Status InsertBlockInCompressedCache(const Slice& block_contents,
145 const CompressionType type,
146 const BlockHandle* handle);
147
148 void WriteFilterBlock(MetaIndexBuilder* meta_index_builder);
149 void WriteIndexBlock(MetaIndexBuilder* meta_index_builder,
150 BlockHandle* index_block_handle);
151 void WritePropertiesBlock(MetaIndexBuilder* meta_index_builder);
152 void WriteCompressionDictBlock(MetaIndexBuilder* meta_index_builder);
153 void WriteRangeDelBlock(MetaIndexBuilder* meta_index_builder);
154 void WriteFooter(BlockHandle& metaindex_block_handle,
155 BlockHandle& index_block_handle);

157 struct Rep;
158 class BlockBasedTablePropertiesCollectorFactory;
159 class BlockBasedTablePropertiesCollector;
160 Rep* rep_;
161
162 struct ParallelCompressionRep;
163
164 // Advanced operation: flush any buffered key/value pairs to file.
165 // Can be used to ensure that two adjacent entries never live in
166 // the same data block. Most clients should not need to use this method.
167 // REQUIRES: Finish(), Abandon() have not been called
168 void Flush();
169
170 // Some compression libraries fail when the uncompressed size is bigger than
171 // int. If uncompressed size is bigger than kCompressionSizeLimit, don't
172 // compress it
173 const uint64_t kCompressionSizeLimit = std::numeric_limits<int>::max();
174
175 // Get blocks from mem-table walking thread, compress them and
176 // pass them to the write thread. Used in parallel compression mode only
177 void BGWorkCompression(const CompressionContext& compression_ctx,
178 UncompressionContext* verify_ctx);
179
180 // Given uncompressed block content, try to compress it and return result and
181 // compression type
182 void CompressAndVerifyBlock(const Slice& uncompressed_block_data,
183 bool is_data_block,
184 const CompressionContext& compression_ctx,
185 UncompressionContext* verify_ctx,
186 std::string* compressed_output,
187 Slice* result_block_contents,
188 CompressionType* result_compression_type,
189 Status* out_status);
190
191 // Get compressed blocks from BGWorkCompression and write them into SST
192 void BGWorkWriteMaybeCompressedBlock();
193
194 // Initialize parallel compression context and
195 // start BGWorkCompression and BGWorkWriteMaybeCompressedBlock threads
196 void StartParallelCompression();
197
198 // Stop BGWorkCompression and BGWorkWriteMaybeCompressedBlock threads
199 void StopParallelCompression();
200};
