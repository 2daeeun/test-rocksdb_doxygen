26class TransactionUtil {
27 public:
28 // Verifies there have been no commits to this key in the db since this
29 // sequence number. If user-defined timestamp is enabled, then also check
30 // no commits to this key in the db since the given ts.
31 //
32 // If cache_only is true, then this function will not attempt to read any
33 // SST files. This will make it more likely this function will
34 // return an error if it is unable to determine if there are any conflicts.
35 //
36 // See comment of CheckKey() for explanation of `snap_seq`, `ts`,
37 // `snap_checker` and `min_uncommitted`.
38 //
39 // Returns OK on success, BUSY if there is a conflicting write, or other error
40 // status for any unexpected errors.
41 static Status CheckKeyForConflicts(
42 DBImpl* db_impl, ColumnFamilyHandle* column_family,
43 const std::string& key, SequenceNumber snap_seq,
44 const std::string* const ts, bool cache_only,
45 ReadCallback* snap_checker = nullptr,
46 SequenceNumber min_uncommitted = kMaxSequenceNumber,
47 bool enable_udt_validation = true);
48
49 // For each key,SequenceNumber pair tracked by the LockTracker, this function
50 // will verify there have been no writes to the key in the db since that
51 // sequence number.
52 //
53 // Returns OK on success, BUSY if there is a conflicting write, or other error
54 // status for any unexpected errors.
55 //
56 // REQUIRED:
57 // This function should only be called on the write thread or if the
58 // mutex is held.
59 // tracker must support point lock.
60 static Status CheckKeysForConflicts(DBImpl* db_impl,
61 const LockTracker& tracker,
62 bool cache_only);
63
64 private:
65 // If `snap_checker` == nullptr, writes are always commited in sequence number
66 // order. All sequence number <= `snap_seq` will not conflict with any
67 // write, and all keys > `snap_seq` of `key` will trigger conflict.
68 // If `snap_checker` != nullptr, writes may not commit in sequence number
69 // order. In this case `min_uncommitted` is a lower bound.
70 // seq < `min_uncommitted`: no conflict
71 // seq > `snap_seq`: applicable to conflict
72 // `min_uncommitted` <= seq <= `snap_seq`: call `snap_checker` to determine.
73 //
74 // If user-defined timestamp is enabled and `enable_udt_validation` is set to
75 // true, a write conflict is detected if an operation for `key` with timestamp
76 // greater than `ts` exists.
77 static Status CheckKey(DBImpl* db_impl, SuperVersion* sv,
78 SequenceNumber earliest_seq, SequenceNumber snap_seq,
79 const std::string& key, const std::string* const ts,
80 bool cache_only, ReadCallback* snap_checker = nullptr,
81 SequenceNumber min_uncommitted = kMaxSequenceNumber,
82 bool enable_udt_validation = true);
83};
