35CompactionFilter::Decision CassandraCompactionFilter::FilterV2(
36 int /*level*/, const Slice& /*key*/, ValueType value_type,
37 const Slice& existing_value, std::string* new_value,
38 std::string* /*skip_until*/) const {
39 bool value_changed = false;
40 RowValue row_value =
41 RowValue::Deserialize(existing_value.data(), existing_value.size());
42 RowValue compacted =
43 options_.purge_ttl_on_expiration
44 ? row_value.RemoveExpiredColumns(&value_changed)
45 : row_value.ConvertExpiredColumnsToTombstones(&value_changed);
46
47 if (value_type == ValueType::kValue) {
48 compacted = compacted.RemoveTombstones(options_.gc_grace_period_in_seconds);
49 }
50
51 if (compacted.Empty()) {
52 return Decision::kRemove;
53 }
54
55 if (value_changed) {
56 compacted.Serialize(new_value);
57 return Decision::kChangeValue;
58 }
59
60 return Decision::kKeep;
61}
