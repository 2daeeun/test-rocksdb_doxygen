39Status PersistRocksDBOptions(const WriteOptions& write_options,
40 const DBOptions& db_opt,
41 const std::vector<std::string>& cf_names,
42 const std::vector<ColumnFamilyOptions>& cf_opts,
43 const std::string& file_name, FileSystem* fs) {
44 ConfigOptions
45 config_options; // Use default for escaped(true) and check (exact)
46 config_options.delimiter = "\n ";
47 // Do not invoke PrepareOptions when we are doing validation.
48 config_options.invoke_prepare_options = false;
49 // If a readahead size was set in the input options, use it
50 if (db_opt.log_readahead_size > 0) {
51 config_options.file_readahead_size = db_opt.log_readahead_size;
52 }
53 return PersistRocksDBOptions(write_options, config_options, db_opt, cf_names,
54 cf_opts, file_name, fs);

57Status PersistRocksDBOptions(const WriteOptions& write_options,
58 const ConfigOptions& config_options_in,
59 const DBOptions& db_opt,
60 const std::vector<std::string>& cf_names,
61 const std::vector<ColumnFamilyOptions>& cf_opts,
62 const std::string& file_name, FileSystem* fs) {
63 ConfigOptions config_options = config_options_in;
64 config_options.delimiter = "\n "; // Override the default to nl
65
66 TEST_SYNC_POINT("PersistRocksDBOptions:start");
67 if (cf_names.size() != cf_opts.size()) {
68 return Status::InvalidArgument(
69 "cf_names.size() and cf_opts.size() must be the same");
70 }
71 std::unique_ptr<FSWritableFile> wf;
72
73 FileOptions file_options;
74 file_options.temperature = db_opt.metadata_write_temperature;
75 Status s = fs->NewWritableFile(file_name, file_options, &wf, nullptr);
76 if (!s.ok()) {
77 return s;
78 }
79 std::unique_ptr<WritableFileWriter> writable;
80 writable.reset(new WritableFileWriter(std::move(wf), file_name, EnvOptions(),
81 nullptr /* statistics */));
82 TEST_SYNC_POINT("PersistRocksDBOptions:create");
83
84 std::string options_file_content;
85
86 IOOptions opts;
87 s = WritableFileWriter::PrepareIOOptions(write_options, opts);
88 if (s.ok()) {
89 s = writable->Append(opts, option_file_header + "[" +
90 opt_section_titles[kOptionSectionVersion] +
91 "]\n"
92 " rocksdb_version=" +
93 std::to_string(ROCKSDB_MAJOR) + "." +
94 std::to_string(ROCKSDB_MINOR) + "." +
95 std::to_string(ROCKSDB_PATCH) + "\n");
96 }
97 if (s.ok()) {
98 s = writable->Append(
99 opts,
100 " options_file_version=" + std::to_string(ROCKSDB_OPTION_FILE_MAJOR) +
101 "." + std::to_string(ROCKSDB_OPTION_FILE_MINOR) + "\n");
102 }
103 if (s.ok()) {
104 s = writable->Append(
105 opts, "\n[" + opt_section_titles[kOptionSectionDBOptions] + "]\n ");
106 }
107
108 if (s.ok()) {
109 s = GetStringFromDBOptions(config_options, db_opt, &options_file_content);
110 }
111 if (s.ok()) {
112 s = writable->Append(opts, options_file_content + "\n");
113 }
114
115 for (size_t i = 0; s.ok() && i < cf_opts.size(); ++i) {
116 // CFOptions section
117 s = writable->Append(
118 opts, "\n[" + opt_section_titles[kOptionSectionCFOptions] + " \"" +
119 EscapeOptionString(cf_names[i]) + "\"]\n ");
120 if (s.ok()) {
121 s = GetStringFromColumnFamilyOptions(config_options, cf_opts[i],
122 &options_file_content);
123 }
124 if (s.ok()) {
125 s = writable->Append(opts, options_file_content + "\n");
126 }
127 // TableOptions section
128 auto* tf = cf_opts[i].table_factory.get();
129 if (tf != nullptr) {
130 if (s.ok()) {
131 s = writable->Append(
132 opts, "[" + opt_section_titles[kOptionSectionTableOptions] +
133 tf->Name() + " \"" + EscapeOptionString(cf_names[i]) +
134 "\"]\n ");
135 }
136 if (s.ok()) {
137 options_file_content.clear();
138 s = tf->GetOptionString(config_options, &options_file_content);
139 }
140 if (s.ok()) {
141 s = writable->Append(opts, options_file_content + "\n");
142 }
143 }
144 }
145 if (s.ok()) {
146 s = writable->Sync(opts, true /* use_fsync */);
147 }
148 if (s.ok()) {
149 s = writable->Close(opts);
150 }
151 TEST_SYNC_POINT("PersistRocksDBOptions:written");
152 if (s.ok()) {
153 return RocksDBOptionsParser::VerifyRocksDBOptionsFromFile(
154 config_options, db_opt, cf_names, cf_opts, file_name, fs);
155 }
156 return s;
157}

176bool RocksDBOptionsParser::IsSection(const std::string& line) {
177 if (line.size() < 2) {
178 return false;
179 }
180 if (line[0] != '[' || line[line.size() - 1] != ']') {
181 return false;
182 }
183 return true;
184}

186Status RocksDBOptionsParser::ParseSection(OptionSection* section,
187 std::string* title,
188 std::string* argument,
189 const std::string& line,
190 const int line_num) {
191 *section = kOptionSectionUnknown;
192 // A section is of the form [<SectionName> "<SectionArg>"], where
193 // "<SectionArg>" is optional.
194 size_t arg_start_pos = line.find('\"');
195 size_t arg_end_pos = line.rfind('\"');
196 // The following if-then check tries to identify whether the input
197 // section has the optional section argument.
198 if (arg_start_pos != std::string::npos && arg_start_pos != arg_end_pos) {
199 *title = TrimAndRemoveComment(line.substr(1, arg_start_pos - 1), true);
200 *argument = UnescapeOptionString(
201 line.substr(arg_start_pos + 1, arg_end_pos - arg_start_pos - 1));
202 } else {
203 *title = TrimAndRemoveComment(line.substr(1, line.size() - 2), true);
204 *argument = "";
205 }
206 for (int i = 0; i < kOptionSectionUnknown; ++i) {
207 if (title->find(opt_section_titles[i]) == 0) {
208 if (i == kOptionSectionVersion || i == kOptionSectionDBOptions ||
209 i == kOptionSectionCFOptions) {
210 if (title->size() == opt_section_titles[i].size()) {
211 // if true, then it indicats equal
212 *section = static_cast<OptionSection>(i);
213 return CheckSection(*section, *argument, line_num);
214 }
215 } else if (i == kOptionSectionTableOptions) {
216 // This type of sections has a sufffix at the end of the
217 // section title
218 if (title->size() > opt_section_titles[i].size()) {
219 *section = static_cast<OptionSection>(i);
220 return CheckSection(*section, *argument, line_num);
221 }
222 }
223 }
224 }
225 return Status::InvalidArgument(std::string("Unknown section ") + line);
226}

235Status RocksDBOptionsParser::ParseStatement(std::string* name,
236 std::string* value,
237 const std::string& line,
238 const int line_num) {
239 size_t eq_pos = line.find('=');
240 if (eq_pos == std::string::npos) {
241 return InvalidArgument(line_num, "A valid statement must have a '='.");
242 }
243
244 *name = TrimAndRemoveComment(line.substr(0, eq_pos), true);
245 *value =
246 TrimAndRemoveComment(line.substr(eq_pos + 1, line.size() - eq_pos - 1));
247 if (name->empty()) {
248 return InvalidArgument(line_num,
249 "A valid statement must have a variable name.");
250 }
251 return Status::OK();
252}

254Status RocksDBOptionsParser::Parse(const std::string& file_name, FileSystem* fs,
255 bool ignore_unknown_options,
256 size_t file_readahead_size) {
257 ConfigOptions
258 config_options; // Use default for escaped(true) and check (exact)
259 config_options.ignore_unknown_options = ignore_unknown_options;
260 if (file_readahead_size > 0) {
261 config_options.file_readahead_size = file_readahead_size;
262 }
263 return Parse(config_options, file_name, fs);
264}

266Status RocksDBOptionsParser::Parse(const ConfigOptions& config_options_in,
267 const std::string& file_name,
268 FileSystem* fs) {
269 Reset();
270 ConfigOptions config_options = config_options_in;
271
272 Status s;
273 bool retry = false;
274 do {
275 std::unique_ptr<FSSequentialFile> seq_file;
276 s = fs->NewSequentialFile(file_name, FileOptions(), &seq_file, nullptr);
277 if (!s.ok()) {
278 return s;
279 }
280
281 LineFileReader lf_reader(
282 std::move(seq_file), file_name, config_options.file_readahead_size,
283 nullptr, std::vector<std::shared_ptr<EventListener>>{}, nullptr, retry);
284
285 OptionSection section = kOptionSectionUnknown;
286 std::string title;
287 std::string argument;
288 std::unordered_map<std::string, std::string> opt_map;
289 std::string line;
290 // we only support single-lined statement.
291 while (
292 lf_reader.ReadLine(&line, Env::IO_TOTAL /* rate_limiter_priority */)) {
293 int line_num = static_cast<int>(lf_reader.GetLineNumber());
294 line = TrimAndRemoveComment(line);
295 if (line.empty()) {
296 continue;
297 }
298 if (IsSection(line)) {
299 s = EndSection(config_options, section, title, argument, opt_map);
300 opt_map.clear();
301 if (!s.ok()) {
302 break;
303 }
304
305 // If the option file is not generated by a higher version, unknown
306 // option should only mean corruption.
307 if (config_options.ignore_unknown_options &&
308 section == kOptionSectionVersion) {
309 using VTuple = std::tuple<int, int, int>;
310 if (VTuple(db_version[0], db_version[1], db_version[2]) <=
311 VTuple(ROCKSDB_MAJOR, ROCKSDB_MINOR, ROCKSDB_PATCH)) {
312 config_options.ignore_unknown_options = false;
313 }
314 }
315
316 s = ParseSection(&section, &title, &argument, line, line_num);
317 if (!s.ok()) {
318 break;
319 }
320 } else {
321 std::string name;
322 std::string value;
323 s = ParseStatement(&name, &value, line, line_num);
324 if (!s.ok()) {
325 break;
326 }
327 opt_map.insert({name, value});
328 }
329 }
330 if (s.ok()) {
331 s = lf_reader.GetStatus();
332 }
333 if (s.ok()) {
334 s = EndSection(config_options, section, title, argument, opt_map);
335 opt_map.clear();
336 }
337 if (s.ok()) {
338 s = ValidityCheck();
339 }
340 if (!s.ok()) {
341 if ((s.IsCorruption() || s.IsInvalidArgument()) && !retry &&
342 CheckFSFeatureSupport(fs,
343 FSSupportedOps::kVerifyAndReconstructRead)) {
344 retry = true;
345 Reset();
346 } else {
347 return s;
348 }
349 } else {
350 return s;
351 }
352 } while (retry);
353
354 return s;
355}

357Status RocksDBOptionsParser::CheckSection(const OptionSection section,
358 const std::string& section_arg,
359 const int line_num) {
360 if (section == kOptionSectionDBOptions) {
361 if (has_db_options_) {
362 return InvalidArgument(
363 line_num,
364 "More than one DBOption section found in the option config file");
365 }
366 has_db_options_ = true;
367 } else if (section == kOptionSectionCFOptions) {
368 bool is_default_cf = (section_arg == kDefaultColumnFamilyName);
369 if (cf_opts_.size() == 0 && !is_default_cf) {
370 return InvalidArgument(
371 line_num,
372 "Default column family must be the first CFOptions section "
373 "in the option config file");
374 } else if (cf_opts_.size() != 0 && is_default_cf) {
375 return InvalidArgument(
376 line_num,
377 "Default column family must be the first CFOptions section "
378 "in the optio/n config file");
379 } else if (GetCFOptions(section_arg) != nullptr) {
380 return InvalidArgument(
381 line_num,
382 "Two identical column families found in option config file");
383 }
384 has_default_cf_options_ |= is_default_cf;
385 } else if (section == kOptionSectionTableOptions) {
386 if (GetCFOptions(section_arg) == nullptr) {
387 return InvalidArgument(
388 line_num, std::string(
389 "Does not find a matched column family name in "
390 "TableOptions section. Column Family Name:") +
391 section_arg);
392 }
393 } else if (section == kOptionSectionVersion) {
394 if (has_version_section_) {
395 return InvalidArgument(
396 line_num,
397 "More than one Version section found in the option config file.");
398 }
399 has_version_section_ = true;
400 }
401 return Status::OK();
402}

404Status RocksDBOptionsParser::ParseVersionNumber(const std::string& ver_name,
405 const std::string& ver_string,
406 const int max_count,
407 int* version) {
408 int version_index = 0;
409 int current_number = 0;
410 int current_digit_count = 0;
411 bool has_dot = false;
412 for (int i = 0; i < max_count; ++i) {
413 version[i] = 0;
414 }
415 constexpr int kBufferSize = 200;
416 char buffer[kBufferSize];
417 for (size_t i = 0; i < ver_string.size(); ++i) {
418 if (ver_string[i] == '.') {
419 if (version_index >= max_count - 1) {
420 snprintf(buffer, sizeof(buffer) - 1,
421 "A valid %s can only contains at most %d dots.",
422 ver_name.c_str(), max_count - 1);
423 return Status::InvalidArgument(buffer);
424 }
425 if (current_digit_count == 0) {
426 snprintf(buffer, sizeof(buffer) - 1,
427 "A valid %s must have at least one digit before each dot.",
428 ver_name.c_str());
429 return Status::InvalidArgument(buffer);
430 }
431 version[version_index++] = current_number;
432 current_number = 0;
433 current_digit_count = 0;
434 has_dot = true;
435 } else if (isdigit(ver_string[i])) {
436 current_number = current_number * 10 + (ver_string[i] - '0');
437 current_digit_count++;
438 } else {
439 snprintf(buffer, sizeof(buffer) - 1,
440 "A valid %s can only contains dots and numbers.",
441 ver_name.c_str());
442 return Status::InvalidArgument(buffer);
443 }
444 }
445 version[version_index] = current_number;
446 if (has_dot && current_digit_count == 0) {
447 snprintf(buffer, sizeof(buffer) - 1,
448 "A valid %s must have at least one digit after each dot.",
449 ver_name.c_str());
450 return Status::InvalidArgument(buffer);
451 }
452 return Status::OK();
453}

455Status RocksDBOptionsParser::EndSection(
456 const ConfigOptions& config_options, const OptionSection section,
457 const std::string& section_title, const std::string& section_arg,
458 const std::unordered_map<std::string, std::string>& opt_map) {
459 Status s;
460 if (section == kOptionSectionDBOptions) {
461 s = GetDBOptionsFromMap(config_options, DBOptions(), opt_map, &db_opt_);
462 if (!s.ok()) {
463 return s;
464 }
465 db_opt_map_ = opt_map;
466 } else if (section == kOptionSectionCFOptions) {
467 // This condition should be ensured earlier in ParseSection
468 // so we make an assertion here.
469 assert(GetCFOptions(section_arg) == nullptr);
470 cf_names_.emplace_back(section_arg);
471 cf_opts_.emplace_back();
472 s = GetColumnFamilyOptionsFromMap(config_options, ColumnFamilyOptions(),
473 opt_map, &cf_opts_.back());
474 if (!s.ok()) {
475 return s;
476 }
477 // keep the parsed string.
478 cf_opt_maps_.emplace_back(opt_map);
479 } else if (section == kOptionSectionTableOptions) {
480 assert(GetCFOptions(section_arg) != nullptr);
481 auto* cf_opt = GetCFOptionsImpl(section_arg);
482 if (cf_opt == nullptr) {
483 return Status::InvalidArgument(
484 "The specified column family must be defined before the "
485 "TableOptions section:",
486 section_arg);
487 }
488 // Ignore error as table factory deserialization is optional
489 cf_opt->table_factory.reset();
490 s = TableFactory::CreateFromString(
491 config_options,
492 section_title.substr(
493 opt_section_titles[kOptionSectionTableOptions].size()),
494 &(cf_opt->table_factory));
495 if (s.ok() && cf_opt->table_factory != nullptr) {
496 s = cf_opt->table_factory->ConfigureFromMap(config_options, opt_map);
497 // Translate any errors (NotFound, NotSupported, to InvalidArgument
498 if (s.ok() || s.IsInvalidArgument()) {
499 return s;
500 } else {
501 return Status::InvalidArgument(s.getState());
502 }
503 } else {
504 // Return OK for not supported table factories as TableFactory
505 // Deserialization is optional.
506 cf_opt->table_factory.reset();
507 return Status::OK();
508 }
509 } else if (section == kOptionSectionVersion) {
510 for (const auto& pair : opt_map) {
511 if (pair.first == "rocksdb_version") {
512 s = ParseVersionNumber(pair.first, pair.second, 3, db_version);
513 if (!s.ok()) {
514 return s;
515 }
516 } else if (pair.first == "options_file_version") {
517 s = ParseVersionNumber(pair.first, pair.second, 2, opt_file_version);
518 if (!s.ok()) {
519 return s;
520 }
521 if (opt_file_version[0] < 1) {
522 return Status::InvalidArgument(
523 "A valid options_file_version must be at least 1.");
524 }
525 }
526 }
527 }
528 return s;
529}

532 if (!has_db_options_) {
533 return Status::Corruption(
534 "A RocksDB Option file must have a single DBOptions section");

537 return Status::Corruption(
538 "A RocksDB Option file must have a single CFOptions:default section");

540
541 return Status::OK();

544std::string RocksDBOptionsParser::TrimAndRemoveComment(const std::string& line,
545 bool trim_only) {
546 size_t start = 0;
547 size_t end = line.size();
548
549 // we only support "#" style comment
550 if (!trim_only) {
551 size_t search_pos = 0;
552 while (search_pos < line.size()) {
553 size_t comment_pos = line.find('#', search_pos);
554 if (comment_pos == std::string::npos) {
555 break;
556 }
557 if (comment_pos == 0 || line[comment_pos - 1] != '\\') {
558 end = comment_pos;
559 break;
560 }
561 search_pos = comment_pos + 1;
562 }
563 }
564
565 while (start < end && isspace(line[start]) != 0) {
566 ++start;
567 }
568
569 // start < end implies end > 0.
570 while (start < end && isspace(line[end - 1]) != 0) {
571 --end;
572 }
573
574 if (start < end) {
575 return line.substr(start, end - start);
576 }
577
578 return "";
579}

581Status RocksDBOptionsParser::VerifyRocksDBOptionsFromFile(
582 const ConfigOptions& config_options_in, const DBOptions& db_opt,
583 const std::vector<std::string>& cf_names,
584 const std::vector<ColumnFamilyOptions>& cf_opts,
585 const std::string& file_name, FileSystem* fs) {
586 RocksDBOptionsParser parser;
587 ConfigOptions config_options = config_options_in;
588 config_options.invoke_prepare_options =
589 false; // No need to do a prepare for verify
590 if (config_options.sanity_level < ConfigOptions::kSanityLevelExactMatch) {
591 // If we are not doing an exact comparison, we should ignore
592 // unsupported options, as they may cause the Parse to fail
593 // (if the ObjectRegistry is not initialized)
594 config_options.ignore_unsupported_options = true;
595 }
596 Status s = parser.Parse(config_options, file_name, fs);
597 if (!s.ok()) {
598 return s;
599 }
600
601 // Verify DBOptions
602 s = VerifyDBOptions(config_options, db_opt, *parser.db_opt(),
603 parser.db_opt_map());
604 if (!s.ok()) {
605 return s;
606 }
607
608 // Verify ColumnFamily Name
609 if (cf_names.size() != parser.cf_names()->size()) {
610 if (config_options.sanity_level >=
611 ConfigOptions::kSanityLevelLooselyCompatible) {
612 return Status::InvalidArgument(
613 "[RocksDBOptionParser Error] The persisted options does not have "
614 "the same number of column family names as the db instance.");
615 } else if (cf_opts.size() > parser.cf_opts()->size()) {
616 return Status::InvalidArgument(
617 "[RocksDBOptionsParser Error]",
618 "The persisted options file has less number of column family "
619 "names than that of the specified one.");
620 }
621 }
622 for (size_t i = 0; i < cf_names.size(); ++i) {
623 if (cf_names[i] != parser.cf_names()->at(i)) {
624 return Status::InvalidArgument(
625 "[RocksDBOptionParser Error] The persisted options and the db"
626 "instance does not have the same name for column family ",
627 std::to_string(i));
628 }
629 }
630
631 // Verify Column Family Options
632 if (cf_opts.size() != parser.cf_opts()->size()) {
633 if (config_options.sanity_level >=
634 ConfigOptions::kSanityLevelLooselyCompatible) {
635 return Status::InvalidArgument(
636 "[RocksDBOptionsParser Error]",
637 "The persisted options does not have the same number of "
638 "column families as the db instance.");
639 } else if (cf_opts.size() > parser.cf_opts()->size()) {
640 return Status::InvalidArgument(
641 "[RocksDBOptionsParser Error]",
642 "The persisted options file has less number of column families "
643 "than that of the specified number.");
644 }
645 }
646 for (size_t i = 0; i < cf_opts.size(); ++i) {
647 s = VerifyCFOptions(config_options, cf_opts[i], parser.cf_opts()->at(i),
648 &(parser.cf_opt_maps()->at(i)));
649 if (!s.ok()) {
650 return s;
651 }
652 s = VerifyTableFactory(config_options, cf_opts[i].table_factory.get(),
653 parser.cf_opts()->at(i).table_factory.get());
654 if (!s.ok()) {
655 return s;
656 }
657 }
658
659 return Status::OK();
660}

662Status RocksDBOptionsParser::VerifyDBOptions(
663 const ConfigOptions& config_options, const DBOptions& base_opt,
664 const DBOptions& file_opt,
665 const std::unordered_map<std::string, std::string>* opt_map) {
666 auto base_config = DBOptionsAsConfigurable(base_opt, opt_map);
667 auto file_config = DBOptionsAsConfigurable(file_opt, opt_map);
668 std::string mismatch;
669 if (!base_config->AreEquivalent(config_options, file_config.get(),
670 &mismatch)) {
671 const size_t kBufferSize = 2048;
672 char buffer[kBufferSize];
673 std::string base_value;
674 std::string file_value;
675 int offset = snprintf(buffer, sizeof(buffer),
676 "[RocksDBOptionsParser]: "
677 "failed the verification on DBOptions::%s -- ",
678 mismatch.c_str());
679 Status s = base_config->GetOption(config_options, mismatch, &base_value);
680 if (s.ok()) {
681 s = file_config->GetOption(config_options, mismatch, &file_value);
682 }
683 assert(offset >= 0);

699Status RocksDBOptionsParser::VerifyCFOptions(
700 const ConfigOptions& config_options, const ColumnFamilyOptions& base_opt,
701 const ColumnFamilyOptions& file_opt,
702 const std::unordered_map<std::string, std::string>* opt_map) {
703 auto base_config = CFOptionsAsConfigurable(base_opt, opt_map);
704 auto file_config = CFOptionsAsConfigurable(file_opt, opt_map);
705 std::string mismatch;
706 if (!base_config->AreEquivalent(config_options, file_config.get(),
707 &mismatch)) {
708 std::string base_value;
709 std::string file_value;
710 // The options do not match
711 const size_t kBufferSize = 2048;
712 char buffer[kBufferSize];
713 Status s = base_config->GetOption(config_options, mismatch, &base_value);
714 if (s.ok()) {
715 s = file_config->GetOption(config_options, mismatch, &file_value);
716 // In file_opt, certain options like MergeOperator may be nullptr due to
717 // factor methods not available. So we use opt_map to get
718 // option value to use in the error message below.
719 if (s.ok() && file_value == kNullptrString && opt_map) {
720 auto const& opt_val_str = (opt_map->find(mismatch));
721 if (opt_val_str != opt_map->end()) {
722 file_value = opt_val_str->second;
723 }
724 }
725 }
726 int offset = snprintf(buffer, sizeof(buffer),
727 "[RocksDBOptionsParser]: "
728 "failed the verification on ColumnFamilyOptions::%s",
729 mismatch.c_str());
730 assert(offset >= 0);
731 assert(static_cast<size_t>(offset) < sizeof(buffer));
732 if (s.ok()) {
733 snprintf(buffer + offset, sizeof(buffer) - static_cast<size_t>(offset),
734 "--- The specified one is %s while the persisted one is %s.\n",
735 base_value.c_str(), file_value.c_str());
736 } else {
737 snprintf(buffer + offset, sizeof(buffer) - static_cast<size_t>(offset),
738 "--- Unable to re-serialize an option: %s.\n",
739 s.ToString().c_str());
740 }
741 return Status::InvalidArgument(Slice(buffer, sizeof(buffer)));
742 } // For each option
743 return Status::OK();
744}
