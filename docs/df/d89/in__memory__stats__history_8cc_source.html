21// Because of garbage collection, the next stats snapshot may or may not be
22// right after the current one. When reading from DBImpl::stats_history_, this
23// call will be protected by DB Mutex so it will not return partial or
24// corrupted results.
25void InMemoryStatsHistoryIterator::Next() {
26 // increment start_time by 1 to avoid infinite loop
27 AdvanceIteratorByTime(GetStatsTime() + 1, end_time_);
28}

37// advance the iterator to the next time between [start_time, end_time)
38// if success, update time_ and stats_map_ with new_time and stats_map
39void InMemoryStatsHistoryIterator::AdvanceIteratorByTime(uint64_t start_time,
40 uint64_t end_time) {
41 // try to find next entry in stats_history_ map
42 if (db_impl_ != nullptr) {
43 valid_ =
44 db_impl_->FindStatsByTime(start_time, end_time, &time_, &stats_map_);
45 } else {
46 valid_ = false;
47 }
48}
