38class Reader {
39 public:
40 // Interface for reporting errors.
41 class Reporter {
42 public:
43 virtual ~Reporter();
44
45 // Some corruption was detected. "size" is the approximate number
46 // of bytes dropped due to the corruption.
47 virtual void Corruption(size_t bytes, const Status& status) = 0;
48
49 virtual void OldLogRecord(size_t /*bytes*/) {}
50 };
51
52 // Create a reader that will return log records from "*file".
53 // "*file" must remain live while this Reader is in use.
54 //
55 // If "reporter" is non-nullptr, it is notified whenever some data is
56 // dropped due to a detected corruption. "*reporter" must remain
57 // live while this Reader is in use.
58 //
59 // If "checksum" is true, verify checksums if available.
60 Reader(std::shared_ptr<Logger> info_log,
61 std::unique_ptr<SequentialFileReader>&& file, Reporter* reporter,
62 bool checksum, uint64_t log_num);
63 // No copying allowed
64 Reader(const Reader&) = delete;
65 void operator=(const Reader&) = delete;
66
67 virtual ~Reader();
68
69 // Read the next record into *record. Returns true if read
70 // successfully, false if we hit end of the input. May use
71 // "*scratch" as temporary storage. The contents filled in *record
72 // will only be valid until the next mutating operation on this
73 // reader or the next mutation to *scratch.
74 // If record_checksum is not nullptr, then this function will calculate the
75 // checksum of the record read and set record_checksum to it. The checksum is
76 // calculated from the original buffers that contain the contents of the
77 // record.
78 virtual bool ReadRecord(Slice* record, std::string* scratch,
79 WALRecoveryMode wal_recovery_mode =
80 WALRecoveryMode::kTolerateCorruptedTailRecords,
81 uint64_t* record_checksum = nullptr);
82
83 // Return the recorded user-defined timestamp size that have been read so
84 // far. This only applies to WAL logs.
85 const UnorderedMap<uint32_t, size_t>& GetRecordedTimestampSize() const {
86 return recorded_cf_to_ts_sz_;
87 }
88
89 // Returns the physical offset of the last record returned by ReadRecord.
90 //
91 // Undefined before the first call to ReadRecord.
92 uint64_t LastRecordOffset();
93
94 // Returns the first physical offset after the last record returned by
95 // ReadRecord, or zero before first call to ReadRecord. This can also be
96 // thought of as the "current" position in processing the file bytes.
97 uint64_t LastRecordEnd();
98
99 // returns true if the reader has encountered an eof condition.
100 bool IsEOF() { return eof_; }
101
102 // returns true if the reader has encountered read error.
103 bool hasReadError() const { return read_error_; }
104
105 // when we know more data has been written to the file. we can use this
106 // function to force the reader to look again in the file.
107 // Also aligns the file position indicator to the start of the next block
108 // by reading the rest of the data from the EOF position to the end of the
109 // block that was partially read.
110 virtual void UnmarkEOF();
111
112 SequentialFileReader* file() { return file_.get(); }
113
114 Reporter* GetReporter() const { return reporter_; }
115
116 uint64_t GetLogNumber() const { return log_number_; }
117
118 size_t GetReadOffset() const {
119 return static_cast<size_t>(end_of_buffer_offset_);
120 }
121
122 bool IsCompressedAndEmptyFile() {
123 return !first_record_read_ && compression_type_record_read_;
124 }
125
126 protected:
127 std::shared_ptr<Logger> info_log_;
128 const std::unique_ptr<SequentialFileReader> file_;
129 Reporter* const reporter_;
130 bool const checksum_;
131 char* const backing_store_;
132
133 // Internal state variables used for reading records
134 Slice buffer_;
135 bool eof_; // Last Read() indicated EOF by returning < kBlockSize
136 bool read_error_; // Error occurred while reading from file
137
138 // Offset of the file position indicator within the last block when an
139 // EOF was detected.
140 size_t eof_offset_;
141
142 // Offset of the last record returned by ReadRecord.
143 uint64_t last_record_offset_;
144 // Offset of the first location past the end of buffer_.
145 uint64_t end_of_buffer_offset_;
146
147 // which log number this is
148 uint64_t const log_number_;
149
150 // Whether this is a recycled log file
151 bool recycled_;
152
153 // Whether the first record has been read or not.
154 bool first_record_read_;
155 // Type of compression used
156 CompressionType compression_type_;
157 // Track whether the compression type record has been read or not.
158 bool compression_type_record_read_;
159 StreamingUncompress* uncompress_;
160 // Reusable uncompressed output buffer
161 std::unique_ptr<char[]> uncompressed_buffer_;
162 // Reusable uncompressed record
163 std::string uncompressed_record_;
164 // Used for stream hashing fragment content in ReadRecord()
165 XXH3_state_t* hash_state_;
166 // Used for stream hashing uncompressed buffer in ReadPhysicalRecord()
167 XXH3_state_t* uncompress_hash_state_;
168
169 // The recorded user-defined timestamp sizes that have been read so far. This
170 // is only for WAL logs.
171 UnorderedMap<uint32_t, size_t> recorded_cf_to_ts_sz_;
172
173 // Extend record types with the following special values
174 enum {
175 kEof = kMaxRecordType + 1,
176 // Returned whenever we find an invalid physical record.
177 // Currently there are three situations in which this happens:
178 // * The record has an invalid CRC (ReadPhysicalRecord reports a drop)
179 // * The record is a 0-length record (No drop is reported)
180 kBadRecord = kMaxRecordType + 2,
181 // Returned when we fail to read a valid header.
182 kBadHeader = kMaxRecordType + 3,
183 // Returned when we read an old record from a previous user of the log.
184 kOldRecord = kMaxRecordType + 4,
185 // Returned when we get a bad record length
186 kBadRecordLen = kMaxRecordType + 5,
187 // Returned when we get a bad record checksum
188 kBadRecordChecksum = kMaxRecordType + 6,
189 };
190
191 // Return type, or one of the preceding special values
192 // If WAL compressioned is enabled, fragment_checksum is the checksum of the
193 // fragment computed from the orginal buffer containinng uncompressed
194 // fragment.
195 unsigned int ReadPhysicalRecord(Slice* result, size_t* drop_size,
196 uint64_t* fragment_checksum = nullptr);
197
198 // Read some more
199 bool ReadMore(size_t* drop_size, int* error);
200
201 void UnmarkEOFInternal();
202
203 // Reports dropped bytes to the reporter.
204 // buffer_ must be updated to remove the dropped bytes prior to invocation.
205 void ReportCorruption(size_t bytes, const char* reason);
206 void ReportDrop(size_t bytes, const Status& reason);
207 void ReportOldLogRecord(size_t bytes);
208
209 void InitCompression(const CompressionTypeRecord& compression_record);
210
211 Status UpdateRecordedTimestampSize(
212 const std::vector<std::pair<uint32_t, size_t>>& cf_to_ts_sz);
213};
