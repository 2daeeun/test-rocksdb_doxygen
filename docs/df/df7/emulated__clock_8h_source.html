17namespace ROCKSDB_NAMESPACE {
18// A SystemClock that can "mock" sleep and counts its operations.
19class EmulatedSystemClock : public SystemClockWrapper {
20 private:
21 // Something to return when mocking current time
22 const int64_t maybe_starting_time_;
23 std::atomic<int> sleep_counter_{0};
24 std::atomic<int> cpu_counter_{0};
25 std::atomic<int64_t> addon_microseconds_{0};
26 // Do not modify in the env of a running DB (could cause deadlock)
27 std::atomic<bool> time_elapse_only_sleep_;
28 bool no_slowdown_;
29
30 public:
31 explicit EmulatedSystemClock(const std::shared_ptr<SystemClock>& base,
32 bool time_elapse_only_sleep = false);
33
34 static const char* kClassName() { return "TimeEmulatedSystemClock"; }
35 const char* Name() const override { return kClassName(); }
36
37 void SleepForMicroseconds(int micros) override {
38 sleep_counter_++;
39 if (no_slowdown_ || time_elapse_only_sleep_) {
40 addon_microseconds_.fetch_add(micros);
41 }
42 if (!no_slowdown_) {
43 SystemClockWrapper::SleepForMicroseconds(micros);
44 }
45 }
46
47 void MockSleepForMicroseconds(int64_t micros) {
48 sleep_counter_++;
49 assert(no_slowdown_);
50 addon_microseconds_.fetch_add(micros);
51 }
52
53 void MockSleepForSeconds(int64_t seconds) {
54 sleep_counter_++;
55 assert(no_slowdown_);
56 addon_microseconds_.fetch_add(seconds * 1000000);
57 }
58
59 void SetTimeElapseOnlySleep(bool enabled) {
60 // We cannot set these before destroying the last DB because they might
61 // cause a deadlock or similar without the appropriate options set in
62 // the DB.
63 time_elapse_only_sleep_ = enabled;
64 no_slowdown_ = enabled;
65 }
66
67 bool IsTimeElapseOnlySleep() const { return time_elapse_only_sleep_.load(); }
68 void SetMockSleep(bool enabled = true) { no_slowdown_ = enabled; }
69 bool IsMockSleepEnabled() const { return no_slowdown_; }
70
71 int GetSleepCounter() const { return sleep_counter_.load(); }
72
73 Status GetCurrentTime(int64_t* unix_time) override {
74 Status s;
75 if (time_elapse_only_sleep_) {
76 *unix_time = maybe_starting_time_;
77 } else {
78 s = SystemClockWrapper::GetCurrentTime(unix_time);
79 }
80 if (s.ok()) {
81 // mock microseconds elapsed to seconds of time
82 *unix_time += addon_microseconds_.load() / 1000000;
83 }
84 return s;
85 }
86
87 uint64_t CPUNanos() override {
88 cpu_counter_++;
89 return SystemClockWrapper::CPUNanos();
90 }
91
92 uint64_t CPUMicros() override {
93 cpu_counter_++;
94 return SystemClockWrapper::CPUMicros();
95 }
96
97 uint64_t NowNanos() override {
98 return (time_elapse_only_sleep_ ? 0 : SystemClockWrapper::NowNanos()) +
99 addon_microseconds_.load() * 1000;
100 }
101
102 uint64_t NowMicros() override {
103 return (time_elapse_only_sleep_ ? 0 : SystemClockWrapper::NowMicros()) +
104 addon_microseconds_.load();
105 }
106
107 int GetCpuCounter() const { return cpu_counter_.load(); }
108
109 void ResetCounters() {
110 cpu_counter_.store(0);
111 sleep_counter_.store(0);
112 }
113};
114} // namespace ROCKSDB_NAMESPACE
