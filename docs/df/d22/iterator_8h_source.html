9// An iterator yields a sequence of key/value pairs from a source.
10// The following class defines the interface. Multiple implementations
11// are provided by this library. In particular, iterators are provided
12// to access the contents of a Table or a DB.
13//
14// Multiple threads can invoke const methods on an Iterator without
15// external synchronization, but if any of the threads may call a
16// non-const method, all threads accessing the same Iterator must use
17// external synchronization.

37 // Return the value for the current entry. If the entry is a plain key-value,
38 // return the value as-is; if it is a wide-column entity, return the value of
39 // the default anonymous column (see kDefaultWideColumnName) if any, or an
40 // empty value otherwise. The underlying storage for the returned slice is
41 // valid only until the next modification of the iterator (i.e. the next
42 // SeekToFirst/SeekToLast/Seek/SeekForPrev/Next/Prev operation).
43 // REQUIRES: Valid()
44 virtual Slice value() const = 0;
45
46 // Return the wide columns for the current entry. If the entry is a
47 // wide-column entity, return it as-is; if it is a plain key-value, return it
48 // as an entity with a single anonymous column (see kDefaultWideColumnName)
49 // which contains the value. The underlying storage for the returned
50 // structure is valid only until the next modification of the iterator (i.e.
51 // the next SeekToFirst/SeekToLast/Seek/SeekForPrev/Next/Prev operation).
52 // REQUIRES: Valid()
53 virtual const WideColumns& columns() const {
54 assert(false);
55 return kNoWideColumns;
56 }
57
58 // Property "rocksdb.iterator.is-key-pinned":
59 // If returning "1", this means that the Slice returned by key() is valid
60 // as long as the iterator is not deleted.
61 // It is guaranteed to always return "1" if
62 // - Iterator created with ReadOptions::pin_data = true
63 // - DB tables were created with
64 // BlockBasedTableOptions::use_delta_encoding = false.
65 // Property "rocksdb.iterator.is-value-pinned":
66 // If returning "1", this means that the Slice returned by value() is valid
67 // as long as the iterator is not deleted.
68 // It is guaranteed to always return "1" if
69 // - Iterator created with ReadOptions::pin_data = true
70 // - The value is found in a `kTypeValue` record
71 // Property "rocksdb.iterator.super-version-number":
72 // LSM version used by the iterator. The same format as DB Property
73 // kCurrentSuperVersionNumber. See its comment for more information.
74 // Property "rocksdb.iterator.internal-key":
75 // Get the user-key portion of the internal key at which the iteration
76 // stopped.
77 // Property "rocksdb.iterator.write-time":
78 // Get the unix time of the best estimate of the write time of the entry.
79 // Returned as 64-bit raw value (8 bytes). It can be converted to uint64_t
80 // with util method `DecodeU64Ts`. The accuracy of the write time depends on
81 // settings like preserve_internal_time_seconds. The actual write time of
82 // the entry should be the same or newer than the returned write time. So
83 // this property can be interpreted as the possible oldest write time for
84 // the entry.
85 // If the seqno to time mapping recording is not enabled,
86 // std::numeric_limits<uint64_t>::max() will be returned to indicate the
87 // write time is unknown. For data entry whose sequence number has
88 // been zeroed out (possible when they reach the last level), 0 is returned
89 // no matter whether the seqno to time recording feature is enabled or not.
90 virtual Status GetProperty(std::string prop_name, std::string* prop);
91
92 virtual Slice timestamp() const {
93 assert(false);
94 return Slice();
95 }
96};
