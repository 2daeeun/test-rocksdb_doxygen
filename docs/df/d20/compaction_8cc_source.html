25int sstableKeyCompare(const Comparator* uc, const Slice& a, const Slice& b) {
26 auto c = uc->CompareWithoutTimestamp(ExtractUserKey(a), ExtractUserKey(b));
27 if (c != 0) {
28 return c;
29 }
30 auto a_footer = ExtractInternalKeyFooter(a);
31 auto b_footer = ExtractInternalKeyFooter(b);
32 if (a_footer == kRangeTombstoneSentinel) {
33 if (b_footer != kRangeTombstoneSentinel) {
34 return -1;
35 }
36 } else if (b_footer == kRangeTombstoneSentinel) {
37 return 1;
38 }
39 return 0;
40}

77void Compaction::GetBoundaryKeys(
78 VersionStorageInfo* vstorage,
79 const std::vector<CompactionInputFiles>& inputs, Slice* smallest_user_key,
80 Slice* largest_user_key, int exclude_level) {
81 bool initialized = false;
82 const Comparator* ucmp = vstorage->InternalComparator()->user_comparator();
83 for (size_t i = 0; i < inputs.size(); ++i) {
84 if (inputs[i].files.empty() || inputs[i].level == exclude_level) {
85 continue;
86 }
87 if (inputs[i].level == 0) {
88 // we need to consider all files on level 0
89 for (const auto* f : inputs[i].files) {
90 const Slice& start_user_key = f->smallest.user_key();
91 if (!initialized ||
92 ucmp->Compare(start_user_key, *smallest_user_key) < 0) {
93 *smallest_user_key = start_user_key;
94 }
95 const Slice& end_user_key = f->largest.user_key();
96 if (!initialized ||
97 ucmp->Compare(end_user_key, *largest_user_key) > 0) {
98 *largest_user_key = end_user_key;
99 }
100 initialized = true;
101 }
102 } else {
103 // we only need to consider the first and last file
104 const Slice& start_user_key = inputs[i].files[0]->smallest.user_key();
105 if (!initialized ||
106 ucmp->Compare(start_user_key, *smallest_user_key) < 0) {
107 *smallest_user_key = start_user_key;
108 }
109 const Slice& end_user_key = inputs[i].files.back()->largest.user_key();
110 if (!initialized || ucmp->Compare(end_user_key, *largest_user_key) > 0) {
111 *largest_user_key = end_user_key;
112 }
113 initialized = true;
114 }
115 }
116}

118void Compaction::GetBoundaryInternalKeys(
119 VersionStorageInfo* vstorage,
120 const std::vector<CompactionInputFiles>& inputs, InternalKey* smallest_key,
121 InternalKey* largest_key, int exclude_level) {
122 bool initialized = false;
123 const InternalKeyComparator* icmp = vstorage->InternalComparator();
124 for (size_t i = 0; i < inputs.size(); ++i) {
125 if (inputs[i].files.empty() || inputs[i].level == exclude_level) {
126 continue;
127 }
128 if (inputs[i].level == 0) {
129 // we need to consider all files on level 0
130 for (const auto* f : inputs[i].files) {
131 if (!initialized || icmp->Compare(f->smallest, *smallest_key) < 0) {
132 *smallest_key = f->smallest;
133 }
134 if (!initialized || icmp->Compare(f->largest, *largest_key) > 0) {
135 *largest_key = f->largest;
136 }
137 initialized = true;
138 }
139 } else {
140 // we only need to consider the first and last file
141 if (!initialized ||
142 icmp->Compare(inputs[i].files[0]->smallest, *smallest_key) < 0) {
143 *smallest_key = inputs[i].files[0]->smallest;
144 }
145 if (!initialized ||
146 icmp->Compare(inputs[i].files.back()->largest, *largest_key) > 0) {
147 *largest_key = inputs[i].files.back()->largest;
148 }
149 initialized = true;
150 }
151 }

154std::vector<CompactionInputFiles> Compaction::PopulateWithAtomicBoundaries(
155 VersionStorageInfo* vstorage, std::vector<CompactionInputFiles> inputs) {
156 const Comparator* ucmp = vstorage->InternalComparator()->user_comparator();
157 for (size_t i = 0; i < inputs.size(); i++) {
158 if (inputs[i].level == 0 || inputs[i].files.empty()) {
159 continue;
160 }
161 inputs[i].atomic_compaction_unit_boundaries.reserve(inputs[i].files.size());
162 AtomicCompactionUnitBoundary cur_boundary;
163 size_t first_atomic_idx = 0;
164 auto add_unit_boundary = [&](size_t to) {
165 if (first_atomic_idx == to) {
166 return;
167 }
168 for (size_t k = first_atomic_idx; k < to; k++) {
169 inputs[i].atomic_compaction_unit_boundaries.push_back(cur_boundary);
170 }
171 first_atomic_idx = to;
172 };
173 for (size_t j = 0; j < inputs[i].files.size(); j++) {
174 const auto* f = inputs[i].files[j];
175 if (j == 0) {
176 // First file in a level.
177 cur_boundary.smallest = &f->smallest;
178 cur_boundary.largest = &f->largest;
179 } else if (sstableKeyCompare(ucmp, *cur_boundary.largest, f->smallest) ==
180 0) {
181 // SSTs overlap but the end key of the previous file was not
182 // artificially extended by a range tombstone. Extend the current
183 // boundary.
184 cur_boundary.largest = &f->largest;
185 } else {
186 // Atomic compaction unit has ended.
187 add_unit_boundary(j);
188 cur_boundary.smallest = &f->smallest;
189 cur_boundary.largest = &f->largest;
190 }
191 }
192 add_unit_boundary(inputs[i].files.size());
193 assert(inputs[i].files.size() ==
194 inputs[i].atomic_compaction_unit_boundaries.size());
195 }
196 return inputs;
197}

199// helper function to determine if compaction is creating files at the
200// bottommost level
201bool Compaction::IsBottommostLevel(
202 int output_level, VersionStorageInfo* vstorage,
203 const std::vector<CompactionInputFiles>& inputs) {
204 int output_l0_idx;
205 if (output_level == 0) {
206 output_l0_idx = 0;
207 for (const auto* file : vstorage->LevelFiles(0)) {
208 if (inputs[0].files.back() == file) {
209 break;
210 }
211 ++output_l0_idx;
212 }
213 assert(static_cast<size_t>(output_l0_idx) < vstorage->LevelFiles(0).size());
214 } else {
215 output_l0_idx = -1;
216 }
217 Slice smallest_key, largest_key;
218 GetBoundaryKeys(vstorage, inputs, &smallest_key, &largest_key);
219 return !vstorage->RangeMightExistAfterSortedRun(smallest_key, largest_key,
220 output_level, output_l0_idx);
221}

231bool Compaction::IsFullCompaction(
232 VersionStorageInfo* vstorage,
233 const std::vector<CompactionInputFiles>& inputs) {
234 size_t num_files_in_compaction = 0;
235 size_t total_num_files = 0;
236 for (int l = 0; l < vstorage->num_levels(); l++) {
237 total_num_files += vstorage->NumLevelFiles(l);
238 }
239 for (size_t i = 0; i < inputs.size(); i++) {
240 num_files_in_compaction += inputs[i].size();
241 }
242 return num_files_in_compaction == total_num_files;
243}

411void Compaction::PopulatePenultimateLevelOutputRange() {
412 if (!SupportsPerKeyPlacement()) {
413 return;
414 }
415
416 // exclude the last level, the range of all input levels is the safe range
417 // of keys that can be moved up.
418 int exclude_level = number_levels_ - 1;
419 penultimate_output_range_type_ = PenultimateOutputRangeType::kNonLastRange;
420
421 // For universal compaction, the penultimate_output_range could be extended if
422 // all penultimate level files are included in the compaction (which includes
423 // the case that the penultimate level is empty).
424 if (immutable_options_.compaction_style == kCompactionStyleUniversal) {
425 exclude_level = kInvalidLevel;
426 penultimate_output_range_type_ = PenultimateOutputRangeType::kFullRange;
427 std::set<uint64_t> penultimate_inputs;
428 for (const auto& input_lvl : inputs_) {
429 if (input_lvl.level == penultimate_level_) {
430 for (const auto& file : input_lvl.files) {
431 penultimate_inputs.emplace(file->fd.GetNumber());
432 }
433 }
434 }
435 auto penultimate_files = input_vstorage_->LevelFiles(penultimate_level_);
436 for (const auto& file : penultimate_files) {
437 if (penultimate_inputs.find(file->fd.GetNumber()) ==
438 penultimate_inputs.end()) {
439 exclude_level = number_levels_ - 1;
440 penultimate_output_range_type_ =
441 PenultimateOutputRangeType::kNonLastRange;
442 break;
443 }
444 }
445 }
446
447 // FIXME: should make use of `penultimate_output_range_type_`.
448 // FIXME: when last level's input range does not overlap with
449 // penultimate level, and penultimate level input is empty,
450 // this call will not set penultimate_level_smallest_ or
451 // penultimate_level_largest_. No keys will be compacted up.
452 GetBoundaryInternalKeys(input_vstorage_, inputs_,
453 &penultimate_level_smallest_,
454 &penultimate_level_largest_, exclude_level);
455}

472// smallest_key and largest_key include timestamps if user-defined timestamp is
473// enabled.
474bool Compaction::OverlapPenultimateLevelOutputRange(
475 const Slice& smallest_key, const Slice& largest_key) const {
476 if (!SupportsPerKeyPlacement()) {
477 return false;
478 }
479
480 // See FIXME in Compaction::PopulatePenultimateLevelOutputRange().
481 // We do not compact any key up in this case.
482 if (penultimate_level_smallest_.size() == 0 ||
483 penultimate_level_largest_.size() == 0) {
484 return false;
485 }
486
487 const Comparator* ucmp =
488 input_vstorage_->InternalComparator()->user_comparator();
489
490 return ucmp->CompareWithoutTimestamp(
491 smallest_key, penultimate_level_largest_.user_key()) <= 0 &&
492 ucmp->CompareWithoutTimestamp(
493 largest_key, penultimate_level_smallest_.user_key()) >= 0;
494}

496// key includes timestamp if user-defined timestamp is enabled.
497bool Compaction::WithinPenultimateLevelOutputRange(
498 const ParsedInternalKey& ikey) const {
499 if (!SupportsPerKeyPlacement()) {
500 return false;
501 }
502
503 if (penultimate_level_smallest_.size() == 0 ||
504 penultimate_level_largest_.size() == 0) {
505 return false;
506 }
507
508 const InternalKeyComparator* icmp = input_vstorage_->InternalComparator();
509
510 // op_type of a key can change during compaction, e.g. Merge -> Put.
511 return icmp->CompareKeySeq(ikey, penultimate_level_smallest_.Encode()) >= 0 &&
512 icmp->CompareKeySeq(ikey, penultimate_level_largest_.Encode()) <= 0;
513}

528bool Compaction::IsTrivialMove() const {
529 // Avoid a move if there is lots of overlapping grandparent data.
530 // Otherwise, the move could create a parent file that will require
531 // a very expensive merge later on.
532 // If start_level_== output_level_, the purpose is to force compaction
533 // filter to be applied to that level, and thus cannot be a trivial move.
534
535 // Check if start level have files with overlapping ranges
536 if (start_level_ == 0 && input_vstorage_->level0_non_overlapping() == false &&
537 l0_files_might_overlap_) {
538 // We cannot move files from L0 to L1 if the L0 files in the LSM-tree are
539 // overlapping, unless we are sure that files picked in L0 don't overlap.
540 return false;
541 }
542
543 if (is_manual_compaction_ &&
544 (immutable_options_.compaction_filter != nullptr ||
545 immutable_options_.compaction_filter_factory != nullptr)) {
546 // This is a manual compaction and we have a compaction filter that should
547 // be executed, we cannot do a trivial move
548 return false;
549 }
550
551 if (start_level_ == output_level_) {
552 // It doesn't make sense if compaction picker picks files just to trivial
553 // move to the same level.
554 return false;
555 }
556
557 if (compaction_reason_ == CompactionReason::kChangeTemperature) {
558 // Changing temperature usually requires rewriting the file.
559 return false;
560 }
561
562 // Used in universal compaction, where trivial move can be done if the
563 // input files are non overlapping
564 if ((mutable_cf_options_.compaction_options_universal.allow_trivial_move) &&
565 (output_level_ != 0) &&
566 (cfd_->ioptions()->compaction_style == kCompactionStyleUniversal)) {
567 return is_trivial_move_;
568 }
569
570 if (!(start_level_ != output_level_ && num_input_levels() == 1 &&
571 input(0, 0)->fd.GetPathId() == output_path_id() &&
572 InputCompressionMatchesOutput())) {
573 return false;
574 }
575
576 // assert inputs_.size() == 1
577
578 if (output_level_ + 1 < number_levels_) {
579 std::unique_ptr<SstPartitioner> partitioner = CreateSstPartitioner();
580 for (const auto& file : inputs_.front().files) {
581 std::vector<FileMetaData*> file_grand_parents;
582 input_vstorage_->GetOverlappingInputs(output_level_ + 1, &file->smallest,
583 &file->largest,
584 &file_grand_parents);
585 const auto compaction_size =
586 file->fd.GetFileSize() + TotalFileSize(file_grand_parents);
587 if (compaction_size > max_compaction_bytes_) {
588 return false;
589 }
590
591 if (partitioner.get() != nullptr) {
592 if (!partitioner->CanDoTrivialMove(file->smallest.user_key(),
593 file->largest.user_key())) {
594 return false;
595 }
596 }
597 }
598 }
599
600 // PerKeyPlacement compaction should never be trivial move.
601 if (SupportsPerKeyPlacement()) {
602 return false;
603 }
604
605 return true;
606}

616bool Compaction::KeyNotExistsBeyondOutputLevel(
617 const Slice& user_key, std::vector<size_t>* level_ptrs) const {
618 assert(input_version_ != nullptr);
619 assert(level_ptrs != nullptr);
620 assert(level_ptrs->size() == static_cast<size_t>(number_levels_));
621 if (bottommost_level_) {
622 return true;
623 } else if (output_level_ != 0 &&
624 cfd_->ioptions()->compaction_style == kCompactionStyleLevel) {
625 // Maybe use binary search to find right entry instead of linear search?
626 const Comparator* user_cmp = cfd_->user_comparator();
627 for (int lvl = output_level_ + 1; lvl < number_levels_; lvl++) {
628 const std::vector<FileMetaData*>& files =
629 input_vstorage_->LevelFiles(lvl);
630 for (; level_ptrs->at(lvl) < files.size(); level_ptrs->at(lvl)++) {
631 auto* f = files[level_ptrs->at(lvl)];
632 if (user_cmp->Compare(user_key, f->largest.user_key()) <= 0) {
633 // We've advanced far enough
634 // In the presence of user-defined timestamp, we may need to handle
635 // the case in which f->smallest.user_key() (including ts) has the
636 // same user key, but the ts part is smaller. If so,
637 // Compare(user_key, f->smallest.user_key()) returns -1.
638 // That's why we need CompareWithoutTimestamp().
639 if (user_cmp->CompareWithoutTimestamp(user_key,
640 f->smallest.user_key()) >= 0) {
641 // Key falls in this file's range, so it may
642 // exist beyond output level
643 return false;
644 }
645 break;
646 }
647 }
648 }
649 return true;
650 }
651 return false;
652}

654bool Compaction::KeyRangeNotExistsBeyondOutputLevel(
655 const Slice& begin_key, const Slice& end_key,
656 std::vector<size_t>* level_ptrs) const {
657 assert(input_version_ != nullptr);
658 assert(level_ptrs != nullptr);
659 assert(level_ptrs->size() == static_cast<size_t>(number_levels_));
660 assert(cfd_->user_comparator()->CompareWithoutTimestamp(begin_key, end_key) <
661 0);
662 if (bottommost_level_) {
663 return true /* does not overlap */;
664 } else if (output_level_ != 0 &&
665 cfd_->ioptions()->compaction_style == kCompactionStyleLevel) {
666 const Comparator* user_cmp = cfd_->user_comparator();
667 for (int lvl = output_level_ + 1; lvl < number_levels_; lvl++) {
668 const std::vector<FileMetaData*>& files =
669 input_vstorage_->LevelFiles(lvl);
670 for (; level_ptrs->at(lvl) < files.size(); level_ptrs->at(lvl)++) {
671 auto* f = files[level_ptrs->at(lvl)];
672 // Advance until the first file with begin_key <= f->largest.user_key()
673 if (user_cmp->CompareWithoutTimestamp(begin_key,
674 f->largest.user_key()) > 0) {
675 continue;
676 }
677 // We know that the previous file prev_f, if exists, has
678 // prev_f->largest.user_key() < begin_key.
679 if (user_cmp->CompareWithoutTimestamp(end_key,
680 f->smallest.user_key()) <= 0) {
681 // not overlapping with this level
682 break;
683 } else {
684 // We have:
685 // - begin_key < end_key,
686 // - begin_key <= f->largest.user_key(), and
687 // - end_key > f->smallest.user_key()
688 return false /* overlap */;
689 }
690 }
691 }
692 return true /* does not overlap */;
693 }
694 return false /* overlaps */;
695};

992int Compaction::EvaluatePenultimateLevel(
993 const VersionStorageInfo* vstorage,
994 const MutableCFOptions& mutable_cf_options,
995 const ImmutableOptions& immutable_options, const int start_level,
996 const int output_level) {
997 // TODO: currently per_key_placement feature only support level and universal
998 // compaction
999 if (immutable_options.compaction_style != kCompactionStyleLevel &&
1000 immutable_options.compaction_style != kCompactionStyleUniversal) {
1001 return kInvalidLevel;
1002 }
1003 if (output_level != immutable_options.num_levels - 1) {
1004 return kInvalidLevel;
1005 }
1006
1007 int penultimate_level = output_level - 1;
1008 assert(penultimate_level < immutable_options.num_levels);
1009 if (penultimate_level <= 0) {
1010 return kInvalidLevel;
1011 }
1012
1013 // If the penultimate level is not within input level -> output level range
1014 // check if the penultimate output level is empty, if it's empty, it could
1015 // also be locked for the penultimate output.
1016 // TODO: ideally, it only needs to check if there's a file within the
1017 // compaction output key range. For simplicity, it just check if there's any
1018 // file on the penultimate level.
1019 if (start_level == immutable_options.num_levels - 1 &&
1020 (immutable_options.compaction_style != kCompactionStyleUniversal ||
1021 !vstorage->LevelFiles(penultimate_level).empty())) {
1022 return kInvalidLevel;
1023 }
1024
1025 bool supports_per_key_placement =
1026 mutable_cf_options.preclude_last_level_data_seconds > 0;
1027
1028 // it could be overridden by unittest
1029 TEST_SYNC_POINT_CALLBACK("Compaction::SupportsPerKeyPlacement:Enabled",
1030 &supports_per_key_placement);
1031 if (!supports_per_key_placement) {
1032 return kInvalidLevel;
1033 }
1034
1035 return penultimate_level;
1036}

1038void Compaction::FilterInputsForCompactionIterator() {
1039 assert(earliest_snapshot_.has_value());
1040 // cfd_ is not populated at Compaction construction time, get it from
1041 // VersionStorageInfo instead.
1042 assert(input_vstorage_);
1043 const auto* ucmp = input_vstorage_->user_comparator();
1044 assert(ucmp);
1045 // Simply comparing file boundaries when user-defined timestamp is defined
1046 // is not as safe because we need to also compare timestamp to know for
1047 // sure. Although entries with higher timestamp is also supposed to have
1048 // higher sequence number for the same user key (without timestamp).
1049 assert(ucmp->timestamp_size() == 0);
1050 size_t num_input_levels = inputs_.size();
1051 // TODO(yuzhangyu): filtering of older L0 file by new L0 file is not
1052 // supported yet.
1053 FileMetaData* rangedel_candidate = inputs_[0].level == 0
1054 ? inputs_[0].files.back()
1055 : inputs_[0].files.front();
1056 assert(rangedel_candidate);
1057 if (!rangedel_candidate->FileIsStandAloneRangeTombstone() ||
1058 !DataIsDefinitelyInSnapshot(rangedel_candidate->fd.smallest_seqno,
1059 earliest_snapshot_.value(),
1060 snapshot_checker_)) {
1061 for (size_t level = 0; level < num_input_levels; level++) {
1062 DoGenerateLevelFilesBrief(&input_levels_[level], inputs_[level].files,
1063 &arena_);
1064 }
1065 return;
1066 }
1067
1068 Slice rangedel_start_ukey = rangedel_candidate->smallest.user_key();
1069 Slice rangedel_end_ukey = rangedel_candidate->largest.user_key();
1070 SequenceNumber rangedel_seqno = rangedel_candidate->fd.smallest_seqno;
1071
1072 std::vector<std::vector<FileMetaData*>> non_start_level_input_files;
1073 non_start_level_input_files.reserve(num_input_levels - 1);
1074 non_start_level_input_files_filtered_.reserve(num_input_levels - 1);
1075 for (size_t level = 1; level < num_input_levels; level++) {
1076 non_start_level_input_files.emplace_back();
1077 non_start_level_input_files_filtered_.emplace_back();
1078 for (FileMetaData* file : inputs_[level].files) {
1079 non_start_level_input_files_filtered_.back().push_back(false);
1080 // When range data and point data has the same sequence number, point
1081 // data wins. Range deletion end key is exclusive, so check it's bigger
1082 // than file right boundary user key.
1083 if (rangedel_seqno > file->fd.largest_seqno &&
1084 ucmp->CompareWithoutTimestamp(rangedel_start_ukey,
1085 file->smallest.user_key()) <= 0 &&
1086 ucmp->CompareWithoutTimestamp(rangedel_end_ukey,
1087 file->largest.user_key()) > 0) {
1088 non_start_level_input_files_filtered_.back().back() = true;
1089 filtered_input_levels_[level].push_back(file);
1090 } else {
1091 non_start_level_input_files.back().push_back(file);
1092 }
1093 }
1094 }
1095
1096 DoGenerateLevelFilesBrief(&input_levels_[0], inputs_[0].files, &arena_);
1097 assert(non_start_level_input_files.size() == num_input_levels - 1);
1098 for (size_t level = 1; level < num_input_levels; level++) {
1099 DoGenerateLevelFilesBrief(&input_levels_[level],
1100 non_start_level_input_files[level - 1], &arena_);
1101 }
1102}
