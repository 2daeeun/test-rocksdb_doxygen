23Random* Random::GetTLSInstance() {
24 STORAGE_DECL Random* tls_instance;
25 STORAGE_DECL aligned_storage<Random>::type tls_instance_bytes;
26
27 auto rv = tls_instance;
28 if (UNLIKELY(rv == nullptr)) {
29 size_t seed = std::hash<std::thread::id>()(std::this_thread::get_id());
30 rv = new (&tls_instance_bytes) Random((uint32_t)seed);
31 tls_instance = rv;
32 }
33 return rv;
34}
