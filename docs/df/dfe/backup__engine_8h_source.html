36 // Best to set this to dbname_ + "/backups"
37 // Required

39
40 // Backup Env object. It will be used for backup file I/O. If it's
41 // nullptr, backups will be written out using DBs Env. If it's
42 // non-nullptr, backup's I/O will be performed using this object.
43 // Default: nullptr
44 Env* backup_env;
45
46 // share_table_files supports table and blob files.
47 //
48 // If share_table_files == true, the backup directory will share table and
49 // blob files among backups, to save space among backups of the same DB and to
50 // enable incremental backups by only copying new files.
51 // If share_table_files == false, each backup will be on its own and will not
52 // share any data with other backups.
53 //
54 // default: true
55 bool share_table_files;
56
57 // Backup info and error messages will be written to info_log
58 // if non-nullptr.
59 // Default: nullptr
60 Logger* info_log;
61
62 // If sync == true, we can guarantee you'll get consistent backup and
63 // restore even on a machine crash/reboot. Backup and restore processes are
64 // slower with sync enabled. If sync == false, we can only guarantee that
65 // other previously synced backups and restores are not modified while
66 // creating a new one.
67 // Default: true
68 bool sync;
69
70 // If true, it will delete whatever backups there are already
71 // Default: false
72 bool destroy_old_data;
73
74 // If false, we won't backup log files. This option can be useful for backing
75 // up in-memory databases where log file are persisted, but table files are in
76 // memory.
77 // Default: true
78 bool backup_log_files;
79
80 // Max bytes that can be transferred in a second during backup.
81 // If 0, go as fast as you can
82 // This limit only applies to writes. To also limit reads,
83 // a rate limiter able to also limit reads (e.g, its mode = kAllIo)
84 // have to be passed in through the option "backup_rate_limiter"
85 // Default: 0
86 uint64_t backup_rate_limit;
87
88 // Backup rate limiter. Used to control transfer speed for backup. If this is
89 // not null, backup_rate_limit is ignored.
90 // Default: nullptr
91 std::shared_ptr<RateLimiter> backup_rate_limiter{nullptr};
92
93 // Max bytes that can be transferred in a second during restore.
94 // If 0, go as fast as you can
95 // This limit only applies to writes. To also limit reads,
96 // a rate limiter able to also limit reads (e.g, its mode = kAllIo)
97 // have to be passed in through the option "restore_rate_limiter"
98 // Default: 0
99 uint64_t restore_rate_limit;
100
101 // Restore rate limiter. Used to control transfer speed during restore. If
102 // this is not null, restore_rate_limit is ignored.
103 // Default: nullptr
104 std::shared_ptr<RateLimiter> restore_rate_limiter{nullptr};
105
106 // share_files_with_checksum supports table and blob files.
107 //
108 // Only used if share_table_files is set to true. Setting to false is
109 // DEPRECATED and potentially dangerous because in that case BackupEngine
110 // can lose data if backing up databases with distinct or divergent
111 // history, for example if restoring from a backup other than the latest,
112 // writing to the DB, and creating another backup. Setting to true (default)
113 // prevents these issues by ensuring that different table files (SSTs) and
114 // blob files with the same number are treated as distinct. See
115 // share_files_with_checksum_naming and ShareFilesNaming.
116 //
117 // Default: true
118 bool share_files_with_checksum;
119
120 // Up to this many background threads will copy files for CreateNewBackup()
121 // and RestoreDBFromBackup()
122 // Default: 1
123 int max_background_operations;
124
125 // During backup user can get callback every time next
126 // callback_trigger_interval_size bytes being copied.
127 // Default: 4194304
128 uint64_t callback_trigger_interval_size;
129
130 // For BackupEngineReadOnly, Open() will open at most this many of the
131 // latest non-corrupted backups.
132 //
133 // Note: this setting is ignored (behaves like INT_MAX) for any kind of
134 // writable BackupEngine because it would inhibit accounting for shared
135 // files for proper backup deletion, including purging any incompletely
136 // created backups on creation of a new backup.
137 //
138 // Default: INT_MAX
139 int max_valid_backups_to_open;
140
141 // ShareFilesNaming describes possible naming schemes for backup
142 // table and blob file names when they are stored in the
143 // shared_checksum directory (i.e., both share_table_files and
144 // share_files_with_checksum are true).
145 enum ShareFilesNaming : uint32_t {
146 // Backup blob filenames are <file_number>_<crc32c>_<file_size>.blob and
147 // backup SST filenames are <file_number>_<crc32c>_<file_size>.sst
148 // where <crc32c> is an unsigned decimal integer. This is the
149 // original/legacy naming scheme for share_files_with_checksum,
150 // with two problems:
151 // * At massive scale, collisions on this triple with different file
152 // contents is plausible.
153 // * Determining the name to use requires computing the checksum,
154 // so generally requires reading the whole file even if the file
155 // is already backed up.
156 //
157 // ** ONLY RECOMMENDED FOR PRESERVING OLD BEHAVIOR **
158 kLegacyCrc32cAndFileSize = 1U,
159
160 // Backup SST filenames are <file_number>_s<db_session_id>.sst. This
161 // pair of values should be very strongly unique for a given SST file
162 // and easily determined before computing a checksum. The 's' indicates
163 // the value is a DB session id, not a checksum.
164 //
165 // Exceptions:
166 // * For blob files, kLegacyCrc32cAndFileSize is used as currently
167 // db_session_id is not supported by the blob file format.
168 // * For old SST files without a DB session id, kLegacyCrc32cAndFileSize
169 // will be used instead, matching the names assigned by RocksDB versions
170 // not supporting the newer naming scheme.
171 // * See also flags below.
172 kUseDbSessionId = 2U,
173
174 kMaskNoNamingFlags = 0xffffU,
175
176 // If not already part of the naming scheme, insert
177 // _<file_size>
178 // before .sst and .blob in the name. In case of user code actually parsing
179 // the last _<whatever> before the .sst and .blob as the file size, this
180 // preserves that feature of kLegacyCrc32cAndFileSize. In other words, this
181 // option makes official that unofficial feature of the backup metadata.
182 //
183 // We do not consider SST and blob file sizes to have sufficient entropy to
184 // contribute significantly to naming uniqueness.
185 kFlagIncludeFileSize = 1U << 31,
186
187 kMaskNamingFlags = ~kMaskNoNamingFlags,
188 };
189
190 // Naming option for share_files_with_checksum table and blob files. See
191 // ShareFilesNaming for details.
192 //
193 // Modifying this option cannot introduce a downgrade compatibility issue
194 // because RocksDB can read, restore, and delete backups using different file
195 // names, and it's OK for a backup directory to use a mixture of table and
196 // blob files naming schemes.
197 //
198 // However, modifying this option and saving more backups to the same
199 // directory can lead to the same file getting saved again to that
200 // directory, under the new shared name in addition to the old shared
201 // name.
202 //
203 // Default: kUseDbSessionId | kFlagIncludeFileSize
204 //
205 // Note: This option comes into effect only if both share_files_with_checksum
206 // and share_table_files are true.
207 ShareFilesNaming share_files_with_checksum_naming;
208
209 // Major schema version to use when writing backup meta files
210 // 1 (default) - compatible with very old versions of RocksDB.
211 // 2 - can be read by RocksDB versions >= 6.19.0. Minimum schema version for
212 // * (Experimental) saving and restoring file temperature metadata
213 int schema_version = 1;
214
215 // (Experimental - subject to change or removal) When taking a backup and
216 // saving file temperature info (minimum schema_version is 2), there are
217 // two potential sources of truth for the placement of files into temperature
218 // tiers: (a) the current file temperature reported by the FileSystem or
219 // (b) the expected file temperature recorded in DB manifest. When this
220 // option is false (default), (b) overrides (a) if both are not UNKNOWN.
221 // When true, (a) overrides (b) if both are not UNKNOWN. Regardless of this
222 // setting, a known temperature overrides UNKNOWN.
223 bool current_temperatures_override_manifest = false;
224
225 void Dump(Logger* logger) const;
226
227 explicit BackupEngineOptions(
228 const std::string& _backup_dir, Env* _backup_env = nullptr,
229 bool _share_table_files = true, Logger* _info_log = nullptr,
230 bool _sync = true, bool _destroy_old_data = false,
231 bool _backup_log_files = true, uint64_t _backup_rate_limit = 0,
232 uint64_t _restore_rate_limit = 0, int _max_background_operations = 1,

234 int _max_valid_backups_to_open = INT_MAX,
235 ShareFilesNaming _share_files_with_checksum_naming =
236 static_cast<ShareFilesNaming>(kUseDbSessionId | kFlagIncludeFileSize))
237 : backup_dir(_backup_dir),
238 backup_env(_backup_env),
239 share_table_files(_share_table_files),
240 info_log(_info_log),
241 sync(_sync),
242 destroy_old_data(_destroy_old_data),
243 backup_log_files(_backup_log_files),
244 backup_rate_limit(_backup_rate_limit),

247 max_background_operations(_max_background_operations),

252 assert((share_files_with_checksum_naming & kMaskNoNamingFlags) != 0);
253 }
254};

256inline BackupEngineOptions::ShareFilesNaming operator&(
257 BackupEngineOptions::ShareFilesNaming lhs,
258 BackupEngineOptions::ShareFilesNaming rhs) {
259 uint32_t l = static_cast<uint32_t>(lhs);
260 uint32_t r = static_cast<uint32_t>(rhs);
261 assert(r == BackupEngineOptions::kMaskNoNamingFlags ||
262 (r & BackupEngineOptions::kMaskNoNamingFlags) == 0);
263 return static_cast<BackupEngineOptions::ShareFilesNaming>(l & r);
264}
265
266inline BackupEngineOptions::ShareFilesNaming operator|(
267 BackupEngineOptions::ShareFilesNaming lhs,
268 BackupEngineOptions::ShareFilesNaming rhs) {
269 uint32_t l = static_cast<uint32_t>(lhs);
270 uint32_t r = static_cast<uint32_t>(rhs);
271 assert((r & BackupEngineOptions::kMaskNoNamingFlags) == 0);
272 return static_cast<BackupEngineOptions::ShareFilesNaming>(l | r);
273}

297struct CreateBackupOptions {
298 // Flush will always trigger if 2PC is enabled.
299 // If write-ahead logs are disabled, set flush_before_backup=true to
300 // avoid losing unflushed key/value pairs from the memtable.
301 bool flush_before_backup = false;
302
303 // Callback for reporting progress, based on callback_trigger_interval_size.
304 //
305 // An exception thrown from the callback will result in Status::Aborted from
306 // the operation.
307 std::function<void()> progress_callback = {};
308
309 // A callback that allows the API user to select files for exclusion, such
310 // as if the files are known to exist in an alternate backup directory.
311 // Only "shared" files can be excluded from backups. This is an advanced
312 // feature because the BackupEngine user is trusted to keep track of files
313 // such that the DB can be restored.
314 //
315 // Input to the callback is a [begin,end) range of sharable files live in
316 // the DB being backed up, and the callback implementation sets
317 // exclude_decision=true for files to exclude. A callback offers maximum
318 // flexibility, e.g. if remote files are unavailable at backup time but
319 // whose existence has been recorded somewhere. In case of an empty or
320 // no-op callback, all files are included in the backup .
321 //
322 // To restore the DB, RestoreOptions::alternate_dirs must be used to provide
323 // the excluded files.
324 //
325 // An exception thrown from the callback will result in Status::Aborted from
326 // the operation.
327 std::function<void(MaybeExcludeBackupFile* files_begin,
328 MaybeExcludeBackupFile* files_end)>
329 exclude_files_callback = {};
330
331 // If false, background_thread_cpu_priority is ignored.
332 // Otherwise, the cpu priority can be decreased,
333 // if you try to increase the priority, the priority will not change.
334 // The initial priority of the threads is CpuPriority::kNormal,
335 // so you can decrease to priorities lower than kNormal.
336 bool decrease_background_thread_cpu_priority = false;
337 CpuPriority background_thread_cpu_priority = CpuPriority::kNormal;
338};

533 // same as CreateNewBackup, but stores extra application metadata.

555 BackupID* new_backup_id = nullptr) {

560 virtual IOStatus CreateNewBackup(
561 DB* db, bool flush_before_backup = false,
562 std::function<void()> progress_callback = []() {}) {
563 CreateBackupOptions options;
564 options.flush_before_backup = flush_before_backup;
565 options.progress_callback = progress_callback;
566 return CreateNewBackup(options, db);
567 }

570 // that is currently happening. It will return immediately, will
571 // not wait for the backup to stop.
572 // The backup will stop ASAP and the call to CreateNewBackup will
573 // return Status::Incomplete(). It will not clean up after itself, but
574 // the state will remain consistent. The state will be cleaned up the
575 // next time you call CreateNewBackup or GarbageCollect.
576 virtual void StopBackup() = 0;
577
578 // Will delete any files left over from incomplete creation or deletion of
579 // a backup. This is not normally needed as those operations also clean up
580 // after prior incomplete calls to the same kind of operation (create or
581 // delete). This does not delete corrupt backups but can delete files that
582 // would be needed to manually recover a corrupt backup or to preserve an
583 // unrecognized (e.g. incompatible future version) backup.
584 // NOTE: This is not designed to delete arbitrary files added to the backup
585 // directory outside of BackupEngine, and clean-up is always subject to

587 // NOTE2: For concurrency and interference purposes (see BackupEngine
588 // comment), GarbageCollect (GC) is like other Append operations, even
589 // though it seems different. Although GC can delete physical data, it does
590 // not delete any logical data read by Read operations. GC can interfere
591 // with Append or Write operations in another BackupEngine on the same
592 // backup_dir, because temporary files will be treated as obsolete and
593 // deleted.
594 virtual IOStatus GarbageCollect() = 0;

597// A backup engine for organizing and managing backups.
598// This class is not user-extensible.
599//
600// This class declaration adds "Write" operations in addition to the
601// operations from BackupEngineAppendOnlyBase and BackupEngineReadOnlyBase.
602//
603// # Concurrency between threads on the same BackupEngine* object
604//
605// As of version 6.20, BackupEngine* operations are generally thread-safe,
606// using a read-write lock, though single-thread operation is still
607// recommended to avoid TOCTOU bugs. Specifically, particular kinds of
608// concurrent operations behave like this:
609//
610// op1\op2| Read | Append | Write
611// -------|-------|--------|--------
612// Read | conc | block | block
613// Append | block | block | block
614// Write | block | block | block
615//
616// conc = operations safely proceed concurrently
617// block = one of the operations safely blocks until the other completes.
618// There is generally no guarantee as to which completes first.
619//
620// StopBackup is the only operation that affects an ongoing operation.
621//
622// # Interleaving operations between BackupEngine* objects open on the
623// same backup_dir
624//
625// It is recommended only to have one BackupEngine* object open for a given
626// backup_dir, but it is possible to mix / interleave some operations
627// (regardless of whether they are concurrent) with these caveats:
628//
629// op1\op2| Open | Read | Append | Write
630// -------|--------|--------|--------|--------
631// Open | conc | conc | atomic | unspec
632// Read | conc | conc | old | unspec
633// Append | atomic | old | unspec | unspec
634// Write | unspec | unspec | unspec | unspec
635//
636// Special case: Open with destroy_old_data=true is really a Write
637//
638// conc = operations safely proceed, concurrently when applicable
639// atomic = operations are effectively atomic; if a concurrent Append
640// operation has not completed at some key point during Open, the
641// opened BackupEngine* will never see the result of the Append op.
642// old = Read operations do not include any state changes from other
643// BackupEngine* objects; they return the state at their Open time.
644// unspec = Behavior is unspecified, including possibly trashing the
645// backup_dir, but is "memory safe" (no C++ undefined behavior)
646//
647class BackupEngine : public BackupEngineReadOnlyBase,
648 public BackupEngineAppendOnlyBase {
649 public:
650 virtual ~BackupEngine() {}
651
652 // BackupEngineOptions have to be the same as the ones used in previous
653 // BackupEngines for the same backup directory.
654 static IOStatus Open(const BackupEngineOptions& options, Env* db_env,
655 BackupEngine** backup_engine_ptr);
656
657 // keep for backward compatibility.
658 static IOStatus Open(Env* db_env, const BackupEngineOptions& options,
659 BackupEngine** backup_engine_ptr) {
660 return BackupEngine::Open(options, db_env, backup_engine_ptr);
661 }
662
663 // Deletes old backups, keeping latest num_backups_to_keep alive.
664 // See also DeleteBackup.
665 virtual IOStatus PurgeOldBackups(uint32_t num_backups_to_keep) = 0;
666
667 // Deletes a specific backup. If this operation (or PurgeOldBackups)
668 // is not completed due to crash, power failure, etc. the state
669 // will be cleaned up the next time you call DeleteBackup,
670 // PurgeOldBackups, or GarbageCollect.
671 virtual IOStatus DeleteBackup(BackupID backup_id) = 0;
672};
673
674// A variant of BackupEngine that only allows "Read" operations. See
675// BackupEngine comment for details. This class is not user-extensible.
676class BackupEngineReadOnly : public BackupEngineReadOnlyBase {
