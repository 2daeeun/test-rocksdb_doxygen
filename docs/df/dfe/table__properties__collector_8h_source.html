117 private:
118 std::shared_ptr<TablePropertiesCollectorFactory> user_collector_factory_;

121// When rocksdb creates a newtable, it will encode all "user keys" into
122// "internal keys". This class collects min/max timestamp from the encoded
123// internal key when Add() is invoked.
124//
125// @param cmp the user comparator to compare the timestamps in internal key.
126class TimestampTablePropertiesCollector : public InternalTblPropColl {
127 public:
128 explicit TimestampTablePropertiesCollector(const Comparator* cmp)
129 : cmp_(cmp),
130 timestamp_min_(kDisableUserTimestamp),
131 timestamp_max_(kDisableUserTimestamp) {}
132
133 Status InternalAdd(const Slice& key, const Slice& /* value */,
134 uint64_t /* file_size */) override {
135 auto user_key = ExtractUserKey(key);
136 assert(cmp_ && cmp_->timestamp_size() > 0);
137 if (user_key.size() < cmp_->timestamp_size()) {
138 return Status::Corruption(
139 "User key size mismatch when comparing to timestamp size.");
140 }
141 auto timestamp_in_key =
142 ExtractTimestampFromUserKey(user_key, cmp_->timestamp_size());
143 if (timestamp_max_ == kDisableUserTimestamp ||
144 cmp_->CompareTimestamp(timestamp_in_key, timestamp_max_) > 0) {
145 timestamp_max_.assign(timestamp_in_key.data(), timestamp_in_key.size());
146 }
147 if (timestamp_min_ == kDisableUserTimestamp ||
148 cmp_->CompareTimestamp(timestamp_min_, timestamp_in_key) > 0) {
149 timestamp_min_.assign(timestamp_in_key.data(), timestamp_in_key.size());
150 }
151 return Status::OK();
152 }
153
154 void BlockAdd(uint64_t /* block_uncomp_bytes */,
155 uint64_t /* block_compressed_bytes_fast */,
156 uint64_t /* block_compressed_bytes_slow */) override {
157 return;
158 }
159
160 Status Finish(UserCollectedProperties* properties) override {
161 // timestamp is empty is table is empty
162 assert(timestamp_min_.size() == timestamp_max_.size() &&
163 (timestamp_min_.empty() ||
164 timestamp_max_.size() == cmp_->timestamp_size()));
165 properties->insert({"rocksdb.timestamp_min", timestamp_min_});
166 properties->insert({"rocksdb.timestamp_max", timestamp_max_});
167 return Status::OK();
168 }
169
170 const char* Name() const override {
171 return "TimestampTablePropertiesCollector";
172 }
173
174 UserCollectedProperties GetReadableProperties() const override {
175 return {{"rocksdb.timestamp_min", Slice(timestamp_min_).ToString(true)},
176 {"rocksdb.timestamp_max", Slice(timestamp_max_).ToString(true)}};
177 }
178
179 protected:
180 const Comparator* const cmp_;
181 std::string timestamp_min_;
182 std::string timestamp_max_;
183};
