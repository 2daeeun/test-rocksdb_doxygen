10// InlineSkipList is derived from SkipList (skiplist.h), but it optimizes
11// the memory layout by requiring that the key storage be allocated through
12// the skip list instance. For the common case of SkipList<const char*,
13// Cmp> this saves 1 pointer per skip list node and gives better cache
14// locality, at the expense of wasted padding from using AllocateAligned
15// instead of Allocate for the keys. The unused padding will be from
16// 0 to sizeof(void*)-1 bytes, and the space savings are sizeof(void*)
17// bytes, so despite the padding the space used is always less than
18// SkipList<const char*, ..>.
19//
20// Thread safety -------------
21//
22// Writes via Insert require external synchronization, most likely a mutex.
23// InsertConcurrently can be safely called concurrently with reads and
24// with other concurrent inserts. Reads require a guarantee that the
25// InlineSkipList will not be destroyed while the read is in progress.
26// Apart from that, reads progress without any internal locking or
27// synchronization.
28//
29// Invariants:
30//
31// (1) Allocated nodes are never deleted until the InlineSkipList is
32// destroyed. This is trivially guaranteed by the code since we never
33// delete any skip list nodes.
34//
35// (2) The contents of a Node except for the next/prev pointers are
36// immutable after the Node has been linked into the InlineSkipList.
37// Only Insert() modifies the list, and it is careful to initialize a
38// node and use release-stores to publish the nodes in one or more lists.
39//
40// ... prev vs. next pointer ordering ...
41//

123 // Inserts a node into the skip list. key must have been allocated by
124 // AllocateKey and then filled in by the caller. If UseCAS is true,
125 // then external synchronization is not required, otherwise this method
126 // may not be called concurrently with any other insertions.
127 //
128 // Regardless of whether UseCAS is true, the splice must be owned
129 // exclusively by the current thread. If allow_partial_splice_fix is
130 // true, then the cost of insertion is amortized O(log D), where D is
131 // the distance from the splice to the inserted key (measured as the
132 // number of intervening nodes). Note that this bound is very good for
133 // sequential insertions! If allow_partial_splice_fix is false then
134 // the existing splice will be ignored unless the current key is being
135 // inserted immediately after the splice. allow_partial_splice_fix ==
136 // false has worse running time for the non-sequential case O(log N),
137 // but a better constant factor.

139 bool Insert(const char* key, Splice* splice, bool allow_partial_splice_fix);

291template <class Comparator>
292struct InlineSkipList<Comparator>::Splice {
293 // The invariant of a Splice is that prev_[i+1].key <= prev_[i].key <
294 // next_[i].key <= next_[i+1].key for all i. That means that if a
295 // key is bracketed by prev_[i] and next_[i] then it is bracketed by
296 // all higher levels. It is _not_ required that prev_[i]->Next(i) ==
297 // next_[i] (it probably did at some point in the past, but intervening
298 // or concurrent operations might have inserted nodes in between).
299 int height_ = 0;
300 Node** prev_;
301 Node** next_;
302};
303
304// The Node data type is more of a pointer into custom-managed memory than
305// a traditional C++ struct. The key is stored in the bytes immediately
306// after the struct, and the next_ pointers for nodes with height > 1 are
307// stored immediately _before_ the struct. This avoids the need to include
308// any pointer or sizing data, which reduces per-node memory overheads.
309template <class Comparator>
310struct InlineSkipList<Comparator>::Node {
311 // Stores the height of the node in the memory location normally used for
312 // next_[0]. This is used for passing data from AllocateKey to Insert.
313 void StashHeight(const int height) {
314 assert(sizeof(int) <= sizeof(next_[0]));
315 memcpy(static_cast<void*>(&next_[0]), &height, sizeof(int));
316 }
317
318 // Retrieves the value passed to StashHeight. Undefined after a call
319 // to SetNext or NoBarrier_SetNext.
320 int UnstashHeight() const {
321 int rv;
322 memcpy(&rv, &next_[0], sizeof(int));
323 return rv;
324 }
325
326 const char* Key() const { return reinterpret_cast<const char*>(&next_[1]); }
327
328 // Accessors/mutators for links. Wrapped in methods so we can add
329 // the appropriate barriers as necessary, and perform the necessary
330 // addressing trickery for storing links below the Node in memory.
331 Node* Next(int n) {
332 assert(n >= 0);
333 // Use an 'acquire load' so that we observe a fully initialized
334 // version of the returned Node.
335 return ((&next_[0] - n)->load(std::memory_order_acquire));
336 }
337
338 void SetNext(int n, Node* x) {
339 assert(n >= 0);
340 // Use a 'release store' so that anybody who reads through this
341 // pointer observes a fully initialized version of the inserted node.
342 (&next_[0] - n)->store(x, std::memory_order_release);
343 }
344
345 bool CASNext(int n, Node* expected, Node* x) {
346 assert(n >= 0);
347 return (&next_[0] - n)->compare_exchange_strong(expected, x);
348 }
349
350 // No-barrier variants that can be safely used in a few locations.
351 Node* NoBarrier_Next(int n) {
352 assert(n >= 0);
353 return (&next_[0] - n)->load(std::memory_order_relaxed);
354 }
355
356 void NoBarrier_SetNext(int n, Node* x) {
357 assert(n >= 0);
358 (&next_[0] - n)->store(x, std::memory_order_relaxed);
359 }
360
361 // Insert node after prev on specific level.
362 void InsertAfter(Node* prev, int level) {
363 // NoBarrier_SetNext() suffices since we will add a barrier when
364 // we publish a pointer to "this" in prev.
365 NoBarrier_SetNext(level, prev->NoBarrier_Next(level));
366 prev->SetNext(level, this);
367 }
368
369 private:
370 // next_[0] is the lowest level link (level 0). Higher levels are
371 // stored _earlier_, so level 1 is at next_[-1].
372 std::atomic<Node*> next_[1];
373};

405template <class Comparator>
406inline Status InlineSkipList<Comparator>::Iterator::NextAndValidate(
407 bool allow_data_in_errors) {
408 assert(Valid());
409 Node* prev_node = node_;
410 node_ = node_->Next(0);
411 // Verify that keys are increasing.
412 if (prev_node != list_->head_ && node_ != nullptr &&
413 list_->compare_(prev_node->Key(), node_->Key()) >= 0) {
414 Node* node = node_;
415 // invalidates the iterator
416 node_ = nullptr;
417 return Corruption(prev_node, node, allow_data_in_errors);
418 }
419 return Status::OK();
420}

422template <class Comparator>
423inline void InlineSkipList<Comparator>::Iterator::Prev() {
424 // Instead of using explicit "prev" links, we just search for the
425 // last node that falls before key.
426 assert(Valid());
427 node_ = list_->FindLessThan(node_->Key(), nullptr);
428 if (node_ == list_->head_) {
429 node_ = nullptr;
430 }
431}

433template <class Comparator>
434inline Status InlineSkipList<Comparator>::Iterator::PrevAndValidate(
435 const bool allow_data_in_errors) {
436 assert(Valid());
437 // Skip list validation is done in FindLessThan().
438 Node* out_of_order_node = nullptr;
439 node_ = list_->FindLessThan(node_->Key(), &out_of_order_node);
440 if (out_of_order_node) {
441 Node* node = node_;
442 node_ = nullptr;
443 return Corruption(node, out_of_order_node, allow_data_in_errors);
444 }
445 if (node_ == list_->head_) {
446 node_ = nullptr;
447 }
448 return Status::OK();
449}

499template <class Comparator>
500int InlineSkipList<Comparator>::RandomHeight() {
501 auto rnd = Random::GetTLSInstance();
502
503 // Increase height with probability 1 in kBranching
504 int height = 1;
505 while (height < kMaxHeight_ && height < kMaxPossibleHeight &&
506 rnd->Next() < kScaledInverseBranching_) {
507 height++;
508 }
509 TEST_SYNC_POINT_CALLBACK("InlineSkipList::RandomHeight::height", &height);
510 assert(height > 0);
511 assert(height <= kMaxHeight_);
512 assert(height <= kMaxPossibleHeight);
513 return height;
514}

532template <class Comparator>
533typename InlineSkipList<Comparator>::Node*
534InlineSkipList<Comparator>::FindGreaterOrEqual(
535 const char* key, Node** const out_of_order_node) const {
536 // Note: It looks like we could reduce duplication by implementing
537 // this function as FindLessThan(key)->Next(0), but we wouldn't be able
538 // to exit early on equality and the result wouldn't even be correct.
539 // A concurrent insert might occur after FindLessThan(key) but before
540 // we get a chance to call Next(0).
541 Node* x = head_;
542 int level = GetMaxHeight() - 1;
543 Node* last_bigger = nullptr;
544 const DecodedKey key_decoded = compare_.decode_key(key);
545 while (true) {
546 Node* next = x->Next(level);
547 if (next != nullptr) {
548 PREFETCH(next->Next(level), 0, 1);
549 if (out_of_order_node && x != head_ &&
550 compare_(x->Key(), next->Key()) >= 0) {
551 *out_of_order_node = next;
552 return x;
553 }
554 }
555 // Make sure the lists are sorted
556 assert(x == head_ || next == nullptr || KeyIsAfterNode(next->Key(), x));
557 // Make sure we haven't overshot during our search
558 assert(x == head_ || KeyIsAfterNode(key_decoded, x));
559 int cmp = (next == nullptr || next == last_bigger)
560 ? 1
561 : compare_(next->Key(), key_decoded);
562 if (cmp == 0 || (cmp > 0 && level == 0)) {
563 return next;
564 } else if (cmp < 0) {
565 // Keep searching in this list
566 x = next;
567 } else {
568 // Switch to next list, reuse compare_() result
569 last_bigger = next;
570 level--;
571 }
572 }
573}

575template <class Comparator>
576typename InlineSkipList<Comparator>::Node*
577InlineSkipList<Comparator>::FindLessThan(const char* key,
578 Node** const out_of_order_node) const {
579 int level = GetMaxHeight() - 1;
580 assert(level >= 0);
581 Node* x = head_;
582 // KeyIsAfter(key, last_not_after) is definitely false
583 Node* last_not_after = nullptr;
584 const DecodedKey key_decoded = compare_.decode_key(key);
585 while (true) {
586 assert(x != nullptr);
587 Node* next = x->Next(level);
588 if (next != nullptr) {
589 PREFETCH(next->Next(level), 0, 1);
590 if (out_of_order_node && x != head_ &&
591 compare_(x->Key(), next->Key()) >= 0) {
592 *out_of_order_node = next;
593 return x;
594 }
595 }
596 assert(x == head_ || next == nullptr || KeyIsAfterNode(next->Key(), x));
597 assert(x == head_ || KeyIsAfterNode(key_decoded, x));
598 if (next != last_not_after && KeyIsAfterNode(key_decoded, next)) {
599 // Keep searching in this list
600 assert(next != nullptr);
601 x = next;
602 } else {
603 if (level == 0) {
604 return x;
605 } else {
606 // Switch to next list, reuse KeyIsAfterNode() result
607 last_not_after = next;
608 level--;
609 }
610 }
611 }
612}

614template <class Comparator>
615typename InlineSkipList<Comparator>::Node*
616InlineSkipList<Comparator>::FindLast() const {
617 Node* x = head_;
618 int level = GetMaxHeight() - 1;
619 while (true) {
620 Node* next = x->Next(level);
621 if (next == nullptr) {
622 if (level == 0) {
623 return x;
624 } else {
625 // Switch to next list
626 level--;
627 }
628 } else {
629 x = next;
630 }
631 }
632}

634template <class Comparator>
635typename InlineSkipList<Comparator>::Node*
636InlineSkipList<Comparator>::FindRandomEntry() const {
637 // TODO(bjlemaire): consider adding PREFETCH calls.
638 Node *x = head_, *scan_node = nullptr, *limit_node = nullptr;
639
640 // We start at the max level.
641 // FOr each level, we look at all the nodes at the level, and
642 // we randomly pick one of them. Then decrement the level
643 // and reiterate the process.
644 // eg: assume GetMaxHeight()=5, and there are #100 elements (nodes).
645 // level 4 nodes: lvl_nodes={#1, #15, #67, #84}. Randomly pick #15.
646 // We will consider all the nodes between #15 (inclusive) and #67
647 // (exclusive). #67 is called 'limit_node' here.
648 // level 3 nodes: lvl_nodes={#15, #21, #45, #51}. Randomly choose
649 // #51. #67 remains 'limit_node'.
650 // [...]
651 // level 0 nodes: lvl_nodes={#56,#57,#58,#59}. Randomly pick $57.
652 // Return Node #57.
653 std::vector<Node*> lvl_nodes;
654 Random* rnd = Random::GetTLSInstance();
655 int level = GetMaxHeight() - 1;
656
657 while (level >= 0) {
658 lvl_nodes.clear();
659 scan_node = x;
660 while (scan_node != limit_node) {
661 lvl_nodes.push_back(scan_node);
662 scan_node = scan_node->Next(level);
663 }
664 uint32_t rnd_idx = rnd->Next() % lvl_nodes.size();
665 x = lvl_nodes[rnd_idx];
666 if (rnd_idx + 1 < lvl_nodes.size()) {
667 limit_node = lvl_nodes[rnd_idx + 1];
668 }
669 level--;
670 }
671 // There is a special case where x could still be the head_
672 // (note that the head_ contains no key).
673 return x == head_ && head_ != nullptr ? head_->Next(0) : x;
674}

676template <class Comparator>
677uint64_t InlineSkipList<Comparator>::ApproximateNumEntries(
678 const Slice& start_ikey, const Slice& end_ikey) const {
679 // The number of entries at a given level for the given range, in terms of
680 // the actual number of entries in that range (level 0), follows a binomial
681 // distribution, which is very well approximated by the Poisson distribution.
682 // That has stddev sqrt(x) where x is the expected number of entries (mean)
683 // at this level, and the best predictor of x is the number of observed
684 // entries (at this level). To predict the number of entries on level 0 we use
685 // x * kBranchinng ^ level. From the standard deviation, the P99+ relative
686 // error is roughly 3 * sqrt(x) / x. Thus, a reasonable approach would be to
687 // find the smallest level with at least some moderate constant number entries
688 // in range. E.g. with at least ~40 entries, we expect P99+ relative error
689 // (approximation accuracy) of ~ 50% = 3 * sqrt(40) / 40; P95 error of
690 // ~30%; P75 error of < 20%.
691 //
692 // However, there are two issues with this approach, and an observation:
693 // * Pointer chasing on the larger (bottom) levels is much slower because of
694 // cache hierarchy effects, so when the result is smaller, getting the result
695 // will be substantially slower, despite traversing a similar number of
696 // entries. (We could be clever about pipelining our pointer chasing but
697 // that's complicated.)
698 // * The larger (bottom) levels also have lower variance because there's a
699 // chance (or certainty) that we reach level 0 and return the exact answer.
700 // * For applications in query planning, we can also tolerate more variance on
701 // small results because the impact of misestimating is likely smaller.
702 //
703 // These factors point us to an approach in which we have a higher minimum
704 // threshold number of samples for higher levels and lower for lower levels
705 // (see sufficient_samples below). This seems to yield roughly consistent
706 // relative error (stddev around 20%, less for large results) and roughly
707 // consistent query time around the time of two memtable point queries.
708 //
709 // Engineering observation: it is tempting to think that taking into account
710 // what we already found in how many entries occur on higher levels, not just
711 // the first iterated level with a sufficient number of samples, would yield
712 // a more accurate estimate. But that doesn't work because of the particular
713 // correlations and independences of the data: each level higher is just an
714 // independently probabilistic filtering of the level below it. That
715 // filtering from level l to l+1 has no more information about levels
716 // 0 .. l-1 than we can get from level l. The structure of RandomHeight() is
717 // a clue to these correlations and independences.
718
719 Node* lb = head_;
720 Node* ub = nullptr;
721 uint64_t count = 0;
722 for (int level = GetMaxHeight() - 1; level >= 0; level--) {
723 auto sufficient_samples = static_cast<uint64_t>(level) * kBranching_ + 10U;
724 if (count >= sufficient_samples) {
725 // No more counting; apply powers of kBranching and avoid floating point
726 count *= kBranching_;
727 continue;
728 }
729 count = 0;
730 Node* next;
731 // Get a more precise lower bound (for start key)
732 for (;;) {
733 next = lb->Next(level);
734 if (next == ub) {
735 break;
736 }
737 assert(next != nullptr);
738 if (compare_(next->Key(), start_ikey) >= 0) {
739 break;
740 }
741 lb = next;
742 }
743 // Count entries on this level until upper bound (for end key)
744 for (;;) {
745 if (next == ub) {
746 break;
747 }
748 assert(next != nullptr);
749 if (compare_(next->Key(), end_ikey) >= 0) {
750 // Save refined upper bound to potentially save key comparison
751 ub = next;
752 break;
753 }
754 count++;
755 next = next->Next(level);
756 }
757 }
758 return count;
759}

789template <class Comparator>
790typename InlineSkipList<Comparator>::Node*
791InlineSkipList<Comparator>::AllocateNode(size_t key_size, int height) {
792 auto prefix = sizeof(std::atomic<Node*>) * (height - 1);
793
794 // prefix is space for the height - 1 pointers that we store before
795 // the Node instance (next_[-(height - 1) .. -1]). Node starts at
796 // raw + prefix, and holds the bottom-mode (level 0) skip list pointer
797 // next_[0]. key_size is the bytes for the key, which comes just after
798 // the Node.
799 char* raw = allocator_->AllocateAligned(prefix + sizeof(Node) + key_size);
800 Node* x = reinterpret_cast<Node*>(raw + prefix);
801
802 // Once we've linked the node into the skip list we don't actually need
803 // to know its height, because we can implicitly use the fact that we
804 // traversed into a node at level h to known that h is a valid level
805 // for that node. We need to convey the height to the Insert step,
806 // however, so that it can perform the proper links. Since we're not
807 // using the pointers at the moment, StashHeight temporarily borrow
808 // storage from next_[0] for that purpose.
809 x->StashHeight(height);
810 return x;
811}

876template <class Comparator>
877template <bool prefetch_before>
878void InlineSkipList<Comparator>::FindSpliceForLevel(const DecodedKey& key,
879 Node* before, Node* after,
880 int level, Node** out_prev,
881 Node** out_next) {
882 while (true) {
883 Node* next = before->Next(level);
884 if (next != nullptr) {
885 PREFETCH(next->Next(level), 0, 1);
886 }
887 if (prefetch_before == true) {
888 if (next != nullptr && level > 0) {
889 PREFETCH(next->Next(level - 1), 0, 1);
890 }
891 }
892 assert(before == head_ || next == nullptr ||
893 KeyIsAfterNode(next->Key(), before));
894 assert(before == head_ || KeyIsAfterNode(key, before));
895 if (next == after || !KeyIsAfterNode(key, next)) {
896 // found it
897 *out_prev = before;
898 *out_next = next;
899 return;
900 }
901 before = next;
902 }
903}

905template <class Comparator>
906void InlineSkipList<Comparator>::RecomputeSpliceLevels(const DecodedKey& key,
907 Splice* splice,
908 int recompute_level) {
909 assert(recompute_level > 0);
910 assert(recompute_level <= splice->height_);
911 for (int i = recompute_level - 1; i >= 0; --i) {
912 FindSpliceForLevel<true>(key, splice->prev_[i + 1], splice->next_[i + 1], i,
913 &splice->prev_[i], &splice->next_[i]);
914 }

917template <class Comparator>
918template <bool UseCAS>
919bool InlineSkipList<Comparator>::Insert(const char* key, Splice* splice,
920 bool allow_partial_splice_fix) {
921 Node* x = reinterpret_cast<Node*>(const_cast<char*>(key)) - 1;
922 const DecodedKey key_decoded = compare_.decode_key(key);
923 int height = x->UnstashHeight();
924 assert(height >= 1 && height <= kMaxHeight_);
925
926 int max_height = max_height_.load(std::memory_order_relaxed);
927 while (height > max_height) {
928 if (max_height_.compare_exchange_weak(max_height, height)) {
929 // successfully updated it
930 max_height = height;
931 break;
932 }
933 // else retry, possibly exiting the loop because somebody else
934 // increased it
935 }
936 assert(max_height <= kMaxPossibleHeight);
937
938 int recompute_height = 0;
939 if (splice->height_ < max_height) {
940 // Either splice has never been used or max_height has grown since
941 // last use. We could potentially fix it in the latter case, but
942 // that is tricky.
943 splice->prev_[max_height] = head_;
944 splice->next_[max_height] = nullptr;
945 splice->height_ = max_height;
946 recompute_height = max_height;
947 } else {
948 // Splice is a valid proper-height splice that brackets some
949 // key, but does it bracket this one? We need to validate it and
950 // recompute a portion of the splice (levels 0..recompute_height-1)
951 // that is a superset of all levels that don't bracket the new key.
952 // Several choices are reasonable, because we have to balance the work
953 // saved against the extra comparisons required to validate the Splice.
954 //
955 // One strategy is just to recompute all of orig_splice_height if the
956 // bottom level isn't bracketing. This pessimistically assumes that
957 // we will either get a perfect Splice hit (increasing sequential
958 // inserts) or have no locality.
959 //
960 // Another strategy is to walk up the Splice's levels until we find
961 // a level that brackets the key. This strategy lets the Splice
962 // hint help for other cases: it turns insertion from O(log N) into
963 // O(log D), where D is the number of nodes in between the key that
964 // produced the Splice and the current insert (insertion is aided
965 // whether the new key is before or after the splice). If you have
966 // a way of using a prefix of the key to map directly to the closest
967 // Splice out of O(sqrt(N)) Splices and we make it so that splices
968 // can also be used as hints during read, then we end up with Oshman's
969 // and Shavit's SkipTrie, which has O(log log N) lookup and insertion
970 // (compare to O(log N) for skip list).
971 //
972 // We control the pessimistic strategy with allow_partial_splice_fix.
973 // A good strategy is probably to be pessimistic for seq_splice_,
974 // optimistic if the caller actually went to the work of providing
975 // a Splice.
976 while (recompute_height < max_height) {
977 if (splice->prev_[recompute_height]->Next(recompute_height) !=
978 splice->next_[recompute_height]) {
979 // splice isn't tight at this level, there must have been some inserts
980 // to this
981 // location that didn't update the splice. We might only be a little
982 // stale, but if
983 // the splice is very stale it would be O(N) to fix it. We haven't used
984 // up any of
985 // our budget of comparisons, so always move up even if we are
986 // pessimistic about
987 // our chances of success.
988 ++recompute_height;
989 } else if (splice->prev_[recompute_height] != head_ &&
990 !KeyIsAfterNode(key_decoded,
991 splice->prev_[recompute_height])) {
992 // key is from before splice
993 if (allow_partial_splice_fix) {
994 // skip all levels with the same node without more comparisons
995 Node* bad = splice->prev_[recompute_height];
996 while (splice->prev_[recompute_height] == bad) {
997 ++recompute_height;
998 }
999 } else {
1000 // we're pessimistic, recompute everything
1001 recompute_height = max_height;
1002 }
1003 } else if (KeyIsAfterNode(key_decoded, splice->next_[recompute_height])) {
1004 // key is from after splice
1005 if (allow_partial_splice_fix) {
1006 Node* bad = splice->next_[recompute_height];
1007 while (splice->next_[recompute_height] == bad) {
1008 ++recompute_height;
1009 }
1010 } else {
1011 recompute_height = max_height;
1012 }
1013 } else {
1014 // this level brackets the key, we won!
1015 break;
1016 }
1017 }
1018 }
1019 assert(recompute_height <= max_height);
1020 if (recompute_height > 0) {
1021 RecomputeSpliceLevels(key_decoded, splice, recompute_height);
1022 }
1023
1024 bool splice_is_valid = true;
1025 if (UseCAS) {
1026 for (int i = 0; i < height; ++i) {
1027 while (true) {
1028 // Checking for duplicate keys on the level 0 is sufficient
1029 if (UNLIKELY(i == 0 && splice->next_[i] != nullptr &&
1030 compare_(splice->next_[i]->Key(), key_decoded) <= 0)) {
1031 // duplicate key
1032 return false;
1033 }
1034 if (UNLIKELY(i == 0 && splice->prev_[i] != head_ &&
1035 compare_(splice->prev_[i]->Key(), key_decoded) >= 0)) {
1036 // duplicate key
1037 return false;
1038 }
1039 assert(splice->next_[i] == nullptr ||
1040 compare_(x->Key(), splice->next_[i]->Key()) < 0);
1041 assert(splice->prev_[i] == head_ ||
1042 compare_(splice->prev_[i]->Key(), x->Key()) < 0);
1043 x->NoBarrier_SetNext(i, splice->next_[i]);
1044 if (splice->prev_[i]->CASNext(i, splice->next_[i], x)) {
1045 // success
1046 break;
1047 }
1048 // CAS failed, we need to recompute prev and next. It is unlikely
1049 // to be helpful to try to use a different level as we redo the
1050 // search, because it should be unlikely that lots of nodes have
1051 // been inserted between prev[i] and next[i]. No point in using
1052 // next[i] as the after hint, because we know it is stale.
1053 FindSpliceForLevel<false>(key_decoded, splice->prev_[i], nullptr, i,
1054 &splice->prev_[i], &splice->next_[i]);
1055
1056 // Since we've narrowed the bracket for level i, we might have
1057 // violated the Splice constraint between i and i-1. Make sure
1058 // we recompute the whole thing next time.
1059 if (i > 0) {
1060 splice_is_valid = false;
1061 }
1062 }
1063 }
1064 } else {
1065 for (int i = 0; i < height; ++i) {
1066 if (i >= recompute_height &&
1067 splice->prev_[i]->Next(i) != splice->next_[i]) {
1068 FindSpliceForLevel<false>(key_decoded, splice->prev_[i], nullptr, i,
1069 &splice->prev_[i], &splice->next_[i]);
1070 }
1071 // Checking for duplicate keys on the level 0 is sufficient
1072 if (UNLIKELY(i == 0 && splice->next_[i] != nullptr &&
1073 compare_(splice->next_[i]->Key(), key_decoded) <= 0)) {
1074 // duplicate key
1075 return false;
1076 }
1077 if (UNLIKELY(i == 0 && splice->prev_[i] != head_ &&
1078 compare_(splice->prev_[i]->Key(), key_decoded) >= 0)) {
1079 // duplicate key
1080 return false;
1081 }
1082 assert(splice->next_[i] == nullptr ||
1083 compare_(x->Key(), splice->next_[i]->Key()) < 0);
1084 assert(splice->prev_[i] == head_ ||
1085 compare_(splice->prev_[i]->Key(), x->Key()) < 0);
1086 assert(splice->prev_[i]->Next(i) == splice->next_[i]);
1087 x->NoBarrier_SetNext(i, splice->next_[i]);
1088 splice->prev_[i]->SetNext(i, x);
1089 }
1090 }
1091 if (splice_is_valid) {
1092 for (int i = 0; i < height; ++i) {
1093 splice->prev_[i] = x;
1094 }
1095 assert(splice->prev_[splice->height_] == head_);
1096 assert(splice->next_[splice->height_] == nullptr);
1097 for (int i = 0; i < splice->height_; ++i) {
1098 assert(splice->next_[i] == nullptr ||
1099 compare_(key, splice->next_[i]->Key()) < 0);
1100 assert(splice->prev_[i] == head_ ||
1101 compare_(splice->prev_[i]->Key(), key) <= 0);
1102 assert(splice->prev_[i + 1] == splice->prev_[i] ||
1103 splice->prev_[i + 1] == head_ ||
1104 compare_(splice->prev_[i + 1]->Key(), splice->prev_[i]->Key()) <
1105 0);
1106 assert(splice->next_[i + 1] == splice->next_[i] ||
1107 splice->next_[i + 1] == nullptr ||
1108 compare_(splice->next_[i]->Key(), splice->next_[i + 1]->Key()) <
1109 0);
1110 }
1111 } else {
1112 splice->height_ = 0;
1113 }
1114 return true;
1115}

1127template <class Comparator>
1128void InlineSkipList<Comparator>::TEST_Validate() const {
1129 // Interate over all levels at the same time, and verify nodes appear in
1130 // the right order, and nodes appear in upper level also appear in lower
1131 // levels.
1132 Node* nodes[kMaxPossibleHeight];
1133 int max_height = GetMaxHeight();
1134 assert(max_height > 0);
1135 for (int i = 0; i < max_height; i++) {
1136 nodes[i] = head_;
1137 }
1138 while (nodes[0] != nullptr) {
1139 Node* l0_next = nodes[0]->Next(0);
1140 if (l0_next == nullptr) {
1141 break;
1142 }
1143 assert(nodes[0] == head_ || compare_(nodes[0]->Key(), l0_next->Key()) < 0);
1144 nodes[0] = l0_next;
1145
1146 int i = 1;
1147 while (i < max_height) {
1148 Node* next = nodes[i]->Next(i);
1149 if (next == nullptr) {
1150 break;
1151 }
1152 auto cmp = compare_(nodes[0]->Key(), next->Key());
1153 assert(cmp <= 0);
1154 if (cmp == 0) {
1155 assert(next == nodes[0]);
1156 nodes[i] = next;
1157 } else {
1158 break;
1159 }
1160 i++;
1161 }
1162 }
1163 for (int i = 1; i < max_height; i++) {
1164 assert(nodes[i] != nullptr && nodes[i]->Next(i) == nullptr);
1165 }
1166}
