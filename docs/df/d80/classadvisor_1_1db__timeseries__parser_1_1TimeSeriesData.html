98 def fetch_aggregated_values(self, entity, statistics, aggregation_op):
99 # this method performs the aggregation specified by 'aggregation_op'
100 # on the timeseries of 'statistics' for 'entity' and returns:
101 # Dict[statistic, aggregated_value]
102 result = {}
103 for stat in statistics:
104 if stat not in self.keys_ts[entity]:
105 continue
106 agg_val = None
107 if aggregation_op is self.AggregationOperator.latest:
108 latest_timestamp = max(list(self.keys_ts[entity][stat].keys()))
109 agg_val = self.keys_ts[entity][stat][latest_timestamp]
110 elif aggregation_op is self.AggregationOperator.oldest:
111 oldest_timestamp = min(list(self.keys_ts[entity][stat].keys()))
112 agg_val = self.keys_ts[entity][stat][oldest_timestamp]
113 elif aggregation_op is self.AggregationOperator.max:
114 agg_val = max(list(self.keys_ts[entity][stat].values()))
115 elif aggregation_op is self.AggregationOperator.min:
116 agg_val = min(list(self.keys_ts[entity][stat].values()))
117 elif aggregation_op is self.AggregationOperator.avg:
118 values = list(self.keys_ts[entity][stat].values())
119 agg_val = sum(values) / len(values)
120 result[stat] = agg_val
121 return result
