26// Fast implementation of extracting the bottom n bits of an integer.
27// To ensure fast implementation, undefined if n bits is full width or more.
28template <typename T>
29inline T BottomNBits(T v, int nbits) {
30 static_assert(std::is_integral_v<T>, "non-integral type");
31 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
32 assert(nbits >= 0);
33 assert(nbits < int{8 * sizeof(T)});
34#ifdef __BMI2__
35 if constexpr (sizeof(T) <= 4) {
36 return static_cast<T>(_bzhi_u32(static_cast<uint32_t>(v), nbits));
37 }
38 if constexpr (sizeof(T) <= 8) {
39 return static_cast<T>(_bzhi_u64(static_cast<uint64_t>(v), nbits));
40 }
41#endif
42 // Newer compilers compile this down to bzhi on x86, but some older
43 // ones don't, thus the need for the intrinsic above.
44 return static_cast<T>(v & ((T{1} << nbits) - 1));
45}

47// Fast implementation of floor(log2(v)). Undefined for 0 or negative
48// numbers (in case of signed type).
49template <typename T>
50inline int FloorLog2(T v) {
51 static_assert(std::is_integral_v<T>, "non-integral type");
52 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
53 assert(v > 0);
54#ifdef _MSC_VER
55 static_assert(sizeof(T) <= sizeof(uint64_t), "type too big");
56 unsigned long idx = 0;
57 if (sizeof(T) <= sizeof(uint32_t)) {
58 _BitScanReverse(&idx, static_cast<uint32_t>(v));
59 } else {
60#if defined(_M_X64) || defined(_M_ARM64)
61 _BitScanReverse64(&idx, static_cast<uint64_t>(v));
62#else
63 const auto vh = static_cast<uint32_t>(static_cast<uint64_t>(v) >> 32);
64 if (vh != 0) {
65 _BitScanReverse(&idx, static_cast<uint32_t>(vh));
66 idx += 32;
67 } else {
68 _BitScanReverse(&idx, static_cast<uint32_t>(v));
69 }
70#endif
71 }
72 return idx;
73#else
74 static_assert(sizeof(T) <= sizeof(unsigned long long), "type too big");
75 if (sizeof(T) <= sizeof(unsigned int)) {
76 int lz = __builtin_clz(static_cast<unsigned int>(v));
77 return int{sizeof(unsigned int)} * 8 - 1 - lz;
78 } else if (sizeof(T) <= sizeof(unsigned long)) {
79 int lz = __builtin_clzl(static_cast<unsigned long>(v));
80 return int{sizeof(unsigned long)} * 8 - 1 - lz;
81 } else {
82 int lz = __builtin_clzll(static_cast<unsigned long long>(v));
83 return int{sizeof(unsigned long long)} * 8 - 1 - lz;
84 }
85#endif
86}

88// Constexpr version of FloorLog2
89template <typename T>
90constexpr int ConstexprFloorLog2(T v) {
91 // NOTE: not checking is_integral so that this works with Unsigned128
92 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
93 int rv = 0;
94 while (v > T{1}) {
95 ++rv;
96 v >>= 1;
97 }
98 return rv;
99}

101// Number of low-order zero bits before the first 1 bit. Undefined for 0.
102template <typename T>
103inline int CountTrailingZeroBits(T v) {
104 static_assert(std::is_integral_v<T>, "non-integral type");
105 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
106 assert(v != 0);
107#ifdef _MSC_VER
108 static_assert(sizeof(T) <= sizeof(uint64_t), "type too big");
109 unsigned long tz = 0;
110 if (sizeof(T) <= sizeof(uint32_t)) {
111 _BitScanForward(&tz, static_cast<uint32_t>(v));
112 } else {
113#if defined(_M_X64) || defined(_M_ARM64)
114 _BitScanForward64(&tz, static_cast<uint64_t>(v));
115#else
116 _BitScanForward(&tz, static_cast<uint32_t>(v));
117 if (tz == 0) {
118 _BitScanForward(&tz,
119 static_cast<uint32_t>(static_cast<uint64_t>(v) >> 32));
120 tz += 32;
121 }
122#endif
123 }
124 return static_cast<int>(tz);
125#else
126 static_assert(sizeof(T) <= sizeof(unsigned long long), "type too big");
127 if (sizeof(T) <= sizeof(unsigned int)) {
128 return __builtin_ctz(static_cast<unsigned int>(v));
129 } else if (sizeof(T) <= sizeof(unsigned long)) {
130 return __builtin_ctzl(static_cast<unsigned long>(v));
131 } else {
132 return __builtin_ctzll(static_cast<unsigned long long>(v));
133 }
134#endif
135}

144template <typename T>
145int BitsSetToOneFallback(T v) {
146 static_assert(std::is_integral_v<T>, "non-integral type");
147 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
148
149 const int kBits = static_cast<int>(sizeof(T)) * 8;
150 static_assert((kBits & (kBits - 1)) == 0, "must be power of two bits");
151 // we static_cast these bit patterns in order to truncate them to the correct
152 // size. Warning C4309 dislikes this technique, so disable it here.
153#pragma warning(disable : 4309)
154 v = static_cast<T>(v - ((v >> 1) & static_cast<T>(0x5555555555555555ull)));
155 v = static_cast<T>((v & static_cast<T>(0x3333333333333333ull)) +
156 ((v >> 2) & static_cast<T>(0x3333333333333333ull)));
157 v = static_cast<T>((v + (v >> 4)) & static_cast<T>(0x0F0F0F0F0F0F0F0Full));
158#pragma warning(default : 4309)
159 for (int shift_bits = 8; shift_bits < kBits; shift_bits <<= 1) {
160 v += static_cast<T>(v >> shift_bits);
161 }
162 // we want the bottom "slot" that's big enough to represent a value up to
163 // (and including) kBits.
164 return static_cast<int>(v & static_cast<T>(kBits | (kBits - 1)));
165}

169#endif // _MSC_VER

171// Number of bits set to 1. Also known as "population count".
172template <typename T>
173inline int BitsSetToOne(T v) {
174 static_assert(std::is_integral_v<T>, "non-integral type");
175 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
176
177#ifdef _MSC_VER
178 static_assert(sizeof(T) <= sizeof(uint64_t), "type too big");
179 if (sizeof(T) < sizeof(uint32_t)) {
180 // This bit mask is to avoid a compiler warning on unused path
181 constexpr auto mm = 8 * sizeof(uint32_t) - 1;
182 // The bit mask is to neutralize sign extension on small signed types
183 constexpr uint32_t m = (uint32_t{1} << ((8 * sizeof(T)) & mm)) - 1;
184#if __POPCNT__
185 return static_cast<int>(__popcnt(static_cast<uint32_t>(v) & m));
186#else
187 return static_cast<int>(detail::BitsSetToOneFallback(v) & m);
188#endif // __POPCNT__
189 } else if (sizeof(T) == sizeof(uint32_t)) {
190#if __POPCNT__
191 return static_cast<int>(__popcnt(static_cast<uint32_t>(v)));
192#else
193 return detail::BitsSetToOneFallback(static_cast<uint32_t>(v));
194#endif // __POPCNT__
195 } else {
196#if __POPCNT__
197#ifdef _M_X64
198 return static_cast<int>(__popcnt64(static_cast<uint64_t>(v)));
199#else
200 return static_cast<int>(
201 __popcnt(static_cast<uint32_t>(static_cast<uint64_t>(v) >> 32) +
202 __popcnt(static_cast<uint32_t>(v))));
203#endif // _M_X64
204#else
205 return detail::BitsSetToOneFallback(static_cast<uint64_t>(v));
206#endif // __POPCNT__
207 }
208#else
209 static_assert(sizeof(T) <= sizeof(unsigned long long), "type too big");
210 if (sizeof(T) < sizeof(unsigned int)) {
211 // This bit mask is to avoid a compiler warning on unused path
212 constexpr auto mm = 8 * sizeof(unsigned int) - 1;
213 // This bit mask is to neutralize sign extension on small signed types
214 constexpr unsigned int m = (1U << ((8 * sizeof(T)) & mm)) - 1;
215 return __builtin_popcount(static_cast<unsigned int>(v) & m);
216 } else if (sizeof(T) == sizeof(unsigned int)) {
217 return __builtin_popcount(static_cast<unsigned int>(v));
218 } else if (sizeof(T) <= sizeof(unsigned long)) {
219 return __builtin_popcountl(static_cast<unsigned long>(v));
220 } else {
221 return __builtin_popcountll(static_cast<unsigned long long>(v));
222 }
223#endif
224}

226template <typename T>
227inline int BitParity(T v) {
228 static_assert(std::is_integral_v<T>, "non-integral type");
229 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
230
231#ifdef _MSC_VER
232 // bit parity == oddness of popcount
233 return BitsSetToOne(v) & 1;
234#else
235 static_assert(sizeof(T) <= sizeof(unsigned long long), "type too big");
236 if (sizeof(T) <= sizeof(unsigned int)) {
237 // On any sane systen, potential sign extension here won't change parity
238 return __builtin_parity(static_cast<unsigned int>(v));
239 } else if (sizeof(T) <= sizeof(unsigned long)) {
240 return __builtin_parityl(static_cast<unsigned long>(v));
241 } else {
242 return __builtin_parityll(static_cast<unsigned long long>(v));
243 }
244#endif
245}

247// Swaps between big and little endian. Can be used in combination with the
248// little-endian encoding/decoding functions in coding_lean.h and coding.h to
249// encode/decode big endian.
250template <typename T>
251inline T EndianSwapValue(T v) {
252 static_assert(std::is_integral_v<T>, "non-integral type");
253 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
254
255#ifdef _MSC_VER
256 if (sizeof(T) == 2) {
257 return static_cast<T>(_byteswap_ushort(static_cast<uint16_t>(v)));
258 } else if (sizeof(T) == 4) {
259 return static_cast<T>(_byteswap_ulong(static_cast<uint32_t>(v)));
260 } else if (sizeof(T) == 8) {
261 return static_cast<T>(_byteswap_uint64(static_cast<uint64_t>(v)));
262 }
263#else
264 if (sizeof(T) == 2) {
265 return static_cast<T>(__builtin_bswap16(static_cast<uint16_t>(v)));
266 } else if (sizeof(T) == 4) {
267 return static_cast<T>(__builtin_bswap32(static_cast<uint32_t>(v)));
268 } else if (sizeof(T) == 8) {
269 return static_cast<T>(__builtin_bswap64(static_cast<uint64_t>(v)));
270 }
271#endif
272 // Recognized by clang as bswap, but not by gcc :(
273 T ret_val = 0;
274 for (std::size_t i = 0; i < sizeof(T); ++i) {
275 ret_val |= ((v >> (8 * i)) & 0xff) << (8 * (sizeof(T) - 1 - i));
276 }
277 return ret_val;
278}

280// Reverses the order of bits in an integral value
281template <typename T>
282inline T ReverseBits(T v) {
283 static_assert(std::is_integral_v<T>, "non-integral type");
284 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
285
286 T r = EndianSwapValue(v);
287 const T kHighestByte = T{1} << ((sizeof(T) - 1) * 8);
288 const T kEveryByte = kHighestByte | (kHighestByte / 255);
289
290 r = ((r & (kEveryByte * 0x0f)) << 4) | ((r >> 4) & (kEveryByte * 0x0f));
291 r = ((r & (kEveryByte * 0x33)) << 2) | ((r >> 2) & (kEveryByte * 0x33));
292 r = ((r & (kEveryByte * 0x55)) << 1) | ((r >> 1) & (kEveryByte * 0x55));
293
294 return r;
295}

297// Every output bit depends on many input bits in the same and higher
298// positions, but not lower positions. Specifically, this function
299// * Output highest bit set to 1 is same as input (same FloorLog2, or
300// equivalently, same number of leading zeros)
301// * Is its own inverse (an involution)
302// * Guarantees that b bottom bits of v and c bottom bits of
303// DownwardInvolution(v) uniquely identify b + c bottom bits of v
304// (which is all of v if v < 2**(b + c)).
305// ** A notable special case is that modifying c adjacent bits at
306// some chosen position in the input is bijective with the bottom c
307// output bits.
308// * Distributes over xor, as in DI(a ^ b) == DI(a) ^ DI(b)
309//
310// This transformation is equivalent to a matrix*vector multiplication in
311// GF(2) where the matrix is recursively defined by the pattern matrix
312// P = | 1 1 |
313// | 0 1 |
314// and replacing 1's with P and 0's with 2x2 zero matices to some depth,
315// e.g. depth of 6 for 64-bit T. An essential feature of this matrix
316// is that all square sub-matrices that include the top row are invertible.
317template <typename T>
318inline T DownwardInvolution(T v) {
319 static_assert(std::is_integral_v<T>, "non-integral type");
320 static_assert(!std::is_reference_v<T>, "use std::remove_reference_t");
321 static_assert(sizeof(T) <= 8, "only supported up to 64 bits");
322
323 uint64_t r = static_cast<uint64_t>(v);
324 if constexpr (sizeof(T) > 4) {
325 r ^= r >> 32;
326 }
327 if constexpr (sizeof(T) > 2) {
328 r ^= (r & 0xffff0000ffff0000U) >> 16;
329 }
330 if constexpr (sizeof(T) > 1) {
331 r ^= (r & 0xff00ff00ff00ff00U) >> 8;
332 }
333 r ^= (r & 0xf0f0f0f0f0f0f0f0U) >> 4;
334 r ^= (r & 0xccccccccccccccccU) >> 2;
335 r ^= (r & 0xaaaaaaaaaaaaaaaaU) >> 1;
336 return static_cast<T>(r);
337}

339// Bitwise-And with typing that allows you to avoid writing an explicit cast
340// to the smaller type, or the type of the right parameter if same size.
341template <typename A, typename B>
342inline std::conditional_t<sizeof(A) < sizeof(B), A, B> BitwiseAnd(A a, B b) {
343 static_assert(std::is_integral_v<A>, "non-integral type");
344 static_assert(std::is_integral_v<B>, "non-integral type");
345 static_assert(!std::is_reference_v<A>, "use std::remove_reference_t");
346 static_assert(!std::is_reference_v<B>, "use std::remove_reference_t");
347 using Smaller = std::conditional_t<sizeof(A) < sizeof(B), A, B>;
348 return static_cast<Smaller>(a & b);
349}
