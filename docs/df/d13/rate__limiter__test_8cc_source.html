264TEST_F(RateLimiterTest, Rate) {
265 auto* env = Env::Default();
266 struct Arg {
267 Arg(int32_t _target_rate, int _burst)
268 : limiter(NewGenericRateLimiter(_target_rate /* rate_bytes_per_sec */,
269 100 * 1000 /* refill_period_us */,
270 10 /* fairness */)),
271 request_size(_target_rate /
272 10 /* refill period here is 1/10 second */),
273 burst(_burst) {}
274 std::unique_ptr<RateLimiter> limiter;
275 int32_t request_size;
276 int burst;
277 };
278
279 auto writer = [](void* p) {
280 const auto& thread_clock = SystemClock::Default();
281 auto* arg = static_cast<Arg*>(p);
282 // Test for 2 seconds
283 auto until = thread_clock->NowMicros() + 2 * 1000000;
284 Random r((uint32_t)(thread_clock->NowNanos() %
285 std::numeric_limits<uint32_t>::max()));
286 while (thread_clock->NowMicros() < until) {
287 for (int i = 0; i < static_cast<int>(r.Skewed(arg->burst * 2) + 1); ++i) {
288 arg->limiter->Request(r.Uniform(arg->request_size - 1) + 1,
289 Env::IO_USER, nullptr /* stats */,
290 RateLimiter::OpType::kWrite);
291 }
292
293 for (int i = 0; i < static_cast<int>(r.Skewed(arg->burst) + 1); ++i) {
294 arg->limiter->Request(r.Uniform(arg->request_size - 1) + 1,
295 Env::IO_HIGH, nullptr /* stats */,
296 RateLimiter::OpType::kWrite);
297 }
298
299 for (int i = 0; i < static_cast<int>(r.Skewed(arg->burst / 2 + 1) + 1);
300 ++i) {
301 arg->limiter->Request(r.Uniform(arg->request_size - 1) + 1, Env::IO_MID,
302 nullptr /* stats */, RateLimiter::OpType::kWrite);
303 }
304
305 arg->limiter->Request(r.Uniform(arg->request_size - 1) + 1, Env::IO_LOW,
306 nullptr /* stats */, RateLimiter::OpType::kWrite);
307 }
308 };
309
310 int samples = 0;
311 int samples_at_minimum = 0;
312
313 for (int i = 1; i <= 16; i *= 2) {
314 int32_t target = i * 1024 * 10;
315 Arg arg(target, i / 4 + 1);
316 int64_t old_total_bytes_through = 0;
317 for (int iter = 1; iter <= 2; ++iter) {
318 // second iteration changes the target dynamically
319 if (iter == 2) {
320 target *= 2;
321 arg.limiter->SetBytesPerSecond(target);
322 }
323 auto start = env->NowMicros();
324 for (int t = 0; t < i; ++t) {
325 env->StartThread(writer, &arg);
326 }
327 env->WaitForJoin();
328
329 auto elapsed = env->NowMicros() - start;
330 double rate =
331 (arg.limiter->GetTotalBytesThrough() - old_total_bytes_through) *
332 1000000.0 / elapsed;
333 old_total_bytes_through = arg.limiter->GetTotalBytesThrough();
334 fprintf(stderr,
335 "request size [1 - %" PRIi32 "], limit %" PRIi32
336 " KB/sec, actual rate: %lf KB/sec, elapsed %.2lf seconds\n",
337 arg.request_size - 1, target / 1024, rate / 1024,
338 elapsed / 1000000.0);
339
340 ++samples;
341 if (rate / target >= 0.80) {
342 ++samples_at_minimum;
343 }
344 ASSERT_LE(rate / target, 1.25);
345 }
346 }
347
348 // This can fail due to slow execution speed, like when using valgrind or in
349 // heavily loaded CI environments
350 bool skip_minimum_rate_check =
351#if defined(ROCKSDB_VALGRIND_RUN)
352 true;
353#elif defined(OS_MACOSX)
354 getenv("CIRCLECI") || getenv("GITHUB_ACTIONS");
355#else
356 getenv("SANDCASTLE");
357#endif
358 if (skip_minimum_rate_check) {
359 fprintf(stderr, "Skipped minimum rate check (%d / %d passed)\n",
360 samples_at_minimum, samples);
361 } else {
362 ASSERT_EQ(samples_at_minimum, samples);
363 }
364}

366TEST_F(RateLimiterTest, LimitChangeTest) {
367 // starvation test when limit changes to a smaller value
368 int64_t refill_period = 1000 * 1000;
369 auto* env = Env::Default();
370 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
371 struct Arg {
372 Arg(int32_t _request_size, Env::IOPriority _pri,
373 std::shared_ptr<RateLimiter> _limiter)
374 : request_size(_request_size), pri(_pri), limiter(_limiter) {}
375 int32_t request_size;
376 Env::IOPriority pri;
377 std::shared_ptr<RateLimiter> limiter;
378 };
379
380 auto writer = [](void* p) {
381 auto* arg = static_cast<Arg*>(p);
382 arg->limiter->Request(arg->request_size, arg->pri, nullptr /* stats */,
383 RateLimiter::OpType::kWrite);
384 };
385
386 for (uint32_t i = 1; i <= 16; i <<= 1) {
387 int32_t target = i * 1024 * 10;
388 // refill per second
389 for (int iter = 0; iter < 2; iter++) {
390 std::shared_ptr<RateLimiter> limiter =
391 std::make_shared<GenericRateLimiter>(
392 target, refill_period, 10, RateLimiter::Mode::kWritesOnly,
393 SystemClock::Default(), false /* auto_tuned */,
394 0 /* single_burst_bytes */);
395 // After "GenericRateLimiter::Request:1" the mutex is held until the bytes
396 // are refilled. This test could be improved to change the limit when lock
397 // is released in `TimedWait()`.
398 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->LoadDependency(
399 {{"GenericRateLimiter::Request",
400 "RateLimiterTest::LimitChangeTest:changeLimitStart"},
401 {"RateLimiterTest::LimitChangeTest:changeLimitEnd",
402 "GenericRateLimiter::Request:1"}});
403 Arg arg(target, Env::IO_HIGH, limiter);
404 // The idea behind is to start a request first, then before it refills,
405 // update limit to a different value (2X/0.5X). No starvation should
406 // be guaranteed under any situation
407 // TODO(lightmark): more test cases are welcome.
408 env->StartThread(writer, &arg);
409 int32_t new_limit = (target << 1) >> (iter << 1);
410 TEST_SYNC_POINT("RateLimiterTest::LimitChangeTest:changeLimitStart");
411 arg.limiter->SetBytesPerSecond(new_limit);
412 TEST_SYNC_POINT("RateLimiterTest::LimitChangeTest:changeLimitEnd");
413 env->WaitForJoin();
414 fprintf(stderr,
415 "[COMPLETE] request size %" PRIi32 " KB, new limit %" PRIi32
416 "KB/sec, refill period %" PRIi64 " ms\n",
417 target / 1024, new_limit / 1024, refill_period / 1000);
418 }
419 }
420 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
421}

423TEST_F(RateLimiterTest, AvailableByteSizeExhaustTest) {
424 SpecialEnv special_env(Env::Default(), /*time_elapse_only_sleep*/ true);
425 const std::chrono::seconds kTimePerRefill(1);
426
427 // This test makes sure available_bytes_ get exhausted first before queuing
428 // any remaining bytes when requested_bytes > available_bytes
429 const int64_t available_bytes_per_period = 500;
430
431 std::shared_ptr<RateLimiter> limiter = std::make_shared<GenericRateLimiter>(

436
437 // Step 1. Request 100 and wait for the refill
438 // so that the remaining available bytes are 400
439 limiter->Request(100, Env::IO_USER, nullptr /* stats */,
440 RateLimiter::OpType::kWrite);
441 special_env.SleepForMicroseconds(
442 static_cast<int>(std::chrono::microseconds(kTimePerRefill).count()));
443

445 "GenericRateLimiter::Request:PostEnqueueRequest", [&](void* arg) {
446 port::Mutex* request_mutex = (port::Mutex*)arg;
447 request_mutex->Unlock();
448 // Step 3. Check GetTotalBytesThrough = available_bytes_per_period
449 // to make sure that the first request (100) and the part of the second
450 // request (400) made through when the remaining of the second request
451 // got queued
452 ASSERT_EQ(available_bytes_per_period,
453 limiter->GetTotalBytesThrough(Env::IO_USER));
454 request_mutex->Lock();
455 });
456 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
457
458 // Step 2. Request 500, which is greater than the remaining available bytes
459 // (400)
460 limiter->Request(500, Env::IO_USER, nullptr /* stats */,
461 RateLimiter::OpType::kWrite);
462
463 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
464 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->ClearCallBack(
465 "GenericRateLimiter::Request:PostEnqueueRequest");

468TEST_F(RateLimiterTest, AutoTuneIncreaseWhenFull) {
469 const std::chrono::seconds kTimePerRefill(1);
470 const int kRefillsPerTune = 100; // needs to match util/rate_limiter.cc
471
472 auto mock_clock =
473 std::make_shared<MockSystemClock>(Env::Default()->GetSystemClock());
474
475 auto stats = CreateDBStatistics();
476 std::unique_ptr<RateLimiter> rate_limiter(new GenericRateLimiter(
477 1000 /* rate_bytes_per_sec */,
478 std::chrono::microseconds(kTimePerRefill).count(), 10 /* fairness */,
479 RateLimiter::Mode::kWritesOnly, mock_clock, true /* auto_tuned */,
480 0 /* single_burst_bytes */));
481
482 // verify rate limit increases after a sequence of periods where rate limiter
483 // is always drained
484 int64_t orig_bytes_per_sec = rate_limiter->GetSingleBurstBytes();
485 rate_limiter->Request(orig_bytes_per_sec, Env::IO_HIGH, stats.get(),
486 RateLimiter::OpType::kWrite);
487 while (std::chrono::microseconds(mock_clock->NowMicros()) <=
488 kRefillsPerTune * kTimePerRefill) {
489 rate_limiter->Request(orig_bytes_per_sec, Env::IO_HIGH, stats.get(),
490 RateLimiter::OpType::kWrite);
491 }
492 int64_t new_bytes_per_sec = rate_limiter->GetSingleBurstBytes();
493 ASSERT_GT(new_bytes_per_sec, orig_bytes_per_sec);
494
495 // decreases after a sequence of periods where rate limiter is not drained
496 orig_bytes_per_sec = new_bytes_per_sec;
497 mock_clock->SleepForMicroseconds(static_cast<int>(
498 kRefillsPerTune * std::chrono::microseconds(kTimePerRefill).count()));
499 // make a request so tuner can be triggered
500 rate_limiter->Request(1 /* bytes */, Env::IO_HIGH, stats.get(),
501 RateLimiter::OpType::kWrite);
502 new_bytes_per_sec = rate_limiter->GetSingleBurstBytes();
503 ASSERT_LT(new_bytes_per_sec, orig_bytes_per_sec);
504}

506TEST_F(RateLimiterTest, WaitHangingBug) {
507 // At t=0: Threads 0 and 1 request `kBytesPerRefill` bytes at low-pri. One
508 // will be granted immediately and the other will enter `TimedWait()`.
509 //
510 // At t=`kMicrosPerRefill`: Thread 2 requests `kBytesPerRefill` bytes at
511 // low-pri. Thread 2's request enters the queue. To expose the bug scenario,
512 // `SyncPoint`s ensure this happens while the lock is temporarily released in
513 // `TimedWait()`. Before the bug fix, Thread 2's request would then hang in
514 // `Wait()` interminably.
515 const int kBytesPerSecond = 100;
516 const int kMicrosPerSecond = 1000 * 1000;
517 const int kMicrosPerRefill = kMicrosPerSecond;
518 const int kBytesPerRefill =
519 kBytesPerSecond * kMicrosPerRefill / kMicrosPerSecond;
520
521 auto mock_clock =
522 std::make_shared<MockSystemClock>(Env::Default()->GetSystemClock());
523 std::unique_ptr<RateLimiter> limiter(new GenericRateLimiter(
524 kBytesPerSecond, kMicrosPerRefill, 10 /* fairness */,
525 RateLimiter::Mode::kWritesOnly, mock_clock, false /* auto_tuned */,
526 0 /* single_burst_bytes */));
527 std::array<std::thread, 3> request_threads;
528
529 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->LoadDependency(
530 {{"RateLimiterTest::WaitHangingBug:InitialRequestsReady",
531 "MockSystemClock::TimedWait:UnlockedPreSleep"},
532 {"MockSystemClock::TimedWait:UnlockedPostSleep1",
533 "RateLimiterTest::WaitHangingBug:TestThreadRequestBegin"},
534 {"RateLimiterTest::WaitHangingBug:TestThreadRequestEnd",
535 "MockSystemClock::TimedWait:UnlockedPostSleep2"}});
536 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
537
538 for (int i = 0; i < 2; i++) {
539 request_threads[i] = std::thread([&]() {
540 limiter->Request(kBytesPerRefill /* bytes */, Env::IOPriority::IO_LOW,
541 nullptr /* stats */, RateLimiter::OpType::kWrite);
542 });
543 }
544 while (limiter->GetTotalRequests() < 2) {
545 }
546 TEST_SYNC_POINT("RateLimiterTest::WaitHangingBug:InitialRequestsReady");
547
548 TEST_SYNC_POINT("RateLimiterTest::WaitHangingBug:TestThreadRequestBegin");
549 request_threads[2] = std::thread([&]() {

564

567

570
571 const int64_t kRefillsPerBurst = 17;
572 const int64_t kBurstBytes = kRefillBytes * kRefillsPerBurst;
573

575 std::make_shared<MockSystemClock>(Env::Default()->GetSystemClock());
576
577 // Zero as `single_burst_bytes` is a special value meaning the refill size
578 std::unique_ptr<RateLimiter> limiter(new GenericRateLimiter(
579 kRateBytesPerSec, kRefillPeriodMicros, 10 /* fairness */,
580 RateLimiter::Mode::kWritesOnly, mock_clock, false /* auto_tuned */,
581 0 /* single_burst_bytes */));
582 ASSERT_EQ(kRefillBytes, limiter->GetSingleBurstBytes());
583
584 // Dynamically setting to zero should change nothing
585 ASSERT_OK(limiter->SetSingleBurstBytes(0));
586 ASSERT_EQ(kRefillBytes, limiter->GetSingleBurstBytes());
587
588 // Negative values are invalid and should change nothing
589 ASSERT_TRUE(limiter->SetSingleBurstBytes(-1).IsInvalidArgument());
590 ASSERT_EQ(kRefillBytes, limiter->GetSingleBurstBytes());
591
592 // Positive values take effect as the new burst size
593 ASSERT_OK(limiter->SetSingleBurstBytes(kBurstBytes));
594 ASSERT_EQ(kBurstBytes, limiter->GetSingleBurstBytes());
595
596 // Initially the supply is full so a request of size `kBurstBytes` needs
597 // `kRefillsPerBurst - 1` refill periods to elapse.
598 limiter->Request(limiter->GetSingleBurstBytes() /* bytes */,
599 Env::IOPriority::IO_USER, nullptr /* stats */,
600 RateLimiter::OpType::kWrite);
601 ASSERT_EQ((kRefillsPerBurst - 1) * kRefillPeriodMicros,
602 mock_clock->NowMicros());
