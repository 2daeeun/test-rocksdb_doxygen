101 // It is called for a couple of endpoints A and B, such that A < B.
102 // If escalation_barrier_check_func(A, B)==true, then there's a lock
103 // escalation barrier between A and B, and lock escalation is not allowed

150struct TransactionDBOptions {
151 // Specifies the maximum number of keys that can be locked at the same time
152 // per column family.
153 // If the number of locked keys is greater than max_num_locks, transaction
154 // writes (or GetForUpdate) will return an error.
155 // If this value is not positive, no limit will be enforced.
156 int64_t max_num_locks = -1;
157
158 // Stores the number of latest deadlocks to track
159 uint32_t max_num_deadlocks = kInitialMaxDeadlocks;
160
161 // Increasing this value will increase the concurrency by dividing the lock
162 // table (per column family) into more sub-tables, each with their own
163 // separate mutex.
164 size_t num_stripes = 16;
165
166 // If positive, specifies the default wait timeout in milliseconds when
167 // a transaction attempts to lock a key if not specified by
168 // TransactionOptions::lock_timeout.
169 //
170 // If 0, no waiting is done if a lock cannot instantly be acquired.
171 // If negative, there is no timeout. Not using a timeout is not recommended
172 // as it can lead to deadlocks. Currently, there is no deadlock-detection to
173 // recover from a deadlock.
174 int64_t transaction_lock_timeout = 1000; // 1 second
175
176 // If positive, specifies the wait timeout in milliseconds when writing a key
177 // OUTSIDE of a transaction (ie by calling DB::Put(),Merge(),Delete(),Write()
178 // directly).
179 // If 0, no waiting is done if a lock cannot instantly be acquired.
180 // If negative, there is no timeout and will block indefinitely when acquiring
181 // a lock.
182 //
183 // Not using a timeout can lead to deadlocks. Currently, there
184 // is no deadlock-detection to recover from a deadlock. While DB writes
185 // cannot deadlock with other DB writes, they can deadlock with a transaction.
186 // A negative timeout should only be used if all transactions have a small
187 // expiration set.
188 int64_t default_lock_timeout = 1000; // 1 second
189
190 // If set, the TransactionDB will use this implementation of a mutex and
191 // condition variable for all transaction locking instead of the default
192 // mutex/condvar implementation.
193 std::shared_ptr<TransactionDBMutexFactory> custom_mutex_factory;
194
195 // The policy for when to write the data into the DB. The default policy is to
196 // write only the committed data (WRITE_COMMITTED). The data could be written
197 // before the commit phase. The DB then needs to provide the mechanisms to
198 // tell apart committed from uncommitted data.
199 TxnDBWritePolicy write_policy = TxnDBWritePolicy::WRITE_COMMITTED;
200
201 // TODO(myabandeh): remove this option
202 // Note: this is a temporary option as a hot fix in rollback of writeprepared
203 // txns in myrocks. MyRocks uses merge operands for autoinc column id without
204 // however obtaining locks. This breaks the assumption behind the rollback
205 // logic in myrocks. This hack of simply not rolling back merge operands works
206 // for the special way that myrocks uses this operands.
207 bool rollback_merge_operands = false;
208
209 // nullptr means use default lock manager.
210 // Other value means the user provides a custom lock manager.
211 std::shared_ptr<LockManagerHandle> lock_mgr_handle;
212
213 // If true, the TransactionDB implementation might skip concurrency control
214 // unless it is overridden by TransactionOptions or
215 // TransactionDBWriteOptimizations. This can be used in conjunction with
216 // DBOptions::unordered_write when the TransactionDB is used solely for write
217 // ordering rather than concurrency control.
218 bool skip_concurrency_control = false;
219
220 // This option is only valid for write unprepared. If a write batch exceeds
221 // this threshold, then the transaction will implicitly flush the currently
222 // pending writes into the database. A value of 0 or less means no limit.
223 int64_t default_write_batch_flush_threshold = 0;
224
225 // This option is valid only for write-prepared/write-unprepared. Transaction
226 // will rely on this callback to determine if a key should be rolled back
227 // with Delete or SingleDelete when necessary. If the callback returns true,
228 // then SingleDelete should be used. If the callback is not callable or the
229 // callback returns false, then a Delete is used.
230 // The application should ensure thread-safety of this callback.
231 // The callback should not throw because RocksDB is not exception-safe.
232 // The callback may be removed if we allow mixing Delete and SingleDelete in
233 // the future.
234 std::function<bool(TransactionDB* /*db*/,
235 ColumnFamilyHandle* /*column_family*/,
236 const Slice& /*key*/)>
237 rollback_deletion_type_callback;
238
239 // A flag to control for the whole DB whether user-defined timestamp based
240 // validation are enabled when applicable. Only WriteCommittedTxn support
241 // user-defined timestamps so this option only applies in this case.
242 bool enable_udt_validation = true;
243
244 // / \ UNDER CONSTRUCTION
245 // / ! \ UNDER CONSTRUCTION
246 // /-----\ UNDER CONSTRUCTION
247 //
248 // The secondary indices to be maintained. See the SecondaryIndex interface
249 // for more details.
250 std::vector<std::shared_ptr<SecondaryIndex>> secondary_indices;
251
252 private:
253 // 128 entries
254 // Should the default value change, please also update wp_snapshot_cache_bits
255 // in db_stress_gflags.cc
256 size_t wp_snapshot_cache_bits = static_cast<size_t>(7);
257 // 8m entry, 64MB size
258 // Should the default value change, please also update wp_commit_cache_bits
259 // in db_stress_gflags.cc
260 size_t wp_commit_cache_bits = static_cast<size_t>(23);
261
262 // For testing, whether transaction name should be auto-generated or not. This
263 // is useful for write unprepared which requires named transactions.
264 bool autogenerate_name = false;
265
266 friend class WritePreparedTxnDB;
267 friend class WriteUnpreparedTxn;
268 friend class WritePreparedTransactionTestBase;
269 friend class TransactionTestBase;
270 friend class MySQLStyleTransactionTest;
271 friend class StressTest;

274struct TransactionOptions {
275 // Setting set_snapshot=true is the same as calling
276 // Transaction::SetSnapshot().
277 bool set_snapshot = false;
278
279 // Setting to true means that before acquiring locks, this transaction will
280 // check if doing so will cause a deadlock. If so, it will return with
281 // Status::Busy. The user should retry their transaction.
282 bool deadlock_detect = false;
283
284 // If set, it states that the CommitTimeWriteBatch represents the latest state
285 // of the application, has only one sub-batch, i.e., no duplicate keys, and
286 // meant to be used later during recovery. It enables an optimization to
287 // postpone updating the memtable with CommitTimeWriteBatch to only
288 // SwitchMemtable or recovery.
289 // This option does not affect write-committed. Only
290 // write-prepared/write-unprepared transactions will be affected.
291 bool use_only_the_last_commit_time_batch_for_recovery = false;
292
293 // TODO(agiardullo): TransactionDB does not yet support comparators that allow
294 // two non-equal keys to be equivalent. Ie, cmp->Compare(a,b) should only
295 // return 0 if
296 // a.compare(b) returns 0.
297
298 // If positive, specifies the wait timeout in milliseconds when
299 // a transaction attempts to lock a key.
300 //
301 // If 0, no waiting is done if a lock cannot instantly be acquired.
302 // If negative, TransactionDBOptions::transaction_lock_timeout will be used.
303 int64_t lock_timeout = -1;
304
305 // Expiration duration in milliseconds. If non-negative, transactions that
306 // last longer than this many milliseconds will fail to commit. If not set,
307 // a forgotten transaction that is never committed, rolled back, or deleted
308 // will never relinquish any locks it holds. This could prevent keys from
309 // being written by other writers.
310 int64_t expiration = -1;
311
312 // The number of traversals to make during deadlock detection.
313 int64_t deadlock_detect_depth = 50;
314
315 // The maximum number of bytes used for the write batch. 0 means no limit.
316 size_t max_write_batch_size = 0;
317
318 // Skip Concurrency Control. This could be as an optimization if the
319 // application knows that the transaction would not have any conflict with
320 // concurrent transactions. It could also be used during recovery if (i)
321 // application guarantees no conflict between prepared transactions in the WAL
322 // (ii) application guarantees that recovered transactions will be rolled
323 // back/commit before new transactions start.
324 // Default: false
325 bool skip_concurrency_control = false;
326
327 // In pessimistic transaction, if this is true, then you can skip Prepare
328 // before Commit, otherwise, you must Prepare before Commit.
329 bool skip_prepare = true;
330
331 // See TransactionDBOptions::default_write_batch_flush_threshold for
332 // description. If a negative value is specified, then the default value from
333 // TransactionDBOptions is used.
334 int64_t write_batch_flush_threshold = -1;
335
336 // DO NOT USE.
337 // This is only a temporary option dedicated for MyRocks that will soon be
338 // removed.
339 // In normal use cases, meta info like column family's timestamp size is
340 // tracked at the transaction layer, so it's not necessary and even
341 // detrimental to track such info inside the internal WriteBatch because it
342 // may let anti-patterns like bypassing Transaction write APIs and directly
343 // write to its internal `WriteBatch` retrieved like this:
344 // https://github.com/facebook/mysql-5.6/blob/fb-mysql-8.0.32/storage/rocksdb/ha_rocksdb.cc#L4949-L4950
345 // Setting this option to true will keep aforementioned use case continue to
346 // work before it's refactored out.
347 // When this flag is enabled, we also intentionally only track the timestamp
348 // size in APIs that MyRocks currently are using, including Put, Merge, Delete
349 // DeleteRange, SingleDelete.
350 bool write_batch_track_timestamp_size = false;
351
352 // EXPERIMENTAL
353 // Only supports write-committed policy. If set to true, the transaction will
354 // skip memtable write and ingest into the DB directly during Commit(). This
355 // makes Commit() much faster for transactions with many operations.
356 // Transactions with Merge() or PutEntity() is not supported yet.
357 //
358 // Note that the transaction will be ingested as an immutable memtable for
359 // CFs it updates, and the current memtable will be switched to a new one.
360 // So ingesting many transactions in a short period of time may cause stall
361 // due to too many memtables.
362 // Note that the ingestion relies on the transaction's underlying index,
363 // (WriteBatchWithIndex), so updates that are added to the transaction
364 // without indexing (e.g. added directly to the transaction underlying
365 // write batch through Transaction::GetWriteBatch()->GetWriteBatch())
366 // are not supported. They will not be applied to the DB.
367 bool commit_bypass_memtable = false;
368};
