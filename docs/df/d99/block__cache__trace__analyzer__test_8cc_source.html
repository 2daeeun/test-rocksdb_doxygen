77 TableReaderCaller GetCaller(uint32_t key_id) {
78 uint32_t n = key_id % 5;
79 switch (n) {
80 case 0:
81 return TableReaderCaller::kPrefetch;
82 case 1:
83 return TableReaderCaller::kCompaction;
84 case 2:
85 return TableReaderCaller::kUserGet;
86 case 3:
87 return TableReaderCaller::kUserMultiGet;
88 case 4:
89 return TableReaderCaller::kUserIterator;
90 }
91 // This cannot happend.
92 assert(false);
93 return TableReaderCaller::kMaxBlockCacheLookupCaller;
94 }

96 void WriteBlockAccess(BlockCacheTraceWriter* writer, uint32_t from_key_id,
97 TraceType block_type, uint32_t nblocks,
98 bool is_referenced_key_null = false) {
99 assert(writer);
100 for (uint32_t i = 0; i < nblocks; i++) {
101 uint32_t key_id = from_key_id + i;
102 uint64_t timestamp = (key_id + 1) * kMicrosInSecond;
103 BlockCacheTraceRecord record;
104 record.block_type = block_type;
105 record.block_size = kBlockSize + key_id;
106 record.block_key = kBlockKeyPrefix + std::to_string(key_id);
107 record.access_timestamp = timestamp;
108 record.cf_id = kCFId;
109 record.cf_name = kDefaultColumnFamilyName;
110 record.caller = GetCaller(key_id);
111 record.level = kLevel;
112 if (key_id % 2 == 0) {
113 record.sst_fd_number = kSSTStoringEvenKeys;
114 } else {
115 record.sst_fd_number = kSSTStoringOddKeys;
116 }
117 record.is_cache_hit = false;
118 record.no_insert = false;
119 // Provide these fields for all block types.
120 // The writer should only write these fields for data blocks and the
121 // caller is either GET or MGET.
122 record.referenced_key =
123 kRefKeyPrefix + std::to_string(key_id) + std::string(8, 0);
124 record.referenced_key_exist_in_block = true;
125 if (is_referenced_key_null &&
126 record.caller == TableReaderCaller::kUserMultiGet) {
127 record.referenced_key = "";
128 record.get_from_user_specified_snapshot = true;
129 }
130 record.num_keys_in_block = kNumKeysInBlock;
131 ASSERT_OK(writer->WriteBlockAccess(
132 record, record.block_key, record.cf_name, record.referenced_key));
133 }
134 }

136 void AssertBlockAccessInfo(
137 uint32_t key_id, TraceType type,
138 const std::map<std::string, BlockAccessInfo>& block_access_info_map) {
139 auto key_id_str = kBlockKeyPrefix + std::to_string(key_id);
140 ASSERT_TRUE(block_access_info_map.find(key_id_str) !=
141 block_access_info_map.end());
142 auto& block_access_info = block_access_info_map.find(key_id_str)->second;
143 ASSERT_EQ(1, block_access_info.num_accesses);
144 ASSERT_EQ(kBlockSize + key_id, block_access_info.block_size);
145 ASSERT_GT(block_access_info.first_access_time, 0);
146 ASSERT_GT(block_access_info.last_access_time, 0);
147 ASSERT_EQ(1, block_access_info.caller_num_access_map.size());
148 TableReaderCaller expected_caller = GetCaller(key_id);
149 ASSERT_TRUE(block_access_info.caller_num_access_map.find(expected_caller) !=
150 block_access_info.caller_num_access_map.end());
151 ASSERT_EQ(
152 1,
153 block_access_info.caller_num_access_map.find(expected_caller)->second);
154
155 if ((expected_caller == TableReaderCaller::kUserGet ||
156 expected_caller == TableReaderCaller::kUserMultiGet) &&
157 type == TraceType::kBlockTraceDataBlock) {
158 ASSERT_EQ(kNumKeysInBlock, block_access_info.num_keys);
159 ASSERT_EQ(1, block_access_info.key_num_access_map.size());
160 ASSERT_EQ(0, block_access_info.non_exist_key_num_access_map.size());
161 ASSERT_EQ(1, block_access_info.num_referenced_key_exist_in_block);
162 }

166 std::vector<std::string> params = {

168 "-block_cache_trace_path=" + trace_file_path_,
169 "-block_cache_sim_config_path=" + block_cache_sim_config_path_,

174 "-cache_sim_warmup_seconds=0",
175 "-analyze_bottom_k_access_count_blocks=5",
176 "-analyze_top_k_access_count_blocks=5",
177 "-analyze_blocks_reuse_k_reuse_window=5",
178 "-timeline_labels=" + timeline_labels_,
179 "-reuse_distance_labels=" + reuse_distance_labels_,
180 "-reuse_distance_buckets=" + reuse_distance_buckets_,
181 "-reuse_interval_labels=" + reuse_interval_labels_,
182 "-reuse_interval_buckets=" + reuse_interval_buckets_,
183 "-reuse_lifetime_labels=" + reuse_lifetime_labels_,
184 "-reuse_lifetime_buckets=" + reuse_lifetime_buckets_,
185 "-analyze_callers=" + analyzing_callers_,
186 "-access_count_buckets=" + access_count_buckets_,
187 "-analyze_get_spatial_locality_labels=" +
188 analyze_get_spatial_locality_labels_,
189 "-analyze_get_spatial_locality_buckets=" +
190 analyze_get_spatial_locality_buckets_,
191 "-analyze_correlation_coefficients_labels=all",
192 "-skew_labels=all",
193 "-skew_buckets=10,50,100"};
194 char arg_buffer[kArgBufferSize];
195 char* argv[kMaxArgCount];
196 int argc = 0;
197 int cursor = 0;
198 for (const auto& arg : params) {
199 ASSERT_LE(cursor + arg.size() + 1, kArgBufferSize);
200 ASSERT_LE(argc + 1, kMaxArgCount);
201 snprintf(arg_buffer + cursor, arg.size() + 1, "%s", arg.c_str());
202
203 argv[argc++] = arg_buffer + cursor;
204 cursor += static_cast<int>(arg.size()) + 1;
205 }
206 ASSERT_EQ(0,
207 ROCKSDB_NAMESPACE::block_cache_trace_analyzer_tool(argc, argv));
208 }
