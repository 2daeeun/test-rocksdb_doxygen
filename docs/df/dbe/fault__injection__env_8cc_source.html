22// Assume a filename, and not a directory name like "/foo/bar/"
23std::string GetDirName(const std::string filename) {
24 size_t found = filename.find_last_of("/\\");
25 if (found == std::string::npos) {
26 return "";
27 } else {
28 return filename.substr(0, found);
29 }
30}

32// A basic file truncation function suitable for this test.
33Status Truncate(Env* env, const std::string& filename, uint64_t length) {
34 std::unique_ptr<SequentialFile> orig_file;
35 const EnvOptions options;
36 Status s = env->NewSequentialFile(filename, &orig_file, options);
37 if (!s.ok()) {
38 fprintf(stderr, "Cannot open file %s for truncation: %s\n",
39 filename.c_str(), s.ToString().c_str());
40 return s;
41 }
42
43 std::unique_ptr<char[]> scratch(new char[length]);
44 ROCKSDB_NAMESPACE::Slice result;
45 s = orig_file->Read(length, &result, scratch.get());
46#ifdef OS_WIN
47 orig_file.reset();
48#endif
49 if (s.ok()) {
50 std::string tmp_name = GetDirName(filename) + "/truncate.tmp";
51 std::unique_ptr<WritableFile> tmp_file;
52 s = env->NewWritableFile(tmp_name, &tmp_file, options);
53 if (s.ok()) {
54 s = tmp_file->Append(result);
55 if (s.ok()) {
56 s = env->RenameFile(tmp_name, filename);
57 } else {
58 fprintf(stderr, "Cannot rename file %s to %s: %s\n", tmp_name.c_str(),
59 filename.c_str(), s.ToString().c_str());
60 env->DeleteFile(tmp_name);
61 }
62 }
63 }
64 if (!s.ok()) {
65 fprintf(stderr, "Cannot truncate file %s: %s\n", filename.c_str(),
66 s.ToString().c_str());
67 }
68
69 return s;
70}

72// Trim the tailing "/" in the end of `str`
73std::string TrimDirname(const std::string& str) {
74 size_t found = str.find_last_not_of('/');
75 if (found == std::string::npos) {
76 return str;
77 }
78 return str.substr(0, found + 1);
79}

81// Return pair <parent directory name, file name> of a full path.
82std::pair<std::string, std::string> GetDirAndName(const std::string& name) {
83 std::string dirname = GetDirName(name);
84 std::string fname = name.substr(dirname.size() + 1);
85 return std::make_pair(dirname, fname);
86}

88Status FileState::DropUnsyncedData(Env* env) const {
89 ssize_t sync_pos = pos_at_last_sync_ == -1 ? 0 : pos_at_last_sync_;
90 return Truncate(env, filename_, sync_pos);
91}

93Status FileState::DropRandomUnsyncedData(Env* env, Random* rand) const {
94 ssize_t sync_pos = pos_at_last_sync_ == -1 ? 0 : pos_at_last_sync_;
95 assert(pos_ >= sync_pos);
96 int range = static_cast<int>(pos_ - sync_pos);
97 uint64_t truncated_size =
98 static_cast<uint64_t>(sync_pos) + rand->Uniform(range);
99 return Truncate(env, filename_, truncated_size);
100}

277Status FaultInjectionTestEnv::NewWritableFile(
278 const std::string& fname, std::unique_ptr<WritableFile>* result,
279 const EnvOptions& soptions) {
280 if (!IsFilesystemActive()) {
281 return GetError();
282 }
283 // Not allow overwriting files
284 Status s = target()->FileExists(fname);
285 if (s.ok()) {
286 return Status::Corruption("File already exists.");
287 } else if (!s.IsNotFound()) {
288 assert(s.IsIOError());
289 return s;
290 }
291 s = target()->NewWritableFile(fname, result, soptions);
292 if (s.ok()) {
293 result->reset(new TestWritableFile(fname, std::move(*result), this));
294 // WritableFileWriter* file is opened
295 // again then it will be truncated - so forget our saved state.
296 UntrackFile(fname);
297 MutexLock l(&mutex_);
298 open_managed_files_.insert(fname);
299 auto dir_and_name = GetDirAndName(fname);
300 auto& list = dir_to_new_files_since_last_sync_[dir_and_name.first];
301 list.insert(dir_and_name.second);
302 }
303 return s;
304}

306Status FaultInjectionTestEnv::ReopenWritableFile(
307 const std::string& fname, std::unique_ptr<WritableFile>* result,
308 const EnvOptions& soptions) {
309 if (!IsFilesystemActive()) {
310 return GetError();
311 }
312
313 bool exists;
314 Status s, exists_s = target()->FileExists(fname);
315 if (exists_s.IsNotFound()) {
316 exists = false;
317 } else if (exists_s.ok()) {
318 exists = true;
319 } else {
320 s = exists_s;
321 exists = false;
322 }
323
324 if (s.ok()) {
325 s = target()->ReopenWritableFile(fname, result, soptions);
326 }
327
328 // Only track files we created. Files created outside of this
329 // `FaultInjectionTestEnv` are not eligible for tracking/data dropping
330 // (for example, they may contain data a previous db_stress run expects to
331 // be recovered). This could be extended to track/drop data appended once
332 // the file is under `FaultInjectionTestEnv`'s control.
333 if (s.ok()) {
334 bool should_track;
335 {
336 MutexLock l(&mutex_);
337 if (db_file_state_.find(fname) != db_file_state_.end()) {
338 // It was written by this `Env` earlier.
339 assert(exists);
340 should_track = true;
341 } else if (!exists) {
342 // It was created by this `Env` just now.
343 should_track = true;
344 open_managed_files_.insert(fname);
345 auto dir_and_name = GetDirAndName(fname);
346 auto& list = dir_to_new_files_since_last_sync_[dir_and_name.first];
347 list.insert(dir_and_name.second);
348 } else {
349 should_track = false;
350 }
351 }
352 if (should_track) {
353 result->reset(new TestWritableFile(fname, std::move(*result), this));
354 }
355 }
356 return s;
357}

410Status FaultInjectionTestEnv::RenameFile(const std::string& s,
411 const std::string& t) {
412 if (!IsFilesystemActive()) {
413 return GetError();
414 }
415 Status ret = EnvWrapper::RenameFile(s, t);
416
417 if (ret.ok()) {
418 MutexLock l(&mutex_);
419 if (db_file_state_.find(s) != db_file_state_.end()) {
420 db_file_state_[t] = db_file_state_[s];
421 db_file_state_.erase(s);
422 }
423
424 auto sdn = GetDirAndName(s);
425 auto tdn = GetDirAndName(t);
426 if (dir_to_new_files_since_last_sync_[sdn.first].erase(sdn.second) != 0) {
427 auto& tlist = dir_to_new_files_since_last_sync_[tdn.first];
428 assert(tlist.find(tdn.second) == tlist.end());
429 tlist.insert(tdn.second);
430 }
431 }
432
433 return ret;
434}

492// For every file that is not fully synced, make a call to `func` with
493// FileState of the file as the parameter.
494Status FaultInjectionTestEnv::DropFileData(
495 std::function<Status(Env*, FileState)> func) {
496 Status s;
497 MutexLock l(&mutex_);
498 for (std::map<std::string, FileState>::const_iterator it =
499 db_file_state_.begin();
500 s.ok() && it != db_file_state_.end(); ++it) {
501 const FileState& state = it->second;
502 if (!state.IsFullySynced()) {
503 s = func(target(), state);
504 }
505 }
506 return s;
507}

521Status FaultInjectionTestEnv::DeleteFilesCreatedAfterLastDirSync() {
522 // Because DeleteFile access this container make a copy to avoid deadlock
523 std::map<std::string, std::set<std::string>> map_copy;
524 {
525 MutexLock l(&mutex_);
526 map_copy.insert(dir_to_new_files_since_last_sync_.begin(),
527 dir_to_new_files_since_last_sync_.end());
528 }
529
530 for (auto& pair : map_copy) {
531 for (const std::string& name : pair.second) {
532 Status s = DeleteFile(pair.first + "/" + name);
533 if (!s.ok()) {
534 return s;
535 }
536 }
537 }

548 MutexLock l(&mutex_);
