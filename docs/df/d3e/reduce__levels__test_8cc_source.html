20class ReduceLevelTest : public testing::Test {
21 public:
22 ReduceLevelTest() {
23 dbname_ = test::PerThreadDBPath("db_reduce_levels_test");

30 Status Put(const std::string& k, const std::string& v) {
31 return db_->Put(WriteOptions(), k, v);
32 }

34 std::string Get(const std::string& k) {
35 ReadOptions options;
36 std::string result;
37 Status s = db_->Get(options, k, &result);
38 if (s.IsNotFound()) {
39 result = "NOT_FOUND";
40 } else if (!s.ok()) {
41 result = s.ToString();
42 }
43 return result;
44 }
45
46 Status Flush() {
47 if (db_ == nullptr) {
48 return Status::InvalidArgument("DB not opened.");
49 }
50 DBImpl* db_impl = static_cast_with_check<DBImpl>(db_);
51 return db_impl->TEST_FlushMemTable();
52 }
53
54 void MoveL0FileToLevel(int level) {
55 DBImpl* db_impl = static_cast_with_check<DBImpl>(db_);
56 for (int i = 0; i < level; ++i) {
57 ASSERT_OK(db_impl->TEST_CompactRange(i, nullptr, nullptr));
58 }
59 }
60
61 void CloseDB() {
62 if (db_ != nullptr) {
63 delete db_;
64 db_ = nullptr;
65 }
66 }

68 bool ReduceLevels(int target_level);

70 int FilesOnLevel(int level) {
71 std::string property;
72 EXPECT_TRUE(db_->GetProperty(
73 "rocksdb.num-files-at-level" + std::to_string(level), &property));
74 return atoi(property.c_str());
75 }

79 DB* db_;

82Status ReduceLevelTest::OpenDB(bool create_if_missing, int num_levels) {
83 ROCKSDB_NAMESPACE::Options opt;
84 opt.level_compaction_dynamic_level_bytes = false;
85 opt.num_levels = num_levels;
86 opt.create_if_missing = create_if_missing;
87 ROCKSDB_NAMESPACE::Status st =
88 ROCKSDB_NAMESPACE::DB::Open(opt, dbname_, &db_);
89 if (!st.ok()) {
90 fprintf(stderr, "Can't open the db:%s\n", st.ToString().c_str());
91 }
92 return st;

95bool ReduceLevelTest::ReduceLevels(int target_level) {
96 std::vector<std::string> args =
97 ROCKSDB_NAMESPACE::ReduceDBLevelsCommand::PrepareArgs(
98 dbname_, target_level, false);
99 LDBCommand* level_reducer = LDBCommand::InitFromCmdLineArgs(
100 args, Options(), LDBOptions(), nullptr, LDBCommand::SelectCommand);
101 level_reducer->Run();
102 bool is_succeed = level_reducer->GetExecuteState().IsSucceed();
103 delete level_reducer;
104 return is_succeed;
105}

107TEST_F(ReduceLevelTest, Last_Level) {
108 ASSERT_OK(OpenDB(true, 4));
