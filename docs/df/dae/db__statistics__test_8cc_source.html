24TEST_F(DBStatisticsTest, CompressionStatsTest) {
25 for (CompressionType type : GetSupportedCompressions()) {
26 if (type == kNoCompression) {
27 continue;
28 }
29 if (type == kBZip2Compression) {
30 // Weird behavior in this test
31 continue;
32 }
33 SCOPED_TRACE("Compression type: " + std::to_string(type));
34
35 Options options = CurrentOptions();
36 options.compression = type;
37 options.statistics = ROCKSDB_NAMESPACE::CreateDBStatistics();
38 options.statistics->set_stats_level(StatsLevel::kExceptTimeForMutex);
39 BlockBasedTableOptions bbto;
40 bbto.enable_index_compression = false;
41 options.table_factory.reset(NewBlockBasedTableFactory(bbto));
42 DestroyAndReopen(options);
43
44 auto PopStat = [&](Tickers t) -> uint64_t {
45 return options.statistics->getAndResetTickerCount(t);
46 };
47
48 int kNumKeysWritten = 100;
49 double compress_to = 0.5;
50 // About three KVs per block
51 int len = static_cast<int>(BlockBasedTableOptions().block_size / 3);
52 int uncomp_est = kNumKeysWritten * (len + 20);
53
54 Random rnd(301);
55 std::string buf;
56
57 // Check that compressions occur and are counted when compression is turned
58 // on
59 for (int i = 0; i < kNumKeysWritten; ++i) {
60 ASSERT_OK(
61 Put(Key(i), test::CompressibleString(&rnd, compress_to, len, &buf)));
62 }
63 ASSERT_OK(Flush());
64 EXPECT_EQ(34, PopStat(NUMBER_BLOCK_COMPRESSED));
65 EXPECT_NEAR2(uncomp_est, PopStat(BYTES_COMPRESSED_FROM), uncomp_est / 10);
66 EXPECT_NEAR2(uncomp_est * compress_to, PopStat(BYTES_COMPRESSED_TO),
67 uncomp_est / 10);
68
69 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_DECOMPRESSED));
70 EXPECT_EQ(0, PopStat(BYTES_DECOMPRESSED_FROM));
71 EXPECT_EQ(0, PopStat(BYTES_DECOMPRESSED_TO));
72
73 // And decompressions
74 for (int i = 0; i < kNumKeysWritten; ++i) {
75 auto r = Get(Key(i));
76 }
77 EXPECT_EQ(34, PopStat(NUMBER_BLOCK_DECOMPRESSED));
78 EXPECT_NEAR2(uncomp_est, PopStat(BYTES_DECOMPRESSED_TO), uncomp_est / 10);
79 EXPECT_NEAR2(uncomp_est * compress_to, PopStat(BYTES_DECOMPRESSED_FROM),
80 uncomp_est / 10);
81
82 EXPECT_EQ(0, PopStat(BYTES_COMPRESSION_BYPASSED));
83 EXPECT_EQ(0, PopStat(BYTES_COMPRESSION_REJECTED));
84 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_COMPRESSION_BYPASSED));
85 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_COMPRESSION_REJECTED));
86
87 // Check when compression is rejected.
88 DestroyAndReopen(options);
89
90 for (int i = 0; i < kNumKeysWritten; ++i) {
91 ASSERT_OK(Put(Key(i), rnd.RandomBinaryString(len)));
92 }
93 ASSERT_OK(Flush());
94 for (int i = 0; i < kNumKeysWritten; ++i) {
95 auto r = Get(Key(i));
96 }
97 EXPECT_EQ(34, PopStat(NUMBER_BLOCK_COMPRESSION_REJECTED));
98 EXPECT_NEAR2(uncomp_est, PopStat(BYTES_COMPRESSION_REJECTED),
99 uncomp_est / 10);
100
101 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_COMPRESSED));
102 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_COMPRESSION_BYPASSED));
103 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_DECOMPRESSED));
104 EXPECT_EQ(0, PopStat(BYTES_COMPRESSED_FROM));
105 EXPECT_EQ(0, PopStat(BYTES_COMPRESSED_TO));
106 EXPECT_EQ(0, PopStat(BYTES_COMPRESSION_BYPASSED));
107 EXPECT_EQ(0, PopStat(BYTES_DECOMPRESSED_FROM));
108 EXPECT_EQ(0, PopStat(BYTES_DECOMPRESSED_TO));
109
110 // Check when compression is disabled.
111 options.compression = kNoCompression;
112 DestroyAndReopen(options);
113
114 for (int i = 0; i < kNumKeysWritten; ++i) {
115 ASSERT_OK(Put(Key(i), rnd.RandomBinaryString(len)));
116 }
117 ASSERT_OK(Flush());
118 for (int i = 0; i < kNumKeysWritten; ++i) {
119 auto r = Get(Key(i));
120 }
121 EXPECT_EQ(34, PopStat(NUMBER_BLOCK_COMPRESSION_BYPASSED));
122 EXPECT_NEAR2(uncomp_est, PopStat(BYTES_COMPRESSION_BYPASSED),
123 uncomp_est / 10);
124
125 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_COMPRESSED));
126 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_COMPRESSION_REJECTED));
127 EXPECT_EQ(0, PopStat(NUMBER_BLOCK_DECOMPRESSED));
128 EXPECT_EQ(0, PopStat(BYTES_COMPRESSED_FROM));
129 EXPECT_EQ(0, PopStat(BYTES_COMPRESSED_TO));
130 EXPECT_EQ(0, PopStat(BYTES_COMPRESSION_REJECTED));
131 EXPECT_EQ(0, PopStat(BYTES_DECOMPRESSED_FROM));
132 EXPECT_EQ(0, PopStat(BYTES_DECOMPRESSED_TO));
133 }

202TEST_F(DBStatisticsTest, VerifyChecksumReadStat) {
203 Options options = CurrentOptions();
204 options.file_checksum_gen_factory = GetFileChecksumGenCrc32cFactory();
205 options.statistics = ROCKSDB_NAMESPACE::CreateDBStatistics();
206 Reopen(options);
207
208 // Expected to be populated regardless of `PerfLevel` in user thread
209 SetPerfLevel(kDisable);
210
211 {
212 // Scenario 0: only WAL data. Not verified so require ticker to be zero.
213 ASSERT_OK(Put("foo", "value"));
214 ASSERT_OK(db_->VerifyFileChecksums(ReadOptions()));
215 ASSERT_OK(db_->VerifyChecksum());
216 ASSERT_EQ(0,
217 options.statistics->getTickerCount(VERIFY_CHECKSUM_READ_BYTES));
218 }
219
220 // Create one SST.
221 ASSERT_OK(Flush());
222 std::unordered_map<std::string, uint64_t> table_files;
223 uint64_t table_files_size = 0;
224 ASSERT_OK(GetAllDataFiles(kTableFile, &table_files, &table_files_size));
225
226 {
227 // Scenario 1: Table verified in `VerifyFileChecksums()`. This should read
228 // the whole file so we require the ticker stat exactly matches the file
229 // size.
230 ASSERT_OK(options.statistics->Reset());
231 ASSERT_OK(db_->VerifyFileChecksums(ReadOptions()));
232 ASSERT_EQ(table_files_size,
233 options.statistics->getTickerCount(VERIFY_CHECKSUM_READ_BYTES));
234 }
235
236 {
237 // Scenario 2: Table verified in `VerifyChecksum()`. This opens a
238 // `TableReader` to verify each block. It can involve duplicate reads of the
239 // same data so we set a lower-bound only.
240 ASSERT_OK(options.statistics->Reset());
241 ASSERT_OK(db_->VerifyChecksum());
242 ASSERT_GE(options.statistics->getTickerCount(VERIFY_CHECKSUM_READ_BYTES),
243 table_files_size);
244 }

247TEST_F(DBStatisticsTest, BlockChecksumStats) {
248 Options options = CurrentOptions();
249 options.statistics = ROCKSDB_NAMESPACE::CreateDBStatistics();
250 Reopen(options);
251
252 // Scenario 0: only WAL data. Not verified so require ticker to be zero.
253 ASSERT_OK(Put("foo", "value"));
254 ASSERT_OK(db_->VerifyChecksum());
255 ASSERT_EQ(0,
256 options.statistics->getTickerCount(BLOCK_CHECKSUM_COMPUTE_COUNT));
257 ASSERT_EQ(0,
258 options.statistics->getTickerCount(BLOCK_CHECKSUM_MISMATCH_COUNT));
259
260 // Scenario 1: Flushed table verified in `VerifyChecksum()`. This opens a
261 // `TableReader` to verify each of the four blocks (meta-index, table
262 // properties, index, and data block).
263 ASSERT_OK(Flush());
264 ASSERT_OK(options.statistics->Reset());
265 ASSERT_OK(db_->VerifyChecksum());
266 ASSERT_EQ(4,
267 options.statistics->getTickerCount(BLOCK_CHECKSUM_COMPUTE_COUNT));
268 ASSERT_EQ(0,
269 options.statistics->getTickerCount(BLOCK_CHECKSUM_MISMATCH_COUNT));
270
271 // Scenario 2: Corrupted table verified in `VerifyChecksum()`. The corruption
272 // is in the fourth and final verified block, i.e., the data block.
273 std::unordered_map<std::string, uint64_t> table_files;
274 ASSERT_OK(GetAllDataFiles(kTableFile, &table_files));
275 ASSERT_EQ(1, table_files.size());
276 std::string table_name = table_files.begin()->first;
277 // Assumes the data block starts at offset zero.
278 ASSERT_OK(test::CorruptFile(options.env, table_name, 0 /* offset */,
279 3 /* bytes_to_corrupt */));
280 ASSERT_OK(options.statistics->Reset());
281 ASSERT_NOK(db_->VerifyChecksum());
282 ASSERT_EQ(4,
283 options.statistics->getTickerCount(BLOCK_CHECKSUM_COMPUTE_COUNT));
284 ASSERT_EQ(1,
285 options.statistics->getTickerCount(BLOCK_CHECKSUM_MISMATCH_COUNT));
286}

288TEST_F(DBStatisticsTest, BytesWrittenStats) {
289 Options options = CurrentOptions();
290 options.statistics = ROCKSDB_NAMESPACE::CreateDBStatistics();
291 options.statistics->set_stats_level(StatsLevel::kExceptHistogramOrTimers);
292 Reopen(options);
293
294 EXPECT_EQ(0, options.statistics->getAndResetTickerCount(WAL_FILE_BYTES));
295 EXPECT_EQ(0, options.statistics->getAndResetTickerCount(BYTES_WRITTEN));
296
297 const int kNumKeysWritten = 100;
298
299 // Scenario 0: Not using transactions.
300 // This will write to WAL and memtable directly.
301 ASSERT_OK(options.statistics->Reset());
302
303 for (int i = 0; i < kNumKeysWritten; ++i) {
304 ASSERT_OK(Put(Key(i), "val"));
305 }
306
307 EXPECT_EQ(options.statistics->getAndResetTickerCount(WAL_FILE_BYTES),
308 options.statistics->getAndResetTickerCount(BYTES_WRITTEN));
309
310 // Scenario 1: Using transactions.
311 // This should not double count BYTES_WRITTEN (issue #12061).
312 for (bool enable_pipelined_write : {false, true}) {
313 ASSERT_OK(options.statistics->Reset());
314
315 // Destroy the DB to recreate as a TransactionDB.
316 Destroy(options, true);
317
318 // Create a TransactionDB.
319 TransactionDB* txn_db = nullptr;
320 TransactionDBOptions txn_db_opts;
321 txn_db_opts.write_policy = TxnDBWritePolicy::WRITE_COMMITTED;
322 options.enable_pipelined_write = enable_pipelined_write;
323 ASSERT_OK(TransactionDB::Open(options, txn_db_opts, dbname_, &txn_db));
324 ASSERT_NE(txn_db, nullptr);
325 db_ = txn_db->GetBaseDB();
326
327 WriteOptions wopts;
328 TransactionOptions txn_opts;
329 Transaction* txn = txn_db->BeginTransaction(wopts, txn_opts, nullptr);
330 ASSERT_NE(txn, nullptr);
331 ASSERT_OK(txn->SetName("txn1"));
332
333 for (int i = 0; i < kNumKeysWritten; ++i) {
334 ASSERT_OK(txn->Put(Key(i), "val"));
335 }
336
337 // Prepare() writes to WAL, but not to memtable. (WriteCommitted)
338 ASSERT_OK(txn->Prepare());
339 EXPECT_NE(0, options.statistics->getTickerCount(WAL_FILE_BYTES));
340 // BYTES_WRITTEN would have been non-zero previously (issue #12061).
341 EXPECT_EQ(0, options.statistics->getTickerCount(BYTES_WRITTEN));
342
343 // Commit() writes to memtable and also a commit marker to WAL.
344 ASSERT_OK(txn->Commit());
345 delete txn;
346
347 // The WAL has an extra header of size `kHeader` written to it,
348 // as we are writing twice to it (first during Prepare, second during
349 // Commit).
350 EXPECT_EQ(options.statistics->getAndResetTickerCount(WAL_FILE_BYTES),
351 options.statistics->getAndResetTickerCount(BYTES_WRITTEN) +
352 WriteBatchInternal::kHeader);
353
354 // Cleanup
355 db_ = nullptr;
356 delete txn_db;
357 }
