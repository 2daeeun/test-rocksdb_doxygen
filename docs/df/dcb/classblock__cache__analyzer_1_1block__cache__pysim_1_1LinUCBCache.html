1082 p = np.zeros(len(self.policies))

1085 ta = x_i.dot(self.A_inv[a]).dot(x_i)
1086 a_upper_ci = self.alph * np.sqrt(ta)
1087 a_mean = self.th_hat[a].dot(x_i)
1088 p[a] = a_mean + a_upper_ci
