608TEST_F(DBBlobCompactionTest, TrackGarbage) {
609 Options options = GetDefaultOptions();
610 options.enable_blob_files = true;
611
612 Reopen(options);
613
614 // First table+blob file pair: 4 blobs with different keys
615 constexpr char first_key[] = "first_key";
616 constexpr char first_value[] = "first_value";
617 constexpr char second_key[] = "second_key";
618 constexpr char second_value[] = "second_value";
619 constexpr char third_key[] = "third_key";
620 constexpr char third_value[] = "third_value";
621 constexpr char fourth_key[] = "fourth_key";
622 constexpr char fourth_value[] = "fourth_value";
623
624 ASSERT_OK(Put(first_key, first_value));
625 ASSERT_OK(Put(second_key, second_value));
626 ASSERT_OK(Put(third_key, third_value));
627 ASSERT_OK(Put(fourth_key, fourth_value));
628 ASSERT_OK(Flush());
629
630 // Second table+blob file pair: overwrite 2 existing keys
631 constexpr char new_first_value[] = "new_first_value";
632 constexpr char new_second_value[] = "new_second_value";
633
634 ASSERT_OK(Put(first_key, new_first_value));
635 ASSERT_OK(Put(second_key, new_second_value));
636 ASSERT_OK(Flush());
637
638 // Compact them together. The first blob file should have 2 garbage blobs
639 // corresponding to the 2 overwritten keys.
640 constexpr Slice* begin = nullptr;
641 constexpr Slice* end = nullptr;
642
643 ASSERT_OK(db_->CompactRange(CompactRangeOptions(), begin, end));
644
645 VersionSet* const versions = dbfull()->GetVersionSet();
646 assert(versions);
647 assert(versions->GetColumnFamilySet());
648
649 ColumnFamilyData* const cfd = versions->GetColumnFamilySet()->GetDefault();
650 assert(cfd);
651
652 Version* const current = cfd->current();
653 assert(current);
654
655 const VersionStorageInfo* const storage_info = current->storage_info();
656 assert(storage_info);
657
658 const auto& blob_files = storage_info->GetBlobFiles();
659 ASSERT_EQ(blob_files.size(), 2);
660
661 {
662 const auto& meta = blob_files.front();
663 assert(meta);
664
665 constexpr uint64_t first_expected_bytes =
666 sizeof(first_value) - 1 +
