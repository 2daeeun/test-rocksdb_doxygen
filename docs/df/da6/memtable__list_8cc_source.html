82 assert(refs_ >= 1);
83 --refs_;
84 if (refs_ == 0) {
85 // if to_delete is equal to nullptr it means we're confident
86 // that refs_ will not be zero

158bool MemTableListVersion::GetFromList(
159 std::list<ReadOnlyMemTable*>* list, const LookupKey& key,
160 std::string* value, PinnableWideColumns* columns, std::string* timestamp,
161 Status* s, MergeContext* merge_context,
162 SequenceNumber* max_covering_tombstone_seq, SequenceNumber* seq,
163 const ReadOptions& read_opts, ReadCallback* callback, bool* is_blob_index) {
164 *seq = kMaxSequenceNumber;
165
166 for (auto& memtable : *list) {
167 assert(memtable->IsFragmentedRangeTombstonesConstructed());
168 SequenceNumber current_seq = kMaxSequenceNumber;
169
170 bool done =
171 memtable->Get(key, value, columns, timestamp, s, merge_context,
172 max_covering_tombstone_seq, &current_seq, read_opts,
173 true /* immutable_memtable */, callback, is_blob_index);
174 if (*seq == kMaxSequenceNumber) {
175 // Store the most recent sequence number of any operation on this key.
176 // Since we only care about the most recent change, we only need to
177 // return the first operation found when searching memtables in
178 // reverse-chronological order.
179 // current_seq would be equal to kMaxSequenceNumber if the value was to be
180 // skipped. This allows seq to be assigned again when the next value is
181 // read.
182 *seq = current_seq;
183 }
184
185 if (done) {
186 assert(*seq != kMaxSequenceNumber ||
187 (!s->ok() && !s->IsMergeInProgress()));
188 return true;
189 }
190 if (!s->ok() && !s->IsMergeInProgress() && !s->IsNotFound()) {
191 return false;
192 }
193 }
194 return false;
195}

197Status MemTableListVersion::AddRangeTombstoneIterators(
198 const ReadOptions& read_opts, Arena* /*arena*/,
199 RangeDelAggregator* range_del_agg) {
200 assert(range_del_agg != nullptr);
201 // Except for snapshot read, using kMaxSequenceNumber is OK because these
202 // are immutable memtables.
203 SequenceNumber read_seq = read_opts.snapshot != nullptr
204 ? read_opts.snapshot->GetSequenceNumber()
205 : kMaxSequenceNumber;
206 for (auto& m : memlist_) {
207 assert(m->IsFragmentedRangeTombstonesConstructed());
208 std::unique_ptr<FragmentedRangeTombstoneIterator> range_del_iter(
209 m->NewRangeTombstoneIterator(read_opts, read_seq,
210 true /* immutable_memtable */));
211 range_del_agg->AddTombstones(std::move(range_del_iter));
212 }
213 return Status::OK();

228void MemTableListVersion::AddIterators(
229 const ReadOptions& options,
230 UnownedPtr<const SeqnoToTimeMapping> seqno_to_time_mapping,
231 const SliceTransform* prefix_extractor,
232 MergeIteratorBuilder* merge_iter_builder, bool add_range_tombstone_iter) {
233 for (auto& m : memlist_) {
234 auto mem_iter =
235 m->NewIterator(options, seqno_to_time_mapping,
236 merge_iter_builder->GetArena(), prefix_extractor,
237 /*for_flush=*/false);
238 if (!add_range_tombstone_iter || options.ignore_range_deletions) {
239 merge_iter_builder->AddIterator(mem_iter);
240 } else {
241 // Except for snapshot read, using kMaxSequenceNumber is OK because these
242 // are immutable memtables.
243 SequenceNumber read_seq = options.snapshot != nullptr
244 ? options.snapshot->GetSequenceNumber()
245 : kMaxSequenceNumber;
246 std::unique_ptr<TruncatedRangeDelIterator> mem_tombstone_iter;
247 auto range_del_iter = m->NewRangeTombstoneIterator(
248 options, read_seq, true /* immutale_memtable */);
249 if (range_del_iter == nullptr || range_del_iter->empty()) {
250 delete range_del_iter;
251 } else {
252 mem_tombstone_iter = std::make_unique<TruncatedRangeDelIterator>(
253 std::unique_ptr<FragmentedRangeTombstoneIterator>(range_del_iter),
254 &m->GetInternalKeyComparator(), nullptr /* smallest */,
255 nullptr /* largest */);
256 }
257 merge_iter_builder->AddPointAndTombstoneIterator(
258 mem_iter, std::move(mem_tombstone_iter));
259 }
260 }

319// Removes m from list of memtables not flushed. Caller should NOT Unref m.
320void MemTableListVersion::Remove(ReadOnlyMemTable* m,
321 autovector<ReadOnlyMemTable*>* to_delete) {
322 assert(refs_ == 1); // only when refs_ == 1 is MemTableListVersion mutable
323 memlist_.remove(m);
324
325 m->MarkFlushed();
326 if (max_write_buffer_size_to_maintain_ > 0 ||
327 max_write_buffer_number_to_maintain_ > 0) {
328 memlist_history_.push_front(m);
329 // Unable to get size of mutable memtable at this point, pass 0 to
330 // TrimHistory as a best effort.
331 TrimHistory(to_delete, 0);
332 } else {
333 UnrefMemTable(to_delete, m);
334 }

352bool MemTableListVersion::MemtableLimitExceeded(size_t usage) {
353 if (max_write_buffer_size_to_maintain_ > 0) {
354 // calculate the total memory usage after dropping the oldest flushed
355 // memtable, compare with max_write_buffer_size_to_maintain_ to decide
356 // whether to trim history
357 return MemoryAllocatedBytesExcludingLast() + usage >=
358 static_cast<size_t>(max_write_buffer_size_to_maintain_);
359 } else if (max_write_buffer_number_to_maintain_ > 0) {
360 return memlist_.size() + memlist_history_.size() >
361 static_cast<size_t>(max_write_buffer_number_to_maintain_);
362 } else {
363 return false;
364 }
365}

404// Returns the memtables that need to be flushed.
405void MemTableList::PickMemtablesToFlush(uint64_t max_memtable_id,
406 autovector<ReadOnlyMemTable*>* ret,
407 uint64_t* max_next_log_number) {
408 AutoThreadOperationStageUpdater stage_updater(
409 ThreadStatus::STAGE_PICK_MEMTABLES_TO_FLUSH);
410 const auto& memlist = current_->memlist_;
411 bool atomic_flush = false;
412
413 // Note: every time MemTableList::Add(mem) is called, it adds the new mem
414 // at the FRONT of the memlist (memlist.push_front(mem)). Therefore, by
415 // iterating through the memlist starting at the end, the vector<MemTable*>
416 // ret is filled with memtables already sorted in increasing MemTable ID.
417 // However, when the mempurge feature is activated, new memtables with older
418 // IDs will be added to the memlist.
419 auto it = memlist.rbegin();
420 for (; it != memlist.rend(); ++it) {
421 ReadOnlyMemTable* m = *it;
422 if (!atomic_flush && m->atomic_flush_seqno_ != kMaxSequenceNumber) {
423 atomic_flush = true;
424 }
425 if (m->GetID() > max_memtable_id) {
426 break;
427 }
428 if (!m->flush_in_progress_) {
429 assert(!m->flush_completed_);
430 num_flush_not_started_--;
431 if (num_flush_not_started_ == 0) {
432 imm_flush_needed.store(false, std::memory_order_release);
433 }
434 m->flush_in_progress_ = true; // flushing will start very soon
435 if (max_next_log_number) {
436 *max_next_log_number =
437 std::max(m->GetNextLogNumber(), *max_next_log_number);
438 }
439 ret->push_back(m);
440 } else if (!ret->empty()) {
441 // This `break` is necessary to prevent picking non-consecutive memtables
442 // in case `memlist` has one or more entries with
443 // `flush_in_progress_ == true` sandwiched between entries with
444 // `flush_in_progress_ == false`. This could happen after parallel flushes
445 // are picked and the one flushing older memtables is rolled back.
446 break;
447 }
448 }
449 if (!ret->empty() && it != memlist.rend()) {
450 // checks that the first memtable not picked to flush is not ingested wbwi.
451 // Ingested memtable should be flushed together with the memtable before it
452 // since they map to the same WAL and have the same NextLogNumber().
453 assert(strcmp((*it)->Name(), "WBWIMemTable") != 0);
454 }
455 if (!atomic_flush || num_flush_not_started_ == 0) {
456 flush_requested_ = false; // start-flush request is complete
457 }
458}

460void MemTableList::RollbackMemtableFlush(
461 const autovector<ReadOnlyMemTable*>& mems,
462 bool rollback_succeeding_memtables) {
463 TEST_SYNC_POINT("RollbackMemtableFlush");
464 AutoThreadOperationStageUpdater stage_updater(
465 ThreadStatus::STAGE_MEMTABLE_ROLLBACK);
466#ifndef NDEBUG
467 for (ReadOnlyMemTable* m : mems) {
468 assert(m->flush_in_progress_);
469 assert(m->file_number_ == 0);
470 }
471#endif
472
473 if (rollback_succeeding_memtables && !mems.empty()) {
474 std::list<ReadOnlyMemTable*>& memlist = current_->memlist_;
475 auto it = memlist.rbegin();
476 for (; *it != mems[0] && it != memlist.rend(); ++it) {
477 }
478 // mems should be in memlist
479 assert(*it == mems[0]);
480 if (*it == mems[0]) {
481 ++it;
482 }
483 while (it != memlist.rend()) {
484 ReadOnlyMemTable* m = *it;
485 // Only rollback complete, not in-progress,
486 // in_progress can be flushes that are still writing SSTs
487 if (m->flush_completed_) {
488 m->flush_in_progress_ = false;
489 m->flush_completed_ = false;
490 m->edit_.Clear();
491 m->file_number_ = 0;
492 num_flush_not_started_++;
493 ++it;
494 } else {
495 break;
496 }
497 }
498 }
499
500 for (ReadOnlyMemTable* m : mems) {
501 if (m->flush_in_progress_) {
502 assert(m->file_number_ == 0);
503 m->file_number_ = 0;
504 m->flush_in_progress_ = false;
505 m->flush_completed_ = false;
506 m->edit_.Clear();
507 num_flush_not_started_++;
508 }
509 }
510 if (!mems.empty()) {
511 imm_flush_needed.store(true, std::memory_order_release);
512 }
513}

515// Try record a successful flush in the manifest file. It might just return
516// Status::OK letting a concurrent flush to do actual the recording..
517Status MemTableList::TryInstallMemtableFlushResults(
518 ColumnFamilyData* cfd, const MutableCFOptions& mutable_cf_options,
519 const autovector<ReadOnlyMemTable*>& mems,
520 LogsWithPrepTracker* prep_tracker, VersionSet* vset, InstrumentedMutex* mu,
521 uint64_t file_number, autovector<ReadOnlyMemTable*>* to_delete,
522 FSDirectory* db_directory, LogBuffer* log_buffer,
523 std::list<std::unique_ptr<FlushJobInfo>>* committed_flush_jobs_info,
524 bool write_edits) {
525 AutoThreadOperationStageUpdater stage_updater(
526 ThreadStatus::STAGE_MEMTABLE_INSTALL_FLUSH_RESULTS);
527 mu->AssertHeld();
528
529 const ReadOptions read_options(Env::IOActivity::kFlush);
530 const WriteOptions write_options(Env::IOActivity::kFlush);
531
532 // Flush was successful
533 // Record the status on the memtable object. Either this call or a call by a
534 // concurrent flush thread will read the status and write it to manifest.
535 for (size_t i = 0; i < mems.size(); ++i) {
536 // All the edits are associated with the first memtable of this batch.
537 assert(i == 0 || mems[i]->GetEdits()->NumEntries() == 0);
538
539 mems[i]->flush_completed_ = true;
540 mems[i]->file_number_ = file_number;
541 }
542
543 // if some other thread is already committing, then return
544 Status s;
545 if (commit_in_progress_) {
546 TEST_SYNC_POINT("MemTableList::TryInstallMemtableFlushResults:InProgress");
547 return s;
548 }
549
550 // Only a single thread can be executing this piece of code
551 commit_in_progress_ = true;
552
553 // Retry until all completed flushes are committed. New flushes can finish
554 // while the current thread is writing manifest where mutex is released.
555 while (s.ok()) {
556 auto& memlist = current_->memlist_;
557 // The back is the oldest; if flush_completed_ is not set to it, it means
558 // that we were assigned a more recent memtable. The memtables' flushes must
559 // be recorded in manifest in order. A concurrent flush thread, who is
560 // assigned to flush the oldest memtable, will later wake up and does all
561 // the pending writes to manifest, in order.
562 if (memlist.empty() || !memlist.back()->flush_completed_) {
563 break;
564 }
565 // scan all memtables from the earliest, and commit those
566 // (in that order) that have finished flushing. Memtables
567 // are always committed in the order that they were created.
568 uint64_t batch_file_number = 0;
569 autovector<VersionEdit*> edit_list;
570 autovector<ReadOnlyMemTable*> memtables_to_flush;
571 // enumerate from the last (earliest) element to see how many batch finished
572 for (auto it = memlist.rbegin(); it != memlist.rend(); ++it) {
573 ReadOnlyMemTable* m = *it;
574 if (!m->flush_completed_) {
575 break;
576 }
577 if (it == memlist.rbegin() || batch_file_number != m->file_number_) {
578 // Oldest memtable in a new batch.
579 batch_file_number = m->file_number_;
580 if (m->edit_.GetBlobFileAdditions().empty()) {
581 ROCKS_LOG_BUFFER(log_buffer,
582 "[%s] Level-0 commit flush result of table #%" PRIu64
583 " started",
584 cfd->GetName().c_str(), m->file_number_);
585 } else {
586 ROCKS_LOG_BUFFER(log_buffer,
587 "[%s] Level-0 commit flush result of table #%" PRIu64
588 " (+%zu blob files) started",
589 cfd->GetName().c_str(), m->file_number_,
590 m->edit_.GetBlobFileAdditions().size());
591 }
592
593 edit_list.push_back(&m->edit_);
594 std::unique_ptr<FlushJobInfo> info = m->ReleaseFlushJobInfo();
595 if (info != nullptr) {
596 committed_flush_jobs_info->push_back(std::move(info));
597 }
598 }
599 memtables_to_flush.push_back(m);
600 }
601
602 size_t num_mem_to_flush = memtables_to_flush.size();
603 // TODO(myabandeh): Not sure how batch_count could be 0 here.
604 if (num_mem_to_flush > 0) {
605 VersionEdit edit;
606#ifdef ROCKSDB_ASSERT_STATUS_CHECKED
607 if (memtables_to_flush.size() == memlist.size()) {
608 // TODO(yuzhangyu): remove this testing code once the
609 // `GetEditForDroppingCurrentVersion` API is used by the atomic data
610 // replacement. This function can get the same edits for wal related
611 // fields, and some duplicated fields as contained already in edit_list
612 // for column family's recovery.
613 edit = GetEditForDroppingCurrentVersion(cfd, vset, prep_tracker);
614 } else {
615 edit = GetDBRecoveryEditForObsoletingMemTables(
616 vset, *cfd, edit_list, memtables_to_flush, prep_tracker);
617 }
618#else
619 edit = GetDBRecoveryEditForObsoletingMemTables(
620 vset, *cfd, edit_list, memtables_to_flush, prep_tracker);
621#endif // ROCKSDB_ASSERT_STATUS_CHECKED
622 TEST_SYNC_POINT_CALLBACK(
623 "MemTableList::TryInstallMemtableFlushResults:"
624 "AfterComputeMinWalToKeep",
625 nullptr);
626 edit_list.push_back(&edit);
627
628 const auto manifest_write_cb = [this, cfd, num_mem_to_flush, log_buffer,
629 to_delete, mu](const Status& status) {
630 RemoveMemTablesOrRestoreFlags(status, cfd, num_mem_to_flush, log_buffer,
631 to_delete, mu);
632 };
633 if (write_edits) {
634 // this can release and reacquire the mutex.
635 s = vset->LogAndApply(
636 cfd, mutable_cf_options, read_options, write_options, edit_list, mu,
637 db_directory, /*new_descriptor_log=*/false,
638 /*column_family_options=*/nullptr, manifest_write_cb);
639 } else {
640 // If write_edit is false (e.g: successful mempurge),
641 // then remove old memtables, wake up manifest write queue threads,
642 // and don't commit anything to the manifest file.
643 RemoveMemTablesOrRestoreFlags(s, cfd, num_mem_to_flush, log_buffer,
644 to_delete, mu);
645 // Note: cfd->SetLogNumber is only called when a VersionEdit
646 // is written to MANIFEST. When mempurge is succesful, we skip
647 // this step, therefore cfd->GetLogNumber is always is
648 // earliest log with data unflushed.
649 // Notify new head of manifest write queue.
650 // wake up all the waiting writers
651 // TODO(bjlemaire): explain full reason WakeUpWaitingManifestWriters
652 // needed or investigate more.
653 vset->WakeUpWaitingManifestWriters();
654 }
655 }
656 }
657 commit_in_progress_ = false;
658 return s;
659}

750void MemTableList::RemoveMemTablesOrRestoreFlags(
751 const Status& s, ColumnFamilyData* cfd, size_t num_mem_to_flush,
752 LogBuffer* log_buffer, autovector<ReadOnlyMemTable*>* to_delete,
753 InstrumentedMutex* mu) {
754 assert(mu);
755 mu->AssertHeld();
756 assert(to_delete);
757 // we will be changing the version in the next code path,
758 // so we better create a new one, since versions are immutable
759 InstallNewVersion();
760
761 // All the later memtables that have the same filenum
762 // are part of the same batch. They can be committed now.
763 uint64_t mem_id = 1; // how many memtables have been flushed.
764
765 // commit new state only if the column family is NOT dropped.
766 // The reason is as follows (refer to
767 // ColumnFamilyTest.FlushAndDropRaceCondition).
768 // If the column family is dropped, then according to LogAndApply, its
769 // corresponding flush operation is NOT written to the MANIFEST. This
770 // means the DB is not aware of the L0 files generated from the flush.
771 // By committing the new state, we remove the memtable from the memtable
772 // list. Creating an iterator on this column family will not be able to
773 // read full data since the memtable is removed, and the DB is not aware
774 // of the L0 files, causing MergingIterator unable to build child
775 // iterators. RocksDB contract requires that the iterator can be created
776 // on a dropped column family, and we must be able to
777 // read full data as long as column family handle is not deleted, even if
778 // the column family is dropped.
779 if (s.ok() && !cfd->IsDropped()) { // commit new state
780 while (num_mem_to_flush-- > 0) {
781 ReadOnlyMemTable* m = current_->memlist_.back();
782 // TODO: The logging can be redundant when we flush multiple memtables
783 // into one SST file. We should only check the edit_ of the oldest
784 // memtable in the group in that case.
785 if (m->edit_.GetBlobFileAdditions().empty()) {
786 ROCKS_LOG_BUFFER(log_buffer,
787 "[%s] Level-0 commit flush result of table #%" PRIu64
788 ": memtable #%" PRIu64 " done",
789 cfd->GetName().c_str(), m->file_number_, mem_id);
790 } else {
791 ROCKS_LOG_BUFFER(log_buffer,
792 "[%s] Level-0 commit flush result of table #%" PRIu64
793 " (+%zu blob files)"
794 ": memtable #%" PRIu64 " done",
795 cfd->GetName().c_str(), m->file_number_,
796 m->edit_.GetBlobFileAdditions().size(), mem_id);
797 }
798
799 assert(m->file_number_ > 0);
800 current_->Remove(m, to_delete);
801 UpdateCachedValuesFromMemTableListVersion();
802 ResetTrimHistoryNeeded();
803 ++mem_id;
804 }
805 } else {
806 for (auto it = current_->memlist_.rbegin(); num_mem_to_flush-- > 0; ++it) {
807 ReadOnlyMemTable* m = *it;
808 // commit failed. setup state so that we can flush again.
809 if (m->edit_.GetBlobFileAdditions().empty()) {
810 ROCKS_LOG_BUFFER(log_buffer,
811 "Level-0 commit table #%" PRIu64 ": memtable #%" PRIu64
812 " failed",
813 m->file_number_, mem_id);
814 } else {
815 ROCKS_LOG_BUFFER(log_buffer,
816 "Level-0 commit table #%" PRIu64
817 " (+%zu blob files)"
818 ": memtable #%" PRIu64 " failed",
819 m->file_number_,
820 m->edit_.GetBlobFileAdditions().size(), mem_id);
821 }
822
823 m->flush_completed_ = false;
824 m->flush_in_progress_ = false;
825 m->edit_.Clear();
826 num_flush_not_started_++;
827 m->file_number_ = 0;
828 imm_flush_needed.store(true, std::memory_order_release);
829 ++mem_id;
830 }
831 }

853// Commit a successful atomic flush in the manifest file.
854Status InstallMemtableAtomicFlushResults(
855 const autovector<MemTableList*>* imm_lists,
856 const autovector<ColumnFamilyData*>& cfds,
857 const autovector<const MutableCFOptions*>& mutable_cf_options_list,
858 const autovector<const autovector<ReadOnlyMemTable*>*>& mems_list,
859 VersionSet* vset, LogsWithPrepTracker* prep_tracker, InstrumentedMutex* mu,
860 const autovector<FileMetaData*>& file_metas,
861 const autovector<std::list<std::unique_ptr<FlushJobInfo>>*>&
862 committed_flush_jobs_info,
863 autovector<ReadOnlyMemTable*>* to_delete, FSDirectory* db_directory,
864 LogBuffer* log_buffer) {
865 AutoThreadOperationStageUpdater stage_updater(
866 ThreadStatus::STAGE_MEMTABLE_INSTALL_FLUSH_RESULTS);
867 mu->AssertHeld();
868
869 const ReadOptions read_options(Env::IOActivity::kFlush);
870 const WriteOptions write_options(Env::IOActivity::kFlush);
871
872 size_t num = mems_list.size();
873 assert(cfds.size() == num);
874 if (imm_lists != nullptr) {
875 assert(imm_lists->size() == num);
876 }
877 if (num == 0) {
878 return Status::OK();
879 }
880
881 for (size_t k = 0; k != num; ++k) {
882#ifndef NDEBUG
883 const auto* imm =
884 (imm_lists == nullptr) ? cfds[k]->imm() : imm_lists->at(k);
885 if (!mems_list[k]->empty()) {
886 assert((*mems_list[k])[0]->GetID() == imm->GetEarliestMemTableID());
887 }
888#endif
889 assert(nullptr != file_metas[k]);
890 for (size_t i = 0; i != mems_list[k]->size(); ++i) {
891 assert(i == 0 || (*mems_list[k])[i]->GetEdits()->NumEntries() == 0);
892 (*mems_list[k])[i]->SetFlushCompleted(true);
893 (*mems_list[k])[i]->SetFileNumber(file_metas[k]->fd.GetNumber());
894 }
895 if (committed_flush_jobs_info[k]) {
896 assert(!mems_list[k]->empty());
897 assert((*mems_list[k])[0]);
898 std::unique_ptr<FlushJobInfo> flush_job_info =
899 (*mems_list[k])[0]->ReleaseFlushJobInfo();
900 committed_flush_jobs_info[k]->push_back(std::move(flush_job_info));
901 }
902 }
903
904 Status s;
905
906 autovector<autovector<VersionEdit*>> edit_lists;
907 uint32_t num_entries = 0;
908 for (const auto mems : mems_list) {
909 assert(mems != nullptr);
910 autovector<VersionEdit*> edits;
911 assert(!mems->empty());
912 edits.emplace_back((*mems)[0]->GetEdits());
913 ++num_entries;
914 edit_lists.emplace_back(edits);
915 }
916
917 WalNumber min_wal_number_to_keep = 0;
918 if (vset->db_options()->allow_2pc) {
919 min_wal_number_to_keep = PrecomputeMinLogNumberToKeep2PC(
920 vset, cfds, edit_lists, mems_list, prep_tracker);
921 } else {
922 min_wal_number_to_keep =
923 PrecomputeMinLogNumberToKeepNon2PC(vset, cfds, edit_lists);
924 }
925
926 VersionEdit wal_deletion;
927 wal_deletion.SetMinLogNumberToKeep(min_wal_number_to_keep);
928 if (vset->db_options()->track_and_verify_wals_in_manifest &&
929 min_wal_number_to_keep > vset->GetWalSet().GetMinWalNumberToKeep()) {
930 wal_deletion.DeleteWalsBefore(min_wal_number_to_keep);
931 }
932 edit_lists.back().push_back(&wal_deletion);
933 ++num_entries;
934
935 // Mark the version edits as an atomic group if the number of version edits
936 // exceeds 1.
937 if (cfds.size() > 1) {
938 for (size_t i = 0; i < edit_lists.size(); i++) {
939 assert((edit_lists[i].size() == 1) ||
940 ((edit_lists[i].size() == 2) && (i == edit_lists.size() - 1)));
941 for (auto& e : edit_lists[i]) {
942 e->MarkAtomicGroup(--num_entries);
943 }
944 }
945 assert(0 == num_entries);
946 }
947
948 // this can release and reacquire the mutex.
949 s = vset->LogAndApply(cfds, mutable_cf_options_list, read_options,
950 write_options, edit_lists, mu, db_directory);
951
952 for (size_t k = 0; k != cfds.size(); ++k) {
953 auto* imm = (imm_lists == nullptr) ? cfds[k]->imm() : imm_lists->at(k);
954 imm->InstallNewVersion();
955 }
956
957 if (s.ok() || s.IsColumnFamilyDropped()) {
958 for (size_t i = 0; i != cfds.size(); ++i) {
959 if (cfds[i]->IsDropped()) {
960 continue;
961 }
962 auto* imm = (imm_lists == nullptr) ? cfds[i]->imm() : imm_lists->at(i);
963 for (auto m : *mems_list[i]) {
964 assert(m->GetFileNumber() > 0);
965 uint64_t mem_id = m->GetID();
966
967 const VersionEdit* const edit = m->GetEdits();
968 assert(edit);
969
970 if (edit->GetBlobFileAdditions().empty()) {
971 ROCKS_LOG_BUFFER(log_buffer,
972 "[%s] Level-0 commit table #%" PRIu64
973 ": memtable #%" PRIu64 " done",
974 cfds[i]->GetName().c_str(), m->GetFileNumber(),
975 mem_id);
976 } else {
977 ROCKS_LOG_BUFFER(log_buffer,
978 "[%s] Level-0 commit table #%" PRIu64
979 " (+%zu blob files)"
980 ": memtable #%" PRIu64 " done",
981 cfds[i]->GetName().c_str(), m->GetFileNumber(),
982 edit->GetBlobFileAdditions().size(), mem_id);
983 }
984
985 imm->current_->Remove(m, to_delete);
986 imm->UpdateCachedValuesFromMemTableListVersion();
987 imm->ResetTrimHistoryNeeded();
988 }
989 }
990 } else {
991 for (size_t i = 0; i != cfds.size(); ++i) {
992 auto* imm = (imm_lists == nullptr) ? cfds[i]->imm() : imm_lists->at(i);
993 for (auto m : *mems_list[i]) {
994 uint64_t mem_id = m->GetID();
995
996 const VersionEdit* const edit = m->GetEdits();
997 assert(edit);
998
999 if (edit->GetBlobFileAdditions().empty()) {
1000 ROCKS_LOG_BUFFER(log_buffer,
1001 "[%s] Level-0 commit table #%" PRIu64
1002 ": memtable #%" PRIu64 " failed",
1003 cfds[i]->GetName().c_str(), m->GetFileNumber(),
1004 mem_id);
1005 } else {
1006 ROCKS_LOG_BUFFER(log_buffer,
1007 "[%s] Level-0 commit table #%" PRIu64
1008 " (+%zu blob files)"
1009 ": memtable #%" PRIu64 " failed",
1010 cfds[i]->GetName().c_str(), m->GetFileNumber(),
1011 edit->GetBlobFileAdditions().size(), mem_id);
1012 }
1013
1014 m->SetFlushCompleted(false);
1015 m->SetFlushInProgress(false);
1016 m->GetEdits()->Clear();
1017 m->SetFileNumber(0);
1018 imm->num_flush_not_started_++;
1019 }
1020 imm->imm_flush_needed.store(true, std::memory_order_release);
1021 }
1022 }
1023
1024 return s;
1025}
