102// Key used by skip list, as the binary searchable index of WriteBatchWithIndex.
103struct WriteBatchIndexEntry {
104 WriteBatchIndexEntry(size_t o, uint32_t c, size_t ko, size_t ksz)
105 : offset(o),
106 column_family(c),
107 has_single_del(false),
108 has_overwritten_single_del(false),
109 key_offset(ko),
110 key_size(ksz),
111 search_key(nullptr) {}
112 // Create a dummy entry as the search key. This index entry won't be backed
113 // by an entry from the write batch, but a pointer to the search key. Or a
114 // special flag of offset can indicate we are seek to first.
115 // @_search_key: the search key
116 // @_column_family: column family
117 // @is_forward_direction: true for Seek(). False for SeekForPrev()
118 // @is_seek_to_first: true if we seek to the beginning of the column family
119 // _search_key should be null in this case.
120 WriteBatchIndexEntry(const Slice* _search_key, uint32_t _column_family,
121 bool is_forward_direction, bool is_seek_to_first)
122 // For SeekForPrev(), we need to make the dummy entry larger than any
123 // entry who has the same search key. Otherwise, we'll miss those entries.
124 : offset(is_forward_direction ? 0 : std::numeric_limits<size_t>::max()),
125 column_family(_column_family),
126 has_single_del(false),
127 has_overwritten_single_del(false),
128 key_offset(0),
129 key_size(is_seek_to_first ? kFlagMinInCf : 0),
130 search_key(_search_key) {
131 assert(_search_key != nullptr || is_seek_to_first);
132 }
133
134 // If this flag appears in the key_size, it indicates a
135 // key that is smaller than any other entry for the same column family.
136 static const size_t kFlagMinInCf = std::numeric_limits<size_t>::max();
137
138 bool is_min_in_cf() const {
139 assert(key_size != kFlagMinInCf ||
140 (key_offset == 0 && search_key == nullptr));
141 return key_size == kFlagMinInCf;
142 }
143
144 // offset of an entry in write batch's string buffer. If this is a dummy
145 // lookup key, in which case search_key != nullptr, offset is set to either
146 // 0 or max, only for comparison purpose. Because when entries have the same
147 // key, the entry with larger offset is larger, offset = 0 will make a seek
148 // key small or equal than all the entries with the seek key, so that Seek()
149 // will find all the entries of the same key. Similarly, offset = MAX will
150 // make the entry just larger than all entries with the search key so
151 // SeekForPrev() will see all the keys with the same key.
152 size_t offset;
153 uint32_t column_family; // column family of the entry.
154 bool has_single_del; // whether single del was issued for this key
155 bool has_overwritten_single_del; // whether a single del for this key was
156 // overwritten by another key
157 size_t key_offset; // offset of the key in write batch's string buffer.
158 size_t key_size; // size of the key. kFlagMinInCf indicates
159 // that this is a dummy look up entry for
160 // SeekToFirst() to the beginning of the column
161 // family. We use the flag here to save a boolean
162 // in the struct.
163
164 const Slice* search_key; // if not null, instead of reading keys from
165 // write batch, use it to compare. This is used
166 // for lookup key.
167};

219class WBWIIteratorImpl final : public WBWIIterator {
220 public:
221 enum Result : uint8_t {
222 kFound,
223 kDeleted,
224 kNotFound,
225 kMergeInProgress,
226 kError
227 };
228 WBWIIteratorImpl(uint32_t column_family_id,
229 WriteBatchEntrySkipList* skip_list,
230 const ReadableWriteBatch* write_batch,
231 WriteBatchEntryComparator* comparator,
232 const Slice* iterate_lower_bound = nullptr,
233 const Slice* iterate_upper_bound = nullptr)
234 : column_family_id_(column_family_id),
235 skip_list_iter_(skip_list),
236 write_batch_(write_batch),
237 comparator_(comparator),
238 iterate_lower_bound_(iterate_lower_bound),
239 iterate_upper_bound_(iterate_upper_bound) {}
240
241 ~WBWIIteratorImpl() override {}
242
243 bool Valid() const override {
244 return !out_of_bound_ && ValidRegardlessOfBoundLimit();
245 }
246
247 void SeekToFirst() override {
248 if (iterate_lower_bound_ != nullptr) {
249 WriteBatchIndexEntry search_entry(
250 iterate_lower_bound_ /* search_key */, column_family_id_,
251 true /* is_forward_direction */, false /* is_seek_to_first */);
252 skip_list_iter_.Seek(&search_entry);
253 } else {
254 WriteBatchIndexEntry search_entry(
255 nullptr /* search_key */, column_family_id_,
256 true /* is_forward_direction */, true /* is_seek_to_first */);
257 skip_list_iter_.Seek(&search_entry);
258 }
259
260 if (ValidRegardlessOfBoundLimit()) {
261 out_of_bound_ = TestOutOfBound();
262 }
263 }
264
265 void SeekToLast() override {
266 WriteBatchIndexEntry search_entry =
267 (iterate_upper_bound_ != nullptr)
268 ? WriteBatchIndexEntry(
269 iterate_upper_bound_ /* search_key */, column_family_id_,
270 true /* is_forward_direction */, false /* is_seek_to_first */)
271 : WriteBatchIndexEntry(
272 nullptr /* search_key */, column_family_id_ + 1,
273 true /* is_forward_direction */, true /* is_seek_to_first */);
274
275 skip_list_iter_.Seek(&search_entry);
276 if (!skip_list_iter_.Valid()) {
277 skip_list_iter_.SeekToLast();
278 } else {
279 skip_list_iter_.Prev();
280 }
281
282 if (ValidRegardlessOfBoundLimit()) {
283 out_of_bound_ = TestOutOfBound();
284 }
285 }
286
287 void Seek(const Slice& key) override {
288 if (BeforeLowerBound(&key)) { // cap to prevent out of bound
289 SeekToFirst();
290 return;
291 }
292
293 WriteBatchIndexEntry search_entry(&key, column_family_id_,
294 true /* is_forward_direction */,
295 false /* is_seek_to_first */);
296 skip_list_iter_.Seek(&search_entry);
297
298 if (ValidRegardlessOfBoundLimit()) {
299 out_of_bound_ = TestOutOfBound();
300 }
301 }
302
303 void SeekForPrev(const Slice& key) override {
304 if (AtOrAfterUpperBound(&key)) { // cap to prevent out of bound
305 SeekToLast();
306 return;
307 }
308
309 WriteBatchIndexEntry search_entry(&key, column_family_id_,
310 false /* is_forward_direction */,
311 false /* is_seek_to_first */);
312 skip_list_iter_.SeekForPrev(&search_entry);
313
314 if (ValidRegardlessOfBoundLimit()) {
315 out_of_bound_ = TestOutOfBound();
316 }
317 }
318
319 void Next() override {
320 skip_list_iter_.Next();
321 if (ValidRegardlessOfBoundLimit()) {
322 out_of_bound_ = TestOutOfBound();
323 }
324 }
325
326 void Prev() override {
327 skip_list_iter_.Prev();
328 if (ValidRegardlessOfBoundLimit()) {
329 out_of_bound_ = TestOutOfBound();
330 }
331 }
332
333 WriteEntry Entry() const override;
334
335 bool HasOverWrittenSingleDel() const override {
336 assert(Valid());
337 return skip_list_iter_.key()->has_overwritten_single_del;
338 }
339
340 Status status() const override {
341 // this is in-memory data structure, so the only way status can be non-ok is
342 // through memory corruption
343 return Status::OK();
344 }
345
346 const WriteBatchIndexEntry* GetRawEntry() const {
347 return skip_list_iter_.key();
348 }
349
350 bool MatchesKey(uint32_t cf_id, const Slice& key);
351
352 // Moves the iterator to first entry of the previous key.
353 void PrevKey();
354 // Moves the iterator to first entry of the next key.
355 void NextKey();
356
357 // Moves the iterator to the Update (Put, PutEntity or Delete) for the current
358 // key. If there is no Put/PutEntity/Delete, the Iterator will point to the
359 // first entry for this key.
360 // @return kFound if a Put/PutEntity was found for the key
361 // @return kDeleted if a delete was found for the key
362 // @return kMergeInProgress if only merges were found for the key
363 // @return kError if an unsupported operation was found for the key
364 // @return kNotFound if no operations were found for this key
365 //
366 Result FindLatestUpdate(const Slice& key, MergeContext* merge_context);
367 Result FindLatestUpdate(MergeContext* merge_context);
368
369 protected:
370 void AdvanceKey(bool forward);
371
372 private:
373 uint32_t column_family_id_;
374 WriteBatchEntrySkipList::Iterator skip_list_iter_;
375 const ReadableWriteBatch* write_batch_;
376 WriteBatchEntryComparator* comparator_;
377 const Slice* iterate_lower_bound_;
378 const Slice* iterate_upper_bound_;
379 bool out_of_bound_ = false;
380
381 bool TestOutOfBound() const {
382 const Slice& curKey = Entry().key;
383 return AtOrAfterUpperBound(&curKey) || BeforeLowerBound(&curKey);
384 }
385
386 bool ValidRegardlessOfBoundLimit() const {
387 if (!skip_list_iter_.Valid()) {
388 return false;
389 }
390 const WriteBatchIndexEntry* iter_entry = skip_list_iter_.key();
391 return iter_entry != nullptr &&
392 iter_entry->column_family == column_family_id_;
393 }
394
395 bool AtOrAfterUpperBound(const Slice* k) const {
396 if (iterate_upper_bound_ == nullptr) {
397 return false;
398 }
399
400 return comparator_->GetComparator(column_family_id_)
401 ->CompareWithoutTimestamp(*k, /*a_has_ts=*/false,
402 *iterate_upper_bound_,
403 /*b_has_ts=*/false) >= 0;
404 }
405
406 bool BeforeLowerBound(const Slice* k) const {
407 if (iterate_lower_bound_ == nullptr) {
408 return false;
409 }
410
411 return comparator_->GetComparator(column_family_id_)
412 ->CompareWithoutTimestamp(*k, /*a_has_ts=*/false,
413 *iterate_lower_bound_,
414 /*b_has_ts=*/false) < 0;
415 }

423 template <typename... ResultTs>
424 static Status MergeKeyWithNoBaseValue(ColumnFamilyHandle* column_family,
425 const Slice& key,
426 const MergeContext& context,
427 ResultTs... results) {
428 const ImmutableOptions* ioptions = nullptr;
429
430 const Status s = CheckAndGetImmutableOptions(column_family, &ioptions);
431 if (!s.ok()) {
432 return s;
433 }
434
435 assert(ioptions);
436
437 // `op_failure_scope` (an output parameter) is not provided (set to
438 // nullptr) since a failure must be propagated regardless of its value.
439 return MergeHelper::TimedFullMerge(
440 ioptions->merge_operator.get(), key, MergeHelper::kNoBaseValue,
441 context.GetOperands(), ioptions->logger, ioptions->stats,
442 ioptions->clock, /* update_num_ops_stats */ false,
443 /* op_failure_scope */ nullptr, results...);
444 }
445
446 template <typename BaseTag, typename BaseT, typename... ResultTs>
447 static Status MergeKeyWithBaseValue(ColumnFamilyHandle* column_family,
448 const Slice& key, const BaseTag& base_tag,
449 const BaseT& value,
450 const MergeContext& context,
451 ResultTs... results) {
452 const ImmutableOptions* ioptions = nullptr;
453
454 const Status s = CheckAndGetImmutableOptions(column_family, &ioptions);
455 if (!s.ok()) {
456 return s;
457 }
458
459 assert(ioptions);
460
461 // `op_failure_scope` (an output parameter) is not provided (set to
462 // nullptr) since a failure must be propagated regardless of its value.
463 return MergeHelper::TimedFullMerge(
464 ioptions->merge_operator.get(), key, base_tag, value,
465 context.GetOperands(), ioptions->logger, ioptions->stats,
466 ioptions->clock, /* update_num_ops_stats */ false,
467 /* op_failure_scope */ nullptr, results...);
468 }
469
470 // If batch contains a value for key, store it in *value and return kFound.
471 // If batch contains a deletion for key, return Deleted.
472 // If batch contains Merge operations as the most recent entry for a key,
473 // and the merge process does not stop (not reaching a value or delete),
474 // prepend the current merge operands to *operands,
475 // and return kMergeInProgress
476 // If batch does not contain this key, return kNotFound
477 // Else, return kError on error with error Status stored in *s.
478 static WBWIIteratorImpl::Result GetFromBatch(
479 WriteBatchWithIndex* batch, ColumnFamilyHandle* column_family,
480 const Slice& key, MergeContext* merge_context, std::string* value,
481 Status* s);
482
483 static WBWIIteratorImpl::Result GetEntityFromBatch(
484 WriteBatchWithIndex* batch, ColumnFamilyHandle* column_family,
485 const Slice& key, MergeContext* merge_context,
486 PinnableWideColumns* columns, Status* s);
487
488 private:
489 static Status CheckAndGetImmutableOptions(ColumnFamilyHandle* column_family,
490 const ImmutableOptions** ioptions);
491
492 template <typename Traits>
493 static WBWIIteratorImpl::Result GetFromBatchImpl(
494 WriteBatchWithIndex* batch, ColumnFamilyHandle* column_family,

496 typename Traits::OutputType* output, Status* s);
