24// Abstract base class that defines the basic interface for a write batch.
25// See WriteBatch for a basic implementation and WrithBatchWithIndex for an
26// indexed implementation.
27class WriteBatchBase {
28 public:
29 virtual ~WriteBatchBase() {}
30
31 // Store the mapping "key->value" in the database.
32 virtual Status Put(ColumnFamilyHandle* column_family, const Slice& key,
33 const Slice& value) = 0;
34 virtual Status Put(const Slice& key, const Slice& value) = 0;
35 virtual Status Put(ColumnFamilyHandle* column_family, const Slice& key,
36 const Slice& ts, const Slice& value) = 0;
37
38 // Variant of Put() that gathers output like writev(2). The key and value
39 // that will be written to the database are concatenations of arrays of
40 // slices.
41 virtual Status Put(ColumnFamilyHandle* column_family, const SliceParts& key,
42 const SliceParts& value);
43 virtual Status Put(const SliceParts& key, const SliceParts& value);
44
45 // EXPERIMENTAL
46 // Store the mapping "key->value" in the database with the specified write
47 // time in the column family. Using some write time that is in the past to
48 // fast track data to their correct placement and preservation is the intended
49 // usage of this API. The DB makes a reasonable best effort to treat the data
50 // as having the given write time for this purpose but doesn't currently make
51 // any guarantees.
52 //
53 // This feature is experimental and one known side effect is that it can break
54 // snapshot immutability. Reading from a snapshot created before
55 // TimedPut(k, v, t) may or may not see that k->v.
56 // Note: this feature is currently not compatible with user-defined timestamps
57 // and wide columns.
58 virtual Status TimedPut(ColumnFamilyHandle* column_family, const Slice& key,
59 const Slice& value, uint64_t write_unix_time) = 0;
60
61 // Store the mapping "key->{column1:value1, column2:value2, ...}" in the
62 // column family specified by "column_family".
63 virtual Status PutEntity(ColumnFamilyHandle* column_family, const Slice& key,
64 const WideColumns& columns) = 0;
65
66 // Split and store wide column entities in multiple column families (a.k.a.
67 // AttributeGroups)
68 virtual Status PutEntity(const Slice& key,
69 const AttributeGroups& attribute_groups) = 0;
70
71 // Merge "value" with the existing value of "key" in the database.
72 // "key->merge(existing, value)"
73 virtual Status Merge(ColumnFamilyHandle* column_family, const Slice& key,
74 const Slice& value) = 0;
75 virtual Status Merge(const Slice& key, const Slice& value) = 0;
76 virtual Status Merge(ColumnFamilyHandle* column_family, const Slice& key,
77 const Slice& ts, const Slice& value) = 0;
78
79 // variant that takes SliceParts
80 virtual Status Merge(ColumnFamilyHandle* column_family, const SliceParts& key,
81 const SliceParts& value);
82 virtual Status Merge(const SliceParts& key, const SliceParts& value);
83
84 // If the database contains a mapping for "key", erase it. Else do nothing.
85 virtual Status Delete(ColumnFamilyHandle* column_family,
86 const Slice& key) = 0;
87 virtual Status Delete(const Slice& key) = 0;
88 virtual Status Delete(ColumnFamilyHandle* column_family, const Slice& key,
89 const Slice& ts) = 0;
90
91 // variant that takes SliceParts
92 virtual Status Delete(ColumnFamilyHandle* column_family,
93 const SliceParts& key);
94 virtual Status Delete(const SliceParts& key);
95
96 // If the database contains a mapping for "key", erase it. Expects that the
97 // key was not overwritten. Else do nothing.
98 virtual Status SingleDelete(ColumnFamilyHandle* column_family,
99 const Slice& key) = 0;
100 virtual Status SingleDelete(const Slice& key) = 0;
101 virtual Status SingleDelete(ColumnFamilyHandle* column_family,
102 const Slice& key, const Slice& ts) = 0;
103
104 // variant that takes SliceParts
105 virtual Status SingleDelete(ColumnFamilyHandle* column_family,
106 const SliceParts& key);
107 virtual Status SingleDelete(const SliceParts& key);
108
109 // If the database contains mappings in the range ["begin_key", "end_key"),
110 // erase them. Else do nothing.
111 virtual Status DeleteRange(ColumnFamilyHandle* column_family,
112 const Slice& begin_key, const Slice& end_key) = 0;
113 virtual Status DeleteRange(const Slice& begin_key, const Slice& end_key) = 0;
114 virtual Status DeleteRange(ColumnFamilyHandle* column_family,
115 const Slice& begin_key, const Slice& end_key,
116 const Slice& ts) = 0;
117
118 // variant that takes SliceParts
119 virtual Status DeleteRange(ColumnFamilyHandle* column_family,
120 const SliceParts& begin_key,
121 const SliceParts& end_key);
122 virtual Status DeleteRange(const SliceParts& begin_key,
123 const SliceParts& end_key);
124
125 // Append a blob of arbitrary size to the records in this batch. The blob will
126 // be stored in the transaction log but not in any other file. In particular,
127 // it will not be persisted to the SST files. When iterating over this
128 // WriteBatch, WriteBatch::Handler::LogData will be called with the contents
129 // of the blob as it is encountered. Blobs, puts, deletes, and merges will be
130 // encountered in the same order in which they were inserted. The blob will
131 // NOT consume sequence number(s) and will NOT increase the count of the batch
132 //
133 // Example application: add timestamps to the transaction log for use in
134 // replication.
135 virtual Status PutLogData(const Slice& blob) = 0;
136
137 // Clear all updates buffered in this batch.
138 virtual void Clear() = 0;
139
140 // Covert this batch into a WriteBatch. This is an abstracted way of
141 // converting any WriteBatchBase(eg WriteBatchWithIndex) into a basic
142 // WriteBatch.
143 virtual WriteBatch* GetWriteBatch() = 0;
144
145 // Records the state of the batch for future calls to RollbackToSavePoint().
146 // May be called multiple times to set multiple save points.
147 virtual void SetSavePoint() = 0;
148
149 // Remove all entries in this batch (Put, Merge, Delete, PutLogData) since the
150 // most recent call to SetSavePoint() and removes the most recent save point.
151 // If there is no previous call to SetSavePoint(), behaves the same as
152 // Clear().
153 virtual Status RollbackToSavePoint() = 0;
154
155 // Pop the most recent save point.
156 // If there is no previous call to SetSavePoint(), Status::NotFound()
157 // will be returned.
158 // Otherwise returns Status::OK().
159 virtual Status PopSavePoint() = 0;
160
161 // Sets the maximum size of the write batch in bytes. 0 means no limit.
162 virtual void SetMaxBytes(size_t max_bytes) = 0;
163};
164
165} // namespace ROCKSDB_NAMESPACE
