262TEST_F(ObjRegistryTest, TestFactoryCount) {
263 std::string msg;
264 auto grand = ObjectRegistry::Default();
265 auto local = ObjectRegistry::NewInstance();
266 std::unordered_set<std::string> grand_types, local_types;
267 std::vector<std::string> grand_names, local_names;
268
269 // Check how many types we have on startup.
270 // Grand should equal local
271 grand->GetFactoryTypes(&grand_types);
272 local->GetFactoryTypes(&local_types);
273 ASSERT_EQ(grand_types, local_types);
274 size_t grand_count = grand->GetFactoryCount(Env::Type());
275 size_t local_count = local->GetFactoryCount(Env::Type());
276
277 ASSERT_EQ(grand_count, local_count);
278 grand->GetFactoryNames(Env::Type(), &grand_names);
279 local->GetFactoryNames(Env::Type(), &local_names);
280 ASSERT_EQ(grand_names.size(), grand_count);
281 ASSERT_EQ(local_names.size(), local_count);
282 ASSERT_EQ(grand_names, local_names);
283
284 // Add an Env to the local registry.
285 // This will add one factory.
286 auto library = local->AddLibrary("local");
287 library->AddFactory<Env>(
288 "A", [](const std::string& /*uri*/, std::unique_ptr<Env>* /*guard */,
289 std::string* /* errmsg */) { return nullptr; });
290 ASSERT_EQ(local_count + 1, local->GetFactoryCount(Env::Type()));
291 ASSERT_EQ(grand_count, grand->GetFactoryCount(Env::Type()));
292 local->GetFactoryTypes(&local_types);
293 local->GetFactoryNames(Env::Type(), &local_names);
294 ASSERT_EQ(grand_names.size() + 1, local_names.size());
295 ASSERT_EQ(local_names.size(), local->GetFactoryCount(Env::Type()));
296
297 if (grand_count == 0) {
298 // There were no Env when we started. Should have one more type
299 // than previously
300 ASSERT_NE(grand_types, local_types);
301 ASSERT_EQ(grand_types.size() + 1, local_types.size());
302 } else {
303 // There was an Env type when we started. The types should match
304 ASSERT_EQ(grand_types, local_types);
305 }
306
307 // Add a MyCustomizable to the registry. This should be a new type
308 library->AddFactory<MyCustomizable>(
309 "MY", [](const std::string& /*uri*/,
310 std::unique_ptr<MyCustomizable>* /*guard */,
311 std::string* /* errmsg */) { return nullptr; });
312 ASSERT_EQ(local_count + 1, local->GetFactoryCount(Env::Type()));
313 ASSERT_EQ(grand_count, grand->GetFactoryCount(Env::Type()));
314 ASSERT_EQ(0U, grand->GetFactoryCount(MyCustomizable::Type()));
315 ASSERT_EQ(1U, local->GetFactoryCount(MyCustomizable::Type()));
316
317 local->GetFactoryNames(MyCustomizable::Type(), &local_names);
318 ASSERT_EQ(1U, local_names.size());
319 ASSERT_EQ(local_names[0], "MY");
320
321 local->GetFactoryTypes(&local_types);
322 ASSERT_EQ(grand_count == 0 ? 2 : grand_types.size() + 1, local_types.size());
323
324 // Add the same name again. We should now have 2 factories.
325 library->AddFactory<MyCustomizable>(
326 "MY", [](const std::string& /*uri*/,
327 std::unique_ptr<MyCustomizable>* /*guard */,
328 std::string* /* errmsg */) { return nullptr; });
329 local->GetFactoryNames(MyCustomizable::Type(), &local_names);
330 ASSERT_EQ(2U, local_names.size());
331}
