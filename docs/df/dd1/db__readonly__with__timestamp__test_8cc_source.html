40 VersionSet* const versions = dbfull()->GetVersionSet();
41 ASSERT_NE(versions, nullptr);
42
43 ColumnFamilyData* const cfd = versions->GetColumnFamilySet()->GetDefault();
44 ASSERT_NE(cfd, nullptr);
45
46 Version* const current = cfd->current();
47 ASSERT_NE(current, nullptr);
48
49 const VersionStorageInfo* const storage_info = current->storage_info();
50 ASSERT_NE(storage_info, nullptr);
51
52 // L0 has no files.
53 ASSERT_EQ(0, NumTableFilesAtLevel(0));
54
55 // All other levels have no files except the highest level with files.

62 ASSERT_TRUE(
63 storage_info->LevelFilesBrief(highest_non_empty_level).num_files > 0);

190TEST_F(DBReadOnlyTestWithTimestamp, IteratorAndGet) {
191 const int kNumKeysPerFile = 128;
192 const uint64_t kMaxKey = 1024;
193 Options options = CurrentOptions();
194 options.env = env_;
195 options.create_if_missing = true;
196 const size_t kTimestampSize = Timestamp(0, 0).size();
197 TestComparator test_cmp(kTimestampSize);
198 options.comparator = &test_cmp;
199 options.memtable_factory.reset(
200 test::NewSpecialSkipListFactory(kNumKeysPerFile));
201 DestroyAndReopen(options);
202 const std::vector<uint64_t> start_keys = {1, 0};
203 const std::vector<std::string> write_timestamps = {Timestamp(1, 0),
204 Timestamp(3, 0)};
205 const std::vector<std::string> read_timestamps = {Timestamp(2, 0),
206 Timestamp(4, 0)};
207 for (size_t i = 0; i < write_timestamps.size(); ++i) {
208 WriteOptions write_opts;
209 for (uint64_t key = start_keys[i]; key <= kMaxKey; ++key) {
210 Status s = db_->Put(write_opts, Key1(key), write_timestamps[i],
211 "value" + std::to_string(i));
212 ASSERT_OK(s);
213 }
214 }
215
216 // Reopen the database in read only mode to test its timestamp support.
217 Close();
218 ASSERT_OK(ReadOnlyReopen(options));
219
220 auto get_value_and_check = [](DB* db, ReadOptions read_opts, Slice key,
221 Slice expected_value, std::string expected_ts) {
222 std::string value_from_get;
223 std::string timestamp;
224 ASSERT_OK(db->Get(read_opts, key.ToString(), &value_from_get, &timestamp));
225 ASSERT_EQ(expected_value, value_from_get);
226 ASSERT_EQ(expected_ts, timestamp);
227 };
228 for (size_t i = 0; i < read_timestamps.size(); ++i) {
229 ReadOptions read_opts;
230 Slice read_ts = read_timestamps[i];
231 read_opts.timestamp = &read_ts;
232 std::unique_ptr<Iterator> it(db_->NewIterator(read_opts));
233 int count = 0;
234 uint64_t key = 0;
235 // Forward iterate.
236 for (it->Seek(Key1(0)), key = start_keys[i]; it->Valid();
237 it->Next(), ++count, ++key) {
238 CheckIterUserEntry(it.get(), Key1(key), kTypeValue,
239 "value" + std::to_string(i), write_timestamps[i]);
240 get_value_and_check(db_, read_opts, it->key(), it->value(),
241 write_timestamps[i]);
242 }
243 ASSERT_OK(it->status());
244 size_t expected_count = kMaxKey - start_keys[i] + 1;
245 ASSERT_EQ(expected_count, count);
246
247 // Backward iterate.
248 count = 0;
249 for (it->SeekForPrev(Key1(kMaxKey)), key = kMaxKey; it->Valid();
250 it->Prev(), ++count, --key) {
251 CheckIterUserEntry(it.get(), Key1(key), kTypeValue,
252 "value" + std::to_string(i), write_timestamps[i]);
253 get_value_and_check(db_, read_opts, it->key(), it->value(),
254 write_timestamps[i]);
255 }
256 ASSERT_OK(it->status());
257 ASSERT_EQ(static_cast<size_t>(kMaxKey) - start_keys[i] + 1, count);
258
259 // SeekToFirst()/SeekToLast() with lower/upper bounds.
260 // Then iter with lower and upper bounds.
261 uint64_t l = 0;
262 uint64_t r = kMaxKey + 1;
263 while (l < r) {
264 std::string lb_str = Key1(l);
265 Slice lb = lb_str;
266 std::string ub_str = Key1(r);
267 Slice ub = ub_str;
268 read_opts.iterate_lower_bound = &lb;
269 read_opts.iterate_upper_bound = &ub;
270 it.reset(db_->NewIterator(read_opts));
271 for (it->SeekToFirst(), key = std::max(l, start_keys[i]), count = 0;
272 it->Valid(); it->Next(), ++key, ++count) {
273 CheckIterUserEntry(it.get(), Key1(key), kTypeValue,
274 "value" + std::to_string(i), write_timestamps[i]);
275 get_value_and_check(db_, read_opts, it->key(), it->value(),
276 write_timestamps[i]);
277 }
278 ASSERT_OK(it->status());
279 ASSERT_EQ(r - std::max(l, start_keys[i]), count);
280
281 for (it->SeekToLast(), key = std::min(r, kMaxKey + 1), count = 0;
282 it->Valid(); it->Prev(), --key, ++count) {
283 CheckIterUserEntry(it.get(), Key1(key - 1), kTypeValue,
284 "value" + std::to_string(i), write_timestamps[i]);
285 get_value_and_check(db_, read_opts, it->key(), it->value(),
286 write_timestamps[i]);
287 }
288 ASSERT_OK(it->status());
289 l += (kMaxKey / 100);
290 r -= (kMaxKey / 100);
291 }
292 }
293 Close();
294}

345 Options options = CurrentOptions();
346 options.env = env_;

367

374
375 // Get()
376 std::string value;
