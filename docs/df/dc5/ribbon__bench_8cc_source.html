14struct KeyMaker {
15 explicit KeyMaker(size_t avg_size)
16 : smallest_size_(avg_size),
17 buf_size_(avg_size + 11), // pad to vary key size and alignment
18 buf_(new char[buf_size_]) {
19 memset(buf_.get(), 0, buf_size_);
20 assert(smallest_size_ > 8);
21 }
22 size_t smallest_size_;
23 size_t buf_size_;
24 std::unique_ptr<char[]> buf_;
25
26 // Returns a unique(-ish) key based on the given parameter values. Each
27 // call returns a Slice from the same buffer so previously returned
28 // Slices should be considered invalidated.
29 Slice Get(uint32_t filter_num, uint32_t val_num) const {
30 size_t start = val_num % 4;
31 size_t len = smallest_size_;
32 // To get range [avg_size - 2, avg_size + 2]
33 // use range [smallest_size, smallest_size + 4]
34 len += FastRange32((val_num >> 5) * 1234567891, 5);
35 char *data = buf_.get() + start;
36 // Populate key data such that all data makes it into a key of at
37 // least 8 bytes. We also don't want all the within-filter key
38 // variance confined to a contiguous 32 bits, because then a 32 bit
39 // hash function can "cheat" the false positive rate by
40 // approximating a perfect hash.
41 EncodeFixed32(data, val_num);
42 EncodeFixed32(data + 4, filter_num + val_num);
43 // ensure clearing leftovers from different alignment
44 EncodeFixed32(data + 8, 0);
45 return {data, len};
46 }
47};

49// benchmark arguments:
50// 0. filter impl (like filter_bench -impl)
51// 1. filter config bits_per_key
52// 2. average data key length
53// 3. data entry number
54static void CustomArguments(benchmark::internal::Benchmark *b) {
55 const auto kImplCount =
56 static_cast<int>(BloomLikeFilterPolicy::GetAllFixedImpls().size());
57 for (int filter_impl = 0; filter_impl < kImplCount; ++filter_impl) {
58 for (int bits_per_key : {10, 20}) {
59 for (int key_len_avg : {10, 100}) {
60 for (int64_t entry_num : {1 << 10, 1 << 20}) {
61 b->Args({filter_impl, bits_per_key, key_len_avg, entry_num});
62 }
63 }
64 }
65 }
66 b->ArgNames({"filter_impl", "bits_per_key", "key_len_avg", "entry_num"});
67}

69static void FilterBuild(benchmark::State &state) {
70 // setup data
71 auto filter = BloomLikeFilterPolicy::Create(
72 BloomLikeFilterPolicy::GetAllFixedImpls().at(state.range(0)),
73 static_cast<double>(state.range(1)));
74 auto tester = std::make_unique<mock::MockBlockBasedTableTester>(filter);
75 KeyMaker km(state.range(2));
76 std::unique_ptr<const char[]> owner;
77 const int64_t kEntryNum = state.range(3);
78 auto rnd = Random32(12345);
79 uint32_t filter_num = rnd.Next();
80 // run the test
81 for (auto _ : state) {
82 std::unique_ptr<FilterBitsBuilder> builder(tester->GetBuilder());
83 for (uint32_t i = 0; i < kEntryNum; i++) {
84 builder->AddKey(km.Get(filter_num, i));
85 }
86 auto ret = builder->Finish(&owner);
87 state.counters["size"] = static_cast<double>(ret.size());
88 }
89}
90BENCHMARK(FilterBuild)->Apply(CustomArguments);
