44// Metadata trailer size for built-in filters. (This is separate from
45// block-based table block trailer.)
46//
47// Originally this was 1 byte for num_probes and 4 bytes for number of
48// cache lines in the Bloom filter, but now the first trailer byte is
49// usually an implementation marker and remaining 4 bytes have various
50// meanings.
51static constexpr uint32_t kMetadataLen = 5;

77 void AddKey(const Slice& key) override {
78 uint64_t hash = GetSliceHash64(key);
79 // Especially with prefixes, it is common to have repetition,
80 // though only adjacent repetition, which we want to immediately
81 // recognize and collapse for estimating true filter space
82 // requirements.
83 if (hash_entries_info_.entries.empty() ||
84 hash != hash_entries_info_.entries.back()) {
85 AddHash(hash);
86 }
87 }
88
89 void AddKeyAndAlt(const Slice& key, const Slice& alt) override {
90 uint64_t key_hash = GetSliceHash64(key);
91 uint64_t alt_hash = GetSliceHash64(alt);
92 std::optional<uint64_t> prev_key_hash;
93 std::optional<uint64_t> prev_alt_hash = hash_entries_info_.prev_alt_hash;
94
95 if (!hash_entries_info_.entries.empty()) {
96 prev_key_hash = hash_entries_info_.entries.back();
97 }
98
99#ifdef ROCKSDB_VALGRIND_RUN
100 // Valgrind can report uninitialized FPs on std::optional usage. See e.g.
101 // https://stackoverflow.com/q/51616179
102 if (!prev_key_hash.has_value()) {
103 std::memset((void*)&prev_key_hash, 0, sizeof(prev_key_hash));
104 prev_key_hash.reset();
105 }
106 if (!prev_alt_hash.has_value()) {
107 std::memset((void*)&prev_alt_hash, 0, sizeof(prev_key_hash));
108 prev_alt_hash.reset();
109 }
110#endif
111
112 // Add alt first, so that entries.back() always contains previous key
113 // ASSUMING a change from one alt to the next implies a change to
114 // corresponding key
115 if (alt_hash != prev_alt_hash && alt_hash != key_hash &&
116 alt_hash != prev_key_hash) {
117 AddHash(alt_hash);
118 }
119 // Overwrite prev_alt_hash for cases like alt_hash == key_hash
120 hash_entries_info_.prev_alt_hash = alt_hash;
121 // NOTE: checking key_hash != prev_alt_hash for cases like
122 // key == prefix(key) at the end of a prefix grouping as in reverse
123 // byte-wise comparator
124 if (key_hash != prev_key_hash && key_hash != prev_alt_hash) {
125 AddHash(key_hash);
126 }
127 }
128
129 size_t EstimateEntriesAdded() override {
130 return hash_entries_info_.entries.size();
131 }

133 Status MaybePostVerify(const Slice& filter_content) override;

138 // Number of hash entries to accumulate before charging their memory usage to
139 // the cache when cache charging is available
140 static const std::size_t kUint64tHashEntryCacheResBucketSize =
141 CacheReservationManagerImpl<
142 CacheEntryRole::kFilterConstruction>::GetDummyEntrySize() /
143 sizeof(uint64_t);

145 void AddHash(uint64_t hash) {
146 if (detect_filter_construct_corruption_) {
147 hash_entries_info_.xor_checksum ^= hash;
148 }
149 hash_entries_info_.entries.push_back(hash);
150 if (cache_res_mgr_ &&
151 // Traditional rounding to whole bucket size
152 ((hash_entries_info_.entries.size() %
153 kUint64tHashEntryCacheResBucketSize) ==
154 kUint64tHashEntryCacheResBucketSize / 2)) {
155 hash_entries_info_.cache_res_bucket_handles.emplace_back(nullptr);
156 Status s = cache_res_mgr_->MakeCacheReservation(
157 kUint64tHashEntryCacheResBucketSize * sizeof(hash),
158 &hash_entries_info_.cache_res_bucket_handles.back());
159 s.PermitUncheckedError();
160 }
161 }

171 virtual size_t RoundDownUsableSpace(size_t available_size) = 0;

173 // To choose size using malloc_usable_size, we have to actually allocate.
174 size_t AllocateMaybeRounding(size_t target_len_with_metadata,
175 size_t num_entries,
176 std::unique_ptr<char[]>* buf) {
177 // Return value set to a default; overwritten in some cases
178 size_t rv = target_len_with_metadata;
179#ifdef ROCKSDB_MALLOC_USABLE_SIZE
180 if (aggregate_rounding_balance_ != nullptr) {
181 // Do optimize_filters_for_memory, using malloc_usable_size.
182 // Approach: try to keep FP rate balance better than or on
183 // target (negative aggregate_rounding_balance_). We can then select a
184 // lower bound filter size (within reasonable limits) that gets us as
185 // close to on target as possible. We request allocation for that filter
186 // size and use malloc_usable_size to "round up" to the actual
187 // allocation size.
188
189 // Although it can be considered bad practice to use malloc_usable_size
190 // to access an object beyond its original size, this approach should be
191 // quite general: working for all allocators that properly support
192 // malloc_usable_size.
193
194 // Race condition on balance is OK because it can only cause temporary
195 // skew in rounding up vs. rounding down, as long as updates are atomic
196 // and relative.
197 int64_t balance = aggregate_rounding_balance_->load();
198
199 double target_fp_rate =
200 EstimatedFpRate(num_entries, target_len_with_metadata);
201 double rv_fp_rate = target_fp_rate;
202
203 if (balance < 0) {
204 // See formula for BloomFilterPolicy::aggregate_rounding_balance_
205 double for_balance_fp_rate =
206 -balance / double{0x100000000} + target_fp_rate;
207
208 // To simplify, we just try a few modified smaller sizes. This also
209 // caps how much we vary filter size vs. target, to avoid outlier
210 // behavior from excessive variance.
211 size_t target_len = target_len_with_metadata - kMetadataLen;
212 assert(target_len < target_len_with_metadata); // check underflow
213 for (uint64_t maybe_len_rough :
214 {uint64_t{3} * target_len / 4, uint64_t{13} * target_len / 16,
215 uint64_t{7} * target_len / 8, uint64_t{15} * target_len / 16}) {
216 size_t maybe_len_with_metadata =
217 RoundDownUsableSpace(maybe_len_rough + kMetadataLen);
218 double maybe_fp_rate =
219 EstimatedFpRate(num_entries, maybe_len_with_metadata);
220 if (maybe_fp_rate <= for_balance_fp_rate) {
221 rv = maybe_len_with_metadata;
222 rv_fp_rate = maybe_fp_rate;
223 break;
224 }
225 }
226 }
227
228 // Filter blocks are loaded into block cache with their block trailer.
229 // We need to make sure that's accounted for in choosing a
230 // fragmentation-friendly size.
231 const size_t kExtraPadding = BlockBasedTable::kBlockTrailerSize;
232 size_t requested = rv + kExtraPadding;
233
234 // Allocate and get usable size
235 buf->reset(new char[requested]);
236 size_t usable = malloc_usable_size(buf->get());
237
238 if (usable - usable / 4 > requested) {
239 // Ratio greater than 4/3 is too much for utilizing, if it's
240 // not a buggy or mislinked malloc_usable_size implementation.
241 // Non-linearity of FP rates with bits/key means rapidly
242 // diminishing returns in overall accuracy for additional
243 // storage on disk.
244 // Nothing to do, except assert that the result is accurate about
245 // the usable size. (Assignment never used.)
246 assert(((*buf)[usable - 1] = 'x'));
247 } else if (usable > requested) {
248 rv = RoundDownUsableSpace(usable - kExtraPadding);
249 assert(rv <= usable - kExtraPadding);
250 rv_fp_rate = EstimatedFpRate(num_entries, rv);
251 } else {
252 // Too small means bad malloc_usable_size
253 assert(usable == requested);
254 }
255 memset(buf->get(), 0, rv);
256
257 // Update balance
258 int64_t diff = static_cast<int64_t>((rv_fp_rate - target_fp_rate) *
259 double{0x100000000});
260 *aggregate_rounding_balance_ += diff;
261 } else {
262 buf->reset(new char[rv]());
263 }
264#else
265 (void)num_entries;
266 buf->reset(new char[rv]());
267#endif // ROCKSDB_MALLOC_USABLE_SIZE
268 return rv;
269 }

271 // TODO: Ideally we want to verify the hash entry
272 // as it is added to the filter and eliminate this function
273 // for speeding up and leaving fewer spaces for undetected memory/CPU
274 // corruption. For Ribbon Filter, it's bit harder.
275 // Possible solution:
276 // pass a custom iterator that tracks the xor checksum as
277 // it iterates to ResetAndFindSeedToSolve
278 Status MaybeVerifyHashEntriesChecksum() {
279 if (!detect_filter_construct_corruption_) {
280 return Status::OK();
281 }
282
283 uint64_t actual_hash_entries_xor_checksum = 0;
284 for (uint64_t h : hash_entries_info_.entries) {
285 actual_hash_entries_xor_checksum ^= h;
286 }
287
288 if (actual_hash_entries_xor_checksum == hash_entries_info_.xor_checksum) {
289 return Status::OK();
290 } else {
291 // Since these hash entries are corrupted and they will not be used
292 // anymore, we can reset them and release memory.
293 ResetEntries();
294 return Status::Corruption("Filter's hash entries checksum mismatched");
295 }
296 }
297
298 // See BloomFilterPolicy::aggregate_rounding_balance_. If nullptr,
299 // always "round up" like historic behavior.
300 std::atomic<int64_t>* aggregate_rounding_balance_;
301
302 // For reserving memory used in (new) Bloom and Ribbon Filter construction
303 std::shared_ptr<CacheReservationManager> cache_res_mgr_;
304
305 // For managing cache charge for final filter in (new) Bloom and Ribbon
306 // Filter construction
307 std::deque<std::unique_ptr<CacheReservationManager::CacheReservationHandle>>
308 final_filter_cache_res_handles_;
309
310 bool detect_filter_construct_corruption_;
311
312 struct HashEntriesInfo {
313 // A deque avoids unnecessary copying of already-saved values
314 // and has near-minimal peak memory use.
315 std::deque<uint64_t> entries;
316
317 // If cache_res_mgr_ != nullptr,
318 // it manages cache charge for buckets of hash entries in (new) Bloom
319 // or Ribbon Filter construction.
320 // Otherwise, it is empty.
321 std::deque<std::unique_ptr<CacheReservationManager::CacheReservationHandle>>
322 cache_res_bucket_handles;
323
324 // If detect_filter_construct_corruption_ == true,
325 // it records the xor checksum of hash entries.
326 // Otherwise, it is 0.
327 uint64_t xor_checksum = 0;
328
329 // A single-element cache to help AddKeyAndAlt
330 std::optional<uint64_t> prev_alt_hash;
331
332 void Swap(HashEntriesInfo* other) {
333 assert(other != nullptr);
334 std::swap(entries, other->entries);
335 std::swap(cache_res_bucket_handles, other->cache_res_bucket_handles);
336 std::swap(xor_checksum, other->xor_checksum);
337 std::swap(prev_alt_hash, other->prev_alt_hash);
338 }
339
340 void Reset() {
341 entries.clear();
342 cache_res_bucket_handles.clear();
343 xor_checksum = 0;
344 prev_alt_hash = {};
345 }
346 };
347
348 HashEntriesInfo hash_entries_info_;
349};

354// See description in FastLocalBloomImpl
355class FastLocalBloomBitsBuilder : public XXPH3FilterBitsBuilder {
356 public:
357 // Non-null aggregate_rounding_balance implies optimize_filters_for_memory
358 explicit FastLocalBloomBitsBuilder(
359 const int millibits_per_key,
360 std::atomic<int64_t>* aggregate_rounding_balance,
361 std::shared_ptr<CacheReservationManager> cache_res_mgr,
362 bool detect_filter_construct_corruption)
363 : XXPH3FilterBitsBuilder(aggregate_rounding_balance, cache_res_mgr,
364 detect_filter_construct_corruption),
365 millibits_per_key_(millibits_per_key) {
366 assert(millibits_per_key >= 1000);
367 }
368
369 // No Copy allowed
370 FastLocalBloomBitsBuilder(const FastLocalBloomBitsBuilder&) = delete;
371 void operator=(const FastLocalBloomBitsBuilder&) = delete;
372
373 ~FastLocalBloomBitsBuilder() override = default;
374
375 using FilterBitsBuilder::Finish;
376
377 Slice Finish(std::unique_ptr<const char[]>* buf) override {
378 return Finish(buf, nullptr);
379 }
380
381 Slice Finish(std::unique_ptr<const char[]>* buf, Status* status) override {
382 size_t num_entries = hash_entries_info_.entries.size();
383 if (num_entries == 0) {
384 // This case migrated from FullFilterBlockBuilder::Finish
385 if (status) {
386 *status = Status::OK();
387 }
388 return FinishAlwaysFalse(buf);
389 }
390
391 size_t len_with_metadata = CalculateSpace(num_entries);
392
393 std::unique_ptr<char[]> mutable_buf;
394 std::unique_ptr<CacheReservationManager::CacheReservationHandle>
395 final_filter_cache_res_handle;
396 len_with_metadata =
397 AllocateMaybeRounding(len_with_metadata, num_entries, &mutable_buf);
398 // Cache charging for mutable_buf
399 if (cache_res_mgr_) {
400 Status s = cache_res_mgr_->MakeCacheReservation(
401 len_with_metadata * sizeof(char), &final_filter_cache_res_handle);
402 s.PermitUncheckedError();
403 }
404
405 assert(mutable_buf);
406 assert(len_with_metadata >= kMetadataLen);
407
408 // Max size supported by implementation
409 assert(len_with_metadata <= 0xffffffffU);
410
411 // Compute num_probes after any rounding / adjustments
412 int num_probes = GetNumProbes(num_entries, len_with_metadata);
413
414 uint32_t len = static_cast<uint32_t>(len_with_metadata - kMetadataLen);
415 if (len > 0) {
416 TEST_SYNC_POINT_CALLBACK(
417 "XXPH3FilterBitsBuilder::Finish::"
418 "TamperHashEntries",
419 &hash_entries_info_.entries);
420 AddAllEntries(mutable_buf.get(), len, num_probes);
421 Status verify_hash_entries_checksum_status =
422 MaybeVerifyHashEntriesChecksum();
423 if (!verify_hash_entries_checksum_status.ok()) {
424 if (status) {
425 *status = verify_hash_entries_checksum_status;
426 }
427 return FinishAlwaysTrue(buf);
428 }
429 }
430
431 bool keep_entries_for_postverify = detect_filter_construct_corruption_;
432 if (!keep_entries_for_postverify) {
433 ResetEntries();
434 }
435
436 // See BloomFilterPolicy::GetBloomBitsReader re: metadata
437 // -1 = Marker for newer Bloom implementations
438 mutable_buf[len] = static_cast<char>(-1);
439 // 0 = Marker for this sub-implementation
440 mutable_buf[len + 1] = static_cast<char>(0);
441 // num_probes (and 0 in upper bits for 64-byte block size)
442 mutable_buf[len + 2] = static_cast<char>(num_probes);
443 // rest of metadata stays zero
444
445 auto TEST_arg_pair __attribute__((__unused__)) =
446 std::make_pair(&mutable_buf, len_with_metadata);
447 TEST_SYNC_POINT_CALLBACK("XXPH3FilterBitsBuilder::Finish::TamperFilter",
448 &TEST_arg_pair);
449
450 Slice rv(mutable_buf.get(), len_with_metadata);
451 *buf = std::move(mutable_buf);
452 final_filter_cache_res_handles_.push_back(
453 std::move(final_filter_cache_res_handle));
454 if (status) {
455 *status = Status::OK();
456 }
457 return rv;
458 }

460 size_t ApproximateNumEntries(size_t bytes) override {
461 size_t bytes_no_meta =
462 bytes >= kMetadataLen ? RoundDownUsableSpace(bytes) - kMetadataLen : 0;
463 return static_cast<size_t>(uint64_t{8000} * bytes_no_meta /
464 millibits_per_key_);
465 }
466
467 size_t CalculateSpace(size_t num_entries) override {
468 // If not for cache line blocks in the filter, what would the target
469 // length in bytes be?
470 size_t raw_target_len = static_cast<size_t>(
471 (uint64_t{num_entries} * millibits_per_key_ + 7999) / 8000);
472
473 if (raw_target_len >= size_t{0xffffffc0}) {
474 // Max supported for this data structure implementation
475 raw_target_len = size_t{0xffffffc0};
476 }
477
478 // Round up to nearest multiple of 64 (block size). This adjustment is
479 // used for target FP rate only so that we don't receive complaints about
480 // lower FP rate vs. historic Bloom filter behavior.
481 return ((raw_target_len + 63) & ~size_t{63}) + kMetadataLen;
482 }
483
484 double EstimatedFpRate(size_t keys, size_t len_with_metadata) override {
485 if (len_with_metadata <= kMetadataLen) {
486 return keys > 0 ? 1.0 : 0.0;
487 }
488 int num_probes = GetNumProbes(keys, len_with_metadata);
489 return FastLocalBloomImpl::EstimatedFpRate(
490 keys, len_with_metadata - kMetadataLen, num_probes, /*hash bits*/ 64);
491 }
492
493 protected:
494 size_t RoundDownUsableSpace(size_t available_size) override {
495 size_t rv = available_size - kMetadataLen;
496
497 if (rv >= size_t{0xffffffc0}) {
498 // Max supported for this data structure implementation
499 rv = size_t{0xffffffc0};
500 }
501
502 // round down to multiple of 64 (block size)
503 rv &= ~size_t{63};
504
505 return rv + kMetadataLen;
506 }
507
508 private:
509 // Compute num_probes after any rounding / adjustments
510 int GetNumProbes(size_t keys, size_t len_with_metadata) {
511 uint64_t millibits = uint64_t{len_with_metadata - kMetadataLen} * 8000;
512 int actual_millibits_per_key =
513 static_cast<int>(millibits / std::max(keys, size_t{1}));
514 // BEGIN XXX/TODO(peterd): preserving old/default behavior for now to
515 // minimize unit test churn. Remove this some time.
516 if (!aggregate_rounding_balance_) {
517 actual_millibits_per_key = millibits_per_key_;
518 }
519 // END XXX/TODO
520 return FastLocalBloomImpl::ChooseNumProbes(actual_millibits_per_key);
521 }

523 void AddAllEntries(char* data, uint32_t len, int num_probes) {
524 // Simple version without prefetching:
525 //
526 // for (auto h : hash_entries_info_.entries) {
527 // FastLocalBloomImpl::AddHash(Lower32of64(h), Upper32of64(h), len,
528 // num_probes, data);
529 // }
530
531 const size_t num_entries = hash_entries_info_.entries.size();
532 constexpr size_t kBufferMask = 7;
533 static_assert(((kBufferMask + 1) & kBufferMask) == 0,
534 "Must be power of 2 minus 1");
535
536 std::array<uint32_t, kBufferMask + 1> hashes;
537 std::array<uint32_t, kBufferMask + 1> byte_offsets;
538
539 // Prime the buffer
540 size_t i = 0;
541 std::deque<uint64_t>::iterator hash_entries_it =
542 hash_entries_info_.entries.begin();
543 for (; i <= kBufferMask && i < num_entries; ++i) {
544 uint64_t h = *hash_entries_it;
545 FastLocalBloomImpl::PrepareHash(Lower32of64(h), len, data,
546 /*out*/ &byte_offsets[i]);
547 hashes[i] = Upper32of64(h);
548 ++hash_entries_it;
549 }
550
551 // Process and buffer
552 for (; i < num_entries; ++i) {
553 uint32_t& hash_ref = hashes[i & kBufferMask];
554 uint32_t& byte_offset_ref = byte_offsets[i & kBufferMask];
555 // Process (add)
556 FastLocalBloomImpl::AddHashPrepared(hash_ref, num_probes,
557 data + byte_offset_ref);
558 // And buffer
559 uint64_t h = *hash_entries_it;
560 FastLocalBloomImpl::PrepareHash(Lower32of64(h), len, data,
561 /*out*/ &byte_offset_ref);
562 hash_ref = Upper32of64(h);
563 ++hash_entries_it;
564 }
565
566 // Finish processing
567 for (i = 0; i <= kBufferMask && i < num_entries; ++i) {
568 FastLocalBloomImpl::AddHashPrepared(hashes[i], num_probes,
569 data + byte_offsets[i]);
570 }
571 }
572
573 // Target allocation per added key, in thousandths of a bit.
574 int millibits_per_key_;
575};

676 Slice Finish(std::unique_ptr<const char[]>* buf, Status* status) override {
677 if (hash_entries_info_.entries.size() > kMaxRibbonEntries) {
678 ROCKS_LOG_WARN(
679 info_log_, "Too many keys for Ribbon filter: %llu",
680 static_cast<unsigned long long>(hash_entries_info_.entries.size()));

686 // Save a conditional in Ribbon queries by using alternate reader
687 // for zero entries added.
688 if (status) {
689 *status = Status::OK();

691 return FinishAlwaysFalse(buf);
692 }
693 uint32_t num_entries =
694 static_cast<uint32_t>(hash_entries_info_.entries.size());
695 uint32_t num_slots;
696 size_t len_with_metadata;
697
698 CalculateSpaceAndSlots(num_entries, &len_with_metadata, &num_slots);
699
700 // Bloom fall-back indicator
701 if (num_slots == 0) {
702 SwapEntriesWith(&bloom_fallback_);
703 assert(hash_entries_info_.entries.empty());
704 return bloom_fallback_.Finish(buf, status);
705 }
706
707 uint32_t entropy = 0;
708 if (!hash_entries_info_.entries.empty()) {
709 entropy = Lower32of64(hash_entries_info_.entries.front());
710 }
711
712 BandingType banding;
713 std::size_t bytes_banding = ribbon::StandardBanding<
714 Standard128RibbonTypesAndSettings>::EstimateMemoryUsage(num_slots);
715 Status status_banding_cache_res = Status::OK();
716
717 // Cache charging for banding
718 std::unique_ptr<CacheReservationManager::CacheReservationHandle>
719 banding_res_handle;
720 if (cache_res_mgr_) {
721 status_banding_cache_res = cache_res_mgr_->MakeCacheReservation(
722 bytes_banding, &banding_res_handle);
723 }
724
725 if (status_banding_cache_res.IsMemoryLimit()) {
726 ROCKS_LOG_WARN(info_log_,
727 "Cache charging for Ribbon filter banding failed due "
728 "to cache full");
729 SwapEntriesWith(&bloom_fallback_);
730 assert(hash_entries_info_.entries.empty());
731 // Release cache for banding since the banding won't be allocated
732 banding_res_handle.reset();
733 return bloom_fallback_.Finish(buf, status);
734 }
735
736 TEST_SYNC_POINT_CALLBACK(
737 "XXPH3FilterBitsBuilder::Finish::"
738 "TamperHashEntries",
739 &hash_entries_info_.entries);
740
741 bool success = banding.ResetAndFindSeedToSolve(
742 num_slots, hash_entries_info_.entries.begin(),
743 hash_entries_info_.entries.end(),
744 /*starting seed*/ entropy & 255, /*seed mask*/ 255);
745 if (!success) {
746 ROCKS_LOG_WARN(
747 info_log_, "Too many re-seeds (256) for Ribbon filter, %llu / %llu",
748 static_cast<unsigned long long>(hash_entries_info_.entries.size()),
749 static_cast<unsigned long long>(num_slots));
750 SwapEntriesWith(&bloom_fallback_);
751 assert(hash_entries_info_.entries.empty());
752 return bloom_fallback_.Finish(buf, status);
753 }
754
755 Status verify_hash_entries_checksum_status =
756 MaybeVerifyHashEntriesChecksum();
757 if (!verify_hash_entries_checksum_status.ok()) {
758 ROCKS_LOG_WARN(info_log_, "Verify hash entries checksum error: %s",
759 verify_hash_entries_checksum_status.getState());
760 if (status) {
761 *status = verify_hash_entries_checksum_status;
762 }
763 return FinishAlwaysTrue(buf);
764 }
765
766 bool keep_entries_for_postverify = detect_filter_construct_corruption_;
767 if (!keep_entries_for_postverify) {
768 ResetEntries();
769 }
770
771 uint32_t seed = banding.GetOrdinalSeed();
772 assert(seed < 256);
773
774 std::unique_ptr<char[]> mutable_buf;
775 std::unique_ptr<CacheReservationManager::CacheReservationHandle>
776 final_filter_cache_res_handle;
777 len_with_metadata =
778 AllocateMaybeRounding(len_with_metadata, num_entries, &mutable_buf);
779 // Cache charging for mutable_buf
780 if (cache_res_mgr_) {
781 Status s = cache_res_mgr_->MakeCacheReservation(
782 len_with_metadata * sizeof(char), &final_filter_cache_res_handle);
783 s.PermitUncheckedError();
784 }
785
786 SolnType soln(mutable_buf.get(), len_with_metadata);
787 soln.BackSubstFrom(banding);
788 uint32_t num_blocks = soln.GetNumBlocks();
789 // This should be guaranteed:
790 // num_entries < 2^30
791 // => (overhead_factor < 2.0)
792 // num_entries * overhead_factor == num_slots < 2^31
793 // => (num_blocks = num_slots / 128)
794 // num_blocks < 2^24
795 assert(num_blocks < 0x1000000U);
796
797 // See BloomFilterPolicy::GetBloomBitsReader re: metadata
798 // -2 = Marker for Standard128 Ribbon
799 mutable_buf[len_with_metadata - 5] = static_cast<char>(-2);
800 // Hash seed
801 mutable_buf[len_with_metadata - 4] = static_cast<char>(seed);
802 // Number of blocks, in 24 bits
803 // (Along with bytes, we can derive other settings)
804 mutable_buf[len_with_metadata - 3] = static_cast<char>(num_blocks & 255);
805 mutable_buf[len_with_metadata - 2] =
806 static_cast<char>((num_blocks >> 8) & 255);
807 mutable_buf[len_with_metadata - 1] =
808 static_cast<char>((num_blocks >> 16) & 255);
809
810 auto TEST_arg_pair __attribute__((__unused__)) =
811 std::make_pair(&mutable_buf, len_with_metadata);
812 TEST_SYNC_POINT_CALLBACK("XXPH3FilterBitsBuilder::Finish::TamperFilter",
813 &TEST_arg_pair);
814

816 *buf = std::move(mutable_buf);
817 final_filter_cache_res_handles_.push_back(
818 std::move(final_filter_cache_res_handle));
819 if (status) {
820 *status = Status::OK();
821 }
822 return rv;

826 // And note this implementation does not support num_entries or num_slots
827 // beyond uint32_t; see kMaxRibbonEntries.
828 void CalculateSpaceAndSlots(size_t num_entries,
829 size_t* target_len_with_metadata,
830 uint32_t* num_slots) {
831 if (num_entries > kMaxRibbonEntries) {
832 // More entries than supported by this Ribbon
833 *num_slots = 0; // use Bloom

871 // This is a somewhat ugly but reasonably fast and reasonably accurate
872 // reversal of CalculateSpace.
873 size_t ApproximateNumEntries(size_t bytes) override {
874 size_t len_no_metadata =
875 RoundDownUsableSpace(std::max(bytes, size_t{kMetadataLen})) -
876 kMetadataLen;
877
878 if (!(desired_one_in_fp_rate_ > 1.0)) {
879 // Effectively asking for 100% FP rate, or NaN etc.
880 // Note that NaN is neither < 1.0 nor > 1.0
881 return kMaxRibbonEntries;
882 }
883
884 // Find a slight under-estimate for actual average bits per slot
885 double min_real_bits_per_slot;
886 if (desired_one_in_fp_rate_ >= 1.0 + std::numeric_limits<uint32_t>::max()) {
887 // Max of 32 solution columns (result bits)
888 min_real_bits_per_slot = 32.0;
889 } else {
890 // Account for mix of b and b+1 solution columns being slightly
891 // suboptimal vs. ideal log2(1/fp_rate) bits.
892 uint32_t rounded = static_cast<uint32_t>(desired_one_in_fp_rate_);
893 int upper_bits_per_key = 1 + FloorLog2(rounded);
894 double fp_rate_for_upper = std::pow(2.0, -upper_bits_per_key);
895 double portion_lower =
896 (1.0 / desired_one_in_fp_rate_ - fp_rate_for_upper) /
897 fp_rate_for_upper;
898 min_real_bits_per_slot = upper_bits_per_key - portion_lower;
899 assert(min_real_bits_per_slot > 0.0);
900 assert(min_real_bits_per_slot <= 32.0);
901 }
902
903 // An overestimate, but this should only be O(1) slots away from truth.
904 double max_slots = len_no_metadata * 8.0 / min_real_bits_per_slot;
905
906 // Let's not bother accounting for overflow to Bloom filter
907 // (Includes NaN case)
908 if (!(max_slots < ConfigHelper::GetNumSlots(kMaxRibbonEntries))) {
909 return kMaxRibbonEntries;
910 }
911
912 // Set up for short iteration
913 uint32_t slots = static_cast<uint32_t>(max_slots);
914 slots = SolnType::RoundUpNumSlots(slots);
915
916 // Assert that we have a valid upper bound on slots
917 assert(SolnType::GetBytesForOneInFpRate(
918 SolnType::RoundUpNumSlots(slots + 1), desired_one_in_fp_rate_,
919 /*rounding*/ 0) > len_no_metadata);
920
921 // Iterate up to a few times to rather precisely account for small effects
922 for (int i = 0; slots > 0; ++i) {
923 size_t reqd_bytes =
924 SolnType::GetBytesForOneInFpRate(slots, desired_one_in_fp_rate_,
925 /*rounding*/ 0);
926 if (reqd_bytes <= len_no_metadata) {
927 break; // done
928 }
929 if (i >= 2) {
930 // should have been enough iterations
931 assert(false);
932 break;
933 }
934 slots = SolnType::RoundDownNumSlots(slots - 1);
935 }
936
937 uint32_t num_entries = ConfigHelper::GetNumToAdd(slots);
938
939 // Consider possible Bloom fallback for small filters
940 if (slots < 1024) {
941 size_t bloom = bloom_fallback_.ApproximateNumEntries(bytes);
942 if (bloom > num_entries) {
943 return bloom;
944 } else {
945 return num_entries;
946 }
947 } else {
948 return std::min(num_entries, kMaxRibbonEntries);
949 }
950 }
951
952 double EstimatedFpRate(size_t num_entries,
953 size_t len_with_metadata) override {
954 if (len_with_metadata <= kMetadataLen) {
955 return num_entries > 0 ? 1.0 : 0.0;
956 }
957 if (num_entries > kMaxRibbonEntries) {
958 // More entries than supported by this Ribbon
959 return bloom_fallback_.EstimatedFpRate(num_entries, len_with_metadata);
960 }
961 uint32_t num_slots =
962 NumEntriesToNumSlots(static_cast<uint32_t>(num_entries));
963 SolnType fake_soln(nullptr, len_with_metadata);
964 fake_soln.ConfigureForNumSlots(num_slots);
965 return fake_soln.ExpectedFpRate();
966 }

969 bool fall_back = (bloom_fallback_.EstimateEntriesAdded() > 0);
970 return fall_back ? bloom_fallback_.MaybePostVerify(filter_content)
971 : XXPH3FilterBitsBuilder::MaybePostVerify(filter_content);
972 }

974 protected:

976 size_t rv = available_size - kMetadataLen;
977
978 // round down to multiple of 16 (segment size)
979 rv &= ~size_t{15};

982 }

996 // is not an inherent limitation but does ensure somewhat graceful Bloom
997 // fallback for crazy high number of entries, since the Bloom implementation
998 // does not support number of bytes bigger than fits in 32 bits. This is
999 // within an order of magnitude of implementation limit on num_slots
1000 // fitting in 32 bits, and even closer for num_blocks fitting in 24 bits
1001 // (for filter metadata).

1168Slice LegacyBloomBitsBuilder::Finish(std::unique_ptr<const char[]>* buf) {
1169 size_t num_entries = hash_entries_.size();
1170 if (num_entries == 0) {
1171 // This case migrated from FullFilterBlockBuilder::Finish
1172 return FinishAlwaysFalse(buf);
1173 }
1174
1175 uint32_t total_bits, num_lines;
1176 char* data =
1177 ReserveSpace(static_cast<int>(num_entries), &total_bits, &num_lines);
1178 assert(data);
1179
1180 if (total_bits != 0 && num_lines != 0) {
1181 for (auto h : hash_entries_) {
1182 AddHash(h, data, num_lines, total_bits);
1183 }
1184
1185 // Check for excessive entries for 32-bit hash function
1186 if (num_entries >= /* minimum of 3 million */ 3000000U) {
1187 // More specifically, we can detect that the 32-bit hash function
1188 // is causing significant increase in FP rate by comparing current
1189 // estimated FP rate to what we would get with a normal number of
1190 // keys at same memory ratio.
1191 double est_fp_rate = LegacyBloomImpl::EstimatedFpRate(
1192 num_entries, total_bits / 8, num_probes_);
1193 double vs_fp_rate = LegacyBloomImpl::EstimatedFpRate(
1194 1U << 16, (1U << 16) * bits_per_key_ / 8, num_probes_);
1195
1196 if (est_fp_rate >= 1.50 * vs_fp_rate) {
1197 // For more details, see
1198 // https://github.com/facebook/rocksdb/wiki/RocksDB-Bloom-Filter
1199 ROCKS_LOG_WARN(
1200 info_log_,
1201 "Using legacy SST/BBT Bloom filter with excessive key count "
1202 "(%.1fM @ %dbpk), causing estimated %.1fx higher filter FP rate. "
1203 "Consider using new Bloom with format_version>=5, smaller SST "
1204 "file size, or partitioned filters.",
1205 num_entries / 1000000.0, bits_per_key_, est_fp_rate / vs_fp_rate);
1206 }
1207 }
1208 }
1209 // See BloomFilterPolicy::GetFilterBitsReader for metadata
1210 data[total_bits / 8] = static_cast<char>(num_probes_);
1211 EncodeFixed32(data + total_bits / 8 + 1, static_cast<uint32_t>(num_lines));
1212
1213 const char* const_data = data;
1214 buf->reset(const_data);
1215 hash_entries_.clear();
1216 prev_alt_hash_ = -1;
1217
1218 return Slice(data, total_bits / 8 + kMetadataLen);
1219}

1221size_t LegacyBloomBitsBuilder::ApproximateNumEntries(size_t bytes) {
1222 assert(bits_per_key_);
1223 assert(bytes > 0);
1224
1225 uint64_t total_bits_tmp = bytes * 8;
1226 // total bits, including temporary computations, cannot exceed 2^32
1227 // for compatibility
1228 total_bits_tmp = std::min(total_bits_tmp, uint64_t{0xffff0000});
1229
1230 uint32_t high = static_cast<uint32_t>(total_bits_tmp) /
1231 static_cast<uint32_t>(bits_per_key_) +
1232 1;
1233 uint32_t low = 1;
1234 uint32_t n = high;
1235 for (; n >= low; n--) {
1236 if (CalculateSpace(n) <= bytes) {
1237 break;
1238 }
1239 }
1240 return n;
1241}

1243uint32_t LegacyBloomBitsBuilder::GetTotalBitsForLocality(uint32_t total_bits) {
1244 uint32_t num_lines =
1245 (total_bits + CACHE_LINE_SIZE * 8 - 1) / (CACHE_LINE_SIZE * 8);
1246
1247 // Make num_lines an odd number to make sure more bits are involved
1248 // when determining which block.
1249 if (num_lines % 2 == 0) {
1250 num_lines++;
1251 }
1252 return num_lines * (CACHE_LINE_SIZE * 8);
1253}

1255uint32_t LegacyBloomBitsBuilder::CalculateSpace(size_t num_entries,
1256 uint32_t* total_bits,
1257 uint32_t* num_lines) {
1258 assert(bits_per_key_);
1259 if (num_entries != 0) {
1260 size_t total_bits_tmp = num_entries * bits_per_key_;
1261 // total bits, including temporary computations, cannot exceed 2^32
1262 // for compatibility
1263 total_bits_tmp = std::min(total_bits_tmp, size_t{0xffff0000});
1264
1265 *total_bits =
1266 GetTotalBitsForLocality(static_cast<uint32_t>(total_bits_tmp));
1267 *num_lines = *total_bits / (CACHE_LINE_SIZE * 8);
1268 assert(*total_bits > 0 && *total_bits % 8 == 0);
1269 } else {
1270 // filter is empty, just leave space for metadata
1271 *total_bits = 0;
1272 *num_lines = 0;
1273 }
1274
1275 // Reserve space for Filter
1276 uint32_t sz = *total_bits / 8;
1277 sz += kMetadataLen; // 4 bytes for num_lines, 1 byte for num_probes
1278 return sz;
1279}

1282 uint32_t* total_bits,
1283 uint32_t* num_lines) {
1284 uint32_t sz = CalculateSpace(num_entries, total_bits, num_lines);
1285 char* data = new char[sz];
1286 memset(data, 0, sz);
1287 return data;
1288}

1302class LegacyBloomBitsReader : public BuiltinFilterBitsReader {
1303 public:
1304 LegacyBloomBitsReader(const char* data, int num_probes, uint32_t num_lines,
1305 uint32_t log2_cache_line_size)

1307 num_probes_(num_probes),
1308 num_lines_(num_lines),

1311 // No Copy allowed
1312 LegacyBloomBitsReader(const LegacyBloomBitsReader&) = delete;
1313 void operator=(const LegacyBloomBitsReader&) = delete;
1314
1315 ~LegacyBloomBitsReader() override = default;
1316
1317 // "contents" contains the data built by a preceding call to
1318 // FilterBitsBuilder::Finish. MayMatch must return true if the key was
1319 // passed to FilterBitsBuilder::AddKey. This method may return true or false
1320 // if the key was not on the list, but it should aim to return false with a
1321 // high probability.
1322 bool MayMatch(const Slice& key) override {

1324 uint32_t byte_offset;
1325 LegacyBloomImpl::PrepareHashMayMatch(
1326 hash, num_lines_, data_, /*out*/ &byte_offset, log2_cache_line_size_);

1328 hash, num_probes_, data_ + byte_offset, log2_cache_line_size_);
1329 }
1330

1332 std::array<uint32_t, MultiGetContext::MAX_BATCH_SIZE> hashes;
1333 std::array<uint32_t, MultiGetContext::MAX_BATCH_SIZE> byte_offsets;
1334 for (int i = 0; i < num_keys; ++i) {
1335 hashes[i] = BloomHash(*keys[i]);
1336 LegacyBloomImpl::PrepareHashMayMatch(hashes[i], num_lines_, data_,
1337 /*out*/ &byte_offsets[i],
1338 log2_cache_line_size_);
1339 }
1340 for (int i = 0; i < num_keys; ++i) {
1341 may_match[i] = LegacyBloomImpl::HashMayMatchPrepared(
1342 hashes[i], num_probes_, data_ + byte_offsets[i],
1343 log2_cache_line_size_);
1344 }
1345 }
1346
1347 bool HashMayMatch(const uint64_t /* h */) override { return false; }
1348
1349 private:
1350 const char* data_;
1351 const int num_probes_;
1352 const uint32_t num_lines_;
1353 const uint32_t log2_cache_line_size_;
1354};

1422 : warned_(false), aggregate_rounding_balance_(0) {
1423 // Sanitize bits_per_key
1424 if (bits_per_key < 0.5) {
1425 // Round down to no filter
1426 bits_per_key = 0;
1427 } else if (bits_per_key < 1.0) {

1430 } else if (!(bits_per_key < 100.0)) { // including NaN

1432 }

1435 // that doubles specified with three decimal digits after the decimal
1436 // point are interpreted accurately.

1440 // memory. (Ribbon bits per key will be ~30% less than Bloom bits per key
1441 // for same FP rate.)

1443 1.0 / BloomMath::CacheLocalFpRate(
1444 bits_per_key,

1449 // e.g. 7.4999999999999 will round up to 8, but that provides more
1450 // predictability against small arithmetic errors in floating point.

1646BuiltinFilterBitsReader* BuiltinFilterPolicy::GetBuiltinFilterBitsReader(
1647 const Slice& contents) {
1648 uint32_t len_with_meta = static_cast<uint32_t>(contents.size());
1649 if (len_with_meta <= kMetadataLen) {
1650 // filter is empty or broken. Treat like zero keys added.
1651 return new AlwaysFalseFilter();
1652 }
1653
1654 // Legacy Bloom filter data:
1655 // 0 +-----------------------------------+
1656 // | Raw Bloom filter data |
1657 // | ... |
1658 // len +-----------------------------------+
1659 // | byte for num_probes or |
1660 // | marker for new implementations |
1661 // len+1 +-----------------------------------+
1662 // | four bytes for number of cache |
1663 // | lines |
1664 // len_with_meta +-----------------------------------+
1665
1666 int8_t raw_num_probes =
1667 static_cast<int8_t>(contents.data()[len_with_meta - kMetadataLen]);
1668 // NB: *num_probes > 30 and < 128 probably have not been used, because of
1669 // BloomFilterPolicy::initialize, unless directly calling
1670 // LegacyBloomBitsBuilder as an API, but we are leaving those cases in
1671 // limbo with LegacyBloomBitsReader for now.
1672
1673 if (raw_num_probes < 1) {
1674 // Note: < 0 (or unsigned > 127) indicate special new implementations
1675 // (or reserved for future use)
1676 switch (raw_num_probes) {
1677 case 0:
1678 // Treat as zero probes (always FP)
1679 return new AlwaysTrueFilter();
1680 case -1:
1681 // Marker for newer Bloom implementations
1682 return GetBloomBitsReader(contents);
1683 case -2:
1684 // Marker for Ribbon implementations
1685 return GetRibbonBitsReader(contents);
1686 default:
1687 // Reserved (treat as zero probes, always FP, for now)
1688 return new AlwaysTrueFilter();
1689 }
1690 }
1691 // else attempt decode for LegacyBloomBitsReader
1692
1693 int num_probes = raw_num_probes;
1694 assert(num_probes >= 1);
1695 assert(num_probes <= 127);
1696
1697 uint32_t len = len_with_meta - kMetadataLen;
1698 assert(len > 0);
1699
1700 uint32_t num_lines = DecodeFixed32(contents.data() + len_with_meta - 4);
1701 uint32_t log2_cache_line_size;
1702
1703 if (num_lines * CACHE_LINE_SIZE == len) {
1704 // Common case
1705 log2_cache_line_size = ConstexprFloorLog2(CACHE_LINE_SIZE);
1706 } else if (num_lines == 0 || len % num_lines != 0) {
1707 // Invalid (no solution to num_lines * x == len)
1708 // Treat as zero probes (always FP) for now.
1709 return new AlwaysTrueFilter();
1710 } else {
1711 // Determine the non-native cache line size (from another system)
1712 log2_cache_line_size = 0;
1713 while ((num_lines << log2_cache_line_size) < len) {
1714 ++log2_cache_line_size;
1715 }
1716 if ((num_lines << log2_cache_line_size) != len) {
1717 // Invalid (block size not a power of two)
1718 // Treat as zero probes (always FP) for now.
1719 return new AlwaysTrueFilter();
1720 }
1721 }
1722 // if not early return
1723 return new LegacyBloomBitsReader(contents.data(), num_probes, num_lines,
1724 log2_cache_line_size);
1725}

1734BuiltinFilterBitsReader* BuiltinFilterPolicy::GetRibbonBitsReader(
1735 const Slice& contents) {
1736 uint32_t len_with_meta = static_cast<uint32_t>(contents.size());
1737 uint32_t len = len_with_meta - kMetadataLen;
1738
1739 assert(len > 0); // precondition
1740
1741 uint32_t seed = static_cast<uint8_t>(contents.data()[len + 1]);
1742 uint32_t num_blocks = static_cast<uint8_t>(contents.data()[len + 2]);
1743 num_blocks |= static_cast<uint8_t>(contents.data()[len + 3]) << 8;
1744 num_blocks |= static_cast<uint8_t>(contents.data()[len + 4]) << 16;
1745 if (num_blocks < 2) {
1746 // Not supported
1747 // num_blocks == 1 is not used because num_starts == 1 is problematic
1748 // for the hashing scheme. num_blocks == 0 is unused because there's
1749 // already a concise encoding of an "always false" filter.
1750 // Return something safe:
1751 return new AlwaysTrueFilter();
1752 }
1753 return new Standard128RibbonBitsReader(contents.data(), len, num_blocks,
1754 seed);
1755}

1757// For newer Bloom filter implementations
1758BuiltinFilterBitsReader* BuiltinFilterPolicy::GetBloomBitsReader(
1759 const Slice& contents) {
1760 uint32_t len_with_meta = static_cast<uint32_t>(contents.size());
1761 uint32_t len = len_with_meta - kMetadataLen;
1762
1763 assert(len > 0); // precondition
1764
1765 // New Bloom filter data:
1766 // 0 +-----------------------------------+
1767 // | Raw Bloom filter data |
1768 // | ... |
1769 // len +-----------------------------------+
1770 // | char{-1} byte -> new Bloom filter |
1771 // len+1 +-----------------------------------+
1772 // | byte for subimplementation |
1773 // | 0: FastLocalBloom |
1774 // | other: reserved |
1775 // len+2 +-----------------------------------+
1776 // | byte for block_and_probes |
1777 // | 0 in top 3 bits -> 6 -> 64-byte |
1778 // | reserved: |
1779 // | 1 in top 3 bits -> 7 -> 128-byte|
1780 // | 2 in top 3 bits -> 8 -> 256-byte|
1781 // | ... |
1782 // | num_probes in bottom 5 bits, |
1783 // | except 0 and 31 reserved |
1784 // len+3 +-----------------------------------+
1785 // | two bytes reserved |
1786 // | possibly for hash seed |
1787 // len_with_meta +-----------------------------------+
1788
1789 // Read more metadata (see above)
1790 char sub_impl_val = contents.data()[len_with_meta - 4];
1791 char block_and_probes = contents.data()[len_with_meta - 3];
1792 int log2_block_bytes = ((block_and_probes >> 5) & 7) + 6;
1793
1794 int num_probes = (block_and_probes & 31);
1795 if (num_probes < 1 || num_probes > 30) {
1796 // Reserved / future safe
1797 return new AlwaysTrueFilter();
1798 }
1799
1800 uint16_t rest = DecodeFixed16(contents.data() + len_with_meta - 2);
1801 if (rest != 0) {
1802 // Reserved, possibly for hash seed
1803 // Future safe
1804 return new AlwaysTrueFilter();
1805 }
1806
1807 if (sub_impl_val == 0) { // FastLocalBloom
1808 if (log2_block_bytes == 6) { // Only block size supported for now
1809 return new FastLocalBloomBitsReader(contents.data(), num_probes, len);
1810 }
1811 }
1812 // otherwise
1813 // Reserved / future safe
1814 return new AlwaysTrueFilter();
1815}

1837FilterBitsBuilder* RibbonFilterPolicy::GetBuilderWithContext(
1838 const FilterBuildingContext& context) const {
1839 if (GetMillibitsPerKey() == 0) {
1840 // "No filter" special case
1841 return nullptr;
1842 }
1843 // Treat unknown same as bottommost, INT_MAX - 1.
1844 // INT_MAX is reserved for "always use Bloom".
1845 int levelish = INT_MAX - 1;
1846
1847 int bloom_before_level = bloom_before_level_.load(std::memory_order_relaxed);
1848 if (bloom_before_level < INT_MAX) {
1849 switch (context.compaction_style) {
1850 case kCompactionStyleLevel:
1851 case kCompactionStyleUniversal: {
1852 if (context.reason == TableFileCreationReason::kFlush) {
1853 // Treat flush as level -1
1854 assert(context.level_at_creation == 0);
1855 levelish = -1;
1856 } else if (context.level_at_creation == -1) {
1857 // Unknown level
1858 assert(levelish == INT_MAX - 1);
1859 } else {
1860 levelish = context.level_at_creation;
1861 }
1862 break;
1863 }
