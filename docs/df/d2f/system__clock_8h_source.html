28// A SystemClock is an interface used by the rocksdb implementation to access
29// operating system time-related functionality.
30class SystemClock : public Customizable {
31 public:
32 ~SystemClock() override {}
33
34 static const char* Type() { return "SystemClock"; }
35 static Status CreateFromString(const ConfigOptions& options,
36 const std::string& value,
37 std::shared_ptr<SystemClock>* result);
38 // The name of this system clock
39 const char* Name() const override = 0;
40
41 // The name/nickname for the Default SystemClock. This name can be used
42 // to determine if the clock is the default one.
43 static const char* kDefaultName() { return "DefaultClock"; }
44
45 // Return a default SystemClock suitable for the current operating
46 // system.
47 static const std::shared_ptr<SystemClock>& Default();
48
49 // Returns the number of micro-seconds since some fixed point in time.
50 // It is often used as system time such as in GenericRateLimiter
51 // and other places so a port needs to return system time in order to work.
52 virtual uint64_t NowMicros() = 0;
53
54 // Returns the number of nano-seconds since some fixed point in time. Only
55 // useful for computing deltas of time in one run.
56 // Default implementation simply relies on NowMicros.
57 // In platform-specific implementations, NowNanos() should return time points
58 // that are MONOTONIC.
59 virtual uint64_t NowNanos() { return NowMicros() * 1000; }
60
61 // Returns the number of micro-seconds of CPU time used by the current thread.
62 // 0 indicates not supported.
63 virtual uint64_t CPUMicros() { return 0; }
64
65 // Returns the number of nano-seconds of CPU time used by the current thread.
66 // Default implementation simply relies on CPUMicros.
67 // 0 indicates not supported.
68 virtual uint64_t CPUNanos() { return CPUMicros() * 1000; }
69
70 // Sleep/delay the thread for the prescribed number of micro-seconds.
71 virtual void SleepForMicroseconds(int micros) = 0;
72
73 // For internal use/extension only.
74 //
75 // Issues a wait on `cv` that times out at `deadline`. May wakeup and return
76 // spuriously.
77 //
78 // Returns true if wait timed out, false otherwise
79 virtual bool TimedWait(port::CondVar* cv, std::chrono::microseconds deadline);
80
81 // Get the number of seconds since the Epoch, 1970-01-01 00:00:00 (UTC).
82 // Only overwrites *unix_time on success.
83 virtual Status GetCurrentTime(int64_t* unix_time) = 0;
84
85 // Converts seconds-since-Jan-01-1970 to a printable string
86 virtual std::string TimeToString(uint64_t time) = 0;
87};

89// Wrapper class for a SystemClock. Redirects all methods (except Name)
90// of the SystemClock interface to the target/wrapped class.
91class SystemClockWrapper : public SystemClock {
92 public:
93 explicit SystemClockWrapper(const std::shared_ptr<SystemClock>& t);
94
95 uint64_t NowMicros() override { return target_->NowMicros(); }
96
97 uint64_t NowNanos() override { return target_->NowNanos(); }
98
99 uint64_t CPUMicros() override { return target_->CPUMicros(); }
100
101 uint64_t CPUNanos() override { return target_->CPUNanos(); }
102
103 void SleepForMicroseconds(int micros) override {
104 return target_->SleepForMicroseconds(micros);
105 }
106
107 bool TimedWait(port::CondVar* cv,
108 std::chrono::microseconds deadline) override {
109 return target_->TimedWait(cv, deadline);
110 }
111
112 Status GetCurrentTime(int64_t* unix_time) override {
113 return target_->GetCurrentTime(unix_time);
114 }
115
116 std::string TimeToString(uint64_t time) override {
117 return target_->TimeToString(time);
118 }
119
120 Status PrepareOptions(const ConfigOptions& options) override;
121 std::string SerializeOptions(const ConfigOptions& config_options,
122 const std::string& header) const override;
123 const Customizable* Inner() const override { return target_.get(); }
124
125 protected:
126 std::shared_ptr<SystemClock> target_;
127};
