19// The file contains two classes PlainTableIndex and PlainTableIndexBuilder
20// The two classes implement the index format of PlainTable.
21// For description of PlainTable format, see comments of class
22// PlainTableFactory
23//
24//
25// PlainTableIndex contains buckets size of index_size_, each is a
26// 32-bit integer. The lower 31 bits contain an offset value (explained below)
27// and the first bit of the integer indicates type of the offset.
28//
29// +--------------+------------------------------------------------------+
30// | Flag (1 bit) | Offset to binary search buffer or file (31 bits) +
31// +--------------+------------------------------------------------------+
32//
33// Explanation for the "flag bit":
34//
35// 0 indicates that the bucket contains only one prefix (no conflict when
36// hashing this prefix), whose first row starts from this offset of the
37// file.
38// 1 indicates that the bucket contains more than one prefixes, or there
39// are too many rows for one prefix so we need a binary search for it. In
40// this case, the offset indicates the offset of sub_index_ holding the
41// binary search indexes of keys for those rows. Those binary search indexes
42// are organized in this way:
43//
44// The first 4 bytes, indicate how many indexes (N) are stored after it. After
45// it, there are N 32-bit integers, each points of an offset of the file,
46// which
47// points to starting of a row. Those offsets need to be guaranteed to be in
48// ascending order so the keys they are pointing to are also in ascending
49// order
50// to make sure we can use them to do binary searches. Below is visual
51// presentation of a bucket.
52//
53// <begin>
54// number_of_records: varint32
55// record 1 file offset: fixedint32
56// record 2 file offset: fixedint32
57// ....
58// record N file offset: fixedint32
59// <end>
60
61// The class loads the index block from a PlainTable SST file, and executes
62// the index lookup.
63// The class is used by PlainTableReader class.
64class PlainTableIndex {
65 public:
66 enum IndexSearchResult {
67 kNoPrefixForBucket = 0,
68 kDirectToFile = 1,
69 kSubindex = 2
70 };
71
72 explicit PlainTableIndex(Slice data) { InitFromRawData(data); }
73
74 PlainTableIndex()
75 : index_size_(0),
76 sub_index_size_(0),
77 num_prefixes_(0),
78 index_(nullptr),
79 sub_index_(nullptr) {}
80
81 // The function that executes the lookup the hash table.
82 // The hash key is `prefix_hash`. The function fills the hash bucket
83 // content in `bucket_value`, which is up to the caller to interpret.
84 IndexSearchResult GetOffset(uint32_t prefix_hash,
85 uint32_t* bucket_value) const;
86
87 // Initialize data from `index_data`, which points to raw data for
88 // index stored in the SST file.
89 Status InitFromRawData(Slice index_data);
90
91 // Decode the sub index for specific hash bucket.
92 // The `offset` is the value returned as `bucket_value` by GetOffset()
93 // and is only valid when the return value is `kSubindex`.
94 // The return value is the pointer to the starting address of the
95 // sub-index. `upper_bound` is filled with the value indicating how many
96 // entries the sub-index has.
97 const char* GetSubIndexBasePtrAndUpperBound(uint32_t offset,
98 uint32_t* upper_bound) const {
99 const char* index_ptr = &sub_index_[offset];
100 return GetVarint32Ptr(index_ptr, index_ptr + 4, upper_bound);
101 }
102
103 uint32_t GetIndexSize() const { return index_size_; }
104
105 uint32_t GetSubIndexSize() const { return sub_index_size_; }
106
107 uint32_t GetNumPrefixes() const { return num_prefixes_; }
108
109 static const uint64_t kMaxFileSize = (1u << 31) - 1;
110 static const uint32_t kSubIndexMask = 0x80000000;
111 static const size_t kOffsetLen = sizeof(uint32_t);
112
113 private:
114 uint32_t index_size_;
115 uint32_t sub_index_size_;
116 uint32_t num_prefixes_;
117
118 uint32_t* index_;
119 char* sub_index_;
120};

169 // Helper class to track all the index records
170 class IndexRecordList {
171 public:
172 explicit IndexRecordList(size_t num_records_per_group)
173 : kNumRecordsPerGroup(num_records_per_group),
174 current_group_(nullptr),
175 num_records_in_current_group_(num_records_per_group) {}
176
177 ~IndexRecordList() {
178 for (size_t i = 0; i < groups_.size(); i++) {
179 delete[] groups_[i];
180 }
181 }
182
183 void AddRecord(uint32_t hash, uint32_t offset);
184
185 size_t GetNumRecords() const {
186 return (groups_.size() - 1) * kNumRecordsPerGroup +
187 num_records_in_current_group_;
188 }
189 IndexRecord* At(size_t index) {
190 return &(
191 groups_[index / kNumRecordsPerGroup][index % kNumRecordsPerGroup]);
192 }
193
194 private:
195 IndexRecord* AllocateNewGroup() {
196 IndexRecord* result = new IndexRecord[kNumRecordsPerGroup];
197 groups_.push_back(result);
198 return result;
199 }
200
201 // Each group in `groups_` contains fix-sized records (determined by
202 // kNumRecordsPerGroup). Which can help us minimize the cost if resizing
203 // occurs.
204 const size_t kNumRecordsPerGroup;
205 IndexRecord* current_group_;
206 // List of arrays allocated
207 std::vector<IndexRecord*> groups_;
208 size_t num_records_in_current_group_;
209 };
