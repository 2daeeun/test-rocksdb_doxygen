135void locktree_manager::locktree_map_remove(locktree *lt) {
136 uint32_t idx;
137 locktree *found_lt;
138 int r = m_locktree_map.find_zero<DICTIONARY_ID, find_by_dict_id>(
139 lt->get_dict_id(), &found_lt, &idx);
140 invariant_zero(r);
141 invariant(found_lt == lt);
142 r = m_locktree_map.delete_at(idx);
143 invariant_zero(r);

146locktree *locktree_manager::get_lt(DICTIONARY_ID dict_id, const comparator &cmp,
147 void *on_create_extra) {
148 // hold the mutex around searching and maybe
149 // inserting into the locktree map
150 mutex_lock();
151
152 locktree *lt = locktree_map_find(dict_id);
153 if (lt == nullptr) {
154 XCALLOC(lt);
155 lt->create(this, dict_id, cmp, mutex_factory);
156
157 // new locktree created - call the on_create callback
158 // and put it in the locktree map
159 if (m_lt_create_callback) {
160 int r = m_lt_create_callback(lt, on_create_extra);
161 if (r != 0) {
162 lt->release_reference();
163 lt->destroy();
164 toku_free(lt);
165 lt = nullptr;
166 }
167 }
168 if (lt) {
169 locktree_map_put(lt);
170 }
171 } else {
172 reference_lt(lt);
173 }
174
175 mutex_unlock();
176
177 return lt;
178}

180void locktree_manager::reference_lt(locktree *lt) {
181 // increment using a sync fetch and add.
182 // the caller guarantees that the lt won't be
183 // destroyed while we increment the count here.
184 //
185 // the caller can do this by already having an lt
186 // reference or by holding the manager mutex.
187 //
188 // if the manager's mutex is held, it is ok for the
189 // reference count to transition from 0 to 1 (no race),
190 // since we're serialized with other opens and closes.
191 lt->add_reference();
192}
193
194void locktree_manager::release_lt(locktree *lt) {
195 bool do_destroy = false;
196 DICTIONARY_ID dict_id = lt->get_dict_id();
197
198 // Release a reference on the locktree. If the count transitions to zero,
199 // then we *may* need to do the cleanup.
200 //
201 // Grab the manager's mutex and look for a locktree with this locktree's
202 // dictionary id. Since dictionary id's never get reused, any locktree
203 // found must be the one we just released a reference on.
204 //
205 // At least two things could have happened since we got the mutex:
206 // - Another thread gets a locktree with the same dict_id, increments
207 // the reference count. In this case, we shouldn't destroy it.
208 // - Another thread gets a locktree with the same dict_id and then
209 // releases it quickly, transitioning the reference count from zero to
210 // one and back to zero. In this case, only one of us should destroy it.
211 // It doesn't matter which. We originally missed this case, see #5776.
212 //
213 // After 5776, the high level rule for release is described below.
214 //
215 // If a thread releases a locktree and notices the reference count transition
216 // to zero, then that thread must immediately:
217 // - assume the locktree object is invalid
218 // - grab the manager's mutex
219 // - search the locktree map for a locktree with the same dict_id and remove
220 // it, if it exists. the destroy may be deferred.
221 // - release the manager's mutex
222 //
223 // This way, if many threads transition the same locktree's reference count
224 // from 1 to zero and wait behind the manager's mutex, only one of them will
225 // do the actual destroy and the others will happily do nothing.
226 uint32_t refs = lt->release_reference();
227 if (refs == 0) {
228 mutex_lock();
229 // lt may not have already been destroyed, so look it up.
230 locktree *find_lt = locktree_map_find(dict_id);
231 if (find_lt != nullptr) {
232 // A locktree is still in the map with that dict_id, so it must be
233 // equal to lt. This is true because dictionary ids are never reused.
234 // If the reference count is zero, it's our responsibility to remove
235 // it and do the destroy. Otherwise, someone still wants it.
236 // If the locktree is still valid then check if it should be deleted.
237 if (find_lt == lt) {
238 if (lt->get_reference_count() == 0) {
239 locktree_map_remove(lt);
240 do_destroy = true;
241 }
242 m_lt_counters.add(lt->get_lock_request_info()->counters);
243 }
244 }
245 mutex_unlock();
246 }
247
248 // if necessary, do the destroy without holding the mutex
249 if (do_destroy) {
250 if (m_lt_destroy_callback) {
251 m_lt_destroy_callback(lt);
252 }
253 lt->destroy();
254 toku_free(lt);
255 }
256}

271void locktree_manager::escalate_all_locktrees(void) {
272 uint64_t t0 = toku_current_time_microsec();
273
274 // get all locktrees
275 mutex_lock();
276 int num_locktrees = m_locktree_map.size();
277 locktree **locktrees = new locktree *[num_locktrees];
278 for (int i = 0; i < num_locktrees; i++) {
279 int r = m_locktree_map.fetch(i, &locktrees[i]);
280 invariant_zero(r);
281 reference_lt(locktrees[i]);
282 }
283 mutex_unlock();
284
285 // escalate them
286 escalate_locktrees(locktrees, num_locktrees);
287
288 delete[] locktrees;
289
290 uint64_t t1 = toku_current_time_microsec();
291 add_escalator_wait_time(t1 - t0);
292}

312int locktree_manager::iterate_pending_lock_requests(
313 lock_request_iterate_callback callback, void *extra) {
314 mutex_lock();
315 int r = 0;
316 uint32_t num_locktrees = m_locktree_map.size();
317 for (uint32_t i = 0; i < num_locktrees && r == 0; i++) {
318 locktree *lt;
319 r = m_locktree_map.fetch(i, &lt);
320 invariant_zero(r);
321 if (r == EINVAL) // Shouldn't happen, avoid compiler warning
322 continue;
323
324 struct lt_lock_request_info *info = lt->get_lock_request_info();
325 toku_external_mutex_lock(&info->mutex);
326
327 uint32_t num_requests = info->pending_lock_requests.size();
328 for (uint32_t k = 0; k < num_requests && r == 0; k++) {
329 lock_request *req;
330 r = info->pending_lock_requests.fetch(k, &req);
331 invariant_zero(r);
332 if (r == EINVAL) /* Shouldn't happen, avoid compiler warning */
333 continue;
334 r = callback(lt->get_dict_id(), req->get_txnid(), req->get_left_key(),
335 req->get_right_key(), req->get_conflicting_txnid(),
336 req->get_start_time(), extra);
337 }
338
339 toku_external_mutex_unlock(&info->mutex);
340 }
341 mutex_unlock();
342 return r;
343}

345int locktree_manager::check_current_lock_constraints(bool big_txn) {
346 int r = 0;
347 if (big_txn && over_big_threshold()) {
348 run_escalation();
349 if (over_big_threshold()) {
350 r = TOKUDB_OUT_OF_LOCKS;
351 }
352 }
353 if (r == 0 && out_of_locks()) {
354 run_escalation();
355 if (out_of_locks()) {
356 // return an error if we're still out of locks after escalation.
357 r = TOKUDB_OUT_OF_LOCKS;
358 }
359 }
360 return r;
361}

392void locktree_manager::escalate_locktrees(locktree **locktrees,
393 int num_locktrees) {
394 // there are too many row locks in the system and we need to tidy up.
395 //
396 // a simple implementation of escalation does not attempt
397 // to reduce the memory foot print of each txn's range buffer.
398 // doing so would require some layering hackery (or a callback)
399 // and more complicated locking. for now, just escalate each
400 // locktree individually, in-place.
401 tokutime_t t0 = toku_time_now();
402 for (int i = 0; i < num_locktrees; i++) {
403 locktrees[i]->escalate(m_lt_escalate_callback,
404 m_lt_escalate_callback_extra);
405 release_lt(locktrees[i]);
406 }
407 tokutime_t t1 = toku_time_now();
408
409 toku_mutex_lock(&m_escalation_mutex);
410 m_escalation_count++;
411 m_escalation_time += (t1 - t0);
412 m_escalation_latest_result = m_current_lock_memory;
413 toku_mutex_unlock(&m_escalation_mutex);
414}

434void locktree_manager::locktree_escalator::run(
435 locktree_manager *mgr, void (*escalate_locktrees_fun)(void *extra),
436 void *extra) {
437 uint64_t t0 = toku_current_time_microsec();
438 toku_mutex_lock(&m_escalator_mutex);
439 if (!m_escalator_running) {
440 // run escalation on this thread
441 m_escalator_running = true;
442 toku_mutex_unlock(&m_escalator_mutex);
443 escalate_locktrees_fun(extra);
444 toku_mutex_lock(&m_escalator_mutex);
445 m_escalator_running = false;
446 toku_cond_broadcast(&m_escalator_done);
447 } else {
448 toku_cond_wait(&m_escalator_done, &m_escalator_mutex);
449 }
450 toku_mutex_unlock(&m_escalator_mutex);
451 uint64_t t1 = toku_current_time_microsec();
452 mgr->add_escalator_wait_time(t1 - t0);
453}

455void locktree_manager::get_status(LTM_STATUS statp) {
456 ltm_status.init();
457 LTM_STATUS_VAL(LTM_SIZE_CURRENT) = m_current_lock_memory;
458 LTM_STATUS_VAL(LTM_SIZE_LIMIT) = m_max_lock_memory;
459 LTM_STATUS_VAL(LTM_ESCALATION_COUNT) = m_escalation_count;
460 LTM_STATUS_VAL(LTM_ESCALATION_TIME) = m_escalation_time;
461 LTM_STATUS_VAL(LTM_ESCALATION_LATEST_RESULT) = m_escalation_latest_result;
462 LTM_STATUS_VAL(LTM_WAIT_ESCALATION_COUNT) = m_wait_escalation_count;
463 LTM_STATUS_VAL(LTM_WAIT_ESCALATION_TIME) = m_wait_escalation_time;
464 LTM_STATUS_VAL(LTM_LONG_WAIT_ESCALATION_COUNT) = m_long_wait_escalation_count;
465 LTM_STATUS_VAL(LTM_LONG_WAIT_ESCALATION_TIME) = m_long_wait_escalation_time;
466
467 uint64_t lock_requests_pending = 0;
468 uint64_t sto_num_eligible = 0;
469 uint64_t sto_end_early_count = 0;
470 tokutime_t sto_end_early_time = 0;
471 uint32_t num_locktrees = 0;
472 struct lt_counters lt_counters;
473 ZERO_STRUCT(lt_counters); // PORT: instead of ={}.
474
475 if (toku_mutex_trylock(&m_mutex) == 0) {
476 lt_counters = m_lt_counters;
477 num_locktrees = m_locktree_map.size();
478 for (uint32_t i = 0; i < num_locktrees; i++) {
479 locktree *lt;
480 int r = m_locktree_map.fetch(i, &lt);
481 invariant_zero(r);
482 if (r == EINVAL) // Shouldn't happen, avoid compiler warning
483 continue;
484 if (toku_external_mutex_trylock(&lt->m_lock_request_info.mutex) == 0) {
485 lock_requests_pending +=
486 lt->m_lock_request_info.pending_lock_requests.size();
487 lt_counters.add(lt->get_lock_request_info()->counters);
488 toku_external_mutex_unlock(&lt->m_lock_request_info.mutex);
489 }
490 sto_num_eligible += lt->sto_txnid_is_valid_unsafe() ? 1 : 0;
491 sto_end_early_count += lt->m_sto_end_early_count;
492 sto_end_early_time += lt->m_sto_end_early_time;
493 }
494 mutex_unlock();
495 }
496
497 LTM_STATUS_VAL(LTM_NUM_LOCKTREES) = num_locktrees;
498 LTM_STATUS_VAL(LTM_LOCK_REQUESTS_PENDING) = lock_requests_pending;
499 LTM_STATUS_VAL(LTM_STO_NUM_ELIGIBLE) = sto_num_eligible;
500 LTM_STATUS_VAL(LTM_STO_END_EARLY_COUNT) = sto_end_early_count;
501 LTM_STATUS_VAL(LTM_STO_END_EARLY_TIME) = sto_end_early_time;
502 LTM_STATUS_VAL(LTM_WAIT_COUNT) = lt_counters.wait_count;
503 LTM_STATUS_VAL(LTM_WAIT_TIME) = lt_counters.wait_time;
504 LTM_STATUS_VAL(LTM_LONG_WAIT_COUNT) = lt_counters.long_wait_count;
505 LTM_STATUS_VAL(LTM_LONG_WAIT_TIME) = lt_counters.long_wait_time;
506 LTM_STATUS_VAL(LTM_TIMEOUT_COUNT) = lt_counters.timeout_count;
507 *statp = ltm_status;
508}
