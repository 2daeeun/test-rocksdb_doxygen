17// A wrapper that allows injection of errors.
18class CorruptionFS : public FileSystemWrapper {
19 public:
20 bool writable_file_error_;
21 int num_writable_file_errors_;
22
23 explicit CorruptionFS(const std::shared_ptr<FileSystem>& _target,
24 bool fs_buffer, bool verify_read)
25 : FileSystemWrapper(_target),
26 writable_file_error_(false),
27 num_writable_file_errors_(0),
28 corruption_trigger_(INT_MAX),
29 read_count_(0),
30 corrupt_offset_(0),
31 corrupt_len_(0),
32 rnd_(300),
33 fs_buffer_(fs_buffer),
34 verify_read_(verify_read) {}
35 ~CorruptionFS() override {
36 // Assert that the corruption was reset, which means it got triggered
37 assert(corruption_trigger_ == INT_MAX || corrupt_len_ > 0);
38 }
39 const char* Name() const override { return "ErrorEnv"; }
40
41 IOStatus NewWritableFile(const std::string& fname, const FileOptions& opts,
42 std::unique_ptr<FSWritableFile>* result,
43 IODebugContext* dbg) override {
44 result->reset();
45 if (writable_file_error_) {
46 ++num_writable_file_errors_;
47 return IOStatus::IOError(fname, "fake error");
48 }
49 return target()->NewWritableFile(fname, opts, result, dbg);
50 }
51
52 void SetCorruptionTrigger(const int trigger) {
53 MutexLock l(&mutex_);
54 corruption_trigger_ = trigger;
55 read_count_ = 0;
56 corrupt_fname_.clear();
57 }
58
59 IOStatus NewRandomAccessFile(const std::string& fname,
60 const FileOptions& opts,
61 std::unique_ptr<FSRandomAccessFile>* result,
62 IODebugContext* dbg) override {
63 class CorruptionRandomAccessFile : public FSRandomAccessFileOwnerWrapper {
64 public:
65 CorruptionRandomAccessFile(CorruptionFS& fs, const std::string& fname,
66 std::unique_ptr<FSRandomAccessFile>& file)
67 : FSRandomAccessFileOwnerWrapper(std::move(file)),
68 fs_(fs),
69 fname_(fname) {}
70
71 IOStatus Read(uint64_t offset, size_t len, const IOOptions& opts,
72 Slice* result, char* scratch,
73 IODebugContext* dbg) const override {
74 IOStatus s = target()->Read(offset, len, opts, result, scratch, dbg);
75 if (opts.verify_and_reconstruct_read) {
76 fs_.MaybeResetOverlapWithCorruptedChunk(fname_, offset,
77 result->size());
78 return s;
79 }
80
81 MutexLock l(&fs_.mutex_);
82 if (s.ok() && ++fs_.read_count_ >= fs_.corruption_trigger_) {
83 fs_.corruption_trigger_ = INT_MAX;
84 char* data = const_cast<char*>(result->data());
85 std::memcpy(
86 data,
87 fs_.rnd_.RandomString(static_cast<int>(result->size())).c_str(),
88 result->size());
89 fs_.SetCorruptedChunk(fname_, offset, result->size());
90 }
91 return s;
92 }
93
94 IOStatus MultiRead(FSReadRequest* reqs, size_t num_reqs,
95 const IOOptions& options,
96 IODebugContext* dbg) override {
97 for (size_t i = 0; i < num_reqs; ++i) {
98 FSReadRequest& req = reqs[i];
99 if (fs_.fs_buffer_) {
100 FSAllocationPtr buffer(new char[req.len], [](void* ptr) {
101 delete[] static_cast<char*>(ptr);
102 });
103 req.fs_scratch = std::move(buffer);
104 req.status = Read(req.offset, req.len, options, &req.result,
105 static_cast<char*>(req.fs_scratch.get()), dbg);
106 } else {
107 req.status = Read(req.offset, req.len, options, &req.result,
108 req.scratch, dbg);
109 }
110 }
111 return IOStatus::OK();
112 }
113
114 IOStatus Prefetch(uint64_t /*offset*/, size_t /*n*/,
115 const IOOptions& /*options*/,
116 IODebugContext* /*dbg*/) override {
117 return IOStatus::NotSupported("Prefetch");
118 }
119
120 private:
121 CorruptionFS& fs_;
122 std::string fname_;
123 };
124
125 std::unique_ptr<FSRandomAccessFile> file;
126 IOStatus s = target()->NewRandomAccessFile(fname, opts, &file, dbg);
127 EXPECT_OK(s);
128 result->reset(new CorruptionRandomAccessFile(*this, fname, file));
129
130 return s;
131 }

133 IOStatus NewSequentialFile(const std::string& fname,
134 const FileOptions& file_opts,
135 std::unique_ptr<FSSequentialFile>* result,
136 IODebugContext* dbg) override {
137 class CorruptionSequentialFile : public FSSequentialFileOwnerWrapper {
138 public:
139 CorruptionSequentialFile(CorruptionFS& fs, const std::string& fname,
140 std::unique_ptr<FSSequentialFile>& file)
141 : FSSequentialFileOwnerWrapper(std::move(file)),
142 fs_(fs),
143 fname_(fname),
144 offset_(0) {}
145
146 IOStatus Read(size_t len, const IOOptions& opts, Slice* result,
147 char* scratch, IODebugContext* dbg) override {
148 IOStatus s = target()->Read(len, opts, result, scratch, dbg);
149 if (result->size() == 0 ||
150 fname_.find("IDENTITY") != std::string::npos) {
151 return s;
152 }
153
154 if (opts.verify_and_reconstruct_read) {
155 fs_.MaybeResetOverlapWithCorruptedChunk(fname_, offset_,
156 result->size());
157 return s;
158 }
159
160 MutexLock l(&fs_.mutex_);
161 if (s.ok() && ++fs_.read_count_ >= fs_.corruption_trigger_) {
162 fs_.corruption_trigger_ = INT_MAX;
163 char* data = const_cast<char*>(result->data());
164 std::memcpy(
165 data,
166 fs_.rnd_.RandomString(static_cast<int>(result->size())).c_str(),
167 result->size());
168 fs_.SetCorruptedChunk(fname_, offset_, result->size());
169 }
170 offset_ += result->size();
171 return s;
172 }
173
174 private:
175 CorruptionFS& fs_;
176 std::string fname_;
177 size_t offset_;
178 };
179
180 std::unique_ptr<FSSequentialFile> file;
181 IOStatus s = target()->NewSequentialFile(fname, file_opts, &file, dbg);
182 EXPECT_OK(s);
183 result->reset(new CorruptionSequentialFile(*this, fname, file));
184
185 return s;
186 }
187
188 void SupportedOps(int64_t& supported_ops) override {
189 supported_ops = 1 << FSSupportedOps::kAsyncIO;
190 if (fs_buffer_) {
191 supported_ops |= 1 << FSSupportedOps::kFSBuffer;
192 }
193 if (verify_read_) {
194 supported_ops |= 1 << FSSupportedOps::kVerifyAndReconstructRead;
195 }
196 }
197
198 void SetCorruptedChunk(const std::string& fname, size_t offset, size_t len) {
199 assert(corrupt_fname_.empty());
200
201 corrupt_fname_ = fname;
202 corrupt_offset_ = offset;
203 corrupt_len_ = len;
204 }
205
206 void MaybeResetOverlapWithCorruptedChunk(const std::string& fname,
207 size_t offset, size_t len) {
208 if (fname == corrupt_fname_ &&
209 ((offset <= corrupt_offset_ && (offset + len) > corrupt_offset_) ||
210 (offset >= corrupt_offset_ &&
211 offset < (corrupt_offset_ + corrupt_len_)))) {
212 corrupt_fname_.clear();
213 }
214 }
215
216 bool VerifyRetry() { return corrupt_len_ > 0 && corrupt_fname_.empty(); }
217
218 int read_count() { return read_count_; }
219
220 int corruption_trigger() { return corruption_trigger_; }
221
222 private:
223 int corruption_trigger_;
224 int read_count_;
225 std::string corrupt_fname_;
226 size_t corrupt_offset_;
227 size_t corrupt_len_;
228 Random rnd_;
229 bool fs_buffer_;
230 bool verify_read_;
231 port::Mutex mutex_;
232};

241TEST_F(DBIOFailureTest, DropWrites) {
242 do {
243 Options options = CurrentOptions();
244 options.env = env_;
245 options.paranoid_checks = false;
246 Reopen(options);
247
248 ASSERT_OK(Put("foo", "v1"));
249 ASSERT_EQ("v1", Get("foo"));
250 Compact("a", "z");
251 const size_t num_files = CountFiles();
252 // Force out-of-space errors
253 env_->drop_writes_.store(true, std::memory_order_release);
254 env_->sleep_counter_.Reset();
255 env_->SetMockSleep();
256 for (int i = 0; i < 5; i++) {
257 if (option_config_ != kUniversalCompactionMultiLevel &&
258 option_config_ != kUniversalSubcompactions) {
259 for (int level = 0; level < dbfull()->NumberLevels(); level++) {
260 if (level > 0 && level == dbfull()->NumberLevels() - 1) {
261 break;
262 }
263 Status s =
264 dbfull()->TEST_CompactRange(level, nullptr, nullptr, nullptr,
265 true /* disallow trivial move */);
266 ASSERT_TRUE(s.ok() || s.IsCorruption());
267 }
268 } else {
269 Status s =
270 dbfull()->CompactRange(CompactRangeOptions(), nullptr, nullptr);
271 ASSERT_TRUE(s.ok() || s.IsCorruption());
272 }
273 }
274
275 std::string property_value;
276 ASSERT_TRUE(db_->GetProperty("rocksdb.background-errors", &property_value));
277 ASSERT_EQ("5", property_value);
278
279 env_->drop_writes_.store(false, std::memory_order_release);
280 const size_t count = CountFiles();
281 ASSERT_LT(count, num_files + 3);
282
283 // Check that compaction attempts slept after errors
284 // TODO @krad: Figure out why ASSERT_EQ 5 keeps failing in certain compiler
285 // versions
286 ASSERT_GE(env_->sleep_counter_.Read(), 4);
287 } while (ChangeCompactOptions());
288}

366TEST_F(DBIOFailureTest, ManifestWriteError) {
367 // Test for the following problem:
368 // (a) Compaction produces file F
369 // (b) Log record containing F is written to MANIFEST file, but Sync() fails
370 // (c) GC deletes F
371 // (d) After reopening DB, reads fail since deleted F is named in log record
372
373 // We iterate twice. In the second iteration, everything is the
374 // same except the log record never makes it to the MANIFEST file.
375 for (int iter = 0; iter < 2; iter++) {
376 std::atomic<bool>* error_type = (iter == 0) ? &env_->manifest_sync_error_
377 : &env_->manifest_write_error_;
378
379 // Insert foo=>bar mapping
380 Options options = CurrentOptions();
381 options.env = env_;
382 options.create_if_missing = true;
383 options.error_if_exists = false;
384 options.paranoid_checks = true;
385 DestroyAndReopen(options);
386 ASSERT_OK(Put("foo", "bar"));
387 ASSERT_EQ("bar", Get("foo"));
388
389 // Memtable compaction (will succeed)
390 ASSERT_OK(Flush());
391 ASSERT_EQ("bar", Get("foo"));
392 const int last = 2;
393 MoveFilesToLevel(2);
394 ASSERT_EQ(NumTableFilesAtLevel(last), 1); // foo=>bar is now in last level
395
396 // Merging compaction (will fail)
397 error_type->store(true, std::memory_order_release);
398 ASSERT_NOK(
399 dbfull()->TEST_CompactRange(last, nullptr, nullptr)); // Should fail
400 ASSERT_EQ("bar", Get("foo"));
401
402 error_type->store(false, std::memory_order_release);
403
404 // Since paranoid_checks=true, writes should fail
405 ASSERT_NOK(Put("foo2", "bar2"));
406
407 // Recovery: should not lose data
408 ASSERT_EQ("bar", Get("foo"));
409
410 // Try again with paranoid_checks=false
411 Close();
412 options.paranoid_checks = false;
413 Reopen(options);
414
415 // Merging compaction (will fail)
416 error_type->store(true, std::memory_order_release);
417 Status s =
418 dbfull()->TEST_CompactRange(last, nullptr, nullptr); // Should fail
419 if (iter == 0) {
420 ASSERT_OK(s);
421 } else {
422 ASSERT_TRUE(s.IsIOError());
423 }
424 ASSERT_EQ("bar", Get("foo"));
425
426 // Recovery: should not lose data
427 error_type->store(false, std::memory_order_release);
428 Reopen(options);
429 ASSERT_EQ("bar", Get("foo"));
430
431 // Since paranoid_checks=false, writes should succeed
432 ASSERT_OK(Put("foo2", "bar2"));
433 ASSERT_EQ("bar", Get("foo"));
434 ASSERT_EQ("bar2", Get("foo2"));
435 }
436}

438TEST_F(DBIOFailureTest, PutFailsParanoid) {
439 // Test the following:
440 // (a) A random put fails in paranoid mode (simulate by sync fail)
441 // (b) All other puts have to fail, even if writes would succeed
442 // (c) All of that should happen ONLY if paranoid_checks = true
443
444 Options options = CurrentOptions();
445 options.env = env_;
446 options.create_if_missing = true;
447 options.error_if_exists = false;
448 options.paranoid_checks = true;
449 DestroyAndReopen(options);
450 CreateAndReopenWithCF({"pikachu"}, options);
451
452 ASSERT_OK(Put(1, "foo", "bar"));
453 ASSERT_OK(Put(1, "foo1", "bar1"));
454 // simulate error
455 env_->log_write_error_.store(true, std::memory_order_release);
456 ASSERT_NOK(Put(1, "foo2", "bar2"));
457 env_->log_write_error_.store(false, std::memory_order_release);
458 // the next put should fail, too
459 ASSERT_NOK(Put(1, "foo3", "bar3"));
460 // but we're still able to read
461 ASSERT_EQ("bar", Get(1, "foo"));
462
463 // do the same thing with paranoid checks off
464 options.paranoid_checks = false;
465 DestroyAndReopen(options);
466 CreateAndReopenWithCF({"pikachu"}, options);
467
468 ASSERT_OK(Put(1, "foo", "bar"));
469 ASSERT_OK(Put(1, "foo1", "bar1"));
470 // simulate error
471 env_->log_write_error_.store(true, std::memory_order_release);
472 ASSERT_NOK(Put(1, "foo2", "bar2"));
473 env_->log_write_error_.store(false, std::memory_order_release);
474 // the next put should NOT fail
475 ASSERT_OK(Put(1, "foo3", "bar3"));
476}

1074TEST_P(DBIOCorruptionTest, DBOpenReadCorruptionRetry) {
1075 if (!std::get<2>(GetParam())) {
1076 return;
1077 }
1078 CorruptionFS* fs =
1079 static_cast<CorruptionFS*>(env_guard_->GetFileSystem().get());
1080
1081 for (int sst = 0; sst < 3; ++sst) {
1082 for (int key = 0; key < 100; ++key) {
1083 std::stringstream ss;
1084 ss << std::setw(3) << 100 * sst + key;
1085 ASSERT_OK(Put("key" + ss.str(), "val" + ss.str()));
1086 }
1087 ASSERT_OK(Flush());
1088 }
1089 Close();
1090
1091 // DB open will create table readers unless we reduce the table cache
1092 // capacity.
1093 // SanitizeOptions will set max_open_files to minimum of 20. Table cache
1094 // is allocated with max_open_files - 10 as capacity. So override
1095 // max_open_files to 11 so table cache capacity will become 1. This will
1096 // prevent file open during DB open and force the file to be opened
1097 // during MultiGet
1098 SyncPoint::GetInstance()->SetCallBack(
1099 "SanitizeOptions::AfterChangeMaxOpenFiles", [&](void* arg) {
1100 int* max_open_files = (int*)arg;
1101 *max_open_files = 11;
1102 });
1103 SyncPoint::GetInstance()->EnableProcessing();
1104
1105 // Progressively increase the IO count trigger for corruption, and verify
1106 // that it was retried
1107 int corruption_trigger = 1;
1108 fs->SetCorruptionTrigger(corruption_trigger);
1109 do {
1110 fs->SetCorruptionTrigger(corruption_trigger);
1111 ASSERT_OK(ReopenDB());
1112 for (int sst = 0; sst < 3; ++sst) {
1113 for (int key = 0; key < 100; ++key) {
1114 std::stringstream ss;
1115 ss << std::setw(3) << 100 * sst + key;
1116 ASSERT_EQ(Get("key" + ss.str()), "val" + ss.str());
1117 }
1118 }
1119 // Verify that the injected corruption was repaired
1120 ASSERT_TRUE(fs->VerifyRetry());
1121 corruption_trigger++;
1122 } while (fs->corruption_trigger() == INT_MAX);
1123}
1124
1125// The parameters are - 1. Use FS provided buffer, 2. Use async IO ReadOption,
1126// 3. Retry with verify_and_reconstruct_read IOOption

1128 testing::Combine(testing::Bool(), testing::Bool(),
