208TEST_F(DBMergeOperatorTest, MergeOperatorFailsWithMustMerge) {
209 // This is like a mini-stress test dedicated to `OpFailureScope::kMustMerge`.
210 // Some or most of it might be deleted upon adding that option to the actual
211 // stress test.
212 //
213 // "k0" and "k2" are stable (uncorrupted) keys before and after a corrupted
214 // key ("k1"). The outer loop (`i`) varies which write (`j`) to "k1" triggers
215 // the corruption. Inside that loop there are three cases:
216 //
217 // - Case 1: pure `Merge()`s
218 // - Case 2: `Merge()`s on top of a `Put()`
219 // - Case 3: `Merge()`s on top of a `Delete()`
220 //
221 // For each case we test query results before flush, after flush, and after
222 // compaction, as well as cleanup after deletion+compaction. The queries
223 // expect "k0" and "k2" to always be readable. "k1" is expected to be readable
224 // only by APIs that do not require merging, such as `GetMergeOperands()`.
225 const int kNumOperands = 3;
226 Options options = CurrentOptions();
227 options.merge_operator.reset(new TestPutOperator());
228 options.env = env_;
229 Reopen(options);
230
231 for (int i = 0; i < kNumOperands; ++i) {
232 auto check_query = [&]() {
233 {
234 std::string value;
235 ASSERT_OK(db_->Get(ReadOptions(), "k0", &value));
236 Status s = db_->Get(ReadOptions(), "k1", &value);
237 ASSERT_TRUE(s.IsCorruption());
238 ASSERT_EQ(Status::SubCode::kMergeOperatorFailed, s.subcode());
239 ASSERT_OK(db_->Get(ReadOptions(), "k2", &value));
240 }
241
242 {
243 std::unique_ptr<Iterator> iter;
244 iter.reset(db_->NewIterator(ReadOptions()));
245 iter->SeekToFirst();
246 ASSERT_TRUE(iter->Valid());
247 ASSERT_EQ("k0", iter->key());
248 iter->Next();
249 ASSERT_TRUE(iter->status().IsCorruption());
250 ASSERT_EQ(Status::SubCode::kMergeOperatorFailed,
251 iter->status().subcode());
252
253 iter->SeekToLast();
254 ASSERT_TRUE(iter->Valid());
255 ASSERT_EQ("k2", iter->key());
256 iter->Prev();
257 ASSERT_TRUE(iter->status().IsCorruption());
258
259 iter->Seek("k2");
260 ASSERT_TRUE(iter->Valid());
261 ASSERT_EQ("k2", iter->key());
262 }
263
264 std::vector<PinnableSlice> values(kNumOperands);
265 GetMergeOperandsOptions merge_operands_info;
266 merge_operands_info.expected_max_number_of_operands = kNumOperands;
267 int num_operands_found = 0;
268 ASSERT_OK(db_->GetMergeOperands(ReadOptions(), db_->DefaultColumnFamily(),
269 "k1", values.data(), &merge_operands_info,
270 &num_operands_found));
271 ASSERT_EQ(kNumOperands, num_operands_found);
272 for (int j = 0; j < num_operands_found; ++j) {
273 if (i == j) {
274 ASSERT_EQ(values[j], "corrupted_must_merge");
275 } else {
276 ASSERT_EQ(values[j], "ok");
277 }
278 }
279 };
280
281 ASSERT_OK(Put("k0", "val"));
282 ASSERT_OK(Put("k2", "val"));
283
284 // Case 1
285 for (int j = 0; j < kNumOperands; ++j) {
286 if (j == i) {
287 ASSERT_OK(Merge("k1", "corrupted_must_merge"));
288 } else {
289 ASSERT_OK(Merge("k1", "ok"));
290 }
291 }
292 check_query();
293 ASSERT_OK(Flush());
294 check_query();
295 {
296 CompactRangeOptions cro;
297 cro.bottommost_level_compaction =
298 BottommostLevelCompaction::kForceOptimized;
299 ASSERT_OK(db_->CompactRange(cro, nullptr, nullptr));
300 }
301 check_query();
302
303 // Case 2
304 for (int j = 0; j < kNumOperands; ++j) {
305 Slice val;
306 if (j == i) {
307 val = "corrupted_must_merge";
308 } else {
309 val = "ok";
310 }
311 if (j == 0) {
312 ASSERT_OK(Put("k1", val));
313 } else {
314 ASSERT_OK(Merge("k1", val));
315 }
316 }
317 check_query();
318 ASSERT_OK(Flush());
319 check_query();
320 {
321 CompactRangeOptions cro;
322 cro.bottommost_level_compaction =
323 BottommostLevelCompaction::kForceOptimized;
324 ASSERT_OK(db_->CompactRange(cro, nullptr, nullptr));
325 }
326 check_query();
327
328 // Case 3
329 ASSERT_OK(Delete("k1"));
330 for (int j = 0; j < kNumOperands; ++j) {
331 if (i == j) {
332 ASSERT_OK(Merge("k1", "corrupted_must_merge"));
333 } else {
334 ASSERT_OK(Merge("k1", "ok"));
335 }
336 }
337 check_query();
338 ASSERT_OK(Flush());
339 check_query();
340 {
341 CompactRangeOptions cro;
342 cro.bottommost_level_compaction =
343 BottommostLevelCompaction::kForceOptimized;
344 ASSERT_OK(db_->CompactRange(cro, nullptr, nullptr));
345 }
346 check_query();
347
348 // Verify obsolete data removal still happens
349 ASSERT_OK(Delete("k0"));
350 ASSERT_OK(Delete("k1"));
351 ASSERT_OK(Delete("k2"));
352 ASSERT_EQ("NOT_FOUND", Get("k0"));
353 ASSERT_EQ("NOT_FOUND", Get("k1"));
354 ASSERT_EQ("NOT_FOUND", Get("k2"));
355 CompactRangeOptions cro;
356 cro.bottommost_level_compaction =
357 BottommostLevelCompaction::kForceOptimized;
358 ASSERT_OK(db_->CompactRange(cro, nullptr, nullptr));
359 ASSERT_EQ("", FilesPerLevel());
360 }
361}

363TEST_F(DBMergeOperatorTest, MergeOperandThresholdExceeded) {
364 Options options = CurrentOptions();
365 options.create_if_missing = true;
366 options.merge_operator = MergeOperators::CreatePutOperator();
367 options.env = env_;
368 Reopen(options);
369
370 std::vector<Slice> keys{"foo", "bar", "baz"};
371
372 // Write base values.
373 for (const auto& key : keys) {
374 ASSERT_OK(Put(key, key.ToString() + "0"));
375 }
376
377 // Write merge operands. Note that the first key has 1 merge operand, the
378 // second one has 2 merge operands, and the third one has 3 merge operands.
379 // Also, we'll take some snapshots to make sure the merge operands are
380 // preserved during flush.
381 std::vector<ManagedSnapshot> snapshots;
382 snapshots.reserve(3);
383
384 for (size_t i = 0; i < keys.size(); ++i) {
385 snapshots.emplace_back(db_);
386
387 const std::string suffix = std::to_string(i + 1);
388
389 for (size_t j = i; j < keys.size(); ++j) {
390 ASSERT_OK(Merge(keys[j], keys[j].ToString() + suffix));
391 }
392 }
393
394 // Verify the results and status codes of various types of point lookups.
395 auto verify = [&](const std::optional<size_t>& threshold) {
396 ReadOptions read_options;
397 read_options.merge_operand_count_threshold = threshold;
398
399 // Check Get()
400 {
401 for (size_t i = 0; i < keys.size(); ++i) {
402 PinnableSlice value;
403 const Status status =
404 db_->Get(read_options, db_->DefaultColumnFamily(), keys[i], &value);
405 ASSERT_OK(status);
406 ASSERT_EQ(status.IsOkMergeOperandThresholdExceeded(),
407 threshold.has_value() && i + 1 > threshold.value());
408 ASSERT_EQ(value, keys[i].ToString() + std::to_string(i + 1));
409 }
410 }
411
412 // Check old-style MultiGet()
413 {
414 std::vector<std::string> values;
415 std::vector<Status> statuses = db_->MultiGet(read_options, keys, &values);
416
417 for (size_t i = 0; i < keys.size(); ++i) {
418 ASSERT_OK(statuses[i]);
419 ASSERT_EQ(statuses[i].IsOkMergeOperandThresholdExceeded(),
420 threshold.has_value() && i + 1 > threshold.value());
421 ASSERT_EQ(values[i], keys[i].ToString() + std::to_string(i + 1));
422 }
423 }
424
425 // Check batched MultiGet()
426 {
427 std::vector<PinnableSlice> values(keys.size());
428 std::vector<Status> statuses(keys.size());
429 db_->MultiGet(read_options, db_->DefaultColumnFamily(), keys.size(),
430 keys.data(), values.data(), statuses.data());
431
432 for (size_t i = 0; i < keys.size(); ++i) {
433 ASSERT_OK(statuses[i]);
434 ASSERT_EQ(statuses[i].IsOkMergeOperandThresholdExceeded(),
435 threshold.has_value() && i + 1 > threshold.value());
436 ASSERT_EQ(values[i], keys[i].ToString() + std::to_string(i + 1));
437 }
438 }
439 };
440
441 // Test the case when the feature is disabled as well as various thresholds.
442 verify(std::nullopt);
443 for (size_t i = 0; i < 5; ++i) {
444 verify(i);
445 }
446
447 // Flush and try again to test the case when results are served from SSTs.
448 ASSERT_OK(Flush());
449 verify(std::nullopt);
450 for (size_t i = 0; i < 5; ++i) {
451 verify(i);
452 }
453}

722TEST_F(DBMergeOperatorTest, TailingIteratorMemtableUnrefedBySomeoneElse) {
723 Options options = CurrentOptions();
724 options.merge_operator = MergeOperators::CreateStringAppendOperator();
725 DestroyAndReopen(options);
726
727 // Overview of the test:
728 // * There are two merge operands for the same key: one in an sst file,
729 // another in a memtable.
730 // * Seek a tailing iterator to this key.
731 // * As part of the seek, the iterator will:
732 // (a) first visit the operand in the memtable and tell ForwardIterator
733 // to pin this operand, then
734 // (b) move on to the operand in the sst file, then pass both operands
735 // to merge operator.
736 // * The memtable may get flushed and unreferenced by another thread between
737 // (a) and (b). The test simulates it by flushing the memtable inside a
738 // SyncPoint callback located between (a) and (b).
739 // * In this case it's ForwardIterator's responsibility to keep the memtable
740 // pinned until (b) is complete. There used to be a bug causing
741 // ForwardIterator to not pin it in some circumstances. This test
742 // reproduces it.
743
744 ASSERT_OK(db_->Merge(WriteOptions(), "key", "sst"));
745 ASSERT_OK(db_->Flush(FlushOptions())); // Switch to SuperVersion A
746 ASSERT_OK(db_->Merge(WriteOptions(), "key", "memtable"));
747
748 // Pin SuperVersion A
749 std::unique_ptr<Iterator> someone_else(db_->NewIterator(ReadOptions()));
750 ASSERT_OK(someone_else->status());
751
752 bool pushed_first_operand = false;
753 bool stepped_to_next_operand = false;
754 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
755 "DBIter::MergeValuesNewToOld:PushedFirstOperand", [&](void*) {
756 EXPECT_FALSE(pushed_first_operand);
757 pushed_first_operand = true;
758 EXPECT_OK(db_->Flush(FlushOptions())); // Switch to SuperVersion B
759 });
760 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
761 "DBIter::MergeValuesNewToOld:SteppedToNextOperand", [&](void*) {
762 EXPECT_FALSE(stepped_to_next_operand);
763 stepped_to_next_operand = true;
764 someone_else.reset(); // Unpin SuperVersion A
765 });
766 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
767
768 ReadOptions ro;
769 ro.tailing = true;
770 std::unique_ptr<Iterator> iter(db_->NewIterator(ro));
771 iter->Seek("key");
772
773 ASSERT_OK(iter->status());
774 ASSERT_TRUE(iter->Valid());
775 EXPECT_EQ(std::string("sst,memtable"), iter->value().ToString());
776 EXPECT_TRUE(pushed_first_operand);
777 EXPECT_TRUE(stepped_to_next_operand);
