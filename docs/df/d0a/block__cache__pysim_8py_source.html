150class HashTable:
151 """
152 A custom implementation of hash table to support fast random sampling.
153 It is closed hashing and uses chaining to resolve hash conflicts.
154 It grows/shrinks the hash table upon insertion/deletion to support
155 fast lookups and random samplings.
156 """
157
158 def __init__(self):
159 self.initial_size = 32
160 self.table = [None] * self.initial_size
161 self.elements = 0
162
163 def random_sample(self, sample_size):
164 """Randomly sample 'sample_size' hash entries from the table."""
165 samples = []
166 index = random.randint(0, len(self.table) - 1)
167 pos = index
168 # Starting from index, adding hash entries to the sample list until
169 # sample_size is met or we ran out of entries.
170 while True:
171 if self.table[pos] is not None:
172 for i in range(len(self.table[pos])):
173 if self.table[pos][i] is None:
174 continue
175 samples.append(self.table[pos][i])
176 if len(samples) == sample_size:
177 break
178 pos += 1
179 pos = pos % len(self.table)
180 if pos == index or len(samples) == sample_size:
181 break
182 assert len(samples) <= sample_size
183 return samples
184
185 def __repr__(self):
186 all_entries = []
187 for i in range(len(self.table)):
188 if self.table[i] is None:
189 continue
190 for j in range(len(self.table[i])):
191 if self.table[i][j] is not None:
192 all_entries.append(self.table[i][j])
193 return f"{all_entries}"
194
195 def values(self):
196 all_values = []
197 for i in range(len(self.table)):
198 if self.table[i] is None:
199 continue
200 for j in range(len(self.table[i])):
201 if self.table[i][j] is not None:
202 all_values.append(self.table[i][j].value)
203 return all_values
204
205 def __len__(self):
206 return self.elements
207
208 def insert(self, key, hash, value):
209 """
210 Insert a hash entry in the table. Replace the old entry if it already
211 exists.
212 """
213 self.grow()
214 inserted = False
215 index = hash % len(self.table)
216 if self.table[index] is None:
217 self.table[index] = []
218 # Search for the entry first.
219 for i in range(len(self.table[index])):
220 if self.table[index][i] is None:
221 continue
222 if self.table[index][i].hash == hash and self.table[index][i].key == key:
223 # The entry already exists in the table.
224 self.table[index][i] = HashEntry(key, hash, value)
225 return
226
227 # Find an empty slot.
228 for i in range(len(self.table[index])):
229 if self.table[index][i] is None:
230 self.table[index][i] = HashEntry(key, hash, value)
231 inserted = True
232 break
233 if not inserted:
234 self.table[index].append(HashEntry(key, hash, value))
235 self.elements += 1
236
237 def resize(self, new_size):
238 if new_size == len(self.table):
239 return
240 if new_size < self.initial_size:
241 return
242 if self.elements < 100:
243 return
244 new_table = [None] * new_size
245 # Copy 'self.table' to new_table.
246 for i in range(len(self.table)):
247 entries = self.table[i]
248 if entries is None:
249 continue
250 for j in range(len(entries)):
251 if entries[j] is None:
252 continue
253 index = entries[j].hash % new_size
254 if new_table[index] is None:
255 new_table[index] = []
256 new_table[index].append(entries[j])
257 self.table = new_table
258 del new_table
259 # Manually call python gc here to free the memory as 'self.table'
260 # might be very large.
261 gc.collect()
262
263 def grow(self):
264 if self.elements < 4 * len(self.table):
265 return
266 new_size = int(len(self.table) * 1.5)
267 self.resize(new_size)
268
269 def delete(self, key, hash):
270 index = hash % len(self.table)
271 deleted = False
272 deleted_entry = None
273 if self.table[index] is None:
274 return
275 for i in range(len(self.table[index])):
276 if (
277 self.table[index][i] is not None
278 and self.table[index][i].hash == hash
279 and self.table[index][i].key == key
280 ):
281 deleted_entry = self.table[index][i]
282 self.table[index][i] = None
283 self.elements -= 1
284 deleted = True
285 break
286 if deleted:
287 self.shrink()
288 return deleted_entry
289
290 def shrink(self):
291 if self.elements * 2 >= len(self.table):
292 return
293 new_size = int(len(self.table) * 0.7)
294 self.resize(new_size)
295
296 def lookup(self, key, hash):
297 index = hash % len(self.table)
298 if self.table[index] is None:
299 return None
300 for i in range(len(self.table[index])):
301 if (
302 self.table[index][i] is not None
303 and self.table[index][i].hash == hash
304 and self.table[index][i].key == key
305 ):
306 return self.table[index][i].value
307 return None

555class HyperbolicPolicy(Policy):
556 """
557 An implementation of Hyperbolic caching.
558
559 Aaron Blankstein, Siddhartha Sen, and Michael J. Freedman. 2017.
560 Hyperbolic caching: flexible caching for web applications. In Proceedings
561 of the 2017 USENIX Conference on Usenix Annual Technical Conference
562 (USENIX ATC '17). USENIX Association, Berkeley, CA, USA, 499-511.
563 """
564
565 def compare(self, e1, e2, now):
566 e1_duration = max(0, (now - e1.value.insertion_time) / kMicrosInSecond) * float(
567 e1.value.value_size
568 )
569 e2_duration = max(0, (now - e2.value.insertion_time) / kMicrosInSecond) * float(
570 e2.value.value_size
571 )
572 if e1_duration == e2_duration:
573 return e1.value.num_hits - e2.value.num_hits
574 if e1_duration == 0:
575 return 1
576 if e2_duration == 0:
577 return 1
578 diff = (float(e1.value.num_hits) / (float(e1_duration))) - (
579 float(e2.value.num_hits) / float(e2_duration)
580 )
581 if diff == 0:
582 return 0
583 elif diff > 0:
584 return 1
585 else:
586 return -1

589 assert len(auxilliary_info) == 3

597class CostClassPolicy(Policy):
598 """
599 We calculate the hit density of a cost class as
600 number of hits / total size in cache * average duration in the cache.
601
602 An entry has a higher priority if its class's hit density is higher.
603 """
604
605 def compare(self, e1, e2, now, cost_classes, cost_class_label):
606 e1_class = e1.value.cost_class(cost_class_label)
607 e2_class = e2.value.cost_class(cost_class_label)
608
609 assert e1_class in cost_classes
610 assert e2_class in cost_classes
611
612 e1_entry = cost_classes[e1_class]
613 e2_entry = cost_classes[e2_class]
614 e1_density = e1_entry.density(now)
615 e2_density = e2_entry.density(now)
616 e1_hits = cost_classes[e1_class].hits
617 e2_hits = cost_classes[e2_class].hits
618
619 if e1_density == e2_density:
620 return e1_hits - e2_hits
621
622 if e1_entry.num_entries_in_cache == 0:
623 return -1
624 if e2_entry.num_entries_in_cache == 0:
625 return 1
626
627 if e1_density == 0:
628 return 1
629 if e2_density == 0:
630 return -1
631 diff = (float(e1_hits) / float(e1_density)) - (
632 float(e2_hits) / float(e2_density)
633 )
634 if diff == 0:
635 return 0
636 elif diff > 0:
637 return 1
638 else:
639 return -1
640
641 def prioritize_samples(self, samples, auxilliary_info):
642 assert len(auxilliary_info) == 3
643 now = auxilliary_info[0]
644 cost_classes = auxilliary_info[1]
645 cost_class_label = auxilliary_info[2]
646 return sorted(
647 samples,
648 cmp=lambda e1, e2: self.compare(
649 e1, e2, now, cost_classes, cost_class_label
650 ),
651 )

780 # We treat future accesses as hits since this get request
781 # completes.
782 # print("row hit 2")

804 trace_record.kv_size > 0

809 trace_record,
810 key=row_key,
811 hash=trace_record.key_id,
812 value_size=trace_record.kv_size,
813 no_insert=False,
814 )

816 self.get_id_row_key_map[trace_record.get_id][row_key] = True

818 def _access_kv(self, trace_record, key, hash, value_size, no_insert):
819 # Sanity checks.
820 assert self.used_size <= self.cache_size
821 if self._lookup(trace_record, key, hash):
822 # A cache hit.
823 return True
824 if no_insert or value_size <= 0:
825 return False
826 # A cache miss.
827 if value_size > self.cache_size:
828 # The block is too large to fit into the cache.
829 return False
830 self._evict(trace_record, key, hash, value_size)
831 if self._should_admit(trace_record, key, hash, value_size):
832 self._insert(trace_record, key, hash, value_size)
833 self.used_size += value_size
834 return False

862 self.sum_last_access_time -= last_access_time / kMicrosInSecond

867 self.sum_last_access_time += trace_record.access_time / kMicrosInSecond

911 if self.cost_class_label is not None:
912 cost_class = value.cost_class(self.cost_class_label)
913 assert cost_class in self.cost_classes

935 def _evict(self, trace_record, key, hash, value_size):
936 # Select a policy, random sample kSampleSize keys from the cache, then
937 # evict keys in the sample set until we have enough room for the new
938 # entry.
939 policy_index = self._select_policy(trace_record, key)
940 assert policy_index < len(self.policies) and policy_index >= 0
941 self.policies[policy_index].delete(key)
942 self.policy_stats.update_metrics(trace_record.access_time, policy_index)
943 self.per_hour_policy_stats.update_metrics(
944 trace_record.access_time, policy_index
945 )

947 # Randomly sample n entries.
948 samples = self.table.random_sample(kSampleSize)
949 samples = self.policies[policy_index].prioritize_samples(
950 samples,
951 [trace_record.access_time, self.cost_classes, self.cost_class_label],
952 )
953 for hash_entry in samples:
954 assert self.table.delete(hash_entry.key, hash_entry.hash) is not None
955 self.used_size -= hash_entry.value.value_size
956 self.policies[policy_index].evict(
957 key=hash_entry.key, max_size=self.table.elements
958 )
959 # Update the entry's cost class statistics.
960 if self.cost_class_label is not None:
961 cost_class = hash_entry.value.cost_class(self.cost_class_label)
962 assert cost_class in self.cost_classes
963 self.cost_classes[cost_class].remove(
964 hash_entry.value.insertion_time,
965 hash_entry.value.last_access_time,
966 key,
967 hash_entry.value.value_size,
968 hash_entry.value.num_hits,
969 )
970 if self.used_size + value_size <= self.cache_size:
971 break

974 assert self.used_size + value_size <= self.cache_size
975 entry = CacheEntry(
976 value_size,
977 trace_record.cf_id,
978 trace_record.level,
979 trace_record.block_type,
980 trace_record.table_id,
981 self.miss_ratio_stats.num_accesses,
982 trace_record.access_time,
983 )
984 # Update the entry's cost class statistics.
985 if self.cost_class_label is not None:
986 cost_class = entry.cost_class(self.cost_class_label)
987 if cost_class not in self.cost_classes:
988 self.cost_classes[cost_class] = CostClassEntry()
989 self.cost_classes[cost_class].insert(trace_record, key, value_size)
990 self.table.insert(key, hash, entry)

1048class LinUCBCache(MLCache):
1049 """
1050 An implementation of LinUCB with disjoint linear models.
1051
1052 Lihong Li, Wei Chu, John Langford, and Robert E. Schapire. 2010.
1053 A contextual-bandit approach to personalized news article recommendation.
1054 In Proceedings of the 19th international conference on World wide web
1055 (WWW '10). ACM, New York, NY, USA, 661-670.
1056 DOI=http://dx.doi.org/10.1145/1772690.1772758
1057 """
1058
1059 def __init__(self, cache_size, enable_cache_row_key, policies, cost_class_label):
1060 super().__init__(
1061 cache_size, enable_cache_row_key, policies, cost_class_label
1062 )
1063 self.nfeatures = 4 # Block type, level, cf.
1064 self.th = np.zeros((len(self.policies), self.nfeatures))
1065 self.eps = 0.2
1066 self.b = np.zeros_like(self.th)
1067 self.A = np.zeros((len(self.policies), self.nfeatures, self.nfeatures))
1068 self.A_inv = np.zeros((len(self.policies), self.nfeatures, self.nfeatures))
1069 for i in range(len(self.policies)):
1070 self.A[i] = np.identity(self.nfeatures)
1071 self.th_hat = np.zeros_like(self.th)
1072 self.p = np.zeros(len(self.policies))
1073 self.alph = 0.2

1075 def _select_policy(self, trace_record, key):
1076 if len(self.policies) == 1:
1077 return 0
1078 x_i = np.zeros(self.nfeatures) # The current context vector
1079 x_i[0] = trace_record.block_type
1080 x_i[1] = trace_record.level
1081 x_i[2] = trace_record.cf_id
1082 p = np.zeros(len(self.policies))
1083 for a in range(len(self.policies)):
1084 self.th_hat[a] = self.A_inv[a].dot(self.b[a])
1085 ta = x_i.dot(self.A_inv[a]).dot(x_i)
1086 a_upper_ci = self.alph * np.sqrt(ta)
1087 a_mean = self.th_hat[a].dot(x_i)
1088 p[a] = a_mean + a_upper_ci
1089 p = p + (np.random.random(len(p)) * 0.000001)
1090 selected_policy = p.argmax()
1091 reward = self.policies[selected_policy].generate_reward(key)
1092 assert reward <= 1 and reward >= 0
1093 self.A[selected_policy] += np.outer(x_i, x_i)
1094 self.b[selected_policy] += reward * x_i
1095 self.A_inv[selected_policy] = np.linalg.inv(self.A[selected_policy])
1096 del x_i
1097 return selected_policy

1131class PQTable:
1132 """
1133 A hash table with a priority queue.
1134 """
1135
1136 def __init__(self):
1137 # A list of entries arranged in a heap sorted based on the entry custom
1138 # implementation of __cmp__
1139 self.pq = []
1140 self.table = {}
1141
1142 def pqinsert(self, entry):
1143 "Add a new key or update the priority of an existing key"
1144 # Remove the entry from the table first.
1145 removed_entry = self.table.pop(entry.key, None)
1146 if removed_entry:
1147 # Mark as removed since there is no 'remove' API in heappq.
1148 # Instead, an entry in pq is removed lazily when calling pop.
1149 removed_entry.is_removed = True
1150 self.table[entry.key] = entry
1151 heapq.heappush(self.pq, entry)
1152 return removed_entry
1153
1154 def pqpop(self):
1155 while self.pq:
1156 entry = heapq.heappop(self.pq)
1157 if not entry.is_removed:
1158 del self.table[entry.key]
1159 return entry
1160 return None
1161
1162 def pqpeek(self):
1163 while self.pq:
1164 entry = self.pq[0]
1165 if not entry.is_removed:
1166 return entry
1167 heapq.heappop(self.pq)
1168 return
1169
1170 def __contains__(self, k):
1171 return k in self.table
1172
1173 def __getitem__(self, k):
1174 return self.table[k]
1175
1176 def __len__(self):
1177 return len(self.table)

1353 2) When we insert an entry, the cache evicts entries in both t1 and t2
1354 queues until it has enough space for the new entry. This also requires
1355 modification of the algorithm to maintain a maximum of 2*c blocks.
1356
1357 Nimrod Megiddo and Dharmendra S. Modha. 2003. ARC: A Self-Tuning, Low
1358 Overhead Replacement Cache. In Proceedings of the 2nd USENIX Conference on
1359 File and Storage Technologies (FAST '03). USENIX Association, Berkeley, CA,
1360 USA, 115-130.

1363 def __init__(self, cache_size, enable_cache_row_key):
1364 super().__init__(cache_size, enable_cache_row_key)
1365 self.table = {}
1366 self.c = cache_size / 16 * 1024 # Number of elements in the cache.
1367 self.p = 0 # Target size for the list T1
1368 # L1: only once recently
1369 self.t1 = Deque() # T1: recent cache entries
1370 self.b1 = Deque() # B1: ghost entries recently evicted from the T1 cache
1371 # L2: at least twice recently
1372 self.t2 = Deque() # T2: frequent entries
1373 self.b2 = Deque() # B2: ghost entries recently evicted from the T2 cache

1375 def _replace(self, key, value_size):

1377 if self.t1 and ((key in self.b2) or (len(self.t1) > self.p)):
1378 old = self.t1.pop()

1380 else:

1383 self.b2.appendleft(old)
1384 else:

1392 # Move key to MRU position in T2.
1393 if key in self.t1:

1396 return True
1397
1398 if key in self.t2:

1401 return True
1402 return False

1404 def _evict(self, trace_record, key, hash, value_size):
1405 # Case II: key is in B1
1406 # Move x from B1 to the MRU position in T2 (also fetch x to the cache).
1407 if key in self.b1:
1408 self.p = min(self.c, self.p + max(len(self.b2) / len(self.b1), 1))

1410 self.b1.remove(key)

1413
1414 # Case III: key is in B2
1415 # Move x from B2 to the MRU position in T2 (also fetch x to the cache).
1416 if key in self.b2:
1417 self.p = max(0, self.p - max(len(self.b1) / len(self.b2), 1))
1418 self._replace(key, value_size)
1419 self.b2.remove(key)
1420 self.t2.appendleft(key)
1421 return
1422
1423 # Case IV: key is not in (T1 u B1 u T2 u B2)
1424 self._replace(key, value_size)
1425 while len(self.t1) + len(self.b1) >= self.c and self.b1:
1426 self.b1.pop()
1427
1428 total = len(self.t1) + len(self.b1) + len(self.t2) + len(self.b2)
1429 while total >= (2 * self.c) and self.b2:
1430 self.b2.pop()
1431 total -= 1
1432 # Finally, move it to MRU position in T1.
1433 self.t1.appendleft(key)
1434 return
