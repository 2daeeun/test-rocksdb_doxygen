72void range_buffer::record_header::init(const DBT *left_key,
73 const DBT *right_key,
74 bool is_exclusive) {
75 is_exclusive_lock = is_exclusive;
76 left_neg_inf = left_key == toku_dbt_negative_infinity();
77 left_pos_inf = left_key == toku_dbt_positive_infinity();
78 left_key_size = toku_dbt_is_infinite(left_key) ? 0 : left_key->size;
79 if (right_key) {
80 right_neg_inf = right_key == toku_dbt_negative_infinity();
81 right_pos_inf = right_key == toku_dbt_positive_infinity();
82 right_key_size = toku_dbt_is_infinite(right_key) ? 0 : right_key->size;
83 } else {
84 right_neg_inf = left_neg_inf;
85 right_pos_inf = left_pos_inf;
86 right_key_size = 0;
87 }
88}

114void range_buffer::iterator::record::deserialize(const char *buf) {
115 size_t current = 0;
116
117 // deserialize the header
118 memcpy(&_header, buf, sizeof(record_header));
119 current += sizeof(record_header);
120
121 // deserialize the left key if necessary
122 if (!_header.left_is_infinite()) {
123 // point the left DBT's buffer into ours
124 toku_fill_dbt(&_left_key, buf + current, _header.left_key_size);
125 current += _header.left_key_size;
126 }
127
128 // deserialize the right key if necessary
129 if (!_header.right_is_infinite()) {
130 if (_header.right_key_size == 0) {
131 toku_copyref_dbt(&_right_key, _left_key);
132 } else {
133 toku_fill_dbt(&_right_key, buf + current, _header.right_key_size);
134 }
135 }
136}

159bool range_buffer::iterator::current(record *rec) {
160 if (_current_chunk_offset < _current_chunk_max) {
161 const char *buf = static_cast<const char *>(_current_chunk_base);
162 rec->deserialize(buf + _current_chunk_offset);
163 _current_rec_size = rec->size();
164 return true;
165 } else {
166 return false;
167 }
168}

170// move the iterator to the next record in the buffer
171void range_buffer::iterator::next(void) {
172 invariant(_current_chunk_offset < _current_chunk_max);
173 invariant(_current_rec_size > 0);
174
175 // the next record is _current_rec_size bytes forward
176 _current_chunk_offset += _current_rec_size;
177 // now, we don't know how big the current is, set it to 0.
178 _current_rec_size = 0;
179
180 if (_current_chunk_offset >= _current_chunk_max) {
181 // current chunk is exhausted, try moving to the next one
182 if (_ma_chunk_iterator.more()) {
183 _ma_chunk_iterator.next();
184 reset_current_chunk();
185 }
186 }
187}

189void range_buffer::create(void) {

192 _arena.create(0);

196void range_buffer::append(const DBT *left_key, const DBT *right_key,
197 bool is_write_request) {
198 // if the keys are equal, then only one copy is stored.
199 if (toku_dbt_equals(left_key, right_key)) {
200 invariant(left_key->size <= MAX_KEY_SIZE);
201 append_point(left_key, is_write_request);
202 } else {
203 invariant(left_key->size <= MAX_KEY_SIZE);
204 invariant(right_key->size <= MAX_KEY_SIZE);
205 append_range(left_key, right_key, is_write_request);
206 }
207 _num_ranges++;
208}

220void range_buffer::append_range(const DBT *left_key, const DBT *right_key,
221 bool is_exclusive) {
222 size_t record_length =
223 sizeof(record_header) + left_key->size + right_key->size;
224 char *buf = static_cast<char *>(_arena.malloc_from_arena(record_length));
225
226 record_header h;
227 h.init(left_key, right_key, is_exclusive);
228
229 // serialize the header
230 memcpy(buf, &h, sizeof(record_header));
231 buf += sizeof(record_header);
232
233 // serialize the left key if necessary
234 if (!h.left_is_infinite()) {
235 memcpy(buf, left_key->data, left_key->size);
236 buf += left_key->size;
237 }
238
239 // serialize the right key if necessary
240 if (!h.right_is_infinite()) {
241 memcpy(buf, right_key->data, right_key->size);
242 }
243}

245void range_buffer::append_point(const DBT *key, bool is_exclusive) {
246 size_t record_length = sizeof(record_header) + key->size;
247 char *buf = static_cast<char *>(_arena.malloc_from_arena(record_length));
248
249 record_header h;
250 h.init(key, nullptr, is_exclusive);
251
252 // serialize the header
253 memcpy(buf, &h, sizeof(record_header));
254 buf += sizeof(record_header);
255
256 // serialize the key if necessary
257 if (!h.left_is_infinite()) {
258 memcpy(buf, key->data, key->size);
259 }
260}
