20// VolatileCacheTier
21//
22// This file provides persistent cache tier implementation for caching
23// key/values in RAM.
24//
25// key/values
26// |
27// V
28// +-------------------+
29// | VolatileCacheTier | Store in an evictable hash table
30// +-------------------+
31// |
32// V
33// on eviction
34// pushed to next tier
35//
36// The implementation is designed to be concurrent. The evictable hash table
37// implementation is not concurrent at this point though.
38//
39// The eviction algorithm is LRU
40namespace ROCKSDB_NAMESPACE {
41
42class VolatileCacheTier : public PersistentCacheTier {
43 public:
44 explicit VolatileCacheTier(
45 const bool is_compressed = true,
46 const size_t max_size = std::numeric_limits<size_t>::max())
47 : is_compressed_(is_compressed), max_size_(max_size) {}
48
49 virtual ~VolatileCacheTier();
50
51 // insert to cache
52 Status Insert(const Slice& page_key, const char* data,
53 const size_t size) override;
54 // lookup key in cache
55 Status Lookup(const Slice& page_key, std::unique_ptr<char[]>* data,
56 size_t* size) override;
57
58 // is compressed cache ?
59 bool IsCompressed() override { return is_compressed_; }
60
61 // erase key from cache
62 bool Erase(const Slice& key) override;
63
64 std::string GetPrintableOptions() const override {
65 return "VolatileCacheTier";
66 }
67
68 // Expose stats as map
69 PersistentCache::StatsType Stats() override;
70
71 private:
72 //
73 // Cache data abstraction
74 //
75 struct CacheData : LRUElement<CacheData> {
76 explicit CacheData(CacheData&& rhs) noexcept
77 : key(std::move(rhs.key)), value(std::move(rhs.value)) {}
78
79 explicit CacheData(const std::string& _key, const std::string& _value = "")
80 : key(_key), value(_value) {}
81
82 virtual ~CacheData() {}
83
84 const std::string key;
85 const std::string value;
86 };
87
88 static void DeleteCacheData(CacheData* data);
89
90 //
91 // Index and LRU definition
92 //
93 struct CacheDataHash {
94 uint64_t operator()(const CacheData* obj) const {
95 assert(obj);
96 return std::hash<std::string>()(obj->key);
97 }
98 };
99
100 struct CacheDataEqual {
101 bool operator()(const CacheData* lhs, const CacheData* rhs) const {
102 assert(lhs);
103 assert(rhs);
104 return lhs->key == rhs->key;
105 }
106 };
107
108 struct Statistics {
109 std::atomic<uint64_t> cache_misses_{0};
110 std::atomic<uint64_t> cache_hits_{0};
111 std::atomic<uint64_t> cache_inserts_{0};
112 std::atomic<uint64_t> cache_evicts_{0};
113
114 double CacheHitPct() const {
115 auto lookups = cache_hits_ + cache_misses_;
116 return lookups ? 100 * cache_hits_ / static_cast<double>(lookups) : 0.0;
117 }
118
119 double CacheMissPct() const {
120 auto lookups = cache_hits_ + cache_misses_;
121 return lookups ? 100 * cache_misses_ / static_cast<double>(lookups) : 0.0;
122 }
123 };
124
125 using IndexType =
126 EvictableHashTable<CacheData, CacheDataHash, CacheDataEqual>;
127
128 // Evict LRU tail
129 bool Evict();
130
131 const bool is_compressed_ = true; // does it store compressed data
132 IndexType index_; // in-memory cache
133 std::atomic<uint64_t> max_size_{0}; // Maximum size of the cache
134 std::atomic<uint64_t> size_{0}; // Size of the cache
135 Statistics stats_;
136};
137
138} // namespace ROCKSDB_NAMESPACE
