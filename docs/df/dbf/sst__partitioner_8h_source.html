40/*
41 * A SstPartitioner is a generic pluggable way of defining the partition
42 * of SST files. Compaction job will split the SST files on partition boundary
43 * to lower the write amplification during SST file promote to higher level.
44 */
45class SstPartitioner {
46 public:
47 virtual ~SstPartitioner() {}
48
49 // Return the name of this partitioner.
50 virtual const char* Name() const = 0;
51
52 // It is called for all keys in compaction. When partitioner want to create
53 // new SST file it needs to return true. It means compaction job will finish
54 // current SST file where last key is "prev_user_key" parameter and start new
55 // SST file where first key is "current_user_key". Returns decision if
56 // partition boundary was detected and compaction should create new file.
57 virtual PartitionerResult ShouldPartition(
58 const PartitionerRequest& request) = 0;
59
60 // Called with smallest and largest keys in SST file when compaction try to do
61 // trivial move. Returns true is partitioner allows to do trivial move.
62 virtual bool CanDoTrivialMove(const Slice& smallest_user_key,
63 const Slice& largest_user_key) = 0;
64
65 // Context information of a compaction run
66 struct Context {
67 // Does this compaction run include all data files
68 bool is_full_compaction;
69 // Is this compaction requested by the client (true),
70 // or is it occurring as an automatic compaction process
71 bool is_manual_compaction;
72 // Output level for this compaction
73 int output_level;
74 // Smallest key for compaction
75 Slice smallest_user_key;
76 // Largest key for compaction
77 Slice largest_user_key;
78 };
79};
