43// The metadata of a dump unit. After it is serialized, its size is fixed 16
44// bytes.
45struct DumpUnitMeta {
46 // sequence number is a monotonically increasing number to indicate the order
47 // of the blocks being written. Header is 0.
48 uint32_t sequence_num;
49 // The Crc32c checksum of its dump unit.
50 uint32_t dump_unit_checksum;
51 // The dump unit size after the dump unit is serialized to a string.
52 uint64_t dump_unit_size;
53
54 void reset() {
55 sequence_num = 0;
56 dump_unit_checksum = 0;
57 dump_unit_size = 0;
58 }
59};
60
61// The data structure to hold a block and its information.
62struct DumpUnit {
63 // The timestamp when the block is identified, copied, and dumped from block
64 // cache
65 uint64_t timestamp;
66 // The type of the block
67 CacheDumpUnitType type;
68 // The key of this block when the block is referenced by this Cache
69 Slice key;
70 // The block size
71 size_t value_len;
72 // The Crc32c checksum of the block
73 uint32_t value_checksum;
74 // Pointer to the block. Note that, in the dump process, it points to a memory
75 // buffer copied from cache block. The buffer is freed when we process the
76 // next block. In the load process, we use an std::string to store the
77 // serialized dump_unit read from the reader. So it points to the memory
78 // address of the begin of the block in this string.
79 void* value;
80
81 DumpUnit() { reset(); }
82
83 void reset() {
84 timestamp = 0;
85 type = CacheDumpUnitType::kBlockTypeMax;
86 key.clear();
87 value_len = 0;
88 value_checksum = 0;
89 value = nullptr;
90 }
91};

157// The default implementation of CacheDumpWriter. We write the blocks to a file
158// sequentially.
159class ToFileCacheDumpWriter : public CacheDumpWriter {
160 public:
161 explicit ToFileCacheDumpWriter(
162 std::unique_ptr<WritableFileWriter>&& file_writer)
163 : file_writer_(std::move(file_writer)) {}
164
165 ~ToFileCacheDumpWriter() { Close().PermitUncheckedError(); }
166
167 // Write the serialized metadata to the file
168 IOStatus WriteMetadata(const Slice& metadata) override {
169 assert(file_writer_ != nullptr);
170 std::string prefix;
171 PutFixed32(&prefix, static_cast<uint32_t>(metadata.size()));
172 const IOOptions opts;
173 IOStatus io_s = file_writer_->Append(opts, Slice(prefix));
174 if (!io_s.ok()) {
175 return io_s;
176 }
177 io_s = file_writer_->Append(opts, metadata);
178 return io_s;
179 }
180
181 // Write the serialized data to the file
182 IOStatus WritePacket(const Slice& data) override {
183 assert(file_writer_ != nullptr);
184 std::string prefix;
185 PutFixed32(&prefix, static_cast<uint32_t>(data.size()));
186 const IOOptions opts;
187 IOStatus io_s = file_writer_->Append(opts, Slice(prefix));
188 if (!io_s.ok()) {
189 return io_s;
190 }
191 io_s = file_writer_->Append(opts, data);
192 return io_s;
193 }
194
195 // Reset the writer
196 IOStatus Close() override {
197 IOStatus io_s;
198 if (file_writer_ != nullptr && !file_writer_->seen_error()) {
199 io_s = file_writer_->Sync(IOOptions(), false /* use_fsync */);
200 }
201 file_writer_.reset();
202 return io_s;
203 }
204
205 private:
206 std::unique_ptr<WritableFileWriter> file_writer_;
207};

209// The default implementation of CacheDumpReader. It is implemented based on
210// RandomAccessFileReader. Note that, we keep an internal variable to remember
211// the current offset.
212class FromFileCacheDumpReader : public CacheDumpReader {
213 public:
214 explicit FromFileCacheDumpReader(
215 std::unique_ptr<RandomAccessFileReader>&& reader)
216 : file_reader_(std::move(reader)),
217 offset_(0),
218 buffer_(new char[kDumpReaderBufferSize]) {}
219
220 ~FromFileCacheDumpReader() { delete[] buffer_; }
221
222 IOStatus ReadMetadata(std::string* metadata) override {
223 uint32_t metadata_len = 0;
224 IOStatus io_s = ReadSizePrefix(&metadata_len);
225 if (!io_s.ok()) {
226 return io_s;
227 }
228 return Read(metadata_len, metadata);
229 }
230
231 IOStatus ReadPacket(std::string* data) override {
232 uint32_t data_len = 0;
233 IOStatus io_s = ReadSizePrefix(&data_len);
234 if (!io_s.ok()) {
235 return io_s;
236 }
237 return Read(data_len, data);
238 }
239
240 private:
241 IOStatus ReadSizePrefix(uint32_t* len) {
242 std::string prefix;
243 IOStatus io_s = Read(kSizePrefixLen, &prefix);
244 if (!io_s.ok()) {
245 return io_s;
246 }
247 Slice encoded_slice(prefix);
248 if (!GetFixed32(&encoded_slice, len)) {
249 return IOStatus::Corruption("Decode size prefix string failed");
250 }
251 return IOStatus::OK();
252 }
253
254 IOStatus Read(size_t len, std::string* data) {
255 assert(file_reader_ != nullptr);
256 IOStatus io_s;
257
258 unsigned int bytes_to_read = static_cast<unsigned int>(len);
259 unsigned int to_read = bytes_to_read > kDumpReaderBufferSize
260 ? kDumpReaderBufferSize
261 : bytes_to_read;
262
263 while (to_read > 0) {
264 io_s = file_reader_->Read(IOOptions(), offset_, to_read, &result_,
265 buffer_, nullptr);
266 if (!io_s.ok()) {
267 return io_s;
268 }
269 if (result_.size() < to_read) {
270 return IOStatus::Corruption("Corrupted cache dump file.");
271 }
272 data->append(result_.data(), result_.size());
273
274 offset_ += to_read;
275 bytes_to_read -= to_read;
276 to_read = bytes_to_read > kDumpReaderBufferSize ? kDumpReaderBufferSize
277 : bytes_to_read;
278 }
279 return io_s;
280 }
281 std::unique_ptr<RandomAccessFileReader> file_reader_;
282 Slice result_;
283 size_t offset_;
284 char* buffer_;
285};

287// The cache dump and load helper class
288class CacheDumperHelper {
289 public:
290 // serialize the dump_unit_meta to a string, it is fixed 16 bytes size.
291 static void EncodeDumpUnitMeta(const DumpUnitMeta& meta, std::string* data) {
292 assert(data);
293 PutFixed32(data, static_cast<uint32_t>(meta.sequence_num));
294 PutFixed32(data, static_cast<uint32_t>(meta.dump_unit_checksum));
295 PutFixed64(data, meta.dump_unit_size);
296 }
297
298 // Serialize the dump_unit to a string.
299 static void EncodeDumpUnit(const DumpUnit& dump_unit, std::string* data) {
300 assert(data);
301 PutFixed64(data, dump_unit.timestamp);
302 data->push_back(dump_unit.type);
303 PutLengthPrefixedSlice(data, dump_unit.key);
304 PutFixed32(data, static_cast<uint32_t>(dump_unit.value_len));
305 PutFixed32(data, dump_unit.value_checksum);
306 PutLengthPrefixedSlice(data,
307 Slice((char*)dump_unit.value, dump_unit.value_len));
308 }
309
310 // Deserialize the dump_unit_meta from a string
311 static Status DecodeDumpUnitMeta(const std::string& encoded_data,
312 DumpUnitMeta* unit_meta) {
313 assert(unit_meta != nullptr);
314 Slice encoded_slice = Slice(encoded_data);
315 if (!GetFixed32(&encoded_slice, &(unit_meta->sequence_num))) {
316 return Status::Incomplete("Decode dumped unit meta sequence_num failed");
317 }
318 if (!GetFixed32(&encoded_slice, &(unit_meta->dump_unit_checksum))) {
319 return Status::Incomplete(
320 "Decode dumped unit meta dump_unit_checksum failed");
321 }
322 if (!GetFixed64(&encoded_slice, &(unit_meta->dump_unit_size))) {
323 return Status::Incomplete(
324 "Decode dumped unit meta dump_unit_size failed");
325 }
326 return Status::OK();
327 }
328
329 // Deserialize the dump_unit from a string.
330 static Status DecodeDumpUnit(const std::string& encoded_data,
331 DumpUnit* dump_unit) {
332 assert(dump_unit != nullptr);
333 Slice encoded_slice = Slice(encoded_data);
334
335 // Decode timestamp
336 if (!GetFixed64(&encoded_slice, &dump_unit->timestamp)) {
337 return Status::Incomplete("Decode dumped unit string failed");
338 }
339 // Decode the block type
340 dump_unit->type = static_cast<CacheDumpUnitType>(encoded_slice[0]);
341 encoded_slice.remove_prefix(1);
342 // Decode the key
343 if (!GetLengthPrefixedSlice(&encoded_slice, &(dump_unit->key))) {
344 return Status::Incomplete("Decode dumped unit string failed");
345 }
346 // Decode the value size
347 uint32_t value_len;
348 if (!GetFixed32(&encoded_slice, &value_len)) {
349 return Status::Incomplete("Decode dumped unit string failed");
350 }
351 dump_unit->value_len = static_cast<size_t>(value_len);
352 // Decode the value checksum
353 if (!GetFixed32(&encoded_slice, &(dump_unit->value_checksum))) {
354 return Status::Incomplete("Decode dumped unit string failed");
355 }
356 // Decode the block content and copy to the memory space whose pointer
357 // will be managed by the cache finally.
358 Slice block;
359 if (!GetLengthPrefixedSlice(&encoded_slice, &block)) {
360 return Status::Incomplete("Decode dumped unit string failed");
361 }
362 dump_unit->value = (void*)block.data();
363 assert(block.size() == dump_unit->value_len);
364 return Status::OK();
365 }
366};
