38// Implementation for the merge operation (concatenates two strings)
39bool StringAppendTESTOperator::FullMergeV2(
40 const MergeOperationInput& merge_in,
41 MergeOperationOutput* merge_out) const {
42 // Clear the *new_value for writing.
43 merge_out->new_value.clear();
44
45 if (merge_in.existing_value == nullptr && merge_in.operand_list.size() == 1) {
46 // Only one operand
47 merge_out->existing_operand = merge_in.operand_list.back();
48 return true;
49 }
50
51 // Compute the space needed for the final result.
52 size_t numBytes = 0;
53
54 for (auto it = merge_in.operand_list.begin();
55 it != merge_in.operand_list.end(); ++it) {
56 numBytes += it->size() + delim_.size();
57 }
58
59 // Only print the delimiter after the first entry has been printed
60 bool printDelim = false;
61
62 // Prepend the *existing_value if one exists.
63 if (merge_in.existing_value) {
64 merge_out->new_value.reserve(numBytes + merge_in.existing_value->size());
65 merge_out->new_value.append(merge_in.existing_value->data(),
66 merge_in.existing_value->size());
67 printDelim = true;
68 } else if (numBytes) {
69 // Without the existing (initial) value, the delimiter before the first of
70 // subsequent operands becomes redundant.
71 merge_out->new_value.reserve(numBytes - delim_.size());
72 }
73
74 // Concatenate the sequence of strings (and add a delimiter between each)
75 for (auto it = merge_in.operand_list.begin();
76 it != merge_in.operand_list.end(); ++it) {
77 if (printDelim) {
78 merge_out->new_value.append(delim_);
79 }
80 merge_out->new_value.append(it->data(), it->size());
81 printDelim = true;
82 }
83
84 return true;
85}

93// A version of PartialMerge that actually performs "partial merging".
94// Use this to simulate the exact behaviour of the StringAppendOperator.
95bool StringAppendTESTOperator::_AssocPartialMergeMulti(
96 const Slice& /*key*/, const std::deque<Slice>& operand_list,
97 std::string* new_value, Logger* /*logger*/) const {
98 // Clear the *new_value for writing
99 assert(new_value);
100 new_value->clear();
101 assert(operand_list.size() >= 2);
102
103 // Generic append
104 // Determine and reserve correct size for *new_value.
105 size_t size = 0;
106 for (const auto& operand : operand_list) {
107 size += operand.size();
108 }
109 size += (operand_list.size() - 1) * delim_.length(); // Delimiters
110 new_value->reserve(size);
111
112 // Apply concatenation
113 new_value->assign(operand_list.front().data(), operand_list.front().size());
114
115 for (std::deque<Slice>::const_iterator it = operand_list.begin() + 1;
116 it != operand_list.end(); ++it) {
117 new_value->append(delim_);
118 new_value->append(it->data(), it->size());
119 }
120
121 return true;
122}
