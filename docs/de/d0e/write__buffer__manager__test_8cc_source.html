20TEST_F(WriteBufferManagerTest, ShouldFlush) {
21 // A write buffer manager of size 10MB
22 std::unique_ptr<WriteBufferManager> wbf(
23 new WriteBufferManager(10 * 1024 * 1024));
24
25 wbf->ReserveMem(8 * 1024 * 1024);
26 ASSERT_FALSE(wbf->ShouldFlush());
27 // 90% of the hard limit will hit the condition
28 wbf->ReserveMem(1 * 1024 * 1024);
29 ASSERT_TRUE(wbf->ShouldFlush());
30 // Scheduling for freeing will release the condition
31 wbf->ScheduleFreeMem(1 * 1024 * 1024);
32 ASSERT_FALSE(wbf->ShouldFlush());
33
34 wbf->ReserveMem(2 * 1024 * 1024);
35 ASSERT_TRUE(wbf->ShouldFlush());
36
37 wbf->ScheduleFreeMem(4 * 1024 * 1024);
38 // 11MB total, 6MB mutable. hard limit still hit
39 ASSERT_TRUE(wbf->ShouldFlush());
40
41 wbf->ScheduleFreeMem(2 * 1024 * 1024);
42 // 11MB total, 4MB mutable. hard limit stills but won't flush because more
43 // than half data is already being flushed.
44 ASSERT_FALSE(wbf->ShouldFlush());
45
46 wbf->ReserveMem(4 * 1024 * 1024);
47 // 15 MB total, 8MB mutable.
48 ASSERT_TRUE(wbf->ShouldFlush());
49
50 wbf->FreeMem(7 * 1024 * 1024);
51 // 8MB total, 8MB mutable.
52 ASSERT_FALSE(wbf->ShouldFlush());
53
54 // change size: 8M limit, 7M mutable limit
55 wbf->SetBufferSize(8 * 1024 * 1024);
56 // 8MB total, 8MB mutable.
57 ASSERT_TRUE(wbf->ShouldFlush());
58
59 wbf->ScheduleFreeMem(2 * 1024 * 1024);
60 // 8MB total, 6MB mutable.
61 ASSERT_TRUE(wbf->ShouldFlush());
62
63 wbf->FreeMem(2 * 1024 * 1024);
64 // 6MB total, 6MB mutable.
65 ASSERT_FALSE(wbf->ShouldFlush());
66
67 wbf->ReserveMem(1 * 1024 * 1024);
68 // 7MB total, 7MB mutable.
69 ASSERT_FALSE(wbf->ShouldFlush());
70
71 wbf->ReserveMem(1 * 1024 * 1024);
72 // 8MB total, 8MB mutable.
73 ASSERT_TRUE(wbf->ShouldFlush());
74
75 wbf->ScheduleFreeMem(1 * 1024 * 1024);
76 wbf->FreeMem(1 * 1024 * 1024);
77 // 7MB total, 7MB mutable.
78 ASSERT_FALSE(wbf->ShouldFlush());
79}

83TEST_F(ChargeWriteBufferTest, Basic) {
84 constexpr std::size_t kMetaDataChargeOverhead = 10000;
85
86 LRUCacheOptions co;
87 // 1GB cache
88 co.capacity = 1024 * 1024 * 1024;
89 co.num_shard_bits = 4;
90 co.metadata_charge_policy = kDontChargeCacheMetadata;
91 std::shared_ptr<Cache> cache = NewLRUCache(co);
92 // A write buffer manager of size 50MB
93 std::unique_ptr<WriteBufferManager> wbf(
94 new WriteBufferManager(50 * 1024 * 1024, cache));
95
96 // Allocate 333KB will allocate 512KB, memory_used_ = 333KB
97 wbf->ReserveMem(333 * 1024);
98 // 2 dummy entries are added for size 333 KB
99 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 2 * kSizeDummyEntry);
100 ASSERT_GE(cache->GetPinnedUsage(), 2 * 256 * 1024);
101 ASSERT_LT(cache->GetPinnedUsage(), 2 * 256 * 1024 + kMetaDataChargeOverhead);
102
103 // Allocate another 512KB, memory_used_ = 845KB
104 wbf->ReserveMem(512 * 1024);
105 // 2 more dummy entries are added for size 512 KB
106 // since ceil((memory_used_ - dummy_entries_in_cache_usage) % kSizeDummyEntry)
107 // = 2
108 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 4 * kSizeDummyEntry);
109 ASSERT_GE(cache->GetPinnedUsage(), 4 * 256 * 1024);
110 ASSERT_LT(cache->GetPinnedUsage(), 4 * 256 * 1024 + kMetaDataChargeOverhead);
111
112 // Allocate another 10MB, memory_used_ = 11085KB
113 wbf->ReserveMem(10 * 1024 * 1024);
114 // 40 more entries are added for size 10 * 1024 * 1024 KB
115 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 44 * kSizeDummyEntry);
116 ASSERT_GE(cache->GetPinnedUsage(), 44 * 256 * 1024);
117 ASSERT_LT(cache->GetPinnedUsage(), 44 * 256 * 1024 + kMetaDataChargeOverhead);
118
119 // Free 1MB, memory_used_ = 10061KB
120 // It will not cause any change in cache cost
121 // since memory_used_ > dummy_entries_in_cache_usage * (3/4)
122 wbf->FreeMem(1 * 1024 * 1024);
123 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 44 * kSizeDummyEntry);
124 ASSERT_GE(cache->GetPinnedUsage(), 44 * 256 * 1024);
125 ASSERT_LT(cache->GetPinnedUsage(), 44 * 256 * 1024 + kMetaDataChargeOverhead);
126 ASSERT_FALSE(wbf->ShouldFlush());
127
128 // Allocate another 41MB, memory_used_ = 52045KB
129 wbf->ReserveMem(41 * 1024 * 1024);
130 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 204 * kSizeDummyEntry);
131 ASSERT_GE(cache->GetPinnedUsage(), 204 * 256 * 1024);
132 ASSERT_LT(cache->GetPinnedUsage(),
133 204 * 256 * 1024 + kMetaDataChargeOverhead);
134 ASSERT_TRUE(wbf->ShouldFlush());
135
136 ASSERT_TRUE(wbf->ShouldFlush());
137
138 // Schedule free 20MB, memory_used_ = 52045KB
139 // It will not cause any change in memory_used and cache cost
140 wbf->ScheduleFreeMem(20 * 1024 * 1024);
141 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 204 * kSizeDummyEntry);
142 ASSERT_GE(cache->GetPinnedUsage(), 204 * 256 * 1024);
143 ASSERT_LT(cache->GetPinnedUsage(),
144 204 * 256 * 1024 + kMetaDataChargeOverhead);
145 // Still need flush as the hard limit hits
146 ASSERT_TRUE(wbf->ShouldFlush());
147
148 // Free 20MB, memory_used_ = 31565KB
149 // It will releae 80 dummy entries from cache since
150 // since memory_used_ < dummy_entries_in_cache_usage * (3/4)
151 // and floor((dummy_entries_in_cache_usage - memory_used_) % kSizeDummyEntry)
152 // = 80
153 wbf->FreeMem(20 * 1024 * 1024);
154 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 124 * kSizeDummyEntry);
155 ASSERT_GE(cache->GetPinnedUsage(), 124 * 256 * 1024);
156 ASSERT_LT(cache->GetPinnedUsage(),
157 124 * 256 * 1024 + kMetaDataChargeOverhead);
158
159 ASSERT_FALSE(wbf->ShouldFlush());
160
161 // Free 16KB, memory_used_ = 31549KB
162 // It will not release any dummy entry since memory_used_ >=
163 // dummy_entries_in_cache_usage * (3/4)
164 wbf->FreeMem(16 * 1024);
165 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 124 * kSizeDummyEntry);
166 ASSERT_GE(cache->GetPinnedUsage(), 124 * 256 * 1024);
167 ASSERT_LT(cache->GetPinnedUsage(),
168 124 * 256 * 1024 + kMetaDataChargeOverhead);
169
170 // Free 20MB, memory_used_ = 11069KB
171 // It will releae 80 dummy entries from cache
172 // since memory_used_ < dummy_entries_in_cache_usage * (3/4)
173 // and floor((dummy_entries_in_cache_usage - memory_used_) % kSizeDummyEntry)
174 // = 80
175 wbf->FreeMem(20 * 1024 * 1024);
176 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 44 * kSizeDummyEntry);
177 ASSERT_GE(cache->GetPinnedUsage(), 44 * 256 * 1024);
178 ASSERT_LT(cache->GetPinnedUsage(), 44 * 256 * 1024 + kMetaDataChargeOverhead);
179
180 // Free 1MB, memory_used_ = 10045KB
181 // It will not cause any change in cache cost
182 // since memory_used_ > dummy_entries_in_cache_usage * (3/4)
183 wbf->FreeMem(1 * 1024 * 1024);
184 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 44 * kSizeDummyEntry);
185 ASSERT_GE(cache->GetPinnedUsage(), 44 * 256 * 1024);
186 ASSERT_LT(cache->GetPinnedUsage(), 44 * 256 * 1024 + kMetaDataChargeOverhead);
187
188 // Reserve 512KB, memory_used_ = 10557KB
189 // It will not casue any change in cache cost
190 // since memory_used_ > dummy_entries_in_cache_usage * (3/4)
191 // which reflects the benefit of saving dummy entry insertion on memory
192 // reservation after delay decrease
193 wbf->ReserveMem(512 * 1024);
194 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 44 * kSizeDummyEntry);
195 ASSERT_GE(cache->GetPinnedUsage(), 44 * 256 * 1024);
196 ASSERT_LT(cache->GetPinnedUsage(), 44 * 256 * 1024 + kMetaDataChargeOverhead);
197
198 // Destroy write buffer manger should free everything
199 wbf.reset();
200 ASSERT_EQ(cache->GetPinnedUsage(), 0);
201}

203TEST_F(ChargeWriteBufferTest, BasicWithNoBufferSizeLimit) {
204 constexpr std::size_t kMetaDataChargeOverhead = 10000;
205 // 1GB cache
206 std::shared_ptr<Cache> cache = NewLRUCache(1024 * 1024 * 1024, 4);
207 // A write buffer manager of size 256MB
208 std::unique_ptr<WriteBufferManager> wbf(new WriteBufferManager(0, cache));
209
210 // Allocate 10MB, memory_used_ = 10240KB
211 // It will allocate 40 dummy entries
212 wbf->ReserveMem(10 * 1024 * 1024);
213 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 40 * kSizeDummyEntry);
214 ASSERT_GE(cache->GetPinnedUsage(), 40 * 256 * 1024);
215 ASSERT_LT(cache->GetPinnedUsage(), 40 * 256 * 1024 + kMetaDataChargeOverhead);
216
217 ASSERT_FALSE(wbf->ShouldFlush());
218
219 // Free 9MB, memory_used_ = 1024KB
220 // It will free 36 dummy entries
221 wbf->FreeMem(9 * 1024 * 1024);
222 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 4 * kSizeDummyEntry);
223 ASSERT_GE(cache->GetPinnedUsage(), 4 * 256 * 1024);
224 ASSERT_LT(cache->GetPinnedUsage(), 4 * 256 * 1024 + kMetaDataChargeOverhead);
225
226 // Free 160KB gradually, memory_used_ = 864KB
227 // It will not cause any change
228 // since memory_used_ > dummy_entries_in_cache_usage * 3/4
229 for (int i = 0; i < 40; i++) {
230 wbf->FreeMem(4 * 1024);
231 }
232 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 4 * kSizeDummyEntry);
233 ASSERT_GE(cache->GetPinnedUsage(), 4 * 256 * 1024);
234 ASSERT_LT(cache->GetPinnedUsage(), 4 * 256 * 1024 + kMetaDataChargeOverhead);
235}

237TEST_F(ChargeWriteBufferTest, BasicWithCacheFull) {
238 constexpr std::size_t kMetaDataChargeOverhead = 20000;
239
240 // 12MB cache size with strict capacity
241 LRUCacheOptions lo;
242 lo.capacity = 12 * 1024 * 1024;
243 lo.num_shard_bits = 0;
244 lo.strict_capacity_limit = true;
245 std::shared_ptr<Cache> cache = NewLRUCache(lo);
246 std::unique_ptr<WriteBufferManager> wbf(new WriteBufferManager(0, cache));
247
248 // Allocate 10MB, memory_used_ = 10240KB
249 wbf->ReserveMem(10 * 1024 * 1024);
250 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 40 * kSizeDummyEntry);
251 ASSERT_GE(cache->GetPinnedUsage(), 40 * kSizeDummyEntry);
252 ASSERT_LT(cache->GetPinnedUsage(),
253 40 * kSizeDummyEntry + kMetaDataChargeOverhead);
254
255 // Allocate 10MB, memory_used_ = 20480KB
256 // Some dummy entry insertion will fail due to full cache
257 wbf->ReserveMem(10 * 1024 * 1024);
258 ASSERT_GE(cache->GetPinnedUsage(), 40 * kSizeDummyEntry);
259 ASSERT_LE(cache->GetPinnedUsage(), 12 * 1024 * 1024);
260 ASSERT_LT(wbf->dummy_entries_in_cache_usage(), 80 * kSizeDummyEntry);
261
262 // Free 15MB after encoutering cache full, memory_used_ = 5120KB
263 wbf->FreeMem(15 * 1024 * 1024);
264 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 20 * kSizeDummyEntry);
265 ASSERT_GE(cache->GetPinnedUsage(), 20 * kSizeDummyEntry);
266 ASSERT_LT(cache->GetPinnedUsage(),
267 20 * kSizeDummyEntry + kMetaDataChargeOverhead);
268
269 // Reserve 15MB, creating cache full again, memory_used_ = 20480KB
270 wbf->ReserveMem(15 * 1024 * 1024);
271 ASSERT_LE(cache->GetPinnedUsage(), 12 * 1024 * 1024);
272 ASSERT_LT(wbf->dummy_entries_in_cache_usage(), 80 * kSizeDummyEntry);
273
274 // Increase capacity so next insert will fully succeed
275 cache->SetCapacity(40 * 1024 * 1024);
276
277 // Allocate 10MB, memory_used_ = 30720KB
278 wbf->ReserveMem(10 * 1024 * 1024);
279 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 120 * kSizeDummyEntry);
280 ASSERT_GE(cache->GetPinnedUsage(), 120 * kSizeDummyEntry);
281 ASSERT_LT(cache->GetPinnedUsage(),
282 120 * kSizeDummyEntry + kMetaDataChargeOverhead);
283
284 // Gradually release 20 MB
285 // It ended up sequentially releasing 32, 24, 18 dummy entries when
286 // memory_used_ decreases to 22528KB, 16384KB, 11776KB.
287 // In total, it releases 74 dummy entries
288 for (int i = 0; i < 40; i++) {
289 wbf->FreeMem(512 * 1024);
290 }
291
292 ASSERT_EQ(wbf->dummy_entries_in_cache_usage(), 46 * kSizeDummyEntry);
293 ASSERT_GE(cache->GetPinnedUsage(), 46 * kSizeDummyEntry);
294 ASSERT_LT(cache->GetPinnedUsage(),
295 46 * kSizeDummyEntry + kMetaDataChargeOverhead);
296}
