20// Retrieves a single block of a given file. Utilizes the prefetch buffer and/or
21// persistent cache provided (if any) to try to avoid reading from the file
22// directly. Note that both the prefetch buffer and the persistent cache are
23// optional; also, note that the persistent cache may be configured to store
24// either compressed or uncompressed blocks.
25//
26// If the retrieved block is compressed and the do_uncompress flag is set,
27// BlockFetcher uncompresses the block (using the uncompression dictionary,
28// if provided, to prime the compression algorithm), and returns the resulting
29// uncompressed block data. Otherwise, it returns the original block.
30//
31// Two read options affect the behavior of BlockFetcher: if verify_checksums is
32// true, the checksum of the (original) block is checked; if fill_cache is true,
33// the block is added to the persistent cache if needed.
34//
35// Memory for uncompressed and compressed blocks is allocated as needed
36// using memory_allocator and memory_allocator_compressed, respectively
37// (if provided; otherwise, the default allocator is used).
38

40 public:

51 MemoryAllocator* memory_allocator = nullptr,

54 : file_(file),
55 prefetch_buffer_(prefetch_buffer),

65 block_size_with_trailer_(block_size_ + footer.GetBlockTrailerSize()),
66 uncompression_dict_(uncompression_dict),
67 cache_options_(cache_options),
68 memory_allocator_(memory_allocator),
69 memory_allocator_compressed_(memory_allocator_compressed),
70 for_compaction_(for_compaction) {
71 io_status_.PermitUncheckedError(); // TODO(AR) can we improve on this?

74 }
75 if (CheckFSFeatureSupport(ioptions_.fs.get(),
76 FSSupportedOps::kVerifyAndReconstructRead)) {
77 retry_corrupt_read_ = true;
78 }
79 }
