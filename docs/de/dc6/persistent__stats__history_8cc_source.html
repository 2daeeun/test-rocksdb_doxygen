19// 10 digit seconds timestamp => [Sep 9, 2001 ~ Nov 20, 2286]
20const int kNowSecondsStringLength = 10;
21const std::string kFormatVersionKeyString =
22 "__persistent_stats_format_version__";
23const std::string kCompatibleVersionKeyString =
24 "__persistent_stats_compatible_version__";
25// Every release maintains two versions numbers for persistents stats: Current
26// format version and compatible format version. Current format version
27// designates what type of encoding will be used when writing to stats CF;
28// compatible format version designates the minimum format version that
29// can decode the stats CF encoded using the current format version.
30const uint64_t kStatsCFCurrentFormatVersion = 1;
31const uint64_t kStatsCFCompatibleFormatVersion = 1;
32
33Status DecodePersistentStatsVersionNumber(DBImpl* db, StatsVersionKeyType type,
34 uint64_t* version_number) {
35 if (type >= StatsVersionKeyType::kKeyTypeMax) {
36 return Status::InvalidArgument("Invalid stats version key type provided");
37 }
38 std::string key;
39 if (type == StatsVersionKeyType::kFormatVersion) {
40 key = kFormatVersionKeyString;
41 } else if (type == StatsVersionKeyType::kCompatibleVersion) {
42 key = kCompatibleVersionKeyString;
43 }
44
45 // TODO: plumb Env::IOActivity, Env::IOPriority
46 ReadOptions options;
47 options.verify_checksums = true;
48 std::string result;
49 Status s = db->Get(options, db->PersistentStatsColumnFamily(), key, &result);
50 if (!s.ok() || result.empty()) {
51 return Status::NotFound("Persistent stats version key " + key +
52 " not found.");
53 }
54
55 // read version_number but do nothing in current version
56 *version_number = ParseUint64(result);
57 return Status::OK();
58}

60int EncodePersistentStatsKey(uint64_t now_seconds, const std::string& key,
61 int size, char* buf) {
62 char timestamp[kNowSecondsStringLength + 1];
63 // make time stamp string equal in length to allow sorting by time
64 snprintf(timestamp, sizeof(timestamp), "%010d",
65 static_cast<int>(now_seconds));
66 timestamp[kNowSecondsStringLength] = '\0';
67 return snprintf(buf, size, "%s#%s", timestamp, key.c_str());
68}

97std::pair<uint64_t, std::string> parseKey(const Slice& key,
98 uint64_t start_time) {
99 std::pair<uint64_t, std::string> result;
100 std::string key_str = key.ToString();
101 std::string::size_type pos = key_str.find('#');
102 // TODO(Zhongyi): add counters to track parse failures?
103 if (pos == std::string::npos) {
104 result.first = std::numeric_limits<uint64_t>::max();
105 result.second.clear();
106 } else {
107 uint64_t parsed_time = ParseUint64(key_str.substr(0, pos));
108 // skip entries with timestamp smaller than start_time
109 if (parsed_time < start_time) {
110 result.first = std::numeric_limits<uint64_t>::max();
111 result.second = "";
112 } else {
113 result.first = parsed_time;
114 std::string key_resize = key_str.substr(pos + 1);
115 result.second = key_resize;
116 }
117 }
118 return result;
119}

121// advance the iterator to the next time between [start_time, end_time)
122// if success, update time_ and stats_map_ with new_time and stats_map
123void PersistentStatsHistoryIterator::AdvanceIteratorByTime(uint64_t start_time,
124 uint64_t end_time) {
125 // try to find next entry in stats_history_ map
126 if (db_impl_ != nullptr) {
127 // TODO: plumb Env::IOActivity, Env::IOPriority
128 ReadOptions ro;
129 Iterator* iter =
130 db_impl_->NewIterator(ro, db_impl_->PersistentStatsColumnFamily());
131
132 char timestamp[kNowSecondsStringLength + 1];
133 snprintf(timestamp, sizeof(timestamp), "%010d",
134 static_cast<int>(std::max(time_, start_time)));
135 timestamp[kNowSecondsStringLength] = '\0';
136
137 iter->Seek(timestamp);
138 // no more entries with timestamp >= start_time is found or version key
139 // is found to be incompatible
140 if (!iter->Valid()) {
141 valid_ = false;
142 delete iter;
143 return;
144 }
145 time_ = parseKey(iter->key(), start_time).first;
146 valid_ = true;
147 // check parsed time and invalid if it exceeds end_time
148 if (time_ > end_time) {
149 valid_ = false;
150 delete iter;
151 return;
152 }
153 // find all entries with timestamp equal to time_
154 std::map<std::string, uint64_t> new_stats_map;
155 std::pair<uint64_t, std::string> kv;
156 for (; iter->Valid(); iter->Next()) {
157 kv = parseKey(iter->key(), start_time);
158 if (kv.first != time_) {
159 break;
160 }
161 if (kv.second.compare(kFormatVersionKeyString) == 0) {
162 continue;
163 }
164 new_stats_map[kv.second] = ParseUint64(iter->value().ToString());
165 }
166 stats_map_.swap(new_stats_map);
167 delete iter;
168 } else {
169 valid_ = false;
170 }
171}
