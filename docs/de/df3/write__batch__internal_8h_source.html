260// LocalSavePoint is similar to a scope guard
261class LocalSavePoint {
262 public:
263 explicit LocalSavePoint(WriteBatch* batch)
264 : batch_(batch),
265 savepoint_(batch->GetDataSize(), batch->Count(),
266 batch->content_flags_.load(std::memory_order_relaxed))
267#ifndef NDEBUG
268 ,
269 committed_(false)
270#endif
271 {
272 }
273
274#ifndef NDEBUG
275 ~LocalSavePoint() { assert(committed_); }
276#endif
277 Status commit() {
278#ifndef NDEBUG
279 committed_ = true;
280#endif
281 if (batch_->max_bytes_ && batch_->rep_.size() > batch_->max_bytes_) {
282 batch_->rep_.resize(savepoint_.size);
283 WriteBatchInternal::SetCount(batch_, savepoint_.count);
284 if (batch_->prot_info_ != nullptr) {
285 batch_->prot_info_->entries_.resize(savepoint_.count);
286 }
287 batch_->content_flags_.store(savepoint_.content_flags,
288 std::memory_order_relaxed);
289 return Status::MemoryLimit();
290 }
291 return Status::OK();
292 }
293
294 private:
295 WriteBatch* batch_;
296 SavePoint savepoint_;
297#ifndef NDEBUG
298 bool committed_;
299#endif
300};

302template <typename TimestampSizeFuncType>
303class TimestampUpdater : public WriteBatch::Handler {
304 public:
305 explicit TimestampUpdater(WriteBatch::ProtectionInfo* prot_info,
306 TimestampSizeFuncType&& ts_sz_func, const Slice& ts)
307 : prot_info_(prot_info),
308 ts_sz_func_(std::move(ts_sz_func)),
309 timestamp_(ts) {
310 assert(!timestamp_.empty());
311 }
312
313 ~TimestampUpdater() override {}
314
315 Status PutCF(uint32_t cf, const Slice& key, const Slice&) override {
316 return UpdateTimestamp(cf, key);
317 }
318
319 Status DeleteCF(uint32_t cf, const Slice& key) override {
320 return UpdateTimestamp(cf, key);
321 }
322
323 Status SingleDeleteCF(uint32_t cf, const Slice& key) override {
324 return UpdateTimestamp(cf, key);
325 }
326
327 Status DeleteRangeCF(uint32_t cf, const Slice& begin_key,
328 const Slice& end_key) override {
329 Status s = UpdateTimestamp(cf, begin_key, true /* is_key */);
330 if (s.ok()) {
331 s = UpdateTimestamp(cf, end_key, false /* is_key */);
332 }
333 return s;
334 }
335
336 Status MergeCF(uint32_t cf, const Slice& key, const Slice&) override {
337 return UpdateTimestamp(cf, key);
338 }
339
340 Status PutBlobIndexCF(uint32_t cf, const Slice& key, const Slice&) override {
341 return UpdateTimestamp(cf, key);
342 }
343
344 Status MarkBeginPrepare(bool) override { return Status::OK(); }
345
346 Status MarkEndPrepare(const Slice&) override { return Status::OK(); }
347
348 Status MarkCommit(const Slice&) override { return Status::OK(); }
349
350 Status MarkCommitWithTimestamp(const Slice&, const Slice&) override {
351 return Status::OK();
352 }
353
354 Status MarkRollback(const Slice&) override { return Status::OK(); }
355
356 Status MarkNoop(bool /*empty_batch*/) override { return Status::OK(); }
357
358 private:
359 // @param is_key specifies whether the update is for key or value.
360 Status UpdateTimestamp(uint32_t cf, const Slice& buf, bool is_key = true) {
361 Status s = UpdateTimestampImpl(cf, buf, idx_, is_key);
362 ++idx_;
363 return s;
364 }
365
366 Status UpdateTimestampImpl(uint32_t cf, const Slice& buf, size_t /*idx*/,
367 bool is_key) {
368 if (timestamp_.empty()) {
369 return Status::InvalidArgument("Timestamp is empty");
370 }
371 size_t cf_ts_sz = ts_sz_func_(cf);
372 if (0 == cf_ts_sz) {
373 // Skip this column family.
374 return Status::OK();
375 } else if (std::numeric_limits<size_t>::max() == cf_ts_sz) {
376 // Column family timestamp info not found.
377 return Status::NotFound();
378 } else if (cf_ts_sz != timestamp_.size()) {
379 return Status::InvalidArgument("timestamp size mismatch");
380 }
381 UpdateProtectionInformationIfNeeded(buf, timestamp_, is_key);
382
383 char* ptr = const_cast<char*>(buf.data() + buf.size() - cf_ts_sz);
384 assert(ptr);
385 memcpy(ptr, timestamp_.data(), timestamp_.size());
386 return Status::OK();
387 }
388
389 void UpdateProtectionInformationIfNeeded(const Slice& buf, const Slice& ts,
390 bool is_key) {
391 if (prot_info_ != nullptr) {
392 const size_t ts_sz = ts.size();
393 SliceParts old(&buf, 1);
394 Slice old_no_ts(buf.data(), buf.size() - ts_sz);
395 std::array<Slice, 2> new_key_cmpts{{old_no_ts, ts}};
396 SliceParts new_parts(new_key_cmpts.data(), 2);
397 if (is_key) {
398 prot_info_->entries_[idx_].UpdateK(old, new_parts);
399 } else {
400 prot_info_->entries_[idx_].UpdateV(old, new_parts);
401 }
402 }
403 }
404
405 // No copy or move.
406 TimestampUpdater(const TimestampUpdater&) = delete;
407 TimestampUpdater(TimestampUpdater&&) = delete;
408 TimestampUpdater& operator=(const TimestampUpdater&) = delete;
409 TimestampUpdater& operator=(TimestampUpdater&&) = delete;
410
411 WriteBatch::ProtectionInfo* const prot_info_ = nullptr;
412 const TimestampSizeFuncType ts_sz_func_{};
413 const Slice timestamp_;
414 size_t idx_ = 0;
415};
