323 void AddMockFile(const mock::KVVector& contents, int level = 0) {
324 assert(contents.size() > 0);
325
326 bool first_key = true;
327 std::string smallest, largest;
328 InternalKey smallest_key, largest_key;
329 SequenceNumber smallest_seqno = kMaxSequenceNumber;
330 SequenceNumber largest_seqno = 0;
331 uint64_t oldest_blob_file_number = kInvalidBlobFileNumber;
332 for (const auto& kv : contents) {
333 ParsedInternalKey key;
334 std::string skey;
335 std::string value;
336 std::tie(skey, value) = kv;
337 const Status pik_status =
338 ParseInternalKey(skey, &key, true /* log_err_key */);
339
340 smallest_seqno = std::min(smallest_seqno, key.sequence);
341 largest_seqno = std::max(largest_seqno, key.sequence);
342
343 if (first_key ||
344 cfd_->user_comparator()->Compare(key.user_key, smallest) < 0) {
345 smallest.assign(key.user_key.data(), key.user_key.size());
346 smallest_key.DecodeFrom(skey);
347 }
348 if (first_key ||
349 cfd_->user_comparator()->Compare(key.user_key, largest) > 0) {
350 largest.assign(key.user_key.data(), key.user_key.size());
351 largest_key.DecodeFrom(skey);
352 }
353
354 first_key = false;
355
356 if (pik_status.ok() && key.type == kTypeBlobIndex) {
357 BlobIndex blob_index;
358 const Status s = blob_index.DecodeFrom(value);
359 if (!s.ok()) {
360 continue;
361 }
362
363 if (blob_index.IsInlined() || blob_index.HasTTL() ||
364 blob_index.file_number() == kInvalidBlobFileNumber) {
365 continue;
366 }
367
368 if (oldest_blob_file_number == kInvalidBlobFileNumber ||
369 oldest_blob_file_number > blob_index.file_number()) {
370 oldest_blob_file_number = blob_index.file_number();
371 }
372 }
373 }
374
375 uint64_t file_number = versions_->NewFileNumber();
376
377 uint64_t file_size = 0;
378 if (table_type_ == TableTypeForTest::kBlockBasedTable) {
379 CreateTable(GenerateFileName(file_number), contents, file_size);
380 } else if (table_type_ == TableTypeForTest::kMockTable) {
381 file_size = 10;
382 EXPECT_OK(mock_table_factory_->CreateMockTable(
383 env_, GenerateFileName(file_number), contents));
384 } else {
385 assert(false);
386 }
387
388 VersionEdit edit;
389 edit.AddFile(
390 level, file_number, 0, file_size, smallest_key, largest_key,
391 smallest_seqno, largest_seqno, false, Temperature::kUnknown,
392 oldest_blob_file_number, kUnknownOldestAncesterTime,
393 kUnknownFileCreationTime,
394 versions_->GetColumnFamilySet()->GetDefault()->NewEpochNumber(),
395 kUnknownFileChecksum, kUnknownFileChecksumFuncName, kNullUniqueId64x2,
396 /*compensated_range_deletion_size=*/0, /*tail_size=*/0,
397 /*user_defined_timestamps_persisted=*/true);
398
399 mutex_.Lock();
400 EXPECT_OK(versions_->LogAndApply(
401 versions_->GetColumnFamilySet()->GetDefault(), mutable_cf_options_,
402 read_options_, write_options_, &edit, &mutex_, nullptr));
403 mutex_.Unlock();
404 }

406 void VerifyTables(int output_level,
407 const std::vector<mock::KVVector>& expected_results,
408 std::vector<uint64_t> expected_oldest_blob_file_numbers) {
409 if (expected_results.empty()) {
410 ASSERT_EQ(compaction_job_stats_.num_output_files, 0U);
411 return;
412 }
413 int expected_output_file_num = 0;
414 for (const auto& e : expected_results) {
415 if (!e.empty()) {
416 ++expected_output_file_num;
417 }
418 }
419 ASSERT_EQ(expected_output_file_num, compaction_job_stats_.num_output_files);
420 if (expected_output_file_num == 0) {
421 return;
422 }
423
424 if (expected_oldest_blob_file_numbers.empty()) {
425 expected_oldest_blob_file_numbers.resize(expected_output_file_num,
426 kInvalidBlobFileNumber);
427 }
428
429 auto cfd = versions_->GetColumnFamilySet()->GetDefault();
430 if (table_type_ == TableTypeForTest::kMockTable) {
431 ASSERT_EQ(compaction_job_stats_.num_output_files,
432 expected_results.size());
433 mock_table_factory_->AssertLatestFiles(expected_results);
434 } else {
435 assert(table_type_ == TableTypeForTest::kBlockBasedTable);
436 }
437
438 auto output_files =
439 cfd->current()->storage_info()->LevelFiles(output_level);
440 ASSERT_EQ(expected_output_file_num, output_files.size());
441
442 if (table_type_ == TableTypeForTest::kMockTable) {
443 assert(output_files.size() ==
444 static_cast<size_t>(expected_output_file_num));
445 const FileMetaData* const output_file = output_files[0];
446 ASSERT_EQ(output_file->oldest_blob_file_number,
447 expected_oldest_blob_file_numbers[0]);
448 return;
449 }
450
451 for (size_t i = 0; i < expected_results.size(); ++i) {
452 const FileMetaData* const output_file = output_files[i];
453 std::string file_name = GenerateFileName(output_file->fd.GetNumber());
454 const auto& fs = env_->GetFileSystem();
455 std::unique_ptr<RandomAccessFileReader> freader;
456 IOStatus ios = RandomAccessFileReader::Create(
457 fs, file_name, FileOptions(), &freader, nullptr);
458 ASSERT_OK(ios);
459 std::unique_ptr<TableReader> table_reader;
460 uint64_t file_size = output_file->fd.GetFileSize();
461 ReadOptions read_opts;
462 Status s = cf_options_.table_factory->NewTableReader(
463 read_opts,
464 TableReaderOptions(*cfd->ioptions(), nullptr, FileOptions(),
465 cfd_->internal_comparator(),
466 0 /* block_protection_bytes_per_key */),
467 std::move(freader), file_size, &table_reader, false);
468 ASSERT_OK(s);
469 assert(table_reader);
470 std::unique_ptr<InternalIterator> iiter(
471 table_reader->NewIterator(read_opts, nullptr, nullptr, true,
472 TableReaderCaller::kUncategorized));
473 assert(iiter);
474
475 mock::KVVector from_db;
476 for (iiter->SeekToFirst(); iiter->Valid(); iiter->Next()) {
477 const Slice key = iiter->key();
478 const Slice value = iiter->value();
479 from_db.emplace_back(
480 make_pair(key.ToString(false), value.ToString(false)));
481 }
482 ASSERT_EQ(expected_results[i], from_db);
483 }
484 }

493 mock::KVVector CreateTwoFiles(bool gen_corrupted_keys) {
494 stl_wrappers::KVMap expected_results;
495 constexpr int kKeysPerFile = 10000;
496 constexpr int kCorruptKeysPerFile = 200;
497 constexpr int kMatchingKeys = kKeysPerFile / 2;
498 SequenceNumber sequence_number = 0;
499
500 auto corrupt_id = [&](int id) {
501 return gen_corrupted_keys && id > 0 && id <= kCorruptKeysPerFile;
502 };
503
504 for (int i = 0; i < 2; ++i) {
505 auto contents = mock::MakeMockFile();
506 for (int k = 0; k < kKeysPerFile; ++k) {
507 auto key = std::to_string(i * kMatchingKeys + k);
508 auto value = std::to_string(i * kKeysPerFile + k);
509 InternalKey internal_key(key, ++sequence_number, kTypeValue);
510
511 // This is how the key will look like once it's written in bottommost
512 // file
513 InternalKey bottommost_internal_key(key, 0, kTypeValue);
514
515 if (corrupt_id(k)) {
516 test::CorruptKeyType(&internal_key);
517 test::CorruptKeyType(&bottommost_internal_key);
518 }
519 contents.push_back({internal_key.Encode().ToString(), value});
520 if (i == 1 || k < kMatchingKeys || corrupt_id(k - kMatchingKeys)) {
521 expected_results.insert(
522 {bottommost_internal_key.Encode().ToString(), value});
523 }
524 }
525 mock::SortKVVector(&contents, ucmp_);
526
527 AddMockFile(contents);
528 }
529
530 SetLastSequence(sequence_number);
531
532 mock::KVVector expected_results_kvvector;
533 for (auto& kv : expected_results) {
534 expected_results_kvvector.push_back({kv.first, kv.second});
535 }
536
537 return expected_results_kvvector;
538 }

613 SequenceNumber earliest_write_conflict_snapshot = kMaxSequenceNumber,

615 std::vector<uint64_t> expected_oldest_blob_file_numbers = {},
616 bool check_get_priority = false,
617 Env::IOPriority read_io_priority = Env::IO_TOTAL,
618 Env::IOPriority write_io_priority = Env::IO_TOTAL,
619 int max_subcompactions = 0) {
620 // For compaction, set fs as MockTestFileSystem to check the io_priority.
621 if (test_io_priority_) {
622 db_options_.fs.reset(
623 new MockTestFileSystem(fs_, read_io_priority, write_io_priority));
624 }
625
626 auto cfd = versions_->GetColumnFamilySet()->GetDefault();
627
628 size_t num_input_files = 0;
629 std::vector<CompactionInputFiles> compaction_input_files;
630 for (size_t i = 0; i < input_files.size(); ++i) {
631 auto level_files = input_files[i];
632 CompactionInputFiles compaction_level;
633 compaction_level.level = input_levels[i];
634 compaction_level.files.insert(compaction_level.files.end(),
635 level_files.begin(), level_files.end());
636 compaction_input_files.push_back(compaction_level);
637 num_input_files += level_files.size();
638 }
639
640 std::vector<FileMetaData*> grandparents;
641 // it should actually be the next non-empty level
642 const int kGrandparentsLevel = output_level + 1;
643 if (kGrandparentsLevel < cf_options_.num_levels) {
644 grandparents =
645 cfd_->current()->storage_info()->LevelFiles(kGrandparentsLevel);
646 }
647
648 Compaction compaction(
649 cfd->current()->storage_info(), *cfd->ioptions(),
650 *cfd->GetLatestMutableCFOptions(), mutable_db_options_,
651 compaction_input_files, output_level,
652 mutable_cf_options_.target_file_size_base,
653 mutable_cf_options_.max_compaction_bytes, 0, kNoCompression,
654 cfd->GetLatestMutableCFOptions()->compression_opts,
655 Temperature::kUnknown, max_subcompactions, grandparents,
656 /*earliest_snapshot*/ std::nullopt, /*snapshot_checker*/ nullptr, true);
657 compaction.FinalizeInputInfo(cfd->current());
658
659 assert(db_options_.info_log);
660 LogBuffer log_buffer(InfoLogLevel::INFO_LEVEL, db_options_.info_log.get());
661 mutex_.Lock();
662 EventLogger event_logger(db_options_.info_log.get());
663 // TODO(yiwu) add a mock snapshot checker and add test for it.
664 SnapshotChecker* snapshot_checker = nullptr;
665 ASSERT_TRUE(full_history_ts_low_.empty() ||
666 ucmp_->timestamp_size() == full_history_ts_low_.size());
667 const std::atomic<bool> kManualCompactionCanceledFalse{false};
668 JobContext job_context(1, false /* create_superversion */);
669 CompactionJob compaction_job(
670 0, &compaction, db_options_, mutable_db_options_, env_options_,
671 versions_.get(), &shutting_down_, &log_buffer, nullptr, nullptr,
672 nullptr, nullptr, &mutex_, &error_handler_, snapshots,
673 earliest_write_conflict_snapshot, snapshot_checker, &job_context,
674 table_cache_, &event_logger, false, false, dbname_,
675 &compaction_job_stats_, Env::Priority::USER, nullptr /* IOTracer */,
676 /*manual_compaction_canceled=*/kManualCompactionCanceledFalse,
677 env_->GenerateUniqueId(), DBImpl::GenerateDbSessionId(nullptr),
678 full_history_ts_low_);
679 VerifyInitializationOfCompactionJobStats(compaction_job_stats_);
680
681 compaction_job.Prepare();
682 mutex_.Unlock();
683 Status s = compaction_job.Run();
684 ASSERT_OK(s);
685 ASSERT_OK(compaction_job.io_status());
686 mutex_.Lock();
687 bool compaction_released = false;
688 ASSERT_OK(compaction_job.Install(*cfd->GetLatestMutableCFOptions(),
689 &compaction_released));
690 ASSERT_OK(compaction_job.io_status());
691 mutex_.Unlock();
692 log_buffer.FlushBufferToLog();
693
694 if (verify) {
695 ASSERT_GE(compaction_job_stats_.elapsed_micros, 0U);
696 ASSERT_EQ(compaction_job_stats_.num_input_files, num_input_files);
697
698 VerifyTables(output_level, expected_results,

700 }
701

704 }
705
706 if (verify_per_key_placement_) {
707 // Verify per_key_placement compaction
708 assert(compaction.SupportsPerKeyPlacement());
709 verify_per_key_placement_(compaction);
710 }
