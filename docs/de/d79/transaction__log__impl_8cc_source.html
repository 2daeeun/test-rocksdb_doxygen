46Status TransactionLogIteratorImpl::OpenLogFile(
47 const WalFile* log_file,
48 std::unique_ptr<SequentialFileReader>* file_reader) {
49 FileSystemPtr fs(options_->fs, io_tracer_);
50 std::unique_ptr<FSSequentialFile> file;
51 std::string fname;
52 Status s;
53 EnvOptions optimized_env_options = fs->OptimizeForLogRead(soptions_);
54 if (log_file->Type() == kArchivedLogFile) {
55 fname = ArchivedLogFileName(dir_, log_file->LogNumber());
56 s = fs->NewSequentialFile(fname, optimized_env_options, &file, nullptr);
57 } else {
58 fname = LogFileName(dir_, log_file->LogNumber());
59 s = fs->NewSequentialFile(fname, optimized_env_options, &file, nullptr);
60 if (!s.ok()) {
61 // If cannot open file in DB directory.
62 // Try the archive dir, as it could have moved in the meanwhile.
63 fname = ArchivedLogFileName(dir_, log_file->LogNumber());
64 s = fs->NewSequentialFile(fname, optimized_env_options, &file, nullptr);
65 }
66 }
67 if (s.ok()) {
68 file_reader->reset(new SequentialFileReader(std::move(file), fname,
69 io_tracer_, options_->listeners,
70 options_->rate_limiter.get()));
71 }
72 return s;
73}

95void TransactionLogIteratorImpl::SeekToStartSequence(uint64_t start_file_index,
96 bool strict) {
97 Slice record;
98 started_ = false;
99 is_valid_ = false;
100 // Check invariant of TransactionLogIterator when SeekToStartSequence()
101 // succeeds.
102 const Defer defer([this]() {
103 if (is_valid_) {
104 assert(current_status_.ok());
105 if (starting_sequence_number_ > current_batch_seq_) {
106 assert(current_batch_seq_ < current_last_seq_);
107 assert(current_last_seq_ >= starting_sequence_number_);
108 }
109 }
110 });
111 if (files_->size() <= start_file_index) {
112 return;
113 } else if (!current_status_.ok()) {
114 return;
115 }
116 Status s =
117 OpenLogReader(files_->at(static_cast<size_t>(start_file_index)).get());
118 if (!s.ok()) {
119 current_status_ = s;
120 reporter_.Info(current_status_.ToString().c_str());
121 return;
122 }
123 while (RestrictedRead(&record)) {
124 if (record.size() < WriteBatchInternal::kHeader) {
125 reporter_.Corruption(record.size(),
126 Status::Corruption("very small log record"));
127 continue;
128 }
129 UpdateCurrentWriteBatch(record);
130 if (current_last_seq_ >= starting_sequence_number_) {
131 if (strict && current_batch_seq_ != starting_sequence_number_) {
132 current_status_ = Status::Corruption(
133 "Gap in sequence number. Could not "
134 "seek to required sequence number");
135 reporter_.Info(current_status_.ToString().c_str());
136 return;
137 } else if (strict) {
138 reporter_.Info(
139 "Could seek required sequence number. Iterator will "
140 "continue.");
141 }
142 is_valid_ = true;
143 started_ = true; // set started_ as we could seek till starting sequence
144 return;
145 } else {
146 is_valid_ = false;
147 }
148 }
149
150 // Could not find start sequence in first file. Normally this must be the
151 // only file. Otherwise log the error and let the iterator return next entry
152 // If strict is set, we want to seek exactly till the start sequence and it
153 // should have been present in the file we scanned above
154 if (strict) {
155 current_status_ = Status::Corruption(
156 "Gap in sequence number. Could not "
157 "seek to required sequence number");
158 reporter_.Info(current_status_.ToString().c_str());
159 } else if (files_->size() != 1) {
160 current_status_ = Status::Corruption(
161 "Start sequence was not found, "
162 "skipping to the next available");
163 reporter_.Info(current_status_.ToString().c_str());
164 // Let NextImpl find the next available entry. started_ remains false
165 // because we don't want to check for gaps while moving to start sequence
166 NextImpl(true);
167 }
168}

177void TransactionLogIteratorImpl::NextImpl(bool internal) {
178 Slice record;
179 is_valid_ = false;
180 if (!internal && !started_) {
181 // Runs every time until we can seek to the start sequence
182 SeekToStartSequence();
183 }
184 while (true) {
185 assert(current_log_reader_);
186 if (current_log_reader_->IsEOF()) {
187 current_log_reader_->UnmarkEOF();
188 }
189 while (RestrictedRead(&record)) {
190 if (record.size() < WriteBatchInternal::kHeader) {
191 reporter_.Corruption(record.size(),
192 Status::Corruption("very small log record"));
193 continue;
194 } else {
195 // started_ should be true if called by application
196 assert(internal || started_);
197 // started_ should be false if called internally
198 assert(!internal || !started_);
199 UpdateCurrentWriteBatch(record);
200 if (internal && !started_) {
201 started_ = true;
202 }
203 return;
204 }
205 }
206
207 // Open the next file
208 if (current_file_index_ < files_->size() - 1) {
209 ++current_file_index_;
210 Status s = OpenLogReader(files_->at(current_file_index_).get());
211 if (!s.ok()) {
212 is_valid_ = false;
213 current_status_ = s;
214 return;
215 }
216 } else {
217 is_valid_ = false;
218 if (current_last_seq_ == versions_->LastSequence()) {
219 current_status_ = Status::OK();
220 } else {
221 const char* msg = "Create a new iterator to fetch the new tail.";
222 current_status_ = Status::TryAgain(msg);
223 }
224 return;
225 }
226 }

229bool TransactionLogIteratorImpl::IsBatchExpected(
230 const WriteBatch* batch, const SequenceNumber expected_seq) {
231 assert(batch);
232 SequenceNumber batchSeq = WriteBatchInternal::Sequence(batch);
233 if (batchSeq != expected_seq) {
234 std::ostringstream oss;
235 oss << "Discontinuity in log records. "
236 << "Got seq=" << batchSeq << ", "
237 << "Expected seq=" << expected_seq << ", "
238 << "Last flushed seq=" << versions_->LastSequence() << ". "
239 << "Log iterator will reseek the correct batch.";
240
241 reporter_.Info(oss.str().c_str());
242 return false;
243 }
244 return true;
245}

247void TransactionLogIteratorImpl::UpdateCurrentWriteBatch(const Slice& record) {
248 std::unique_ptr<WriteBatch> batch(new WriteBatch());
249 Status s = WriteBatchInternal::SetContents(batch.get(), record);
250 s.PermitUncheckedError(); // TODO: What should we do with this error?
251
252 SequenceNumber expected_seq = current_last_seq_ + 1;
253 // If the iterator has started, then confirm that we get continuous batches
254 if (started_ && !IsBatchExpected(batch.get(), expected_seq)) {
255 // Seek to the batch having expected sequence number
256 if (expected_seq < files_->at(current_file_index_)->StartSequence()) {
257 // Expected batch must lie in the previous log file
258 // Avoid underflow.
259 if (current_file_index_ != 0) {
260 current_file_index_--;
261 }
262 }
263 starting_sequence_number_ = expected_seq;
264 // currentStatus_ will be set to Ok if reseek succeeds
265 // Note: this is still ok in seq_pre_batch_ && two_write_queuesp_ mode
266 // that allows gaps in the WAL since it will still skip over the gap.
267 current_status_ = Status::NotFound("Gap in sequence numbers");
268 // In seq_per_batch_ mode, gaps in the seq are possible so the strict mode
269 // should be disabled
270 return SeekToStartSequence(current_file_index_, !seq_per_batch_);
271 }
272
273 current_batch_seq_ = WriteBatchInternal::Sequence(batch.get());
274 assert(!seq_per_batch_);
275 current_last_seq_ =
276 current_batch_seq_ + WriteBatchInternal::Count(batch.get()) - 1;
277 // currentBatchSeq_ can only change here
278 assert(current_last_seq_ <= versions_->LastSequence());
279
280 current_batch_ = std::move(batch);
281 is_valid_ = true;
282 current_status_ = Status::OK();
283}

285Status TransactionLogIteratorImpl::OpenLogReader(const WalFile* log_file) {
286 std::unique_ptr<SequentialFileReader> file;
287 Status s = OpenLogFile(log_file, &file);
288 if (!s.ok()) {
289 return s;
