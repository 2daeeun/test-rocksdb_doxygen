23TEST_F(PointLockManagerTest, LockStatus) {
24 MockColumnFamilyHandle cf1(1024), cf2(2048);
25 locker_->AddColumnFamily(&cf1);
26 locker_->AddColumnFamily(&cf2);
27
28 auto txn1 = NewTxn();
29 ASSERT_OK(locker_->TryLock(txn1, 1024, "k1", env_, true));
30 ASSERT_OK(locker_->TryLock(txn1, 2048, "k1", env_, true));
31
32 auto txn2 = NewTxn();
33 ASSERT_OK(locker_->TryLock(txn2, 1024, "k2", env_, false));
34 ASSERT_OK(locker_->TryLock(txn2, 2048, "k2", env_, false));
35
36 auto s = locker_->GetPointLockStatus();
37 ASSERT_EQ(s.size(), 4u);
38 for (uint32_t cf_id : {1024, 2048}) {
39 ASSERT_EQ(s.count(cf_id), 2u);
40 auto range = s.equal_range(cf_id);
41 for (auto it = range.first; it != range.second; it++) {
42 ASSERT_TRUE(it->second.key == "k1" || it->second.key == "k2");
43 if (it->second.key == "k1") {
44 ASSERT_EQ(it->second.exclusive, true);
45 ASSERT_EQ(it->second.ids.size(), 1u);
46 ASSERT_EQ(it->second.ids[0], txn1->GetID());
47 } else if (it->second.key == "k2") {
48 ASSERT_EQ(it->second.exclusive, false);
49 ASSERT_EQ(it->second.ids.size(), 1u);
50 ASSERT_EQ(it->second.ids[0], txn2->GetID());
51 }
52 }
53 }
54
55 // Cleanup
56 locker_->UnLock(txn1, 1024, "k1", env_);
57 locker_->UnLock(txn1, 2048, "k1", env_);
58 locker_->UnLock(txn2, 1024, "k2", env_);
59 locker_->UnLock(txn2, 2048, "k2", env_);
60
61 delete txn1;
62 delete txn2;
63}

104TEST_F(PointLockManagerTest, DeadlockDepthExceeded) {
105 // Tests that when detecting deadlock, if the detection depth is exceeded,
106 // it's also viewed as deadlock.
107 MockColumnFamilyHandle cf(1);
108 locker_->AddColumnFamily(&cf);
109 TransactionOptions txn_opt;
110 txn_opt.deadlock_detect = true;
111 txn_opt.deadlock_detect_depth = 1;
112 txn_opt.lock_timeout = 1000000;
113 auto txn1 = NewTxn(txn_opt);
114 auto txn2 = NewTxn(txn_opt);
115 auto txn3 = NewTxn(txn_opt);
116 auto txn4 = NewTxn(txn_opt);
117 // "a ->(k) b" means transaction a is waiting for transaction b to release
118 // the held lock on key k.
119 // txn4 ->(k3) -> txn3 ->(k2) txn2 ->(k1) txn1
120 // txn3's deadlock detection will exceed the detection depth 1,
121 // which will be viewed as a deadlock.
122 // NOTE:
123 // txn4 ->(k3) -> txn3 must be set up before
124 // txn3 ->(k2) -> txn2, because to trigger deadlock detection for txn3,
125 // it must have another txn waiting on it, which is txn4 in this case.
126 ASSERT_OK(locker_->TryLock(txn1, 1, "k1", env_, true));
127
128 port::Thread t1 = BlockUntilWaitingTxn(wait_sync_point_name_, [&]() {
129 ASSERT_OK(locker_->TryLock(txn2, 1, "k2", env_, true));
130 // block because txn1 is holding a lock on k1.
131 ASSERT_OK(locker_->TryLock(txn2, 1, "k1", env_, true));
132 });
133
134 ASSERT_OK(locker_->TryLock(txn3, 1, "k3", env_, true));
135
136 port::Thread t2 = BlockUntilWaitingTxn(wait_sync_point_name_, [&]() {
137 // block because txn3 is holding a lock on k1.
138 ASSERT_OK(locker_->TryLock(txn4, 1, "k3", env_, true));
139 });
140
141 auto s = locker_->TryLock(txn3, 1, "k2", env_, true);
142 ASSERT_TRUE(s.IsBusy());
143 ASSERT_EQ(s.subcode(), Status::SubCode::kDeadlock);
144
145 std::vector<DeadlockPath> deadlock_paths = locker_->GetDeadlockInfoBuffer();
146 ASSERT_EQ(deadlock_paths.size(), 1u);
147 ASSERT_TRUE(deadlock_paths[0].limit_exceeded);
148
149 locker_->UnLock(txn1, 1, "k1", env_);
150 locker_->UnLock(txn3, 1, "k3", env_);
151 t1.join();
152 t2.join();
153
154 delete txn4;
155 delete txn3;
156 delete txn2;
157 delete txn1;
158}
