43// BlockReadAmpBitmap is a bitmap that map the ROCKSDB_NAMESPACE::Block data
44// bytes to a bitmap with ratio bytes_per_bit. Whenever we access a range of
45// bytes in the Block we update the bitmap and increment
46// READ_AMP_ESTIMATE_USEFUL_BYTES.
47class BlockReadAmpBitmap {
48 public:
49 explicit BlockReadAmpBitmap(size_t block_size, size_t bytes_per_bit,
50 Statistics* statistics)
51 : bitmap_(nullptr),
52 bytes_per_bit_pow_(0),
53 statistics_(statistics),
54 rnd_(Random::GetTLSInstance()->Uniform(
55 static_cast<int>(bytes_per_bit))) {
56 TEST_SYNC_POINT_CALLBACK("BlockReadAmpBitmap:rnd", &rnd_);
57 assert(block_size > 0 && bytes_per_bit > 0);
58
59 // convert bytes_per_bit to be a power of 2
60 while (bytes_per_bit >>= 1) {
61 bytes_per_bit_pow_++;
62 }
63
64 // num_bits_needed = ceil(block_size / bytes_per_bit)
65 size_t num_bits_needed = ((block_size - 1) >> bytes_per_bit_pow_) + 1;
66 assert(num_bits_needed > 0);
67
68 // bitmap_size = ceil(num_bits_needed / kBitsPerEntry)
69 size_t bitmap_size = (num_bits_needed - 1) / kBitsPerEntry + 1;
70
71 // Create bitmap and set all the bits to 0
72 bitmap_ = new std::atomic<uint32_t>[bitmap_size]();
73
74 RecordTick(GetStatistics(), READ_AMP_TOTAL_READ_BYTES, block_size);
75 }
76
77 ~BlockReadAmpBitmap() { delete[] bitmap_; }
78
79 void Mark(uint32_t start_offset, uint32_t end_offset) {
80 assert(end_offset >= start_offset);
81 // Index of first bit in mask
82 uint32_t start_bit =
83 (start_offset + (1 << bytes_per_bit_pow_) - rnd_ - 1) >>
84 bytes_per_bit_pow_;
85 // Index of last bit in mask + 1
86 uint32_t exclusive_end_bit =
87 (end_offset + (1 << bytes_per_bit_pow_) - rnd_) >> bytes_per_bit_pow_;
88 if (start_bit >= exclusive_end_bit) {
89 return;
90 }
91 assert(exclusive_end_bit > 0);
92
93 if (GetAndSet(start_bit) == 0) {
94 uint32_t new_useful_bytes = (exclusive_end_bit - start_bit)
95 << bytes_per_bit_pow_;
96 RecordTick(GetStatistics(), READ_AMP_ESTIMATE_USEFUL_BYTES,
97 new_useful_bytes);
98 }
99 }
100
101 Statistics* GetStatistics() {
102 return statistics_.load(std::memory_order_relaxed);
103 }
104
105 void SetStatistics(Statistics* stats) { statistics_.store(stats); }
106
107 uint32_t GetBytesPerBit() { return 1 << bytes_per_bit_pow_; }
108
109 size_t ApproximateMemoryUsage() const {
110#ifdef ROCKSDB_MALLOC_USABLE_SIZE
111 return malloc_usable_size((void*)this);
112#endif // ROCKSDB_MALLOC_USABLE_SIZE
113 return sizeof(*this);
114 }
115
116 private:
117 // Get the current value of bit at `bit_idx` and set it to 1
118 inline bool GetAndSet(uint32_t bit_idx) {
119 const uint32_t byte_idx = bit_idx / kBitsPerEntry;
120 const uint32_t bit_mask = 1 << (bit_idx % kBitsPerEntry);
121
122 return bitmap_[byte_idx].fetch_or(bit_mask, std::memory_order_relaxed) &
123 bit_mask;
124 }
125
126 const uint32_t kBytesPersEntry = sizeof(uint32_t); // 4 bytes
127 const uint32_t kBitsPerEntry = kBytesPersEntry * 8; // 32 bits
128
129 // Bitmap used to record the bytes that we read, use atomic to protect
130 // against multiple threads updating the same bit
131 std::atomic<uint32_t>* bitmap_;
132 // (1 << bytes_per_bit_pow_) is bytes_per_bit. Use power of 2 to optimize
133 // muliplication and division
134 uint8_t bytes_per_bit_pow_;
135 // Pointer to DB Statistics object, Since this bitmap may outlive the DB
136 // this pointer maybe invalid, but the DB will update it to a valid pointer
137 // by using SetStatistics() before calling Mark()
138 std::atomic<Statistics*> statistics_;
139 uint32_t rnd_;
140};

293// A `BlockIter` iterates over the entries in a `Block`'s data buffer. The
294// format of this data buffer is an uncompressed, sorted sequence of key-value
295// pairs (see `Block` API for more details).
296//
297// Notably, the keys may either be in internal key format or user key format.
298// Subclasses are responsible for configuring the key format.
299//
300// `BlockIter` intends to provide final overrides for all of
301// `InternalIteratorBase` functions that can move the iterator. It does
302// this to guarantee `UpdateKey()` is called exactly once after each key
303// movement potentially visible to users. In this step, the key is prepared
304// (e.g., serialized if global seqno is in effect) so it can be returned
305// immediately when the user asks for it via calling `key() const`.
306//
307// For its subclasses, it provides protected variants of the above-mentioned
308// final-overridden methods. They are named with the "Impl" suffix, e.g.,
309// `Seek()` logic would be implemented by subclasses in `SeekImpl()`. These
310// "Impl" functions are responsible for positioning `raw_key_` but not
311// invoking `UpdateKey()`.
312//
313// Per key-value checksum is enabled if relevant states are passed in during
314// `InitializeBase()`. The checksum verification is done in each call to
315// UpdateKey() for the current key. Each subclass is responsible for keeping
316// track of cur_entry_idx_, the index of the current key within the block.
317// BlockIter uses this index to get the corresponding checksum for current key.
318// Additional checksum verification may be done in subclasses if they read keys
319// other than the key being processed in UpdateKey().
320template <class TValue>
321class BlockIter : public InternalIteratorBase<TValue> {
322 public:
323 // Makes Valid() return false, status() return `s`, and Seek()/Prev()/etc do
324 // nothing. Calls cleanup functions.
325 virtual void Invalidate(const Status& s) {
326 // Assert that the BlockIter is never deleted while Pinning is Enabled.
327 assert(!pinned_iters_mgr_ || !pinned_iters_mgr_->PinningEnabled());
328
329 data_ = nullptr;
330 current_ = restarts_;
331 status_ = s;
332
333 // Call cleanup callbacks.
334 Cleanable::Reset();
335 }
336

338 // When status_ is not ok, iter should be invalid.
339 assert(status_.ok() || current_ >= restarts_);
340 return current_ < restarts_;

482 // Returns the restart interval of this block.
483 // Returns 0 if num_restarts_ <= 1 or if the BlockIter is not initialized.
484 virtual uint32_t GetRestartInterval() {
485 if (num_restarts_ <= 1 || data_ == nullptr) {
486 return 0;
487 }
488 SeekToFirstImpl();
489 uint32_t end_index = GetRestartPoint(1);
490 uint32_t count = 1;
491 while (NextEntryOffset() < end_index && status_.ok()) {
492 assert(Valid());
493 NextImpl();
494 ++count;
495 }
496 return count;
497 }

499 // Returns the number of keys in this block.
500 virtual uint32_t NumberOfKeys(uint32_t block_restart_interval) {
501 if (num_restarts_ == 0 || data_ == nullptr) {
502 return 0;
503 }
504 uint32_t count = (num_restarts_ - 1) * block_restart_interval;
505 // Add number of keys from the last restart interval
506 SeekToRestartPoint(num_restarts_ - 1);
507 while (NextEntryOffset() < restarts_ && status_.ok()) {
508 NextImpl();
509 ++count;
510 }
511 return count;
512 }

522 // protection_bytes_per_key, kv_checksum, and block_restart_interval
523 // are needed only for per kv checksum verification.
524 void InitializeBase(const Comparator* raw_ucmp, const char* data,
525 uint32_t restarts, uint32_t num_restarts,
526 SequenceNumber global_seqno, bool block_contents_pinned,
527 bool user_defined_timestamp_persisted,
528
529 uint8_t protection_bytes_per_key, const char* kv_checksum,
530 uint32_t block_restart_interval) {
531 assert(data_ == nullptr); // Ensure it is called only once
532 assert(num_restarts > 0); // Ensure the param is valid
533
534 icmp_ = std::make_unique<InternalKeyComparator>(raw_ucmp);
535 data_ = data;
536 restarts_ = restarts;
537 num_restarts_ = num_restarts;
538 current_ = restarts_;
539 restart_index_ = num_restarts_;
540 global_seqno_ = global_seqno;
541 if (raw_ucmp != nullptr) {
542 ts_sz_ = raw_ucmp->timestamp_size();
543 }
544 pad_min_timestamp_ = ts_sz_ > 0 && !user_defined_timestamp_persisted;
545 block_contents_pinned_ = block_contents_pinned;
546 cache_handle_ = nullptr;
547 cur_entry_idx_ = -1;
548 protection_bytes_per_key_ = protection_bytes_per_key;
549 kv_checksum_ = kv_checksum;
550 block_restart_interval_ = block_restart_interval;
551 // Checksum related states are either all 0/nullptr or all non-zero.
552 // One exception is when num_restarts == 0, block_restart_interval can be 0
553 // since we are not able to compute it.
554 assert((protection_bytes_per_key == 0 && kv_checksum == nullptr) ||
555 (protection_bytes_per_key > 0 && kv_checksum != nullptr &&
556 (block_restart_interval > 0 || num_restarts == 1)));
557 }

584 // Must be called every time a key is found that needs to be returned to user,
585 // and may be called when no key is found (as a no-op). Updates `key_`,
586 // `key_buf_`, and `key_pinned_` with info about the found key.
587 // Per key-value checksum verification is done if available for the key to be
588 // returned. Iterator is invalidated with corruption status if checksum
589 // verification fails.
590 void UpdateKey() {
591 key_buf_.Clear();
592 if (!Valid()) {
593 return;
594 }
595 if (raw_key_.IsUserKey()) {
596 assert(global_seqno_ == kDisableGlobalSequenceNumber);
597 key_ = raw_key_.GetUserKey();
598 key_pinned_ = raw_key_.IsKeyPinned();
599 } else if (global_seqno_ == kDisableGlobalSequenceNumber) {
600 key_ = raw_key_.GetInternalKey();
601 key_pinned_ = raw_key_.IsKeyPinned();
602 } else {
603 key_buf_.SetInternalKey(raw_key_.GetUserKey(), global_seqno_,
604 ExtractValueType(raw_key_.GetInternalKey()));
605 key_ = key_buf_.GetInternalKey();
606 key_pinned_ = false;
607 }
608 TEST_SYNC_POINT_CALLBACK("BlockIter::UpdateKey::value",
609 (void*)value_.data());
610 TEST_SYNC_POINT_CALLBACK("Block::VerifyChecksum::checksum_len",
611 &protection_bytes_per_key_);
612 if (protection_bytes_per_key_ > 0) {
613 if (!ProtectionInfo64()
614 .ProtectKV(raw_key_.GetKey(), value_)
615 .Verify(
616 protection_bytes_per_key_,
617 kv_checksum_ + protection_bytes_per_key_ * cur_entry_idx_)) {
618 PerKVChecksumCorruptionError();
619 }
620 }
621 }

623 // Returns the result of `Comparator::Compare()`, where the appropriate
624 // comparator is used for the block contents, the LHS argument is the current
625 // key with global seqno applied, and the RHS argument is `other`.
626 int CompareCurrentKey(const Slice& other) {
627 if (raw_key_.IsUserKey()) {
628 assert(global_seqno_ == kDisableGlobalSequenceNumber);
629 return icmp_->user_comparator()->Compare(raw_key_.GetUserKey(), other);
630 } else if (global_seqno_ == kDisableGlobalSequenceNumber) {
631 return icmp_->Compare(raw_key_.GetInternalKey(), other);
632 }
633 return icmp_->Compare(raw_key_.GetInternalKey(), global_seqno_, other,
634 kDisableGlobalSequenceNumber);
