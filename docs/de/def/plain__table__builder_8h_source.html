30// The builder class of PlainTable. For description of PlainTable format
31// See comments of class PlainTableFactory, where instances of
32// PlainTableReader are created.
33class PlainTableBuilder : public TableBuilder {
34 public:
35 // Create a builder that will store the contents of the table it is
36 // building in *file. Does not close the file. It is up to the
37 // caller to close the file after calling Finish(). The output file
38 // will be part of level specified by 'level'. A value of -1 means
39 // that the caller does not know which level the output file will reside.
40 PlainTableBuilder(
41 const ImmutableOptions& ioptions, const MutableCFOptions& moptions,
42 const InternalTblPropCollFactories* internal_tbl_prop_coll_factories,
43 uint32_t column_family_id, int level_at_creation,
44 WritableFileWriter* file, uint32_t user_key_size,
45 EncodingType encoding_type, size_t index_sparseness,
46 uint32_t bloom_bits_per_key, const std::string& column_family_name,
47 uint32_t num_probes = 6, size_t huge_page_tlb_size = 0,
48 double hash_table_ratio = 0, bool store_index_in_file = false,
49 const std::string& db_id = "", const std::string& db_session_id = "",
50 uint64_t file_number = 0);
51 // No copying allowed
52 PlainTableBuilder(const PlainTableBuilder&) = delete;
53 void operator=(const PlainTableBuilder&) = delete;
54
55 // REQUIRES: Either Finish() or Abandon() has been called.
56 ~PlainTableBuilder();
57
58 // Add key,value to the table being constructed.
59 // REQUIRES: key is after any previously added key according to comparator.
60 // REQUIRES: Finish(), Abandon() have not been called
61 void Add(const Slice& key, const Slice& value) override;
62
63 // Return non-ok iff some error has been detected.
64 Status status() const override { return status_; }
65
66 // Return non-ok iff some error happens during IO.
67 IOStatus io_status() const override { return io_status_; }
68
69 // Finish building the table. Stops using the file passed to the
70 // constructor after this function returns.
71 // REQUIRES: Finish(), Abandon() have not been called
72 Status Finish() override;
73
74 // Indicate that the contents of this builder should be abandoned. Stops
75 // using the file passed to the constructor after this function returns.
76 // If the caller is not going to call Finish(), it must call Abandon()
77 // before destroying this builder.
78 // REQUIRES: Finish(), Abandon() have not been called
79 void Abandon() override;
80
81 // Number of calls to Add() so far.
82 uint64_t NumEntries() const override;
83
84 // Size of the file generated so far. If invoked after a successful
85 // Finish() call, returns the size of the final generated file.
86 uint64_t FileSize() const override;
87
88 TableProperties GetTableProperties() const override { return properties_; }
89
90 bool SaveIndexInFile() const { return store_index_in_file_; }
91
92 // Get file checksum
93 std::string GetFileChecksum() const override;
94
95 // Get file checksum function name
96 const char* GetFileChecksumFuncName() const override;
97
98 void SetSeqnoTimeTableProperties(const SeqnoToTimeMapping& relevant_mapping,
99 uint64_t uint_64) override;
100
101 private:
102 Arena arena_;
103 const ImmutableOptions& ioptions_;
104 const MutableCFOptions& moptions_;
105 std::vector<std::unique_ptr<InternalTblPropColl>>
106 table_properties_collectors_;
107
108 BloomBlockBuilder bloom_block_;
109 std::unique_ptr<PlainTableIndexBuilder> index_builder_;
110
111 WritableFileWriter* file_;
112 uint64_t offset_ = 0;
113 uint32_t bloom_bits_per_key_;
114 size_t huge_page_tlb_size_;
115 Status status_;
116 IOStatus io_status_;
117 TableProperties properties_;
118 PlainTableKeyEncoder encoder_;
119
120 bool store_index_in_file_;
121
122 std::vector<uint32_t> keys_or_prefixes_hashes_;
123 bool closed_ = false; // Either Finish() or Abandon() has been called.
124
125 const SliceTransform* prefix_extractor_;
126
127 Slice GetPrefix(const Slice& target) const {
128 assert(target.size() >= 8); // target is internal key
129 return GetPrefixFromUserKey(ExtractUserKey(target));
130 }
131
132 Slice GetPrefix(const ParsedInternalKey& target) const {
133 return GetPrefixFromUserKey(target.user_key);
134 }
135
136 Slice GetPrefixFromUserKey(const Slice& user_key) const {
137 if (!IsTotalOrderMode()) {
138 return prefix_extractor_->Transform(user_key);
139 } else {
140 // Use empty slice as prefix if prefix_extractor is not set.
141 // In that case,
142 // it falls back to pure binary search and
143 // total iterator seek is supported.
144 return Slice();
145 }
146 }
147
148 bool IsTotalOrderMode() const { return (prefix_extractor_ == nullptr); }
149};
