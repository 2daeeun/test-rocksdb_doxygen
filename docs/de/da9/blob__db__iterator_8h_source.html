21class BlobDBIterator : public Iterator {
22 public:
23 BlobDBIterator(ManagedSnapshot* snapshot, ArenaWrappedDBIter* iter,
24 BlobDBImpl* blob_db, SystemClock* clock,
25 Statistics* statistics)
26 : snapshot_(snapshot),
27 iter_(iter),
28 blob_db_(blob_db),
29 clock_(clock),
30 statistics_(statistics) {}
31
32 virtual ~BlobDBIterator() = default;
33
34 bool Valid() const override {
35 if (!iter_->Valid()) {
36 return false;
37 }
38 return status_.ok();
39 }
40
41 Status status() const override {
42 if (!iter_->status().ok()) {
43 return iter_->status();
44 }
45 return status_;
46 }
47
48 void SeekToFirst() override {
49 StopWatch seek_sw(clock_, statistics_, BLOB_DB_SEEK_MICROS);
50 RecordTick(statistics_, BLOB_DB_NUM_SEEK);
51 iter_->SeekToFirst();
52 while (UpdateBlobValue()) {
53 iter_->Next();
54 }
55 }
56
57 void SeekToLast() override {
58 StopWatch seek_sw(clock_, statistics_, BLOB_DB_SEEK_MICROS);
59 RecordTick(statistics_, BLOB_DB_NUM_SEEK);
60 iter_->SeekToLast();
61 while (UpdateBlobValue()) {
62 iter_->Prev();
63 }
64 }
65
66 void Seek(const Slice& target) override {
67 StopWatch seek_sw(clock_, statistics_, BLOB_DB_SEEK_MICROS);
68 RecordTick(statistics_, BLOB_DB_NUM_SEEK);
69 iter_->Seek(target);
70 while (UpdateBlobValue()) {
71 iter_->Next();
72 }
73 }
74
75 void SeekForPrev(const Slice& target) override {
76 StopWatch seek_sw(clock_, statistics_, BLOB_DB_SEEK_MICROS);
77 RecordTick(statistics_, BLOB_DB_NUM_SEEK);
78 iter_->SeekForPrev(target);
79 while (UpdateBlobValue()) {
80 iter_->Prev();
81 }
82 }
83
84 void Next() override {
85 assert(Valid());
86 StopWatch next_sw(clock_, statistics_, BLOB_DB_NEXT_MICROS);
87 RecordTick(statistics_, BLOB_DB_NUM_NEXT);
88 iter_->Next();
89 while (UpdateBlobValue()) {
90 iter_->Next();
91 }
92 }
93
94 void Prev() override {
95 assert(Valid());
96 StopWatch prev_sw(clock_, statistics_, BLOB_DB_PREV_MICROS);
97 RecordTick(statistics_, BLOB_DB_NUM_PREV);
98 iter_->Prev();
99 while (UpdateBlobValue()) {
100 iter_->Prev();
101 }
102 }
103
104 Slice key() const override {
105 assert(Valid());
106 return iter_->key();
107 }
108
109 Slice value() const override {
110 assert(Valid());
111 if (!iter_->IsBlob()) {
112 return iter_->value();
113 }
114 return value_;
115 }
116
117 // Iterator::Refresh() not supported.
118
119 private:
120 // Return true if caller should continue to next value.
121 bool UpdateBlobValue() {
122 value_.Reset();
123 status_ = Status::OK();
124 if (iter_->Valid() && iter_->status().ok() && iter_->IsBlob()) {
125 Status s = blob_db_->GetBlobValue(iter_->key(), iter_->value(), &value_);
126 if (s.IsNotFound()) {
127 return true;
128 } else {
129 if (!s.ok()) {
130 status_ = s;
131 }
132 return false;
133 }
134 } else {
135 return false;
136 }
137 }
138

141 BlobDBImpl* blob_db_;
142 SystemClock* clock_;
143 Statistics* statistics_;
144 Status status_;
145 PinnableSlice value_;
146};
