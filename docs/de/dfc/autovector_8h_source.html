19// A vector that leverages pre-allocated stack-based array to achieve better
20// performance for array with small amount of items.
21//
22// The interface resembles that of vector, but with less features since we aim
23// to solve the problem that we have in hand, rather than implementing a
24// full-fledged generic container.
25//
26// Currently we don't support:
27// * shrink_to_fit()
28// If used correctly, in most cases, people should not touch the
29// underlying vector at all.
30// * random insert()/erase(), please only use push_back()/pop_back().
31// * No move/swap operations. Each autovector instance has a
32// stack-allocated array and if we want support move/swap operations, we
33// need to copy the arrays other than just swapping the pointers. In this
34// case we'll just explicitly forbid these operations since they may
35// lead users to make false assumption by thinking they are inexpensive
36// operations.
37//
38// Naming style of public methods almost follows that of the STL's.
39template <class T, size_t kSize = 8>
40class autovector {
41 public:
42 // General STL-style container member types.
43 using value_type = T;
44 using difference_type = typename std::vector<T>::difference_type;
45 using size_type = typename std::vector<T>::size_type;
46 using reference = value_type&;
47 using const_reference = const value_type&;
48 using pointer = value_type*;
49 using const_pointer = const value_type*;

181 // -- Immutable operations
182 // Indicate if all data resides in in-stack data structure.
183 bool only_in_stack() const {
184 // If no element was inserted at all, the vector's capacity will be `0`.
185 return vect_.capacity() == 0;
186 }
187
188 size_type size() const { return num_stack_items_ + vect_.size(); }
189
190 // resize does not guarantee anything about the contents of the newly
191 // available elements
192 void resize(size_type n) {
193 if (n > kSize) {
194 vect_.resize(n - kSize);
195 while (num_stack_items_ < kSize) {
196 new ((void*)(&values_[num_stack_items_++])) value_type();
197 }
198 num_stack_items_ = kSize;
199 } else {
200 vect_.clear();
201 while (num_stack_items_ < n) {
202 new ((void*)(&values_[num_stack_items_++])) value_type();
203 }
204 while (num_stack_items_ > n) {
205 values_[--num_stack_items_].~value_type();
206 }
207 }
208 }

268 // -- Mutable Operations
269 void push_back(T&& item) {
270 if (num_stack_items_ < kSize) {
271 new ((void*)(&values_[num_stack_items_])) value_type();
272 values_[num_stack_items_++] = std::move(item);
273 } else {
274 vect_.push_back(item);
275 }
276 }

278 void push_back(const T& item) {
279 if (num_stack_items_ < kSize) {

281 values_[num_stack_items_++] = item;
282 } else {
283 vect_.push_back(item);
284 }
285 }

287 template <class... Args>
288#if _LIBCPP_STD_VER > 14

290 if (num_stack_items_ < kSize) {
291 return *(new ((void*)(&values_[num_stack_items_++]))
292 value_type(std::forward<Args>(args)...));
293 } else {
294 return vect_.emplace_back(std::forward<Args>(args)...);

296 }
297#else

299 if (num_stack_items_ < kSize) {

302 } else {
303 vect_.emplace_back(std::forward<Args>(args)...);
304 }

308 void pop_back() {
309 assert(!empty());
310 if (!vect_.empty()) {
311 vect_.pop_back();
312 } else {
313 values_[--num_stack_items_].~value_type();
314 }
315 }

318 while (num_stack_items_ > 0) {
319 values_[--num_stack_items_].~value_type();
320 }
321 vect_.clear();
322 }

356 size_type num_stack_items_ = 0; // current number of items
357 alignas(alignof(
358 value_type)) char buf_[kSize *
359 sizeof(value_type)]; // the first `kSize` items
360 pointer values_;
361 // used only if there are more than `kSize` items.
362 std::vector<T> vect_;

365template <class T, size_t kSize>
366autovector<T, kSize>& autovector<T, kSize>::assign(
367 const autovector<T, kSize>& other) {
368 values_ = reinterpret_cast<pointer>(buf_);
369 // copy the internal vector
370 vect_.assign(other.vect_.begin(), other.vect_.end());
371
372 // copy array
373 num_stack_items_ = other.num_stack_items_;
374 for (size_t i = 0; i < num_stack_items_; ++i) {
375 new ((void*)(&values_[i])) value_type();
376 }
377 std::copy(other.values_, other.values_ + num_stack_items_, values_);
378
379 return *this;
380}
381
382template <class T, size_t kSize>
383autovector<T, kSize>& autovector<T, kSize>::operator=(
384 autovector<T, kSize>&& other) {
385 values_ = reinterpret_cast<pointer>(buf_);
386 vect_ = std::move(other.vect_);
387 size_t n = other.num_stack_items_;
388 num_stack_items_ = n;
389 other.num_stack_items_ = 0;
390 for (size_t i = 0; i < n; ++i) {
391 new ((void*)(&values_[i])) value_type();
392 values_[i] = std::move(other.values_[i]);
393 }
394 return *this;
395}
