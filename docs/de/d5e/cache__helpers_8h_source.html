31// Generic resource management object for cache handles that releases the handle
32// when destroyed. Has unique ownership of the handle, so copying it is not
33// allowed, while moving it transfers ownership.
34template <typename T>
35class CacheHandleGuard {
36 public:
37 CacheHandleGuard() = default;
38
39 CacheHandleGuard(Cache* cache, Cache::Handle* handle)
40 : cache_(cache),
41 handle_(handle),
42 value_(GetFromCacheHandle<T>(cache, handle)) {
43 assert(cache_ && handle_ && value_);
44 }
45
46 CacheHandleGuard(const CacheHandleGuard&) = delete;
47 CacheHandleGuard& operator=(const CacheHandleGuard&) = delete;
48
49 CacheHandleGuard(CacheHandleGuard&& rhs) noexcept
50 : cache_(rhs.cache_), handle_(rhs.handle_), value_(rhs.value_) {
51 assert((!cache_ && !handle_ && !value_) || (cache_ && handle_ && value_));
52
53 rhs.ResetFields();
54 }
55
56 CacheHandleGuard& operator=(CacheHandleGuard&& rhs) noexcept {
57 if (this == &rhs) {
58 return *this;
59 }
60
61 ReleaseHandle();
62
63 cache_ = rhs.cache_;
64 handle_ = rhs.handle_;
65 value_ = rhs.value_;
66
67 assert((!cache_ && !handle_ && !value_) || (cache_ && handle_ && value_));
68
69 rhs.ResetFields();
70
71 return *this;
72 }
73
74 ~CacheHandleGuard() { ReleaseHandle(); }
75
76 bool IsEmpty() const { return !handle_; }
77
78 Cache* GetCache() const { return cache_; }
79 Cache::Handle* GetCacheHandle() const { return handle_; }
80 T* GetValue() const { return value_; }
81
82 void TransferTo(Cleanable* cleanable) {
83 if (cleanable) {
84 if (handle_ != nullptr) {
85 assert(cache_);
86 cleanable->RegisterCleanup(&ReleaseCacheHandleCleanup, cache_, handle_);
87 }
88 }
89 ResetFields();
90 }
91
92 void Reset() {
93 ReleaseHandle();
94 ResetFields();
95 }
96
97 private:
98 void ReleaseHandle() {
99 if (IsEmpty()) {
100 return;
101 }
102
103 assert(cache_);
104 cache_->Release(handle_);
105 }
106
107 void ResetFields() {
108 cache_ = nullptr;
109 handle_ = nullptr;
110 value_ = nullptr;
111 }
112
113 private:
114 Cache* cache_ = nullptr;
115 Cache::Handle* handle_ = nullptr;
116 T* value_ = nullptr;
117};

119// Build an aliasing shared_ptr that keeps `handle` in cache while there
120// are references, but the pointer is to the value for that cache entry,
121// which must be of type T. This is copyable, unlike CacheHandleGuard, but
122// does not provide access to caching details.
123template <typename T>
124std::shared_ptr<T> MakeSharedCacheHandleGuard(Cache* cache,
125 Cache::Handle* handle) {
126 auto wrapper = std::make_shared<CacheHandleGuard<T>>(cache, handle);
127 return std::shared_ptr<T>(wrapper, GetFromCacheHandle<T>(cache, handle));
128}
