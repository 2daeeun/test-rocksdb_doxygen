15// Instead of reconstructing a Transaction object, and calling rollback on it,
16// we can be more efficient with RollbackRecoveredTransaction by skipping
17// unnecessary steps (eg. updating CommitMap, reconstructing keyset)
18Status WriteUnpreparedTxnDB::RollbackRecoveredTransaction(
19 const DBImpl::RecoveredTransaction* rtxn) {
20 // TODO(lth): Reduce duplicate code with WritePrepared rollback logic.
21 assert(rtxn->unprepared_);
22 auto cf_map_shared_ptr = WritePreparedTxnDB::GetCFHandleMap();
23 auto cf_comp_map_shared_ptr = WritePreparedTxnDB::GetCFComparatorMap();
24 // In theory we could write with disableWAL = true during recovery, and
25 // assume that if we crash again during recovery, we can just replay from
26 // the very beginning. Unfortunately, the XIDs from the application may not
27 // necessarily be unique across restarts, potentially leading to situations
28 // like this:
29 //
30 // BEGIN_PREPARE(unprepared) Put(a) END_PREPARE(xid = 1)
31 // -- crash and recover with Put(a) rolled back as it was not prepared
32 // BEGIN_PREPARE(prepared) Put(b) END_PREPARE(xid = 1)
33 // COMMIT(xid = 1)
34 // -- crash and recover with both a, b
35 //
36 // We could just write the rollback marker, but then we would have to extend
37 // MemTableInserter during recovery to actually do writes into the DB
38 // instead of just dropping the in-memory write batch.
39 //
40 // TODO: plumb Env::IOActivity, Env::IOPriority
41 WriteOptions w_options;
42
43 class InvalidSnapshotReadCallback : public ReadCallback {
44 public:
45 InvalidSnapshotReadCallback(SequenceNumber snapshot)
46 : ReadCallback(snapshot) {}
47
48 inline bool IsVisibleFullCheck(SequenceNumber) override {
49 // The seq provided as snapshot is the seq right before we have locked and
50 // wrote to it, so whatever is there, it is committed.
51 return true;
52 }
53
54 // Ignore the refresh request since we are confident that our snapshot seq
55 // is not going to be affected by concurrent compactions (not enabled yet.)
56 void Refresh(SequenceNumber) override {}
57 };
58
59 // Iterate starting with largest sequence number.
60 for (auto it = rtxn->batches_.rbegin(); it != rtxn->batches_.rend(); ++it) {
61 auto last_visible_txn = it->first - 1;
62 const auto& batch = it->second.batch_;
63 WriteBatch rollback_batch(0 /* reserved_bytes */, 0 /* max_bytes */,
64 w_options.protection_bytes_per_key,
65 0 /* default_cf_ts_sz */);

67 struct RollbackWriteBatchBuilder : public WriteBatch::Handler {
68 DBImpl* db_;
69 ReadOptions roptions;
70 InvalidSnapshotReadCallback callback;
71 WriteBatch* rollback_batch_;
72 std::map<uint32_t, const Comparator*>& comparators_;
73 std::map<uint32_t, ColumnFamilyHandle*>& handles_;
74 using CFKeys = std::set<Slice, SetComparator>;
75 std::map<uint32_t, CFKeys> keys_;
76 bool rollback_merge_operands_;
77 RollbackWriteBatchBuilder(
78 DBImpl* db, SequenceNumber snap_seq, WriteBatch* dst_batch,
79 std::map<uint32_t, const Comparator*>& comparators,
80 std::map<uint32_t, ColumnFamilyHandle*>& handles,
81 bool rollback_merge_operands)
82 : db_(db),
83 callback(snap_seq),
84 // disable min_uncommitted optimization
85 rollback_batch_(dst_batch),
86 comparators_(comparators),
87 handles_(handles),
88 rollback_merge_operands_(rollback_merge_operands) {}

90 Status Rollback(uint32_t cf, const Slice& key) {
91 Status s;
92 CFKeys& cf_keys = keys_[cf];
93 if (cf_keys.size() == 0) { // just inserted
94 auto cmp = comparators_[cf];
95 keys_[cf] = CFKeys(SetComparator(cmp));
96 }
97 auto res = cf_keys.insert(key);
98 if (res.second ==
99 false) { // second is false if a element already existed.
100 return s;
101 }
102
103 PinnableSlice pinnable_val;
104 bool not_used;
105 auto cf_handle = handles_[cf];
106 DBImpl::GetImplOptions get_impl_options;
107 get_impl_options.column_family = cf_handle;
108 get_impl_options.value = &pinnable_val;
109 get_impl_options.value_found = &not_used;
110 get_impl_options.callback = &callback;
111 s = db_->GetImpl(roptions, key, get_impl_options);
112 assert(s.ok() || s.IsNotFound());
113 if (s.ok()) {
114 s = rollback_batch_->Put(cf_handle, key, pinnable_val);
115 assert(s.ok());
116 } else if (s.IsNotFound()) {
117 // There has been no readable value before txn. By adding a delete we
118 // make sure that there will be none afterwards either.
119 s = rollback_batch_->Delete(cf_handle, key);
120 assert(s.ok());
121 } else {
122 // Unexpected status. Return it to the user.
123 }
124 return s;
125 }

128 const Slice& /*val*/) override {

142 if (rollback_merge_operands_) {
143 return Rollback(cf, key);
144 } else {
145 return Status::OK();
146 }

149 // Recovered batches do not contain 2PC markers.
150 Status MarkNoop(bool) override { return Status::InvalidArgument(); }
151 Status MarkBeginPrepare(bool) override {
152 return Status::InvalidArgument();
153 }
154 Status MarkEndPrepare(const Slice&) override {
155 return Status::InvalidArgument();
156 }
157 Status MarkCommit(const Slice&) override {
158 return Status::InvalidArgument();
159 }

161 return Status::InvalidArgument();
162 }
163 } rollback_handler(db_impl_, last_visible_txn, &rollback_batch,
164 *cf_comp_map_shared_ptr.get(), *cf_map_shared_ptr.get(),
165 txn_db_options_.rollback_merge_operands);
166
167 auto s = batch->Iterate(&rollback_handler);
168 if (!s.ok()) {
169 return s;
170 }
171
172 // The Rollback marker will be used as a batch separator
173 s = WriteBatchInternal::MarkRollback(&rollback_batch, rtxn->name_);
174 if (!s.ok()) {
175 return s;
176 }
177
178 const uint64_t kNoLogRef = 0;
179 const bool kDisableMemtable = true;
180 const size_t kOneBatch = 1;
181 uint64_t seq_used = kMaxSequenceNumber;
182 s = db_impl_->WriteImpl(w_options, &rollback_batch, nullptr, nullptr,
183 nullptr, kNoLogRef, !kDisableMemtable, &seq_used,
184 kOneBatch);
185 if (!s.ok()) {
186 return s;
187 }
188
189 // If two_write_queues, we must manually release the sequence number to
190 // readers.
191 if (db_impl_->immutable_db_options().two_write_queues) {
192 db_impl_->SetLastPublishedSequence(seq_used);
193 }
194 }
195
196 return Status::OK();
197}

199Status WriteUnpreparedTxnDB::Initialize(
200 const std::vector<size_t>& compaction_enabled_cf_indices,
201 const std::vector<ColumnFamilyHandle*>& handles) {
202 // TODO(lth): Reduce code duplication in this function.
203 auto dbimpl = static_cast_with_check<DBImpl>(GetRootDB());
204 assert(dbimpl != nullptr);
205
206 db_impl_->SetSnapshotChecker(new WritePreparedSnapshotChecker(this));
207 // A callback to commit a single sub-batch
208 class CommitSubBatchPreReleaseCallback : public PreReleaseCallback {
209 public:
210 explicit CommitSubBatchPreReleaseCallback(WritePreparedTxnDB* db)
211 : db_(db) {}
212 Status Callback(SequenceNumber commit_seq,
213 bool is_mem_disabled __attribute__((__unused__)), uint64_t,
214 size_t /*index*/, size_t /*total*/) override {
215 assert(!is_mem_disabled);
216 db_->AddCommitted(commit_seq, commit_seq);
217 return Status::OK();
218 }
219
220 private:
221 WritePreparedTxnDB* db_;
222 };
223 db_impl_->SetRecoverableStatePreReleaseCallback(
224 new CommitSubBatchPreReleaseCallback(this));
225
226 // PessimisticTransactionDB::Initialize
227 for (auto cf_ptr : handles) {
228 AddColumnFamily(cf_ptr);
229 }
230 // Verify cf options
231 for (auto handle : handles) {
232 ColumnFamilyDescriptor cfd;
233 Status s = handle->GetDescriptor(&cfd);
234 if (!s.ok()) {
235 return s;
236 }
237 s = VerifyCFOptions(cfd.options);
238 if (!s.ok()) {
239 return s;
240 }
241 }
242
243 // Re-enable compaction for the column families that initially had
244 // compaction enabled.
245 std::vector<ColumnFamilyHandle*> compaction_enabled_cf_handles;
246 compaction_enabled_cf_handles.reserve(compaction_enabled_cf_indices.size());
247 for (auto index : compaction_enabled_cf_indices) {
248 compaction_enabled_cf_handles.push_back(handles[index]);
249 }
250
251 // create 'real' transactions from recovered shell transactions
252 auto rtxns = dbimpl->recovered_transactions();
253 std::map<SequenceNumber, SequenceNumber> ordered_seq_cnt;
254 for (const auto& rtxn : rtxns) {
255 auto recovered_trx = rtxn.second;
256 assert(recovered_trx);
257 assert(recovered_trx->batches_.size() >= 1);
258 assert(recovered_trx->name_.length());
259
260 // We can only rollback transactions after AdvanceMaxEvictedSeq is called,
261 // but AddPrepared must occur before AdvanceMaxEvictedSeq, which is why
262 // two iterations is required.
263 if (recovered_trx->unprepared_) {
264 continue;
265 }
266
267 // TODO: plumb Env::IOActivity, Env::IOPriority
268 WriteOptions w_options;
269 w_options.sync = true;
270 TransactionOptions t_options;
271
272 auto first_log_number = recovered_trx->batches_.begin()->second.log_number_;
273 auto first_seq = recovered_trx->batches_.begin()->first;
274 auto last_prepare_batch_cnt =
275 recovered_trx->batches_.begin()->second.batch_cnt_;
276
277 Transaction* real_trx = BeginTransaction(w_options, t_options, nullptr);
278 assert(real_trx);
279 auto wupt = static_cast_with_check<WriteUnpreparedTxn>(real_trx);
280 wupt->recovered_txn_ = true;
281
282 real_trx->SetLogNumber(first_log_number);
283 real_trx->SetId(first_seq);
284 Status s = real_trx->SetName(recovered_trx->name_);
285 if (!s.ok()) {
286 return s;
287 }
288 wupt->prepare_batch_cnt_ = last_prepare_batch_cnt;
289
290 for (auto batch : recovered_trx->batches_) {
291 const auto& seq = batch.first;
292 const auto& batch_info = batch.second;
293 auto cnt = batch_info.batch_cnt_ ? batch_info.batch_cnt_ : 1;
294 assert(batch_info.log_number_);
295
296 ordered_seq_cnt[seq] = cnt;
297 assert(wupt->unprep_seqs_.count(seq) == 0);
298 wupt->unprep_seqs_[seq] = cnt;
299
300 s = wupt->RebuildFromWriteBatch(batch_info.batch_);
301 assert(s.ok());
302 if (!s.ok()) {
303 return s;
304 }
305 }
306
307 const bool kClear = true;
308 wupt->InitWriteBatch(kClear);
309
310 real_trx->SetState(Transaction::PREPARED);
311 if (!s.ok()) {
312 return s;
313 }
314 }
315 // AddPrepared must be called in order
316 for (auto seq_cnt : ordered_seq_cnt) {
317 auto seq = seq_cnt.first;
318 auto cnt = seq_cnt.second;
319 for (size_t i = 0; i < cnt; i++) {
320 AddPrepared(seq + i);
321 }
322 }
323
324 SequenceNumber prev_max = max_evicted_seq_;
325 SequenceNumber last_seq = db_impl_->GetLatestSequenceNumber();
326 AdvanceMaxEvictedSeq(prev_max, last_seq);
327 // Create a gap between max and the next snapshot. This simplifies the logic
328 // in IsInSnapshot by not having to consider the special case of max ==
329 // snapshot after recovery. This is tested in IsInSnapshotEmptyMapTest.
330 if (last_seq) {
331 db_impl_->versions_->SetLastAllocatedSequence(last_seq + 1);
332 db_impl_->versions_->SetLastSequence(last_seq + 1);
333 db_impl_->versions_->SetLastPublishedSequence(last_seq + 1);
334 }
335
336 Status s;
337 // Rollback unprepared transactions.
338 for (const auto& rtxn : rtxns) {
339 auto recovered_trx = rtxn.second;
340 if (recovered_trx->unprepared_) {
341 s = RollbackRecoveredTransaction(recovered_trx);
342 if (!s.ok()) {
343 return s;
344 }
345 continue;
346 }
347 }
348
349 if (s.ok()) {
350 dbimpl->DeleteAllRecoveredTransactions();
351
352 // Compaction should start only after max_evicted_seq_ is set AND recovered
353 // transactions are either added to PrepareHeap or rolled back.
354 s = EnableAutoCompaction(compaction_enabled_cf_handles);
355 }
356
357 return s;
358}

391Iterator* WriteUnpreparedTxnDB::NewIterator(const ReadOptions& _read_options,
392 ColumnFamilyHandle* column_family,
393 WriteUnpreparedTxn* txn) {
394 if (_read_options.io_activity != Env::IOActivity::kUnknown &&
395 _read_options.io_activity != Env::IOActivity::kDBIterator) {
396 return NewErrorIterator(Status::InvalidArgument(
397 "Can only call NewIterator with `ReadOptions::io_activity` is "
398 "`Env::IOActivity::kUnknown` or `Env::IOActivity::kDBIterator`"));
399 }
400
401 ReadOptions read_options(_read_options);
402 if (read_options.io_activity == Env::IOActivity::kUnknown) {
403 read_options.io_activity = Env::IOActivity::kDBIterator;
404 }
405 // TODO(lth): Refactor so that this logic is shared with WritePrepared.
406 constexpr bool expose_blob_index = false;
407 constexpr bool allow_refresh = false;
408 std::shared_ptr<ManagedSnapshot> own_snapshot = nullptr;
409 SequenceNumber snapshot_seq = kMaxSequenceNumber;
410 SequenceNumber min_uncommitted = 0;
411
412 // Currently, the Prev() iterator logic does not work well without snapshot
413 // validation. The logic simply iterates through values of a key in
414 // ascending seqno order, stopping at the first non-visible value and
415 // returning the last visible value.
416 //
417 // For example, if snapshot sequence is 3, and we have the following keys:
418 // foo: v1 1
419 // foo: v2 2
420 // foo: v3 3
421 // foo: v4 4
422 // foo: v5 5
423 //
424 // Then 1, 2, 3 will be visible, but 4 will be non-visible, so we return v3,
425 // which is the last visible value.
426 //
427 // For unprepared transactions, if we have snap_seq = 3, but the current
428 // transaction has unprep_seq 5, then returning the first non-visible value
429 // would be incorrect, as we should return v5, and not v3. The problem is that
430 // there are committed values at snapshot_seq < commit_seq < unprep_seq.
431 //
432 // Snapshot validation can prevent this problem by ensuring that no committed
433 // values exist at snapshot_seq < commit_seq, and thus any value with a
434 // sequence number greater than snapshot_seq must be unprepared values. For
435 // example, if the transaction had a snapshot at 3, then snapshot validation
436 // would be performed during the Put(v5) call. It would find v4, and the Put
437 // would fail with snapshot validation failure.
438 //
439 // TODO(lth): Improve Prev() logic to continue iterating until
440 // max_visible_seq, and then return the last visible value, so that this
441 // restriction can be lifted.
442 const Snapshot* snapshot = nullptr;
443 if (read_options.snapshot == nullptr) {
444 snapshot = GetSnapshot();
445 own_snapshot = std::make_shared<ManagedSnapshot>(db_impl_, snapshot);
446 } else {
447 snapshot = read_options.snapshot;
448 }
449
450 snapshot_seq = snapshot->GetSequenceNumber();
451 assert(snapshot_seq != kMaxSequenceNumber);
452 // Iteration is safe as long as largest_validated_seq <= snapshot_seq. We are
453 // guaranteed that for keys that were modified by this transaction (and thus
454 // might have unprepared values), no committed values exist at
455 // largest_validated_seq < commit_seq (or the contrapositive: any committed
456 // value must exist at commit_seq <= largest_validated_seq). This implies
457 // that commit_seq <= largest_validated_seq <= snapshot_seq or commit_seq <=
458 // snapshot_seq. As explained above, the problem with Prev() only happens when
459 // snapshot_seq < commit_seq.
460 //
461 // For keys that were not modified by this transaction, largest_validated_seq_
462 // is meaningless, and Prev() should just work with the existing visibility
463 // logic.
464 if (txn->largest_validated_seq_ > snapshot->GetSequenceNumber() &&
465 !txn->unprep_seqs_.empty()) {
466 ROCKS_LOG_ERROR(info_log_,
467 "WriteUnprepared iterator creation failed since the "
468 "transaction has performed unvalidated writes");
469 return nullptr;
470 }
471 min_uncommitted =
472 static_cast_with_check<const SnapshotImpl>(snapshot)->min_uncommitted_;
473
474 auto* cfh = static_cast_with_check<ColumnFamilyHandleImpl>(column_family);
475 auto* cfd = cfh->cfd();
476 auto* state =
477 new IteratorState(this, snapshot_seq, own_snapshot, min_uncommitted, txn);
478 SuperVersion* super_version = cfd->GetReferencedSuperVersion(db_impl_);
479 auto* db_iter = db_impl_->NewIteratorImpl(
480 read_options, cfh, super_version, state->MaxVisibleSeq(),
481 &state->callback, expose_blob_index, allow_refresh);
482 db_iter->RegisterCleanup(CleanupWriteUnpreparedTxnDBIterator, state, nullptr);
483 return db_iter;
484}
