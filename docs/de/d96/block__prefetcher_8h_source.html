12namespace ROCKSDB_NAMESPACE {
13class BlockPrefetcher {
14 public:
15 explicit BlockPrefetcher(size_t compaction_readahead_size,
16 size_t initial_auto_readahead_size)
17 : compaction_readahead_size_(compaction_readahead_size),
18 readahead_size_(initial_auto_readahead_size),
19 initial_auto_readahead_size_(initial_auto_readahead_size) {}
20
21 void PrefetchIfNeeded(
22 const BlockBasedTable::Rep* rep, const BlockHandle& handle,
23 size_t readahead_size, bool is_for_compaction,
24 const bool no_sequential_checking, const ReadOptions& read_options,
25 const std::function<void(bool, uint64_t&, uint64_t&)>& readaheadsize_cb,
26 bool is_async_io_prefetch);
27 FilePrefetchBuffer* prefetch_buffer() { return prefetch_buffer_.get(); }
28
29 void UpdateReadPattern(const uint64_t& offset, const size_t& len) {
30 prev_offset_ = offset;
31 prev_len_ = len;
32 }
33
34 bool IsBlockSequential(const uint64_t& offset) {
35 return (prev_len_ == 0 || (prev_offset_ + prev_len_ == offset));
36 }
37
38 void ResetValues(size_t initial_auto_readahead_size) {
39 num_file_reads_ = 1;
40 // Since initial_auto_readahead_size_ can be different from
41 // the value passed to BlockBasedTableOptions.initial_auto_readahead_size in
42 // case of adaptive_readahead, so fallback the readahead_size_ to that value
43 // in case of reset.
44 initial_auto_readahead_size_ = initial_auto_readahead_size;
45 readahead_size_ = initial_auto_readahead_size_;
46 readahead_limit_ = 0;
47 return;
48 }
49
50 void SetReadaheadState(ReadaheadFileInfo::ReadaheadInfo* readahead_info) {
51 num_file_reads_ = readahead_info->num_file_reads;
52 initial_auto_readahead_size_ = readahead_info->readahead_size;
53 TEST_SYNC_POINT_CALLBACK("BlockPrefetcher::SetReadaheadState",
54 &initial_auto_readahead_size_);
55 }
56
57 private:
58 // Readahead size used in compaction, its value is used only if
59 // lookup_context_.caller = kCompaction.
60 size_t compaction_readahead_size_;
61
62 // readahead_size_ is used in non-compaction read if underlying FS supports
63 // prefetching.
64 size_t readahead_size_;
65 size_t readahead_limit_ = 0;
66 // initial_auto_readahead_size_ is used in non-compaction read if RocksDB uses
67 // internal prefetch buffer.
68 uint64_t initial_auto_readahead_size_;
69 uint64_t num_file_reads_ = 0;
70 uint64_t prev_offset_ = 0;
71 size_t prev_len_ = 0;
72 std::unique_ptr<FilePrefetchBuffer> prefetch_buffer_;
73};
74} // namespace ROCKSDB_NAMESPACE
