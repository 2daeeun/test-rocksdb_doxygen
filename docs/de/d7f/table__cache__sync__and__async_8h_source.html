16// Batched version of TableCache::MultiGet.
17DEFINE_SYNC_AND_ASYNC(Status, TableCache::MultiGet)
18(const ReadOptions& options, const InternalKeyComparator& internal_comparator,
19 const FileMetaData& file_meta, const MultiGetContext::Range* mget_range,
20 const MutableCFOptions& mutable_cf_options, HistogramImpl* file_read_hist,
21 bool skip_filters, bool skip_range_deletions, int level, TypedHandle* handle) {
22 auto& fd = file_meta.fd;
23 Status s;
24 TableReader* t = fd.table_reader;
25 MultiGetRange table_range(*mget_range, mget_range->begin(),
26 mget_range->end());
27 if (handle != nullptr && t == nullptr) {
28 t = cache_.Value(handle);
29 }
30 autovector<std::string, MultiGetContext::MAX_BATCH_SIZE> row_cache_entries;
31 IterKey row_cache_key;
32 size_t row_cache_key_prefix_size = 0;
33 KeyContext& first_key = *table_range.begin();
34 bool lookup_row_cache =
35 ioptions_.row_cache && !first_key.get_context->NeedToReadSequence();
36
37 // Check row cache if enabled. Since row cache does not currently store
38 // sequence numbers, we cannot use it if we need to fetch the sequence.
39 if (lookup_row_cache) {
40 GetContext* first_context = first_key.get_context;
41 CreateRowCacheKeyPrefix(options, fd, first_key.ikey, first_context,
42 row_cache_key);
43 row_cache_key_prefix_size = row_cache_key.Size();
44
45 for (auto miter = table_range.begin(); miter != table_range.end();
46 ++miter) {
47 const Slice& user_key = miter->ukey_with_ts;
48
49 GetContext* get_context = miter->get_context;
50
51 Status read_status;
52 bool ret =
53 GetFromRowCache(user_key, row_cache_key, row_cache_key_prefix_size,
54 get_context, &read_status);
55 if (!read_status.ok()) {
56 CO_RETURN read_status;
57 }
58 if (ret) {
59 table_range.SkipKey(miter);
60 } else {
61 row_cache_entries.emplace_back();
62 get_context->SetReplayLog(&(row_cache_entries.back()));
63 }
64 }
65 }
66
67 // Check that table_range is not empty. Its possible all keys may have been
68 // found in the row cache and thus the range may now be empty
69 if (s.ok() && !table_range.empty()) {
70 if (t == nullptr) {
71 assert(handle == nullptr);
72 s = FindTable(options, file_options_, internal_comparator, file_meta,
73 &handle, mutable_cf_options,
74 options.read_tier == kBlockCacheTier /* no_io */,
75 file_read_hist, skip_filters, level,
76 true /* prefetch_index_and_filter_in_cache */,
77 0 /*max_file_size_for_l0_meta_pin*/, file_meta.temperature);
78 TEST_SYNC_POINT_CALLBACK("TableCache::MultiGet:FindTable", &s);
79 if (s.ok()) {
80 t = cache_.Value(handle);
81 assert(t);
82 }
83 }
84 if (s.ok() && !options.ignore_range_deletions && !skip_range_deletions) {
85 UpdateRangeTombstoneSeqnums(options, t, table_range);
86 }
87 if (s.ok()) {
88 CO_AWAIT(t->MultiGet)
89 (options, &table_range, mutable_cf_options.prefix_extractor.get(),
90 skip_filters);
91 } else if (options.read_tier == kBlockCacheTier && s.IsIncomplete()) {
92 for (auto iter = table_range.begin(); iter != table_range.end(); ++iter) {
93 Status* status = iter->s;
94 if (status->IsIncomplete()) {
95 // Couldn't find Table in cache but treat as kFound if no_io set
96 iter->get_context->MarkKeyMayExist();
97 s = Status::OK();
98 }
99 }
100 }
101 }
102
103 if (lookup_row_cache) {
104 size_t row_idx = 0;
105 RowCacheInterface row_cache{ioptions_.row_cache.get()};
106
107 for (auto miter = table_range.begin(); miter != table_range.end();
108 ++miter) {
109 std::string& row_cache_entry = row_cache_entries[row_idx++];
110 const Slice& user_key = miter->ukey_with_ts;
111 GetContext* get_context = miter->get_context;
112
113 get_context->SetReplayLog(nullptr);
114 // Compute row cache key.
115 row_cache_key.TrimAppend(row_cache_key_prefix_size, user_key.data(),
116 user_key.size());
117 // Put the replay log in row cache only if something was found.
118 if (s.ok() && !row_cache_entry.empty()) {
119 size_t charge = row_cache_entry.capacity() + sizeof(std::string);
120 auto row_ptr = new std::string(std::move(row_cache_entry));
121 // If row cache is full, it's OK.
122 row_cache.Insert(row_cache_key.GetUserKey(), row_ptr, charge)
123 .PermitUncheckedError();
124 }
125 }
126 }
127
128 if (handle != nullptr) {
129 cache_.Release(handle);
130 }
131 CO_RETURN s;
132}
