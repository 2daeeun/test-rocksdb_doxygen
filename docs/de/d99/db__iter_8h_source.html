27// This file declares the factory functions of DBIter, in its original form
28// or a wrapped form with class ArenaWrappedDBIter, which is defined here.
29// Class DBIter, which is declared and implemented inside db_iter.cc, is
30// an iterator that converts internal keys (yielded by an InternalIterator)
31// that were live at the specified sequence number into appropriate user
32// keys.
33// Each internal key consists of a user key, a sequence number, and a value
34// type. DBIter deals with multiple key versions, tombstones, merge operands,
35// etc, and exposes an Iterator.
36// For example, DBIter may wrap following InternalIterator:
37// user key: AAA value: v3 seqno: 100 type: Put
38// user key: AAA value: v2 seqno: 97 type: Put
39// user key: AAA value: v1 seqno: 95 type: Put
40// user key: BBB value: v1 seqno: 90 type: Put
41// user key: BBC value: N/A seqno: 98 type: Delete
42// user key: BBC value: v1 seqno: 95 type: Put
43// If the snapshot passed in is 102, then the DBIter is expected to
44// expose the following iterator:
45// key: AAA value: v3
46// key: BBB value: v1
47// If the snapshot passed in is 96, then it should expose:
48// key: AAA value: v1
49// key: BBB value: v1
50// key: BBC value: v1
51//
52
53// Memtables and sstables that make the DB representation contain
54// (userkey,seq,type) => uservalue entries. DBIter
55// combines multiple entries for the same userkey found in the DB
56// representation into a single entry while accounting for sequence
57// numbers, deletion markers, overwrites, etc.
58class DBIter final : public Iterator {
59 public:
60 // The following is grossly complicated. TODO: clean it up
61 // Which direction is the iterator currently moving?
62 // (1) When moving forward:
63 // (1a) if current_entry_is_merged_ = false, the internal iterator is
64 // positioned at the exact entry that yields this->key(), this->value()
65 // (1b) if current_entry_is_merged_ = true, the internal iterator is
66 // positioned immediately after the last entry that contributed to the
67 // current this->value(). That entry may or may not have key equal to
68 // this->key().
69 // (2) When moving backwards, the internal iterator is positioned
70 // just before all entries whose user key == this->key().
71 enum Direction : uint8_t { kForward, kReverse };
72
73 // LocalStatistics contain Statistics counters that will be aggregated per
74 // each iterator instance and then will be sent to the global statistics when
75 // the iterator is destroyed.
76 //
77 // The purpose of this approach is to avoid perf regression happening
78 // when multiple threads bump the atomic counters from a DBIter::Next().
79 struct LocalStatistics {
80 explicit LocalStatistics() { ResetCounters(); }
81
82 void ResetCounters() {
83 next_count_ = 0;
84 next_found_count_ = 0;
85 prev_count_ = 0;
86 prev_found_count_ = 0;
87 bytes_read_ = 0;
88 skip_count_ = 0;
89 }
90
91 void BumpGlobalStatistics(Statistics* global_statistics) {
92 RecordTick(global_statistics, NUMBER_DB_NEXT, next_count_);
93 RecordTick(global_statistics, NUMBER_DB_NEXT_FOUND, next_found_count_);
94 RecordTick(global_statistics, NUMBER_DB_PREV, prev_count_);
95 RecordTick(global_statistics, NUMBER_DB_PREV_FOUND, prev_found_count_);
96 RecordTick(global_statistics, ITER_BYTES_READ, bytes_read_);
97 RecordTick(global_statistics, NUMBER_ITER_SKIP, skip_count_);
98 PERF_COUNTER_ADD(iter_read_bytes, bytes_read_);
99 ResetCounters();
100 }
101
102 // Map to Tickers::NUMBER_DB_NEXT
103 uint64_t next_count_;
104 // Map to Tickers::NUMBER_DB_NEXT_FOUND
105 uint64_t next_found_count_;
106 // Map to Tickers::NUMBER_DB_PREV
107 uint64_t prev_count_;
108 // Map to Tickers::NUMBER_DB_PREV_FOUND
109 uint64_t prev_found_count_;
110 // Map to Tickers::ITER_BYTES_READ
111 uint64_t bytes_read_;
112 // Map to Tickers::NUMBER_ITER_SKIP
113 uint64_t skip_count_;
114 };
