512std::string StatisticsImpl::ToString() const {
513 MutexLock lock(&aggregate_lock_);
514 std::string res;
515 res.reserve(20000);
516 for (const auto& t : TickersNameMap) {
517 assert(t.first < TICKER_ENUM_MAX);
518 char buffer[kTmpStrBufferSize];
519 snprintf(buffer, kTmpStrBufferSize, "%s COUNT : %" PRIu64 "\n",
520 t.second.c_str(), getTickerCountLocked(t.first));
521 res.append(buffer);
522 }
523 for (const auto& h : HistogramsNameMap) {
524 assert(h.first < HISTOGRAM_ENUM_MAX);
525 char buffer[kTmpStrBufferSize];
526 HistogramData hData;
527 getHistogramImplLocked(h.first)->Data(&hData);
528 // don't handle failures - buffer should always be big enough and arguments
529 // should be provided correctly
530 int ret =
531 snprintf(buffer, kTmpStrBufferSize,
532 "%s P50 : %f P95 : %f P99 : %f P100 : %f COUNT : %" PRIu64
533 " SUM : %" PRIu64 "\n",
534 h.second.c_str(), hData.median, hData.percentile95,
535 hData.percentile99, hData.max, hData.count, hData.sum);
536 if (ret < 0 || ret >= kTmpStrBufferSize) {
537 assert(false);
538 continue;
539 }
540 res.append(buffer);
541 }
542 res.shrink_to_fit();
543 return res;
544}
