168// TableBuilder provides the interface used to build a Table
169// (an immutable and sorted map from keys to values).
170//
171// Multiple threads can invoke const methods on a TableBuilder without
172// external synchronization, but if any of the threads may call a
173// non-const method, all threads accessing the same TableBuilder must use
174// external synchronization.
175class TableBuilder {
176 public:
177 // REQUIRES: Either Finish() or Abandon() has been called.
178 virtual ~TableBuilder() {}
179
180 // Add key,value to the table being constructed.
181 // REQUIRES: key is after any previously added key according to comparator.
182 // REQUIRES: Finish(), Abandon() have not been called
183 virtual void Add(const Slice& key, const Slice& value) = 0;
184
185 // Return non-ok iff some error has been detected.
186 virtual Status status() const = 0;
187
188 // Return non-ok iff some error happens during IO.
189 virtual IOStatus io_status() const = 0;
190
191 // Finish building the table.
192 // REQUIRES: Finish(), Abandon() have not been called
193 virtual Status Finish() = 0;
194
195 // Indicate that the contents of this builder should be abandoned.
196 // If the caller is not going to call Finish(), it must call Abandon()
197 // before destroying this builder.
198 // REQUIRES: Finish(), Abandon() have not been called
199 virtual void Abandon() = 0;
200
201 // Number of calls to Add() so far.
202 virtual uint64_t NumEntries() const = 0;
203
204 // Whether the output file is completely empty. It has neither entries
205 // or tombstones.
206 virtual bool IsEmpty() const {
207 return NumEntries() == 0 && GetTableProperties().num_range_deletions == 0;
208 }
209
210 // Size of the file generated so far. If invoked after a successful
211 // Finish() call, returns the size of the final generated file.
212 virtual uint64_t FileSize() const = 0;
213
214 // Estimated size of the file generated so far. This is used when
215 // FileSize() cannot estimate final SST size, e.g. parallel compression
216 // is enabled.
217 virtual uint64_t EstimatedFileSize() const { return FileSize(); }
218
219 virtual uint64_t GetTailSize() const { return 0; }
220
221 // If the user defined table properties collector suggest the file to
222 // be further compacted.
223 virtual bool NeedCompact() const { return false; }
224
225 // Returns table properties
226 virtual TableProperties GetTableProperties() const = 0;
227
228 // Return file checksum
229 virtual std::string GetFileChecksum() const = 0;
230
231 // Return file checksum function name
232 virtual const char* GetFileChecksumFuncName() const = 0;
233
234 // Set the sequence number to time mapping. `relevant_mapping` must be in
235 // enforced state (ready to encode to string).
236 virtual void SetSeqnoTimeTableProperties(
237 const SeqnoToTimeMapping& /*relevant_mapping*/,
238 uint64_t /*oldest_ancestor_time*/){}
239};
