35// Given a path, flatten the path name by replacing all chars not in
36// {[0-9,a-z,A-Z,-,_,.]} with _. And append '_LOG\0' at the end.
37// Return the number of chars stored in dest not including the trailing '\0'.
38static size_t GetInfoLogPrefix(const std::string& path, char* dest, int len) {
39 const char suffix[] = "_LOG";
40
41 size_t write_idx = 0;
42 size_t i = 0;
43 size_t src_len = path.size();
44
45 while (i < src_len && write_idx < len - sizeof(suffix)) {
46 if ((path[i] >= 'a' && path[i] <= 'z') ||
47 (path[i] >= '0' && path[i] <= '9') ||
48 (path[i] >= 'A' && path[i] <= 'Z') || path[i] == '-' ||
49 path[i] == '.' || path[i] == '_') {
50 dest[write_idx++] = path[i];
51 } else {
52 if (i > 0) {
53 dest[write_idx++] = '_';
54 }
55 }
56 i++;
57 }
58 assert(sizeof(suffix) <= len - write_idx);
59 // "\0" is automatically added by snprintf
60 snprintf(dest + write_idx, len - write_idx, suffix);
61 write_idx += sizeof(suffix) - 1;
62 return write_idx;
63}

129uint64_t TableFileNameToNumber(const std::string& name) {
130 uint64_t number = 0;
131 uint64_t base = 1;
132 int pos = static_cast<int>(name.find_last_of('.'));
133 while (--pos >= 0 && name[pos] >= '0' && name[pos] <= '9') {
134 number += (name[pos] - '0') * base;
135 base *= 10;
136 }
137 return number;
138}

273bool ParseFileName(const std::string& fname, uint64_t* number,
274 const Slice& info_log_name_prefix, FileType* type,
275 WalFileType* log_type) {
276 Slice rest(fname);
277 if (fname.length() > 1 && fname[0] == '/') {
278 rest.remove_prefix(1);
279 }
280 if (rest == "IDENTITY") {
281 *number = 0;
282 *type = kIdentityFile;
283 } else if (rest == "CURRENT") {
284 *number = 0;
285 *type = kCurrentFile;
286 } else if (rest == "LOCK") {
287 *number = 0;
288 *type = kDBLockFile;
289 } else if (info_log_name_prefix.size() > 0 &&
290 rest.starts_with(info_log_name_prefix)) {
291 rest.remove_prefix(info_log_name_prefix.size());
292 if (rest == "" || rest == ".old") {
293 *number = 0;
294 *type = kInfoLogFile;
295 } else if (rest.starts_with(".old.")) {
296 uint64_t ts_suffix;
297 // sizeof also counts the trailing '\0'.
298 rest.remove_prefix(sizeof(".old.") - 1);
299 if (!ConsumeDecimalNumber(&rest, &ts_suffix)) {
300 return false;
301 }
302 *number = ts_suffix;
303 *type = kInfoLogFile;
304 }
305 } else if (rest.starts_with("MANIFEST-")) {
306 rest.remove_prefix(strlen("MANIFEST-"));
307 uint64_t num;
308 if (!ConsumeDecimalNumber(&rest, &num)) {
309 return false;
310 }
311 if (!rest.empty()) {
312 return false;
313 }
314 *type = kDescriptorFile;
315 *number = num;
316 } else if (rest.starts_with("METADB-")) {
317 rest.remove_prefix(strlen("METADB-"));
318 uint64_t num;
319 if (!ConsumeDecimalNumber(&rest, &num)) {
320 return false;
321 }
322 if (!rest.empty()) {
323 return false;
324 }
325 *type = kMetaDatabase;
326 *number = num;
327 } else if (rest.starts_with(kOptionsFileNamePrefix)) {
328 uint64_t ts_suffix;
329 bool is_temp_file = false;
330 rest.remove_prefix(kOptionsFileNamePrefix.size());
331 const std::string kTempFileNameSuffixWithDot =
332 std::string(".") + kTempFileNameSuffix;
333 if (rest.ends_with(kTempFileNameSuffixWithDot)) {
334 rest.remove_suffix(kTempFileNameSuffixWithDot.size());
335 is_temp_file = true;
336 }
337 if (!ConsumeDecimalNumber(&rest, &ts_suffix)) {
338 return false;
339 }
340 *number = ts_suffix;
341 *type = is_temp_file ? kTempFile : kOptionsFile;
342 } else {
343 // Avoid strtoull() to keep filename format independent of the
344 // current locale
345 bool archive_dir_found = false;
346 if (rest.starts_with(kArchivalDirName)) {
347 if (rest.size() <= kArchivalDirName.size()) {
348 return false;
349 }
350 rest.remove_prefix(kArchivalDirName.size() +
351 1); // Add 1 to remove / also
352 if (log_type) {
353 *log_type = kArchivedLogFile;
354 }
355 archive_dir_found = true;
356 }
357 uint64_t num;
358 if (!ConsumeDecimalNumber(&rest, &num)) {
359 return false;
360 }
361 if (rest.size() <= 1 || rest[0] != '.') {
362 return false;
363 }
364 rest.remove_prefix(1);
365
366 Slice suffix = rest;
367 if (suffix == Slice("log")) {
368 *type = kWalFile;
369 if (log_type && !archive_dir_found) {
370 *log_type = kAliveLogFile;
371 }
372 } else if (archive_dir_found) {
373 return false; // Archive dir can contain only log files
374 } else if (suffix == Slice(kRocksDbTFileExt) ||
375 suffix == Slice(kLevelDbTFileExt)) {
376 *type = kTableFile;
377 } else if (suffix == Slice(kRocksDBBlobFileExt)) {
378 *type = kBlobFile;
379 } else if (suffix == Slice(kTempFileNameSuffix)) {
380 *type = kTempFile;
381 } else {
382 return false;
383 }
384 *number = num;
385 }
386 return true;
387}

389IOStatus SetCurrentFile(const WriteOptions& write_options, FileSystem* fs,
390 const std::string& dbname, uint64_t descriptor_number,
391 Temperature temp,
392 FSDirectory* dir_contains_current_file) {
393 // Remove leading "dbname/" and add newline to manifest file name
394 std::string manifest = DescriptorFileName(dbname, descriptor_number);
395 Slice contents = manifest;
396 assert(contents.starts_with(dbname + "/"));
397 contents.remove_prefix(dbname.size() + 1);
398 std::string tmp = TempFileName(dbname, descriptor_number);
399 IOOptions opts;
400 IOStatus s = PrepareIOFromWriteOptions(write_options, opts);
401 FileOptions file_opts;
402 file_opts.temperature = temp;
403 if (s.ok()) {
404 s = WriteStringToFile(fs, contents.ToString() + "\n", tmp, true, opts,
405 file_opts);
406 }
407 TEST_SYNC_POINT_CALLBACK("SetCurrentFile:BeforeRename", &s);
408 if (s.ok()) {
409 TEST_KILL_RANDOM_WITH_WEIGHT("SetCurrentFile:0", REDUCE_ODDS2);
410 s = fs->RenameFile(tmp, CurrentFileName(dbname), opts, nullptr);
411 TEST_KILL_RANDOM_WITH_WEIGHT("SetCurrentFile:1", REDUCE_ODDS2);
412 TEST_SYNC_POINT_CALLBACK("SetCurrentFile:AfterRename", &s);
413 }
414 if (s.ok()) {
415 if (dir_contains_current_file != nullptr) {
416 s = dir_contains_current_file->FsyncWithDirOptions(
417 opts, nullptr, DirFsyncOptions(CurrentFileName(dbname)));
418 }
419 } else {
420 fs->DeleteFile(tmp, opts, nullptr)
421 .PermitUncheckedError(); // NOTE: PermitUncheckedError is acceptable
422 // here as we are already handling an error
423 // case, and this is just a best-attempt
424 // effort at some cleanup
425 }
426 return s;
427}

429Status SetIdentityFile(const WriteOptions& write_options, Env* env,
430 const std::string& dbname, Temperature temp,
431 const std::string& db_id) {
432 std::string id;
433 if (db_id.empty()) {
434 id = env->GenerateUniqueId();
435 } else {
436 id = db_id;
437 }
438 assert(!id.empty());
439 // Reserve the filename dbname/000000.dbtmp for the temporary identity file
440 std::string tmp = TempFileName(dbname, 0);
441 std::string identify_file_name = IdentityFileName(dbname);
442 Status s;
443 IOOptions opts;
444 s = PrepareIOFromWriteOptions(write_options, opts);
445 FileOptions file_opts;
446 file_opts.temperature = temp;
447 if (s.ok()) {
448 s = WriteStringToFile(env->GetFileSystem().get(), id, tmp,
449 /*should_sync=*/true, opts, file_opts);
450 }
451 if (s.ok()) {
452 s = env->RenameFile(tmp, identify_file_name);
453 }
454 std::unique_ptr<FSDirectory> dir_obj;
455 if (s.ok()) {
456 s = env->GetFileSystem()->NewDirectory(dbname, opts, &dir_obj, nullptr);
457 }
458 if (s.ok()) {
459 s = dir_obj->FsyncWithDirOptions(opts, nullptr,
460 DirFsyncOptions(identify_file_name));
461 }
462
463 // The default Close() could return "NotSupported" and we bypass it
464 // if it is not impelmented. Detailed explanations can be found in
465 // db/db_impl/db_impl.h
466 if (s.ok()) {
467 Status temp_s = dir_obj->Close(opts, nullptr);
468 if (!temp_s.ok()) {
469 if (temp_s.IsNotSupported()) {
470 temp_s.PermitUncheckedError();
471 } else {
472 s = temp_s;
473 }
474 }
475 }
476 if (!s.ok()) {
477 env->DeleteFile(tmp).PermitUncheckedError();
478 }
479 return s;
480}

495Status GetInfoLogFiles(const std::shared_ptr<FileSystem>& fs,
496 const std::string& db_log_dir, const std::string& dbname,
497 std::string* parent_dir,
498 std::vector<std::string>* info_log_list) {
499 assert(parent_dir != nullptr);
500 assert(info_log_list != nullptr);
501 uint64_t number = 0;
502 FileType type = kWalFile;
503
504 if (!db_log_dir.empty()) {
505 *parent_dir = db_log_dir;
506 } else {
507 *parent_dir = dbname;
508 }
509
510 InfoLogPrefix info_log_prefix(!db_log_dir.empty(), dbname);
511
512 std::vector<std::string> file_names;
513 Status s = fs->GetChildren(*parent_dir, IOOptions(), &file_names, nullptr);
514
515 if (!s.ok()) {
516 return s;
517 }
518
519 for (auto& f : file_names) {
520 if (ParseFileName(f, &number, info_log_prefix.prefix, &type) &&
521 (type == kInfoLogFile)) {
522 info_log_list->push_back(f);
523 }
524 }
525 return Status::OK();
526}

528std::string NormalizePath(const std::string& path) {
529 std::string dst;
530
531 if (path.length() > 2 && path[0] == kFilePathSeparator &&
532 path[1] == kFilePathSeparator) { // Handle UNC names
533 dst.append(2, kFilePathSeparator);
534 }
535
536 for (auto c : path) {
537 if (!dst.empty() && (c == kFilePathSeparator || c == '/') &&
538 (dst.back() == kFilePathSeparator || dst.back() == '/')) {
539 continue;
540 }
541 dst.push_back(c);
542 }
543 return dst;
544}
