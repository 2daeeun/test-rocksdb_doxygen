218 struct JoinThreadsOnExit {
219 explicit JoinThreadsOnExit(PosixEnv& _deflt) : deflt(_deflt) {}
220 ~JoinThreadsOnExit() {
221 for (const auto tid : deflt.threads_to_join_) {
222 pthread_join(tid, nullptr);
223 }

225 deflt.thread_pools_[pool_id].JoinAllThreads();

228 // free after use when Env::Default() is destructed while some other
229 // child threads are still trying to update thread status. All
230 // PosixEnv instances use the same thread_status_updater_, so never
231 // explicitly delete it.
232 }
233 PosixEnv& deflt;
234 };

242#ifndef ROCKSDB_NO_DYNAMIC_EXTENSION
243 // Loads the named library into the result.
244 // If the input name is empty, the current executable is loaded
245 // On *nix systems, a "lib" prefix is added to the name if one is not supplied
246 // Comparably, the appropriate shared library extension is added to the name
247 // if not supplied. If search_path is not specified, the shared library will
248 // be loaded using the default path (LD_LIBRARY_PATH) If search_path is
249 // specified, the shared library will be searched for in the directories
250 // provided by the search path
251 Status LoadLibrary(const std::string& name, const std::string& path,
252 std::shared_ptr<DynamicLibrary>* result) override {
253 assert(result != nullptr);
254 if (name.empty()) {
255 void* hndl = dlopen(NULL, RTLD_NOW);
256 if (hndl != nullptr) {
257 result->reset(new PosixDynamicLibrary(name, hndl));
258 return Status::OK();
259 }
260 } else {
261 std::string library_name = name;
262 if (library_name.find(kSharedLibExt) == std::string::npos) {
263 library_name = library_name + kSharedLibExt;
264 }
265#if !defined(OS_WIN)
266 if (library_name.find('/') == std::string::npos &&
267 library_name.compare(0, 3, "lib") != 0) {
268 library_name = "lib" + library_name;
269 }
270#endif
271 if (path.empty()) {
272 void* hndl = dlopen(library_name.c_str(), RTLD_NOW);
273 if (hndl != nullptr) {
274 result->reset(new PosixDynamicLibrary(library_name, hndl));
275 return Status::OK();
276 }
277 } else {
278 std::string local_path;
279 std::stringstream ss(path);
280 while (getline(ss, local_path, kPathSeparator)) {
281 if (!path.empty()) {
282 std::string full_name = local_path + "/" + library_name;
283 void* hndl = dlopen(full_name.c_str(), RTLD_NOW);
284 if (hndl != nullptr) {
285 result->reset(new PosixDynamicLibrary(full_name, hndl));
286 return Status::OK();
287 }
288 }
289 }
290 }
291 }
292 return Status::IOError(
293 IOErrorMsg("Failed to open shared library: xs", name), dlerror());
294 }
295#endif // !ROCKSDB_NO_DYNAMIC_EXTENSION

474void PosixEnv::StartThread(void (*function)(void* arg), void* arg) {
475 pthread_t t;
476 StartThreadState* state = new StartThreadState;
477 state->user_function = function;
478 state->arg = arg;
479 ThreadPoolImpl::PthreadCall(
480 "start thread", pthread_create(&t, nullptr, &StartThreadWrapper, state));
481 ThreadPoolImpl::PthreadCall("lock", pthread_mutex_lock(&mu_));
482 threads_to_join_.push_back(t);
483 ThreadPoolImpl::PthreadCall("unlock", pthread_mutex_unlock(&mu_));
484}

495//
496// Default Posix Env
497//
498Env* Env::Default() {
499 // The following function call initializes the singletons of ThreadLocalPtr
500 // right before the static default_env. This guarantees default_env will
501 // always being destructed before the ThreadLocalPtr singletons get
502 // destructed as C++ guarantees that the destructions of static variables
503 // is in the reverse order of their constructions.
504 //
505 // Since static members are destructed in the reverse order
506 // of their construction, having this call here guarantees that
507 // the destructor of static PosixEnv will go first, then the
508 // the singletons of ThreadLocalPtr.
509 ThreadLocalPtr::InitSingletons();
510 CompressionContextCache::InitSingleton();
511 INIT_SYNC_POINT_SINGLETONS();
512 // Avoid problems with accessing most members of Env::Default() during
513 // static destruction.
514 STATIC_AVOID_DESTRUCTION(PosixEnv, default_env);
515 // This destructor must be called on exit
516 static PosixEnv::JoinThreadsOnExit thread_joiner(default_env);
517 return &default_env;
518}
