13void WideColumnsHelper::DumpWideColumns(const WideColumns& columns,
14 std::ostream& os, bool hex) {
15 if (columns.empty()) {
16 return;
17 }
18
19 const std::ios_base::fmtflags orig_flags = os.flags();
20
21 if (hex) {
22 os << std::hex;
23 }
24 auto it = columns.begin();
25 os << *it;
26 for (++it; it != columns.end(); ++it) {
27 os << ' ' << *it;
28 }
29
30 os.flags(orig_flags);
31}
