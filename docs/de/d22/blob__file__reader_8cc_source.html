28Status BlobFileReader::Create(
29 const ImmutableOptions& immutable_options, const ReadOptions& read_options,
30 const FileOptions& file_options, uint32_t column_family_id,
31 HistogramImpl* blob_file_read_hist, uint64_t blob_file_number,
32 const std::shared_ptr<IOTracer>& io_tracer,
33 std::unique_ptr<BlobFileReader>* blob_file_reader) {
34 assert(blob_file_reader);
35 assert(!*blob_file_reader);
36
37 uint64_t file_size = 0;
38 std::unique_ptr<RandomAccessFileReader> file_reader;
39
40 {
41 const Status s =
42 OpenFile(immutable_options, file_options, blob_file_read_hist,
43 blob_file_number, io_tracer, &file_size, &file_reader);
44 if (!s.ok()) {
45 return s;
46 }
47 }
48
49 assert(file_reader);
50
51 Statistics* const statistics = immutable_options.stats;
52
53 CompressionType compression_type = kNoCompression;
54
55 {
56 const Status s =
57 ReadHeader(file_reader.get(), read_options, column_family_id,
58 statistics, &compression_type);
59 if (!s.ok()) {
60 return s;
61 }
62 }
63
64 {
65 const Status s =
66 ReadFooter(file_reader.get(), read_options, file_size, statistics);
67 if (!s.ok()) {
68 return s;
69 }
70 }
71
72 blob_file_reader->reset(
73 new BlobFileReader(std::move(file_reader), file_size, compression_type,
74 immutable_options.clock, statistics));
75
76 return Status::OK();
77}

79Status BlobFileReader::OpenFile(
80 const ImmutableOptions& immutable_options, const FileOptions& file_opts,
81 HistogramImpl* blob_file_read_hist, uint64_t blob_file_number,
82 const std::shared_ptr<IOTracer>& io_tracer, uint64_t* file_size,
83 std::unique_ptr<RandomAccessFileReader>* file_reader) {
84 assert(file_size);
85 assert(file_reader);
86
87 const auto& cf_paths = immutable_options.cf_paths;
88 assert(!cf_paths.empty());
89
90 const std::string blob_file_path =
91 BlobFileName(cf_paths.front().path, blob_file_number);
92
93 FileSystem* const fs = immutable_options.fs.get();
94 assert(fs);
95
96 constexpr IODebugContext* dbg = nullptr;
97
98 {
99 TEST_SYNC_POINT("BlobFileReader::OpenFile:GetFileSize");
100
101 const Status s =
102 fs->GetFileSize(blob_file_path, IOOptions(), file_size, dbg);
103 if (!s.ok()) {
104 return s;
105 }
106 }
107
108 if (*file_size < BlobLogHeader::kSize + BlobLogFooter::kSize) {
109 return Status::Corruption("Malformed blob file");
110 }
111
112 std::unique_ptr<FSRandomAccessFile> file;
113
114 {
115 TEST_SYNC_POINT("BlobFileReader::OpenFile:NewRandomAccessFile");
116
117 const Status s =
118 fs->NewRandomAccessFile(blob_file_path, file_opts, &file, dbg);
119 if (!s.ok()) {
120 return s;
121 }
122 }
123
124 assert(file);
125
126 if (immutable_options.advise_random_on_open) {
127 file->Hint(FSRandomAccessFile::kRandom);
128 }
129
130 file_reader->reset(new RandomAccessFileReader(
131 std::move(file), blob_file_path, immutable_options.clock, io_tracer,
132 immutable_options.stats, BLOB_DB_BLOB_FILE_READ_MICROS,
133 blob_file_read_hist, immutable_options.rate_limiter.get(),
134 immutable_options.listeners));
135
136 return Status::OK();
137}

139Status BlobFileReader::ReadHeader(const RandomAccessFileReader* file_reader,
140 const ReadOptions& read_options,
141 uint32_t column_family_id,
142 Statistics* statistics,
143 CompressionType* compression_type) {
144 assert(file_reader);
145 assert(compression_type);
146
147 Slice header_slice;
148 Buffer buf;
149 AlignedBuf aligned_buf;
150
151 {
152 TEST_SYNC_POINT("BlobFileReader::ReadHeader:ReadFromFile");
153
154 constexpr uint64_t read_offset = 0;
155 constexpr size_t read_size = BlobLogHeader::kSize;
156
157 const Status s =
158 ReadFromFile(file_reader, read_options, read_offset, read_size,
159 statistics, &header_slice, &buf, &aligned_buf);
160 if (!s.ok()) {
161 return s;
162 }
163
164 TEST_SYNC_POINT_CALLBACK("BlobFileReader::ReadHeader:TamperWithResult",
165 &header_slice);
166 }
167
168 BlobLogHeader header;
169
170 {
171 const Status s = header.DecodeFrom(header_slice);
172 if (!s.ok()) {
173 return s;
174 }
175 }
176
177 constexpr ExpirationRange no_expiration_range;
178
179 if (header.has_ttl || header.expiration_range != no_expiration_range) {
180 return Status::Corruption("Unexpected TTL blob file");
181 }
182
183 if (header.column_family_id != column_family_id) {
184 return Status::Corruption("Column family ID mismatch");
185 }
186
187 *compression_type = header.compression;
188
189 return Status::OK();
190}

193 const ReadOptions& read_options,

195 assert(file_size >= BlobLogHeader::kSize + BlobLogFooter::kSize);

217 }
218

220
221 {

223 if (!s.ok()) {

229

231 return Status::Corruption("Unexpected TTL blob file");

237Status BlobFileReader::ReadFromFile(const RandomAccessFileReader* file_reader,
238 const ReadOptions& read_options,
239 uint64_t read_offset, size_t read_size,
240 Statistics* statistics, Slice* slice,
241 Buffer* buf, AlignedBuf* aligned_buf) {
242 assert(slice);
243 assert(buf);
244 assert(aligned_buf);
245
246 assert(file_reader);
247
248 RecordTick(statistics, BLOB_DB_BLOB_FILE_BYTES_READ, read_size);
249
250 Status s;
251
252 IOOptions io_options;
253 s = file_reader->PrepareIOOptions(read_options, io_options);
254 if (!s.ok()) {
255 return s;
256 }
257
258 if (file_reader->use_direct_io()) {
259 constexpr char* scratch = nullptr;
260
261 s = file_reader->Read(io_options, read_offset, read_size, slice, scratch,
262 aligned_buf);
263 } else {
264 buf->reset(new char[read_size]);
265 constexpr AlignedBuf* aligned_scratch = nullptr;
266
267 s = file_reader->Read(io_options, read_offset, read_size, slice, buf->get(),
268 aligned_scratch);
269 }
270
271 if (!s.ok()) {
272 return s;
273 }
274
275 if (slice->size() != read_size) {
276 return Status::Corruption("Failed to read data from blob file");
277 }
278
279 return Status::OK();
280}

296Status BlobFileReader::GetBlob(
297 const ReadOptions& read_options, const Slice& user_key, uint64_t offset,
298 uint64_t value_size, CompressionType compression_type,
299 FilePrefetchBuffer* prefetch_buffer, MemoryAllocator* allocator,
300 std::unique_ptr<BlobContents>* result, uint64_t* bytes_read) const {
301 assert(result);
302
303 const uint64_t key_size = user_key.size();
304
305 if (!IsValidBlobOffset(offset, key_size, value_size, file_size_)) {
306 return Status::Corruption("Invalid blob offset");
307 }
308
309 if (compression_type != compression_type_) {
310 return Status::Corruption("Compression type mismatch when reading blob");
311 }
312
313 // Note: if verify_checksum is set, we read the entire blob record to be able
314 // to perform the verification; otherwise, we just read the blob itself. Since
315 // the offset in BlobIndex actually points to the blob value, we need to make
316 // an adjustment in the former case.
317 const uint64_t adjustment =
318 read_options.verify_checksums
319 ? BlobLogRecord::CalculateAdjustmentForRecordHeader(key_size)
320 : 0;
321 assert(offset >= adjustment);
322
323 const uint64_t record_offset = offset - adjustment;
324 const uint64_t record_size = value_size + adjustment;
325
326 Slice record_slice;
327 Buffer buf;
328 AlignedBuf aligned_buf;
329
330 bool prefetched = false;
331
332 if (prefetch_buffer) {
333 Status s;
334 constexpr bool for_compaction = true;
335
336 IOOptions io_options;
337 s = file_reader_->PrepareIOOptions(read_options, io_options);
338 if (!s.ok()) {
339 return s;
340 }
341 prefetched = prefetch_buffer->TryReadFromCache(
342 io_options, file_reader_.get(), record_offset,
343 static_cast<size_t>(record_size), &record_slice, &s, for_compaction);
344 if (!s.ok()) {
345 return s;
346 }
347 }
348
349 if (!prefetched) {
350 TEST_SYNC_POINT("BlobFileReader::GetBlob:ReadFromFile");
351 PERF_COUNTER_ADD(blob_read_count, 1);
352 PERF_COUNTER_ADD(blob_read_byte, record_size);
353 PERF_TIMER_GUARD(blob_read_time);
354 const Status s =
355 ReadFromFile(file_reader_.get(), read_options, record_offset,
356 static_cast<size_t>(record_size), statistics_,
357 &record_slice, &buf, &aligned_buf);
358 if (!s.ok()) {
359 return s;
360 }
361 }
362
363 TEST_SYNC_POINT_CALLBACK("BlobFileReader::GetBlob:TamperWithResult",
364 &record_slice);
365
366 if (read_options.verify_checksums) {
367 const Status s = VerifyBlob(record_slice, user_key, value_size);
368 if (!s.ok()) {
369 return s;
370 }
371 }
372
373 const Slice value_slice(record_slice.data() + adjustment, value_size);
374
375 {
376 const Status s = UncompressBlobIfNeeded(
377 value_slice, compression_type, allocator, clock_, statistics_, result);
378 if (!s.ok()) {
379 return s;
380 }
381 }
382
383 if (bytes_read) {
384 *bytes_read = record_size;
385 }
386
387 return Status::OK();
388}

390void BlobFileReader::MultiGetBlob(
391 const ReadOptions& read_options, MemoryAllocator* allocator,
392 autovector<std::pair<BlobReadRequest*, std::unique_ptr<BlobContents>>>&
393 blob_reqs,
394 uint64_t* bytes_read) const {
395 const size_t num_blobs = blob_reqs.size();
396 assert(num_blobs > 0);
397 assert(num_blobs <= MultiGetContext::MAX_BATCH_SIZE);
398
399#ifndef NDEBUG
400 for (size_t i = 0; i < num_blobs - 1; ++i) {
401 assert(blob_reqs[i].first->offset <= blob_reqs[i + 1].first->offset);
402 }
403#endif // !NDEBUG
404
405 std::vector<FSReadRequest> read_reqs;
406 autovector<uint64_t> adjustments;
407 uint64_t total_len = 0;
408 read_reqs.reserve(num_blobs);
409 for (size_t i = 0; i < num_blobs; ++i) {
410 BlobReadRequest* const req = blob_reqs[i].first;
411 assert(req);
412 assert(req->user_key);
413 assert(req->status);
414
415 const size_t key_size = req->user_key->size();
416 const uint64_t offset = req->offset;
417 const uint64_t value_size = req->len;
418
419 if (!IsValidBlobOffset(offset, key_size, value_size, file_size_)) {
420 *req->status = Status::Corruption("Invalid blob offset");
421 continue;
422 }
423 if (req->compression != compression_type_) {
424 *req->status =
425 Status::Corruption("Compression type mismatch when reading a blob");
426 continue;
427 }
428
429 const uint64_t adjustment =
430 read_options.verify_checksums
431 ? BlobLogRecord::CalculateAdjustmentForRecordHeader(key_size)
432 : 0;
433 assert(req->offset >= adjustment);
434 adjustments.push_back(adjustment);
435
436 FSReadRequest read_req;
437 read_req.offset = req->offset - adjustment;
438 read_req.len = req->len + adjustment;
439 total_len += read_req.len;
440 read_reqs.emplace_back(std::move(read_req));
441 }
442
443 RecordTick(statistics_, BLOB_DB_BLOB_FILE_BYTES_READ, total_len);
444
445 Buffer buf;
446 AlignedBuf aligned_buf;
447
448 Status s;
449 bool direct_io = file_reader_->use_direct_io();
450 if (direct_io) {
451 for (size_t i = 0; i < read_reqs.size(); ++i) {
452 read_reqs[i].scratch = nullptr;
453 }
454 } else {
455 buf.reset(new char[total_len]);
456 std::ptrdiff_t pos = 0;
457 for (size_t i = 0; i < read_reqs.size(); ++i) {
458 read_reqs[i].scratch = buf.get() + pos;
459 pos += read_reqs[i].len;
460 }
461 }
462 TEST_SYNC_POINT("BlobFileReader::MultiGetBlob:ReadFromFile");
463 PERF_COUNTER_ADD(blob_read_count, num_blobs);
464 PERF_COUNTER_ADD(blob_read_byte, total_len);
465 IOOptions opts;
466 s = file_reader_->PrepareIOOptions(read_options, opts);
467 if (s.ok()) {
468 s = file_reader_->MultiRead(opts, read_reqs.data(), read_reqs.size(),
469 direct_io ? &aligned_buf : nullptr);
470 }
471 if (!s.ok()) {
472 for (auto& req : read_reqs) {
473 req.status.PermitUncheckedError();
474 }
475 for (auto& blob_req : blob_reqs) {
476 BlobReadRequest* const req = blob_req.first;
477 assert(req);
478 assert(req->status);
479
480 if (!req->status->IsCorruption()) {
481 // Avoid overwriting corruption status.
482 *req->status = s;
483 }
484 }
485 return;
486 }
487
488 assert(s.ok());
489
490 uint64_t total_bytes = 0;
491 for (size_t i = 0, j = 0; i < num_blobs; ++i) {
492 BlobReadRequest* const req = blob_reqs[i].first;
493 assert(req);
494 assert(req->user_key);
495 assert(req->status);
496
497 if (!req->status->ok()) {
498 continue;
499 }
500
501 assert(j < read_reqs.size());
502 auto& read_req = read_reqs[j++];
503 const auto& record_slice = read_req.result;
504 if (read_req.status.ok() && record_slice.size() != read_req.len) {
505 read_req.status =
506 IOStatus::Corruption("Failed to read data from blob file");
507 }
508
509 *req->status = read_req.status;
510 if (!req->status->ok()) {
511 continue;
512 }
513
514 // Verify checksums if enabled
515 if (read_options.verify_checksums) {
516 *req->status = VerifyBlob(record_slice, *req->user_key, req->len);
517 if (!req->status->ok()) {
518 continue;
519 }
520 }
521
522 // Uncompress blob if needed
523 Slice value_slice(record_slice.data() + adjustments[i], req->len);
524 *req->status =
525 UncompressBlobIfNeeded(value_slice, compression_type_, allocator,
526 clock_, statistics_, &blob_reqs[i].second);
527 if (req->status->ok()) {
528 total_bytes += record_slice.size();
529 }
530 }
531
532 if (bytes_read) {
533 *bytes_read = total_bytes;
534 }
535}

537Status BlobFileReader::VerifyBlob(const Slice& record_slice,
538 const Slice& user_key, uint64_t value_size) {
539 PERF_TIMER_GUARD(blob_checksum_time);
540
541 BlobLogRecord record;
542
543 const Slice header_slice(record_slice.data(), BlobLogRecord::kHeaderSize);
544
545 {
546 const Status s = record.DecodeHeaderFrom(header_slice);
547 if (!s.ok()) {
548 return s;
549 }
550 }
551
552 if (record.key_size != user_key.size()) {
553 return Status::Corruption("Key size mismatch when reading blob");
554 }
555
556 if (record.value_size != value_size) {
557 return Status::Corruption("Value size mismatch when reading blob");
558 }
559
560 record.key =
561 Slice(record_slice.data() + BlobLogRecord::kHeaderSize, record.key_size);
562 if (record.key != user_key) {
563 return Status::Corruption("Key mismatch when reading blob");
564 }
565
566 record.value = Slice(record.key.data() + record.key_size, value_size);
567
568 {
569 TEST_SYNC_POINT_CALLBACK("BlobFileReader::VerifyBlob:CheckBlobCRC",
570 &record);
571
572 const Status s = record.CheckBlobCRC();
573 if (!s.ok()) {
574 return s;
575 }
576 }
577
578 return Status::OK();
579}

581Status BlobFileReader::UncompressBlobIfNeeded(
582 const Slice& value_slice, CompressionType compression_type,
583 MemoryAllocator* allocator, SystemClock* clock, Statistics* statistics,
584 std::unique_ptr<BlobContents>* result) {
585 assert(result);
586
587 if (compression_type == kNoCompression) {
588 BlobContentsCreator::Create(result, nullptr, value_slice, kNoCompression,
589 allocator);
590 return Status::OK();
591 }
592
593 UncompressionContext context(compression_type);
594 UncompressionInfo info(context, UncompressionDict::GetEmptyDict(),
595 compression_type);
596
597 size_t uncompressed_size = 0;
598 constexpr uint32_t compression_format_version = 2;
599
600 CacheAllocationPtr output;
601
602 {
603 PERF_TIMER_GUARD(blob_decompress_time);
604 StopWatch stop_watch(clock, statistics, BLOB_DB_DECOMPRESSION_MICROS);
605 output = UncompressData(info, value_slice.data(), value_slice.size(),
606 &uncompressed_size, compression_format_version,
607 allocator);
608 }
609
610 TEST_SYNC_POINT_CALLBACK(
611 "BlobFileReader::UncompressBlobIfNeeded:TamperWithResult", &output);
612
613 if (!output) {
614 return Status::Corruption("Unable to uncompress blob");
615 }
616
617 result->reset(new BlobContents(std::move(output), uncompressed_size));
618
619 return Status::OK();
620}
