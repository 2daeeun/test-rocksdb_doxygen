51 // A guard to prepare file operations, such as mutex and skip
52 // I/O context.
53 class FileOpGuard {
54 public:
55 explicit FileOpGuard(EnvLogger& logger)
56 : logger_(logger), prev_perf_level_(GetPerfLevel()) {
57 // Preserve iostats not to pollute writes from user writes. We might
58 // need a better solution than this.
59 SetPerfLevel(PerfLevel::kDisable);
60 IOSTATS_SET_DISABLE(true);
61 logger.mutex_.Lock();
62 }
63 ~FileOpGuard() {
64 logger_.mutex_.Unlock();
65 IOSTATS_SET_DISABLE(false);
66 SetPerfLevel(prev_perf_level_);
67 }
68
69 private:
70 EnvLogger& logger_;
71 PerfLevel prev_perf_level_;
72 };

74 void FlushLocked() {
75 mutex_.AssertHeld();
76 if (flush_pending_) {
77 flush_pending_ = false;
78 file_.Flush(IOOptions()).PermitUncheckedError();
79 file_.reset_seen_error();
80 }
81 last_flush_micros_ = clock_->NowMicros();
82 }

84 void Flush() override {
85 TEST_SYNC_POINT("EnvLogger::Flush:Begin1");
86 TEST_SYNC_POINT("EnvLogger::Flush:Begin2");
87
88 FileOpGuard guard(*this);
89 FlushLocked();

92 Status CloseImpl() override { return CloseHelper(); }

94 Status CloseHelper() {
95 FileOpGuard guard(*this);
96 const auto close_status = file_.Close(IOOptions());
97
98 if (close_status.ok()) {
99 return close_status;
100 }
101 return Status::IOError("Close of log file failed with error:" +
102 (close_status.getState()
103 ? std::string(close_status.getState())
104 : std::string()));
105 }

107 using Logger::Logv;
108 void Logv(const char* format, va_list ap) override {
109 IOSTATS_TIMER_GUARD(logger_nanos);
110
111 const uint64_t thread_id = env_->GetThreadID();
112
113 // We try twice: the first time with a fixed-size stack allocated buffer,
114 // and the second time with a much larger dynamically allocated buffer.
115 char buffer[500];
116 for (int iter = 0; iter < 2; iter++) {
117 char* base;
118 int bufsize;
119 if (iter == 0) {
120 bufsize = sizeof(buffer);
121 base = buffer;
122 } else {
123 bufsize = 65536;
124 base = new char[bufsize];
125 }
126 char* p = base;
127 char* limit = base + bufsize;
128
129 port::TimeVal now_tv;
130 port::GetTimeOfDay(&now_tv, nullptr);
131 const time_t seconds = now_tv.tv_sec;
132 struct tm t;
133 port::LocalTimeR(&seconds, &t);
134 p += snprintf(p, limit - p, "%04d/%02d/%02d-%02d:%02d:%02d.%06d %llu ",
135 t.tm_year + 1900, t.tm_mon + 1, t.tm_mday, t.tm_hour,
136 t.tm_min, t.tm_sec, static_cast<int>(now_tv.tv_usec),
137 static_cast<long long unsigned int>(thread_id));
138
139 // Print the message
140 if (p < limit) {
141 va_list backup_ap;
142 va_copy(backup_ap, ap);
143 p += vsnprintf(p, limit - p, format, backup_ap);
144 va_end(backup_ap);
145 }
146
147 // Truncate to available space if necessary
148 if (p >= limit) {
149 if (iter == 0) {
150 continue; // Try again with larger buffer
151 } else {
152 p = limit - 1;
153 }
154 }
155
156 // Add newline if necessary
157 if (p == base || p[-1] != '\n') {
158 *p++ = '\n';
159 }
160
161 assert(p <= limit);
162 {
163 FileOpGuard guard(*this);
164 // We will ignore any error returned by Append().
165 file_.Append(IOOptions(), Slice(base, p - base)).PermitUncheckedError();
166 file_.reset_seen_error();
167 flush_pending_ = true;
168 const uint64_t now_micros = clock_->NowMicros();
169 if (now_micros - last_flush_micros_ >= flush_every_seconds_ * 1000000) {
170 FlushLocked();
171 }
172 }
173 if (base != buffer) {
174 delete[] base;
175 }
176 break;
177 }
178 }
179
180 size_t GetLogFileSize() const override {
181 MutexLock l(&mutex_);
182 return file_.GetFileSize();
183 }
184
185 private:
186 Env* env_;
187 SystemClock* clock_;
188 WritableFileWriter file_;
189 mutable port::Mutex mutex_; // Mutex to protect the shared variables below.
190 const static uint64_t flush_every_seconds_ = 5;
191 std::atomic_uint_fast64_t last_flush_micros_;
192 std::atomic<bool> flush_pending_;
193};
