62// For getting SecondaryCache-compatible helpers from a BlockType. This is
63// useful for accessing block cache in untyped contexts, such as for generic
64// cache warming in table builder.
65const std::array<const Cache::CacheItemHelper*,
66 static_cast<unsigned>(BlockType::kInvalid) + 1>
67 kCacheItemFullHelperForBlockType{{
68 BlockCacheInterface<Block_kData>::GetFullHelper(),
69 BlockCacheInterface<ParsedFullFilterBlock>::GetFullHelper(),
70 BlockCacheInterface<Block_kFilterPartitionIndex>::GetFullHelper(),
71 nullptr, // kProperties
72 BlockCacheInterface<UncompressionDict>::GetFullHelper(),
73 BlockCacheInterface<Block_kRangeDeletion>::GetFullHelper(),
74 nullptr, // kHashIndexPrefixes
75 nullptr, // kHashIndexMetadata
76 nullptr, // kMetaIndex (not yet stored in block cache)
77 BlockCacheInterface<Block_kIndex>::GetFullHelper(),
78 nullptr, // kInvalid
79 }};
80
81// For getting basic helpers from a BlockType (no SecondaryCache support)
82const std::array<const Cache::CacheItemHelper*,
83 static_cast<unsigned>(BlockType::kInvalid) + 1>
84 kCacheItemBasicHelperForBlockType{{
85 BlockCacheInterface<Block_kData>::GetBasicHelper(),
86 BlockCacheInterface<ParsedFullFilterBlock>::GetBasicHelper(),
87 BlockCacheInterface<Block_kFilterPartitionIndex>::GetBasicHelper(),
88 nullptr, // kProperties
89 BlockCacheInterface<UncompressionDict>::GetBasicHelper(),
90 BlockCacheInterface<Block_kRangeDeletion>::GetBasicHelper(),
91 nullptr, // kHashIndexPrefixes
92 nullptr, // kHashIndexMetadata
93 nullptr, // kMetaIndex (not yet stored in block cache)
94 BlockCacheInterface<Block_kIndex>::GetBasicHelper(),
95 nullptr, // kInvalid
96 }};
97} // namespace

99const Cache::CacheItemHelper* GetCacheItemHelper(
100 BlockType block_type, CacheTier lowest_used_cache_tier) {
101 if (lowest_used_cache_tier > CacheTier::kVolatileTier) {
102 return kCacheItemFullHelperForBlockType[static_cast<unsigned>(block_type)];
103 } else {
104 return kCacheItemBasicHelperForBlockType[static_cast<unsigned>(block_type)];
105 }
106}
