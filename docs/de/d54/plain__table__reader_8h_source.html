58// The reader class of PlainTable. For description of PlainTable format
59// See comments of class PlainTableFactory, where instances of
60// PlainTableReader are created.
61class PlainTableReader : public TableReader {
62 public:
63 // Based on following output file format shown in plain_table_factory.h
64 // When opening the output file, PlainTableReader creates a hash table
65 // from key prefixes to offset of the output file. PlainTable will decide
66 // whether it points to the data offset of the first key with the key prefix
67 // or the offset of it. If there are too many keys share this prefix, it will
68 // create a binary search-able index from the suffix to offset on disk.
69 static Status Open(const ImmutableOptions& ioptions,
70 const EnvOptions& env_options,
71 const InternalKeyComparator& internal_comparator,
72 std::unique_ptr<RandomAccessFileReader>&& file,
73 uint64_t file_size, std::unique_ptr<TableReader>* table,
74 const int bloom_bits_per_key, double hash_table_ratio,
75 size_t index_sparseness, size_t huge_page_tlb_size,
76 bool full_scan_mode, const bool immortal_table = false,
77 const SliceTransform* prefix_extractor = nullptr);
78
79 // Returns new iterator over table contents
80 // compaction_readahead_size: its value will only be used if for_compaction =
81 // true
82 InternalIterator* NewIterator(const ReadOptions&,
83 const SliceTransform* prefix_extractor,
84 Arena* arena, bool skip_filters,
85 TableReaderCaller caller,
86 size_t compaction_readahead_size = 0,
87 bool allow_unprepared_value = false) override;
88
89 void Prepare(const Slice& target) override;
90
91 Status Get(const ReadOptions& readOptions, const Slice& key,
92 GetContext* get_context, const SliceTransform* prefix_extractor,
93 bool skip_filters = false) override;
94
95 uint64_t ApproximateOffsetOf(const ReadOptions& read_options,
96 const Slice& key,
97 TableReaderCaller caller) override;
98
99 uint64_t ApproximateSize(const ReadOptions& read_options, const Slice& start,
100 const Slice& end, TableReaderCaller caller) override;
101
102 uint32_t GetIndexSize() const { return index_.GetIndexSize(); }
103 void SetupForCompaction() override;
104
105 std::shared_ptr<const TableProperties> GetTableProperties() const override {
106 return table_properties_;
107 }
108
109 size_t ApproximateMemoryUsage() const override {
110 return arena_.MemoryAllocatedBytes();
111 }
112
113 PlainTableReader(const ImmutableOptions& ioptions,
114 std::unique_ptr<RandomAccessFileReader>&& file,
115 const EnvOptions& env_options,
116 const InternalKeyComparator& internal_comparator,
117 EncodingType encoding_type, uint64_t file_size,
118 const TableProperties* table_properties,
119 const SliceTransform* prefix_extractor);
120 virtual ~PlainTableReader();
121
122 protected:
123 // Check bloom filter to see whether it might contain this prefix.
124 // The hash of the prefix is given, since it can be reused for index lookup
125 // too.
126 virtual bool MatchBloom(uint32_t hash) const;
127
128 // PopulateIndex() builds index of keys. It must be called before any query
129 // to the table.
130 //
131 // props: the table properties object that need to be stored. Ownership of
132 // the object will be passed.
133 //
134
135 Status PopulateIndex(TableProperties* props, int bloom_bits_per_key,
136 double hash_table_ratio, size_t index_sparseness,
137 size_t huge_page_tlb_size);
138
139 Status MmapDataIfNeeded();
140
141 private:
142 const InternalKeyComparator internal_comparator_;
143 EncodingType encoding_type_;
144 // represents plain table's current status.
145 Status status_;
146
147 PlainTableIndex index_;
148 bool full_scan_mode_;
149
150 // data_start_offset_ and data_end_offset_ defines the range of the
151 // sst file that stores data.
152 const uint32_t data_start_offset_ = 0;
153 const uint32_t user_key_len_;
154 const SliceTransform* prefix_extractor_;
155
156 static const size_t kNumInternalBytes = 8;
157
158 // Bloom filter is used to rule out non-existent key
159 bool enable_bloom_;
160 PlainTableBloomV1 bloom_;
161 PlainTableReaderFileInfo file_info_;
162 Arena arena_;
163 CacheAllocationPtr index_block_alloc_;
164 CacheAllocationPtr bloom_block_alloc_;
165
166 const ImmutableOptions& ioptions_;
167 std::unique_ptr<Cleanable> dummy_cleanable_;
168 uint64_t file_size_;
169
170 protected: // for testing
171 std::shared_ptr<const TableProperties> table_properties_;
172
173 private:
174 bool IsFixedLength() const {
175 return user_key_len_ != kPlainTableVariableLength;
176 }
177
178 size_t GetFixedInternalKeyLength() const {
179 return user_key_len_ + kNumInternalBytes;
180 }
181
182 Slice GetPrefix(const Slice& target) const {
183 assert(target.size() >= 8); // target is internal key
184 return GetPrefixFromUserKey(ExtractUserKey(target));
185 }
186
187 Slice GetPrefix(const ParsedInternalKey& target) const {
188 return GetPrefixFromUserKey(target.user_key);
189 }
190
191 Slice GetPrefixFromUserKey(const Slice& user_key) const {
192 if (!IsTotalOrderMode()) {
193 return prefix_extractor_->Transform(user_key);
194 } else {
195 // Use empty slice as prefix if prefix_extractor is not set.
196 // In that case,
197 // it falls back to pure binary search and
198 // total iterator seek is supported.
199 return Slice();
200 }
201 }

203 friend class TableCache;
204 friend class PlainTableIterator;
205
206 // Internal helper function to generate an IndexRecordList object from all
207 // the rows, which contains index records as a list.
208 // If bloom_ is not null, all the keys' full-key hash will be added to the
209 // bloom filter.
210 Status PopulateIndexRecordList(PlainTableIndexBuilder* index_builder,
211 std::vector<uint32_t>* prefix_hashes);
212
213 // Internal helper function to allocate memory for bloom filter
214 void AllocateBloom(int bloom_bits_per_key, int num_prefixes,
215 size_t huge_page_tlb_size);
216
217 void FillBloom(const std::vector<uint32_t>& prefix_hashes);
218
219 // Read the key and value at `offset` to parameters for keys, the and
220 // `seekable`.
221 // On success, `offset` will be updated as the offset for the next key.
222 // `parsed_key` will be key in parsed format.
223 // if `internal_key` is not empty, it will be filled with key with slice
224 // format.
225 // if `seekable` is not null, it will return whether we can directly read
226 // data using this offset.
227 Status Next(PlainTableKeyDecoder* decoder, uint32_t* offset,
228 ParsedInternalKey* parsed_key, Slice* internal_key, Slice* value,
229 bool* seekable = nullptr) const;
230 // Get file offset for key target.
231 // return value prefix_matched is set to true if the offset is confirmed
232 // for a key with the same prefix as target.
233 Status GetOffset(PlainTableKeyDecoder* decoder, const Slice& target,
234 const Slice& prefix, uint32_t prefix_hash,
235 bool& prefix_matched, uint32_t* offset) const;
236
237 bool IsTotalOrderMode() const { return (prefix_extractor_ == nullptr); }
238
239 // No copying allowed
240 explicit PlainTableReader(const TableReader&) = delete;
241 void operator=(const TableReader&) = delete;
242};
