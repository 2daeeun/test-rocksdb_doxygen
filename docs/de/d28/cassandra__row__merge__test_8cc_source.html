48TEST(RowValueMergeTest, MergeWithRowTombstone) {
49 std::vector<RowValue> row_values;
50
51 // A row tombstone.
52 row_values.push_back(CreateRowTombstone(11));
53
54 // This row's timestamp is smaller than tombstone.
55 row_values.push_back(CreateTestRowValue({
56 CreateTestColumnSpec(kColumn, 0, 5),
57 CreateTestColumnSpec(kColumn, 1, 6),
58 }));
59
60 // Some of the column's row is smaller, some is larger.
61 row_values.push_back(CreateTestRowValue({
62 CreateTestColumnSpec(kColumn, 2, 10),
63 CreateTestColumnSpec(kColumn, 3, 12),
64 }));
65
66 // All of the column's rows are larger than tombstone.
67 row_values.push_back(CreateTestRowValue({
68 CreateTestColumnSpec(kColumn, 4, 13),
69 CreateTestColumnSpec(kColumn, 5, 14),
70 }));
71
72 RowValue merged = RowValue::Merge(std::move(row_values));
73 EXPECT_FALSE(merged.IsTombstone());
74 EXPECT_EQ(merged.get_columns().size(), 3);
75 VerifyRowValueColumns(merged.get_columns(), 0, kColumn, 3, 12);
76 VerifyRowValueColumns(merged.get_columns(), 1, kColumn, 4, 13);
77 VerifyRowValueColumns(merged.get_columns(), 2, kColumn, 5, 14);
78
79 // If the tombstone's timestamp is the latest, then it returns a
80 // row tombstone.
81 row_values.push_back(CreateRowTombstone(15));
82
83 row_values.push_back(CreateRowTombstone(17));
84
85 merged = RowValue::Merge(std::move(row_values));
86 EXPECT_TRUE(merged.IsTombstone());
87 EXPECT_EQ(merged.LastModifiedTime(), 17);
88}
