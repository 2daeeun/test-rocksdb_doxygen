13void FaultInjectionSecondaryCache::ResultHandle::UpdateHandleValue(
14 FaultInjectionSecondaryCache::ResultHandle* handle) {
15 ErrorContext* ctx = handle->cache_->GetErrorContext();
16 if (!ctx->rand.OneIn(handle->cache_->prob_)) {
17 handle->value_ = handle->base_->Value();
18 handle->size_ = handle->base_->Size();
19 }
20 handle->base_.reset();
21}

45void FaultInjectionSecondaryCache::ResultHandle::WaitAll(
46 FaultInjectionSecondaryCache* cache,
47 std::vector<SecondaryCacheResultHandle*> handles) {
48 std::vector<SecondaryCacheResultHandle*> base_handles;
49 for (SecondaryCacheResultHandle* hdl : handles) {
50 FaultInjectionSecondaryCache::ResultHandle* handle =
51 static_cast<FaultInjectionSecondaryCache::ResultHandle*>(hdl);
52 if (!handle->base_) {
53 continue;
54 }
55 base_handles.emplace_back(handle->base_.get());
56 }
57
58 cache->base_->WaitAll(base_handles);
59 for (SecondaryCacheResultHandle* hdl : handles) {
60 FaultInjectionSecondaryCache::ResultHandle* handle =
61 static_cast<FaultInjectionSecondaryCache::ResultHandle*>(hdl);
62 if (handle->base_) {
63 UpdateHandleValue(handle);
64 }
65 }
66}

79Status FaultInjectionSecondaryCache::Insert(
80 const Slice& key, Cache::ObjectPtr value,
81 const Cache::CacheItemHelper* helper, bool force_insert) {
82 ErrorContext* ctx = GetErrorContext();

90std::unique_ptr<SecondaryCacheResultHandle>
91FaultInjectionSecondaryCache::Lookup(const Slice& key,
92 const Cache::CacheItemHelper* helper,
93 Cache::CreateContext* create_context,
94 bool wait, bool advise_erase,
95 Statistics* stats,
96 bool& kept_in_sec_cache) {
97 ErrorContext* ctx = GetErrorContext();
98 if (base_is_compressed_sec_cache_) {
99 if (ctx->rand.OneIn(prob_)) {
100 return nullptr;
101 } else {
102 return base_->Lookup(key, helper, create_context, wait, advise_erase,
103 stats, kept_in_sec_cache);
104 }
105 } else {
106 std::unique_ptr<SecondaryCacheResultHandle> hdl =
107 base_->Lookup(key, helper, create_context, wait, advise_erase, stats,
108 kept_in_sec_cache);
109 if (wait && ctx->rand.OneIn(prob_)) {
110 hdl.reset();
111 }
112 return std::unique_ptr<FaultInjectionSecondaryCache::ResultHandle>(
113 new FaultInjectionSecondaryCache::ResultHandle(this, std::move(hdl)));
114 }
115}

121void FaultInjectionSecondaryCache::WaitAll(
122 std::vector<SecondaryCacheResultHandle*> handles) {
123 if (base_is_compressed_sec_cache_) {
124 ErrorContext* ctx = GetErrorContext();
125 std::vector<SecondaryCacheResultHandle*> base_handles;
126 for (SecondaryCacheResultHandle* hdl : handles) {
127 if (ctx->rand.OneIn(prob_)) {
128 continue;
129 }
130 base_handles.push_back(hdl);
131 }
132 base_->WaitAll(base_handles);
133 } else {
134 FaultInjectionSecondaryCache::ResultHandle::WaitAll(this, handles);
135 }
136}
