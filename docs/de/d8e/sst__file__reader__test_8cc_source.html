24std::string EncodeAsString(uint64_t v) {
25 char buf[16];
26 snprintf(buf, sizeof(buf), "%08" PRIu64, v);
27 return std::string(buf);
28}

30std::string EncodeAsUint64(uint64_t v) {
31 std::string dst;
32 PutFixed64(&dst, v);
33 return dst;
34}

56 const std::vector<std::string>& keys) {
57 SstFileWriter writer(soptions_, options_);
58 ASSERT_OK(writer.Open(file_name));
59 for (size_t i = 0; i + 2 < keys.size(); i += 3) {
60 ASSERT_OK(writer.Put(keys[i], keys[i]));

67 void CheckFile(const std::string& file_name,
68 const std::vector<std::string>& keys,
69 bool check_global_seqno = false) {
70 ReadOptions ropts;
71 SstFileReader reader(options_);
72 ASSERT_OK(reader.Open(file_name));
73 ASSERT_OK(reader.VerifyChecksum());
74 std::unique_ptr<Iterator> iter(reader.NewIterator(ropts));
75 iter->SeekToFirst();
76 for (size_t i = 0; i + 2 < keys.size(); i += 3) {
77 ASSERT_TRUE(iter->Valid());
78 ASSERT_EQ(iter->key().compare(keys[i]), 0);
79 ASSERT_EQ(iter->value().compare(keys[i]), 0);
80 iter->Next();
81 ASSERT_TRUE(iter->Valid());
82 ASSERT_EQ(iter->key().compare(keys[i + 1]), 0);
83 ASSERT_EQ(iter->value().compare(EncodeAsUint64(i + 1)), 0);
84 iter->Next();
85 }
86 ASSERT_FALSE(iter->Valid());
87 if (check_global_seqno) {
88 auto properties = reader.GetTableProperties();
89 ASSERT_TRUE(properties);
90 std::string hostname;
91 ASSERT_OK(env_->GetHostNameString(&hostname));
92 ASSERT_EQ(properties->db_host_id, hostname);
93 auto& user_properties = properties->user_collected_properties;
94 ASSERT_TRUE(
95 user_properties.count(ExternalSstFilePropertyNames::kGlobalSeqno));
96 }

253 void CreateFile(const std::vector<InputKeyValueDesc>& descs,
254 ExternalSstFileInfo* file_info = nullptr) {
255 SstFileWriter writer(soptions_, options_);
256
257 ASSERT_OK(writer.Open(sst_name_));
258
259 for (const auto& desc : descs) {
260 if (desc.is_delete) {
261 if (desc.use_contiguous_buffer) {
262 std::string key_with_ts(desc.key + desc.timestamp);
263 ASSERT_OK(writer.Delete(Slice(key_with_ts.data(), desc.key.size()),
264 Slice(key_with_ts.data() + desc.key.size(),
265 desc.timestamp.size())));
266 } else {
267 ASSERT_OK(writer.Delete(desc.key, desc.timestamp));
268 }
269 } else {
270 if (desc.use_contiguous_buffer) {
271 std::string key_with_ts(desc.key + desc.timestamp);
272 ASSERT_OK(writer.Put(Slice(key_with_ts.data(), desc.key.size()),
273 Slice(key_with_ts.data() + desc.key.size(),
274 desc.timestamp.size()),
275 desc.value));
276 } else {
277 ASSERT_OK(writer.Put(desc.key, desc.timestamp, desc.value));
278 }
279 }
280 }
281
282 ASSERT_OK(writer.Finish(file_info));
283 }

320TEST_F(SstFileReaderTimestampTest, Basic) {
321 std::vector<InputKeyValueDesc> input_descs;
322
323 for (uint64_t k = 0; k < kNumKeys; k += 4) {
324 // A Put with key k, timestamp k that gets overwritten by a subsequent Put
325 // with timestamp (k + 1). Note that the comparator uses descending order
326 // for the timestamp part, so we add the later Put first.
327 input_descs.emplace_back(
328 /* key */ EncodeAsString(k), /* timestamp */ EncodeAsUint64(k + 1),
329 /* value */ EncodeAsString(k * 2), /* is_delete */ false,
330 /* use_contiguous_buffer */ false);
331 input_descs.emplace_back(
332 /* key */ EncodeAsString(k), /* timestamp */ EncodeAsUint64(k),
333 /* value */ EncodeAsString(k * 3), /* is_delete */ false,
334 /* use_contiguous_buffer */ true);
335
336 // A Put with key (k + 2), timestamp (k + 2) that gets cancelled out by a
337 // Delete with timestamp (k + 3). Note that the comparator uses descending
338 // order for the timestamp part, so we add the Delete first.
339 input_descs.emplace_back(/* key */ EncodeAsString(k + 2),
340 /* timestamp */ EncodeAsUint64(k + 3),
341 /* value */ std::string(), /* is_delete */ true,
342 /* use_contiguous_buffer */ (k % 8) == 0);
343 input_descs.emplace_back(
344 /* key */ EncodeAsString(k + 2), /* timestamp */ EncodeAsUint64(k + 2),
345 /* value */ EncodeAsString(k * 5), /* is_delete */ false,
346 /* use_contiguous_buffer */ (k % 8) != 0);
347 }
348
349 CreateFile(input_descs);
350
351 // Note: below, we check the results as of each timestamp in the range,
352 // updating the expected result as needed.
353 std::vector<OutputKeyValueDesc> output_descs;
354
355 for (uint64_t ts = 0; ts < kNumKeys; ++ts) {
356 const uint64_t k = ts - (ts % 4);
357
358 switch (ts % 4) {
359 case 0: // Initial Put for key k
360 output_descs.emplace_back(/* key */ EncodeAsString(k),
361 /* timestamp */ EncodeAsUint64(ts),
362 /* value */ EncodeAsString(k * 3));
363 break;
364
365 case 1: // Second Put for key k
366 assert(output_descs.back().key == EncodeAsString(k));
367 assert(output_descs.back().timestamp == EncodeAsUint64(ts - 1));
368 assert(output_descs.back().value == EncodeAsString(k * 3));
369 output_descs.back().timestamp = EncodeAsUint64(ts);
370 output_descs.back().value = EncodeAsString(k * 2);
371 break;
372
373 case 2: // Put for key (k + 2)
374 output_descs.emplace_back(/* key */ EncodeAsString(k + 2),
375 /* timestamp */ EncodeAsUint64(ts),
376 /* value */ EncodeAsString(k * 5));
377 break;
378
379 case 3: // Delete for key (k + 2)
380 assert(output_descs.back().key == EncodeAsString(k + 2));
381 assert(output_descs.back().timestamp == EncodeAsUint64(ts - 1));
382 assert(output_descs.back().value == EncodeAsString(k * 5));
383 output_descs.pop_back();
384 break;
385 }
386
387 CheckFile(EncodeAsUint64(ts), output_descs);
388 }
389}

466TEST_F(SstFileReaderTimestampNotPersistedTest, NonMinTimestampNotAllowed) {
467 SstFileWriter writer(soptions_, options_);
468
469 ASSERT_OK(writer.Open(sst_name_));
470
471 ASSERT_NOK(writer.Delete("baz", EncodeAsUint64(2)));
472 ASSERT_OK(writer.Put("baz", EncodeAsUint64(0), "foo_val"));
473
474 ASSERT_NOK(writer.Put("key", EncodeAsUint64(2), "value1"));
475 ASSERT_OK(writer.Put("key", EncodeAsUint64(0), "value2"));
476
477 // The `SstFileWriter::DeleteRange` API documentation specifies that
478 // a range deletion tombstone added in the file does NOT delete point
479 // (Put/Merge/Delete) keys in the same file. While there is no checks in
480 // `SstFileWriter` to ensure this requirement is met, when such a range
481 // deletion does exist, it will get over-written by point data in the same
482 // file after ingestion because they have the same sequence number.
483 // We allow having a point data entry and having a range deletion entry for
484 // a key in the same file when timestamps are removed for the same reason.
485 // After the file is ingested, the range deletion will effectively get
486 // over-written by the point data since they will have the same sequence
487 // number and the same user-defined timestamps.
488 ASSERT_NOK(writer.DeleteRange("bar", "foo", EncodeAsUint64(2)));
489 ASSERT_OK(writer.DeleteRange("bar", "foo", EncodeAsUint64(0)));
490
491 ExternalSstFileInfo external_sst_file_info;
492
493 ASSERT_OK(writer.Finish(&external_sst_file_info));
494 ASSERT_EQ(external_sst_file_info.smallest_key, "baz");
495 ASSERT_EQ(external_sst_file_info.largest_key, "key");
496 ASSERT_EQ(external_sst_file_info.smallest_range_del_key, "bar");
497 ASSERT_EQ(external_sst_file_info.largest_range_del_key, "foo");
498}

610TEST_F(SstFileReaderTableIteratorTest, Basic) {
611 Options options = CurrentOptions();
612 const Comparator* ucmp = BytewiseComparator();
613 options.comparator = ucmp;
614 options.disable_auto_compactions = true;
615
616 DestroyAndReopen(options);
617
618 // Create a L0 sst file with 4 entries, two for each user key.
619 // The file should have these entries in ascending internal key order:
620 // 'bar, seq: 4, type: kTypeValue => val2'
621 // 'bar, seq: 3, type: kTypeDeletion'
622 // 'foo, seq: 2, type: kTypeDeletion'
623 // 'foo, seq: 1, type: kTypeValue => val1'
624 ASSERT_OK(Put("foo", "val1"));
625 const Snapshot* snapshot1 = dbfull()->GetSnapshot();
