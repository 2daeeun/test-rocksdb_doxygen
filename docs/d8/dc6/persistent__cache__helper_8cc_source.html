45Status PersistentCacheHelper::LookupSerialized(
46 const PersistentCacheOptions& cache_options, const BlockHandle& handle,
47 std::unique_ptr<char[]>* out_data, const size_t expected_data_size) {
48#ifdef NDEBUG
49 (void)expected_data_size;
50#endif
51 assert(cache_options.persistent_cache);
52 assert(cache_options.persistent_cache->IsCompressed());
53
54 CacheKey key =
55 BlockBasedTable::GetCacheKey(cache_options.base_cache_key, handle);
56
57 size_t size;
58 Status s =
59 cache_options.persistent_cache->Lookup(key.AsSlice(), out_data, &size);
60 if (!s.ok()) {
61 // cache miss
62 RecordTick(cache_options.statistics, PERSISTENT_CACHE_MISS);
63 return s;
64 }
65
66 // cache hit
67 // Block-based table is assumed
68 assert(expected_data_size ==
69 handle.size() + BlockBasedTable::kBlockTrailerSize);
70 assert(size == expected_data_size);
71 RecordTick(cache_options.statistics, PERSISTENT_CACHE_HIT);
72 return Status::OK();
73}

75Status PersistentCacheHelper::LookupUncompressed(
76 const PersistentCacheOptions& cache_options, const BlockHandle& handle,
77 BlockContents* contents) {
78 assert(cache_options.persistent_cache);
79 assert(!cache_options.persistent_cache->IsCompressed());
80 if (!contents) {
81 // We shouldn't lookup in the cache. Either
82 // (1) Nowhere to store
83 return Status::NotFound();
84 }
85
86 CacheKey key =
87 BlockBasedTable::GetCacheKey(cache_options.base_cache_key, handle);
88
89 std::unique_ptr<char[]> data;
90 size_t size;
91 Status s =
92 cache_options.persistent_cache->Lookup(key.AsSlice(), &data, &size);
93 if (!s.ok()) {
94 // cache miss
95 RecordTick(cache_options.statistics, PERSISTENT_CACHE_MISS);
96 return s;
97 }
98
99 // please note we are potentially comparing compressed data size with
100 // uncompressed data size
101 assert(handle.size() <= size);
102
103 // update stats
104 RecordTick(cache_options.statistics, PERSISTENT_CACHE_HIT);
105 // construct result and return
106 *contents = BlockContents(std::move(data), size);
107 return Status::OK();
108}
