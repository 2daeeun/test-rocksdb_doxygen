93// Value in block-based table file index.
94//
95// The index entry for block n is: y -> h, [x],
96// where: y is some key between the last key of block n (inclusive) and the
97// first key of block n+1 (exclusive); h is BlockHandle pointing to block n;
98// x, if present, is the first key of block n (unshortened).
99// This struct represents the "h, [x]" part.
100struct IndexValue {
101 BlockHandle handle;
102 // Empty means unknown.
103 Slice first_internal_key;
104
105 IndexValue() = default;
106 IndexValue(BlockHandle _handle, Slice _first_internal_key)
107 : handle(_handle), first_internal_key(_first_internal_key) {}
108
109 // have_first_key indicates whether the `first_internal_key` is used.
110 // If previous_handle is not null, delta encoding is used;
111 // in this case, the two handles must point to consecutive blocks:
112 // handle.offset() ==
113 // previous_handle->offset() + previous_handle->size() + kBlockTrailerSize
114 void EncodeTo(std::string* dst, bool have_first_key,
115 const BlockHandle* previous_handle) const;
116 Status DecodeFrom(Slice* input, bool have_first_key,
117 const BlockHandle* previous_handle);
118
119 std::string ToString(bool hex, bool have_first_key) const;
120};

122// Given a file's base_context_checksum and an offset of a block within that
123// file, choose a 32-bit value that is as unique as possible. This value will
124// be added to the standard checksum to get a checksum "with context," or can
125// be subtracted to "remove" context. Returns zero (no modifier) if feature is
126// disabled with base_context_checksum == 0.
127inline uint32_t ChecksumModifierForContext(uint32_t base_context_checksum,
128 uint64_t offset) {
129 // To disable on base_context_checksum == 0, we could write
130 // `if (base_context_checksum == 0) return 0;` but benchmarking shows
131 // measurable performance penalty vs. this: compute the modifier
132 // unconditionally and use an "all or nothing" bit mask to enable
133 // or disable.
134 uint32_t all_or_nothing = uint32_t{0} - (base_context_checksum != 0);
135
136 // Desired properties:
137 // (call this function f(b, o) where b = base and o = offset)
138 // 1. Fast
139 // 2. f(b1, o) == f(b2, o) iff b1 == b2
140 // (Perfectly preserve base entropy)
141 // 3. f(b, o1) == f(b, o2) only if o1 == o2 or |o1-o2| >= 4 billion
142 // (Guaranteed uniqueness for nearby offsets)
143 // 3. f(b, o + j * 2**32) == f(b, o + k * 2**32) only if j == k
144 // (Upper bits matter, and *aligned* misplacement fails check)
145 // 4. f(b1, o) == f(b2, o + x) then preferably not
146 // f(b1, o + y) == f(b2, o + x + y)
147 // (Avoid linearly correlated matches)
148 // 5. f(b, o) == 0 depends on both b and o
149 // (No predictable overlap with non-context checksums)
150 uint32_t modifier =
151 base_context_checksum ^ (Lower32of64(offset) + Upper32of64(offset));
152
153 return modifier & all_or_nothing;
154}

239 // Encoded lengths of Footers. Bytes for serialized Footer will always be
240 // >= kMinEncodedLength and <= kMaxEncodedLength.

331// Represents the contents of a block read from an SST file. Depending on how
332// it's created, it may or may not own the actual block bytes. As an example,
333// BlockContents objects representing data read from mmapped files only point
334// into the mmapped region. Depending on context, it might be a serialized
335// (potentially compressed) block, including a trailer beyond `size`, or an
336// uncompressed block.
337//
338// Please try to use this terminology when dealing with blocks:
339// * "Serialized block" - bytes that go into storage. For block-based table
340// (usually the case) this includes the block trailer. Here the `size` does
341// not include the trailer, but other places in code might include the trailer
342// in the size.
343// * "Maybe compressed block" - like a serialized block, but without the
344// trailer (or no promise of including a trailer). Must be accompanied by a
345// CompressionType in some other variable or field.
346// * "Uncompressed block" - "payload" bytes that are either stored with no
347// compression, used as input to compression function, or result of
348// decompression function.
349// * "Parsed block" - an in-memory form of a block in block cache, as it is
350// used by the table reader. Different C++ types are used depending on the
351// block type (see block_cache.h). Only trivially parsable block types
352// use BlockContents as the parsed form.
353//
354struct BlockContents {
355 // Points to block payload (without trailer)
356 Slice data;
357 CacheAllocationPtr allocation;
358
359#ifndef NDEBUG
360 // Whether there is a known trailer after what is pointed to by `data`.
361 // See BlockBasedTable::GetCompressionType.
362 bool has_trailer = false;
363#endif // NDEBUG
364
365 BlockContents() {}
366
367 // Does not take ownership of the underlying data bytes.
368 BlockContents(const Slice& _data) : data(_data) {}
369
370 // Takes ownership of the underlying data bytes.
371 BlockContents(CacheAllocationPtr&& _data, size_t _size)
372 : data(_data.get(), _size), allocation(std::move(_data)) {}
373
374 // Takes ownership of the underlying data bytes.
375 BlockContents(std::unique_ptr<char[]>&& _data, size_t _size)
376 : data(_data.get(), _size) {
377 allocation.reset(_data.release());
378 }
379
380 // Returns whether the object has ownership of the underlying data bytes.
381 bool own_bytes() const { return allocation.get() != nullptr; }
382
383 // The additional memory space taken by the block data.
384 size_t usable_size() const {
385 if (allocation.get() != nullptr) {
386 auto allocator = allocation.get_deleter().allocator;
387 if (allocator) {
388 return allocator->UsableSize(allocation.get(), data.size());
389 }
390#ifdef ROCKSDB_MALLOC_USABLE_SIZE
391 return malloc_usable_size(allocation.get());
392#else
393 return data.size();
394#endif // ROCKSDB_MALLOC_USABLE_SIZE
395 } else {
396 return 0; // no extra memory is occupied by the data
397 }
398 }
399
400 size_t ApproximateMemoryUsage() const {
401 return usable_size() + sizeof(*this);
402 }
403
404 BlockContents(BlockContents&& other) noexcept { *this = std::move(other); }
405
406 BlockContents& operator=(BlockContents&& other) {
407 data = std::move(other.data);
408 allocation = std::move(other.allocation);
409#ifndef NDEBUG
410 has_trailer = other.has_trailer;
411#endif // NDEBUG
412 return *this;
413 }
414};
