16def my_check_output(*popenargs, **kwargs):
17 """
18 If we had python 2.7, we should simply use subprocess.check_output.
19 This is a stop-gap solution for python 2.6
20 """
21 if "stdout" in kwargs:
22 raise ValueError("stdout argument not allowed, it will be overridden.")
23 process = subprocess.Popen(
24 stderr=subprocess.PIPE, stdout=subprocess.PIPE, *popenargs, **kwargs
25 )
26 output, unused_err = process.communicate()
27 retcode = process.poll()
28 if retcode:
29 cmd = kwargs.get("args")
30 if cmd is None:
31 cmd = popenargs[0]
32 raise Exception("Exit code is not 0. It is %d. Command: %s" % (retcode, cmd))
33 return output.decode("utf-8")

57 def assertRunOKFull(
58 self, params, expectedOutput, unexpected=False, isPattern=False
59 ):
60 """
61 All command-line params must be specified.
62 Allows full flexibility in testing; for example: missing db param.
63 """
64 output = my_check_output(
65 './ldb %s |grep -v "Created bg thread"' % params, shell=True
66 )
67 if not unexpected:
68 if isPattern:
69 self.assertNotEqual(expectedOutput.search(output.strip()), None)
70 else:
71 self.assertEqual(output.strip(), expectedOutput.strip())
72 else:
73 if isPattern:
74 self.assertEqual(expectedOutput.search(output.strip()), None)
75 else:
76 self.assertNotEqual(output.strip(), expectedOutput.strip())
