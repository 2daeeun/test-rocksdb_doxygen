49bool SumAggregator::Aggregate(const std::vector<Slice>& item_list,
50 std::string& result) const {
51 int64_t sum = 0;
52 for (const Slice& item : item_list) {
53 int64_t ivalue;
54 Slice v = item;
55 if (!GetVarsignedint64(&v, &ivalue) || !v.empty()) {
56 return false;
57 }
58 sum += ivalue;
59 }
60 result = EncodeHelper::EncodeInt(sum);
61 return true;
62}

79bool Last3Aggregator::Aggregate(const std::vector<Slice>& item_list,
80 std::string& result) const {
81 std::vector<Slice> last3;
82 last3.reserve(3);
83 for (auto it = item_list.rbegin(); it != item_list.rend(); ++it) {
84 Slice input = *it;
85 Slice entity;
86 bool ret;
87 while ((ret = GetLengthPrefixedSlice(&input, &entity)) == true) {
88 last3.push_back(entity);
89 if (last3.size() >= 3) {
90 break;
91 }
92 }
93 if (last3.size() >= 3) {
94 break;
95 }
96 if (!ret) {
97 continue;
98 }
99 }
100 result = EncodeHelper::EncodeList(last3);
101 return true;
102}
