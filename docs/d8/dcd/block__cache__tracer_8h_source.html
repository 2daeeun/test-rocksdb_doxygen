50// Lookup context for tracing block cache accesses.
51// We trace block accesses at five places:
52// 1. BlockBasedTable::GetFilter
53// 2. BlockBasedTable::GetUncompressedDict.
54// 3. BlockBasedTable::MaybeReadAndLoadToCache. (To trace access on data, index,
55// and range deletion block.)
56// 4. BlockBasedTable::Get. (To trace the referenced key and whether the
57// referenced key exists in a fetched data block.)
58// 5. BlockBasedTable::MultiGet. (To trace the referenced key and whether the
59// referenced key exists in a fetched data block.)
60// The context is created at:
61// 1. BlockBasedTable::Get. (kUserGet)
62// 2. BlockBasedTable::MultiGet. (kUserMGet)
63// 3. BlockBasedTable::NewIterator. (either kUserIterator, kCompaction, or
64// external SST ingestion calls this function.)
65// 4. BlockBasedTable::Open. (kPrefetch)
66// 5. Index/Filter::CacheDependencies. (kPrefetch)
67// 6. BlockBasedTable::ApproximateOffsetOf. (kCompaction or
68// kUserApproximateSize).
69struct BlockCacheLookupContext {
70 BlockCacheLookupContext(const TableReaderCaller& _caller) : caller(_caller) {}
71 BlockCacheLookupContext(const TableReaderCaller& _caller, uint64_t _get_id,
72 bool _get_from_user_specified_snapshot)
73 : caller(_caller),
74 get_id(_get_id),
75 get_from_user_specified_snapshot(_get_from_user_specified_snapshot) {}
76 const TableReaderCaller caller;
77 // These are populated when we perform lookup/insert on block cache. The block
78 // cache tracer uses these inforation when logging the block access at
79 // BlockBasedTable::GET and BlockBasedTable::MultiGet.
80 bool is_cache_hit = false;
81 bool no_insert = false;
82 TraceType block_type = TraceType::kTraceMax;
83 uint64_t block_size = 0;
84 std::string block_key;
85 uint64_t num_keys_in_block = 0;
86 // The unique id associated with Get and MultiGet. This enables us to track
87 // how many blocks a Get/MultiGet request accesses. We can also measure the
88 // impact of row cache vs block cache.
89 uint64_t get_id = 0;
90 std::string referenced_key;
91 bool get_from_user_specified_snapshot = false;
92
93 void FillLookupContext(bool _is_cache_hit, bool _no_insert,
94 TraceType _block_type, uint64_t _block_size,
95 const std::string& _block_key,
96 uint64_t _num_keys_in_block) {
97 is_cache_hit = _is_cache_hit;
98 no_insert = _no_insert;
99 block_type = _block_type;
100 block_size = _block_size;
101 block_key = _block_key;
102 num_keys_in_block = _num_keys_in_block;
103 }
104};

112// BlockCacheTraceWriter captures all RocksDB block cache accesses using a
113// user-provided TraceWriter. Every RocksDB operation is written as a single
114// trace. Each trace will have a timestamp and type, followed by the trace
115// payload.
116class BlockCacheTraceWriterImpl : public BlockCacheTraceWriter {
117 public:
118 BlockCacheTraceWriterImpl(SystemClock* clock,
119 const BlockCacheTraceWriterOptions& trace_options,
120 std::unique_ptr<TraceWriter>&& trace_writer);
121 ~BlockCacheTraceWriterImpl() = default;
122 // No copy and move.
123 BlockCacheTraceWriterImpl(const BlockCacheTraceWriterImpl&) = delete;
124 BlockCacheTraceWriterImpl& operator=(const BlockCacheTraceWriterImpl&) =
125 delete;
126 BlockCacheTraceWriterImpl(BlockCacheTraceWriterImpl&&) = delete;
127 BlockCacheTraceWriterImpl& operator=(BlockCacheTraceWriterImpl&&) = delete;
128
129 // Pass Slice references to avoid copy.
130 Status WriteBlockAccess(const BlockCacheTraceRecord& record,
131 const Slice& block_key, const Slice& cf_name,
132 const Slice& referenced_key) override;
133
134 // Write a trace header at the beginning, typically on initiating a trace,
135 // with some metadata like a magic number and RocksDB version.
136 Status WriteHeader() override;
137
138 private:
139 SystemClock* clock_;
140 BlockCacheTraceWriterOptions trace_options_;
141 std::unique_ptr<TraceWriter> trace_writer_;
142};

200// A block cache tracer. It downsamples the accesses according to
201// trace_options and uses BlockCacheTraceWriter to write the access record to
202// the trace file.
203class BlockCacheTracer {
204 public:
205 BlockCacheTracer();
206 ~BlockCacheTracer();
207 // No copy and move.
208 BlockCacheTracer(const BlockCacheTracer&) = delete;
209 BlockCacheTracer& operator=(const BlockCacheTracer&) = delete;
210 BlockCacheTracer(BlockCacheTracer&&) = delete;
211 BlockCacheTracer& operator=(BlockCacheTracer&&) = delete;
212
213 // Start writing block cache accesses to the trace_writer.
214 Status StartTrace(const BlockCacheTraceOptions& trace_options,
215 std::unique_ptr<BlockCacheTraceWriter>&& trace_writer);
216
217 // Stop writing block cache accesses to the trace_writer.
218 void EndTrace();
219
220 bool is_tracing_enabled() const {
221 return writer_.load(std::memory_order_relaxed);
222 }
223
224 Status WriteBlockAccess(const BlockCacheTraceRecord& record,
225 const Slice& block_key, const Slice& cf_name,
226 const Slice& referenced_key);
227
228 // GetId cycles from 1 to std::numeric_limits<uint64_t>::max().
229 uint64_t NextGetId();
230
231 private:
232 BlockCacheTraceOptions trace_options_;
233 // A mutex protects the writer_.
234 InstrumentedMutex trace_writer_mutex_;
235 std::atomic<BlockCacheTraceWriter*> writer_;
236 std::atomic<uint64_t> get_id_counter_;
237};
