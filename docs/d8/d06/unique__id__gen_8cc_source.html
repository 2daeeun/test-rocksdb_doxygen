43// Each of these "tracks" below should be sufficient for generating 128 bits
44// of entropy, after hashing the raw bytes. The tracks are separable for
45// testing purposes, but in production we combine as many tracks as possible
46// to ensure quality results even if some environments have degraded
47// capabilities or quality in some APIs.
48//
49// This approach has not been validated for use in cryptography. The goal is
50// generating globally unique values with high probability without coordination
51// between instances.
52//
53// Linux performance: EntropyTrackRandomDevice is much faster than
54// EntropyTrackEnvDetails, which is much faster than EntropyTrackPortUuid.
55
56struct EntropyTrackPortUuid {
57 std::array<char, 36> uuid;
58
59 void Populate(const GenerateRawUniqueIdOpts& opts) {
60 if (opts.exclude_port_uuid) {
61 return;
62 }
63 std::string s;
64 port::GenerateRfcUuid(&s);
65 if (s.size() >= uuid.size()) {
66 std::copy_n(s.begin(), uuid.size(), uuid.begin());
67 }
68 }
69};

71struct EntropyTrackEnvDetails {
72 std::array<char, 64> hostname_buf;
73 int64_t process_id;
74 uint64_t thread_id;
75 int64_t unix_time;
76 uint64_t nano_time;
77
78 void Populate(const GenerateRawUniqueIdOpts& opts) {
79 if (opts.exclude_env_details) {
80 return;
81 }
82 opts.env->GetHostName(hostname_buf.data(), hostname_buf.size())
83 .PermitUncheckedError();
84 process_id = port::GetProcessID();
85 thread_id = opts.env->GetThreadID();
86 opts.env->GetCurrentTime(&unix_time).PermitUncheckedError();
87 nano_time = opts.env->NowNanos();
88 }
89};

91struct EntropyTrackRandomDevice {
92 using RandType = std::random_device::result_type;
93 static constexpr size_t kNumRandVals =
94 /* generous bits */ 192U / (8U * sizeof(RandType));
95 std::array<RandType, kNumRandVals> rand_vals;
96
97 void Populate(const GenerateRawUniqueIdOpts& opts) {
98 if (opts.exclude_random_device) {
99 return;
100 }
101 std::random_device r;
102 for (auto& val : rand_vals) {
103 val = r();
104 }
105 }
106};

108struct Entropy {
109 uint64_t version_identifier;
110 EntropyTrackRandomDevice et1;
111 EntropyTrackEnvDetails et2;
112 EntropyTrackPortUuid et3;
113
114 void Populate(const GenerateRawUniqueIdOpts& opts) {
115 // If we change the format of what goes into the entropy inputs, it's
116 // conceivable there could be a physical collision in the hash input
117 // even though they are logically different. This value should change
118 // if there's a change to the "schema" here, including byte order.
119 version_identifier = (uint64_t{ROCKSDB_MAJOR} << 32) +
120 (uint64_t{ROCKSDB_MINOR} << 16) +
121 uint64_t{ROCKSDB_PATCH};
122 et1.Populate(opts);
123 et2.Populate(opts);
124 et3.Populate(opts);
125 }
126};

128void GenerateRawUniqueIdImpl(uint64_t* a, uint64_t* b,
129 const GenerateRawUniqueIdOpts& opts) {
130 Entropy e;
131 std::memset(&e, 0, sizeof(e));
132 e.Populate(opts);
133 Hash2x64(reinterpret_cast<const char*>(&e), sizeof(e), a, b);
134}

158void SemiStructuredUniqueIdGen::Reset() {
159 saved_process_id_ = port::GetProcessID();
160 GenerateRawUniqueId(&base_upper_, &base_lower_);
161 counter_ = 0;
162}

164void SemiStructuredUniqueIdGen::GenerateNext(uint64_t* upper, uint64_t* lower) {
165 if (port::GetProcessID() == saved_process_id_) {
166 // Safe to increment the atomic for guaranteed uniqueness within this
167 // process lifetime. Xor slightly better than +. See
168 // https://github.com/pdillinger/unique_id
169 *lower = base_lower_ ^ counter_.fetch_add(1);
170 *upper = base_upper_;
171 } else {
172 // There must have been a fork() or something. Rather than attempting to
173 // update in a thread-safe way, simply fall back on GenerateRawUniqueId.
174 GenerateRawUniqueId(upper, lower);
175 }
176}

188void UnpredictableUniqueIdGen::GenerateNext(uint64_t* upper, uint64_t* lower) {
189 uint64_t extra_entropy;
190 // Use timing information (if available) to add to entropy. (Not a disaster
191 // if unavailable on some platforms. High performance is important.)
192#ifdef __SSE4_2__ // More than enough to guarantee rdtsc instruction
193 extra_entropy = static_cast<uint64_t>(_rdtsc());
194#else
195 extra_entropy = SystemClock::Default()->NowNanos();
196#endif
197
198 GenerateNextWithEntropy(upper, lower, extra_entropy);
199}

201void UnpredictableUniqueIdGen::GenerateNextWithEntropy(uint64_t* upper,
202 uint64_t* lower,
203 uint64_t extra_entropy) {
204 // To efficiently ensure unique inputs to the hash function in the presence
205 // of multithreading, we do not require atomicity on the whole entropy pool,
206 // but instead only a piece of it (a 64-bit counter) that is sufficient to
207 // guarantee uniqueness.
208 uint64_t count = counter_.fetch_add(1, std::memory_order_relaxed);
209 uint64_t a = count;
210 uint64_t b = extra_entropy;
211 // Invoking the hash function several times avoids copying all the inputs
212 // to a contiguous, non-atomic buffer.
213 BijectiveHash2x64(a, b, &a, &b); // Based on XXH128
214
215 // In hashing the rest of the pool with that, we don't need to worry about
216 // races, but use atomic operations for sanitizer-friendliness.
217 for (size_t i = 0; i < pool_.size(); i += 2) {
218 assert(i + 1 < pool_.size());
219 a ^= pool_[i].load(std::memory_order_relaxed);
220 b ^= pool_[i + 1].load(std::memory_order_relaxed);
221 BijectiveHash2x64(a, b, &a, &b); // Based on XXH128
222 }
223
224 // Return result
225 *lower = a;
226 *upper = b;
227
228 // Add some back into pool. We don't really care that there's a race in
229 // storing the result back and another thread computing the next value.
230 // It's just an entropy pool.
231 pool_[count & (pool_.size() - 1)].fetch_add(a, std::memory_order_relaxed);
232}
