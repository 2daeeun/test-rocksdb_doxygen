15Slice CompactionState::SmallestUserKey() {
16 for (const auto& sub_compact_state : sub_compact_states) {
17 Slice smallest = sub_compact_state.SmallestUserKey();
18 if (!smallest.empty()) {

22 // If there is no finished output, return an empty slice.
23 return Slice{nullptr, 0};
24}

26Slice CompactionState::LargestUserKey() {
