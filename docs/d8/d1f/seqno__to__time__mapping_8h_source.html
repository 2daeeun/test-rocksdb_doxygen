26// Maximum number of entries can be encoded into SST. The data is delta encode
27// so the maximum data usage for each SST is < 0.3K
28constexpr uint64_t kMaxSeqnoTimePairsPerSST = 100;
29
30// Maximum number of entries per CF. If there's only CF with this feature on,
31// the max span divided by this number, so for example, if
32// preclude_last_level_data_seconds = 100000 (~1day), then it will sample the
33// seqno -> time every 1000 seconds (~17minutes). Then the maximum entry it
34// needs is 100.
35// When there are multiple CFs having this feature on, the sampling cadence is
36// determined by the smallest setting, the capacity is determined the largest
37// setting, also it's caped by kMaxSeqnoTimePairsPerCF * 10.
38constexpr uint64_t kMaxSeqnoTimePairsPerCF = 100;
39
40constexpr uint64_t kMaxSeqnoToTimeEntries = kMaxSeqnoTimePairsPerCF * 10;

42// SeqnoToTimeMapping stores a sampled mapping from sequence numbers to
43// unix times (seconds since epoch). This information provides rough bounds
44// between sequence numbers and their write times, but is primarily designed
45// for getting a best lower bound on the sequence number of data written no
46// later than a specified time.
47//
48// For ease of sampling, it is assumed that the recorded time in each pair
49// comes at or after the sequence number and before the next sequence number,
50// so this example:
51//
52// Seqno: 10, 11, ... 20, 21, ... 30, 31, ...
53// Time: ... 500 ... 600 ... 700 ...
54//
55// would be represented as
56// 10 -> 500
57// 20 -> 600
58// 30 -> 700
59//
60// In typical operation, the list is in "enforced" operation to maintain
61// invariants on sortedness, capacity, and time span of entries. However, some
62// operations will put the object into "unenforced" mode where those invariants
63// are relaxed until explicitly or implicitly re-enforced (which will sort and
64// filter the data).
65//
66// NOT thread safe - requires external synchronization, except a const
67// object allows concurrent reads.
68class SeqnoToTimeMapping {
69 public:
70 // A simple struct for sequence number to time pair
71 struct SeqnoTimePair {
72 SequenceNumber seqno = 0;
73 uint64_t time = 0;
74
75 SeqnoTimePair() = default;
76 SeqnoTimePair(SequenceNumber _seqno, uint64_t _time)
77 : seqno(_seqno), time(_time) {}
78
79 // Encode to dest string
80 void Encode(std::string& dest) const;
81
82 // Decode the value from input Slice and remove it from the input
83 Status Decode(Slice& input);
84
85 // For delta encoding
86 SeqnoTimePair ComputeDelta(const SeqnoTimePair& base) const {
87 return {seqno - base.seqno, time - base.time};
88 }
89
90 // For delta decoding
91 void ApplyDelta(const SeqnoTimePair& delta_or_base) {
92 seqno += delta_or_base.seqno;
93 time += delta_or_base.time;
94 }
95
96 // If another pair can be combined into this one (for optimizing
97 // normal SeqnoToTimeMapping behavior), then this mapping is modified
98 // and true is returned, indicating the other mapping can be discarded.
99 // Otherwise false is returned and nothing is changed.
100 bool Merge(const SeqnoTimePair& other);
101
102 // Ordering used for Sort()
103 bool operator<(const SeqnoTimePair& other) const {
104 return std::tie(seqno, time) < std::tie(other.seqno, other.time);
105 }
106
107 bool operator==(const SeqnoTimePair& other) const {
108 return std::tie(seqno, time) == std::tie(other.seqno, other.time);
109 }
110
111 static bool SeqnoLess(const SeqnoTimePair& a, const SeqnoTimePair& b) {
112 return a.seqno < b.seqno;
113 }
114
115 static bool TimeLess(const SeqnoTimePair& a, const SeqnoTimePair& b) {
116 return a.time < b.time;
117 }
118 };
119
120 // Construct an empty SeqnoToTimeMapping with no limits.
121 SeqnoToTimeMapping() {}
122
123 // ==== Configuration for enforced state ==== //
124 // Set a time span beyond which old entries can be deleted. Specifically,
125 // under enforcement mode, the structure will maintian only one entry older
126 // than the newest entry time minus max_time_span, so that
127 // GetProximalSeqnoBeforeTime queries back to that time return a good result.
128 // UINT64_MAX == unlimited. 0 == retain just one latest entry. Returns *this.
129 SeqnoToTimeMapping& SetMaxTimeSpan(uint64_t max_time_span);
130
131 // Set the nominal capacity under enforcement mode. The structure is allowed
132 // to grow some reasonable fraction larger but will automatically compact
133 // down to this size. UINT64_MAX == unlimited. 0 == retain nothing.
134 // Returns *this.
135 SeqnoToTimeMapping& SetCapacity(uint64_t capacity);
136
137 // ==== Modifiers, enforced ==== //
138 // Adds a series of mappings interpolating from from_seqno->from_time to
139 // to_seqno->to_time. This can only be called on an empty object and both
140 // seqno range and time range are inclusive.
141 bool PrePopulate(SequenceNumber from_seqno, SequenceNumber to_seqno,
142 uint64_t from_time, uint64_t to_time);
143
144 // Append a new entry to the list. The `seqno` should be >= all previous
145 // entries. This operation maintains enforced mode invariants, and will
146 // automatically (re-)enter enforced mode if not already in that state.
147 // Returns false if the entry was merged into the most recent entry
148 // rather than creating a new entry.
149 bool Append(SequenceNumber seqno, uint64_t time);
150
151 // Clear all entries and (re-)enter enforced mode if not already in that
152 // state. Enforced limits are unchanged.
153 void Clear() {
154 pairs_.clear();
155 enforced_ = true;
156 }
157
158 // Enters the "enforced" state if not already in that state, which is
159 // useful before copying or querying. This will
160 // * Sort the entries
161 // * Discard any obsolete entries, which is aided if the caller specifies
162 // the `now` time so that entries older than now minus the max time span can
163 // be discarded.
164 // * Compact the entries to the configured capacity.
165 // Returns *this.
166 SeqnoToTimeMapping& Enforce(uint64_t now = 0);
167
168 // ==== Modifiers, unenforced ==== //
169 // Add a new random entry and enter "unenforced" state. Unlike Append(), it
170 // can be any historical data.
171 void AddUnenforced(SequenceNumber seqno, uint64_t time);
172
173 // Decode and add the entries to this mapping object. Unless starting from
174 // an empty mapping with no configured enforcement limits, this operation
175 // enters the unenforced state.
176 Status DecodeFrom(const std::string& pairs_str);
177
178 // Copies entries from the src mapping object to this one, limited to entries
179 // needed to answer GetProximalTimeBeforeSeqno() queries for the given
180 // *inclusive* seqno range. The source structure must be in enforced
181 // state as a precondition. Unless starting with this object as empty mapping
182 // with no configured enforcement limits, this object enters the unenforced
183 // state.
184 void CopyFromSeqnoRange(const SeqnoToTimeMapping& src,
185 SequenceNumber from_seqno,
186 SequenceNumber to_seqno = kMaxSequenceNumber);
187 void CopyFrom(const SeqnoToTimeMapping& src) {
188 CopyFromSeqnoRange(src, kUnknownSeqnoBeforeAll, kMaxSequenceNumber);
189 }
190
191 // ==== Accessors ==== //
192 // Given a sequence number, return the best (largest / newest) known time
193 // that is no later than the write time of that given sequence number.
194 // If no such specific time is known, returns kUnknownTimeBeforeAll.
195 // Using the example in the class comment above,
196 // GetProximalTimeBeforeSeqno(10) -> kUnknownTimeBeforeAll
197 // GetProximalTimeBeforeSeqno(11) -> 500
198 // GetProximalTimeBeforeSeqno(20) -> 500
199 // GetProximalTimeBeforeSeqno(21) -> 600
200 // Because this is a const operation depending on sortedness, the structure
201 // must be in enforced state as a precondition.
202 uint64_t GetProximalTimeBeforeSeqno(SequenceNumber seqno) const;
203
204 // Given a time, return the best (largest) sequence number whose write time
205 // is no later than that given time. If no such specific sequence number is
206 // known, returns kUnknownSeqnoBeforeAll. Using the example in the class
207 // comment above,
208 // GetProximalSeqnoBeforeTime(499) -> kUnknownSeqnoBeforeAll
209 // GetProximalSeqnoBeforeTime(500) -> 10
210 // GetProximalSeqnoBeforeTime(599) -> 10
211 // GetProximalSeqnoBeforeTime(600) -> 20
212 // Because this is a const operation depending on sortedness, the structure
213 // must be in enforced state as a precondition.
214 SequenceNumber GetProximalSeqnoBeforeTime(uint64_t time) const;
215
216 // Given current time, the configured `preserve_internal_time_seconds`, and
217 // `preclude_last_level_data_seconds`, find the relevant cutoff sequence
218 // numbers for tiering.
219 void GetCurrentTieringCutoffSeqnos(
220 uint64_t current_time, uint64_t preserve_internal_time_seconds,
221 uint64_t preclude_last_level_data_seconds,
222 SequenceNumber* preserve_time_min_seqno,
223 SequenceNumber* preclude_last_level_min_seqno) const;
224
225 // Encode to a binary string by appending to `dest`.
226 // Because this is a const operation depending on sortedness, the structure
227 // must be in enforced state as a precondition.
228 void EncodeTo(std::string& dest) const;
229
230 // Return the number of entries
231 size_t Size() const { return pairs_.size(); }
232
233 uint64_t GetCapacity() const { return capacity_; }
234
235 // If the internal list is empty
236 bool Empty() const { return pairs_.empty(); }
237
238 // return the string for user message
239 // Note: Not efficient, okay for print
240 std::string ToHumanString() const;
241
242#ifndef NDEBUG
243 const SeqnoTimePair& TEST_GetLastEntry() const { return pairs_.back(); }
244 const std::deque<SeqnoTimePair>& TEST_GetInternalMapping() const {
245 return pairs_;
246 }
247 bool TEST_IsEnforced() const { return enforced_; }
248#endif
249
250 private:
251 uint64_t max_time_span_ = UINT64_MAX;
252 uint64_t capacity_ = UINT64_MAX;
253
254 std::deque<SeqnoTimePair> pairs_;
255
256 // Whether this object is in the "enforced" state. Between calls to public
257 // functions, enforced_==true means that
258 // * `pairs_` is sorted
259 // * The capacity limit (non-strict) is met
260 // * The time span limit is met
261 // However, some places within the implementation (Append()) will temporarily
262 // violate those last two conditions while enforced_==true. See also the
263 // Enforce*() and Sort*() private functions below.
264 bool enforced_ = true;
265
266 void EnforceMaxTimeSpan(uint64_t now = 0);
267 void EnforceCapacity(bool strict);
268 void SortAndMerge();
269
270 using pair_const_iterator =
271 std::deque<SeqnoToTimeMapping::SeqnoTimePair>::const_iterator;
272 pair_const_iterator FindGreaterTime(uint64_t time) const;
273 pair_const_iterator FindGreaterSeqno(SequenceNumber seqno) const;
274 pair_const_iterator FindGreaterEqSeqno(SequenceNumber seqno) const;
275};
