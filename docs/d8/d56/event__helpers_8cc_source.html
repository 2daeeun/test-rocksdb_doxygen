51void EventHelpers::NotifyOnBackgroundError(
52 const std::vector<std::shared_ptr<EventListener>>& listeners,
53 BackgroundErrorReason reason, Status* bg_error, InstrumentedMutex* db_mutex,
54 bool* auto_recovery) {
55 if (listeners.empty()) {
56 return;
57 }
58 db_mutex->AssertHeld();
59 // release lock while notifying events
60 db_mutex->Unlock();
61 for (auto& listener : listeners) {
62 listener->OnBackgroundError(reason, bg_error);
63 bg_error->PermitUncheckedError();
64 if (*auto_recovery) {
65 listener->OnErrorRecoveryBegin(reason, *bg_error, auto_recovery);
66 }
67 }
68 db_mutex->Lock();
69}

71void EventHelpers::LogAndNotifyTableFileCreationFinished(
72 EventLogger* event_logger,
73 const std::vector<std::shared_ptr<EventListener>>& listeners,
74 const std::string& db_name, const std::string& cf_name,
75 const std::string& file_path, int job_id, const FileDescriptor& fd,
76 uint64_t oldest_blob_file_number, const TableProperties& table_properties,
77 TableFileCreationReason reason, const Status& s,
78 const std::string& file_checksum,
79 const std::string& file_checksum_func_name) {

84 << "table_file_creation"

88 << file_checksum_func_name << "smallest_seqno" << fd.smallest_seqno
89 << "largest_seqno" << fd.largest_seqno;
90
91 // table_properties
92 {
93 jwriter << "table_properties";
94 jwriter.StartObject();
95
96 // basic properties:
97 jwriter << "data_size" << table_properties.data_size << "index_size"
98 << table_properties.index_size << "index_partitions"
99 << table_properties.index_partitions << "top_level_index_size"
100 << table_properties.top_level_index_size
101 << "index_key_is_user_key"
102 << table_properties.index_key_is_user_key
103 << "index_value_is_delta_encoded"
104 << table_properties.index_value_is_delta_encoded << "filter_size"
105 << table_properties.filter_size << "raw_key_size"
106 << table_properties.raw_key_size << "raw_average_key_size"
107 << SafeDivide(table_properties.raw_key_size,
108 table_properties.num_entries)
109 << "raw_value_size" << table_properties.raw_value_size
110 << "raw_average_value_size"
111 << SafeDivide(table_properties.raw_value_size,
112 table_properties.num_entries)
113 << "num_data_blocks" << table_properties.num_data_blocks
114 << "num_entries" << table_properties.num_entries
115 << "num_filter_entries" << table_properties.num_filter_entries
116 << "num_deletions" << table_properties.num_deletions
117 << "num_merge_operands" << table_properties.num_merge_operands
118 << "num_range_deletions" << table_properties.num_range_deletions
119 << "format_version" << table_properties.format_version
120 << "fixed_key_len" << table_properties.fixed_key_len
121 << "filter_policy" << table_properties.filter_policy_name
122 << "column_family_name" << table_properties.column_family_name
123 << "column_family_id" << table_properties.column_family_id
124 << "comparator" << table_properties.comparator_name
125 << "user_defined_timestamps_persisted"
126 << table_properties.user_defined_timestamps_persisted
127 << "key_largest_seqno" << table_properties.key_largest_seqno
128 << "merge_operator" << table_properties.merge_operator_name
129 << "prefix_extractor_name"
130 << table_properties.prefix_extractor_name << "property_collectors"
131 << table_properties.property_collectors_names << "compression"
132 << table_properties.compression_name << "compression_options"
133 << table_properties.compression_options << "creation_time"
134 << table_properties.creation_time << "oldest_key_time"
135 << table_properties.newest_key_time << "newest_key_time"
136 << table_properties.oldest_key_time << "file_creation_time"
137 << table_properties.file_creation_time
138 << "slow_compression_estimated_data_size"
139 << table_properties.slow_compression_estimated_data_size
140 << "fast_compression_estimated_data_size"
141 << table_properties.fast_compression_estimated_data_size
142 << "db_id" << table_properties.db_id << "db_session_id"
143 << table_properties.db_session_id << "orig_file_number"
144 << table_properties.orig_file_number << "seqno_to_time_mapping";
145
146 if (table_properties.seqno_to_time_mapping.empty()) {
147 jwriter << "N/A";
148 } else {
149 SeqnoToTimeMapping tmp;
150 Status status = tmp.DecodeFrom(table_properties.seqno_to_time_mapping);
151 if (status.ok()) {
152 jwriter << tmp.ToHumanString();
153 } else {
154 jwriter << "Invalid";
155 }
156 }
157
158 // user collected properties
159 for (const auto& prop : table_properties.readable_properties) {
160 jwriter << prop.first << prop.second;
161 }
162 jwriter.EndObject();
163 }
164
165 if (oldest_blob_file_number != kInvalidBlobFileNumber) {
166 jwriter << "oldest_blob_file_number" << oldest_blob_file_number;
167 }
168
169 jwriter.EndObject();
170
171 event_logger->Log(jwriter);
172 }
173
174 if (listeners.empty()) {
175 return;
176 }
177 TableFileCreationInfo info;
178 info.db_name = db_name;
179 info.cf_name = cf_name;
180 info.file_path = file_path;
181 info.file_size = fd.file_size;
182 info.job_id = job_id;
183 info.table_properties = table_properties;
184 info.reason = reason;
185 info.status = s;
186 info.file_checksum = file_checksum;
187 info.file_checksum_func_name = file_checksum_func_name;
188 for (auto& listener : listeners) {
189 listener->OnTableFileCreated(info);
190 }
191 info.status.PermitUncheckedError();
192}

227void EventHelpers::NotifyOnErrorRecoveryEnd(
228 const std::vector<std::shared_ptr<EventListener>>& listeners,
229 const Status& old_bg_error, const Status& new_bg_error,
230 InstrumentedMutex* db_mutex) {
231 if (!listeners.empty()) {
232 db_mutex->AssertHeld();
233 // Make copies before releasing mutex to avoid race.
234 Status old_bg_error_cp = old_bg_error;
235 Status new_bg_error_cp = new_bg_error;
236 // release lock while notifying events
237 db_mutex->Unlock();
238 TEST_SYNC_POINT("NotifyOnErrorRecoveryEnd:MutexUnlocked:1");
239 TEST_SYNC_POINT("NotifyOnErrorRecoveryEnd:MutexUnlocked:2");
240 for (auto& listener : listeners) {
241 BackgroundErrorRecoveryInfo info;
242 info.old_bg_error = old_bg_error_cp;
243 info.new_bg_error = new_bg_error_cp;
244 listener->OnErrorRecoveryCompleted(old_bg_error_cp);
245 listener->OnErrorRecoveryEnd(info);
246 info.old_bg_error.PermitUncheckedError();
247 info.new_bg_error.PermitUncheckedError();
248 }
249 db_mutex->Lock();
250 } else {
251 old_bg_error.PermitUncheckedError();
252 }
253}
