239void VerifySimilar(uint64_t a, uint64_t b, double bias) {
240 ASSERT_EQ(a == 0U, b == 0U);
241 if (a == 0) {
242 return;
243 }
244 double dbl_a = static_cast<double>(a);
245 double dbl_b = static_cast<double>(b);
246 if (dbl_a > dbl_b) {
247 ASSERT_LT(static_cast<double>(dbl_a - dbl_b) / (dbl_a + dbl_b), bias);
248 } else {
249 ASSERT_LT(static_cast<double>(dbl_b - dbl_a) / (dbl_a + dbl_b), bias);
250 }
251}

253void VerifyTableProperties(
254 const TableProperties& base_tp, const TableProperties& new_tp,
255 double filter_size_bias = CACHE_LINE_SIZE >= 256 ? 0.18 : 0.1,
256 double index_size_bias = 0.1, double data_size_bias = 0.1,
257 double num_data_blocks_bias = 0.05) {
258 VerifySimilar(base_tp.data_size, new_tp.data_size, data_size_bias);
259 VerifySimilar(base_tp.index_size, new_tp.index_size, index_size_bias);
260 VerifySimilar(base_tp.filter_size, new_tp.filter_size, filter_size_bias);
261 VerifySimilar(base_tp.num_data_blocks, new_tp.num_data_blocks,
262 num_data_blocks_bias);
263
264 ASSERT_EQ(base_tp.raw_key_size, new_tp.raw_key_size);
265 ASSERT_EQ(base_tp.raw_value_size, new_tp.raw_value_size);
266 ASSERT_EQ(base_tp.num_entries, new_tp.num_entries);
267 ASSERT_EQ(base_tp.num_deletions, new_tp.num_deletions);
268 ASSERT_EQ(base_tp.num_range_deletions, new_tp.num_range_deletions);
269
270 // Merge operands may become Puts, so we only have an upper bound the exact
271 // number of merge operands.
272 ASSERT_GE(base_tp.num_merge_operands, new_tp.num_merge_operands);
273}

275void GetExpectedTableProperties(
276 TableProperties* expected_tp, const int kKeySize, const int kValueSize,
277 const int kPutsPerTable, const int kDeletionsPerTable,
278 const int kMergeOperandsPerTable, const int kRangeDeletionsPerTable,
279 const int kTableCount, const int kBloomBitsPerKey, const size_t kBlockSize,
280 const bool index_key_is_user_key, const bool value_delta_encoding) {
281 const int kKeysPerTable =
282 kPutsPerTable + kDeletionsPerTable + kMergeOperandsPerTable;
283 const int kPutCount = kTableCount * kPutsPerTable;
284 const int kDeletionCount = kTableCount * kDeletionsPerTable;
285 const int kMergeCount = kTableCount * kMergeOperandsPerTable;
286 const int kRangeDeletionCount = kTableCount * kRangeDeletionsPerTable;
287 const int kKeyCount =
288 kPutCount + kDeletionCount + kMergeCount + kRangeDeletionCount;
289 const int kAvgSuccessorSize = kKeySize / 5;
290 const int kEncodingSavePerKey = kKeySize / 4;
291 expected_tp->raw_key_size = kKeyCount * (kKeySize + 8);
292 expected_tp->raw_value_size =
293 (kPutCount + kMergeCount + kRangeDeletionCount) * kValueSize;
294 expected_tp->num_entries = kKeyCount;
295 expected_tp->num_deletions = kDeletionCount + kRangeDeletionCount;
296 expected_tp->num_merge_operands = kMergeCount;
297 expected_tp->num_range_deletions = kRangeDeletionCount;
298 expected_tp->num_data_blocks =
299 kTableCount *
300 (kKeysPerTable * (kKeySize - kEncodingSavePerKey + kValueSize)) /
301 kBlockSize;
302 expected_tp->data_size =
303 kTableCount * (kKeysPerTable * (kKeySize + 8 + kValueSize));
304 expected_tp->index_size =
305 expected_tp->num_data_blocks *
306 (kAvgSuccessorSize + (index_key_is_user_key ? 0 : 8) -
307 // discount 1 byte as value size is not encoded in value delta encoding
308 (value_delta_encoding ? 1 : 0));
309 expected_tp->filter_size =
310 kTableCount * ((kKeysPerTable * kBloomBitsPerKey + 7) / 8 +
311 /*average-ish overhead*/ CACHE_LINE_SIZE / 2);
312}
