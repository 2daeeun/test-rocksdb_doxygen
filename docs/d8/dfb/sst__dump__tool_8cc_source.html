33void print_help(bool to_stderr) {
34 std::string supported_compressions;
35 for (CompressionType ct : GetSupportedCompressions()) {
36 if (!supported_compressions.empty()) {
37 supported_compressions += ", ";
38 }
39 std::string str;
40 Status s = GetStringFromCompressionType(&str, ct);
41 assert(s.ok());
42 supported_compressions += str;
43 }
44 fprintf(
45 to_stderr ? stderr : stdout,
46 R"(sst_dump --file=<data_dir_OR_sst_file> [--command=check|scan|raw|recompress|identify]
47 --file=<data_dir_OR_sst_file>
48 Path to SST file or directory containing SST files
49
50 --env_uri=<uri of underlying Env>
51 URI of underlying Env, mutually exclusive with fs_uri
52
53 --fs_uri=<uri of underlying FileSystem>
54 URI of underlying FileSystem, mutually exclusive with env_uri
55
56 --command=check|scan|raw|verify|identify
57 check: Iterate over entries in files but don't print anything except if an error is encountered (default command)
58 When read_num, from and to are not set, it compares the number of keys read with num_entries in table
59 property and will report corruption if there is a mismatch.
60 scan: Iterate over entries in files and print them to screen
61 raw: Dump all the table contents to <file_name>_dump.txt
62 verify: Iterate all the blocks in files verifying checksum to detect possible corruption but don't print anything except if a corruption is encountered
63 recompress: reports the SST file size if recompressed with different
64 compression types
65 identify: Reports a file is a valid SST file or lists all valid SST files under a directory
66
67 --output_hex
68 Can be combined with scan command to print the keys and values in Hex
69
70 --decode_blob_index
71 Decode blob indexes and print them in a human-readable format during scans.
72
73 --from=<user_key>
74 Key to start reading from when executing check|scan
75
76 --to=<user_key>
77 Key to stop reading at when executing check|scan
78
79 --prefix=<user_key>
80 Returns all keys with this prefix when executing check|scan
81 Cannot be used in conjunction with --from
82
83 --read_num=<num>
84 Maximum number of entries to read when executing check|scan
85
86 --verify_checksum
87 Verify file checksum when executing check|scan
88
89 --input_key_hex
90 Can be combined with --from and --to to indicate that these values are encoded in Hex
91
92 --show_properties
93 Print table properties after iterating over the file when executing
94 check|scan|raw|identify
95
96 --set_block_size=<block_size>
97 Can be combined with --command=recompress to set the block size that will
98 be used when trying different compression algorithms
99
100 --compression_types=<comma-separated list of CompressionType members, e.g.,
101 kSnappyCompression>
102 Can be combined with --command=recompress to run recompression for this
103 list of compression types
104 Supported compression types: %s
105
106 --parse_internal_key=<0xKEY>
107 Convenience option to parse an internal key on the command line. Dumps the
108 internal key in hex format {'key' @ SN: type}
109
110 --compression_level_from=<compression_level>
111 Compression level to start compressing when executing recompress. One compression type

133// arg_name would include all prefix, e.g. "--my_arg="
134// arg_val is the parses value.
135// True if there is a match. False otherwise.
136// Woud exit after printing errmsg if cannot be parsed.
137bool ParseIntArg(const char* arg, const std::string arg_name,
138 const std::string err_msg, int64_t* arg_val) {
139 if (strncmp(arg, arg_name.c_str(), arg_name.size()) == 0) {
140 std::string input_str = arg + arg_name.size();
141 std::istringstream iss(input_str);
142 iss >> *arg_val;
143 if (iss.fail()) {
144 fprintf(stderr, "%s\n", err_msg.c_str());
145 exit(1);
146 }
147 return true;
148 }
149 return false;
150}

190 ROCKSDB_NAMESPACE::CompressionOptions().max_dict_bytes;
191 uint32_t compression_zstd_max_train_bytes =
192 ROCKSDB_NAMESPACE::CompressionOptions().zstd_max_train_bytes;
193 uint64_t compression_max_dict_buffer_bytes =

207 } else if (strcmp(argv[i], "--output_hex") == 0) {
208 output_hex = true;
209 } else if (strcmp(argv[i], "--decode_blob_index") == 0) {
210 decode_blob_index = true;
211 } else if (strcmp(argv[i], "--input_key_hex") == 0) {
212 input_key_hex = true;
213 } else if (sscanf(argv[i], "--read_num=%lu%c", (unsigned long*)&n, &junk) ==
214 1) {
215 read_num = n;
216 } else if (strcmp(argv[i], "--verify_checksum") == 0) {
217 verify_checksum = true;
218 } else if (strncmp(argv[i], "--command=", 10) == 0) {
219 command = argv[i] + 10;
220 } else if (strncmp(argv[i], "--from=", 7) == 0) {
221 from_key = argv[i] + 7;
222 has_from = true;
223 } else if (strncmp(argv[i], "--to=", 5) == 0) {
224 to_key = argv[i] + 5;
225 has_to = true;

227 from_key = argv[i] + 9;

229 } else if (strcmp(argv[i], "--show_properties") == 0) {

231 } else if (strcmp(argv[i], "--show_summary") == 0) {

237 } else if (ParseIntArg(argv[i], "--readahead_size=",

239 readahead_size = static_cast<size_t>(tmp_val);
240 } else if (strncmp(argv[i], "--compression_types=", 20) == 0) {
241 std::string compression_types_csv = argv[i] + 20;
242 std::istringstream iss(compression_types_csv);
243 std::string compression_type;
244 has_specified_compression_types = true;
245 while (std::getline(iss, compression_type, ',')) {
246 auto iter = std::find_if(
247 kCompressions.begin(), kCompressions.end(),
248 [&compression_type](std::pair<CompressionType, const char*> curr) {
249 return curr.second == compression_type;
250 });
251 if (iter == kCompressions.end()) {
252 fprintf(stderr, "%s is not a valid CompressionType\n",
253 compression_type.c_str());
254 exit(1);
255 }
256 compression_types.emplace_back(*iter);
257 }
258 } else if (strncmp(argv[i], "--parse_internal_key=", 21) == 0) {
259 std::string in_key(argv[i] + 21);
260 try {
261 in_key = ROCKSDB_NAMESPACE::LDBCommand::HexToString(in_key);
262 } catch (...) {
263 std::cerr << "ERROR: Invalid key input '" << in_key
264 << "' Use 0x{hex representation of internal rocksdb key}"

268 Slice sl_key = ROCKSDB_NAMESPACE::Slice(in_key);
269 ParsedInternalKey ikey;
270 int retc = 0;
271 Status pik_status =
272 ParseInternalKey(sl_key, &ikey, true /* log_err_key */);
273 if (!pik_status.ok()) {
274 std::cerr << pik_status.getState() << "\n";
275 retc = -1;
276 }
277 fprintf(stdout, "key=%s\n", ikey.DebugString(true, true).c_str());
278 return retc;
279 } else if (ParseIntArg(argv[i], "--compression_level_from=",
280 "compression_level_from must be numeric",
281 &tmp_val)) {
282 has_compression_level_from = true;
283 compress_level_from = static_cast<int>(tmp_val);
284 } else if (ParseIntArg(argv[i], "--compression_level_to=",
285 "compression_level_to must be numeric", &tmp_val)) {
286 has_compression_level_to = true;
287 compress_level_to = static_cast<int>(tmp_val);
288 } else if (ParseIntArg(argv[i], "--compression_max_dict_bytes=",
289 "compression_max_dict_bytes must be numeric",
290 &tmp_val)) {
291 if (tmp_val < 0 || tmp_val > std::numeric_limits<uint32_t>::max()) {
292 fprintf(stderr, "compression_max_dict_bytes must be a uint32_t: '%s'\n",
293 argv[i]);
294 print_help(/*to_stderr*/ true);
295 return 1;
296 }
297 compression_max_dict_bytes = static_cast<uint32_t>(tmp_val);
298 } else if (ParseIntArg(argv[i], "--compression_zstd_max_train_bytes=",
299 "compression_zstd_max_train_bytes must be numeric",
300 &tmp_val)) {
301 if (tmp_val < 0 || tmp_val > std::numeric_limits<uint32_t>::max()) {
302 fprintf(stderr,
303 "compression_zstd_max_train_bytes must be a uint32_t: '%s'\n",
304 argv[i]);
305 print_help(/*to_stderr*/ true);
306 return 1;
307 }
308 compression_zstd_max_train_bytes = static_cast<uint32_t>(tmp_val);
309 } else if (ParseIntArg(argv[i], "--compression_max_dict_buffer_bytes=",
310 "compression_max_dict_buffer_bytes must be numeric",
311 &tmp_val)) {
312 if (tmp_val < 0) {
313 fprintf(stderr,
314 "compression_max_dict_buffer_bytes must be positive: '%s'\n",
315 argv[i]);
316 print_help(/*to_stderr*/ true);
317 return 1;
318 }
319 compression_max_dict_buffer_bytes = static_cast<uint64_t>(tmp_val);
320 } else if (strcmp(argv[i], "--compression_use_zstd_finalize_dict") == 0) {
321 compression_use_zstd_finalize_dict = true;
322 } else if (strcmp(argv[i], "--help") == 0) {
323 print_help(/*to_stderr*/ false);
324 return 0;
325 } else if (strcmp(argv[i], "--version") == 0) {
326 printf("%s\n", GetRocksBuildInfoAsString("sst_dump").c_str());
327 return 0;
328 } else {
329 fprintf(stderr, "Unrecognized argument '%s'\n\n", argv[i]);
330 print_help(/*to_stderr*/ true);
331 return 1;
332 }
333 }
334
335 if (has_compression_level_from && has_compression_level_to) {
336 if (!has_specified_compression_types || compression_types.size() != 1) {
337 fprintf(stderr, "Specify one compression type.\n\n");
338 exit(1);
339 }
340 } else if (has_compression_level_from || has_compression_level_to) {
341 fprintf(stderr,
342 "Specify both --compression_level_from and "
343 "--compression_level_to.\n\n");
344 exit(1);
345 }
346
347 if (use_from_as_prefix && has_from) {
348 fprintf(stderr, "Cannot specify --prefix and --from\n\n");
349 exit(1);
350 }
351
352 if (input_key_hex) {
353 if (has_from || use_from_as_prefix) {
354 from_key = ROCKSDB_NAMESPACE::LDBCommand::HexToString(from_key);
355 }
356 if (has_to) {
357 to_key = ROCKSDB_NAMESPACE::LDBCommand::HexToString(to_key);
358 }
359 }
360
361 if (dir_or_file == nullptr) {
362 fprintf(stderr, "file or directory must be specified.\n\n");
363 print_help(/*to_stderr*/ true);
364 exit(1);
365 }
366
367 std::shared_ptr<ROCKSDB_NAMESPACE::Env> env_guard;
368
369 // If caller of SSTDumpTool::Run(...) does not specify a different env other
370 // than Env::Default(), then try to load custom env based on env_uri/fs_uri.
371 // Otherwise, the caller is responsible for creating custom env.
372 {
373 ConfigOptions config_options;
374 config_options.env = options.env;
375 Status s = Env::CreateFromUri(config_options, env_uri, fs_uri, &options.env,
376 &env_guard);
377 if (!s.ok()) {
378 fprintf(stderr, "CreateEnvFromUri: %s\n", s.ToString().c_str());
379 exit(1);
380 } else {
381 fprintf(stdout, "options.env is %p\n", options.env);
382 }
383 }
384
385 std::vector<std::string> filenames;
386 ROCKSDB_NAMESPACE::Env* env = options.env;
387 ROCKSDB_NAMESPACE::Status st = env->GetChildren(dir_or_file, &filenames);
388 bool dir = true;
389 if (!st.ok() || filenames.empty()) {
390 // dir_or_file does not exist or does not contain children
391 // Check its existence first
392 Status s = env->FileExists(dir_or_file);
393 // dir_or_file does not exist
394 if (!s.ok()) {
395 fprintf(stderr, "%s%s: No such file or directory\n", s.ToString().c_str(),
396 dir_or_file);
397 return 1;
398 }
399 // dir_or_file exists and is treated as a "file"
400 // since it has no children
401 // This is ok since later it will be checked
402 // that whether it is a valid sst or not
403 // (A directory "file" is not a valid sst)
404 filenames.clear();
405 filenames.emplace_back(dir_or_file);
406 dir = false;
407 }
408
409 uint64_t total_read = 0;
410 // List of RocksDB SST file without corruption
411 std::vector<std::string> valid_sst_files;
412 for (size_t i = 0; i < filenames.size(); i++) {
413 std::string filename = filenames.at(i);
414 if (filename.length() <= 4 ||
415 filename.rfind(".sst") != filename.length() - 4) {
416 // ignore
417 continue;
418 }
419
420 if (dir) {
421 filename = std::string(dir_or_file) + "/" + filename;
422 }
423
424 if (command == "verify") {
425 verify_checksum = true;
426 }
427
428 ROCKSDB_NAMESPACE::SstFileDumper dumper(
429 options, filename, Temperature::kUnknown, readahead_size,
430 verify_checksum, output_hex, decode_blob_index);
431 // Not a valid SST
432 if (!dumper.getStatus().ok()) {
433 fprintf(stderr, "%s: %s\n", filename.c_str(),
434 dumper.getStatus().ToString().c_str());
435 continue;
436 } else {
437 valid_sst_files.push_back(filename);
438 // Print out from and to key information once
439 // where there is at least one valid SST
440 if (valid_sst_files.size() == 1) {
441 // from_key and to_key are only used for "check", "scan", or ""
442 if (command == "check" || command == "scan" || command == "") {
443 fprintf(stdout, "from [%s] to [%s]\n",
444 ROCKSDB_NAMESPACE::Slice(from_key).ToString(true).c_str(),
445 ROCKSDB_NAMESPACE::Slice(to_key).ToString(true).c_str());
446 }
447 }
448 }
449
450 if (command == "recompress") {
451 st = dumper.ShowAllCompressionSizes(
452 set_block_size ? block_size : 16384,
453 compression_types.empty() ? kCompressions : compression_types,
454 compress_level_from, compress_level_to, compression_max_dict_bytes,
455 compression_zstd_max_train_bytes, compression_max_dict_buffer_bytes,
456 !compression_use_zstd_finalize_dict);
457 if (!st.ok()) {
458 fprintf(stderr, "Failed to recompress: %s\n", st.ToString().c_str());
459 exit(1);
460 }
461 return 0;
462 }
463
464 if (command == "raw") {
465 std::string out_filename = filename.substr(0, filename.length() - 4);
466 out_filename.append("_dump.txt");
467
468 st = dumper.DumpTable(out_filename);
469 if (!st.ok()) {
470 fprintf(stderr, "%s: %s\n", filename.c_str(), st.ToString().c_str());
471 exit(1);
472 } else {
473 fprintf(stdout, "raw dump written to file %s\n", out_filename.data());
474 }
475 continue;
476 }
477
478 // scan all files in give file path.
479 if (command == "" || command == "scan" || command == "check") {
480 st = dumper.ReadSequential(
481 command == "scan", read_num > 0 ? (read_num - total_read) : read_num,
482 has_from || use_from_as_prefix, from_key, has_to, to_key,
483 use_from_as_prefix);
484 if (!st.ok()) {
485 fprintf(stderr, "%s: %s\n", filename.c_str(), st.ToString().c_str());
486 }
487 total_read += dumper.GetReadNumber();
488 if (read_num > 0 && total_read > read_num) {
489 break;
490 }
491 }
492
493 if (command == "verify") {
494 st = dumper.VerifyChecksum();
495 if (!st.ok()) {
496 fprintf(stderr, "%s is corrupted: %s\n", filename.c_str(),
497 st.ToString().c_str());
498 } else {
499 fprintf(stdout, "The file is ok\n");
500 }
501 continue;
502 }
503
504 if (show_properties || show_summary) {
505 const ROCKSDB_NAMESPACE::TableProperties* table_properties;
506
507 std::shared_ptr<const ROCKSDB_NAMESPACE::TableProperties>
508 table_properties_from_reader;
509 st = dumper.ReadTableProperties(&table_properties_from_reader);
510 if (!st.ok()) {
511 fprintf(stderr, "%s: %s\n", filename.c_str(), st.ToString().c_str());
512 fprintf(stderr, "Try to use initial table properties\n");
513 table_properties = dumper.GetInitTableProperties();
514 } else {
515 table_properties = table_properties_from_reader.get();
516 }
517 if (table_properties != nullptr) {
518 if (show_properties) {
519 fprintf(stdout,
520 "Table Properties:\n"
521 "------------------------------\n"
522 " %s",
523 table_properties->ToString("\n ", ": ").c_str());
524 }
525 total_num_files += 1;
526 total_num_data_blocks += table_properties->num_data_blocks;
527 total_data_block_size += table_properties->data_size;
528 total_index_block_size += table_properties->index_size;
529 total_filter_block_size += table_properties->filter_size;
530 if (show_properties) {
531 fprintf(stdout,
532 "Raw user collected properties\n"
533 "------------------------------\n");
534 for (const auto& kv : table_properties->user_collected_properties) {
535 std::string prop_name = kv.first;
536 std::string prop_val = Slice(kv.second).ToString(true);
537 fprintf(stdout, " # %s: 0x%s\n", prop_name.c_str(),
538 prop_val.c_str());
539 }
540 }
541 } else {
542 fprintf(stderr, "Reader unexpectedly returned null properties\n");
543 }
544 }
545 }
546 if (show_summary) {
547 fprintf(stdout, "total number of files: %" PRIu64 "\n", total_num_files);
548 fprintf(stdout, "total number of data blocks: %" PRIu64 "\n",
549 total_num_data_blocks);
550 fprintf(stdout, "total data block size: %" PRIu64 "\n",
551 total_data_block_size);
552 fprintf(stdout, "total index block size: %" PRIu64 "\n",
553 total_index_block_size);
554 fprintf(stdout, "total filter block size: %" PRIu64 "\n",
555 total_filter_block_size);
556 }
557
558 if (valid_sst_files.empty()) {
559 // No valid SST files are found
560 // Exit with an error state
561 if (dir) {
562 fprintf(stdout, "------------------------------\n");
563 fprintf(stderr, "No valid SST files found in %s\n", dir_or_file);
564 } else {
565 fprintf(stderr, "%s is not a valid SST file\n", dir_or_file);
566 }
567 return 1;
568 } else {
569 if (command == "identify") {
570 if (dir) {
571 fprintf(stdout, "------------------------------\n");
572 fprintf(stdout, "List of valid SST files found in %s:\n", dir_or_file);
573 for (const auto& f : valid_sst_files) {
574 fprintf(stdout, "%s\n", f.c_str());
575 }
576 fprintf(stdout, "Number of valid SST files: %zu\n",
577 valid_sst_files.size());
578 } else {
579 fprintf(stdout, "%s is a valid SST file\n", dir_or_file);
580 }
581 }
582 // At least one valid SST
583 // exit with a success state
584 return 0;
585 }
