17// Helper class defining static methods for supporting the Configurable
18// class. The purpose of this class is to keep the Configurable class
19// as tight as possible and provide methods for doing the actual work
20// of configuring the objects.
21class ConfigurableHelper {
22 public:
23 // Configures the input Configurable object based on the parameters.
24 // On successful completion, the Configurable is updated with the settings
25 // from the opt_map.
26 //
27 // The acceptable values of the name/value pairs are documented with the
28 // specific class/instance.
29 //
30 // @param config_options Controls how the arguments are processed.
31 // @param opt_map Name/value pairs of the options to update
32 // @param unused If specified, this value will return the name/value
33 // pairs from opt_map that were NotFound for this object.
34 // @return OK If all values in the map were successfully updated
35 // @return NotFound If any of the names in the opt_map were not valid
36 // for this object. If unused is specified, it will contain the
37 // collection of NotFound entries
38 // @return NotSupported If any of the names are valid but the object does
39 // not know how to convert the value. This can happen if, for example,
40 // there is some nested Configurable that cannot be created.
41 // @return InvalidArgument If any of the values cannot be successfully
42 // parsed. This can also be returned if PrepareOptions encounters an
43 // error.
44 static Status ConfigureOptions(
45 const ConfigOptions& config_options, Configurable& configurable,
46 const std::unordered_map<std::string, std::string>& options,
47 std::unordered_map<std::string, std::string>* unused);
48
49 // Internal method to configure a set of options for this object.
50 // Classes may override this value to change its behavior.
51 // @param config_options Controls how the options are being configured
52 // @param type_name The name that was registered for this set of options
53 // @param type_map The map of options for this name
54 // @param opt_ptr Pointer to the object being configured for this option set.
55 // @param options The option name/values being updated. On return, any
56 // option that was found is removed from the list.
57 // @return OK If all of the options were successfully updated.
58 // @return InvalidArgument If an option was found but the value could not
59 // be updated.
60 // @return NotFound If an option name was not found in type_mape
61 // @return NotSupported If the option was found but no rule for converting
62 // the value could be found.
63 static Status ConfigureSomeOptions(
64 const ConfigOptions& config_options, Configurable& configurable,
65 const std::unordered_map<std::string, OptionTypeInfo>& type_map,
66 std::unordered_map<std::string, std::string>* options, void* opt_ptr);
67
68 // Configures a single option in the input Configurable.
69 // This method will look through the set of option names for this
70 // Configurable searching for one with the input name. If such an option
71 // is found, it will be configured via the input value.
72 //
73 // @param config_options Controls how the option is being configured
74 // @param configurable The object to configure
75 // @param name For options with sub-options (like Structs or
76 // Configurables),
77 // this value may be the name of the sub-field of the option being
78 // updated. For example, if the option is
79 // "compaction_options_fifo.allow_compaction", then field name would be
80 // "allow_compaction". For most options, field_name and opt_name will be
81 // equivalent.
82 // @param value The new value for this option.
83 // @param See ConfigureOptions for the possible return values
84 static Status ConfigureSingleOption(const ConfigOptions& config_options,
85 Configurable& configurable,
86 const std::string& name,
87 const std::string& value);
88
89 // Configures the option referenced by opt_info for this configurable
90 // This method configures the option based on opt_info for the input
91 // configurable.
92 // @param config_options Controls how the option is being configured
93 // @param configurable The object to configure
94 // @param opt_name The full option name
95 // @param name For options with sub-options (like Structs or
96 // Configurables),
97 // this value may be the name of the sub-field of the option being
98 // updated. For example, if the option is
99 // "compaction_options_fifo.allow_compaction", then field name would be
100 // "allow_compaction". For most options, field_name and opt_name will be
101 // equivalent.
102 // @param value The new value for this option.
103 // @param See ConfigureOptions for the possible return values
104 static Status ConfigureOption(const ConfigOptions& config_options,
105 Configurable& configurable,
106 const OptionTypeInfo& opt_info,
107 const std::string& opt_name,
108 const std::string& name,
109 const std::string& value, void* opt_ptr);
110
111 // Returns the value of the option associated with the input name
112 // This method is the functional inverse of ConfigureOption
113 // @param config_options Controls how the value is returned
114 // @param configurable The object from which to get the option.
115 // @param name The name of the option to return a value for.
116 // @param value The returned value associated with the named option.
117 // Note that value will be only the serialized version
118 // of the option and not "name=value"
119 // @return OK If the named field was successfully updated to value.
120 // @return NotFound If the name is not valid for this object.
121 // @param InvalidArgument If the name is valid for this object but
122 // its value cannot be serialized.
123 static Status GetOption(const ConfigOptions& config_options,
124 const Configurable& configurable,
125 const std::string& name, std::string* value);
126
127 // Serializes the input Configurable into the output result.
128 // This is the inverse of ConfigureOptions
129 // @param config_options Controls how serialization happens.
130 // @param configurable The object to serialize
131 // @param prefix A prefix to add to the each option as it is serialized.
132 // @param result The string representation of the configurable.
133 // @return OK If the options for this object wer successfully serialized.
134 // @return InvalidArgument If one or more of the options could not be
135 // serialized.
136 static Status SerializeOptions(const ConfigOptions& config_options,
137 const Configurable& configurable,
138 const std::string& prefix,
139 std::string* result);
140
141 // Internal method to list the option names for this object.
142 // Classes may override this value to change its behavior.
143 // @see ListOptions for more details
144 static Status ListOptions(const ConfigOptions& config_options,
145 const Configurable& configurable,
146 const std::string& prefix,
147 std::unordered_set<std::string>* result);
148
149 // Checks to see if the two configurables are equivalent to one other.
150 // This method assumes that the two objects are of the same class.
151 // @param config_options Controls how the options are compared.
152 // @param this_one The object to compare to.
153 // @param that_one The other object being compared.
154 // @param mismatch If the objects do not match, this parameter contains
155 // the name of the option that triggered the match failure.
156 // @param True if the objects match, false otherwise.
157 static bool AreEquivalent(const ConfigOptions& config_options,
158 const Configurable& this_one,
159 const Configurable& that_one,
160 std::string* mismatch);
161
162 private:
163 // Looks for the option specified by name in the RegisteredOptions of a
164 // configurable. If an entry matching name is found, that entry, opt_name,
165 // and pointer are returned.
166 // @param options The vector of options to search through
167 // @param name The name of the option to search for in the OptionType map
168 // @param opt_name If the name was found, this value is set to the option name
169 // associated with the input name/type.
170 // @param opt_ptr If the name was found, this value is set to the option
171 // pointer
172 // in the RegisteredOptions vector associated with this entry
173 // @return A pointer to the OptionTypeInfo from the options if found,
174 // nullptr if the name was not found in the input options
175 static const OptionTypeInfo* FindOption(const Configurable& configurable,
176 const std::string& name,
177 std::string* opt_name,
178 void** opt_ptr);
179
180 static Status ConfigureCustomizableOption(
181 const ConfigOptions& config_options, Configurable& configurable,
182 const OptionTypeInfo& opt_info, const std::string& opt_name,
183 const std::string& name, const std::string& value, void* opt_ptr);
184};
