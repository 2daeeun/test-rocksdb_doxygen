108class TtlIterator : public Iterator {
109 public:
110 explicit TtlIterator(Iterator* iter) : iter_(iter) { assert(iter_); }
111
112 ~TtlIterator() { delete iter_; }
113
114 bool Valid() const override { return iter_->Valid(); }
115
116 void SeekToFirst() override { iter_->SeekToFirst(); }
117
118 void SeekToLast() override { iter_->SeekToLast(); }
119
120 void Seek(const Slice& target) override { iter_->Seek(target); }
121
122 void SeekForPrev(const Slice& target) override { iter_->SeekForPrev(target); }
123
124 void Next() override { iter_->Next(); }
125
126 void Prev() override { iter_->Prev(); }
127
128 Slice key() const override { return iter_->key(); }
129
130 int32_t ttl_timestamp() const {
131 return DecodeFixed32(iter_->value().data() + iter_->value().size() -
132 DBWithTTLImpl::kTSLength);
133 }
134
135 Slice value() const override {
136 // TODO: handle timestamp corruption like in general iterator semantics
137 assert(DBWithTTLImpl::SanityCheckTimestamp(iter_->value()).ok());
138 Slice trimmed_value = iter_->value();
139 trimmed_value.size_ -= DBWithTTLImpl::kTSLength;
140 return trimmed_value;
141 }
142
143 Status status() const override { return iter_->status(); }
144
145 private:
146 Iterator* iter_;
147};
