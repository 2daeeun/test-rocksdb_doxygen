23TruncatedRangeDelIterator::TruncatedRangeDelIterator(
24 std::unique_ptr<FragmentedRangeTombstoneIterator> iter,
25 const InternalKeyComparator* icmp, const InternalKey* smallest,
26 const InternalKey* largest)
27 : iter_(std::move(iter)),
28 icmp_(icmp),
29 smallest_ikey_(smallest),
30 largest_ikey_(largest) {
31 // Set up bounds such that range tombstones from this iterator are
32 // truncated to range [smallest_, largest_).
33 if (smallest != nullptr) {
34 pinned_bounds_.emplace_back();
35 auto& parsed_smallest = pinned_bounds_.back();
36 Status pik_status = ParseInternalKey(smallest->Encode(), &parsed_smallest,
37 false /* log_err_key */); // TODO
38 pik_status.PermitUncheckedError();
39 parsed_smallest.type = kTypeMaxValid;
40 assert(pik_status.ok());
41 smallest_ = &parsed_smallest;
42 }
43 if (largest != nullptr) {
44 pinned_bounds_.emplace_back();
45 auto& parsed_largest = pinned_bounds_.back();
46
47 Status pik_status = ParseInternalKey(largest->Encode(), &parsed_largest,
48 false /* log_err_key */); // TODO
49 pik_status.PermitUncheckedError();
50 assert(pik_status.ok());
51
52 if (parsed_largest.type == kTypeRangeDeletion &&
53 parsed_largest.sequence == kMaxSequenceNumber) {
54 // The file boundary has been artificially extended by a range tombstone.
55 // We do not need to adjust largest to properly truncate range
56 // tombstones that extend past the boundary.
57 } else if (parsed_largest.sequence == 0) {
58 // The largest key in the sstable has a sequence number of 0. Since we
59 // guarantee that no internal keys with the same user key and sequence
60 // number can exist in a DB, we know that the largest key in this sstable
61 // cannot exist as the smallest key in the next sstable. This further
62 // implies that no range tombstone in this sstable covers largest;
63 // otherwise, the file boundary would have been artificially extended.
64 //
65 // Therefore, we will never truncate a range tombstone at largest, so we
66 // can leave it unchanged.
67 // TODO: maybe use kMaxValid here to ensure range tombstone having
68 // distinct key from point keys.
69 } else {
70 // The same user key may straddle two sstable boundaries. To ensure that
71 // the truncated end key can cover the largest key in this sstable, reduce
72 // its sequence number by 1.
73 parsed_largest.sequence -= 1;
74 // This line is not needed for correctness, but it ensures that the
75 // truncated end key is not covering keys from the next SST file.
76 parsed_largest.type = kTypeMaxValid;
77 }
78 largest_ = &parsed_largest;
79 }
80}

82bool TruncatedRangeDelIterator::Valid() const {
83 assert(iter_ != nullptr);
84 return iter_->Valid() &&
85 (smallest_ == nullptr ||
86 icmp_->Compare(*smallest_, iter_->parsed_end_key()) < 0) &&
87 (largest_ == nullptr ||
88 icmp_->Compare(iter_->parsed_start_key(), *largest_) < 0);
89}

92void TruncatedRangeDelIterator::Seek(const Slice& target) {
93 if (largest_ != nullptr &&
94 icmp_->Compare(*largest_, ParsedInternalKey(target, kMaxSequenceNumber,
95 kTypeRangeDeletion)) <= 0) {
96 iter_->Invalidate();
97 return;
98 }
99 if (smallest_ != nullptr &&
100 icmp_->user_comparator()->Compare(target, smallest_->user_key) < 0) {
101 iter_->Seek(smallest_->user_key);
102 return;
103 }
104 iter_->Seek(target);
105}

107void TruncatedRangeDelIterator::SeekInternalKey(const Slice& target) {
108 if (largest_ && icmp_->Compare(*largest_, target) <= 0) {
109 iter_->Invalidate();
110 return;
111 }
112 if (smallest_ && icmp_->Compare(target, *smallest_) < 0) {
113 // Since target < smallest, target < largest_.
114 // This seek must land on a range tombstone where end_key() > target,
115 // so there is no need to check again.
116 iter_->Seek(smallest_->user_key);
117 } else {
118 iter_->Seek(ExtractUserKey(target));
119 while (Valid() && icmp_->Compare(end_key(), target) <= 0) {
120 Next();
121 }
122 }
123}

125// NOTE: target is a user key, with timestamp if enabled.
126void TruncatedRangeDelIterator::SeekForPrev(const Slice& target) {
127 if (smallest_ != nullptr &&
128 icmp_->Compare(ParsedInternalKey(target, 0, kTypeRangeDeletion),
129 *smallest_) < 0) {
130 iter_->Invalidate();
131 return;
132 }
133 if (largest_ != nullptr &&
134 icmp_->user_comparator()->Compare(largest_->user_key, target) < 0) {
135 iter_->SeekForPrev(largest_->user_key);
136 return;
137 }
138 iter_->SeekForPrev(target);
139}

157std::map<SequenceNumber, std::unique_ptr<TruncatedRangeDelIterator>>
158TruncatedRangeDelIterator::SplitBySnapshot(
159 const std::vector<SequenceNumber>& snapshots) {
160 using FragmentedIterPair =
161 std::pair<const SequenceNumber,
162 std::unique_ptr<FragmentedRangeTombstoneIterator>>;
163
164 auto split_untruncated_iters = iter_->SplitBySnapshot(snapshots);
165 std::map<SequenceNumber, std::unique_ptr<TruncatedRangeDelIterator>>
166 split_truncated_iters;
167 std::for_each(
168 split_untruncated_iters.begin(), split_untruncated_iters.end(),
169 [&](FragmentedIterPair& iter_pair) {
170 auto truncated_iter = std::make_unique<TruncatedRangeDelIterator>(
171 std::move(iter_pair.second), icmp_, smallest_ikey_, largest_ikey_);
172 split_truncated_iters.emplace(iter_pair.first,
173 std::move(truncated_iter));
174 });
175 return split_truncated_iters;
176}

186bool ForwardRangeDelIterator::ShouldDelete(const ParsedInternalKey& parsed) {
187 // Move active iterators that end before parsed.
188 while (!active_iters_.empty() &&
189 icmp_->Compare((*active_iters_.top())->end_key(), parsed) <= 0) {
190 TruncatedRangeDelIterator* iter = PopActiveIter();
191 do {
192 iter->Next();
193 } while (iter->Valid() && icmp_->Compare(iter->end_key(), parsed) <= 0);
194 PushIter(iter, parsed);
195 assert(active_iters_.size() == active_seqnums_.size());
196 }
197
198 // Move inactive iterators that start before parsed.
199 while (!inactive_iters_.empty() &&
200 icmp_->Compare(inactive_iters_.top()->start_key(), parsed) <= 0) {
201 TruncatedRangeDelIterator* iter = PopInactiveIter();
202 while (iter->Valid() && icmp_->Compare(iter->end_key(), parsed) <= 0) {
203 iter->Next();
204 }
205 PushIter(iter, parsed);
206 assert(active_iters_.size() == active_seqnums_.size());
207 }
208
209 return active_seqnums_.empty()
210 ? false
211 : (*active_seqnums_.begin())->seq() > parsed.sequence;
212}

229bool ReverseRangeDelIterator::ShouldDelete(const ParsedInternalKey& parsed) {
230 // Move active iterators that start after parsed.
231 while (!active_iters_.empty() &&
232 icmp_->Compare(parsed, (*active_iters_.top())->start_key()) < 0) {
233 TruncatedRangeDelIterator* iter = PopActiveIter();
234 do {
235 iter->Prev();
236 } while (iter->Valid() && icmp_->Compare(parsed, iter->start_key()) < 0);
237 PushIter(iter, parsed);
238 assert(active_iters_.size() == active_seqnums_.size());
239 }
240
241 // Move inactive iterators that end after parsed.
242 while (!inactive_iters_.empty() &&
243 icmp_->Compare(parsed, inactive_iters_.top()->end_key()) < 0) {
244 TruncatedRangeDelIterator* iter = PopInactiveIter();
245 while (iter->Valid() && icmp_->Compare(parsed, iter->start_key()) < 0) {
246 iter->Prev();
247 }
248 PushIter(iter, parsed);
249 assert(active_iters_.size() == active_seqnums_.size());
250 }
251
252 return active_seqnums_.empty()
253 ? false
254 : (*active_seqnums_.begin())->seq() > parsed.sequence;
255}

264bool RangeDelAggregator::StripeRep::ShouldDelete(
265 const ParsedInternalKey& parsed, RangeDelPositioningMode mode) {
266 if (!InStripe(parsed.sequence) || IsEmpty()) {
267 return false;
268 }
269 switch (mode) {
270 case RangeDelPositioningMode::kForwardTraversal:
271 InvalidateReverseIter();
272
273 // Pick up previously unseen iterators.
274 for (auto it = std::next(iters_.begin(), forward_iter_.UnusedIdx());
275 it != iters_.end(); ++it, forward_iter_.IncUnusedIdx()) {
276 auto& iter = *it;
277 forward_iter_.AddNewIter(iter.get(), parsed);
278 }
279
280 return forward_iter_.ShouldDelete(parsed);
281 case RangeDelPositioningMode::kBackwardTraversal:
282 InvalidateForwardIter();
283
284 // Pick up previously unseen iterators.
285 for (auto it = std::next(iters_.begin(), reverse_iter_.UnusedIdx());
286 it != iters_.end(); ++it, reverse_iter_.IncUnusedIdx()) {
287 auto& iter = *it;
288 reverse_iter_.AddNewIter(iter.get(), parsed);
289 }
290
291 return reverse_iter_.ShouldDelete(parsed);
292 default:
293 assert(false);
294 return false;
295 }

298bool RangeDelAggregator::StripeRep::IsRangeOverlapped(const Slice& start,
299 const Slice& end) {
300 Invalidate();
301
302 // Set the internal start/end keys so that:
303 // - if start_ikey has the same user key and sequence number as the
304 // current end key, start_ikey will be considered greater; and
305 // - if end_ikey has the same user key and sequence number as the current
306 // start key, end_ikey will be considered greater.
307 ParsedInternalKey start_ikey(start, kMaxSequenceNumber,
308 static_cast<ValueType>(0));
309 ParsedInternalKey end_ikey(end, 0, static_cast<ValueType>(0));
310 for (auto& iter : iters_) {
311 bool checked_candidate_tombstones = false;
312 for (iter->SeekForPrev(start);
313 iter->Valid() && icmp_->Compare(iter->start_key(), end_ikey) <= 0;
314 iter->Next()) {
315 checked_candidate_tombstones = true;
316 if (icmp_->Compare(start_ikey, iter->end_key()) < 0 &&
317 icmp_->Compare(iter->start_key(), end_ikey) <= 0) {
318 return true;
319 }
320 }
321
322 if (!checked_candidate_tombstones) {
323 // Do an additional check for when the end of the range is the begin
324 // key of a tombstone, which we missed earlier since SeekForPrev'ing
325 // to the start was invalid.
326 iter->SeekForPrev(end);
327 if (iter->Valid() && icmp_->Compare(start_ikey, iter->end_key()) < 0 &&
328 icmp_->Compare(iter->start_key(), end_ikey) <= 0) {
329 return true;
330 }
331 }
332 }
333 return false;
334}

357void CompactionRangeDelAggregator::AddTombstones(
358 std::unique_ptr<FragmentedRangeTombstoneIterator> input_iter,
359 const InternalKey* smallest, const InternalKey* largest) {
360 if (input_iter == nullptr || input_iter->empty()) {
361 return;
362 }
363 // This bounds output of CompactionRangeDelAggregator::NewIterator.
364 if (!trim_ts_.empty()) {
365 assert(icmp_->user_comparator()->timestamp_size() > 0);
366 input_iter->SetTimestampUpperBound(&trim_ts_);
367 }
368
369 assert(input_iter->lower_bound() == 0);
370 assert(input_iter->upper_bound() == kMaxSequenceNumber);
371 parent_iters_.emplace_back(new TruncatedRangeDelIterator(
372 std::move(input_iter), icmp_, smallest, largest));
373
374 Slice* ts_upper_bound = nullptr;
375 if (!ts_upper_bound_.empty()) {
376 assert(icmp_->user_comparator()->timestamp_size() > 0);
377 ts_upper_bound = &ts_upper_bound_;
378 }
379 auto split_iters = parent_iters_.back()->SplitBySnapshot(*snapshots_);
380 for (auto& split_iter : split_iters) {
381 auto it = reps_.find(split_iter.first);
382 if (it == reps_.end()) {
383 bool inserted;
384 SequenceNumber upper_bound = split_iter.second->upper_bound();
385 SequenceNumber lower_bound = split_iter.second->lower_bound();
386 std::tie(it, inserted) = reps_.emplace(
387 split_iter.first, StripeRep(icmp_, upper_bound, lower_bound));
388 assert(inserted);
389 }
390 assert(it != reps_.end());
391 // ts_upper_bound is used to bound ShouldDelete() to only consider
392 // range tombstones under full_history_ts_low_ and trim_ts_. Keys covered by
393 // range tombstones that are above full_history_ts_low_ should not be
394 // dropped prematurely: user may read with a timestamp between the range
395 // tombstone and the covered key. Note that we cannot set timestamp
396 // upperbound on the original `input_iter` since `input_iter`s are later
397 // used in CompactionRangeDelAggregator::NewIterator to output range
398 // tombstones for persistence. We do not want to only persist range
399 // tombstones with timestamp lower than ts_upper_bound.
400 split_iter.second->SetTimestampUpperBound(ts_upper_bound);
401 it->second.AddTombstones(std::move(split_iter.second));
402 }
403}

416// Produce a sorted (by start internal key) stream of range tombstones from
417// `children`. lower_bound and upper_bound on internal key can be
418// optionally specified. Range tombstones that ends before lower_bound or starts
419// after upper_bound are excluded.
420// If user-defined timestamp is enabled, lower_bound and upper_bound should
421// contain timestamp.
422class TruncatedRangeDelMergingIter : public InternalIterator {
423 public:
424 TruncatedRangeDelMergingIter(
425 const InternalKeyComparator* icmp, const Slice* lower_bound,
426 const Slice* upper_bound,
427 const std::vector<std::unique_ptr<TruncatedRangeDelIterator>>& children)
428 : icmp_(icmp),
429 lower_bound_(lower_bound),
430 upper_bound_(upper_bound),
431 heap_(StartKeyMinComparator(icmp)),
432 ts_sz_(icmp_->user_comparator()->timestamp_size()) {
433 for (auto& child : children) {
434 if (child != nullptr) {
435 assert(child->lower_bound() == 0);
436 assert(child->upper_bound() == kMaxSequenceNumber);
437 children_.push_back(child.get());
438 }
439 }
440 }
441
442 bool Valid() const override {
443 return !heap_.empty() && !AfterEndKey(heap_.top());
444 }
445 Status status() const override { return Status::OK(); }
446
447 void SeekToFirst() override {
448 heap_.clear();
449 for (auto& child : children_) {
450 if (lower_bound_ != nullptr) {
451 child->Seek(ExtractUserKey(*lower_bound_));
452 // Since the above `Seek()` operates on a user key while `lower_bound_`
453 // is an internal key, we may need to advance `child` farther for it to
454 // be in bounds.
455 while (child->Valid() && BeforeStartKey(child)) {
456 child->InternalNext();
457 }
458 } else {
459 child->SeekToFirst();
460 }
461 if (child->Valid()) {
462 heap_.push(child);
463 }
464 }
465 }
466
467 void Next() override {
468 auto* top = heap_.top();
469 top->InternalNext();
470 if (top->Valid()) {
471 heap_.replace_top(top);
472 } else {
473 heap_.pop();
474 }
475 }
476
477 Slice key() const override {
478 auto* top = heap_.top();
479 if (ts_sz_) {
480 cur_start_key_.Set(top->start_key().user_key, top->seq(),
481 kTypeRangeDeletion, top->timestamp());
482 } else {
483 cur_start_key_.Set(top->start_key().user_key, top->seq(),
484 kTypeRangeDeletion);
485 }
486 assert(top->start_key().user_key.size() >= ts_sz_);
487 return cur_start_key_.Encode();
488 }
489
490 Slice value() const override {
491 auto* top = heap_.top();
492 if (!ts_sz_) {
493 return top->end_key().user_key;
494 }
495 assert(top->timestamp().size() == ts_sz_);
496 cur_end_key_.clear();
497 cur_end_key_.append(top->end_key().user_key.data(),
498 top->end_key().user_key.size() - ts_sz_);
499 cur_end_key_.append(top->timestamp().data(), ts_sz_);
500 return cur_end_key_;
501 }
502
503 // Unused InternalIterator methods
504 void Prev() override { assert(false); }
505 void Seek(const Slice& /* target */) override { assert(false); }
506 void SeekForPrev(const Slice& /* target */) override { assert(false); }
507 void SeekToLast() override { assert(false); }
508
509 private:
510 bool BeforeStartKey(const TruncatedRangeDelIterator* iter) const {
511 if (lower_bound_ == nullptr) {
512 return false;
513 }
514 return icmp_->Compare(iter->end_key(), *lower_bound_) <= 0;
515 }
516
517 bool AfterEndKey(const TruncatedRangeDelIterator* iter) const {
518 if (upper_bound_ == nullptr) {
519 return false;
520 }
521 return icmp_->Compare(iter->start_key(), *upper_bound_) > 0;
522 }
523
524 const InternalKeyComparator* icmp_;
525 const Slice* lower_bound_;
526 const Slice* upper_bound_;
527 BinaryHeap<TruncatedRangeDelIterator*, StartKeyMinComparator> heap_;
528 std::vector<TruncatedRangeDelIterator*> children_;
529
530 mutable InternalKey cur_start_key_;
531 mutable std::string cur_end_key_;
532 size_t ts_sz_;
533};

537std::unique_ptr<FragmentedRangeTombstoneIterator>
538CompactionRangeDelAggregator::NewIterator(const Slice* lower_bound,
539 const Slice* upper_bound) {
540 InvalidateRangeDelMapPositions();
541 auto merging_iter = std::make_unique<TruncatedRangeDelMergingIter>(
542 icmp_, lower_bound, upper_bound, parent_iters_);
543
544 auto fragmented_tombstone_list =
545 std::make_shared<FragmentedRangeTombstoneList>(
546 std::move(merging_iter), *icmp_, true /* for_compaction */,
547 *snapshots_);
548
549 return std::make_unique<FragmentedRangeTombstoneIterator>(
550 fragmented_tombstone_list, *icmp_, kMaxSequenceNumber /* upper_bound */);
551}
