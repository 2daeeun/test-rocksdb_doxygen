101 std::vector<std::string> PrepareKeys(const std::string* orig_keys,
102 int number_of_keys) {
103 std::vector<std::string> user_keys;
104 if (ts_sz_ == 0) {
105 user_keys.assign(orig_keys, orig_keys + number_of_keys);
106 } else {
107 for (int i = 0; i < number_of_keys; i++) {
108 std::string key_with_ts;
109 AppendKeyWithMinTimestamp(&key_with_ts, orig_keys[i], ts_sz_);
110 user_keys.push_back(std::move(key_with_ts));
111 }
112 }

116 uint64_t MaxIndexSize() {
117 uint64_t max_key_size = 0;
118 for (int i = 0; i < kKeyNum; i++) {
119 // If UDT is enabled, the size of each key would be increased by a
120 // timestamp size.
121 max_key_size = std::max(
122 max_key_size, static_cast<uint64_t>(keys_without_ts[i].size()) +
123 ts_sz_ * sizeof(static_cast<unsigned char>(0)));
124 }
125 uint64_t max_index_size = kKeyNum * (max_key_size + 8 /*handle*/);
126 return max_index_size;
127 }

169 PartitionedFilterBlockReader* NewReader(
170 PartitionedFilterBlockBuilder* builder, PartitionedIndexBuilder* pib,
171 bool expect_empty = false) {
172 BlockHandle bh;
173 Status status;
174 Slice slice;
175 std::unique_ptr<const char[]> filter_data;
176 do {
177 status = builder->Finish(bh, &slice, &filter_data);
178 bh = Write(slice);
179 if (expect_empty) {
180 // Ensure most efficient "empty" filter is used
181 EXPECT_OK(status);
182 EXPECT_EQ(0, slice.size());
183 }
184 } while (status.IsIncomplete());
185
186 constexpr bool skip_filters = false;
187 constexpr uint64_t file_size = 12345;
188 constexpr int level = 0;
189 constexpr bool immortal_table = false;
190 table_.reset(new MockedBlockBasedTable(
191 new BlockBasedTable::Rep(ioptions_, env_options_, table_options_,
192 icomp_, skip_filters, file_size, level,
193 immortal_table,
194 user_defined_timestamps_persisted_),
195 pib));
196 BlockContents contents(slice);
197 CachableEntry<Block> block(
198 new Block(std::move(contents), 0 /* read_amp_bytes_per_bit */, nullptr),
199 nullptr /* cache */, nullptr /* cache_handle */, true /* own_value */);
200 auto reader =
201 new MyPartitionedFilterBlockReader(table_.get(), std::move(block));
202 return reader;
203 }

205 void VerifyReader(PartitionedFilterBlockBuilder* builder,
206 PartitionedIndexBuilder* pib, bool empty = false) {
207 std::unique_ptr<PartitionedFilterBlockReader> reader(
208 NewReader(builder, pib, empty));
209 // Querying added keys
210 std::vector<std::string> keys = PrepareKeys(keys_without_ts, kKeyNum);
211 for (const auto& key : keys) {
212 auto ikey = InternalKey(key, 0, ValueType::kTypeValue);
213 const Slice ikey_slice = Slice(*ikey.rep());
214 ASSERT_TRUE(reader->KeyMayMatch(
215 StripTimestampFromUserKey(key, ts_sz_), &ikey_slice,
216 /*get_context=*/nullptr,
217 /*lookup_context=*/nullptr, test::kReadOptionsNoIo));
218 }
219 {
220 // querying a key twice
221 auto ikey = InternalKey(keys[0], 0, ValueType::kTypeValue);
222 const Slice ikey_slice = Slice(*ikey.rep());
223 ASSERT_TRUE(reader->KeyMayMatch(
224 StripTimestampFromUserKey(keys[0], ts_sz_), &ikey_slice,
225 /*get_context=*/nullptr,
226 /*lookup_context=*/nullptr, test::kReadOptionsNoIo));
227 }
228 // querying missing keys
229 std::vector<std::string> missing_keys =
230 PrepareKeys(missing_keys_without_ts, kMissingKeyNum);
231 for (const auto& key : missing_keys) {
232 auto ikey = InternalKey(key, 0, ValueType::kTypeValue);
233 const Slice ikey_slice = Slice(*ikey.rep());
234 if (empty) {
235 ASSERT_TRUE(reader->KeyMayMatch(
236 StripTimestampFromUserKey(key, ts_sz_), &ikey_slice,
237 /*get_context=*/nullptr,
238 /*lookup_context=*/nullptr, test::kReadOptionsNoIo));
239 } else {
240 // assuming a good hash function
241 ASSERT_FALSE(reader->KeyMayMatch(
242 StripTimestampFromUserKey(key, ts_sz_), &ikey_slice,
243 /*get_context=*/nullptr,
244 /*lookup_context=*/nullptr, test::kReadOptionsNoIo));
245 }
246 }
247 }

311 void CutABlock(PartitionedIndexBuilder* builder,

313 // Assuming a block is cut, add an entry to the index

315 std::string(*InternalKey(user_key, 0, ValueType::kTypeValue).rep());
316 BlockHandle dont_care_block_handle(1, 1);

318 builder->AddIndexEntry(key, nullptr, dont_care_block_handle, &scratch);
319 }
320
321 void CutABlock(PartitionedIndexBuilder* builder, const std::string& user_key,
322 const std::string& next_user_key) {
323 // Assuming a block is cut, add an entry to the index
324 std::string key = *InternalKey(user_key, 0, ValueType::kTypeValue).rep();
325 std::string next_key =
326 *InternalKey(next_user_key, 0, ValueType::kTypeValue).rep();
327 BlockHandle dont_care_block_handle(1, 1);
328 Slice slice = Slice(next_key.data(), next_key.size());
329 std::string scratch;
330 builder->AddIndexEntry(key, &slice, dont_care_block_handle, &scratch);
331 }

333 int CountNumOfIndexPartitions(PartitionedIndexBuilder* builder) {
334 IndexBuilder::IndexBlocks dont_care_ib;

421// This reproduces the bug in format_version=3 that the seeking the prefix will
422// lead us to the partition before the one that has filter for the prefix.
423TEST_P(PartitionedFilterBlockTest, PrefixInWrongPartitionBug) {
424 // some small number to cause partition cuts
425 table_options_.metadata_block_size = 1;
426 std::unique_ptr<const SliceTransform> prefix_extractor(
427 ROCKSDB_NAMESPACE::NewFixedPrefixTransform(2));
428 std::unique_ptr<PartitionedIndexBuilder> pib(NewIndexBuilder());
429 std::unique_ptr<PartitionedFilterBlockBuilder> builder(
430 NewBuilder(pib.get(), prefix_extractor.get()));
431 // In the bug, searching for prefix "p3" on an index with format version 3,
432 // will give the key "p3" and the partition of the keys that are <= p3, i.e.,
433 // p2-keys, where the filter for prefix "p3" does not exist.
434 const std::string pkeys_without_ts[] = {"p1-key1", "p2-key2", "p3-key3",
435 "p4-key3", "p5-key3"};
436 std::vector<std::string> pkeys =
437 PrepareKeys(pkeys_without_ts, 5 /* number_of_keys */);
438 builder->Add(StripTimestampFromUserKey(pkeys[0], ts_sz_));
439 CutABlock(pib.get(), pkeys[0], pkeys[1]);
440 builder->Add(StripTimestampFromUserKey(pkeys[1], ts_sz_));
441 CutABlock(pib.get(), pkeys[1], pkeys[2]);
442 builder->Add(StripTimestampFromUserKey(pkeys[2], ts_sz_));
443 CutABlock(pib.get(), pkeys[2], pkeys[3]);
444 builder->Add(StripTimestampFromUserKey(pkeys[3], ts_sz_));
445 CutABlock(pib.get(), pkeys[3], pkeys[4]);
446 builder->Add(StripTimestampFromUserKey(pkeys[4], ts_sz_));
447 CutABlock(pib.get(), pkeys[4]);
448 std::unique_ptr<PartitionedFilterBlockReader> reader(
449 NewReader(builder.get(), pib.get()));
450 for (const auto& key : pkeys) {
451 auto prefix = prefix_extractor->Transform(key);
452 auto ikey = InternalKey(key, 0, ValueType::kTypeValue);
453 const Slice ikey_slice = Slice(*ikey.rep());
454 ASSERT_TRUE(reader->PrefixMayMatch(prefix, &ikey_slice,
455 /*get_context=*/nullptr,
456 /*lookup_context=*/nullptr,
457 ReadOptions()));
458 }
