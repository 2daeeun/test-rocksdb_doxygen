35std::string GenerateInternalKey(int primary_key, int secondary_key,
36 int padding_size, Random *rnd,
37 size_t ts_sz = 0) {
38 char buf[50];
39 char *p = &buf[0];
40 snprintf(buf, sizeof(buf), "%6d%4d", primary_key, secondary_key);
41 std::string k(p);
42 if (padding_size) {
43 k += rnd->RandomString(padding_size);
44 }
45 AppendInternalKeyFooter(&k, 0 /* seqno */, kTypeValue);
46 std::string key_with_ts;
47 if (ts_sz > 0) {
48 PadInternalKeyWithMinTimestamp(&key_with_ts, k, ts_sz);
49 return key_with_ts;
50 }
51
52 return k;
53}

55// Generate random key value pairs.
56// The generated key will be sorted. You can tune the parameters to generated
57// different kinds of test key/value pairs for different scenario.
58void GenerateRandomKVs(std::vector<std::string> *keys,
59 std::vector<std::string> *values, const int from,
60 const int len, const int step = 1,
61 const int padding_size = 0,
62 const int keys_share_prefix = 1, size_t ts_sz = 0) {
63 Random rnd(302);

67 // generating keys that shares the prefix

70 keys->emplace_back(GenerateInternalKey(i, j, padding_size, &rnd, ts_sz));
71
72 // 100 bytes values
73 values->emplace_back(rnd.RandomString(100));
74 }
75 }

304// A slow and accurate version of BlockReadAmpBitmap that simply store
305// all the marked ranges in a set.
306class BlockReadAmpBitmapSlowAndAccurate {
307 public:
308 void Mark(size_t start_offset, size_t end_offset) {
309 assert(end_offset >= start_offset);
310 marked_ranges_.emplace(end_offset, start_offset);
311 }
312
313 void ResetCheckSequence() { iter_valid_ = false; }
314
315 // Return true if any byte in this range was Marked
316 // This does linear search from the previous position. When calling
317 // multiple times, `offset` needs to be incremental to get correct results.
318 // Call ResetCheckSequence() to reset it.
319 bool IsPinMarked(size_t offset) {
320 if (iter_valid_) {
321 // Has existing iterator, try linear search from
322 // the iterator.
323 for (int i = 0; i < 64; i++) {
324 if (offset < iter_->second) {
325 return false;
326 }
327 if (offset <= iter_->first) {
328 return true;
329 }
330
331 iter_++;
332 if (iter_ == marked_ranges_.end()) {
333 iter_valid_ = false;
334 return false;
335 }
336 }
337 }
338 // Initial call or have linear searched too many times.
339 // Do binary search.
340 iter_ = marked_ranges_.lower_bound(
341 std::make_pair(offset, static_cast<size_t>(0)));
342 if (iter_ == marked_ranges_.end()) {
343 iter_valid_ = false;
344 return false;
345 }
346 iter_valid_ = true;
347 return offset <= iter_->first && offset >= iter_->second;
348 }
349
350 private:
351 std::set<std::pair<size_t, size_t>> marked_ranges_;
352 std::set<std::pair<size_t, size_t>>::iterator iter_;
353 bool iter_valid_ = false;
354};

431 stats->getTickerCount(READ_AMP_ESTIMATE_USEFUL_BYTES);
432 ASSERT_EQ(expected_estimate_useful, got_estimate_useful);
433 }
434 }

436 SyncPoint::GetInstance()->ClearAllCallBacks();

657 std::string first_key_to_persist_buf;
658 Slice first_internal_key = first_keys[i];
659 if (ts_sz > 0 && !shouldPersistUDT()) {
660 StripTimestampFromInternalKey(&first_key_to_persist_buf, first_keys[i],
661 ts_sz);
662 first_internal_key = first_key_to_persist_buf;
663 }
664 IndexValue entry(block_handles[i], first_internal_key);
665 std::string encoded_entry;
666 std::string delta_encoded_entry;
667 entry.EncodeTo(&encoded_entry, includeFirstKey(), nullptr);
668 if (useValueDeltaEncoding() && i > 0) {
669 entry.EncodeTo(&delta_encoded_entry, includeFirstKey(),
670 &last_encoded_handle);

692 IndexBlockIter *kNullIter = nullptr;
693 Statistics *kNullStats = nullptr;
694 // read contents of block sequentially
695 InternalIteratorBase<IndexValue> *iter = reader.NewIndexIterator(
696 options.comparator, kDisableGlobalSequenceNumber, kNullIter, kNullStats,
697 kTotalOrderSeek, includeFirstKey(), keyIncludesSeq(),
698 !useValueDeltaEncoding(), false /* block_contents_pinned */,
699 shouldPersistUDT());
700 iter->SeekToFirst();
701 for (int index = 0; index < num_records; ++index) {
702 ASSERT_TRUE(iter->Valid());
703
704 Slice k = iter->key();
705 IndexValue v = iter->value();
706

708 EXPECT_EQ(separators[index], k.ToString());
709 } else {
710 const Slice user_key = ExtractUserKey(separators[index]);
711 EXPECT_EQ(user_key, k);
712 }
713 EXPECT_EQ(block_handles[index].offset(), v.handle.offset());
714 EXPECT_EQ(block_handles[index].size(), v.handle.size());
715 EXPECT_EQ(includeFirstKey() ? first_keys[index] : "",
716 v.first_internal_key.ToString());
717
718 iter->Next();
719 }
720 delete iter;
721

723 iter = reader.NewIndexIterator(
724 options.comparator, kDisableGlobalSequenceNumber, kNullIter, kNullStats,
725 kTotalOrderSeek, includeFirstKey(), keyIncludesSeq(),
726 !useValueDeltaEncoding(), false /* block_contents_pinned */,
727 shouldPersistUDT());
