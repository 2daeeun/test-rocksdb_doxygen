27void PopulateHistogram(Histogram& histogram, uint64_t low, uint64_t high,
28 uint64_t loop = 1) {
29 Random rnd(test::RandomSeed());
30 for (; loop > 0; loop--) {
31 for (uint64_t i = low; i <= high; i++) {
32 histogram.Add(i);
33 // sleep a random microseconds [0-10)
34 clock->SleepForMicroseconds(rnd.Uniform(10));
35 }
36 }
37 // make sure each data population at least take some time
38 clock->SleepForMicroseconds(1);
39}

41void BasicOperation(Histogram& histogram) {
42 PopulateHistogram(histogram, 1, 110, 10); // fill up to bucket [70, 110)
43
44 HistogramData data;
45 histogram.Data(&data);
46
47 ASSERT_LE(fabs(histogram.Percentile(100.0) - 110.0), kIota);
48 ASSERT_LE(fabs(data.percentile99 - 108.9), kIota); // 99 * 110 / 100
49 ASSERT_LE(fabs(data.percentile95 - 104.5), kIota); // 95 * 110 / 100
50 ASSERT_LE(fabs(data.median - 55.0), kIota); // 50 * 110 / 100
51 ASSERT_EQ(data.average, 55.5); // (1 + 110) / 2
52}

54void MergeHistogram(Histogram& histogram, Histogram& other) {
55 PopulateHistogram(histogram, 1, 100);
56 PopulateHistogram(other, 101, 250);

64 ASSERT_LE(fabs(data.percentile95 - 237.5), kIota); // 95 * 250 / 100
65 ASSERT_LE(fabs(data.median - 125.0), kIota); // 50 * 250 / 100
66 ASSERT_EQ(data.average, 125.5); // (1 + 250) / 2
67}

137TEST_F(HistogramTest, HistogramWindowingExpire) {
138 uint64_t num_windows = 3;
139 int micros_per_window = 1000000;
140 uint64_t min_num_per_window = 0;
141
142 HistogramWindowingImpl histogramWindowing(num_windows, micros_per_window,
143 min_num_per_window);
144 histogramWindowing.TEST_UpdateClock(clock);
145 PopulateHistogram(histogramWindowing, 1, 1, 100);
146 clock->SleepForMicroseconds(micros_per_window);
147 ASSERT_EQ(histogramWindowing.num(), 100);
148 ASSERT_EQ(histogramWindowing.min(), 1);
149 ASSERT_EQ(histogramWindowing.max(), 1);
150 ASSERT_EQ(histogramWindowing.Average(), 1.0);
151 ASSERT_EQ(histogramWindowing.StandardDeviation(), 0.0);
152
153 PopulateHistogram(histogramWindowing, 2, 2, 100);
154 clock->SleepForMicroseconds(micros_per_window);
155 ASSERT_EQ(histogramWindowing.num(), 200);
156 ASSERT_EQ(histogramWindowing.min(), 1);
157 ASSERT_EQ(histogramWindowing.max(), 2);
158 ASSERT_EQ(histogramWindowing.Average(), 1.5);
159 ASSERT_GT(histogramWindowing.StandardDeviation(), 0.0);
160
161 PopulateHistogram(histogramWindowing, 3, 3, 100);
162 clock->SleepForMicroseconds(micros_per_window);
163 ASSERT_EQ(histogramWindowing.num(), 300);
164 ASSERT_EQ(histogramWindowing.min(), 1);
165 ASSERT_EQ(histogramWindowing.max(), 3);
166 ASSERT_EQ(histogramWindowing.Average(), 2.0);
167 ASSERT_GT(histogramWindowing.StandardDeviation(), 0.0);
168
169 // dropping oldest window with value 1, remaining 2 ~ 4
170 PopulateHistogram(histogramWindowing, 4, 4, 100);
171 clock->SleepForMicroseconds(micros_per_window);
172 ASSERT_EQ(histogramWindowing.num(), 300);
173 ASSERT_EQ(histogramWindowing.min(), 2);
174 ASSERT_EQ(histogramWindowing.max(), 4);
175 ASSERT_EQ(histogramWindowing.Average(), 3.0);
176 ASSERT_GT(histogramWindowing.StandardDeviation(), 0.0);
177
178 // dropping oldest window with value 2, remaining 3 ~ 5
179 PopulateHistogram(histogramWindowing, 5, 5, 100);
180 clock->SleepForMicroseconds(micros_per_window);
181 ASSERT_EQ(histogramWindowing.num(), 300);
182 ASSERT_EQ(histogramWindowing.min(), 3);
183 ASSERT_EQ(histogramWindowing.max(), 5);
184 ASSERT_EQ(histogramWindowing.Average(), 4.0);
185 ASSERT_GT(histogramWindowing.StandardDeviation(), 0.0);
186}

188TEST_F(HistogramTest, HistogramWindowingMerge) {

190 int micros_per_window = 1000000;
191 uint64_t min_num_per_window = 0;
192
193 HistogramWindowingImpl histogramWindowing(num_windows, micros_per_window,
194 min_num_per_window);
195 HistogramWindowingImpl otherWindowing(num_windows, micros_per_window,

199

203

205 PopulateHistogram(otherWindowing, 2, 2, 100);
206 clock->SleepForMicroseconds(micros_per_window);
207
208 PopulateHistogram(histogramWindowing, 3, 3, 100);
209 PopulateHistogram(otherWindowing, 3, 3, 100);
210 clock->SleepForMicroseconds(micros_per_window);
211
212 histogramWindowing.Merge(otherWindowing);
213 ASSERT_EQ(histogramWindowing.num(), 600);
214 ASSERT_EQ(histogramWindowing.min(), 1);
215 ASSERT_EQ(histogramWindowing.max(), 3);
216 ASSERT_EQ(histogramWindowing.Average(), 2.0);
217
218 // dropping oldest window with value 1, remaining 2 ~ 4
219 PopulateHistogram(histogramWindowing, 4, 4, 100);
220 clock->SleepForMicroseconds(micros_per_window);
221 ASSERT_EQ(histogramWindowing.num(), 500);
222 ASSERT_EQ(histogramWindowing.min(), 2);
223 ASSERT_EQ(histogramWindowing.max(), 4);
224
225 // dropping oldest window with value 2, remaining 3 ~ 5
