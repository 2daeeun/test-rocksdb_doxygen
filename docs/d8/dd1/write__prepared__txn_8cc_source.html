42void WritePreparedTxn::MultiGet(const ReadOptions& _read_options,
43 ColumnFamilyHandle* column_family,
44 const size_t num_keys, const Slice* keys,
45 PinnableSlice* values, Status* statuses,
46 const bool sorted_input) {
47 if (_read_options.io_activity != Env::IOActivity::kUnknown &&
48 _read_options.io_activity != Env::IOActivity::kMultiGet) {
49 Status s = Status::InvalidArgument(
50 "Can only call MultiGet with `ReadOptions::io_activity` is "
51 "`Env::IOActivity::kUnknown` or `Env::IOActivity::kMultiGet`");
52
53 for (size_t i = 0; i < num_keys; ++i) {
54 if (statuses[i].ok()) {
55 statuses[i] = s;
56 }
57 }
58 return;
59 }
60 ReadOptions read_options(_read_options);
61 if (read_options.io_activity == Env::IOActivity::kUnknown) {
62 read_options.io_activity = Env::IOActivity::kMultiGet;
63 }
64
65 SequenceNumber min_uncommitted, snap_seq;
66 const SnapshotBackup backed_by_snapshot = wpt_db_->AssignMinMaxSeqs(
67 read_options.snapshot, &min_uncommitted, &snap_seq);
68 WritePreparedTxnReadCallback callback(wpt_db_, snap_seq, min_uncommitted,
69 backed_by_snapshot);
70 write_batch_.MultiGetFromBatchAndDB(db_, read_options, column_family,
71 num_keys, keys, values, statuses,
72 sorted_input, &callback);
73 if (UNLIKELY(!callback.valid() ||
74 !wpt_db_->ValidateSnapshot(snap_seq, backed_by_snapshot))) {
75 wpt_db_->WPRecordTick(TXN_GET_TRY_AGAIN);
76 for (size_t i = 0; i < num_keys; i++) {
77 statuses[i] = Status::TryAgain();
78 }
79 }
80}

99Status WritePreparedTxn::GetImpl(const ReadOptions& options,
100 ColumnFamilyHandle* column_family,
101 const Slice& key,
102 PinnableSlice* pinnable_val) {
103 SequenceNumber min_uncommitted, snap_seq;
104 const SnapshotBackup backed_by_snapshot =
105 wpt_db_->AssignMinMaxSeqs(options.snapshot, &min_uncommitted, &snap_seq);
106 WritePreparedTxnReadCallback callback(wpt_db_, snap_seq, min_uncommitted,
107 backed_by_snapshot);
108 Status res = write_batch_.GetFromBatchAndDB(db_, options, column_family, key,
109 pinnable_val, &callback);
110 const bool callback_valid =
111 callback.valid(); // NOTE: validity of callback must always be checked
112 // before it is destructed
113 if (res.ok()) {
114 if (!LIKELY(callback_valid &&
115 wpt_db_->ValidateSnapshot(callback.max_visible_seq(),
116 backed_by_snapshot))) {
117 wpt_db_->WPRecordTick(TXN_GET_TRY_AGAIN);
118 res = Status::TryAgain();
119 }
120 }
121
122 return res;
123}

155Status WritePreparedTxn::PrepareInternal() {
156 WriteOptions write_options = write_options_;
157 write_options.disableWAL = false;
158 const bool WRITE_AFTER_COMMIT = true;
159 const bool kFirstPrepareBatch = true;
160 auto s = WriteBatchInternal::MarkEndPrepare(GetWriteBatch()->GetWriteBatch(),
161 name_, !WRITE_AFTER_COMMIT);
162 assert(s.ok());
163 // For each duplicate key we account for a new sub-batch
164 prepare_batch_cnt_ = GetWriteBatch()->SubBatchCnt();
165 // Having AddPrepared in the PreReleaseCallback allows in-order addition of
166 // prepared entries to PreparedHeap and hence enables an optimization. Refer
167 // to SmallestUnCommittedSeq for more details.
168 AddPreparedCallback add_prepared_callback(
169 wpt_db_, db_impl_, prepare_batch_cnt_,
170 db_impl_->immutable_db_options().two_write_queues, kFirstPrepareBatch);
171 const bool DISABLE_MEMTABLE = true;
172 uint64_t seq_used = kMaxSequenceNumber;
173 s = db_impl_->WriteImpl(write_options, GetWriteBatch()->GetWriteBatch(),
174 /*callback*/ nullptr, /*user_write_cb=*/nullptr,
175 &log_number_, /*log ref*/ 0, !DISABLE_MEMTABLE,
176 &seq_used, prepare_batch_cnt_,
177 &add_prepared_callback);
178 assert(!s.ok() || seq_used != kMaxSequenceNumber);
179 auto prepare_seq = seq_used;
180 SetId(prepare_seq);
181 return s;
182}

195Status WritePreparedTxn::CommitInternal() {
196 ROCKS_LOG_DETAILS(db_impl_->immutable_db_options().info_log,
197 "CommitInternal prepare_seq: %" PRIu64, GetID());
198 // We take the commit-time batch and append the Commit marker.
199 // The Memtable will ignore the Commit marker in non-recovery mode
200 WriteBatch* working_batch = GetCommitTimeWriteBatch();
201 const bool empty = working_batch->Count() == 0;
202 auto s = WriteBatchInternal::MarkCommit(working_batch, name_);
203 assert(s.ok());
204
205 const bool for_recovery = use_only_the_last_commit_time_batch_for_recovery_;
206 if (!empty) {
207 // When not writing to memtable, we can still cache the latest write batch.
208 // The cached batch will be written to memtable in WriteRecoverableState
209 // during FlushMemTable
210 if (for_recovery) {
211 WriteBatchInternal::SetAsLatestPersistentState(working_batch);
212 } else {
213 return Status::InvalidArgument(
214 "Commit-time-batch can only be used if "
215 "use_only_the_last_commit_time_batch_for_recovery is true");
216 }
217 }
218
219 auto prepare_seq = GetId();
220 const bool includes_data = !empty && !for_recovery;
221 assert(prepare_batch_cnt_);
222 size_t commit_batch_cnt = 0;
223 if (UNLIKELY(includes_data)) {
224 ROCKS_LOG_WARN(db_impl_->immutable_db_options().info_log,
225 "Duplicate key overhead");
226 SubBatchCounter counter(*wpt_db_->GetCFComparatorMap());
227 s = working_batch->Iterate(&counter);
228 assert(s.ok());
229 commit_batch_cnt = counter.BatchCount();
230 }
231 const bool disable_memtable = !includes_data;
232 const bool do_one_write =
233 !db_impl_->immutable_db_options().two_write_queues || disable_memtable;
234 WritePreparedCommitEntryPreReleaseCallback update_commit_map(
235 wpt_db_, db_impl_, prepare_seq, prepare_batch_cnt_, commit_batch_cnt);
236 // This is to call AddPrepared on CommitTimeWriteBatch
237 const bool kFirstPrepareBatch = true;
238 AddPreparedCallback add_prepared_callback(
239 wpt_db_, db_impl_, commit_batch_cnt,
240 db_impl_->immutable_db_options().two_write_queues, !kFirstPrepareBatch);
241 PreReleaseCallback* pre_release_callback;
242 if (do_one_write) {
243 pre_release_callback = &update_commit_map;
244 } else {
245 pre_release_callback = &add_prepared_callback;
246 }
247 uint64_t seq_used = kMaxSequenceNumber;
248 // Since the prepared batch is directly written to memtable, there is already
249 // a connection between the memtable and its WAL, so there is no need to
250 // redundantly reference the log that contains the prepared data.
251 const uint64_t zero_log_number = 0ull;
252 size_t batch_cnt = UNLIKELY(commit_batch_cnt) ? commit_batch_cnt : 1;
253 // If `two_write_queues && includes_data`, then `do_one_write` is false. The
254 // following `WriteImpl` will insert the data of the commit-time-batch into
255 // the database before updating the commit cache. Therefore, the data of the
256 // commmit-time-batch is considered uncommitted. Furthermore, since data of
257 // the commit-time-batch are not locked, it is possible for two uncommitted
258 // versions of the same key to co-exist for a (short) period of time until
259 // the commit cache is updated by the second write. If the two uncommitted
260 // keys are compacted to the bottommost level in the meantime, it is possible
261 // that compaction iterator will zero out the sequence numbers of both, thus
262 // violating the invariant that an SST does not have two identical internal
263 // keys. To prevent this situation, we should allow the usage of
264 // commit-time-batch only if the user sets
265 // TransactionOptions::use_only_the_last_commit_time_batch_for_recovery to
266 // true. See the comments about GetCommitTimeWriteBatch() in
267 // include/rocksdb/utilities/transaction.h.
268 s = db_impl_->WriteImpl(write_options_, working_batch, nullptr,
269 /*user_write_cb=*/nullptr, nullptr, zero_log_number,
270 disable_memtable, &seq_used, batch_cnt,
271 pre_release_callback);
272 assert(!s.ok() || seq_used != kMaxSequenceNumber);
273 const SequenceNumber commit_batch_seq = seq_used;
274 if (LIKELY(do_one_write || !s.ok())) {
275 if (UNLIKELY(!db_impl_->immutable_db_options().two_write_queues &&
276 s.ok())) {
277 // Note: RemovePrepared should be called after WriteImpl that publishsed
278 // the seq. Otherwise SmallestUnCommittedSeq optimization breaks.
279 wpt_db_->RemovePrepared(prepare_seq, prepare_batch_cnt_);
280 } // else RemovePrepared is called from within PreReleaseCallback
281 if (UNLIKELY(!do_one_write)) {
282 assert(!s.ok());
283 // Cleanup the prepared entry we added with add_prepared_callback
284 wpt_db_->RemovePrepared(commit_batch_seq, commit_batch_cnt);
285 }
286 return s;
287 } // else do the 2nd write to publish seq
288 // Note: the 2nd write comes with a performance penality. So if we have too
289 // many of commits accompanied with ComitTimeWriteBatch and yet we cannot
290 // enable use_only_the_last_commit_time_batch_for_recovery_ optimization,
291 // two_write_queues should be disabled to avoid many additional writes here.
292 const size_t kZeroData = 0;
293 // Update commit map only from the 2nd queue
294 WritePreparedCommitEntryPreReleaseCallback update_commit_map_with_aux_batch(
295 wpt_db_, db_impl_, prepare_seq, prepare_batch_cnt_, kZeroData,
296 commit_batch_seq, commit_batch_cnt);
297 WriteBatch empty_batch;
298 s = empty_batch.PutLogData(Slice());
299 assert(s.ok());
300 // In the absence of Prepare markers, use Noop as a batch separator
301 s = WriteBatchInternal::InsertNoop(&empty_batch);
302 assert(s.ok());
303 const bool DISABLE_MEMTABLE = true;
304 const size_t ONE_BATCH = 1;
305 const uint64_t NO_REF_LOG = 0;
306 s = db_impl_->WriteImpl(write_options_, &empty_batch, nullptr,
307 /*user_write_cb=*/nullptr, nullptr, NO_REF_LOG,
308 DISABLE_MEMTABLE, &seq_used, ONE_BATCH,
309 &update_commit_map_with_aux_batch);
310 assert(!s.ok() || seq_used != kMaxSequenceNumber);
311 return s;
312}

314Status WritePreparedTxn::RollbackInternal() {
315 ROCKS_LOG_WARN(db_impl_->immutable_db_options().info_log,
316 "RollbackInternal prepare_seq: %" PRIu64, GetId());
317
318 assert(db_impl_);
319 assert(wpt_db_);
320
321 WriteBatch rollback_batch(0 /* reserved_bytes */, 0 /* max_bytes */,
322 write_options_.protection_bytes_per_key,
323 0 /* default_cf_ts_sz */);
324 assert(GetId() != kMaxSequenceNumber);
325 assert(GetId() > 0);
326 auto cf_map_shared_ptr = wpt_db_->GetCFHandleMap();
327 auto cf_comp_map_shared_ptr = wpt_db_->GetCFComparatorMap();
328 auto read_at_seq = kMaxSequenceNumber;
329 // TODO: plumb Env::IOActivity, Env::IOPriority
330 ReadOptions roptions;
331 // to prevent callback's seq to be overrriden inside DBImpk::Get
332 roptions.snapshot = wpt_db_->GetMaxSnapshot();
333 struct RollbackWriteBatchBuilder : public WriteBatch::Handler {
334 DBImpl* const db_;
335 WritePreparedTxnDB* const wpt_db_;
336 WritePreparedTxnReadCallback callback_;
337 WriteBatch* rollback_batch_;
338 std::map<uint32_t, const Comparator*>& comparators_;
339 std::map<uint32_t, ColumnFamilyHandle*>& handles_;
340 using CFKeys = std::set<Slice, SetComparator>;
341 std::map<uint32_t, CFKeys> keys_;
342 bool rollback_merge_operands_;
343 ReadOptions roptions_;
344
345 RollbackWriteBatchBuilder(
346 DBImpl* db, WritePreparedTxnDB* wpt_db, SequenceNumber snap_seq,
347 WriteBatch* dst_batch,
348 std::map<uint32_t, const Comparator*>& comparators,
349 std::map<uint32_t, ColumnFamilyHandle*>& handles,
350 bool rollback_merge_operands, const ReadOptions& _roptions)
351 : db_(db),
352 wpt_db_(wpt_db),
353 callback_(wpt_db, snap_seq), // disable min_uncommitted optimization
354 rollback_batch_(dst_batch),
355 comparators_(comparators),
356 handles_(handles),
357 rollback_merge_operands_(rollback_merge_operands),
358 roptions_(_roptions) {}
359
360 Status Rollback(uint32_t cf, const Slice& key) {
361 Status s;
362 CFKeys& cf_keys = keys_[cf];
363 if (cf_keys.size() == 0) { // just inserted
364 auto cmp = comparators_[cf];
365 keys_[cf] = CFKeys(SetComparator(cmp));
366 }
367 auto it = cf_keys.insert(key);
368 // second is false if a element already existed.
369 if (it.second == false) {
370 return s;
371 }
372
373 PinnableSlice pinnable_val;
374 bool not_used;
375 auto cf_handle = handles_[cf];
376 DBImpl::GetImplOptions get_impl_options;
377 get_impl_options.column_family = cf_handle;
378 get_impl_options.value = &pinnable_val;
379 get_impl_options.value_found = &not_used;
380 get_impl_options.callback = &callback_;
381 s = db_->GetImpl(roptions_, key, get_impl_options);
382 assert(s.ok() || s.IsNotFound());
383 if (s.ok()) {
384 s = rollback_batch_->Put(cf_handle, key, pinnable_val);
385 assert(s.ok());
386 } else if (s.IsNotFound()) {
387 // There has been no readable value before txn. By adding a delete we
388 // make sure that there will be none afterwards either.
389 if (wpt_db_->ShouldRollbackWithSingleDelete(cf_handle, key)) {
390 s = rollback_batch_->SingleDelete(cf_handle, key);
391 } else {
392 s = rollback_batch_->Delete(cf_handle, key);
393 }
394 assert(s.ok());
395 } else {
396 // Unexpected status. Return it to the user.
397 }
398 return s;
399 }
400
401 Status PutCF(uint32_t cf, const Slice& key, const Slice& /*val*/) override {
402 return Rollback(cf, key);
403 }
404
405 Status DeleteCF(uint32_t cf, const Slice& key) override {
406 return Rollback(cf, key);
407 }
408
409 Status SingleDeleteCF(uint32_t cf, const Slice& key) override {
410 return Rollback(cf, key);
411 }
412
413 Status MergeCF(uint32_t cf, const Slice& key,
414 const Slice& /*val*/) override {
415 if (rollback_merge_operands_) {
416 return Rollback(cf, key);
417 } else {
418 return Status::OK();
419 }
420 }
421
422 Status MarkNoop(bool) override { return Status::OK(); }
423 Status MarkBeginPrepare(bool) override { return Status::OK(); }
424 Status MarkEndPrepare(const Slice&) override { return Status::OK(); }
425 Status MarkCommit(const Slice&) override { return Status::OK(); }
426 Status MarkRollback(const Slice&) override {
427 return Status::InvalidArgument();
428 }
429
430 protected:
431 Handler::OptionState WriteAfterCommit() const override {
432 return Handler::OptionState::kDisabled;
433 }
434 } rollback_handler(db_impl_, wpt_db_, read_at_seq, &rollback_batch,
435 *cf_comp_map_shared_ptr.get(), *cf_map_shared_ptr.get(),
436 wpt_db_->txn_db_options_.rollback_merge_operands,
437 roptions);
438 auto s = GetWriteBatch()->GetWriteBatch()->Iterate(&rollback_handler);
439 if (!s.ok()) {
440 return s;
441 }
442 // The Rollback marker will be used as a batch separator
443 s = WriteBatchInternal::MarkRollback(&rollback_batch, name_);
444 assert(s.ok());
445 bool do_one_write = !db_impl_->immutable_db_options().two_write_queues;
446 const bool DISABLE_MEMTABLE = true;
447 const uint64_t NO_REF_LOG = 0;
448 uint64_t seq_used = kMaxSequenceNumber;
449 const size_t ONE_BATCH = 1;
450 const bool kFirstPrepareBatch = true;
451 // We commit the rolled back prepared batches. Although this is
452 // counter-intuitive, i) it is safe to do so, since the prepared batches are
453 // already canceled out by the rollback batch, ii) adding the commit entry to
454 // CommitCache will allow us to benefit from the existing mechanism in
455 // CommitCache that keeps an entry evicted due to max advance and yet overlaps
456 // with a live snapshot around so that the live snapshot properly skips the
457 // entry even if its prepare seq is lower than max_evicted_seq_.
458 AddPreparedCallback add_prepared_callback(
459 wpt_db_, db_impl_, ONE_BATCH,
460 db_impl_->immutable_db_options().two_write_queues, !kFirstPrepareBatch);
461 WritePreparedCommitEntryPreReleaseCallback update_commit_map(
462 wpt_db_, db_impl_, GetId(), prepare_batch_cnt_, ONE_BATCH);
463 PreReleaseCallback* pre_release_callback;
464 if (do_one_write) {
465 pre_release_callback = &update_commit_map;
466 } else {
467 pre_release_callback = &add_prepared_callback;
468 }
469 // Note: the rollback batch does not need AddPrepared since it is written to
470 // DB in one shot. min_uncommitted still works since it requires capturing
471 // data that is written to DB but not yet committed, while
472 // the rollback batch commits with PreReleaseCallback.
473 s = db_impl_->WriteImpl(write_options_, &rollback_batch, nullptr,
474 /*user_write_cb=*/nullptr, nullptr, NO_REF_LOG,
475 !DISABLE_MEMTABLE, &seq_used, ONE_BATCH,
476 pre_release_callback);
477 assert(!s.ok() || seq_used != kMaxSequenceNumber);
478 if (!s.ok()) {
479 return s;
480 }
481 if (do_one_write) {
482 assert(!db_impl_->immutable_db_options().two_write_queues);
483 wpt_db_->RemovePrepared(GetId(), prepare_batch_cnt_);
484 return s;
485 } // else do the 2nd write for commit
486 uint64_t rollback_seq = seq_used;
487 ROCKS_LOG_DETAILS(db_impl_->immutable_db_options().info_log,
488 "RollbackInternal 2nd write rollback_seq: %" PRIu64,
489 rollback_seq);
490 // Commit the batch by writing an empty batch to the queue that will release
491 // the commit sequence number to readers.
492 WritePreparedRollbackPreReleaseCallback update_commit_map_with_prepare(
493 wpt_db_, db_impl_, GetId(), rollback_seq, prepare_batch_cnt_);
494 WriteBatch empty_batch;
495 s = empty_batch.PutLogData(Slice());
496 assert(s.ok());
497 // In the absence of Prepare markers, use Noop as a batch separator
498 s = WriteBatchInternal::InsertNoop(&empty_batch);
499 assert(s.ok());
500 s = db_impl_->WriteImpl(write_options_, &empty_batch, nullptr,
501 /*user_write_cb=*/nullptr, nullptr, NO_REF_LOG,
502 DISABLE_MEMTABLE, &seq_used, ONE_BATCH,
503 &update_commit_map_with_prepare);
504 assert(!s.ok() || seq_used != kMaxSequenceNumber);
505 ROCKS_LOG_DETAILS(db_impl_->immutable_db_options().info_log,
506 "RollbackInternal (status=%s) commit: %" PRIu64,
507 s.ToString().c_str(), GetId());
508 // TODO(lth): For WriteUnPrepared that rollback is called frequently,
509 // RemovePrepared could be moved to the callback to reduce lock contention.
510 if (s.ok()) {
511 wpt_db_->RemovePrepared(GetId(), prepare_batch_cnt_);
512 }
513 // Note: RemovePrepared for prepared batch is called from within
514 // PreReleaseCallback
515 wpt_db_->RemovePrepared(rollback_seq, ONE_BATCH);
516
517 return s;
518}

520Status WritePreparedTxn::ValidateSnapshot(ColumnFamilyHandle* column_family,
521 const Slice& key,
522 SequenceNumber* tracked_at_seq) {
523 assert(snapshot_);
524
525 SequenceNumber min_uncommitted =
526 static_cast_with_check<const SnapshotImpl>(snapshot_.get())
527 ->min_uncommitted_;
528 SequenceNumber snap_seq = snapshot_->GetSequenceNumber();
529 // tracked_at_seq is either max or the last snapshot with which this key was
530 // trackeed so there is no need to apply the IsInSnapshot to this comparison
531 // here as tracked_at_seq is not a prepare seq.
532 if (*tracked_at_seq <= snap_seq) {
533 // If the key has been previous validated at a sequence number earlier
534 // than the curent snapshot's sequence number, we already know it has not
535 // been modified.
536 return Status::OK();
537 }
538
539 *tracked_at_seq = snap_seq;
540
541 ColumnFamilyHandle* cfh =
542 column_family ? column_family : db_impl_->DefaultColumnFamily();
543
544 WritePreparedTxnReadCallback snap_checker(wpt_db_, snap_seq, min_uncommitted,
545 kBackedByDBSnapshot);
546 // TODO(yanqin): support user-defined timestamp
547 return TransactionUtil::CheckKeyForConflicts(
548 db_impl_, cfh, key.ToString(), snap_seq, /*ts=*/nullptr,
549 false /* cache_only */, &snap_checker, min_uncommitted,
550 txn_db_impl_->GetTxnDBOptions().enable_udt_validation);
551}
