55 // REQUIRES: Finish(), Abandon() have not been called

57
58 // Indicate that the contents of this builder should be abandoned. Stops
59 // using the file passed to the constructor after this function returns.
60 // If the caller is not going to call Finish(), it must call Abandon()
61 // before destroying this builder.
62 // REQUIRES: Finish(), Abandon() have not been called
63 void Abandon() override;
64
65 // Number of calls to Add() so far.

68 // Size of the file generated so far. If invoked after a successful
