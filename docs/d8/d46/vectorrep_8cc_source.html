107void VectorRep::Insert(KeyHandle handle) {
108 auto* key = static_cast<char*>(handle);
109 WriteLock l(&rwlock_);
110 assert(!immutable_);
111 bucket_->push_back(key);
112}

114// Returns true iff an entry that compares equal to key is in the collection.
115bool VectorRep::Contains(const char* key) const {
116 ReadLock l(&rwlock_);
117 return std::find(bucket_->begin(), bucket_->end(), key) != bucket_->end();
118}

125size_t VectorRep::ApproximateMemoryUsage() {

151void VectorRep::Iterator::DoSort() const {
152 // vrep is non-null means that we are working on an immutable memtable
153 if (!sorted_ && vrep_ != nullptr) {
154 WriteLock l(&vrep_->rwlock_);
155 if (!vrep_->sorted_) {
156 std::sort(bucket_->begin(), bucket_->end(),
157 stl_wrappers::Compare(compare_));
158 cit_ = bucket_->begin();
159 vrep_->sorted_ = true;
160 }
161 sorted_ = true;
162 }
163 if (!sorted_) {
164 std::sort(bucket_->begin(), bucket_->end(),
165 stl_wrappers::Compare(compare_));
166 cit_ = bucket_->begin();
167 sorted_ = true;
168 }
169 assert(sorted_);
170 assert(vrep_ == nullptr || vrep_->sorted_);
171}

173// Returns true iff the iterator is positioned at a valid node.
174bool VectorRep::Iterator::Valid() const {
175 DoSort();
176 return cit_ != bucket_->end();
177}
178
179// Returns the key at the current position.
180// REQUIRES: Valid()
181const char* VectorRep::Iterator::key() const {
182 assert(sorted_);
183 return *cit_;
184}
185
186// Advances to the next position.
187// REQUIRES: Valid()
188void VectorRep::Iterator::Next() {
189 assert(sorted_);
190 if (cit_ == bucket_->end()) {
191 return;
192 }
193 ++cit_;
194}
195
196// Advances to the previous position.
197// REQUIRES: Valid()
198void VectorRep::Iterator::Prev() {
199 assert(sorted_);
200 if (cit_ == bucket_->begin()) {
201 // If you try to go back from the first element, the iterator should be
202 // invalidated. So we set it to past-the-end. This means that you can
203 // treat the container circularly.
204 cit_ = bucket_->end();
205 } else {
206 --cit_;
207 }
208}

210// Advance to the first entry with a key >= target
211void VectorRep::Iterator::Seek(const Slice& user_key,
212 const char* memtable_key) {
213 DoSort();
214 // Do binary search to find first value not less than the target
215 const char* encoded_key =
216 (memtable_key != nullptr) ? memtable_key : EncodeKey(&tmp_, user_key);
217 cit_ = std::equal_range(bucket_->begin(), bucket_->end(), encoded_key,
218 [this](const char* a, const char* b) {
219 return compare_(a, b) < 0;
220 })
221 .first;
222}
223
224// Advance to the first entry with a key <= target
225void VectorRep::Iterator::SeekForPrev(const Slice& /*user_key*/,
226 const char* /*memtable_key*/) {
227 assert(false);
228}

230// Position at the first entry in collection.
231// Final state of iterator is Valid() iff collection is not empty.
232void VectorRep::Iterator::SeekToFirst() {
233 DoSort();
234 cit_ = bucket_->begin();
235}
236
237// Position at the last entry in collection.
238// Final state of iterator is Valid() iff collection is not empty.
239void VectorRep::Iterator::SeekToLast() {
240 DoSort();
241 cit_ = bucket_->end();
242 if (bucket_->size() != 0) {
243 --cit_;
244 }
245}

247void VectorRep::Get(const LookupKey& k, void* callback_args,
248 bool (*callback_func)(void* arg, const char* entry)) {
249 rwlock_.ReadLock();
250 VectorRep* vector_rep;
251 std::shared_ptr<Bucket> bucket;
252 if (immutable_) {
253 vector_rep = this;
254 } else {
255 vector_rep = nullptr;
256 bucket.reset(new Bucket(*bucket_)); // make a copy
257 }
258 VectorRep::Iterator iter(vector_rep, immutable_ ? bucket_ : bucket, compare_);
259 rwlock_.ReadUnlock();
260
261 for (iter.Seek(k.user_key(), k.memtable_key().data());
262 iter.Valid() && callback_func(callback_args, iter.key()); iter.Next()) {
263 }
264}

266MemTableRep::Iterator* VectorRep::GetIterator(Arena* arena) {
267 char* mem = nullptr;
268 if (arena != nullptr) {
269 mem = arena->AllocateAligned(sizeof(Iterator));
270 }
271 ReadLock l(&rwlock_);
272 // Do not sort here. The sorting would be done the first time
273 // a Seek is performed on the iterator.
274 if (immutable_) {
275 if (arena == nullptr) {
276 return new Iterator(this, bucket_, compare_);
277 } else {
278 return new (mem) Iterator(this, bucket_, compare_);
279 }
280 } else {
281 std::shared_ptr<Bucket> tmp;
282 tmp.reset(new Bucket(*bucket_)); // make a copy
283 if (arena == nullptr) {
284 return new Iterator(nullptr, tmp, compare_);
285 } else {
286 return new (mem) Iterator(nullptr, tmp, compare_);
287 }
288 }
