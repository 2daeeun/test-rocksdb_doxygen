16// Table Properties that are specific to tables created by SstFileWriter.
17struct ExternalSstFilePropertyNames {
18 // value of this property is a fixed uint32 number.
19 static const std::string kVersion;
20 // value of this property is a fixed uint64 number.
21 static const std::string kGlobalSeqno;
22};
23
24// PropertiesCollector used to add properties specific to tables
25// generated by SstFileWriter
26class SstFileWriterPropertiesCollector : public InternalTblPropColl {
27 public:
28 explicit SstFileWriterPropertiesCollector(int32_t version,
29 SequenceNumber global_seqno)
30 : version_(version), global_seqno_(global_seqno) {}
31
32 Status InternalAdd(const Slice& /*key*/, const Slice& /*value*/,
33 uint64_t /*file_size*/) override {
34 // Intentionally left blank. Have no interest in collecting stats for
35 // individual key/value pairs.
36 return Status::OK();
37 }
38
39 void BlockAdd(uint64_t /* block_uncomp_bytes */,
40 uint64_t /* block_compressed_bytes_fast */,
41 uint64_t /* block_compressed_bytes_slow */) override {
42 // Intentionally left blank. No interest in collecting stats for
43 // blocks.
44 return;
45 }
46
47 Status Finish(UserCollectedProperties* properties) override {
48 // File version
49 std::string version_val;
50 PutFixed32(&version_val, static_cast<uint32_t>(version_));
51 properties->insert({ExternalSstFilePropertyNames::kVersion, version_val});
52
53 // Global Sequence number
54 std::string seqno_val;
55 PutFixed64(&seqno_val, static_cast<uint64_t>(global_seqno_));
56 properties->insert({ExternalSstFilePropertyNames::kGlobalSeqno, seqno_val});
57
58 return Status::OK();
59 }
60
61 const char* Name() const override {
62 return "SstFileWriterPropertiesCollector";
63 }
64
65 UserCollectedProperties GetReadableProperties() const override {
66 return {{ExternalSstFilePropertyNames::kVersion, std::to_string(version_)}};
67 }
68
69 private:
70 int32_t version_;
71 SequenceNumber global_seqno_;
72};
