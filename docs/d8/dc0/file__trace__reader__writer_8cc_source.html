42Status FileTraceReader::Read(std::string* data) {
43 assert(file_reader_ != nullptr);
44 Status s = file_reader_->Read(IOOptions(), offset_, kTraceMetadataSize,
45 &result_, buffer_, nullptr);
46 if (!s.ok()) {
47 return s;
48 }
49 if (result_.size() == 0) {
50 // No more data to read
51 // Todo: Come up with a better way to indicate end of data. May be this
52 // could be avoided once footer is introduced.
53 return Status::Incomplete();
54 }
55 if (result_.size() < kTraceMetadataSize) {
56 return Status::Corruption("Corrupted trace file.");
57 }
58 *data = result_.ToString();
59 offset_ += kTraceMetadataSize;
60
61 uint32_t payload_len =
62 DecodeFixed32(&buffer_[kTraceTimestampSize + kTraceTypeSize]);
63
64 // Read Payload
65 unsigned int bytes_to_read = payload_len;
66 unsigned int to_read =
67 bytes_to_read > kBufferSize ? kBufferSize : bytes_to_read;
68 while (to_read > 0) {
69 s = file_reader_->Read(IOOptions(), offset_, to_read, &result_, buffer_,
70 nullptr);
71 if (!s.ok()) {
72 return s;
73 }
74 if (result_.size() < to_read) {
75 return Status::Corruption("Corrupted trace file.");
76 }
77 data->append(result_.data(), result_.size());
78
79 offset_ += to_read;
80 bytes_to_read -= to_read;
81 to_read = bytes_to_read > kBufferSize ? kBufferSize : bytes_to_read;
82 }
83
84 return s;
85}

104Status NewFileTraceReader(Env* env, const EnvOptions& env_options,
105 const std::string& trace_filename,
106 std::unique_ptr<TraceReader>* trace_reader) {
107 std::unique_ptr<RandomAccessFileReader> file_reader;
108 Status s = RandomAccessFileReader::Create(
109 env->GetFileSystem(), trace_filename, FileOptions(env_options),
110 &file_reader, nullptr);
111 if (!s.ok()) {
112 return s;
113 }
114 trace_reader->reset(new FileTraceReader(std::move(file_reader)));
115 return s;
116}
