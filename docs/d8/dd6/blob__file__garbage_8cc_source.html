48Status BlobFileGarbage::DecodeFrom(Slice* input) {
49 constexpr char class_name[] = "BlobFileGarbage";
50
51 if (!GetVarint64(input, &blob_file_number_)) {
52 return Status::Corruption(class_name, "Error decoding blob file number");
53 }
54
55 if (!GetVarint64(input, &garbage_blob_count_)) {
56 return Status::Corruption(class_name, "Error decoding garbage blob count");
57 }
58
59 if (!GetVarint64(input, &garbage_blob_bytes_)) {
60 return Status::Corruption(class_name, "Error decoding garbage blob bytes");
61 }
62
63 while (true) {
64 uint32_t custom_field_tag = 0;
65 if (!GetVarint32(input, &custom_field_tag)) {
66 return Status::Corruption(class_name, "Error decoding custom field tag");
67 }
68
69 if (custom_field_tag == kEndMarker) {
70 break;
71 }
72
73 if (custom_field_tag & kForwardIncompatibleMask) {
74 return Status::Corruption(
75 class_name, "Forward incompatible custom field encountered");
76 }
77
78 Slice custom_field_value;
79 if (!GetLengthPrefixedSlice(input, &custom_field_value)) {
80 return Status::Corruption(class_name,
81 "Error decoding custom field value");
82 }
83 }
84
85 return Status::OK();
86}
