121 template <typename CompareOp>
122 class MultiCfHeapItemComparator {
123 public:
124 explicit MultiCfHeapItemComparator(const Comparator* comparator)
125 : comparator_(comparator) {}
126 bool operator()(const MultiCfIteratorInfo& a,
127 const MultiCfIteratorInfo& b) const {
128 assert(a.iterator);
129 assert(b.iterator);
130 assert(a.iterator->Valid());
131 assert(b.iterator->Valid());
132 int c = comparator_->Compare(a.iterator->key(), b.iterator->key());
133 assert(c != 0 || a.order != b.order);
134 return c == 0 ? a.order - b.order > 0 : CompareOp()(c, 0);
135 }
136
137 private:
138 const Comparator* comparator_;
139 };

183 template <typename BinaryHeap, typename ChildSeekFuncType>
184 void SeekCommon(BinaryHeap& heap, ChildSeekFuncType child_seek_func) {
185 reset_func_();
186 heap.clear();
187 int i = 0;
188 for (auto& [cfh, iter] : cfh_iter_pairs_) {
189 child_seek_func(iter.get());
190 if (iter->Valid()) {
191 assert(iter->status().ok());
192 heap.push(MultiCfIteratorInfo{cfh, iter.get(), i});
193 } else {
194 considerStatus(iter->status());
195 if (!status_.ok()) {
196 // Non-OK status from the iterator. Bail out early
197 heap.clear();
198 return;
199 }
200 }
201 ++i;
202 }
203 if (!allow_unprepared_value_ && !heap.empty()) {
204 [[maybe_unused]] const bool result = PopulateIterator(heap);
205 assert(result || (!Valid() && !status_.ok()));
206 }
207 }

209 template <typename BinaryHeap, typename AdvanceFuncType>
210 void AdvanceIterator(BinaryHeap& heap, AdvanceFuncType advance_func) {
211 reset_func_();
212 // It is possible for one or more child iters are at invalid keys due to
213 // manual prefix iteration. For such cases, we consider the result of the
214 // multi-cf-iter is also undefined.
215 // https://github.com/facebook/rocksdb/wiki/Prefix-Seek#manual-prefix-iterating
216 // for details about manual prefix iteration
217 if (heap.empty()) {
218 return;
219 }
220
221 // 1. Keep the top iterator (by popping it from the heap)
222 // 2. Make sure all others have iterated past the top iterator key slice
223 // 3. Advance the top iterator, and add it back to the heap if valid
224 auto top = heap.top();
225 assert(top.iterator);
226 assert(top.iterator->Valid());
227 assert(top.iterator->status().ok());
228
229 heap.pop();
230
231 while (!heap.empty()) {
232 auto current = heap.top();
233 assert(current.iterator);
234 assert(current.iterator->Valid());
235 assert(current.iterator->status().ok());
236
237 if (comparator_->Compare(current.iterator->key(), top.iterator->key()) !=
238 0) {
239 break;
240 }
241
242 advance_func(current.iterator);
243
244 if (current.iterator->Valid()) {
245 assert(current.iterator->status().ok());
246 heap.replace_top(current);
247 } else {
248 considerStatus(current.iterator->status());
249 if (!status_.ok()) {
250 heap.clear();
251 return;
252 } else {
253 heap.pop();
254 }
255 }
256 }
257
258 advance_func(top.iterator);
259
260 if (top.iterator->Valid()) {
261 assert(top.iterator->status().ok());
262 heap.push(top);
263 } else {
264 considerStatus(top.iterator->status());
265 if (!status_.ok()) {
266 heap.clear();
267 return;
268 }
269 }
270
271 if (!allow_unprepared_value_ && !heap.empty()) {
272 [[maybe_unused]] const bool result = PopulateIterator(heap);
273 assert(result || (!Valid() && !status_.ok()));
274 }
275 }

277 template <typename BinaryHeap>
278 bool PopulateIterator(BinaryHeap& heap) {
279 // 1. Keep the top iterator (by popping it from the heap) and add it to list
280 // to populate
281 // 2. For all non-top iterators having the same key as top iter popped
282 // from the previous step, add them to the same list and pop it
283 // temporarily from the heap
284 // 3. Once no other iters have the same key as the top iter from step 1,
285 // populate the value/columns and attribute_groups from the list
286 // collected in step 1 and 2 and add all the iters back to the heap
287 assert(!heap.empty());
288
289 auto prepare_value = [this, &heap](Iterator* iterator) {
290 assert(iterator);
291 assert(iterator->Valid());
292 assert(iterator->status().ok());
293
294 if (!iterator->PrepareValue()) {
295 assert(!iterator->Valid());
296 assert(!iterator->status().ok());
297
298 considerStatus(iterator->status());
299 heap.clear();
300
301 assert(!Valid());
302 assert(!status_.ok());
303
304 return false;
305 }
306
307 return true;
308 };
309
310 auto top = heap.top();
311 assert(top.iterator);
312 assert(top.iterator->Valid());
313 assert(top.iterator->status().ok());
314
315 if (!prepare_value(top.iterator)) {
316 return false;
317 }
318
319 autovector<MultiCfIteratorInfo> to_populate;
320
321 to_populate.push_back(top);
322 heap.pop();
323
324 while (!heap.empty()) {
325 auto current = heap.top();
326 assert(current.iterator);
327 assert(current.iterator->Valid());
328 assert(current.iterator->status().ok());
329
330 if (comparator_->Compare(current.iterator->key(), top.iterator->key()) !=
331 0) {
332 break;
333 }
334
335 if (!prepare_value(current.iterator)) {
336 return false;
337 }
338
339 to_populate.push_back(current);
340 heap.pop();
341 }
342
343 // Add the items back to the heap
344 for (auto& item : to_populate) {
345 heap.push(item);
346 }
347
348 populate_func_(to_populate);
349
350 return true;
351 }
352};
