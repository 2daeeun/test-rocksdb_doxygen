32// Compaction filter that deletes expired blob indexes from the base DB.
33// Comes into two varieties, one for the non-GC case and one for the GC case.
34class BlobIndexCompactionFilterBase : public LayeredCompactionFilterBase {
35 public:
36 BlobIndexCompactionFilterBase(
37 BlobCompactionContext&& _context,
38 const CompactionFilter* _user_comp_filter,
39 std::unique_ptr<const CompactionFilter> _user_comp_filter_from_factory,
40 uint64_t current_time, Statistics* stats)
41 : LayeredCompactionFilterBase(_user_comp_filter,
42 std::move(_user_comp_filter_from_factory)),
43 context_(std::move(_context)),
44 current_time_(current_time),
45 statistics_(stats) {}
46
47 ~BlobIndexCompactionFilterBase() override;
48
49 // Filter expired blob indexes regardless of snapshots.
50 bool IgnoreSnapshots() const override { return true; }
51
52 Decision FilterV2(int level, const Slice& key, ValueType value_type,
53 const Slice& value, std::string* new_value,
54 std::string* skip_until) const override;
55
56 bool IsStackedBlobDbInternalCompactionFilter() const override { return true; }
57
58 protected:
59 bool IsBlobFileOpened() const;
60 virtual bool OpenNewBlobFileIfNeeded() const;
61 bool ReadBlobFromOldFile(const Slice& key, const BlobIndex& blob_index,
62 PinnableSlice* blob, bool need_decompress,
63 CompressionType* compression_type) const;
64 bool WriteBlobToNewFile(const Slice& key, const Slice& blob,
65 uint64_t* new_blob_file_number,
66 uint64_t* new_blob_offset) const;
67 bool CloseAndRegisterNewBlobFileIfNeeded() const;
68 bool CloseAndRegisterNewBlobFile() const;
69
70 Statistics* statistics() const { return statistics_; }
71 const BlobCompactionContext& context() const { return context_; }
72
73 private:
74 Decision HandleValueChange(const Slice& key, std::string* new_value) const;
75
76 private:
77 BlobCompactionContext context_;
78 const uint64_t current_time_;
79 Statistics* statistics_;
80
81 mutable std::shared_ptr<BlobFile> blob_file_;
82 mutable std::shared_ptr<BlobLogWriter> writer_;
83
84 // It is safe to not using std::atomic since the compaction filter, created
85 // from a compaction filter factroy, will not be called from multiple threads.
86 mutable uint64_t expired_count_ = 0;
87 mutable uint64_t expired_size_ = 0;
88 mutable uint64_t evicted_count_ = 0;
89 mutable uint64_t evicted_size_ = 0;
90};
