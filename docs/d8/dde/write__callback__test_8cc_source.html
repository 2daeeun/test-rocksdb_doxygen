130TEST_P(WriteCallbackPTest, WriteWithCallbackTest) {
131 struct WriteOP {
132 WriteOP(bool should_fail = false) { callback_.should_fail_ = should_fail; }
133
134 void Put(const string& key, const string& val) {
135 kvs_.emplace_back(key, val);
136 ASSERT_OK(write_batch_.Put(key, val));
137 }
138
139 void Clear() {
140 kvs_.clear();
141 write_batch_.Clear();
142 callback_.was_called_.store(false);
143 user_write_cb_.Reset();
144 }
145
146 MockWriteCallback callback_;
147 MockUserWriteCallback user_write_cb_;
148 WriteBatch write_batch_;
149 std::vector<std::pair<string, string>> kvs_;
150 };
151
152 // In each scenario we'll launch multiple threads to write.
153 // The size of each array equals to number of threads, and
154 // each boolean in it denote whether callback of corresponding
155 // thread should succeed or fail.
156 std::vector<std::vector<WriteOP>> write_scenarios = {
157 {true},
158 {false},
159 {false, false},
160 {true, true},
161 {true, false},
162 {false, true},
163 {false, false, false},
164 {true, true, true},
165 {false, true, false},
166 {true, false, true},
167 {true, false, false, false, false},
168 {false, false, false, false, true},
169 {false, false, true, false, true},
170 };
171
172 for (auto& write_group : write_scenarios) {
173 Options options;
174 options.create_if_missing = true;
175 options.unordered_write = unordered_write_;
176 options.allow_concurrent_memtable_write = allow_parallel_;
177 options.enable_pipelined_write = enable_pipelined_write_;
178 options.two_write_queues = two_queues_;
179 // Skip unsupported combinations
180 if (options.enable_pipelined_write && seq_per_batch_) {
181 continue;
182 }
183 if (options.enable_pipelined_write && options.two_write_queues) {
184 continue;
185 }
186 if (options.unordered_write && !options.allow_concurrent_memtable_write) {
187 continue;
188 }
189 if (options.unordered_write && options.enable_pipelined_write) {
190 continue;
191 }
192
193 ReadOptions read_options;
194 DB* db;
195 DBImpl* db_impl;
196
197 ASSERT_OK(DestroyDB(dbname, options));
198
199 DBOptions db_options(options);
200 ColumnFamilyOptions cf_options(options);
201 std::vector<ColumnFamilyDescriptor> column_families;
202 column_families.emplace_back(kDefaultColumnFamilyName, cf_options);
203 std::vector<ColumnFamilyHandle*> handles;
204 auto open_s = DBImpl::Open(db_options, dbname, column_families, &handles,
205 &db, seq_per_batch_, true /* batch_per_txn */,
206 false /* is_retry */, nullptr /* can_retry */);
207 ASSERT_OK(open_s);
208 assert(handles.size() == 1);
209 delete handles[0];
210
211 db_impl = dynamic_cast<DBImpl*>(db);
212 ASSERT_TRUE(db_impl);
213
214 // Writers that have called JoinBatchGroup.
215 std::atomic<uint64_t> threads_joining(0);
216 // Writers that have linked to the queue
217 std::atomic<uint64_t> threads_linked(0);
218 // Writers that pass WriteThread::JoinBatchGroup:Wait sync-point.
219 std::atomic<uint64_t> threads_verified(0);
220
221 std::atomic<uint64_t> seq(db_impl->GetLatestSequenceNumber());
222 ASSERT_EQ(db_impl->GetLatestSequenceNumber(), 0);
223
224 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
225 "WriteThread::JoinBatchGroup:Start", [&](void*) {
226 uint64_t cur_threads_joining = threads_joining.fetch_add(1);
227 // Wait for the last joined writer to link to the queue.
228 // In this way the writers link to the queue one by one.
229 // This allows us to confidently detect the first writer
230 // who increases threads_linked as the leader.
231 while (threads_linked.load() < cur_threads_joining) {
232 }
233 });
234
235 // Verification once writers call JoinBatchGroup.
236 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
237 "WriteThread::JoinBatchGroup:Wait", [&](void* arg) {
238 uint64_t cur_threads_linked = threads_linked.fetch_add(1);
239 bool is_leader = false;
240 bool is_last = false;
241
242 // who am i
243 is_leader = (cur_threads_linked == 0);
244 is_last = (cur_threads_linked == write_group.size() - 1);
245
246 // check my state
247 auto* writer = static_cast<WriteThread::Writer*>(arg);
248
249 if (is_leader) {
250 ASSERT_TRUE(writer->state ==
251 WriteThread::State::STATE_GROUP_LEADER);
252 } else {
253 ASSERT_TRUE(writer->state == WriteThread::State::STATE_INIT);
254 }
255
256 // (meta test) the first WriteOP should indeed be the first
257 // and the last should be the last (all others can be out of
258 // order)
259 if (is_leader) {
260 ASSERT_TRUE(writer->callback->Callback(nullptr).ok() ==
261 !write_group.front().callback_.should_fail_);
262 } else if (is_last) {
263 ASSERT_TRUE(writer->callback->Callback(nullptr).ok() ==
264 !write_group.back().callback_.should_fail_);
265 }
266
267 threads_verified.fetch_add(1);
268 // Wait here until all verification in this sync-point
269 // callback finish for all writers.
270 while (threads_verified.load() < write_group.size()) {
271 }
272 });
273
274 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
275 "WriteThread::JoinBatchGroup:DoneWaiting", [&](void* arg) {
276 // check my state
277 auto* writer = static_cast<WriteThread::Writer*>(arg);
278
279 if (!allow_batching_) {
280 // no batching so everyone should be a leader
281 ASSERT_TRUE(writer->state ==
282 WriteThread::State::STATE_GROUP_LEADER);
283 } else if (!allow_parallel_) {
284 ASSERT_TRUE(writer->state == WriteThread::State::STATE_COMPLETED ||
285 (enable_pipelined_write_ &&
286 writer->state ==
287 WriteThread::State::STATE_MEMTABLE_WRITER_LEADER));
288 }
289 });
290
291 std::atomic<uint32_t> thread_num(0);
292 std::atomic<char> dummy_key(0);
293
294 // Each write thread create a random write batch and write to DB
295 // with a write callback.
296 std::function<void()> write_with_callback_func = [&]() {
297 uint32_t i = thread_num.fetch_add(1);
298 Random rnd(i);
299
300 // leaders gotta lead
301 while (i > 0 && threads_verified.load() < 1) {
302 }
303
304 // loser has to lose
305 while (i == write_group.size() - 1 &&
306 threads_verified.load() < write_group.size() - 1) {
307 }
308
309 auto& write_op = write_group.at(i);
310 write_op.Clear();
311 write_op.callback_.allow_batching_ = allow_batching_;
312
313 // insert some keys
314 for (uint32_t j = 0; j < rnd.Next() % 50; j++) {
315 // grab unique key
316 char my_key = dummy_key.fetch_add(1);
317
318 string skey(5, my_key);
319 string sval(10, my_key);
320 write_op.Put(skey, sval);
321
322 if (!write_op.callback_.should_fail_ && !seq_per_batch_) {
323 seq.fetch_add(1);
324 }
325 }
326 if (!write_op.callback_.should_fail_ && seq_per_batch_) {
327 seq.fetch_add(1);
328 }
329
330 WriteOptions woptions;
331 woptions.disableWAL = !enable_WAL_;
332 woptions.sync = enable_WAL_;
333 if (woptions.protection_bytes_per_key > 0) {
334 ASSERT_OK(WriteBatchInternal::UpdateProtectionInfo(
335 &write_op.write_batch_, woptions.protection_bytes_per_key));
336 }
337 Status s;
338 if (seq_per_batch_) {
339 class PublishSeqCallback : public PreReleaseCallback {
340 public:
341 PublishSeqCallback(DBImpl* db_impl_in) : db_impl_(db_impl_in) {}
342 Status Callback(SequenceNumber last_seq, bool /*not used*/, uint64_t,
343 size_t /*index*/, size_t /*total*/) override {
344 db_impl_->SetLastPublishedSequence(last_seq);
345 return Status::OK();
346 }
347 DBImpl* db_impl_;
348 } publish_seq_callback(db_impl);
349 // seq_per_batch_ requires a natural batch separator or Noop
350 ASSERT_OK(WriteBatchInternal::InsertNoop(&write_op.write_batch_));
351 const size_t ONE_BATCH = 1;
352 s = db_impl->WriteImpl(woptions, &write_op.write_batch_,
353 &write_op.callback_, &write_op.user_write_cb_,
354 nullptr, 0, false, nullptr, ONE_BATCH,
355 two_queues_ ? &publish_seq_callback : nullptr);
356 } else {
357 s = db_impl->WriteWithCallback(woptions, &write_op.write_batch_,
358 &write_op.callback_,
359 &write_op.user_write_cb_);
360 }
361
362 ASSERT_TRUE(write_op.user_write_cb_.write_enqueued_.load());
363 if (write_op.callback_.should_fail_) {
364 ASSERT_TRUE(s.IsBusy());
365 ASSERT_FALSE(write_op.user_write_cb_.wal_write_done_.load());
366 } else {
367 ASSERT_OK(s);
368 if (enable_WAL_) {
369 ASSERT_TRUE(write_op.user_write_cb_.wal_write_done_.load());
370 } else {
371 ASSERT_FALSE(write_op.user_write_cb_.wal_write_done_.load());
372 }
373 }
374 };
375
376 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
377
378 // do all the writes
379 std::vector<port::Thread> threads;
380 for (uint32_t i = 0; i < write_group.size(); i++) {
381 threads.emplace_back(write_with_callback_func);
382 }
383 for (auto& t : threads) {
384 t.join();
385 }
386
387 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
388
389 // check for keys
390 string value;
391 for (auto& w : write_group) {
392 ASSERT_TRUE(w.callback_.was_called_.load());
393 for (auto& kvp : w.kvs_) {
394 if (w.callback_.should_fail_) {
395 ASSERT_TRUE(db->Get(read_options, kvp.first, &value).IsNotFound());
396 } else {
397 ASSERT_OK(db->Get(read_options, kvp.first, &value));
398 ASSERT_EQ(value, kvp.second);
399 }
400 }
401 }
402
403 ASSERT_EQ(seq.load(), db_impl->TEST_GetLastVisibleSequence());
404
405 delete db;
406 ASSERT_OK(DestroyDB(dbname, options));
407 }
408}
