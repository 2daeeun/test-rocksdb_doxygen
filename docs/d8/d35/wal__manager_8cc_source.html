46Status WalManager::GetSortedWalFiles(VectorWalPtr& files, bool need_seqnos,
47 bool include_archived) {
48 // First get sorted files in db dir, then get sorted files from archived
49 // dir, to avoid a race condition where a log file is moved to archived
50 // dir in between.
51 Status s;
52 // list wal files in main db dir.
53 VectorWalPtr logs;
54 s = GetSortedWalsOfType(wal_dir_, logs, kAliveLogFile, need_seqnos);
55
56 if (!include_archived || !s.ok()) {
57 return s;
58 }
59
60 // Reproduce the race condition where a log file is moved
61 // to archived dir, between these two sync points, used in
62 // (DBTest,TransactionLogIteratorRace)
63 TEST_SYNC_POINT("WalManager::GetSortedWalFiles:1");
64 TEST_SYNC_POINT("WalManager::GetSortedWalFiles:2");
65
66 files.clear();
67 // list wal files in archive dir.
68 std::string archivedir = ArchivalDirectory(wal_dir_);
69 Status exists = env_->FileExists(archivedir);
70 if (exists.ok()) {
71 s = GetSortedWalsOfType(archivedir, files, kArchivedLogFile, need_seqnos);
72 if (!s.ok()) {
73 return s;
74 }
75 } else if (!exists.IsNotFound()) {
76 assert(s.ok());
77 return exists;
78 }
79
80 uint64_t latest_archived_log_number = 0;
81 if (!files.empty()) {
82 latest_archived_log_number = files.back()->LogNumber();
83 ROCKS_LOG_INFO(db_options_.info_log, "Latest Archived log: %" PRIu64,
84 latest_archived_log_number);
85 }
86
87 files.reserve(files.size() + logs.size());
88 for (auto& log : logs) {
89 if (log->LogNumber() > latest_archived_log_number) {
90 files.push_back(std::move(log));
91 } else {
92 // When the race condition happens, we could see the
93 // same log in both db dir and archived dir. Simply
94 // ignore the one in db dir. Note that, if we read
95 // archived dir first, we would have missed the log file.
96 ROCKS_LOG_WARN(db_options_.info_log, "%s already moved to archive",
97 log->PathName().c_str());
98 }
99 }
100
101 return s;
102}

104Status WalManager::GetUpdatesSince(
105 SequenceNumber seq, std::unique_ptr<TransactionLogIterator>* iter,
106 const TransactionLogIterator::ReadOptions& read_options,
107 VersionSet* version_set) {
108 if (seq_per_batch_) {
109 return Status::NotSupported();
110 }
111
112 assert(!seq_per_batch_);
113
114 // Get all sorted Wal Files.
115 // Do binary search and open files and find the seq number.
116
117 std::unique_ptr<VectorWalPtr> wal_files(new VectorWalPtr);
118 Status s = GetSortedWalFiles(*wal_files);
119 if (!s.ok()) {
120 return s;
121 }
122
123 s = RetainProbableWalFiles(*wal_files, seq);
124 if (!s.ok()) {
125 return s;
126 }
127 iter->reset(new TransactionLogIteratorImpl(
128 wal_dir_, &db_options_, read_options, file_options_, seq,
129 std::move(wal_files), version_set, seq_per_batch_, io_tracer_));
130 return (*iter)->status();
131}
132
133// 1. Go through all archived files and
134// a. if ttl is enabled, delete outdated files
135// b. if archive size limit is enabled, delete empty files,
136// compute file number and size.
137// 2. If size limit is enabled:
138// a. compute how many files should be deleted
139// b. get sorted non-empty archived logs
140// c. delete what should be deleted
141void WalManager::PurgeObsoleteWALFiles() {
142 bool const ttl_enabled = db_options_.WAL_ttl_seconds > 0;
143 bool const size_limit_enabled = db_options_.WAL_size_limit_MB > 0;
144 if (!ttl_enabled && !size_limit_enabled) {
145 return;
146 }
147
148 int64_t current_time = 0;
149 Status s = db_options_.clock->GetCurrentTime(&current_time);
150 if (!s.ok()) {
151 ROCKS_LOG_ERROR(db_options_.info_log, "Can't get current time: %s",
152 s.ToString().c_str());
153 assert(false);
154 return;
155 }
156 uint64_t const now_seconds = static_cast<uint64_t>(current_time);
157 uint64_t const time_to_check =
158 ttl_enabled
159 ? std::min(kDefaultIntervalToDeleteObsoleteWAL,
160 std::max(uint64_t{1}, db_options_.WAL_ttl_seconds / 2))
161 : kDefaultIntervalToDeleteObsoleteWAL;
162 uint64_t old_last_run_time = purge_wal_files_last_run_.LoadRelaxed();
163 do {
164 if (old_last_run_time + time_to_check > now_seconds) {
165 // last run is recent enough, no need to purge
166 return;
167 }
168 } while (!purge_wal_files_last_run_.CasWeakRelaxed(
169 /*expected=*/old_last_run_time, /*desired=*/now_seconds));
170
171 std::string archival_dir = ArchivalDirectory(wal_dir_);
172 std::vector<std::string> files;
173 s = env_->GetChildren(archival_dir, &files);
174 if (!s.ok()) {
175 ROCKS_LOG_ERROR(db_options_.info_log, "Can't get archive files: %s",
176 s.ToString().c_str());
177 return;
178 }
179
180 size_t log_files_num = 0;
181 uint64_t log_file_size = 0;
182 for (auto& f : files) {
183 uint64_t number;
184 FileType type;
185 if (ParseFileName(f, &number, &type) && type == kWalFile) {
186 std::string const file_path = archival_dir + "/" + f;
187 if (ttl_enabled) {
188 uint64_t file_m_time;
189 s = env_->GetFileModificationTime(file_path, &file_m_time);
190 if (!s.ok()) {
191 ROCKS_LOG_WARN(db_options_.info_log,
192 "Can't get file mod time: %s: %s", file_path.c_str(),
193 s.ToString().c_str());
194 continue;
195 }
196 if (now_seconds - file_m_time > db_options_.WAL_ttl_seconds) {
197 s = DeleteDBFile(&db_options_, file_path, archival_dir, false,
198 /*force_fg=*/!wal_in_db_path_);
199 if (!s.ok()) {
200 ROCKS_LOG_WARN(db_options_.info_log, "Can't delete file: %s: %s",
201 file_path.c_str(), s.ToString().c_str());
202 continue;
203 } else {
204 MutexLock l(&read_first_record_cache_mutex_);
205 read_first_record_cache_.erase(number);
206 }
207 continue;
208 }
209 }
210
211 if (size_limit_enabled) {
212 uint64_t file_size;
213 s = env_->GetFileSize(file_path, &file_size);
214 if (!s.ok()) {
215 ROCKS_LOG_ERROR(db_options_.info_log,
216 "Unable to get file size: %s: %s", file_path.c_str(),
217 s.ToString().c_str());
218 return;
219 } else {
220 if (file_size > 0) {
221 log_file_size = std::max(log_file_size, file_size);
222 ++log_files_num;
223 } else {
224 s = DeleteDBFile(&db_options_, file_path, archival_dir, false,
225 /*force_fg=*/!wal_in_db_path_);
226 if (!s.ok()) {
227 ROCKS_LOG_WARN(db_options_.info_log,
228 "Unable to delete file: %s: %s", file_path.c_str(),
229 s.ToString().c_str());
230 continue;
231 } else {
232 MutexLock l(&read_first_record_cache_mutex_);
233 read_first_record_cache_.erase(number);
234 }
235 }
236 }
237 }
238 }
239 }
240
241 if (0 == log_files_num || !size_limit_enabled) {
242 return;
243 }
244
245 size_t const files_keep_num = static_cast<size_t>(
246 db_options_.WAL_size_limit_MB * 1024 * 1024 / log_file_size);
247 if (log_files_num <= files_keep_num) {
248 return;
249 }
250
251 size_t files_del_num = log_files_num - files_keep_num;
252 VectorWalPtr archived_logs;
253 s = GetSortedWalsOfType(archival_dir, archived_logs, kArchivedLogFile,
254 /*need_seqno=*/false);
255 if (!s.ok()) {
256 ROCKS_LOG_WARN(db_options_.info_log,
257 "Unable to get archived WALs from: %s: %s",
258 archival_dir.c_str(), s.ToString().c_str());
259 files_del_num = 0;
260 } else if (files_del_num > archived_logs.size()) {
261 ROCKS_LOG_WARN(db_options_.info_log,
262 "Trying to delete more archived log files than "
263 "exist. Deleting all");
264 files_del_num = archived_logs.size();
265 }
266
267 for (size_t i = 0; i < files_del_num; ++i) {
268 std::string const file_path = archived_logs[i]->PathName();
269 s = DeleteDBFile(&db_options_, wal_dir_ + "/" + file_path, wal_dir_, false,
270 /*force_fg=*/!wal_in_db_path_);
271 if (!s.ok()) {
272 ROCKS_LOG_WARN(db_options_.info_log, "Unable to delete file: %s: %s",
273 file_path.c_str(), s.ToString().c_str());
274 continue;
275 } else {
276 MutexLock l(&read_first_record_cache_mutex_);
277 read_first_record_cache_.erase(archived_logs[i]->LogNumber());
278 }
279 }
280}

297Status WalManager::GetSortedWalsOfType(const std::string& path,
298 VectorWalPtr& log_files,
299 WalFileType log_type, bool need_seqnos) {
300 std::vector<std::string> all_files;
301 const Status status = env_->GetChildren(path, &all_files);
302 if (!status.ok()) {
303 return status;
304 }
305 log_files.reserve(all_files.size());
306 for (const auto& f : all_files) {
307 uint64_t number;
308 FileType type;
309 if (ParseFileName(f, &number, &type) && type == kWalFile) {
310 SequenceNumber sequence;
311 if (need_seqnos) {
312 Status s = ReadFirstRecord(log_type, number, &sequence);
313 if (!s.ok()) {
314 return s;
315 }
316 if (sequence == 0) {
317 // empty file
318 continue;
319 }
320 } else {
321 sequence = 0;
322 }
323
324 // Reproduce the race condition where a log file is moved
325 // to archived dir, between these two sync points, used in
326 // (DBTest,TransactionLogIteratorRace)
327 TEST_SYNC_POINT("WalManager::GetSortedWalsOfType:1");
328 TEST_SYNC_POINT("WalManager::GetSortedWalsOfType:2");
329
330 uint64_t size_bytes;
331 Status s = env_->GetFileSize(LogFileName(path, number), &size_bytes);
332 // re-try in case the alive log file has been moved to archive.
333 if (!s.ok() && log_type == kAliveLogFile) {
334 std::string archived_file = ArchivedLogFileName(path, number);
335 if (env_->FileExists(archived_file).ok()) {
336 s = env_->GetFileSize(archived_file, &size_bytes);
337 if (!s.ok() && env_->FileExists(archived_file).IsNotFound()) {
338 // oops, the file just got deleted from archived dir! move on
339 s = Status::OK();
340 continue;
341 }
342 }
343 }
344 if (!s.ok()) {
345 return s;
346 }
347
348 log_files.emplace_back(
349 new WalFileImpl(number, log_type, sequence, size_bytes));
350 }
351 }
352 std::sort(
353 log_files.begin(), log_files.end(),
354 [](const std::unique_ptr<WalFile>& a, const std::unique_ptr<WalFile>& b) {
355 WalFileImpl* a_impl = static_cast_with_check<WalFileImpl>(a.get());
356 WalFileImpl* b_impl = static_cast_with_check<WalFileImpl>(b.get());
357 return *a_impl < *b_impl;
358 });
359 return status;
360}

362Status WalManager::RetainProbableWalFiles(VectorWalPtr& all_logs,
363 const SequenceNumber target) {
364 int64_t start = 0; // signed to avoid overflow when target is < first file.
365 int64_t end = static_cast<int64_t>(all_logs.size()) - 1;
366 // Binary Search. avoid opening all files.
367 while (end >= start) {
368 int64_t mid = start + (end - start) / 2; // Avoid overflow.
369 SequenceNumber current_seq_num =
370 all_logs.at(static_cast<size_t>(mid))->StartSequence();
371 if (current_seq_num == target) {
372 end = mid;
373 break;
374 } else if (current_seq_num < target) {
375 start = mid + 1;
376 } else {
377 end = mid - 1;
378 }
379 }
380 // end could be -ve.
381 size_t start_index =
382 static_cast<size_t>(std::max(static_cast<int64_t>(0), end));
383 // The last wal file is always included
384 all_logs.erase(all_logs.begin(), all_logs.begin() + start_index);
385 return Status::OK();
386}

388Status WalManager::ReadFirstRecord(const WalFileType type,
389 const uint64_t number,
390 SequenceNumber* sequence) {
391 *sequence = 0;
392 if (type != kAliveLogFile && type != kArchivedLogFile) {
393 ROCKS_LOG_ERROR(db_options_.info_log, "[WalManger] Unknown file type %s",
394 std::to_string(type).c_str());
395 return Status::NotSupported("File Type Not Known " + std::to_string(type));
396 }
397 {
398 MutexLock l(&read_first_record_cache_mutex_);
399 auto itr = read_first_record_cache_.find(number);
400 if (itr != read_first_record_cache_.end()) {
401 *sequence = itr->second;
402 return Status::OK();
403 }
404 }
405 Status s;
406 if (type == kAliveLogFile) {
407 std::string fname = LogFileName(wal_dir_, number);
408 s = ReadFirstLine(fname, number, sequence);
409 if (!s.ok() && env_->FileExists(fname).ok()) {
410 // return any error that is not caused by non-existing file
411 return s;
412 }
413 }
414
415 if (type == kArchivedLogFile || !s.ok()) {
416 // check if the file got moved to archive.
417 std::string archived_file = ArchivedLogFileName(wal_dir_, number);
418 s = ReadFirstLine(archived_file, number, sequence);
419 // maybe the file was deleted from archive dir. If that's the case, return
420 // Status::OK(). The caller with identify this as empty file because
421 // *sequence == 0
422 if (!s.ok() && env_->FileExists(archived_file).IsNotFound()) {
423 return Status::OK();
424 }
425 }
426
427 if (s.ok() && *sequence != 0) {
428 MutexLock l(&read_first_record_cache_mutex_);
429 read_first_record_cache_.insert({number, *sequence});
430 }
431 return s;
432}

460// the function returns status.ok() and sequence == 0 if the file exists, but is
461// empty
462Status WalManager::ReadFirstLine(const std::string& fname,
463 const uint64_t number,
464 SequenceNumber* sequence) {
465 struct LogReporter : public log::Reader::Reporter {
466 Env* env;
467 Logger* info_log;
468 const char* fname;
469
470 Status* status;
471 bool ignore_error; // true if db_options_.paranoid_checks==false
472 void Corruption(size_t bytes, const Status& s) override {
473 ROCKS_LOG_WARN(info_log, "[WalManager] %s%s: dropping %d bytes; %s",
474 (this->ignore_error ? "(ignoring error) " : ""), fname,
475 static_cast<int>(bytes), s.ToString().c_str());
476 if (this->status->ok()) {
477 // only keep the first error
478 *this->status = s;
479 }
480 }
481 };
482
483 std::unique_ptr<FSSequentialFile> file;
484 Status status = fs_->NewSequentialFile(
485 fname, fs_->OptimizeForLogRead(file_options_), &file, nullptr);
486 std::unique_ptr<SequentialFileReader> file_reader(
487 new SequentialFileReader(std::move(file), fname, io_tracer_));
488
489 if (!status.ok()) {
490 return status;
491 }
492
493 LogReporter reporter;
494 reporter.env = env_;
495 reporter.info_log = db_options_.info_log.get();
496 reporter.fname = fname.c_str();
497 reporter.status = &status;
498 reporter.ignore_error = !db_options_.paranoid_checks;
499 log::Reader reader(db_options_.info_log, std::move(file_reader), &reporter,
500 true /*checksum*/, number);
501 std::string scratch;
502 Slice record;
503
504 if (reader.ReadRecord(&record, &scratch) &&
505 (status.ok() || !db_options_.paranoid_checks)) {
506 if (record.size() < WriteBatchInternal::kHeader) {
507 reporter.Corruption(record.size(),
508 Status::Corruption("log record too small"));
509 // TODO read record's till the first no corrupt entry?
510 } else {
511 WriteBatch batch;
512 // We can overwrite an existing non-OK Status since it'd only reach here
513 // with `paranoid_checks == false`.
514 status = WriteBatchInternal::SetContents(&batch, record);
515 if (status.ok()) {
516 *sequence = WriteBatchInternal::Sequence(&batch);
517 return status;
518 }
519 }
520 }
521
522 if (status.ok() && reader.IsCompressedAndEmptyFile()) {
523 // In case of wal_compression, it writes a `kSetCompressionType` record
524 // which is not associated with any sequence number. As result for an empty
525 // file, GetSortedWalsOfType() will skip these WALs causing the operations
526 // to fail.
527 // Therefore, in order to avoid that failure, it sets sequence_number to 1
528 // indicating those WALs should be included.
529 *sequence = 1;
530 } else {
531 // ReadRecord might have returned false on EOF, which means that the log
532 // file is empty. Or, a failure may have occurred while processing the first
533 // entry. In any case, return status and set sequence number to 0.
534 *sequence = 0;
535 }
536 return status;
537}
