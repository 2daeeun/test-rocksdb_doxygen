40def parse_src_mk(repo_path):
41 src_mk = repo_path + "/src.mk"
42 src_files = {}
43 for line in open(src_mk):
44 line = line.strip()
45 if len(line) == 0 or line[0] == "#":
46 continue
47 if "=" in line:
48 current_src = line.split("=")[0].strip()
49 src_files[current_src] = []
50 elif ".c" in line:
51 src_path = line.split("\\")[0].strip()
52 src_files[current_src].append(src_path)
