18static void RegisterTableFactories(const std::string& /*arg*/) {
19 static std::once_flag loaded;
20 std::call_once(loaded, []() {
21 auto library = ObjectLibrary::Default();
22 library->AddFactory<TableFactory>(
23 TableFactory::kBlockBasedTableName(),
24 [](const std::string& /*uri*/, std::unique_ptr<TableFactory>* guard,
25 std::string* /* errmsg */) {
26 guard->reset(new BlockBasedTableFactory());
27 return guard->get();
28 });
29 library->AddFactory<TableFactory>(
30 TableFactory::kPlainTableName(),
31 [](const std::string& /*uri*/, std::unique_ptr<TableFactory>* guard,
32 std::string* /* errmsg */) {
33 guard->reset(new PlainTableFactory());
34 return guard->get();
35 });
36 library->AddFactory<TableFactory>(
37 TableFactory::kCuckooTableName(),
38 [](const std::string& /*uri*/, std::unique_ptr<TableFactory>* guard,
39 std::string* /* errmsg */) {
40 guard->reset(new CuckooTableFactory());
41 return guard->get();
42 });
43 });
44}

46Status TableFactory::CreateFromString(const ConfigOptions& config_options,
47 const std::string& value,

49 RegisterTableFactories("");
