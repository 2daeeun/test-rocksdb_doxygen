103 {
104 // When there is a next partition, add the prefix of the first key in the
105 // next partition before closing this one out. This is needed to support
106 // prefix Seek, because there could exist a key k where
107 // * last_key < k < next_key
108 // * prefix(last_key) != prefix(k)
109 // * prefix(k) == prefix(next_key)
110 // * seeking to k lands in this partition, not the next
111 // in which case the iterator needs to find next_key despite starting in
112 // the partition before it. (This fixes a bug in the original implementation
113 // of format_version=3.)
114 if (next_prefix) {
115 if (whole_key_filtering()) {
116 // NOTE: At the end of building filter bits, we need a special case for
117 // treating prefix as an "alt" entry. See AddKeyAndAlt() comment. This is
118 // a reasonable hack for that.
119 filter_bits_builder_->AddKeyAndAlt(*next_prefix, *next_prefix);
120 } else {
121 filter_bits_builder_->AddKey(*next_prefix);
122 }
123 }
124

128 Status filter_construction_status = Status::OK();

131 if (filter_construction_status.ok()) {
132 filter_construction_status = filter_bits_builder_->MaybePostVerify(filter);
133 }
