17// Evictable Hash Table
18//
19// Hash table index where least accessed (or one of the least accessed) elements
20// can be evicted.
21//
22// Please note EvictableHashTable can only be created for pointer type objects
23template <class T, class Hash, class Equal>
24class EvictableHashTable : private HashTable<T*, Hash, Equal> {
25 public:
26 using hash_table = HashTable<T*, Hash, Equal>;
27
28 explicit EvictableHashTable(const size_t capacity = 1024 * 1024,
29 const float load_factor = 2.0,
30 const uint32_t nlocks = 256)
31 : HashTable<T*, Hash, Equal>(capacity, load_factor, nlocks),
32 lru_lists_(new LRUList<T>[hash_table::nlocks_]) {
33 assert(lru_lists_);
34 }
35
36 virtual ~EvictableHashTable() { AssertEmptyLRU(); }
37
38 //
39 // Insert given record to hash table (and LRU list)
40 //
41 bool Insert(T* t) {
42 const uint64_t h = Hash()(t);
43 typename hash_table::Bucket& bucket = GetBucket(h);
44 LRUListType& lru = GetLRUList(h);
45 port::RWMutex& lock = GetMutex(h);
46
47 WriteLock _(&lock);
48 if (hash_table::Insert(&bucket, t)) {
49 lru.Push(t);
50 return true;
51 }
52 return false;
53 }
54
55 //
56 // Lookup hash table
57 //
58 // Please note that read lock should be held by the caller. This is because
59 // the caller owns the data, and should hold the read lock as long as he
60 // operates on the data.
61 bool Find(T* t, T** ret) {
62 const uint64_t h = Hash()(t);
63 typename hash_table::Bucket& bucket = GetBucket(h);
64 LRUListType& lru = GetLRUList(h);
65 port::RWMutex& lock = GetMutex(h);
66
67 ReadLock _(&lock);
68 if (hash_table::Find(&bucket, t, ret)) {
69 ++(*ret)->refs_;
70 lru.Touch(*ret);
71 return true;
72 }
73 return false;
74 }
75
76 //
77 // Evict one of the least recently used object
78 //
79 T* Evict(const std::function<void(T*)>& fn = nullptr) {
80 uint32_t random = Random::GetTLSInstance()->Next();
81 const size_t start_idx = random % hash_table::nlocks_;
82 T* t = nullptr;
83
84 // iterate from start_idx .. 0 .. start_idx
85 for (size_t i = 0; !t && i < hash_table::nlocks_; ++i) {
86 const size_t idx = (start_idx + i) % hash_table::nlocks_;
87
88 WriteLock _(&hash_table::locks_[idx]);
89 LRUListType& lru = lru_lists_[idx];
90 if (!lru.IsEmpty() && (t = lru.Pop()) != nullptr) {
91 assert(!t->refs_);
92 // We got an item to evict, erase from the bucket
93 const uint64_t h = Hash()(t);
94 typename hash_table::Bucket& bucket = GetBucket(h);
95 T* tmp = nullptr;
96 bool status = hash_table::Erase(&bucket, t, &tmp);
97 assert(t == tmp);
98 (void)status;
99 assert(status);
100 if (fn) {
101 fn(t);
102 }
103 break;
104 }
105 assert(!t);
106 }
107 return t;
108 }
109
110 void Clear(void (*fn)(T*)) {
111 for (uint32_t i = 0; i < hash_table::nbuckets_; ++i) {
112 const uint32_t lock_idx = i % hash_table::nlocks_;
113 WriteLock _(&hash_table::locks_[lock_idx]);
114 auto& lru_list = lru_lists_[lock_idx];
115 auto& bucket = hash_table::buckets_[i];
116 for (auto* t : bucket.list_) {
117 lru_list.Unlink(t);
118 (*fn)(t);
119 }
120 bucket.list_.clear();
121 }
122 // make sure that all LRU lists are emptied
123 AssertEmptyLRU();
124 }
125
126 void AssertEmptyLRU() {
127#ifndef NDEBUG
128 for (uint32_t i = 0; i < hash_table::nlocks_; ++i) {
129 WriteLock _(&hash_table::locks_[i]);
130 auto& lru_list = lru_lists_[i];
131 assert(lru_list.IsEmpty());
132 }
133#endif
134 }
135
136 //
137 // Fetch the mutex associated with a key
138 // This call is used to hold the lock for a given data for extended period of
139 // time.
140 port::RWMutex* GetMutex(T* t) { return hash_table::GetMutex(t); }
141
142 private:
143 using LRUListType = LRUList<T>;
144
145 typename hash_table::Bucket& GetBucket(const uint64_t h) {
146 const uint32_t bucket_idx = h % hash_table::nbuckets_;
147 return hash_table::buckets_[bucket_idx];
148 }
149
150 LRUListType& GetLRUList(const uint64_t h) {
151 const uint32_t bucket_idx = h % hash_table::nbuckets_;
152 const uint32_t lock_idx = bucket_idx % hash_table::nlocks_;
153 return lru_lists_[lock_idx];
154 }
155
156 port::RWMutex& GetMutex(const uint64_t h) {
157 const uint32_t bucket_idx = h % hash_table::nbuckets_;
158 const uint32_t lock_idx = bucket_idx % hash_table::nlocks_;
159 return hash_table::locks_[lock_idx];
160 }
161
162 std::unique_ptr<LRUListType[]> lru_lists_;
163};
