10// Repairer does best effort recovery to recover as much data as possible after
11// a disaster without compromising consistency. It does not guarantee bringing
12// the database to a time consistent state.
13//
14// Repair process is broken into 4 phases:
15// (a) Find files
16// (b) Convert logs to tables
17// (c) Extract metadata
18// (d) Write Descriptor
19//
20// (a) Find files
21//
22// The repairer goes through all the files in the directory, and classifies them
23// based on their file name. Any file that cannot be identified by name will be
24// ignored.
25//
26// (b) Convert logs to table
27//
28// Every log file that is active is replayed. All sections of the file where the
29// checksum does not match is skipped over. We intentionally give preference to
30// data consistency.
31//
32// (c) Extract metadata
33//
34// We scan every table to compute
35// (1) smallest/largest for the table
36// (2) largest sequence number in the table
37// (3) oldest blob file referred to by the table (if applicable)
38//
39// If we are unable to scan the file, then we ignore the table.
40//
41// (d) Write Descriptor
42//
43// We generate descriptor contents:
44// - log number is set to zero
45// - next-file-number is set to 1 + largest file number we found
46// - last-sequence-number is set to largest sequence# found across
47// all tables (see 2c)
48// - compaction pointers are cleared
49// - every table file is added at level 0
50//
51// Possible optimization 1:
52// (a) Compute total size and use to pick appropriate max-level M
53// (b) Sort tables by largest sequence# in the table
54// (c) For each table: if it overlaps earlier table, place in level-0,
55// else place in level-M.
56// (d) We can provide options for time consistent recovery and unsafe recovery
57// (ignore checksum failure when applicable)
58// Possible optimization 2:
59// Store per-table metadata (smallest, largest, largest-seq#, ...)
60// in the table's meta section to speed up ScanTable.

148 Status AddColumnFamily(const std::string& cf_name, uint32_t cf_id) {
149 // TODO: plumb Env::IOActivity, Env::IOPriority;
150 const ReadOptions read_options;
151 const WriteOptions write_options;
152
153 const auto* cf_opts = GetColumnFamilyOptions(cf_name);
154 if (cf_opts == nullptr) {
155 return Status::Corruption("Encountered unknown column family with name=" +

157 }
158 Options opts(db_options_, *cf_opts);
159 MutableCFOptions mut_cf_opts(opts);
160
161 VersionEdit edit;
162 edit.SetComparatorName(opts.comparator->Name());
163 edit.SetPersistUserDefinedTimestamps(opts.persist_user_defined_timestamps);
164 edit.SetLogNumber(0);
165 edit.SetColumnFamily(cf_id);
166 ColumnFamilyData* cfd;
167 cfd = nullptr;
168 edit.AddColumnFamily(cf_name);
169
170 mutex_.Lock();
171 std::unique_ptr<FSDirectory> db_dir;
172 Status status = env_->GetFileSystem()->NewDirectory(dbname_, IOOptions(),
173 &db_dir, nullptr);
174 if (status.ok()) {
175 status = vset_.LogAndApply(cfd, mut_cf_opts, read_options, write_options,
176 &edit, &mutex_, db_dir.get(),
177 false /* new_descriptor_log */, cf_opts);
178 }
179 mutex_.Unlock();
180 return status;
181 }

197 Status Run() {
198 Status status = env_->LockFile(LockFileName(dbname_), &db_lock_);
199 if (!status.ok()) {
200 return status;
201 }
202 status = FindFiles();
203 DBImpl* db_impl = nullptr;
204 if (status.ok()) {
205 // Discard older manifests and start a fresh one
206 for (size_t i = 0; i < manifests_.size(); i++) {
207 ArchiveFile(dbname_ + "/" + manifests_[i]);
208 }
209 // Just create a DBImpl temporarily so we can reuse NewDB()
210 db_impl = new DBImpl(db_options_, dbname_);
211 status = db_impl->NewDB(/*new_filenames=*/nullptr);
212 }
213 delete db_impl;
214
215 if (status.ok()) {
216 // Recover using the fresh manifest created by NewDB()
217 status =
218 vset_.Recover({{kDefaultColumnFamilyName, default_cf_opts_}}, false);
219 }
220 if (status.ok()) {
221 // Need to scan existing SST files first so the column families are
222 // created before we process WAL files
223 ExtractMetaData();
224
225 // ExtractMetaData() uses table_fds_ to know which SST files' metadata to
226 // extract -- we need to clear it here since metadata for existing SST
227 // files has been extracted already
228 table_fds_.clear();
229 ConvertLogFilesToTables();
230 ExtractMetaData();
231 status = AddTables();
232 }
233 if (status.ok()) {
234 uint64_t bytes = 0;
235 for (size_t i = 0; i < tables_.size(); i++) {
236 bytes += tables_[i].meta.fd.GetFileSize();
237 }
238 ROCKS_LOG_WARN(db_options_.info_log,
239 "**** Repaired rocksdb %s; "
240 "recovered %" ROCKSDB_PRIszt " files; %" PRIu64
241 " bytes. "
242 "Some data may have been lost. "
243 "****",
244 dbname_.c_str(), tables_.size(), bytes);
245 }
246 return status;
247 }

286 Status FindFiles() {
287 std::vector<std::string> filenames;
288 bool found_file = false;
289 std::vector<std::string> to_search_paths;
290
291 for (size_t path_id = 0; path_id < db_options_.db_paths.size(); path_id++) {
292 to_search_paths.push_back(db_options_.db_paths[path_id].path);
293 }
294
295 // search wal_dir if user uses a customize wal_dir
296 bool same = immutable_db_options_.IsWalDirSameAsDBPath(dbname_);
297 if (!same) {
298 to_search_paths.push_back(immutable_db_options_.wal_dir);
299 }
300
301 for (size_t path_id = 0; path_id < to_search_paths.size(); path_id++) {
302 ROCKS_LOG_INFO(db_options_.info_log, "Searching path %s\n",
303 to_search_paths[path_id].c_str());
304 Status status = env_->GetChildren(to_search_paths[path_id], &filenames);
305 if (!status.ok()) {
306 return status;
307 }
308 if (!filenames.empty()) {
309 found_file = true;
310 }
311
312 uint64_t number;
313 FileType type;
314 for (size_t i = 0; i < filenames.size(); i++) {
315 if (ParseFileName(filenames[i], &number, &type)) {
316 if (type == kDescriptorFile) {
317 manifests_.push_back(filenames[i]);
318 } else {
319 if (number + 1 > next_file_number_) {
320 next_file_number_ = number + 1;
321 }
322 if (type == kWalFile) {
323 logs_.push_back(number);
324 } else if (type == kTableFile) {
325 table_fds_.emplace_back(number, static_cast<uint32_t>(path_id),
326 0);
327 } else {
328 // Ignore other files
329 }
330 }
331 }
332 }
333 }
334 if (!found_file) {
335 return Status::Corruption(dbname_, "repair found no files");
336 }
337 return Status::OK();
338 }

341 const auto& wal_dir = immutable_db_options_.GetWalDir();
342 for (size_t i = 0; i < logs_.size(); i++) {
343 // we should use LogFileName(wal_dir, logs_[i]) here. user might uses
344 // wal_dir option.
345 std::string logname = LogFileName(wal_dir, logs_[i]);
346 Status status = ConvertLogToTable(wal_dir, logs_[i]);
347 if (!status.ok()) {
348 ROCKS_LOG_WARN(db_options_.info_log,

356 Status ConvertLogToTable(const std::string& wal_dir, uint64_t log) {
357 struct LogReporter : public log::Reader::Reporter {
358 Env* env;
359 std::shared_ptr<Logger> info_log;
360 uint64_t lognum;
361 void Corruption(size_t bytes, const Status& s) override {
362 // We print error messages for corruption, but continue repairing.
363 ROCKS_LOG_ERROR(info_log, "Log #%" PRIu64 ": dropping %d bytes; %s",
364 lognum, static_cast<int>(bytes), s.ToString().c_str());
365 }
366 };
367
368 // Open the log file
369 std::string logname = LogFileName(wal_dir, log);
370 const auto& fs = env_->GetFileSystem();
371 std::unique_ptr<SequentialFileReader> lfile_reader;
372 Status status = SequentialFileReader::Create(
373 fs, logname, fs->OptimizeForLogRead(file_options_), &lfile_reader,
374 nullptr /* dbg */, nullptr /* rate limiter */);
375 if (!status.ok()) {
376 return status;
377 }
378
379 // Create the log reader.
380 LogReporter reporter;
381 reporter.env = env_;
382 reporter.info_log = db_options_.info_log;
383 reporter.lognum = log;
384 // We intentionally make log::Reader do checksumming so that
385 // corruptions cause entire commits to be skipped instead of
386 // propagating bad information (like overly large sequence
387 // numbers).
388 log::Reader reader(db_options_.info_log, std::move(lfile_reader), &reporter,
389 true /*enable checksum*/, log);
390
391 // Initialize per-column family memtables
392 for (auto* cfd : *vset_.GetColumnFamilySet()) {
393 cfd->CreateNewMemtable(*cfd->GetLatestMutableCFOptions(),
394 kMaxSequenceNumber);
395 }
396 auto cf_mems = new ColumnFamilyMemTablesImpl(vset_.GetColumnFamilySet());
397
398 // Read all the records and add to a memtable
399 const UnorderedMap<uint32_t, size_t>& running_ts_sz =
400 vset_.GetRunningColumnFamiliesTimestampSize();
401 std::string scratch;
402 Slice record;
403 WriteBatch batch;
404
405 int counter = 0;
406 while (reader.ReadRecord(&record, &scratch)) {
407 if (record.size() < WriteBatchInternal::kHeader) {
408 reporter.Corruption(record.size(),
409 Status::Corruption("log record too small"));
410 continue;
411 }
412 Status record_status = WriteBatchInternal::SetContents(&batch, record);
413 if (record_status.ok()) {
414 const UnorderedMap<uint32_t, size_t>& record_ts_sz =
415 reader.GetRecordedTimestampSize();
416 // Use same value for `seq_per_batch` and `batch_per_txn` as
417 // WriteBatchInternal::InsertInto does below.
418 record_status = HandleWriteBatchTimestampSizeDifference(
419 &batch, running_ts_sz, record_ts_sz,
420 TimestampSizeConsistencyMode::kVerifyConsistency,
421 /* seq_per_batch */ false, /* batch_per_txn */ true);
422 if (record_status.ok()) {
423 record_status =
424 WriteBatchInternal::InsertInto(&batch, cf_mems, nullptr, nullptr);
425 }
426 }
427 if (record_status.ok()) {
428 counter += WriteBatchInternal::Count(&batch);
429 } else {
430 ROCKS_LOG_WARN(db_options_.info_log, "Log #%" PRIu64 ": ignoring %s",
431 log, record_status.ToString().c_str());
432 }
433 }
434
435 // Dump a table for each column family with entries in this log file.
436 for (auto* cfd : *vset_.GetColumnFamilySet()) {
437 // Do not record a version edit for this conversion to a Table
438 // since ExtractMetaData() will also generate edits.
439 MemTable* mem = cfd->mem();
440 if (mem->IsEmpty()) {
441 continue;
442 }
443
444 FileMetaData meta;
445 meta.fd = FileDescriptor(next_file_number_++, 0, 0);
446 // TODO: plumb Env::IOActivity, Env::IOPriority
447 ReadOptions ro;
448 ro.total_order_seek = true;
449 Arena arena;
450 ScopedArenaPtr<InternalIterator> iter(
451 mem->NewIterator(ro, /*seqno_to_time_mapping=*/nullptr, &arena,
452 /*prefix_extractor=*/nullptr, /*for_flush=*/true));
453 int64_t _current_time = 0;
454 immutable_db_options_.clock->GetCurrentTime(&_current_time)
455 .PermitUncheckedError(); // ignore error
456 const uint64_t current_time = static_cast<uint64_t>(_current_time);
457 meta.file_creation_time = current_time;
458 SnapshotChecker* snapshot_checker = DisableGCSnapshotChecker::Instance();
459
460 auto write_hint =
461 cfd->current()->storage_info()->CalculateSSTWriteHint(/*level=*/0);
462 std::vector<std::unique_ptr<FragmentedRangeTombstoneIterator>>
463 range_del_iters;
464 auto range_del_iter = mem->NewRangeTombstoneIterator(
465 ro, kMaxSequenceNumber, false /* immutable_memtable */);
466 if (range_del_iter != nullptr) {
467 range_del_iters.emplace_back(range_del_iter);
468 }
469
470 IOStatus io_s;
471 CompressionOptions default_compression;
472 // TODO: plumb Env::IOActivity, Env::IOPriority
473 const ReadOptions read_options;
474 const WriteOptions write_option(Env::IO_HIGH);
475 TableBuilderOptions tboptions(
476 *cfd->ioptions(), *cfd->GetLatestMutableCFOptions(), read_options,
477 write_option, cfd->internal_comparator(),
478 cfd->internal_tbl_prop_coll_factories(), kNoCompression,
479 default_compression, cfd->GetID(), cfd->GetName(), -1 /* level */,
480 current_time /* newest_key_time */, false /* is_bottommost */,
481 TableFileCreationReason::kRecovery, 0 /* oldest_key_time */,
482 0 /* file_creation_time */, "DB Repairer" /* db_id */, db_session_id_,
483 0 /*target_file_size*/, meta.fd.GetNumber());
484
485 SeqnoToTimeMapping empty_seqno_to_time_mapping;
486 status = BuildTable(
487 dbname_, /* versions */ nullptr, immutable_db_options_, tboptions,
488 file_options_, table_cache_.get(), iter.get(),
489 std::move(range_del_iters), &meta, nullptr /* blob_file_additions */,
490 {}, kMaxSequenceNumber, kMaxSequenceNumber, kMaxSequenceNumber,
491 snapshot_checker, false /* paranoid_file_checks*/,
492 nullptr /* internal_stats */, &io_s, nullptr /*IOTracer*/,
493 BlobFileCreationReason::kRecovery,
494 nullptr /* seqno_to_time_mapping */, nullptr /* event_logger */,
495 0 /* job_id */, nullptr /* table_properties */, write_hint);
496 ROCKS_LOG_INFO(db_options_.info_log,
497 "Log #%" PRIu64 ": %d ops saved to Table #%" PRIu64 " %s",
498 log, counter, meta.fd.GetNumber(),
499 status.ToString().c_str());
500 if (status.ok()) {
501 if (meta.fd.GetFileSize() > 0) {
502 table_fds_.push_back(meta.fd);
503 }
504 } else {
505 break;
506 }
507 }
508 delete cf_mems;
509 return status;
510 }

532 Status ScanTable(TableInfo* t) {
533 std::string fname = TableFileName(
534 db_options_.db_paths, t->meta.fd.GetNumber(), t->meta.fd.GetPathId());
535 int counter = 0;
536 uint64_t file_size;
537 Status status = env_->GetFileSize(fname, &file_size);
538 t->meta.fd = FileDescriptor(t->meta.fd.GetNumber(), t->meta.fd.GetPathId(),
539 file_size);
540 std::shared_ptr<const TableProperties> props;
541 if (status.ok()) {
542 // TODO: plumb Env::IOActivity, Env::IOPriority
543 const ReadOptions read_options;
544 status = table_cache_->GetTableProperties(
545 file_options_, read_options, icmp_, t->meta, &props, default_mopts_);
546 }
547 if (status.ok()) {
548 auto s =
549 GetSstInternalUniqueId(props->db_id, props->db_session_id,
550 props->orig_file_number, &t->meta.unique_id);
551 if (!s.ok()) {
552 ROCKS_LOG_WARN(db_options_.info_log,
553 "Table #%" PRIu64
554 ": unable to get unique id, default to Unknown.",
555 t->meta.fd.GetNumber());
556 }
557 t->column_family_id = static_cast<uint32_t>(props->column_family_id);
558 if (t->column_family_id ==
559 TablePropertiesCollectorFactory::Context::kUnknownColumnFamily) {
560 ROCKS_LOG_WARN(
561 db_options_.info_log,
562 "Table #%" PRIu64
563 ": column family unknown (probably due to legacy format); "
564 "adding to default column family id 0.",
565 t->meta.fd.GetNumber());
566 t->column_family_id = 0;
567 }
568
569 if (vset_.GetColumnFamilySet()->GetColumnFamily(t->column_family_id) ==
570 nullptr) {
571 status =
572 AddColumnFamily(props->column_family_name, t->column_family_id);
573 }
574 t->meta.oldest_ancester_time = props->creation_time;
575 t->meta.user_defined_timestamps_persisted =
576 static_cast<bool>(props->user_defined_timestamps_persisted);
577 }
578 if (status.ok()) {
579 uint64_t tail_size = 0;
580 bool contain_no_data_blocks =
581 props->num_entries > 0 &&
582 (props->num_entries == props->num_range_deletions);
583 if (props->tail_start_offset > 0 || contain_no_data_blocks) {
584 assert(props->tail_start_offset <= file_size);
585 tail_size = file_size - props->tail_start_offset;
586 }
587 t->meta.tail_size = tail_size;
588 }
589 ColumnFamilyData* cfd = nullptr;
590 if (status.ok()) {
591 cfd = vset_.GetColumnFamilySet()->GetColumnFamily(t->column_family_id);
592 if (cfd->GetName() != props->column_family_name) {
593 ROCKS_LOG_ERROR(
594 db_options_.info_log,
595 "Table #%" PRIu64
596 ": inconsistent column family name '%s'; expected '%s' for column "
597 "family id %" PRIu32 ".",
598 t->meta.fd.GetNumber(), props->column_family_name.c_str(),
599 cfd->GetName().c_str(), t->column_family_id);
600 status = Status::Corruption(dbname_, "inconsistent column family name");
601 }
602 }
603 if (status.ok()) {
604 // TODO: plumb Env::IOActivity, Env::IOPriority
605 ReadOptions ropts;
606 ropts.total_order_seek = true;
607 InternalIterator* iter = table_cache_->NewIterator(
608 ropts, file_options_, cfd->internal_comparator(), t->meta,
609 nullptr /* range_del_agg */, *cfd->GetLatestMutableCFOptions(),
610 /*table_reader_ptr=*/nullptr, /*file_read_hist=*/nullptr,
611 TableReaderCaller::kRepair, /*arena=*/nullptr, /*skip_filters=*/false,
612 /*level=*/-1, /*max_file_size_for_l0_meta_pin=*/0,
613 /*smallest_compaction_key=*/nullptr,
614 /*largest_compaction_key=*/nullptr,
615 /*allow_unprepared_value=*/false);
616 ParsedInternalKey parsed;
617 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
618 Slice key = iter->key();
619 Status pik_status =
620 ParseInternalKey(key, &parsed, db_options_.allow_data_in_errors);
621 if (!pik_status.ok()) {
622 ROCKS_LOG_ERROR(db_options_.info_log,
623 "Table #%" PRIu64 ": unparsable key - %s",
624 t->meta.fd.GetNumber(), pik_status.getState());
625 continue;
626 }
627
628 counter++;
629
630 status = t->meta.UpdateBoundaries(key, iter->value(), parsed.sequence,
631 parsed.type);
632 if (!status.ok()) {
633 break;
634 }
635 }
636 if (status.ok() && !iter->status().ok()) {
637 status = iter->status();
638 }
639 delete iter;
640
641 ROCKS_LOG_INFO(db_options_.info_log, "Table #%" PRIu64 ": %d entries %s",
642 t->meta.fd.GetNumber(), counter,
643 status.ToString().c_str());
644 }
645 if (status.ok()) {
646 // XXX/FIXME: This is just basic, naive handling of range tombstones,
647 // like call to UpdateBoundariesForRange in builder.cc where we assume
648 // an SST file is a full sorted run. This probably needs the extra logic
649 // from compaction_job.cc around call to UpdateBoundariesForRange (to
650 // handle range tombstones extendingg beyond range of other entries).
651 // TODO: plumb Env::IOActivity, Env::IOPriority
652 ReadOptions ropts;
653 std::unique_ptr<FragmentedRangeTombstoneIterator> r_iter;
654 status = table_cache_->GetRangeTombstoneIterator(
655 ropts, cfd->internal_comparator(), t->meta,
656 *cfd->GetLatestMutableCFOptions(), &r_iter);
657
658 if (r_iter) {
659 r_iter->SeekToFirst();
660
661 while (r_iter->Valid()) {
662 auto tombstone = r_iter->Tombstone();
663 auto kv = tombstone.Serialize();
664 t->meta.UpdateBoundariesForRange(
665 kv.first, tombstone.SerializeEndKey(), tombstone.seq_,
666 cfd->internal_comparator());
667 r_iter->Next();
668 }
669 }
670 }
671 return status;
672 }

674 Status AddTables() {
675 // TODO: plumb Env::IOActivity, Env::IOPriority;
676 const ReadOptions read_options;
677 const WriteOptions write_options;
678
679 std::unordered_map<uint32_t, std::vector<const TableInfo*>> cf_id_to_tables;
680 SequenceNumber max_sequence = 0;
681 for (size_t i = 0; i < tables_.size(); i++) {
682 cf_id_to_tables[tables_[i].column_family_id].push_back(&tables_[i]);
683 if (max_sequence < tables_[i].meta.fd.largest_seqno) {
684 max_sequence = tables_[i].meta.fd.largest_seqno;
685 }
686 }
687 vset_.SetLastAllocatedSequence(max_sequence);
688 vset_.SetLastPublishedSequence(max_sequence);
689 vset_.SetLastSequence(max_sequence);
690
691 for (const auto& cf_id_and_tables : cf_id_to_tables) {
692 auto* cfd =
693 vset_.GetColumnFamilySet()->GetColumnFamily(cf_id_and_tables.first);
694
695 // Recover files' epoch number using dummy VersionStorageInfo
696 VersionBuilder dummy_version_builder(
697 cfd->current()->version_set()->file_options(), cfd->ioptions(),
698 cfd->table_cache(), cfd->current()->storage_info(),
699 cfd->current()->version_set(),
700 cfd->GetFileMetadataCacheReservationManager());
701 VersionStorageInfo dummy_vstorage(
702 &cfd->internal_comparator(), cfd->user_comparator(),
703 cfd->NumberLevels(), cfd->ioptions()->compaction_style,
704 nullptr /* src_vstorage */, cfd->ioptions()->force_consistency_checks,
705 EpochNumberRequirement::kMightMissing, cfd->ioptions()->clock,
706 /*bottommost_file_compaction_delay=*/0,
707 cfd->current()->version_set()->offpeak_time_option());
708 Status s;
709 VersionEdit dummy_edit;
710 for (const auto* table : cf_id_and_tables.second) {
711 // TODO(opt): separate out into multiple levels
712 dummy_edit.AddFile(
713 0, table->meta.fd.GetNumber(), table->meta.fd.GetPathId(),
714 table->meta.fd.GetFileSize(), table->meta.smallest,
715 table->meta.largest, table->meta.fd.smallest_seqno,
716 table->meta.fd.largest_seqno, table->meta.marked_for_compaction,
717 table->meta.temperature, table->meta.oldest_blob_file_number,
718 table->meta.oldest_ancester_time, table->meta.file_creation_time,
719 table->meta.epoch_number, table->meta.file_checksum,
720 table->meta.file_checksum_func_name, table->meta.unique_id,
721 table->meta.compensated_range_deletion_size, table->meta.tail_size,
722 table->meta.user_defined_timestamps_persisted);
723 }
724 s = dummy_version_builder.Apply(&dummy_edit);
725 if (s.ok()) {
726 s = dummy_version_builder.SaveTo(&dummy_vstorage);
727 }
728 if (s.ok()) {
729 dummy_vstorage.RecoverEpochNumbers(cfd);
730 }
731 if (s.ok()) {
732 // Record changes from this repair in VersionEdit, including files with
733 // recovered epoch numbers
734 VersionEdit edit;
735 edit.SetComparatorName(cfd->user_comparator()->Name());
736 edit.SetPersistUserDefinedTimestamps(
737 cfd->ioptions()->persist_user_defined_timestamps);
738 edit.SetLogNumber(0);
739 edit.SetNextFile(next_file_number_);
740 edit.SetColumnFamily(cfd->GetID());
741 for (int level = 0; level < dummy_vstorage.num_levels(); ++level) {
742 for (FileMetaData* file_meta : dummy_vstorage.LevelFiles(level)) {
743 edit.AddFile(level, *file_meta);
744 }
745 }
746
747 // Release resources occupied by the dummy VersionStorageInfo
748 for (int level = 0; level < dummy_vstorage.num_levels(); ++level) {
749 for (FileMetaData* file_meta : dummy_vstorage.LevelFiles(level)) {
750 file_meta->refs--;
751 if (file_meta->refs <= 0) {
752 delete file_meta;
753 }
754 }
755 }
756
757 // Persist record of changes
758 assert(next_file_number_ > 0);
759 vset_.MarkFileNumberUsed(next_file_number_ - 1);
760 mutex_.Lock();
761 std::unique_ptr<FSDirectory> db_dir;
762 s = env_->GetFileSystem()->NewDirectory(dbname_, IOOptions(), &db_dir,
763 nullptr);
764 if (s.ok()) {
765 s = vset_.LogAndApply(cfd, *cfd->GetLatestMutableCFOptions(),
766 read_options, write_options, &edit, &mutex_,
767 db_dir.get(), false /* new_descriptor_log */);
768 }
769 mutex_.Unlock();
770 }
771 if (!s.ok()) {
772 return s;
773 }
774 }
775 return Status::OK();
776 }
