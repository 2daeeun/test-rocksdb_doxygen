124Status GetPlainTableOptionsFromString(const ConfigOptions& config_options,
125 const PlainTableOptions& table_options,
126 const std::string& opts_str,
127 PlainTableOptions* new_table_options) {
128 std::unordered_map<std::string, std::string> opts_map;
129 Status s = StringToMap(opts_str, &opts_map);
130 if (!s.ok()) {
131 return s;
132 }
133
134 s = GetPlainTableOptionsFromMap(config_options, table_options, opts_map,
135 new_table_options);
136 // Translate any errors (NotFound, NotSupported, to InvalidArgument
137 if (s.ok() || s.IsInvalidArgument()) {
138 return s;
139 } else {
140 return Status::InvalidArgument(s.getState());
141 }

144static int RegisterBuiltinMemTableRepFactory(ObjectLibrary& library,
145 const std::string& /*arg*/) {
146 // The MemTableRepFactory built-in classes will be either a class
147 // (VectorRepFactory) or a nickname (vector), followed optionally by ":#",
148 // where # is the "size" of the factory.
149 auto AsPattern = [](const std::string& name, const std::string& alt) {
150 auto pattern = ObjectLibrary::PatternEntry(name, true);
151 pattern.AnotherName(alt);
152 pattern.AddNumber(":");
153 return pattern;
154 };
155 library.AddFactory<MemTableRepFactory>(
156 AsPattern(VectorRepFactory::kClassName(), VectorRepFactory::kNickName()),
157 [](const std::string& uri, std::unique_ptr<MemTableRepFactory>* guard,
158 std::string* /*errmsg*/) {
159 auto colon = uri.find(':');
160 if (colon != std::string::npos) {
161 size_t count = ParseSizeT(uri.substr(colon + 1));
162 guard->reset(new VectorRepFactory(count));
163 } else {
164 guard->reset(new VectorRepFactory());
165 }
166 return guard->get();
167 });
168 library.AddFactory<MemTableRepFactory>(
169 AsPattern(SkipListFactory::kClassName(), SkipListFactory::kNickName()),
170 [](const std::string& uri, std::unique_ptr<MemTableRepFactory>* guard,
171 std::string* /*errmsg*/) {
172 auto colon = uri.find(':');
173 if (colon != std::string::npos) {
174 size_t lookahead = ParseSizeT(uri.substr(colon + 1));
175 guard->reset(new SkipListFactory(lookahead));
176 } else {
177 guard->reset(new SkipListFactory());
178 }
179 return guard->get();
180 });
181 library.AddFactory<MemTableRepFactory>(
182 AsPattern("HashLinkListRepFactory", "hash_linkedlist"),
183 [](const std::string& uri, std::unique_ptr<MemTableRepFactory>* guard,
184 std::string* /*errmsg*/) {
185 // Expecting format: hash_linkedlist:<hash_bucket_count>
186 auto colon = uri.find(':');
187 if (colon != std::string::npos) {
188 size_t hash_bucket_count = ParseSizeT(uri.substr(colon + 1));
189 guard->reset(NewHashLinkListRepFactory(hash_bucket_count));
190 } else {
191 guard->reset(NewHashLinkListRepFactory());
192 }
193 return guard->get();
194 });
195 library.AddFactory<MemTableRepFactory>(
196 AsPattern("HashSkipListRepFactory", "prefix_hash"),
197 [](const std::string& uri, std::unique_ptr<MemTableRepFactory>* guard,
198 std::string* /*errmsg*/) {
199 // Expecting format: prefix_hash:<hash_bucket_count>
200 auto colon = uri.find(':');
201 if (colon != std::string::npos) {
202 size_t hash_bucket_count = ParseSizeT(uri.substr(colon + 1));
203 guard->reset(NewHashSkipListRepFactory(hash_bucket_count));
204 } else {
205 guard->reset(NewHashSkipListRepFactory());
206 }
207 return guard->get();
208 });
209 library.AddFactory<MemTableRepFactory>(
210 "cuckoo",
211 [](const std::string& /*uri*/,
212 std::unique_ptr<MemTableRepFactory>* /*guard*/, std::string* errmsg) {
213 *errmsg = "cuckoo hash memtable is not supported anymore.";
214 return nullptr;
215 });
216
217 size_t num_types;
218 return static_cast<int>(library.GetFactoryCount(&num_types));
219}

229Status MemTableRepFactory::CreateFromString(
230 const ConfigOptions& config_options, const std::string& value,
231 std::unique_ptr<MemTableRepFactory>* result) {
232 static std::once_flag once;
233 std::call_once(once, [&]() {
234 RegisterBuiltinMemTableRepFactory(*(ObjectLibrary::Default().get()), "");
235 });
236 std::string id;
237 std::unordered_map<std::string, std::string> opt_map;
238 Status status = Customizable::GetOptionsMap(config_options, result->get(),
239 value, &id, &opt_map);
240 if (!status.ok()) { // GetOptionsMap failed
241 return status;
242 } else if (value.empty()) {
243 // No Id and no options. Clear the object
244 result->reset();
245 return Status::OK();
246 } else if (id.empty()) { // We have no Id but have options. Not good
247 return Status::NotSupported("Cannot reset object ", id);
248 } else {
249 status = NewUniqueObject<MemTableRepFactory>(config_options, id, opt_map,
250 result);
251 }
252 return status;
253}

255Status MemTableRepFactory::CreateFromString(
256 const ConfigOptions& config_options, const std::string& value,
257 std::shared_ptr<MemTableRepFactory>* result) {
258 std::unique_ptr<MemTableRepFactory> factory;
259 Status s = CreateFromString(config_options, value, &factory);
260 if (factory && s.ok()) {
261 result->reset(factory.release());
262 }
263 return s;
264}

266Status GetPlainTableOptionsFromMap(
267 const ConfigOptions& config_options, const PlainTableOptions& table_options,
268 const std::unordered_map<std::string, std::string>& opts_map,
269 PlainTableOptions* new_table_options) {
270 assert(new_table_options);
271 PlainTableFactory ptf(table_options);
272 Status s = ptf.ConfigureFromMap(config_options, opts_map);
273 if (s.ok()) {
274 *new_table_options = *(ptf.GetOptions<PlainTableOptions>());
275 } else {
276 // Restore "new_options" to the default "base_options".
277 *new_table_options = table_options;
278 }
279 return s;
280}
