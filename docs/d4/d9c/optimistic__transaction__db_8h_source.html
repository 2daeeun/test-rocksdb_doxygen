34enum class OccValidationPolicy {
35 // Validate serially at commit stage, AFTER entering the write-group.
36 // Isolation validation is processed single-threaded(since in the
37 // write-group).
38 // May suffer from high mutex contention, as per this link:
39 // https://github.com/facebook/rocksdb/issues/4402
40 kValidateSerial = 0,
41 // Validate parallelly before commit stage, BEFORE entering the write-group to
42 // reduce mutex contention. Each txn acquires locks for its write-set
43 // records in some well-defined order.
44 kValidateParallel = 1
45};

60// An object for sharing a pool of locks across DB instances.

68struct OptimisticTransactionDBOptions {
69 OccValidationPolicy validate_policy = OccValidationPolicy::kValidateParallel;
70
71 // Number of striped/bucketed mutex locks for validating transactions.
72 // Used on only if validate_policy == OccValidationPolicy::kValidateParallel
73 // and shared_lock_buckets (below) is empty. Larger number potentially
74 // reduces contention but uses more memory.
75 uint32_t occ_lock_buckets = (1 << 20);
76
77 // A pool of mutex locks for validating transactions. Can be shared among
78 // DBs. Ignored if validate_policy != OccValidationPolicy::kValidateParallel.
79 // If empty and validate_policy == OccValidationPolicy::kValidateParallel,
80 // an OccLockBuckets will be created using the count in occ_lock_buckets.
81 // See MakeSharedOccLockBuckets()
82 std::shared_ptr<OccLockBuckets> shared_lock_buckets;
83};
