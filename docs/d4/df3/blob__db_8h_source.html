22// A wrapped database which puts values of KV pairs in a separate log
23// and store location to the log in the underlying DB.
24//
25// The factory needs to be moved to include/rocksdb/utilities to allow
26// users to use blob DB.
27
28constexpr uint64_t kNoExpiration = std::numeric_limits<uint64_t>::max();
29
30struct BlobDBOptions {
31 // Name of the directory under the base DB where blobs will be stored. Using
32 // a directory where the base DB stores its SST files is not supported.
33 // Default is "blob_dir"
34 std::string blob_dir = "blob_dir";
35
36 // whether the blob_dir path is relative or absolute.
37 bool path_relative = true;
38
39 // When max_db_size is reached, evict blob files to free up space
40 // instead of returnning NoSpace error on write. Blob files will be
41 // evicted from oldest to newest, based on file creation time.
42 bool is_fifo = false;
43
44 // Maximum size of the database (including SST files and blob files).
45 //
46 // Default: 0 (no limits)
47 uint64_t max_db_size = 0;
48
49 // a new bucket is opened, for ttl_range. So if ttl_range is 600seconds
50 // (10 minutes), and the first bucket starts at 1471542000
51 // then the blob buckets will be
52 // first bucket is 1471542000 - 1471542600
53 // second bucket is 1471542600 - 1471543200
54 // and so on
55 uint64_t ttl_range_secs = 3600;
56
57 // The smallest value to store in blob log. Values smaller than this threshold
58 // will be inlined in base DB together with the key.
59 uint64_t min_blob_size = 0;
60
61 // Allows OS to incrementally sync blob files to disk for every
62 // bytes_per_sync bytes written. Users shouldn't rely on it for
63 // persistency guarantee.
64 uint64_t bytes_per_sync = 512 * 1024;
65
66 // the target size of each blob file. File will become immutable
67 // after it exceeds that size
68 uint64_t blob_file_size = 256 * 1024 * 1024;
69
70 // what compression to use for Blob's
71 CompressionType compression = kNoCompression;
72
73 // If enabled, BlobDB cleans up stale blobs in non-TTL files during compaction
74 // by rewriting the remaining live blobs to new files.
75 bool enable_garbage_collection = false;
76
77 // The cutoff in terms of blob file age for garbage collection. Blobs in
78 // the oldest N non-TTL blob files will be rewritten when encountered during
79 // compaction, where N = garbage_collection_cutoff * number_of_non_TTL_files.
80 double garbage_collection_cutoff = 0.25;
81
82 // Disable all background job. Used for test only.
83 bool disable_background_tasks = false;
84
85 void Dump(Logger* log) const;
86};
