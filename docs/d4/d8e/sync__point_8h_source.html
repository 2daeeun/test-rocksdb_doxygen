65// This class provides facility to reproduce race conditions deterministically
66// in unit tests.
67// Developer could specify sync points in the codebase via TEST_SYNC_POINT.
68// Each sync point represents a position in the execution stream of a thread.
69// In the unit test, 'Happens After' relationship among sync points could be
70// setup via SyncPoint::LoadDependency, to reproduce a desired interleave of
71// threads execution.
72// Refer to (DBTest,TransactionLogIteratorRace), for an example use case.
73
74class SyncPoint {
75 public:
76 static SyncPoint* GetInstance();
77
78 SyncPoint(const SyncPoint&) = delete;
79 SyncPoint& operator=(const SyncPoint&) = delete;
80 ~SyncPoint();
81
82 struct SyncPointPair {
83 std::string predecessor;
84 std::string successor;
85 };
86
87 // call once at the beginning of a test to setup the dependency between
88 // sync points. Specifically, execution will not be allowed to proceed past
89 // each successor until execution has reached the corresponding predecessor,
90 // in any thread.
91 void LoadDependency(const std::vector<SyncPointPair>& dependencies);
92
93 // call once at the beginning of a test to setup the dependency between
94 // sync points and setup markers indicating the successor is only enabled
95 // when it is processed on the same thread as the predecessor.
96 // When adding a marker, it implicitly adds a dependency for the marker pair.
97 void LoadDependencyAndMarkers(const std::vector<SyncPointPair>& dependencies,
98 const std::vector<SyncPointPair>& markers);
99
100 // The argument to the callback is passed through from
101 // TEST_SYNC_POINT_CALLBACK(); nullptr if TEST_SYNC_POINT or
102 // TEST_IDX_SYNC_POINT was used.
103 void SetCallBack(const std::string& point,
104 const std::function<void(void*)>& callback);
105
106 // Clear callback function by point
107 void ClearCallBack(const std::string& point);
108
109 // Clear all call back functions.
110 void ClearAllCallBacks();
111
112 // enable sync point processing (disabled on startup)
113 void EnableProcessing();
114
115 // disable sync point processing
116 void DisableProcessing();
117
118 // remove the execution trace of all sync points
119 void ClearTrace();
120
121 // triggered by TEST_SYNC_POINT, blocking execution until all predecessors
122 // are executed.
123 // And/or call registered callback function, with argument `cb_arg`
124 void Process(const Slice& point, void* cb_arg = nullptr);
125
126 // template gets length of const string at compile time,
127 // avoiding strlen() at runtime
128 template <size_t kLen>
129 void Process(const char (&point)[kLen], void* cb_arg = nullptr) {
130 static_assert(kLen > 0, "Must not be empty");
131 assert(point[kLen - 1] == '\0');
132 Process(Slice(point, kLen - 1), cb_arg);
133 }
134
135 // TODO: it might be useful to provide a function that blocks until all

138 // We want this to be public so we can
139 // subclass the implementation
140 struct Data;
141
142 private:
143 // Singleton
144 SyncPoint();
145 Data* impl_;
146};

153// Use TEST_SYNC_POINT to specify sync points inside code base.
154// Sync points can have happens-after dependency on other sync points,
155// configured at runtime via SyncPoint::LoadDependency. This could be
156// utilized to re-produce race conditions between threads.
157// See TransactionLogIteratorRace in db_test.cc for an example use case.
158// TEST_SYNC_POINT is no op in release build.
159#define TEST_SYNC_POINT(x) \
160 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->Process(x)
161#define TEST_IDX_SYNC_POINT(x, index) \
162 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->Process(x + \
163 std::to_string(index))
164#define TEST_SYNC_POINT_CALLBACK(x, y) \
165 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->Process(x, y)
166#define INIT_SYNC_POINT_SINGLETONS() \
167 (void)ROCKSDB_NAMESPACE::SyncPoint::GetInstance();
168#endif // NDEBUG
