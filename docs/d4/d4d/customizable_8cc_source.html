19std::string Customizable::GetOptionName(const std::string& long_name) const {
20 const std::string& name = Name();
21 size_t name_len = name.size();
22 if (long_name.size() > name_len + 1 &&
23 long_name.compare(0, name_len, name) == 0 &&
24 long_name.at(name_len) == '.') {
25 return long_name.substr(name_len + 1);
26 } else {
27 return Configurable::GetOptionName(long_name);
28 }
29}

49std::string Customizable::SerializeOptions(const ConfigOptions& config_options,
50 const std::string& prefix) const {
51 std::string result;
52 std::string parent;
53 std::string id = GetId();
54 if (!config_options.IsShallow() && !id.empty()) {
55 parent = Configurable::SerializeOptions(config_options, "");
56 }
57 if (parent.empty()) {
58 result = id;
59 } else {
60 result.append(prefix);
61 result.append(OptionTypeInfo::kIdPropName());
62 result.append("=");
63 result.append(id);
64 result.append(config_options.delimiter);
65 result.append(parent);
66 }
67 return result;
68}

71bool Customizable::AreEquivalent(const ConfigOptions& config_options,
72 const Configurable* other,
73 std::string* mismatch) const {
74 if (config_options.sanity_level > ConfigOptions::kSanityLevelNone &&
75 this != other) {
76 const Customizable* custom = static_cast<const Customizable*>(other);
77 if (custom == nullptr) { // Cast failed
78 return false;
79 } else if (GetId() != custom->GetId()) {
80 *mismatch = OptionTypeInfo::kIdPropName();
81 return false;
82 } else if (config_options.sanity_level >
83 ConfigOptions::kSanityLevelLooselyCompatible) {
84 bool matches =
85 Configurable::AreEquivalent(config_options, other, mismatch);
86 return matches;
87 }
88 }
89 return true;
90}

92Status Customizable::GetOptionsMap(
93 const ConfigOptions& config_options, const Customizable* customizable,
94 const std::string& value, std::string* id,
95 std::unordered_map<std::string, std::string>* props) {
96 Status status;
97 if (value.empty() || value == kNullptrString) {
98 *id = "";
99 props->clear();
100 } else if (customizable != nullptr) {
101 status =
102 Configurable::GetOptionsMap(value, customizable->GetId(), id, props);
103 if (status.ok() && customizable->IsInstanceOf(*id)) {
104 // The new ID and the old ID match, so the objects are the same type.
105 // Try to get the existing options, ignoring any errors
106 ConfigOptions embedded = config_options;
107 embedded.delimiter = ";";
108 std::string curr_opts;
109 if (customizable->GetOptionString(embedded, &curr_opts).ok()) {
110 std::unordered_map<std::string, std::string> curr_props;
111 if (StringToMap(curr_opts, &curr_props).ok()) {
112 props->insert(curr_props.begin(), curr_props.end());
113 }
114 }
115 }
116 } else {
117 status = Configurable::GetOptionsMap(value, "", id, props);
118 }
119 return status;
120}
