46 const std::string& file_path,
47 Temperature file_temp, size_t readahead_size,

73Status SstFileDumper::GetTableReader(const std::string& file_path) {
74 // Warning about 'magic_number' being uninitialized shows up only in UBsan
75 // builds. Though access is guarded by 's.ok()' checks, fix the issue to
76 // avoid any warnings.
77 uint64_t magic_number = Footer::kNullTableMagicNumber;
78
79 // read table magic number
80 Footer footer;
81
82 const auto& fs = options_.env->GetFileSystem();
83 std::unique_ptr<FSRandomAccessFile> file;
84 uint64_t file_size = 0;
85 FileOptions fopts = soptions_;
86 fopts.temperature = file_temp_;
87 Status s = fs->NewRandomAccessFile(file_path, fopts, &file, nullptr);
88 if (s.ok()) {
89 // check empty file
90 // if true, skip further processing of this file
91 s = fs->GetFileSize(file_path, IOOptions(), &file_size, nullptr);
92 if (s.ok()) {
93 if (file_size == 0) {
94 return Status::Aborted(file_path, "Empty file");
95 }
96 }
97 }
98
99 file_.reset(new RandomAccessFileReader(std::move(file), file_path));
100
101 FilePrefetchBuffer prefetch_buffer(ReadaheadParams(),
102 !fopts.use_mmap_reads /* enable */,
103 false /* track_min_offset */);
104 if (s.ok()) {
105 const uint64_t kSstDumpTailPrefetchSize = 512 * 1024;
106 uint64_t prefetch_size = (file_size > kSstDumpTailPrefetchSize)
107 ? kSstDumpTailPrefetchSize
108 : file_size;
109 uint64_t prefetch_off = file_size - prefetch_size;
110 IOOptions opts;
111 s = prefetch_buffer.Prefetch(opts, file_.get(), prefetch_off,
112 static_cast<size_t>(prefetch_size));
113
114 s = ReadFooterFromFile(opts, file_.get(), *fs, &prefetch_buffer, file_size,
115 &footer);
116 }
117 if (s.ok()) {
118 magic_number = footer.table_magic_number();
119 }
120
121 if (s.ok()) {
122 if (magic_number == kPlainTableMagicNumber ||
123 magic_number == kLegacyPlainTableMagicNumber ||
124 magic_number == kCuckooTableMagicNumber) {
125 soptions_.use_mmap_reads = true;
126 fopts.use_mmap_reads = soptions_.use_mmap_reads;
127
128 if (magic_number == kCuckooTableMagicNumber) {
129 fopts = soptions_;
130 fopts.temperature = file_temp_;
131 }
132
133 fs->NewRandomAccessFile(file_path, fopts, &file, nullptr);
134 file_.reset(new RandomAccessFileReader(std::move(file), file_path));
135 }
136
137 // For old sst format, ReadTableProperties might fail but file can be read
138 if (ReadTableProperties(magic_number, file_.get(), file_size,
139 (magic_number == kBlockBasedTableMagicNumber)
140 ? &prefetch_buffer
141 : nullptr)
142 .ok()) {
143 s = SetTableOptionsByMagicNumber(magic_number);
144 if (s.ok()) {
145 if (table_properties_ && !table_properties_->comparator_name.empty()) {
146 ConfigOptions config_options;
147 const Comparator* user_comparator = nullptr;
148 s = Comparator::CreateFromString(config_options,
149 table_properties_->comparator_name,
150 &user_comparator);
151 if (s.ok()) {
152 assert(user_comparator);
153 internal_comparator_ = InternalKeyComparator(user_comparator);
154 }
155 }
156 }
157 } else {
158 s = SetOldTableOptions();
159 }
160 options_.comparator = internal_comparator_.user_comparator();
161 }
162
163 if (s.ok()) {
164 s = NewTableReader(ioptions_, soptions_, internal_comparator_, file_size,
165 &table_reader_);
166 }
167 return s;
168}

170Status SstFileDumper::NewTableReader(
171 const ImmutableOptions& /*ioptions*/, const EnvOptions& /*soptions*/,
172 const InternalKeyComparator& /*internal_comparator*/, uint64_t file_size,
173 std::unique_ptr<TableReader>* /*table_reader*/) {
174 auto t_opt = TableReaderOptions(
175 ioptions_, moptions_.prefix_extractor, soptions_, internal_comparator_,
176 0 /* block_protection_bytes_per_key */, false /* skip_filters */,
177 false /* immortal */, true /* force_direct_prefetch */, -1 /* level */,
178 nullptr /* block_cache_tracer */, 0 /* max_file_size_for_l0_meta_pin */,
179 "" /* cur_db_session_id */, 0 /* cur_file_num */, {} /* unique_id */,
180 0 /* largest_seqno */, 0 /* tail_size */,
181 table_properties_ == nullptr
182 ? true
183 : static_cast<bool>(
184 table_properties_->user_defined_timestamps_persisted));
185 // Allow open file with global sequence number for backward compatibility.
186 t_opt.largest_seqno = kMaxSequenceNumber;
187
188 // We need to turn off pre-fetching of index and filter nodes for
189 // BlockBasedTable
190 if (options_.table_factory->IsInstanceOf(
191 TableFactory::kBlockBasedTableName())) {
192 return options_.table_factory->NewTableReader(t_opt, std::move(file_),
193 file_size, &table_reader_,
194 /*enable_prefetch=*/false);
195 }
196
197 // For all other factory implementation
198 return options_.table_factory->NewTableReader(t_opt, std::move(file_),
199 file_size, &table_reader_);
200}

209Status SstFileDumper::DumpTable(const std::string& out_filename) {
210 std::unique_ptr<WritableFile> out_file;
211 Env* env = options_.env;
212 Status s = env->NewWritableFile(out_filename, &out_file, soptions_);
213 if (s.ok()) {
214 s = table_reader_->DumpTable(out_file.get());
215 }
216 if (!s.ok()) {
217 // close the file before return error, ignore the close error if there's any
218 out_file->Close().PermitUncheckedError();
219 return s;
220 }
221 return out_file->Close();
222}

224Status SstFileDumper::CalculateCompressedTableSize(
225 const TableBuilderOptions& tb_options, size_t block_size,
226 uint64_t* num_data_blocks, uint64_t* compressed_table_size) {
227 std::unique_ptr<Env> env(NewMemEnv(options_.env));
228 std::unique_ptr<WritableFileWriter> dest_writer;
229 Status s =
230 WritableFileWriter::Create(env->GetFileSystem(), testFileName,
231 FileOptions(soptions_), &dest_writer, nullptr);
232 if (!s.ok()) {
233 return s;
234 }
235 BlockBasedTableOptions table_options;
236 table_options.block_size = block_size;
237 BlockBasedTableFactory block_based_tf(table_options);
238 std::unique_ptr<TableBuilder> table_builder;
239 table_builder.reset(
240 block_based_tf.NewTableBuilder(tb_options, dest_writer.get()));
241 std::unique_ptr<InternalIterator> iter(table_reader_->NewIterator(
242 read_options_, moptions_.prefix_extractor.get(), /*arena=*/nullptr,
243 /*skip_filters=*/false, TableReaderCaller::kSSTDumpTool));
244 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
245 table_builder->Add(iter->key(), iter->value());
246 }
247 s = iter->status();
248 if (!s.ok()) {
249 return s;
250 }
251 s = table_builder->Finish();
252 if (!s.ok()) {
253 return s;
254 }
255 *compressed_table_size = table_builder->FileSize();
256 assert(num_data_blocks != nullptr);
257 *num_data_blocks = table_builder->GetTableProperties().num_data_blocks;
258 return env->DeleteFile(testFileName);
259}

261Status SstFileDumper::ShowAllCompressionSizes(
262 size_t block_size,
263 const std::vector<std::pair<CompressionType, const char*>>&
264 compression_types,
265 int32_t compress_level_from, int32_t compress_level_to,
266 uint32_t max_dict_bytes, uint32_t zstd_max_train_bytes,
267 uint64_t max_dict_buffer_bytes, bool use_zstd_dict_trainer) {
268 fprintf(stdout, "Block Size: %" ROCKSDB_PRIszt "\n", block_size);
269 for (auto& i : compression_types) {
270 if (CompressionTypeSupported(i.first)) {
271 fprintf(stdout, "Compression: %-24s\n", i.second);
272 CompressionOptions compress_opt;
273 compress_opt.max_dict_bytes = max_dict_bytes;
274 compress_opt.zstd_max_train_bytes = zstd_max_train_bytes;
275 compress_opt.max_dict_buffer_bytes = max_dict_buffer_bytes;
276 compress_opt.use_zstd_dict_trainer = use_zstd_dict_trainer;
277 for (int32_t j = compress_level_from; j <= compress_level_to; j++) {
278 fprintf(stdout, "Compression level: %d", j);
279 compress_opt.level = j;
280 Status s = ShowCompressionSize(block_size, i.first, compress_opt);
281 if (!s.ok()) {
282 return s;
283 }
284 }
285 } else {
286 fprintf(stdout, "Unsupported compression type: %s.\n", i.second);
287 }
288 }
289 return Status::OK();

292Status SstFileDumper::ShowCompressionSize(
293 size_t block_size, CompressionType compress_type,
294 const CompressionOptions& compress_opt) {
295 Options opts;
296 opts.statistics = ROCKSDB_NAMESPACE::CreateDBStatistics();
297 opts.statistics->set_stats_level(StatsLevel::kAll);
298 const ImmutableOptions imoptions(opts);
299 const ColumnFamilyOptions cfo(opts);
300 const MutableCFOptions moptions(cfo);
301 // TODO: plumb Env::IOActivity, Env::IOPriority
302 const ReadOptions read_options;
303 const WriteOptions write_options;
304 ROCKSDB_NAMESPACE::InternalKeyComparator ikc(opts.comparator);
305 InternalTblPropCollFactories block_based_table_factories;
306
307 std::string column_family_name;
308 int unknown_level = -1;
309
310 TableBuilderOptions tb_opts(
311 imoptions, moptions, read_options, write_options, ikc,
312 &block_based_table_factories, compress_type, compress_opt,
313 TablePropertiesCollectorFactory::Context::kUnknownColumnFamily,
314 column_family_name, unknown_level, kUnknownNewestKeyTime);
315 uint64_t num_data_blocks = 0;
316 std::chrono::steady_clock::time_point start =
317 std::chrono::steady_clock::now();
318 uint64_t file_size;
319 Status s = CalculateCompressedTableSize(tb_opts, block_size, &num_data_blocks,
320 &file_size);
321 if (!s.ok()) {
322 return s;
323 }
324
325 std::chrono::steady_clock::time_point end = std::chrono::steady_clock::now();
326 fprintf(stdout, " Size: %10" PRIu64, file_size);
327 fprintf(stdout, " Blocks: %6" PRIu64, num_data_blocks);
328 fprintf(stdout, " Time Taken: %10s microsecs",
329 std::to_string(
330 std::chrono::duration_cast<std::chrono::microseconds>(end - start)
331 .count())
332 .c_str());
333 const uint64_t compressed_blocks =
334 opts.statistics->getAndResetTickerCount(NUMBER_BLOCK_COMPRESSED);
335 const uint64_t not_compressed_blocks =
336 opts.statistics->getAndResetTickerCount(
337 NUMBER_BLOCK_COMPRESSION_REJECTED);
338 // When the option enable_index_compression is true,
339 // NUMBER_BLOCK_COMPRESSED is incremented for index block(s).
340 if ((compressed_blocks + not_compressed_blocks) > num_data_blocks) {
341 num_data_blocks = compressed_blocks + not_compressed_blocks;
342 }
343
344 const uint64_t ratio_not_compressed_blocks =
345 (num_data_blocks - compressed_blocks) - not_compressed_blocks;
346 const double compressed_pcnt =
347 (0 == num_data_blocks) ? 0.0
348 : ((static_cast<double>(compressed_blocks) /
349 static_cast<double>(num_data_blocks)) *
350 100.0);
351 const double ratio_not_compressed_pcnt =
352 (0 == num_data_blocks)
353 ? 0.0
354 : ((static_cast<double>(ratio_not_compressed_blocks) /
355 static_cast<double>(num_data_blocks)) *
356 100.0);
357 const double not_compressed_pcnt =
358 (0 == num_data_blocks) ? 0.0
359 : ((static_cast<double>(not_compressed_blocks) /
360 static_cast<double>(num_data_blocks)) *
361 100.0);
362 fprintf(stdout, " Compressed: %6" PRIu64 " (%5.1f%%)", compressed_blocks,
363 compressed_pcnt);
364 fprintf(stdout, " Not compressed (ratio): %6" PRIu64 " (%5.1f%%)",
365 ratio_not_compressed_blocks, ratio_not_compressed_pcnt);
366 fprintf(stdout, " Not compressed (abort): %6" PRIu64 " (%5.1f%%)\n",
367 not_compressed_blocks, not_compressed_pcnt);
368 return Status::OK();
369}

389Status SstFileDumper::SetTableOptionsByMagicNumber(
390 uint64_t table_magic_number) {
391 assert(table_properties_);
392 if (table_magic_number == kBlockBasedTableMagicNumber ||
393 table_magic_number == kLegacyBlockBasedTableMagicNumber) {
394 BlockBasedTableFactory* bbtf = new BlockBasedTableFactory();
395 // To force tail prefetching, we fake reporting two useful reads of 512KB
396 // from the tail.
397 // It needs at least two data points to warm up the stats.
398 bbtf->tail_prefetch_stats()->RecordEffectiveSize(512 * 1024);
399 bbtf->tail_prefetch_stats()->RecordEffectiveSize(512 * 1024);
400
401 options_.table_factory.reset(bbtf);
402 if (!silent_) {
403 fprintf(stdout, "Sst file format: block-based\n");
404 }
405
406 auto& props = table_properties_->user_collected_properties;
407 auto pos = props.find(BlockBasedTablePropertyNames::kIndexType);
408 if (pos != props.end()) {
409 auto index_type_on_file = static_cast<BlockBasedTableOptions::IndexType>(
410 DecodeFixed32(pos->second.c_str()));
411 if (index_type_on_file ==
412 BlockBasedTableOptions::IndexType::kHashSearch) {
413 options_.prefix_extractor.reset(NewNoopTransform());
414 }
415 }
416 } else if (table_magic_number == kPlainTableMagicNumber ||
417 table_magic_number == kLegacyPlainTableMagicNumber) {
418 options_.allow_mmap_reads = true;
419
420 PlainTableOptions plain_table_options;
421 plain_table_options.user_key_len = kPlainTableVariableLength;
422 plain_table_options.bloom_bits_per_key = 0;
423 plain_table_options.hash_table_ratio = 0;
424 plain_table_options.index_sparseness = 1;
425 plain_table_options.huge_page_tlb_size = 0;
426 plain_table_options.encoding_type = kPlain;
427 plain_table_options.full_scan_mode = true;
428
429 options_.table_factory.reset(NewPlainTableFactory(plain_table_options));
430 if (!silent_) {
431 fprintf(stdout, "Sst file format: plain table\n");
432 }
433 } else if (table_magic_number == kCuckooTableMagicNumber) {
434 ioptions_.allow_mmap_reads = true;
435
436 options_.table_factory.reset(NewCuckooTableFactory());
437 if (!silent_) {
438 fprintf(stdout, "Sst file format: cuckoo table\n");
439 }
440 } else {
441 char error_msg_buffer[80];
442 snprintf(error_msg_buffer, sizeof(error_msg_buffer) - 1,
443 "Unsupported table magic number --- %lx",
444 (long)table_magic_number);
445 return Status::InvalidArgument(error_msg_buffer);
446 }
447
448 return Status::OK();
449}

461Status SstFileDumper::ReadSequential(bool print_kv, uint64_t read_num_limit,
462 bool has_from, const std::string& from_key,
463 bool has_to, const std::string& to_key,
464 bool use_from_as_prefix) {
465 if (!table_reader_) {
466 return init_result_;
467 }
468
469 InternalIterator* iter = table_reader_->NewIterator(
470 read_options_, moptions_.prefix_extractor.get(),
471 /*arena=*/nullptr, /*skip_filters=*/false,
472 TableReaderCaller::kSSTDumpTool);
473
474 const Comparator* ucmp = internal_comparator_.user_comparator();
475 size_t ts_sz = ucmp->timestamp_size();
476
477 Slice from_slice = from_key;
478 Slice to_slice = to_key;
479 std::string from_key_buf, to_key_buf;
480 auto [from, to] = MaybeAddTimestampsToRange(
481 has_from ? &from_slice : nullptr, has_to ? &to_slice : nullptr, ts_sz,
482 &from_key_buf, &to_key_buf);
483 uint64_t i = 0;
484 if (from.has_value()) {
485 InternalKey ikey;
486 ikey.SetMinPossibleForUserKey(from.value());
487 iter->Seek(ikey.Encode());
488 } else {
489 iter->SeekToFirst();
490 }
491 for (; iter->Valid(); iter->Next()) {
492 Slice key = iter->key();
493 Slice value = iter->value();
494 ++i;
495 if (read_num_limit > 0 && i > read_num_limit) {
496 break;
497 }
498
499 ParsedInternalKey ikey;
500 Status pik_status = ParseInternalKey(key, &ikey, true /* log_err_key */);
501 if (!pik_status.ok()) {
502 std::cerr << pik_status.getState() << "\n";
503 continue;
504 }
505
506 // the key returned is not prefixed with out 'from' key
507 if (use_from_as_prefix && !ikey.user_key.starts_with(from_key)) {
508 break;
509 }
510
511 // If end marker was specified, we stop before it
512 if (to.has_value() && ucmp->Compare(ikey.user_key, to.value()) >= 0) {
513 break;
514 }
515
516 if (print_kv) {
517 if (!decode_blob_index_ || ikey.type != kTypeBlobIndex) {
518 if (ikey.type == kTypeWideColumnEntity) {
519 std::ostringstream oss;
520 const Status s = WideColumnsHelper::DumpSliceAsWideColumns(
521 iter->value(), oss, output_hex_);
522 if (!s.ok()) {
523 fprintf(stderr, "%s => error deserializing wide columns\n",
524 ikey.DebugString(true, output_hex_, ucmp).c_str());
525 continue;
526 }
527 fprintf(stdout, "%s => %s\n",
528 ikey.DebugString(true, output_hex_, ucmp).c_str(),
529 oss.str().c_str());
530 } else if (ikey.type == kTypeValuePreferredSeqno) {
531 auto [unpacked_value, preferred_seqno] =
532 ParsePackedValueWithSeqno(value);
533 fprintf(stdout, "%s => %s, %llu\n",
534 ikey.DebugString(true, output_hex_, ucmp).c_str(),
535 unpacked_value.ToString(output_hex_).c_str(),
536 static_cast<unsigned long long>(preferred_seqno));
537 } else {
538 fprintf(stdout, "%s => %s\n",
539 ikey.DebugString(true, output_hex_, ucmp).c_str(),
540 value.ToString(output_hex_).c_str());
541 }
542 } else {
543 BlobIndex blob_index;
544
545 const Status s = blob_index.DecodeFrom(value);
546 if (!s.ok()) {
547 fprintf(stderr, "%s => error decoding blob index\n",
548 ikey.DebugString(true, output_hex_, ucmp).c_str());
549 continue;
550 }
551
552 fprintf(stdout, "%s => %s\n",
553 ikey.DebugString(true, output_hex_, ucmp).c_str(),
554 blob_index.DebugString(output_hex_).c_str());
555 }
556 }
557 }
558
559 read_num_ += i;
560
561 Status ret = iter->status();
562
563 bool verify_num_entries =
564 (read_num_limit == 0 ||
565 read_num_limit == std::numeric_limits<uint64_t>::max()) &&
566 !has_from && !has_to;
567 if (verify_num_entries && ret.ok()) {
568 // Compare the number of entries
569 if (!table_properties_) {
570 fprintf(stderr, "Table properties not available.");
571 } else {
572 // TODO: verify num_range_deletions
573 if (i != table_properties_->num_entries -
574 table_properties_->num_range_deletions) {
575 std::ostringstream oss;
576 oss << "Table property expects "
577 << table_properties_->num_entries -
578 table_properties_->num_range_deletions
579 << " entries when excluding range deletions,"
580 << " but scanning the table returned " << std::to_string(i)
581 << " entries";
582 ret = Status::Corruption(oss.str());
583 }
584 }
585 }
586
587 delete iter;
588 return ret;
589}
