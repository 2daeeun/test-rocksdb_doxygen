51 bool CheckFileLock(bool lock_expected) {
52 // We need to fork to check the fcntl lock as we need
53 // to open and close the file from a different process
54 // to avoid either releasing the lock on close, or not
55 // contending for it when requesting a lock.
56
57#ifdef OS_WIN
58
59 // WaitForSingleObject and GetExitCodeProcess can do what waitpid does.
60 // TODO - implement on Windows
61 return true;
62
63#else
64
65 pid_t pid = fork();
66 if (0 == pid) {
67 // child process
68 int exit_val = EXIT_FAILURE;
69 int fd = open(file_.c_str(), O_RDWR | O_CREAT, 0644);
70 if (fd < 0) {
71 // could not open file, could not check if it was locked
72 fprintf(stderr, "Open on on file %s failed.\n", file_.c_str());
73 exit(exit_val);
74 }
75
76 struct flock f;
77 memset(&f, 0, sizeof(f));
78 f.l_type = (F_WRLCK);
79 f.l_whence = SEEK_SET;
80 f.l_start = 0;
81 f.l_len = 0; // Lock/unlock entire file
82 int value = fcntl(fd, F_SETLK, &f);
83 if (value == -1) {
84 if (lock_expected) {
85 exit_val = EXIT_SUCCESS;
86 }
87 } else {
88 if (!lock_expected) {
89 exit_val = EXIT_SUCCESS;
90 }
91 }
92 close(fd); // lock is released for child process
93 exit(exit_val);
94 } else if (pid > 0) {
95 // parent process
96 int status;
97 while (-1 == waitpid(pid, &status, 0)) {
98 }
99 if (!WIFEXITED(status) || WEXITSTATUS(status) != 0) {
100 // child process exited with non success status
101 return false;
102 } else {
103 return true;
104 }
105 } else {
106 fprintf(stderr, "Fork failed\n");
107 return false;
108 }
109 return false;
110
111#endif
112 }
113};

116TEST_F(LockTest, LockBySameThread) {
117 FileLock* lock1;
118 FileLock* lock2;
119
120 // acquire a lock on a file
121 ASSERT_OK(LockFile(&lock1));
122
123 // check the file is locked
124 ASSERT_TRUE(AssertFileIsLocked());
125
126 // re-acquire the lock on the same file. This should fail.
127 Status s = LockFile(&lock2);
128 ASSERT_TRUE(s.IsIOError());
129#ifndef OS_WIN
130 // Validate that error message contains current thread ID.
131 ASSERT_TRUE(s.ToString().find(std::to_string(
132 Env::Default()->GetThreadID())) != std::string::npos);
133#endif
134
135 // check the file is locked
136 ASSERT_TRUE(AssertFileIsLocked());
137
138 // release the lock
139 ASSERT_OK(UnlockFile(lock1));
140
141 // check the file is not locked
142 ASSERT_TRUE(AssertFileIsNotLocked());
143}
