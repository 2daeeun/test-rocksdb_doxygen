28class MockTableIterator : public InternalIterator {
29 public:
30 explicit MockTableIterator(const KVVector& table) : table_(table) {
31 itr_ = table_.end();
32 }
33
34 bool Valid() const override { return itr_ != table_.end(); }
35
36 void SeekToFirst() override { itr_ = table_.begin(); }
37
38 void SeekToLast() override {
39 itr_ = table_.end();
40 --itr_;
41 }
42
43 void Seek(const Slice& target) override {
44 KVPair target_pair(target.ToString(), "");
45 InternalKeyComparator icmp(BytewiseComparator());
46 itr_ = std::lower_bound(table_.begin(), table_.end(), target_pair,
47 [icmp](KVPair a, KVPair b) -> bool {
48 return icmp.Compare(a.first, b.first) < 0;
49 });
50 }
51
52 void SeekForPrev(const Slice& target) override {
53 KVPair target_pair(target.ToString(), "");
54 InternalKeyComparator icmp(BytewiseComparator());
55 itr_ = std::upper_bound(table_.begin(), table_.end(), target_pair,
56 [icmp](KVPair a, KVPair b) -> bool {
57 return icmp.Compare(a.first, b.first) < 0;
58 });
59 Prev();
60 }
61
62 void Next() override { ++itr_; }
63
64 void Prev() override {
65 if (itr_ == table_.begin()) {
66 itr_ = table_.end();
67 } else {
68 --itr_;
69 }
70 }
71
72 Slice key() const override { return Slice(itr_->first); }
73
74 Slice value() const override { return Slice(itr_->second); }
75
76 Status status() const override { return Status::OK(); }
77
78 private:
79 const KVVector& table_;
80 KVVector::const_iterator itr_;
81};

83class MockTableBuilder : public TableBuilder {
84 public:
85 MockTableBuilder(uint32_t id, MockTableFileSystem* file_system,
86 MockTableFactory::MockCorruptionMode corrupt_mode =
87 MockTableFactory::kCorruptNone,
88 size_t key_value_size = 1)
89 : id_(id),
90 file_system_(file_system),
91 corrupt_mode_(corrupt_mode),
92 key_value_size_(key_value_size) {
93 table_ = MakeMockFile({});
94 }
95
96 // REQUIRES: Either Finish() or Abandon() has been called.
97 ~MockTableBuilder() = default;
98
99 // Add key,value to the table being constructed.
100 // REQUIRES: key is after any previously added key according to comparator.
101 // REQUIRES: Finish(), Abandon() have not been called
102 void Add(const Slice& key, const Slice& value) override {
103 if (corrupt_mode_ == MockTableFactory::kCorruptValue) {
104 // Corrupt the value
105 table_.push_back({key.ToString(), value.ToString() + " "});
106 corrupt_mode_ = MockTableFactory::kCorruptNone;
107 } else if (corrupt_mode_ == MockTableFactory::kCorruptKey) {
108 table_.push_back({key.ToString() + " ", value.ToString()});
109 corrupt_mode_ = MockTableFactory::kCorruptNone;
110 } else if (corrupt_mode_ == MockTableFactory::kCorruptReorderKey) {
111 if (prev_key_.empty()) {
112 prev_key_ = key.ToString();
113 prev_value_ = value.ToString();
114 } else {
115 table_.push_back({key.ToString(), value.ToString()});
116 table_.push_back({prev_key_, prev_value_});
117 corrupt_mode_ = MockTableFactory::kCorruptNone;
118 }
119 } else {
120 table_.push_back({key.ToString(), value.ToString()});
121 }
122 }
123
124 // Return non-ok iff some error has been detected.
125 Status status() const override { return Status::OK(); }
126
127 // Return non-ok iff some error happens during IO.
128 IOStatus io_status() const override { return IOStatus::OK(); }
129
130 Status Finish() override {
131 MutexLock lock_guard(&file_system_->mutex);
132 file_system_->files.insert({id_, table_});
133 return Status::OK();
134 }
135
136 void Abandon() override {}
137
138 uint64_t NumEntries() const override { return table_.size(); }
139
140 uint64_t FileSize() const override { return table_.size() * key_value_size_; }
141
142 TableProperties GetTableProperties() const override {
143 return TableProperties();
144 }
145
146 // Get file checksum
147 std::string GetFileChecksum() const override { return kUnknownFileChecksum; }
148 // Get file checksum function name
149 const char* GetFileChecksumFuncName() const override {
150 return kUnknownFileChecksumFuncName;
151 }
152
153 private:
154 uint32_t id_;
155 std::string prev_key_;
156 std::string prev_value_;
157 MockTableFileSystem* file_system_;
158 int corrupt_mode_;
159 KVVector table_;
160 size_t key_value_size_;
161};

170Status MockTableReader::Get(const ReadOptions&, const Slice& key,
171 GetContext* get_context,
172 const SliceTransform* /*prefix_extractor*/,
173 bool /*skip_filters*/) {
174 std::unique_ptr<MockTableIterator> iter(new MockTableIterator(table_));
175 for (iter->Seek(key); iter->Valid(); iter->Next()) {
176 ParsedInternalKey parsed_key;
177 Status pik_status =
178 ParseInternalKey(iter->key(), &parsed_key, true /* log_err_key */);
179 if (!pik_status.ok()) {
180 return pik_status;
181 }
182
183 bool dont_care __attribute__((__unused__));
184 Status read_status;
185 bool ret = get_context->SaveValue(parsed_key, iter->value(), &dont_care,
186 &read_status);
187 if (!read_status.ok()) {
188 return read_status;
189 }
190 if (!ret) {
191 break;
192 }
193 }
194 return Status::OK();
195}

200Status MockTableFactory::NewTableReader(
201 const ReadOptions& /*ro*/,
202 const TableReaderOptions& /*table_reader_options*/,
203 std::unique_ptr<RandomAccessFileReader>&& file, uint64_t /*file_size*/,
204 std::unique_ptr<TableReader>* table_reader,
205 bool /*prefetch_index_and_filter_in_cache*/) const {
206 uint32_t id;

208 if (!s.ok()) {
209 return s;
210 }
211
212 MutexLock lock_guard(&file_system_.mutex);
213
214 auto it = file_system_.files.find(id);
215 if (it == file_system_.files.end()) {
216 return Status::IOError("Mock file not found");
217 }
218
219 table_reader->reset(new MockTableReader(it->second));
220
221 return Status::OK();
222}
