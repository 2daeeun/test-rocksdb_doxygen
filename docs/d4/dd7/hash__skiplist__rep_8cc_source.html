79 class Iterator : public MemTableRep::Iterator {
80 public:
81 explicit Iterator(Bucket* list, bool own_list = true,
82 Arena* arena = nullptr)
83 : list_(list), iter_(list), own_list_(own_list), arena_(arena) {}
84
85 ~Iterator() override {
86 // if we own the list, we should also delete it
87 if (own_list_) {
88 assert(list_ != nullptr);
89 delete list_;
90 }
91 }
92
93 // Returns true iff the iterator is positioned at a valid node.
94 bool Valid() const override { return list_ != nullptr && iter_.Valid(); }
95
96 // Returns the key at the current position.
97 // REQUIRES: Valid()
98 const char* key() const override {
99 assert(Valid());
100 return iter_.key();
101 }
102
103 // Advances to the next position.
104 // REQUIRES: Valid()
105 void Next() override {
106 assert(Valid());
107 iter_.Next();
108 }
109
110 // Advances to the previous position.
111 // REQUIRES: Valid()
112 void Prev() override {
113 assert(Valid());
114 iter_.Prev();
115 }
116
117 // Advance to the first entry with a key >= target
118 void Seek(const Slice& internal_key, const char* memtable_key) override {
119 if (list_ != nullptr) {
120 const char* encoded_key = (memtable_key != nullptr)
121 ? memtable_key
122 : EncodeKey(&tmp_, internal_key);
123 iter_.Seek(encoded_key);
124 }
125 }
126
127 // Retreat to the last entry with a key <= target
128 void SeekForPrev(const Slice& /*internal_key*/,
129 const char* /*memtable_key*/) override {
130 // not supported
131 assert(false);
132 }
133
134 // Position at the first entry in collection.
135 // Final state of iterator is Valid() iff collection is not empty.
136 void SeekToFirst() override {
137 if (list_ != nullptr) {
138 iter_.SeekToFirst();
139 }
140 }
141
142 // Position at the last entry in collection.
143 // Final state of iterator is Valid() iff collection is not empty.
144 void SeekToLast() override {
145 if (list_ != nullptr) {
146 iter_.SeekToLast();
147 }
148 }
149
150 protected:
151 void Reset(Bucket* list) {
152 if (own_list_) {
153 assert(list_ != nullptr);
154 delete list_;
155 }
156 list_ = list;
157 iter_.SetList(list);
158 own_list_ = false;
159 }
160
161 private:
162 // if list_ is nullptr, we should NEVER call any methods on iter_
163 // if list_ is nullptr, this Iterator is not Valid()
164 Bucket* list_;
165 Bucket::Iterator iter_;
166 // here we track if we own list_. If we own it, we are also
167 // responsible for it's cleaning. This is a poor man's std::shared_ptr
168 bool own_list_;
169 std::unique_ptr<Arena> arena_;
170 std::string tmp_; // For passing to EncodeKey
171 };

173 class DynamicIterator : public HashSkipListRep::Iterator {
174 public:
175 explicit DynamicIterator(const HashSkipListRep& memtable_rep)
176 : HashSkipListRep::Iterator(nullptr, false),
177 memtable_rep_(memtable_rep) {}
178
179 // Advance to the first entry with a key >= target
180 void Seek(const Slice& k, const char* memtable_key) override {
181 auto transformed = memtable_rep_.transform_->Transform(ExtractUserKey(k));
182 Reset(memtable_rep_.GetBucket(transformed));
183 HashSkipListRep::Iterator::Seek(k, memtable_key);
184 }
185
186 // Position at the first entry in collection.
187 // Final state of iterator is Valid() iff collection is not empty.
188 void SeekToFirst() override {
189 // Prefix iterator does not support total order.
190 // We simply set the iterator to invalid state
191 Reset(nullptr);
192 }
193
194 // Position at the last entry in collection.
195 // Final state of iterator is Valid() iff collection is not empty.
196 void SeekToLast() override {
197 // Prefix iterator does not support total order.
198 // We simply set the iterator to invalid state
199 Reset(nullptr);
200 }
201
202 private:
203 // the underlying memtable
204 const HashSkipListRep& memtable_rep_;
205 };

230HashSkipListRep::HashSkipListRep(const MemTableRep::KeyComparator& compare,
231 Allocator* allocator,
232 const SliceTransform* transform,
233 size_t bucket_size, int32_t skiplist_height,
234 int32_t skiplist_branching_factor)
235 : MemTableRep(allocator),
236 bucket_size_(bucket_size),
237 skiplist_height_(skiplist_height),
238 skiplist_branching_factor_(skiplist_branching_factor),
239 transform_(transform),
240 compare_(compare),
241 allocator_(allocator) {
242 auto mem =
243 allocator->AllocateAligned(sizeof(std::atomic<void*>) * bucket_size);
244 buckets_ = new (mem) std::atomic<Bucket*>[bucket_size];
245
246 for (size_t i = 0; i < bucket_size_; ++i) {
247 buckets_[i].store(nullptr, std::memory_order_relaxed);
248 }
249}

253HashSkipListRep::Bucket* HashSkipListRep::GetInitializedBucket(
254 const Slice& transformed) {
255 size_t hash = GetHash(transformed);
256 auto bucket = GetBucket(hash);
257 if (bucket == nullptr) {
258 auto addr = allocator_->AllocateAligned(sizeof(Bucket));
259 bucket = new (addr) Bucket(compare_, allocator_, skiplist_height_,
260 skiplist_branching_factor_);
261 buckets_[hash].store(bucket, std::memory_order_release);
262 }
263 return bucket;
264}

266void HashSkipListRep::Insert(KeyHandle handle) {
267 auto* key = static_cast<char*>(handle);
268 assert(!Contains(key));
269 auto transformed = transform_->Transform(UserKey(key));
270 auto bucket = GetInitializedBucket(transformed);

285void HashSkipListRep::Get(const LookupKey& k, void* callback_args,
286 bool (*callback_func)(void* arg, const char* entry)) {
287 auto transformed = transform_->Transform(k.user_key());
288 auto bucket = GetBucket(transformed);
289 if (bucket != nullptr) {
290 Bucket::Iterator iter(bucket);
291 for (iter.Seek(k.memtable_key().data());
292 iter.Valid() && callback_func(callback_args, iter.key());
293 iter.Next()) {
294 }
295 }
296}

298MemTableRep::Iterator* HashSkipListRep::GetIterator(Arena* arena) {
299 // allocate a new arena of similar size to the one currently in use
300 Arena* new_arena = new Arena(allocator_->BlockSize());
301 auto list = new Bucket(compare_, new_arena);
302 for (size_t i = 0; i < bucket_size_; ++i) {
303 auto bucket = GetBucket(i);
304 if (bucket != nullptr) {
305 Bucket::Iterator itr(bucket);
306 for (itr.SeekToFirst(); itr.Valid(); itr.Next()) {
307 list->Insert(itr.key());
308 }
309 }
310 }
311 if (arena == nullptr) {
312 return new Iterator(list, true, new_arena);
313 } else {
314 auto mem = arena->AllocateAligned(sizeof(Iterator));
315 return new (mem) Iterator(list, true, new_arena);
316 }
317}
