30// Creates a test blob file with a single blob in it.
31void WriteBlobFile(uint32_t column_family_id,
32 const ImmutableOptions& immutable_options,
33 uint64_t blob_file_number) {
34 assert(!immutable_options.cf_paths.empty());
35
36 const std::string blob_file_path =
37 BlobFileName(immutable_options.cf_paths.front().path, blob_file_number);
38
39 std::unique_ptr<FSWritableFile> file;
40 ASSERT_OK(NewWritableFile(immutable_options.fs.get(), blob_file_path, &file,
41 FileOptions()));
42
43 std::unique_ptr<WritableFileWriter> file_writer(new WritableFileWriter(
44 std::move(file), blob_file_path, FileOptions(), immutable_options.clock));
45
46 constexpr Statistics* statistics = nullptr;
47 constexpr bool use_fsync = false;
48 constexpr bool do_flush = false;
49
50 BlobLogWriter blob_log_writer(std::move(file_writer), immutable_options.clock,
51 statistics, blob_file_number, use_fsync,
52 do_flush);
53
54 constexpr bool has_ttl = false;
55 constexpr ExpirationRange expiration_range;
56
57 BlobLogHeader header(column_family_id, kNoCompression, has_ttl,
58 expiration_range);
59
60 ASSERT_OK(blob_log_writer.WriteHeader(WriteOptions(), header));
61
62 constexpr char key[] = "key";
63 constexpr char blob[] = "blob";
64
65 std::string compressed_blob;
66
67 uint64_t key_offset = 0;
68 uint64_t blob_offset = 0;
69
70 ASSERT_OK(blob_log_writer.AddRecord(WriteOptions(), key, blob, &key_offset,
71 &blob_offset));
72
73 BlobLogFooter footer;
74 footer.blob_count = 1;
75 footer.expiration_range = expiration_range;
76
77 std::string checksum_method;
78 std::string checksum_value;
79
80 ASSERT_OK(blob_log_writer.AppendFooter(WriteOptions(), footer,
81 &checksum_method, &checksum_value));
82}

93TEST_F(BlobFileCacheTest, GetBlobFileReader) {
94 Options options;
95 options.env = mock_env_.get();
96 options.statistics = CreateDBStatistics();
97 options.cf_paths.emplace_back(
98 test::PerThreadDBPath(mock_env_.get(),
99 "BlobFileCacheTest_GetBlobFileReader"),
100 0);
101 options.enable_blob_files = true;
102
103 constexpr uint32_t column_family_id = 1;
104 ImmutableOptions immutable_options(options);
105 constexpr uint64_t blob_file_number = 123;
106
107 WriteBlobFile(column_family_id, immutable_options, blob_file_number);
108
109 constexpr size_t capacity = 10;
110 std::shared_ptr<Cache> backing_cache = NewLRUCache(capacity);
111
112 FileOptions file_options;
113 constexpr HistogramImpl* blob_file_read_hist = nullptr;
114
115 BlobFileCache blob_file_cache(backing_cache.get(), &immutable_options,
116 &file_options, column_family_id,
117 blob_file_read_hist, nullptr /*IOTracer*/);
118
119 // First try: reader should be opened and put in cache
120 CacheHandleGuard<BlobFileReader> first;
121
122 const ReadOptions read_options;
123 ASSERT_OK(blob_file_cache.GetBlobFileReader(read_options, blob_file_number,
124 &first));
125 ASSERT_NE(first.GetValue(), nullptr);
126 ASSERT_EQ(options.statistics->getTickerCount(NO_FILE_OPENS), 1);
127 ASSERT_EQ(options.statistics->getTickerCount(NO_FILE_ERRORS), 0);
128
129 // Second try: reader should be served from cache
130 CacheHandleGuard<BlobFileReader> second;
131
132 ASSERT_OK(blob_file_cache.GetBlobFileReader(read_options, blob_file_number,
133 &second));
134 ASSERT_NE(second.GetValue(), nullptr);
135 ASSERT_EQ(options.statistics->getTickerCount(NO_FILE_OPENS), 1);
136 ASSERT_EQ(options.statistics->getTickerCount(NO_FILE_ERRORS), 0);
137
138 ASSERT_EQ(first.GetValue(), second.GetValue());
139}

159

161 constexpr HistogramImpl* blob_file_read_hist = nullptr;
162

166

168 CacheHandleGuard<BlobFileReader> second;
169
170 const ReadOptions read_options;
171 SyncPoint::GetInstance()->SetCallBack(
172 "BlobFileCache::GetBlobFileReader:DoubleCheck", [&](void* /* arg */) {
173 // Disabling sync points to prevent infinite recursion
174 SyncPoint::GetInstance()->DisableProcessing();
175 ASSERT_OK(blob_file_cache.GetBlobFileReader(read_options,
176 blob_file_number, &second));
177 ASSERT_NE(second.GetValue(), nullptr);
178 ASSERT_EQ(options.statistics->getTickerCount(NO_FILE_OPENS), 1);
179 ASSERT_EQ(options.statistics->getTickerCount(NO_FILE_ERRORS), 0);
180 });
181 SyncPoint::GetInstance()->EnableProcessing();
182
183 ASSERT_OK(blob_file_cache.GetBlobFileReader(read_options, blob_file_number,
184 &first));
185 ASSERT_NE(first.GetValue(), nullptr);
186 ASSERT_EQ(options.statistics->getTickerCount(NO_FILE_OPENS), 1);
187 ASSERT_EQ(options.statistics->getTickerCount(NO_FILE_ERRORS), 0);
188
189 ASSERT_EQ(first.GetValue(), second.GetValue());
190
191 SyncPoint::GetInstance()->DisableProcessing();
192 SyncPoint::GetInstance()->ClearAllCallBacks();
193}
