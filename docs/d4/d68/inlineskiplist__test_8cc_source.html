129 Random rnd(1000);
130 std::set<Key> keys;
131 ConcurrentArena arena;
132 TestComparator cmp;
133 InlineSkipList<TestComparator> list(cmp, &arena);

138 memcpy(buf, &key, sizeof(Key));

147 ASSERT_EQ(keys.count(i), 0U);

154 ASSERT_TRUE(!iter.Valid());

313// We want to make sure that with a single writer and multiple
314// concurrent readers (with no synchronization other than when a
315// reader's iterator is created), the reader always observes all the
316// data that was present in the skip list when the iterator was
317// constructor. Because insertions are happening concurrently, we may
318// also observe new values that were inserted since the iterator was
319// constructed, but we should never miss any values that were present
320// at iterator construction time.
321//
322// We generate multi-part keys:
323// <key,gen,hash>
324// where:
325// key is in range [0..K-1]
326// gen is a generation number for key
327// hash is hash(key,gen)
328//
329// The insertion code picks a random key, sets gen to be 1 + the last
330// generation number inserted for that key, and sets hash to Hash(key,gen).
331//
332// At the beginning of a read, we snapshot the last inserted
333// generation number for each key. We then iterate, including random
334// calls to Next() and Seek(). For every key we encounter, we
335// check that it is either expected given the initial snapshot or has
336// been concurrently added since the iterator started.
337class ConcurrentTest {
338 public:
339 static const uint32_t K = 8;
340
341 private:
342 static uint64_t key(Key key) { return (key >> 40); }
343 static uint64_t gen(Key key) { return (key >> 8) & 0xffffffffu; }
344 static uint64_t hash(Key key) { return key & 0xff; }
345
346 static uint64_t HashNumbers(uint64_t k, uint64_t g) {
347 uint64_t data[2] = {k, g};
348 return Hash(reinterpret_cast<char*>(data), sizeof(data), 0);
349 }
350
351 static Key MakeKey(uint64_t k, uint64_t g) {
352 assert(sizeof(Key) == sizeof(uint64_t));
353 assert(k <= K); // We sometimes pass K to seek to the end of the skiplist
354 assert(g <= 0xffffffffu);
355 return ((k << 40) | (g << 8) | (HashNumbers(k, g) & 0xff));
356 }
357
358 static bool IsValidKey(Key k) {
359 return hash(k) == (HashNumbers(key(k), gen(k)) & 0xff);
360 }
361
362 static Key RandomTarget(Random* rnd) {
363 switch (rnd->Next() % 10) {
364 case 0:
365 // Seek to beginning
366 return MakeKey(0, 0);
367 case 1:
368 // Seek to end
369 return MakeKey(K, 0);
370 default:
371 // Seek to middle
372 return MakeKey(rnd->Next() % K, 0);
373 }
374 }
375
376 // Per-key generation
377 struct State {
378 std::atomic<int> generation[K];
379 void Set(int k, int v) {
380 generation[k].store(v, std::memory_order_release);
381 }
382 int Get(int k) { return generation[k].load(std::memory_order_acquire); }
383
384 State() {
385 for (unsigned int k = 0; k < K; k++) {
386 Set(k, 0);
387 }
388 }
389 };
390
391 // Current state of the test
392 State current_;
393
394 ConcurrentArena arena_;
395
396 // InlineSkipList is not protected by mu_. We just use a single writer
397 // thread to modify it.
398 InlineSkipList<TestComparator> list_;
399
400 public:
401 ConcurrentTest() : list_(TestComparator(), &arena_) {}
402
403 // REQUIRES: No concurrent calls to WriteStep or ConcurrentWriteStep
404 void WriteStep(Random* rnd) {
405 const uint32_t k = rnd->Next() % K;
406 const int g = current_.Get(k) + 1;
407 const Key new_key = MakeKey(k, g);
408 char* buf = list_.AllocateKey(sizeof(Key));
409 memcpy(buf, &new_key, sizeof(Key));
410 list_.Insert(buf);
411 current_.Set(k, g);
412 }
413
414 // REQUIRES: No concurrent calls for the same k
415 void ConcurrentWriteStep(uint32_t k, bool use_hint = false) {
416 const int g = current_.Get(k) + 1;
417 const Key new_key = MakeKey(k, g);
418 char* buf = list_.AllocateKey(sizeof(Key));
419 memcpy(buf, &new_key, sizeof(Key));
420 if (use_hint) {
421 void* hint = nullptr;
422 list_.InsertWithHintConcurrently(buf, &hint);
423 delete[] reinterpret_cast<char*>(hint);
424 } else {
425 list_.InsertConcurrently(buf);
426 }
427 ASSERT_EQ(g, current_.Get(k) + 1);
428 current_.Set(k, g);
429 }
430
431 void ReadStep(Random* rnd) {
432 // Remember the initial committed state of the skiplist.
433 State initial_state;
434 for (unsigned int k = 0; k < K; k++) {
435 initial_state.Set(k, current_.Get(k));
436 }
437
438 Key pos = RandomTarget(rnd);
439 InlineSkipList<TestComparator>::Iterator iter(&list_);
440 iter.Seek(Encode(&pos));
441 while (true) {
442 Key current;
443 if (!iter.Valid()) {
444 current = MakeKey(K, 0);
445 } else {
446 current = Decode(iter.key());
447 ASSERT_TRUE(IsValidKey(current)) << current;
448 }
449 ASSERT_LE(pos, current) << "should not go backwards";
450
451 // Verify that everything in [pos,current) was not present in
452 // initial_state.
453 while (pos < current) {
454 ASSERT_LT(key(pos), K) << pos;
455
456 // Note that generation 0 is never inserted, so it is ok if
457 // <*,0,*> is missing.
458 ASSERT_TRUE((gen(pos) == 0U) ||
459 (gen(pos) > static_cast<uint64_t>(initial_state.Get(
460 static_cast<int>(key(pos))))))
461 << "key: " << key(pos) << "; gen: " << gen(pos)
462 << "; initgen: " << initial_state.Get(static_cast<int>(key(pos)));
463
464 // Advance to next key in the valid key space
465 if (key(pos) < key(current)) {
466 pos = MakeKey(key(pos) + 1, 0);
467 } else {
468 pos = MakeKey(key(pos), gen(pos) + 1);
469 }
470 }
471
472 if (!iter.Valid()) {
473 break;
474 }
475
476 if (rnd->Next() % 2) {
477 iter.Next();
478 pos = MakeKey(key(pos), gen(pos) + 1);
479 } else {
480 Key new_target = RandomTarget(rnd);
481 if (new_target > pos) {
482 pos = new_target;
483 iter.Seek(Encode(&new_target));
484 }
485 }
486 }
487 }
488};

514class TestState {
515 public:
516 ConcurrentTest t_;
517 bool use_hint_;
518 int seed_;
519 std::atomic<bool> quit_flag_;
520 std::atomic<uint32_t> next_writer_;
521
522 enum ReaderState { STARTING, RUNNING, DONE };
523
524 explicit TestState(int s)
525 : seed_(s),
526 quit_flag_(false),
527 state_(STARTING),
528 pending_writers_(0),
529 state_cv_(&mu_) {}
530
531 void Wait(ReaderState s) {
532 mu_.Lock();
533 while (state_ != s) {
534 state_cv_.Wait();
535 }
536 mu_.Unlock();
537 }
538
539 void Change(ReaderState s) {
540 mu_.Lock();
541 state_ = s;
542 state_cv_.Signal();
543 mu_.Unlock();
544 }
545
546 void AdjustPendingWriters(int delta) {
547 mu_.Lock();
548 pending_writers_ += delta;
549 if (pending_writers_ == 0) {
550 state_cv_.Signal();
551 }
552 mu_.Unlock();
553 }
554
555 void WaitForPendingWriters() {
556 mu_.Lock();
557 while (pending_writers_ != 0) {
558 state_cv_.Wait();
559 }
560 mu_.Unlock();
561 }
562
563 private:
564 port::Mutex mu_;
565 ReaderState state_;
566 int pending_writers_;
567 port::CondVar state_cv_;
568};

570static void ConcurrentReader(void* arg) {
571 TestState* state = static_cast<TestState*>(arg);
572 Random rnd(state->seed_);
573 int64_t reads = 0;
574 state->Change(TestState::RUNNING);
575 while (!state->quit_flag_.load(std::memory_order_acquire)) {
576 state->t_.ReadStep(&rnd);
577 ++reads;
578 }
579 (void)reads;
580 state->Change(TestState::DONE);
581}
582
583static void ConcurrentWriter(void* arg) {
584 TestState* state = static_cast<TestState*>(arg);
585 uint32_t k = state->next_writer_++ % ConcurrentTest::K;
586 state->t_.ConcurrentWriteStep(k, state->use_hint_);
587 state->AdjustPendingWriters(-1);
588}

590static void RunConcurrentRead(int run) {
591 const int seed = test::RandomSeed() + (run * 100);
592 Random rnd(seed);
593 const int N = 1000;
594 const int kSize = 1000;
595 for (int i = 0; i < N; i++) {
596 if ((i % 100) == 0) {
597 fprintf(stderr, "Run %d of %d\n", i, N);
598 }
599 TestState state(seed + 1);
600 Env::Default()->SetBackgroundThreads(1);
601 Env::Default()->Schedule(ConcurrentReader, &state);
602 state.Wait(TestState::RUNNING);
603 for (int k = 0; k < kSize; ++k) {
604 state.t_.WriteStep(&rnd);
605 }
606 state.quit_flag_.store(true, std::memory_order_release);
607 state.Wait(TestState::DONE);
608 }
609}

611static void RunConcurrentInsert(int run, bool use_hint = false,
612 int write_parallelism = 4) {
613 Env::Default()->SetBackgroundThreads(1 + write_parallelism,
614 Env::Priority::LOW);
615 const int seed = test::RandomSeed() + (run * 100);
616 Random rnd(seed);
617 const int N = 1000;
618 const int kSize = 1000;
619 for (int i = 0; i < N; i++) {
620 if ((i % 100) == 0) {
621 fprintf(stderr, "Run %d of %d\n", i, N);
622 }
623 TestState state(seed + 1);
624 state.use_hint_ = use_hint;
625 Env::Default()->Schedule(ConcurrentReader, &state);
626 state.Wait(TestState::RUNNING);
627 for (int k = 0; k < kSize; k += write_parallelism) {
628 state.next_writer_ = rnd.Next();
629 state.AdjustPendingWriters(write_parallelism);
630 for (int p = 0; p < write_parallelism; ++p) {
631 Env::Default()->Schedule(ConcurrentWriter, &state);
632 }
633 state.WaitForPendingWriters();
634 }
635 state.quit_flag_.store(true, std::memory_order_release);
636 state.Wait(TestState::DONE);
637 }
