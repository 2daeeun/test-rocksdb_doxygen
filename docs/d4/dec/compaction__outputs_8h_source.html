203 void SetOutputSlitKey(const std::optional<Slice> start,
204 const std::optional<Slice> end) {
205 const InternalKeyComparator* icmp =
206 &compaction_->column_family_data()->internal_comparator();
207
208 const InternalKey* output_split_key = compaction_->GetOutputSplitKey();
209 // Invalid output_split_key indicates that we do not need to split
210 if (output_split_key != nullptr) {
211 // We may only split the output when the cursor is in the range. Split
212 if ((!end.has_value() ||
213 icmp->user_comparator()->Compare(
214 ExtractUserKey(output_split_key->Encode()), *end) < 0) &&
215 (!start.has_value() ||
216 icmp->user_comparator()->Compare(
217 ExtractUserKey(output_split_key->Encode()), *start) > 0)) {
218 local_output_split_key_ = output_split_key;
219 }
220 }
221 }

259 // Close the current output. `open_file_func` is needed for creating new file
260 // for range-dels only output file.
261 Status CloseOutput(const Status& curr_status,
262 const CompactionFileOpenFunc& open_file_func,
263 const CompactionFileCloseFunc& close_file_func) {
264 Status status = curr_status;
265 // handle subcompaction containing only range deletions
266 if (status.ok() && !HasBuilder() && !HasOutput() && HasRangeDel()) {
267 status = open_file_func(*this);
268 }
269 if (HasBuilder()) {
270 const Slice empty_key{};
271 Status s = close_file_func(*this, status, empty_key);
272 if (!s.ok() && status.ok()) {
273 status = s;
274 }
275 }
276
277 return status;
278 }

280 // This subcompaction's output could be empty if compaction was aborted before
281 // this subcompaction had a chance to generate any output files. When
282 // subcompactions are executed sequentially this is more likely and will be
283 // particularly likely for the later subcompactions to be empty. Once they are
284 // run in parallel however it should be much rarer.
285 // It's caller's responsibility to make sure it's not empty.

376// helper struct to concatenate the last level and penultimate level outputs
377// which could be replaced by std::ranges::join_view() in c++20
378struct OutputIterator {
379 public:
380 explicit OutputIterator(const std::vector<CompactionOutputs::Output>& a,
381 const std::vector<CompactionOutputs::Output>& b)
382 : a_(a), b_(b) {
383 within_a = !a_.empty();
384 idx_ = 0;
385 }
386
387 OutputIterator begin() { return *this; }
388
389 OutputIterator end() { return *this; }
390
391 size_t size() { return a_.size() + b_.size(); }
392
393 const CompactionOutputs::Output& operator*() const {
394 return within_a ? a_[idx_] : b_[idx_];
395 }
396
397 OutputIterator& operator++() {
398 idx_++;
399 if (within_a && idx_ >= a_.size()) {
400 within_a = false;
401 idx_ = 0;
402 }
403 assert(within_a || idx_ <= b_.size());
404 return *this;
405 }
406
407 bool operator!=(const OutputIterator& /*rhs*/) const {
408 return within_a || idx_ < b_.size();
409 }
410
411 private:
412 const std::vector<CompactionOutputs::Output>& a_;
413 const std::vector<CompactionOutputs::Output>& b_;
414 bool within_a;
415 size_t idx_;
416};
