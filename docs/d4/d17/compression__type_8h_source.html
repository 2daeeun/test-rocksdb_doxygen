13// DB contents are stored in a set of blocks, each of which holds a
14// sequence of key,value pairs. Each block may be compressed before
15// being stored in a file. The following enum describes which
16// compression method (if any) is used to compress a block.
17
18enum CompressionType : unsigned char {
19 // NOTE: do not change the values of existing entries, as these are
20 // part of the persistent format on disk.
21 kNoCompression = 0x0,
22 kSnappyCompression = 0x1,
23 kZlibCompression = 0x2,
24 kBZip2Compression = 0x3,
25 kLZ4Compression = 0x4,
26 kLZ4HCCompression = 0x5,
27 kXpressCompression = 0x6,
28 kZSTD = 0x7,
29
30 // Only use kZSTDNotFinalCompression if you have to use ZSTD lib older than
31 // 0.8.0 or consider a possibility of downgrading the service or copying
32 // the database files to another service running with an older version of
33 // RocksDB that doesn't have kZSTD. Otherwise, you should use kZSTD. We will
34 // eventually remove the option from the public API.
35 kZSTDNotFinalCompression = 0x40,
36
37 // kDisableCompressionOption is used to disable some compression options.
38 kDisableCompressionOption = 0xff,
39};

41// Compression options for different compression algorithms like Zlib
42struct CompressionOptions {
43 // ==> BEGIN options that can be set by deprecated configuration syntax, <==
44 // ==> e.g. compression_opts=5:6:7:8:9:10:true:11:false <==
45 // ==> Please use compression_opts={level=6;strategy=7;} form instead. <==
46
47 // RocksDB's generic default compression level. Internally it'll be translated
48 // to the default compression level specific to the library being used (see
49 // comment above `ColumnFamilyOptions::compression`).
50 //
51 // The default value is the max 16-bit int as it'll be written out in OPTIONS
52 // file, which should be portable.
53 static constexpr int kDefaultCompressionLevel = 32767;
54
55 // zlib only: windowBits parameter. See https://www.zlib.net/manual.html
56 int window_bits = -14;
57
58 // Compression "level" applicable to zstd, zlib, LZ4, and LZ4HC. Except for
59 // kDefaultCompressionLevel (see above), the meaning of each value depends
60 // on the compression algorithm. Decreasing across non-
61 // `kDefaultCompressionLevel` values will either favor speed over
62 // compression ratio or have no effect.
63 //
64 // In LZ4 specifically, the absolute value of a negative `level` internally
65 // configures the `acceleration` parameter. For example, set `level=-10` for
66 // `acceleration=10`. This negation is necessary to ensure decreasing `level`
67 // values favor speed over compression ratio.
68 int level = kDefaultCompressionLevel;
69
70 // zlib only: strategy parameter. See https://www.zlib.net/manual.html
71 int strategy = 0;
72
73 // Maximum size of dictionaries used to prime the compression library.
74 // Enabling dictionary can improve compression ratios when there are
75 // repetitions across data blocks.
76 //
77 // The dictionary is created by sampling the SST file data. If
78 // `zstd_max_train_bytes` is nonzero, the samples are passed through zstd's
79 // dictionary generator (see comments for option `use_zstd_dict_trainer` for
80 // detail on dictionary generator). If `zstd_max_train_bytes` is zero, the
81 // random samples are used directly as the dictionary.
82 //
83 // When compression dictionary is disabled, we compress and write each block
84 // before buffering data for the next one. When compression dictionary is
85 // enabled, we buffer SST file data in-memory so we can sample it, as data
86 // can only be compressed and written after the dictionary has been finalized.
87 //
88 // The amount of data buffered can be limited by `max_dict_buffer_bytes`. This
89 // buffered memory is charged to the block cache when there is a block cache.
90 // If block cache insertion fails with `Status::MemoryLimit` (i.e., it is
91 // full), we finalize the dictionary with whatever data we have and then stop
92 // buffering.
93 uint32_t max_dict_bytes = 0;
94
95 // Maximum size of training data passed to zstd's dictionary trainer. Using
96 // zstd's dictionary trainer can achieve even better compression ratio
97 // improvements than using `max_dict_bytes` alone.
98 //
99 // The training data will be used to generate a dictionary of max_dict_bytes.
100 uint32_t zstd_max_train_bytes = 0;
101
102 // Number of threads for parallel compression.
103 // Parallel compression is enabled only if threads > 1.
104 // THE FEATURE IS STILL EXPERIMENTAL
105 //
106 // This option is valid only when BlockBasedTable is used.
107 //
108 // When parallel compression is enabled, SST size file sizes might be
109 // more inflated compared to the target size, because more data of unknown
110 // compressed size is in flight when compression is parallelized. To be
111 // reasonably accurate, this inflation is also estimated by using historical
112 // compression ratio and current bytes inflight.
113 uint32_t parallel_threads = 1;
114
115 // When the compression options are set by the user, it will be set to "true".
116 // For bottommost_compression_opts, to enable it, user must set enabled=true.
117 // Otherwise, bottommost compression will use compression_opts as default
118 // compression options.
119 //
120 // For compression_opts, if compression_opts.enabled=false, it is still
121 // used as compression options for compression process.
122 bool enabled = false;
123
124 // Limit on data buffering when gathering samples to build a dictionary. Zero
125 // means no limit. When dictionary is disabled (`max_dict_bytes == 0`),
126 // enabling this limit (`max_dict_buffer_bytes != 0`) has no effect.
127 //
128 // In compaction, the buffering is limited to the target file size (see
129 // `target_file_size_base` and `target_file_size_multiplier`) even if this
130 // setting permits more buffering. Since we cannot determine where the file
131 // should be cut until data blocks are compressed with dictionary, buffering
132 // more than the target file size could lead to selecting samples that belong
133 // to a later output SST.
134 //
135 // Limiting too strictly may harm dictionary effectiveness since it forces
136 // RocksDB to pick samples from the initial portion of the output SST, which
137 // may not be representative of the whole file. Configuring this limit below
138 // `zstd_max_train_bytes` (when enabled) can restrict how many samples we can
139 // pass to the dictionary trainer. Configuring it below `max_dict_bytes` can
140 // restrict the size of the final dictionary.
141 uint64_t max_dict_buffer_bytes = 0;
142
143 // Use zstd trainer to generate dictionaries. When this option is set to true,
144 // zstd_max_train_bytes of training data sampled from max_dict_buffer_bytes
145 // buffered data will be passed to zstd dictionary trainer to generate a
146 // dictionary of size max_dict_bytes.
147 //
148 // When this option is false, zstd's API ZDICT_finalizeDictionary() will be
149 // called to generate dictionaries. zstd_max_train_bytes of training sampled
150 // data will be passed to this API. Using this API should save CPU time on
151 // dictionary training, but the compression ratio may not be as good as using
152 // a dictionary trainer.
153 bool use_zstd_dict_trainer = true;
154
155 // ===> END options that can be set by deprecated configuration syntax <===
156 // ===> Use compression_opts={level=6;strategy=7;} form for below opts <===
157
158 // Essentially specifies a minimum acceptable compression ratio. A block is
159 // stored uncompressed if the compressed block does not achieve this ratio,
160 // because the downstream cost of decompression is not considered worth such
161 // a small savings (if any).
162 // However, the ratio is specified in a way that is efficient for checking.
163 // An integer from 1 to 1024 indicates the maximum allowable compressed bytes
164 // per 1KB of input, so the minimum acceptable ratio is 1024.0 / this value.
165 // For example, for a minimum ratio of 1.5:1, set to 683. See SetMinRatio().
166 // Default: abandon use of compression for a specific block or entry if
167 // compressed by less than 12.5% (minimum ratio of 1.143:1).
168 int max_compressed_bytes_per_kb = 1024 * 7 / 8;
169
170 // ZSTD only.
171 // Enable compression algorithm's checksum feature.
172 // (https://github.com/facebook/zstd/blob/d857369028d997c92ff1f1861a4d7f679a125464/lib/zstd.h#L428)
173 // Each compressed frame will have a 32-bit checksum attached. The checksum
174 // computed from the uncompressed data and can be verified during
175 // decompression.
176 bool checksum = false;
177
178 // A convenience function for setting max_compressed_bytes_per_kb based on a
179 // minimum acceptable compression ratio (uncompressed size over compressed
180 // size).
181 void SetMinRatio(double min_ratio) {
182 max_compressed_bytes_per_kb = static_cast<int>(1024.0 / min_ratio + 0.5);
183 }
184};
