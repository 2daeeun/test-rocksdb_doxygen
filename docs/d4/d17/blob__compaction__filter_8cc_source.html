28CompactionFilter::Decision BlobIndexCompactionFilterBase::FilterV2(
29 int level, const Slice& key, ValueType value_type, const Slice& value,
30 std::string* new_value, std::string* skip_until) const {
31 const CompactionFilter* ucf = user_comp_filter();
32 if (value_type != kBlobIndex) {
33 if (ucf == nullptr) {
34 return Decision::kKeep;
35 }
36 // Apply user compaction filter for inlined data.
37 CompactionFilter::Decision decision =
38 ucf->FilterV2(level, key, value_type, value, new_value, skip_until);
39 if (decision == Decision::kChangeValue) {
40 return HandleValueChange(key, new_value);
41 }
42 return decision;
43 }
44 BlobIndex blob_index;
45 Status s = blob_index.DecodeFrom(value);
46 if (!s.ok()) {
47 // Unable to decode blob index. Keeping the value.
48 return Decision::kKeep;
49 }
50 if (blob_index.HasTTL() && blob_index.expiration() <= current_time_) {
51 // Expired
52 expired_count_++;
53 expired_size_ += key.size() + value.size();
54 return Decision::kRemove;
55 }
56 if (!blob_index.IsInlined() &&
57 blob_index.file_number() < context_.next_file_number &&
58 context_.current_blob_files.count(blob_index.file_number()) == 0) {
59 // Corresponding blob file gone (most likely, evicted by FIFO eviction).
60 evicted_count_++;
61 evicted_size_ += key.size() + value.size();
62 return Decision::kRemove;
63 }
64 if (context_.fifo_eviction_seq > 0 && blob_index.HasTTL() &&
65 blob_index.expiration() < context_.evict_expiration_up_to) {
66 // Hack: Internal key is passed to BlobIndexCompactionFilter for it to
67 // get sequence number.
68 ParsedInternalKey ikey;
69 if (!ParseInternalKey(
70 key, &ikey,
71 context_.blob_db_impl->db_options_.allow_data_in_errors)
72 .ok()) {
73 assert(false);
74 return Decision::kKeep;
75 }
76 // Remove keys that could have been remove by last FIFO eviction.
77 // If get error while parsing key, ignore and continue.
78 if (ikey.sequence < context_.fifo_eviction_seq) {
79 evicted_count_++;
80 evicted_size_ += key.size() + value.size();
81 return Decision::kRemove;
82 }
83 }
84 // Apply user compaction filter for all non-TTL blob data.
85 if (ucf != nullptr && !blob_index.HasTTL()) {
86 // Hack: Internal key is passed to BlobIndexCompactionFilter for it to
87 // get sequence number.
88 ParsedInternalKey ikey;
89 if (!ParseInternalKey(
90 key, &ikey,
91 context_.blob_db_impl->db_options_.allow_data_in_errors)
92 .ok()) {
93 assert(false);
94 return Decision::kKeep;
95 }
96 // Read value from blob file.
97 PinnableSlice blob;
98 CompressionType compression_type = kNoCompression;
99 constexpr bool need_decompress = true;
100 if (!ReadBlobFromOldFile(ikey.user_key, blob_index, &blob, need_decompress,
101 &compression_type)) {
102 return Decision::kIOError;
103 }
104 CompactionFilter::Decision decision = ucf->FilterV2(
105 level, ikey.user_key, kValue, blob, new_value, skip_until);
106 if (decision == Decision::kChangeValue) {
107 return HandleValueChange(ikey.user_key, new_value);
108 }
109 return decision;
110 }
111 return Decision::kKeep;
112}

114CompactionFilter::Decision BlobIndexCompactionFilterBase::HandleValueChange(
115 const Slice& key, std::string* new_value) const {
116 BlobDBImpl* const blob_db_impl = context_.blob_db_impl;
117 assert(blob_db_impl);
118
119 if (new_value->size() < blob_db_impl->bdb_options_.min_blob_size) {
120 // Keep new_value inlined.
121 return Decision::kChangeValue;
122 }
123 if (!OpenNewBlobFileIfNeeded()) {
124 return Decision::kIOError;
125 }
126 Slice new_blob_value(*new_value);
127 std::string compression_output;
128 if (blob_db_impl->bdb_options_.compression != kNoCompression) {
129 new_blob_value =
130 blob_db_impl->GetCompressedSlice(new_blob_value, &compression_output);
131 }
132 uint64_t new_blob_file_number = 0;
133 uint64_t new_blob_offset = 0;
134 if (!WriteBlobToNewFile(key, new_blob_value, &new_blob_file_number,
135 &new_blob_offset)) {
136 return Decision::kIOError;
137 }
138 if (!CloseAndRegisterNewBlobFileIfNeeded()) {
139 return Decision::kIOError;
140 }
141 BlobIndex::EncodeBlob(new_value, new_blob_file_number, new_blob_offset,
142 new_blob_value.size(),
143 blob_db_impl->bdb_options_.compression);
144 return Decision::kChangeBlobIndex;
145}

175bool BlobIndexCompactionFilterBase::OpenNewBlobFileIfNeeded() const {
176 if (IsBlobFileOpened()) {

178 }
179

204bool BlobIndexCompactionFilterBase::ReadBlobFromOldFile(
205 const Slice& key, const BlobIndex& blob_index, PinnableSlice* blob,
206 bool need_decompress, CompressionType* compression_type) const {
207 BlobDBImpl* const blob_db_impl = context_.blob_db_impl;
208 assert(blob_db_impl);
209
210 Status s = blob_db_impl->GetRawBlobFromFile(
211 key, blob_index.file_number(), blob_index.offset(), blob_index.size(),
212 blob, compression_type);
213
214 if (!s.ok()) {
215 ROCKS_LOG_ERROR(
216 blob_db_impl->db_options_.info_log,
217 "Error reading blob during compaction/GC, key: %s (%s), status: %s",
218 key.ToString(/* output_hex */ true).c_str(),
219 blob_index.DebugString(/* output_hex */ true).c_str(),
220 s.ToString().c_str());
221
222 return false;
223 }
224
225 if (need_decompress && *compression_type != kNoCompression) {
226 s = blob_db_impl->DecompressSlice(*blob, *compression_type, blob);
227 if (!s.ok()) {
228 ROCKS_LOG_ERROR(
229 blob_db_impl->db_options_.info_log,
230 "Uncompression error during blob read from file: %" PRIu64
231 " blob_offset: %" PRIu64 " blob_size: %" PRIu64
232 " key: %s status: '%s'",
233 blob_index.file_number(), blob_index.offset(), blob_index.size(),
234 key.ToString(/* output_hex */ true).c_str(), s.ToString().c_str());
235
236 return false;
237 }
238 }
239
240 return true;
241}

243bool BlobIndexCompactionFilterBase::WriteBlobToNewFile(
244 const Slice& key, const Slice& blob, uint64_t* new_blob_file_number,
245 uint64_t* new_blob_offset) const {
246 TEST_SYNC_POINT("BlobIndexCompactionFilterBase::WriteBlobToNewFile");
247 assert(new_blob_file_number);
248 assert(new_blob_offset);
249
250 assert(blob_file_);
251 *new_blob_file_number = blob_file_->BlobFileNumber();
252
253 assert(writer_);
254 uint64_t new_key_offset = 0;
255 // TODO: plumb Env::IOActivity, Env::IOPriority
256 const Status s = writer_->AddRecord(WriteOptions(), key, blob, kNoExpiration,
257 &new_key_offset, new_blob_offset);
258
259 if (!s.ok()) {
260 const BlobDBImpl* const blob_db_impl = context_.blob_db_impl;
261 assert(blob_db_impl);
262
263 ROCKS_LOG_ERROR(blob_db_impl->db_options_.info_log,
264 "Error writing blob to new file %s during compaction/GC, "
265 "key: %s, status: %s",
266 blob_file_->PathName().c_str(),
267 key.ToString(/* output_hex */ true).c_str(),
268 s.ToString().c_str());
269 return false;
270 }
271
272 const uint64_t new_size =
273 BlobLogRecord::kHeaderSize + key.size() + blob.size();
274 blob_file_->BlobRecordAdded(new_size);
275
276 BlobDBImpl* const blob_db_impl = context_.blob_db_impl;
277 assert(blob_db_impl);
278
279 blob_db_impl->total_blob_size_ += new_size;
280
281 return true;
282}

297bool BlobIndexCompactionFilterBase::CloseAndRegisterNewBlobFile() const {
298 BlobDBImpl* const blob_db_impl = context_.blob_db_impl;
299 assert(blob_db_impl);
300 assert(blob_file_);
301
302 Status s;
303
304 {
305 WriteLock wl(&blob_db_impl->mutex_);
306
307 // TODO: plumb Env::IOActivity, Env::IOPriority
308 s = blob_db_impl->CloseBlobFile(WriteOptions(), blob_file_);
309
310 // Note: we delay registering the new blob file until it's closed to
311 // prevent FIFO eviction from processing it during compaction/GC.
312 blob_db_impl->RegisterBlobFile(blob_file_);
313 }
314
315 assert(blob_file_->Immutable());
316
317 if (!s.ok()) {
318 ROCKS_LOG_ERROR(
319 blob_db_impl->db_options_.info_log,
320 "Error closing new blob file %s during compaction/GC, status: %s",
321 blob_file_->PathName().c_str(), s.ToString().c_str());
322 }
323
324 blob_file_.reset();
325 return s.ok();
326}

328CompactionFilter::BlobDecision BlobIndexCompactionFilterGC::PrepareBlobOutput(
329 const Slice& key, const Slice& existing_value,
330 std::string* new_value) const {
331 assert(new_value);
332
333 const BlobDBImpl* const blob_db_impl = context().blob_db_impl;
334 (void)blob_db_impl;
335
336 assert(blob_db_impl);
337 assert(blob_db_impl->bdb_options_.enable_garbage_collection);
338
339 BlobIndex blob_index;
340 const Status s = blob_index.DecodeFrom(existing_value);
341 if (!s.ok()) {
342 gc_stats_.SetError();
343 return BlobDecision::kCorruption;
344 }
345
346 if (blob_index.IsInlined()) {
347 gc_stats_.AddBlob(blob_index.value().size());
348
349 return BlobDecision::kKeep;
350 }
351
352 gc_stats_.AddBlob(blob_index.size());
353
354 if (blob_index.HasTTL()) {
355 return BlobDecision::kKeep;
356 }
357
358 if (blob_index.file_number() >= context_gc_.cutoff_file_number) {
359 return BlobDecision::kKeep;
360 }
361
362 // Note: each compaction generates its own blob files, which, depending on the
363 // workload, might result in many small blob files. The total number of files
364 // is bounded though (determined by the number of compactions and the blob
365 // file size option).
366 if (!OpenNewBlobFileIfNeeded()) {
367 gc_stats_.SetError();
368 return BlobDecision::kIOError;
369 }
370
371 PinnableSlice blob;
372 CompressionType compression_type = kNoCompression;
373 std::string compression_output;
374 if (!ReadBlobFromOldFile(key, blob_index, &blob, false, &compression_type)) {
375 gc_stats_.SetError();
376 return BlobDecision::kIOError;
377 }
378
379 // If the compression_type is changed, re-compress it with the new compression
380 // type.
381 if (compression_type != blob_db_impl->bdb_options_.compression) {
382 if (compression_type != kNoCompression) {
383 const Status status =
384 blob_db_impl->DecompressSlice(blob, compression_type, &blob);
385 if (!status.ok()) {
386 gc_stats_.SetError();
387 return BlobDecision::kCorruption;
388 }
389 }
390 if (blob_db_impl->bdb_options_.compression != kNoCompression) {
391 blob_db_impl->GetCompressedSlice(blob, &compression_output);
392 blob = PinnableSlice(&compression_output);
393 blob.PinSelf();
394 }
395 }
396
397 uint64_t new_blob_file_number = 0;
398 uint64_t new_blob_offset = 0;
399 if (!WriteBlobToNewFile(key, blob, &new_blob_file_number, &new_blob_offset)) {
400 gc_stats_.SetError();
401 return BlobDecision::kIOError;
402 }
403
404 if (!CloseAndRegisterNewBlobFileIfNeeded()) {
405 gc_stats_.SetError();
406 return BlobDecision::kIOError;
407 }
408
409 BlobIndex::EncodeBlob(new_value, new_blob_file_number, new_blob_offset,
410 blob.size(), compression_type);
411
412 gc_stats_.AddRelocatedBlob(blob_index.size());
413
414 return BlobDecision::kChangeValue;
415}

439std::unique_ptr<CompactionFilter>
440BlobIndexCompactionFilterFactory::CreateCompactionFilter(
441 const CompactionFilter::Context& _context) {
442 assert(clock());
443
444 int64_t current_time = 0;
445 Status s = clock()->GetCurrentTime(&current_time);
446 if (!s.ok()) {
447 return nullptr;
448 }
449 assert(current_time >= 0);
450
451 assert(blob_db_impl());
452
453 BlobCompactionContext context;
454 blob_db_impl()->GetCompactionContext(&context);
455
456 std::unique_ptr<CompactionFilter> user_comp_filter_from_factory =
457 CreateUserCompactionFilterFromFactory(_context);
458
459 return std::unique_ptr<CompactionFilter>(new BlobIndexCompactionFilter(
460 std::move(context), user_comp_filter(),
461 std::move(user_comp_filter_from_factory), current_time, statistics()));
462}
