15// An internal iterator that wraps another one and ensures that any keys
16// returned are strictly within a range [start, end). If the underlying
17// iterator has already performed the bounds checking, it relies on that result;
18// otherwise, it performs the necessary key comparisons itself. Both bounds
19// are optional.
20class ClippingIterator : public InternalIterator {
21 public:
22 ClippingIterator(InternalIterator* iter, const Slice* start, const Slice* end,
23 const CompareInterface* cmp)
24 : iter_(iter), start_(start), end_(end), cmp_(cmp), valid_(false) {
25 assert(iter_);
26 assert(cmp_);
27 assert(!start_ || !end_ || cmp_->Compare(*start_, *end_) <= 0);
28
29 UpdateAndEnforceBounds();
30 }
31
32 bool Valid() const override { return valid_; }
33
34 void SeekToFirst() override {
35 if (start_) {
36 iter_->Seek(*start_);
37 } else {
38 iter_->SeekToFirst();
39 }
40
41 UpdateAndEnforceUpperBound();
42 }
43
44 void SeekToLast() override {
45 if (end_) {
46 iter_->SeekForPrev(*end_);
47
48 // Upper bound is exclusive, so we need a key which is strictly smaller
49 if (iter_->Valid() && cmp_->Compare(iter_->key(), *end_) == 0) {
50 iter_->Prev();
51 }
52 } else {
53 iter_->SeekToLast();
54 }
55
56 UpdateAndEnforceLowerBound();
57 }
58
59 void Seek(const Slice& target) override {
60 if (start_ && cmp_->Compare(target, *start_) < 0) {
61 iter_->Seek(*start_);
62 UpdateAndEnforceUpperBound();
63 return;
64 }
65
66 if (end_ && cmp_->Compare(target, *end_) >= 0) {
67 valid_ = false;
68 return;
69 }
70
71 iter_->Seek(target);
72 UpdateAndEnforceUpperBound();
73 }
74
75 void SeekForPrev(const Slice& target) override {
76 if (start_ && cmp_->Compare(target, *start_) < 0) {
77 valid_ = false;
78 return;
79 }
80
81 if (end_ && cmp_->Compare(target, *end_) >= 0) {
82 iter_->SeekForPrev(*end_);
83
84 // Upper bound is exclusive, so we need a key which is strictly smaller
85 if (iter_->Valid() && cmp_->Compare(iter_->key(), *end_) == 0) {
86 iter_->Prev();
87 }
88
89 UpdateAndEnforceLowerBound();
90 return;
91 }
92
93 iter_->SeekForPrev(target);
94 UpdateAndEnforceLowerBound();
95 }
96
97 void Next() override {
98 assert(valid_);
99 iter_->Next();
100 UpdateAndEnforceUpperBound();
101 }
102
103 bool NextAndGetResult(IterateResult* result) override {
104 assert(valid_);
105 assert(result);
106
107 IterateResult res;
108 valid_ = iter_->NextAndGetResult(&res);
109
110 if (!valid_) {
111 return false;
112 }
113
114 if (end_) {
115 EnforceUpperBoundImpl(res.bound_check_result);
116
117 if (!valid_) {
118 return false;
119 }
120 }
121
122 res.bound_check_result = IterBoundCheck::kInbound;
123 *result = res;
124
125 return true;
126 }
127
128 void Prev() override {
129 assert(valid_);
130 iter_->Prev();
131 UpdateAndEnforceLowerBound();
132 }
133
134 Slice key() const override {
135 assert(valid_);
136 return iter_->key();
137 }
138
139 Slice user_key() const override {
140 assert(valid_);
141 return iter_->user_key();
142 }
143
144 Slice value() const override {
145 assert(valid_);
146 return iter_->value();
147 }
148
149 Status status() const override { return iter_->status(); }
150
151 bool PrepareValue() override {
152 assert(valid_);
153
154 if (iter_->PrepareValue()) {
155 return true;
156 }
157
158 assert(!iter_->Valid());
159 valid_ = false;
160 return false;
161 }
162
163 bool MayBeOutOfLowerBound() override {
164 assert(valid_);
165 return false;
166 }
167
168 IterBoundCheck UpperBoundCheckResult() override {
169 assert(valid_);
170 return IterBoundCheck::kInbound;
171 }
172
173 void SetPinnedItersMgr(PinnedIteratorsManager* pinned_iters_mgr) override {
174 iter_->SetPinnedItersMgr(pinned_iters_mgr);
175 }
176
177 bool IsKeyPinned() const override {
178 assert(valid_);
179 return iter_->IsKeyPinned();
180 }
181
182 bool IsValuePinned() const override {
183 assert(valid_);
184 return iter_->IsValuePinned();
185 }
186
187 Status GetProperty(std::string prop_name, std::string* prop) override {
188 return iter_->GetProperty(prop_name, prop);
189 }
190
191 bool IsDeleteRangeSentinelKey() const override {
192 assert(valid_);
193 return iter_->IsDeleteRangeSentinelKey();
194 }
195
196 private:
197 void UpdateValid() {
198 assert(!iter_->Valid() || iter_->status().ok());
199
200 valid_ = iter_->Valid();
201 }
202
203 void EnforceUpperBoundImpl(IterBoundCheck bound_check_result) {
204 if (bound_check_result == IterBoundCheck::kInbound) {
205 return;
206 }
207
208 if (bound_check_result == IterBoundCheck::kOutOfBound) {
209 valid_ = false;
210 return;
211 }
212
213 assert(bound_check_result == IterBoundCheck::kUnknown);
214
215 if (cmp_->Compare(key(), *end_) >= 0) {
216 valid_ = false;
217 }
218 }
219
220 void EnforceUpperBound() {
221 if (!valid_) {
222 return;
223 }
224
225 if (!end_) {
226 return;
227 }
228
229 EnforceUpperBoundImpl(iter_->UpperBoundCheckResult());
230 }
231
232 void EnforceLowerBound() {
233 if (!valid_) {
234 return;
235 }
236
237 if (!start_) {
238 return;
239 }
240
241 if (!iter_->MayBeOutOfLowerBound()) {
242 return;
243 }
244
245 if (cmp_->Compare(key(), *start_) < 0) {
246 valid_ = false;
247 }
248 }
249
250 void AssertBounds() {
251 assert(!valid_ || !start_ || cmp_->Compare(key(), *start_) >= 0);
252 assert(!valid_ || !end_ || cmp_->Compare(key(), *end_) < 0);
253 }
254
255 void UpdateAndEnforceBounds() {
256 UpdateValid();
257 EnforceUpperBound();
258 EnforceLowerBound();
259 AssertBounds();
260 }
261
262 void UpdateAndEnforceUpperBound() {
263 UpdateValid();
264 EnforceUpperBound();
265 AssertBounds();
266 }
267
268 void UpdateAndEnforceLowerBound() {
269 UpdateValid();
270 EnforceLowerBound();
271 AssertBounds();
272 }
273
274 InternalIterator* iter_;
275 const Slice* start_;
276 const Slice* end_;
277 const CompareInterface* cmp_;
278 bool valid_;
279};
