53static Slice Key(int i, char* buffer) {
54 std::string s;
55 PutFixed32(&s, static_cast<uint32_t>(i));
56 memcpy(buffer, s.c_str(), sizeof(i));
57 return Slice(buffer, sizeof(i));
58}

60static int NextLength(int length) {
61 if (length < 10) {
62 length += 1;
63 } else if (length < 100) {
64 length += 10;
65 } else if (length < 1000) {
66 length += 100;
67 } else {
68 length += 1000;
69 }
70 return length;
71}

73class FullBloomTest : public testing::TestWithParam<std::string> {
74 protected:
75 BlockBasedTableOptions table_options_;
76
77 private:
78 std::shared_ptr<const FilterPolicy>& policy_;
79 std::unique_ptr<FilterBitsBuilder> bits_builder_;
80 std::unique_ptr<FilterBitsReader> bits_reader_;
81 std::unique_ptr<const char[]> buf_;
82 size_t filter_size_;
83
84 public:
85 FullBloomTest() : policy_(table_options_.filter_policy), filter_size_(0) {
86 ResetPolicy();
87 }
88
89 BuiltinFilterBitsBuilder* GetBuiltinFilterBitsBuilder() {
90 // Throws on bad cast
91 return dynamic_cast<BuiltinFilterBitsBuilder*>(bits_builder_.get());
92 }
93
94 const BloomLikeFilterPolicy* GetBloomLikeFilterPolicy() {
95 // Throws on bad cast
96 return &dynamic_cast<const BloomLikeFilterPolicy&>(*policy_);
97 }
98
99 void Reset() {
100 bits_builder_.reset(BloomFilterPolicy::GetBuilderFromContext(
101 FilterBuildingContext(table_options_)));
102 bits_reader_.reset(nullptr);
103 buf_.reset(nullptr);
104 filter_size_ = 0;
105 }
106
107 void ResetPolicy(double bits_per_key) {
108 policy_ = BloomLikeFilterPolicy::Create(GetParam(), bits_per_key);
109 Reset();
110 }
111
112 void ResetPolicy() { ResetPolicy(FLAGS_bits_per_key); }
113
114 void Add(const Slice& s) { bits_builder_->AddKey(s); }
115
116 void OpenRaw(const Slice& s) {
117 bits_reader_.reset(policy_->GetFilterBitsReader(s));
118 }
119
120 void Build() {
121 Slice filter = bits_builder_->Finish(&buf_);
122 bits_reader_.reset(policy_->GetFilterBitsReader(filter));
123 filter_size_ = filter.size();
124 }
125
126 size_t FilterSize() const { return filter_size_; }
127
128 Slice FilterData() { return Slice(buf_.get(), filter_size_); }
129
130 int GetNumProbesFromFilterData() {
131 assert(filter_size_ >= 5);
132 int8_t raw_num_probes = static_cast<int8_t>(buf_.get()[filter_size_ - 5]);
133 if (raw_num_probes == -1) { // New bloom filter marker
134 return static_cast<uint8_t>(buf_.get()[filter_size_ - 3]);
135 } else {
136 return raw_num_probes;
137 }
138 }
139
140 int GetRibbonSeedFromFilterData() {
141 assert(filter_size_ >= 5);
142 // Check for ribbon marker
143 assert(-2 == static_cast<int8_t>(buf_.get()[filter_size_ - 5]));
144 return static_cast<uint8_t>(buf_.get()[filter_size_ - 4]);
145 }
146
147 bool Matches(const Slice& s) {
148 if (bits_reader_ == nullptr) {
149 Build();
150 }
151 return bits_reader_->MayMatch(s);
152 }
153
154 // Provides a kind of fingerprint on the Bloom filter's
155 // behavior, for reasonbly high FP rates.
156 uint64_t PackedMatches() {
157 char buffer[sizeof(int)];
158 uint64_t result = 0;
159 for (int i = 0; i < 64; i++) {
160 if (Matches(Key(i + 12345, buffer))) {
161 result |= uint64_t{1} << i;
162 }
163 }
164 return result;
165 }
166
167 // Provides a kind of fingerprint on the Bloom filter's
168 // behavior, for lower FP rates.
169 std::string FirstFPs(int count) {
170 char buffer[sizeof(int)];
171 std::string rv;
172 int fp_count = 0;
173 for (int i = 0; i < 1000000; i++) {
174 // Pack four match booleans into each hexadecimal digit
175 if (Matches(Key(i + 1000000, buffer))) {
176 ++fp_count;
177 rv += std::to_string(i);
178 if (fp_count == count) {
179 break;
180 }
181 rv += ',';
182 }
183 }
184 return rv;
185 }
186
187 double FalsePositiveRate() {
188 char buffer[sizeof(int)];
189 int result = 0;
190 for (int i = 0; i < 10000; i++) {
191 if (Matches(Key(i + 1000000000, buffer))) {
192 result++;
193 }
194 }
195 return result / 10000.0;
196 }
197};

199TEST_P(FullBloomTest, FilterSize) {
200 // In addition to checking the consistency of space computation, we are
201 // checking that denoted and computed doubles are interpreted as expected
202 // as bits_per_key values.
203 bool some_computed_less_than_denoted = false;
204 // Note: to avoid unproductive configurations, bits_per_key < 0.5 is rounded
205 // down to 0 (no filter), and 0.5 <= bits_per_key < 1.0 is rounded up to 1
206 // bit per key (1000 millibits). Also, enforced maximum is 100 bits per key
207 // (100000 millibits).
208 for (auto bpk : std::vector<std::pair<double, int> >{{-HUGE_VAL, 0},
209 {-INFINITY, 0},
210 {0.0, 0},
211 {0.499, 0},
212 {0.5, 1000},
213 {1.234, 1234},
214 {3.456, 3456},
215 {9.5, 9500},
216 {10.0, 10000},
217 {10.499, 10499},
218 {21.345, 21345},
219 {99.999, 99999},
220 {1234.0, 100000},
221 {HUGE_VAL, 100000},
222 {INFINITY, 100000},
223 {NAN, 100000}}) {
224 ResetPolicy(bpk.first);
225 auto bfp = GetBloomLikeFilterPolicy();
226 EXPECT_EQ(bpk.second, bfp->GetMillibitsPerKey());
227 EXPECT_EQ((bpk.second + 500) / 1000, bfp->GetWholeBitsPerKey());
228
229 double computed = bpk.first;
230 // This transforms e.g. 9.5 -> 9.499999999999998, which we still
231 // round to 10 for whole bits per key.
232 computed += 0.5;
233 computed /= 1234567.0;
234 computed *= 1234567.0;
235 computed -= 0.5;
236 some_computed_less_than_denoted |= (computed < bpk.first);
237 ResetPolicy(computed);
238 bfp = GetBloomLikeFilterPolicy();
239 EXPECT_EQ(bpk.second, bfp->GetMillibitsPerKey());
240 EXPECT_EQ((bpk.second + 500) / 1000, bfp->GetWholeBitsPerKey());
241
242 auto bits_builder = GetBuiltinFilterBitsBuilder();
243 if (bpk.second == 0) {
244 ASSERT_EQ(bits_builder, nullptr);
245 continue;
246 }
247
248 size_t n = 1;
249 size_t space = 0;
250 for (; n < 1000000; n += 1 + n / 1000) {
251 // Ensure consistency between CalculateSpace and ApproximateNumEntries
252 space = bits_builder->CalculateSpace(n);
253 size_t n2 = bits_builder->ApproximateNumEntries(space);
254 EXPECT_GE(n2, n);
255 size_t space2 = bits_builder->CalculateSpace(n2);
256 if (n > 12000 && GetParam() == kStandard128Ribbon) {
257 // TODO(peterd): better approximation?
258 EXPECT_GE(space2, space);
259 EXPECT_LE(space2 * 0.998, space * 1.0);
260 } else {
261 EXPECT_EQ(space2, space);
262 }
263 }
264 // Until size_t overflow
265 for (; n < (n + n / 3); n += n / 3) {
266 // Ensure space computation is not overflowing; capped is OK
267 size_t space2 = bits_builder->CalculateSpace(n);
268 EXPECT_GE(space2, space);
269 space = space2;
270 }
271 }
272 // Check that the compiler hasn't optimized our computation into nothing
273 EXPECT_TRUE(some_computed_less_than_denoted);
274 ResetPolicy();
275}

292TEST_P(FullBloomTest, FullVaryingLengths) {
293 // Match how this test was originally built
294 table_options_.optimize_filters_for_memory = false;
295
296 char buffer[sizeof(int)];
297
298 // Count number of filters that significantly exceed the false positive rate
299 int mediocre_filters = 0;
300 int good_filters = 0;
301
302 for (int length = 1; length <= 10000; length = NextLength(length)) {
303 Reset();
304 for (int i = 0; i < length; i++) {
305 Add(Key(i, buffer));
306 }
307 Build();
308
309 EXPECT_LE(FilterSize(), (size_t)((length * FLAGS_bits_per_key / 8) +
310 CACHE_LINE_SIZE * 2 + 5));
311
312 // All added keys must match
313 for (int i = 0; i < length; i++) {
314 ASSERT_TRUE(Matches(Key(i, buffer)))
315 << "Length " << length << "; key " << i;
316 }
317
318 // Check false positive rate
319 double rate = FalsePositiveRate();
320 if (kVerbose >= 1) {
321 fprintf(stderr, "False positives: %5.2f%% @ length = %6d ; bytes = %6d\n",
322 rate * 100.0, length, static_cast<int>(FilterSize()));
323 }
324 if (FLAGS_bits_per_key == 10) {
325 EXPECT_LE(rate, 0.02); // Must not be over 2%
326 if (rate > 0.0125) {
327 mediocre_filters++; // Allowed, but not too often
328 } else {
329 good_filters++;
330 }
331 }
332 }
333 if (kVerbose >= 1) {
334 fprintf(stderr, "Filters: %d good, %d mediocre\n", good_filters,
335 mediocre_filters);
336 }
337 EXPECT_LE(mediocre_filters, good_filters / 5);
338}

340TEST_P(FullBloomTest, OptimizeForMemory) {
341 // Verify default option
342 EXPECT_EQ(BlockBasedTableOptions().optimize_filters_for_memory, true);
343
344 char buffer[sizeof(int)];
345 for (bool offm : {true, false}) {
346 table_options_.optimize_filters_for_memory = offm;
347 ResetPolicy();
348 Random32 rnd(12345);
349 uint64_t total_size = 0;
350 uint64_t total_mem = 0;
351 int64_t total_keys = 0;
352 double total_fp_rate = 0;
353 constexpr int nfilters = 100;
354 for (int i = 0; i < nfilters; ++i) {
355 int nkeys = static_cast<int>(rnd.Uniformish(10000)) + 100;
356 Reset();
357 for (int j = 0; j < nkeys; ++j) {
358 Add(Key(j, buffer));
359 }
360 Build();
361 size_t size = FilterData().size();
362 total_size += size;
363 // optimize_filters_for_memory currently only has an effect with
364 // malloc_usable_size support, but we run the rest of the test to ensure
365 // no bad behavior without it.
366#ifdef ROCKSDB_MALLOC_USABLE_SIZE
367 size = malloc_usable_size(const_cast<char*>(FilterData().data()));
368#endif // ROCKSDB_MALLOC_USABLE_SIZE
369 total_mem += size;
370 total_keys += nkeys;
371 total_fp_rate += FalsePositiveRate();
372 }
373 if (FLAGS_bits_per_key == 10) {
374 EXPECT_LE(total_fp_rate / double{nfilters}, 0.011);
375 EXPECT_GE(total_fp_rate / double{nfilters},
376 CACHE_LINE_SIZE >= 256 ? 0.007 : 0.008);
377 }
378
379 int64_t ex_min_total_size = int64_t{FLAGS_bits_per_key} * total_keys / 8;
380 if (GetParam() == kStandard128Ribbon) {
381 // ~ 30% savings vs. Bloom filter
382 ex_min_total_size = 7 * ex_min_total_size / 10;
383 }
384 EXPECT_GE(static_cast<int64_t>(total_size), ex_min_total_size);
385
386 int64_t blocked_bloom_overhead = nfilters * (CACHE_LINE_SIZE + 5);
387 if (GetParam() == kLegacyBloom) {
388 // this config can add extra cache line to make odd number
389 blocked_bloom_overhead += nfilters * CACHE_LINE_SIZE;
390 }
391
392 EXPECT_GE(total_mem, total_size);
393
394 // optimize_filters_for_memory not implemented with legacy Bloom
395 if (offm && GetParam() != kLegacyBloom) {
396 // This value can include a small extra penalty for kExtraPadding
397 fprintf(stderr, "Internal fragmentation (optimized): %g%%\n",
398 (total_mem - total_size) * 100.0 / total_size);
399 // Less than 1% internal fragmentation
400 EXPECT_LE(total_mem, total_size * 101 / 100);
401 // Up to 2% storage penalty
402 EXPECT_LE(static_cast<int64_t>(total_size),
403 ex_min_total_size * 102 / 100 + blocked_bloom_overhead);
404 } else {
405 fprintf(stderr, "Internal fragmentation (not optimized): %g%%\n",
406 (total_mem - total_size) * 100.0 / total_size);
407 // TODO: add control checks for more allocators?
408#ifdef ROCKSDB_JEMALLOC
409 fprintf(stderr, "Jemalloc detected? %d\n", HasJemalloc());
410 if (HasJemalloc()) {
411#ifdef ROCKSDB_MALLOC_USABLE_SIZE
412 // More than 5% internal fragmentation
413 EXPECT_GE(total_mem, total_size * 105 / 100);
414#endif // ROCKSDB_MALLOC_USABLE_SIZE
415 }
416#endif // ROCKSDB_JEMALLOC
417 // No storage penalty, just usual overhead
418 EXPECT_LE(static_cast<int64_t>(total_size),
419 ex_min_total_size + blocked_bloom_overhead);
420 }
421 }

425TEST_F(ChargeFilterConstructionTest, RibbonFilterFallBackOnLargeBanding) {
426 constexpr std::size_t kCacheCapacity =
427 8 * CacheReservationManagerImpl<
428 CacheEntryRole::kFilterConstruction>::GetDummyEntrySize();
429 constexpr std::size_t num_entries_for_cache_full = kCacheCapacity / 8;
430
431 for (CacheEntryRoleOptions::Decision charge_filter_construction_mem :
432 {CacheEntryRoleOptions::Decision::kEnabled,
433 CacheEntryRoleOptions::Decision::kDisabled}) {
434 bool will_fall_back = charge_filter_construction_mem ==
435 CacheEntryRoleOptions::Decision::kEnabled;
436
437 BlockBasedTableOptions table_options;
438 table_options.cache_usage_options.options_overrides.insert(
439 {CacheEntryRole::kFilterConstruction,
440 {/*.charged = */ charge_filter_construction_mem}});
441 LRUCacheOptions lo;
442 lo.capacity = kCacheCapacity;
443 lo.num_shard_bits = 0; // 2^0 shard
444 lo.strict_capacity_limit = true;
445 std::shared_ptr<Cache> cache(NewLRUCache(lo));
446 table_options.block_cache = cache;
447 table_options.filter_policy =
448 BloomLikeFilterPolicy::Create(kStandard128Ribbon, FLAGS_bits_per_key);
449 FilterBuildingContext ctx(table_options);
450 std::unique_ptr<FilterBitsBuilder> filter_bits_builder(
451 table_options.filter_policy->GetBuilderWithContext(ctx));
452
453 char key_buffer[sizeof(int)];
454 for (std::size_t i = 0; i < num_entries_for_cache_full; ++i) {
455 filter_bits_builder->AddKey(Key(static_cast<int>(i), key_buffer));
456 }
457
458 std::unique_ptr<const char[]> buf;
459 Slice filter = filter_bits_builder->Finish(&buf);
460
461 // To verify Ribbon Filter fallbacks to Bloom Filter properly
462 // based on cache charging result
463 // See BloomFilterPolicy::GetBloomBitsReader re: metadata
464 // -1 = Marker for newer Bloom implementations
465 // -2 = Marker for Standard128 Ribbon
466 if (will_fall_back) {
467 EXPECT_EQ(filter.data()[filter.size() - 5], static_cast<char>(-1));
468 } else {
469 EXPECT_EQ(filter.data()[filter.size() - 5], static_cast<char>(-2));
470 }
471
472 if (charge_filter_construction_mem ==
473 CacheEntryRoleOptions::Decision::kEnabled) {
474 const size_t dummy_entry_num = static_cast<std::size_t>(std::ceil(
475 filter.size() * 1.0 /
476 CacheReservationManagerImpl<
477 CacheEntryRole::kFilterConstruction>::GetDummyEntrySize()));
478 EXPECT_GE(
479 cache->GetPinnedUsage(),
480 dummy_entry_num *
481 CacheReservationManagerImpl<
482 CacheEntryRole::kFilterConstruction>::GetDummyEntrySize());
483 EXPECT_LT(
484 cache->GetPinnedUsage(),
485 (dummy_entry_num + 1) *
486 CacheReservationManagerImpl<
487 CacheEntryRole::kFilterConstruction>::GetDummyEntrySize());
488 } else {
489 EXPECT_EQ(cache->GetPinnedUsage(), 0);
490 }
491 }
492}

815// A helper class for testing custom or corrupt filter bits as read by
816// built-in FilterBitsReaders.
817struct RawFilterTester {
818 // Buffer, from which we always return a tail Slice, so the
819 // last five bytes are always the metadata bytes.
820 std::array<char, 3000> data_{};
821 // Points five bytes from the end
822 char* metadata_ptr_;
823
824 RawFilterTester() : metadata_ptr_(&*(data_.end() - 5)) {}
825
826 Slice ResetNoFill(uint32_t len_without_metadata, uint32_t num_lines,
827 uint32_t num_probes) {
828 metadata_ptr_[0] = static_cast<char>(num_probes);
829 EncodeFixed32(metadata_ptr_ + 1, num_lines);
830 uint32_t len = len_without_metadata + /*metadata*/ 5;
831 assert(len <= data_.size());
832 return Slice(metadata_ptr_ - len_without_metadata, len);
833 }
834
835 Slice Reset(uint32_t len_without_metadata, uint32_t num_lines,
836 uint32_t num_probes, bool fill_ones) {
837 data_.fill(fill_ones ? 0xff : 0);
838 return ResetNoFill(len_without_metadata, num_lines, num_probes);
839 }
840
841 Slice ResetWeirdFill(uint32_t len_without_metadata, uint32_t num_lines,
842 uint32_t num_probes) {
843 for (uint32_t i = 0; i < data_.size(); ++i) {
844 data_[i] = static_cast<char>(0x7b7b >> (i % 7));
845 }
846 return ResetNoFill(len_without_metadata, num_lines, num_probes);
847 }
848};

1091static double GetEffectiveBitsPerKey(FilterBitsBuilder* builder) {
1092 union {
1093 uint64_t key_value = 0;
1094 char key_bytes[8];
1095 };
1096
1097 const unsigned kNumKeys = 1000;
1098
1099 Slice key_slice{key_bytes, 8};
1100 for (key_value = 0; key_value < kNumKeys; ++key_value) {
1101 builder->AddKey(key_slice);
1102 }
1103
1104 std::unique_ptr<const char[]> buf;
1105 auto filter = builder->Finish(&buf);
1106 return filter.size() * /*bits per byte*/ 8 / (1.0 * kNumKeys);
1107}
