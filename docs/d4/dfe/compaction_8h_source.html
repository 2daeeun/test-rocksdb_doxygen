23// Utility for comparing sstable boundary keys. Returns -1 if either a or b is
24// null which provides the property that a==null indicates a key that is less
25// than any key and b==null indicates a key that is greater than any key. Note
26// that the comparison is performed primarily on the user-key portion of the
27// key. If the user-keys compare equal, an additional test is made to sort
28// range tombstone sentinel keys before other keys with the same user-key. The
29// result is that 2 user-keys will compare equal if they differ purely on
30// their sequence number and value, but the range tombstone sentinel for that
31// user-key will compare not equal. This is necessary because the range
32// tombstone sentinel key is set as the largest key for an sstable even though
33// that key never appears in the database. We don't want adjacent sstables to
34// be considered overlapping if they are separated by the range tombstone
35// sentinel.
36int sstableKeyCompare(const Comparator* user_cmp, const Slice&, const Slice&);
37inline int sstableKeyCompare(const Comparator* user_cmp, const Slice& a,
38 const InternalKey& b) {
39 return sstableKeyCompare(user_cmp, a, b.Encode());
40}
41inline int sstableKeyCompare(const Comparator* user_cmp, const InternalKey& a,
42 const Slice& b) {
43 return sstableKeyCompare(user_cmp, a.Encode(), b);
44}
45inline int sstableKeyCompare(const Comparator* user_cmp, const InternalKey& a,
46 const InternalKey& b) {
47 return sstableKeyCompare(user_cmp, a.Encode(), b.Encode());
48}
49int sstableKeyCompare(const Comparator* user_cmp, const InternalKey* a,
50 const InternalKey& b);
51int sstableKeyCompare(const Comparator* user_cmp, const InternalKey& a,
52 const InternalKey* b);

54// An AtomicCompactionUnitBoundary represents a range of keys [smallest,
55// largest] that exactly spans one ore more neighbouring SSTs on the same
56// level. Every pair of SSTs in this range "overlap" (i.e., the largest
57// user key of one file is the smallest user key of the next file). These
58// boundaries are propagated down to RangeDelAggregator during compaction
59// to provide safe truncation boundaries for range tombstones.
60struct AtomicCompactionUnitBoundary {
61 const InternalKey* smallest = nullptr;
62 const InternalKey* largest = nullptr;
63};

65// The structure that manages compaction input files associated
66// with the same physical level.
67struct CompactionInputFiles {
68 int level;
69 std::vector<FileMetaData*> files;
70 std::vector<AtomicCompactionUnitBoundary> atomic_compaction_unit_boundaries;
71 inline bool empty() const { return files.empty(); }
72 inline size_t size() const { return files.size(); }
73 inline void clear() { files.clear(); }
74 inline FileMetaData* operator[](size_t i) const { return files[i]; }
75};
