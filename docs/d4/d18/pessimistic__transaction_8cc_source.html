61void PessimisticTransaction::Initialize(const TransactionOptions& txn_options) {
62 // Range lock manager uses address of transaction object as TXNID
63 const TransactionDBOptions& db_options = txn_db_impl_->GetTxnDBOptions();
64 if (db_options.lock_mgr_handle &&
65 db_options.lock_mgr_handle->getLockManager()->IsRangeLockSupported()) {
66 txn_id_ = reinterpret_cast<TransactionID>(this);
67 } else {
68 txn_id_ = GenTxnID();
69 }
70
71 txn_state_ = STARTED;
72
73 deadlock_detect_ = txn_options.deadlock_detect;
74 deadlock_detect_depth_ = txn_options.deadlock_detect_depth;
75 write_batch_.SetMaxBytes(txn_options.max_write_batch_size);
76 write_batch_.GetWriteBatch()->SetTrackTimestampSize(
77 txn_options.write_batch_track_timestamp_size);
78 skip_concurrency_control_ = txn_options.skip_concurrency_control;
79
80 lock_timeout_ = txn_options.lock_timeout * 1000;
81 if (lock_timeout_ < 0) {
82 // Lock timeout not set, use default
83 lock_timeout_ =
84 txn_db_impl_->GetTxnDBOptions().transaction_lock_timeout * 1000;
85 }
86
87 if (txn_options.expiration >= 0) {
88 expiration_time_ = start_time_ + txn_options.expiration * 1000;
89 } else {
90 expiration_time_ = 0;
91 }
92
93 if (txn_options.set_snapshot) {
94 SetSnapshot();
95 }
96
97 if (expiration_time_ > 0) {
98 txn_db_impl_->InsertExpirableTransaction(txn_id_, this);
99 }
100 use_only_the_last_commit_time_batch_for_recovery_ =
101 txn_options.use_only_the_last_commit_time_batch_for_recovery;
102 skip_prepare_ = txn_options.skip_prepare;
103
104 read_timestamp_ = kMaxTxnTimestamp;
105 commit_timestamp_ = kMaxTxnTimestamp;
106
107 commit_bypass_memtable_ = txn_options.commit_bypass_memtable;
108 write_batch_.SetTrackPerCFStat(txn_options.commit_bypass_memtable);
109}

169template <typename TValue>
170inline Status WriteCommittedTxn::GetForUpdateImpl(
171 const ReadOptions& read_options, ColumnFamilyHandle* column_family,
172 const Slice& key, TValue* value, bool exclusive, const bool do_validate) {
173 if (read_options.io_activity != Env::IOActivity::kUnknown) {
174 return Status::InvalidArgument(
175 "Cannot call GetForUpdate with `ReadOptions::io_activity` != "
176 "`Env::IOActivity::kUnknown`");
177 }
178 column_family =
179 column_family ? column_family : db_impl_->DefaultColumnFamily();
180 assert(column_family);
181 if (!read_options.timestamp) {
182 const Comparator* const ucmp = column_family->GetComparator();
183 assert(ucmp);
184 size_t ts_sz = ucmp->timestamp_size();
185 if (0 == ts_sz) {
186 return TransactionBaseImpl::GetForUpdate(read_options, column_family, key,
187 value, exclusive, do_validate);
188 }
189 } else {
190 Status s =
191 db_impl_->FailIfTsMismatchCf(column_family, *(read_options.timestamp));
192 if (!s.ok()) {
193 return s;
194 }
195 }
196
197 Status s = SanityCheckReadTimestamp(do_validate);
198 if (!s.ok()) {
199 return s;
200 }
201
202 if (!read_options.timestamp) {
203 ReadOptions read_opts_copy = read_options;
204 char ts_buf[sizeof(kMaxTxnTimestamp)];
205 EncodeFixed64(ts_buf, read_timestamp_);
206 Slice ts(ts_buf, sizeof(ts_buf));
207 read_opts_copy.timestamp = &ts;
208 return TransactionBaseImpl::GetForUpdate(read_opts_copy, column_family, key,
209 value, exclusive, do_validate);
210 }
211 assert(read_options.timestamp);
212 const char* const ts_buf = read_options.timestamp->data();
213 assert(read_options.timestamp->size() == sizeof(kMaxTxnTimestamp));
214 TxnTimestamp ts = DecodeFixed64(ts_buf);
215 if (ts != read_timestamp_) {
216 return Status::InvalidArgument("Must read from the same read_timestamp");
217 }
218 return TransactionBaseImpl::GetForUpdate(read_options, column_family, key,
219 value, exclusive, do_validate);
220}

222Status WriteCommittedTxn::GetEntityForUpdate(const ReadOptions& read_options,
223 ColumnFamilyHandle* column_family,
224 const Slice& key,
225 PinnableWideColumns* columns,
226 bool exclusive, bool do_validate) {
227 if (!column_family) {
228 return Status::InvalidArgument(
229 "Cannot call GetEntityForUpdate without a column family handle");
230 }
231
232 const Comparator* const ucmp = column_family->GetComparator();
233 assert(ucmp);
234 const size_t ts_sz = ucmp->timestamp_size();
235
236 if (ts_sz == 0) {
237 return TransactionBaseImpl::GetEntityForUpdate(
238 read_options, column_family, key, columns, exclusive, do_validate);
239 }
240
241 assert(ts_sz > 0);
242 Status s = SanityCheckReadTimestamp(do_validate);
243 if (!s.ok()) {
244 return s;
245 }
246
247 std::string ts_buf;
248 PutFixed64(&ts_buf, read_timestamp_);
249 Slice ts(ts_buf);
250
251 if (!read_options.timestamp) {
252 ReadOptions read_options_copy = read_options;
253 read_options_copy.timestamp = &ts;
254
255 return TransactionBaseImpl::GetEntityForUpdate(
256 read_options_copy, column_family, key, columns, exclusive, do_validate);
257 }
258
259 assert(read_options.timestamp);
260 if (*read_options.timestamp != ts) {
261 return Status::InvalidArgument("Must read from the same read timestamp");
262 }
263
264 return TransactionBaseImpl::GetEntityForUpdate(
265 read_options, column_family, key, columns, exclusive, do_validate);
266}

476template <typename TKey, typename TOperation>
477Status WriteCommittedTxn::Operate(ColumnFamilyHandle* column_family,
478 const TKey& key, const bool do_validate,
479 const bool assume_tracked,
480 TOperation&& operation) {
481 Status s;
482 if constexpr (std::is_same_v<Slice, TKey>) {
483 s = TryLock(column_family, key, /*read_only=*/false, /*exclusive=*/true,
484 do_validate, assume_tracked);
485 } else if constexpr (std::is_same_v<SliceParts, TKey>) {
486 std::string key_buf;
487 Slice contiguous_key(key, &key_buf);
488 s = TryLock(column_family, contiguous_key, /*read_only=*/false,
489 /*exclusive=*/true, do_validate, assume_tracked);
490 }
491 if (!s.ok()) {
492 return s;
493 }
494 column_family =
495 column_family ? column_family : db_impl_->DefaultColumnFamily();
496 assert(column_family);
497 const Comparator* const ucmp = column_family->GetComparator();
498 assert(ucmp);
499 size_t ts_sz = ucmp->timestamp_size();
500 if (ts_sz > 0) {
501 assert(ts_sz == sizeof(TxnTimestamp));
502 if (!IndexingEnabled()) {
503 cfs_with_ts_tracked_when_indexing_disabled_.insert(
504 column_family->GetID());
505 }
506 }
507 return operation();
508}

529Status PessimisticTransaction::CommitBatch(WriteBatch* batch) {
530 if (batch && WriteBatchInternal::HasKeyWithTimestamp(*batch)) {
531 // CommitBatch() needs to lock the keys in the batch.
532 // However, the application also needs to specify the timestamp for the
533 // keys in batch before calling this API.
534 // This means timestamp order may violate the order of locking, thus
535 // violate the sequence number order for the same user key.
536 // Therefore, we disallow this operation for now.
537 return Status::NotSupported(
538 "Batch to commit includes timestamp assigned before locking");
539 }
540
541 std::unique_ptr<LockTracker> keys_to_unlock(lock_tracker_factory_.Create());
542 Status s = LockBatch(batch, keys_to_unlock.get());
543
544 if (!s.ok()) {
545 return s;
546 }
547
548 bool can_commit = false;
549
550 if (IsExpired()) {
551 s = Status::Expired();
552 } else if (expiration_time_ > 0) {
553 TransactionState expected = STARTED;
554 can_commit = std::atomic_compare_exchange_strong(&txn_state_, &expected,
555 AWAITING_COMMIT);
556 } else if (txn_state_ == STARTED) {
557 // lock stealing is not a concern
558 can_commit = true;
559 }
560
561 if (can_commit) {
562 txn_state_.store(AWAITING_COMMIT);
563 s = CommitBatchInternal(batch);
564 if (s.ok()) {
565 txn_state_.store(COMMITTED);
566 }
567 } else if (txn_state_ == LOCKS_STOLEN) {
568 s = Status::Expired();
569 } else {
570 s = Status::InvalidArgument("Transaction is not in state for commit.");
571 }
572
573 txn_db_impl_->UnLock(this, *keys_to_unlock);
574
575 return s;
576}

578Status PessimisticTransaction::Prepare() {
579 if (name_.empty()) {
580 return Status::InvalidArgument(
581 "Cannot prepare a transaction that has not been named.");
582 }
583
584 if (IsExpired()) {
585 return Status::Expired();
586 }
587
588 Status s;
589 bool can_prepare = false;
590
591 if (expiration_time_ > 0) {
592 // must concern ourselves with expiraton and/or lock stealing
593 // need to compare/exchange bc locks could be stolen under us here
594 TransactionState expected = STARTED;
595 can_prepare = std::atomic_compare_exchange_strong(&txn_state_, &expected,
596 AWAITING_PREPARE);
597 } else if (txn_state_ == STARTED) {
598 // expiration and lock stealing is not possible
599 txn_state_.store(AWAITING_PREPARE);
600 can_prepare = true;
601 }
602
603 if (can_prepare) {
604 // transaction can't expire after preparation
605 expiration_time_ = 0;
606 assert(log_number_ == 0 ||
607 txn_db_impl_->GetTxnDBOptions().write_policy == WRITE_UNPREPARED);
608
609 s = PrepareInternal();
610 if (s.ok()) {
611 txn_state_.store(PREPARED);
612 }
613 } else if (txn_state_ == LOCKS_STOLEN) {
614 s = Status::Expired();
615 } else if (txn_state_ == PREPARED) {
616 s = Status::InvalidArgument("Transaction has already been prepared.");
617 } else if (txn_state_ == COMMITTED) {
618 s = Status::InvalidArgument("Transaction has already been committed.");
619 } else if (txn_state_ == ROLLEDBACK) {
620 s = Status::InvalidArgument("Transaction has already been rolledback.");
621 } else {
622 s = Status::InvalidArgument("Transaction is not in state for commit.");
623 }
624
625 return s;
626}

669Status PessimisticTransaction::Commit() {
670 bool commit_without_prepare = false;
671 bool commit_prepared = false;
672
673 if (IsExpired()) {
674 return Status::Expired();
675 }
676
677 if (expiration_time_ > 0) {
678 // we must atomicaly compare and exchange the state here because at
679 // this state in the transaction it is possible for another thread
680 // to change our state out from under us in the even that we expire and have
681 // our locks stolen. In this case the only valid state is STARTED because
682 // a state of PREPARED would have a cleared expiration_time_.
683 TransactionState expected = STARTED;
684 commit_without_prepare = std::atomic_compare_exchange_strong(
685 &txn_state_, &expected, AWAITING_COMMIT);
686 TEST_SYNC_POINT("TransactionTest::ExpirableTransactionDataRace:1");
687 } else if (txn_state_ == PREPARED) {
688 // expiration and lock stealing is not a concern
689 commit_prepared = true;
690 } else if (txn_state_ == STARTED) {
691 // expiration and lock stealing is not a concern
692 if (skip_prepare_) {
693 commit_without_prepare = true;
694 } else {
695 return Status::TxnNotPrepared();

1001Status PessimisticTransaction::LockBatch(WriteBatch* batch,
1002 LockTracker* keys_to_unlock) {

1004 return Status::InvalidArgument("batch is nullptr");
1005 }
1006
1007 class Handler : public WriteBatch::Handler {
1008 public:
1009 // Sorted map of column_family_id to sorted set of keys.
1010 // Since LockBatch() always locks keys in sorted order, it cannot deadlock
1011 // with itself. We're not using a comparator here since it doesn't matter
1012 // what the sorting is as long as it's consistent.
1013 std::map<uint32_t, std::set<std::string>> keys_;
1014
1015 Handler() = default;
1016
1017 void RecordKey(uint32_t column_family_id, const Slice& key) {
1018 auto& cfh_keys = keys_[column_family_id];
1019 cfh_keys.insert(key.ToString());
1020 }
1021
1022 Status PutCF(uint32_t column_family_id, const Slice& key,
1023 const Slice& /* unused */) override {
1024 RecordKey(column_family_id, key);
1025 return Status::OK();
1026 }
1027 Status PutEntityCF(uint32_t column_family_id, const Slice& key,
1028 const Slice& /* unused */) override {

1030 return Status::OK();
1031 }

1038 RecordKey(column_family_id, key);
1039 return Status::OK();

1041 };
1042
1043 // Iterating on this handler will add all keys in this batch into keys
1044 Handler handler;
1045 Status s = batch->Iterate(&handler);
1046 if (!s.ok()) {
1047 return s;
1048 }
1049
1050 // Attempt to lock all keys
1051 for (const auto& cf_iter : handler.keys_) {
1052 uint32_t cfh_id = cf_iter.first;
1053 auto& cfh_keys = cf_iter.second;
1054
1055 for (const auto& key_iter : cfh_keys) {
1056 const std::string& key = key_iter;
1057
1058 s = txn_db_impl_->TryLock(this, cfh_id, key, true /* exclusive */);
1059 if (!s.ok()) {
1060 break;
1061 }
1062 PointLockRequest r;
1063 r.column_family_id = cfh_id;
1064 r.key = key;
1065 r.seq = kMaxSequenceNumber;
1066 r.read_only = false;
1067 r.exclusive = true;
1068 keys_to_unlock->Track(r);
1069 }
1070
1071 if (!s.ok()) {
1072 break;
1073 }
1074 }
1075
1076 if (!s.ok()) {
1077 txn_db_impl_->UnLock(this, *keys_to_unlock);
1078 }

1080 return s;
1081}

1083// Attempt to lock this key.
1084// Returns OK if the key has been successfully locked. Non-ok, otherwise.
1085// If check_shapshot is true and this transaction has a snapshot set,
1086// this key will only be locked if there have been no writes to this key since
1087// the snapshot time.
1088Status PessimisticTransaction::TryLock(ColumnFamilyHandle* column_family,
1089 const Slice& key, bool read_only,
1090 bool exclusive, const bool do_validate,
1091 const bool assume_tracked) {
1092 assert(!assume_tracked || !do_validate);
1093 Status s;
1094 if (UNLIKELY(skip_concurrency_control_)) {
1095 return s;
1096 }
1097 uint32_t cfh_id = GetColumnFamilyID(column_family);
1098 std::string key_str = key.ToString();
1099
1100 PointLockStatus status;
1101 bool lock_upgrade;
1102 bool previously_locked;
1103 if (tracked_locks_->IsPointLockSupported()) {
1104 status = tracked_locks_->GetPointLockStatus(cfh_id, key_str);
1105 previously_locked = status.locked;
1106 lock_upgrade = previously_locked && exclusive && !status.exclusive;
1107 } else {
1108 // If the record is tracked, we can assume it was locked, too.
1109 previously_locked = assume_tracked;
1110 status.locked = false;
1111 lock_upgrade = false;
1112 }
1113
1114 // Lock this key if this transactions hasn't already locked it or we require
1115 // an upgrade.
1116 if (!previously_locked || lock_upgrade) {
1117 s = txn_db_impl_->TryLock(this, cfh_id, key_str, exclusive);
1118 }
1119
1120 const ColumnFamilyHandle* const cfh =
1121 column_family ? column_family : db_impl_->DefaultColumnFamily();
1122 assert(cfh);
1123 const Comparator* const ucmp = cfh->GetComparator();
1124 assert(ucmp);
1125 size_t ts_sz = ucmp->timestamp_size();
1126
1127 SetSnapshotIfNeeded();
1128
1129 // Even though we do not care about doing conflict checking for this write,
1130 // we still need to take a lock to make sure we do not cause a conflict with
1131 // some other write. However, we do not need to check if there have been
1132 // any writes since this transaction's snapshot.
1133 // TODO(agiardullo): could optimize by supporting shared txn locks in the
1134 // future.
1135 SequenceNumber tracked_at_seq =
1136 status.locked ? status.seq : kMaxSequenceNumber;
1137 if (!do_validate || (snapshot_ == nullptr &&
1138 (0 == ts_sz || kMaxTxnTimestamp == read_timestamp_))) {
1139 if (assume_tracked && !previously_locked &&
1140 tracked_locks_->IsPointLockSupported()) {
1141 s = Status::InvalidArgument(
1142 "assume_tracked is set but it is not tracked yet");
1143 }
1144 // Need to remember the earliest sequence number that we know that this
1145 // key has not been modified after. This is useful if this same
1146 // transaction later tries to lock this key again.
1147 if (tracked_at_seq == kMaxSequenceNumber) {
1148 // Since we haven't checked a snapshot, we only know this key has not
1149 // been modified since after we locked it.
1150 // Note: when last_seq_same_as_publish_seq_==false this is less than the
1151 // latest allocated seq but it is ok since i) this is just a heuristic
1152 // used only as a hint to avoid actual check for conflicts, ii) this would
1153 // cause a false positive only if the snapthot is taken right after the
1154 // lock, which would be an unusual sequence.
1155 tracked_at_seq = db_->GetLatestSequenceNumber();
1156 }
1157 } else if (s.ok()) {
1158 // If a snapshot is set, we need to make sure the key hasn't been modified
1159 // since the snapshot. This must be done after we locked the key.
1160 // If we already have validated an earilier snapshot it must has been
1161 // reflected in tracked_at_seq and ValidateSnapshot will return OK.
1162 s = ValidateSnapshot(column_family, key, &tracked_at_seq);
1163
1164 if (!s.ok()) {
1165 // Failed to validate key
1166 // Unlock key we just locked
1167 if (lock_upgrade) {
1168 s = txn_db_impl_->TryLock(this, cfh_id, key_str, false /* exclusive */);
1169 assert(s.ok());
1170 } else if (!previously_locked) {
1171 txn_db_impl_->UnLock(this, cfh_id, key.ToString());
1172 }
1173 }
1174 }
1175
1176 if (s.ok()) {
1177 // We must track all the locked keys so that we can unlock them later. If
1178 // the key is already locked, this func will update some stats on the
1179 // tracked key. It could also update the tracked_at_seq if it is lower
1180 // than the existing tracked key seq. These stats are necessary for
1181 // RollbackToSavePoint to determine whether a key can be safely removed
1182 // from tracked_keys_. Removal can only be done if a key was only locked
1183 // during the current savepoint.
1184 //
1185 // Recall that if assume_tracked is true, we assume that TrackKey has been
1186 // called previously since the last savepoint, with the same exclusive
1187 // setting, and at a lower sequence number, so skipping here should be
1188 // safe.
1189 if (!assume_tracked) {
1190 TrackKey(cfh_id, key_str, tracked_at_seq, read_only, exclusive);
1191 } else {
1192#ifndef NDEBUG
1193 if (tracked_locks_->IsPointLockSupported()) {
1194 PointLockStatus lock_status =
1195 tracked_locks_->GetPointLockStatus(cfh_id, key_str);
1196 assert(lock_status.locked);
1197 assert(lock_status.seq <= tracked_at_seq);
1198 assert(lock_status.exclusive == exclusive);
1199 }
1200#endif
1201 }
1202 }
1203
1204 return s;
1205}
