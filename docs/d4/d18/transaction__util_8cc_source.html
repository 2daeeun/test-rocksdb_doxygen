21Status TransactionUtil::CheckKeyForConflicts(
22 DBImpl* db_impl, ColumnFamilyHandle* column_family, const std::string& key,
23 SequenceNumber snap_seq, const std::string* const read_ts, bool cache_only,
24 ReadCallback* snap_checker, SequenceNumber min_uncommitted,
25 bool enable_udt_validation) {
26 Status result;
27
28 auto cfh = static_cast_with_check<ColumnFamilyHandleImpl>(column_family);
29 auto cfd = cfh->cfd();
30 SuperVersion* sv = db_impl->GetAndRefSuperVersion(cfd);
31
32 if (sv == nullptr) {
33 result = Status::InvalidArgument("Could not access column family " +
34 cfh->GetName());
35 }
36
37 if (result.ok()) {
38 SequenceNumber earliest_seq =
39 db_impl->GetEarliestMemTableSequenceNumber(sv, true);
40
41 result =
42 CheckKey(db_impl, sv, earliest_seq, snap_seq, key, read_ts, cache_only,
43 snap_checker, min_uncommitted, enable_udt_validation);
44
45 db_impl->ReturnAndCleanupSuperVersion(cfd, sv);
46 }
47
48 return result;
49}

51Status TransactionUtil::CheckKey(DBImpl* db_impl, SuperVersion* sv,
52 SequenceNumber earliest_seq,
53 SequenceNumber snap_seq,
54 const std::string& key,
55 const std::string* const read_ts,
56 bool cache_only, ReadCallback* snap_checker,
57 SequenceNumber min_uncommitted,
58 bool enable_udt_validation) {
59 // When `min_uncommitted` is provided, keys are not always committed
60 // in sequence number order, and `snap_checker` is used to check whether
61 // specific sequence number is in the database is visible to the transaction.
62 // So `snap_checker` must be provided.
63 assert(min_uncommitted == kMaxSequenceNumber || snap_checker != nullptr);
64
65 Status result;
66 bool need_to_read_sst = false;
67
68 // Since it would be too slow to check the SST files, we will only use
69 // the memtables to check whether there have been any recent writes
70 // to this key after it was accessed in this transaction. But if the
71 // Memtables do not contain a long enough history, we must fail the
72 // transaction.
73 if (earliest_seq == kMaxSequenceNumber) {
74 // The age of this memtable is unknown. Cannot rely on it to check
75 // for recent writes. This error shouldn't happen often in practice as
76 // the Memtable should have a valid earliest sequence number except in some
77 // corner cases (such as error cases during recovery).
78 need_to_read_sst = true;
79
80 if (cache_only) {
81 result = Status::TryAgain(
82 "Transaction could not check for conflicts as the MemTable does not "
83 "contain a long enough history to check write at SequenceNumber: ",
84 std::to_string(snap_seq));
85 }
86 } else if (snap_seq < earliest_seq || min_uncommitted <= earliest_seq) {
87 // Use <= for min_uncommitted since earliest_seq is actually the largest sec
88 // before this memtable was created
89 need_to_read_sst = true;
90
91 if (cache_only) {
92 // The age of this memtable is too new to use to check for recent
93 // writes.
94 char msg[300];
95 snprintf(msg, sizeof(msg),
96 "Transaction could not check for conflicts for operation at "
97 "SequenceNumber %" PRIu64
98 " as the MemTable only contains changes newer than "
99 "SequenceNumber %" PRIu64
100 ". Increasing the value of the "
101 "max_write_buffer_size_to_maintain option could reduce the "
102 "frequency "
103 "of this error.",
104 snap_seq, earliest_seq);
105 result = Status::TryAgain(msg);
106 }
107 }
108
109 if (result.ok()) {
110 SequenceNumber seq = kMaxSequenceNumber;
111 std::string timestamp;
112 bool found_record_for_key = false;
113
114 // When min_uncommitted == kMaxSequenceNumber, writes are committed in
115 // sequence number order, so only keys larger than `snap_seq` can cause
116 // conflict.
117 // When min_uncommitted != kMaxSequenceNumber, keys lower than
118 // min_uncommitted will not triggered conflicts, while keys larger than
119 // min_uncommitted might create conflicts, so we need to read them out
120 // from the DB, and call callback to snap_checker to determine. So only
121 // keys lower than min_uncommitted can be skipped.
122 SequenceNumber lower_bound_seq =
123 (min_uncommitted == kMaxSequenceNumber) ? snap_seq : min_uncommitted;
124 Status s = db_impl->GetLatestSequenceForKey(
125 sv, key, !need_to_read_sst, lower_bound_seq, &seq,
126 !read_ts ? nullptr : &timestamp, &found_record_for_key,
127 /*is_blob_index=*/nullptr);
128
129 if (!(s.ok() || s.IsNotFound() || s.IsMergeInProgress())) {
130 result = s;
131 } else if (found_record_for_key) {
132 bool write_conflict = snap_checker == nullptr
133 ? snap_seq < seq
134 : !snap_checker->IsVisible(seq);
135 // Perform conflict checking based on timestamp if applicable.
136 if (enable_udt_validation && !write_conflict && read_ts != nullptr) {
137 ColumnFamilyData* cfd = sv->cfd;
138 assert(cfd);
139 const Comparator* const ucmp = cfd->user_comparator();
140 assert(ucmp);
141 assert(read_ts->size() == ucmp->timestamp_size());
142 assert(read_ts->size() == timestamp.size());
143 // Write conflict if *ts < timestamp.
144 write_conflict = ucmp->CompareTimestamp(*read_ts, timestamp) < 0;
145 }
146 if (write_conflict) {
147 result = Status::Busy();
148 }
149 }
150 }
151
152 return result;
153}

155Status TransactionUtil::CheckKeysForConflicts(DBImpl* db_impl,
156 const LockTracker& tracker,
157 bool cache_only) {
158 Status result;
159
160 std::unique_ptr<LockTracker::ColumnFamilyIterator> cf_it(
161 tracker.GetColumnFamilyIterator());
162 assert(cf_it != nullptr);
163 while (cf_it->HasNext()) {
164 ColumnFamilyId cf = cf_it->Next();
165
166 SuperVersion* sv = db_impl->GetAndRefSuperVersion(cf);
167 if (sv == nullptr) {
168 result = Status::InvalidArgument("Could not access column family " +
169 std::to_string(cf));
170 break;
171 }
172
173 SequenceNumber earliest_seq =
174 db_impl->GetEarliestMemTableSequenceNumber(sv, true);
175
176 // For each of the keys in this transaction, check to see if someone has
177 // written to this key since the start of the transaction.
178 std::unique_ptr<LockTracker::KeyIterator> key_it(
179 tracker.GetKeyIterator(cf));
180 assert(key_it != nullptr);
181 while (key_it->HasNext()) {
182 const std::string& key = key_it->Next();
183 PointLockStatus status = tracker.GetPointLockStatus(cf, key);
184 const SequenceNumber key_seq = status.seq;
185
186 // TODO: support timestamp-based conflict checking.
187 // CheckKeysForConflicts() is currently used only by optimistic
188 // transactions.
189 result = CheckKey(db_impl, sv, earliest_seq, key_seq, key,
190 /*read_ts=*/nullptr, cache_only);
191 if (!result.ok()) {
192 break;
193 }
194 }
195
196 db_impl->ReturnAndCleanupSuperVersion(cf, sv);
197
198 if (!result.ok()) {
199 break;
200 }
201 }
202
203 return result;
204}
