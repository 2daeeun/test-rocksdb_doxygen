41 // Request for token to write bytes. If this request can not be satisfied,
42 // the call is blocked. Caller is responsible to make sure
43 // bytes <= GetSingleBurstBytes() and bytes >= 0. Negative bytes
44 // passed in will be rounded up to 0.
45 using RateLimiter::Request;
46 void Request(const int64_t bytes, const Env::IOPriority pri,
47 Statistics* stats) override;
48
49 int64_t GetSingleBurstBytes() const override {
50 int64_t raw_single_burst_bytes =
51 raw_single_burst_bytes_.load(std::memory_order_relaxed);
52 if (raw_single_burst_bytes == 0) {
53 return refill_bytes_per_period_.load(std::memory_order_relaxed);
54 }
55 return raw_single_burst_bytes;
56 }
57
58 int64_t GetTotalBytesThrough(
59 const Env::IOPriority pri = Env::IO_TOTAL) const override {
60 MutexLock g(&request_mutex_);
61 if (pri == Env::IO_TOTAL) {
62 int64_t total_bytes_through_sum = 0;
63 for (int i = Env::IO_LOW; i < Env::IO_TOTAL; ++i) {
64 total_bytes_through_sum += total_bytes_through_[i];
65 }
66 return total_bytes_through_sum;
67 }
68 return total_bytes_through_[pri];
69 }
70
71 int64_t GetTotalRequests(
72 const Env::IOPriority pri = Env::IO_TOTAL) const override {
73 MutexLock g(&request_mutex_);
74 if (pri == Env::IO_TOTAL) {
75 int64_t total_requests_sum = 0;
76 for (int i = Env::IO_LOW; i < Env::IO_TOTAL; ++i) {
77 total_requests_sum += total_requests_[i];
78 }
79 return total_requests_sum;
80 }
81 return total_requests_[pri];
82 }
83
84 Status GetTotalPendingRequests(
85 int64_t* total_pending_requests,
86 const Env::IOPriority pri = Env::IO_TOTAL) const override {
87 assert(total_pending_requests != nullptr);
88 MutexLock g(&request_mutex_);
89 if (pri == Env::IO_TOTAL) {
90 int64_t total_pending_requests_sum = 0;
91 for (int i = Env::IO_LOW; i < Env::IO_TOTAL; ++i) {
92 total_pending_requests_sum += static_cast<int64_t>(queue_[i].size());
93 }
94 *total_pending_requests = total_pending_requests_sum;
95 } else {
96 *total_pending_requests = static_cast<int64_t>(queue_[pri].size());
97 }
98 return Status::OK();
99 }
100
101 int64_t GetBytesPerSecond() const override {
102 return rate_bytes_per_sec_.load(std::memory_order_relaxed);
103 }
104
105 virtual void TEST_SetClock(std::shared_ptr<SystemClock> clock) {
106 MutexLock g(&request_mutex_);
107 clock_ = std::move(clock);
108 next_refill_us_ = NowMicrosMonotonicLocked();
109 }
110
111 private:
112 static constexpr int kMicrosecondsPerSecond = 1000000;
113 void RefillBytesAndGrantRequestsLocked();
114 std::vector<Env::IOPriority> GeneratePriorityIterationOrderLocked();
115 int64_t CalculateRefillBytesPerPeriodLocked(int64_t rate_bytes_per_sec);
116 Status TuneLocked();
117 void SetBytesPerSecondLocked(int64_t bytes_per_second);
118
119 uint64_t NowMicrosMonotonicLocked() {
120 return clock_->NowNanos() / std::milli::den;
121 }
