16struct TrackedKeyInfo {
17 // Earliest sequence number that is relevant to this transaction for this key
18 SequenceNumber seq;
19
20 uint32_t num_writes;
21 uint32_t num_reads;
22
23 bool exclusive;
24
25 explicit TrackedKeyInfo(SequenceNumber seq_no)
26 : seq(seq_no), num_writes(0), num_reads(0), exclusive(false) {}
27
28 void Merge(const TrackedKeyInfo& info) {
29 assert(seq <= info.seq);
30 num_reads += info.num_reads;
31 num_writes += info.num_writes;
32 exclusive = exclusive || info.exclusive;
33 }
34};
