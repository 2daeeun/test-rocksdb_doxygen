41 void ResetArgs() {
42 argc_ = 0;

47 void AppendArgs(const std::vector<std::string>& args) {
48 for (const auto& arg : args) {
49 ASSERT_LE(cursor_ + arg.size() + 1, kArgBufferSize);
50 ASSERT_LE(argc_ + 1, kMaxArgCount);
51 snprintf(arg_buffer_ + cursor_, arg.size() + 1, "%s", arg.c_str());
52
53 argv_[argc_++] = arg_buffer_ + cursor_;
54 cursor_ += arg.size() + 1;
55 }
56 }

58 // Gets the default options for this test/db_bench.
59 // Note that db_bench may change some of the default option values and that
60 // the database might as well. The options changed by db_bench are
61 // specified here; the ones by the DB are set via SanitizeOptions
62 Options GetDefaultOptions(CompactionStyle style = kCompactionStyleLevel,
63 int levels = 7) const {
64 Options opt;
65
66 opt.create_if_missing = true;
67 opt.max_open_files = 256;
68 opt.max_background_compactions = 10;
69 opt.dump_malloc_stats = true; // db_bench uses a different default
70 opt.compaction_style = style;
71 opt.num_levels = levels;
72 opt.compression = kNoCompression;
73 opt.arena_block_size = 8388608;
74
75 return SanitizeOptions(db_path_, opt);
76 }
77
78 void RunDbBench(const std::string& options_file_name) {
79 AppendArgs({"./db_bench", "--benchmarks=fillseq", "--use_existing_db=0",
80 "--num=1000", "--compression_type=none",
81 std::string(std::string("--db=") + db_path_).c_str(),
82 std::string(std::string("--wal_dir=") + wal_path_).c_str(),
83 std::string(std::string("--options_file=") + options_file_name)
84 .c_str()});
85 ASSERT_EQ(0, db_bench_tool(argc(), argv()));
86 }
87
88 void VerifyOptions(const Options& opt) {
89 DBOptions loaded_db_opts;
90 ConfigOptions config_opts;
91 config_opts.ignore_unknown_options = false;
92 config_opts.input_strings_escaped = true;
93 config_opts.env = Env::Default();
94 std::vector<ColumnFamilyDescriptor> cf_descs;
95 ASSERT_OK(
96 LoadLatestOptions(config_opts, db_path_, &loaded_db_opts, &cf_descs));
97
98 ConfigOptions exact;
99 exact.input_strings_escaped = false;
100 exact.sanity_level = ConfigOptions::kSanityLevelExactMatch;
101 ASSERT_OK(RocksDBOptionsParser::VerifyDBOptions(exact, DBOptions(opt),
102 loaded_db_opts));
103 ASSERT_OK(RocksDBOptionsParser::VerifyCFOptions(
104 exact, ColumnFamilyOptions(opt), cf_descs[0].options));
105
106 // check with the default rocksdb options and expect failure
107 ASSERT_NOK(RocksDBOptionsParser::VerifyDBOptions(exact, DBOptions(),
108 loaded_db_opts));
109 ASSERT_NOK(RocksDBOptionsParser::VerifyCFOptions(
110 exact, ColumnFamilyOptions(), cf_descs[0].options));
111 }
