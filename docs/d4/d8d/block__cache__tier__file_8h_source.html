24// The io code path of persistent cache uses pipelined architecture
25//
26// client -> In Queue <-- BlockCacheTier --> Out Queue <-- Writer <--> Kernel
27//
28// This would enable the system to scale for GB/s of throughput which is
29// expected with modern devies like NVM.
30//
31// The file level operations are encapsulated in the following abstractions
32//
33// BlockCacheFile
34// ^
35// |
36// |
37// RandomAccessCacheFile (For reading)
38// ^
39// |
40// |
41// WriteableCacheFile (For writing)
42//
43// Write IO code path :
44//
45namespace ROCKSDB_NAMESPACE {

50// Represents a logical record on device

52// (L)ogical (B)lock (Address = { cache-file-id, offset, size }
53struct LogicalBlockAddress {

169// class WriteableCacheFile
170//
171// All writes to the files are cached in buffers. The buffers are flushed to
172// disk as they get filled up. When file size reaches a certain size, a new file
173// will be created provided there is free space
174class WriteableCacheFile : public RandomAccessCacheFile {
175 public:
176 explicit WriteableCacheFile(Env* const env, CacheWriteBufferAllocator* alloc,
177 Writer* writer, const std::string& dir,
178 const uint32_t cache_id, const uint32_t max_size,
179 const std::shared_ptr<Logger>& log)
180 : RandomAccessCacheFile(env, dir, cache_id, log),
181 alloc_(alloc),
182 writer_(writer),
183 max_size_(max_size) {}
184
185 virtual ~WriteableCacheFile();
186
187 // create file on disk
188 bool Create(const bool enable_direct_writes, const bool enable_direct_reads);
189
190 // read data from logical file
191 bool Read(const LBA& lba, Slice* key, Slice* block, char* scratch) override {
192 ReadLock _(&rwlock_);
193 const bool closed = eof_ && bufs_.empty();
194 if (closed) {
195 // the file is closed, read from disk
196 return RandomAccessCacheFile::Read(lba, key, block, scratch);
197 }
198 // file is still being written, read from buffers
199 return ReadBuffer(lba, key, block, scratch);
200 }
201
202 // append data to end of file
203 bool Append(const Slice&, const Slice&, LBA* const) override;
204 // End-of-file
205 bool Eof() const { return eof_; }
206
207 private:
208 friend class ThreadedWriter;
209
210 static const size_t kFileAlignmentSize = 4 * 1024; // align file size
211
212 bool ReadBuffer(const LBA& lba, Slice* key, Slice* block, char* scratch);
213 bool ReadBuffer(const LBA& lba, char* data);
214 bool ExpandBuffer(const size_t size);
215 void DispatchBuffer();
216 void BufferWriteDone();
217 void CloseAndOpenForReading();
218 void ClearBuffers();
219 void Close();
220
221 // File layout in memory
222 //
223 // +------+------+------+------+------+------+
224 // | b0 | b1 | b2 | b3 | b4 | b5 |
225 // +------+------+------+------+------+------+
226 // ^ ^
227 // | |
228 // buf_doff_ buf_woff_
229 // (next buffer to (next buffer to fill)
230 // flush to disk)
231 //
232 // The buffers are flushed to disk serially for a given file
233
234 CacheWriteBufferAllocator* const alloc_ = nullptr; // Buffer provider
235 Writer* const writer_ = nullptr; // File writer thread
236 std::unique_ptr<WritableFile> file_; // RocksDB Env file abstraction
237 std::vector<CacheWriteBuffer*> bufs_; // Written buffers
238 uint32_t size_ = 0; // Size of the file
239 const uint32_t max_size_; // Max size of the file
240 bool eof_ = false; // End of file
241 uint32_t disk_woff_ = 0; // Offset to write on disk
242 size_t buf_woff_ = 0; // off into bufs_ to write
243 size_t buf_doff_ = 0; // off into bufs_ to dispatch
244 size_t pending_ios_ = 0; // Number of ios to disk in-progress
245 bool enable_direct_reads_ = false; // Should we enable direct reads
246 // when reading from disk
247};

252class ThreadedWriter : public Writer {
253 public:
254 // Representation of IO to device
255 struct IO {
256 explicit IO(const bool signal) : signal_(signal) {}
257 explicit IO(WritableFile* const file, CacheWriteBuffer* const buf,
258 const uint64_t file_off, const std::function<void()> callback)
259 : file_(file), buf_(buf), file_off_(file_off), callback_(callback) {}
260
261 IO(const IO&) = default;
262 IO& operator=(const IO&) = default;
263 size_t Size() const { return sizeof(IO); }
264
265 WritableFile* file_ = nullptr; // File to write to
266 CacheWriteBuffer* buf_ = nullptr; // buffer to write
267 uint64_t file_off_ = 0; // file offset
268 bool signal_ = false; // signal to exit thread loop
269 std::function<void()> callback_; // Callback on completion
270 };
271
272 explicit ThreadedWriter(PersistentCacheTier* const cache, const size_t qdepth,
273 const size_t io_size);
274 virtual ~ThreadedWriter() { assert(threads_.empty()); }
275
276 void Stop() override;
277 void Write(WritableFile* const file, CacheWriteBuffer* buf,
278 const uint64_t file_off,
279 const std::function<void()> callback) override;
280
281 private:
282 void ThreadMain();
283 void DispatchIO(const IO& io);
284
285 const size_t io_size_ = 0;
286 BoundedQueue<IO> q_;
287 std::vector<port::Thread> threads_;
288};
