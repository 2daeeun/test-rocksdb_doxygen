21// Class representing a wide column, which is defined as a pair of column name
22// and column value.
23class WideColumn {
24 public:
25 WideColumn() = default;
26
27 // Initializes a WideColumn object by forwarding the name and value
28 // arguments to the corresponding member Slices. This makes it possible to
29 // construct a WideColumn using combinations of const char*, const
30 // std::string&, const Slice& etc., for example:
31 //
32 // constexpr char foo[] = "foo";
33 // const std::string bar("bar");
34 // WideColumn column(foo, bar);
35 template <typename N, typename V>
36 WideColumn(N&& name, V&& value)
37 : name_(std::forward<N>(name)), value_(std::forward<V>(value)) {}
38
39 // Initializes a WideColumn object by forwarding the elements of
40 // name_tuple and value_tuple to the constructors of the corresponding member
41 // Slices. This makes it possible to initialize the Slices using the Slice
42 // constructors that take more than one argument, for example:
43 //
44 // constexpr char foo_name[] = "foo_name";
45 // constexpr char bar_value[] = "bar_value";
46 // WideColumn column(std::piecewise_construct,
47 // std::forward_as_tuple(foo_name, 3),
48 // std::forward_as_tuple(bar_value, 3));
49 template <typename NTuple, typename VTuple>
50 WideColumn(std::piecewise_construct_t, NTuple&& name_tuple,
51 VTuple&& value_tuple)
52 : name_(std::make_from_tuple<Slice>(std::forward<NTuple>(name_tuple))),
53 value_(std::make_from_tuple<Slice>(std::forward<VTuple>(value_tuple))) {
54 }
55
56 const Slice& name() const { return name_; }
57 const Slice& value() const { return value_; }
58
59 Slice& name() { return name_; }
60 Slice& value() { return value_; }
61
62 private:
63 Slice name_;
64 Slice value_;
65};
66
67// Note: column names and values are compared bytewise.
68inline bool operator==(const WideColumn& lhs, const WideColumn& rhs) {
69 return lhs.name() == rhs.name() && lhs.value() == rhs.value();
70}
71
72inline bool operator!=(const WideColumn& lhs, const WideColumn& rhs) {
73 return !(lhs == rhs);
74}
75
76inline std::ostream& operator<<(std::ostream& os, const WideColumn& column) {
77 const bool hex =
78 (os.flags() & std::ios_base::basefield) == std::ios_base::hex;
79 if (!column.name().empty()) {
80 if (hex) {
81 os << "0x";
82 }
83 os << column.name().ToString(hex);
84 }
85 os << ':';
86 if (!column.value().empty()) {
87 if (hex) {
88 os << "0x";
89 }
90 os << column.value().ToString(hex);
91 }
92 return os;
93}

147inline void PinnableWideColumns::Reset() {
148 value_.Reset();
149 columns_.clear();

152inline void PinnableWideColumns::Move(PinnableWideColumns&& other) {
153 assert(columns_.empty());
154
155 if (other.columns_.empty()) {
156 return;
157 }
158
159 const char* const data = other.value_.data();
160 const bool is_plain_value =
161 other.columns_.size() == 1 &&
162 other.columns_.front().name() == kDefaultWideColumnName &&
163 other.columns_.front().value() == other.value_;
164
165 MoveValue(std::move(other.value_));
166
167 if (value_.data() == data) {
168 columns_ = std::move(other.columns_);
169 } else {
170 if (is_plain_value) {
171 CreateIndexForPlainValue();
172 } else {
173 const Status s = CreateIndexForWideColumns();
174 assert(s.ok());
175
176 s.PermitUncheckedError();
177 }
178 }
179
180 other.Reset();
181}
