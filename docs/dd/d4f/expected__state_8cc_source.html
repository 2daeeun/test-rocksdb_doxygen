28void ExpectedState::Precommit(int cf, int64_t key, const ExpectedValue& value) {
29 Value(cf, key).store(value.Read());
30 // To prevent low-level instruction reordering that results
31 // in db write happens before setting pending state in expected value
32 std::atomic_thread_fence(std::memory_order_release);
33}

35PendingExpectedValue ExpectedState::PreparePut(int cf, int64_t key) {
36 ExpectedValue expected_value = Load(cf, key);
37
38 // Calculate the original expected value
39 const ExpectedValue orig_expected_value = expected_value;
40
41 // Calculate the pending expected value
42 expected_value.Put(true /* pending */);
43 const ExpectedValue pending_expected_value = expected_value;
44
45 // Calculate the final expected value
46 expected_value.Put(false /* pending */);
47 const ExpectedValue final_expected_value = expected_value;
48
49 // Precommit
50 Precommit(cf, key, pending_expected_value);
51 return PendingExpectedValue(&Value(cf, key), orig_expected_value,
52 final_expected_value);
53}

57PendingExpectedValue ExpectedState::PrepareDelete(int cf, int64_t key) {
58 ExpectedValue expected_value = Load(cf, key);
59
60 // Calculate the original expected value
61 const ExpectedValue orig_expected_value = expected_value;
62
63 // Calculate the pending expected value
64 bool res = expected_value.Delete(true /* pending */);
65 if (!res) {
66 PendingExpectedValue ret = PendingExpectedValue(
67 &Value(cf, key), orig_expected_value, orig_expected_value);
68 return ret;
69 }
70 const ExpectedValue pending_expected_value = expected_value;
71
72 // Calculate the final expected value
73 expected_value.Delete(false /* pending */);
74 const ExpectedValue final_expected_value = expected_value;
75
76 // Precommit
77 Precommit(cf, key, pending_expected_value);
78 return PendingExpectedValue(&Value(cf, key), orig_expected_value,
79 final_expected_value);
80}

101void ExpectedState::Reset() {
102 const uint32_t del_mask = ExpectedValue::GetDelMask();
103 for (size_t i = 0; i < num_column_families_; ++i) {
104 for (size_t j = 0; j < max_key_; ++j) {
105 Value(static_cast<int>(i), j).store(del_mask, std::memory_order_relaxed);
106 }
107 }

140Status FileExpectedState::Open(bool create) {
141 size_t expected_values_size = GetValuesLen();
142
143 Env* default_env = Env::Default();
144
145 Status status;
146 if (create) {
147 std::unique_ptr<WritableFile> wfile;
148 const EnvOptions soptions;
149 status = default_env->NewWritableFile(expected_state_file_path_, &wfile,
150 soptions);
151 if (status.ok()) {
152 std::string buf(expected_values_size, '\0');
153 status = wfile->Append(buf);
154 }
155 }
156 if (status.ok()) {
157 status = default_env->NewMemoryMappedFileBuffer(
158 expected_state_file_path_, &expected_state_mmap_buffer_);
159 }
160 if (status.ok()) {
161 assert(expected_state_mmap_buffer_->GetLen() == expected_values_size);
162 values_ = static_cast<std::atomic<uint32_t>*>(
163 expected_state_mmap_buffer_->GetBase());
164 assert(values_ != nullptr);
165 if (create) {
166 Reset();
167 }
168 } else {
169 assert(values_ == nullptr);
170 }
171 return status;
172}

214Status FileExpectedStateManager::Open() {
215 // Before doing anything, sync directory state with ours. That is, determine
216 // `saved_seqno_`, and create any necessary missing files.
217 std::vector<std::string> expected_state_dir_children;
218 Status s = Env::Default()->GetChildren(expected_state_dir_path_,
219 &expected_state_dir_children);
220 bool found_trace = false;
221 if (s.ok()) {
222 for (size_t i = 0; i < expected_state_dir_children.size(); ++i) {
223 const auto& filename = expected_state_dir_children[i];
224 if (filename.size() >= kStateFilenameSuffix.size() &&
225 filename.rfind(kStateFilenameSuffix) ==
226 filename.size() - kStateFilenameSuffix.size() &&
227 filename.rfind(kLatestBasename, 0) == std::string::npos) {
228 SequenceNumber found_seqno = ParseUint64(
229 filename.substr(0, filename.size() - kStateFilenameSuffix.size()));
230 if (saved_seqno_ == kMaxSequenceNumber || found_seqno > saved_seqno_) {
231 saved_seqno_ = found_seqno;
232 }
233 }
234 }
235 // Check if crash happened after creating state file but before creating
236 // trace file.
237 if (saved_seqno_ != kMaxSequenceNumber) {
238 std::string saved_seqno_trace_path = GetPathForFilename(
239 std::to_string(saved_seqno_) + kTraceFilenameSuffix);
240 Status exists_status = Env::Default()->FileExists(saved_seqno_trace_path);
241 if (exists_status.ok()) {
242 found_trace = true;
243 } else if (exists_status.IsNotFound()) {
244 found_trace = false;
245 } else {
246 s = exists_status;
247 }
248 }
249 }
250 if (s.ok() && saved_seqno_ != kMaxSequenceNumber && !found_trace) {
251 // Create an empty trace file so later logic does not need to distinguish
252 // missing vs. empty trace file.
253 std::unique_ptr<WritableFile> wfile;
254 const EnvOptions soptions;
255 std::string saved_seqno_trace_path =
256 GetPathForFilename(std::to_string(saved_seqno_) + kTraceFilenameSuffix);
257 s = Env::Default()->NewWritableFile(saved_seqno_trace_path, &wfile,
258 soptions);
259 }
260
261 if (s.ok()) {
262 s = Clean();
263 }
264
265 std::string expected_state_file_path =
266 GetPathForFilename(kLatestBasename + kStateFilenameSuffix);
267 bool found = false;
268 if (s.ok()) {
269 Status exists_status = Env::Default()->FileExists(expected_state_file_path);
270 if (exists_status.ok()) {
271 found = true;
272 } else if (exists_status.IsNotFound()) {
273 found = false;
274 } else {
275 s = exists_status;
276 }
277 }
278
279 if (!found) {
280 // Initialize the file in a temp path and then rename it. That way, in case
281 // this process is killed during setup, `Clean()` will take care of removing
282 // the incomplete expected values file.
283 std::string temp_expected_state_file_path =
284 GetTempPathForFilename(kLatestBasename + kStateFilenameSuffix);
285 FileExpectedState temp_expected_state(temp_expected_state_file_path,
286 max_key_, num_column_families_);
287 if (s.ok()) {
288 s = temp_expected_state.Open(true /* create */);
289 }
290 if (s.ok()) {
291 s = Env::Default()->RenameFile(temp_expected_state_file_path,
292 expected_state_file_path);
293 }
294 }
295
296 if (s.ok()) {
297 latest_.reset(new FileExpectedState(std::move(expected_state_file_path),
298 max_key_, num_column_families_));
299 s = latest_->Open(false /* create */);
300 }
301 return s;
302}

304Status FileExpectedStateManager::SaveAtAndAfter(DB* db) {
305 SequenceNumber seqno = db->GetLatestSequenceNumber();
306
307 std::string state_filename = std::to_string(seqno) + kStateFilenameSuffix;
308 std::string state_file_temp_path = GetTempPathForFilename(state_filename);
309 std::string state_file_path = GetPathForFilename(state_filename);
310
311 std::string latest_file_path =
312 GetPathForFilename(kLatestBasename + kStateFilenameSuffix);
313
314 std::string trace_filename = std::to_string(seqno) + kTraceFilenameSuffix;
315 std::string trace_file_path = GetPathForFilename(trace_filename);
316
317 // Populate a tempfile and then rename it to atomically create "<seqno>.state"
318 // with contents from "LATEST.state"
319 Status s =
320 CopyFile(FileSystem::Default(), latest_file_path, Temperature::kUnknown,
321 state_file_temp_path, Temperature::kUnknown, 0 /* size */,
322 false /* use_fsync */, nullptr /* io_tracer */);
323 if (s.ok()) {
324 s = FileSystem::Default()->RenameFile(state_file_temp_path, state_file_path,
325 IOOptions(), nullptr /* dbg */);
326 }
327 SequenceNumber old_saved_seqno = 0;
328 if (s.ok()) {
329 old_saved_seqno = saved_seqno_;
330 saved_seqno_ = seqno;
331 }
332
333 // If there is a crash now, i.e., after "<seqno>.state" was created but before
334 // "<seqno>.trace" is created, it will be treated as if "<seqno>.trace" were
335 // present but empty.
336
337 // Create "<seqno>.trace" directly. It is initially empty so no need for
338 // tempfile.
339 std::unique_ptr<TraceWriter> trace_writer;
340 if (s.ok()) {
341 EnvOptions soptions;
342 // Disable buffering so traces will not get stuck in application buffer.
343 soptions.writable_file_max_buffer_size = 0;
344 s = NewFileTraceWriter(Env::Default(), soptions, trace_file_path,
345 &trace_writer);
346 }
347 if (s.ok()) {
348 TraceOptions trace_opts;
349 trace_opts.filter |= kTraceFilterGet;
350 trace_opts.filter |= kTraceFilterMultiGet;
351 trace_opts.filter |= kTraceFilterIteratorSeek;
352 trace_opts.filter |= kTraceFilterIteratorSeekForPrev;
353 trace_opts.preserve_write_order = true;
354 s = db->StartTrace(trace_opts, std::move(trace_writer));
355 }
356
357 // Delete old state/trace files. Deletion order does not matter since we only
358 // delete after successfully saving new files, so old files will never be used
359 // again, even if we crash.
360 if (s.ok() && old_saved_seqno != kMaxSequenceNumber &&
361 old_saved_seqno != saved_seqno_) {
362 s = Env::Default()->DeleteFile(GetPathForFilename(
363 std::to_string(old_saved_seqno) + kStateFilenameSuffix));
364 }
365 if (s.ok() && old_saved_seqno != kMaxSequenceNumber &&
366 old_saved_seqno != saved_seqno_) {
367 s = Env::Default()->DeleteFile(GetPathForFilename(
368 std::to_string(old_saved_seqno) + kTraceFilenameSuffix));
369 }
370 return s;
371}

379// An `ExpectedStateTraceRecordHandler` applies a configurable number of
380// write operation trace records to the configured expected state. It is used in
381// `FileExpectedStateManager::Restore()` to sync the expected state with the
382// DB's post-recovery state.
383class ExpectedStateTraceRecordHandler : public TraceRecord::Handler,
384 public WriteBatch::Handler {
385 public:
386 ExpectedStateTraceRecordHandler(uint64_t max_write_ops, ExpectedState* state)
387 : max_write_ops_(max_write_ops),
388 state_(state),
389 buffered_writes_(nullptr) {}
390
391 ~ExpectedStateTraceRecordHandler() { assert(IsDone()); }
392
393 // True if we have already reached the limit on write operations to apply.
394 bool IsDone() { return num_write_ops_ == max_write_ops_; }
395
396 Status Handle(const WriteQueryTraceRecord& record,
397 std::unique_ptr<TraceRecordResult>* /* result */) override {
398 if (IsDone()) {
399 return Status::OK();
400 }
401 WriteBatch batch(record.GetWriteBatchRep().ToString());
402 return batch.Iterate(this);
403 }
404
405 // Ignore reads.
406 Status Handle(const GetQueryTraceRecord& /* record */,
407 std::unique_ptr<TraceRecordResult>* /* result */) override {
408 return Status::OK();
409 }
410
411 // Ignore reads.
412 Status Handle(const IteratorSeekQueryTraceRecord& /* record */,
413 std::unique_ptr<TraceRecordResult>* /* result */) override {
414 return Status::OK();
415 }
416
417 // Ignore reads.
418 Status Handle(const MultiGetQueryTraceRecord& /* record */,
419 std::unique_ptr<TraceRecordResult>* /* result */) override {
420 return Status::OK();
421 }
422
423 // Below are the WriteBatch::Handler overrides. We could use a separate
424 // object, but it's convenient and works to share state with the
425 // `TraceRecord::Handler`.
426
427 Status PutCF(uint32_t column_family_id, const Slice& key_with_ts,
428 const Slice& value) override {
429 Slice key =
430 StripTimestampFromUserKey(key_with_ts, FLAGS_user_timestamp_size);
431 uint64_t key_id;
432 if (!GetIntVal(key.ToString(), &key_id)) {
433 return Status::Corruption("unable to parse key", key.ToString());
434 }
435 uint32_t value_base = GetValueBase(value);
436
437 bool should_buffer_write = !(buffered_writes_ == nullptr);
438 if (should_buffer_write) {
439 return WriteBatchInternal::Put(buffered_writes_.get(), column_family_id,
440 key, value);
441 }
442
443 state_->SyncPut(column_family_id, static_cast<int64_t>(key_id), value_base);
444 ++num_write_ops_;
445 return Status::OK();
446 }
447
448 Status TimedPutCF(uint32_t column_family_id, const Slice& key_with_ts,
449 const Slice& value, uint64_t write_unix_time) override {
450 Slice key =
451 StripTimestampFromUserKey(key_with_ts, FLAGS_user_timestamp_size);
452 uint64_t key_id;
453 if (!GetIntVal(key.ToString(), &key_id)) {
454 return Status::Corruption("unable to parse key", key.ToString());
455 }
456 uint32_t value_base = GetValueBase(value);
457
458 bool should_buffer_write = !(buffered_writes_ == nullptr);
459 if (should_buffer_write) {
460 return WriteBatchInternal::TimedPut(buffered_writes_.get(),
461 column_family_id, key, value,
462 write_unix_time);
463 }
464
465 state_->SyncPut(column_family_id, static_cast<int64_t>(key_id), value_base);
466 ++num_write_ops_;
467 return Status::OK();
468 }
469
470 Status PutEntityCF(uint32_t column_family_id, const Slice& key_with_ts,
471 const Slice& entity) override {
472 Slice key =
473 StripTimestampFromUserKey(key_with_ts, FLAGS_user_timestamp_size);
474
475 uint64_t key_id = 0;
476 if (!GetIntVal(key.ToString(), &key_id)) {
477 return Status::Corruption("Unable to parse key", key.ToString());
478 }
479
480 Slice entity_copy = entity;
481 WideColumns columns;
482 if (!WideColumnSerialization::Deserialize(entity_copy, columns).ok()) {
483 return Status::Corruption("Unable to deserialize entity",
484 entity.ToString(/* hex */ true));
485 }
486
487 if (!VerifyWideColumns(columns)) {
488 return Status::Corruption("Wide columns in entity inconsistent",
489 entity.ToString(/* hex */ true));
490 }
491
492 if (buffered_writes_) {
493 return WriteBatchInternal::PutEntity(buffered_writes_.get(),
494 column_family_id, key, columns);
495 }
496
497 const uint32_t value_base =
498 GetValueBase(WideColumnsHelper::GetDefaultColumn(columns));
499
500 state_->SyncPut(column_family_id, static_cast<int64_t>(key_id), value_base);
501
502 ++num_write_ops_;
503
504 return Status::OK();
505 }
506
507 Status DeleteCF(uint32_t column_family_id,
508 const Slice& key_with_ts) override {
509 Slice key =
510 StripTimestampFromUserKey(key_with_ts, FLAGS_user_timestamp_size);
511 uint64_t key_id;
512 if (!GetIntVal(key.ToString(), &key_id)) {
513 return Status::Corruption("unable to parse key", key.ToString());
514 }
515
516 bool should_buffer_write = !(buffered_writes_ == nullptr);
517 if (should_buffer_write) {
518 return WriteBatchInternal::Delete(buffered_writes_.get(),
519 column_family_id, key);
520 }
521
522 state_->SyncDelete(column_family_id, static_cast<int64_t>(key_id));
523 ++num_write_ops_;
524 return Status::OK();
525 }
526
527 Status SingleDeleteCF(uint32_t column_family_id,
528 const Slice& key_with_ts) override {
529 bool should_buffer_write = !(buffered_writes_ == nullptr);
530 if (should_buffer_write) {
531 Slice key =
532 StripTimestampFromUserKey(key_with_ts, FLAGS_user_timestamp_size);
533 Slice ts =
534 ExtractTimestampFromUserKey(key_with_ts, FLAGS_user_timestamp_size);
535 std::array<Slice, 2> key_with_ts_arr{{key, ts}};
536 return WriteBatchInternal::SingleDelete(
537 buffered_writes_.get(), column_family_id,
538 SliceParts(key_with_ts_arr.data(), 2));
539 }
540
541 return DeleteCF(column_family_id, key_with_ts);
542 }
543
544 Status DeleteRangeCF(uint32_t column_family_id,
545 const Slice& begin_key_with_ts,
546 const Slice& end_key_with_ts) override {
547 Slice begin_key =
548 StripTimestampFromUserKey(begin_key_with_ts, FLAGS_user_timestamp_size);
549 Slice end_key =
550 StripTimestampFromUserKey(end_key_with_ts, FLAGS_user_timestamp_size);
551 uint64_t begin_key_id, end_key_id;
552 if (!GetIntVal(begin_key.ToString(), &begin_key_id)) {
553 return Status::Corruption("unable to parse begin key",
554 begin_key.ToString());
555 }
556 if (!GetIntVal(end_key.ToString(), &end_key_id)) {
557 return Status::Corruption("unable to parse end key", end_key.ToString());
558 }
559
560 bool should_buffer_write = !(buffered_writes_ == nullptr);
561 if (should_buffer_write) {
562 return WriteBatchInternal::DeleteRange(
563 buffered_writes_.get(), column_family_id, begin_key, end_key);
564 }
565
566 state_->SyncDeleteRange(column_family_id,
567 static_cast<int64_t>(begin_key_id),
568 static_cast<int64_t>(end_key_id));
569 ++num_write_ops_;
570 return Status::OK();
571 }
572
573 Status MergeCF(uint32_t column_family_id, const Slice& key_with_ts,
574 const Slice& value) override {
575 Slice key =
576 StripTimestampFromUserKey(key_with_ts, FLAGS_user_timestamp_size);
577
578 bool should_buffer_write = !(buffered_writes_ == nullptr);
579 if (should_buffer_write) {
580 return WriteBatchInternal::Merge(buffered_writes_.get(), column_family_id,
581 key, value);
582 }
583
584 return PutCF(column_family_id, key, value);
585 }
586
587 Status MarkBeginPrepare(bool = false) override {
588 assert(!buffered_writes_);
589 buffered_writes_.reset(new WriteBatch());
590 return Status::OK();
591 }
592
593 Status MarkEndPrepare(const Slice& xid) override {
594 assert(buffered_writes_);
595 std::string xid_str = xid.ToString();
596 assert(xid_to_buffered_writes_.find(xid_str) ==
597 xid_to_buffered_writes_.end());
598
599 xid_to_buffered_writes_[xid_str].swap(buffered_writes_);
600
601 buffered_writes_.reset();
602
603 return Status::OK();
604 }
605
606 Status MarkCommit(const Slice& xid) override {
607 std::string xid_str = xid.ToString();
608 assert(xid_to_buffered_writes_.find(xid_str) !=
609 xid_to_buffered_writes_.end());
610 assert(xid_to_buffered_writes_.at(xid_str));
611
612 Status s = xid_to_buffered_writes_.at(xid_str)->Iterate(this);
613 xid_to_buffered_writes_.erase(xid_str);
614
615 return s;
616 }
617
618 Status MarkRollback(const Slice& xid) override {
619 std::string xid_str = xid.ToString();
620 assert(xid_to_buffered_writes_.find(xid_str) !=
621 xid_to_buffered_writes_.end());
622 assert(xid_to_buffered_writes_.at(xid_str));
623 xid_to_buffered_writes_.erase(xid_str);
624
625 return Status::OK();
626 }
627
628 private:
629 uint64_t num_write_ops_ = 0;
630 uint64_t max_write_ops_;
631 ExpectedState* state_;
632 std::unordered_map<std::string, std::unique_ptr<WriteBatch>>
633 xid_to_buffered_writes_;
634 std::unique_ptr<WriteBatch> buffered_writes_;
635};

639Status FileExpectedStateManager::Restore(DB* db) {
640 assert(HasHistory());
641 SequenceNumber seqno = db->GetLatestSequenceNumber();
642 if (seqno < saved_seqno_) {
643 return Status::Corruption("DB is older than any restorable expected state");
644 }
645
646 std::string state_filename =
647 std::to_string(saved_seqno_) + kStateFilenameSuffix;
648 std::string state_file_path = GetPathForFilename(state_filename);
649
650 std::string latest_file_temp_path =
651 GetTempPathForFilename(kLatestBasename + kStateFilenameSuffix);
652 std::string latest_file_path =
653 GetPathForFilename(kLatestBasename + kStateFilenameSuffix);
654
655 std::string trace_filename =
656 std::to_string(saved_seqno_) + kTraceFilenameSuffix;
657 std::string trace_file_path = GetPathForFilename(trace_filename);
658
659 std::unique_ptr<TraceReader> trace_reader;
660 Status s = NewFileTraceReader(Env::Default(), EnvOptions(), trace_file_path,
661 &trace_reader);
662
663 if (s.ok()) {
664 // We are going to replay on top of "`seqno`.state" to create a new
665 // "LATEST.state". Start off by creating a tempfile so we can later make the
666 // new "LATEST.state" appear atomically using `RenameFile()`.
667 s = CopyFile(FileSystem::Default(), state_file_path, Temperature::kUnknown,
668 latest_file_temp_path, Temperature::kUnknown, 0 /* size */,
669 false /* use_fsync */, nullptr /* io_tracer */);
670 }
671
672 {
673 std::unique_ptr<Replayer> replayer;
674 std::unique_ptr<ExpectedState> state;
675 std::unique_ptr<ExpectedStateTraceRecordHandler> handler;
676 if (s.ok()) {
677 state.reset(new FileExpectedState(latest_file_temp_path, max_key_,
678 num_column_families_));
679 s = state->Open(false /* create */);
680 }
681 if (s.ok()) {
682 handler.reset(new ExpectedStateTraceRecordHandler(seqno - saved_seqno_,
683 state.get()));
684 // TODO(ajkr): An API limitation requires we provide `handles` although
685 // they will be unused since we only use the replayer for reading records.
686 // Just give a default CFH for now to satisfy the requirement.
687 s = db->NewDefaultReplayer({db->DefaultColumnFamily()} /* handles */,
688 std::move(trace_reader), &replayer);
689 }
690
691 if (s.ok()) {
692 s = replayer->Prepare();
693 }
694 for (; s.ok();) {
695 std::unique_ptr<TraceRecord> record;
696 s = replayer->Next(&record);
697 if (!s.ok()) {
698 if (s.IsCorruption() && handler->IsDone()) {
699 // There could be a corruption reading the tail record of the trace
700 // due to `db_stress` crashing while writing it. It shouldn't matter
701 // as long as we already found all the write ops we need to catch up
702 // the expected state.
703 s = Status::OK();
704 }
705 if (s.IsIncomplete()) {
706 // OK because `Status::Incomplete` is expected upon finishing all the
707 // trace records.
708 s = Status::OK();
709 }
710 break;
711 }
712 std::unique_ptr<TraceRecordResult> res;
713 s = record->Accept(handler.get(), &res);
714 }
715 }
716
717 if (s.ok()) {
718 s = FileSystem::Default()->RenameFile(latest_file_temp_path,
719 latest_file_path, IOOptions(),
720 nullptr /* dbg */);
721 }
722 if (s.ok()) {
723 latest_.reset(new FileExpectedState(latest_file_path, max_key_,
724 num_column_families_));
725 s = latest_->Open(false /* create */);
726 }
727
728 // Delete old state/trace files. We must delete the state file first.
729 // Otherwise, a crash-recovery immediately after deleting the trace file could
730 // lead to `Restore()` unable to replay to `seqno`.
731 if (s.ok()) {
732 s = Env::Default()->DeleteFile(state_file_path);
733 }
734 if (s.ok()) {
735 std::vector<std::string> expected_state_dir_children;
736 s = Env::Default()->GetChildren(expected_state_dir_path_,
737 &expected_state_dir_children);
738 if (s.ok()) {
739 for (size_t i = 0; i < expected_state_dir_children.size(); ++i) {
740 const auto& filename = expected_state_dir_children[i];
741 if (filename.size() >= kTraceFilenameSuffix.size() &&
742 filename.rfind(kTraceFilenameSuffix) ==
743 filename.size() - kTraceFilenameSuffix.size()) {
744 SequenceNumber found_seqno = ParseUint64(filename.substr(
745 0, filename.size() - kTraceFilenameSuffix.size()));
746 // Delete older trace files, but keep the one we just replayed for
747 // debugging purposes
748 if (found_seqno < saved_seqno_) {
749 s = Env::Default()->DeleteFile(GetPathForFilename(filename));
750 }
751 }
752 if (!s.ok()) {
753 break;
754 }
755 }
756 }
757 if (s.ok()) {
758 saved_seqno_ = kMaxSequenceNumber;
759 }
760 }
761 return s;
762}

764Status FileExpectedStateManager::Clean() {
765 std::vector<std::string> expected_state_dir_children;
766 Status s = Env::Default()->GetChildren(expected_state_dir_path_,
767 &expected_state_dir_children);
768 // An incomplete `Open()` or incomplete `SaveAtAndAfter()` could have left
769 // behind invalid temporary files. An incomplete `SaveAtAndAfter()` could have
770 // also left behind stale state/trace files. An incomplete `Restore()` could
771 // have left behind stale trace files.
772 for (size_t i = 0; s.ok() && i < expected_state_dir_children.size(); ++i) {
773 const auto& filename = expected_state_dir_children[i];
774 if (filename.rfind(kTempFilenamePrefix, 0 /* pos */) == 0 &&
775 filename.size() >= kTempFilenameSuffix.size() &&
776 filename.rfind(kTempFilenameSuffix) ==
777 filename.size() - kTempFilenameSuffix.size()) {
778 // Delete all temp files.
779 s = Env::Default()->DeleteFile(GetPathForFilename(filename));
780 } else if (filename.size() >= kStateFilenameSuffix.size() &&
781 filename.rfind(kStateFilenameSuffix) ==
782 filename.size() - kStateFilenameSuffix.size() &&
783 filename.rfind(kLatestBasename, 0) == std::string::npos &&
784 ParseUint64(filename.substr(
785 0, filename.size() - kStateFilenameSuffix.size())) <
786 saved_seqno_) {
787 assert(saved_seqno_ != kMaxSequenceNumber);
788 // Delete stale state files.
789 s = Env::Default()->DeleteFile(GetPathForFilename(filename));
790 } else if (filename.size() >= kTraceFilenameSuffix.size() &&
791 filename.rfind(kTraceFilenameSuffix) ==
792 filename.size() - kTraceFilenameSuffix.size() &&
793 ParseUint64(filename.substr(
794 0, filename.size() - kTraceFilenameSuffix.size())) <
795 saved_seqno_) {
796 // Delete stale trace files.
797 s = Env::Default()->DeleteFile(GetPathForFilename(filename));
798 }
799 }
800 return s;
801}
