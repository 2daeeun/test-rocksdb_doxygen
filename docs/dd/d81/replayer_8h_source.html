19struct ReplayOptions {
20 // Number of threads used for replaying. If 0 or 1, replay using
21 // single thread.
22 uint32_t num_threads;
23
24 // Enables fast forwarding a replay by increasing/reducing the delay between
25 // the ingested traces.
26 // If > 0.0 and < 1.0, slow down the replay by this amount.
27 // If 1.0, replay the operations at the same rate as in the trace stream.
28 // If > 1, speed up the replay by this amount.
29 double fast_forward;
30
31 ReplayOptions() : num_threads(1), fast_forward(1.0) {}
32
33 ReplayOptions(uint32_t num_of_threads, double fast_forward_ratio)
34 : num_threads(num_of_threads), fast_forward(fast_forward_ratio) {}
35};

37// Replayer helps to replay the captured RocksDB query level operations.
38// The Replayer can either be created from DB::NewReplayer method, or be
39// instantiated via db_bench today, on using "replay" benchmark.
40class Replayer {
41 public:
42 virtual ~Replayer() = default;
43
44 // Make some preparation before replaying the trace. This will also reset the
45 // replayer in order to restart replaying.
46 virtual Status Prepare() = 0;
47
48 // Return the timestamp when the trace recording was started.
49 virtual uint64_t GetHeaderTimestamp() const = 0;
50
51 // Atomically read one trace into a TraceRecord (excluding the header and
52 // footer traces).
53 // Return Status::OK() on success;
54 // Status::Incomplete() if Prepare() was not called or no more available
55 // trace;
56 // Status::NotSupported() if the read trace type is not supported.
57 virtual Status Next(std::unique_ptr<TraceRecord>* record) = 0;
58
59 // Execute one TraceRecord.
60 // Return Status::OK() if the execution was successful. Get/MultiGet traces
61 // will still return Status::OK() even if they got Status::NotFound()
62 // from DB::Get() or DB::MultiGet();
63 // Status::Incomplete() if Prepare() was not called or no more available
64 // trace;
65 // Status::NotSupported() if the operation is not supported;
66 // Otherwise, return the corresponding error status.
67 //
68 // The actual operation execution status and result(s) will be saved in
69 // result. For example, a GetQueryTraceRecord will have its DB::Get() status
70 // and the returned value saved in a SingleValueTraceExecutionResult.
71 virtual Status Execute(const std::unique_ptr<TraceRecord>& record,
72 std::unique_ptr<TraceRecordResult>* result) = 0;
73
74 // Replay all the traces from the provided trace stream, taking the delay
75 // between the traces into consideration.
76 //
77 // result_callback reports the status of executing a trace record, and the
78 // actual operation execution result (See the description for Execute()).
79 virtual Status Replay(
80 const ReplayOptions& options,
81 const std::function<void(Status, std::unique_ptr<TraceRecordResult>&&)>&
82 result_callback) = 0;
83};
