19// The merge context for merging a user key.
20// When doing a Get(), DB will create such a class and pass it when
21// issuing Get() operation to memtables and version_set. The operands
22// will be fetched from the context when issuing partial of full merge.
23class MergeContext {
24 public:
25 GetMergeOperandsOptions* get_merge_operands_options = nullptr;
26
27 // Clear all the operands
28 void Clear() {
29 if (operand_list_) {
30 operand_list_->clear();
31 copied_operands_->clear();
32 }
33 }
34
35 // Push a merge operand
36 void PushOperand(const Slice& operand_slice, bool operand_pinned = false) {
37 Initialize();
38 SetDirectionBackward();
39
40 if (operand_pinned) {
41 operand_list_->push_back(operand_slice);
42 } else {
43 // We need to have our own copy of the operand since it's not pinned
44 copied_operands_->emplace_back(
45 new std::string(operand_slice.data(), operand_slice.size()));
46 operand_list_->push_back(*copied_operands_->back());
47 }
48 }
49
50 // Push back a merge operand
51 void PushOperandBack(const Slice& operand_slice,
52 bool operand_pinned = false) {
53 Initialize();
54 SetDirectionForward();
55
56 if (operand_pinned) {
57 operand_list_->push_back(operand_slice);
58 } else {
59 // We need to have our own copy of the operand since it's not pinned
60 copied_operands_->emplace_back(
61 new std::string(operand_slice.data(), operand_slice.size()));
62 operand_list_->push_back(*copied_operands_->back());
63 }
64 }
65
66 // return total number of operands in the list
67 size_t GetNumOperands() const {
68 if (!operand_list_) {
69 return 0;
70 }
71 return operand_list_->size();
72 }
73
74 // Get the operand at the index.
75 Slice GetOperand(int index) const {
76 assert(operand_list_);
77
78 SetDirectionForward();
79 return (*operand_list_)[index];
80 }
81
82 // Same as GetOperandsDirectionForward
83 //
84 // Note that the returned reference is only good until another call
85 // to this MergeContext. If the returned value is needed for longer,
86 // a copy must be made.
87 const std::vector<Slice>& GetOperands() const {
88 return GetOperandsDirectionForward();
89 }
90
91 // Return all the operands in the order as they were merged (passed to
92 // FullMerge or FullMergeV2)
93 //
94 // Note that the returned reference is only good until another call
95 // to this MergeContext. If the returned value is needed for longer,
96 // a copy must be made.
97 const std::vector<Slice>& GetOperandsDirectionForward() const {
98 if (!operand_list_) {
99 return empty_operand_list;
100 }
101
102 SetDirectionForward();
103 return *operand_list_;
104 }
105
106 // Return all the operands in the reversed order relative to how they were
107 // merged (passed to FullMerge or FullMergeV2)
108 //
109 // Note that the returned reference is only good until another call
110 // to this MergeContext. If the returned value is needed for longer,
111 // a copy must be made.
112 const std::vector<Slice>& GetOperandsDirectionBackward() const {
113 if (!operand_list_) {
114 return empty_operand_list;
115 }
116
117 SetDirectionBackward();
118 return *operand_list_;
119 }
120
121 private:
122 void Initialize() {
123 if (!operand_list_) {
124 operand_list_.reset(new std::vector<Slice>());
125 copied_operands_.reset(new std::vector<std::unique_ptr<std::string>>());
126 }
127 }
128
129 void SetDirectionForward() const {
130 if (operands_reversed_ == true) {
131 std::reverse(operand_list_->begin(), operand_list_->end());
132 operands_reversed_ = false;
133 }
134 }
135
136 void SetDirectionBackward() const {
137 if (operands_reversed_ == false) {
138 std::reverse(operand_list_->begin(), operand_list_->end());
139 operands_reversed_ = true;
140 }
141 }
142
143 // List of operands
144 mutable std::unique_ptr<std::vector<Slice>> operand_list_;
145 // Copy of operands that are not pinned.
146 std::unique_ptr<std::vector<std::unique_ptr<std::string>>> copied_operands_;
147 mutable bool operands_reversed_ = true;
148};
