23void UpdateOptionsFiles(DB* db,
24 std::unordered_set<std::string>* filename_history,
25 int* options_files_count) {
26 std::vector<std::string> filenames;
27 EXPECT_OK(db->GetEnv()->GetChildren(db->GetName(), &filenames));
28 uint64_t number;
29 FileType type;
30 *options_files_count = 0;
31 for (const auto& filename : filenames) {
32 if (ParseFileName(filename, &number, &type) && type == kOptionsFile) {
33 filename_history->insert(filename);
34 (*options_files_count)++;

39// Verify whether the current Options Files are the latest ones.
40void VerifyOptionsFileName(
41 DB* db, const std::unordered_set<std::string>& past_filenames) {
42 std::vector<std::string> filenames;
43 std::unordered_set<std::string> current_filenames;
44 EXPECT_OK(db->GetEnv()->GetChildren(db->GetName(), &filenames));
45 uint64_t number;
46 FileType type;
47 for (const auto& filename : filenames) {
48 if (ParseFileName(filename, &number, &type) && type == kOptionsFile) {
49 current_filenames.insert(filename);
50 }
51 }
52 for (const auto& past_filename : past_filenames) {
53 if (current_filenames.find(past_filename) != current_filenames.end()) {
54 continue;
55 }
56 for (const auto& filename : current_filenames) {
57 ASSERT_GT(filename, past_filename);
58 }
59 }
60}

63TEST_F(OptionsFileTest, NumberOfOptionsFiles) {
64 const int kReopenCount = 20;
65 Options opt;
66 opt.create_if_missing = true;
67 ASSERT_OK(DestroyDB(dbname_, opt));
68 std::unordered_set<std::string> filename_history;
69 DB* db;
70 for (int i = 0; i < kReopenCount; ++i) {
71 ASSERT_OK(DB::Open(opt, dbname_, &db));
72 int num_options_files = 0;
73 UpdateOptionsFiles(db, &filename_history, &num_options_files);
74 ASSERT_GT(num_options_files, 0);
75 ASSERT_LE(num_options_files, 2);
76 // Make sure we always keep the latest option files.
77 VerifyOptionsFileName(db, filename_history);
78 delete db;
79 }
80}
