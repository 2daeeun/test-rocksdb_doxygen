19void StderrLogger::Logv(const char* format, va_list ap) {
20 const uint64_t thread_id = Env::Default()->GetThreadID();
21
22 port::TimeVal now_tv;
23 port::GetTimeOfDay(&now_tv, nullptr);
24 const time_t seconds = now_tv.tv_sec;
25 struct tm t;
26 port::LocalTimeR(&seconds, &t);
27
28 // The string we eventually log has three parts: the context (time, thread),
29 // optional user-supplied prefix, and the actual log message (the "suffix").
30 //
31 // We compute their lengths so that we can allocate a buffer big enough to
32 // print it. The context string (with the date and thread id) is really only
33 // 44 bytes, but we allocate 50 to be safe.
34 //
35 // ctx_len = 44 = ( 4+ 1+ 2+1+2+ 1+2+ 1+2+ 1+ 2+1+6+ 1+16+1)
36 const char* ctx_prefix_fmt = "%04d/%02d/%02d-%02d:%02d:%02d.%06d %llx %s";
37 size_t ctx_len = 50;
38
39 va_list ap_copy;
40 va_copy(ap_copy, ap);
41 const size_t log_suffix_len = vsnprintf(nullptr, 0, format, ap_copy) + 1;
42 va_end(ap_copy);
43
44 // Allocate space for the context, log_prefix, and log itself
45 // Extra byte for null termination
46 size_t buf_len = ctx_len + log_prefix_len + log_suffix_len;
47 std::unique_ptr<char[]> buf(new char[buf_len]);
48
49 // If the logger was created without a prefix, the prefix is a nullptr
50 const char* prefix = log_prefix == nullptr ? "" : log_prefix;
51
52 // Write out the context and prefix string
53 int written =
54 snprintf(buf.get(), ctx_len + log_prefix_len, ctx_prefix_fmt,
55 t.tm_year + 1900, t.tm_mon + 1, t.tm_mday, t.tm_hour, t.tm_min,
56 t.tm_sec, static_cast<int>(now_tv.tv_usec),
57 static_cast<long long unsigned int>(thread_id), prefix);
58 vsnprintf(buf.get() + written, log_suffix_len, format, ap);
59
60 fprintf(stderr, "%s%c", buf.get(), '\n');
61}
