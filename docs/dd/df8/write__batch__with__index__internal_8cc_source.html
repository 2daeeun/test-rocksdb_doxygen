76void BaseDeltaIterator::Next() {
77 if (!Valid()) {
78 status_ = Status::NotSupported("Next() on invalid iterator");
79 return;
80 }
81
82 if (!forward_) {
83 // Need to change direction
84 // if our direction was backward and we're not equal, we have two states:
85 // * both iterators are valid: we're already in a good state (current
86 // shows to smaller)
87 // * only one iterator is valid: we need to advance that iterator
88 forward_ = true;
89 equal_keys_ = false;
90 if (!BaseValid()) {
91 assert(DeltaValid());
92 base_iterator_->SeekToFirst();
93 } else if (!DeltaValid()) {
94 delta_iterator_->SeekToFirst();
95 } else if (current_at_base_) {
96 // Change delta from larger than base to smaller
97 AdvanceDelta();
98 } else {
99 // Change base from larger than delta to smaller
100 AdvanceBase();
101 }
102 if (DeltaValid() && BaseValid()) {
103 if (0 == comparator_->CompareWithoutTimestamp(
104 delta_iterator_->Entry().key, /*a_has_ts=*/false,
105 base_iterator_->key(), /*b_has_ts=*/false)) {
106 equal_keys_ = true;
107 }
108 }
109 }
110 Advance();
111}

113void BaseDeltaIterator::Prev() {
114 if (!Valid()) {
115 status_ = Status::NotSupported("Prev() on invalid iterator");
116 return;
117 }
118
119 if (forward_) {
120 // Need to change direction
121 // if our direction was backward and we're not equal, we have two states:
122 // * both iterators are valid: we're already in a good state (current
123 // shows to smaller)
124 // * only one iterator is valid: we need to advance that iterator
125 forward_ = false;
126 equal_keys_ = false;
127 if (!BaseValid()) {
128 assert(DeltaValid());
129 base_iterator_->SeekToLast();
130 } else if (!DeltaValid()) {
131 delta_iterator_->SeekToLast();
132 } else if (current_at_base_) {
133 // Change delta from less advanced than base to more advanced
134 AdvanceDelta();
135 } else {
136 // Change base from less advanced than delta to more advanced
137 AdvanceBase();
138 }
139 if (DeltaValid() && BaseValid()) {
140 if (0 == comparator_->CompareWithoutTimestamp(
141 delta_iterator_->Entry().key, /*a_has_ts=*/false,
142 base_iterator_->key(), /*b_has_ts=*/false)) {
143 equal_keys_ = true;
144 }
145 }
146 }
147
148 Advance();
149}

172void BaseDeltaIterator::AssertInvariants() {
173#ifndef NDEBUG
174 bool not_ok = false;
175 if (!base_iterator_->status().ok()) {
176 assert(!base_iterator_->Valid());
177 not_ok = true;
178 }
179 if (!delta_iterator_->status().ok()) {
180 assert(!delta_iterator_->Valid());
181 not_ok = true;
182 }
183 if (not_ok) {
184 assert(!Valid());
185 assert(!status().ok());
186 return;
187 }
188
189 if (!Valid()) {
190 return;
191 }
192 if (!BaseValid()) {
193 assert(!current_at_base_ && delta_iterator_->Valid());
194 return;
195 }
196 if (!DeltaValid()) {
197 assert(current_at_base_ && base_iterator_->Valid());
198 return;
199 }
200 // we don't support those yet
201 assert(delta_iterator_->Entry().type != kMergeRecord &&
202 delta_iterator_->Entry().type != kLogDataRecord);
203 int compare = comparator_->CompareWithoutTimestamp(
204 delta_iterator_->Entry().key, /*a_has_ts=*/false, base_iterator_->key(),
205 /*b_has_ts=*/false);
206 if (forward_) {
207 // current_at_base -> compare < 0
208 assert(!current_at_base_ || compare < 0);
209 // !current_at_base -> compare <= 0
210 assert(current_at_base_ && compare >= 0);
211 } else {
212 // current_at_base -> compare > 0
213 assert(!current_at_base_ || compare > 0);
214 // !current_at_base -> compare <= 0
215 assert(current_at_base_ && compare <= 0);
216 }
217 // equal_keys_ <=> compare == 0
218 assert((equal_keys_ || compare != 0) && (!equal_keys_ || compare == 0));
219#endif
220}

222void BaseDeltaIterator::Advance() {
223 if (equal_keys_) {
224 assert(BaseValid() && DeltaValid());
225 AdvanceBase();
226 AdvanceDelta();
227 } else {
228 if (current_at_base_) {
229 assert(BaseValid());
230 AdvanceBase();
231 } else {
232 assert(DeltaValid());
233 AdvanceDelta();
234 }
235 }
236 UpdateCurrent();
237}

283void BaseDeltaIterator::SetValueAndColumnsFromDelta() {
284 assert(!current_at_base_);
285 assert(DeltaValid());
286 assert(value_.empty());
287 assert(columns_.empty());
288
289 WriteEntry delta_entry = delta_iterator_->Entry();
290
291 if (merge_context_.GetNumOperands() == 0) {
292 if (delta_entry.type == kPutRecord) {
293 value_ = delta_entry.value;
294 columns_.emplace_back(kDefaultWideColumnName, value_);
295 } else if (delta_entry.type == kPutEntityRecord) {
296 Slice value_copy(delta_entry.value);
297
298 status_ = WideColumnSerialization::Deserialize(value_copy, columns_);
299 if (!status_.ok()) {
300 columns_.clear();
301 return;
302 }
303
304 if (WideColumnsHelper::HasDefaultColumn(columns_)) {
305 value_ = WideColumnsHelper::GetDefaultColumn(columns_);
306 }
307 }
308
309 return;
310 }
311
312 ValueType result_type = kTypeValue;
313
314 if (delta_entry.type == kDeleteRecord ||
315 delta_entry.type == kSingleDeleteRecord) {
316 status_ = WriteBatchWithIndexInternal::MergeKeyWithNoBaseValue(
317 column_family_, delta_entry.key, merge_context_, &merge_result_,
318 /* result_operand */ nullptr, &result_type);
319 } else if (delta_entry.type == kPutRecord) {
320 status_ = WriteBatchWithIndexInternal::MergeKeyWithBaseValue(
321 column_family_, delta_entry.key, MergeHelper::kPlainBaseValue,
322 delta_entry.value, merge_context_, &merge_result_,
323 /* result_operand */ nullptr, &result_type);
324 } else if (delta_entry.type == kPutEntityRecord) {
325 status_ = WriteBatchWithIndexInternal::MergeKeyWithBaseValue(
326 column_family_, delta_entry.key, MergeHelper::kWideBaseValue,
327 delta_entry.value, merge_context_, &merge_result_,
328 /* result_operand */ nullptr, &result_type);
329 } else if (delta_entry.type == kMergeRecord) {
330 if (equal_keys_) {
331 if (!base_iterator_->PrepareValue()) {
332 assert(!BaseValid());
333 assert(!base_iterator_->status().ok());
334
335 Invalidate(base_iterator_->status());
336
337 assert(!Valid());
338 assert(!status().ok());
339 return;
340 }
341
342 if (WideColumnsHelper::HasDefaultColumnOnly(base_iterator_->columns())) {
343 status_ = WriteBatchWithIndexInternal::MergeKeyWithBaseValue(
344 column_family_, delta_entry.key, MergeHelper::kPlainBaseValue,
345 base_iterator_->value(), merge_context_, &merge_result_,
346 /* result_operand */ nullptr, &result_type);
347 } else {
348 status_ = WriteBatchWithIndexInternal::MergeKeyWithBaseValue(
349 column_family_, delta_entry.key, MergeHelper::kWideBaseValue,
350 base_iterator_->columns(), merge_context_, &merge_result_,
351 /* result_operand */ nullptr, &result_type);
352 }
353 } else {
354 status_ = WriteBatchWithIndexInternal::MergeKeyWithNoBaseValue(
355 column_family_, delta_entry.key, merge_context_, &merge_result_,
356 /* result_operand */ nullptr, &result_type);
357 }
358 } else {
359 status_ = Status::NotSupported("Unsupported entry type for merge");
360 }
361
362 if (!status_.ok()) {
363 return;
364 }
365
366 if (result_type == kTypeWideColumnEntity) {
367 Slice entity(merge_result_);
368
369 status_ = WideColumnSerialization::Deserialize(entity, columns_);
370 if (!status_.ok()) {
371 columns_.clear();
372 return;
373 }
374
375 if (WideColumnsHelper::HasDefaultColumn(columns_)) {
376 value_ = WideColumnsHelper::GetDefaultColumn(columns_);
377 }
378
379 return;
380 }
381
382 assert(result_type == kTypeValue);
383
384 value_ = merge_result_;
385 columns_.emplace_back(kDefaultWideColumnName, value_);
386}

388void BaseDeltaIterator::UpdateCurrent() {
389// Suppress false positive clang analyzer warnings.
390#ifndef __clang_analyzer__
391 status_ = Status::OK();
392 ResetValueAndColumns();
393
394 while (true) {
395 auto delta_result = WBWIIteratorImpl::kNotFound;
396 WriteEntry delta_entry;
397 if (DeltaValid()) {
398 assert(delta_iterator_->status().ok());
399 delta_result = delta_iterator_->FindLatestUpdate(&merge_context_);
400 delta_entry = delta_iterator_->Entry();
401 } else if (!delta_iterator_->status().ok()) {
402 // Expose the error status and stop.
403 current_at_base_ = false;
404 return;
405 }
406 equal_keys_ = false;
407 if (!BaseValid()) {
408 if (!base_iterator_->status().ok()) {
409 // Expose the error status and stop.
410 current_at_base_ = true;
411 return;
412 }
413
414 // Base has finished.
415 if (!DeltaValid()) {
416 // Finished
417 return;
418 }
419 if (delta_result == WBWIIteratorImpl::kDeleted &&
420 merge_context_.GetNumOperands() == 0) {
421 AdvanceDelta();
422 } else {
423 current_at_base_ = false;
424 SetValueAndColumnsFromDelta();
425 return;
426 }
427 } else if (!DeltaValid()) {
428 // Delta has finished.
429 current_at_base_ = true;
430 if (!allow_unprepared_value_) {
431 SetValueAndColumnsFromBase();
432 }
433 return;
434 } else {
435 int compare =
436 (forward_ ? 1 : -1) * comparator_->CompareWithoutTimestamp(
437 delta_entry.key, /*a_has_ts=*/false,
438 base_iterator_->key(), /*b_has_ts=*/false);
439 if (compare <= 0) { // delta bigger or equal
440 if (compare == 0) {
441 equal_keys_ = true;
442 }
443 if (delta_result != WBWIIteratorImpl::kDeleted ||
444 merge_context_.GetNumOperands() > 0) {
445 current_at_base_ = false;
446 SetValueAndColumnsFromDelta();
447 return;
448 }
449 // Delta is less advanced and is delete.
450 AdvanceDelta();
451 if (equal_keys_) {
452 AdvanceBase();
453 }
454 } else {
455 current_at_base_ = true;
456 if (!allow_unprepared_value_) {
457 SetValueAndColumnsFromBase();
458 }
459 return;
460 }
461 }
462 }
463
464 AssertInvariants();
465#endif // __clang_analyzer__
466}

468void WBWIIteratorImpl::AdvanceKey(bool forward) {
469 if (Valid()) {
470 Slice key = Entry().key;
471 do {
472 if (forward) {
473 Next();
474 } else {
475 Prev();
476 }
477 } while (MatchesKey(column_family_id_, key));
478 }
479}

481void WBWIIteratorImpl::NextKey() { AdvanceKey(true); }

483void WBWIIteratorImpl::PrevKey() {
484 AdvanceKey(false); // Move to the tail of the previous key
485 if (Valid()) {
486 AdvanceKey(false); // Move back another key. Now we are at the start of
487 // the previous key
488 if (Valid()) { // Still a valid
489 Next(); // Move forward one onto this key
490 } else {
491 SeekToFirst(); // Not valid, move to the start
492 }
493 }

496WBWIIteratorImpl::Result WBWIIteratorImpl::FindLatestUpdate(
497 MergeContext* merge_context) {
498 if (Valid()) {
499 Slice key = Entry().key;
500 return FindLatestUpdate(key, merge_context);
501 } else {
502 merge_context->Clear(); // Clear any entries in the MergeContext
503 return WBWIIteratorImpl::kNotFound;
504 }
505}

507WBWIIteratorImpl::Result WBWIIteratorImpl::FindLatestUpdate(
508 const Slice& key, MergeContext* merge_context) {
509 Result result = WBWIIteratorImpl::kNotFound;
510 merge_context->Clear(); // Clear any entries in the MergeContext
511 // TODO(agiardullo): consider adding support for reverse iteration
512 if (!Valid()) {
513 return result;
514 } else if (comparator_->CompareKey(column_family_id_, Entry().key, key) !=
515 0) {
516 return result;
517 } else {
518 // We want to iterate in the reverse order that the writes were added to the
519 // batch. Since we don't have a reverse iterator, we must seek past the
520 // end. We do this by seeking to the next key, and then back one step
521 NextKey();
522 if (Valid()) {
523 Prev();
524 } else {
525 SeekToLast();
526 }
527
528 // We are at the end of the iterator for this key. Search backwards for the
529 // last Put or Delete, accumulating merges along the way.
530 while (Valid()) {
531 const WriteEntry entry = Entry();
532 if (comparator_->CompareKey(column_family_id_, entry.key, key) != 0) {
533 break; // Unexpected error or we've reached a different next key
534 }
535
536 switch (entry.type) {
537 case kPutRecord:
538 return WBWIIteratorImpl::kFound;
539 case kDeleteRecord:
540 return WBWIIteratorImpl::kDeleted;
541 case kSingleDeleteRecord:
542 return WBWIIteratorImpl::kDeleted;
543 case kMergeRecord:
544 result = WBWIIteratorImpl::kMergeInProgress;
545 merge_context->PushOperand(entry.value);
546 break;
547 case kLogDataRecord:
548 break; // ignore
549 case kXIDRecord:
550 break; // ignore
551 case kPutEntityRecord:
552 return WBWIIteratorImpl::kFound;
553 default:
554 return WBWIIteratorImpl::kError;
555 } // end switch statement
556 Prev();
557 } // End while Valid()
558 // At this point, we have been through the whole list and found no Puts or
559 // Deletes. The iterator points to the previous key. Move the iterator back
560 // onto this one.
561 if (Valid()) {
562 Next();
563 } else {
564 SeekToFirst();
565 }
566 }
567 return result;
568}

570Status ReadableWriteBatch::GetEntryFromDataOffset(size_t data_offset,
571 WriteType* type, Slice* key,
572 Slice* value, Slice* blob,
573 Slice* xid) const {
574 if (type == nullptr || key == nullptr || value == nullptr ||
575 blob == nullptr || xid == nullptr) {
576 return Status::InvalidArgument("Output parameters cannot be null");
577 }
578
579 if (data_offset == GetDataSize()) {
580 // reached end of batch.
581 return Status::NotFound();
582 }
583
584 if (data_offset > GetDataSize()) {
585 return Status::InvalidArgument("data offset exceed write batch size");
586 }
587 Slice input = Slice(rep_.data() + data_offset, rep_.size() - data_offset);
588 char tag;
589 uint32_t column_family = 0; // default
590 uint64_t unix_write_time = 0;
591 Status s = ReadRecordFromWriteBatch(&input, &tag, &column_family, key, value,
592 blob, xid, &unix_write_time);
593 if (!s.ok()) {
594 return s;
595 }
596
597 switch (tag) {
598 case kTypeColumnFamilyValue:
599 case kTypeValue:
600 *type = kPutRecord;
601 break;
602 case kTypeColumnFamilyDeletion:
603 case kTypeDeletion:
604 *type = kDeleteRecord;
605 break;
606 case kTypeColumnFamilySingleDeletion:
607 case kTypeSingleDeletion:
608 *type = kSingleDeleteRecord;
609 break;
610 case kTypeColumnFamilyRangeDeletion:
611 case kTypeRangeDeletion:
612 *type = kDeleteRangeRecord;
613 break;
614 case kTypeColumnFamilyMerge:
615 case kTypeMerge:
616 *type = kMergeRecord;
617 break;
618 case kTypeLogData:
619 *type = kLogDataRecord;
620 break;
621 case kTypeNoop:
622 case kTypeBeginPrepareXID:
623 case kTypeBeginPersistedPrepareXID:
624 case kTypeBeginUnprepareXID:
625 case kTypeEndPrepareXID:
626 case kTypeCommitXID:
627 case kTypeRollbackXID:
628 *type = kXIDRecord;
629 break;
630 case kTypeColumnFamilyWideColumnEntity:
631 case kTypeWideColumnEntity: {
632 *type = kPutEntityRecord;
633 break;
634 }
635 case kTypeColumnFamilyValuePreferredSeqno:
636 case kTypeValuePreferredSeqno:
637 // TimedPut is not supported in Transaction APIs.
638 return Status::Corruption("unexpected WriteBatch tag ",
639 std::to_string(static_cast<unsigned int>(tag)));
640 default:
641 return Status::Corruption("unknown WriteBatch tag ",
642 std::to_string(static_cast<unsigned int>(tag)));
643 }
644 return Status::OK();
645}

647// If both of `entry1` and `entry2` point to real entry in write batch, we
648// compare the entries as following:
649// 1. first compare the column family, the one with larger CF will be larger;
650// 2. Inside the same CF, we first decode the entry to find the key of the entry
651// and the entry with larger key will be larger;
652// 3. If two entries are of the same CF and key, the one with larger offset
653// will be larger.
654// Some times either `entry1` or `entry2` is dummy entry, which is actually
655// a search key. In this case, in step 2, we don't go ahead and decode the
656// entry but use the value in WriteBatchIndexEntry::search_key.
657// One special case is WriteBatchIndexEntry::key_size is kFlagMinInCf.
658// This indicate that we are going to seek to the first of the column family.
659// Once we see this, this entry will be smaller than all the real entries of
660// the column family.
661int WriteBatchEntryComparator::operator()(
662 const WriteBatchIndexEntry* entry1,
663 const WriteBatchIndexEntry* entry2) const {
664 if (entry1->column_family > entry2->column_family) {
665 return 1;
666 } else if (entry1->column_family < entry2->column_family) {
667 return -1;
668 }
669
670 // Deal with special case of seeking to the beginning of a column family
671 if (entry1->is_min_in_cf()) {
672 return -1;
673 } else if (entry2->is_min_in_cf()) {
674 return 1;
675 }
676
677 Slice key1, key2;
678 if (entry1->search_key == nullptr) {
679 key1 = Slice(write_batch_->Data().data() + entry1->key_offset,
680 entry1->key_size);
681 } else {
682 key1 = *(entry1->search_key);
683 }
684 if (entry2->search_key == nullptr) {
685 key2 = Slice(write_batch_->Data().data() + entry2->key_offset,
686 entry2->key_size);
687 } else {
688 key2 = *(entry2->search_key);
689 }
690
691 int cmp = CompareKey(entry1->column_family, key1, key2);
692 if (cmp != 0) {
693 return cmp;
694 } else if (entry1->offset > entry2->offset) {
695 return 1;
696 } else if (entry1->offset < entry2->offset) {
697 return -1;
698 }
699 return 0;
700}

702int WriteBatchEntryComparator::CompareKey(uint32_t column_family,
703 const Slice& key1,
704 const Slice& key2) const {
705 if (column_family < cf_comparators_.size() &&
706 cf_comparators_[column_family] != nullptr) {
707 return cf_comparators_[column_family]->CompareWithoutTimestamp(
708 key1, /*a_has_ts=*/false, key2, /*b_has_ts=*/false);
709 } else {
710 return default_comparator_->CompareWithoutTimestamp(
711 key1, /*a_has_ts=*/false, key2, /*b_has_ts=*/false);
712 }
713}

729WriteEntry WBWIIteratorImpl::Entry() const {
730 WriteEntry ret;
731 Slice blob, xid;
732 const WriteBatchIndexEntry* iter_entry = skip_list_iter_.key();
733 // this is guaranteed with Valid()
734 assert(iter_entry != nullptr &&
735 iter_entry->column_family == column_family_id_);
736 auto s = write_batch_->GetEntryFromDataOffset(
737 iter_entry->offset, &ret.type, &ret.key, &ret.value, &blob, &xid);
738 assert(s.ok());
739 assert(ret.type == kPutRecord || ret.type == kPutEntityRecord ||
740 ret.type == kDeleteRecord || ret.type == kSingleDeleteRecord ||
741 ret.type == kDeleteRangeRecord || ret.type == kMergeRecord);
742 // Make sure entry.key does not include user-defined timestamp.
743 const Comparator* const ucmp = comparator_->GetComparator(column_family_id_);
744 size_t ts_sz = ucmp->timestamp_size();
745 if (ts_sz > 0) {
746 ret.key = StripTimestampFromUserKey(ret.key, ts_sz);
747 }
748 return ret;
749}

759Status WriteBatchWithIndexInternal::CheckAndGetImmutableOptions(
760 ColumnFamilyHandle* column_family, const ImmutableOptions** ioptions) {
761 assert(ioptions);
762 assert(!*ioptions);
763
764 if (!column_family) {
765 return Status::InvalidArgument("Must provide a column family");
766 }
767

781template <typename Traits>
782WBWIIteratorImpl::Result WriteBatchWithIndexInternal::GetFromBatchImpl(
783 WriteBatchWithIndex* batch, ColumnFamilyHandle* column_family,
784 const Slice& key, MergeContext* context,
785 typename Traits::OutputType* output, Status* s) {
786 assert(batch);
787 assert(context);
788 assert(output);
789 assert(s);
790
791 std::unique_ptr<WBWIIteratorImpl> iter(
792 static_cast_with_check<WBWIIteratorImpl>(
793 batch->NewIterator(column_family)));
794
795 iter->Seek(key);
796 auto result = iter->FindLatestUpdate(key, context);
797
798 if (result == WBWIIteratorImpl::kError) {
799 Traits::ClearOutput(output);
800 *s = Status::Corruption("Unexpected entry in WriteBatchWithIndex:",
801 std::to_string(iter->Entry().type));
802 return result;
803 }
804
805 if (result == WBWIIteratorImpl::kNotFound) {
806 Traits::ClearOutput(output);
807 *s = Status::OK();
808 return result;
809 }
810
811 auto resolve_merge_outputs = [](auto out) {
812 std::string* output_value = nullptr;
813 PinnableWideColumns* output_entity = nullptr;
814
815 if constexpr (std::is_same_v<typename Traits::OutputType, std::string>) {
816 output_value = out;
817 } else {
818 static_assert(
819 std::is_same_v<typename Traits::OutputType, PinnableWideColumns>,
820 "unexpected type");
821 output_entity = out;
822 }
823
824 return std::pair<std::string*, PinnableWideColumns*>(output_value,
825 output_entity);
826 };
827
828 if (result == WBWIIteratorImpl::Result::kFound) { // Put/PutEntity
829 WriteEntry entry = iter->Entry();
830
831 if (context->GetNumOperands() > 0) {
832 auto [output_value, output_entity] = resolve_merge_outputs(output);
833
834 if (entry.type == kPutRecord) {
835 *s = MergeKeyWithBaseValue(column_family, key,
836 MergeHelper::kPlainBaseValue, entry.value,
837 *context, output_value, output_entity);
838 } else {
839 assert(entry.type == kPutEntityRecord);
840
841 *s = MergeKeyWithBaseValue(column_family, key,
842 MergeHelper::kWideBaseValue, entry.value,
843 *context, output_value, output_entity);
844 }
845 } else {
846 if (entry.type == kPutRecord) {
847 *s = Traits::SetPlainValue(entry.value, output);
848 } else {
849 assert(entry.type == kPutEntityRecord);
850 *s = Traits::SetWideColumnValue(entry.value, output);
851 }
852 }
853
854 if (!s->ok()) {
855 Traits::ClearOutput(output);
856 result = WBWIIteratorImpl::Result::kError;
857 }
858
859 return result;
860 }
861
862 if (result == WBWIIteratorImpl::kDeleted) {
863 if (context->GetNumOperands() > 0) {
864 auto [output_value, output_entity] = resolve_merge_outputs(output);
865
866 *s = MergeKeyWithNoBaseValue(column_family, key, *context, output_value,
867 output_entity);
868 if (s->ok()) {
869 result = WBWIIteratorImpl::Result::kFound;
870 } else {
871 Traits::ClearOutput(output);
872 result = WBWIIteratorImpl::Result::kError;
873 }
874 } else {
875 Traits::ClearOutput(output);
876 *s = Status::OK();
877 }
878
879 return result;
880 }
881
882 assert(result == WBWIIteratorImpl::Result::kMergeInProgress);
883
884 Traits::ClearOutput(output);
885 *s = Status::OK();
886 return result;
887}
