31std::unique_ptr<SecondaryCacheResultHandle> CompressedSecondaryCache::Lookup(
32 const Slice& key, const Cache::CacheItemHelper* helper,
33 Cache::CreateContext* create_context, bool /*wait*/, bool advise_erase,
34 Statistics* stats, bool& kept_in_sec_cache) {
35 assert(helper);
36 // This is a minor optimization. Its ok to skip it in TSAN in order to
37 // avoid a false positive.
38#ifndef __SANITIZE_THREAD__
39 if (disable_cache_) {
40 return nullptr;
41 }
42#endif
43
44 std::unique_ptr<SecondaryCacheResultHandle> handle;
45 kept_in_sec_cache = false;
46 Cache::Handle* lru_handle = cache_->Lookup(key);
47 if (lru_handle == nullptr) {
48 return nullptr;
49 }
50
51 void* handle_value = cache_->Value(lru_handle);
52 if (handle_value == nullptr) {
53 cache_->Release(lru_handle, /*erase_if_last_ref=*/false);
54 RecordTick(stats, COMPRESSED_SECONDARY_CACHE_DUMMY_HITS);
55 return nullptr;
56 }
57
58 CacheAllocationPtr* ptr{nullptr};
59 CacheAllocationPtr merged_value;
60 size_t handle_value_charge{0};
61 const char* data_ptr = nullptr;
62 CacheTier source = CacheTier::kVolatileCompressedTier;
63 CompressionType type = cache_options_.compression_type;
64 if (cache_options_.enable_custom_split_merge) {
65 CacheValueChunk* value_chunk_ptr =
66 reinterpret_cast<CacheValueChunk*>(handle_value);
67 merged_value = MergeChunksIntoValue(value_chunk_ptr, handle_value_charge);
68 ptr = &merged_value;
69 data_ptr = ptr->get();
70 } else {
71 uint32_t type_32 = static_cast<uint32_t>(type);
72 uint32_t source_32 = static_cast<uint32_t>(source);
73 ptr = reinterpret_cast<CacheAllocationPtr*>(handle_value);
74 handle_value_charge = cache_->GetCharge(lru_handle);
75 data_ptr = ptr->get();
76 data_ptr = GetVarint32Ptr(data_ptr, data_ptr + 1,
77 static_cast<uint32_t*>(&type_32));
78 type = static_cast<CompressionType>(type_32);
79 data_ptr = GetVarint32Ptr(data_ptr, data_ptr + 1,
80 static_cast<uint32_t*>(&source_32));
81 source = static_cast<CacheTier>(source_32);
82 uint64_t data_size = 0;
83 data_ptr = GetVarint64Ptr(data_ptr, ptr->get() + handle_value_charge,
84 static_cast<uint64_t*>(&data_size));
85 assert(handle_value_charge > data_size);
86 handle_value_charge = data_size;
87 }
88 MemoryAllocator* allocator = cache_options_.memory_allocator.get();
89
90 Status s;
91 Cache::ObjectPtr value{nullptr};
92 size_t charge{0};
93 if (source == CacheTier::kVolatileCompressedTier) {
94 if (cache_options_.compression_type == kNoCompression ||
95 cache_options_.do_not_compress_roles.Contains(helper->role)) {
96 s = helper->create_cb(Slice(data_ptr, handle_value_charge),
97 kNoCompression, CacheTier::kVolatileTier,
98 create_context, allocator, &value, &charge);
99 } else {
100 UncompressionContext uncompression_context(
101 cache_options_.compression_type);
102 UncompressionInfo uncompression_info(uncompression_context,
103 UncompressionDict::GetEmptyDict(),
104 cache_options_.compression_type);
105
106 size_t uncompressed_size{0};
107 CacheAllocationPtr uncompressed =
108 UncompressData(uncompression_info, (char*)data_ptr,
109 handle_value_charge, &uncompressed_size,
110 cache_options_.compress_format_version, allocator);
111
112 if (!uncompressed) {
113 cache_->Release(lru_handle, /*erase_if_last_ref=*/true);
114 return nullptr;
115 }
116 s = helper->create_cb(Slice(uncompressed.get(), uncompressed_size),
117 kNoCompression, CacheTier::kVolatileTier,
118 create_context, allocator, &value, &charge);
119 }
120 } else {
121 // The item was not compressed by us. Let the helper create_cb
122 // uncompress it
123 s = helper->create_cb(Slice(data_ptr, handle_value_charge), type, source,
124 create_context, allocator, &value, &charge);
125 }
126
127 if (!s.ok()) {
128 cache_->Release(lru_handle, /*erase_if_last_ref=*/true);
129 return nullptr;
130 }
131
132 if (advise_erase) {
133 cache_->Release(lru_handle, /*erase_if_last_ref=*/true);
134 // Insert a dummy handle.
135 cache_
136 ->Insert(key, /*obj=*/nullptr,
137 GetHelper(cache_options_.enable_custom_split_merge),
138 /*charge=*/0)
139 .PermitUncheckedError();
140 } else {
141 kept_in_sec_cache = true;
142 cache_->Release(lru_handle, /*erase_if_last_ref=*/false);
143 }
144 handle.reset(new CompressedSecondaryCacheResultHandle(value, charge));
145 RecordTick(stats, COMPRESSED_SECONDARY_CACHE_HITS);
146 return handle;
147}

149bool CompressedSecondaryCache::MaybeInsertDummy(const Slice& key) {
150 auto internal_helper = GetHelper(cache_options_.enable_custom_split_merge);
151 Cache::Handle* lru_handle = cache_->Lookup(key);
152 if (lru_handle == nullptr) {
153 PERF_COUNTER_ADD(compressed_sec_cache_insert_dummy_count, 1);
154 // Insert a dummy handle if the handle is evicted for the first time.
155 cache_->Insert(key, /*obj=*/nullptr, internal_helper, /*charge=*/0)
156 .PermitUncheckedError();
157 return true;
158 } else {
159 cache_->Release(lru_handle, /*erase_if_last_ref=*/false);
160 }
161
162 return false;
163}

165Status CompressedSecondaryCache::InsertInternal(
166 const Slice& key, Cache::ObjectPtr value,
167 const Cache::CacheItemHelper* helper, CompressionType type,
168 CacheTier source) {
169 if (source != CacheTier::kVolatileCompressedTier &&
170 cache_options_.enable_custom_split_merge) {
171 // We don't support custom split/merge for the tiered case
172 return Status::OK();
173 }
174
175 auto internal_helper = GetHelper(cache_options_.enable_custom_split_merge);
176 char header[20];
177 char* payload = header;
178 payload = EncodeVarint32(payload, static_cast<uint32_t>(type));
179 payload = EncodeVarint32(payload, static_cast<uint32_t>(source));
180 size_t data_size = (*helper->size_cb)(value);
181 char* data_size_ptr = payload;
182 payload = EncodeVarint64(payload, data_size);
183
184 size_t header_size = payload - header;
185 size_t total_size = data_size + header_size;
186 CacheAllocationPtr ptr =
187 AllocateBlock(total_size, cache_options_.memory_allocator.get());
188 char* data_ptr = ptr.get() + header_size;
189
190 Status s = (*helper->saveto_cb)(value, 0, data_size, data_ptr);
191 if (!s.ok()) {
192 return s;
193 }
194 Slice val(data_ptr, data_size);
195
196 std::string compressed_val;
197 if (cache_options_.compression_type != kNoCompression &&
198 type == kNoCompression &&
199 !cache_options_.do_not_compress_roles.Contains(helper->role)) {
200 PERF_COUNTER_ADD(compressed_sec_cache_uncompressed_bytes, data_size);
201 CompressionContext compression_context(cache_options_.compression_type,
202 cache_options_.compression_opts);
203 uint64_t sample_for_compression{0};
204 CompressionInfo compression_info(
205 cache_options_.compression_opts, compression_context,
206 CompressionDict::GetEmptyDict(), cache_options_.compression_type,
207 sample_for_compression);
208
209 bool success =
210 CompressData(val, compression_info,
211 cache_options_.compress_format_version, &compressed_val);
212
213 if (!success) {
214 return Status::Corruption("Error compressing value.");
215 }
216
217 val = Slice(compressed_val);
218 data_size = compressed_val.size();
219 payload = EncodeVarint64(data_size_ptr, data_size);
220 header_size = payload - header;
221 total_size = header_size + data_size;
222 PERF_COUNTER_ADD(compressed_sec_cache_compressed_bytes, data_size);
223
224 if (!cache_options_.enable_custom_split_merge) {
225 ptr = AllocateBlock(total_size, cache_options_.memory_allocator.get());
226 data_ptr = ptr.get() + header_size;
227 memcpy(data_ptr, compressed_val.data(), data_size);
228 }
229 }
230
231 PERF_COUNTER_ADD(compressed_sec_cache_insert_real_count, 1);
232 if (cache_options_.enable_custom_split_merge) {
233 size_t split_charge{0};
234 CacheValueChunk* value_chunks_head = SplitValueIntoChunks(
235 val, cache_options_.compression_type, split_charge);
236 return cache_->Insert(key, value_chunks_head, internal_helper,
237 split_charge);
238 } else {
239#ifdef ROCKSDB_MALLOC_USABLE_SIZE
240 size_t charge = malloc_usable_size(ptr.get());
241#else
242 size_t charge = total_size;
243#endif
244 std::memcpy(ptr.get(), header, header_size);
245 CacheAllocationPtr* buf = new CacheAllocationPtr(std::move(ptr));
246 charge += sizeof(CacheAllocationPtr);
247 return cache_->Insert(key, buf, internal_helper, charge);
248 }
249}

251Status CompressedSecondaryCache::Insert(const Slice& key,
252 Cache::ObjectPtr value,
253 const Cache::CacheItemHelper* helper,
254 bool force_insert) {
255 if (value == nullptr) {
256 return Status::InvalidArgument();
257 }
258
259 if (!force_insert && MaybeInsertDummy(key)) {
260 return Status::OK();
261 }
262
263 return InsertInternal(key, value, helper, kNoCompression,
264 CacheTier::kVolatileCompressedTier);
265}

267Status CompressedSecondaryCache::InsertSaved(
268 const Slice& key, const Slice& saved, CompressionType type = kNoCompression,
269 CacheTier source = CacheTier::kVolatileTier) {
270 if (type == kNoCompression) {
271 return Status::OK();
272 }
273
274 auto slice_helper = &kSliceCacheItemHelper;
275 if (MaybeInsertDummy(key)) {
276 return Status::OK();
277 }
278
279 return InsertInternal(
280 key, static_cast<Cache::ObjectPtr>(const_cast<Slice*>(&saved)),
281 slice_helper, type, source);
282}

320CompressedSecondaryCache::CacheValueChunk*
321CompressedSecondaryCache::SplitValueIntoChunks(const Slice& value,
322 CompressionType compression_type,
323 size_t& charge) {
324 assert(!value.empty());
325 const char* src_ptr = value.data();
326 size_t src_size{value.size()};
327
328 CacheValueChunk dummy_head = CacheValueChunk();
329 CacheValueChunk* current_chunk = &dummy_head;
330 // Do not split when value size is large or there is no compression.
331 size_t predicted_chunk_size{0};
332 size_t actual_chunk_size{0};
333 size_t tmp_size{0};
334 while (src_size > 0) {
335 predicted_chunk_size = sizeof(CacheValueChunk) - 1 + src_size;
336 auto upper =
337 std::upper_bound(malloc_bin_sizes_.begin(), malloc_bin_sizes_.end(),
338 predicted_chunk_size);
339 // Do not split when value size is too small, too large, close to a bin
340 // size, or there is no compression.
341 if (upper == malloc_bin_sizes_.begin() ||
342 upper == malloc_bin_sizes_.end() ||
343 *upper - predicted_chunk_size < malloc_bin_sizes_.front() ||
344 compression_type == kNoCompression) {
345 tmp_size = predicted_chunk_size;
346 } else {
347 tmp_size = *(--upper);
348 }
349
350 CacheValueChunk* new_chunk =
351 reinterpret_cast<CacheValueChunk*>(new char[tmp_size]);
352 current_chunk->next = new_chunk;
353 current_chunk = current_chunk->next;
354 actual_chunk_size = tmp_size - sizeof(CacheValueChunk) + 1;
355 memcpy(current_chunk->data, src_ptr, actual_chunk_size);
356 current_chunk->size = actual_chunk_size;
357 src_ptr += actual_chunk_size;
358 src_size -= actual_chunk_size;
359 charge += tmp_size;
360 }
361 current_chunk->next = nullptr;
362
363 return dummy_head.next;
364}

366CacheAllocationPtr CompressedSecondaryCache::MergeChunksIntoValue(
367 const void* chunks_head, size_t& charge) {
368 const CacheValueChunk* head =
369 reinterpret_cast<const CacheValueChunk*>(chunks_head);
370 const CacheValueChunk* current_chunk = head;
371 charge = 0;
372 while (current_chunk != nullptr) {
373 charge += current_chunk->size;
374 current_chunk = current_chunk->next;
375 }
376
377 CacheAllocationPtr ptr =
378 AllocateBlock(charge, cache_options_.memory_allocator.get());
379 current_chunk = head;
380 size_t pos{0};
381 while (current_chunk != nullptr) {
382 memcpy(ptr.get() + pos, current_chunk->data, current_chunk->size);
383 pos += current_chunk->size;
384 current_chunk = current_chunk->next;
385 }
386
387 return ptr;
388}

390const Cache::CacheItemHelper* CompressedSecondaryCache::GetHelper(
391 bool enable_custom_split_merge) const {
392 if (enable_custom_split_merge) {
393 static const Cache::CacheItemHelper kHelper{
394 CacheEntryRole::kMisc,
395 [](Cache::ObjectPtr obj, MemoryAllocator* /*alloc*/) {
396 CacheValueChunk* chunks_head = static_cast<CacheValueChunk*>(obj);
397 while (chunks_head != nullptr) {
398 CacheValueChunk* tmp_chunk = chunks_head;
399 chunks_head = chunks_head->next;
400 tmp_chunk->Free();
401 obj = nullptr;
402 }
403 }};
404 return &kHelper;
405 } else {
406 static const Cache::CacheItemHelper kHelper{
407 CacheEntryRole::kMisc,
408 [](Cache::ObjectPtr obj, MemoryAllocator* /*alloc*/) {
409 delete static_cast<CacheAllocationPtr*>(obj);
410 obj = nullptr;
411 }};
412 return &kHelper;
413 }

416size_t CompressedSecondaryCache::TEST_GetCharge(const Slice& key) {
417 Cache::Handle* lru_handle = cache_->Lookup(key);
418 if (lru_handle == nullptr) {
419 return 0;
420 }
421
422 size_t charge = cache_->GetCharge(lru_handle);
423 if (cache_->Value(lru_handle) != nullptr &&
424 !cache_options_.enable_custom_split_merge) {
425 charge -= 10;
426 }
427 cache_->Release(lru_handle, /*erase_if_last_ref=*/false);
428 return charge;
429}
