38// In Level-based compaction, it Determines which file from a level to be
39// picked to merge to the next level. We suggest people try
40// kMinOverlappingRatio first when you tune your database.
41enum CompactionPri : char {
42 // Slightly prioritize larger files by size compensated by #deletes
43 kByCompensatedSize = 0x0,
44 // First compact files whose data's latest update time is oldest.
45 // Try this if you only update some hot keys in small ranges.
46 kOldestLargestSeqFirst = 0x1,
47 // First compact files whose range hasn't been compacted to the next level
48 // for the longest. If your updates are random across the key space,
49 // write amplification is slightly better with this option.
50 kOldestSmallestSeqFirst = 0x2,
51 // First compact files whose ratio between overlapping size in next level
52 // and its size is the smallest. It in many cases can optimize write
53 // amplification.
54 // Files marked for compaction will be prioritized over files that are not
55 // marked.
56 kMinOverlappingRatio = 0x3,
57 // Keeps a cursor(s) of the successor of the file (key range) was/were
58 // compacted before, and always picks the next files (key range) in that
59 // level. The file picking process will cycle through all the files in a
60 // round-robin manner.
61 kRoundRobin = 0x4,
62};

75 // If true, try to do compaction to compact smaller files into larger ones.
76 // Minimum files to compact follows options.level0_file_num_compaction_trigger
77 // and compaction won't trigger if average compact bytes per del file is
78 // larger than options.write_buffer_size. This is to protect large files
79 // from being compacted again.
80 // Default: false;
81 bool allow_compaction = false;
82
83 // DEPRECATED
84 // When not 0, if the data in the file is older than this threshold, RocksDB
85 // will soon move the file to warm temperature.
86 uint64_t age_for_warm = 0;
87
88 // EXPERIMENTAL
89 // Age (in seconds) threshold for different file temperatures.
90 // When not empty, each element specifies an age threshold `age` and a
91 // temperature such that if all the data in a file is older than `age`,
92 // RocksDB will compact the file to the specified `temperature`. Oldest file
93 // will be considered first. Only one file is compacted at a time,
94 // so multiple files qualifying to be compacted to be same temperature
95 // won't be merged together.
96 //
97 // Note:
98 // - Flushed files will always have temperature kUnknown.
99 // - Compaction output files will have temperature kUnknown by default, so
100 // only temperatures other than kUnknown needs to be specified.
101 // - The elements should be in increasing order with respect to `age` field.
102 //
103 // Dynamically changeable through SetOptions() API, e.g.,
104 // SetOptions("compaction_options_fifo",
105 // "{file_temperature_age_thresholds={
106 // {age=10;temperature=kWarm}:{age=20;temperature=kCold}}}")
107 // In this example, all files that are at least 20 seconds old will be
108 // compacted and output files will have temperature kCold. All files that are
109 // at least 10 seconds old but younger than 20 seconds will be compacted to
110 // files with temperature kWarm.
111 //
112 // Default: empty
113 std::vector<FileTemperatureAge> file_temperature_age_thresholds{};
114
115 CompactionOptionsFIFO() : max_table_files_size(1 * 1024 * 1024 * 1024) {}
116 CompactionOptionsFIFO(uint64_t _max_table_files_size, bool _allow_compaction)
117 : max_table_files_size(_max_table_files_size),
118 allow_compaction(_allow_compaction) {}

141struct AdvancedColumnFamilyOptions {
142 // The maximum number of write buffers that are built up in memory.
143 // The default and the minimum number is 2, so that when 1 write buffer
144 // is being flushed to storage, new writes can continue to the other
145 // write buffer.
146 // If max_write_buffer_number > 3, writing will be slowed down to
147 // options.delayed_write_rate if we are writing to the last write buffer
148 // allowed.
149 //
150 // Default: 2
151 //
152 // Dynamically changeable through SetOptions() API
153 int max_write_buffer_number = 2;
154
155 // The minimum number of write buffers that will be merged together
156 // before writing to storage. If set to 1, then
157 // all write buffers are flushed to L0 as individual files and this increases
158 // read amplification because a get request has to check in all of these
159 // files. Also, an in-memory merge may result in writing lesser
160 // data to storage if there are duplicate records in each of these
161 // individual write buffers.
162 // If atomic flush is enabled (options.atomic_flush == true), then this
163 // option will be sanitized to 1.
164 // Default: 1
165 int min_write_buffer_number_to_merge = 1;
166
167 // DEPRECATED
168 // The total maximum number of write buffers to maintain in memory including
169 // copies of buffers that have already been flushed. Unlike
170 // max_write_buffer_number, this parameter does not affect flushing.
171 // This parameter is being replaced by max_write_buffer_size_to_maintain.
172 // If both parameters are set to non-zero values, this parameter will be
173 // ignored.
174 int max_write_buffer_number_to_maintain = 0;
175
176 // The target number of write history bytes to hold in memory. Write history
177 // comprises the latest write buffers (memtables). To reach the target, write
178 // buffers that were most recently flushed to SST files may be retained in
179 // memory.
180 //
181 // This controls the target amount of write history that will be available
182 // in memory for conflict checking when Transactions are used.
183 //
184 // This target may be undershot when the CF first opens and has not recovered
185 // or received enough writes to reach the target. After reaching the target
186 // once, it is guaranteed to never undershoot again. That guarantee is
187 // implemented by retaining flushed write buffers in-memory until the oldest
188 // one can be trimmed without dropping below the target.
189 //
190 // Examples with `max_write_buffer_size_to_maintain` set to 32MB:
191 //
192 // - One mutable memtable of 64MB, one unflushed immutable memtable of 64MB,
193 // and zero flushed immutable memtables. Nothing trimmable exists.
194 // - One mutable memtable of 16MB, zero unflushed immutable memtables, and
195 // one flushed immutable memtable of 64MB. Trimming is disallowed because
196 // dropping the earliest (only) flushed immutable memtable would result in
197 // write history of 16MB < 32MB.
198 // - One mutable memtable of 24MB, one unflushed immutable memtable of 16MB,
199 // and one flushed immutable memtable of 16MB. The earliest (only) flushed
200 // immutable memtable is trimmed because without it we still have
201 // 16MB + 24MB = 40MB > 32MB of write history.
202 //
203 // When using an OptimisticTransactionDB:
204 // If this value is too low, some transactions may fail at commit time due
205 // to not being able to determine whether there were any write conflicts.
206 //
207 // When using a TransactionDB:
208 // If Transaction::SetSnapshot is used, TransactionDB will read either
209 // in-memory write buffers or SST files to do write-conflict checking.
210 // Increasing this value can reduce the number of reads to SST files
211 // done for conflict detection.
212 //
213 // Setting this value to 0 will cause write buffers to be freed immediately
214 // after they are flushed. If this value is set to -1,
215 // 'max_write_buffer_number * write_buffer_size' will be used.
216 //
217 // Default:
218 // If using a TransactionDB/OptimisticTransactionDB, the default value will
219 // be set to the value of 'max_write_buffer_number * write_buffer_size'
220 // if it is not explicitly set by the user. Otherwise, the default is 0.
221 int64_t max_write_buffer_size_to_maintain = 0;

223 // Allows thread-safe inplace updates.
224 //
225 // If this is true, there is no way to
226 // achieve point-in-time consistency using snapshot or iterator (assuming
227 // concurrent updates). Hence iterator and multi-get will return results
228 // which are not consistent as of any point-in-time.
229 //
230 // Backward iteration on memtables will not work either.
231 //
232 // It is intended to work or be compatible with a limited set of features:
233 // (1) Non-snapshot Get()
234 //
235 // If inplace_callback function is not set,
236 // Put(key, new_value) will update inplace the existing_value iff
237 // * key exists in current memtable
238 // * new sizeof(new_value) <= sizeof(existing_value)
239 // * existing_value for that key is a put i.e. kTypeValue
240 // If inplace_callback function is set, check doc for inplace_callback.
241 // Default: false.
242 bool inplace_update_support = false;
243
244 // Number of locks used for inplace update
245 // Default: 10000, if inplace_update_support = true, else 0.
246 //
247 // Dynamically changeable through SetOptions() API
248 size_t inplace_update_num_locks = 10000;

250 // [experimental]
251 // Used to activate or deactive the Mempurge feature (memtable garbage
252 // collection). (deactivated by default). At every flush, the total useful
253 // payload (total entries minus garbage entries) is estimated as a ratio
254 // [useful payload bytes]/[size of a memtable (in bytes)]. This ratio is then
255 // compared to this `threshold` value:
256 // - if ratio<threshold: the flush is replaced by a mempurge operation
257 // - else: a regular flush operation takes place.
258 // Threshold values:
259 // 0.0: mempurge deactivated (default).
260 // 1.0: recommended threshold value.
261 // >1.0 : aggressive mempurge.
262 // 0 < threshold < 1.0: mempurge triggered only for very low useful payload
263 // ratios.
264 // [experimental]
265 double experimental_mempurge_threshold = 0.0;

267 // existing_value - pointer to previous value (from both memtable and sst).
268 // nullptr if key doesn't exist
269 // existing_value_size - pointer to size of existing_value).
270 // nullptr if key doesn't exist
271 // delta_value - Delta value to be merged with the existing_value.
272 // Stored in transaction logs.
273 // merged_value - Set when delta is applied on the previous value.
274 //
275 // Applicable only when inplace_update_support is true,
276 // this callback function is called at the time of updating the memtable
277 // as part of a Put operation, lets say Put(key, delta_value). It allows the
278 // 'delta_value' specified as part of the Put operation to be merged with
279 // an 'existing_value' of the key in the database.
280 //
281 // If the merged value is smaller in size that the 'existing_value',
282 // then this function can update the 'existing_value' buffer inplace and
283 // the corresponding 'existing_value'_size pointer, if it wishes to.
284 // The callback should return UpdateStatus::UPDATED_INPLACE.
285 // In this case. (In this case, the snapshot-semantics of the rocksdb
286 // Iterator is not atomic anymore).
287 //
288 // If the merged value is larger in size than the 'existing_value' or the
289 // application does not wish to modify the 'existing_value' buffer inplace,
290 // then the merged value should be returned via *merge_value. It is set by
291 // merging the 'existing_value' and the Put 'delta_value'. The callback should
292 // return UpdateStatus::UPDATED in this case. This merged value will be added
293 // to the memtable.
294 //
295 // If merging fails or the application does not wish to take any action,
296 // then the callback should return UpdateStatus::UPDATE_FAILED.
297 //
298 // Please remember that the original call from the application is Put(key,
299 // delta_value). So the transaction log (if enabled) will still contain (key,
300 // delta_value). The 'merged_value' is not stored in the transaction log.
301 // Hence the inplace_callback function should be consistent across db reopens.
302 //
303 // RocksDB callbacks are NOT exception-safe. A callback completing with an
304 // exception can lead to undefined behavior in RocksDB, including data loss,
305 // unreported corruption, deadlocks, and more.
306 //
307 // Default: nullptr
308 UpdateStatus (*inplace_callback)(char* existing_value,
309 uint32_t* existing_value_size,
310 Slice delta_value,
311 std::string* merged_value) = nullptr;

397 // Different levels can have different compression policies. There
398 // are cases where most lower levels would like to use quick compression
399 // algorithms while the higher levels (which have more data) use
400 // compression algorithms that have better compression but could
401 // be slower. This array, if non-empty, should have an entry for
402 // each level of the database; these override the value specified in
403 // the previous field 'compression'.
404 //
405 // NOTICE if level_compaction_dynamic_level_bytes=true,
406 // compression_per_level[0] still determines L0, but other elements
407 // of the array are based on base level (the level L0 files are merged
408 // to), and may not match the level users see from info log for metadata.
409 // If L0 files are merged to level-n, then, for i>0, compression_per_level[i]
410 // determines compaction type for level n+i-1.
411 // For example, if we have three 5 levels, and we determine to merge L0
412 // data to L4 (which means L1..L3 will be empty), then the new files go to
413 // L4 uses compression type compression_per_level[1].
414 // If now L0 is merged to L2. Data goes to L2 will be compressed
415 // according to compression_per_level[1], L3 using compression_per_level[2]
416 // and L4 using compression_per_level[3]. Compaction for each level can
417 // change when data grows.
418 //
419 // NOTE: if the vector size is smaller than the level number, the undefined
420 // lower level uses the last option in the vector, for example, for 3 level
421 // LSM tree the following settings are the same:
422 // {kNoCompression, kSnappyCompression}
423 // {kNoCompression, kSnappyCompression, kSnappyCompression}
424 //
425 // Dynamically changeable through SetOptions() API
426 std::vector<CompressionType> compression_per_level;

447 // Target file size for compaction.
448 // target_file_size_base is per-file size for level-1.
449 // Target file size for level L can be calculated by
450 // target_file_size_base * (target_file_size_multiplier ^ (L-1))
451 // For example, if target_file_size_base is 2MB and
452 // target_file_size_multiplier is 10, then each file on level-1 will
453 // be 2MB, and each file on level 2 will be 20MB,
454 // and each file on level-3 will be 200MB.

456 // Default: 64MB.

462 // by default files in different levels will have similar size.

464 // Dynamically changeable through SetOptions() API
465 int target_file_size_multiplier = 1;

467 // If true, RocksDB will pick target size of each level dynamically.
468 // We will pick a base level b >= 1. L0 will be directly merged into level b,
469 // instead of always into level 1. Level 1 to b-1 need to be empty.
470 // We try to pick b and its target size so that
471 // 1. target size is in the range of
472 // (max_bytes_for_level_base / max_bytes_for_level_multiplier,
473 // max_bytes_for_level_base]
474 // 2. target size of the last level (level num_levels-1) equals to the max
475 // size of a level in the LSM (typically the last level).
476 // At the same time max_bytes_for_level_multiplier is still satisfied.
477 // Note that max_bytes_for_level_multiplier_additional is ignored with this
478 // flag on.
479 //
480 // With this option on, from an empty DB, we make last level the base level,
481 // which means merging L0 data into the last level, until it exceeds
482 // max_bytes_for_level_base. And then we make the second last level to be
483 // base level, to start to merge L0 data to second last level, with its
484 // target size to be 1/max_bytes_for_level_multiplier of the last level's
485 // extra size. After the data accumulates more so that we need to move the
486 // base level to the third last one, and so on.
487 //
488 // For example, assume max_bytes_for_level_multiplier=10, num_levels=6,
489 // and max_bytes_for_level_base=10MB.
490 // Target sizes of level 1 to 5 starts with:
491 // [- - - - 10MB]
492 // with base level is level 5. Target sizes of level 1 to 4 are not applicable
493 // because they will not be used.
494 // Until the size of Level 5 grows to more than 10MB, say 11MB, we make
495 // base target to level 4 and now the targets looks like:
496 // [- - - 1.1MB 11MB]
497 // While data are accumulated, size targets are tuned based on actual data
498 // of level 5. When level 5 has 50MB of data, the target is like:
499 // [- - - 5MB 50MB]
500 // Until level 5's actual size is more than 100MB, say 101MB. Now if we keep
501 // level 4 to be the base level, its target size needs to be 10.1MB, which
502 // doesn't satisfy the target size range. So now we make level 3 the target
503 // size and the target sizes of the levels look like:
504 // [- - 1.01MB 10.1MB 101MB]
505 // In the same way, while level 5 further grows, all levels' targets grow,
506 // like
507 // [- - 5MB 50MB 500MB]
508 // Until level 5 exceeds 1000MB and becomes 1001MB, we make level 2 the
509 // base level and make levels' target sizes like this:
510 // [- 1.001MB 10.01MB 100.1MB 1001MB]
511 // and go on...
512 //
513 // By doing it, we give max_bytes_for_level_multiplier a priority against
514 // max_bytes_for_level_base, for a more predictable LSM tree shape. It is
515 // useful to limit worse case space amplification.
516 // If `allow_ingest_behind=true` or `preclude_last_level_data_seconds > 0`,
517 // then the last level is reserved, and we will start filling LSM from the
518 // second last level.
519 //
520 // With this option on, compaction is more adaptive to write traffic:
521 // Compaction priority will take into account estimated bytes to be compacted
522 // down to a level and favors compacting lower levels when there is a write
523 // traffic spike (and hence more compaction debt). Refer to
524 // https://github.com/facebook/rocksdb/wiki/Leveled-Compactio#option-level_compaction_dynamic_level_bytes-and-levels-target-size
525 // for more detailed description. See more implementation detail in:
526 // VersionStorageInfo::ComputeCompactionScore().
527 //
528 // With this option on, unneeded levels will be drained automatically:
529 // Note that there may be excessive levels (where target level size is 0 when
530 // computed based on this feature) in the LSM. This can happen after a user
531 // migrates to turn this feature on or deletes a lot of data. This is
532 // especially likely when a user migrates from leveled compaction with a
533 // smaller multiplier or from universal compaction. RocksDB will gradually
534 // drain these unnecessary levels by compacting files down the LSM. Smaller
535 // number of levels should help to reduce read amplification.
536 //
537 // Migration to turn on this option:
538 // - Before RocksDB v8.2, users are expected to do a full manual compaction
539 // and then restart DB to turn on this option.
540 // - Since RocksDB v8.2, users can just restart DB with this option on, as
541 // long as num_levels is no smaller than number of non-empty levels in the
542 // LSM. Migration will be done automatically by RocksDB. See more in
543 // https://github.com/facebook/rocksdb/wiki/Leveled-Compaction#migrating-from-level_compaction_dynamic_level_bytesfalse-to-level_compaction_dynamic_level_bytestrue
544 //
545 // Default: true
546 bool level_compaction_dynamic_level_bytes = true;
547
548 // Default: 10.
549 //
550 // Dynamically changeable through SetOptions() API
551 double max_bytes_for_level_multiplier = 10;
552
553 // Different max-size multipliers for different levels.
554 // These are multiplied by max_bytes_for_level_multiplier to arrive
555 // at the max-size of each level.
556 // This option only applies to leveled compaction with
557 // `level_compaction_dynamic_level_bytes = false`.
558 //
559 // Default: 1
560 //
561 // Dynamically changeable through SetOptions() API
562 std::vector<int> max_bytes_for_level_multiplier_additional =
563 std::vector<int>(static_cast<size_t>(num_levels), 1);

565 // We try to limit number of bytes in one compaction to be lower than this
566 // threshold. But it's not guaranteed.
567 // Value 0 will be sanitized.
568 //
569 // Default: target_file_size_base * 25
570 //

572 uint64_t max_compaction_bytes = 0;
573
574 // All writes will be slowed down to at least delayed_write_rate if estimated
575 // bytes needed to be compaction exceed this threshold.
576 //
577 // Default: 64GB
578 //
579 // Dynamically changeable through SetOptions() API
