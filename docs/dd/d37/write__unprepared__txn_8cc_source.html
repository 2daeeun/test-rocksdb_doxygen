16bool WriteUnpreparedTxnReadCallback::IsVisibleFullCheck(SequenceNumber seq) {
17 // Since unprep_seqs maps prep_seq => prepare_batch_cnt, to check if seq is
18 // in unprep_seqs, we have to check if seq is equal to prep_seq or any of
19 // the prepare_batch_cnt seq nums after it.
20 //
21 // TODO(lth): Can be optimized with std::lower_bound if unprep_seqs is
22 // large.
23 for (const auto& it : unprep_seqs_) {
24 if (it.first <= seq && seq < it.first + it.second) {
25 return true;
26 }
27 }
28
29 bool snap_released = false;
30 auto ret =
31 db_->IsInSnapshot(seq, wup_snapshot_, min_uncommitted_, &snap_released);
32 assert(!snap_released || backed_by_snapshot_ == kUnbackedByDBSnapshot);
33 snap_released_ |= snap_released;
34 return ret;
35}

53WriteUnpreparedTxn::~WriteUnpreparedTxn() {
54 if (!unprep_seqs_.empty()) {
55 assert(log_number_ > 0);
56 assert(GetId() > 0);
57 assert(!name_.empty());
58
59 // We should rollback regardless of GetState, but some unit tests that
60 // test crash recovery run the destructor assuming that rollback does not
61 // happen, so that rollback during recovery can be exercised.
62 if (GetState() == STARTED || GetState() == LOCKS_STOLEN) {
63 auto s = RollbackInternal();
64 assert(s.ok());
65 if (!s.ok()) {
66 ROCKS_LOG_FATAL(
67 wupt_db_->info_log_,
68 "Rollback of WriteUnprepared transaction failed in destructor: %s",
69 s.ToString().c_str());
70 }
71 dbimpl_->logs_with_prep_tracker()->MarkLogAsHavingPrepSectionFlushed(
72 log_number_);
73 }
74 }
75
76 // Clear the tracked locks so that ~PessimisticTransaction does not
77 // try to unlock keys for recovered transactions.
78 if (recovered_txn_) {
79 tracked_locks_->Clear();
80 }

102Status WriteUnpreparedTxn::HandleWrite(std::function<Status()> do_write) {
103 Status s;
104 if (active_iterators_.empty()) {
105 s = MaybeFlushWriteBatchToDB();
106 if (!s.ok()) {
107 return s;
108 }
109 }
110 s = do_write();
111 if (s.ok()) {
112 if (snapshot_) {
113 largest_validated_seq_ =
114 std::max(largest_validated_seq_, snapshot_->GetSequenceNumber());
115 } else {
116 // TODO(lth): We should use the same number as tracked_at_seq in TryLock,
117 // because what is actually being tracked is the sequence number at which
118 // this key was locked at.
119 largest_validated_seq_ = db_impl_->GetLastPublishedSequence();
120 }
121 }
122 return s;
123}

183// WriteUnpreparedTxn::RebuildFromWriteBatch is only called on recovery. For
184// WriteUnprepared, the write batches have already been written into the
185// database during WAL replay, so all we have to do is just to "retrack" the key
186// so that rollbacks are possible.
187//
188// Calling TryLock instead of TrackKey is also possible, but as an optimization,
189// recovered transactions do not hold locks on their keys. This follows the
190// implementation in PessimisticTransactionDB::Initialize where we set
191// skip_concurrency_control to true.
192Status WriteUnpreparedTxn::RebuildFromWriteBatch(WriteBatch* wb) {
193 struct TrackKeyHandler : public WriteBatch::Handler {
194 WriteUnpreparedTxn* txn_;
195 bool rollback_merge_operands_;
196
197 TrackKeyHandler(WriteUnpreparedTxn* txn, bool rollback_merge_operands)
198 : txn_(txn), rollback_merge_operands_(rollback_merge_operands) {}
199
200 Status PutCF(uint32_t cf, const Slice& key, const Slice&) override {
201 txn_->TrackKey(cf, key.ToString(), kMaxSequenceNumber,
202 false /* read_only */, true /* exclusive */);
203 return Status::OK();
204 }
205
206 Status DeleteCF(uint32_t cf, const Slice& key) override {
207 txn_->TrackKey(cf, key.ToString(), kMaxSequenceNumber,
208 false /* read_only */, true /* exclusive */);
209 return Status::OK();
210 }
211
212 Status SingleDeleteCF(uint32_t cf, const Slice& key) override {
213 txn_->TrackKey(cf, key.ToString(), kMaxSequenceNumber,
214 false /* read_only */, true /* exclusive */);
215 return Status::OK();
216 }
217
218 Status MergeCF(uint32_t cf, const Slice& key, const Slice&) override {
219 if (rollback_merge_operands_) {
220 txn_->TrackKey(cf, key.ToString(), kMaxSequenceNumber,
221 false /* read_only */, true /* exclusive */);
222 }
223 return Status::OK();
224 }
225
226 // Recovered batches do not contain 2PC markers.
227 Status MarkBeginPrepare(bool) override { return Status::InvalidArgument(); }
228
229 Status MarkEndPrepare(const Slice&) override {
230 return Status::InvalidArgument();
231 }
232
233 Status MarkNoop(bool) override { return Status::InvalidArgument(); }
234
235 Status MarkCommit(const Slice&) override {
236 return Status::InvalidArgument();
237 }
238
239 Status MarkRollback(const Slice&) override {
240 return Status::InvalidArgument();
241 }
242 };
243
244 TrackKeyHandler handler(this,
245 wupt_db_->txn_db_options_.rollback_merge_operands);
246 return wb->Iterate(&handler);
247}

354 // TODO(lth): Reduce duplicate code with WritePrepared prepare logic.
355 WriteOptions write_options = write_options_;
356 write_options.disableWAL = false;
357 const bool WRITE_AFTER_COMMIT = true;
358 const bool first_prepare_batch = log_number_ == 0;
359 // MarkEndPrepare will change Noop marker to the appropriate marker.
360 s = WriteBatchInternal::MarkEndPrepare(GetWriteBatch()->GetWriteBatch(),
361 name_, !WRITE_AFTER_COMMIT, !prepared);
362 assert(s.ok());
363 // For each duplicate key we account for a new sub-batch
364 prepare_batch_cnt_ = GetWriteBatch()->SubBatchCnt();
365 // AddPrepared better to be called in the pre-release callback otherwise there
366 // is a non-zero chance of max advancing prepare_seq and readers assume the
367 // data as committed.
368 // Also having it in the PreReleaseCallback allows in-order addition of
369 // prepared entries to PreparedHeap and hence enables an optimization. Refer
370 // to SmallestUnCommittedSeq for more details.
371 AddPreparedCallback add_prepared_callback(
372 wpt_db_, db_impl_, prepare_batch_cnt_,
373 db_impl_->immutable_db_options().two_write_queues, first_prepare_batch);
374 const bool DISABLE_MEMTABLE = true;
375 uint64_t seq_used = kMaxSequenceNumber;
376 // log_number_ should refer to the oldest log containing uncommitted data
377 // from the current transaction. This means that if log_number_ is set,
378 // WriteImpl should not overwrite that value, so set log_used to nullptr if
379 // log_number_ is already set.
380 s = db_impl_->WriteImpl(write_options, GetWriteBatch()->GetWriteBatch(),
381 /*callback*/ nullptr, /*user_write_cb=*/nullptr,
382 &last_log_number_,
383 /*log ref*/ 0, !DISABLE_MEMTABLE, &seq_used,
384 prepare_batch_cnt_, &add_prepared_callback);
385 if (log_number_ == 0) {
386 log_number_ = last_log_number_;
387 }
388 assert(!s.ok() || seq_used != kMaxSequenceNumber);
389 auto prepare_seq = seq_used;
390
391 // Only call SetId if it hasn't been set yet.
392 if (GetId() == 0) {
393 SetId(prepare_seq);
394 }
395 // unprep_seqs_ will also contain prepared seqnos since they are treated in
396 // the same way in the prepare/commit callbacks. See the comment on the
397 // definition of unprep_seqs_.
398 if (s.ok()) {
399 unprep_seqs_[prepare_seq] = prepare_batch_cnt_;
400 }
401
402 // Reset transaction state.
403 if (!prepared) {
404 prepare_batch_cnt_ = 0;
405 const bool kClear = true;
406 TransactionBaseImpl::InitWriteBatch(kClear);
407 }
408
409 return s;
410}

418 // Handler class for creating an unprepared batch from a savepoint.
419 struct SavePointBatchHandler : public WriteBatch::Handler {
420 WriteBatchWithIndex* wb_;
421 const std::map<uint32_t, ColumnFamilyHandle*>& handles_;
422
423 SavePointBatchHandler(
424 WriteBatchWithIndex* wb,
425 const std::map<uint32_t, ColumnFamilyHandle*>& handles)
426 : wb_(wb), handles_(handles) {}

428 Status PutCF(uint32_t cf, const Slice& key, const Slice& value) override {
429 return wb_->Put(handles_.at(cf), key, value);

433 return wb_->Delete(handles_.at(cf), key);
434 }

436 Status SingleDeleteCF(uint32_t cf, const Slice& key) override {
437 return wb_->SingleDelete(handles_.at(cf), key);
438 }
439
440 Status MergeCF(uint32_t cf, const Slice& key, const Slice& value) override {
441 return wb_->Merge(handles_.at(cf), key, value);
442 }
443
444 // The only expected 2PC marker is the initial Noop marker.
445 Status MarkNoop(bool empty_batch) override {
446 return empty_batch ? Status::OK() : Status::InvalidArgument();
447 }
448
449 Status MarkBeginPrepare(bool) override { return Status::InvalidArgument(); }
450
451 Status MarkEndPrepare(const Slice&) override {
452 return Status::InvalidArgument();
453 }
454
455 Status MarkCommit(const Slice&) override {
456 return Status::InvalidArgument();
457 }
458
459 Status MarkRollback(const Slice&) override {
460 return Status::InvalidArgument();
461 }
462 };
463
464 // The comparator of the default cf is passed in, similar to the
465 // initialization of TransactionBaseImpl::write_batch_. This comparator is
466 // only used if the write batch encounters an invalid cf id, and falls back to
467 // this comparator.
468 WriteBatchWithIndex wb(wpt_db_->DefaultColumnFamily()->GetComparator(), 0,
469 true, 0, write_options_.protection_bytes_per_key);
470 // Swap with write_batch_ so that wb contains the complete write batch. The
471 // actual write batch that will be flushed to DB will be built in
472 // write_batch_, and will be read by FlushWriteBatchToDBInternal.
473 std::swap(wb, write_batch_);
474 TransactionBaseImpl::InitWriteBatch();
475
476 size_t prev_boundary = WriteBatchInternal::kHeader;
477 const bool kPrepared = true;
478 for (size_t i = 0; i < unflushed_save_points_->size() + 1; i++) {
479 bool trailing_batch = i == unflushed_save_points_->size();
480 SavePointBatchHandler sp_handler(&write_batch_,
481 *wupt_db_->GetCFHandleMap().get());
482 size_t curr_boundary = trailing_batch ? wb.GetWriteBatch()->GetDataSize()
483 : (*unflushed_save_points_)[i];
484
485 // Construct the partial write batch up to the savepoint.
486 //
487 // Theoretically, a memcpy between the write batches should be sufficient
488 // since the rewriting into the batch should produce the exact same byte
489 // representation. Rebuilding the WriteBatchWithIndex index is still
490 // necessary though, and would imply doing two passes over the batch though.
491 Status s = WriteBatchInternal::Iterate(wb.GetWriteBatch(), &sp_handler,
492 prev_boundary, curr_boundary);
493 if (!s.ok()) {
494 return s;
495 }
496
497 if (write_batch_.GetWriteBatch()->Count() > 0) {
498 // Flush the write batch.
499 s = FlushWriteBatchToDBInternal(!kPrepared);
500 if (!s.ok()) {
501 return s;
502 }
503 }
504
505 if (!trailing_batch) {
506 if (flushed_save_points_ == nullptr) {
507 flushed_save_points_.reset(
508 new autovector<WriteUnpreparedTxn::SavePoint>());
509 }
510 flushed_save_points_->emplace_back(
511 unprep_seqs_, new ManagedSnapshot(db_impl_, wupt_db_->GetSnapshot()));
512 }
513
514 prev_boundary = curr_boundary;
515 const bool kClear = true;
516 TransactionBaseImpl::InitWriteBatch(kClear);
517 }
518
519 unflushed_save_points_->clear();
520 return Status::OK();
521}

544Status WriteUnpreparedTxn::CommitInternal() {
545 // TODO(lth): Reduce duplicate code with WritePrepared commit logic.
546
547 // We take the commit-time batch and append the Commit marker. The Memtable
548 // will ignore the Commit marker in non-recovery mode
549 WriteBatch* working_batch = GetCommitTimeWriteBatch();
550 const bool empty = working_batch->Count() == 0;
551 auto s = WriteBatchInternal::MarkCommit(working_batch, name_);
552 assert(s.ok());
553
554 const bool for_recovery = use_only_the_last_commit_time_batch_for_recovery_;
555 if (!empty) {
556 // When not writing to memtable, we can still cache the latest write batch.
557 // The cached batch will be written to memtable in WriteRecoverableState
558 // during FlushMemTable
559 if (for_recovery) {
560 WriteBatchInternal::SetAsLatestPersistentState(working_batch);
561 } else {
562 return Status::InvalidArgument(
563 "Commit-time-batch can only be used if "
564 "use_only_the_last_commit_time_batch_for_recovery is true");
565 }
566 }
567
568 const bool includes_data = !empty && !for_recovery;
569 size_t commit_batch_cnt = 0;
570 if (UNLIKELY(includes_data)) {
571 ROCKS_LOG_WARN(db_impl_->immutable_db_options().info_log,
572 "Duplicate key overhead");
573 SubBatchCounter counter(*wpt_db_->GetCFComparatorMap());
574 s = working_batch->Iterate(&counter);
575 assert(s.ok());
576 commit_batch_cnt = counter.BatchCount();
577 }
578 const bool disable_memtable = !includes_data;
579 const bool do_one_write =
580 !db_impl_->immutable_db_options().two_write_queues || disable_memtable;
581
582 WriteUnpreparedCommitEntryPreReleaseCallback update_commit_map(
583 wpt_db_, db_impl_, unprep_seqs_, commit_batch_cnt);
584 const bool kFirstPrepareBatch = true;
585 AddPreparedCallback add_prepared_callback(
586 wpt_db_, db_impl_, commit_batch_cnt,
587 db_impl_->immutable_db_options().two_write_queues, !kFirstPrepareBatch);
588 PreReleaseCallback* pre_release_callback;
589 if (do_one_write) {
590 pre_release_callback = &update_commit_map;
591 } else {
592 pre_release_callback = &add_prepared_callback;
593 }
594 uint64_t seq_used = kMaxSequenceNumber;
595 // Since the prepared batch is directly written to memtable, there is
596 // already a connection between the memtable and its WAL, so there is no
597 // need to redundantly reference the log that contains the prepared data.
598 const uint64_t zero_log_number = 0ull;
599 size_t batch_cnt = UNLIKELY(commit_batch_cnt) ? commit_batch_cnt : 1;
600 s = db_impl_->WriteImpl(write_options_, working_batch, nullptr, nullptr,
601 nullptr, zero_log_number, disable_memtable, &seq_used,
602 batch_cnt, pre_release_callback);
603 assert(!s.ok() || seq_used != kMaxSequenceNumber);
604 const SequenceNumber commit_batch_seq = seq_used;
605 if (LIKELY(do_one_write || !s.ok())) {
606 if (LIKELY(s.ok())) {
607 // Note RemovePrepared should be called after WriteImpl that publishsed
608 // the seq. Otherwise SmallestUnCommittedSeq optimization breaks.
609 for (const auto& seq : unprep_seqs_) {
610 wpt_db_->RemovePrepared(seq.first, seq.second);
611 }
612 }
613 if (UNLIKELY(!do_one_write)) {
614 wpt_db_->RemovePrepared(commit_batch_seq, commit_batch_cnt);
615 }
616 unprep_seqs_.clear();
617 flushed_save_points_.reset(nullptr);
618 unflushed_save_points_.reset(nullptr);
619 return s;
620 } // else do the 2nd write to publish seq
621
622 // Populate unprep_seqs_ with commit_batch_seq, since we treat data in the
623 // commit write batch as just another "unprepared" batch. This will also
624 // update the unprep_seqs_ in the update_commit_map callback.
625 unprep_seqs_[commit_batch_seq] = commit_batch_cnt;
626 WriteUnpreparedCommitEntryPreReleaseCallback
627 update_commit_map_with_commit_batch(wpt_db_, db_impl_, unprep_seqs_, 0);
628
629 // Note: the 2nd write comes with a performance penality. So if we have too
630 // many of commits accompanied with ComitTimeWriteBatch and yet we cannot
631 // enable use_only_the_last_commit_time_batch_for_recovery_ optimization,
632 // two_write_queues should be disabled to avoid many additional writes here.
633
634 // Update commit map only from the 2nd queue
635 WriteBatch empty_batch;
636 s = empty_batch.PutLogData(Slice());
637 assert(s.ok());
638 // In the absence of Prepare markers, use Noop as a batch separator
639 s = WriteBatchInternal::InsertNoop(&empty_batch);
640 assert(s.ok());
641 const bool DISABLE_MEMTABLE = true;
642 const size_t ONE_BATCH = 1;
643 const uint64_t NO_REF_LOG = 0;
644 s = db_impl_->WriteImpl(write_options_, &empty_batch, nullptr, nullptr,
645 nullptr, NO_REF_LOG, DISABLE_MEMTABLE, &seq_used,
646 ONE_BATCH, &update_commit_map_with_commit_batch);
647 assert(!s.ok() || seq_used != kMaxSequenceNumber);
648 // Note RemovePrepared should be called after WriteImpl that publishsed the
649 // seq. Otherwise SmallestUnCommittedSeq optimization breaks.
650 for (const auto& seq : unprep_seqs_) {
651 wpt_db_->RemovePrepared(seq.first, seq.second);
652 }

659Status WriteUnpreparedTxn::WriteRollbackKeys(
660 const LockTracker& lock_tracker, WriteBatchWithIndex* rollback_batch,
661 ReadCallback* callback, const ReadOptions& roptions) {
662 // This assertion can be removed when range lock is supported.
663 assert(lock_tracker.IsPointLockSupported());
664 const auto& cf_map = *wupt_db_->GetCFHandleMap();
665 auto WriteRollbackKey = [&](const std::string& key, uint32_t cfid) {
666 const auto& cf_handle = cf_map.at(cfid);
667 PinnableSlice pinnable_val;
668 bool not_used;
669 DBImpl::GetImplOptions get_impl_options;
670 get_impl_options.column_family = cf_handle;
671 get_impl_options.value = &pinnable_val;
672 get_impl_options.value_found = &not_used;
673 get_impl_options.callback = callback;
674 auto s = db_impl_->GetImpl(roptions, key, get_impl_options);
675
676 if (s.ok()) {
677 s = rollback_batch->Put(cf_handle, key, pinnable_val);
678 assert(s.ok());
679 } else if (s.IsNotFound()) {
680 if (wupt_db_->ShouldRollbackWithSingleDelete(cf_handle, key)) {
681 s = rollback_batch->SingleDelete(cf_handle, key);
682 } else {
683 s = rollback_batch->Delete(cf_handle, key);
684 }
685 assert(s.ok());
686 } else {
687 return s;
688 }
689
690 return Status::OK();
691 };
692
693 std::unique_ptr<LockTracker::ColumnFamilyIterator> cf_it(
694 lock_tracker.GetColumnFamilyIterator());
695 assert(cf_it != nullptr);
696 while (cf_it->HasNext()) {
697 ColumnFamilyId cf = cf_it->Next();
698 std::unique_ptr<LockTracker::KeyIterator> key_it(
699 lock_tracker.GetKeyIterator(cf));
700 assert(key_it != nullptr);
701 while (key_it->HasNext()) {
702 const std::string& key = key_it->Next();
703 auto s = WriteRollbackKey(key, cf);
704 if (!s.ok()) {
705 return s;
706 }
707 }
708 }
709
710 for (const auto& cfkey : untracked_keys_) {
711 const auto cfid = cfkey.first;
712 const auto& keys = cfkey.second;
713 for (const auto& key : keys) {
714 auto s = WriteRollbackKey(key, cfid);
715 if (!s.ok()) {
716 return s;
717 }
718 }
719 }
720
721 return Status::OK();

724Status WriteUnpreparedTxn::RollbackInternal() {
725 // TODO(lth): Reduce duplicate code with WritePrepared rollback logic.
726 WriteBatchWithIndex rollback_batch(
727 wpt_db_->DefaultColumnFamily()->GetComparator(), 0, true, 0,
728 write_options_.protection_bytes_per_key);
729 assert(GetId() != kMaxSequenceNumber);
730 assert(GetId() > 0);
731 Status s;
732 auto read_at_seq = kMaxSequenceNumber;
733 // TODO: plumb Env::IOActivity, Env::IOPriority
734 ReadOptions roptions;
735 // to prevent callback's seq to be overrriden inside DBImpk::Get
736 roptions.snapshot = wpt_db_->GetMaxSnapshot();
737 // Note that we do not use WriteUnpreparedTxnReadCallback because we do not
738 // need to read our own writes when reading prior versions of the key for
739 // rollback.
740 WritePreparedTxnReadCallback callback(wpt_db_, read_at_seq);
741 // TODO(lth): We write rollback batch all in a single batch here, but this
742 // should be subdivded into multiple batches as well. In phase 2, when key
743 // sets are read from WAL, this will happen naturally.
744 s = WriteRollbackKeys(*tracked_locks_, &rollback_batch, &callback, roptions);
745 if (!s.ok()) {
746 return s;
747 }
748
749 // The Rollback marker will be used as a batch separator
750 s = WriteBatchInternal::MarkRollback(rollback_batch.GetWriteBatch(), name_);
751 assert(s.ok());
752 bool do_one_write = !db_impl_->immutable_db_options().two_write_queues;
753 const bool DISABLE_MEMTABLE = true;
754 const uint64_t NO_REF_LOG = 0;
755 uint64_t seq_used = kMaxSequenceNumber;
756 // Rollback batch may contain duplicate keys, because tracked_keys_ is not
757 // comparator aware.
758 auto rollback_batch_cnt = rollback_batch.SubBatchCnt();
759 // We commit the rolled back prepared batches. Although this is
760 // counter-intuitive, i) it is safe to do so, since the prepared batches are
761 // already canceled out by the rollback batch, ii) adding the commit entry to
762 // CommitCache will allow us to benefit from the existing mechanism in
763 // CommitCache that keeps an entry evicted due to max advance and yet overlaps
764 // with a live snapshot around so that the live snapshot properly skips the
765 // entry even if its prepare seq is lower than max_evicted_seq_.
766 //
767 // TODO(lth): RollbackInternal is conceptually very similar to
768 // CommitInternal, with the rollback batch simply taking on the role of
769 // CommitTimeWriteBatch. We should be able to merge the two code paths.
770 WriteUnpreparedCommitEntryPreReleaseCallback update_commit_map(
771 wpt_db_, db_impl_, unprep_seqs_, rollback_batch_cnt);
772 // Note: the rollback batch does not need AddPrepared since it is written to
773 // DB in one shot. min_uncommitted still works since it requires capturing
774 // data that is written to DB but not yet committed, while the rollback
775 // batch commits with PreReleaseCallback.
776 s = db_impl_->WriteImpl(write_options_, rollback_batch.GetWriteBatch(),
777 nullptr, nullptr, nullptr, NO_REF_LOG,
778 !DISABLE_MEMTABLE, &seq_used, rollback_batch_cnt,
779 do_one_write ? &update_commit_map : nullptr);
780 assert(!s.ok() || seq_used != kMaxSequenceNumber);
781 if (!s.ok()) {
782 return s;
783 }
784 if (do_one_write) {
785 for (const auto& seq : unprep_seqs_) {
786 wpt_db_->RemovePrepared(seq.first, seq.second);
787 }
788 unprep_seqs_.clear();
789 flushed_save_points_.reset(nullptr);
790 unflushed_save_points_.reset(nullptr);
791 return s;
792 } // else do the 2nd write for commit
793
794 uint64_t& prepare_seq = seq_used;
795 // Populate unprep_seqs_ with rollback_batch_cnt, since we treat data in the
796 // rollback write batch as just another "unprepared" batch. This will also
797 // update the unprep_seqs_ in the update_commit_map callback.
798 unprep_seqs_[prepare_seq] = rollback_batch_cnt;
799 WriteUnpreparedCommitEntryPreReleaseCallback
800 update_commit_map_with_rollback_batch(wpt_db_, db_impl_, unprep_seqs_, 0);
801
802 ROCKS_LOG_DETAILS(db_impl_->immutable_db_options().info_log,
803 "RollbackInternal 2nd write prepare_seq: %" PRIu64,
804 prepare_seq);
805 WriteBatch empty_batch;
806 const size_t ONE_BATCH = 1;
807 s = empty_batch.PutLogData(Slice());
808 assert(s.ok());
809 // In the absence of Prepare markers, use Noop as a batch separator
810 s = WriteBatchInternal::InsertNoop(&empty_batch);
811 assert(s.ok());
812 s = db_impl_->WriteImpl(write_options_, &empty_batch, nullptr, nullptr,
813 nullptr, NO_REF_LOG, DISABLE_MEMTABLE, &seq_used,
814 ONE_BATCH, &update_commit_map_with_rollback_batch);
815 assert(!s.ok() || seq_used != kMaxSequenceNumber);
816 // Mark the txn as rolled back
817 if (s.ok()) {
818 for (const auto& seq : unprep_seqs_) {
819 wpt_db_->RemovePrepared(seq.first, seq.second);
820 }
821 }
822
823 unprep_seqs_.clear();
824 flushed_save_points_.reset(nullptr);
825 unflushed_save_points_.reset(nullptr);
826 return s;
827}
