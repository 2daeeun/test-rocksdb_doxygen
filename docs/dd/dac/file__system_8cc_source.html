29static int RegisterBuiltinFileSystems(ObjectLibrary& library,
30 const std::string& /*arg*/) {
31 library.AddFactory<FileSystem>(
32 TimedFileSystem::kClassName(),
33 [](const std::string& /*uri*/, std::unique_ptr<FileSystem>* guard,
34 std::string* /* errmsg */) {
35 guard->reset(new TimedFileSystem(nullptr));
36 return guard->get();
37 });
38 library.AddFactory<FileSystem>(
39 ReadOnlyFileSystem::kClassName(),
40 [](const std::string& /*uri*/, std::unique_ptr<FileSystem>* guard,
41 std::string* /* errmsg */) {
42 guard->reset(new ReadOnlyFileSystem(nullptr));
43 return guard->get();
44 });
45 library.AddFactory<FileSystem>(
46 EncryptedFileSystem::kClassName(),
47 [](const std::string& /*uri*/, std::unique_ptr<FileSystem>* guard,
48 std::string* errmsg) {
49 Status s = NewEncryptedFileSystemImpl(nullptr, nullptr, guard);
50 if (!s.ok()) {
51 *errmsg = s.ToString();
52 }
53 return guard->get();
54 });
55 library.AddFactory<FileSystem>(
56 CountedFileSystem::kClassName(),
57 [](const std::string& /*uri*/, std::unique_ptr<FileSystem>* guard,
58 std::string* /*errmsg*/) {
59 guard->reset(new CountedFileSystem(FileSystem::Default()));
60 return guard->get();
61 });
62 library.AddFactory<FileSystem>(
63 MockFileSystem::kClassName(),
64 [](const std::string& /*uri*/, std::unique_ptr<FileSystem>* guard,
65 std::string* /*errmsg*/) {
66 guard->reset(new MockFileSystem(SystemClock::Default()));
67 return guard->get();
68 });
69#ifndef OS_WIN
70 library.AddFactory<FileSystem>(
71 ChrootFileSystem::kClassName(),
72 [](const std::string& /*uri*/, std::unique_ptr<FileSystem>* guard,
73 std::string* /* errmsg */) {
74 guard->reset(new ChrootFileSystem(nullptr, ""));
75 return guard->get();
76 });
77#endif // OS_WIN
78 size_t num_types;
79 return static_cast<int>(library.GetFactoryCount(&num_types));
80}

82Status FileSystem::CreateFromString(const ConfigOptions& config_options,
83 const std::string& value,
84 std::shared_ptr<FileSystem>* result) {
85 auto default_fs = FileSystem::Default();
86 if (default_fs->IsInstanceOf(value)) {
87 *result = default_fs;
88 return Status::OK();
89 } else {
90 static std::once_flag once;
91 std::call_once(once, [&]() {
92 RegisterBuiltinFileSystems(*(ObjectLibrary::Default().get()), "");
93 });
94 return LoadSharedObject<FileSystem>(config_options, value, result);
95 }
96}

110IOStatus FileSystem::NewLogger(const std::string& fname,
111 const IOOptions& io_opts,
112 std::shared_ptr<Logger>* result,
113 IODebugContext* dbg) {
114 FileOptions options;
115 options.io_options = io_opts;
116 // TODO: Tune the buffer size.
117 options.writable_file_max_buffer_size = 1024 * 1024;
118 std::unique_ptr<FSWritableFile> writable_file;
119 const IOStatus status = NewWritableFile(fname, options, &writable_file, dbg);
120 if (!status.ok()) {
121 return status;
122 }
123
124 *result = std::make_shared<EnvLogger>(std::move(writable_file), fname,
125 options, Env::Default());
126 return IOStatus::OK();
127}

182IOStatus WriteStringToFile(FileSystem* fs, const Slice& data,
183 const std::string& fname, bool should_sync,
184 const IOOptions& io_options,
185 const FileOptions& file_options) {
186 std::unique_ptr<FSWritableFile> file;
187 IOStatus s = fs->NewWritableFile(fname, file_options, &file, nullptr);
188 if (!s.ok()) {
189 return s;
190 }
191 s = file->Append(data, io_options, nullptr);
192 if (s.ok() && should_sync) {
193 s = file->Sync(io_options, nullptr);
194 }
195 if (!s.ok()) {
196 fs->DeleteFile(fname, io_options, nullptr);
197 }
198 return s;
199}

206IOStatus ReadFileToString(FileSystem* fs, const std::string& fname,
207 const IOOptions& opts, std::string* data) {
208 FileOptions soptions;
209 data->clear();
210 std::unique_ptr<FSSequentialFile> file;
211 IOStatus s = status_to_io_status(
212 fs->NewSequentialFile(fname, soptions, &file, nullptr));
213 if (!s.ok()) {
214 return s;
215 }
216 static const int kBufferSize = 8192;
217 char* space = new char[kBufferSize];
218 while (true) {
219 Slice fragment;
220 s = file->Read(kBufferSize, opts, &fragment, space, nullptr);
221 if (!s.ok()) {
222 break;
223 }
224 data->append(fragment.data(), fragment.size());
225 if (fragment.empty()) {
226 break;
227 }
228 }
229 delete[] space;
230 return s;
231}

252std::string FileSystemWrapper::SerializeOptions(
253 const ConfigOptions& config_options, const std::string& header) const {
254 auto parent = FileSystem::SerializeOptions(config_options, "");
255 if (config_options.IsShallow() || target_ == nullptr ||
256 target_->IsInstanceOf(FileSystem::kDefaultName())) {
257 return parent;
258 } else {
259 std::string result = header;
260 if (!StartsWith(parent, OptionTypeInfo::kIdPropName())) {
261 result.append(OptionTypeInfo::kIdPropName()).append("=");
262 }
263 result.append(parent);
264 if (!EndsWith(result, config_options.delimiter)) {
265 result.append(config_options.delimiter);
266 }
267 result.append("target=").append(target_->ToString(config_options));
268 return result;
269 }
