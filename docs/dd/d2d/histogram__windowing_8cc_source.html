39void HistogramWindowingImpl::Clear() {
40 std::lock_guard<std::mutex> lock(mutex_);
41
42 stats_.Clear();
43 for (size_t i = 0; i < num_windows_; i++) {
44 window_stats_[i].Clear();
45 }
46 current_window_.store(0, std::memory_order_relaxed);
47 last_swap_time_.store(clock_->NowMicros(), std::memory_order_relaxed);

52// This function is designed to be lock free, as it's in the critical path
53// of any operation.
54// Each individual value is atomic, it is just that some samples can go
55// in the older bucket which is tolerable.
56void HistogramWindowingImpl::Add(uint64_t value) {
57 TimerTick();
58
59 // Parent (global) member update
60 stats_.Add(value);
61
62 // Current window update
63 window_stats_[static_cast<size_t>(current_window())].Add(value);
64}

72void HistogramWindowingImpl::Merge(const HistogramWindowingImpl& other) {
73 std::lock_guard<std::mutex> lock(mutex_);
74 stats_.Merge(other.stats_);
75
76 if (stats_.num_buckets_ != other.stats_.num_buckets_ ||
77 micros_per_window_ != other.micros_per_window_) {
78 return;
79 }
80
81 uint64_t cur_window = current_window();
82 uint64_t other_cur_window = other.current_window();
83 // going backwards for alignment
84 for (unsigned int i = 0; i < std::min(num_windows_, other.num_windows_);
85 i++) {
86 uint64_t window_index = (cur_window + num_windows_ - i) % num_windows_;
87 uint64_t other_window_index =
88 (other_cur_window + other.num_windows_ - i) % other.num_windows_;
89 size_t windex = static_cast<size_t>(window_index);
90 size_t other_windex = static_cast<size_t>(other_window_index);
91
92 window_stats_[windex].Merge(other.window_stats_[other_windex]);
93 }
94}

102double HistogramWindowingImpl::Percentile(double p) const {
103 // Retry 3 times in total
104 for (int retry = 0; retry < 3; retry++) {
105 uint64_t start_num = stats_.num();
106 double result = stats_.Percentile(p);
107 // Detect if swap buckets or Clear() was called during calculation
108 if (stats_.num() >= start_num) {
109 return result;
110 }
111 }
112 return 0.0;
113}

125void HistogramWindowingImpl::TimerTick() {
126 uint64_t curr_time = clock_->NowMicros();
127 size_t curr_window_ = static_cast<size_t>(current_window());
128 if (curr_time - last_swap_time() > micros_per_window_ &&
129 window_stats_[curr_window_].num() >= min_num_per_window_) {
130 SwapHistoryBucket();
131 }
132}

134void HistogramWindowingImpl::SwapHistoryBucket() {
135 // Threads executing Add() would be competing for this mutex, the first one
136 // who got the metex would take care of the bucket swap, other threads
137 // can skip this.
138 // If mutex is held by Merge() or Clear(), next Add() will take care of the
139 // swap, if needed.
140 if (mutex_.try_lock()) {
141 last_swap_time_.store(clock_->NowMicros(), std::memory_order_relaxed);
142
143 uint64_t curr_window = current_window();
144 uint64_t next_window =
145 (curr_window == num_windows_ - 1) ? 0 : curr_window + 1;
146
147 // subtract next buckets from totals and swap to next buckets
148 HistogramStat& stats_to_drop =
149 window_stats_[static_cast<size_t>(next_window)];
150
151 if (!stats_to_drop.Empty()) {
152 for (size_t b = 0; b < stats_.num_buckets_; b++) {
153 stats_.buckets_[b].fetch_sub(stats_to_drop.bucket_at(b),
154 std::memory_order_relaxed);
155 }
156
157 if (stats_.min() == stats_to_drop.min()) {
158 uint64_t new_min = std::numeric_limits<uint64_t>::max();
159 for (unsigned int i = 0; i < num_windows_; i++) {
160 if (i != next_window) {
161 uint64_t m = window_stats_[i].min();
162 if (m < new_min) {
163 new_min = m;
164 }
165 }
166 }
167 stats_.min_.store(new_min, std::memory_order_relaxed);
168 }
169
170 if (stats_.max() == stats_to_drop.max()) {
171 uint64_t new_max = 0;
172 for (unsigned int i = 0; i < num_windows_; i++) {
173 if (i != next_window) {
174 uint64_t m = window_stats_[i].max();
175 if (m > new_max) {
176 new_max = m;
177 }
178 }
179 }
180 stats_.max_.store(new_max, std::memory_order_relaxed);
181 }
182
183 stats_.num_.fetch_sub(stats_to_drop.num(), std::memory_order_relaxed);
184 stats_.sum_.fetch_sub(stats_to_drop.sum(), std::memory_order_relaxed);
185 stats_.sum_squares_.fetch_sub(stats_to_drop.sum_squares(),
186 std::memory_order_relaxed);
187
188 stats_to_drop.Clear();
189 }
190
191 // advance to next window bucket
192 current_window_.store(next_window, std::memory_order_relaxed);
193
194 mutex_.unlock();
195 }
196}
