89 Status GenerateAndAddExternalFile(
90 const Options options, std::vector<int> keys,
91 const std::vector<ValueType>& value_types,
92 std::vector<std::pair<int, int>> range_deletions, int file_id,
93 bool write_global_seqno, bool verify_checksums_before_ingest,
94 std::map<std::string, std::string>* true_data) {
95 assert(value_types.size() == 1 || keys.size() == value_types.size());
96 std::string file_path = sst_files_dir_ + std::to_string(file_id);
97 SstFileWriter sst_file_writer(EnvOptions(), options);
98
99 Status s = sst_file_writer.Open(file_path);
100 if (!s.ok()) {
101 return s;
102 }
103 for (size_t i = 0; i < range_deletions.size(); i++) {
104 // Account for the effect of range deletions on true_data before
105 // all point operators, even though sst_file_writer.DeleteRange
106 // must be called before other sst_file_writer methods. This is
107 // because point writes take precedence over range deletions
108 // in the same ingested sst. This precedence is part of
109 // `SstFileWriter::DeleteRange()`'s API contract.
110 std::string start_key = Key(range_deletions[i].first);
111 std::string end_key = Key(range_deletions[i].second);
112 s = sst_file_writer.DeleteRange(start_key, end_key);
113 if (!s.ok()) {
114 sst_file_writer.Finish();
115 return s;
116 }
117 auto start_key_it = true_data->find(start_key);
118 if (start_key_it == true_data->end()) {
119 start_key_it = true_data->upper_bound(start_key);
120 }
121 auto end_key_it = true_data->find(end_key);
122 if (end_key_it == true_data->end()) {
123 end_key_it = true_data->upper_bound(end_key);
124 }
125 true_data->erase(start_key_it, end_key_it);
126 }
127 for (size_t i = 0; i < keys.size(); i++) {
128 std::string key = Key(keys[i]);
129 std::string value = Key(keys[i]) + std::to_string(file_id);
130 ValueType value_type =
131 (value_types.size() == 1 ? value_types[0] : value_types[i]);
132 switch (value_type) {
133 case ValueType::kTypeValue:
134 s = sst_file_writer.Put(key, value);
135 (*true_data)[key] = value;
136 break;
137 case ValueType::kTypeMerge:
138 s = sst_file_writer.Merge(key, value);
139 // we only use TestPutOperator in this test
140 (*true_data)[key] = value;
141 break;
142 case ValueType::kTypeDeletion:
143 s = sst_file_writer.Delete(key);
144 true_data->erase(key);
145 break;
146 default:
147 return Status::InvalidArgument("Value type is not supported");
148 }
149 if (!s.ok()) {
150 sst_file_writer.Finish();
151 return s;
152 }
153 }
154 s = sst_file_writer.Finish();
155
156 if (s.ok()) {
157 IngestExternalFileOptions ifo;
158 ifo.allow_global_seqno = true;
159 ifo.write_global_seqno = write_global_seqno;
160 ifo.verify_checksums_before_ingest = verify_checksums_before_ingest;
161 s = db_->IngestExternalFile({file_path}, ifo);
162 }
163 return s;
164 }

347class ChecksumVerifyHelper {
348 private:
349 Options options_;
350
351 public:
352 ChecksumVerifyHelper(Options& options) : options_(options) {}
353 ~ChecksumVerifyHelper() = default;
354
355 Status GetSingleFileChecksumAndFuncName(
356 const std::string& file_path, std::string* file_checksum,
357 std::string* file_checksum_func_name) {
358 Status s;
359 EnvOptions soptions;
360 std::unique_ptr<SequentialFile> file_reader;
361 s = options_.env->NewSequentialFile(file_path, &file_reader, soptions);
362 if (!s.ok()) {
363 return s;
364 }
365 std::unique_ptr<char[]> scratch(new char[2048]);
366 Slice result;
367 FileChecksumGenFactory* file_checksum_gen_factory =
368 options_.file_checksum_gen_factory.get();
369 if (file_checksum_gen_factory == nullptr) {
370 *file_checksum = kUnknownFileChecksum;
371 *file_checksum_func_name = kUnknownFileChecksumFuncName;
372 return Status::OK();
373 } else {
374 FileChecksumGenContext gen_context;
375 std::unique_ptr<FileChecksumGenerator> file_checksum_gen =
376 file_checksum_gen_factory->CreateFileChecksumGenerator(gen_context);
377 *file_checksum_func_name = file_checksum_gen->Name();
378 s = file_reader->Read(2048, &result, scratch.get());
379 if (!s.ok()) {
380 return s;
381 }
382 while (result.size() != 0) {
383 file_checksum_gen->Update(scratch.get(), result.size());
384 s = file_reader->Read(2048, &result, scratch.get());
385 if (!s.ok()) {
386 return s;
387 }
388 }
389 file_checksum_gen->Finalize();
390 *file_checksum = file_checksum_gen->GetChecksum();
391 }
392 return Status::OK();
393 }
