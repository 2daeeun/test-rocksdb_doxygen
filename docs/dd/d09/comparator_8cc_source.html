42 void FindShortestSeparator(std::string* start,
43 const Slice& limit) const override {
44 // Find length of common prefix
45 size_t min_length = std::min(start->size(), limit.size());
46 size_t diff_index = 0;
47 while ((diff_index < min_length) &&
48 ((*start)[diff_index] == limit[diff_index])) {
49 diff_index++;
50 }
51
52 if (diff_index >= min_length) {
53 // Do not shorten if one string is a prefix of the other
54 } else {
55 uint8_t start_byte = static_cast<uint8_t>((*start)[diff_index]);
56 uint8_t limit_byte = static_cast<uint8_t>(limit[diff_index]);
57 if (start_byte >= limit_byte) {
58 // Cannot shorten since limit is smaller than start or start is
59 // already the shortest possible.
60 return;
61 }
62 assert(start_byte < limit_byte);
63
64 if (diff_index < limit.size() - 1 || start_byte + 1 < limit_byte) {
65 (*start)[diff_index]++;
66 start->resize(diff_index + 1);
67 } else {
68 // v
69 // A A 1 A A A
70 // A A 2
71 //
72 // Incrementing the current byte will make start bigger than limit, we
73 // will skip this byte, and find the first non 0xFF byte in start and
74 // increment it.
75 diff_index++;
76
77 while (diff_index < start->size()) {
78 // Keep moving until we find the first non 0xFF byte to
79 // increment it
80 if (static_cast<uint8_t>((*start)[diff_index]) <
81 static_cast<uint8_t>(0xff)) {
82 (*start)[diff_index]++;
83 start->resize(diff_index + 1);
84 break;
85 }
86 diff_index++;
87 }
88 }
89 assert(Compare(*start, limit) < 0);
90 }
91 }
92
93 void FindShortSuccessor(std::string* key) const override {
94 // Find first character that can be incremented
95 size_t n = key->size();
96 for (size_t i = 0; i < n; i++) {
97 const uint8_t byte = (*key)[i];
98 if (byte != static_cast<uint8_t>(0xff)) {
99 (*key)[i] = byte + 1;
100 key->resize(i + 1);
101 return;
102 }
103 }
104 // *key is a run of 0xffs. Leave it alone.
105 }

107 bool IsSameLengthImmediateSuccessor(const Slice& s,
108 const Slice& t) const override {
109 if (s.size() != t.size() || s.size() == 0) {
110 return false;
111 }
112 size_t diff_ind = s.difference_offset(t);
113 // same slice
114 if (diff_ind >= s.size()) {
115 return false;
116 }
117 uint8_t byte_s = static_cast<uint8_t>(s[diff_ind]);
118 uint8_t byte_t = static_cast<uint8_t>(t[diff_ind]);
119 // first different byte must be consecutive, and remaining bytes must be
120 // 0xff for s and 0x00 for t
121 if (byte_s != uint8_t{0xff} && byte_s + 1 == byte_t) {
122 for (size_t i = diff_ind + 1; i < s.size(); ++i) {
123 byte_s = static_cast<uint8_t>(s[i]);
124 byte_t = static_cast<uint8_t>(t[i]);
125 if (byte_s != uint8_t{0xff} || byte_t != uint8_t{0x00}) {
126 return false;
127 }
128 }
129 return true;
130 } else {
131 return false;
132 }
133 }
134
135 bool CanKeysWithDifferentByteContentsBeEqual() const override {
136 return false;
137 }
138
139 using Comparator::CompareWithoutTimestamp;
140 int CompareWithoutTimestamp(const Slice& a, bool /*a_has_ts*/, const Slice& b,
141 bool /*b_has_ts*/) const override {
142 return a.compare(b);
143 }
144
145 bool EqualWithoutTimestamp(const Slice& a, const Slice& b) const override {
146 return a == b;
147 }

150class ReverseBytewiseComparatorImpl : public BytewiseComparatorImpl {

159 int Compare(const Slice& a, const Slice& b) const override {
160 return -a.compare(b);

163 void FindShortestSeparator(std::string* start,
164 const Slice& limit) const override {
165 // Find length of common prefix
166 size_t min_length = std::min(start->size(), limit.size());
167 size_t diff_index = 0;
168 while ((diff_index < min_length) &&
169 ((*start)[diff_index] == limit[diff_index])) {
170 diff_index++;
171 }
172
173 assert(diff_index <= min_length);
174 if (diff_index == min_length) {
175 // Do not shorten if one string is a prefix of the other
176 //
177 // We could handle cases like:
178 // V
179 // A A 2 X Y
180 // A A 2
181 // in a similar way as BytewiseComparator::FindShortestSeparator().
182 // We keep it simple by not implementing it. We can come back to it
183 // later when needed.
184 } else {
185 uint8_t start_byte = static_cast<uint8_t>((*start)[diff_index]);
186 uint8_t limit_byte = static_cast<uint8_t>(limit[diff_index]);
187 if (start_byte > limit_byte && diff_index < start->size() - 1) {
188 // Case like
189 // V
190 // A A 3 A A
191 // A A 1 B B
192 //
193 // or
194 // v
195 // A A 2 A A
196 // A A 1 B B
197 // In this case "AA2" will be good.
198#ifndef NDEBUG
199 std::string old_start = *start;
200#endif
201 start->resize(diff_index + 1);
202#ifndef NDEBUG
203 assert(old_start >= *start);
204#endif
205 assert(Slice(*start).compare(limit) > 0);
206 }
207 }
208 }

235// Comparator with 64-bit integer timestamp.
236// We did not performance test this yet.
237template <typename TComparator>
238class ComparatorWithU64TsImpl : public Comparator {
239 static_assert(std::is_base_of<Comparator, TComparator>::value,
240 "template type must be a inherited type of comparator");
241
242 public:
243 explicit ComparatorWithU64TsImpl() : Comparator(/*ts_sz=*/sizeof(uint64_t)) {
244 assert(cmp_without_ts_.timestamp_size() == 0);
245 }
246
247 static const char* kClassName() {
248 static std::string class_name = kClassNameInternal();
249 return class_name.c_str();
250 }
251
252 const char* Name() const override { return kClassName(); }
253
254 // The comparator that compares the user key without timestamp part is treated
255 // as the root comparator.
256 const Comparator* GetRootComparator() const override {
257 return &cmp_without_ts_;
258 }
259
260 void FindShortSuccessor(std::string*) const override {}
261 void FindShortestSeparator(std::string*, const Slice&) const override {}
262 int Compare(const Slice& a, const Slice& b) const override {
263 int ret = CompareWithoutTimestamp(a, b);
264 size_t ts_sz = timestamp_size();
265 if (ret != 0) {
266 return ret;
267 }
268 // Compare timestamp.
269 // For the same user key with different timestamps, larger (newer) timestamp
270 // comes first.
271 return -CompareTimestamp(ExtractTimestampFromUserKey(a, ts_sz),
272 ExtractTimestampFromUserKey(b, ts_sz));
273 }
274
275 Slice GetMaxTimestamp() const override { return MaxU64Ts(); }
276
277 Slice GetMinTimestamp() const override { return MinU64Ts(); }
278
279 std::string TimestampToString(const Slice& timestamp) const override {
280 assert(timestamp.size() == sizeof(uint64_t));
281 uint64_t ts = 0;
282 DecodeU64Ts(timestamp, &ts).PermitUncheckedError();
283 return std::to_string(ts);
284 }
285
286 using Comparator::CompareWithoutTimestamp;
287 int CompareWithoutTimestamp(const Slice& a, bool a_has_ts, const Slice& b,
288 bool b_has_ts) const override {
289 const size_t ts_sz = timestamp_size();
290 assert(!a_has_ts || a.size() >= ts_sz);
291 assert(!b_has_ts || b.size() >= ts_sz);
292 Slice lhs = a_has_ts ? StripTimestampFromUserKey(a, ts_sz) : a;
293 Slice rhs = b_has_ts ? StripTimestampFromUserKey(b, ts_sz) : b;
294 return cmp_without_ts_.Compare(lhs, rhs);
295 }
296 int CompareTimestamp(const Slice& ts1, const Slice& ts2) const override {
297 assert(ts1.size() == sizeof(uint64_t));
298 assert(ts2.size() == sizeof(uint64_t));
299 uint64_t lhs = DecodeFixed64(ts1.data());
300 uint64_t rhs = DecodeFixed64(ts2.data());
301 if (lhs < rhs) {
302 return -1;
303 } else if (lhs > rhs) {
304 return 1;
305 } else {
306 return 0;
307 }
308 }
309
310 private:
311 static std::string kClassNameInternal() {
312 std::stringstream ss;
313 ss << TComparator::kClassName() << ".u64ts";
314 return ss.str();
315 }
316
317 TComparator cmp_without_ts_;
318};

344Status DecodeU64Ts(const Slice& ts, uint64_t* int_ts) {
345 if (ts.size() != sizeof(uint64_t)) {
346 return Status::InvalidArgument("U64Ts timestamp size mismatch.");
347 }
348 *int_ts = DecodeFixed64(ts.data());
349 return Status::OK();
350}

352Slice EncodeU64Ts(uint64_t ts, std::string* ts_buf) {
353 char buf[sizeof(ts)];
354 EncodeFixed64(buf, ts);
355 ts_buf->assign(buf, sizeof(buf));
356 return Slice(*ts_buf);
357}

396Status Comparator::CreateFromString(const ConfigOptions& config_options,
397 const std::string& value,
398 const Comparator** result) {
399 static std::once_flag once;
400 std::call_once(once, [&]() {
401 RegisterBuiltinComparators(*(ObjectLibrary::Default().get()), "");
402 });
403 std::string id;
404 std::unordered_map<std::string, std::string> opt_map;
405 Status status = Customizable::GetOptionsMap(config_options, *result, value,
406 &id, &opt_map);
407 if (!status.ok()) { // GetOptionsMap failed
408 return status;
409 }
410 if (id == BytewiseComparatorImpl::kClassName()) {
411 *result = BytewiseComparator();
412 } else if (id == ReverseBytewiseComparatorImpl::kClassName()) {
413 *result = ReverseBytewiseComparator();
414 } else if (id ==
415 ComparatorWithU64TsImpl<BytewiseComparatorImpl>::kClassName()) {
416 *result = BytewiseComparatorWithU64Ts();
417 } else if (id == ComparatorWithU64TsImpl<
418 ReverseBytewiseComparatorImpl>::kClassName()) {
419 *result = ReverseBytewiseComparatorWithU64Ts();
420 } else if (value.empty()) {
421 // No Id and no options. Clear the object
422 *result = nullptr;
423 return Status::OK();
424 } else if (id.empty()) { // We have no Id but have options. Not good
425 return Status::NotSupported("Cannot reset object ", id);
426 } else {
427 status = config_options.registry->NewStaticObject(id, result);
428 if (!status.ok()) {
429 if (config_options.ignore_unsupported_options &&
430 status.IsNotSupported()) {
431 return Status::OK();
432 } else {
433 return status;
434 }
435 } else {
436 Comparator* comparator = const_cast<Comparator*>(*result);
437 status =
438 Customizable::ConfigureNewObject(config_options, comparator, opt_map);
439 }
440 }
441 return status;
442}
