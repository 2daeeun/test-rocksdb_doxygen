16template <typename TBlocklike>
17Status FilterBlockReaderCommon<TBlocklike>::ReadFilterBlock(
18 const BlockBasedTable* table, FilePrefetchBuffer* prefetch_buffer,
19 const ReadOptions& read_options, bool use_cache, GetContext* get_context,
20 BlockCacheLookupContext* lookup_context,
21 CachableEntry<TBlocklike>* filter_block) {
22 PERF_TIMER_GUARD(read_filter_block_nanos);
23
24 assert(table);
25 assert(filter_block);
26 assert(filter_block->IsEmpty());
27
28 const BlockBasedTable::Rep* const rep = table->get_rep();
29 assert(rep);
30
31 const Status s = table->RetrieveBlock(
32 prefetch_buffer, read_options, rep->filter_handle,
33 UncompressionDict::GetEmptyDict(), filter_block, get_context,
34 lookup_context,
35 /* for_compaction */ false, use_cache,
36 /* async_read */ false, /* use_block_cache_for_lookup */ true);
37
38 return s;
39}

94template <typename TBlocklike>
95bool FilterBlockReaderCommon<TBlocklike>::RangeMayExist(
96 const Slice* iterate_upper_bound, const Slice& user_key_without_ts,
97 const SliceTransform* prefix_extractor, const Comparator* comparator,
98 const Slice* const const_ikey_ptr, bool* filter_checked,
99 bool need_upper_bound_check, BlockCacheLookupContext* lookup_context,
100 const ReadOptions& read_options) {
101 if (!prefix_extractor || !prefix_extractor->InDomain(user_key_without_ts)) {
102 *filter_checked = false;
103 return true;
104 }
105 Slice prefix = prefix_extractor->Transform(user_key_without_ts);
106 if (need_upper_bound_check &&
107 !IsFilterCompatible(iterate_upper_bound, prefix, comparator)) {
108 *filter_checked = false;
109 return true;
110 } else {
111 *filter_checked = true;
112 return PrefixMayMatch(prefix, const_ikey_ptr,
113 /* get_context */ nullptr, lookup_context,
114 read_options);
115 }
116}

118template <typename TBlocklike>
119bool FilterBlockReaderCommon<TBlocklike>::IsFilterCompatible(
120 const Slice* iterate_upper_bound, const Slice& prefix,
121 const Comparator* comparator) const {
122 // Try to reuse the bloom filter in the SST table if prefix_extractor in
123 // mutable_cf_options has changed. If range [user_key, upper_bound) all
124 // share the same prefix then we may still be able to use the bloom filter.
125 const SliceTransform* const prefix_extractor = table_prefix_extractor();
126 if (iterate_upper_bound != nullptr && prefix_extractor) {
127 if (!prefix_extractor->InDomain(*iterate_upper_bound)) {
128 return false;
129 }
130 Slice upper_bound_xform = prefix_extractor->Transform(*iterate_upper_bound);
131 // first check if user_key and upper_bound all share the same prefix
132 if (comparator->CompareWithoutTimestamp(prefix, false, upper_bound_xform,
133 false) != 0) {
134 // second check if user_key's prefix is the immediate predecessor of
135 // upper_bound and have the same length. If so, we know for sure all
136 // keys in the range [user_key, upper_bound) share the same prefix.
137 // Also need to make sure upper_bound are full length to ensure
138 // correctness
139 if (!full_length_enabled_ ||
140 iterate_upper_bound->size() != prefix_extractor_full_length_ ||
141 !comparator->IsSameLengthImmediateSuccessor(prefix,
142 *iterate_upper_bound)) {
143 return false;
144 }
145 }
146 return true;
147 } else {
148 return false;
149 }
150}

152template <typename TBlocklike>
153void FilterBlockReaderCommon<TBlocklike>::EraseFromCacheBeforeDestruction(
154 uint32_t uncache_aggressiveness) {
155 if (uncache_aggressiveness > 0) {
156 if (filter_block_.IsCached()) {
157 filter_block_.ResetEraseIfLastRef();
158 } else {
159 table()->EraseFromCache(table()->get_rep()->filter_handle);
160 }
