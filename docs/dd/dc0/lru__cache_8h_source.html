25// LRU cache implementation. This class is not thread-safe.
26
27// An entry is a variable length heap-allocated structure.
28// Entries are referenced by cache and/or by any external entity.
29// The cache keeps all its entries in a hash table. Some elements
30// are also stored on LRU list.
31//
32// LRUHandle can be in these states:
33// 1. Referenced externally AND in hash table.
34// In that case the entry is *not* in the LRU list
35// (refs >= 1 && in_cache == true)
36// 2. Not referenced externally AND in hash table.
37// In that case the entry is in the LRU list and can be freed.
38// (refs == 0 && in_cache == true)
39// 3. Referenced externally AND not in hash table.
40// In that case the entry is not in the LRU list and not in hash table.
41// The entry must be freed if refs becomes 0 in this state.
42// (refs >= 1 && in_cache == false)
43// If you call LRUCacheShard::Release enough times on an entry in state 1, it
44// will go into state 2. To move from state 1 to state 3, either call
45// LRUCacheShard::Erase or LRUCacheShard::Insert with the same key (but
46// possibly different value). To move from state 2 to state 1, use
47// LRUCacheShard::Lookup.
48// While refs > 0, public properties like value and deleter must not change.
49
50struct LRUHandle : public Cache::Handle {
51 Cache::ObjectPtr value;
52 const Cache::CacheItemHelper* helper;
53 LRUHandle* next_hash;
54 LRUHandle* next;
55 LRUHandle* prev;
56 size_t total_charge; // TODO(opt): Only allow uint32_t?
57 size_t key_length;
58 // The hash of key(). Used for fast sharding and comparisons.
59 uint32_t hash;
60 // The number of external refs to this entry. The cache itself is not counted.
61 uint32_t refs;
62
63 // Mutable flags - access controlled by mutex
64 // The m_ and M_ prefixes (and im_ and IM_ later) are to hopefully avoid
65 // checking an M_ flag on im_flags or an IM_ flag on m_flags.
66 uint8_t m_flags;
67 enum MFlags : uint8_t {
68 // Whether this entry is referenced by the hash table.
69 M_IN_CACHE = (1 << 0),
70 // Whether this entry has had any lookups (hits).
71 M_HAS_HIT = (1 << 1),
72 // Whether this entry is in high-pri pool.
73 M_IN_HIGH_PRI_POOL = (1 << 2),
74 // Whether this entry is in low-pri pool.
75 M_IN_LOW_PRI_POOL = (1 << 3),
76 };
77
78 // "Immutable" flags - only set in single-threaded context and then
79 // can be accessed without mutex
80 uint8_t im_flags;
81 enum ImFlags : uint8_t {
82 // Whether this entry is high priority entry.
83 IM_IS_HIGH_PRI = (1 << 0),
84 // Whether this entry is low priority entry.
85 IM_IS_LOW_PRI = (1 << 1),
86 // Marks result handles that should not be inserted into cache
87 IM_IS_STANDALONE = (1 << 2),
88 };
89
90 // Beginning of the key (MUST BE THE LAST FIELD IN THIS STRUCT!)
91 char key_data[1];
92
93 Slice key() const { return Slice(key_data, key_length); }
94
95 // For HandleImpl concept
96 uint32_t GetHash() const { return hash; }
97
98 // Increase the reference count by 1.
99 void Ref() { refs++; }
100
101 // Just reduce the reference count by 1. Return true if it was last reference.
102 bool Unref() {
103 assert(refs > 0);
104 refs--;
105 return refs == 0;
106 }
107
108 // Return true if there are external refs, false otherwise.
109 bool HasRefs() const { return refs > 0; }
110
111 bool InCache() const { return m_flags & M_IN_CACHE; }
112 bool IsHighPri() const { return im_flags & IM_IS_HIGH_PRI; }
113 bool InHighPriPool() const { return m_flags & M_IN_HIGH_PRI_POOL; }
114 bool IsLowPri() const { return im_flags & IM_IS_LOW_PRI; }
115 bool InLowPriPool() const { return m_flags & M_IN_LOW_PRI_POOL; }
116 bool HasHit() const { return m_flags & M_HAS_HIT; }
117 bool IsStandalone() const { return im_flags & IM_IS_STANDALONE; }
118
119 void SetInCache(bool in_cache) {
120 if (in_cache) {
121 m_flags |= M_IN_CACHE;
122 } else {
123 m_flags &= ~M_IN_CACHE;
124 }
125 }
126
127 void SetPriority(Cache::Priority priority) {
128 if (priority == Cache::Priority::HIGH) {
129 im_flags |= IM_IS_HIGH_PRI;
130 im_flags &= ~IM_IS_LOW_PRI;
131 } else if (priority == Cache::Priority::LOW) {
132 im_flags &= ~IM_IS_HIGH_PRI;
133 im_flags |= IM_IS_LOW_PRI;
134 } else {
135 im_flags &= ~IM_IS_HIGH_PRI;
136 im_flags &= ~IM_IS_LOW_PRI;
137 }
138 }
139
140 void SetInHighPriPool(bool in_high_pri_pool) {
141 if (in_high_pri_pool) {
142 m_flags |= M_IN_HIGH_PRI_POOL;
143 } else {
144 m_flags &= ~M_IN_HIGH_PRI_POOL;
145 }
146 }
147
148 void SetInLowPriPool(bool in_low_pri_pool) {
149 if (in_low_pri_pool) {
150 m_flags |= M_IN_LOW_PRI_POOL;
151 } else {
152 m_flags &= ~M_IN_LOW_PRI_POOL;
153 }
154 }
155
156 void SetHit() { m_flags |= M_HAS_HIT; }
157
158 void SetIsStandalone(bool is_standalone) {
159 if (is_standalone) {
160 im_flags |= IM_IS_STANDALONE;
161 } else {
162 im_flags &= ~IM_IS_STANDALONE;
163 }
164 }
165
166 void Free(MemoryAllocator* allocator) {
167 assert(refs == 0);
168 assert(helper);
169 if (helper->del_cb) {
170 helper->del_cb(value, allocator);
171 }
172
173 free(this);
174 }
175
176 inline size_t CalcuMetaCharge(
177 CacheMetadataChargePolicy metadata_charge_policy) const {
178 if (metadata_charge_policy != kFullChargeCacheMetadata) {
179 return 0;
180 } else {
181#ifdef ROCKSDB_MALLOC_USABLE_SIZE
182 return malloc_usable_size(
183 const_cast<void*>(static_cast<const void*>(this)));
184#else
185 // This is the size that is used when a new handle is created.
186 return sizeof(LRUHandle) - 1 + key_length;
187#endif
188 }
189 }
190
191 // Calculate the memory usage by metadata.
192 inline void CalcTotalCharge(
193 size_t charge, CacheMetadataChargePolicy metadata_charge_policy) {
194 total_charge = charge + CalcuMetaCharge(metadata_charge_policy);
195 }
196
197 inline size_t GetCharge(
198 CacheMetadataChargePolicy metadata_charge_policy) const {
199 size_t meta_charge = CalcuMetaCharge(metadata_charge_policy);
200 assert(total_charge >= meta_charge);
201 return total_charge - meta_charge;
202 }
203};

205// We provide our own simple hash table since it removes a whole bunch
206// of porting hacks and is also faster than some of the built-in hash
207// table implementations in some of the compiler/runtime combinations
208// we have tested. E.g., readrandom speeds up by ~5% over the g++
209// 4.4.3's builtin hashtable.
210class LRUHandleTable {
211 public:
212 explicit LRUHandleTable(int max_upper_hash_bits, MemoryAllocator* allocator);
213 ~LRUHandleTable();
214
215 LRUHandle* Lookup(const Slice& key, uint32_t hash);
216 LRUHandle* Insert(LRUHandle* h);
217 LRUHandle* Remove(const Slice& key, uint32_t hash);
218
219 template <typename T>
220 void ApplyToEntriesRange(T func, size_t index_begin, size_t index_end) {
221 for (size_t i = index_begin; i < index_end; i++) {
222 LRUHandle* h = list_[i];
223 while (h != nullptr) {
224 auto n = h->next_hash;
225 assert(h->InCache());
226 func(h);
227 h = n;
228 }
229 }
230 }
231
232 int GetLengthBits() const { return length_bits_; }
233
234 size_t GetOccupancyCount() const { return elems_; }
235
236 MemoryAllocator* GetAllocator() const { return allocator_; }
237
238 private:
239 // Return a pointer to slot that points to a cache entry that
240 // matches key/hash. If there is no such cache entry, return a
241 // pointer to the trailing slot in the corresponding linked list.
242 LRUHandle** FindPointer(const Slice& key, uint32_t hash);
243
244 void Resize();
245
246 // Number of hash bits (upper because lower bits used for sharding)
247 // used for table index. Length == 1 << length_bits_
248 int length_bits_;
249
250 // The table consists of an array of buckets where each bucket is
251 // a linked list of cache entries that hash into the bucket.
252 std::unique_ptr<LRUHandle*[]> list_;
253
254 // Number of elements currently in the table.
255 uint32_t elems_;
256
257 // Set from max_upper_hash_bits (see constructor).
258 const int max_length_bits_;
259
260 // From Cache, needed for delete
261 MemoryAllocator* const allocator_;
262};
