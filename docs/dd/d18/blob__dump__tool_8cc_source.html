28Status BlobDumpTool::Run(const std::string& filename, DisplayType show_key,
29 DisplayType show_blob,
30 DisplayType show_uncompressed_blob,
31 bool show_summary) {
32 constexpr size_t kReadaheadSize = 2 * 1024 * 1024;
33 Status s;
34 const auto fs = FileSystem::Default();
35 IOOptions io_opts;
36 s = fs->FileExists(filename, io_opts, nullptr);
37 if (!s.ok()) {
38 return s;
39 }
40 uint64_t file_size = 0;
41 s = fs->GetFileSize(filename, io_opts, &file_size, nullptr);
42 if (!s.ok()) {
43 return s;
44 }
45 std::unique_ptr<FSRandomAccessFile> file;
46 s = fs->NewRandomAccessFile(filename, FileOptions(), &file, nullptr);
47 if (!s.ok()) {
48 return s;
49 }
50 file = NewReadaheadRandomAccessFile(std::move(file), kReadaheadSize);
51 if (file_size == 0) {
52 return Status::Corruption("File is empty.");
53 }
54 reader_.reset(new RandomAccessFileReader(std::move(file), filename));
55 uint64_t offset = 0;
56 uint64_t footer_offset = 0;
57 CompressionType compression = kNoCompression;
58 s = DumpBlobLogHeader(&offset, &compression);
59 if (!s.ok()) {
60 return s;
61 }
62 s = DumpBlobLogFooter(file_size, &footer_offset);
63 if (!s.ok()) {
64 return s;
65 }
66 uint64_t total_records = 0;
67 uint64_t total_key_size = 0;
68 uint64_t total_blob_size = 0;
69 uint64_t total_uncompressed_blob_size = 0;
70 if (show_key != DisplayType::kNone || show_summary) {
71 while (offset < footer_offset) {
72 s = DumpRecord(show_key, show_blob, show_uncompressed_blob, show_summary,
73 compression, &offset, &total_records, &total_key_size,
74 &total_blob_size, &total_uncompressed_blob_size);
75 if (!s.ok()) {
76 break;
77 }
78 }
79 }
80 if (show_summary) {
81 fprintf(stdout, "Summary:\n");
82 fprintf(stdout, " total records: %" PRIu64 "\n", total_records);
83 fprintf(stdout, " total key size: %" PRIu64 "\n", total_key_size);
84 fprintf(stdout, " total blob size: %" PRIu64 "\n", total_blob_size);
85 if (compression != kNoCompression) {
86 fprintf(stdout, " total raw blob size: %" PRIu64 "\n",
87 total_uncompressed_blob_size);
88 }
89 }
90 return s;
91}

93Status BlobDumpTool::Read(uint64_t offset, size_t size, Slice* result) {
94 if (buffer_size_ < size) {
95 if (buffer_size_ == 0) {
96 buffer_size_ = 4096;
97 }
98 while (buffer_size_ < size) {
99 buffer_size_ *= 2;
100 }
101 buffer_.reset(new char[buffer_size_]);
102 }
103 Status s =
104 reader_->Read(IOOptions(), offset, size, result, buffer_.get(), nullptr);
105 if (!s.ok()) {
106 return s;
107 }
108 if (result->size() != size) {
109 return Status::Corruption("Reach the end of the file unexpectedly.");
110 }
111 return s;
112}

114Status BlobDumpTool::DumpBlobLogHeader(uint64_t* offset,
115 CompressionType* compression) {
116 Slice slice;
117 Status s = Read(0, BlobLogHeader::kSize, &slice);
118 if (!s.ok()) {
119 return s;
120 }
121 BlobLogHeader header;
122 s = header.DecodeFrom(slice);
123 if (!s.ok()) {
124 return s;
125 }
126 fprintf(stdout, "Blob log header:\n");
127 fprintf(stdout, " Version : %" PRIu32 "\n", header.version);
128 fprintf(stdout, " Column Family ID : %" PRIu32 "\n",
129 header.column_family_id);
130 std::string compression_str;
131 if (!GetStringFromCompressionType(&compression_str, header.compression)
132 .ok()) {
133 compression_str = "Unrecongnized compression type (" +
134 std::to_string((int)header.compression) + ")";
135 }
136 fprintf(stdout, " Compression : %s\n", compression_str.c_str());
137 fprintf(stdout, " Expiration range : %s\n",
138 GetString(header.expiration_range).c_str());
139 *offset = BlobLogHeader::kSize;
140 *compression = header.compression;
141 return s;
142}

144Status BlobDumpTool::DumpBlobLogFooter(uint64_t file_size,
145 uint64_t* footer_offset) {
146 auto no_footer = [&]() {
147 *footer_offset = file_size;
148 fprintf(stdout, "No blob log footer.\n");
149 return Status::OK();
150 };
151 if (file_size < BlobLogHeader::kSize + BlobLogFooter::kSize) {
152 return no_footer();
153 }
154 Slice slice;
155 *footer_offset = file_size - BlobLogFooter::kSize;
156 Status s = Read(*footer_offset, BlobLogFooter::kSize, &slice);
157 if (!s.ok()) {
158 return s;
159 }
160 BlobLogFooter footer;
161 s = footer.DecodeFrom(slice);
162 if (!s.ok()) {
163 return no_footer();
164 }
165 fprintf(stdout, "Blob log footer:\n");
166 fprintf(stdout, " Blob count : %" PRIu64 "\n", footer.blob_count);
167 fprintf(stdout, " Expiration Range : %s\n",
168 GetString(footer.expiration_range).c_str());
169 return s;
170}

172Status BlobDumpTool::DumpRecord(DisplayType show_key, DisplayType show_blob,
173 DisplayType show_uncompressed_blob,
174 bool show_summary, CompressionType compression,
175 uint64_t* offset, uint64_t* total_records,
176 uint64_t* total_key_size,
177 uint64_t* total_blob_size,
178 uint64_t* total_uncompressed_blob_size) {
179 if (show_key != DisplayType::kNone) {
180 fprintf(stdout, "Read record with offset 0x%" PRIx64 " (%" PRIu64 "):\n",
181 *offset, *offset);
182 }
183 Slice slice;
184 Status s = Read(*offset, BlobLogRecord::kHeaderSize, &slice);
185 if (!s.ok()) {
186 return s;
187 }
188 BlobLogRecord record;
189 s = record.DecodeHeaderFrom(slice);
190 if (!s.ok()) {
191 return s;
192 }
193 uint64_t key_size = record.key_size;
194 uint64_t value_size = record.value_size;
195 if (show_key != DisplayType::kNone) {
196 fprintf(stdout, " key size : %" PRIu64 "\n", key_size);
197 fprintf(stdout, " value size : %" PRIu64 "\n", value_size);
198 fprintf(stdout, " expiration : %" PRIu64 "\n", record.expiration);
199 }
200 *offset += BlobLogRecord::kHeaderSize;
201 s = Read(*offset, static_cast<size_t>(key_size + value_size), &slice);
202 if (!s.ok()) {
203 return s;
204 }
205 // Decompress value
206 std::string uncompressed_value;
207 if (compression != kNoCompression &&
208 (show_uncompressed_blob != DisplayType::kNone || show_summary)) {
209 BlockContents contents;
210 UncompressionContext context(compression);
211 UncompressionInfo info(context, UncompressionDict::GetEmptyDict(),
212 compression);
213 s = UncompressBlockData(
214 info, slice.data() + key_size, static_cast<size_t>(value_size),
215 &contents, 2 /*compress_format_version*/, ImmutableOptions(Options()));
216 if (!s.ok()) {
217 return s;
218 }
219 uncompressed_value = contents.data.ToString();
220 }
221 if (show_key != DisplayType::kNone) {
222 fprintf(stdout, " key : ");
223 DumpSlice(Slice(slice.data(), static_cast<size_t>(key_size)), show_key);
224 if (show_blob != DisplayType::kNone) {
225 fprintf(stdout, " blob : ");
226 DumpSlice(Slice(slice.data() + static_cast<size_t>(key_size),
227 static_cast<size_t>(value_size)),
228 show_blob);
229 }
230 if (show_uncompressed_blob != DisplayType::kNone) {
231 fprintf(stdout, " raw blob : ");
232 DumpSlice(Slice(uncompressed_value), show_uncompressed_blob);
233 }
234 }
235 *offset += key_size + value_size;
236 *total_records += 1;
237 *total_key_size += key_size;
238 *total_blob_size += value_size;
239 *total_uncompressed_blob_size += uncompressed_value.size();
240 return s;
241}

243void BlobDumpTool::DumpSlice(const Slice s, DisplayType type) {
244 if (type == DisplayType::kRaw) {
245 fprintf(stdout, "%s\n", s.ToString().c_str());
246 } else if (type == DisplayType::kHex) {
247 fprintf(stdout, "%s\n", s.ToString(true /*hex*/).c_str());
248 } else if (type == DisplayType::kDetail) {
249 char buf[100];
250 for (size_t i = 0; i < s.size(); i += 16) {
251 memset(buf, 0, sizeof(buf));
252 for (size_t j = 0; j < 16 && i + j < s.size(); j++) {
253 unsigned char c = s[i + j];
254 snprintf(buf + j * 3 + 15, 2, "%x", c >> 4);
255 snprintf(buf + j * 3 + 16, 2, "%x", c & 0xf);
256 snprintf(buf + j + 65, 2, "%c", (0x20 <= c && c <= 0x7e) ? c : '.');
257 }
258 for (size_t p = 0; p + 1 < sizeof(buf); p++) {
259 if (buf[p] == 0) {
260 buf[p] = ' ';
261 }
262 }
263 fprintf(stdout, "%s\n", i == 0 ? buf + 15 : buf);
264 }
265 }
