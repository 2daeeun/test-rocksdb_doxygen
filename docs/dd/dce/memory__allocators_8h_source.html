12// A memory allocator using new/delete
13class DefaultMemoryAllocator : public MemoryAllocator {
14 public:
15 static const char* kClassName() { return "DefaultMemoryAllocator"; }
16 const char* Name() const override { return kClassName(); }
17 void* Allocate(size_t size) override {
18 return static_cast<void*>(new char[size]);
19 }
20
21 void Deallocate(void* p) override { delete[] static_cast<char*>(p); }
22};

24// Base class for a MemoryAllocator. This implementation does nothing
25// and implements the methods in failuse mode (assert if the methods are
26// invoked). Implementations can extend this class and override these methods
27// when they are enabled via compiler switches (e.g., the
28// JeMallocMemoryAllocator can define these methods if ROCKSDB_JEMALLOC is
29// defined at compile time. If compiled in "disabled" mode, this class provides
30// default/failure implementations. If compiled in "enabled" mode, the derived
31// class needs to provide the appopriate "enabled" methods for the "real"
32// implementation. Failure of the "real" implementation to implement ovreride
33// any of these methods will result in an assert failure.
34class BaseMemoryAllocator : public MemoryAllocator {
35 public:
36 void* Allocate(size_t /*size*/) override {
37 assert(false);
38 return nullptr;
39 }
40
41 void Deallocate(void* /*p*/) override { assert(false); }
42};

44// A Wrapped MemoryAllocator. Delegates the memory allcator functions to the
45// wrapped one.
46class MemoryAllocatorWrapper : public MemoryAllocator {
47 public:
48 // Initialize an MemoryAllocatorWrapper that delegates all calls to *t
49 explicit MemoryAllocatorWrapper(const std::shared_ptr<MemoryAllocator>& t);
50 ~MemoryAllocatorWrapper() override {}
51
52 // Return the target to which to forward all calls
53 MemoryAllocator* target() const { return target_.get(); }
54 // Allocate a block of at least size. Has to be thread-safe.
55 void* Allocate(size_t size) override { return target_->Allocate(size); }
56
57 // Deallocate previously allocated block. Has to be thread-safe.
58 void Deallocate(void* p) override { return target_->Deallocate(p); }
59
60 // Returns the memory size of the block allocated at p. The default
61 // implementation that just returns the original allocation_size is fine.
62 size_t UsableSize(void* p, size_t allocation_size) const override {
63 return target_->UsableSize(p, allocation_size);
64 }
65
66 const Customizable* Inner() const override { return target_.get(); }
67
68 protected:
69 std::shared_ptr<MemoryAllocator> target_;
70};

72// A memory allocator that counts the number of allocations and deallocations
73// This class is useful if the number of memory allocations/dellocations is
74// important.
75class CountedMemoryAllocator : public MemoryAllocatorWrapper {
76 public:
77 CountedMemoryAllocator()
78 : MemoryAllocatorWrapper(std::make_shared<DefaultMemoryAllocator>()),
79 allocations_(0),
80 deallocations_(0) {}
81
82 explicit CountedMemoryAllocator(const std::shared_ptr<MemoryAllocator>& t)
83 : MemoryAllocatorWrapper(t), allocations_(0), deallocations_(0) {}
84 static const char* kClassName() { return "CountedMemoryAllocator"; }
85 const char* Name() const override { return kClassName(); }
86 std::string GetId() const override { return std::string(Name()); }
87 void* Allocate(size_t size) override {
88 allocations_++;
89 return MemoryAllocatorWrapper::Allocate(size);
90 }
91
92 void Deallocate(void* p) override {
93 deallocations_++;
94 MemoryAllocatorWrapper::Deallocate(p);
95 }
96 uint64_t GetNumAllocations() const { return allocations_; }
97 uint64_t GetNumDeallocations() const { return deallocations_; }
98
99 private:
100 std::atomic<uint64_t> allocations_;
101 std::atomic<uint64_t> deallocations_;
102};
