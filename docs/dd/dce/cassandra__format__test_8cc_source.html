16TEST(ColumnTest, Column) {

18 int8_t mask = 0;
19 int8_t index = 1;
20 int64_t timestamp = 1494022807044;

26

30 c.Serialize(&dest);

34 EXPECT_EQ(Deserialize<int8_t>(dest.c_str(), offset), mask);

43
