54ColumnFamilyHandleImpl::~ColumnFamilyHandleImpl() {
55 if (cfd_ != nullptr) {
56 for (auto& listener : cfd_->ioptions()->listeners) {
57 listener->OnColumnFamilyHandleDeletionStarted(this);
58 }
59 // Job id == 0 means that this is not our background process, but rather
60 // user thread
61 // Need to hold some shared pointers owned by the initial_cf_options
62 // before final cleaning up finishes.
63 ColumnFamilyOptions initial_cf_options_copy = cfd_->initial_cf_options();
64 JobContext job_context(0);
65 mutex_->Lock();
66 bool dropped = cfd_->IsDropped();
67 if (cfd_->UnrefAndTryDelete()) {
68 if (dropped) {
69 db_->FindObsoleteFiles(&job_context, false, true);
70 }
71 }
72 mutex_->Unlock();
73 if (job_context.HaveSomethingToDelete()) {
74 bool defer_purge =
75 db_->immutable_db_options().avoid_unnecessary_blocking_io;
76 db_->PurgeObsoleteFiles(job_context, defer_purge);
77 }
78 job_context.Clean();
79 }

113Status CheckCompressionSupported(const ColumnFamilyOptions& cf_options) {
114 if (!cf_options.compression_per_level.empty()) {
115 for (size_t level = 0; level < cf_options.compression_per_level.size();
116 ++level) {
117 if (!CompressionTypeSupported(cf_options.compression_per_level[level])) {
118 return Status::InvalidArgument(
119 "Compression type " +
120 CompressionTypeToString(cf_options.compression_per_level[level]) +
121 " is not linked with the binary.");
122 }
123 }
124 } else {
125 if (!CompressionTypeSupported(cf_options.compression)) {
126 return Status::InvalidArgument(
127 "Compression type " +
128 CompressionTypeToString(cf_options.compression) +
129 " is not linked with the binary.");
130 }
131 }
132 if (cf_options.compression_opts.zstd_max_train_bytes > 0) {
133 if (cf_options.compression_opts.use_zstd_dict_trainer) {
134 if (!ZSTD_TrainDictionarySupported()) {
135 return Status::InvalidArgument(
136 "zstd dictionary trainer cannot be used because ZSTD 1.1.3+ "
137 "is not linked with the binary.");
138 }
139 } else if (!ZSTD_FinalizeDictionarySupported()) {
140 return Status::InvalidArgument(
141 "zstd finalizeDictionary cannot be used because ZSTD 1.4.5+ "
142 "is not linked with the binary.");
143 }
144 if (cf_options.compression_opts.max_dict_bytes == 0) {
145 return Status::InvalidArgument(
146 "The dictionary size limit (`CompressionOptions::max_dict_bytes`) "
147 "should be nonzero if we're using zstd's dictionary generator.");
148 }
149 }
150
151 if (!CompressionTypeSupported(cf_options.blob_compression_type)) {
152 std::ostringstream oss;
153 oss << "The specified blob compression type "
154 << CompressionTypeToString(cf_options.blob_compression_type)
155 << " is not available.";
156
157 return Status::InvalidArgument(oss.str());
158 }
159
160 return Status::OK();
161}

163Status CheckConcurrentWritesSupported(const ColumnFamilyOptions& cf_options) {
164 if (cf_options.inplace_update_support) {
165 return Status::InvalidArgument(
166 "In-place memtable updates (inplace_update_support) is not compatible "
167 "with concurrent writes (allow_concurrent_memtable_write)");
168 }
169 if (!cf_options.memtable_factory->IsInsertConcurrentlySupported()) {
170 return Status::InvalidArgument(
171 "Memtable doesn't concurrent writes (allow_concurrent_memtable_write)");
172 }
173 return Status::OK();
174}

176Status CheckCFPathsSupported(const DBOptions& db_options,
177 const ColumnFamilyOptions& cf_options) {
178 // More than one cf_paths are supported only in universal
179 // and level compaction styles. This function also checks the case
180 // in which cf_paths is not specified, which results in db_paths
181 // being used.
182 if ((cf_options.compaction_style != kCompactionStyleUniversal) &&
183 (cf_options.compaction_style != kCompactionStyleLevel)) {
184 if (cf_options.cf_paths.size() > 1) {
185 return Status::NotSupported(
186 "More than one CF paths are only supported in "
187 "universal and level compaction styles. ");
188 } else if (cf_options.cf_paths.empty() && db_options.db_paths.size() > 1) {
189 return Status::NotSupported(
190 "More than one DB paths are only supported in "
191 "universal and level compaction styles. ");
192 }
193 }
194 return Status::OK();
195}

202ColumnFamilyOptions SanitizeOptions(const ImmutableDBOptions& db_options,
203 const ColumnFamilyOptions& src) {
204 ColumnFamilyOptions result = src;
205 size_t clamp_max = std::conditional<
206 sizeof(size_t) == 4, std::integral_constant<size_t, 0xffffffff>,
207 std::integral_constant<uint64_t, 64ull << 30>>::type::value;
208 ClipToRange(&result.write_buffer_size, (static_cast<size_t>(64)) << 10,
209 clamp_max);
210 // if user sets arena_block_size, we trust user to use this value. Otherwise,
211 // calculate a proper value from writer_buffer_size;
212 if (result.arena_block_size <= 0) {
213 result.arena_block_size =
214 std::min(size_t{1024 * 1024}, result.write_buffer_size / 8);
215
216 // Align up to 4k
217 const size_t align = 4 * 1024;
218 result.arena_block_size =
219 ((result.arena_block_size + align - 1) / align) * align;
220 }
221 result.min_write_buffer_number_to_merge =
222 std::min(result.min_write_buffer_number_to_merge,
223 result.max_write_buffer_number - 1);
224 if (result.min_write_buffer_number_to_merge < 1) {
225 result.min_write_buffer_number_to_merge = 1;
226 }
227
228 if (db_options.atomic_flush && result.min_write_buffer_number_to_merge > 1) {
229 ROCKS_LOG_WARN(
230 db_options.logger,
231 "Currently, if atomic_flush is true, then triggering flush for any "
232 "column family internally (non-manual flush) will trigger flushing "
233 "all column families even if the number of memtables is smaller "
234 "min_write_buffer_number_to_merge. Therefore, configuring "
235 "min_write_buffer_number_to_merge > 1 is not compatible and should "
236 "be satinized to 1. Not doing so will lead to data loss and "
237 "inconsistent state across multiple column families when WAL is "
238 "disabled, which is a common setting for atomic flush");
239
240 result.min_write_buffer_number_to_merge = 1;
241 }
242
243 if (result.num_levels < 1) {
244 result.num_levels = 1;
245 }
246 if (result.compaction_style == kCompactionStyleLevel &&
247 result.num_levels < 2) {
248 result.num_levels = 2;
249 }
250
251 if (result.compaction_style == kCompactionStyleUniversal &&
252 db_options.allow_ingest_behind && result.num_levels < 3) {
253 result.num_levels = 3;
254 }
255
256 if (result.max_write_buffer_number < 2) {
257 result.max_write_buffer_number = 2;
258 }
259 // fall back max_write_buffer_number_to_maintain if
260 // max_write_buffer_size_to_maintain is not set
261 if (result.max_write_buffer_size_to_maintain < 0) {
262 result.max_write_buffer_size_to_maintain =
263 result.max_write_buffer_number *
264 static_cast<int64_t>(result.write_buffer_size);
265 } else if (result.max_write_buffer_size_to_maintain == 0 &&
266 result.max_write_buffer_number_to_maintain < 0) {
267 result.max_write_buffer_number_to_maintain = result.max_write_buffer_number;
268 }
269 // bloom filter size shouldn't exceed 1/4 of memtable size.
270 if (result.memtable_prefix_bloom_size_ratio > 0.25) {
271 result.memtable_prefix_bloom_size_ratio = 0.25;
272 } else if (result.memtable_prefix_bloom_size_ratio < 0) {
273 result.memtable_prefix_bloom_size_ratio = 0;
274 }
275
276 if (!result.prefix_extractor) {
277 assert(result.memtable_factory);
278 Slice name = result.memtable_factory->Name();
279 if (name.compare("HashSkipListRepFactory") == 0 ||
280 name.compare("HashLinkListRepFactory") == 0) {
281 result.memtable_factory = std::make_shared<SkipListFactory>();
282 }
283 }
284
285 if (result.compaction_style == kCompactionStyleFIFO) {
286 // since we delete level0 files in FIFO compaction when there are too many
287 // of them, these options don't really mean anything
288 result.level0_slowdown_writes_trigger = std::numeric_limits<int>::max();
289 result.level0_stop_writes_trigger = std::numeric_limits<int>::max();
290 }
291
292 if (result.max_bytes_for_level_multiplier <= 0) {
293 result.max_bytes_for_level_multiplier = 1;
294 }
295
296 if (result.level0_file_num_compaction_trigger == 0) {
297 ROCKS_LOG_WARN(db_options.logger,
298 "level0_file_num_compaction_trigger cannot be 0");
299 result.level0_file_num_compaction_trigger = 1;
300 }
301
302 if (result.level0_stop_writes_trigger <
303 result.level0_slowdown_writes_trigger ||
304 result.level0_slowdown_writes_trigger <
305 result.level0_file_num_compaction_trigger) {
306 ROCKS_LOG_WARN(db_options.logger,
307 "This condition must be satisfied: "
308 "level0_stop_writes_trigger(%d) >= "
309 "level0_slowdown_writes_trigger(%d) >= "
310 "level0_file_num_compaction_trigger(%d)",
311 result.level0_stop_writes_trigger,
312 result.level0_slowdown_writes_trigger,
313 result.level0_file_num_compaction_trigger);
314 if (result.level0_slowdown_writes_trigger <
315 result.level0_file_num_compaction_trigger) {
316 result.level0_slowdown_writes_trigger =
317 result.level0_file_num_compaction_trigger;
318 }
319 if (result.level0_stop_writes_trigger <
320 result.level0_slowdown_writes_trigger) {
321 result.level0_stop_writes_trigger = result.level0_slowdown_writes_trigger;
322 }
323 ROCKS_LOG_WARN(db_options.logger,
324 "Adjust the value to "
325 "level0_stop_writes_trigger(%d) "
326 "level0_slowdown_writes_trigger(%d) "
327 "level0_file_num_compaction_trigger(%d)",
328 result.level0_stop_writes_trigger,
329 result.level0_slowdown_writes_trigger,
330 result.level0_file_num_compaction_trigger);
331 }
332
333 if (result.soft_pending_compaction_bytes_limit == 0) {
334 result.soft_pending_compaction_bytes_limit =
335 result.hard_pending_compaction_bytes_limit;
336 } else if (result.hard_pending_compaction_bytes_limit > 0 &&
337 result.soft_pending_compaction_bytes_limit >
338 result.hard_pending_compaction_bytes_limit) {
339 result.soft_pending_compaction_bytes_limit =
340 result.hard_pending_compaction_bytes_limit;
341 }
342
343 // When the DB is stopped, it's possible that there are some .trash files that
344 // were not deleted yet, when we open the DB we will find these .trash files
345 // and schedule them to be deleted (or delete immediately if SstFileManager
346 // was not used)
347 auto sfm =
348 static_cast<SstFileManagerImpl*>(db_options.sst_file_manager.get());
349 for (size_t i = 0; i < result.cf_paths.size(); i++) {
350 DeleteScheduler::CleanupDirectory(db_options.env, sfm,
351 result.cf_paths[i].path)
352 .PermitUncheckedError();
353 }
354
355 if (result.cf_paths.empty()) {
356 result.cf_paths = db_options.db_paths;
357 }
358
359 if (result.level_compaction_dynamic_level_bytes) {
360 if (result.compaction_style != kCompactionStyleLevel) {
361 ROCKS_LOG_INFO(db_options.info_log.get(),
362 "level_compaction_dynamic_level_bytes only makes sense "
363 "for level-based compaction");
364 result.level_compaction_dynamic_level_bytes = false;
365 } else if (result.cf_paths.size() > 1U) {
366 // we don't yet know how to make both of this feature and multiple
367 // DB path work.
368 ROCKS_LOG_WARN(db_options.info_log.get(),
369 "multiple cf_paths/db_paths and "
370 "level_compaction_dynamic_level_bytes "
371 "can't be used together");
372 result.level_compaction_dynamic_level_bytes = false;
373 }
374 }
375
376 if (result.max_compaction_bytes == 0) {
377 result.max_compaction_bytes = result.target_file_size_base * 25;
378 }
379
380 bool is_block_based_table = (result.table_factory->IsInstanceOf(
381 TableFactory::kBlockBasedTableName()));
382
383 const uint64_t kAdjustedTtl = 30 * 24 * 60 * 60;
384 if (result.ttl == kDefaultTtl) {
385 if (is_block_based_table) {
386 // FIFO also requires max_open_files=-1, which is checked in
387 // ValidateOptions().
388 result.ttl = kAdjustedTtl;
389 } else {
390 result.ttl = 0;
391 }
392 }
393
394 const uint64_t kAdjustedPeriodicCompSecs = 30 * 24 * 60 * 60;
395 if (result.compaction_style == kCompactionStyleLevel) {
396 if ((result.compaction_filter != nullptr ||
397 result.compaction_filter_factory != nullptr) &&
398 result.periodic_compaction_seconds == kDefaultPeriodicCompSecs &&
399 is_block_based_table) {
400 result.periodic_compaction_seconds = kAdjustedPeriodicCompSecs;
401 }
402 } else if (result.compaction_style == kCompactionStyleUniversal) {
403 if (result.periodic_compaction_seconds == kDefaultPeriodicCompSecs &&
404 is_block_based_table) {
405 result.periodic_compaction_seconds = kAdjustedPeriodicCompSecs;
406 }
407 } else if (result.compaction_style == kCompactionStyleFIFO) {
408 if (result.periodic_compaction_seconds != kDefaultPeriodicCompSecs) {
409 ROCKS_LOG_WARN(
410 db_options.info_log.get(),
411 "periodic_compaction_seconds does not support FIFO compaction. You"
412 "may want to set option TTL instead.");
413 }
414 if (result.last_level_temperature != Temperature::kUnknown) {
415 ROCKS_LOG_WARN(
416 db_options.info_log.get(),
417 "last_level_temperature is ignored with FIFO compaction. Consider "
418 "CompactionOptionsFIFO::file_temperature_age_thresholds.");
419 result.last_level_temperature = Temperature::kUnknown;
420 }
421 }
422
423 // For universal compaction, `ttl` and `periodic_compaction_seconds` mean the
424 // same thing, take the stricter value.
425 if (result.compaction_style == kCompactionStyleUniversal) {
426 if (result.periodic_compaction_seconds == 0) {
427 result.periodic_compaction_seconds = result.ttl;
428 } else if (result.ttl != 0) {
429 result.periodic_compaction_seconds =
430 std::min(result.ttl, result.periodic_compaction_seconds);
431 }
432 }
433
434 if (result.periodic_compaction_seconds == kDefaultPeriodicCompSecs) {
435 result.periodic_compaction_seconds = 0;
436 }
437
438 return result;
439}

463void SuperVersion::Cleanup() {
464 assert(refs.load(std::memory_order_relaxed) == 0);
465 // Since this SuperVersion object is being deleted,
466 // decrement reference to the immutable MemtableList
467 // this SV object was pointing to.
468 imm->Unref(&to_delete);
469 ReadOnlyMemTable* m = mem->Unref();
470 if (m != nullptr) {
471 auto* memory_usage = current->cfd()->imm()->current_memory_usage();
472 assert(*memory_usage >= m->ApproximateMemoryUsage());
473 *memory_usage -= m->ApproximateMemoryUsage();
474 to_delete.push_back(m);
475 }
476 current->Unref();
477 cfd->UnrefAndTryDelete();
478}

498void SuperVersionUnrefHandle(void* ptr) {
499 // UnrefHandle is called when a thread exits or a ThreadLocalPtr gets
500 // destroyed. When the former happens, the thread shouldn't see kSVInUse.
501 // When the latter happens, only super_version_ holds a reference
502 // to ColumnFamilyData, so no further queries are possible.
503 SuperVersion* sv = static_cast<SuperVersion*>(ptr);
504 bool was_last_ref __attribute__((__unused__));
505 was_last_ref = sv->Unref();
506 // Thread-local SuperVersions can't outlive ColumnFamilyData::super_version_.
507 // This is important because we can't do SuperVersion cleanup here.
508 // That would require locking DB mutex, which would deadlock because
509 // SuperVersionUnrefHandle is called with locked ThreadLocalPtr mutex.
510 assert(!was_last_ref);
511}

538 refs_(0),

540 dropped_(false),
541 flush_skip_reschedule_(false),
542 internal_comparator_(cf_options.comparator),
543 initial_cf_options_(SanitizeOptions(db_options, cf_options)),
544 ioptions_(db_options, initial_cf_options_),
545 mutable_cf_options_(initial_cf_options_),
546 is_delete_range_supported_(

548 write_buffer_manager_(write_buffer_manager),
549 mem_(nullptr),
550 imm_(ioptions_.min_write_buffer_number_to_merge,
551 ioptions_.max_write_buffer_number_to_maintain,
552 ioptions_.max_write_buffer_size_to_maintain),
553 super_version_(nullptr),
554 super_version_number_(0),
555 local_sv_(new ThreadLocalPtr(&SuperVersionUnrefHandle)),
556 next_(nullptr),
557 prev_(nullptr),
558 log_number_(0),
559 column_family_set_(column_family_set),
560 queued_for_flush_(false),
561 queued_for_compaction_(false),
562 prev_compaction_needed_bytes_(0),
563 allow_2pc_(db_options.allow_2pc),
564 last_memtable_id_(0),
565 db_paths_registered_(false),
566 mempurge_used_(false),
567 next_epoch_number_(1) {
568 if (id_ != kDummyColumnFamilyDataId) {
569 // TODO(cc): RegisterDbPaths can be expensive, considering moving it
570 // outside of this constructor which might be called with db mutex held.
571 // TODO(cc): considering using ioptions_.fs, currently some tests rely on
572 // EnvWrapper, that's the main reason why we use env here.
573 Status s = ioptions_.env->RegisterDbPaths(GetDbPaths());
574 if (s.ok()) {
575 db_paths_registered_ = true;
576 } else {
577 ROCKS_LOG_ERROR(
578 ioptions_.logger,
579 "Failed to register data paths of column family (id: %d, name: %s)",
580 id_, name_.c_str());
581 }
582 }
583 Ref();
584
585 // Convert user defined table properties collector factories to internal ones.
586 GetInternalTblPropCollFactory(ioptions_, &internal_tbl_prop_coll_factories_);
587
588 // if _dummy_versions is nullptr, then this is a dummy column family.
589 if (_dummy_versions != nullptr) {
590 internal_stats_.reset(
591 new InternalStats(ioptions_.num_levels, ioptions_.clock, this));
592 table_cache_.reset(new TableCache(ioptions_, file_options, _table_cache,
593 block_cache_tracer, io_tracer,
594 db_session_id));
595 blob_file_cache_.reset(
596 new BlobFileCache(_table_cache, ioptions(), soptions(), id_,
597 internal_stats_->GetBlobFileReadHist(), io_tracer));
598 blob_source_.reset(new BlobSource(ioptions_, mutable_cf_options_, db_id,
599 db_session_id, blob_file_cache_.get()));
600
601 if (ioptions_.compaction_style == kCompactionStyleLevel) {
602 compaction_picker_.reset(
603 new LevelCompactionPicker(ioptions_, &internal_comparator_));
604 } else if (ioptions_.compaction_style == kCompactionStyleUniversal) {
605 compaction_picker_.reset(
606 new UniversalCompactionPicker(ioptions_, &internal_comparator_));
607 } else if (ioptions_.compaction_style == kCompactionStyleFIFO) {
608 compaction_picker_.reset(
609 new FIFOCompactionPicker(ioptions_, &internal_comparator_));
610 } else if (ioptions_.compaction_style == kCompactionStyleNone) {
611 compaction_picker_.reset(

623 new LevelCompactionPicker(ioptions_, &internal_comparator_));

634 }

636 RecalculateWriteStallConditions(mutable_cf_options_);

638 if (cf_options.table_factory->IsInstanceOf(

640 cf_options.table_factory->GetOptions<BlockBasedTableOptions>()) {

777const double kIncSlowdownRatio = 0.8;
778const double kDecSlowdownRatio = 1 / kIncSlowdownRatio;
779const double kNearStopSlowdownRatio = 0.6;
780const double kDelayRecoverSlowdownRatio = 1.4;
781
782namespace {
783// If penalize_stop is true, we further reduce slowdown rate.
784std::unique_ptr<WriteControllerToken> SetupDelay(
785 WriteController* write_controller, uint64_t compaction_needed_bytes,
786 uint64_t prev_compaction_need_bytes, bool penalize_stop,
787 bool auto_compactions_disabled) {
788 const uint64_t kMinWriteRate = 16 * 1024u; // Minimum write rate 16KB/s.
789
790 uint64_t max_write_rate = write_controller->max_delayed_write_rate();
791 uint64_t write_rate = write_controller->delayed_write_rate();
792
793 if (auto_compactions_disabled) {
794 // When auto compaction is disabled, always use the value user gave.
795 write_rate = max_write_rate;
796 } else if (write_controller->NeedsDelay() && max_write_rate > kMinWriteRate) {
797 // If user gives rate less than kMinWriteRate, don't adjust it.
798 //
799 // If already delayed, need to adjust based on previous compaction debt.
800 // When there are two or more column families require delay, we always
801 // increase or reduce write rate based on information for one single
802 // column family. It is likely to be OK but we can improve if there is a
803 // problem.
804 // Ignore compaction_needed_bytes = 0 case because compaction_needed_bytes
805 // is only available in level-based compaction
806 //
807 // If the compaction debt stays the same as previously, we also further slow
808 // down. It usually means a mem table is full. It's mainly for the case
809 // where both of flush and compaction are much slower than the speed we
810 // insert to mem tables, so we need to actively slow down before we get
811 // feedback signal from compaction and flushes to avoid the full stop
812 // because of hitting the max write buffer number.
813 //
814 // If DB just falled into the stop condition, we need to further reduce
815 // the write rate to avoid the stop condition.
816 if (penalize_stop) {
817 // Penalize the near stop or stop condition by more aggressive slowdown.
818 // This is to provide the long term slowdown increase signal.
819 // The penalty is more than the reward of recovering to the normal
820 // condition.
821 write_rate = static_cast<uint64_t>(static_cast<double>(write_rate) *
822 kNearStopSlowdownRatio);
823 if (write_rate < kMinWriteRate) {
824 write_rate = kMinWriteRate;
825 }
826 } else if (prev_compaction_need_bytes > 0 &&
827 prev_compaction_need_bytes <= compaction_needed_bytes) {
828 write_rate = static_cast<uint64_t>(static_cast<double>(write_rate) *
829 kIncSlowdownRatio);
830 if (write_rate < kMinWriteRate) {
831 write_rate = kMinWriteRate;
832 }
833 } else if (prev_compaction_need_bytes > compaction_needed_bytes) {
834 // We are speeding up by ratio of kSlowdownRatio when we have paid
835 // compaction debt. But we'll never speed up to faster than the write rate
836 // given by users.
837 write_rate = static_cast<uint64_t>(static_cast<double>(write_rate) *
838 kDecSlowdownRatio);
839 if (write_rate > max_write_rate) {
840 write_rate = max_write_rate;
841 }
842 }
843 }
844 return write_controller->GetDelayToken(write_rate);
845}

847int GetL0FileCountForCompactionSpeedup(int level0_file_num_compaction_trigger,
848 int level0_slowdown_writes_trigger) {
849 // SanitizeOptions() ensures it.
850 assert(level0_file_num_compaction_trigger <= level0_slowdown_writes_trigger);
851
852 if (level0_file_num_compaction_trigger < 0) {
853 return std::numeric_limits<int>::max();
854 }
855
856 const int64_t twice_level0_trigger =
857 static_cast<int64_t>(level0_file_num_compaction_trigger) * 2;
858
859 const int64_t one_fourth_trigger_slowdown =
860 static_cast<int64_t>(level0_file_num_compaction_trigger) +
861 ((level0_slowdown_writes_trigger - level0_file_num_compaction_trigger) /
862 4);
863
864 assert(twice_level0_trigger >= 0);
865 assert(one_fourth_trigger_slowdown >= 0);
866
867 // 1/4 of the way between L0 compaction trigger threshold and slowdown
868 // condition.
869 // Or twice as compaction trigger, if it is smaller.
870 int64_t res = std::min(twice_level0_trigger, one_fourth_trigger_slowdown);
871 if (res >= std::numeric_limits<int32_t>::max()) {
872 return std::numeric_limits<int32_t>::max();
873 } else {
874 // res fits in int
875 return static_cast<int>(res);
876 }
877}

879uint64_t GetPendingCompactionBytesForCompactionSpeedup(
880 const MutableCFOptions& mutable_cf_options,
881 const VersionStorageInfo* vstorage) {
882 // Compaction debt relatively large compared to the stable (bottommost) data
883 // size indicates compaction fell behind.
884 const uint64_t kBottommostSizeDivisor = 8;
885 // Meaningful progress toward the slowdown trigger is another good indication.
886 const uint64_t kSlowdownTriggerDivisor = 4;
887
888 uint64_t bottommost_files_size = 0;
889 for (const auto& level_and_file : vstorage->BottommostFiles()) {
890 bottommost_files_size += level_and_file.second->fd.GetFileSize();
891 }
892
893 // Slowdown trigger might be zero but that means compaction speedup should
894 // always happen (undocumented/historical), so no special treatment is needed.
895 uint64_t slowdown_threshold =
896 mutable_cf_options.soft_pending_compaction_bytes_limit /
897 kSlowdownTriggerDivisor;
898
899 // Size of zero, however, should not be used to decide to speedup compaction.
900 if (bottommost_files_size == 0) {
901 return slowdown_threshold;
902 }
903
904 // Prevent a small CF from triggering parallel compactions for other CFs.
905 // Require compaction debt to be more than a full L0 to Lbase compaction.
906 const uint64_t kMinDebtSize = 2 * mutable_cf_options.max_bytes_for_level_base;
907 uint64_t size_threshold =
908 std::max(bottommost_files_size / kBottommostSizeDivisor, kMinDebtSize);
909 return std::min(size_threshold, slowdown_threshold);
910}

912uint64_t GetMarkedFileCountForCompactionSpeedup() {
913 // When just one file is marked, it is not clear that parallel compaction will
914 // help the compaction that the user nicely requested to happen sooner. When
915 // multiple files are marked, however, it is pretty clearly helpful, except
916 // for the rare case in which a single compaction grabs all the marked files.
917 return 2;
918}

921std::pair<WriteStallCondition, WriteStallCause>
922ColumnFamilyData::GetWriteStallConditionAndCause(
923 int num_unflushed_memtables, int num_l0_files,
924 uint64_t num_compaction_needed_bytes,
925 const MutableCFOptions& mutable_cf_options,
926 const ImmutableCFOptions& immutable_cf_options) {
927 if (num_unflushed_memtables >= mutable_cf_options.max_write_buffer_number) {
928 return {WriteStallCondition::kStopped, WriteStallCause::kMemtableLimit};
929 } else if (!mutable_cf_options.disable_auto_compactions &&
930 num_l0_files >= mutable_cf_options.level0_stop_writes_trigger) {
931 return {WriteStallCondition::kStopped, WriteStallCause::kL0FileCountLimit};
932 } else if (!mutable_cf_options.disable_auto_compactions &&
933 mutable_cf_options.hard_pending_compaction_bytes_limit > 0 &&
934 num_compaction_needed_bytes >=
935 mutable_cf_options.hard_pending_compaction_bytes_limit) {
936 return {WriteStallCondition::kStopped,
937 WriteStallCause::kPendingCompactionBytes};
938 } else if (mutable_cf_options.max_write_buffer_number > 3 &&
939 num_unflushed_memtables >=
940 mutable_cf_options.max_write_buffer_number - 1 &&
941 num_unflushed_memtables - 1 >=
942 immutable_cf_options.min_write_buffer_number_to_merge) {
943 return {WriteStallCondition::kDelayed, WriteStallCause::kMemtableLimit};
944 } else if (!mutable_cf_options.disable_auto_compactions &&
945 mutable_cf_options.level0_slowdown_writes_trigger >= 0 &&
946 num_l0_files >=
947 mutable_cf_options.level0_slowdown_writes_trigger) {
948 return {WriteStallCondition::kDelayed, WriteStallCause::kL0FileCountLimit};
949 } else if (!mutable_cf_options.disable_auto_compactions &&
950 mutable_cf_options.soft_pending_compaction_bytes_limit > 0 &&
951 num_compaction_needed_bytes >=
952 mutable_cf_options.soft_pending_compaction_bytes_limit) {
953 return {WriteStallCondition::kDelayed,
954 WriteStallCause::kPendingCompactionBytes};
955 }
956 return {WriteStallCondition::kNormal, WriteStallCause::kNone};
957}

959WriteStallCondition ColumnFamilyData::RecalculateWriteStallConditions(
960 const MutableCFOptions& mutable_cf_options) {
961 auto write_stall_condition = WriteStallCondition::kNormal;
962 if (current_ != nullptr) {
963 auto* vstorage = current_->storage_info();
964 auto write_controller = column_family_set_->write_controller_;
965 uint64_t compaction_needed_bytes =
966 vstorage->estimated_compaction_needed_bytes();
967
968 auto write_stall_condition_and_cause = GetWriteStallConditionAndCause(
969 imm()->NumNotFlushed(), vstorage->l0_delay_trigger_count(),
970 vstorage->estimated_compaction_needed_bytes(), mutable_cf_options,
971 *ioptions());
972 write_stall_condition = write_stall_condition_and_cause.first;
973 auto write_stall_cause = write_stall_condition_and_cause.second;
974
975 bool was_stopped = write_controller->IsStopped();
976 bool needed_delay = write_controller->NeedsDelay();
977
978 if (write_stall_condition == WriteStallCondition::kStopped &&
979 write_stall_cause == WriteStallCause::kMemtableLimit) {
980 write_controller_token_ = write_controller->GetStopToken();
981 internal_stats_->AddCFStats(InternalStats::MEMTABLE_LIMIT_STOPS, 1);
982 ROCKS_LOG_WARN(
983 ioptions_.logger,
984 "[%s] Stopping writes because we have %d immutable memtables "
985 "(waiting for flush), max_write_buffer_number is set to %d",
986 name_.c_str(), imm()->NumNotFlushed(),
987 mutable_cf_options.max_write_buffer_number);
988 } else if (write_stall_condition == WriteStallCondition::kStopped &&
989 write_stall_cause == WriteStallCause::kL0FileCountLimit) {
990 write_controller_token_ = write_controller->GetStopToken();
991 internal_stats_->AddCFStats(InternalStats::L0_FILE_COUNT_LIMIT_STOPS, 1);
992 if (compaction_picker_->IsLevel0CompactionInProgress()) {
993 internal_stats_->AddCFStats(
994 InternalStats::L0_FILE_COUNT_LIMIT_STOPS_WITH_ONGOING_COMPACTION,
995 1);
996 }
997 ROCKS_LOG_WARN(ioptions_.logger,
998 "[%s] Stopping writes because we have %d level-0 files",
999 name_.c_str(), vstorage->l0_delay_trigger_count());
1000 } else if (write_stall_condition == WriteStallCondition::kStopped &&
1001 write_stall_cause == WriteStallCause::kPendingCompactionBytes) {
1002 write_controller_token_ = write_controller->GetStopToken();
1003 internal_stats_->AddCFStats(
1004 InternalStats::PENDING_COMPACTION_BYTES_LIMIT_STOPS, 1);
1005 ROCKS_LOG_WARN(
1006 ioptions_.logger,
1007 "[%s] Stopping writes because of estimated pending compaction "
1008 "bytes %" PRIu64,
1009 name_.c_str(), compaction_needed_bytes);
1010 } else if (write_stall_condition == WriteStallCondition::kDelayed &&
1011 write_stall_cause == WriteStallCause::kMemtableLimit) {
1012 write_controller_token_ =
1013 SetupDelay(write_controller, compaction_needed_bytes,
1014 prev_compaction_needed_bytes_, was_stopped,
1015 mutable_cf_options.disable_auto_compactions);
1016 internal_stats_->AddCFStats(InternalStats::MEMTABLE_LIMIT_DELAYS, 1);
1017 ROCKS_LOG_WARN(
1018 ioptions_.logger,
1019 "[%s] Stalling writes because we have %d immutable memtables "
1020 "(waiting for flush), max_write_buffer_number is set to %d "
1021 "rate %" PRIu64,
1022 name_.c_str(), imm()->NumNotFlushed(),
1023 mutable_cf_options.max_write_buffer_number,
1024 write_controller->delayed_write_rate());
1025 } else if (write_stall_condition == WriteStallCondition::kDelayed &&
1026 write_stall_cause == WriteStallCause::kL0FileCountLimit) {
1027 // L0 is the last two files from stopping.
1028 bool near_stop = vstorage->l0_delay_trigger_count() >=
1029 mutable_cf_options.level0_stop_writes_trigger - 2;
1030 write_controller_token_ =
1031 SetupDelay(write_controller, compaction_needed_bytes,
1032 prev_compaction_needed_bytes_, was_stopped || near_stop,
1033 mutable_cf_options.disable_auto_compactions);
1034 internal_stats_->AddCFStats(InternalStats::L0_FILE_COUNT_LIMIT_DELAYS, 1);
1035 if (compaction_picker_->IsLevel0CompactionInProgress()) {
1036 internal_stats_->AddCFStats(
1037 InternalStats::L0_FILE_COUNT_LIMIT_DELAYS_WITH_ONGOING_COMPACTION,
1038 1);
1039 }
1040 ROCKS_LOG_WARN(ioptions_.logger,
1041 "[%s] Stalling writes because we have %d level-0 files "
1042 "rate %" PRIu64,
1043 name_.c_str(), vstorage->l0_delay_trigger_count(),
1044 write_controller->delayed_write_rate());
1045 } else if (write_stall_condition == WriteStallCondition::kDelayed &&
1046 write_stall_cause == WriteStallCause::kPendingCompactionBytes) {
1047 // If the distance to hard limit is less than 1/4 of the gap between soft
1048 // and
1049 // hard bytes limit, we think it is near stop and speed up the slowdown.
1050 bool near_stop =
1051 mutable_cf_options.hard_pending_compaction_bytes_limit > 0 &&
1052 (compaction_needed_bytes -
1053 mutable_cf_options.soft_pending_compaction_bytes_limit) >
1054 3 *
1055 (mutable_cf_options.hard_pending_compaction_bytes_limit -
1056 mutable_cf_options.soft_pending_compaction_bytes_limit) /
1057 4;
1058
1059 write_controller_token_ =
1060 SetupDelay(write_controller, compaction_needed_bytes,
1061 prev_compaction_needed_bytes_, was_stopped || near_stop,
1062 mutable_cf_options.disable_auto_compactions);
1063 internal_stats_->AddCFStats(
1064 InternalStats::PENDING_COMPACTION_BYTES_LIMIT_DELAYS, 1);
1065 ROCKS_LOG_WARN(
1066 ioptions_.logger,
1067 "[%s] Stalling writes because of estimated pending compaction "
1068 "bytes %" PRIu64 " rate %" PRIu64,
1069 name_.c_str(), vstorage->estimated_compaction_needed_bytes(),
1070 write_controller->delayed_write_rate());
1071 } else {
1072 assert(write_stall_condition == WriteStallCondition::kNormal);
1073 if (vstorage->l0_delay_trigger_count() >=
1074 GetL0FileCountForCompactionSpeedup(
1075 mutable_cf_options.level0_file_num_compaction_trigger,
1076 mutable_cf_options.level0_slowdown_writes_trigger)) {
1077 write_controller_token_ =
1078 write_controller->GetCompactionPressureToken();
1079 ROCKS_LOG_INFO(
1080 ioptions_.logger,
1081 "[%s] Increasing compaction threads because we have %d level-0 "
1082 "files ",
1083 name_.c_str(), vstorage->l0_delay_trigger_count());
1084 } else if (mutable_cf_options.soft_pending_compaction_bytes_limit == 0) {
1085 // If soft pending compaction byte limit is not set, always speed up
1086 // compaction.
1087 write_controller_token_ =
1088 write_controller->GetCompactionPressureToken();
1089 } else if (vstorage->estimated_compaction_needed_bytes() >=
1090 GetPendingCompactionBytesForCompactionSpeedup(
1091 mutable_cf_options, vstorage)) {
1092 write_controller_token_ =
1093 write_controller->GetCompactionPressureToken();
1094 ROCKS_LOG_INFO(
1095 ioptions_.logger,
1096 "[%s] Increasing compaction threads because of estimated pending "
1097 "compaction "
1098 "bytes %" PRIu64,
1099 name_.c_str(), vstorage->estimated_compaction_needed_bytes());
1100 } else if (uint64_t(vstorage->FilesMarkedForCompaction().size()) >=
1101 GetMarkedFileCountForCompactionSpeedup()) {
1102 write_controller_token_ =
1103 write_controller->GetCompactionPressureToken();
1104 ROCKS_LOG_INFO(
1105 ioptions_.logger,
1106 "[%s] Increasing compaction threads because we have %" PRIu64
1107 " files marked for compaction",
1108 name_.c_str(),
1109 uint64_t(vstorage->FilesMarkedForCompaction().size()));
1110 } else {
1111 write_controller_token_.reset();
1112 }
1113 // If the DB recovers from delay conditions, we reward with reducing
1114 // double the slowdown ratio. This is to balance the long term slowdown
1115 // increase signal.
1116 if (needed_delay) {
1117 uint64_t write_rate = write_controller->delayed_write_rate();
1118 write_controller->set_delayed_write_rate(static_cast<uint64_t>(
1119 static_cast<double>(write_rate) * kDelayRecoverSlowdownRatio));
1120 // Set the low pri limit to be 1/4 the delayed write rate.
1121 // Note we don't reset this value even after delay condition is relased.
1122 // Low-pri rate will continue to apply if there is a compaction
1123 // pressure.
1124 write_controller->low_pri_rate_limiter()->SetBytesPerSecond(write_rate /
1125 4);
1126 }
1127 }
1128 prev_compaction_needed_bytes_ = compaction_needed_bytes;
1129 }
1130 return write_stall_condition;
1131}

1198Status ColumnFamilyData::RangesOverlapWithMemtables(
1199 const autovector<UserKeyRange>& ranges, SuperVersion* super_version,
1200 bool allow_data_in_errors, bool* overlap) {
1201 assert(overlap != nullptr);
1202 *overlap = false;
1203 // Create an InternalIterator over all unflushed memtables
1204 Arena arena;
1205 // TODO: plumb Env::IOActivity, Env::IOPriority
1206 ReadOptions read_opts;
1207 read_opts.total_order_seek = true;
1208 MergeIteratorBuilder merge_iter_builder(&internal_comparator_, &arena);
1209 merge_iter_builder.AddIterator(super_version->mem->NewIterator(
1210 read_opts, /*seqno_to_time_mapping=*/nullptr, &arena,
1211 /*prefix_extractor=*/nullptr, /*for_flush=*/false));
1212 super_version->imm->AddIterators(read_opts, /*seqno_to_time_mapping=*/nullptr,
1213 /*prefix_extractor=*/nullptr,
1214 &merge_iter_builder,
1215 false /* add_range_tombstone_iter */);
1216 ScopedArenaPtr<InternalIterator> memtable_iter(merge_iter_builder.Finish());
1217
1218 auto read_seq = super_version->current->version_set()->LastSequence();
1219 ReadRangeDelAggregator range_del_agg(&internal_comparator_, read_seq);
1220 auto* active_range_del_iter = super_version->mem->NewRangeTombstoneIterator(
1221 read_opts, read_seq, false /* immutable_memtable */);
1222 range_del_agg.AddTombstones(
1223 std::unique_ptr<FragmentedRangeTombstoneIterator>(active_range_del_iter));
1224 Status status;
1225 status = super_version->imm->AddRangeTombstoneIterators(
1226 read_opts, nullptr /* arena */, &range_del_agg);
1227 // AddRangeTombstoneIterators always return Status::OK.
1228 assert(status.ok());
1229
1230 for (size_t i = 0; i < ranges.size() && status.ok() && !*overlap; ++i) {
1231 auto* vstorage = super_version->current->storage_info();
1232 auto* ucmp = vstorage->InternalComparator()->user_comparator();
1233 InternalKey range_start(ranges[i].start, kMaxSequenceNumber,
1234 kValueTypeForSeek);
1235 memtable_iter->Seek(range_start.Encode());
1236 status = memtable_iter->status();
1237 ParsedInternalKey seek_result;
1238
1239 if (status.ok() && memtable_iter->Valid()) {
1240 status = ParseInternalKey(memtable_iter->key(), &seek_result,
1241 allow_data_in_errors);
1242 }
1243
1244 if (status.ok()) {
1245 if (memtable_iter->Valid() &&
1246 ucmp->CompareWithoutTimestamp(seek_result.user_key,
1247 ranges[i].limit) <= 0) {
1248 *overlap = true;
1249 } else if (range_del_agg.IsRangeOverlapped(ranges[i].start,
1250 ranges[i].limit)) {
1251 *overlap = true;
1252 }
1253 }
1254 }
1255 return status;
1256}

1292
1293SuperVersion* ColumnFamilyData::GetThreadLocalSuperVersion(DBImpl* db) {
1294 // The SuperVersion is cached in thread local storage to avoid acquiring
1295 // mutex when SuperVersion does not change since the last use. When a new
1296 // SuperVersion is installed, the compaction or flush thread cleans up
1297 // cached SuperVersion in all existing thread local storage. To avoid
1298 // acquiring mutex for this operation, we use atomic Swap() on the thread
1299 // local pointer to guarantee exclusive access. If the thread local pointer
1300 // is being used while a new SuperVersion is installed, the cached
1301 // SuperVersion can become stale. In that case, the background thread would
1302 // have swapped in kSVObsolete. We re-check the value at when returning
1303 // SuperVersion back to thread local, with an atomic compare and swap.
1304 // The superversion will need to be released if detected to be stale.
1305 void* ptr = local_sv_->Swap(SuperVersion::kSVInUse);
1306 // Invariant:
1307 // (1) Scrape (always) installs kSVObsolete in ThreadLocal storage
1308 // (2) the Swap above (always) installs kSVInUse, ThreadLocal storage
1309 // should only keep kSVInUse before ReturnThreadLocalSuperVersion call
1310 // (if no Scrape happens).
1311 assert(ptr != SuperVersion::kSVInUse);
1312 SuperVersion* sv = static_cast<SuperVersion*>(ptr);
1313 if (sv == SuperVersion::kSVObsolete) {
1314 RecordTick(ioptions_.stats, NUMBER_SUPERVERSION_ACQUIRES);
1315 db->mutex()->Lock();
1316 sv = super_version_->Ref();
1317 db->mutex()->Unlock();
1318 }
1319 assert(sv != nullptr);
1320 return sv;
1321}
1322
1323bool ColumnFamilyData::ReturnThreadLocalSuperVersion(SuperVersion* sv) {
1324 assert(sv != nullptr);
1325 // Put the SuperVersion back
1326 void* expected = SuperVersion::kSVInUse;
1327 if (local_sv_->CompareAndSwap(static_cast<void*>(sv), expected)) {
1328 // When we see kSVInUse in the ThreadLocal, we are sure ThreadLocal
1329 // storage has not been altered and no Scrape has happened. The
1330 // SuperVersion is still current.
1331 return true;
1332 } else {
1333 // ThreadLocal scrape happened in the process of this GetImpl call (after
1334 // thread local Swap() at the beginning and before CompareAndSwap()).
1335 // This means the SuperVersion it holds is obsolete.
1336 assert(expected == SuperVersion::kSVObsolete);
1337 }
1338 return false;
1339}

1347void ColumnFamilyData::InstallSuperVersion(
1348 SuperVersionContext* sv_context,
1349 const MutableCFOptions& mutable_cf_options) {
1350 SuperVersion* new_superversion = sv_context->new_superversion.release();
1351 new_superversion->mutable_cf_options = mutable_cf_options;
1352 new_superversion->Init(this, mem_, imm_.current(), current_,
1353 sv_context->new_seqno_to_time_mapping
1354 ? std::move(sv_context->new_seqno_to_time_mapping)
1355 : super_version_
1356 ? super_version_->ShareSeqnoToTimeMapping()
1357 : nullptr);
1358 SuperVersion* old_superversion = super_version_;
1359 super_version_ = new_superversion;
1360 if (old_superversion == nullptr || old_superversion->current != current() ||
1361 old_superversion->mem != mem_ ||
1362 old_superversion->imm != imm_.current()) {
1363 // Should not recalculate slow down condition if nothing has changed, since
1364 // currently RecalculateWriteStallConditions() treats it as further slowing
1365 // down is needed.
1366 super_version_->write_stall_condition =
1367 RecalculateWriteStallConditions(mutable_cf_options);
1368 } else {
1369 super_version_->write_stall_condition =
1370 old_superversion->write_stall_condition;
1371 }
1372 if (old_superversion != nullptr) {
1373 // Reset SuperVersions cached in thread local storage.
1374 // This should be done before old_superversion->Unref(). That's to ensure
1375 // that local_sv_ never holds the last reference to SuperVersion, since
1376 // it has no means to safely do SuperVersion cleanup.
1377 ResetThreadLocalSuperVersions();
1378
1379 if (old_superversion->mutable_cf_options.write_buffer_size !=
1380 mutable_cf_options.write_buffer_size) {
1381 mem_->UpdateWriteBufferSize(mutable_cf_options.write_buffer_size);
1382 }
1383 if (old_superversion->write_stall_condition !=
1384 new_superversion->write_stall_condition) {
1385 sv_context->PushWriteStallNotification(
1386 old_superversion->write_stall_condition,
1387 new_superversion->write_stall_condition, GetName(), ioptions());
1388 }
1389 if (old_superversion->Unref()) {
1390 old_superversion->Cleanup();
1391 sv_context->superversions_to_free.push_back(old_superversion);
1392 }
1393 }
1394 ++super_version_number_;
1395 super_version_->version_number = super_version_number_;
1396}
1397
1398void ColumnFamilyData::ResetThreadLocalSuperVersions() {
1399 autovector<void*> sv_ptrs;
1400 local_sv_->Scrape(&sv_ptrs, SuperVersion::kSVObsolete);
1401 for (auto ptr : sv_ptrs) {
1402 assert(ptr);
1403 if (ptr == SuperVersion::kSVInUse) {
1404 continue;
1405 }
1406 auto sv = static_cast<SuperVersion*>(ptr);
1407 bool was_last_ref __attribute__((__unused__));
1408 was_last_ref = sv->Unref();
1409 // sv couldn't have been the last reference because
1410 // ResetThreadLocalSuperVersions() is called before
1411 // unref'ing super_version_.
1412 assert(!was_last_ref);
1413 }

1416Status ColumnFamilyData::ValidateOptions(
1417 const DBOptions& db_options, const ColumnFamilyOptions& cf_options) {
1418 Status s;
1419 s = CheckCompressionSupported(cf_options);
1420 if (s.ok() && db_options.allow_concurrent_memtable_write) {
1421 s = CheckConcurrentWritesSupported(cf_options);
1422 }
1423 if (s.ok() && db_options.unordered_write &&
1424 cf_options.max_successive_merges != 0) {
1425 s = Status::InvalidArgument(
1426 "max_successive_merges > 0 is incompatible with unordered_write");
1427 }
1428 if (s.ok()) {
1429 s = CheckCFPathsSupported(db_options, cf_options);
1430 }
1431 if (!s.ok()) {
1432 return s;
1433 }
1434
1435 if (cf_options.ttl > 0 && cf_options.ttl != kDefaultTtl) {
1436 if (!cf_options.table_factory->IsInstanceOf(
1437 TableFactory::kBlockBasedTableName())) {
1438 return Status::NotSupported(
1439 "TTL is only supported in Block-Based Table format. ");
1440 }
1441 }
1442
1443 if (cf_options.periodic_compaction_seconds > 0 &&
1444 cf_options.periodic_compaction_seconds != kDefaultPeriodicCompSecs) {
1445 if (!cf_options.table_factory->IsInstanceOf(
1446 TableFactory::kBlockBasedTableName())) {
1447 return Status::NotSupported(
1448 "Periodic Compaction is only supported in "
1449 "Block-Based Table format. ");
1450 }
1451 }
1452
1453 const auto* ucmp = cf_options.comparator;
1454 assert(ucmp);
1455 if (ucmp->timestamp_size() > 0 &&
1456 !cf_options.persist_user_defined_timestamps) {
1457 if (db_options.atomic_flush) {
1458 return Status::NotSupported(
1459 "Not persisting user-defined timestamps feature is not supported"
1460 "in combination with atomic flush.");
1461 }
1462 if (db_options.allow_concurrent_memtable_write) {
1463 return Status::NotSupported(
1464 "Not persisting user-defined timestamps feature is not supported"
1465 " in combination with concurrent memtable write.");
1466 }
1467 const char* comparator_name = cf_options.comparator->Name();
1468 size_t name_size = strlen(comparator_name);
1469 const char* suffix = ".u64ts";
1470 size_t suffix_size = strlen(suffix);
1471 if (name_size <= suffix_size ||
1472 strcmp(comparator_name + name_size - suffix_size, suffix) != 0) {
1473 return Status::NotSupported(
1474 "Not persisting user-defined timestamps"
1475 "feature only support user-defined timestamps formatted as "
1476 "uint64_t.");
1477 }
1478 }
1479
1480 if (cf_options.enable_blob_garbage_collection) {
1481 if (cf_options.blob_garbage_collection_age_cutoff < 0.0 ||
1482 cf_options.blob_garbage_collection_age_cutoff > 1.0) {
1483 return Status::InvalidArgument(
1484 "The age cutoff for blob garbage collection should be in the range "
1485 "[0.0, 1.0].");
1486 }
1487 if (cf_options.blob_garbage_collection_force_threshold < 0.0 ||
1488 cf_options.blob_garbage_collection_force_threshold > 1.0) {
1489 return Status::InvalidArgument(
1490 "The garbage ratio threshold for forcing blob garbage collection "
1491 "should be in the range [0.0, 1.0].");
1492 }
1493 }
1494
1495 if (cf_options.compaction_style == kCompactionStyleFIFO &&
1496 db_options.max_open_files != -1 && cf_options.ttl > 0) {
1497 return Status::NotSupported(
1498 "FIFO compaction only supported with max_open_files = -1.");
1499 }
1500
1501 std::vector<uint32_t> supported{0, 1, 2, 4, 8};
1502 if (std::find(supported.begin(), supported.end(),
1503 cf_options.memtable_protection_bytes_per_key) ==
1504 supported.end()) {
1505 return Status::NotSupported(
1506 "Memtable per key-value checksum protection only supports 0, 1, 2, 4 "
1507 "or 8 bytes per key.");
1508 }
1509 if (std::find(supported.begin(), supported.end(),
1510 cf_options.block_protection_bytes_per_key) == supported.end()) {
1511 return Status::NotSupported(
1512 "Block per key-value checksum protection only supports 0, 1, 2, 4 "
1513 "or 8 bytes per key.");
1514 }
1515
1516 if (!cf_options.compaction_options_fifo.file_temperature_age_thresholds
1517 .empty()) {
1518 if (cf_options.compaction_style != kCompactionStyleFIFO) {
1519 return Status::NotSupported(
1520 "Option file_temperature_age_thresholds only supports FIFO "
1521 "compaction.");
1522 } else if (cf_options.num_levels > 1) {
1523 return Status::NotSupported(
1524 "Option file_temperature_age_thresholds is only supported when "
1525 "num_levels = 1.");
1526 } else {
1527 const auto& ages =
1528 cf_options.compaction_options_fifo.file_temperature_age_thresholds;
1529 assert(ages.size() >= 1);
1530 // check that age is sorted
1531 for (size_t i = 0; i < ages.size() - 1; ++i) {
1532 if (ages[i].age >= ages[i + 1].age) {
1533 return Status::NotSupported(
1534 "Option file_temperature_age_thresholds requires elements to be "
1535 "sorted in increasing order with respect to `age` field.");
1536 }
1537 }
1538 }
1539 }
1540
1541 if (cf_options.compaction_style == kCompactionStyleUniversal) {
1542 int max_read_amp = cf_options.compaction_options_universal.max_read_amp;
1543 if (max_read_amp < -1) {
1544 return Status::NotSupported(
1545 "CompactionOptionsUniversal::max_read_amp should be at least -1.");
1546 } else if (0 < max_read_amp &&
1547 max_read_amp < cf_options.level0_file_num_compaction_trigger) {
1548 return Status::NotSupported(
1549 "CompactionOptionsUniversal::max_read_amp limits the number of sorted"
1550 " runs but is smaller than the compaction trigger "
1551 "level0_file_num_compaction_trigger.");
1552 }
1553 }
1554 return s;
1555}

1651 // `max_memtable_id`. Return immediately on finding the first MemTable that
1652 // needs postponing.
1653 if (ucmp->CompareTimestamp(table_newest_udt, full_history_ts_low) >= 0) {
1654 return true;
1655 }
1656 }
