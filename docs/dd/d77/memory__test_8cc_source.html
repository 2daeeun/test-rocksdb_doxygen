37 void GetCachePointers(const std::vector<DB*>& dbs,
38 std::unordered_set<const Cache*>* cache_set) {
39 cache_set->clear();
40
41 for (auto* db : dbs) {
42 assert(db);
43
44 // Cache from DBImpl
45 StackableDB* sdb = dynamic_cast<StackableDB*>(db);
46 DBImpl* db_impl = dynamic_cast<DBImpl*>(sdb ? sdb->GetBaseDB() : db);
47 if (db_impl != nullptr) {
48 cache_set->insert(db_impl->TEST_table_cache());
49 }
50
51 // Cache from DBOptions
52 cache_set->insert(db->GetDBOptions().row_cache.get());
53
54 // Cache from table factories
55 if (db_impl != nullptr) {
56 db_impl->TEST_GetAllBlockCaches(cache_set);
57 }
58 }
59 }

76TEST_F(MemoryTest, SharedBlockCacheTotal) {
77 std::vector<DB*> dbs;
78 std::vector<uint64_t> usage_by_type;
79 const int kNumDBs = 10;
80 const int kKeySize = 100;
81 const int kValueSize = 500;
82 Options opt;
83 opt.create_if_missing = true;
84 opt.write_buffer_size = kKeySize + kValueSize;
85 opt.max_write_buffer_number = 10;
86 opt.min_write_buffer_number_to_merge = 10;
87 opt.disable_auto_compactions = true;
88 BlockBasedTableOptions bbt_opts;
89 bbt_opts.block_cache = NewLRUCache(4096 * 1000 * 10);
90 for (int i = 0; i < kNumDBs; ++i) {
91 ASSERT_OK(DestroyDB(GetDBName(i), opt));
92 DB* db = nullptr;
93 ASSERT_OK(DB::Open(opt, GetDBName(i), &db));
94 dbs.push_back(db);
95 }
96
97 std::vector<std::string> keys_by_db[kNumDBs];
98
99 // Fill one memtable per Put to make memtable use more memory.
100 for (int p = 0; p < opt.min_write_buffer_number_to_merge / 2; ++p) {
101 for (int i = 0; i < kNumDBs; ++i) {
102 for (int j = 0; j < 100; ++j) {
103 keys_by_db[i].emplace_back(rnd_.RandomString(kKeySize));
104 ASSERT_OK(dbs[i]->Put(WriteOptions(), keys_by_db[i].back(),
105 rnd_.RandomString(kValueSize)));
106 }
107 ASSERT_OK(dbs[i]->Flush(FlushOptions()));
108 }
109 }
110 for (int i = 0; i < kNumDBs; ++i) {
111 for (auto& key : keys_by_db[i]) {
112 std::string value;
113 ASSERT_OK(dbs[i]->Get(ReadOptions(), key, &value));
114 }
115 UpdateUsagesHistory(dbs);
116 }
117 for (size_t i = 1; i < usage_history_[MemoryUtil::kMemTableTotal].size();
118 ++i) {
119 // Expect EQ as we didn't flush more memtables.
120 ASSERT_EQ(usage_history_[MemoryUtil::kTableReadersTotal][i],
121 usage_history_[MemoryUtil::kTableReadersTotal][i - 1]);
122 }
123 for (int i = 0; i < kNumDBs; ++i) {
124 delete dbs[i];
125 }
126}

128TEST_F(MemoryTest, MemTableAndTableReadersTotal) {
129 std::vector<DB*> dbs;
130 std::vector<uint64_t> usage_by_type;
131 std::vector<std::vector<ColumnFamilyHandle*>> vec_handles;
132 const int kNumDBs = 10;
133 // These key/value sizes ensure each KV has its own memtable. Note that the
134 // minimum write_buffer_size allowed is 64 KB.
135 const int kKeySize = 100;
136 const int kValueSize = 1 << 16;
137 Options opt;
138 opt.create_if_missing = true;
139 opt.create_missing_column_families = true;
140 opt.write_buffer_size = kKeySize + kValueSize;
141 opt.max_write_buffer_number = 10;
142 opt.min_write_buffer_number_to_merge = 10;
143 opt.disable_auto_compactions = true;
144
145 std::vector<ColumnFamilyDescriptor> cf_descs = {
146 {kDefaultColumnFamilyName, ColumnFamilyOptions(opt)},
147 {"one", ColumnFamilyOptions(opt)},
148 {"two", ColumnFamilyOptions(opt)},
149 };
150
151 for (int i = 0; i < kNumDBs; ++i) {
152 ASSERT_OK(DestroyDB(GetDBName(i), opt));
153 std::vector<ColumnFamilyHandle*> handles;
154 dbs.emplace_back();
155 vec_handles.emplace_back();
156 ASSERT_OK(DB::Open(DBOptions(opt), GetDBName(i), cf_descs,
157 &vec_handles.back(), &dbs.back()));
158 }
159
160 // Fill one memtable per Put to make memtable use more memory.
161 for (int p = 0; p < opt.min_write_buffer_number_to_merge / 2; ++p) {
162 for (int i = 0; i < kNumDBs; ++i) {
163 for (auto* handle : vec_handles[i]) {
164 ASSERT_OK(dbs[i]->Put(WriteOptions(), handle,
165 rnd_.RandomString(kKeySize),
166 rnd_.RandomString(kValueSize)));
167 UpdateUsagesHistory(dbs);
168 }
169 }
170 }
171 // Expect the usage history is monotonically increasing
172 for (size_t i = 1; i < usage_history_[MemoryUtil::kMemTableTotal].size();
173 ++i) {
174 ASSERT_GT(usage_history_[MemoryUtil::kMemTableTotal][i],
175 usage_history_[MemoryUtil::kMemTableTotal][i - 1]);
176 ASSERT_GT(usage_history_[MemoryUtil::kMemTableUnFlushed][i],
177 usage_history_[MemoryUtil::kMemTableUnFlushed][i - 1]);
178 ASSERT_EQ(usage_history_[MemoryUtil::kTableReadersTotal][i],
179 usage_history_[MemoryUtil::kTableReadersTotal][i - 1]);
180 }
181
182 size_t usage_check_point = usage_history_[MemoryUtil::kMemTableTotal].size();
183 std::vector<Iterator*> iters;
184
185 // Create an iterator and flush all memtables for each db
186 for (int i = 0; i < kNumDBs; ++i) {
187 iters.push_back(dbs[i]->NewIterator(ReadOptions()));
188 ASSERT_OK(dbs[i]->Flush(FlushOptions()));
189
190 for (int j = 0; j < 100; ++j) {
191 std::string value;
192 ASSERT_NOK(
193 dbs[i]->Get(ReadOptions(), rnd_.RandomString(kKeySize), &value));
194 }
195
196 UpdateUsagesHistory(dbs);
197 }
198 for (size_t i = usage_check_point;
199 i < usage_history_[MemoryUtil::kMemTableTotal].size(); ++i) {
200 // Since memtables are pinned by iterators, we don't expect the
201 // memory usage of all the memtables decreases as they are pinned
202 // by iterators.
203 ASSERT_GE(usage_history_[MemoryUtil::kMemTableTotal][i],
204 usage_history_[MemoryUtil::kMemTableTotal][i - 1]);
205 // Expect the usage history from the "usage_decay_point" is
206 // monotonically decreasing.
207 ASSERT_LT(usage_history_[MemoryUtil::kMemTableUnFlushed][i],
208 usage_history_[MemoryUtil::kMemTableUnFlushed][i - 1]);
209 // Expect the usage history of the table readers increases
210 // as we flush tables.
211 ASSERT_GT(usage_history_[MemoryUtil::kTableReadersTotal][i],
212 usage_history_[MemoryUtil::kTableReadersTotal][i - 1]);
213 ASSERT_GT(usage_history_[MemoryUtil::kCacheTotal][i],
214 usage_history_[MemoryUtil::kCacheTotal][i - 1]);
215 }
216 usage_check_point = usage_history_[MemoryUtil::kMemTableTotal].size();
217 for (int i = 0; i < kNumDBs; ++i) {
218 // iterator is not used.
219 ASSERT_OK(iters[i]->status());
220 delete iters[i];
221 UpdateUsagesHistory(dbs);
222 }
223 for (size_t i = usage_check_point;
224 i < usage_history_[MemoryUtil::kMemTableTotal].size(); ++i) {
225 // Expect the usage of all memtables decreasing as we delete iterators.
226 ASSERT_LT(usage_history_[MemoryUtil::kMemTableTotal][i],
227 usage_history_[MemoryUtil::kMemTableTotal][i - 1]);
228 // Since the memory usage of un-flushed memtables is only affected
229 // by Put and flush, we expect EQ here as we only delete iterators.
230 ASSERT_EQ(usage_history_[MemoryUtil::kMemTableUnFlushed][i],
231 usage_history_[MemoryUtil::kMemTableUnFlushed][i - 1]);
232 // Expect EQ as we didn't flush more memtables.
233 ASSERT_EQ(usage_history_[MemoryUtil::kTableReadersTotal][i],
234 usage_history_[MemoryUtil::kTableReadersTotal][i - 1]);
235 }
236
237 for (int i = 0; i < kNumDBs; ++i) {
238 for (auto* handle : vec_handles[i]) {
239 delete handle;
240 }
241 delete dbs[i];
242 }
243}
