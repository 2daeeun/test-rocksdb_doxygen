26LRUHandleTable::LRUHandleTable(int max_upper_hash_bits,
27 MemoryAllocator* allocator)
28 : length_bits_(/* historical starting size*/ 4),
29 list_(new LRUHandle* [size_t{1} << length_bits_] {}),
30 elems_(0),
31 max_length_bits_(max_upper_hash_bits),
32 allocator_(allocator) {}
33
34LRUHandleTable::~LRUHandleTable() {
35 auto alloc = allocator_;
36 ApplyToEntriesRange(
37 [alloc](LRUHandle* h) {
38 if (!h->HasRefs()) {
39 h->Free(alloc);
40 }
41 },
42 0, size_t{1} << length_bits_);
43}
44
45LRUHandle* LRUHandleTable::Lookup(const Slice& key, uint32_t hash) {
46 return *FindPointer(key, hash);
47}
48
49LRUHandle* LRUHandleTable::Insert(LRUHandle* h) {
50 LRUHandle** ptr = FindPointer(h->key(), h->hash);
51 LRUHandle* old = *ptr;
52 h->next_hash = (old == nullptr ? nullptr : old->next_hash);
53 *ptr = h;
54 if (old == nullptr) {
55 ++elems_;
56 if ((elems_ >> length_bits_) > 0) { // elems_ >= length
57 // Since each cache entry is fairly large, we aim for a small
58 // average linked list length (<= 1).
59 Resize();
60 }
61 }
62 return old;
63}
64
65LRUHandle* LRUHandleTable::Remove(const Slice& key, uint32_t hash) {
66 LRUHandle** ptr = FindPointer(key, hash);
67 LRUHandle* result = *ptr;
68 if (result != nullptr) {
69 *ptr = result->next_hash;
70 --elems_;
71 }
72 return result;
73}
74
75LRUHandle** LRUHandleTable::FindPointer(const Slice& key, uint32_t hash) {
76 LRUHandle** ptr = &list_[hash >> (32 - length_bits_)];
77 while (*ptr != nullptr && ((*ptr)->hash != hash || key != (*ptr)->key())) {
78 ptr = &(*ptr)->next_hash;
79 }
80 return ptr;
81}

83void LRUHandleTable::Resize() {
84 if (length_bits_ >= max_length_bits_) {
85 // Due to reaching limit of hash information, if we made the table bigger,
86 // we would allocate more addresses but only the same number would be used.
87 return;
88 }
89 if (length_bits_ >= 31) {
90 // Avoid undefined behavior shifting uint32_t by 32.
91 return;
92 }
93
94 uint32_t old_length = uint32_t{1} << length_bits_;
95 int new_length_bits = length_bits_ + 1;
96 std::unique_ptr<LRUHandle* []> new_list {
97 new LRUHandle* [size_t{1} << new_length_bits] {}
98 };
99 [[maybe_unused]] uint32_t count = 0;
100 for (uint32_t i = 0; i < old_length; i++) {
101 LRUHandle* h = list_[i];
102 while (h != nullptr) {
103 LRUHandle* next = h->next_hash;
104 uint32_t hash = h->hash;
105 LRUHandle** ptr = &new_list[hash >> (32 - new_length_bits)];
106 h->next_hash = *ptr;
107 *ptr = h;
108 h = next;
109 count++;
110 }
111 }
112 assert(elems_ == count);
113 list_ = std::move(new_list);
114 length_bits_ = new_length_bits;
115}

117LRUCacheShard::LRUCacheShard(size_t capacity, bool strict_capacity_limit,
118 double high_pri_pool_ratio,
119 double low_pri_pool_ratio, bool use_adaptive_mutex,
120 CacheMetadataChargePolicy metadata_charge_policy,
121 int max_upper_hash_bits,
122 MemoryAllocator* allocator,
123 const Cache::EvictionCallback* eviction_callback)
124 : CacheShardBase(metadata_charge_policy),
125 capacity_(0),
126 high_pri_pool_usage_(0),
127 low_pri_pool_usage_(0),

130 high_pri_pool_capacity_(0),
131 low_pri_pool_ratio_(low_pri_pool_ratio),
132 low_pri_pool_capacity_(0),
133 table_(max_upper_hash_bits, allocator),
134 usage_(0),
135 lru_usage_(0),
136 mutex_(use_adaptive_mutex),
137 eviction_callback_(*eviction_callback) {
138 // Make empty circular linked list.
139 lru_.next = &lru_;
140 lru_.prev = &lru_;

143 SetCapacity(capacity);
144}

146void LRUCacheShard::EraseUnRefEntries() {
147 autovector<LRUHandle*> last_reference_list;
148 {
149 DMutexLock l(mutex_);
150 while (lru_.next != &lru_) {
151 LRUHandle* old = lru_.next;
152 // LRU list contains only elements which can be evicted.
153 assert(old->InCache() && !old->HasRefs());
154 LRU_Remove(old);
155 table_.Remove(old->key(), old->hash);
156 old->SetInCache(false);
157 assert(usage_ >= old->total_charge);
158 usage_ -= old->total_charge;
159 last_reference_list.push_back(old);
160 }
161 }
162
163 for (auto entry : last_reference_list) {
164 entry->Free(table_.GetAllocator());
165 }
166}

168void LRUCacheShard::ApplyToSomeEntries(
169 const std::function<void(const Slice& key, Cache::ObjectPtr value,
170 size_t charge,
171 const Cache::CacheItemHelper* helper)>& callback,
172 size_t average_entries_per_lock, size_t* state) {
173 // The state is essentially going to be the starting hash, which works
174 // nicely even if we resize between calls because we use upper-most
175 // hash bits for table indexes.
176 DMutexLock l(mutex_);
177 int length_bits = table_.GetLengthBits();
178 size_t length = size_t{1} << length_bits;
179
180 assert(average_entries_per_lock > 0);
181 // Assuming we are called with same average_entries_per_lock repeatedly,
182 // this simplifies some logic (index_end will not overflow).
183 assert(average_entries_per_lock < length || *state == 0);
184
185 size_t index_begin = *state >> (sizeof(size_t) * 8u - length_bits);
186 size_t index_end = index_begin + average_entries_per_lock;
187 if (index_end >= length) {
188 // Going to end
189 index_end = length;
190 *state = SIZE_MAX;
191 } else {
192 *state = index_end << (sizeof(size_t) * 8u - length_bits);
193 }
194
195 table_.ApplyToEntriesRange(
196 [callback,
197 metadata_charge_policy = metadata_charge_policy_](LRUHandle* h) {
198 callback(h->key(), h->value, h->GetCharge(metadata_charge_policy),
199 h->helper);
200 },
201 index_begin, index_end);
202}

233void LRUCacheShard::LRU_Remove(LRUHandle* e) {
234 assert(e->next != nullptr);
235 assert(e->prev != nullptr);
236 if (lru_low_pri_ == e) {
237 lru_low_pri_ = e->prev;
238 }
239 if (lru_bottom_pri_ == e) {
240 lru_bottom_pri_ = e->prev;
241 }
242 e->next->prev = e->prev;
243 e->prev->next = e->next;
244 e->prev = e->next = nullptr;
245 assert(lru_usage_ >= e->total_charge);
246 lru_usage_ -= e->total_charge;
247 assert(!e->InHighPriPool() || !e->InLowPriPool());
248 if (e->InHighPriPool()) {
249 assert(high_pri_pool_usage_ >= e->total_charge);
250 high_pri_pool_usage_ -= e->total_charge;
251 } else if (e->InLowPriPool()) {
252 assert(low_pri_pool_usage_ >= e->total_charge);
253 low_pri_pool_usage_ -= e->total_charge;
254 }
255}
256
257void LRUCacheShard::LRU_Insert(LRUHandle* e) {
258 assert(e->next == nullptr);
259 assert(e->prev == nullptr);
260 if (high_pri_pool_ratio_ > 0 && (e->IsHighPri() || e->HasHit())) {
261 // Inset "e" to head of LRU list.
262 e->next = &lru_;
263 e->prev = lru_.prev;
264 e->prev->next = e;
265 e->next->prev = e;
266 e->SetInHighPriPool(true);
267 e->SetInLowPriPool(false);
268 high_pri_pool_usage_ += e->total_charge;
269 MaintainPoolSize();
270 } else if (low_pri_pool_ratio_ > 0 &&
271 (e->IsHighPri() || e->IsLowPri() || e->HasHit())) {
272 // Insert "e" to the head of low-pri pool.
273 e->next = lru_low_pri_->next;
274 e->prev = lru_low_pri_;
275 e->prev->next = e;
276 e->next->prev = e;
277 e->SetInHighPriPool(false);
278 e->SetInLowPriPool(true);
279 low_pri_pool_usage_ += e->total_charge;
280 lru_low_pri_ = e;
281 MaintainPoolSize();
282 } else {
283 // Insert "e" to the head of bottom-pri pool.
284 e->next = lru_bottom_pri_->next;
285 e->prev = lru_bottom_pri_;
286 e->prev->next = e;
287 e->next->prev = e;
288 e->SetInHighPriPool(false);
289 e->SetInLowPriPool(false);
290 // if the low-pri pool is empty, lru_low_pri_ also needs to be updated.
291 if (lru_bottom_pri_ == lru_low_pri_) {
292 lru_low_pri_ = e;
293 }
294 lru_bottom_pri_ = e;
295 }
296 lru_usage_ += e->total_charge;
297}

299void LRUCacheShard::MaintainPoolSize() {
300 while (high_pri_pool_usage_ > high_pri_pool_capacity_) {
301 // Overflow last entry in high-pri pool to low-pri pool.
302 lru_low_pri_ = lru_low_pri_->next;
303 assert(lru_low_pri_ != &lru_);
304 assert(lru_low_pri_->InHighPriPool());
305 lru_low_pri_->SetInHighPriPool(false);
306 lru_low_pri_->SetInLowPriPool(true);
307 assert(high_pri_pool_usage_ >= lru_low_pri_->total_charge);
308 high_pri_pool_usage_ -= lru_low_pri_->total_charge;
309 low_pri_pool_usage_ += lru_low_pri_->total_charge;
310 }
311
312 while (low_pri_pool_usage_ > low_pri_pool_capacity_) {
313 // Overflow last entry in low-pri pool to bottom-pri pool.
314 lru_bottom_pri_ = lru_bottom_pri_->next;
315 assert(lru_bottom_pri_ != &lru_);
316 assert(lru_bottom_pri_->InLowPriPool());
317 lru_bottom_pri_->SetInHighPriPool(false);
318 lru_bottom_pri_->SetInLowPriPool(false);
319 assert(low_pri_pool_usage_ >= lru_bottom_pri_->total_charge);
320 low_pri_pool_usage_ -= lru_bottom_pri_->total_charge;
321 }
322}

324void LRUCacheShard::EvictFromLRU(size_t charge,
325 autovector<LRUHandle*>* deleted) {
326 while ((usage_ + charge) > capacity_ && lru_.next != &lru_) {
327 LRUHandle* old = lru_.next;
328 // LRU list contains only elements which can be evicted.
329 assert(old->InCache() && !old->HasRefs());
330 LRU_Remove(old);
331 table_.Remove(old->key(), old->hash);
332 old->SetInCache(false);
333 assert(usage_ >= old->total_charge);
334 usage_ -= old->total_charge;
335 deleted->push_back(old);
336 }
337}

339void LRUCacheShard::NotifyEvicted(
340 const autovector<LRUHandle*>& evicted_handles) {
341 MemoryAllocator* alloc = table_.GetAllocator();
342 for (LRUHandle* entry : evicted_handles) {
343 if (eviction_callback_ &&
344 eviction_callback_(entry->key(), static_cast<Cache::Handle*>(entry),
345 entry->HasHit())) {
346 // Callback took ownership of obj; just free handle
347 free(entry);
348 } else {
349 // Free the entries here outside of mutex for performance reasons.
350 entry->Free(alloc);
351 }
352 }
353}

355void LRUCacheShard::SetCapacity(size_t capacity) {
356 autovector<LRUHandle*> last_reference_list;
357 {
358 DMutexLock l(mutex_);
359 capacity_ = capacity;
360 high_pri_pool_capacity_ = capacity_ * high_pri_pool_ratio_;
361 low_pri_pool_capacity_ = capacity_ * low_pri_pool_ratio_;
362 EvictFromLRU(0, &last_reference_list);

365 NotifyEvicted(last_reference_list);

373Status LRUCacheShard::InsertItem(LRUHandle* e, LRUHandle** handle) {
374 Status s = Status::OK();
375 autovector<LRUHandle*> last_reference_list;
376
377 {
378 DMutexLock l(mutex_);
379
380 // Free the space following strict LRU policy until enough space
381 // is freed or the lru list is empty.
382 EvictFromLRU(e->total_charge, &last_reference_list);
383
384 if ((usage_ + e->total_charge) > capacity_ &&
385 (strict_capacity_limit_ || handle == nullptr)) {
386 e->SetInCache(false);
387 if (handle == nullptr) {
388 // Don't insert the entry but still return ok, as if the entry inserted
389 // into cache and get evicted immediately.
390 last_reference_list.push_back(e);
391 } else {
392 free(e);
393 e = nullptr;
394 *handle = nullptr;
395 s = Status::MemoryLimit("Insert failed due to LRU cache being full.");
396 }
397 } else {
398 // Insert into the cache. Note that the cache might get larger than its
399 // capacity if not enough space was freed up.
400 LRUHandle* old = table_.Insert(e);
401 usage_ += e->total_charge;
402 if (old != nullptr) {
403 s = Status::OkOverwritten();
404 assert(old->InCache());
405 old->SetInCache(false);
406 if (!old->HasRefs()) {
407 // old is on LRU because it's in cache and its reference count is 0.
408 LRU_Remove(old);
409 assert(usage_ >= old->total_charge);
410 usage_ -= old->total_charge;
411 last_reference_list.push_back(old);
412 }
413 }
414 if (handle == nullptr) {
415 LRU_Insert(e);
416 } else {
417 // If caller already holds a ref, no need to take one here.
418 if (!e->HasRefs()) {
419 e->Ref();
420 }
421 *handle = e;
422 }
423 }
424 }
425
426 NotifyEvicted(last_reference_list);
427
428 return s;
429}

431LRUHandle* LRUCacheShard::Lookup(const Slice& key, uint32_t hash,
432 const Cache::CacheItemHelper* /*helper*/,
433 Cache::CreateContext* /*create_context*/,
434 Cache::Priority /*priority*/,
435 Statistics* /*stats*/) {
436 DMutexLock l(mutex_);
437 LRUHandle* e = table_.Lookup(key, hash);
438 if (e != nullptr) {
439 assert(e->InCache());
440 if (!e->HasRefs()) {
441 // The entry is in LRU since it's in hash and has no external
442 // references.
443 LRU_Remove(e);
444 }
445 e->Ref();
446 e->SetHit();
447 }
448 return e;
449}

452 DMutexLock l(mutex_);

454 assert(e->HasRefs());
455 e->Ref();
456 return true;
457}

473bool LRUCacheShard::Release(LRUHandle* e, bool /*useful*/,
474 bool erase_if_last_ref) {
475 if (e == nullptr) {
476 return false;
477 }
478 bool must_free;
479 bool was_in_cache;
480 {
481 DMutexLock l(mutex_);
482 must_free = e->Unref();
483 was_in_cache = e->InCache();
484 if (must_free && was_in_cache) {
485 // The item is still in cache, and nobody else holds a reference to it.
486 if (usage_ > capacity_ || erase_if_last_ref) {
487 // The LRU list must be empty since the cache is full.
488 assert(lru_.next == &lru_ || erase_if_last_ref);
489 // Take this opportunity and remove the item.
490 table_.Remove(e->key(), e->hash);
491 e->SetInCache(false);
492 } else {
493 // Put the item back on the LRU list, and don't free it.
494 LRU_Insert(e);
495 must_free = false;
496 }
497 }
498 // If about to be freed, then decrement the cache usage.
499 if (must_free) {
500 assert(usage_ >= e->total_charge);
501 usage_ -= e->total_charge;
502 }
503 }
504
505 // Free the entry here outside of mutex for performance reasons.
506 if (must_free) {
507 // Only call eviction callback if we're sure no one requested erasure
508 // FIXME: disabled because of test churn
509 if (false && was_in_cache && !erase_if_last_ref && eviction_callback_ &&
510 eviction_callback_(e->key(), static_cast<Cache::Handle*>(e),
511 e->HasHit())) {
512 // Callback took ownership of obj; just free handle
513 free(e);
514 } else {
515 e->Free(table_.GetAllocator());
516 }
517 }
518 return must_free;
519}

521LRUHandle* LRUCacheShard::CreateHandle(const Slice& key, uint32_t hash,
522 Cache::ObjectPtr value,
523 const Cache::CacheItemHelper* helper,
524 size_t charge) {
525 assert(helper);
526 // value == nullptr is reserved for indicating failure in SecondaryCache
527 assert(!(helper->IsSecondaryCacheCompatible() && value == nullptr));
528
529 // Allocate the memory here outside of the mutex.
530 // If the cache is full, we'll have to release it.
531 // It shouldn't happen very often though.
532 LRUHandle* e =
533 static_cast<LRUHandle*>(malloc(sizeof(LRUHandle) - 1 + key.size()));
534
535 e->value = value;
536 e->m_flags = 0;
537 e->im_flags = 0;
538 e->helper = helper;
539 e->key_length = key.size();
540 e->hash = hash;
541 e->refs = 0;
542 e->next = e->prev = nullptr;
543 memcpy(e->key_data, key.data(), key.size());
544 e->CalcTotalCharge(charge, metadata_charge_policy_);
545
546 return e;
547}

560LRUHandle* LRUCacheShard::CreateStandalone(const Slice& key, uint32_t hash,
561 Cache::ObjectPtr value,
562 const Cache::CacheItemHelper* helper,
563 size_t charge,
564 bool allow_uncharged) {
565 LRUHandle* e = CreateHandle(key, hash, value, helper, charge);
566 e->SetIsStandalone(true);
567 e->Ref();
568
569 autovector<LRUHandle*> last_reference_list;
570
571 {
572 DMutexLock l(mutex_);
573
574 EvictFromLRU(e->total_charge, &last_reference_list);
575
576 if (strict_capacity_limit_ && (usage_ + e->total_charge) > capacity_) {
577 if (allow_uncharged) {
578 e->total_charge = 0;
579 } else {
580 free(e);
581 e = nullptr;
582 }
583 } else {
584 usage_ += e->total_charge;
585 }
586 }
587
588 NotifyEvicted(last_reference_list);
589 return e;
590}

592void LRUCacheShard::Erase(const Slice& key, uint32_t hash) {
593 LRUHandle* e;
594 bool last_reference = false;
595 {
596 DMutexLock l(mutex_);
597 e = table_.Remove(key, hash);
598 if (e != nullptr) {
599 assert(e->InCache());
600 e->SetInCache(false);
601 if (!e->HasRefs()) {
602 // The entry is in LRU since it's in hash and has no external references
603 LRU_Remove(e);
604 assert(usage_ >= e->total_charge);
605 usage_ -= e->total_charge;
606 last_reference = true;
607 }
608 }
609 }
610
611 // Free the entry here outside of mutex for performance reasons.
612 // last_reference will only be true if e != nullptr.
613 if (last_reference) {
614 e->Free(table_.GetAllocator());
615 }
616}

680void LRUCache::ApplyToHandle(
681 Cache* cache, Handle* handle,

701std::shared_ptr<Cache> LRUCacheOptions::MakeSharedCache() const {
702 if (num_shard_bits >= 20) {
703 return nullptr; // The cache cannot be sharded into too many fine pieces.
704 }
705 if (high_pri_pool_ratio < 0.0 || high_pri_pool_ratio > 1.0) {
706 // Invalid high_pri_pool_ratio
707 return nullptr;
708 }
709 if (low_pri_pool_ratio < 0.0 || low_pri_pool_ratio > 1.0) {
710 // Invalid low_pri_pool_ratio
711 return nullptr;
712 }
713 if (low_pri_pool_ratio + high_pri_pool_ratio > 1.0) {
714 // Invalid high_pri_pool_ratio and low_pri_pool_ratio combination
715 return nullptr;
716 }
717 // For sanitized options
718 LRUCacheOptions opts = *this;
719 if (opts.num_shard_bits < 0) {
720 opts.num_shard_bits = GetDefaultCacheShardBits(capacity);
721 }
722 std::shared_ptr<Cache> cache = std::make_shared<LRUCache>(opts);
723 if (secondary_cache) {
724 cache = std::make_shared<CacheWithSecondaryAdapter>(cache, secondary_cache);
725 }
726 return cache;
727}

729std::shared_ptr<RowCache> LRUCacheOptions::MakeSharedRowCache() const {
730 if (secondary_cache) {
731 // Not allowed for a RowCache
732 return nullptr;
733 }
734 // Works while RowCache is an alias for Cache
735 return MakeSharedCache();
736}
