6// Currently we support two types of tables: plain table and block-based table.
7// 1. Block-based table: this is the default table type that we inherited from
8// LevelDB, which was designed for storing data in hard disk or flash
9// device.
10// 2. Plain table: it is one of RocksDB's SST file format optimized
11// for low query latency on pure-memory or really low-latency media.
12//
13// A tutorial of rocksdb table formats is available here:

48// Types of checksums to use for checking integrity of logical blocks within
49// files. All checksums currently use 32 bits of checking power (1 in 4B
50// chance of failing to detect random corruption). Traditionally, the actual
51// checking power can be far from ideal if the corruption is due to misplaced
52// data (e.g. physical blocks out of order in a file, or from another file),
53// which is fixed in format_version=6 (see below).
54enum ChecksumType : char {
55 kNoChecksum = 0x0,
56 kCRC32c = 0x1,
57 kxxHash = 0x2,
58 kxxHash64 = 0x3,
59 kXXH3 = 0x4, // Supported since RocksDB 6.27
60};

62// `PinningTier` is used to specify which tier of block-based tables should
63// be affected by a block cache pinning setting (see
64// `MetadataCacheOptions` below).
65enum class PinningTier {
66 // For compatibility, this value specifies to fallback to the behavior
67 // indicated by the deprecated options,
68 // `pin_l0_filter_and_index_blocks_in_cache` and
69 // `pin_top_level_index_and_filter`.
70 kFallback,
71
72 // This tier contains no block-based tables.
73 kNone,
74
75 // This tier contains block-based tables that may have originated from a
76 // memtable flush. In particular, it includes tables from L0 that are smaller
77 // than 1.5 times the current `write_buffer_size`. Note these criteria imply
78 // it can include intra-L0 compaction outputs and ingested files, as long as
79 // they are not abnormally large compared to flushed files in L0.
80 kFlushedAndSimilar,
81
82 // This tier contains all block-based tables.
83 kAll,
84};

226 // The index type that will be used for this table.
227 enum IndexType : char {
228 // A space efficient index block that is optimized for
229 // binary-search-based index.
230 kBinarySearch = 0x00,
231
232 // The hash index, if enabled, will do the hash lookup when
233 // `Options.prefix_extractor` is provided.
234 kHashSearch = 0x01,
235
236 // A two-level index implementation. Both levels are binary search indexes.
237 // Second level index blocks ("partitions") use block cache even when
238 // cache_index_and_filter_blocks=false.
239 kTwoLevelIndexSearch = 0x02,
240
241 // Like kBinarySearch, but index also contains first key of each block.
242 // This allows iterators to defer reading the block until it's actually
243 // needed. May significantly reduce read amplification of short range scans.
244 // Without it, iterator seek usually reads one block from each level-0 file
245 // and from each level, which may be expensive.
246 // Works best in combination with:
247 // - IndexShorteningMode::kNoShortening,
248 // - custom FlushBlockPolicy to cut blocks at some meaningful boundaries,
249 // e.g. when prefix changes.
250 // Makes the index significantly bigger (2x or more), especially when keys
251 // are long.
252 kBinarySearchWithFirstKey = 0x03,
253 };

326 // `cache_usage_options` allows users to specify the default
327 // options (`cache_usage_options.options`) and the overriding
328 // options (`cache_usage_options.options_overrides`)
329 // for different `CacheEntryRole` under various features related to cache
330 // usage.
331 //
332 // For a certain `CacheEntryRole role` and a certain feature `f` of
333 // `CacheEntryRoleOptions`:
334 // 1. If `options_overrides` has an entry for `role` and
335 // `options_overrides[role].f != kFallback`, we use
336 // `options_overrides[role].f`
337 // 2. Otherwise, if `options[role].f != kFallback`, we use `options[role].f`
338 // 3. Otherwise, we follow the compatible existing behavior for `f` (see
339 // each feature's comment for more)
340 //
341 // `cache_usage_options` currently supports specifying options for the
342 // following features:
343 //
344 // 1. Memory charging to block cache (`CacheEntryRoleOptions::charged`)
345 // Memory charging is a feature of accounting memory usage of specific area
346 // (represented by `CacheEntryRole`) toward usage in block cache (if
347 // available), by updating a dynamical charge to the block cache loosely based
348 // on the actual memory usage of that area.
349 //
350 // (a) CacheEntryRole::kCompressionDictionaryBuildingBuffer
351 // (i) If kEnabled:
352 // Charge memory usage of the buffered data used as training samples for
353 // dictionary compression.
354 // If such memory usage exceeds the avaible space left in the block cache
355 // at some point (i.e, causing a cache full under
356 // `LRUCacheOptions::strict_capacity_limit` = true), the data will then be
357 // unbuffered.
358 // (ii) If kDisabled:
359 // Does not charge the memory usage mentioned above.
360 // (iii) Compatible existing behavior:
361 // Same as kEnabled.
362 //
363 // (b) CacheEntryRole::kFilterConstruction
364 // (i) If kEnabled:
365 // Charge memory usage of Bloom Filter
366 // (format_version >= 5) and Ribbon Filter construction.
367 // If additional temporary memory of Ribbon Filter exceeds the avaible
368 // space left in the block cache at some point (i.e, causing a cache full
369 // under `LRUCacheOptions::strict_capacity_limit` = true),
370 // construction will fall back to Bloom Filter.
371 // (ii) If kDisabled:
372 // Does not charge the memory usage mentioned above.
373 // (iii) Compatible existing behavior:
374 // Same as kDisabled.
375 //
376 // (c) CacheEntryRole::kBlockBasedTableReader
377 // (i) If kEnabled:
378 // Charge memory usage of table properties +
379 // index block/filter block/uncompression dictionary (when stored in table
380 // reader i.e, BlockBasedTableOptions::cache_index_and_filter_blocks ==
381 // false) + some internal data structures during table reader creation.
382 // If such a table reader exceeds
383 // the avaible space left in the block cache at some point (i.e, causing
384 // a cache full under `LRUCacheOptions::strict_capacity_limit` = true),
385 // creation will fail with Status::MemoryLimit().
386 // (ii) If kDisabled:
387 // Does not charge the memory usage mentioned above.
388 // (iii) Compatible existing behavior:
389 // Same as kDisabled.
390 //
391 // (d) CacheEntryRole::kFileMetadata
392 // (i) If kEnabled:
393 // Charge memory usage of file metadata. RocksDB holds one file metadata
394 // structure in-memory per on-disk table file.
395 // If such file metadata's
396 // memory exceeds the avaible space left in the block cache at some point
397 // (i.e, causing a cache full under `LRUCacheOptions::strict_capacity_limit` =
398 // true), creation will fail with Status::MemoryLimit().
399 // (ii) If kDisabled:
400 // Does not charge the memory usage mentioned above.
401 // (iii) Compatible existing behavior:
402 // Same as kDisabled.
403 //
404 // (e) Other CacheEntryRole
405 // Not supported.
406 // `Status::kNotSupported` will be returned if
407 // `CacheEntryRoleOptions::charged` is set to {`kEnabled`, `kDisabled`}.
408 //
409 //
410 // 2. More to come ...
411 //

439 // Option to generate Bloom/Ribbon filters that minimize memory
440 // internal fragmentation.
441 //
442 // When false, malloc_usable_size is not available, or format_version < 5,
443 // filters are generated without regard to internal fragmentation when
444 // loaded into memory (historical behavior). When true (and
445 // malloc_usable_size is available and format_version >= 5), then
446 // filters are generated to "round up" and "round down" their sizes to
447 // minimize internal fragmentation when loaded into memory, assuming the
448 // reading DB has the same memory allocation characteristics as the
449 // generating DB. This option does not break forward or backward
450 // compatibility.
451 //
452 // While individual filters will vary in bits/key and false positive rate
453 // when setting is true, the implementation attempts to maintain a weighted
454 // average FP rate for filters consistent with this option set to false.
455 //
456 // With Jemalloc for example, this setting is expected to save about 10% of
457 // the memory footprint and block cache charge of filters, while increasing
458 // disk usage of filters by about 1-2% due to encoding efficiency losses

460 //
461 // NOTE: Because some memory counted by block cache might be unmapped pages
462 // within internal fragmentation, this option can increase observed RSS
463 // memory usage. With cache_index_and_filter_blocks=true, this option makes
464 // the block cache better at using space it is allowed. (These issues
465 // should not arise with partitioned filters.)

573 // This enum allows trading off increased index size for improved iterator
574 // seek performance in some situations, particularly when block cache is
575 // disabled (ReadOptions::fill_cache = false) and direct IO is
576 // enabled (DBOptions::use_direct_reads = true).
577 // The default mode is the best tradeoff for most use cases.
578 // This option only affects newly written tables.
579 //
580 // The index contains a key separating each pair of consecutive blocks.
581 // Let A be the highest key in one block, B the lowest key in the next block,
582 // and I the index entry separating these two blocks:
583 // [ ... A] I [B ...]
584 // I is allowed to be anywhere in [A, B).
585 // If an iterator is seeked to a key in (A, I], we'll unnecessarily read the
586 // first block, then immediately fall through to the second block.
587 // However, if I=A, this can't happen, and we'll read only the second block.
588 // In kNoShortening mode, we use I=A. In other modes, we use the shortest
589 // key in [A, B), which usually significantly reduces index size.
590 //
591 // There's a similar story for the last index entry, which is an upper bound
592 // of the highest key in the file. If it's shortened and therefore
593 // overestimated, iterator is likely to unnecessarily read the last data block
594 // from each file on each seek.
595 enum class IndexShorteningMode : char {
596 // Use full keys.
597 kNoShortening,
598 // Shorten index keys between blocks, but use full key for the last index
599 // key, which is the upper bound of the whole file.
