134 {
135 // Threads executing Add() would be competing for this mutex, the first one

142

145 (curr_window == num_windows_ - 1) ? 0 : curr_window + 1;
146
147 // subtract next buckets from totals and swap to next buckets
148 HistogramStat& stats_to_drop =
149 window_stats_[static_cast<size_t>(next_window)];
150
151 if (!stats_to_drop.Empty()) {
152 for (size_t b = 0; b < stats_.num_buckets_; b++) {
153 stats_.buckets_[b].fetch_sub(stats_to_drop.bucket_at(b),

162 if (m < new_min) {

164 }

168 }
169
