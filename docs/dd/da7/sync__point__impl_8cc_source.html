15void KillPoint::TestKillRandom(std::string kill_point, int odds_weight,
16 const std::string& srcfile, int srcline) {
17 if (rocksdb_kill_odds <= 0) {
18 return;
19 }
20 int odds = rocksdb_kill_odds * odds_weight;
21 for (auto& p : rocksdb_kill_exclude_prefixes) {
22 if (kill_point.substr(0, p.length()) == p) {
23 return;
24 }
25 }
26
27 assert(odds > 0);
28 if (odds % 7 == 0) {
29 // class Random uses multiplier 16807, which is 7^5. If odds are
30 // multiplier of 7, there might be limited values generated.
31 odds++;
32 }
33 auto* r = Random::GetTLSInstance();
34 bool crash = r->OneIn(odds);
35 if (crash) {
36 port::Crash(srcfile, srcline);
37 }
38}

40void SyncPoint::Data::LoadDependency(
41 const std::vector<SyncPointPair>& dependencies) {
42 std::lock_guard<std::mutex> lock(mutex_);
43 successors_.clear();
44 predecessors_.clear();
45 cleared_points_.clear();
46 for (const auto& dependency : dependencies) {
47 successors_[dependency.predecessor].push_back(dependency.successor);
48 predecessors_[dependency.successor].push_back(dependency.predecessor);
49 point_filter_.Add(dependency.successor);
50 point_filter_.Add(dependency.predecessor);
51 }
52 cv_.notify_all();
53}

105void SyncPoint::Data::Process(const Slice& point, void* cb_arg) {
106 if (!enabled_) {
107 return;
108 }
109
110 // Use a filter to prevent mutex lock if possible.
111 if (!point_filter_.MayContain(point)) {
112 return;
113 }
114
115 // Must convert to std::string for remaining work. Take
116 // heap hit.
117 std::string point_string(point.ToString());
118 std::unique_lock<std::mutex> lock(mutex_);
119 auto thread_id = std::this_thread::get_id();
120
121 auto marker_iter = markers_.find(point_string);
122 if (marker_iter != markers_.end()) {
123 for (auto& marked_point : marker_iter->second) {
124 marked_thread_id_.emplace(marked_point, thread_id);
125 point_filter_.Add(marked_point);
126 }
127 }
128
129 if (DisabledByMarker(point_string, thread_id)) {
130 return;
131 }
132
133 while (!PredecessorsAllCleared(point_string)) {
134 cv_.wait(lock);
135 if (DisabledByMarker(point_string, thread_id)) {
136 return;
137 }
138 }
139
140 auto callback_pair = callbacks_.find(point_string);
141 if (callback_pair != callbacks_.end()) {
142 num_callbacks_running_++;
143 mutex_.unlock();
144 callback_pair->second(cb_arg);
145 mutex_.lock();
146 num_callbacks_running_--;
147 }
148 cleared_points_.insert(point_string);
149 cv_.notify_all();
150}
