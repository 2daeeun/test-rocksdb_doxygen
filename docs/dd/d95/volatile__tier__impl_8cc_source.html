40Status VolatileCacheTier::Insert(const Slice& page_key, const char* data,
41 const size_t size) {
42 // precondition
43 assert(data);
44 assert(size);
45
46 // increment the size
47 size_ += size;
48
49 // check if we have overshot the limit, if so evict some space
50 while (size_ > max_size_) {
51 if (!Evict()) {
52 // unable to evict data, we give up so we don't spike read
53 // latency
54 assert(size_ >= size);
55 size_ -= size;
56 return Status::TryAgain("Unable to evict any data");
57 }
58 }
59
60 assert(size_ >= size);
61
62 // insert order: LRU, followed by index
63 std::string key(page_key.data(), page_key.size());
64 std::string value(data, size);
65 std::unique_ptr<CacheData> cache_data(
66 new CacheData(std::move(key), std::move(value)));
67 bool ok = index_.Insert(cache_data.get());
68 if (!ok) {
69 // decrement the size that we incremented ahead of time
70 assert(size_ >= size);
71 size_ -= size;
72 // failed to insert to cache, block already in cache
73 return Status::TryAgain("key already exists in volatile cache");
74 }
75
76 cache_data.release();
77 stats_.cache_inserts_++;
78 return Status::OK();
79}

81Status VolatileCacheTier::Lookup(const Slice& page_key,
82 std::unique_ptr<char[]>* result,
83 size_t* size) {
84 CacheData key(std::move(page_key.ToString()));
85 CacheData* kv;
86 bool ok = index_.Find(&key, &kv);
87 if (ok) {
88 // set return data
89 result->reset(new char[kv->value.size()]);
90 memcpy(result->get(), kv->value.c_str(), kv->value.size());
91 *size = kv->value.size();
92 // drop the reference on cache data
93 kv->refs_--;
94 // update stats
95 stats_.cache_hits_++;
96 return Status::OK();
97 }
98
99 stats_.cache_misses_++;
100
101 if (next_tier()) {
102 return next_tier()->Lookup(page_key, result, size);
103 }
104
105 return Status::NotFound("key not found in volatile cache");
106}

113bool VolatileCacheTier::Evict() {
114 CacheData* edata = index_.Evict();
115 if (!edata) {
116 // not able to evict any object
117 return false;
118 }
119
120 stats_.cache_evicts_++;
121
122 // push the evicted object to the next level
123 if (next_tier()) {
124 // TODO: Should the insert error be ignored?
125 Status s = next_tier()->Insert(Slice(edata->key), edata->value.c_str(),
126 edata->value.size());
127 s.PermitUncheckedError();
128 }
129
130 // adjust size and destroy data
131 size_ -= edata->value.size();
132 delete edata;
133
134 return true;
135}
