133 if (use_direct_io_ && !IsDirectIOSupported()) {

200TEST_P(DBRateLimiterOnReadTest, Iterator) {
201 if (use_direct_io_ && !IsDirectIOSupported()) {
202 return;
203 }
204 Init();
205
206 std::unique_ptr<Iterator> iter(db_->NewIterator(GetReadOptions()));
207 ASSERT_EQ(0, options_.rate_limiter->GetTotalRequests(Env::IO_USER));
208
209 int expected = 0;
210 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
211 ++expected;
212 ASSERT_EQ(expected, options_.rate_limiter->GetTotalRequests(Env::IO_USER));
213 }
214
215 for (iter->SeekToLast(); iter->Valid(); iter->Prev()) {
216 // When `use_block_cache_ == true`, the reverse scan will access the blocks
217 // loaded to cache during the above forward scan, in which case no further
218 // file reads are expected.
219 if (!use_block_cache_) {
220 ++expected;
221 }
222 }
223 ASSERT_OK(iter->status());
224 // Reverse scan does not read evenly (one block per iteration) due to
225 // descending seqno ordering, so wait until after the loop to check total.
226 ASSERT_EQ(expected, options_.rate_limiter->GetTotalRequests(Env::IO_USER));
227}

230TEST_P(DBRateLimiterOnReadTest, VerifyChecksum) {
231 if (use_direct_io_ && !IsDirectIOSupported()) {
232 return;
233 }
234 Init();
235
236 ASSERT_EQ(0, options_.rate_limiter->GetTotalRequests(Env::IO_USER));
237
238 ASSERT_OK(db_->VerifyChecksum(GetReadOptions()));
239 // In BufferedIO,
240 // there are 7 reads per file, each of which will be rate-limited.
241 // During open: read footer, meta index block, properties block, index block.
242 // During actual checksum verification: read meta index block, verify checksum
243 // in meta blocks and verify checksum in file blocks.
244 //
245 // In DirectIO, where we support tail prefetching, during table open, we only
246 // do 1 read instead of 4 as described above. Actual checksum verification
247 // reads stay the same.
248#ifdef OS_WIN
249 // No file system prefetch implemented for OS Win. During table open,
250 // we only do 1 read for BufferedIO.
251 int num_read_per_file = 4;
252#else
253 int num_read_per_file = (!use_direct_io_) ? 7 : 4;
254#endif
255 int expected = kNumFiles * num_read_per_file;
256
257 ASSERT_EQ(expected, options_.rate_limiter->GetTotalRequests(Env::IO_USER));
258}
