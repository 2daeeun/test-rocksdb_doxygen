25// Creates a new shared customizable instance object based on the
26// input parameters using the object registry.
27//
28// The id parameter specifies the instance class of the object to create.
29// The opt_map parameter specifies the configuration of the new instance.
30//
31// The config_options parameter controls the process and how errors are
32// returned. If ignore_unknown_options=true, unknown values are ignored during
33// the configuration. If ignore_unsupported_options=true, unknown instance types
34// are ignored. If invoke_prepare_options=true, the resulting instance will be
35// initialized (via PrepareOptions)
36//
37// @param config_options Controls how the instance is created and errors are
38// handled
39// @param id The identifier of the new object being created. This string
40// will be used by the object registry to locate the appropriate object to
41// create.
42// @param opt_map Optional name-value pairs of properties to set for the newly
43// created object
44// @param result The newly created and configured instance.
45template <typename T>
46static Status NewSharedObject(
47 const ConfigOptions& config_options, const std::string& id,
48 const std::unordered_map<std::string, std::string>& opt_map,
49 std::shared_ptr<T>* result) {
50 if (!id.empty()) {
51 Status status;
52 status = config_options.registry->NewSharedObject(id, result);
53 if (config_options.ignore_unsupported_options && status.IsNotSupported()) {
54 status = Status::OK();
55 } else if (status.ok()) {
56 status = Customizable::ConfigureNewObject(config_options, result->get(),
57 opt_map);
58 }
59 return status;
60 } else if (opt_map.empty()) {
61 // There was no ID and no map (everything empty), so reset/clear the result
62 result->reset();
63 return Status::OK();
64 } else {
65 return Status::NotSupported("Cannot reset object ");
66 }
67}

69// Creates a new managed customizable instance object based on the
70// input parameters using the object registry. Unlike "shared" objects,
71// managed objects are limited to a single instance per ID.
72//
73// The id parameter specifies the instance class of the object to create.
74// If an object with this id exists in the registry, the existing object
75// will be returned. If the object does not exist, a new one will be created.
76//
77// The opt_map parameter specifies the configuration of the new instance.
78// If the object already exists, the existing object is returned "as is" and
79// this parameter is ignored.
80//
81// The config_options parameter controls the process and how errors are
82// returned. If ignore_unknown_options=true, unknown values are ignored during
83// the configuration. If ignore_unsupported_options=true, unknown instance types
84// are ignored. If invoke_prepare_options=true, the resulting instance will be
85// initialized (via PrepareOptions)
86//
87// @param config_options Controls how the instance is created and errors are
88// handled
89// @param id The identifier of the object. This string
90// will be used by the object registry to locate the appropriate object to
91// create or return.
92// @param opt_map Optional name-value pairs of properties to set for the newly
93// created object
94// @param result The managed instance.
95template <typename T>
96static Status NewManagedObject(
97 const ConfigOptions& config_options, const std::string& id,
98 const std::unordered_map<std::string, std::string>& opt_map,
99 std::shared_ptr<T>* result) {
100 Status status;
101 if (!id.empty()) {
102 status = config_options.registry->GetOrCreateManagedObject<T>(
103 id, result, [config_options, opt_map](T* object) {
104 return object->ConfigureFromMap(config_options, opt_map);
105 });
106 if (config_options.ignore_unsupported_options && status.IsNotSupported()) {
107 return Status::OK();
108 }
109 } else {
110 status = Status::NotSupported("Cannot reset object ");
111 }
112 return status;
113}

115// Creates a new shared Customizable object based on the input parameters.
116// This method parses the input value to determine the type of instance to
117// create. If there is an existing instance (in result) and it is the same ID
118// as the object being created, the existing configuration is stored and used as
119// the default for the new object.
120//
121// The value parameter specified the instance class of the object to create.
122// If it is a simple string (e.g. BlockBasedTable), then the instance will be
123// created using the default settings. If the value is a set of name-value
124// pairs, then the "id" value is used to determine the instance to create and
125// the remaining parameters are used to configure the object. Id name-value
126// pairs are specified, there should be an "id=value" pairing or an error may
127// result.
128//
129// The config_options parameter controls the process and how errors are
130// returned. If ignore_unknown_options=true, unknown values are ignored during
131// the configuration. If ignore_unsupported_options=true, unknown instance types
132// are ignored. If invoke_prepare_options=true, the resulting instance will be
133// initialized (via PrepareOptions)
134//
135// @param config_options Controls how the instance is created and errors are
136// handled
137// @param value Either the simple name of the instance to create, or a set of
138// name-value pairs to create and initailize the object
139// @param result The newly created instance.
140template <typename T>
141static Status LoadSharedObject(const ConfigOptions& config_options,
142 const std::string& value,
143 std::shared_ptr<T>* result) {
144 std::string id;
145 std::unordered_map<std::string, std::string> opt_map;
146
147 Status status = Customizable::GetOptionsMap(config_options, result->get(),
148 value, &id, &opt_map);
149 if (!status.ok()) { // GetOptionsMap failed
150 return status;
151 } else {
152 return NewSharedObject(config_options, id, opt_map, result);
153 }
154}

156// Creates a new shared Customizable object based on the input parameters.
157//
158// The value parameter specified the instance class of the object to create.
159// If it is a simple string (e.g. BlockBasedTable), then the instance will be
160// created using the default settings. If the value is a set of name-value
161// pairs, then the "id" value is used to determine the instance to create and
162// the remaining parameters are used to configure the object. Id name-value
163// pairs are specified, there should be an "id=value" pairing or an error may
164// result.
165//
166// The "id" field from the value (either the whole field or "id=XX") is used
167// to determine the type/id of the object to return. For a given id, there
168// the same instance of the object will be returned from this method (as opposed
169// to LoadSharedObject which would create different objects for the same id.
170//
171// The config_options parameter controls the process and how errors are
172// returned. If ignore_unknown_options=true, unknown values are ignored during
173// the configuration. If ignore_unsupported_options=true, unknown instance types
174// are ignored. If invoke_prepare_options=true, the resulting instance will be
175// initialized (via PrepareOptions)
176//
177// @param config_options Controls how the instance is created and errors are
178// handled
179// @param value Either the simple name of the instance to create, or a set of
180// name-value pairs to create and initailize the object
181// @param result The newly created instance.
182template <typename T>
183static Status LoadManagedObject(const ConfigOptions& config_options,
184 const std::string& value,
185 std::shared_ptr<T>* result) {
186 std::string id;
187 std::unordered_map<std::string, std::string> opt_map;
188 Status status = Customizable::GetOptionsMap(config_options, nullptr, value,
189 &id, &opt_map);
190 if (!status.ok()) { // GetOptionsMap failed
191 return status;
192 } else if (value.empty()) { // No Id and no options. Clear the object
193 *result = nullptr;
194 return Status::OK();
195 } else {
196 return NewManagedObject(config_options, id, opt_map, result);
197 }
198}
