21intptr_t GetOffset(const Configurable* holder, void* field) {
22 return reinterpret_cast<intptr_t>(field) -
23 reinterpret_cast<intptr_t>(static_cast<const void*>(holder));
24}

26void* ApplyOffset(const Configurable* holder, intptr_t offset) {
27 return reinterpret_cast<void*>(
28 reinterpret_cast<intptr_t>(static_cast<const void*>(holder)) + offset);

48Status Configurable::PrepareOptions(const ConfigOptions& opts) {
49 // We ignore the invoke_prepare_options here intentionally,
50 // as if you are here, you must have called PrepareOptions explicitly.
51 Status status = Status::OK();
52 for (const auto& opt_iter : options_) {
53 if (opt_iter.type_map != nullptr) {
54 for (const auto& map_iter : *(opt_iter.type_map)) {
55 auto& opt_info = map_iter.second;
56 if (opt_info.ShouldPrepare()) {
57 status = opt_info.Prepare(opts, map_iter.first,
58 ApplyOffset(this, opt_iter.opt_offset));
59 if (!status.ok()) {
60 return status;
61 }
62 }
63 }
64 }
65 }
66 return status;
67}

69Status Configurable::ValidateOptions(const DBOptions& db_opts,
70 const ColumnFamilyOptions& cf_opts) const {
71 Status status;
72 for (const auto& opt_iter : options_) {
73 if (opt_iter.type_map != nullptr) {
74 for (const auto& map_iter : *(opt_iter.type_map)) {
75 auto& opt_info = map_iter.second;
76 if (opt_info.ShouldValidate()) {
77 status = opt_info.Validate(db_opts, cf_opts, map_iter.first,
78 ApplyOffset(this, opt_iter.opt_offset));
79 if (!status.ok()) {
80 return status;
81 }
82 }
83 }
84 }
85 }
86 return status;
87}

108const OptionTypeInfo* ConfigurableHelper::FindOption(
109 const Configurable& configurable, const std::string& short_name,
110 std::string* opt_name, void** opt_ptr) {
111 for (const auto& iter : configurable.options_) {
112 if (iter.type_map != nullptr) {
113 const auto opt_info =
114 OptionTypeInfo::Find(short_name, *(iter.type_map), opt_name);
115 if (opt_info != nullptr) {
116 *opt_ptr = ApplyOffset(&configurable, iter.opt_offset);
117 return opt_info;
118 }
119 }
120 }
121 return nullptr;

144Status Configurable::ConfigureOptions(
145 const ConfigOptions& config_options,
146 const std::unordered_map<std::string, std::string>& opts_map,
147 std::unordered_map<std::string, std::string>* unused) {
148 std::string curr_opts;
149 Status s;
150 if (!opts_map.empty()) {
151 // There are options in the map.
152 // Save the current configuration in curr_opts and then configure the
153 // options, but do not prepare them now. We will do all the prepare when
154 // the configuration is complete.
155 ConfigOptions copy = config_options;
156 copy.invoke_prepare_options = false;
157 if (!config_options.ignore_unknown_options) {
158 // If we are not ignoring unused, get the defaults in case we need to
159 // reset
160 copy.depth = ConfigOptions::kDepthDetailed;
161 copy.delimiter = "; ";
162 GetOptionString(copy, &curr_opts).PermitUncheckedError();
163 }
164
165 s = ConfigurableHelper::ConfigureOptions(copy, *this, opts_map, unused);
166 }
167 if (config_options.invoke_prepare_options && s.ok()) {
168 s = PrepareOptions(config_options);
169 }
170 if (!s.ok() && !curr_opts.empty()) {
171 ConfigOptions reset = config_options;
172 reset.ignore_unknown_options = true;
173 reset.invoke_prepare_options = true;
174 reset.ignore_unsupported_options = true;
175 // There are some options to reset from this current error
176 ConfigureFromString(reset, curr_opts).PermitUncheckedError();
177 }
178 return s;
179}

186Status Configurable::ConfigureFromString(const ConfigOptions& config_options,
187 const std::string& opts_str) {
188 Status s;
189 if (!opts_str.empty()) {
190 if (opts_str.find(';') != std::string::npos ||
191 opts_str.find('=') != std::string::npos) {
192 std::unordered_map<std::string, std::string> opt_map;
193 s = StringToMap(opts_str, &opt_map);
194 if (s.ok()) {
195 s = ConfigureFromMap(config_options, opt_map, nullptr);
196 }
197 } else {
198 s = ParseStringOptions(config_options, opts_str);
199 if (s.ok() && config_options.invoke_prepare_options) {
200 s = PrepareOptions(config_options);
201 }
202 }
203 } else if (config_options.invoke_prepare_options) {
204 s = PrepareOptions(config_options);
205 } else {
206 s = Status::OK();
207 }
208 return s;
209}

229Status Configurable::ParseOption(const ConfigOptions& config_options,
230 const OptionTypeInfo& opt_info,
231 const std::string& opt_name,
232 const std::string& opt_value, void* opt_ptr) {
233 if (opt_info.IsMutable()) {
234 if (config_options.mutable_options_only) {
235 // This option is mutable. Treat all of its children as mutable as well
236 ConfigOptions copy = config_options;
237 copy.mutable_options_only = false;
238 return opt_info.Parse(copy, opt_name, opt_value, opt_ptr);
239 } else {
240 return opt_info.Parse(config_options, opt_name, opt_value, opt_ptr);
241 }
242 } else if (config_options.mutable_options_only) {
243 return Status::InvalidArgument("Option not changeable: " + opt_name);
244 } else {
245 return opt_info.Parse(config_options, opt_name, opt_value, opt_ptr);
246 }

250Status ConfigurableHelper::ConfigureOptions(
251 const ConfigOptions& config_options, Configurable& configurable,
252 const std::unordered_map<std::string, std::string>& opts_map,
253 std::unordered_map<std::string, std::string>* unused) {
254 std::unordered_map<std::string, std::string> remaining = opts_map;
255 Status s = Status::OK();
256 if (!opts_map.empty()) {
257 for (const auto& iter : configurable.options_) {
258 if (iter.type_map != nullptr) {
259 s = ConfigureSomeOptions(config_options, configurable, *(iter.type_map),
260 &remaining,
261 ApplyOffset(&configurable, iter.opt_offset));
262 if (remaining.empty()) { // Are there more options left?
263 break;
264 } else if (!s.ok()) {
265 break;
266 }
267 }
268 }
269 }
270 if (unused != nullptr && !remaining.empty()) {
271 unused->insert(remaining.begin(), remaining.end());
272 }
273 if (config_options.ignore_unknown_options) {
274 s = Status::OK();
275 } else if (s.ok() && unused == nullptr && !remaining.empty()) {
276 s = Status::NotFound("Could not find option: ", remaining.begin()->first);
277 }
278 return s;
279}

293Status ConfigurableHelper::ConfigureSomeOptions(
294 const ConfigOptions& config_options, Configurable& configurable,
295 const std::unordered_map<std::string, OptionTypeInfo>& type_map,
296 std::unordered_map<std::string, std::string>* options, void* opt_ptr) {
297 Status result = Status::OK(); // The last non-OK result (if any)
298 Status notsup = Status::OK(); // The last NotSupported result (if any)
299 std::string elem_name;
300 int found = 1;
301 std::unordered_set<std::string> unsupported;
302 // While there are unused properties and we processed at least one,
303 // go through the remaining unused properties and attempt to configure them.
304 while (found > 0 && !options->empty()) {
305 found = 0;
306 notsup = Status::OK();
307 for (auto it = options->begin(); it != options->end();) {
308 const std::string& opt_name = configurable.GetOptionName(it->first);
309 const std::string& opt_value = it->second;
310 const auto opt_info =
311 OptionTypeInfo::Find(opt_name, type_map, &elem_name);
312 if (opt_info == nullptr) { // Did not find the option. Skip it
313 ++it;
314 } else {
315 Status s = ConfigureOption(config_options, configurable, *opt_info,
316 opt_name, elem_name, opt_value, opt_ptr);
317 if (s.IsNotFound()) {
318 ++it;
319 } else if (s.IsNotSupported()) {
320 notsup = s;
321 unsupported.insert(it->first);
322 ++it; // Skip it for now
323 } else {
324 found++;
325 it = options->erase(it);
326 if (!s.ok()) {
327 result = s;
328 }
329 }
330 }
331 } // End for all remaining options
332 } // End while found one or options remain
333
334 // Now that we have been through the list, remove any unsupported
335 for (const auto& u : unsupported) {
336 auto it = options->find(u);
337 if (it != options->end()) {
338 options->erase(it);
339 }
340 }
341 if (config_options.ignore_unknown_options) {
342 if (!result.ok()) {
343 result.PermitUncheckedError();
344 }
345 if (!notsup.ok()) {
346 notsup.PermitUncheckedError();
347 }
348 return Status::OK();
349 } else if (!result.ok()) {
350 if (!notsup.ok()) {
351 notsup.PermitUncheckedError();
352 }
353 return result;
354 } else if (config_options.ignore_unsupported_options) {
355 if (!notsup.ok()) {
356 notsup.PermitUncheckedError();
357 }
358 return Status::OK();
359 } else {
360 return notsup;
361 }
362}

364Status ConfigurableHelper::ConfigureSingleOption(
365 const ConfigOptions& config_options, Configurable& configurable,
366 const std::string& name, const std::string& value) {
367 const std::string& opt_name = configurable.GetOptionName(name);
368 std::string elem_name;
369 void* opt_ptr = nullptr;
370 const auto opt_info =
371 FindOption(configurable, opt_name, &elem_name, &opt_ptr);
372 if (opt_info == nullptr) {
373 return Status::NotFound("Could not find option: ", name);
374 } else {
375 return ConfigureOption(config_options, configurable, *opt_info, opt_name,
376 elem_name, value, opt_ptr);

379Status ConfigurableHelper::ConfigureCustomizableOption(
380 const ConfigOptions& config_options, Configurable& configurable,
381 const OptionTypeInfo& opt_info, const std::string& opt_name,
382 const std::string& name, const std::string& value, void* opt_ptr) {
383 Customizable* custom = opt_info.AsRawPointer<Customizable>(opt_ptr);
384 ConfigOptions copy = config_options;
385 if (opt_info.IsMutable()) {
386 // This option is mutable. Pass that property on to any subsequent calls
387 copy.mutable_options_only = false;
388 }
389
390 if (opt_info.IsMutable() || !config_options.mutable_options_only) {
391 // Either the option is mutable, or we are processing all of the options
392 if (opt_name == name || name == OptionTypeInfo::kIdPropName() ||
393 EndsWith(opt_name, OptionTypeInfo::kIdPropSuffix())) {
394 return configurable.ParseOption(copy, opt_info, name, value, opt_ptr);
395 } else if (value.empty()) {
396 return Status::OK();
397 } else if (custom == nullptr || !StartsWith(name, custom->GetId() + ".")) {
398 return configurable.ParseOption(copy, opt_info, name, value, opt_ptr);
399 } else if (value.find('=') != std::string::npos) {
400 return custom->ConfigureFromString(copy, value);
401 } else {
402 return custom->ConfigureOption(copy, name, value);
403 }
404 } else {
405 // We are processing immutable options, which means that we cannot change
406 // the Customizable object itself, but could change its mutable properties.
407 // Check to make sure that nothing is trying to change the Customizable
408 if (custom == nullptr) {
409 // We do not have a Customizable to configure. This is OK if the
410 // value is empty (nothing being configured) but an error otherwise
411 if (value.empty()) {
412 return Status::OK();
413 } else {
414 return Status::InvalidArgument("Option not changeable: " + opt_name);
415 }
416 } else if (EndsWith(opt_name, OptionTypeInfo::kIdPropSuffix()) ||
417 name == OptionTypeInfo::kIdPropName()) {
418 // We have a property of the form "id=value" or "table.id=value"
419 // This is OK if we ID/value matches the current customizable object
420 if (custom->GetId() == value) {
421 return Status::OK();
422 } else {
423 return Status::InvalidArgument("Option not changeable: " + opt_name);
424 }
425 } else if (opt_name == name) {
426 // The properties are of one of forms:
427 // name = { id = id; prop1 = value1; ... }
428 // name = { prop1=value1; prop2=value2; ... }
429 // name = ID
430 // Convert the value to a map and extract the ID
431 // If the ID does not match that of the current customizable, return an
432 // error. Otherwise, update the current customizable via the properties
433 // map
434 std::unordered_map<std::string, std::string> props;
435 std::string id;
436 Status s =
437 Configurable::GetOptionsMap(value, custom->GetId(), &id, &props);
438 if (!s.ok()) {
439 return s;
440 } else if (custom->GetId() != id) {
441 return Status::InvalidArgument("Option not changeable: " + opt_name);
442 } else if (props.empty()) {
443 return Status::OK();
444 } else {
445 return custom->ConfigureFromMap(copy, props);
446 }
447 } else {
448 // Attempting to configure one of the properties of the customizable
449 // Let it through
450 return custom->ConfigureOption(copy, name, value);
451 }
452 }
453}

455Status ConfigurableHelper::ConfigureOption(
456 const ConfigOptions& config_options, Configurable& configurable,
457 const OptionTypeInfo& opt_info, const std::string& opt_name,
458 const std::string& name, const std::string& value, void* opt_ptr) {
459 if (opt_info.IsCustomizable()) {
460 return ConfigureCustomizableOption(config_options, configurable, opt_info,
461 opt_name, name, value, opt_ptr);
462 } else if (opt_name == name) {
463 return configurable.ParseOption(config_options, opt_info, opt_name, value,
464 opt_ptr);
465 } else if (opt_info.IsStruct() || opt_info.IsConfigurable()) {
466 return configurable.ParseOption(config_options, opt_info, name, value,
467 opt_ptr);
468 } else {
469 return Status::NotFound("Could not find option: ", name);
470 }
471}

513Status ConfigurableHelper::GetOption(const ConfigOptions& config_options,
514 const Configurable& configurable,
515 const std::string& short_name,
516 std::string* value) {
517 // Look for option directly
518 assert(value);
519 value->clear();
520
521 std::string opt_name;
522 void* opt_ptr = nullptr;
523 const auto opt_info =
524 FindOption(configurable, short_name, &opt_name, &opt_ptr);
525 if (opt_info != nullptr) {
526 ConfigOptions embedded = config_options;
527 embedded.delimiter = ";";
528 if (short_name == opt_name) {
529 return opt_info->Serialize(embedded, opt_name, opt_ptr, value);
530 } else if (opt_info->IsStruct()) {
531 return opt_info->Serialize(embedded, opt_name, opt_ptr, value);
532 } else if (opt_info->IsConfigurable()) {
533 auto const* config = opt_info->AsRawPointer<Configurable>(opt_ptr);
534 if (config != nullptr) {
535 return config->GetOption(embedded, opt_name, value);
536 }
537 }
538 }
539 return Status::NotFound("Cannot find option: ", short_name);
540}

542Status ConfigurableHelper::SerializeOptions(const ConfigOptions& config_options,
543 const Configurable& configurable,
544 const std::string& prefix,
545 std::string* result) {
546 assert(result);
547 for (auto const& opt_iter : configurable.options_) {
548 if (opt_iter.type_map != nullptr) {
549 for (const auto& map_iter : *(opt_iter.type_map)) {
550 const auto& opt_name = map_iter.first;
551 const auto& opt_info = map_iter.second;
552 if (opt_info.ShouldSerialize()) {
553 std::string value;
554 Status s;
555 void* opt_ptr = ApplyOffset(&configurable, opt_iter.opt_offset);
556 if (!config_options.mutable_options_only) {
557 s = opt_info.Serialize(config_options, prefix + opt_name, opt_ptr,
558 &value);
559 } else if (opt_info.IsMutable()) {
560 ConfigOptions copy = config_options;
561 copy.mutable_options_only = false;
562 s = opt_info.Serialize(copy, prefix + opt_name, opt_ptr, &value);
563 } else if (opt_info.IsConfigurable()) {
564 // If it is a Configurable and we are either printing all of the
565 // details or not printing only the name, this option should be
566 // included in the list
567 if (config_options.IsDetailed() ||
568 !opt_info.IsEnabled(OptionTypeFlags::kStringNameOnly)) {
569 s = opt_info.Serialize(config_options, prefix + opt_name, opt_ptr,
570 &value);
571 }
572 }
573 if (!s.ok()) {
574 return s;
575 } else if (!value.empty()) {
576 // <prefix><opt_name>=<value><delimiter>
577 result->append(prefix + opt_name + "=" + value +
578 config_options.delimiter);
579 }
580 }
581 }
582 }
583 }
584 return Status::OK();
585}

599Status ConfigurableHelper::ListOptions(
600 const ConfigOptions& config_options, const Configurable& configurable,
601 const std::string& prefix, std::unordered_set<std::string>* result) {
602 Status status;
603 for (auto const& opt_iter : configurable.options_) {
604 if (opt_iter.type_map != nullptr) {
605 for (const auto& map_iter : *(opt_iter.type_map)) {
606 const auto& opt_name = map_iter.first;
607 const auto& opt_info = map_iter.second;
608 // If the option is no longer used in rocksdb and marked as deprecated,
609 // we skip it in the serialization.
610 if (!opt_info.IsDeprecated() && !opt_info.IsAlias()) {
611 if (!config_options.mutable_options_only) {
612 result->emplace(prefix + opt_name);
613 } else if (opt_info.IsMutable()) {
614 result->emplace(prefix + opt_name);
615 }
616 }
617 }
618 }
619 }
620 return status;

633 name->clear();
634 if (this == other || config_options.IsCheckDisabled()) {

644bool Configurable::OptionsAreEqual(const ConfigOptions& config_options,
645 const OptionTypeInfo& opt_info,
646 const std::string& opt_name,
647 const void* const this_ptr,
648 const void* const that_ptr,
649 std::string* mismatch) const {
650 if (opt_info.AreEqual(config_options, opt_name, this_ptr, that_ptr,
651 mismatch)) {
652 return true;
653 } else if (opt_info.AreEqualByName(config_options, opt_name, this_ptr,
654 that_ptr)) {
655 *mismatch = "";
656 return true;
657 } else {
658 return false;
659 }
660}
661
662bool ConfigurableHelper::AreEquivalent(const ConfigOptions& config_options,
663 const Configurable& this_one,
664 const Configurable& that_one,
665 std::string* mismatch) {
666 assert(mismatch != nullptr);
667 for (auto const& o : this_one.options_) {
668 const auto this_offset = this_one.GetOptionsPtr(o.name);
669 const auto that_offset = that_one.GetOptionsPtr(o.name);
670 if (this_offset != that_offset) {
671 if (this_offset == nullptr || that_offset == nullptr) {
672 return false;
673 } else if (o.type_map != nullptr) {
674 for (const auto& map_iter : *(o.type_map)) {
675 const auto& opt_info = map_iter.second;
676 if (config_options.IsCheckEnabled(opt_info.GetSanityLevel())) {
677 if (!config_options.mutable_options_only) {
678 if (!this_one.OptionsAreEqual(config_options, opt_info,
679 map_iter.first, this_offset,
680 that_offset, mismatch)) {
681 return false;
682 }
683 } else if (opt_info.IsMutable()) {
684 ConfigOptions copy = config_options;
685 copy.mutable_options_only = false;
686 if (!this_one.OptionsAreEqual(copy, opt_info, map_iter.first,
687 this_offset, that_offset,
688 mismatch)) {
689 return false;
690 }
691 }
692 }
693 }
694 }
695 }
696 }
697 return true;
698}

700Status Configurable::GetOptionsMap(
701 const std::string& value, const std::string& default_id, std::string* id,
702 std::unordered_map<std::string, std::string>* props) {
703 assert(id);
704 assert(props);
705 Status status;
706 if (value.empty() || value == kNullptrString) {
707 *id = default_id;
708 } else if (value.find('=') == std::string::npos) {
709 *id = value;
710 } else {
711 status = StringToMap(value, props);
712 if (!status.ok()) { // There was an error creating the map.
713 *id = value; // Treat the value as id
714 props->clear(); // Clear the properties
715 status = Status::OK(); // and ignore the error
716 } else {
717 auto iter = props->find(OptionTypeInfo::kIdPropName());
718 if (iter != props->end()) {
