85 void BarrierInit(int count) {
86 MutexLock l(&mutex_);

88 count_ = count;

91 void BarrierWait() {
92 MutexLock l(&mutex_);
93 if (!barrier_) {
94 return;
95 }
96 if (--count_ == 0) {
97 if (reinit_count_ > 0) {
98 count_ = reinit_count_;
99 reinit_count_ = 0;
100 } else {
101 barrier_ = false;
102 }
103 cv_.SignalAll();
104 } else {
105 cv_.Wait();
106 }
107 }

109 void BarrierWaitAndReinit(int count) {
110 MutexLock l(&mutex_);
111 if (!barrier_) {
112 return;
113 }
114 reinit_count_ = count;
115 if (--count_ == 0) {
116 if (reinit_count_ > 0) {
117 count_ = reinit_count_;
118 reinit_count_ = 0;
119 } else {
120 barrier_ = false;
121 }
122 cv_.SignalAll();
123 } else {
124 cv_.Wait();
125 }
126 }

371// This test creates 4 L0 files and compacts them. The follower, during catchup,
372// successfully instantiates 4 Versions corresponding to the 4 files (but
373// donesn't install them yet), followed by deleting those 4 and adding a new
374// file from compaction. The test verifies that the 4 L0 files are deleted
375// correctly by the follower.
376// We use teh Barrier* functions to ensure that the follower first sees the 4
377// L0 files and is able to link them, and then sees the compaction that
378// obsoletes those L0 files (so those L0 files are intermediates that it has
379// to explicitly delete). Suppose we don't have any barriers, its possible
380// the follower reads the L0 records and compaction records from the MANIFEST
381// in one read, which means those L0 files would have already been deleted
382// by the leader and the follower cannot link to them.
383TEST_F(DBFollowerTest, IntermediateObsoleteFiles) {
384 Options opts = CurrentOptions();
385 opts.disable_auto_compactions = true;
386 Reopen(opts);
387 ASSERT_OK(OpenAsFollower());
388
389 follower_fs()->BarrierInit(2);
390 ASSERT_OK(Put("k1", "v1"));
391 ASSERT_OK(Flush());
392 ASSERT_OK(Put("k1", "v2"));
393 ASSERT_OK(Flush());
394 ASSERT_OK(Put("k1", "v3"));
395 ASSERT_OK(Flush());
396 ASSERT_OK(Put("k1", "v4"));
397 ASSERT_OK(Flush());
398 follower_fs()->BarrierWaitAndReinit(2);
399
400 ASSERT_OK(dbfull()->TEST_CompactRange(0, nullptr, nullptr, nullptr, true));
401 follower_fs()->BarrierWait();
402
403 SyncPoint::GetInstance()->LoadDependency({
404 {"DBImplFollower::TryCatchupWithLeader:End",
405 "Follower::WaitForCatchup:1"},
406 });
407 SyncPoint::GetInstance()->EnableProcessing();
408 TEST_SYNC_POINT("Follower::WaitForCatchup:1");
409 CheckDirs();
410 ASSERT_EQ(FollowerGet("k1"), "v4");
411}

413// This test verifies a scenario where the follower can recover a Version
414// partially (i.e some of the additions cannot be found), and the files
415// that are found are obsoleted by a subsequent VersionEdit.
416TEST_F(DBFollowerTest, PartialVersionRecovery) {
417 Options opts = CurrentOptions();
418 opts.disable_auto_compactions = true;
419 opts.sst_partitioner_factory =
420 std::make_shared<DBFollowerTestSstPartitionerFactory>(1);
421 Reopen(opts);
422
423 ASSERT_OK(Put("k1", "v1"));
424 ASSERT_OK(Put("k2", "v1"));
425 ASSERT_OK(Put("k3", "v1"));
426 ASSERT_OK(Flush());
427 MoveFilesToLevel(2);
428
429 ASSERT_OK(Put("k1", "v2"));
430 ASSERT_OK(Flush());
431 ASSERT_OK(Put("k3", "v2"));
432 ASSERT_OK(Flush());
433 MoveFilesToLevel(1);
434
435 ASSERT_OK(OpenAsFollower());
436 ASSERT_OK(dbfull()->SetOptions(dbfull()->DefaultColumnFamily(),
437 {{"max_compaction_bytes", "1"}}));
438
439 follower_fs()->BarrierInit(2);
440 Slice key("k1");
441 ASSERT_OK(dbfull()->TEST_CompactRange(1, &key, &key, nullptr, true));
442
443 follower_fs()->BarrierWaitAndReinit(2);
444
445 // The second compaction input overlaps the previous compaction outputs
446 // by one file. This file is never added to VersionStorageInfo since it
447 // was added and deleted before the catch up completes. We later verify that
448 // the follower correctly deleted this file.
449 key = Slice("k3");
450 ASSERT_OK(dbfull()->TEST_CompactRange(1, &key, &key, nullptr, true));
451 follower_fs()->BarrierWait();
452
453 SyncPoint::GetInstance()->LoadDependency({
454 {"DBImplFollower::TryCatchupWithLeader:End",
455 "Follower::WaitForCatchup:1"},
456 });
457 SyncPoint::GetInstance()->EnableProcessing();
458 TEST_SYNC_POINT("Follower::WaitForCatchup:1");
459 CheckDirs();
460 ASSERT_EQ(FollowerGet("k1"), "v2");
461 ASSERT_EQ(FollowerGet("k2"), "v1");
462 ASSERT_EQ(FollowerGet("k3"), "v2");
463 SyncPoint::GetInstance()->DisableProcessing();
464}
