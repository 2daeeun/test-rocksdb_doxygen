38// Manages caching for TableReader objects for a column family. The actual
39// cache is allocated separately and passed to the constructor. TableCache
40// wraps around the underlying SST file readers by providing Get(),
41// MultiGet() and NewIterator() methods that hide the instantiation,
42// caching and access to the TableReader. The main purpose of this is
43// performance - by caching the TableReader, it avoids unnecessary file opens
44// and object allocation and instantiation. One exception is compaction, where
45// a new TableReader may be instantiated - see NewIterator() comments
46//
47// Another service provided by TableCache is managing the row cache - if the
48// DB is configured with a row cache, and the lookup key is present in the row
49// cache, lookup is very fast. The row cache is obtained from
50// ioptions.row_cache
51class TableCache {

58 ~TableCache();

70 // Return an iterator for the specified file number (the corresponding
71 // file length must be exactly "file_size" bytes). If "table_reader_ptr"
72 // is non-nullptr, also sets "*table_reader_ptr" to point to the Table object
73 // underlying the returned iterator, or nullptr if no Table object underlies
74 // the returned iterator. The returned "*table_reader_ptr" object is owned
75 // by the cache and should not be deleted, and is valid for as long as the
76 // returned iterator is live.
77 // If !options.ignore_range_deletions, and range_del_iter is non-nullptr,
78 // then range_del_iter is set to a TruncatedRangeDelIterator for range
79 // tombstones in the SST file corresponding to the specified file number. The
80 // upper/lower bounds for the TruncatedRangeDelIterator are set to the SST
81 // file's boundary.
82 // @param options Must outlive the returned iterator.
83 // @param range_del_agg If non-nullptr, adds range deletions to the
84 // aggregator. If an error occurs, returns it in a NewErrorInternalIterator
85 // @param for_compaction If true, a new TableReader may be allocated (but
86 // not cached), depending on the CF options
87 // @param skip_filters Disables loading/accessing the filter block
88 // @param level The level this table is at, -1 for "not set / don't know"
89 // @param range_del_read_seqno If non-nullptr, will be used to create
90 // *range_del_iter.
91 InternalIterator* NewIterator(
92 const ReadOptions& options, const FileOptions& toptions,
93 const InternalKeyComparator& internal_comparator,

96 TableReader** table_reader_ptr, HistogramImpl* file_read_hist,

100 const InternalKey* largest_compaction_key, bool allow_unprepared_value,
101 const SequenceNumber* range_del_read_seqno = nullptr,
102 std::unique_ptr<TruncatedRangeDelIterator>* range_del_iter = nullptr);

234 // Capacity of the backing Cache that indicates infinite TableCache capacity.
235 // For example when max_open_files is -1 we set the backing Cache to this.
236 static const int kInfiniteCapacity = 0x400000;
237
238 // The tables opened with this TableCache will be immortal, i.e., their
239 // lifetime is as long as that of the DB.
240 void SetTablesAreImmortal() {
241 if (cache_.get()->GetCapacity() >= kInfiniteCapacity) {
242 immortal_tables_ = true;
243 }
