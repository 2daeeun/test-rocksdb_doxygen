15std::string EncodeSessionId(uint64_t upper, uint64_t lower) {
16 std::string db_session_id(20U, '\0');
17 char *buf = db_session_id.data();
18 // Preserving `lower` is slightly tricky. 36^12 is slightly more than
19 // 62 bits, so we use 12 chars plus the bottom two bits of one more.
20 // (A tiny fraction of 20 digit strings go unused.)
21 uint64_t a = (upper << 2) | (lower >> 62);
22 uint64_t b = lower & (UINT64_MAX >> 2);
23 PutBaseChars<36>(&buf, 8, a, /*uppercase*/ true);
24 PutBaseChars<36>(&buf, 12, b, /*uppercase*/ true);
25 assert(buf == &db_session_id.back() + 1);
26 return db_session_id;
27}

29Status DecodeSessionId(const std::string &db_session_id, uint64_t *upper,
30 uint64_t *lower) {
31 const size_t len = db_session_id.size();
32 if (len == 0) {
33 return Status::NotSupported("Missing db_session_id");
34 }
35 // Anything from 13 to 24 chars is reasonable. We don't have to limit to
36 // exactly 20.
37 if (len < 13) {
38 return Status::NotSupported("Too short db_session_id");
39 }
40 if (len > 24) {
41 return Status::NotSupported("Too long db_session_id");
42 }
43 uint64_t a = 0, b = 0;
44 const char *buf = &db_session_id.front();
45 bool success = ParseBaseChars<36>(&buf, len - 12U, &a);
46 if (!success) {
47 return Status::NotSupported("Bad digit in db_session_id");
48 }
49 success = ParseBaseChars<36>(&buf, 12U, &b);
50 if (!success) {
51 return Status::NotSupported("Bad digit in db_session_id");
52 }
53 assert(buf == &db_session_id.back() + 1);
54 *upper = a >> 2;
55 *lower = (b & (UINT64_MAX >> 2)) | (a << 62);
56 return Status::OK();
57}

59Status GetSstInternalUniqueId(const std::string &db_id,
60 const std::string &db_session_id,
61 uint64_t file_number, UniqueIdPtr out,
62 bool force) {
63 if (!force) {
64 if (db_id.empty()) {
65 return Status::NotSupported("Missing db_id");
66 }
67 if (file_number == 0) {
68 return Status::NotSupported("Missing or bad file number");
69 }
70 if (db_session_id.empty()) {
71 return Status::NotSupported("Missing db_session_id");
72 }
73 }
74 uint64_t session_upper = 0; // Assignment to appease clang-analyze
75 uint64_t session_lower = 0; // Assignment to appease clang-analyze
76 {
77 Status s = DecodeSessionId(db_session_id, &session_upper, &session_lower);
78 if (!s.ok()) {
79 if (!force) {
80 return s;
81 } else {
82 // A reasonable fallback in case malformed
83 Hash2x64(db_session_id.data(), db_session_id.size(), &session_upper,
84 &session_lower);
85 if (session_lower == 0) {
86 session_lower = session_upper | 1;
87 }
88 }
89 }
90 }
91
92 // Exactly preserve session lower to ensure that session ids generated
93 // during the same process lifetime are guaranteed unique.
94 // DBImpl also guarantees (in recent versions) that this is not zero,
95 // so that we can guarantee unique ID is never all zeros. (Can't assert
96 // that here because of testing and old versions.)
97 // We put this first in anticipation of matching a small-ish set of cache
98 // key prefixes to cover entries relevant to any DB.
99 out.ptr[0] = session_lower;
100
101 // Hash the session upper (~39 bits entropy) and DB id (120+ bits entropy)
102 // for very high global uniqueness entropy.
103 // (It is possible that many DBs descended from one common DB id are copied
104 // around and proliferate, in which case session id is critical, but it is
105 // more common for different DBs to have different DB ids.)
106 uint64_t db_a, db_b;
107 Hash2x64(db_id.data(), db_id.size(), session_upper, &db_a, &db_b);
108
109 // Xor in file number for guaranteed uniqueness by file number for a given
110 // session and DB id. (Xor slightly better than + here. See
111 // https://github.com/pdillinger/unique_id )
112 out.ptr[1] = db_a ^ file_number;
113
114 // Extra (optional) global uniqueness
115 if (out.extended) {
116 out.ptr[2] = db_b;
117 }
118
119 return Status::OK();
120}

123// For InternalUniqueIdToExternal / ExternalUniqueIdToInternal we want all
124// zeros in first 128 bits to map to itself, so that excluding zero in
125// internal IDs (session_lower != 0 above) does the same for external IDs.
126// These values are meaningless except for making that work.
127constexpr uint64_t kHiOffsetForZero = 17391078804906429400U;
128constexpr uint64_t kLoOffsetForZero = 6417269962128484497U;
129} // namespace
130
131void InternalUniqueIdToExternal(UniqueIdPtr in_out) {
132 uint64_t hi, lo;
133 BijectiveHash2x64(in_out.ptr[1] + kHiOffsetForZero,
134 in_out.ptr[0] + kLoOffsetForZero, &hi, &lo);
135 in_out.ptr[0] = lo;
136 in_out.ptr[1] = hi;
137 if (in_out.extended) {
138 in_out.ptr[2] += lo + hi;
139 }
140}
141
142void ExternalUniqueIdToInternal(UniqueIdPtr in_out) {
143 uint64_t lo = in_out.ptr[0];
144 uint64_t hi = in_out.ptr[1];
145 if (in_out.extended) {
146 in_out.ptr[2] -= lo + hi;
147 }
148 BijectiveUnhash2x64(hi, lo, &hi, &lo);
149 in_out.ptr[0] = lo - kLoOffsetForZero;
150 in_out.ptr[1] = hi - kHiOffsetForZero;
151}

153std::string EncodeUniqueIdBytes(UniqueIdPtr in) {
154 std::string ret(in.extended ? 24U : 16U, '\0');
155 EncodeFixed64(ret.data(), in.ptr[0]);
156 EncodeFixed64(&ret[8], in.ptr[1]);
157 if (in.extended) {
158 EncodeFixed64(&ret[16], in.ptr[2]);
159 }
160 return ret;
161}

163Status DecodeUniqueIdBytes(const std::string &unique_id, UniqueIdPtr out) {
164 if (unique_id.size() != (out.extended ? 24 : 16)) {
165 return Status::NotSupported("Not a valid unique_id");
166 }
167 const char *buf = &unique_id.front();
168 out.ptr[0] = DecodeFixed64(&buf[0]);
169 out.ptr[1] = DecodeFixed64(&buf[8]);
170 if (out.extended) {
171 out.ptr[2] = DecodeFixed64(&buf[16]);
172 }
173 return Status::OK();
174}

176template <typename ID>
177Status GetUniqueIdFromTablePropertiesHelper(const TableProperties &props,
178 std::string *out_id) {
179 ID tmp{};
180 Status s = GetSstInternalUniqueId(props.db_id, props.db_session_id,
181 props.orig_file_number, &tmp);
182 if (s.ok()) {
183 InternalUniqueIdToExternal(&tmp);
184 *out_id = EncodeUniqueIdBytes(&tmp);
185 } else {
186 out_id->clear();
187 }
188 return s;
189}

201std::string UniqueIdToHumanString(const std::string &id) {
202 // Not so efficient, but that's OK
