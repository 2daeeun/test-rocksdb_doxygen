48bool RandomTransactionInserter::TransactionDBInsert(
49 TransactionDB* db, const TransactionOptions& txn_options) {
50 txn_ = db->BeginTransaction(write_options_, txn_options, txn_);
51
52 std::hash<std::thread::id> hasher;
53 char name[64];
54 snprintf(name, 64, "txn%" ROCKSDB_PRIszt "-%" PRIu64,
55 hasher(std::this_thread::get_id()), txn_id_++);
56 assert(strlen(name) < 64 - 1);
57 assert(txn_->SetName(name).ok());
58
59 // Take a snapshot if set_snapshot was not set or with 50% change otherwise
60 bool take_snapshot = txn_->GetSnapshot() == nullptr || rand_->OneIn(2);
61 if (take_snapshot) {
62 txn_->SetSnapshot();
63 read_options_.snapshot = txn_->GetSnapshot();
64 }
65 auto res = DoInsert(db, txn_, false);
66 if (take_snapshot) {
67 read_options_.snapshot = nullptr;
68 }
69 return res;
70}

85Status RandomTransactionInserter::DBGet(
86 DB* db, Transaction* txn, ReadOptions& read_options, uint16_t set_i,
87 uint64_t ikey, bool get_for_update, uint64_t* int_value,
88 std::string* full_key, bool* unexpected_error) {
89 Status s;
90 // Five digits (since the largest uint16_t is 65535) plus the NUL
91 // end char.
92 char prefix_buf[6] = {0};
93 // Pad prefix appropriately so we can iterate over each set
94 assert(set_i + 1 <= 9999);
95 snprintf(prefix_buf, sizeof(prefix_buf), "%.4u", set_i + 1);
96 // key format: [SET#][random#]
97 std::string skey = std::to_string(ikey);
98 Slice base_key(skey);
99 *full_key = std::string(prefix_buf) + base_key.ToString();
100 Slice key(*full_key);
101
102 std::string value;
103 if (txn != nullptr) {
104 if (get_for_update) {
105 s = txn->GetForUpdate(read_options, key, &value);
106 } else {
107 s = txn->Get(read_options, key, &value);
108 }
109 } else {
110 s = db->Get(read_options, key, &value);
111 }
112
113 if (s.ok()) {
114 // Found key, parse its value
115 *int_value = std::stoull(value);
116 if (*int_value == 0 || *int_value == ULONG_MAX) {
117 *unexpected_error = true;
118 fprintf(stderr, "Get returned unexpected value: %s\n", value.c_str());
119 s = Status::Corruption();
120 }
121 } else if (s.IsNotFound()) {
122 // Have not yet written to this key, so assume its value is 0
123 *int_value = 0;
124 s = Status::OK();
125 }
126 return s;
127}

129bool RandomTransactionInserter::DoInsert(DB* db, Transaction* txn,
130 bool is_optimistic) {
131 Status s;
132 WriteBatch batch;
133
134 // pick a random number to use to increment a key in each set
135 uint64_t incr = (rand_->Next() % 100) + 1;
136 bool unexpected_error = false;
137
138 std::vector<uint16_t> set_vec(num_sets_);
139 std::iota(set_vec.begin(), set_vec.end(), static_cast<uint16_t>(0));
140 RandomShuffle(set_vec.begin(), set_vec.end());
141
142 // For each set, pick a key at random and increment it
143 for (uint16_t set_i : set_vec) {
144 uint64_t int_value = 0;
145 std::string full_key;
146 uint64_t rand_key = rand_->Next() % num_keys_;
147 const bool get_for_update = txn ? rand_->OneIn(2) : false;
148 s = DBGet(db, txn, read_options_, set_i, rand_key, get_for_update,
149 &int_value, &full_key, &unexpected_error);
150 Slice key(full_key);
151 if (!s.ok()) {
152 // Optimistic transactions should never return non-ok status here.
153 // Non-optimistic transactions may return write-coflict/timeout errors.
154 if (is_optimistic || !(s.IsBusy() || s.IsTimedOut() || s.IsTryAgain())) {
155 fprintf(stderr, "Get returned an unexpected error: %s\n",
156 s.ToString().c_str());
157 unexpected_error = true;
158 }
159 break;
160 }
161
162 if (s.ok()) {
163 // Increment key
164 std::string sum = std::to_string(int_value + incr);
165 if (txn != nullptr) {
166 if ((set_i % 4) != 0) {
167 s = txn->SingleDelete(key);
168 } else {
169 s = txn->Delete(key);
170 }
171 if (!get_for_update && (s.IsBusy() || s.IsTimedOut())) {
172 // If the initial get was not for update, then the key is not locked
173 // before put and put could fail due to concurrent writes.
174 break;
175 } else if (!s.ok()) {
176 // Since we did a GetForUpdate, SingleDelete should not fail.
177 fprintf(stderr, "SingleDelete returned an unexpected error: %s\n",
178 s.ToString().c_str());
179 unexpected_error = true;
180 }
181 s = txn->Put(key, sum);
182 if (!s.ok()) {
183 // Since we did a GetForUpdate, Put should not fail.
184 fprintf(stderr, "Put returned an unexpected error: %s\n",
185 s.ToString().c_str());
186 unexpected_error = true;
187 }
188 } else {
189 batch.Put(key, sum);
190 }
191 bytes_inserted_ += key.size() + sum.size();
192 }
193 if (txn != nullptr) {
194 ROCKS_LOG_DEBUG(db->GetDBOptions().info_log,
195 "Insert (%s) %s snap: %" PRIu64 " key:%s value: %" PRIu64
196 "+%" PRIu64 "=%" PRIu64,
197 txn->GetName().c_str(), s.ToString().c_str(),
198 txn->GetSnapshot()->GetSequenceNumber(), full_key.c_str(),
199 int_value, incr, int_value + incr);
200 }
201 }
202
203 if (s.ok()) {
204 if (txn != nullptr) {
205 bool with_prepare = !is_optimistic && !rand_->OneIn(10);
206 if (with_prepare) {
207 // Also try commit without prepare
208 s = txn->Prepare();
209 if (!s.ok()) {
210 fprintf(stderr, "Prepare returned an unexpected error: %s\n",
211 s.ToString().c_str());
212 }
213 assert(s.ok());
214 ROCKS_LOG_DEBUG(db->GetDBOptions().info_log,
215 "Prepare of %" PRIu64 " %s (%s)", txn->GetId(),
216 s.ToString().c_str(), txn->GetName().c_str());
217 if (rand_->OneIn(20)) {
218 // This currently only tests the mechanics of writing commit time
219 // write batch so the exact values would not matter.
220 s = txn_->GetCommitTimeWriteBatch()->Put("cat", "dog");
221 assert(s.ok());
222 }
223 db->GetDBOptions().env->SleepForMicroseconds(
224 static_cast<int>(cmt_delay_ms_ * 1000));
225 }
226 if (!rand_->OneIn(20)) {
227 s = txn->Commit();
228 assert(!with_prepare || s.ok());
229 ROCKS_LOG_DEBUG(db->GetDBOptions().info_log,
230 "Commit of %" PRIu64 " %s (%s)", txn->GetId(),
231 s.ToString().c_str(), txn->GetName().c_str());
232 } else {
233 // Also try 5% rollback
234 s = txn->Rollback();
235 ROCKS_LOG_DEBUG(db->GetDBOptions().info_log,
236 "Rollback %" PRIu64 " %s %s", txn->GetId(),
237 txn->GetName().c_str(), s.ToString().c_str());
238 assert(s.ok());
239 }
240 assert(is_optimistic || s.ok());
241
242 if (!s.ok()) {
243 if (is_optimistic) {
244 // Optimistic transactions can have write-conflict errors on commit.
245 // Any other error is unexpected.
246 if (!(s.IsBusy() || s.IsTimedOut() || s.IsTryAgain())) {
247 unexpected_error = true;
248 }
249 } else {
250 // Non-optimistic transactions should only fail due to expiration
251 // or write failures. For testing purproses, we do not expect any
252 // write failures.
253 if (!s.IsExpired()) {
254 unexpected_error = true;
255 }
256 }
257
258 if (unexpected_error) {
259 fprintf(stderr, "Commit returned an unexpected error: %s\n",
260 s.ToString().c_str());
261 }
262 }
263 } else {
264 s = db->Write(write_options_, &batch);
265 if (!s.ok()) {
266 unexpected_error = true;
267 fprintf(stderr, "Write returned an unexpected error: %s\n",
268 s.ToString().c_str());
269 }
270 }
271 } else {
272 if (txn != nullptr) {
273 assert(txn->Rollback().ok());
274 ROCKS_LOG_DEBUG(db->GetDBOptions().info_log, "Error %s for txn %s",
275 s.ToString().c_str(), txn->GetName().c_str());
276 }
277 }
278
279 if (s.ok()) {
280 success_count_++;
281 } else {
282 failure_count_++;
283 }
284
285 last_status_ = s;
286
287 // return success if we didn't get any unexpected errors
288 return !unexpected_error;
289}

291// Verify that the sum of the keys in each set are equal
292Status RandomTransactionInserter::Verify(DB* db, uint16_t num_sets,
293 uint64_t num_keys_per_set,
294 bool take_snapshot, Random64* rand,
295 uint64_t delay_ms) {
296 // delay_ms is the delay between taking a snapshot and doing the reads. It
297 // emulates reads from a long-running backup job.
298 assert(delay_ms == 0 || take_snapshot);
299 uint64_t prev_total = 0;
300 uint32_t prev_i = 0;
301 bool prev_assigned = false;
302
303 ReadOptions roptions;
304 if (take_snapshot) {
305 roptions.snapshot = db->GetSnapshot();
306 db->GetDBOptions().env->SleepForMicroseconds(
307 static_cast<int>(delay_ms * 1000));
308 }
309
310 std::vector<uint16_t> set_vec(num_sets);
311 std::iota(set_vec.begin(), set_vec.end(), static_cast<uint16_t>(0));
312 RandomShuffle(set_vec.begin(), set_vec.end());
313
314 // For each set of keys with the same prefix, sum all the values
315 for (uint16_t set_i : set_vec) {
316 // Five digits (since the largest uint16_t is 65535) plus the NUL
317 // end char.
318 char prefix_buf[6];
319 assert(set_i + 1 <= 9999);
320 snprintf(prefix_buf, sizeof(prefix_buf), "%.4u", set_i + 1);
321 uint64_t total = 0;
322
323 // Use either point lookup or iterator. Point lookups are slower so we use
324 // it less often.
325 const bool use_point_lookup =
326 num_keys_per_set != 0 && rand && rand->OneIn(10);
327 if (use_point_lookup) {
328 ReadOptions read_options;
329 for (uint64_t k = 0; k < num_keys_per_set; k++) {
330 std::string dont_care;
331 uint64_t int_value = 0;
332 bool unexpected_error = false;
333 const bool FOR_UPDATE = false;
334 Status s = DBGet(db, nullptr, roptions, set_i, k, FOR_UPDATE,
335 &int_value, &dont_care, &unexpected_error);
336 assert(s.ok());
337 assert(!unexpected_error);
338 total += int_value;
339 }
340 } else { // user iterators
341 Iterator* iter = db->NewIterator(roptions);
342 for (iter->Seek(Slice(prefix_buf, 4)); iter->Valid(); iter->Next()) {
343 Slice key = iter->key();
344 // stop when we reach a different prefix
345 if (key.ToString().compare(0, 4, prefix_buf) != 0) {
346 break;
347 }
348 Slice value = iter->value();
349 uint64_t int_value = std::stoull(value.ToString());
350 if (int_value == 0 || int_value == ULONG_MAX) {
351 fprintf(stderr, "Iter returned unexpected value: %s\n",
352 value.ToString().c_str());
353 return Status::Corruption();
354 }
355 ROCKS_LOG_DEBUG(
356 db->GetDBOptions().info_log,
357 "VerifyRead at %" PRIu64 " (%" PRIu64 "): %.*s value: %" PRIu64,
358 roptions.snapshot ? roptions.snapshot->GetSequenceNumber() : 0ul,
359 roptions.snapshot
360 ? ((SnapshotImpl*)roptions.snapshot)->min_uncommitted_
361 : 0ul,
362 static_cast<int>(key.size()), key.data(), int_value);
363 total += int_value;
364 }
365 iter->status().PermitUncheckedError();
366 delete iter;
367 }
368
369 if (prev_assigned && total != prev_total) {
370 db->GetDBOptions().info_log->Flush();
371 fprintf(stdout,
372 "RandomTransactionVerify found inconsistent totals using "
373 "pointlookup? %d "
374 "Set[%" PRIu32 "]: %" PRIu64 ", Set[%" PRIu32 "]: %" PRIu64
375 " at snapshot %" PRIu64 "\n",
376 use_point_lookup, prev_i, prev_total, set_i, total,
377 roptions.snapshot ? roptions.snapshot->GetSequenceNumber() : 0ul);
378 fflush(stdout);
379 return Status::Corruption();
380 } else {
381 ROCKS_LOG_DEBUG(
382 db->GetDBOptions().info_log,
383 "RandomTransactionVerify pass pointlookup? %d total: %" PRIu64
384 " snap: %" PRIu64,
385 use_point_lookup, total,
386 roptions.snapshot ? roptions.snapshot->GetSequenceNumber() : 0ul);
387 }
388 prev_total = total;
389 prev_i = set_i;
390 prev_assigned = true;
391 }
392 if (take_snapshot) {
393 db->ReleaseSnapshot(roptions.snapshot);
394 }
395
396 return Status::OK();
397}
