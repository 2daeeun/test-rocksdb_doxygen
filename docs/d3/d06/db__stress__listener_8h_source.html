185 void OnTableFileCreated(const TableFileCreationInfo& info) override {
186 assert(info.db_name == db_name_);
187 assert(IsValidColumnFamilyName(info.cf_name));
188 assert(info.job_id > 0 || FLAGS_compact_files_one_in > 0);
189 if (info.status.ok()) {
190 assert(info.file_size > 0);
191 VerifyFilePath(info.file_path);
192 assert(info.table_properties.data_size > 0 ||
193 info.table_properties.num_range_deletions > 0);
194 assert(info.table_properties.raw_key_size > 0);
195 assert(info.table_properties.num_entries > 0);
196 VerifyTableFileUniqueId(info.table_properties, info.file_path);
197 }
198 --num_pending_file_creations_;

227 void OnFileReadFinish(const FileOperationInfo& info) override {
228 // Even empty callback is valuable because sometimes some locks are
229 // released in order to make the callback.
230
231 // Sleep carefully here as it is a frequent operation and we don't want
232 // to slow down the tests. We always sleep when the read is large.
233 // When read is small, sleep in a small chance.
234 size_t length_read = info.length;
235 if (length_read >= 1000000 || Random::GetTLSInstance()->OneIn(1000)) {
236 RandomSleep();
237 }
238 }

240 void OnFileWriteFinish(const FileOperationInfo& info) override {
241 // Even empty callback is valuable because sometimes some locks are
242 // released in order to make the callback.
243
244 // Sleep carefully here as it is a frequent operation and we don't want
245 // to slow down the tests. When the write is large, always sleep.
246 // Otherwise, sleep in a relatively small chance.
247 size_t length_write = info.length;
248 if (length_write >= 1000000 || Random::GetTLSInstance()->OneIn(64)) {
249 RandomSleep();

281 bool IsValidColumnFamilyName(const std::string& cf_name) const {
282 if (cf_name == kDefaultColumnFamilyName) {
283 return true;
284 }
285 // The column family names in the stress tests are numbers.
286 for (size_t i = 0; i < cf_name.size(); ++i) {
287 if (cf_name[i] < '0' || cf_name[i] > '9') {
288 return false;
289 }
290 }
291 return true;
292 }

294 void VerifyFileDir(const std::string& file_dir) {
295#ifndef NDEBUG
296 if (db_name_ == file_dir) {
297 return;
298 }
299 for (const auto& db_path : db_paths_) {
300 if (db_path.path == file_dir) {
301 return;
302 }
303 }
304 for (auto& cf : column_families_) {
305 for (const auto& cf_path : cf.options.cf_paths) {
306 if (cf_path.path == file_dir) {
307 return;
308 }
309 }
310 }
311 assert(false);
312#else
313 (void)file_dir;
314#endif // !NDEBUG
315 }

317 void VerifyFileName(const std::string& file_name) {
318#ifndef NDEBUG
319 uint64_t file_number;
320 FileType file_type;
321 bool result = ParseFileName(file_name, &file_number, &file_type);
322 assert(result);
323 assert(file_type == kTableFile);
324#else
325 (void)file_name;
326#endif // !NDEBUG
327 }
328
329 void VerifyFilePath(const std::string& file_path) {
330#ifndef NDEBUG
331 size_t pos = file_path.find_last_of("/");
332 if (pos == std::string::npos) {
333 VerifyFileName(file_path);
334 } else {
335 if (pos > 0) {
336 VerifyFileDir(file_path.substr(0, pos));
337 }
338 VerifyFileName(file_path.substr(pos));
339 }
340#else
341 (void)file_path;
342#endif // !NDEBUG
343 }
