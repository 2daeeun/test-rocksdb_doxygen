28Status ImportColumnFamilyJob::Prepare(uint64_t next_file_number,
29 SuperVersion* sv) {
30 Status status;
31 std::vector<ColumnFamilyIngestFileInfo> cf_ingest_infos;
32 for (const auto& metadata_per_cf : metadatas_) {
33 // Read the information of files we are importing
34 ColumnFamilyIngestFileInfo cf_file_info;
35 InternalKey smallest, largest;
36 int num_files = 0;
37 std::vector<IngestedFileInfo> files_to_import_per_cf;
38 for (size_t i = 0; i < metadata_per_cf.size(); i++) {
39 auto file_metadata = *metadata_per_cf[i];
40 const auto file_path = file_metadata.db_path + "/" + file_metadata.name;
41 IngestedFileInfo file_to_import;
42 status = GetIngestedFileInfo(file_path, next_file_number++, sv,
43 file_metadata, &file_to_import);
44 if (!status.ok()) {
45 return status;
46 }
47
48 if (file_to_import.num_entries == 0) {
49 status = Status::InvalidArgument("File contain no entries");
50 return status;
51 }
52
53 if (!file_to_import.smallest_internal_key.Valid() ||
54 !file_to_import.largest_internal_key.Valid()) {
55 status = Status::Corruption("File has corrupted keys");
56 return status;
57 }
58
59 files_to_import_per_cf.push_back(file_to_import);
60 num_files++;
61
62 // Calculate the smallest and largest keys of all files in this CF
63 if (i == 0) {
64 smallest = file_to_import.smallest_internal_key;
65 largest = file_to_import.largest_internal_key;
66 } else {
67 if (cfd_->internal_comparator().Compare(
68 smallest, file_to_import.smallest_internal_key) > 0) {
69 smallest = file_to_import.smallest_internal_key;
70 }
71 if (cfd_->internal_comparator().Compare(
72 largest, file_to_import.largest_internal_key) < 0) {
73 largest = file_to_import.largest_internal_key;
74 }
75 }
76 }
77
78 if (num_files == 0) {
79 status = Status::InvalidArgument("The list of files is empty");
80 return status;
81 }
82 files_to_import_.push_back(files_to_import_per_cf);
83 cf_file_info.smallest_internal_key = smallest;
84 cf_file_info.largest_internal_key = largest;
85 cf_ingest_infos.push_back(cf_file_info);
86 }
87
88 std::sort(cf_ingest_infos.begin(), cf_ingest_infos.end(),
89 [this](const ColumnFamilyIngestFileInfo& info1,
90 const ColumnFamilyIngestFileInfo& info2) {
91 return cfd_->user_comparator()->Compare(
92 info1.smallest_internal_key.user_key(),
93 info2.smallest_internal_key.user_key()) < 0;
94 });
95
96 for (size_t i = 0; i + 1 < cf_ingest_infos.size(); i++) {
97 if (cfd_->user_comparator()->Compare(
98 cf_ingest_infos[i].largest_internal_key.user_key(),
99 cf_ingest_infos[i + 1].smallest_internal_key.user_key()) >= 0) {
100 status = Status::InvalidArgument("CFs have overlapping ranges");
101 return status;
102 }
103 }
104
105 // Copy/Move external files into DB
106 auto hardlink_files = import_options_.move_files;
107
108 for (auto& files_to_import_per_cf : files_to_import_) {
109 for (auto& f : files_to_import_per_cf) {
110 const auto path_outside_db = f.external_file_path;
111 const auto path_inside_db = TableFileName(
112 cfd_->ioptions()->cf_paths, f.fd.GetNumber(), f.fd.GetPathId());
113
114 if (hardlink_files) {
115 status = fs_->LinkFile(path_outside_db, path_inside_db, IOOptions(),
116 nullptr);
117 if (status.IsNotSupported()) {
118 // Original file is on a different FS, use copy instead of hard
119 // linking
120 hardlink_files = false;
121 ROCKS_LOG_INFO(db_options_.info_log,
122 "Try to link file %s but it's not supported : %s",
123 f.internal_file_path.c_str(),
124 status.ToString().c_str());
125 }
126 }
127 if (!hardlink_files) {
128 // FIXME: temperature handling (like ExternalSstFileIngestionJob)
129 status = CopyFile(fs_.get(), path_outside_db, Temperature::kUnknown,
130 path_inside_db, Temperature::kUnknown, 0,
131 db_options_.use_fsync, io_tracer_);
132 }
133 if (!status.ok()) {
134 break;
135 }
136 f.copy_file = !hardlink_files;
137 f.internal_file_path = path_inside_db;
138 }
139 if (!status.ok()) {
140 break;
141 }
142 }
143
144 if (!status.ok()) {
145 // We failed, remove all files that we copied into the db
146 for (auto& files_to_import_per_cf : files_to_import_) {
147 for (auto& f : files_to_import_per_cf) {
148 if (f.internal_file_path.empty()) {
149 break;
150 }
151 const auto s =
152 fs_->DeleteFile(f.internal_file_path, IOOptions(), nullptr);
153 if (!s.ok()) {
154 ROCKS_LOG_WARN(db_options_.info_log,
155 "AddFile() clean up for file %s failed : %s",
156 f.internal_file_path.c_str(), s.ToString().c_str());
157 }
158 }
159 }
160 }
161
162 return status;
163}

167Status ImportColumnFamilyJob::Run() {
168 // We use the import time as the ancester time. This is the time the data
169 // is written to the database.
170 int64_t temp_current_time = 0;
171 uint64_t oldest_ancester_time = kUnknownOldestAncesterTime;
172 uint64_t current_time = kUnknownOldestAncesterTime;
173 if (clock_->GetCurrentTime(&temp_current_time).ok()) {
174 current_time = oldest_ancester_time =
175 static_cast<uint64_t>(temp_current_time);
176 }
177
178 Status s;
179 // When importing multiple CFs, we should not reuse epoch number from ingested
180 // files. Since these epoch numbers were assigned by different CFs, there may
181 // be different files from different CFs with the same epoch number. With a
182 // subsequent intra-L0 compaction we may end up with files with overlapping
183 // key range but the same epoch number. Here we will create a dummy
184 // VersionStorageInfo per CF being imported. Each CF's files will be assigned
185 // increasing epoch numbers to avoid duplicated epoch number. This is done by
186 // only resetting epoch number of the new CF in the first call to
187 // RecoverEpochNumbers() below.
188 for (size_t i = 0; s.ok() && i < files_to_import_.size(); ++i) {
189 VersionBuilder dummy_version_builder(
190 cfd_->current()->version_set()->file_options(), cfd_->ioptions(),
191 cfd_->table_cache(), cfd_->current()->storage_info(),
192 cfd_->current()->version_set(),
193 cfd_->GetFileMetadataCacheReservationManager());
194 VersionStorageInfo dummy_vstorage(
195 &cfd_->internal_comparator(), cfd_->user_comparator(),
196 cfd_->NumberLevels(), cfd_->ioptions()->compaction_style,
197 nullptr /* src_vstorage */, cfd_->ioptions()->force_consistency_checks,
198 EpochNumberRequirement::kMightMissing, cfd_->ioptions()->clock,
199 cfd_->GetLatestMutableCFOptions()->bottommost_file_compaction_delay,
200 cfd_->current()->version_set()->offpeak_time_option());
201 for (size_t j = 0; s.ok() && j < files_to_import_[i].size(); ++j) {
202 const auto& f = files_to_import_[i][j];
203 const auto& file_metadata = *metadatas_[i][j];
204
205 uint64_t tail_size = 0;
206 bool contain_no_data_blocks = f.table_properties.num_entries > 0 &&
207 (f.table_properties.num_entries ==
208 f.table_properties.num_range_deletions);
209 if (f.table_properties.tail_start_offset > 0 || contain_no_data_blocks) {
210 uint64_t file_size = f.fd.GetFileSize();
211 assert(f.table_properties.tail_start_offset <= file_size);
212 tail_size = file_size - f.table_properties.tail_start_offset;
213 }
214
215 VersionEdit dummy_version_edit;
216 dummy_version_edit.AddFile(
217 file_metadata.level, f.fd.GetNumber(), f.fd.GetPathId(),
218 f.fd.GetFileSize(), f.smallest_internal_key, f.largest_internal_key,
219 file_metadata.smallest_seqno, file_metadata.largest_seqno, false,
220 file_metadata.temperature, kInvalidBlobFileNumber,
221 oldest_ancester_time, current_time, file_metadata.epoch_number,
222 kUnknownFileChecksum, kUnknownFileChecksumFuncName, f.unique_id, 0,
223 tail_size,
224 static_cast<bool>(
225 f.table_properties.user_defined_timestamps_persisted));
226 s = dummy_version_builder.Apply(&dummy_version_edit);
227 }
228 if (s.ok()) {
229 s = dummy_version_builder.SaveTo(&dummy_vstorage);
230 }
231 if (s.ok()) {
232 // force resetting epoch number for each file
233 dummy_vstorage.RecoverEpochNumbers(cfd_, /*restart_epoch=*/i == 0,
234 /*force=*/true);
235 edit_.SetColumnFamily(cfd_->GetID());
236
237 for (int level = 0; level < dummy_vstorage.num_levels(); level++) {
238 for (FileMetaData* file_meta : dummy_vstorage.LevelFiles(level)) {
239 edit_.AddFile(level, *file_meta);
240 // If incoming sequence number is higher, update local sequence
241 // number.
242 if (file_meta->fd.largest_seqno > versions_->LastSequence()) {
243 versions_->SetLastAllocatedSequence(file_meta->fd.largest_seqno);
244 versions_->SetLastPublishedSequence(file_meta->fd.largest_seqno);
245 versions_->SetLastSequence(file_meta->fd.largest_seqno);
246 }
247 }
248 }
249 }
250 // Release resources occupied by the dummy VersionStorageInfo
251 for (int level = 0; level < dummy_vstorage.num_levels(); level++) {
252 for (FileMetaData* file_meta : dummy_vstorage.LevelFiles(level)) {
253 file_meta->refs--;
254 if (file_meta->refs <= 0) {
255 delete file_meta;
256 }
257 }
258 }
259 }
260
261 return s;
262}

264void ImportColumnFamilyJob::Cleanup(const Status& status) {
265 if (!status.ok()) {
266 // We failed to add files to the database remove all the files we copied.
267 for (auto& files_to_import_per_cf : files_to_import_) {
268 for (auto& f : files_to_import_per_cf) {
269 const auto s =
270 fs_->DeleteFile(f.internal_file_path, IOOptions(), nullptr);
271 if (!s.ok()) {
272 ROCKS_LOG_WARN(db_options_.info_log,
273 "AddFile() clean up for file %s failed : %s",
274 f.internal_file_path.c_str(), s.ToString().c_str());
275 }
276 }
277 }
278 } else if (status.ok() && import_options_.move_files) {
279 // The files were moved and added successfully, remove original file links
280 for (auto& files_to_import_per_cf : files_to_import_) {
281 for (auto& f : files_to_import_per_cf) {
282 const auto s =
283 fs_->DeleteFile(f.external_file_path, IOOptions(), nullptr);
284 if (!s.ok()) {
285 ROCKS_LOG_WARN(
286 db_options_.info_log,
287 "%s was added to DB successfully but failed to remove original "
288 "file link : %s",
289 f.external_file_path.c_str(), s.ToString().c_str());
290 }
291 }
292 }
293 }

296Status ImportColumnFamilyJob::GetIngestedFileInfo(
297 const std::string& external_file, uint64_t new_file_number,
298 SuperVersion* sv, const LiveFileMetaData& file_meta,
299 IngestedFileInfo* file_to_import) {
300 file_to_import->external_file_path = external_file;
301 Status status;
302 if (file_meta.size > 0) {
303 file_to_import->file_size = file_meta.size;
304 } else {
305 // Get external file size
306 status = fs_->GetFileSize(external_file, IOOptions(),
307 &file_to_import->file_size, nullptr);
308 if (!status.ok()) {
309 return status;
310 }
311 }
312 // Assign FD with number
313 file_to_import->fd =
314 FileDescriptor(new_file_number, 0, file_to_import->file_size);
315
316 // Create TableReader for external file
317 std::unique_ptr<TableReader> table_reader;
318 std::unique_ptr<FSRandomAccessFile> sst_file;
319 std::unique_ptr<RandomAccessFileReader> sst_file_reader;
320
321 status =
322 fs_->NewRandomAccessFile(external_file, env_options_, &sst_file, nullptr);
323 if (!status.ok()) {
324 return status;
325 }
326 sst_file_reader.reset(new RandomAccessFileReader(
327 std::move(sst_file), external_file, nullptr /*Env*/, io_tracer_));
328
329 // TODO(yuzhangyu): User-defined timestamps doesn't support importing column
330 // family. Pass in the correct `user_defined_timestamps_persisted` flag for
331 // creating `TableReaderOptions` when the support is there.
332 status = sv->mutable_cf_options.table_factory->NewTableReader(
333 TableReaderOptions(
334 *cfd_->ioptions(), sv->mutable_cf_options.prefix_extractor,
335 env_options_, cfd_->internal_comparator(),
336 sv->mutable_cf_options.block_protection_bytes_per_key,
337 /*skip_filters*/ false, /*immortal*/ false,
338 /*force_direct_prefetch*/ false, /*level*/ -1,
339 /*block_cache_tracer*/ nullptr,
340 /*max_file_size_for_l0_meta_pin*/ 0, versions_->DbSessionId(),
341 /*cur_file_num*/ new_file_number),
342 std::move(sst_file_reader), file_to_import->file_size, &table_reader);
343 if (!status.ok()) {
344 return status;
345 }
346
347 // Get the external file properties
348 auto props = table_reader->GetTableProperties();
349
350 // Set original_seqno to 0.
351 file_to_import->original_seqno = 0;
352
353 // Get number of entries in table
354 file_to_import->num_entries = props->num_entries;
355
356 // If the importing files were exported with Checkpoint::ExportColumnFamily(),
357 // we cannot simply recompute smallest and largest used to truncate range
358 // tombstones from file content, and we expect smallest and largest populated
359 // in file_meta.
360 if (file_meta.smallest.empty()) {
361 assert(file_meta.largest.empty());
362 // TODO: plumb Env::IOActivity, Env::IOPriority
363 ReadOptions ro;
364 std::unique_ptr<InternalIterator> iter(table_reader->NewIterator(
365 ro, sv->mutable_cf_options.prefix_extractor.get(), /*arena=*/nullptr,
366 /*skip_filters=*/false, TableReaderCaller::kExternalSSTIngestion));
367
368 // Get first (smallest) key from file
369 iter->SeekToFirst();
370 bool bound_set = false;
371 if (iter->Valid()) {
372 file_to_import->smallest_internal_key.DecodeFrom(iter->key());
373 Slice largest;
374 if (strcmp(sv->mutable_cf_options.table_factory->Name(), "PlainTable") ==
375 0) {
376 // PlainTable iterator does not support SeekToLast().
377 largest = iter->key();
378 for (; iter->Valid(); iter->Next()) {
379 if (cfd_->internal_comparator().Compare(iter->key(), largest) > 0) {
380 largest = iter->key();
381 }
382 }
383 if (!iter->status().ok()) {
384 return iter->status();
385 }
386 } else {
387 iter->SeekToLast();
388 if (!iter->Valid()) {
389 if (iter->status().ok()) {
390 // The file contains at least 1 key since iter is valid after
391 // SeekToFirst().
392 return Status::Corruption("Can not find largest key in sst file");
393 } else {
394 return iter->status();
395 }
396 }
397 largest = iter->key();
398 }
399 file_to_import->largest_internal_key.DecodeFrom(largest);
400 bound_set = true;
401 } else if (!iter->status().ok()) {
402 return iter->status();
403 }
404
405 std::unique_ptr<InternalIterator> range_del_iter{
406 table_reader->NewRangeTombstoneIterator(ro)};
407 if (range_del_iter != nullptr) {
408 range_del_iter->SeekToFirst();
409 if (range_del_iter->Valid()) {
410 ParsedInternalKey key;
411 Status pik_status = ParseInternalKey(range_del_iter->key(), &key,
412 db_options_.allow_data_in_errors);
413 if (!pik_status.ok()) {
414 return Status::Corruption("Corrupted key in external file. ",
415 pik_status.getState());
416 }
417 RangeTombstone first_tombstone(key, range_del_iter->value());
418 InternalKey start_key = first_tombstone.SerializeKey();
419 const InternalKeyComparator* icmp = &cfd_->internal_comparator();
420 if (!bound_set ||
421 icmp->Compare(start_key, file_to_import->smallest_internal_key) <
422 0) {
423 file_to_import->smallest_internal_key = start_key;
424 }
425
426 range_del_iter->SeekToLast();
427 pik_status = ParseInternalKey(range_del_iter->key(), &key,
428 db_options_.allow_data_in_errors);
429 if (!pik_status.ok()) {
430 return Status::Corruption("Corrupted key in external file. ",
431 pik_status.getState());
432 }
433 RangeTombstone last_tombstone(key, range_del_iter->value());
434 InternalKey end_key = last_tombstone.SerializeEndKey();
435 if (!bound_set ||
436 icmp->Compare(end_key, file_to_import->largest_internal_key) > 0) {
437 file_to_import->largest_internal_key = end_key;
438 }
439 bound_set = true;
440 }
441 }
442 assert(bound_set);
443 } else {
444 assert(!file_meta.largest.empty());
445 file_to_import->smallest_internal_key.DecodeFrom(file_meta.smallest);
446 file_to_import->largest_internal_key.DecodeFrom(file_meta.largest);
447 }
448
449 file_to_import->cf_id = static_cast<uint32_t>(props->column_family_id);
450
451 file_to_import->table_properties = *props;
452
453 auto s = GetSstInternalUniqueId(props->db_id, props->db_session_id,
454 props->orig_file_number,
455 &(file_to_import->unique_id));
456 if (!s.ok()) {
457 ROCKS_LOG_WARN(db_options_.info_log,
458 "Failed to get SST unique id for file %s",
459 file_to_import->internal_file_path.c_str());
460 }
461
462 return status;
463}
