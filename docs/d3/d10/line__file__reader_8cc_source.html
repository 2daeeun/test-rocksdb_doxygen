30bool LineFileReader::ReadLine(std::string* out,
31 Env::IOPriority rate_limiter_priority) {
32 assert(out);
33 if (!io_status_.ok()) {
34 // Status should be checked (or permit unchecked) any time we return false.
35 io_status_.MustCheck();
36 return false;
37 }
38 out->clear();
39 for (;;) {
40 // Look for line delimiter
41 const char* found = static_cast<const char*>(
42 std::memchr(buf_begin_, '\n', buf_end_ - buf_begin_));
43 if (found) {
44 size_t len = found - buf_begin_;
45 out->append(buf_begin_, len);
46 buf_begin_ += len + /*delim*/ 1;
47 ++line_number_;
48 return true;
49 }
50 if (at_eof_) {
51 io_status_.MustCheck();
52 return false;
53 }
54 // else flush and reload buffer
55 out->append(buf_begin_, buf_end_ - buf_begin_);
56 Slice result;
57 io_status_ =
58 sfr_.Read(buf_.size(), &result, buf_.data(), rate_limiter_priority);
59 IOSTATS_ADD(bytes_read, result.size());
60 if (!io_status_.ok()) {
61 io_status_.MustCheck();
62 return false;
63 }
64 if (result.size() != buf_.size()) {
65 // The obscure way of indicating EOF
66 at_eof_ = true;
67 }
68 buf_begin_ = result.data();
69 buf_end_ = result.data() + result.size();
70 }
71}
