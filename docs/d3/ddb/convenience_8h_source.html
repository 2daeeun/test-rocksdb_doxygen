26// ConfigOptions containing the parameters/controls for
27// comparing objects and converting to/from strings.
28// These settings control how the methods
29// treat errors (e.g. ignore_unknown_objects), the format
30// of the serialization (e.g. delimiter), and how to compare
31// options (sanity_level).
32// NOTE: members of this struct make it potentially problematic for
33// static storage duration ("static initialization order fiasco")
34struct ConfigOptions {
35 // Constructs a new ConfigOptions with a new object registry.
36 // This method should only be used when a DBOptions is not available,
37 // else registry settings may be lost
38 ConfigOptions();
39
40 // Constructs a new ConfigOptions using the settings from
41 // the input DBOptions. Currently constructs a new object registry.
42 explicit ConfigOptions(const DBOptions&);
43
44 // This enum defines the RocksDB options sanity level.
45 enum SanityLevel : unsigned char {
46 kSanityLevelNone = 0x01, // Performs no sanity check at all.
47 // Performs minimum check to ensure the RocksDB instance can be
48 // opened without corrupting / mis-interpreting the data.
49 kSanityLevelLooselyCompatible = 0x02,
50 // Perform exact match sanity check.
51 kSanityLevelExactMatch = 0xFF,
52 };
53
54 enum Depth {
55 kDepthDefault, // Traverse nested options that are not flagged as "shallow"
56 kDepthShallow, // Do not traverse into any nested options
57 kDepthDetailed, // Traverse nested options, overriding the options shallow
58 // setting
59 };
60
61 // When true, any unused options will be ignored and OK will be returned.
62 // For options files that appear to be from the current version or earlier,
63 // unknown options are considered corruption regardless of this setting.
64 bool ignore_unknown_options = false;
65
66 // When true, any unsupported options will be ignored and OK will be returned
67 bool ignore_unsupported_options = true;
68
69 // If the strings are escaped (old-style?)
70 bool input_strings_escaped = true;
71
72 // Whether or not to invoke PrepareOptions after configure is called.
73 bool invoke_prepare_options = true;
74
75 // Options can be marked as Mutable (OptionTypeInfo::IsMutable()) or not.
76 // When "mutable_options_only=false", all options are evaluated.
77 // When "mutable_options_only="true", any option not marked as Mutable is
78 // either ignored (in the case of string/equals methods) or results in an
79 // error (in the case of Configure).
80 bool mutable_options_only = false;
81
82 // The separator between options when converting to a string
83 std::string delimiter = ";";
84
85 // Controls how to traverse options during print/match stages
86 Depth depth = Depth::kDepthDefault;
87
88 // Controls how options are serialized
89 // Controls how pedantic the comparison must be for equivalency
90 SanityLevel sanity_level = SanityLevel::kSanityLevelExactMatch;
91 // `file_readahead_size` is used for readahead for the option file.
92 size_t file_readahead_size = 512 * 1024;
93
94 // The environment to use for this option
95 Env* env = Env::Default();
96
97 // The object registry to use for this options
98 std::shared_ptr<ObjectRegistry> registry;
99
100 bool IsShallow() const { return depth == Depth::kDepthShallow; }
101 bool IsDetailed() const { return depth == Depth::kDepthDetailed; }
102
103 bool IsCheckDisabled() const {
104 return sanity_level == SanityLevel::kSanityLevelNone;
105 }
106
107 bool IsCheckEnabled(SanityLevel level) const {
108 return (level > SanityLevel::kSanityLevelNone && level <= sanity_level);
109 }
110};

113// The following set of functions provide a way to construct RocksDB Options
114// from a string or a string-to-string map. Here is the general rule of
115// setting option values from strings by type. Some RocksDB types are also
116// supported in these APIs. Please refer to the comment of the function itself
117// to find more information about how to config those RocksDB types.
118//
119// * Strings:
120// Strings will be used as values directly without any truncating or
121// trimming.
122//
123// * Booleans:
124// - "true" or "1" => true
125// - "false" or "0" => false.
126// [Example]:
127// - {"optimize_filters_for_hits", "1"} in GetColumnFamilyOptionsFromMap, or
128// - "optimize_filters_for_hits=true" in GetColumnFamilyOptionsFromString.
129//
130// * Integers:
131// Integers are converted directly from string, in addition to the following
132// units that we support:
133// - 'k' or 'K' => 2^10
134// - 'm' or 'M' => 2^20
135// - 'g' or 'G' => 2^30
136// - 't' or 'T' => 2^40 // only for unsigned int with sufficient bits.
137// [Example]:
138// - {"arena_block_size", "19G"} in GetColumnFamilyOptionsFromMap, or
139// - "arena_block_size=19G" in GetColumnFamilyOptionsFromString.
140//
141// * Doubles / Floating Points:
142// Doubles / Floating Points are converted directly from string. Note that
143// currently we do not support units.
144// [Example]:
145// - {"memtable_prefix_bloom_size_ratio", "0.1"} in
146// GetColumnFamilyOptionsFromMap, or
147// - "memtable_prefix_bloom_size_ratio=0.1" in
148// GetColumnFamilyOptionsFromString.
149// * Array / Vectors:
150// An array is specified by a list of values, where ':' is used as
151// the delimiter to separate each value.
152// [Example]:
153// - {"compression_per_level", "kNoCompression:kSnappyCompression"}
154// in GetColumnFamilyOptionsFromMap, or
155// - "compression_per_level=kNoCompression:kSnappyCompression" in
156// GetColumnFamilyOptionsFromMapString
157// * Enums:
158// The valid values of each enum are identical to the names of its constants.
159// [Example]:
160// - CompressionType: valid values are "kNoCompression",
161// "kSnappyCompression", "kZlibCompression", "kBZip2Compression", ...
162// - CompactionStyle: valid values are "kCompactionStyleLevel",
163// "kCompactionStyleUniversal", "kCompactionStyleFIFO", and
164// "kCompactionStyleNone".
165//
166
167// Take a ConfigOptions `config_options` and a ColumnFamilyOptions
168// "base_options" as the default option in addition to a map "opts_map" of
169// option name to option value to construct the new ColumnFamilyOptions
170// "new_options".
171//
172// Below are the instructions of how to config some non-primitive-typed
173// options in ColumnFamilyOptions:
174//
175// * table_factory:
176// table_factory can be configured using our custom nested-option syntax.
177//
178// {option_a=value_a; option_b=value_b; option_c=value_c; ... }
179//
180// A nested option is enclosed by two curly braces, within which there are
181// multiple option assignments. Each assignment is of the form
182// "variable_name=value;".
183//
184// Currently we support the following types of TableFactory:
185// - BlockBasedTableFactory:
186// Use name "block_based_table_factory" to initialize table_factory with
187// BlockBasedTableFactory. Its BlockBasedTableFactoryOptions can be
188// configured using the nested-option syntax.
189// [Example]:
190// * {"block_based_table_factory", "{block_cache=1M;block_size=4k;}"}
191// is equivalent to assigning table_factory with a BlockBasedTableFactory
192// that has 1M LRU block-cache with block size equals to 4k:
193// ColumnFamilyOptions cf_opt;
194// BlockBasedTableOptions blk_opt;
195// blk_opt.block_cache = NewLRUCache(1 * 1024 * 1024);
196// blk_opt.block_size = 4 * 1024;
197// cf_opt.table_factory.reset(NewBlockBasedTableFactory(blk_opt));
198// - PlainTableFactory:
199// Use name "plain_table_factory" to initialize table_factory with
200// PlainTableFactory. Its PlainTableFactoryOptions can be configured using
201// the nested-option syntax.
202// [Example]:
203// * {"plain_table_factory", "{user_key_len=66;bloom_bits_per_key=20;}"}
204//
205// * memtable_factory:
206// Use "memtable" to config memtable_factory. Here are the supported
207// memtable factories:
208// - SkipList:
209// Pass "skip_list:<lookahead>" to config memtable to use SkipList,
210// or simply "skip_list" to use the default SkipList.
211// [Example]:
212// * {"memtable", "skip_list:5"} is equivalent to setting
213// memtable to SkipListFactory(5).
214// - PrefixHash:
215// Pass "prefix_hash:<hash_bucket_count>" to config memtable
216// to use PrefixHash, or simply "prefix_hash" to use the default
217// PrefixHash.
218// [Example]:
219// * {"memtable", "prefix_hash:1000"} is equivalent to setting
220// memtable to NewHashSkipListRepFactory(hash_bucket_count).
221// - HashLinkedList:
222// Pass "hash_linkedlist:<hash_bucket_count>" to config memtable
223// to use HashLinkedList, or simply "hash_linkedlist" to use the default
224// HashLinkedList.
225// [Example]:
226// * {"memtable", "hash_linkedlist:1000"} is equivalent to
227// setting memtable to NewHashLinkListRepFactory(1000).
228// - VectorRepFactory:
229// Pass "vector:<count>" to config memtable to use VectorRepFactory,
230// or simply "vector" to use the default Vector memtable.
231// [Example]:
232// * {"memtable", "vector:1024"} is equivalent to setting memtable
233// to VectorRepFactory(1024).
234//
235// * compression_opts:
236// Use "compression_opts" to config compression_opts. The value format
237// is of the form "<window_bits>:<level>:<strategy>:<max_dict_bytes>".
238// [Example]:
239// * {"compression_opts", "4:5:6:7"} is equivalent to setting:
240// ColumnFamilyOptions cf_opt;
241// cf_opt.compression_opts.window_bits = 4;
242// cf_opt.compression_opts.level = 5;
243// cf_opt.compression_opts.strategy = 6;
244// cf_opt.compression_opts.max_dict_bytes = 7;
245//
246// @param config_options controls how the map is processed.
247// @param base_options the default options of the output "new_options".
248// @param opts_map an option name to value map for specifying how "new_options"
249// should be set.
250// @param new_options the resulting options based on "base_options" with the
251// change specified in "opts_map".
252// @param input_strings_escaped when set to true, each escaped characters
253// prefixed by '\' in the values of the opts_map will be further converted
254// back to the raw string before assigning to the associated options.
255// @param ignore_unknown_options when set to true, unknown options are ignored
256// instead of resulting in an unknown-option error.
257// @return Status::OK() on success. Otherwise, a non-ok status indicating
258// error will be returned, and "new_options" will be set to "base_options".
259// @return Status::NotFound means the one (or more) of the option name in
260// the opts_map is not valid for this option
261// @return Status::NotSupported means we do not know how to parse one of the
262// value for this option
263// @return Status::InvalidArgument means the one of the option values is not
264// valid for this option.
265Status GetColumnFamilyOptionsFromMap(
266 const ConfigOptions& config_options,
267 const ColumnFamilyOptions& base_options,
268 const std::unordered_map<std::string, std::string>& opts_map,
269 ColumnFamilyOptions* new_options);
