18class Slice;

20// The general interface for comparing two Slices are defined for both of
21// Comparator and some internal data structures.
22class CompareInterface {
23 public:
24 virtual ~CompareInterface() {}
25
26 // Three-way comparison. Returns value:
27 // < 0 iff "a" < "b",
28 // == 0 iff "a" == "b",
29 // > 0 iff "a" > "b"
30 // Note that Compare(a, b) also compares timestamp if timestamp size is
31 // non-zero. For the same user key with different timestamps, larger (newer)
32 // timestamp comes first.
33 virtual int Compare(const Slice& a, const Slice& b) const = 0;
34};
35
36// A Comparator object provides a total order across slices that are
37// used as keys in an sstable or a database. A Comparator implementation
38// must be thread-safe since rocksdb may invoke its methods concurrently
39// from multiple threads.
40//
41// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
42// because RocksDB is not exception-safe. This could cause undefined behavior
43// including data loss, unreported corruption, deadlocks, and more.
44class Comparator : public Customizable, public CompareInterface {
45 public:
46 Comparator() : timestamp_size_(0) {}
47
48 Comparator(size_t ts_sz) : timestamp_size_(ts_sz) {}
49
50 Comparator(const Comparator& orig) : timestamp_size_(orig.timestamp_size_) {}
51
52 Comparator& operator=(const Comparator& rhs) {
53 if (this != &rhs) {
54 timestamp_size_ = rhs.timestamp_size_;
55 }
56 return *this;
57 }
58
59 ~Comparator() override {}
60
61 static Status CreateFromString(const ConfigOptions& opts,
62 const std::string& id,
63 const Comparator** comp);
64 static const char* Type() { return "Comparator"; }
65
66 // The name of the comparator. Used to check for comparator
67 // mismatches (i.e., a DB created with one comparator is
68 // accessed using a different comparator.
69 //
70 // The client of this package should switch to a new name whenever
71 // the comparator implementation changes in a way that will cause
72 // the relative ordering of any two keys to change.
73 //
74 // Names starting with "rocksdb." are reserved and should not be used
75 // by any clients of this package.
76 const char* Name() const override = 0;
77
78 // Compares two slices for equality. The following invariant should always
79 // hold (and is the default implementation):
80 // Equal(a, b) iff Compare(a, b) == 0
81 // Overwrite only if equality comparisons can be done more efficiently than
82 // three-way comparisons.
83 virtual bool Equal(const Slice& a, const Slice& b) const {
84 return Compare(a, b) == 0;
85 }
86
87 // Advanced functions: these are used to reduce the space requirements
88 // for internal data structures like index blocks.
89
90 // If *start < limit, changes *start to a short string in [start,limit).
91 // Simple comparator implementations may return with *start unchanged,
92 // i.e., an implementation of this method that does nothing is correct.
93 virtual void FindShortestSeparator(std::string* start,
94 const Slice& limit) const = 0;
95
96 // Changes *key to a short string >= *key.
97 // Simple comparator implementations may return with *key unchanged,
98 // i.e., an implementation of this method that does nothing is correct.
99 virtual void FindShortSuccessor(std::string* key) const = 0;
100
101 // given two keys, determine if t is the successor of s
102 // BUG: only return true if no other keys starting with `t` are ordered
103 // before `t`. Otherwise, the auto_prefix_mode can omit entries within
104 // iterator bounds that have same prefix as upper bound but different
105 // prefix from seek key.
106 virtual bool IsSameLengthImmediateSuccessor(const Slice& /*s*/,
107 const Slice& /*t*/) const {
108 return false;
109 }
110
111 // return true if two keys with different byte sequences can be regarded
112 // as equal by this comparator.
113 // The major use case is to determine if DataBlockHashIndex is compatible
114 // with the customized comparator.
115 virtual bool CanKeysWithDifferentByteContentsBeEqual() const { return true; }
116
117 // if it is a wrapped comparator, may return the root one.
118 // return itself it is not wrapped.
119 virtual const Comparator* GetRootComparator() const { return this; }
120
121 inline size_t timestamp_size() const { return timestamp_size_; }
122
123 // Return what this Comparator considers as the maximum timestamp.
124 // The default implementation only works for when `timestamp_size_` is 0,
125 // subclasses for which this is not the case needs to override this function.
126 virtual Slice GetMaxTimestamp() const {
127 if (timestamp_size_ == 0) {
128 return Slice();
129 }
130 assert(false);
131 return Slice();
132 }
133
134 // Return what this Comparator considers as the min timestamp.
135 // The default implementation only works for when `timestamp_size_` is 0,
136 // subclasses for which this is not the case needs to override this function.
137 virtual Slice GetMinTimestamp() const {
138 if (timestamp_size_ == 0) {
139 return Slice();
140 }
141 assert(false);
142 return Slice();
143 }
144
145 // Return a human readable user-defined timestamp for debugging.
146 virtual std::string TimestampToString(const Slice& /*timestamp*/) const {
147 return "";
148 }
149
150 int CompareWithoutTimestamp(const Slice& a, const Slice& b) const {
151 return CompareWithoutTimestamp(a, /*a_has_ts=*/true, b, /*b_has_ts=*/true);
152 }
153
154 // For two events e1 and e2 whose timestamps are t1 and t2 respectively,
155 // Returns value:
156 // < 0 iff t1 < t2
157 // == 0 iff t1 == t2
158 // > 0 iff t1 > t2
159 // Note that an all-zero byte array will be the smallest (oldest) timestamp
160 // of the same length, and a byte array with all bits 1 will be the largest.
161 // In the future, we can extend Comparator so that subclasses can specify
162 // both largest and smallest timestamps.
163 virtual int CompareTimestamp(const Slice& /*ts1*/,
164 const Slice& /*ts2*/) const {
165 return 0;
166 }
167
168 virtual int CompareWithoutTimestamp(const Slice& a, bool /*a_has_ts*/,
169 const Slice& b, bool /*b_has_ts*/) const {
170 return Compare(a, b);
171 }
172
173 virtual bool EqualWithoutTimestamp(const Slice& a, const Slice& b) const {
174 return 0 ==
175 CompareWithoutTimestamp(a, /*a_has_ts=*/true, b, /*b_has_ts=*/true);
176 }
177
178 private:
179 size_t timestamp_size_;
180};

182// Return a builtin comparator that uses lexicographic ordering
183// on unsigned bytes, so the empty string is ordered before everything
184// else and a sufficiently long string of \xFF orders after anything.
185// CanKeysWithDifferentByteContentsBeEqual() == false
186// Returns an immortal pointer that must not be deleted by the caller.
187const Comparator* BytewiseComparator();
188
189// Return a builtin comparator that is the reverse ordering of
190// BytewiseComparator(), so the empty string is ordered after everything
191// else and a sufficiently long string of \xFF orders before anything.
192// CanKeysWithDifferentByteContentsBeEqual() == false
193// Returns an immortal pointer that must not be deleted by the caller.
194const Comparator* ReverseBytewiseComparator();
195
196// Returns a builtin comparator that enables user-defined timestamps (formatted
197// as uint64_t) while ordering the user key part without UDT with a
198// BytewiseComparator.
199// For the same user key with different timestamps, larger (newer) timestamp
200// comes first.
201const Comparator* BytewiseComparatorWithU64Ts();
202
203// Returns a builtin comparator that enables user-defined timestamps (formatted
204// as uint64_t) while ordering the user key part without UDT with a
205// ReverseBytewiseComparator.
206// For the same user key with different timestamps, larger (newer) timestamp
207// comes first.
208const Comparator* ReverseBytewiseComparatorWithU64Ts();

210// Decode a `U64Ts` timestamp returned by RocksDB to uint64_t.
211// When a column family enables user-defined timestamp feature
212// with `BytewiseComparatorWithU64Ts` or `ReverseBytewiseComparatorWithU64Ts`
213// comparator, the `Iterator::timestamp()` API returns timestamp in `Slice`
214// format. This util function helps to translate that `Slice` into an uint64_t
215// type.
216Status DecodeU64Ts(const Slice& ts, uint64_t* int_ts);
217
218// Encode an uint64_t timestamp into a U64Ts `Slice`, to be used as
219// `ReadOptions.timestamp` for a column family that enables user-defined
220// timestamp feature with `BytewiseComparatorWithU64Ts` or
221// `ReverseBytewiseComparatorWithU64Ts` comparator.
222// Be mindful that the returned `Slice` is backed by `ts_buf`. When `ts_buf`
223// is deconstructed, the returned `Slice` can no longer be used.
224Slice EncodeU64Ts(uint64_t ts, std::string* ts_buf);
225
226// Returns a `Slice` representing the maximum U64Ts timestamp.
227// The returned `Slice` is backed by some static storage, so it's valid until
228// program destruction.
229Slice MaxU64Ts();
230

232// The returned `Slice` is backed by some static storage, so it's valid until
233// program destruction.
