29Status IOTraceWriter::WriteIOOp(const IOTraceRecord& record,
30 IODebugContext* dbg) {
31 uint64_t trace_file_size = trace_writer_->GetFileSize();
32 if (trace_file_size > trace_options_.max_trace_file_size) {
33 return Status::OK();
34 }
35 Trace trace;
36 trace.ts = record.access_timestamp;
37 trace.type = record.trace_type;
38 PutFixed64(&trace.payload, record.io_op_data);
39 Slice file_operation(record.file_operation);
40 PutLengthPrefixedSlice(&trace.payload, file_operation);
41 PutFixed64(&trace.payload, record.latency);
42 Slice io_status(record.io_status);
43 PutLengthPrefixedSlice(&trace.payload, io_status);
44 Slice file_name(record.file_name);
45 PutLengthPrefixedSlice(&trace.payload, file_name);
46
47 // Each bit in io_op_data stores which corresponding info from IOTraceOp will
48 // be added in the trace. Foreg, if bit at position 1 is set then
49 // IOTraceOp::kIOLen (length) will be logged in the record (Since
50 // IOTraceOp::kIOLen = 1 in the enum). So find all the set positions in
51 // io_op_data one by one and, update corresponsing info in the trace record,
52 // unset that bit to find other set bits until io_op_data = 0.
53 /* Write remaining options based on io_op_data set by file operation */
54 int64_t io_op_data = static_cast<int64_t>(record.io_op_data);
55 while (io_op_data) {
56 // Find the rightmost set bit.
57 uint32_t set_pos = static_cast<uint32_t>(log2(io_op_data & -io_op_data));
58 switch (set_pos) {
59 case IOTraceOp::kIOFileSize:
60 PutFixed64(&trace.payload, record.file_size);
61 break;
62 case IOTraceOp::kIOLen:
63 PutFixed64(&trace.payload, record.len);
64 break;
65 case IOTraceOp::kIOOffset:
66 PutFixed64(&trace.payload, record.offset);
67 break;
68 default:
69 assert(false);
70 }
71 // unset the rightmost bit.
72 io_op_data &= (io_op_data - 1);
73 }
74
75 int64_t trace_data = 0;
76 if (dbg) {
77 trace_data = static_cast<int64_t>(dbg->trace_data);
78 }
79 PutFixed64(&trace.payload, trace_data);
80 while (trace_data) {
81 // Find the rightmost set bit.
82 uint32_t set_pos = static_cast<uint32_t>(log2(trace_data & -trace_data));
83 switch (set_pos) {
84 case IODebugContext::TraceData::kRequestID: {
85 Slice request_id(dbg->request_id);
86 PutLengthPrefixedSlice(&trace.payload, request_id);
87 } break;
88 default:
89 assert(false);
90 }
91 // unset the rightmost bit.
92 trace_data &= (trace_data - 1);
93 }
94
95 std::string encoded_trace;
96 TracerHelper::EncodeTrace(trace, &encoded_trace);
97 return trace_writer_->Write(encoded_trace);
98}

100Status IOTraceWriter::WriteHeader() {
101 Trace trace;
102 trace.ts = clock_->NowMicros();
103 trace.type = TraceType::kTraceBegin;
104 PutLengthPrefixedSlice(&trace.payload, kTraceMagic);
105 PutFixed32(&trace.payload, kMajorVersion);
106 PutFixed32(&trace.payload, kMinorVersion);
107 std::string encoded_trace;
108 TracerHelper::EncodeTrace(trace, &encoded_trace);
109 return trace_writer_->Write(encoded_trace);
110}

115Status IOTraceReader::ReadHeader(IOTraceHeader* header) {
116 assert(header != nullptr);
117 std::string encoded_trace;
118 Status s = trace_reader_->Read(&encoded_trace);
119 if (!s.ok()) {
120 return s;
121 }
122 Trace trace;
123 s = TracerHelper::DecodeTrace(encoded_trace, &trace);
124 if (!s.ok()) {
125 return s;
126 }
127 header->start_time = trace.ts;
128 Slice enc_slice = Slice(trace.payload);
129 Slice magic_number;
130 if (!GetLengthPrefixedSlice(&enc_slice, &magic_number)) {
131 return Status::Corruption(
132 "Corrupted header in the trace file: Failed to read the magic number.");
133 }
134 if (magic_number.ToString() != kTraceMagic) {
135 return Status::Corruption(
136 "Corrupted header in the trace file: Magic number does not match.");
137 }
138 if (!GetFixed32(&enc_slice, &header->rocksdb_major_version)) {
139 return Status::Corruption(
140 "Corrupted header in the trace file: Failed to read rocksdb major "
141 "version number.");
142 }
143 if (!GetFixed32(&enc_slice, &header->rocksdb_minor_version)) {
144 return Status::Corruption(
145 "Corrupted header in the trace file: Failed to read rocksdb minor "
146 "version number.");
147 }
148 // We should have retrieved all information in the header.
149 if (!enc_slice.empty()) {
150 return Status::Corruption(
151 "Corrupted header in the trace file: The length of header is too "
152 "long.");
153 }
154 return Status::OK();
155}

157Status IOTraceReader::ReadIOOp(IOTraceRecord* record) {
158 assert(record);
159 std::string encoded_trace;
160 Status s = trace_reader_->Read(&encoded_trace);
161 if (!s.ok()) {
162 return s;
163 }
164 Trace trace;
165 s = TracerHelper::DecodeTrace(encoded_trace, &trace);
166 if (!s.ok()) {
167 return s;
168 }
169 record->access_timestamp = trace.ts;
170 record->trace_type = trace.type;
171 Slice enc_slice = Slice(trace.payload);
172
173 if (!GetFixed64(&enc_slice, &record->io_op_data)) {
174 return Status::Incomplete(
175 "Incomplete access record: Failed to read trace data.");
176 }
177 Slice file_operation;
178 if (!GetLengthPrefixedSlice(&enc_slice, &file_operation)) {
179 return Status::Incomplete(
180 "Incomplete access record: Failed to read file operation.");
181 }
182 record->file_operation = file_operation.ToString();
183 if (!GetFixed64(&enc_slice, &record->latency)) {
184 return Status::Incomplete(
185 "Incomplete access record: Failed to read latency.");
186 }
187 Slice io_status;
188 if (!GetLengthPrefixedSlice(&enc_slice, &io_status)) {
189 return Status::Incomplete(
190 "Incomplete access record: Failed to read IO status.");
191 }
192 record->io_status = io_status.ToString();
193 Slice file_name;
194 if (!GetLengthPrefixedSlice(&enc_slice, &file_name)) {
195 return Status::Incomplete(
196 "Incomplete access record: Failed to read file name.");
197 }
198 record->file_name = file_name.ToString();
199
200 // Each bit in io_op_data stores which corresponding info from IOTraceOp will
201 // be added in the trace. Foreg, if bit at position 1 is set then
202 // IOTraceOp::kIOLen (length) will be logged in the record (Since
203 // IOTraceOp::kIOLen = 1 in the enum). So find all the set positions in
204 // io_op_data one by one and, update corresponsing info in the trace record,
205 // unset that bit to find other set bits until io_op_data = 0.
206 /* Read remaining options based on io_op_data set by file operation */
207 // Assuming 63 bits will be used at max.
208 int64_t io_op_data = static_cast<int64_t>(record->io_op_data);
209 while (io_op_data) {
210 // Find the rightmost set bit.
211 uint32_t set_pos = static_cast<uint32_t>(log2(io_op_data & -io_op_data));
212 switch (set_pos) {
213 case IOTraceOp::kIOFileSize:
214 if (!GetFixed64(&enc_slice, &record->file_size)) {
215 return Status::Incomplete(
216 "Incomplete access record: Failed to read file size.");
217 }
218 break;
219 case IOTraceOp::kIOLen:
220 if (!GetFixed64(&enc_slice, &record->len)) {
221 return Status::Incomplete(
222 "Incomplete access record: Failed to read length.");
223 }
224 break;
225 case IOTraceOp::kIOOffset:
226 if (!GetFixed64(&enc_slice, &record->offset)) {
227 return Status::Incomplete(
228 "Incomplete access record: Failed to read offset.");
229 }
230 break;
231 default:
232 assert(false);
233 }
234 // unset the rightmost bit.
235 io_op_data &= (io_op_data - 1);
236 }
237
238 if (!GetFixed64(&enc_slice, &record->trace_data)) {
239 return Status::Incomplete(
240 "Incomplete access record: Failed to read trace op.");
241 }
242 int64_t trace_data = static_cast<int64_t>(record->trace_data);
243 while (trace_data) {
244 // Find the rightmost set bit.
245 uint32_t set_pos = static_cast<uint32_t>(log2(trace_data & -trace_data));
246 switch (set_pos) {
247 case IODebugContext::TraceData::kRequestID: {
248 Slice request_id;
249 if (!GetLengthPrefixedSlice(&enc_slice, &request_id)) {
250 return Status::Incomplete(
251 "Incomplete access record: Failed to request id.");
252 }
253 record->request_id = request_id.ToString();
254 } break;
255 default:
256 assert(false);
257 }
258 // unset the rightmost bit.
259 trace_data &= (trace_data - 1);
260 }
261
262 return Status::OK();
263}

269Status IOTracer::StartIOTrace(SystemClock* clock,
270 const TraceOptions& trace_options,
271 std::unique_ptr<TraceWriter>&& trace_writer) {
272 InstrumentedMutexLock lock_guard(&trace_writer_mutex_);
273 if (writer_.load()) {
274 return Status::Busy();
275 }
276 trace_options_ = trace_options;
277 writer_.store(
278 new IOTraceWriter(clock, trace_options, std::move(trace_writer)));
279 tracing_enabled = true;
280 return writer_.load()->WriteHeader();
281}
282
283void IOTracer::EndIOTrace() {
284 InstrumentedMutexLock lock_guard(&trace_writer_mutex_);
285 if (!writer_.load()) {
286 return;
287 }
288 delete writer_.load();
289 writer_.store(nullptr);
290 tracing_enabled = false;
291}

293void IOTracer::WriteIOOp(const IOTraceRecord& record, IODebugContext* dbg) {
294 if (!writer_.load()) {
295 return;
296 }
297 InstrumentedMutexLock lock_guard(&trace_writer_mutex_);
298 if (!writer_.load()) {
299 return;
300 }
301 writer_.load()->WriteIOOp(record, dbg).PermitUncheckedError();
302}
