66 // Returns the number of micro-seconds since some fixed point in time.

148 IOStatus MultiRead(FSReadRequest* fs_reqs, size_t num_reqs,
149 const IOOptions& /*options*/,
150 IODebugContext* /*dbg*/) override {
151 std::vector<ReadRequest> reqs;
152 Status status;
153
154 reqs.reserve(num_reqs);
155 for (size_t i = 0; i < num_reqs; ++i) {
156 ReadRequest req;
157
158 req.offset = fs_reqs[i].offset;
159 req.len = fs_reqs[i].len;
160 req.scratch = fs_reqs[i].scratch;
161 req.status = Status::OK();
162 reqs.emplace_back(std::move(req));
163 }
164 status = target_->MultiRead(reqs.data(), num_reqs);
165 for (size_t i = 0; i < num_reqs; ++i) {
166 fs_reqs[i].result = reqs[i].result;
167 fs_reqs[i].status = status_to_io_status(std::move(reqs[i].status));
168 }
169 return status_to_io_status(std::move(status));
170 }
171
172 IOStatus Prefetch(uint64_t offset, size_t n, const IOOptions& /*options*/,

786std::string Env::GenerateUniqueId() {
787 std::string result;
788 bool success = port::GenerateRfcUuid(&result);
789 if (!success) {
790 // Fall back on our own way of generating a unique ID and adapt it to
791 // RFC 4122 variant 1 version 4 (a random ID).
792 // https://en.wikipedia.org/wiki/Universally_unique_identifier
793 // We already tried GenerateRfcUuid so no need to try it again in
794 // GenerateRawUniqueId
795 constexpr bool exclude_port_uuid = true;
796 uint64_t upper, lower;
797 GenerateRawUniqueId(&upper, &lower, exclude_port_uuid);
798
799 // Set 4-bit version to 4
800 upper = (upper & (~uint64_t{0xf000})) | 0x4000;
801 // Set unary-encoded variant to 1 (0b10)
802 lower = (lower & (~(uint64_t{3} << 62))) | (uint64_t{2} << 62);
803
804 // Use 36 character format of RFC 4122
805 result.resize(36U);
806 char* buf = result.data();
807 PutBaseChars<16>(&buf, 8, upper >> 32, false);
808 *(buf++) = '-';
809 PutBaseChars<16>(&buf, 4, upper >> 16, false);
810 *(buf++) = '-';
811 PutBaseChars<16>(&buf, 4, upper, false);
812 *(buf++) = '-';
813 PutBaseChars<16>(&buf, 4, lower >> 48, false);
814 *(buf++) = '-';
815 PutBaseChars<16>(&buf, 12, lower, false);
816 assert(buf == &result[36]);
817
818 // Verify variant 1 version 4
819 assert(result[14] == '4');
820 assert(result[19] == '8' || result[19] == '9' || result[19] == 'a' ||
821 result[19] == 'b');
822 }
823 return result;
824}

877void Logger::Logv(const InfoLogLevel log_level, const char* format,
878 va_list ap) {
879 static const char* kInfoLogLevelNames[5] = {"DEBUG", "INFO", "WARN", "ERROR",
880 "FATAL"};
881 if (log_level < log_level_) {
882 return;
883 }
884
885 if (log_level == InfoLogLevel::INFO_LEVEL) {
886 // Doesn't print log level if it is INFO level.
887 // This is to avoid unexpected performance regression after we add
888 // the feature of log level. All the logs before we add the feature
889 // are INFO level. We don't want to add extra costs to those existing
890 // logging.
891 Logv(format, ap);
892 } else if (log_level == InfoLogLevel::HEADER_LEVEL) {
893 LogHeader(format, ap);
894 } else {
895 char new_format[500];
896 snprintf(new_format, sizeof(new_format) - 1, "[%s] %s",
897 kInfoLogLevelNames[log_level], format);
898 Logv(new_format, ap);
899 }
900
901 if (log_level >= InfoLogLevel::WARN_LEVEL &&
902 log_level != InfoLogLevel::HEADER_LEVEL) {
903 // Log messages with severity of warning or higher should be rare and are
904 // sometimes followed by an unclean crash. We want to be sure important
905 // messages are not lost in an application buffer when that happens.
906 Flush();
907 }
908}
