48Status SstFileReader::Open(const std::string& file_path) {
49 auto r = rep_.get();
50 Status s;
51 uint64_t file_size = 0;
52 std::unique_ptr<FSRandomAccessFile> file;
53 std::unique_ptr<RandomAccessFileReader> file_reader;
54 FileOptions fopts(r->soptions);
55 const auto& fs = r->options.env->GetFileSystem();
56
57 s = fs->GetFileSize(file_path, fopts.io_options, &file_size, nullptr);
58 if (s.ok()) {
59 s = fs->NewRandomAccessFile(file_path, fopts, &file, nullptr);
60 }
61 if (s.ok()) {
62 file_reader.reset(new RandomAccessFileReader(std::move(file), file_path));
63 }
64 if (s.ok()) {
65 TableReaderOptions t_opt(
66 r->ioptions, r->moptions.prefix_extractor, r->soptions,
67 r->ioptions.internal_comparator,
68 r->moptions.block_protection_bytes_per_key,
69 /*skip_filters*/ false, /*immortal*/ false,
70 /*force_direct_prefetch*/ false, /*level*/ -1,
71 /*block_cache_tracer*/ nullptr,
72 /*max_file_size_for_l0_meta_pin*/ 0, /*cur_db_session_id*/ "",
73 /*cur_file_num*/ 0,
74 /* unique_id */ {}, /* largest_seqno */ 0,
75 /* tail_size */ 0, r->ioptions.persist_user_defined_timestamps);
76 // Allow open file with global sequence number for backward compatibility.
77 t_opt.largest_seqno = kMaxSequenceNumber;
78 s = r->options.table_factory->NewTableReader(t_opt, std::move(file_reader),
79 file_size, &r->table_reader);
80 }
81 return s;
82}

84Iterator* SstFileReader::NewIterator(const ReadOptions& roptions) {
85 assert(roptions.io_activity == Env::IOActivity::kUnknown);
86 auto r = rep_.get();
87 auto sequence = roptions.snapshot != nullptr
88 ? roptions.snapshot->GetSequenceNumber()
89 : kMaxSequenceNumber;
90 ArenaWrappedDBIter* res = new ArenaWrappedDBIter();
91 res->Init(
92 r->options.env, roptions, r->ioptions, r->moptions, nullptr /* version */,
93 sequence, r->moptions.max_sequential_skip_in_iterations,
94 0 /* version_number */, nullptr /* read_callback */, nullptr /* cfh */,
95 true /* expose_blob_index */, false /* allow_refresh */);
96 auto internal_iter = r->table_reader->NewIterator(
97 res->GetReadOptions(), r->moptions.prefix_extractor.get(),
98 res->GetArena(), false /* skip_filters */,
99 TableReaderCaller::kSSTFileReader);
100 res->SetIterUnderDBIter(internal_iter);
101 return res;
102}

104std::unique_ptr<Iterator> SstFileReader::NewTableIterator() {
105 auto r = rep_.get();
106 InternalIterator* internal_iter = r->table_reader->NewIterator(
107 r->roptions_for_table_iter, r->moptions.prefix_extractor.get(),
108 /*arena*/ nullptr, false /* skip_filters */,
109 TableReaderCaller::kSSTFileReader);
110 assert(internal_iter);
111 if (internal_iter == nullptr) {
112 // Do not attempt to create a TableIterator if we cannot get a valid
113 // InternalIterator.
114 return nullptr;
115 }
116 return std::make_unique<TableIterator>(internal_iter);
117}

130Status SstFileReader::VerifyNumEntries(const ReadOptions& read_options) {
131 Rep* r = rep_.get();
132 std::unique_ptr<InternalIterator> internal_iter{r->table_reader->NewIterator(
133 read_options, r->moptions.prefix_extractor.get(), nullptr,
134 false /* skip_filters */, TableReaderCaller::kSSTFileReader)};
135 internal_iter->SeekToFirst();
136 Status s = internal_iter->status();
137 if (!s.ok()) {
138 return s;
139 }
140 uint64_t num_read = 0;
141 for (; internal_iter->Valid(); internal_iter->Next()) {
142 ++num_read;
143 }
144 s = internal_iter->status();
145 if (!s.ok()) {
146 return s;
147 }
148 std::shared_ptr<const TableProperties> tp = GetTableProperties();
149 if (!tp) {
150 s = Status::Corruption("table properties not available");
151 } else {
152 // TODO: verify num_range_deletions
153 uint64_t expected = tp->num_entries - tp->num_range_deletions;
154 if (num_read != expected) {
155 std::ostringstream oss;
156 oss << "Table property expects " << expected
157 << " entries when excluding range deletions,"
158 << " but scanning the table returned " << std::to_string(num_read)
159 << " entries";
160 s = Status::Corruption(oss.str());
161 }
162 }
163 return s;
164}
