287// An escalation barrier function. Allow escalation iff the first two bytes are
288// identical.
289static bool escalation_barrier(const Endpoint& a, const Endpoint& b) {
290 assert(a.slice.size() > 2);
291 assert(b.slice.size() > 2);
292 if (memcmp(a.slice.data(), b.slice.data(), 2)) {
293 return true; // This is a barrier
294 } else {
295 return false; // No barrier
296 }
297}

299TEST_F(RangeLockingTest, LockEscalationBarrier) {
300 auto cf = db->DefaultColumnFamily();
301
302 auto counters = range_lock_mgr->GetStatus();
303
304 // Initially not using any lock memory
305 ASSERT_EQ(counters.escalation_count, 0);
306
307 range_lock_mgr->SetMaxLockMemory(8000);
308 range_lock_mgr->SetEscalationBarrierFunc(escalation_barrier);
309
310 // Insert enough locks to cause lock escalations to happen
311 auto txn = NewTxn();
312 const int N = 2000;
313 for (int i = 0; i < N; i++) {
314 std::ostringstream buf;
315 buf << std::setw(4) << std::setfill('0') << i;
316 std::string buf_str = buf.str();
317 ASSERT_OK(txn->GetRangeLock(cf, Endpoint(buf_str), Endpoint(buf_str)));
318 }
319 counters = range_lock_mgr->GetStatus();
320 ASSERT_GT(counters.escalation_count, 0);
321
322 // Check that lock escalation was not performed across escalation barriers:
323 // Use another txn to acquire locks near the barriers.
324 auto txn2 = NewTxn();
325 range_lock_mgr->SetMaxLockMemory(500000);
326 for (int i = 100; i < N; i += 100) {
327 std::ostringstream buf;
328 buf << std::setw(4) << std::setfill('0') << i - 1 << "-a";
329 std::string buf_str = buf.str();
330 // Check that we CAN get a lock near the escalation barrier
331 ASSERT_OK(txn2->GetRangeLock(cf, Endpoint(buf_str), Endpoint(buf_str)));
332 }
333
334 txn->Rollback();
335 txn2->Rollback();
336 delete txn;
337 delete txn2;
338}

368TEST_F(RangeLockingTest, LockWaiteeAccess) {
369 TransactionOptions txn_options;
370 auto cf = db->DefaultColumnFamily();
371 txn_options.lock_timeout = 60;
372 Transaction* txn0 = db->BeginTransaction(WriteOptions(), txn_options);
373 Transaction* txn1 = db->BeginTransaction(WriteOptions(), txn_options);
374

376 ASSERT_OK(txn0->GetRangeLock(cf, Endpoint("a"), Endpoint("c")));

378 std::atomic<bool> reached(false);
379 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
380 "RangeTreeLockManager::TryRangeLock:EnterWaitingTxn", [&](void* /*arg*/) {
381 reached.store(true);
382 std::this_thread::sleep_for(std::chrono::milliseconds(2000));
383 });
384 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
385
386 port::Thread t([&]() {
387 // Attempt to get a conflicting lock
388 auto s = txn1->GetRangeLock(cf, Endpoint("b"), Endpoint("z"));
389 ASSERT_TRUE(s.ok());
390 txn1->Rollback();
391 });
392

394 std::this_thread::sleep_for(std::chrono::milliseconds(100));
395 }
396 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
397 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->ClearAllCallBacks();
398
399 // Release locks and free the transaction
400 txn0->Rollback();
401 delete txn0;
402
403 t.join();
404
