126// Options shared betweeen various cache implementations that
127// divide the key space into shards using hashing.
128struct ShardedCacheOptions {
129 // Capacity of the cache, in the same units as the `charge` of each entry.
130 // This is typically measured in bytes, but can be a different unit if using
131 // kDontChargeCacheMetadata.
132 size_t capacity = 0;
133
134 // Cache is sharded into 2^num_shard_bits shards, by hash of key.
135 // If < 0, a good default is chosen based on the capacity and the
136 // implementation. (Mutex-based implementations are much more reliant
137 // on many shards for parallel scalability.)
138 int num_shard_bits = -1;
139
140 // If strict_capacity_limit is set, Insert() will fail if there is not
141 // enough capacity for the new entry along with all the existing referenced
142 // (pinned) cache entries. (Unreferenced cache entries are evicted as
143 // needed, sometimes immediately.) If strict_capacity_limit == false
144 // (default), Insert() never fails.
145 bool strict_capacity_limit = false;
146
147 // If non-nullptr, RocksDB will use this allocator instead of system
148 // allocator when allocating memory for cache blocks.
149 //
150 // Caveat: when the cache is used as block cache, the memory allocator is
151 // ignored when dealing with compression libraries that allocate memory
152 // internally (currently only XPRESS).
153 std::shared_ptr<MemoryAllocator> memory_allocator;
154
155 // See CacheMetadataChargePolicy
156 CacheMetadataChargePolicy metadata_charge_policy =
157 kDefaultCacheMetadataChargePolicy;
158
159 // A SecondaryCache instance to use the non-volatile tier. For a RowCache
160 // this option must be kept as default empty.
161 std::shared_ptr<SecondaryCache> secondary_cache;
162
163 // See hash_seed comments below
164 static constexpr int32_t kQuasiRandomHashSeed = -1;
165 static constexpr int32_t kHostHashSeed = -2;
166
167 // EXPERT OPTION: Specifies how a hash seed should be determined for the
168 // cache, or specifies a specific seed (only recommended for diagnostics or
169 // testing).
170 //
171 // Background: it could be dangerous to have different cache instances
172 // access the same SST files with the same hash seed, as correlated unlucky
173 // hashing across hosts or restarts could cause a widespread issue, rather
174 // than an isolated one. For example, with smaller block caches, it is
175 // possible for large full Bloom filters in a set of SST files to be randomly
176 // clustered into one cache shard, causing mutex contention or a thrashing
177 // condition as there's little or no space left for other entries assigned to
178 // the shard. If a set of SST files is broadcast and used on many hosts, we
179 // should ensure all have an independent chance of balanced shards.
180 //
181 // Values >= 0 will be treated as fixed hash seeds. Values < 0 are reserved
182 // for methods of dynamically choosing a seed, currently:
183 // * kQuasiRandomHashSeed - Each cache created chooses a seed mostly randomly,
184 // except that within a process, no seed is repeated until all have been
185 // issued.
186 // * kHostHashSeed - The seed is determined based on hashing the host name.
187 // Although this is arguably slightly worse for production reliability, it
188 // solves the essential problem of cross-host correlation while ensuring
189 // repeatable behavior on a host, for diagnostic purposes.
190 int32_t hash_seed = kHostHashSeed;
191
192 ShardedCacheOptions() {}
193 ShardedCacheOptions(
194 size_t _capacity, int _num_shard_bits, bool _strict_capacity_limit,
195 std::shared_ptr<MemoryAllocator> _memory_allocator = nullptr,
196 CacheMetadataChargePolicy _metadata_charge_policy =
197 kDefaultCacheMetadataChargePolicy)
198 : capacity(_capacity),
199 num_shard_bits(_num_shard_bits),
200 strict_capacity_limit(_strict_capacity_limit),
201 memory_allocator(std::move(_memory_allocator)),
202 metadata_charge_policy(_metadata_charge_policy) {}
203 // Make ShardedCacheOptions polymorphic
204 virtual ~ShardedCacheOptions() = default;
205};

207// LRUCache - A cache using LRU eviction to stay at or below a set capacity.
208// The cache is sharded to 2^num_shard_bits shards, by hash of the key.
209// The total capacity is divided and evenly assigned to each shard, and each
210// shard has its own LRU list for evictions. Each shard also has a mutex for
211// exclusive access during operations; even read operations need exclusive
212// access in order to update the LRU list. Mutex contention is usually low
213// with enough shards.
214struct LRUCacheOptions : public ShardedCacheOptions {
215 // Ratio of cache reserved for high-priority and low-priority entries,
216 // respectively. (See Cache::Priority below more information on the levels.)
217 // Valid values are between 0 and 1 (inclusive), and the sum of the two
218 // values cannot exceed 1.
219 //
220 // If high_pri_pool_ratio is greater than zero, a dedicated high-priority LRU
221 // list is maintained by the cache. A ratio of 0.5 means non-high-priority
222 // entries will use midpoint insertion. Similarly, if low_pri_pool_ratio is
223 // greater than zero, a dedicated low-priority LRU list is maintained.
224 // There is also a bottom-priority LRU list, which is always enabled and not
225 // explicitly configurable. Entries are spilled over to the next available
226 // lower-priority pool if a certain pool's capacity is exceeded.
227 //
228 // Entries with cache hits are inserted into the highest priority LRU list
229 // available regardless of the entry's priority. Entries without hits
230 // are inserted into highest priority LRU list available whose priority
231 // does not exceed the entry's priority. (For example, high-priority items
232 // with no hits are placed in the high-priority pool if available;
233 // otherwise, they are placed in the low-priority pool if available;
234 // otherwise, they are placed in the bottom-priority pool.) This results
235 // in lower-priority entries without hits getting evicted from the cache
236 // sooner.
237 double high_pri_pool_ratio = 0.5;
238 double low_pri_pool_ratio = 0.0;
239
240 // Whether to use adaptive mutexes for cache shards. Note that adaptive
241 // mutexes need to be supported by the platform in order for this to have any
242 // effect. The default value is true if RocksDB is compiled with
243 // -DROCKSDB_DEFAULT_TO_ADAPTIVE_MUTEX, false otherwise.
244 bool use_adaptive_mutex = kDefaultToAdaptiveMutex;
245
246 LRUCacheOptions() {}
247 LRUCacheOptions(size_t _capacity, int _num_shard_bits,
248 bool _strict_capacity_limit, double _high_pri_pool_ratio,
249 std::shared_ptr<MemoryAllocator> _memory_allocator = nullptr,
250 bool _use_adaptive_mutex = kDefaultToAdaptiveMutex,
251 CacheMetadataChargePolicy _metadata_charge_policy =
252 kDefaultCacheMetadataChargePolicy,
253 double _low_pri_pool_ratio = 0.0)
254 : ShardedCacheOptions(_capacity, _num_shard_bits, _strict_capacity_limit,
255 std::move(_memory_allocator),
256 _metadata_charge_policy),
257 high_pri_pool_ratio(_high_pri_pool_ratio),
258 low_pri_pool_ratio(_low_pri_pool_ratio),
259 use_adaptive_mutex(_use_adaptive_mutex) {}
260
261 // Construct an instance of LRUCache using these options
262 std::shared_ptr<Cache> MakeSharedCache() const;
263
264 // Construct an instance of LRUCache for use as a row cache, typically for
265 // `DBOptions::row_cache`. Some options are not relevant to row caches.
266 std::shared_ptr<RowCache> MakeSharedRowCache() const;
267};

374// HyperClockCache - A lock-free Cache alternative for RocksDB block cache
375// that offers much improved CPU efficiency vs. LRUCache under high parallel
376// load or high contention, with some caveats:
377// * Not a general Cache implementation: can only be used for
378// BlockBasedTableOptions::block_cache, which RocksDB uses in a way that is
379// compatible with HyperClockCache.
380// * Requires an extra tuning parameter: see estimated_entry_charge below.
381// Similarly, substantially changing the capacity with SetCapacity could
382// harm efficiency. -> EXPERIMENTAL: the tuning parameter can be set to 0
383// to find the appropriate balance automatically.
384// * Cache priorities are less aggressively enforced, which could cause
385// cache dilution from long range scans (unless they use fill_cache=false).
386//
387// See internal cache/clock_cache.h for full description.
388struct HyperClockCacheOptions : public ShardedCacheOptions {
389 // The estimated average `charge` associated with cache entries.
390 //
391 // EXPERIMENTAL: the field can be set to 0 to size the table dynamically
392 // and automatically. See also min_avg_entry_charge. This feature requires
393 // platform support for lazy anonymous memory mappings (incl Linux, Windows).
394 // Performance is very similar to choosing the best configuration parameter.
395 //
396 // PRODUCTION-TESTED: This is a critical configuration parameter for good
397 // performance, because having a table size that is fixed at creation time
398 // greatly reduces the required synchronization between threads.
399 // * If the estimate is substantially too low (e.g. less than half the true
400 // average) then metadata space overhead with be substantially higher (e.g.
401 // 200 bytes per entry rather than 100). With kFullChargeCacheMetadata, this
402 // can slightly reduce cache hit rates, and slightly reduce access times due
403 // to the larger working memory size.
404 // * If the estimate is substantially too high (e.g. 25% higher than the true
405 // average) then there might not be sufficient slots in the hash table for
406 // both efficient operation and capacity utilization (hit rate). The hyper
407 // cache will evict entries to prevent load factors that could dramatically
408 // affect lookup times, instead letting the hit rate suffer by not utilizing
409 // the full capacity.
410 //
411 // A reasonable choice is the larger of block_size and metadata_block_size.
412 // When WriteBufferManager (and similar) charge memory usage to the block
413 // cache, this can lead to the same effect as estimate being too low, which
414 // is better than the opposite. Therefore, the general recommendation is to
415 // assume that other memory charged to block cache could be negligible, and
416 // ignore it in making the estimate.
417 //
418 // The best parameter choice based on a cache in use is given by
419 // GetUsage() / GetOccupancyCount(), ignoring metadata overheads such as
420 // with kDontChargeCacheMetadata. More precisely with
421 // kFullChargeCacheMetadata is (GetUsage() - 64 * GetTableAddressCount()) /
422 // GetOccupancyCount(). However, when the average value size might vary
423 // (e.g. balance between metadata and data blocks in cache), it is better
424 // to estimate toward the lower side than the higher side.
425 size_t estimated_entry_charge;
426
427 // EXPERIMENTAL: When estimated_entry_charge == 0, this parameter establishes
428 // a promised lower bound on the average charge of all entries in the table,
429 // which is roughly the average uncompressed SST block size of block cache
430 // entries, typically > 4KB. The default should generally suffice with almost
431 // no cost. (This option is ignored for estimated_entry_charge > 0.)
432 //
433 // More detail: The table for indexing cache entries will grow automatically
434 // as needed, but a hard upper bound on that size is needed at creation time.
435 // The reason is that a contiguous memory mapping for the maximum size is
436 // created, but memory pages are only mapped to physical (RSS) memory as
437 // needed. If the average charge of all entries in the table falls below
438 // this value, the table will operate below its full logical capacity (total
439 // memory usage) because it has reached its physical capacity for efficiently
440 // indexing entries. The hash table is never allowed to exceed a certain safe
441 // load factor for efficient Lookup, Insert, etc.
442 size_t min_avg_entry_charge = 450;
443
444 // A tuning parameter to cap eviction CPU usage in a "thrashing" situation
445 // by allowing the memory capacity to be exceeded slightly as needed. The
446 // default setting should offer balanced protection against excessive CPU
447 // and memory usage under extreme stress conditions, with no effect on
448 // normal operation. Such stress conditions are proportionally more likely
449 // with small caches (10s of MB or less) vs. large caches (GB-scale).
450 // (NOTE: With the unusual setting of strict_capacity_limit=true, this
451 // parameter is ignored.)
452 //
453 // BACKGROUND: Without some kind of limiter, inserting into a CLOCK-based
454 // cache with no evictable entries (all "pinned") requires scanning the
455 // entire cache to determine that nothing can be evicted. (By contrast,
456 // LRU caches can determine no entries are evictable in O(1) time, but
457 // require more synchronization/coordination on that eviction metadata.)
458 // This aspect of a CLOCK cache can make a stressed situation worse by
459 // bogging down the CPU with repeated scans of the cache. And with
460 // strict_capacity_limit=false (normal setting), finding something evictable
461 // doesn't change the outcome of insertion: the entry is inserted anyway
462 // and the cache is allowed to exceed its target capacity if necessary.
463 //
464 // SOLUTION: Eviction is aborted upon seeing some number of pinned
465 // entries before evicting anything, or if the ratio of pinned to evicted
466 // is too high. This setting `eviction_effort_cap` essentially controls both
467 // that allowed initial number of pinned entries and the maximum allowed
468 // ratio. As the pinned size approaches the target cache capacity, roughly
469 // 1/eviction_effort_cap additional portion of the capacity might be kept
470 // in memory and evictable in order to keep CLOCK eviction reasonably
471 // performant. Under the default setting and high stress conditions, this
472 // memory overhead is around 3-5%. Under normal or even moderate stress
473 // conditions, the memory overhead is negligible to zero.
474 //
475 // A large value like 1000 offers some protection with essentially no
476 // memory overhead, while the minimum value of 1 could be useful for a
477 // small cache where roughly doubling in size under stress could be OK to
478 // keep operations very fast.
479 int eviction_effort_cap = 30;
480
481 HyperClockCacheOptions(
482 size_t _capacity, size_t _estimated_entry_charge,
483 int _num_shard_bits = -1, bool _strict_capacity_limit = false,
484 std::shared_ptr<MemoryAllocator> _memory_allocator = nullptr,
485 CacheMetadataChargePolicy _metadata_charge_policy =
486 kDefaultCacheMetadataChargePolicy)
487 : ShardedCacheOptions(_capacity, _num_shard_bits, _strict_capacity_limit,
488 std::move(_memory_allocator),
489 _metadata_charge_policy),
490 estimated_entry_charge(_estimated_entry_charge) {}

494};
