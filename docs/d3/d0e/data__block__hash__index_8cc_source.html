16void DataBlockHashIndexBuilder::Add(const Slice& key,
17 const size_t restart_index) {
18 assert(Valid());
19 if (restart_index > kMaxRestartSupportedByHashIndex) {
20 valid_ = false;
21 return;
22 }
23
24 uint32_t hash_value = GetSliceHash(key);
25 hash_and_restart_pairs_.emplace_back(hash_value,
26 static_cast<uint8_t>(restart_index));
27 estimated_num_buckets_ += bucket_per_key_;
28}

30void DataBlockHashIndexBuilder::Finish(std::string& buffer) {
31 assert(Valid());
32 uint16_t num_buckets = static_cast<uint16_t>(estimated_num_buckets_);
33
34 if (num_buckets == 0) {
35 num_buckets = 1; // sanity check
36 }
37
38 // The build-in hash cannot well distribute strings when into different
39 // buckets when num_buckets is power of two, resulting in high hash
40 // collision.
41 // We made the num_buckets to be odd to avoid this issue.
42 num_buckets |= 1;
43
44 std::vector<uint8_t> buckets(num_buckets, kNoEntry);
45 // write the restart_index array
46 for (auto& entry : hash_and_restart_pairs_) {
47 uint32_t hash_value = entry.first;
48 uint8_t restart_index = entry.second;
49 uint16_t buck_idx = static_cast<uint16_t>(hash_value % num_buckets);
50 if (buckets[buck_idx] == kNoEntry) {
51 buckets[buck_idx] = restart_index;
52 } else if (buckets[buck_idx] != restart_index) {
53 // same bucket cannot store two different restart_index, mark collision
54 buckets[buck_idx] = kCollision;
55 }
56 }
57
58 for (uint8_t restart_index : buckets) {
59 buffer.append(
60 const_cast<const char*>(reinterpret_cast<char*>(&restart_index)),
61 sizeof(restart_index));
62 }
63
64 // write NUM_BUCK
65 PutFixed16(&buffer, num_buckets);
66
67 assert(buffer.size() <= kMaxBlockSizeSupportedByHashIndex);
68}

76void DataBlockHashIndex::Initialize(const char* data, uint16_t size,
77 uint16_t* map_offset) {
78 assert(size >= sizeof(uint16_t)); // NUM_BUCKETS
79 num_buckets_ = DecodeFixed16(data + size - sizeof(uint16_t));
80 assert(num_buckets_ > 0);
81 assert(size > num_buckets_ * sizeof(uint8_t));
82 *map_offset = static_cast<uint16_t>(size - sizeof(uint16_t) -
83 num_buckets_ * sizeof(uint8_t));
84}

86uint8_t DataBlockHashIndex::Lookup(const char* data, uint32_t map_offset,
87 const Slice& key) const {
88 uint32_t hash_value = GetSliceHash(key);
89 uint16_t idx = static_cast<uint16_t>(hash_value % num_buckets_);
90 const char* bucket_table = data + map_offset;
91 return static_cast<uint8_t>(*(bucket_table + idx * sizeof(uint8_t)));
92}
