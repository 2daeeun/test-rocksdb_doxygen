24PartitionedFilterBlockBuilder::PartitionedFilterBlockBuilder(
25 const SliceTransform* _prefix_extractor, bool whole_key_filtering,
26 FilterBitsBuilder* filter_bits_builder, int index_block_restart_interval,
27 const bool use_value_delta_encoding,
28 PartitionedIndexBuilder* const p_index_builder,
29 const uint32_t partition_size, size_t ts_sz,
30 const bool persist_user_defined_timestamps,
31 bool decouple_from_index_partitions)
32 : FullFilterBlockBuilder(_prefix_extractor, whole_key_filtering,
33 filter_bits_builder),
34 p_index_builder_(p_index_builder),
35 ts_sz_(ts_sz),
36 decouple_from_index_partitions_(decouple_from_index_partitions),
37 index_on_filter_block_builder_(
38 index_block_restart_interval, true /*use_delta_encoding*/,
39 use_value_delta_encoding,
40 BlockBasedTableOptions::kDataBlockBinarySearch /* index_type */,
41 0.75 /* data_block_hash_table_util_ratio */, ts_sz,
42 persist_user_defined_timestamps, false /* is_user_key */),
43 index_on_filter_block_builder_without_seq_(
44 index_block_restart_interval, true /*use_delta_encoding*/,
45 use_value_delta_encoding,
46 BlockBasedTableOptions::kDataBlockBinarySearch /* index_type */,
47 0.75 /* data_block_hash_table_util_ratio */, ts_sz,
48 persist_user_defined_timestamps, true /* is_user_key */) {
49 // Compute keys_per_partition_
50 keys_per_partition_ = static_cast<uint32_t>(
51 filter_bits_builder_->ApproximateNumEntries(partition_size));
52 if (keys_per_partition_ < 1) {
53 // partition_size (minus buffer, ~10%) might be smaller than minimum
54 // filter size, sometimes based on cache line size. Try to find that
55 // minimum size without CalculateSpace (not necessarily available).
56 uint32_t larger = std::max(partition_size + 4, uint32_t{16});
57 for (;;) {
58 keys_per_partition_ = static_cast<uint32_t>(
59 filter_bits_builder_->ApproximateNumEntries(larger));
60 if (keys_per_partition_ >= 1) {
61 break;
62 }
63 larger += larger / 4;
64 if (larger > 100000) {
65 // might be a broken implementation. substitute something reasonable:
66 // 1 key / byte.
67 keys_per_partition_ = partition_size;
68 break;
69 }
70 }
71 }
72 if (keys_per_partition_ > 1 && prefix_extractor()) {
73 // Correct for adding next prefix in CutAFilterBlock *after* checking
74 // against this threshold
75 keys_per_partition_--;
76 }
77}

83bool PartitionedFilterBlockBuilder::DecideCutAFilterBlock() {
84 size_t added = filter_bits_builder_->EstimateEntriesAdded();
85 if (decouple_from_index_partitions_) {
86 // NOTE: Can't just use ==, because estimated might be incremented by more
87 // than one.
88 return added >= keys_per_partition_;
89 } else {
90 // NOTE: Can't just use ==, because estimated might be incremented by more
91 // than one.
92 if (added >= keys_per_partition_) {
93 // Currently only index builder is in charge of cutting a partition. We
94 // keep requesting until it is granted.
95 p_index_builder_->RequestPartitionCut();
96 }
97 return p_index_builder_->ShouldCutFilterBlock();
98 }

101void PartitionedFilterBlockBuilder::CutAFilterBlock(const Slice* next_key,
102 const Slice* next_prefix,
103 const Slice& prev_key) {
104 // When there is a next partition, add the prefix of the first key in the
105 // next partition before closing this one out. This is needed to support
106 // prefix Seek, because there could exist a key k where
107 // * last_key < k < next_key
108 // * prefix(last_key) != prefix(k)
109 // * prefix(k) == prefix(next_key)
110 // * seeking to k lands in this partition, not the next
111 // in which case the iterator needs to find next_key despite starting in
112 // the partition before it. (This fixes a bug in the original implementation
113 // of format_version=3.)
114 if (next_prefix) {
115 if (whole_key_filtering()) {
116 // NOTE: At the end of building filter bits, we need a special case for
117 // treating prefix as an "alt" entry. See AddKeyAndAlt() comment. This is
118 // a reasonable hack for that.
119 filter_bits_builder_->AddKeyAndAlt(*next_prefix, *next_prefix);
120 } else {
121 filter_bits_builder_->AddKey(*next_prefix);
122 }
123 }
124
125 // Cut the partition
126 total_added_in_built_ += filter_bits_builder_->EstimateEntriesAdded();
127 std::unique_ptr<const char[]> filter_data;
128 Status filter_construction_status = Status::OK();
129 Slice filter =
130 filter_bits_builder_->Finish(&filter_data, &filter_construction_status);
131 if (filter_construction_status.ok()) {
132 filter_construction_status = filter_bits_builder_->MaybePostVerify(filter);
133 }
134 std::string ikey;
135 if (decouple_from_index_partitions_) {
136 if (ts_sz_ > 0) {
137 AppendKeyWithMinTimestamp(&ikey, prev_key, ts_sz_);
138 } else {
139 ikey = prev_key.ToString();
140 }
141 AppendInternalKeyFooter(&ikey, /*seqno*/ 0, ValueType::kTypeDeletion);
142 } else {
143 ikey = p_index_builder_->GetPartitionKey();
144 }
145 filters_.push_back({std::move(ikey), std::move(filter_data), filter});
146 partitioned_filters_construction_status_.UpdateIfOk(
147 filter_construction_status);
148
149 // If we are building another filter partition, the last prefix in the
150 // previous partition should be added to support prefix SeekForPrev.
151 // (Analogous to above fix for prefix Seek.)
152 if (next_key && prefix_extractor() &&
153 prefix_extractor()->InDomain(prev_key)) {
154 // NOTE: At the beginning of building filter bits, we don't need a special
155 // case for treating prefix as an "alt" entry.
156 // See DBBloomFilterTest.FilterBitsBuilderDedup
157 filter_bits_builder_->AddKey(prefix_extractor()->Transform(prev_key));
158 }
159}

182void PartitionedFilterBlockBuilder::AddImpl(const Slice& key_without_ts,
183 const Slice& prev_key_without_ts) {
184 // When filter partitioning is coupled to index partitioning, we need to
185 // check for cutting a block even if we aren't adding anything this time.
186 bool cut = DecideCutAFilterBlock();
187 if (prefix_extractor() && prefix_extractor()->InDomain(key_without_ts)) {
188 Slice prefix = prefix_extractor()->Transform(key_without_ts);
189 if (cut) {
190 CutAFilterBlock(&key_without_ts, &prefix, prev_key_without_ts);
191 }
192 if (whole_key_filtering()) {
193 filter_bits_builder_->AddKeyAndAlt(key_without_ts, prefix);
194 } else {
195 filter_bits_builder_->AddKey(prefix);
196 }
197 } else {
198 if (cut) {
199 CutAFilterBlock(&key_without_ts, nullptr /*no prefix*/,
200 prev_key_without_ts);
201 }
202 if (whole_key_filtering()) {
203 filter_bits_builder_->AddKey(key_without_ts);
204 }
205 }
206}

223Status PartitionedFilterBlockBuilder::Finish(
224 const BlockHandle& last_partition_block_handle, Slice* filter,
225 std::unique_ptr<const char[]>* filter_owner) {
226 if (finishing_front_filter_) {
227 assert(!filters_.empty());
228 auto& e = filters_.front();
229
230 assert(last_partition_block_handle != BlockHandle{});
231 // Record the handle of the last written filter block in the index
232 std::string handle_encoding;
233 last_partition_block_handle.EncodeTo(&handle_encoding);
234 std::string handle_delta_encoding;
235 PutVarsignedint64(
236 &handle_delta_encoding,
237 last_partition_block_handle.size() - last_encoded_handle_.size());
238 last_encoded_handle_ = last_partition_block_handle;
239 const Slice handle_delta_encoding_slice(handle_delta_encoding);
240
241 index_on_filter_block_builder_.Add(e.ikey, handle_encoding,
242 &handle_delta_encoding_slice);
243 if (!p_index_builder_->seperator_is_key_plus_seq()) {
244 index_on_filter_block_builder_without_seq_.Add(
245 ExtractUserKey(e.ikey), handle_encoding,
246 &handle_delta_encoding_slice);
247 }
248
249 filters_.pop_front();
250 } else {
251 assert(last_partition_block_handle == BlockHandle{});
252 if (filter_bits_builder_->EstimateEntriesAdded() > 0) {
253 // PrevKeyBeforeFinish was not called
254 assert(!DEBUG_add_with_prev_key_called_);
255 CutAFilterBlock(nullptr, nullptr, prev_key_without_ts_);
256 }
257 // Nothing uncommitted
258 assert(filter_bits_builder_->EstimateEntriesAdded() == 0);
259 }
260
261 Status s = partitioned_filters_construction_status_;
262 assert(!s.IsIncomplete());
263
264 if (s.ok()) {
265 // If there is no filter partition left, then return the index on filter
266 // partitions
267 if (UNLIKELY(filters_.empty())) {
268 if (!index_on_filter_block_builder_.empty()) {
269 // Simplest to just add them all at the end
270 if (p_index_builder_->seperator_is_key_plus_seq()) {
271 *filter = index_on_filter_block_builder_.Finish();
272 } else {
273 *filter = index_on_filter_block_builder_without_seq_.Finish();
274 }
275 } else {
276 // This is the rare case where no key was added to the filter
277 *filter = Slice{};
278 }
279 } else {
280 // Return the next filter partition in line and set Incomplete() status to
281 // indicate we expect more calls to Finish
282 s = Status::Incomplete();
283 finishing_front_filter_ = true;
284
285 auto& e = filters_.front();
286 if (filter_owner != nullptr) {
287 *filter_owner = std::move(e.filter_owner);
288 }
289 *filter = e.filter;
290 }
291 }
292 return s;
293}

300std::unique_ptr<FilterBlockReader> PartitionedFilterBlockReader::Create(
301 const BlockBasedTable* table, const ReadOptions& ro,
302 FilePrefetchBuffer* prefetch_buffer, bool use_cache, bool prefetch,
303 bool pin, BlockCacheLookupContext* lookup_context) {
304 assert(table);
305 assert(table->get_rep());
306 assert(!pin || prefetch);
307
308 CachableEntry<Block_kFilterPartitionIndex> filter_block;
309 if (prefetch || !use_cache) {
310 const Status s = ReadFilterBlock(table, prefetch_buffer, ro, use_cache,
311 nullptr /* get_context */, lookup_context,
312 &filter_block);
313 if (!s.ok()) {
314 IGNORE_STATUS_IF_ERROR(s);
315 return std::unique_ptr<FilterBlockReader>();
316 }
317
318 if (use_cache && !pin) {
319 filter_block.Reset();
320 }
321 }
322
323 return std::unique_ptr<FilterBlockReader>(
324 new PartitionedFilterBlockReader(table, std::move(filter_block)));
325}

368BlockHandle PartitionedFilterBlockReader::GetFilterPartitionHandle(
369 const CachableEntry<Block_kFilterPartitionIndex>& filter_block,
370 const Slice& entry) const {
371 IndexBlockIter iter;
372 const InternalKeyComparator* const comparator = internal_comparator();
373 Statistics* kNullStats = nullptr;
374 filter_block.GetValue()->NewIndexIterator(
375 comparator->user_comparator(),
376 table()->get_rep()->get_global_seqno(BlockType::kFilterPartitionIndex),
377 &iter, kNullStats, true /* total_order_seek */,
378 false /* have_first_key */, index_key_includes_seq(),
379 index_value_is_full(), false /* block_contents_pinned */,
380 user_defined_timestamps_persisted());
381 iter.Seek(entry);
382 if (UNLIKELY(!iter.Valid())) {
383 // entry is larger than all the keys. However its prefix might still be
384 // present in the last partition. If this is called by PrefixMayMatch this
385 // is necessary for correct behavior. Otherwise it is unnecessary but safe.
386 // Assuming this is an unlikely case for full key search, the performance
387 // overhead should be negligible.
388 iter.SeekToLast();
389 }
390 assert(iter.Valid());
391 BlockHandle fltr_blk_handle = iter.value().handle;
392 return fltr_blk_handle;
393}

395Status PartitionedFilterBlockReader::GetFilterPartitionBlock(
396 FilePrefetchBuffer* prefetch_buffer, const BlockHandle& fltr_blk_handle,
397 GetContext* get_context, BlockCacheLookupContext* lookup_context,
398 const ReadOptions& read_options,
399 CachableEntry<ParsedFullFilterBlock>* filter_block) const {
400 assert(table());
401 assert(filter_block);
402 assert(filter_block->IsEmpty());
403
404 if (!filter_map_.empty()) {
405 auto iter = filter_map_.find(fltr_blk_handle.offset());
406 // This is a possible scenario since block cache might not have had space
407 // for the partition
408 if (iter != filter_map_.end()) {
409 filter_block->SetUnownedValue(iter->second.GetValue());
410 return Status::OK();
411 }
412 }
413
414 const Status s = table()->RetrieveBlock(
415 prefetch_buffer, read_options, fltr_blk_handle,
416 UncompressionDict::GetEmptyDict(), filter_block, get_context,
417 lookup_context,
418 /* for_compaction */ false, /* use_cache */ true,
419 /* async_read */ false, /* use_block_cache_for_lookup */ true);
420
421 return s;
422}

424bool PartitionedFilterBlockReader::MayMatch(
425 const Slice& slice, const Slice* const_ikey_ptr, GetContext* get_context,
426 BlockCacheLookupContext* lookup_context, const ReadOptions& read_options,
427 FilterFunction filter_function) const {
428 CachableEntry<Block_kFilterPartitionIndex> filter_block;
429 Status s = GetOrReadFilterBlock(get_context, lookup_context, &filter_block,
430 read_options);
431 if (UNLIKELY(!s.ok())) {
432 IGNORE_STATUS_IF_ERROR(s);
433 return true;
434 }
435
436 if (UNLIKELY(filter_block.GetValue()->size() == 0)) {
437 return true;
438 }
439
440 auto filter_handle = GetFilterPartitionHandle(filter_block, *const_ikey_ptr);
441 if (UNLIKELY(filter_handle.size() == 0)) { // key is out of range
442 return false;
443 }
444
445 CachableEntry<ParsedFullFilterBlock> filter_partition_block;
446 s = GetFilterPartitionBlock(nullptr /* prefetch_buffer */, filter_handle,
447 get_context, lookup_context, read_options,
448 &filter_partition_block);
449 if (UNLIKELY(!s.ok())) {
450 IGNORE_STATUS_IF_ERROR(s);
451 return true;
452 }
453
454 FullFilterBlockReader filter_partition(table(),
455 std::move(filter_partition_block));
456 return (filter_partition.*filter_function)(slice, const_ikey_ptr, get_context,
457 lookup_context, read_options);
458}

460void PartitionedFilterBlockReader::MayMatch(
461 MultiGetRange* range, const SliceTransform* prefix_extractor,
462 BlockCacheLookupContext* lookup_context, const ReadOptions& read_options,
463 FilterManyFunction filter_function) const {
464 CachableEntry<Block_kFilterPartitionIndex> filter_block;
465 Status s = GetOrReadFilterBlock(range->begin()->get_context, lookup_context,
466 &filter_block, read_options);
467 if (UNLIKELY(!s.ok())) {
468 IGNORE_STATUS_IF_ERROR(s);
469 return; // Any/all may match
470 }
471
472 if (UNLIKELY(filter_block.GetValue()->size() == 0)) {
473 return; // Any/all may match
474 }
475
476 auto start_iter_same_handle = range->begin();
477 BlockHandle prev_filter_handle = BlockHandle::NullBlockHandle();
478
479 // For all keys mapping to same partition (must be adjacent in sorted order)
480 // share block cache lookup and use full filter multiget on the partition
481 // filter.
482 for (auto iter = start_iter_same_handle; iter != range->end(); ++iter) {
483 // TODO: re-use one top-level index iterator
484 BlockHandle this_filter_handle =
485 GetFilterPartitionHandle(filter_block, iter->ikey);
486 if (!prev_filter_handle.IsNull() &&
487 this_filter_handle != prev_filter_handle) {
488 MultiGetRange subrange(*range, start_iter_same_handle, iter);
489 MayMatchPartition(&subrange, prefix_extractor, prev_filter_handle,
490 lookup_context, read_options, filter_function);
491 range->AddSkipsFrom(subrange);
492 start_iter_same_handle = iter;
493 }
494 if (UNLIKELY(this_filter_handle.size() == 0)) { // key is out of range
495 // Not reachable with current behavior of GetFilterPartitionHandle
496 assert(false);
497 range->SkipKey(iter);
498 prev_filter_handle = BlockHandle::NullBlockHandle();
499 } else {
500 prev_filter_handle = this_filter_handle;
501 }
502 }
503 if (!prev_filter_handle.IsNull()) {
504 MultiGetRange subrange(*range, start_iter_same_handle, range->end());
505 MayMatchPartition(&subrange, prefix_extractor, prev_filter_handle,
506 lookup_context, read_options, filter_function);
507 range->AddSkipsFrom(subrange);
508 }
509}

541// TODO(myabandeh): merge this with the same function in IndexReader
542Status PartitionedFilterBlockReader::CacheDependencies(
543 const ReadOptions& ro, bool pin, FilePrefetchBuffer* tail_prefetch_buffer) {
544 assert(table());
545
546 const BlockBasedTable::Rep* const rep = table()->get_rep();
547 assert(rep);
548
549 BlockCacheLookupContext lookup_context{TableReaderCaller::kPrefetch};
550
551 CachableEntry<Block_kFilterPartitionIndex> filter_block;
552
553 Status s = GetOrReadFilterBlock(nullptr /* get_context */, &lookup_context,
554 &filter_block, ro);
555 if (!s.ok()) {
556 ROCKS_LOG_ERROR(rep->ioptions.logger,
557 "Error retrieving top-level filter block while trying to "
558 "cache filter partitions: %s",
559 s.ToString().c_str());
560 return s;
561 }
562
563 // Before read partitions, prefetch them to avoid lots of IOs
564 assert(filter_block.GetValue());
565
566 IndexBlockIter biter;
567 const InternalKeyComparator* const comparator = internal_comparator();
568 Statistics* kNullStats = nullptr;
569 filter_block.GetValue()->NewIndexIterator(
570 comparator->user_comparator(),
571 rep->get_global_seqno(BlockType::kFilterPartitionIndex), &biter,
572 kNullStats, true /* total_order_seek */, false /* have_first_key */,
573 index_key_includes_seq(), index_value_is_full(),
574 false /* block_contents_pinned */, user_defined_timestamps_persisted());
575 // Index partitions are assumed to be consecuitive. Prefetch them all.
576 // Read the first block offset
577 biter.SeekToFirst();
578 BlockHandle handle = biter.value().handle;
579 uint64_t prefetch_off = handle.offset();
580
581 // Read the last block's offset
582 biter.SeekToLast();
583 handle = biter.value().handle;
584 uint64_t last_off =
585 handle.offset() + handle.size() + BlockBasedTable::kBlockTrailerSize;
586 uint64_t prefetch_len = last_off - prefetch_off;
587 std::unique_ptr<FilePrefetchBuffer> prefetch_buffer;
588 if (tail_prefetch_buffer == nullptr || !tail_prefetch_buffer->Enabled() ||
589 tail_prefetch_buffer->GetPrefetchOffset() > prefetch_off) {
590 rep->CreateFilePrefetchBuffer(ReadaheadParams(), &prefetch_buffer,
591 /*readaheadsize_cb*/ nullptr,
592 /*usage=*/FilePrefetchBufferUsage::kUnknown);
593
594 IOOptions opts;
595 s = rep->file->PrepareIOOptions(ro, opts);
596 if (s.ok()) {
597 s = prefetch_buffer->Prefetch(opts, rep->file.get(), prefetch_off,
598 static_cast<size_t>(prefetch_len));
599 }
600 if (!s.ok()) {
601 return s;
602 }
603 }
604 // After prefetch, read the partitions one by one
605 for (biter.SeekToFirst(); biter.Valid(); biter.Next()) {
606 handle = biter.value().handle;
607
608 CachableEntry<ParsedFullFilterBlock> block;
609 // TODO: Support counter batch update for partitioned index and
610 // filter blocks
611 s = table()->MaybeReadBlockAndLoadToCache(
612 prefetch_buffer ? prefetch_buffer.get() : tail_prefetch_buffer, ro,
613 handle, UncompressionDict::GetEmptyDict(),
614 /* for_compaction */ false, &block, nullptr /* get_context */,
615 &lookup_context, nullptr /* contents */, false,
616 /* use_block_cache_for_lookup */ true);
617 if (!s.ok()) {
618 return s;
619 }
620 assert(s.ok() || block.GetValue() == nullptr);
621
622 if (block.GetValue() != nullptr) {
623 if (block.IsCached()) {
624 if (pin) {
625 filter_map_[handle.offset()] = std::move(block);
626 }
627 }
628 }

633void PartitionedFilterBlockReader::EraseFromCacheBeforeDestruction(
634 uint32_t uncache_aggressiveness) {
635 // NOTE: essentially a copy of
636 // PartitionIndexReader::EraseFromCacheBeforeDestruction
637 if (uncache_aggressiveness > 0) {
638 CachableEntry<Block_kFilterPartitionIndex> top_level_block;
639
640 ReadOptions ro;
641 ro.read_tier = ReadTier::kBlockCacheTier;
642 GetOrReadFilterBlock(/*get_context=*/nullptr,
643 /*lookup_context=*/nullptr, &top_level_block, ro)
644 .PermitUncheckedError();
645
646 if (!filter_map_.empty()) {
647 // All partitions present if any
648 for (auto& e : filter_map_) {
649 e.second.ResetEraseIfLastRef();
650 }
651 } else if (!top_level_block.IsEmpty()) {
652 IndexBlockIter biter;
653 const InternalKeyComparator* const comparator = internal_comparator();
654 Statistics* kNullStats = nullptr;
655 top_level_block.GetValue()->NewIndexIterator(
656 comparator->user_comparator(),
657 table()->get_rep()->get_global_seqno(
658 BlockType::kFilterPartitionIndex),
659 &biter, kNullStats, true /* total_order_seek */,
660 false /* have_first_key */, index_key_includes_seq(),
661 index_value_is_full(), false /* block_contents_pinned */,
662 user_defined_timestamps_persisted());
663
664 UncacheAggressivenessAdvisor advisor(uncache_aggressiveness);
665 for (biter.SeekToFirst(); biter.Valid() && advisor.ShouldContinue();
666 biter.Next()) {
667 bool erased = table()->EraseFromCache(biter.value().handle);
668 advisor.Report(erased);
669 }
670 biter.status().PermitUncheckedError();
671 }
672 top_level_block.ResetEraseIfLastRef();
673 }
674 // Might be needed to un-cache a pinned top-level block
675 FilterBlockReaderCommon<Block_kFilterPartitionIndex>::
676 EraseFromCacheBeforeDestruction(uncache_aggressiveness);
677}
