90void CuckooTableBuilder::Add(const Slice& key, const Slice& value) {
91 if (num_entries_ >= kMaxVectorIdx - 1) {
92 status_ = Status::NotSupported("Number of keys in a file must be < 2^32-1");
93 return;
94 }
95 ParsedInternalKey ikey;
96 Status pik_status =
97 ParseInternalKey(key, &ikey, false /* log_err_key */); // TODO
98 if (!pik_status.ok()) {
99 status_ = Status::Corruption("Unable to parse key into internal key. ",
100 pik_status.getState());
101 return;
102 }
103 if (ikey.type != kTypeDeletion && ikey.type != kTypeValue) {
104 status_ = Status::NotSupported("Unsupported key type " +
105 std::to_string(ikey.type));
106 return;
107 }
108
109 // Determine if we can ignore the sequence number and value type from
110 // internal keys by looking at sequence number from first key. We assume
111 // that if first key has a zero sequence number, then all the remaining
112 // keys will have zero seq. no.
113 if (!has_seen_first_key_) {
114 is_last_level_file_ = ikey.sequence == 0;
115 has_seen_first_key_ = true;
116 smallest_user_key_.assign(ikey.user_key.data(), ikey.user_key.size());
117 largest_user_key_.assign(ikey.user_key.data(), ikey.user_key.size());
118 key_size_ = is_last_level_file_ ? ikey.user_key.size() : key.size();
119 }
120 if (key_size_ != (is_last_level_file_ ? ikey.user_key.size() : key.size())) {
121 status_ = Status::NotSupported("all keys have to be the same size");
122 return;
123 }
124
125 if (ikey.type == kTypeValue) {
126 if (!has_seen_first_value_) {
127 has_seen_first_value_ = true;
128 value_size_ = value.size();
129 }
130 if (value_size_ != value.size()) {
131 status_ = Status::NotSupported("all values have to be the same size");
132 return;
133 }
134
135 if (is_last_level_file_) {
136 kvs_.append(ikey.user_key.data(), ikey.user_key.size());
137 } else {
138 kvs_.append(key.data(), key.size());
139 }
140 kvs_.append(value.data(), value.size());
141 ++num_values_;
142 } else {
143 if (is_last_level_file_) {
144 deleted_keys_.append(ikey.user_key.data(), ikey.user_key.size());
145 } else {
146 deleted_keys_.append(key.data(), key.size());
147 }
148 }
149 ++num_entries_;
150
151 // In order to fill the empty buckets in the hash table, we identify a
152 // key which is not used so far (unused_user_key). We determine this by
153 // maintaining smallest and largest keys inserted so far in bytewise order
154 // and use them to find a key outside this range in Finish() operation.
155 // Note that this strategy is independent of user comparator used here.
156 if (ikey.user_key.compare(smallest_user_key_) < 0) {
157 smallest_user_key_.assign(ikey.user_key.data(), ikey.user_key.size());
158 } else if (ikey.user_key.compare(largest_user_key_) > 0) {
159 largest_user_key_.assign(ikey.user_key.data(), ikey.user_key.size());
160 }
161 if (!use_module_hash_) {
162 if (hash_table_size_ < num_entries_ / max_hash_table_ratio_) {
163 hash_table_size_ *= 2;
164 }
165 }
166}

173Slice CuckooTableBuilder::GetKey(uint64_t idx) const {
174 assert(closed_);
175 if (IsDeletedKey(idx)) {
176 return Slice(
177 &deleted_keys_[static_cast<size_t>((idx - num_values_) * key_size_)],
178 static_cast<size_t>(key_size_));
179 }
180 return Slice(&kvs_[static_cast<size_t>(idx * (key_size_ + value_size_))],
181 static_cast<size_t>(key_size_));
182}

184Slice CuckooTableBuilder::GetUserKey(uint64_t idx) const {
185 assert(closed_);
186 return is_last_level_file_ ? GetKey(idx) : ExtractUserKey(GetKey(idx));
187}

189Slice CuckooTableBuilder::GetValue(uint64_t idx) const {
190 assert(closed_);
191 if (IsDeletedKey(idx)) {
192 static std::string empty_value(static_cast<unsigned int>(value_size_), 'a');
193 return Slice(empty_value);
194 }
195 return Slice(
196 &kvs_[static_cast<size_t>(idx * (key_size_ + value_size_) + key_size_)],
197 static_cast<size_t>(value_size_));
198}

200Status CuckooTableBuilder::MakeHashTable(std::vector<CuckooBucket>* buckets) {
201 buckets->resize(
202 static_cast<size_t>(hash_table_size_ + cuckoo_block_size_ - 1));
203 uint32_t make_space_for_key_call_id = 0;
204 for (uint32_t vector_idx = 0; vector_idx < num_entries_; vector_idx++) {
205 uint64_t bucket_id = 0;
206 bool bucket_found = false;
207 autovector<uint64_t> hash_vals;
208 Slice user_key = GetUserKey(vector_idx);
209 for (uint32_t hash_cnt = 0; hash_cnt < num_hash_func_ && !bucket_found;
210 ++hash_cnt) {
211 uint64_t hash_val =
212 CuckooHash(user_key, hash_cnt, use_module_hash_, hash_table_size_,
213 identity_as_first_hash_, get_slice_hash_);
214 // If there is a collision, check next cuckoo_block_size_ locations for
215 // empty locations. While checking, if we reach end of the hash table,
216 // stop searching and proceed for next hash function.
217 for (uint32_t block_idx = 0; block_idx < cuckoo_block_size_;
218 ++block_idx, ++hash_val) {
219 if ((*buckets)[static_cast<size_t>(hash_val)].vector_idx ==
220 kMaxVectorIdx) {
221 bucket_id = hash_val;
222 bucket_found = true;
223 break;
224 } else {
225 if (ucomp_->Compare(
226 user_key, GetUserKey((*buckets)[static_cast<size_t>(hash_val)]
227 .vector_idx)) == 0) {
228 return Status::NotSupported("Same key is being inserted again.");
229 }
230 hash_vals.push_back(hash_val);
231 }
232 }
233 }
234 while (!bucket_found &&
235 !MakeSpaceForKey(hash_vals, ++make_space_for_key_call_id, buckets,
236 &bucket_id)) {
237 // Rehash by increashing number of hash tables.
238 if (num_hash_func_ >= max_num_hash_func_) {
239 return Status::NotSupported("Too many collisions. Unable to hash.");
240 }
241 // We don't really need to rehash the entire table because old hashes are
242 // still valid and we only increased the number of hash functions.
243 uint64_t hash_val = CuckooHash(user_key, num_hash_func_, use_module_hash_,
244 hash_table_size_, identity_as_first_hash_,
245 get_slice_hash_);
246 ++num_hash_func_;
247 for (uint32_t block_idx = 0; block_idx < cuckoo_block_size_;
248 ++block_idx, ++hash_val) {
249 if ((*buckets)[static_cast<size_t>(hash_val)].vector_idx ==
250 kMaxVectorIdx) {
251 bucket_found = true;
252 bucket_id = hash_val;
253 break;
254 } else {
255 hash_vals.push_back(hash_val);
256 }
257 }
258 }
259 (*buckets)[static_cast<size_t>(bucket_id)].vector_idx = vector_idx;
260 }
261 return Status::OK();
262}

264Status CuckooTableBuilder::Finish() {
265 assert(!closed_);
266 closed_ = true;
267 std::vector<CuckooBucket> buckets;
268 std::string unused_bucket;
269 if (num_entries_ > 0) {
270 // Calculate the real hash size if module hash is enabled.
271 if (use_module_hash_) {
272 hash_table_size_ =
273 static_cast<uint64_t>(num_entries_ / max_hash_table_ratio_);
274 }
275 status_ = MakeHashTable(&buckets);
276 if (!status_.ok()) {
277 return status_;
278 }
279 // Determine unused_user_key to fill empty buckets.
280 std::string unused_user_key = smallest_user_key_;
281 int curr_pos = static_cast<int>(unused_user_key.size()) - 1;
282 while (curr_pos >= 0) {
283 --unused_user_key[curr_pos];
284 if (Slice(unused_user_key).compare(smallest_user_key_) < 0) {
285 break;
286 }
287 --curr_pos;
288 }
289 if (curr_pos < 0) {
290 // Try using the largest key to identify an unused key.
291 unused_user_key = largest_user_key_;
292 curr_pos = static_cast<int>(unused_user_key.size()) - 1;
293 while (curr_pos >= 0) {
294 ++unused_user_key[curr_pos];
295 if (Slice(unused_user_key).compare(largest_user_key_) > 0) {
296 break;
297 }
298 --curr_pos;
299 }
300 }
301 if (curr_pos < 0) {
302 return Status::Corruption("Unable to find unused key");
303 }
304 if (is_last_level_file_) {
305 unused_bucket = unused_user_key;
306 } else {
307 ParsedInternalKey ikey(unused_user_key, 0, kTypeValue);
308 AppendInternalKey(&unused_bucket, ikey);
309 }
310 }
311 properties_.num_entries = num_entries_;
312 properties_.num_deletions = num_entries_ - num_values_;
313 properties_.fixed_key_len = key_size_;
314 properties_.user_collected_properties[CuckooTablePropertyNames::kValueLength]
315 .assign(reinterpret_cast<const char*>(&value_size_), sizeof(value_size_));
316
317 uint64_t bucket_size = key_size_ + value_size_;
318 unused_bucket.resize(static_cast<size_t>(bucket_size), 'a');
319 // Write the table.
320 uint32_t num_added = 0;
321 const IOOptions opts;
322 for (auto& bucket : buckets) {
323 if (bucket.vector_idx == kMaxVectorIdx) {
324 io_status_ = file_->Append(opts, Slice(unused_bucket));
325 } else {
326 ++num_added;
327 io_status_ = file_->Append(opts, GetKey(bucket.vector_idx));
328 if (io_status_.ok()) {
329 if (value_size_ > 0) {
330 io_status_ = file_->Append(opts, GetValue(bucket.vector_idx));
331 }
332 }
333 }
334 if (!io_status_.ok()) {
335 status_ = io_status_;
336 return status_;
337 }
338 }
339 assert(num_added == NumEntries());
340 properties_.raw_key_size = num_added * properties_.fixed_key_len;
341 properties_.raw_value_size = num_added * value_size_;
342
343 uint64_t offset = buckets.size() * bucket_size;
344 properties_.data_size = offset;
345 unused_bucket.resize(static_cast<size_t>(properties_.fixed_key_len));
346 properties_.user_collected_properties[CuckooTablePropertyNames::kEmptyKey] =
347 unused_bucket;
348 properties_.user_collected_properties[CuckooTablePropertyNames::kNumHashFunc]
349 .assign(reinterpret_cast<char*>(&num_hash_func_), sizeof(num_hash_func_));
350
351 properties_
352 .user_collected_properties[CuckooTablePropertyNames::kHashTableSize]
353 .assign(reinterpret_cast<const char*>(&hash_table_size_),
354 sizeof(hash_table_size_));
355 properties_.user_collected_properties[CuckooTablePropertyNames::kIsLastLevel]
356 .assign(reinterpret_cast<const char*>(&is_last_level_file_),
357 sizeof(is_last_level_file_));
358 properties_
359 .user_collected_properties[CuckooTablePropertyNames::kCuckooBlockSize]
360 .assign(reinterpret_cast<const char*>(&cuckoo_block_size_),
361 sizeof(cuckoo_block_size_));
362 properties_
363 .user_collected_properties[CuckooTablePropertyNames::kIdentityAsFirstHash]
364 .assign(reinterpret_cast<const char*>(&identity_as_first_hash_),
365 sizeof(identity_as_first_hash_));
366 properties_
367 .user_collected_properties[CuckooTablePropertyNames::kUseModuleHash]
368 .assign(reinterpret_cast<const char*>(&use_module_hash_),
369 sizeof(use_module_hash_));
370 uint32_t user_key_len = static_cast<uint32_t>(smallest_user_key_.size());
371 properties_
372 .user_collected_properties[CuckooTablePropertyNames::kUserKeyLength]
373 .assign(reinterpret_cast<const char*>(&user_key_len),
374 sizeof(user_key_len));
375
376 // Write meta blocks.
377 MetaIndexBuilder meta_index_builder;
378 PropertyBlockBuilder property_block_builder;
379
380 property_block_builder.AddTableProperty(properties_);
381 property_block_builder.Add(properties_.user_collected_properties);
382 Slice property_block = property_block_builder.Finish();
383 BlockHandle property_block_handle;
384 property_block_handle.set_offset(offset);
385 property_block_handle.set_size(property_block.size());
386 io_status_ = file_->Append(opts, property_block);
387 offset += property_block.size();
388 if (!io_status_.ok()) {
389 status_ = io_status_;
390 return status_;
391 }
392
393 meta_index_builder.Add(kPropertiesBlockName, property_block_handle);
394 Slice meta_index_block = meta_index_builder.Finish();
395
396 BlockHandle meta_index_block_handle;
397 meta_index_block_handle.set_offset(offset);
398 meta_index_block_handle.set_size(meta_index_block.size());
399 io_status_ = file_->Append(opts, meta_index_block);
400 if (!io_status_.ok()) {
401 status_ = io_status_;
402 return status_;
403 }
404
405 FooterBuilder footer;
406 Status s = footer.Build(kCuckooTableMagicNumber, /* format_version */ 1,
407 offset, kNoChecksum, meta_index_block_handle);
408 if (!s.ok()) {
409 status_ = s;
410 return status_;
411 }
412 io_status_ = file_->Append(opts, footer.GetSlice());
413 status_ = io_status_;
414 return status_;
415}

424uint64_t CuckooTableBuilder::FileSize() const {
425 if (closed_) {
426 return file_->GetFileSize();
427 } else if (num_entries_ == 0) {
428 return 0;
429 }
430
431 if (use_module_hash_) {
432 return static_cast<uint64_t>((key_size_ + value_size_) * num_entries_ /
433 max_hash_table_ratio_);
434 } else {
435 // Account for buckets being a power of two.
436 // As elements are added, file size remains constant for a while and
437 // doubles its size. Since compaction algorithm stops adding elements
438 // only after it exceeds the file limit, we account for the extra element
439 // being added here.
440 uint64_t expected_hash_table_size = hash_table_size_;
441 if (expected_hash_table_size < (num_entries_ + 1) / max_hash_table_ratio_) {
442 expected_hash_table_size *= 2;
443 }
444 return (key_size_ + value_size_) * expected_hash_table_size - 1;
445 }
446}

448// This method is invoked when there is no place to insert the target key.
449// It searches for a set of elements that can be moved to accommodate target
450// key. The search is a BFS graph traversal with first level (hash_vals)
451// being all the buckets target key could go to.
452// Then, from each node (curr_node), we find all the buckets that curr_node
453// could go to. They form the children of curr_node in the tree.
454// We continue the traversal until we find an empty bucket, in which case, we
455// move all elements along the path from first level to this empty bucket, to
456// make space for target key which is inserted at first level (*bucket_id).
457// If tree depth exceedes max depth, we return false indicating failure.
458bool CuckooTableBuilder::MakeSpaceForKey(
459 const autovector<uint64_t>& hash_vals,
460 const uint32_t make_space_for_key_call_id,
461 std::vector<CuckooBucket>* buckets, uint64_t* bucket_id) {
462 struct CuckooNode {
463 uint64_t bucket_id;
464 uint32_t depth;
465 uint32_t parent_pos;
466 CuckooNode(uint64_t _bucket_id, uint32_t _depth, int _parent_pos)
467 : bucket_id(_bucket_id), depth(_depth), parent_pos(_parent_pos) {}
468 };
469 // This is BFS search tree that is stored simply as a vector.
470 // Each node stores the index of parent node in the vector.
471 std::vector<CuckooNode> tree;
472 // We want to identify already visited buckets in the current method call so
473 // that we don't add same buckets again for exploration in the tree.
474 // We do this by maintaining a count of current method call in
475 // make_space_for_key_call_id, which acts as a unique id for this invocation
476 // of the method. We store this number into the nodes that we explore in
477 // current method call.
478 // It is unlikely for the increment operation to overflow because the maximum
479 // no. of times this will be called is <= max_num_hash_func_ + num_entries_.
480 for (uint32_t hash_cnt = 0; hash_cnt < num_hash_func_; ++hash_cnt) {
481 uint64_t bid = hash_vals[hash_cnt];
482 (*buckets)[static_cast<size_t>(bid)].make_space_for_key_call_id =
483 make_space_for_key_call_id;
484 tree.emplace_back(bid, 0, 0);
485 }
486 bool null_found = false;
487 uint32_t curr_pos = 0;
488 while (!null_found && curr_pos < tree.size()) {
489 CuckooNode& curr_node = tree[curr_pos];
490 uint32_t curr_depth = curr_node.depth;
491 if (curr_depth >= max_search_depth_) {
492 break;
493 }
494 CuckooBucket& curr_bucket =
495 (*buckets)[static_cast<size_t>(curr_node.bucket_id)];
496 for (uint32_t hash_cnt = 0; hash_cnt < num_hash_func_ && !null_found;
497 ++hash_cnt) {
498 uint64_t child_bucket_id = CuckooHash(
499 GetUserKey(curr_bucket.vector_idx), hash_cnt, use_module_hash_,
500 hash_table_size_, identity_as_first_hash_, get_slice_hash_);
501 // Iterate inside Cuckoo Block.
502 for (uint32_t block_idx = 0; block_idx < cuckoo_block_size_;
503 ++block_idx, ++child_bucket_id) {
504 if ((*buckets)[static_cast<size_t>(child_bucket_id)]
505 .make_space_for_key_call_id == make_space_for_key_call_id) {
506 continue;
507 }
508 (*buckets)[static_cast<size_t>(child_bucket_id)]
509 .make_space_for_key_call_id = make_space_for_key_call_id;
510 tree.emplace_back(child_bucket_id, curr_depth + 1, curr_pos);
511 if ((*buckets)[static_cast<size_t>(child_bucket_id)].vector_idx ==
512 kMaxVectorIdx) {
513 null_found = true;
514 break;
515 }
516 }
517 }
518 ++curr_pos;
519 }
520
521 if (null_found) {
522 // There is an empty node in tree.back(). Now, traverse the path from this
523 // empty node to top of the tree and at every node in the path, replace
524 // child with the parent. Stop when first level is reached in the tree
525 // (happens when 0 <= bucket_to_replace_pos < num_hash_func_) and return
526 // this location in first level for target key to be inserted.
527 uint32_t bucket_to_replace_pos = static_cast<uint32_t>(tree.size()) - 1;
528 while (bucket_to_replace_pos >= num_hash_func_) {
529 CuckooNode& curr_node = tree[bucket_to_replace_pos];
530 (*buckets)[static_cast<size_t>(curr_node.bucket_id)] =
531 (*buckets)[static_cast<size_t>(tree[curr_node.parent_pos].bucket_id)];
532 bucket_to_replace_pos = curr_node.parent_pos;
533 }
534 *bucket_id = tree[bucket_to_replace_pos].bucket_id;
535 }
536 return null_found;
537}
