119// A copyable structure contains information needed to read data from an SST
120// file. It can contain a pointer to a table reader opened for the file, or
121// file number and size, which can be used to create a new table reader for it.
122// The behavior is undefined when a copied of the structure is used when the
123// file is not in any live version any more.
124struct FileDescriptor {
125 // Table reader in table_reader_handle
126 TableReader* table_reader;
127 uint64_t packed_number_and_path_id;
128 uint64_t file_size; // File size in bytes
129 SequenceNumber smallest_seqno; // The smallest seqno in this file
130 SequenceNumber largest_seqno; // The largest seqno in this file
131
132 FileDescriptor() : FileDescriptor(0, 0, 0) {}
133
134 FileDescriptor(uint64_t number, uint32_t path_id, uint64_t _file_size)
135 : FileDescriptor(number, path_id, _file_size, kMaxSequenceNumber, 0) {}
136
137 FileDescriptor(uint64_t number, uint32_t path_id, uint64_t _file_size,
138 SequenceNumber _smallest_seqno, SequenceNumber _largest_seqno)
139 : table_reader(nullptr),
140 packed_number_and_path_id(PackFileNumberAndPathId(number, path_id)),
141 file_size(_file_size),
142 smallest_seqno(_smallest_seqno),
143 largest_seqno(_largest_seqno) {}
144
145 FileDescriptor(const FileDescriptor& fd) { *this = fd; }
146
147 FileDescriptor& operator=(const FileDescriptor& fd) {
148 table_reader = fd.table_reader;
149 packed_number_and_path_id = fd.packed_number_and_path_id;
150 file_size = fd.file_size;
151 smallest_seqno = fd.smallest_seqno;
152 largest_seqno = fd.largest_seqno;
153 return *this;
154 }
155
156 uint64_t GetNumber() const {
157 return packed_number_and_path_id & kFileNumberMask;
158 }
159 uint32_t GetPathId() const {
160 return static_cast<uint32_t>(packed_number_and_path_id /
161 (kFileNumberMask + 1));
162 }
163 uint64_t GetFileSize() const { return file_size; }
164};

298 // Unlike UpdateBoundaries, ranges do not need to be presented in any
299 // particular order.
300 void UpdateBoundariesForRange(const InternalKey& start,
301 const InternalKey& end, SequenceNumber seqno,
302 const InternalKeyComparator& icmp) {
303 if (smallest.size() == 0 || icmp.Compare(start, smallest) < 0) {
304 smallest = start;
305 }
306 if (largest.size() == 0 || icmp.Compare(largest, end) < 0) {
307 largest = end;
308 }
309 assert(icmp.Compare(smallest, largest) <= 0);
310 fd.smallest_seqno = std::min(fd.smallest_seqno, seqno);
311 fd.largest_seqno = std::max(fd.largest_seqno, seqno);
312 }

314 // Try to get oldest ancester time from the class itself or table properties
315 // if table reader is already pinned.
316 // 0 means the information is not available.
317 uint64_t TryGetOldestAncesterTime() {
318 if (oldest_ancester_time != kUnknownOldestAncesterTime) {
319 return oldest_ancester_time;
320 } else if (fd.table_reader != nullptr &&
321 fd.table_reader->GetTableProperties() != nullptr) {
322 return fd.table_reader->GetTableProperties()->creation_time;
323 }
324 return kUnknownOldestAncesterTime;
325 }
326
327 uint64_t TryGetFileCreationTime() {
328 if (file_creation_time != kUnknownFileCreationTime) {
329 return file_creation_time;
330 } else if (fd.table_reader != nullptr &&
331 fd.table_reader->GetTableProperties() != nullptr) {
332 return fd.table_reader->GetTableProperties()->file_creation_time;
333 }
334 return kUnknownFileCreationTime;
335 }
336
337 // Tries to get the newest key time from the current file
338 // Falls back on oldest ancestor time of previous (newer) file
339 uint64_t TryGetNewestKeyTime(FileMetaData* prev_file = nullptr) {
340 if (fd.table_reader != nullptr &&
341 fd.table_reader->GetTableProperties() != nullptr) {
342 uint64_t newest_key_time =
343 fd.table_reader->GetTableProperties()->newest_key_time;
344 if (newest_key_time != kUnknownNewestKeyTime) {
345 return newest_key_time;
346 }
347 }
348 if (prev_file != nullptr) {
349 uint64_t prev_oldest_ancestor_time =
350 prev_file->TryGetOldestAncesterTime();
351 if (prev_oldest_ancestor_time != kUnknownOldestAncesterTime) {
352 return prev_oldest_ancestor_time;
353 }
354 }
355 return kUnknownNewestKeyTime;
356 }

358 // WARNING: manual update to this function is needed
359 // whenever a new string property is added to FileMetaData
360 // to reduce approximation error.
361 //
362 // TODO: eliminate the need of manually updating this function

364 size_t ApproximateMemoryUsage() const {
365 size_t usage = 0;
366#ifdef ROCKSDB_MALLOC_USABLE_SIZE
367 usage += malloc_usable_size(const_cast<FileMetaData*>(this));
