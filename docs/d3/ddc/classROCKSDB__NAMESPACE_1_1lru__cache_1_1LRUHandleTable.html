83 {
84 if (length_bits_ >= max_length_bits_) {
85 // Due to reaching limit of hash information, if we made the table bigger,
86 // we would allocate more addresses but only the same number would be used.
87 return;
88 }
89 if (length_bits_ >= 31) {
90 // Avoid undefined behavior shifting uint32_t by 32.
91 return;
92 }
93
94 uint32_t old_length = uint32_t{1} << length_bits_;
95 int new_length_bits = length_bits_ + 1;
96 std::unique_ptr<LRUHandle* []> new_list {
97 new LRUHandle* [size_t{1} << new_length_bits] {}
98 };
99 [[maybe_unused]] uint32_t count = 0;
100 for (uint32_t i = 0; i < old_length; i++) {
101 LRUHandle* h = list_[i];
102 while (h != nullptr) {
103 LRUHandle* next = h->next_hash;
104 uint32_t hash = h->hash;
105 LRUHandle** ptr = &new_list[hash >> (32 - new_length_bits)];
106 h->next_hash = *ptr;
107 *ptr = h;
108 h = next;
109 count++;
110 }
111 }
112 assert(elems_ == count);
113 list_ = std::move(new_list);
114 length_bits_ = new_length_bits;
115}
