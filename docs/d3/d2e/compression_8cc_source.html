43int ZSTDStreamingCompress::Compress(const char* input, size_t input_size,
44 char* output, size_t* output_pos) {
45 assert(input != nullptr && output != nullptr && output_pos != nullptr);
46 *output_pos = 0;
47 // Don't need to compress an empty input
48 if (input_size == 0) {
49 return 0;
50 }
51#ifndef ZSTD_ADVANCED
52 (void)input;
53 (void)input_size;
54 (void)output;
55 return -1;
56#else
57 if (input_buffer_.src == nullptr || input_buffer_.src != input) {
58 // New input
59 // Catch errors where the previous input was not fully decompressed.
60 assert(input_buffer_.pos == input_buffer_.size);
61 input_buffer_ = {input, input_size, /*pos=*/0};
62 } else if (input_buffer_.src == input) {
63 // Same input, not fully compressed.
64 }
65 ZSTD_outBuffer output_buffer = {output, max_output_len_, /*pos=*/0};
66 const size_t remaining =
67 ZSTD_compressStream2(cctx_, &output_buffer, &input_buffer_, ZSTD_e_end);
68 if (ZSTD_isError(remaining)) {
69 // Failure
70 Reset();
71 return -1;
72 }
73 // Success
74 *output_pos = output_buffer.pos;
75 return (int)remaining;
76#endif
77}

86int ZSTDStreamingUncompress::Uncompress(const char* input, size_t input_size,
87 char* output, size_t* output_pos) {
88 assert(output != nullptr && output_pos != nullptr);
89 *output_pos = 0;
90 // Don't need to uncompress an empty input
91 if (input_size == 0) {
92 return 0;
93 }
94#ifdef ZSTD_ADVANCED
95 if (input) {
96 // New input
97 input_buffer_ = {input, input_size, /*pos=*/0};
98 }
99 ZSTD_outBuffer output_buffer = {output, max_output_len_, /*pos=*/0};
100 size_t ret = ZSTD_decompressStream(dctx_, &output_buffer, &input_buffer_);
101 if (ZSTD_isError(ret)) {
102 Reset();
103 return -1;
104 }
105 *output_pos = output_buffer.pos;
106 return (int)(input_buffer_.size - input_buffer_.pos);
107#else
108 (void)input;
109 (void)input_size;
110 (void)output;
111 return -1;
112#endif
113}
