16enum class RecoveryType {
17 kNoop,
18 kUnrecoverable,
19 kStripTimestamp,
20 kPadTimestamp,
21};

23RecoveryType GetRecoveryType(const size_t running_ts_sz,
24 const std::optional<size_t>& recorded_ts_sz) {
25 if (running_ts_sz == 0) {
26 if (!recorded_ts_sz.has_value()) {
27 // A column family id not recorded is equivalent to that column family has
28 // zero timestamp size.
29 return RecoveryType::kNoop;
30 }
31 return RecoveryType::kStripTimestamp;
32 }
33
34 assert(running_ts_sz != 0);
35
36 if (!recorded_ts_sz.has_value()) {
37 return RecoveryType::kPadTimestamp;
38 }
39
40 if (running_ts_sz != *recorded_ts_sz) {
41 return RecoveryType::kUnrecoverable;
42 }
43
44 return RecoveryType::kNoop;
45}

47bool AllRunningColumnFamiliesConsistent(
48 const UnorderedMap<uint32_t, size_t>& running_ts_sz,
49 const UnorderedMap<uint32_t, size_t>& record_ts_sz) {
50 for (const auto& [cf_id, ts_sz] : running_ts_sz) {
51 auto record_it = record_ts_sz.find(cf_id);
52 RecoveryType recovery_type =
53 GetRecoveryType(ts_sz, record_it != record_ts_sz.end()
54 ? std::optional<size_t>(record_it->second)
55 : std::nullopt);
56 if (recovery_type != RecoveryType::kNoop) {
57 return false;
58 }
59 }
60 return true;
61}

63Status CheckWriteBatchTimestampSizeConsistency(
64 const WriteBatch* batch,
65 const UnorderedMap<uint32_t, size_t>& running_ts_sz,
66 const UnorderedMap<uint32_t, size_t>& record_ts_sz,
67 TimestampSizeConsistencyMode check_mode, bool* ts_need_recovery) {
68 std::vector<uint32_t> column_family_ids;
69 Status status =
70 CollectColumnFamilyIdsFromWriteBatch(*batch, &column_family_ids);
71 if (!status.ok()) {
72 return status;
73 }
74 for (const auto& cf_id : column_family_ids) {
75 auto running_iter = running_ts_sz.find(cf_id);
76 if (running_iter == running_ts_sz.end()) {
77 // Ignore dropped column family referred to in a WriteBatch regardless of
78 // its consistency.
79 continue;
80 }
81 auto record_iter = record_ts_sz.find(cf_id);
82 RecoveryType recovery_type = GetRecoveryType(
83 running_iter->second, record_iter != record_ts_sz.end()
84 ? std::optional<size_t>(record_iter->second)
85 : std::nullopt);
86 if (recovery_type != RecoveryType::kNoop) {
87 if (check_mode == TimestampSizeConsistencyMode::kVerifyConsistency) {
88 return Status::InvalidArgument(
89 "WriteBatch contains timestamp size inconsistency.");
90 }
91
92 if (recovery_type == RecoveryType::kUnrecoverable) {
93 return Status::InvalidArgument(
94 "WriteBatch contains unrecoverable timestamp size inconsistency.");
95 }
96
97 // If any column family needs reconciliation, it will mark the whole
98 // WriteBatch to need recovery and rebuilt.
99 *ts_need_recovery = true;
100 }
101 }
102 return Status::OK();
103}

105enum class ToggleUDT {
106 kUnchanged,
107 kEnableUDT,
108 kDisableUDT,
109 kInvalidChange,

112ToggleUDT CompareComparator(const Comparator* new_comparator,
113 const std::string& old_comparator_name) {
114 static const char* kUDTSuffix = ".u64ts";
115 static const Slice kSuffixSlice = kUDTSuffix;
116 static const size_t kSuffixSize = 6;
117 size_t ts_sz = new_comparator->timestamp_size();
118 (void)ts_sz;
119 Slice new_ucmp_name(new_comparator->Name());
120 Slice old_ucmp_name(old_comparator_name);
121 if (new_ucmp_name.compare(old_ucmp_name) == 0) {
122 return ToggleUDT::kUnchanged;
123 }
124 if (new_ucmp_name.size() == old_ucmp_name.size() + kSuffixSize &&
125 new_ucmp_name.starts_with(old_ucmp_name) &&
126 new_ucmp_name.ends_with(kSuffixSlice)) {
127 assert(ts_sz == 8);
128 return ToggleUDT::kEnableUDT;
129 }
130 if (old_ucmp_name.size() == new_ucmp_name.size() + kSuffixSize &&
131 old_ucmp_name.starts_with(new_ucmp_name) &&
132 old_ucmp_name.ends_with(kSuffixSlice)) {
133 assert(ts_sz == 0);
134 return ToggleUDT::kDisableUDT;
135 }
136 return ToggleUDT::kInvalidChange;
137}

306Status TimestampRecoveryHandler::ReconcileTimestampDiscrepancy(
307 uint32_t cf, const Slice& key, std::string* new_key_buf, Slice* new_key) {
308 assert(handler_valid_);
309 auto running_iter = running_ts_sz_.find(cf);
310 if (running_iter == running_ts_sz_.end()) {
311 // The column family referred to by the WriteBatch is no longer running.
312 // Copy over the entry as is to the new WriteBatch.
313 *new_key = key;
314 return Status::OK();
315 }
316 size_t running_ts_sz = running_iter->second;
317 auto record_iter = record_ts_sz_.find(cf);
318 std::optional<size_t> record_ts_sz =
319 record_iter != record_ts_sz_.end()
320 ? std::optional<size_t>(record_iter->second)
321 : std::nullopt;
322 RecoveryType recovery_type = GetRecoveryType(running_ts_sz, record_ts_sz);
323
324 switch (recovery_type) {
325 case RecoveryType::kNoop:
326 *new_key = key;
327 break;
328 case RecoveryType::kStripTimestamp:
329 assert(record_ts_sz.has_value());
330 *new_key = StripTimestampFromUserKey(key, *record_ts_sz);
331 new_batch_diff_from_orig_batch_ = true;
332 break;
333 case RecoveryType::kPadTimestamp:
334 AppendKeyWithMinTimestamp(new_key_buf, key, running_ts_sz);
335 *new_key = *new_key_buf;
336 new_batch_diff_from_orig_batch_ = true;
337 break;
338 case RecoveryType::kUnrecoverable:
339 return Status::InvalidArgument(
340 "Unrecoverable timestamp size inconsistency encountered by "
341 "TimestampRecoveryHandler.");
342 default:
343 assert(false);
344 }
345 return Status::OK();
346}

348Status HandleWriteBatchTimestampSizeDifference(
349 const WriteBatch* batch,
350 const UnorderedMap<uint32_t, size_t>& running_ts_sz,
351 const UnorderedMap<uint32_t, size_t>& record_ts_sz,
352 TimestampSizeConsistencyMode check_mode, bool seq_per_batch,
353 bool batch_per_txn, std::unique_ptr<WriteBatch>* new_batch) {
354 // Quick path to bypass checking the WriteBatch.
355 if (AllRunningColumnFamiliesConsistent(running_ts_sz, record_ts_sz)) {
356 return Status::OK();
357 }
358 bool need_recovery = false;
359 Status status = CheckWriteBatchTimestampSizeConsistency(
360 batch, running_ts_sz, record_ts_sz, check_mode, &need_recovery);
361 if (!status.ok()) {
362 return status;
363 } else if (need_recovery) {
364 assert(new_batch);
365 SequenceNumber sequence = WriteBatchInternal::Sequence(batch);
366 TimestampRecoveryHandler recovery_handler(running_ts_sz, record_ts_sz,
367 seq_per_batch, batch_per_txn);
368 status = batch->Iterate(&recovery_handler);
369 if (!status.ok()) {
370 return status;
371 } else {
372 *new_batch = recovery_handler.TransferNewBatch();
373 WriteBatchInternal::SetSequence(new_batch->get(), sequence);
374 }
375 }
376 return Status::OK();
377}

379Status ValidateUserDefinedTimestampsOptions(
380 const Comparator* new_comparator, const std::string& old_comparator_name,
381 bool new_persist_udt, bool old_persist_udt,
382 bool* mark_sst_files_has_no_udt) {
383 size_t ts_sz = new_comparator->timestamp_size();
384 ToggleUDT res = CompareComparator(new_comparator, old_comparator_name);
385 switch (res) {
386 case ToggleUDT::kUnchanged:
387 if (old_persist_udt == new_persist_udt) {
388 return Status::OK();
389 }
390 if (ts_sz == 0) {
391 return Status::OK();
392 }
393 return Status::InvalidArgument(
394 "Cannot toggle the persist_user_defined_timestamps flag for a column "
395 "family with user-defined timestamps feature enabled.");
396 case ToggleUDT::kEnableUDT:
397 if (!new_persist_udt) {
398 *mark_sst_files_has_no_udt = true;
399 return Status::OK();
400 }
401 return Status::InvalidArgument(
402 "Cannot open a column family and enable user-defined timestamps "
403 "feature without setting persist_user_defined_timestamps flag to "
404 "false.");
405 case ToggleUDT::kDisableUDT:
406 if (!old_persist_udt) {
407 return Status::OK();
408 }
409 return Status::InvalidArgument(
410 "Cannot open a column family and disable user-defined timestamps "
411 "feature if its existing persist_user_defined_timestamps flag is not "
412 "false.");
413 case ToggleUDT::kInvalidChange:
414 return Status::InvalidArgument(
415 new_comparator->Name(),
416 "does not match existing comparator " + old_comparator_name);
417 default:
418 break;
419 }
420 return Status::InvalidArgument(
421 "Unsupported user defined timestamps settings change.");
422}

432std::tuple<std::optional<Slice>, std::optional<Slice>>
433MaybeAddTimestampsToRange(const Slice* start, const Slice* end, size_t ts_sz,
434 std::string* start_with_ts, std::string* end_with_ts,
435 bool exclusive_end) {
436 std::optional<Slice> ret_start, ret_end;
437 if (start) {
438 if (ts_sz == 0) {
439 ret_start = *start;
440 } else {
441 // Maximum timestamp means including all keys with any timestamp for start
442 AppendKeyWithMaxTimestamp(start_with_ts, *start, ts_sz);
443 ret_start = Slice(*start_with_ts);
444 }
445 }
446 if (end) {
447 if (ts_sz == 0) {
448 ret_end = *end;
449 } else {
450 if (exclusive_end) {
451 // Append a maximum timestamp as the range limit is exclusive:
452 // [start, end)
453 AppendKeyWithMaxTimestamp(end_with_ts, *end, ts_sz);
454 } else {
455 // Append a minimum timestamp to end so the range limit is inclusive:
456 // [start, end]
457 AppendKeyWithMinTimestamp(end_with_ts, *end, ts_sz);
458 }
459 ret_end = Slice(*end_with_ts);
460 }
461 }
462 return std::make_tuple(ret_start, ret_end);
463}
