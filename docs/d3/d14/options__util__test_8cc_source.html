45TEST_F(OptionsUtilTest, SaveAndLoad) {
46 const size_t kCFCount = 5;
47
48 DBOptions db_opt;
49 std::vector<std::string> cf_names;
50 std::vector<ColumnFamilyOptions> cf_opts;

57 }
58

63 DBOptions loaded_db_opt;
64 std::vector<ColumnFamilyDescriptor> loaded_cf_descs;
65 ConfigOptions config_options;
66 config_options.ignore_unknown_options = false;
67 config_options.input_strings_escaped = true;
68 config_options.env = env_.get();

70 &loaded_cf_descs));
71 ConfigOptions exact;
72 exact.sanity_level = ConfigOptions::kSanityLevelExactMatch;
73 ASSERT_OK(
74 RocksDBOptionsParser::VerifyDBOptions(exact, db_opt, loaded_db_opt));
75 test::RandomInitDBOptions(&db_opt, &rnd_);
76 ASSERT_NOK(
77 RocksDBOptionsParser::VerifyDBOptions(exact, db_opt, loaded_db_opt));
78
79 for (size_t i = 0; i < kCFCount; ++i) {
80 ASSERT_EQ(cf_names[i], loaded_cf_descs[i].name);
81 ASSERT_OK(RocksDBOptionsParser::VerifyCFOptions(
82 exact, cf_opts[i], loaded_cf_descs[i].options));
83 ASSERT_OK(RocksDBOptionsParser::VerifyTableFactory(
84 exact, cf_opts[i].table_factory.get(),
85 loaded_cf_descs[i].options.table_factory.get()));
86 test::RandomInitCFOptions(&cf_opts[i], db_opt, &rnd_);
87 ASSERT_NOK(RocksDBOptionsParser::VerifyCFOptions(
88 exact, cf_opts[i], loaded_cf_descs[i].options));
89 }
90
91 ASSERT_OK(DestroyDB(dbname_, Options(db_opt, cf_opts[0])));
92 for (size_t i = 0; i < kCFCount; ++i) {
93 if (cf_opts[i].compaction_filter) {
94 delete cf_opts[i].compaction_filter;
95 }
96 }

226TEST_F(OptionsUtilTest, SanityCheck) {
227 DBOptions db_opt;
228 std::vector<ColumnFamilyDescriptor> cf_descs;
229 const size_t kCFCount = 5;
230 for (size_t i = 0; i < kCFCount; ++i) {
231 cf_descs.emplace_back();

233 (i == 0) ? kDefaultColumnFamilyName : test::RandomName(&rnd_, 10);
234
235 cf_descs.back().options.table_factory.reset(NewBlockBasedTableFactory());
236 // Assign non-null values to prefix_extractors except the first cf.
237 cf_descs.back().options.prefix_extractor.reset(
238 i != 0 ? test::RandomSliceTransform(&rnd_) : nullptr);
239 cf_descs.back().options.merge_operator.reset(
240 test::RandomMergeOperator(&rnd_));
241 }
242
243 db_opt.create_missing_column_families = true;
244 db_opt.create_if_missing = true;
245
246 ASSERT_OK(DestroyDB(dbname_, Options(db_opt, cf_descs[0].options)));
247 DB* db;
248 std::vector<ColumnFamilyHandle*> handles;
249 // open and persist the options
250 ASSERT_OK(DB::Open(db_opt, dbname_, cf_descs, &handles, &db));
251
252 // close the db
253 for (auto* handle : handles) {
254 delete handle;
255 }
256 delete db;
257
258 ConfigOptions config_options;
259 config_options.ignore_unknown_options = false;
260 config_options.input_strings_escaped = true;
261 config_options.sanity_level = ConfigOptions::kSanityLevelLooselyCompatible;
262 // perform sanity check
263 ASSERT_OK(
264 CheckOptionsCompatibility(config_options, dbname_, db_opt, cf_descs));
265
266 ASSERT_GE(kCFCount, 5);
267 // merge operator
268 {

270 cf_descs[0].options.merge_operator;

283 CheckOptionsCompatibility(config_options, dbname_, db_opt, cf_descs));
284 }
285
286 // prefix extractor

289 cf_descs[1].options.prefix_extractor;
290
291 // It's okay to set prefix_extractor to nullptr.
292 ASSERT_NE(prefix_extractor, nullptr);
293 cf_descs[1].options.prefix_extractor.reset();

295 CheckOptionsCompatibility(config_options, dbname_, db_opt, cf_descs));

310

431TEST_F(OptionsUtilTest, LoadLatestOptions) {
432 Options options;
433 options.OptimizeForSmallDb();
434 ColumnFamilyDescriptor cf_desc;
435 ConfigOptions config_opts;
436 DBOptions db_opts;
437 std::vector<ColumnFamilyDescriptor> cf_descs;
438 std::vector<ColumnFamilyHandle*> handles;
439 DB* db;
440 options.create_if_missing = true;
441
442 ASSERT_OK(DestroyDB(dbname_, options));
443
444 cf_descs.emplace_back();
445 cf_descs.back().name = kDefaultColumnFamilyName;
446 cf_descs.back().options.table_factory.reset(NewBlockBasedTableFactory());
447 cf_descs.emplace_back();
448 cf_descs.back().name = "Plain";
449 cf_descs.back().options.table_factory.reset(NewPlainTableFactory());
450 db_opts.create_missing_column_families = true;
451 db_opts.create_if_missing = true;
452
453 // open and persist the options
454 ASSERT_OK(DB::Open(db_opts, dbname_, cf_descs, &handles, &db));
455
456 std::string options_file_name;
457 std::string new_options_file;
458
459 ASSERT_OK(GetLatestOptionsFileName(dbname_, options.env, &options_file_name));
460 ASSERT_OK(LoadLatestOptions(config_opts, dbname_, &db_opts, &cf_descs));
461 ASSERT_EQ(cf_descs.size(), 2U);
462 ASSERT_OK(RocksDBOptionsParser::VerifyDBOptions(config_opts,
463 db->GetDBOptions(), db_opts));
464 ASSERT_OK(handles[0]->GetDescriptor(&cf_desc));
465 ASSERT_OK(RocksDBOptionsParser::VerifyCFOptions(config_opts, cf_desc.options,
466 cf_descs[0].options));
467 ASSERT_OK(handles[1]->GetDescriptor(&cf_desc));
468 ASSERT_OK(RocksDBOptionsParser::VerifyCFOptions(config_opts, cf_desc.options,
469 cf_descs[1].options));
470
471 // Now change some of the DBOptions
472 ASSERT_OK(db->SetDBOptions(
473 {{"delayed_write_rate", "1234"}, {"bytes_per_sync", "32768"}}));
474 ASSERT_OK(GetLatestOptionsFileName(dbname_, options.env, &new_options_file));
475 ASSERT_NE(options_file_name, new_options_file);
476 ASSERT_OK(LoadLatestOptions(config_opts, dbname_, &db_opts, &cf_descs));
477 ASSERT_OK(RocksDBOptionsParser::VerifyDBOptions(config_opts,
478 db->GetDBOptions(), db_opts));
479 options_file_name = new_options_file;
480
481 // Now change some of the ColumnFamilyOptions
482 ASSERT_OK(db->SetOptions(handles[1], {{"write_buffer_size", "32768"}}));
483 ASSERT_OK(GetLatestOptionsFileName(dbname_, options.env, &new_options_file));
484 ASSERT_NE(options_file_name, new_options_file);
485 ASSERT_OK(LoadLatestOptions(config_opts, dbname_, &db_opts, &cf_descs));
486 ASSERT_OK(RocksDBOptionsParser::VerifyDBOptions(config_opts,
487 db->GetDBOptions(), db_opts));
488 ASSERT_OK(handles[0]->GetDescriptor(&cf_desc));
489 ASSERT_OK(RocksDBOptionsParser::VerifyCFOptions(config_opts, cf_desc.options,
490 cf_descs[0].options));
491 ASSERT_OK(handles[1]->GetDescriptor(&cf_desc));
492 ASSERT_OK(RocksDBOptionsParser::VerifyCFOptions(config_opts, cf_desc.options,
493 cf_descs[1].options));
494
495 // close the db
496 for (auto* handle : handles) {
497 delete handle;
498 }
499 delete db;
500 ASSERT_OK(DestroyDB(dbname_, options, cf_descs));
501}

503static void WriteOptionsFile(Env* env, const std::string& path,
504 const std::string& options_file, int major,
505 int minor, const std::string& db_opts,
506 const std::string& cf_opts,
507 const std::string& bbt_opts = "") {
508 std::string options_file_header =
509 "\n"
510 "[Version]\n"
511 " rocksdb_version=" +
512 std::to_string(major) + "." + std::to_string(minor) +
513 ".0\n"
514 " options_file_version=1\n";
515
516 std::unique_ptr<WritableFile> wf;
517 ASSERT_OK(env->NewWritableFile(path + "/" + options_file, &wf, EnvOptions()));
518 ASSERT_OK(
519 wf->Append(options_file_header + "[ DBOptions ]\n" + db_opts + "\n"));
520 ASSERT_OK(wf->Append(
521 "[CFOptions \"default\"] # column family must be specified\n" +
522 cf_opts + "\n"));
523 ASSERT_OK(wf->Append("[TableOptions/BlockBasedTable \"default\"]\n" +
524 bbt_opts + "\n"));
525 ASSERT_OK(wf->Close());
526
527 std::string latest_options_file;
528 ASSERT_OK(GetLatestOptionsFileName(path, env, &latest_options_file));
529 ASSERT_EQ(latest_options_file, options_file);
530}
