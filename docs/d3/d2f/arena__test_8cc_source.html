30bool CheckMemoryAllocated(size_t allocated, size_t expected) {
31 // The value returned by Arena::MemoryAllocatedBytes() may be greater than
32 // the requested memory. We choose a somewhat arbitrary upper bound of
33 // max_expected = expected * 1.1 to detect critical overallocation.
34 size_t max_expected = expected + expected / 10;
35 return allocated >= expected && allocated <= max_expected;
36}

38void MemoryAllocatedBytesTest(size_t huge_page_size) {
39 const int N = 17;
40 size_t req_sz; // requested size
41 size_t bsz = 32 * 1024; // block size
42 size_t expected_memory_allocated;
43
44 Arena arena(bsz, nullptr, huge_page_size);
45
46 // requested size > quarter of a block:
47 // allocate requested size separately
48 req_sz = 12 * 1024;
49 for (int i = 0; i < N; i++) {
50 arena.Allocate(req_sz);
51 }
52 expected_memory_allocated = req_sz * N + Arena::kInlineSize;
53 ASSERT_PRED2(CheckMemoryAllocated, arena.MemoryAllocatedBytes(),
54 expected_memory_allocated);
55
56 arena.Allocate(Arena::kInlineSize - 1);
57
58 // requested size < quarter of a block:
59 // allocate a block with the default size, then try to use unused part
60 // of the block. So one new block will be allocated for the first
61 // Allocate(99) call. All the remaining calls won't lead to new allocation.
62 req_sz = 99;
63 for (int i = 0; i < N; i++) {
64 arena.Allocate(req_sz);
65 }
66 if (huge_page_size) {
67 ASSERT_TRUE(
68 CheckMemoryAllocated(arena.MemoryAllocatedBytes(),
69 expected_memory_allocated + bsz) ||
70 CheckMemoryAllocated(arena.MemoryAllocatedBytes(),
71 expected_memory_allocated + huge_page_size));
72 } else {
73 expected_memory_allocated += bsz;
74 ASSERT_PRED2(CheckMemoryAllocated, arena.MemoryAllocatedBytes(),
75 expected_memory_allocated);
76 }
77
78 // requested size > size of a block:
79 // allocate requested size separately
80 expected_memory_allocated = arena.MemoryAllocatedBytes();
81 req_sz = 8 * 1024 * 1024;
82 for (int i = 0; i < N; i++) {
83 arena.Allocate(req_sz);
84 }
85 expected_memory_allocated += req_sz * N;
86 ASSERT_PRED2(CheckMemoryAllocated, arena.MemoryAllocatedBytes(),
87 expected_memory_allocated);
88}

90// Make sure we didn't count the allocate but not used memory space in
91// Arena::ApproximateMemoryUsage()
92static void ApproximateMemoryUsageTest(size_t huge_page_size) {
93 const size_t kBlockSize = 4096;
94 const size_t kEntrySize = kBlockSize / 8;
95 const size_t kZero = 0;
96 Arena arena(kBlockSize, nullptr, huge_page_size);
97 ASSERT_EQ(kZero, arena.ApproximateMemoryUsage());
98
99 // allocate inline bytes
100 const size_t kAlignUnit = alignof(max_align_t);
101 EXPECT_TRUE(arena.IsInInlineBlock());
102 arena.AllocateAligned(kAlignUnit);
103 EXPECT_TRUE(arena.IsInInlineBlock());
104 arena.AllocateAligned(Arena::kInlineSize / 2 - (2 * kAlignUnit));
105 EXPECT_TRUE(arena.IsInInlineBlock());
106 arena.AllocateAligned(Arena::kInlineSize / 2);
107 EXPECT_TRUE(arena.IsInInlineBlock());
108 ASSERT_EQ(arena.ApproximateMemoryUsage(), Arena::kInlineSize - kAlignUnit);
109 ASSERT_PRED2(CheckMemoryAllocated, arena.MemoryAllocatedBytes(),
110 Arena::kInlineSize);
111
112 auto num_blocks = kBlockSize / kEntrySize;
113
114 // first allocation
115 arena.AllocateAligned(kEntrySize);
116 EXPECT_FALSE(arena.IsInInlineBlock());
117 auto mem_usage = arena.MemoryAllocatedBytes();
118 if (huge_page_size) {
119 ASSERT_TRUE(
120 CheckMemoryAllocated(mem_usage, kBlockSize + Arena::kInlineSize) ||
121 CheckMemoryAllocated(mem_usage, huge_page_size + Arena::kInlineSize));
122 } else {
123 ASSERT_PRED2(CheckMemoryAllocated, mem_usage,
124 kBlockSize + Arena::kInlineSize);
125 }
126 auto usage = arena.ApproximateMemoryUsage();
127 ASSERT_LT(usage, mem_usage);
128 for (size_t i = 1; i < num_blocks; ++i) {
129 arena.AllocateAligned(kEntrySize);
130 ASSERT_EQ(mem_usage, arena.MemoryAllocatedBytes());
131 ASSERT_EQ(arena.ApproximateMemoryUsage(), usage + kEntrySize);
132 EXPECT_FALSE(arena.IsInInlineBlock());
133 usage = arena.ApproximateMemoryUsage();
134 }
135 if (huge_page_size) {
136 ASSERT_TRUE(usage > mem_usage ||
137 usage + huge_page_size - kBlockSize == mem_usage);
138 } else {
139 ASSERT_GT(usage, mem_usage);
140 }
141}

143static void SimpleTest(size_t huge_page_size) {
144 std::vector<std::pair<size_t, char*>> allocated;
145 Arena arena(Arena::kMinBlockSize, nullptr, huge_page_size);
146 const int N = 100000;
147 size_t bytes = 0;
148 Random rnd(301);
149 for (int i = 0; i < N; i++) {
150 size_t s;
151 if (i % (N / 10) == 0) {
152 s = i;
153 } else {
154 s = rnd.OneIn(4000)
155 ? rnd.Uniform(6000)
156 : (rnd.OneIn(10) ? rnd.Uniform(100) : rnd.Uniform(20));
157 }
158 if (s == 0) {
159 // Our arena disallows size 0 allocations.
160 s = 1;
161 }
162 char* r;
163 if (rnd.OneIn(10)) {
164 r = arena.AllocateAligned(s);
165 } else {
166 r = arena.Allocate(s);
167 }
168
169 for (unsigned int b = 0; b < s; b++) {
170 // Fill the "i"th allocation with a known bit pattern
171 r[b] = i % 256;
172 }
173 bytes += s;
174 allocated.emplace_back(s, r);
175 ASSERT_GE(arena.ApproximateMemoryUsage(), bytes);
176 if (i > N / 10) {
177 ASSERT_LE(arena.ApproximateMemoryUsage(), bytes * 1.10);
178 }
179 }
180 for (unsigned int i = 0; i < allocated.size(); i++) {
181 size_t num_bytes = allocated[i].first;
182 const char* p = allocated[i].second;
183 for (unsigned int b = 0; b < num_bytes; b++) {
184 // Check the "i"th allocation for the known bit pattern
185 ASSERT_EQ(int(p[b]) & 0xff, (int)(i % 256));
186 }
187 }
188}

206// Number of minor page faults since last call
207size_t PopMinorPageFaultCount() {
208#ifdef RUSAGE_SELF
209 static long prev = 0;
210 struct rusage usage;
211 EXPECT_EQ(getrusage(RUSAGE_SELF, &usage), 0);
212 size_t rv = usage.ru_minflt - prev;
213 prev = usage.ru_minflt;
214 return rv;
215#else
216 // Conservative
217 return SIZE_MAX;
218#endif // RUSAGE_SELF
219}

221TEST(MmapTest, AllocateLazyZeroed) {
222 // Doesn't have to be page aligned
223 constexpr size_t len = 1234567; // in bytes
224 constexpr size_t count = len / 8; // in uint64_t objects
225 // Implicit conversion move
226 TypedMemMapping<uint64_t> pre_arr = MemMapping::AllocateLazyZeroed(len);
227 // Move from same type
228 TypedMemMapping<uint64_t> arr = std::move(pre_arr);
229
230 ASSERT_NE(arr.Get(), nullptr);
231 ASSERT_EQ(arr.Get(), &arr[0]);
232 ASSERT_EQ(arr.Get(), arr.MemMapping::Get());
233
234 ASSERT_EQ(arr.Length(), len);
235 ASSERT_EQ(arr.Count(), count);
236
237 // Start counting page faults
238 PopMinorPageFaultCount();
239
240 // Access half of the allocation
241 size_t i = 0;
242 for (; i < count / 2; ++i) {
243 ASSERT_EQ(arr[i], 0);
244 arr[i] = i;
245 }
246
247 // Appropriate page faults (maybe more)
248 size_t faults = PopMinorPageFaultCount();
249 ASSERT_GE(faults, len / 2 / port::kPageSize);
250
251 // Access rest of the allocation
252 for (; i < count; ++i) {
253 ASSERT_EQ(arr[i], 0);
254 arr[i] = i;
255 }
256
257 // Appropriate page faults (maybe more)
258 faults = PopMinorPageFaultCount();
259 ASSERT_GE(faults, len / 2 / port::kPageSize);
260
261 // Verify data
262 for (i = 0; i < count; ++i) {
263 ASSERT_EQ(arr[i], i);
264 }
265}

267TEST_F(ArenaTest, UnmappedAllocation) {
268 // Verify that it's possible to get unmapped pages in large allocations,
269 // for memory efficiency and to ensure we don't accidentally waste time &
270 // space initializing the memory.
271
272#ifdef ROCKSDB_JEMALLOC
273 // With Jemalloc config.fill, the pages are written to before we get them
274 uint8_t fill = 0;
275 size_t fill_sz = sizeof(fill);
276 mallctl("config.fill", &fill, &fill_sz, nullptr, 0);
277 if (fill) {
278 ROCKSDB_GTEST_BYPASS("Test skipped because of config.fill==true");
279 return;
280 }
281#endif // ROCKSDB_JEMALLOC
282
283 // This block size value is smaller than the smallest x86 huge page size,
284 // so should not be fulfilled by a transparent huge page mapping.
285 constexpr size_t kBlockSize = 1U << 20;
286 Arena arena(kBlockSize);
287
288 // The allocator might give us back recycled memory for a while, but
289 // shouldn't last forever.
290 for (int i = 0;; ++i) {
291 char* p = arena.Allocate(kBlockSize);
292
293 // Start counting page faults
294 PopMinorPageFaultCount();
295
296 // Overwrite the whole allocation
297 for (size_t j = 0; j < kBlockSize; ++j) {
298 p[j] = static_cast<char>(j & 255);
299 }
300
301 size_t faults = PopMinorPageFaultCount();
302 if (faults >= kBlockSize * 3 / 4 / port::kPageSize) {
303 // Most of the access generated page faults => GOOD
304 break;
305 }
306 // Should have succeeded after enough tries
307 ASSERT_LT(i, 1000);
308 }
309}
