24// Construct a string of the specified length made out of the supplied
25// partial string.
26static std::string BigString(const std::string& partial_string, size_t n) {
27 std::string result;
28 while (result.size() < n) {
29 result.append(partial_string);
30 }
31 result.resize(n);
32 return result;
33}

35// Construct a string from a number
36static std::string NumberString(int n) {
37 char buf[50];
38 snprintf(buf, sizeof(buf), "%d.", n);
39 return std::string(buf);
40}

42// Return a skewed potentially long string
43static std::string RandomSkewedString(int i, Random* rnd) {
44 return BigString(NumberString(i), rnd->Skewed(17));
45}

54 class StringSource : public FSSequentialFile {
55 public:
56 Slice& contents_;
57 bool force_error_;
58 size_t force_error_position_;
59 bool force_eof_;
60 size_t force_eof_position_;
61 bool returned_partial_;
62 bool fail_after_read_partial_;
63 explicit StringSource(Slice& contents, bool fail_after_read_partial)
64 : contents_(contents),
65 force_error_(false),
66 force_error_position_(0),
67 force_eof_(false),
68 force_eof_position_(0),
69 returned_partial_(false),
70 fail_after_read_partial_(fail_after_read_partial) {}
71
72 IOStatus Read(size_t n, const IOOptions& /*opts*/, Slice* result,
73 char* scratch, IODebugContext* /*dbg*/) override {
74 if (fail_after_read_partial_) {
75 EXPECT_TRUE(!returned_partial_) << "must not Read() after eof/error";
76 }
77
78 if (force_error_) {
79 if (force_error_position_ >= n) {
80 force_error_position_ -= n;
81 } else {
82 *result = Slice(contents_.data(), force_error_position_);
83 contents_.remove_prefix(force_error_position_);
84 force_error_ = false;
85 returned_partial_ = true;
86 return IOStatus::Corruption("read error");
87 }
88 }
89
90 if (contents_.size() < n) {
91 n = contents_.size();
92 returned_partial_ = true;
93 }
94
95 if (force_eof_) {
96 if (force_eof_position_ >= n) {
97 force_eof_position_ -= n;
98 } else {
99 force_eof_ = false;
100 n = force_eof_position_;
101 returned_partial_ = true;
102 }
103 }
104
105 // By using scratch we ensure that caller has control over the
106 // lifetime of result.data()
107 memcpy(scratch, contents_.data(), n);
108 *result = Slice(scratch, n);
109
110 contents_.remove_prefix(n);
111 return IOStatus::OK();
112 }
113
114 IOStatus Skip(uint64_t n) override {
115 if (n > contents_.size()) {
116 contents_.clear();
117 return IOStatus::NotFound("in-memory file skipepd past end");
118 }
119
120 contents_.remove_prefix(n);
121
122 return IOStatus::OK();
123 }
124 };

194
195 std::string Read(const WALRecoveryMode wal_recovery_mode =
196 WALRecoveryMode::kTolerateCorruptedTailRecords,
197 UnorderedMap<uint32_t, size_t>* cf_to_ts_sz = nullptr) {
198 std::string scratch;
199 Slice record;
200 bool ret = false;
201 uint64_t record_checksum;
202 ret = reader_->ReadRecord(&record, &scratch, wal_recovery_mode,
203 &record_checksum);
204 if (cf_to_ts_sz != nullptr) {
205 *cf_to_ts_sz = reader_->GetRecordedTimestampSize();
206 }
207 if (ret) {
208 if (!allow_retry_read_) {
209 // allow_retry_read_ means using FragmentBufferedReader which does not
210 // support record checksum yet.
211 uint64_t actual_record_checksum =
212 XXH3_64bits(record.data(), record.size());
213 assert(actual_record_checksum == record_checksum);
214 }
215 return record.ToString();
216 } else {
217 return "EOF";
218 }
219 }

231 void FixChecksum(int header_offset, int len, bool recyclable) {
232 // Compute crc of type/len/data
233 int header_size = recyclable ? kRecyclableHeaderSize : kHeaderSize;
234 uint32_t crc = crc32c::Value(&dest_contents()[header_offset + 6],
235 header_size - 6 + len);
236 crc = crc32c::Mask(crc);
237 EncodeFixed32(&dest_contents()[header_offset], crc);
238 }
