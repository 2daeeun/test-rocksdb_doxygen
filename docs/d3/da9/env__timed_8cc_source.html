172// An environment that measures function call times for filesystem
173// operations, reporting results to variables in PerfContext.
174Env* NewTimedEnv(Env* base_env) {
175 std::shared_ptr<FileSystem> timed_fs =
176 NewTimedFileSystem(base_env->GetFileSystem());
177 return new CompositeEnvWrapper(base_env, timed_fs);
178}
