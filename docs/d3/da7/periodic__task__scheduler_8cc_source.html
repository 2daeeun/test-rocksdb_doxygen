13// `timer_mutex` is a global mutex serves 3 purposes currently:
14// (1) to ensure calls to `Start()` and `Shutdown()` are serialized, as
15// they are currently not implemented in a thread-safe way; and
16// (2) to ensure the `Timer::Add()`s and `Timer::Start()` run atomically, and
17// the `Timer::Cancel()`s and `Timer::Shutdown()` run atomically.
18// (3) protect tasks_map_ in PeriodicTaskScheduler
19// Note: It's not efficient to have a static global mutex, for
20// PeriodicTaskScheduler it should be okay, as the operations are called
21// infrequently.
22static port::Mutex timer_mutex;

43Status PeriodicTaskScheduler::Register(PeriodicTaskType task_type,
44 const PeriodicTaskFunc& fn,
45 uint64_t repeat_period_seconds) {
46 MutexLock l(&timer_mutex);
47 static std::atomic<uint64_t> initial_delay(0);
48
49 if (repeat_period_seconds == kInvalidPeriodSec) {
50 return Status::InvalidArgument("Invalid task repeat period");
51 }
52 auto it = tasks_map_.find(task_type);
53 if (it != tasks_map_.end()) {
54 // the task already exists and it's the same, no update needed
55 if (it->second.repeat_every_sec == repeat_period_seconds) {
56 return Status::OK();
57 }
58 // cancel the existing one before register new one
59 timer_->Cancel(it->second.name);
60 tasks_map_.erase(it);
61 }
62
63 timer_->Start();
64 // put task type name as prefix, for easy debug
65 std::string unique_id =
66 kPeriodicTaskTypeNames.at(task_type) + std::to_string(id_++);
67
68 bool succeeded = timer_->Add(
69 fn, unique_id,
70 (initial_delay.fetch_add(1) % repeat_period_seconds) * kMicrosInSecond,
71 repeat_period_seconds * kMicrosInSecond);
72 if (!succeeded) {
73 return Status::Aborted("Failed to register periodic task");
74 }
75 auto result = tasks_map_.try_emplace(
76 task_type, TaskInfo{unique_id, repeat_period_seconds});
77 if (!result.second) {
78 return Status::Aborted("Failed to add periodic task");
79 }
80 return Status::OK();
81}

83Status PeriodicTaskScheduler::Unregister(PeriodicTaskType task_type) {
84 MutexLock l(&timer_mutex);
85 auto it = tasks_map_.find(task_type);
86 if (it != tasks_map_.end()) {
87 timer_->Cancel(it->second.name);
88 tasks_map_.erase(it);
89 }
90 if (!timer_->HasPendingTask()) {
91 timer_->Shutdown();
92 }
93 return Status::OK();
94}
