11// newest version is called "current". Older versions may be kept
12// around to provide a consistent view to live iterators.
13//
14// Each Version keeps track of a set of table files per level, as well as a
15// set of blob files. The entire set of versions is maintained in a
16// VersionSet.
17//
18// Version,VersionSet are thread-compatible, but require external
19// synchronization on all accesses.

380 const auto it = file_locations_.find(file_number);

385
386 assert(it->second.GetLevel() < num_levels_);
387 assert(it->second.GetPosition() < files_[it->second.GetLevel()].size());

390 ->fd.GetNumber() == file_number);

433 double space_amp = 0.0;

435
436 BlobStats GetBlobStats() const {
437 uint64_t total_file_size = 0;
438 uint64_t total_garbage_size = 0;
439
440 for (const auto& meta : blob_files_) {
441 assert(meta);
442
443 total_file_size += meta->GetBlobFileSize();
444 total_garbage_size += meta->GetGarbageBlobBytes();
445 }
446
447 double space_amp = 0.0;
448 if (total_file_size > total_garbage_size) {
449 space_amp = static_cast<double>(total_file_size) /
450 (total_file_size - total_garbage_size);
451 }
452
453 return BlobStats{total_file_size, total_garbage_size, space_amp};
454 }
