201 // be added in the trace. Foreg, if bit at position 1 is set then
202 // IOTraceOp::kIOLen (length) will be logged in the record (Since
203 // IOTraceOp::kIOLen = 1 in the enum). So find all the set positions in
204 // io_op_data one by one and, update corresponsing info in the trace record,
205 // unset that bit to find other set bits until io_op_data = 0.
206 /* Read remaining options based on io_op_data set by file operation */
207 // Assuming 63 bits will be used at max.
208 int64_t io_op_data = static_cast<int64_t>(record->io_op_data);
209 while (io_op_data) {
210 // Find the rightmost set bit.
211 uint32_t set_pos = static_cast<uint32_t>(log2(io_op_data & -io_op_data));
212 switch (set_pos) {
213 case IOTraceOp::kIOFileSize:
214 if (!GetFixed64(&enc_slice, &record->file_size)) {
