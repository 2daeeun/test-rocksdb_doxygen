19// Basic identifiers and metadata for a file in a DB. This only includes
20// information considered relevant for taking backups, checkpoints, or other
21// services relating to DB file storage.
22// This is only appropriate for immutable files, such as SST files or all
23// files in a backup. See also LiveFileStorageInfo.
24struct FileStorageInfo {
25 // The name of the file within its directory (e.g. "123456.sst")
26 std::string relative_filename;
27 // The directory containing the file, without a trailing '/'. This could be
28 // a DB path, wal_dir, etc.
29 std::string directory;
30
31 // The id of the file within a single DB. Set to 0 if the file does not have
32 // a number (e.g. CURRENT)
33 uint64_t file_number = 0;
34 // The type of the file as part of a DB.
35 FileType file_type = kTempFile;
36
37 // File size in bytes. See also `trim_to_size`.
38 uint64_t size = 0;
39
40 // This feature is experimental and subject to change.
41 Temperature temperature = Temperature::kUnknown;
42
43 // The checksum of a SST file, the value is decided by the file content and
44 // the checksum algorithm used for this SST file. The checksum function is
45 // identified by the file_checksum_func_name. If the checksum function is
46 // not specified, file_checksum is "0" by default.
47 std::string file_checksum;
48
49 // The name of the checksum function used to generate the file checksum
50 // value. If file checksum is not enabled (e.g., sst_file_checksum_func is
51 // null), file_checksum_func_name is UnknownFileChecksumFuncName, which is
52 // "Unknown".
53 std::string file_checksum_func_name;
54};

56// Adds to FileStorageInfo the ability to capture the state of files that
57// might change in a running DB.
58struct LiveFileStorageInfo : public FileStorageInfo {
59 // If non-empty, this string represents the "saved" contents of the file
60 // for the current context. (This field is used for checkpointing CURRENT
61 // file.) In that case, size == replacement_contents.size() and file on disk
62 // should be ignored. If empty string, the file on disk should still have
63 // "saved" contents. (See trim_to_size.)
64 std::string replacement_contents;
65
66 // If true, the file on disk is allowed to be larger than `size` but only
67 // the first `size` bytes should be used for the current context. If false,
68 // the file is corrupt if size on disk does not equal `size`.
69 bool trim_to_size = false;
70};

72// The metadata that describes an SST file. (Does not need to extend
73// LiveFileStorageInfo because SST files are always immutable.)
74struct SstFileMetaData : public FileStorageInfo {
75 SstFileMetaData() { file_type = kTableFile; }
76
77 SstFileMetaData(const std::string& _file_name, uint64_t _file_number,
78 const std::string& _directory, uint64_t _size,
79 SequenceNumber _smallest_seqno, SequenceNumber _largest_seqno,
80 const std::string& _smallestkey,
81 const std::string& _largestkey, uint64_t _num_reads_sampled,
82 bool _being_compacted, Temperature _temperature,
83 uint64_t _oldest_blob_file_number,
84 uint64_t _oldest_ancester_time, uint64_t _file_creation_time,
85 uint64_t _epoch_number, std::string& _file_checksum,
86 std::string& _file_checksum_func_name)
87 : smallest_seqno(_smallest_seqno),
88 largest_seqno(_largest_seqno),
89 smallestkey(_smallestkey),
90 largestkey(_largestkey),
91 num_reads_sampled(_num_reads_sampled),
92 being_compacted(_being_compacted),
93 num_entries(0),
94 num_deletions(0),
95 oldest_blob_file_number(_oldest_blob_file_number),
96 oldest_ancester_time(_oldest_ancester_time),
97 file_creation_time(_file_creation_time),
98 epoch_number(_epoch_number) {
99 if (!_file_name.empty()) {
100 if (_file_name[0] == '/') {
101 relative_filename = _file_name.substr(1);
102 name = _file_name; // Deprecated field
103 } else {
104 relative_filename = _file_name;
105 name = std::string("/") + _file_name; // Deprecated field
106 }
107 assert(relative_filename.size() + 1 == name.size());
108 assert(relative_filename[0] != '/');
109 assert(name[0] == '/');
110 }
111 directory = _directory;
112 db_path = _directory; // Deprecated field
113 file_number = _file_number;
114 file_type = kTableFile;
115 size = _size;
116 temperature = _temperature;
117 file_checksum = _file_checksum;
118 file_checksum_func_name = _file_checksum_func_name;
119 }
120
121 SequenceNumber smallest_seqno = 0; // Smallest sequence number in file.
122 SequenceNumber largest_seqno = 0; // Largest sequence number in file.
123 std::string smallestkey; // Smallest user defined key in the file.
124 std::string largestkey; // Largest user defined key in the file.
125 uint64_t num_reads_sampled = 0; // How many times the file is read.
126 bool being_compacted =
127 false; // true if the file is currently being compacted.
128
129 uint64_t num_entries = 0;
130 uint64_t num_deletions = 0;
131
132 uint64_t oldest_blob_file_number = 0; // The id of the oldest blob file
133 // referenced by the file.
134 // An SST file may be generated by compactions whose input files may
135 // in turn be generated by earlier compactions. The creation time of the
136 // oldest SST file that is the compaction ancestor of this file.
137 // The timestamp is provided SystemClock::GetCurrentTime().
138 // 0 if the information is not available.
139 //
140 // Note: for TTL blob files, it contains the start of the expiration range.
141 uint64_t oldest_ancester_time = 0;
142 // Timestamp when the SST file is created, provided by
143 // SystemClock::GetCurrentTime(). 0 if the information is not available.
144 uint64_t file_creation_time = 0;
145 // The order of a file being flushed or ingested/imported.
146 // Compaction output file will be assigned with the minimum `epoch_number`
147 // among input files'.
148 // For L0, larger `epoch_number` indicates newer L0 file.
149 // 0 if the information is not available.
150 uint64_t epoch_number = 0;
151
152 // These bounds define the effective key range for range tombstones
153 // in this file.
154 // Currently only used by CreateColumnFamilyWithImport().
155 std::string smallest{}; // Smallest internal key served by table
156 std::string largest{}; // Largest internal key served by table
157
158 // DEPRECATED: The name of the file within its directory with a
159 // leading slash (e.g. "/123456.sst"). Use relative_filename from base struct
160 // instead.
161 std::string name;
162
163 // DEPRECATED: replaced by `directory` in base struct
164 std::string db_path;
165};
