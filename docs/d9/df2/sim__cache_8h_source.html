23// For instrumentation purpose, use NewSimCache instead of NewLRUCache API
24// NewSimCache is a wrapper function returning a SimCache instance that can
25// have additional interface provided in Simcache class besides Cache interface
26// to predict block cache hit rate without actually allocating the memory. It
27// can help users tune their current block cache size, and determine how
28// efficient they are using the memory.
29//
30// Since GetSimCapacity() returns the capacity for simulation, it differs from
31// actual memory usage, which can be estimated as:
32// sim_capacity * entry_size / (entry_size + block_size),
33// where 76 <= entry_size <= 104,
34// BlockBasedTableOptions.block_size = 4096 by default but is configurable,
35// Therefore, generally the actual memory overhead of SimCache is Less than
36// sim_capacity * 2%
37std::shared_ptr<SimCache> NewSimCache(std::shared_ptr<Cache> cache,
38 size_t sim_capacity, int num_shard_bits);

44// An abstract base class (public interface) to the SimCache implementation
45class SimCache : public CacheWrapper {
46 public:
47 using CacheWrapper::CacheWrapper;
48
49 // returns the maximum configured capacity of the simcache for simulation
50 virtual size_t GetSimCapacity() const = 0;
51
52 // simcache doesn't provide internal handler reference to user, so always
53 // PinnedUsage = 0 and the behavior will be not exactly consistent the
54 // with real cache.
55 // returns the memory size for the entries residing in the simcache.
56 virtual size_t GetSimUsage() const = 0;
57
58 // sets the maximum configured capacity of the simcache. When the new
59 // capacity is less than the old capacity and the existing usage is
60 // greater than new capacity, the implementation will purge old entries
61 // to fit new capacity.
62 virtual void SetSimCapacity(size_t capacity) = 0;
63
64 // returns the lookup times of simcache
65 virtual uint64_t get_miss_counter() const = 0;
66 // returns the hit times of simcache
67 virtual uint64_t get_hit_counter() const = 0;
68 // reset the lookup and hit counters
69 virtual void reset_counter() = 0;
70 // String representation of the statistics of the simcache
71 virtual std::string ToString() const = 0;
72
73 // Start storing logs of the cache activity (Add/Lookup) into
74 // a file located at activity_log_file, max_logging_size option can be used to
75 // stop logging to the file automatically after reaching a specific size in
76 // bytes, a values of 0 disable this feature
77 virtual Status StartActivityLogging(const std::string& activity_log_file,
78 Env* env,
79 uint64_t max_logging_size = 0) = 0;
80
81 // Stop cache activity logging if any
82 virtual void StopActivityLogging() = 0;
83
84 // Status of cache logging happening in background
85 virtual Status GetActivityLoggingStatus() = 0;
86
87 private:
88 SimCache(const SimCache&);
89 SimCache& operator=(const SimCache&);
90};
