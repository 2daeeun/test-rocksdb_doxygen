12// Instead of opening a `DB` to perform certain manifest updates, this
13// uses the underlying `VersionSet` API to read and modify the MANIFEST. This
14// allows us to use the user's real options, while not having to worry about
15// the DB persisting new SST files via flush/compaction or attempting to read/
16// compact files which may fail, particularly for the file we intend to remove
17// (the user may want to remove an already deleted file from MANIFEST).
18class OfflineManifestWriter {
19 public:
20 OfflineManifestWriter(const DBOptions& options, const std::string& db_path)
21 : wc_(options.delayed_write_rate),
22 wb_(options.db_write_buffer_size),
23 immutable_db_options_(WithDbPath(options, db_path)),
24 tc_(NewLRUCache(1 << 20 /* capacity */,
25 options.table_cache_numshardbits)),
26 versions_(db_path, &immutable_db_options_, sopt_, tc_.get(), &wb_, &wc_,
27 /*block_cache_tracer=*/nullptr, /*io_tracer=*/nullptr,
28 /*db_id=*/"", /*db_session_id=*/"",
29 options.daily_offpeak_time_utc,
30 /*error_handler=*/nullptr,
31 /*read_only=*/false) {}
32
33 Status Recover(const std::vector<ColumnFamilyDescriptor>& column_families) {
34 return versions_.Recover(column_families, /*read_only*/ false,
35 /*db_id*/ nullptr,
36 /*no_error_if_files_missing*/ true);
37 }
38
39 Status LogAndApply(const ReadOptions& read_options,
40 const WriteOptions& write_options, ColumnFamilyData* cfd,
41 VersionEdit* edit,
42 FSDirectory* dir_contains_current_file) {
43 // Use `mutex` to imitate a locked DB mutex when calling `LogAndApply()`.
44 InstrumentedMutex mutex;
45 mutex.Lock();
46 Status s = versions_.LogAndApply(cfd, *cfd->GetLatestMutableCFOptions(),
47 read_options, write_options, edit, &mutex,
48 dir_contains_current_file,
49 false /* new_descriptor_log */);
50 mutex.Unlock();
51 return s;
52 }
53
54 VersionSet& Versions() { return versions_; }
55 const ImmutableDBOptions& IOptions() { return immutable_db_options_; }
56
57 private:
58 WriteController wc_;
59 WriteBufferManager wb_;
60 ImmutableDBOptions immutable_db_options_;
61 std::shared_ptr<Cache> tc_;
62 EnvOptions sopt_;
63 VersionSet versions_;
64
65 static ImmutableDBOptions WithDbPath(const DBOptions& options,
66 const std::string& db_path) {
67 ImmutableDBOptions rv(options);
68 if (rv.db_paths.empty()) {
69 // `VersionSet` expects options that have been through
70 // `SanitizeOptions()`, which would sanitize an empty `db_paths`.
71 rv.db_paths.emplace_back(db_path, 0 /* target_size */);
72 }
73 return rv;
74 }
75};
