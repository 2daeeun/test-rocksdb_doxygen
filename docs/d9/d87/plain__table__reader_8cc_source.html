42// Safely getting a uint32_t element from a char array, where, starting from
43// `base`, every 4 bytes are considered as an fixed 32 bit integer.
44inline uint32_t GetFixed32Element(const char* base, size_t offset) {
45 return DecodeFixed32(base + offset * sizeof(uint32_t));
46}

115Status PlainTableReader::Open(
116 const ImmutableOptions& ioptions, const EnvOptions& env_options,
117 const InternalKeyComparator& internal_comparator,
118 std::unique_ptr<RandomAccessFileReader>&& file, uint64_t file_size,
119 std::unique_ptr<TableReader>* table_reader, const int bloom_bits_per_key,
120 double hash_table_ratio, size_t index_sparseness, size_t huge_page_tlb_size,
121 bool full_scan_mode, const bool immortal_table,
122 const SliceTransform* prefix_extractor) {
123 if (file_size > PlainTableIndex::kMaxFileSize) {
124 return Status::NotSupported("File is too large for PlainTableReader!");
125 }
126
127 std::unique_ptr<TableProperties> props;
128 // TODO: plumb Env::IOActivity, Env::IOPriority
129 const ReadOptions read_options;
130 auto s = ReadTableProperties(file.get(), file_size, kPlainTableMagicNumber,
131 ioptions, read_options, &props);
132 if (!s.ok()) {
133 return s;
134 }
135
136 assert(hash_table_ratio >= 0.0);
137 auto& user_props = props->user_collected_properties;
138 auto prefix_extractor_in_file = props->prefix_extractor_name;
139
140 if (!full_scan_mode &&
141 !prefix_extractor_in_file.empty() /* old version sst file*/
142 && prefix_extractor_in_file != "nullptr") {
143 if (!prefix_extractor) {
144 return Status::InvalidArgument(
145 "Prefix extractor is missing when opening a PlainTable built "
146 "using a prefix extractor");
147 } else if (prefix_extractor_in_file != prefix_extractor->AsString()) {
148 return Status::InvalidArgument(
149 "Prefix extractor given doesn't match the one used to build "
150 "PlainTable");
151 }
152 }
153
154 EncodingType encoding_type = kPlain;
155 auto encoding_type_prop =
156 user_props.find(PlainTablePropertyNames::kEncodingType);
157 if (encoding_type_prop != user_props.end()) {
158 encoding_type = static_cast<EncodingType>(
159 DecodeFixed32(encoding_type_prop->second.c_str()));
160 }
161
162 std::unique_ptr<PlainTableReader> new_reader(new PlainTableReader(
163 ioptions, std::move(file), env_options, internal_comparator,
164 encoding_type, file_size, props.get(), prefix_extractor));
165
166 s = new_reader->MmapDataIfNeeded();
167 if (!s.ok()) {
168 return s;
169 }
170
171 if (!full_scan_mode) {
172 s = new_reader->PopulateIndex(props.get(), bloom_bits_per_key,
173 hash_table_ratio, index_sparseness,
174 huge_page_tlb_size);
175 if (!s.ok()) {
176 return s;
177 }
178 } else {
179 // Flag to indicate it is a full scan mode so that none of the indexes
180 // can be used.
181 new_reader->full_scan_mode_ = true;
182 }
183 // PopulateIndex can add to the props, so don't store them until now
184 new_reader->table_properties_ = std::move(props);
185
186 if (immortal_table && new_reader->file_info_.is_mmap_mode) {
187 new_reader->dummy_cleanable_.reset(new Cleanable());
188 }
189
190 *table_reader = std::move(new_reader);
191 return s;
192}

196InternalIterator* PlainTableReader::NewIterator(
197 const ReadOptions& options, const SliceTransform* /* prefix_extractor */,
198 Arena* arena, bool /*skip_filters*/, TableReaderCaller /*caller*/,
199 size_t /*compaction_readahead_size*/, bool /* allow_unprepared_value */) {
200 // Not necessarily used here, but make sure this has been initialized
201 assert(table_properties_);
202
203 // Auto prefix mode is not implemented in PlainTable.
204 bool use_prefix_seek =
205 !IsTotalOrderMode() &&
206 (options.prefix_same_as_start ||
207 (!options.total_order_seek && !options.auto_prefix_mode));
208 if (arena == nullptr) {
209 return new PlainTableIterator(this, use_prefix_seek);
210 } else {
211 auto mem = arena->AllocateAligned(sizeof(PlainTableIterator));
212 return new (mem) PlainTableIterator(this, use_prefix_seek);
213 }

216Status PlainTableReader::PopulateIndexRecordList(
217 PlainTableIndexBuilder* index_builder,
218 std::vector<uint32_t>* prefix_hashes) {
219 Slice prev_key_prefix_slice;
220 std::string prev_key_prefix_buf;
221 uint32_t pos = data_start_offset_;
222
223 bool is_first_record = true;
224 Slice key_prefix_slice;
225 PlainTableKeyDecoder decoder(&file_info_, encoding_type_, user_key_len_,
226 prefix_extractor_);
227 while (pos < file_info_.data_end_offset) {
228 uint32_t key_offset = pos;
229 ParsedInternalKey key;
230 Slice value_slice;
231 bool seekable = false;
232 Status s = Next(&decoder, &pos, &key, nullptr, &value_slice, &seekable);
233 if (!s.ok()) {
234 return s;
235 }
236
237 key_prefix_slice = GetPrefix(key);
238 if (enable_bloom_) {
239 bloom_.AddHash(GetSliceHash(key.user_key));
240 } else {
241 if (is_first_record || prev_key_prefix_slice != key_prefix_slice) {
242 if (!is_first_record) {
243 prefix_hashes->push_back(GetSliceHash(prev_key_prefix_slice));
244 }
245 if (file_info_.is_mmap_mode) {
246 prev_key_prefix_slice = key_prefix_slice;
247 } else {
248 prev_key_prefix_buf = key_prefix_slice.ToString();
249 prev_key_prefix_slice = prev_key_prefix_buf;
250 }
251 }
252 }
253
254 index_builder->AddKeyPrefix(GetPrefix(key), key_offset);
255
256 if (!seekable && is_first_record) {
257 return Status::Corruption("Key for a prefix is not seekable");
258 }
259
260 is_first_record = false;
261 }
262
263 prefix_hashes->push_back(GetSliceHash(key_prefix_slice));
264 auto s = index_.InitFromRawData(index_builder->Finish());
265 return s;
266}

268void PlainTableReader::AllocateBloom(int bloom_bits_per_key, int num_keys,
269 size_t huge_page_tlb_size) {
270 uint32_t bloom_total_bits = num_keys * bloom_bits_per_key;
271 if (bloom_total_bits > 0) {
272 enable_bloom_ = true;
273 bloom_.SetTotalBits(&arena_, bloom_total_bits, ioptions_.bloom_locality,
274 huge_page_tlb_size, ioptions_.logger);
275 }
276}

278void PlainTableReader::FillBloom(const std::vector<uint32_t>& prefix_hashes) {
279 assert(bloom_.IsInitialized());
280 for (const auto prefix_hash : prefix_hashes) {
281 bloom_.AddHash(prefix_hash);
282 }

295Status PlainTableReader::PopulateIndex(TableProperties* props,
296 int bloom_bits_per_key,
297 double hash_table_ratio,
298 size_t index_sparseness,
299 size_t huge_page_tlb_size) {
300 assert(props != nullptr);
301
302 BlockContents index_block_contents;
303
304 // TODO: plumb Env::IOActivity, Env::IOPriority
305 const ReadOptions read_options;
306 Status s =
307 ReadMetaBlock(file_info_.file.get(), nullptr /* prefetch_buffer */,
308 file_size_, kPlainTableMagicNumber, ioptions_, read_options,
309 PlainTableIndexBuilder::kPlainTableIndexBlock,
310 BlockType::kIndex, &index_block_contents);
311
312 bool index_in_file = s.ok();
313
314 BlockContents bloom_block_contents;
315 bool bloom_in_file = false;
316 // We only need to read the bloom block if index block is in file.
317 if (index_in_file) {
318 s = ReadMetaBlock(file_info_.file.get(), nullptr /* prefetch_buffer */,
319 file_size_, kPlainTableMagicNumber, ioptions_,
320 read_options, BloomBlockBuilder::kBloomBlock,
321 BlockType::kFilter, &bloom_block_contents);
322 bloom_in_file = s.ok() && bloom_block_contents.data.size() > 0;
323 }
324
325 Slice* bloom_block;
326 if (bloom_in_file) {
327 // If bloom_block_contents.allocation is not empty (which will be the case
328 // for non-mmap mode), it holds the alloated memory for the bloom block.
329 // It needs to be kept alive to keep `bloom_block` valid.
330 bloom_block_alloc_ = std::move(bloom_block_contents.allocation);
331 bloom_block = &bloom_block_contents.data;
332 } else {
333 bloom_block = nullptr;
334 }
335
336 Slice* index_block;
337 if (index_in_file) {
338 // If index_block_contents.allocation is not empty (which will be the case
339 // for non-mmap mode), it holds the alloated memory for the index block.
340 // It needs to be kept alive to keep `index_block` valid.
341 index_block_alloc_ = std::move(index_block_contents.allocation);
342 index_block = &index_block_contents.data;
343 } else {
344 index_block = nullptr;
345 }
346
347 if ((prefix_extractor_ == nullptr) && (hash_table_ratio != 0)) {
348 // moptions.prefix_extractor is requried for a hash-based look-up.
349 return Status::NotSupported(
350 "PlainTable requires a prefix extractor enable prefix hash mode.");
351 }
352
353 // First, read the whole file, for every kIndexIntervalForSamePrefixKeys rows
354 // for a prefix (starting from the first one), generate a record of (hash,
355 // offset) and append it to IndexRecordList, which is a data structure created
356 // to store them.
357
358 if (!index_in_file) {
359 // Allocate bloom filter here for total order mode.
360 if (IsTotalOrderMode()) {
361 AllocateBloom(bloom_bits_per_key,
362 static_cast<uint32_t>(props->num_entries),
363 huge_page_tlb_size);
364 }
365 } else if (bloom_in_file) {
366 enable_bloom_ = true;
367 auto num_blocks_property = props->user_collected_properties.find(
368 PlainTablePropertyNames::kNumBloomBlocks);
369
370 uint32_t num_blocks = 0;
371 if (num_blocks_property != props->user_collected_properties.end()) {
372 Slice temp_slice(num_blocks_property->second);
373 if (!GetVarint32(&temp_slice, &num_blocks)) {
374 num_blocks = 0;
375 }
376 }
377 // cast away const qualifier, because bloom_ won't be changed
378 bloom_.SetRawData(const_cast<char*>(bloom_block->data()),
379 static_cast<uint32_t>(bloom_block->size()) * 8,
380 num_blocks);
381 } else {
382 // Index in file but no bloom in file. Disable bloom filter in this case.
383 enable_bloom_ = false;
384 bloom_bits_per_key = 0;
385 }
386
387 PlainTableIndexBuilder index_builder(&arena_, ioptions_, prefix_extractor_,
388 index_sparseness, hash_table_ratio,
389 huge_page_tlb_size);
390
391 std::vector<uint32_t> prefix_hashes;
392 if (!index_in_file) {
393 // Populates _bloom if enabled (total order mode)
394 s = PopulateIndexRecordList(&index_builder, &prefix_hashes);
395 if (!s.ok()) {
396 return s;
397 }
398 } else {
399 s = index_.InitFromRawData(*index_block);
400 if (!s.ok()) {
401 return s;
402 }
403 }
404
405 if (!index_in_file) {
406 if (!IsTotalOrderMode()) {
407 // Calculated bloom filter size and allocate memory for
408 // bloom filter based on the number of prefixes, then fill it.
409 AllocateBloom(bloom_bits_per_key, index_.GetNumPrefixes(),
410 huge_page_tlb_size);
411 if (enable_bloom_) {
412 FillBloom(prefix_hashes);
413 }
414 }
415 }
416
417 // Fill two table properties.
418 if (!index_in_file) {
419 props->user_collected_properties["plain_table_hash_table_size"] =
420 std::to_string(index_.GetIndexSize() * PlainTableIndex::kOffsetLen);
421 props->user_collected_properties["plain_table_sub_index_size"] =
422 std::to_string(index_.GetSubIndexSize());
423 } else {
424 props->user_collected_properties["plain_table_hash_table_size"] =
425 std::to_string(0);
426 props->user_collected_properties["plain_table_sub_index_size"] =
427 std::to_string(0);
428 }
429
430 return Status::OK();
431}

433Status PlainTableReader::GetOffset(PlainTableKeyDecoder* decoder,
434 const Slice& target, const Slice& prefix,
435 uint32_t prefix_hash, bool& prefix_matched,
436 uint32_t* offset) const {
437 prefix_matched = false;
438 uint32_t prefix_index_offset;
439 auto res = index_.GetOffset(prefix_hash, &prefix_index_offset);
440 if (res == PlainTableIndex::kNoPrefixForBucket) {
441 *offset = file_info_.data_end_offset;
442 return Status::OK();
443 } else if (res == PlainTableIndex::kDirectToFile) {
444 *offset = prefix_index_offset;
445 return Status::OK();
446 }
447
448 // point to sub-index, need to do a binary search
449 uint32_t upper_bound = 0;
450 const char* base_ptr =
451 index_.GetSubIndexBasePtrAndUpperBound(prefix_index_offset, &upper_bound);
452 uint32_t low = 0;
453 uint32_t high = upper_bound;
454 ParsedInternalKey mid_key;
455 ParsedInternalKey parsed_target;
456 Status s = ParseInternalKey(target, &parsed_target,
457 false /* log_err_key */); // TODO
458 if (!s.ok()) {
459 return s;
460 }
461
462 // The key is between [low, high). Do a binary search between it.
463 while (high - low > 1) {
464 uint32_t mid = (high + low) / 2;
465 uint32_t file_offset = GetFixed32Element(base_ptr, mid);
466 uint32_t tmp;
467 s = decoder->NextKeyNoValue(file_offset, &mid_key, nullptr, &tmp);
468 if (!s.ok()) {
469 return s;
470 }
471 int cmp_result = internal_comparator_.Compare(mid_key, parsed_target);
472 if (cmp_result < 0) {
473 low = mid;
474 } else {
475 if (cmp_result == 0) {
476 // Happen to have found the exact key or target is smaller than the
477 // first key after base_offset.
478 prefix_matched = true;
479 *offset = file_offset;
480 return Status::OK();
481 } else {
482 high = mid;
483 }
484 }
485 }
486 // Both of the key at the position low or low+1 could share the same
487 // prefix as target. We need to rule out one of them to avoid to go
488 // to the wrong prefix.
489 ParsedInternalKey low_key;
490 uint32_t tmp;
491 uint32_t low_key_offset = GetFixed32Element(base_ptr, low);
492 s = decoder->NextKeyNoValue(low_key_offset, &low_key, nullptr, &tmp);
493 if (!s.ok()) {
494 return s;
495 }
496
497 if (GetPrefix(low_key) == prefix) {
498 prefix_matched = true;
499 *offset = low_key_offset;
500 } else if (low + 1 < upper_bound) {
501 // There is possible a next prefix, return it
502 prefix_matched = false;
503 *offset = GetFixed32Element(base_ptr, low + 1);
504 } else {
505 // target is larger than a key of the last prefix in this bucket
506 // but with a different prefix. Key does not exist.
507 *offset = file_info_.data_end_offset;
508 }
509 return Status::OK();
510}

512bool PlainTableReader::MatchBloom(uint32_t hash) const {
513 if (!enable_bloom_) {
514 return true;
515 }
516
517 if (bloom_.MayContainHash(hash)) {
518 PERF_COUNTER_ADD(bloom_sst_hit_count, 1);
519 return true;
520 } else {
521 PERF_COUNTER_ADD(bloom_sst_miss_count, 1);
522 return false;
523 }
524}

526Status PlainTableReader::Next(PlainTableKeyDecoder* decoder, uint32_t* offset,
527 ParsedInternalKey* parsed_key,
528 Slice* internal_key, Slice* value,
529 bool* seekable) const {
530 if (*offset == file_info_.data_end_offset) {
531 *offset = file_info_.data_end_offset;
532 return Status::OK();
533 }
534
535 if (*offset > file_info_.data_end_offset) {
536 return Status::Corruption("Offset is out of file size");
537 }
538
539 uint32_t bytes_read;
540 Status s = decoder->NextKey(*offset, parsed_key, internal_key, value,
541 &bytes_read, seekable);
542 if (!s.ok()) {
543 return s;
544 }
545 *offset = *offset + bytes_read;
546 return Status::OK();
547}

556Status PlainTableReader::Get(const ReadOptions& /*ro*/, const Slice& target,
557 GetContext* get_context,
558 const SliceTransform* /* prefix_extractor */,
559 bool /*skip_filters*/) {
560 // Check bloom filter first.
561 Slice prefix_slice;
562 uint32_t prefix_hash;
563 if (IsTotalOrderMode()) {
564 if (full_scan_mode_) {
565 status_ =
566 Status::InvalidArgument("Get() is not allowed in full scan mode.");
567 }
568 // Match whole user key for bloom filter check.
569 if (!MatchBloom(GetSliceHash(ExtractUserKey(target)))) {
570 return Status::OK();
571 }
572 // in total order mode, there is only one bucket 0, and we always use empty
573 // prefix.
574 prefix_slice = Slice();
575 prefix_hash = 0;
576 } else {
577 prefix_slice = GetPrefix(target);
578 prefix_hash = GetSliceHash(prefix_slice);
579 if (!MatchBloom(prefix_hash)) {
580 return Status::OK();
581 }
582 }
583 uint32_t offset;
584 bool prefix_match;
585 PlainTableKeyDecoder decoder(&file_info_, encoding_type_, user_key_len_,
586 prefix_extractor_);
587 Status s = GetOffset(&decoder, target, prefix_slice, prefix_hash,
588 prefix_match, &offset);
589
590 if (!s.ok()) {
591 return s;
592 }
593 ParsedInternalKey found_key;
594 ParsedInternalKey parsed_target;
595 s = ParseInternalKey(target, &parsed_target,
596 false /* log_err_key */); // TODO
597 if (!s.ok()) {
598 return s;
599 }
600
601 Slice found_value;
602 while (offset < file_info_.data_end_offset) {
603 s = Next(&decoder, &offset, &found_key, nullptr, &found_value);
604 if (!s.ok()) {
605 return s;
606 }
607 if (!prefix_match) {
608 // Need to verify prefix for the first key found if it is not yet
609 // checked.
610 if (GetPrefix(found_key) != prefix_slice) {
611 return Status::OK();
612 }
613 prefix_match = true;
614 }
615 // TODO(ljin): since we know the key comparison result here,
616 // can we enable the fast path?
617 if (internal_comparator_.Compare(found_key, parsed_target) >= 0) {
618 bool dont_care __attribute__((__unused__));
619 bool ret = get_context->SaveValue(found_key, found_value, &dont_care, &s,
620 dummy_cleanable_.get());
621 if (!s.ok()) {
622 return s;
623 }
624 if (!ret) {
625 break;
626 }
627 }
628 }
629 return Status::OK();
630}

677void PlainTableIterator::Seek(const Slice& target) {
678 if (use_prefix_seek_ != !table_->IsTotalOrderMode()) {
679 // This check is done here instead of NewIterator() to permit creating an
680 // iterator with total_order_seek = true even if we won't be able to Seek()
681 // it. This is needed for compaction: it creates iterator with
682 // total_order_seek = true but usually never does Seek() on it,
683 // only SeekToFirst().
684 status_ = Status::InvalidArgument(
685 "total_order_seek not implemented for PlainTable.");
686 offset_ = next_offset_ = table_->file_info_.data_end_offset;
687 return;
688 }

690 // If the user doesn't set prefix seek option and we are not able to do a
691 // total Seek(). assert failure.
692 if (table_->IsTotalOrderMode()) {
693 if (table_->full_scan_mode_) {
694 status_ =
695 Status::InvalidArgument("Seek() is not allowed in full scan mode.");
696 offset_ = next_offset_ = table_->file_info_.data_end_offset;
697 return;
698 } else if (table_->GetIndexSize() > 1) {
699 assert(false);
700 status_ = Status::NotSupported(
701 "PlainTable cannot issue non-prefix seek unless in total order "
702 "mode.");
703 offset_ = next_offset_ = table_->file_info_.data_end_offset;
704 return;
705 }
706 }
707
708 Slice prefix_slice = table_->GetPrefix(target);
709 uint32_t prefix_hash = 0;
710 // Bloom filter is ignored in total-order mode.
711 if (!table_->IsTotalOrderMode()) {
712 prefix_hash = GetSliceHash(prefix_slice);
713 if (!table_->MatchBloom(prefix_hash)) {
714 status_ = Status::OK();
715 offset_ = next_offset_ = table_->file_info_.data_end_offset;
716 return;
717 }
718 }
719 bool prefix_match;
720 status_ = table_->GetOffset(&decoder_, target, prefix_slice, prefix_hash,
721 prefix_match, &next_offset_);
722 if (!status_.ok()) {
723 offset_ = next_offset_ = table_->file_info_.data_end_offset;
724 return;
725 }
726
727 if (next_offset_ < table_->file_info_.data_end_offset) {
728 for (Next(); status_.ok() && Valid(); Next()) {
729 if (!prefix_match) {
730 // Need to verify the first key's prefix
731 if (table_->GetPrefix(key()) != prefix_slice) {
732 offset_ = next_offset_ = table_->file_info_.data_end_offset;
733 break;
734 }
735 prefix_match = true;
736 }
737 if (table_->internal_comparator_.Compare(key(), target) >= 0) {
738 break;
739 }
740 }
741 } else {
742 offset_ = table_->file_info_.data_end_offset;
743 }
744}

747 assert(false);
748 status_ =
749 Status::NotSupported("SeekForPrev() is not supported in PlainTable");
750 offset_ = next_offset_ = table_->file_info_.data_end_offset;
751}

754 offset_ = next_offset_;
755 if (offset_ < table_->file_info_.data_end_offset) {
756 Slice tmp_slice;
757 ParsedInternalKey parsed_key;
758 status_ =
759 table_->Next(&decoder_, &next_offset_, &parsed_key, &key_, &value_);
760 if (!status_.ok()) {
761 offset_ = next_offset_ = table_->file_info_.data_end_offset;
762 }
763 }
764}
