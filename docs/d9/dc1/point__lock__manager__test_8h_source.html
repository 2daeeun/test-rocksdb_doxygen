157TEST_P(AnyLockManagerTest, LockConflict) {
158 // Tests that lock conflicts lead to lock timeout.
159 MockColumnFamilyHandle cf(1);
160 locker_->AddColumnFamily(&cf);
161 auto txn1 = NewTxn();
162 auto txn2 = NewTxn();
163
164 {
165 // exclusive-exclusive conflict.
166 ASSERT_OK(locker_->TryLock(txn1, 1, "k1", env_, true));
167 auto s = locker_->TryLock(txn2, 1, "k1", env_, true);
168 ASSERT_TRUE(s.IsTimedOut());
169 }
170
171 {
172 // exclusive-shared conflict.
173 ASSERT_OK(locker_->TryLock(txn1, 1, "k2", env_, true));
174 auto s = locker_->TryLock(txn2, 1, "k2", env_, false);
175 ASSERT_TRUE(s.IsTimedOut());
176 }
177
178 {
179 // shared-exclusive conflict.
180 ASSERT_OK(locker_->TryLock(txn1, 1, "k2", env_, false));
181 auto s = locker_->TryLock(txn2, 1, "k2", env_, true);
182 ASSERT_TRUE(s.IsTimedOut());
183 }
184
185 // Cleanup
186 locker_->UnLock(txn1, 1, "k1", env_);
187 locker_->UnLock(txn1, 1, "k2", env_);
188
189 delete txn1;
190 delete txn2;

193port::Thread BlockUntilWaitingTxn(const char* sync_point_name,
194 std::function<void()> f) {
195 std::atomic<bool> reached(false);
196 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
197 sync_point_name, [&](void* /*arg*/) { reached.store(true); });
198 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
199
200 port::Thread t(f);
201
202 while (!reached.load()) {
203 std::this_thread::sleep_for(std::chrono::milliseconds(100));
204 }
205 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
206 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->ClearAllCallBacks();
207
208 return t;
209}

228TEST_P(AnyLockManagerTest, Deadlock) {
229 // Tests that deadlock can be detected.
230 // Deadlock scenario:
231 // txn1 exclusively locks k1, and wants to lock k2;
232 // txn2 exclusively locks k2, and wants to lock k1.
233 MockColumnFamilyHandle cf(1);
234 locker_->AddColumnFamily(&cf);
235 TransactionOptions txn_opt;
236 txn_opt.deadlock_detect = true;
237 txn_opt.lock_timeout = 1000000;
238 auto txn1 = NewTxn(txn_opt);
239 auto txn2 = NewTxn(txn_opt);
240
241 ASSERT_OK(locker_->TryLock(txn1, 1, "k1", env_, true));
242 ASSERT_OK(locker_->TryLock(txn2, 1, "k2", env_, true));
243
244 // txn1 tries to lock k2, will block forever.
245 port::Thread t = BlockUntilWaitingTxn(wait_sync_point_name_, [&]() {
246 // block because txn2 is holding a lock on k2.
247 ASSERT_OK(locker_->TryLock(txn1, 1, "k2", env_, true));
248 });
249
250 auto s = locker_->TryLock(txn2, 1, "k1", env_, true);
251 ASSERT_TRUE(s.IsBusy());
252 ASSERT_EQ(s.subcode(), Status::SubCode::kDeadlock);
253
254 std::vector<DeadlockPath> deadlock_paths = locker_->GetDeadlockInfoBuffer();
255 ASSERT_EQ(deadlock_paths.size(), 1u);
256 ASSERT_FALSE(deadlock_paths[0].limit_exceeded);
257
258 std::vector<DeadlockInfo> deadlocks = deadlock_paths[0].path;
259 ASSERT_EQ(deadlocks.size(), 2u);
260
261 ASSERT_EQ(deadlocks[0].m_txn_id, txn1->GetID());
262 ASSERT_EQ(deadlocks[0].m_cf_id, 1u);
263 ASSERT_TRUE(deadlocks[0].m_exclusive);
264 ASSERT_EQ(deadlocks[0].m_waiting_key, "k2");
265
266 ASSERT_EQ(deadlocks[1].m_txn_id, txn2->GetID());
267 ASSERT_EQ(deadlocks[1].m_cf_id, 1u);
268 ASSERT_TRUE(deadlocks[1].m_exclusive);
269 ASSERT_EQ(deadlocks[1].m_waiting_key, "k1");
270
271 locker_->UnLock(txn2, 1, "k2", env_);
272 t.join();
273
274 // Cleanup
275 locker_->UnLock(txn1, 1, "k1", env_);
276 locker_->UnLock(txn1, 1, "k2", env_);
277 delete txn2;
278 delete txn1;
279}

281TEST_P(AnyLockManagerTest, GetWaitingTxns_MultipleTxns) {
282 MockColumnFamilyHandle cf(1);
283 locker_->AddColumnFamily(&cf);
284
285 auto txn1 = NewTxn();
286 ASSERT_OK(locker_->TryLock(txn1, 1, "k", env_, false));
287
288 auto txn2 = NewTxn();
289 ASSERT_OK(locker_->TryLock(txn2, 1, "k", env_, false));
290
291 auto txn3 = NewTxn();
292 txn3->SetLockTimeout(10000);
293 port::Thread t1 = BlockUntilWaitingTxn(wait_sync_point_name_, [&]() {
294 ASSERT_OK(locker_->TryLock(txn3, 1, "k", env_, true));
295 locker_->UnLock(txn3, 1, "k", env_);
296 });
297
298 // Ok, now txn3 is waiting for lock on "k", which is owned by two
299 // transactions. Check that GetWaitingTxns reports this correctly
300 uint32_t wait_cf_id;
301 std::string wait_key;
302 auto waiters = txn3->GetWaitingTxns(&wait_cf_id, &wait_key);
303
304 ASSERT_EQ(wait_cf_id, 1u);
305 ASSERT_EQ(wait_key, "k");
306 ASSERT_EQ(waiters.size(), 2);
307 bool waits_correct =
308 (waiters[0] == txn1->GetID() && waiters[1] == txn2->GetID()) ||
309 (waiters[1] == txn1->GetID() && waiters[0] == txn2->GetID());
310 ASSERT_EQ(waits_correct, true);
311
312 // Release locks so txn3 can proceed with execution
313 locker_->UnLock(txn1, 1, "k", env_);
314 locker_->UnLock(txn2, 1, "k", env_);
315
316 // Wait until txn3 finishes
317 t1.join();
318
319 delete txn1;
320 delete txn2;
321 delete txn3;
322}
