57class TestDatabaseLogs(unittest.TestCase):
58 def test_check_and_trigger_conditions(self):
59 this_path = os.path.abspath(os.path.dirname(__file__))
60 logs_path_prefix = os.path.join(this_path, "input_files/LOG-0")
61 column_families = ["default", "col-fam-A", "col-fam-B"]
62 db_logs = DatabaseLogs(logs_path_prefix, column_families)
63 # matches, has 2 col_fams
64 condition1 = LogCondition.create(Condition("cond-A"))
65 condition1.set_parameter("regex", "random log message")
66 # matches, multiple lines message
67 condition2 = LogCondition.create(Condition("cond-B"))
68 condition2.set_parameter("regex", "continuing on next line")
69 # does not match
70 condition3 = LogCondition.create(Condition("cond-C"))
71 condition3.set_parameter("regex", "this should match no log")
72 db_logs.check_and_trigger_conditions([condition1, condition2, condition3])
73 cond1_trigger = condition1.get_trigger()
74 self.assertEqual(2, len(cond1_trigger.keys()))
75 self.assertSetEqual({"col-fam-A", NO_COL_FAMILY}, set(cond1_trigger.keys()))
76 self.assertEqual(2, len(cond1_trigger["col-fam-A"]))
77 messages = [
78 "[db/db_impl.cc:563] [col-fam-A] random log message for testing",
79 "[db/db_impl.cc:653] [col-fam-A] another random log message",
80 ]
81 self.assertIn(cond1_trigger["col-fam-A"][0].get_message(), messages)
82 self.assertIn(cond1_trigger["col-fam-A"][1].get_message(), messages)
83 self.assertEqual(1, len(cond1_trigger[NO_COL_FAMILY]))
84 self.assertEqual(
85 cond1_trigger[NO_COL_FAMILY][0].get_message(),
86 "[db/db_impl.cc:331] [unknown] random log message no column family",
87 )
88 cond2_trigger = condition2.get_trigger()
89 self.assertEqual(["col-fam-B"], list(cond2_trigger.keys()))
90 self.assertEqual(1, len(cond2_trigger["col-fam-B"]))
91 self.assertEqual(
92 cond2_trigger["col-fam-B"][0].get_message(),
93 "[db/db_impl.cc:234] [col-fam-B] log continuing on next line\n"
94 + "remaining part of the log",
95 )
96 self.assertIsNone(condition3.get_trigger())
