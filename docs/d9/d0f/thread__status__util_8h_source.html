19// The static utility class for updating thread-local status.
20//
21// The thread-local status is updated via the thread-local cached
22// pointer thread_updater_local_cache_. During each function call,
23// when ThreadStatusUtil finds thread_updater_local_cache_ is
24// left uninitialized (determined by thread_updater_initialized_),
25// it will tries to initialize it using the return value of
26// Env::GetThreadStatusUpdater(). When thread_updater_local_cache_
27// is initialized by a non-null pointer, each function call will
28// then update the status of the current thread. Otherwise,
29// all function calls to ThreadStatusUtil will be no-op.
30class ThreadStatusUtil {
31 public:
32 // Register the current thread for tracking.
33 static void RegisterThread(const Env* env,
34 ThreadStatus::ThreadType thread_type);
35
36 // Unregister the current thread.
37 static void UnregisterThread();
38
39 // Create an entry in the global ColumnFamilyInfo table for the
40 // specified column family. This function should be called only
41 // when the current thread does not hold db_mutex.
42 static void NewColumnFamilyInfo(const DB* db, const ColumnFamilyData* cfd,
43 const std::string& cf_name, const Env* env);
44
45 // Erase the ConstantColumnFamilyInfo that is associated with the
46 // specified ColumnFamilyData. This function should be called only
47 // when the current thread does not hold db_mutex.
48 static void EraseColumnFamilyInfo(const ColumnFamilyData* cfd);
49
50 // Erase all ConstantColumnFamilyInfo that is associated with the
51 // specified db instance. This function should be called only when
52 // the current thread does not hold db_mutex.
53 static void EraseDatabaseInfo(const DB* db);
54
55 static void SetEnableTracking(bool enable_tracking);
56
57 // Update the thread status to indicate the current thread is doing
58 // something related to the specified column family.
59 //
60 // REQUIRES: cfd != nullptr
61 static void SetColumnFamily(const ColumnFamilyData* cfd);
62
63 static void SetThreadOperation(ThreadStatus::OperationType type);
64
65 static ThreadStatus::OperationType GetThreadOperation();
66
67 static ThreadStatus::OperationStage SetThreadOperationStage(
68 ThreadStatus::OperationStage stage);
69
70 static void SetThreadOperationProperty(int code, uint64_t value);
71
72 static void IncreaseThreadOperationProperty(int code, uint64_t delta);
73
74 static void SetThreadState(ThreadStatus::StateType type);
75
76 static void ResetThreadStatus();
77
78#ifndef NDEBUG
79 static void TEST_SetStateDelay(const ThreadStatus::StateType state,
80 int micro);
81 static void TEST_StateDelay(const ThreadStatus::StateType state);
82
83 static Env::IOActivity TEST_GetExpectedIOActivity(
84 ThreadStatus::OperationType thread_op);
85#endif
86
87 protected:
88 // Initialize the thread-local ThreadStatusUpdater when it finds
89 // the cached value is nullptr. Returns true if it has cached
90 // a non-null pointer.
91 static bool MaybeInitThreadLocalUpdater(const Env* env);
92
93#ifdef ROCKSDB_USING_THREAD_STATUS
94 // A boolean flag indicating whether thread_updater_local_cache_
95 // is initialized. It is set to true when an Env uses any
96 // ThreadStatusUtil functions using the current thread other
97 // than UnregisterThread(). It will be set to false when
98 // UnregisterThread() is called.
99 //
100 // When this variable is set to true, thread_updater_local_cache_
101 // will not be updated until this variable is again set to false
102 // in UnregisterThread().
103 static thread_local bool thread_updater_initialized_;
104
105 // The thread-local cached ThreadStatusUpdater that caches the
106 // thread_status_updater_ of the first Env that uses any ThreadStatusUtil
107 // function other than UnregisterThread(). This variable will
108 // be cleared when UnregisterThread() is called.
109 //
110 // When this variable is set to a non-null pointer, then the status
111 // of the current thread will be updated when a function of
112 // ThreadStatusUtil is called. Otherwise, all functions of
113 // ThreadStatusUtil will be no-op.
114 //
115 // When thread_updater_initialized_ is set to true, this variable
116 // will not be updated until this thread_updater_initialized_ is
117 // again set to false in UnregisterThread().
118 static thread_local ThreadStatusUpdater* thread_updater_local_cache_;
119#else
120 static bool thread_updater_initialized_;
121 static ThreadStatusUpdater* thread_updater_local_cache_;
122#endif
123};
