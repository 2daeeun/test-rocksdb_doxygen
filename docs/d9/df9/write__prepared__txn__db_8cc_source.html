42Status WritePreparedTxnDB::Initialize(
43 const std::vector<size_t>& compaction_enabled_cf_indices,
44 const std::vector<ColumnFamilyHandle*>& handles) {
45 auto dbimpl = static_cast_with_check<DBImpl>(GetRootDB());
46 assert(dbimpl != nullptr);
47 auto rtxns = dbimpl->recovered_transactions();
48 std::map<SequenceNumber, SequenceNumber> ordered_seq_cnt;
49 for (const auto& rtxn : rtxns) {
50 // There should only one batch for WritePrepared policy.
51 assert(rtxn.second->batches_.size() == 1);
52 const auto& seq = rtxn.second->batches_.begin()->first;
53 const auto& batch_info = rtxn.second->batches_.begin()->second;
54 auto cnt = batch_info.batch_cnt_ ? batch_info.batch_cnt_ : 1;
55 ordered_seq_cnt[seq] = cnt;
56 }
57 // AddPrepared must be called in order
58 for (auto seq_cnt : ordered_seq_cnt) {
59 auto seq = seq_cnt.first;
60 auto cnt = seq_cnt.second;
61 for (size_t i = 0; i < cnt; i++) {
62 AddPrepared(seq + i);
63 }
64 }
65 SequenceNumber prev_max = max_evicted_seq_;
66 SequenceNumber last_seq = db_impl_->GetLatestSequenceNumber();
67 AdvanceMaxEvictedSeq(prev_max, last_seq);
68 // Create a gap between max and the next snapshot. This simplifies the logic
69 // in IsInSnapshot by not having to consider the special case of max ==
70 // snapshot after recovery. This is tested in IsInSnapshotEmptyMapTest.
71 if (last_seq) {
72 db_impl_->versions_->SetLastAllocatedSequence(last_seq + 1);
73 db_impl_->versions_->SetLastSequence(last_seq + 1);
74 db_impl_->versions_->SetLastPublishedSequence(last_seq + 1);
75 }
76
77 db_impl_->SetSnapshotChecker(new WritePreparedSnapshotChecker(this));
78 // A callback to commit a single sub-batch
79 class CommitSubBatchPreReleaseCallback : public PreReleaseCallback {
80 public:
81 explicit CommitSubBatchPreReleaseCallback(WritePreparedTxnDB* db)
82 : db_(db) {}
83 Status Callback(SequenceNumber commit_seq,
84 bool is_mem_disabled __attribute__((__unused__)), uint64_t,
85 size_t /*index*/, size_t /*total*/) override {
86 assert(!is_mem_disabled);
87 db_->AddCommitted(commit_seq, commit_seq);
88 return Status::OK();
89 }
90
91 private:
92 WritePreparedTxnDB* db_;
93 };
94 db_impl_->SetRecoverableStatePreReleaseCallback(
95 new CommitSubBatchPreReleaseCallback(this));
96
97 auto s = PessimisticTransactionDB::Initialize(compaction_enabled_cf_indices,
98 handles);
99 return s;
100}

158Status WritePreparedTxnDB::WriteInternal(const WriteOptions& write_options_orig,
159 WriteBatch* batch, size_t batch_cnt,
160 WritePreparedTxn* txn) {
161 ROCKS_LOG_DETAILS(db_impl_->immutable_db_options().info_log,
162 "CommitBatchInternal");
163 if (batch->Count() == 0) {
164 // Otherwise our 1 seq per batch logic will break since there is no seq
165 // increased for this batch.
166 return Status::OK();
167 }
168
169 if (write_options_orig.protection_bytes_per_key > 0) {
170 auto s = WriteBatchInternal::UpdateProtectionInfo(
171 batch, write_options_orig.protection_bytes_per_key);
172 if (!s.ok()) {
173 return s;
174 }
175 }
176
177 if (batch_cnt == 0) { // not provided, then compute it
178 // TODO(myabandeh): add an option to allow user skipping this cost
179 SubBatchCounter counter(*GetCFComparatorMap());
180 auto s = batch->Iterate(&counter);
181 if (!s.ok()) {
182 return s;
183 }
184 batch_cnt = counter.BatchCount();
185 WPRecordTick(TXN_DUPLICATE_KEY_OVERHEAD);
186 ROCKS_LOG_DETAILS(info_log_, "Duplicate key overhead: %" PRIu64 " batches",
187 static_cast<uint64_t>(batch_cnt));
188 }
189 assert(batch_cnt);
190
191 bool do_one_write = !db_impl_->immutable_db_options().two_write_queues;
192 WriteOptions write_options(write_options_orig);
193 // In the absence of Prepare markers, use Noop as a batch separator
194 auto s = WriteBatchInternal::InsertNoop(batch);
195 assert(s.ok());
196 const bool DISABLE_MEMTABLE = true;
197 const uint64_t no_log_ref = 0;
198 uint64_t seq_used = kMaxSequenceNumber;
199 const size_t ZERO_PREPARES = 0;
200 const bool kSeperatePrepareCommitBatches = true;
201 // Since this is not 2pc, there is no need for AddPrepared but having it in
202 // the PreReleaseCallback enables an optimization. Refer to
203 // SmallestUnCommittedSeq for more details.
204 AddPreparedCallback add_prepared_callback(
205 this, db_impl_, batch_cnt,
206 db_impl_->immutable_db_options().two_write_queues,
207 !kSeperatePrepareCommitBatches);
208 WritePreparedCommitEntryPreReleaseCallback update_commit_map(
209 this, db_impl_, kMaxSequenceNumber, ZERO_PREPARES, batch_cnt);
210 PreReleaseCallback* pre_release_callback;
211 if (do_one_write) {
212 pre_release_callback = &update_commit_map;
213 } else {
214 pre_release_callback = &add_prepared_callback;
215 }
216 s = db_impl_->WriteImpl(write_options, batch, nullptr, nullptr, nullptr,
217 no_log_ref, !DISABLE_MEMTABLE, &seq_used, batch_cnt,
218 pre_release_callback);
219 assert(!s.ok() || seq_used != kMaxSequenceNumber);
220 uint64_t prepare_seq = seq_used;
221 if (txn != nullptr) {
222 txn->SetId(prepare_seq);
223 }
224 if (!s.ok()) {
225 return s;
226 }
227 if (do_one_write) {
228 return s;
229 } // else do the 2nd write for commit
230 ROCKS_LOG_DETAILS(db_impl_->immutable_db_options().info_log,
231 "CommitBatchInternal 2nd write prepare_seq: %" PRIu64,
232 prepare_seq);
233 // Commit the batch by writing an empty batch to the 2nd queue that will
234 // release the commit sequence number to readers.
235 const size_t ZERO_COMMITS = 0;
236 WritePreparedCommitEntryPreReleaseCallback update_commit_map_with_prepare(
237 this, db_impl_, prepare_seq, batch_cnt, ZERO_COMMITS);
238 WriteBatch empty_batch;
239 write_options.disableWAL = true;
240 write_options.sync = false;
241 const size_t ONE_BATCH = 1; // Just to inc the seq
242 s = db_impl_->WriteImpl(write_options, &empty_batch, nullptr, nullptr,
243 nullptr, no_log_ref, DISABLE_MEMTABLE, &seq_used,
244 ONE_BATCH, &update_commit_map_with_prepare);
245 assert(!s.ok() || seq_used != kMaxSequenceNumber);
246 // Note: RemovePrepared is called from within PreReleaseCallback
247 return s;
248}

272Status WritePreparedTxnDB::GetImpl(const ReadOptions& options,
273 ColumnFamilyHandle* column_family,
274 const Slice& key, PinnableSlice* value) {
275 SequenceNumber min_uncommitted, snap_seq;
276 const SnapshotBackup backed_by_snapshot =
277 AssignMinMaxSeqs(options.snapshot, &min_uncommitted, &snap_seq);
278 WritePreparedTxnReadCallback callback(this, snap_seq, min_uncommitted,
279 backed_by_snapshot);
280 bool* dont_care = nullptr;
281 DBImpl::GetImplOptions get_impl_options;
282 get_impl_options.column_family = column_family;
283 get_impl_options.value = value;
284 get_impl_options.value_found = dont_care;
285 get_impl_options.callback = &callback;
286 auto res = db_impl_->GetImpl(options, key, get_impl_options);
287 if (LIKELY(callback.valid() && ValidateSnapshot(callback.max_visible_seq(),
288 backed_by_snapshot))) {
289 return res;
290 } else {
291 res.PermitUncheckedError();
292 WPRecordTick(TXN_GET_TRY_AGAIN);
293 return Status::TryAgain();
294 }

297void WritePreparedTxnDB::UpdateCFComparatorMap(
298 const std::vector<ColumnFamilyHandle*>& handles) {
299 auto cf_map = new std::map<uint32_t, const Comparator*>();
300 auto handle_map = new std::map<uint32_t, ColumnFamilyHandle*>();
301 for (auto h : handles) {
302 auto id = h->GetID();
303 const Comparator* comparator = h->GetComparator();
304 (*cf_map)[id] = comparator;
305 if (id != 0) {
306 (*handle_map)[id] = h;
307 } else {
308 // The pointer to the default cf handle in the handles will be deleted.
309 // Use the pointer maintained by the db instead.
310 (*handle_map)[id] = DefaultColumnFamily();
311 }
312 }
313 cf_map_.reset(cf_map);
314 handle_map_.reset(handle_map);
315}

501void WritePreparedTxnDB::CheckPreparedAgainstMax(SequenceNumber new_max,
502 bool locked) {
503 // When max_evicted_seq_ advances, move older entries from prepared_txns_
504 // to delayed_prepared_. This guarantees that if a seq is lower than max,
505 // then it is not in prepared_txns_ and save an expensive, synchronized
506 // lookup from a shared set. delayed_prepared_ is expected to be empty in
507 // normal cases.
508 ROCKS_LOG_DETAILS(
509 info_log_,
510 "CheckPreparedAgainstMax prepared_txns_.empty() %d top: %" PRIu64,
511 prepared_txns_.empty(),
512 prepared_txns_.empty() ? 0 : prepared_txns_.top());
513 const SequenceNumber prepared_top = prepared_txns_.top();
514 const bool empty = prepared_top == kMaxSequenceNumber;
515 // Preliminary check to avoid the synchronization cost
516 if (!empty && prepared_top <= new_max) {
517 if (locked) {
518 // Needed to avoid double locking in pop().
519 prepared_txns_.push_pop_mutex()->Unlock();
520 }
521 WriteLock wl(&prepared_mutex_);
522 // Need to fetch fresh values of ::top after mutex is acquired
523 while (!prepared_txns_.empty() && prepared_txns_.top() <= new_max) {
524 auto to_be_popped = prepared_txns_.top();
525 delayed_prepared_.insert(to_be_popped);
526 ROCKS_LOG_WARN(info_log_,
527 "prepared_mutex_ overhead %" PRIu64 " (prep=%" PRIu64
528 " new_max=%" PRIu64 ")",
529 static_cast<uint64_t>(delayed_prepared_.size()),
530 to_be_popped, new_max);
531 delayed_prepared_empty_.store(false, std::memory_order_release);
532 // Update prepared_txns_ after updating delayed_prepared_empty_ otherwise
533 // there will be a point in time that the entry is neither in
534 // prepared_txns_ nor in delayed_prepared_, which will not be checked if
535 // delayed_prepared_empty_ is false.
536 prepared_txns_.pop();
537 }
538 if (locked) {
539 prepared_txns_.push_pop_mutex()->Lock();
540 }
541 }
542}

544void WritePreparedTxnDB::AddPrepared(uint64_t seq, bool locked) {
545 ROCKS_LOG_DETAILS(info_log_, "Txn %" PRIu64 " Preparing with max %" PRIu64,
546 seq, max_evicted_seq_.load());
547 TEST_SYNC_POINT("AddPrepared::begin:pause");
548 TEST_SYNC_POINT("AddPrepared::begin:resume");
549 if (!locked) {
550 prepared_txns_.push_pop_mutex()->Lock();
551 }
552 prepared_txns_.push_pop_mutex()->AssertHeld();
553 prepared_txns_.push(seq);
554 auto new_max = future_max_evicted_seq_.load();
555 if (UNLIKELY(seq <= new_max)) {
556 // This should not happen in normal case
557 ROCKS_LOG_ERROR(
558 info_log_,
559 "Added prepare_seq is not larger than max_evicted_seq_: %" PRIu64
560 " <= %" PRIu64,
561 seq, new_max);
562 CheckPreparedAgainstMax(new_max, true /*locked*/);
563 }
564 if (!locked) {
565 prepared_txns_.push_pop_mutex()->Unlock();
566 }
567 TEST_SYNC_POINT("AddPrepared::end");
568}

570void WritePreparedTxnDB::AddCommitted(uint64_t prepare_seq, uint64_t commit_seq,
571 uint8_t loop_cnt) {
572 ROCKS_LOG_DETAILS(info_log_, "Txn %" PRIu64 " Committing with %" PRIu64,
573 prepare_seq, commit_seq);
574 TEST_SYNC_POINT("WritePreparedTxnDB::AddCommitted:start");
575 TEST_SYNC_POINT("WritePreparedTxnDB::AddCommitted:start:pause");
576 auto indexed_seq = prepare_seq % COMMIT_CACHE_SIZE;
577 CommitEntry64b evicted_64b;
578 CommitEntry evicted;
579 bool to_be_evicted = GetCommitEntry(indexed_seq, &evicted_64b, &evicted);
580 if (LIKELY(to_be_evicted)) {
581 assert(evicted.prep_seq != prepare_seq);
582 auto prev_max = max_evicted_seq_.load(std::memory_order_acquire);
583 ROCKS_LOG_DETAILS(info_log_,
584 "Evicting %" PRIu64 ",%" PRIu64 " with max %" PRIu64,
585 evicted.prep_seq, evicted.commit_seq, prev_max);
586 if (prev_max < evicted.commit_seq) {
587 auto last = db_impl_->GetLastPublishedSequence(); // could be 0
588 SequenceNumber max_evicted_seq;
589 if (LIKELY(evicted.commit_seq < last)) {
590 assert(last > 0);
591 // Inc max in larger steps to avoid frequent updates
592 max_evicted_seq =
593 std::min(evicted.commit_seq + INC_STEP_FOR_MAX_EVICTED, last - 1);
594 } else {
595 // legit when a commit entry in a write batch overwrite the previous one
596 max_evicted_seq = evicted.commit_seq;
597 }
598#ifdef OS_LINUX
599 if (rocksdb_write_prepared_TEST_ShouldClearCommitCache &&
600 rocksdb_write_prepared_TEST_ShouldClearCommitCache()) {
601 max_evicted_seq = last;
602 }
603#endif // OS_LINUX
604 ROCKS_LOG_DETAILS(info_log_,
605 "%lu Evicting %" PRIu64 ",%" PRIu64 " with max %" PRIu64
606 " => %lu",
607 prepare_seq, evicted.prep_seq, evicted.commit_seq,
608 prev_max, max_evicted_seq);
609 AdvanceMaxEvictedSeq(prev_max, max_evicted_seq);
610 }
611 if (UNLIKELY(!delayed_prepared_empty_.load(std::memory_order_acquire))) {
612 WriteLock wl(&prepared_mutex_);
613 auto dp_iter = delayed_prepared_.find(evicted.prep_seq);
614 if (dp_iter != delayed_prepared_.end()) {
615 // This is a rare case that txn is committed but prepared_txns_ is not
616 // cleaned up yet. Refer to delayed_prepared_commits_ definition for
617 // why it should be kept updated.
618 delayed_prepared_commits_[evicted.prep_seq] = evicted.commit_seq;
619 ROCKS_LOG_DEBUG(info_log_,
620 "delayed_prepared_commits_[%" PRIu64 "]=%" PRIu64,
621 evicted.prep_seq, evicted.commit_seq);
622 }
623 }
624 // After each eviction from commit cache, check if the commit entry should
625 // be kept around because it overlaps with a live snapshot.
626 CheckAgainstSnapshots(evicted);
627 }
628 bool succ =
629 ExchangeCommitEntry(indexed_seq, evicted_64b, {prepare_seq, commit_seq});
630 if (UNLIKELY(!succ)) {
631 ROCKS_LOG_ERROR(info_log_,
632 "ExchangeCommitEntry failed on [%" PRIu64 "] %" PRIu64
633 ",%" PRIu64 " retrying...",
634 indexed_seq, prepare_seq, commit_seq);
635 // A very rare event, in which the commit entry is updated before we do.
636 // Here we apply a very simple solution of retrying.
637 if (loop_cnt > 100) {
638 throw std::runtime_error("Infinite loop in AddCommitted!");
639 }
640 AddCommitted(prepare_seq, commit_seq, ++loop_cnt);
641 return;
642 }
643 TEST_SYNC_POINT("WritePreparedTxnDB::AddCommitted:end");
644 TEST_SYNC_POINT("WritePreparedTxnDB::AddCommitted:end:pause");

708void WritePreparedTxnDB::AdvanceMaxEvictedSeq(const SequenceNumber& prev_max,
709 const SequenceNumber& new_max) {
710 ROCKS_LOG_DETAILS(info_log_,
711 "AdvanceMaxEvictedSeq overhead %" PRIu64 " => %" PRIu64,
712 prev_max, new_max);
713 // Declare the intention before getting snapshot from the DB. This helps a
714 // concurrent GetSnapshot to wait to catch up with future_max_evicted_seq_ if
715 // it has not already. Otherwise the new snapshot is when we ask DB for
716 // snapshots smaller than future max.
717 auto updated_future_max = prev_max;
718 while (updated_future_max < new_max &&
719 !future_max_evicted_seq_.compare_exchange_weak(
720 updated_future_max, new_max, std::memory_order_acq_rel,
721 std::memory_order_relaxed)) {
722 };
723
724 CheckPreparedAgainstMax(new_max, false /*locked*/);
725
726 // With each change to max_evicted_seq_ fetch the live snapshots behind it.
727 // We use max as the version of snapshots to identify how fresh are the
728 // snapshot list. This works because the snapshots are between 0 and
729 // max, so the larger the max, the more complete they are.
730 SequenceNumber new_snapshots_version = new_max;
731 std::vector<SequenceNumber> snapshots;
732 bool update_snapshots = false;
733 if (new_snapshots_version > snapshots_version_) {
734 // This is to avoid updating the snapshots_ if it already updated
735 // with a more recent vesion by a concrrent thread
736 update_snapshots = true;
737 // We only care about snapshots lower then max
738 snapshots = GetSnapshotListFromDB(new_max);
739 }
740 if (update_snapshots) {
741 UpdateSnapshots(snapshots, new_snapshots_version);
742 if (!snapshots.empty()) {
743 WriteLock wl(&old_commit_map_mutex_);
744 for (auto snap : snapshots) {
745 // This allows IsInSnapshot to tell apart the reads from in valid
746 // snapshots from the reads from committed values in valid snapshots.
747 old_commit_map_[snap];
748 }
749 old_commit_map_empty_.store(false, std::memory_order_release);
750 }
751 }
752 auto updated_prev_max = prev_max;
753 TEST_SYNC_POINT("AdvanceMaxEvictedSeq::update_max:pause");
754 TEST_SYNC_POINT("AdvanceMaxEvictedSeq::update_max:resume");
755 while (updated_prev_max < new_max &&
756 !max_evicted_seq_.compare_exchange_weak(updated_prev_max, new_max,
757 std::memory_order_acq_rel,
758 std::memory_order_relaxed)) {
759 };
760}

767SnapshotImpl* WritePreparedTxnDB::GetSnapshotInternal(
768 bool for_ww_conflict_check) {
769 // Note: for this optimization setting the last sequence number and obtaining
770 // the smallest uncommitted seq should be done atomically. However to avoid
771 // the mutex overhead, we call SmallestUnCommittedSeq BEFORE taking the
772 // snapshot. Since we always updated the list of unprepared seq (via
773 // AddPrepared) AFTER the last sequence is updated, this guarantees that the
774 // smallest uncommitted seq that we pair with the snapshot is smaller or equal
775 // the value that would be obtained otherwise atomically. That is ok since
776 // this optimization works as long as min_uncommitted is less than or equal
777 // than the smallest uncommitted seq when the snapshot was taken.
778 auto min_uncommitted = WritePreparedTxnDB::SmallestUnCommittedSeq();
779 SnapshotImpl* snap_impl = db_impl_->GetSnapshotImpl(for_ww_conflict_check);
780 TEST_SYNC_POINT("WritePreparedTxnDB::GetSnapshotInternal:first");
781 assert(snap_impl);
782 SequenceNumber snap_seq = snap_impl->GetSequenceNumber();
783 // Note: Check against future_max_evicted_seq_ (in contrast with
784 // max_evicted_seq_) in case there is a concurrent AdvanceMaxEvictedSeq.
785 if (UNLIKELY(snap_seq != 0 && snap_seq <= future_max_evicted_seq_)) {
786 // There is a very rare case in which the commit entry evicts another commit
787 // entry that is not published yet thus advancing max evicted seq beyond the
788 // last published seq. This case is not likely in real-world setup so we
789 // handle it with a few retries.
790 size_t retry = 0;
791 SequenceNumber max;
792 while ((max = future_max_evicted_seq_.load()) != 0 &&
793 snap_impl->GetSequenceNumber() <= max && retry < 100) {
794 ROCKS_LOG_WARN(info_log_,
795 "GetSnapshot snap: %" PRIu64 " max: %" PRIu64
796 " retry %" ROCKSDB_PRIszt,
797 snap_impl->GetSequenceNumber(), max, retry);
798 ReleaseSnapshot(snap_impl);
799 // Wait for last visible seq to catch up with max, and also go beyond it
800 // by one.
801 AdvanceSeqByOne();
802 snap_impl = db_impl_->GetSnapshotImpl(for_ww_conflict_check);
803 assert(snap_impl);
804 retry++;
805 }
806 assert(snap_impl->GetSequenceNumber() > max);
807 if (snap_impl->GetSequenceNumber() <= max) {
808 throw std::runtime_error(
809 "Snapshot seq " + std::to_string(snap_impl->GetSequenceNumber()) +
810 " after " + std::to_string(retry) +
811 " retries is still less than futre_max_evicted_seq_" +
812 std::to_string(max));
813 }
814 }
815 EnhanceSnapshot(snap_impl, min_uncommitted);
816 ROCKS_LOG_DETAILS(
817 db_impl_->immutable_db_options().info_log,
818 "GetSnapshot %" PRIu64 " ww:%" PRIi32 " min_uncommitted: %" PRIu64,
819 snap_impl->GetSequenceNumber(), for_ww_conflict_check, min_uncommitted);
820 TEST_SYNC_POINT("WritePreparedTxnDB::GetSnapshotInternal:end");
821 return snap_impl;
822}

826 // published, i.e., max == last_published, increase the last published to

888void WritePreparedTxnDB::CleanupReleasedSnapshots(
889 const std::vector<SequenceNumber>& new_snapshots,
890 const std::vector<SequenceNumber>& old_snapshots) {
891 auto newi = new_snapshots.begin();
892 auto oldi = old_snapshots.begin();
893 for (; newi != new_snapshots.end() && oldi != old_snapshots.end();) {
894 assert(*newi >= *oldi); // cannot have new snapshots with lower seq
895 if (*newi == *oldi) { // still not released
896 auto value = *newi;
897 while (newi != new_snapshots.end() && *newi == value) {
898 newi++;
899 }
900 while (oldi != old_snapshots.end() && *oldi == value) {
901 oldi++;
902 }
903 } else {
904 assert(*newi > *oldi); // *oldi is released
905 ReleaseSnapshotInternal(*oldi);
906 oldi++;
907 }
908 }
909 // Everything remained in old_snapshots is released and must be cleaned up
910 for (; oldi != old_snapshots.end(); oldi++) {
911 ReleaseSnapshotInternal(*oldi);
912 }

915void WritePreparedTxnDB::UpdateSnapshots(
916 const std::vector<SequenceNumber>& snapshots,
917 const SequenceNumber& version) {
918 ROCKS_LOG_DETAILS(info_log_, "UpdateSnapshots with version %" PRIu64,
919 version);
920 TEST_SYNC_POINT("WritePreparedTxnDB::UpdateSnapshots:p:start");
921 TEST_SYNC_POINT("WritePreparedTxnDB::UpdateSnapshots:s:start");
922#ifndef NDEBUG
923 size_t sync_i = 0;
924#endif
925 ROCKS_LOG_DETAILS(info_log_, "snapshots_mutex_ overhead");
926 WriteLock wl(&snapshots_mutex_);
927 snapshots_version_ = version;
928 // We update the list concurrently with the readers.
929 // Both new and old lists are sorted and the new list is subset of the
930 // previous list plus some new items. Thus if a snapshot repeats in
931 // both new and old lists, it will appear upper in the new list. So if
932 // we simply insert the new snapshots in order, if an overwritten item
933 // is still valid in the new list is either written to the same place in
934 // the array or it is written in a higher palce before it gets
935 // overwritten by another item. This guarantess a reader that reads the
936 // list bottom-up will eventaully see a snapshot that repeats in the
937 // update, either before it gets overwritten by the writer or
938 // afterwards.
939 size_t i = 0;
940 auto it = snapshots.begin();
941 for (; it != snapshots.end() && i < SNAPSHOT_CACHE_SIZE; ++it, ++i) {
942 snapshot_cache_[i].store(*it, std::memory_order_release);
943 TEST_IDX_SYNC_POINT("WritePreparedTxnDB::UpdateSnapshots:p:", ++sync_i);
944 TEST_IDX_SYNC_POINT("WritePreparedTxnDB::UpdateSnapshots:s:", sync_i);
945 }
946#ifndef NDEBUG
947 // Release the remaining sync points since they are useless given that the
948 // reader would also use lock to access snapshots
949 for (++sync_i; sync_i <= 10; ++sync_i) {
950 TEST_IDX_SYNC_POINT("WritePreparedTxnDB::UpdateSnapshots:p:", sync_i);
951 TEST_IDX_SYNC_POINT("WritePreparedTxnDB::UpdateSnapshots:s:", sync_i);
952 }
953#endif
954 snapshots_.clear();
955 for (; it != snapshots.end(); ++it) {
956 // Insert them to a vector that is less efficient to access
957 // concurrently
958 snapshots_.push_back(*it);
959 }
960 // Update the size at the end. Otherwise a parallel reader might read
961 // items that are not set yet.
962 snapshots_total_.store(snapshots.size(), std::memory_order_release);
963
964 // Note: this must be done after the snapshots data structures are updated
965 // with the new list of snapshots.
966 CleanupReleasedSnapshots(snapshots, snapshots_all_);
967 snapshots_all_ = snapshots;
968
969 TEST_SYNC_POINT("WritePreparedTxnDB::UpdateSnapshots:p:end");
970 TEST_SYNC_POINT("WritePreparedTxnDB::UpdateSnapshots:s:end");
971}

973void WritePreparedTxnDB::CheckAgainstSnapshots(const CommitEntry& evicted) {
974 TEST_SYNC_POINT("WritePreparedTxnDB::CheckAgainstSnapshots:p:start");
975 TEST_SYNC_POINT("WritePreparedTxnDB::CheckAgainstSnapshots:s:start");
976#ifndef NDEBUG
977 size_t sync_i = 0;
978#endif
979 // First check the snapshot cache that is efficient for concurrent access
980 auto cnt = snapshots_total_.load(std::memory_order_acquire);
981 // The list might get updated concurrently as we are reading from it. The
982 // reader should be able to read all the snapshots that are still valid
983 // after the update. Since the survived snapshots are written in a higher
984 // place before gets overwritten the reader that reads bottom-up will
985 // eventully see it.
986 const bool next_is_larger = true;
987 // We will set to true if the border line snapshot suggests that.
988 bool search_larger_list = false;
989 size_t ip1 = std::min(cnt, SNAPSHOT_CACHE_SIZE);
990 for (; 0 < ip1; ip1--) {
991 SequenceNumber snapshot_seq =
992 snapshot_cache_[ip1 - 1].load(std::memory_order_acquire);
993 TEST_IDX_SYNC_POINT("WritePreparedTxnDB::CheckAgainstSnapshots:p:",
994 ++sync_i);
995 TEST_IDX_SYNC_POINT("WritePreparedTxnDB::CheckAgainstSnapshots:s:", sync_i);
996 if (ip1 == SNAPSHOT_CACHE_SIZE) { // border line snapshot
997 // snapshot_seq < commit_seq => larger_snapshot_seq <= commit_seq
998 // then later also continue the search to larger snapshots
999 search_larger_list = snapshot_seq < evicted.commit_seq;
1000 }
1001 if (!MaybeUpdateOldCommitMap(evicted.prep_seq, evicted.commit_seq,
1002 snapshot_seq, !next_is_larger)) {
1003 break;
1004 }
1005 }
1006#ifndef NDEBUG
1007 // Release the remaining sync points before accquiring the lock
1008 for (++sync_i; sync_i <= 10; ++sync_i) {
1009 TEST_IDX_SYNC_POINT("WritePreparedTxnDB::CheckAgainstSnapshots:p:", sync_i);
1010 TEST_IDX_SYNC_POINT("WritePreparedTxnDB::CheckAgainstSnapshots:s:", sync_i);
1011 }
1012#endif
1013 TEST_SYNC_POINT("WritePreparedTxnDB::CheckAgainstSnapshots:p:end");
1014 TEST_SYNC_POINT("WritePreparedTxnDB::CheckAgainstSnapshots:s:end");
1015 if (UNLIKELY(SNAPSHOT_CACHE_SIZE < cnt && search_larger_list)) {
1016 // Then access the less efficient list of snapshots_
1017 WPRecordTick(TXN_SNAPSHOT_MUTEX_OVERHEAD);
1018 ROCKS_LOG_WARN(info_log_,
1019 "snapshots_mutex_ overhead for <%" PRIu64 ",%" PRIu64
1020 "> with %" ROCKSDB_PRIszt " snapshots",
1021 evicted.prep_seq, evicted.commit_seq, cnt);
1022 ReadLock rl(&snapshots_mutex_);
1023 // Items could have moved from the snapshots_ to snapshot_cache_ before
1024 // accquiring the lock. To make sure that we do not miss a valid snapshot,
1025 // read snapshot_cache_ again while holding the lock.
1026 for (size_t i = 0; i < SNAPSHOT_CACHE_SIZE; i++) {
1027 SequenceNumber snapshot_seq =
1028 snapshot_cache_[i].load(std::memory_order_acquire);
1029 if (!MaybeUpdateOldCommitMap(evicted.prep_seq, evicted.commit_seq,
1030 snapshot_seq, next_is_larger)) {
1031 break;
1032 }
1033 }
1034 for (auto snapshot_seq_2 : snapshots_) {
1035 if (!MaybeUpdateOldCommitMap(evicted.prep_seq, evicted.commit_seq,
1036 snapshot_seq_2, next_is_larger)) {
1037 break;
1038 }
1039 }
1040 }
1041}

1043bool WritePreparedTxnDB::MaybeUpdateOldCommitMap(
1044 const uint64_t& prep_seq, const uint64_t& commit_seq,
1045 const uint64_t& snapshot_seq, const bool next_is_larger = true) {
1046 // If we do not store an entry in old_commit_map_ we assume it is committed in
1047 // all snapshots. If commit_seq <= snapshot_seq, it is considered already in
1048 // the snapshot so we need not to keep the entry around for this snapshot.
1049 if (commit_seq <= snapshot_seq) {
1050 // continue the search if the next snapshot could be smaller than commit_seq
1051 return !next_is_larger;
1052 }
1053 // then snapshot_seq < commit_seq
1054 if (prep_seq <= snapshot_seq) { // overlapping range
1055 WPRecordTick(TXN_OLD_COMMIT_MAP_MUTEX_OVERHEAD);
1056 ROCKS_LOG_WARN(info_log_,
1057 "old_commit_map_mutex_ overhead for %" PRIu64
1058 " commit entry: <%" PRIu64 ",%" PRIu64 ">",
1059 snapshot_seq, prep_seq, commit_seq);
1060 WriteLock wl(&old_commit_map_mutex_);
1061 old_commit_map_empty_.store(false, std::memory_order_release);
1062 auto& vec = old_commit_map_[snapshot_seq];
1063 vec.insert(std::upper_bound(vec.begin(), vec.end(), prep_seq), prep_seq);
1064 // We need to store it once for each overlapping snapshot. Returning true to
1065 // continue the search if there is more overlapping snapshot.
1066 return true;
1067 }
1068 // continue the search if the next snapshot could be larger than prep_seq
1069 return next_is_larger;
1070}
