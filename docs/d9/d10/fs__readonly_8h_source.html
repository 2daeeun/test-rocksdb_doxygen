13// A FileSystem wrapper that only allows read-only operation.

15// This class has not been fully analyzed for providing strong security
16// guarantees.

19 IOStatus s = IOStatus::IOError("Attempted write to ReadOnlyFileSystem");
20 assert(s.GetRetryable() == false);

22 }
