29// A wrapper of internal iterator whose purpose is to count how
30// many entries there are in the iterator.
31class SequenceIterWrapper : public InternalIterator {
32 public:
33 SequenceIterWrapper(InternalIterator* iter, const Comparator* cmp,
34 bool need_count_entries)
35 : icmp_(cmp),
36 inner_iter_(iter),
37 need_count_entries_(need_count_entries) {}
38 bool Valid() const override { return inner_iter_->Valid(); }
39 Status status() const override { return inner_iter_->status(); }
40 void Next() override {
41 if (!inner_iter_->IsDeleteRangeSentinelKey()) {
42 num_itered_++;
43 }
44 inner_iter_->Next();
45 }
46 void Seek(const Slice& target) override {
47 if (!need_count_entries_) {
48 has_num_itered_ = false;
49 inner_iter_->Seek(target);
50 } else {
51 // Need to count total number of entries,
52 // so we do Next() rather than Seek().
53 while (inner_iter_->Valid() &&
54 icmp_.Compare(inner_iter_->key(), target) < 0) {
55 Next();
56 }
57 }
58 }
59 Slice key() const override { return inner_iter_->key(); }
60 Slice value() const override { return inner_iter_->value(); }
61
62 // Unused InternalIterator methods
63 void SeekToFirst() override { assert(false); }
64 void Prev() override { assert(false); }
65 void SeekForPrev(const Slice& /* target */) override { assert(false); }
66 void SeekToLast() override { assert(false); }
67
68 uint64_t NumItered() const { return num_itered_; }
69 bool HasNumItered() const { return has_num_itered_; }
70 bool IsDeleteRangeSentinelKey() const override {
71 assert(Valid());
72 return inner_iter_->IsDeleteRangeSentinelKey();
73 }
74
75 private:
76 InternalKeyComparator icmp_;
77 InternalIterator* inner_iter_; // not owned
78 uint64_t num_itered_ = 0;
79 bool need_count_entries_;
80 bool has_num_itered_ = true;
81};
