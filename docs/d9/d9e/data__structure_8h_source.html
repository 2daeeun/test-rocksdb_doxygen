22// Represents a set of values of some enum type with a small number of
23// possible enumerators. For now, it supports enums where no enumerator
24// exceeds 63 when converted to int.
25template <typename ENUM_TYPE, ENUM_TYPE MAX_ENUMERATOR>
26class SmallEnumSet {
27 private:
28 using StateT = uint64_t;
29 static constexpr int kStateBits = sizeof(StateT) * 8;
30 static constexpr int kMaxMax = kStateBits - 1;
31 static constexpr int kMaxValue = static_cast<int>(MAX_ENUMERATOR);
32 static_assert(kMaxValue >= 0);
33 static_assert(kMaxValue <= kMaxMax);
34
35 public:
36 // construct / create
37 SmallEnumSet() : state_(0) {}
38
39 template <class... TRest>
40 /*implicit*/ constexpr SmallEnumSet(const ENUM_TYPE e, TRest... rest) {
41 *this = SmallEnumSet(rest...).With(e);
42 }
43
44 // Return the set that includes all valid values, assuming the enum
45 // is "dense" (includes all values converting to 0 through kMaxValue)
46 static constexpr SmallEnumSet All() {
47 StateT tmp = StateT{1} << kMaxValue;
48 return SmallEnumSet(RawStateMarker(), tmp | (tmp - 1));
49 }
50
51 // equality
52 bool operator==(const SmallEnumSet& that) const {
53 return this->state_ == that.state_;
54 }
55 bool operator!=(const SmallEnumSet& that) const { return !(*this == that); }
56
57 // query
58
59 // Return true if the input enum is contained in the "Set".
60 bool Contains(const ENUM_TYPE e) const {
61 int value = static_cast<int>(e);
62 assert(value >= 0 && value <= kMaxValue);
63 StateT tmp = 1;
64 return state_ & (tmp << value);
65 }
66
67 bool empty() const { return state_ == 0; }
68
69 // iterator
70 class const_iterator {
71 public:
72 // copy
73 const_iterator(const const_iterator& that) = default;
74 const_iterator& operator=(const const_iterator& that) = default;
75
76 // move
77 const_iterator(const_iterator&& that) noexcept = default;
78 const_iterator& operator=(const_iterator&& that) noexcept = default;
79
80 // equality
81 bool operator==(const const_iterator& that) const {
82 assert(set_ == that.set_);
83 return this->pos_ == that.pos_;
84 }
85
86 bool operator!=(const const_iterator& that) const {
87 return !(*this == that);
88 }
89
90 // ++iterator
91 const_iterator& operator++() {
92 if (pos_ < kMaxValue) {
93 pos_ = set_->SkipUnset(pos_ + 1);
94 } else {
95 pos_ = kStateBits;
96 }
97 return *this;
98 }
99
100 // iterator++
101 const_iterator operator++(int) {
102 auto old = *this;
103 ++*this;
104 return old;
105 }
106
107 ENUM_TYPE operator*() const {
108 assert(pos_ <= kMaxValue);
109 return static_cast<ENUM_TYPE>(pos_);
110 }
111
112 private:
113 friend class SmallEnumSet;
114 const_iterator(const SmallEnumSet* set, int pos) : set_(set), pos_(pos) {}
115 const SmallEnumSet* set_;
116 int pos_;
117 };
118
119 const_iterator begin() const { return const_iterator(this, SkipUnset(0)); }
120
121 const_iterator end() const { return const_iterator(this, kStateBits); }
122
123 // mutable ops
124
125 // Modifies the set (if needed) to include the given value. Returns true
126 // iff the set was modified.
127 bool Add(const ENUM_TYPE e) {
128 int value = static_cast<int>(e);
129 assert(value >= 0 && value <= kMaxValue);
130 StateT old_state = state_;
131 state_ |= (StateT{1} << value);
132 return old_state != state_;
133 }
134
135 // Modifies the set (if needed) not to include the given value. Returns true
136 // iff the set was modified.
137 bool Remove(const ENUM_TYPE e) {
138 int value = static_cast<int>(e);
139 assert(value >= 0 && value <= kMaxValue);
140 StateT old_state = state_;
141 state_ &= ~(StateT{1} << value);
142 return old_state != state_;
143 }
144
145 // applicative ops
146
147 // Return a new set based on this one with the additional value(s) inserted
148 constexpr SmallEnumSet With(const ENUM_TYPE e) const {
149 int value = static_cast<int>(e);
150 assert(value >= 0 && value <= kMaxValue);
151 return SmallEnumSet(RawStateMarker(), state_ | (StateT{1} << value));
152 }
153 template <class... TRest>
154 constexpr SmallEnumSet With(const ENUM_TYPE e1, const ENUM_TYPE e2,
155 TRest... rest) const {
156 return With(e1).With(e2, rest...);
157 }
158
159 // Return a new set based on this one excluding the given value(s)
160 constexpr SmallEnumSet Without(const ENUM_TYPE e) const {
161 int value = static_cast<int>(e);
162 assert(value >= 0 && value <= kMaxValue);
163 return SmallEnumSet(RawStateMarker(), state_ & ~(StateT{1} << value));
164 }
165 template <class... TRest>
166 constexpr SmallEnumSet Without(const ENUM_TYPE e1, const ENUM_TYPE e2,
167 TRest... rest) const {
168 return Without(e1).Without(e2, rest...);
169 }
170
171 private:
172 int SkipUnset(int pos) const {
173 StateT tmp = state_ >> pos;
174 if (tmp == 0) {
175 return kStateBits;
176 } else {
177 return pos + detail::CountTrailingZeroBitsForSmallEnumSet(tmp);
178 }
179 }
180 struct RawStateMarker {};
181 explicit SmallEnumSet(RawStateMarker, StateT state) : state_(state) {}
182
183 StateT state_;
184};
