19// A internal wrapper class with an interface similar to Iterator that caches
20// the valid() and key() results for an underlying iterator.
21// This can help avoid virtual function calls and also gives better
22// cache locality.
23template <class TValue = Slice>
24class IteratorWrapperBase {
25 public:
26 IteratorWrapperBase() : iter_(nullptr), valid_(false) {}
27 explicit IteratorWrapperBase(InternalIteratorBase<TValue>* _iter)
28 : iter_(nullptr) {
29 Set(_iter);
30 }
31 ~IteratorWrapperBase() {}
32 InternalIteratorBase<TValue>* iter() const { return iter_; }
33 void SetRangeDelReadSeqno(SequenceNumber read_seqno) {
34 if (iter_) {
35 iter_->SetRangeDelReadSeqno(read_seqno);
36 }
37 }
38
39 // Set the underlying Iterator to _iter and return
40 // previous underlying Iterator.
41 InternalIteratorBase<TValue>* Set(InternalIteratorBase<TValue>* _iter) {
42 InternalIteratorBase<TValue>* old_iter = iter_;
43
44 iter_ = _iter;
45 if (iter_ == nullptr) {
46 valid_ = false;
47 } else {
48 Update();
49 }
50 return old_iter;
51 }
52
53 void DeleteIter(bool is_arena_mode) {
54 if (iter_) {
55#ifdef ROCKSDB_ASSERT_STATUS_CHECKED
56 if (!status_checked_after_invalid_) {
57 // If this assertion fails, it is likely that you did not check
58 // iterator status after Valid() returns false.
59 fprintf(stderr,
60 "Failed to check status after Valid() returned false from this "
61 "iterator.\n");
62 port::PrintStack();
63 std::abort();
64 }
65#endif
66 if (!is_arena_mode) {
67 delete iter_;
68 } else {
69 iter_->~InternalIteratorBase<TValue>();
70 }
71 }
72 }
73
74 // Iterator interface methods
75 bool Valid() const {
76#ifdef ROCKSDB_ASSERT_STATUS_CHECKED
77 status_checked_after_invalid_ = valid_;
78#endif
79 return valid_;
80 }
81 Slice key() const {
82 assert(Valid());
83 return result_.key;
84 }
85
86 uint64_t write_unix_time() const {
87 assert(Valid());
88 return iter_->write_unix_time();
89 }
90
91 TValue value() const {
92 assert(Valid());
93 return iter_->value();
94 }
95 // Methods below require iter() != nullptr
96 Status status() const {
97#ifdef ROCKSDB_ASSERT_STATUS_CHECKED
98 status_checked_after_invalid_ = true;
99#endif
100 assert(iter_);
101 return iter_->status();
102 }
103 bool PrepareValue() {
104 assert(Valid());
105 if (result_.value_prepared) {
106 return true;
107 }
108 if (iter_->PrepareValue()) {
109 result_.value_prepared = true;
110 result_.key = iter_->key();
111 return true;
112 }
113
114 assert(!iter_->Valid());
115 valid_ = false;
116 return false;
117 }
118 void Next() {
119 assert(iter_);
120 valid_ = iter_->NextAndGetResult(&result_);
121 assert(!valid_ || iter_->status().ok());
122 }
123 bool NextAndGetResult(IterateResult* result) {
124 assert(iter_);
125 valid_ = iter_->NextAndGetResult(&result_);
126 *result = result_;
127 assert(!valid_ || iter_->status().ok());
128 return valid_;
129 }
130 void Prev() {
131 assert(iter_);
132 iter_->Prev();
133 Update();
134 }
135 void Seek(const Slice& k) {
136 assert(iter_);
137 iter_->Seek(k);
138 Update();
139 }
140 void SeekForPrev(const Slice& k) {
141 assert(iter_);
142 iter_->SeekForPrev(k);
143 Update();
144 }
145 void SeekToFirst() {
146 assert(iter_);
147 iter_->SeekToFirst();
148 Update();
149 }
150 void SeekToLast() {
151 assert(iter_);
152 iter_->SeekToLast();
153 Update();
154 }
155
156 bool MayBeOutOfLowerBound() {
157 assert(Valid());
158 return iter_->MayBeOutOfLowerBound();
159 }
160
161 IterBoundCheck UpperBoundCheckResult() {
162 assert(Valid());
163 return result_.bound_check_result;
164 }
165
166 void SetPinnedItersMgr(PinnedIteratorsManager* pinned_iters_mgr) {
167 assert(iter_);
168 iter_->SetPinnedItersMgr(pinned_iters_mgr);
169 }
170 bool IsKeyPinned() const {
171 assert(Valid());
172 return iter_->IsKeyPinned();
173 }
174 bool IsValuePinned() const {
175 assert(Valid());
176 return iter_->IsValuePinned();
177 }
178
179 bool IsValuePrepared() const { return result_.value_prepared; }
180
181 Slice user_key() const {
182 assert(Valid());
183 return iter_->user_key();
184 }
185
186 void UpdateReadaheadState(InternalIteratorBase<TValue>* old_iter) {
187 if (old_iter && iter_) {
188 ReadaheadFileInfo readahead_file_info;
189 old_iter->GetReadaheadState(&readahead_file_info);
190 iter_->SetReadaheadState(&readahead_file_info);
191 }
192 }
193
194 bool IsDeleteRangeSentinelKey() const {
195 return iter_->IsDeleteRangeSentinelKey();
196 }
197
198 private:
199 void Update() {
200 valid_ = iter_->Valid();
201 if (valid_) {
202 assert(iter_->status().ok());
203 result_.key = iter_->key();
204 result_.bound_check_result = IterBoundCheck::kUnknown;
205 result_.value_prepared = false;
206 }
207 }
208
209 InternalIteratorBase<TValue>* iter_;
210 IterateResult result_;
211 bool valid_;
212
213#ifdef ROCKSDB_ASSERT_STATUS_CHECKED
214 mutable bool status_checked_after_invalid_ = true;
215#endif
216};
