5/*
6 A wrapper around ROCKSDB_NAMESPACE::TransactionDBMutexFactory-provided
7 condition and mutex that provides toku_pthread_*-like interface. The functions
8 are named
9
10 toku_external_{mutex|cond}_XXX
11
12 Lock Tree uses this mutex and condition for interruptible (long) lock waits.
13
14 (It also still uses toku_pthread_XXX calls for mutexes/conditions for
15 shorter waits on internal objects)
16*/
17

54inline int toku_external_cond_timedwait(toku_external_cond_t *cond,
55 toku_external_mutex_t *mutex,
56 int64_t timeout_microsec) {
57 auto res = (*cond)->WaitFor(*mutex, timeout_microsec);
58 if (res.ok())
59 return 0;
60 else
61 return ETIMEDOUT;
62}

64inline void toku_external_mutex_init(toku_external_mutex_factory_t factory,
65 toku_external_mutex_t *mutex) {
66 // Use placement new: the memory has been allocated but constructor wasn't
67 // called
68 new (mutex) toku_external_mutex_t;
69 *mutex = factory->AllocateMutex();
70}

81inline void toku_external_mutex_unlock(toku_external_mutex_t *mutex) {

83}
