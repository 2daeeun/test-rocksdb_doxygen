10#define __PPC_RA(a) (((a)&0x1f) << 16)
11#define __PPC_RB(b) (((b)&0x1f) << 11)
12#define __PPC_XA(a) ((((a)&0x1f) << 16) | (((a)&0x20) >> 3))
13#define __PPC_XB(b) ((((b)&0x1f) << 11) | (((b)&0x20) >> 4))
14#define __PPC_XS(s) ((((s)&0x1f) << 21) | (((s)&0x20) >> 5))
15#define __PPC_XT(s) __PPC_XS(s)
16#define VSX_XX3(t, a, b) (__PPC_XT(t) | __PPC_XA(a) | __PPC_XB(b))
17#define VSX_XX1(s, a, b) (__PPC_XS(s) | __PPC_RA(a) | __PPC_RB(b))
