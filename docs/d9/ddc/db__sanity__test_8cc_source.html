25class SanityTest {
26 public:
27 explicit SanityTest(const std::string& path)
28 : env_(Env::Default()), path_(path) {
29 env_->CreateDirIfMissing(path);
30 }
31 virtual ~SanityTest() {}
32
33 virtual std::string Name() const = 0;
34 virtual Options GetOptions() const = 0;
35
36 Status Create() {
37 Options options = GetOptions();
38 options.create_if_missing = true;
39 std::string dbname = path_ + Name();
40 Status s = DestroyDB(dbname, options);
41 if (!s.ok()) {
42 return s;
43 }
44 DB* db = nullptr;
45 s = DB::Open(options, dbname, &db);
46 std::unique_ptr<DB> db_guard(db);
47 if (!s.ok()) {
48 return s;
49 }
50 for (int i = 0; i < 1000000; ++i) {
51 std::string k = "key" + std::to_string(i);
52 std::string v = "value" + std::to_string(i);
53 s = db->Put(WriteOptions(), Slice(k), Slice(v));
54 if (!s.ok()) {
55 return s;
56 }
57 }
58 return db->Flush(FlushOptions());
59 }
60 Status Verify() {
61 DB* db = nullptr;
62 std::string dbname = path_ + Name();
63 Status s = DB::Open(GetOptions(), dbname, &db);
64 std::unique_ptr<DB> db_guard(db);
65 if (!s.ok()) {
66 return s;
67 }
68 for (int i = 0; i < 1000000; ++i) {
69 std::string k = "key" + std::to_string(i);
70 std::string v = "value" + std::to_string(i);
71 std::string result;
72 s = db->Get(ReadOptions(), Slice(k), &result);
73 if (!s.ok()) {
74 return s;
75 }
76 if (result != v) {
77 return Status::Corruption("Unexpected value for key " + k);
78 }
79 }
80 return Status::OK();
81 }
82
83 private:
84 Env* env_;
85 std::string const path_;
86};

228bool RunSanityTests(const std::string& command, const std::string& path) {
229 bool result = true;
230// Suppress false positive clang static anaylzer warnings.
231#ifndef __clang_analyzer__
232 std::vector<SanityTest*> sanity_tests = {
233 new SanityTestBasic(path),
234 new SanityTestSpecialComparator(path),
235 new SanityTestZlibCompression(path),
236 new SanityTestZlibCompressionVersion2(path),
237 new SanityTestLZ4Compression(path),
238 new SanityTestLZ4HCCompression(path),
239 new SanityTestZSTDCompression(path),
240 new SanityTestPlainTableFactory(path),
241 new SanityTestBloomFilter(path)};
242
243 if (command == "create") {
244 fprintf(stderr, "Creating...\n");
245 } else {
246 fprintf(stderr, "Verifying...\n");
247 }
248 for (auto sanity_test : sanity_tests) {
249 Status s;
250 fprintf(stderr, "%s -- ", sanity_test->Name().c_str());
251 if (command == "create") {
252 s = sanity_test->Create();
253 } else {
254 assert(command == "verify");
255 s = sanity_test->Verify();
256 }
257 fprintf(stderr, "%s\n", s.ToString().c_str());
258 if (!s.ok()) {
259 fprintf(stderr, "FAIL\n");
260 result = false;
261 }
262
263 delete sanity_test;
264 }
265#endif // __clang_analyzer__
266 return result;
267}

272int main(int argc, char** argv) {
273 std::string path, command;
274 bool ok = (argc == 3);
275 if (ok) {
276 path = std::string(argv[1]);
277 command = std::string(argv[2]);
278 ok = (command == "create" || command == "verify");
279 }
280 if (!ok) {
281 fprintf(stderr, "Usage: %s <path> [create|verify] \n", argv[0]);
282 exit(1);
283 }
284 if (path.back() != '/') {
285 path += "/";
286 }
287
288 bool sanity_ok = ROCKSDB_NAMESPACE::RunSanityTests(command, path);
289
290 return sanity_ok ? 0 : 1;
291}
