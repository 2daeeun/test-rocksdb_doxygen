60 void GenerateIOTrace() {
61 WriteOptions write_opt;
62 TraceOptions trace_opt;
63 std::unique_ptr<TraceWriter> trace_writer;
64
65 ASSERT_OK(NewFileTraceWriter(env_, env_options_, trace_file_path_,
66 &trace_writer));
67
68 ASSERT_OK(db_->StartIOTrace(trace_opt, std::move(trace_writer)));
69
70 for (int i = 0; i < 10; i++) {
71 ASSERT_OK(db_->Put(write_opt, "key_" + std::to_string(i),
72 "value_" + std::to_string(i)));
73 ASSERT_OK(db_->Flush(FlushOptions()));
74 }
75
76 ASSERT_OK(db_->EndIOTrace());
77 ASSERT_OK(env_->FileExists(trace_file_path_));
78 }

80 void RunIOTracerParserTool() {

82 "-io_trace_file=" + trace_file_path_};
83
84 char arg_buffer[kArgBufferSize];

86 int argc = 0;
87 int cursor = 0;

90 ASSERT_LE(argc + 1, kMaxArgCount);
91
92 snprintf(arg_buffer + cursor, arg.size() + 1, "%s", arg.c_str());
93
94 argv[argc++] = arg_buffer + cursor;
95 cursor += static_cast<int>(arg.size()) + 1;
96 }
97 ASSERT_EQ(0, ROCKSDB_NAMESPACE::io_tracer_parser(argc, argv));
98 }

134TEST_F(IOTracerParserTest, NoRecordingAfterEndIOTrace) {
135 uint64_t file_size = 0;
136 // Generate IO trace records and parse them.
137 {
138 GenerateIOTrace();
139 RunIOTracerParserTool();
140 ASSERT_OK(env_->GetFileSize(trace_file_path_, &file_size));
141 }
142 // Once DB::EndIOTrace is invoked in GenerateIOTrace(), no new records should
143 // be appended.
144 {
145 WriteOptions write_opt;
146 for (int i = 10; i < 20; i++) {
147 ASSERT_OK(db_->Put(write_opt, "key_" + std::to_string(i),
148 "value_" + std::to_string(i)));
149 ASSERT_OK(db_->Flush(FlushOptions()));
150 }
151 }
152
153 uint64_t new_file_size = 0;
154 ASSERT_OK(env_->GetFileSize(trace_file_path_, &new_file_size));
155 ASSERT_EQ(file_size, new_file_size);
156}
