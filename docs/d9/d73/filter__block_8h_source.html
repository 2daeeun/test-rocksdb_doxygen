40// A FilterBlockBuilder is used to construct all of the filters for a
41// particular Table. It generates a single string which is stored as
42// a special block in the Table, or partitioned into smaller filters.
43//
44// The sequence of calls to FilterBlockBuilder must match the regexp:
45// Add* Finish
46class FilterBlockBuilder {
47 public:
48 explicit FilterBlockBuilder() {}
49 // No copying allowed
50 FilterBlockBuilder(const FilterBlockBuilder&) = delete;
51 void operator=(const FilterBlockBuilder&) = delete;
52
53 virtual ~FilterBlockBuilder() {}
54
55 // Add a key to current filter.
56 virtual void Add(const Slice& key_without_ts) = 0;
57 // A potentially more efficient version of Add(), though you cannot go back
58 // to Add() after using AddWithPrevKey() on a FilterBlockBuilder.
59 // prev_key_without_ts should be the empty Slice for the first key added
60 // (regardless of comparator; e.g. for bootstrapping delta encoding).
61 // More detail: The previous key is used when filters are key-range
62 // partitioned, and the PartitionedFilterBlockBuilder doesn't need to buffer
63 // the previous key when it is provided by calling this function.
64 virtual void AddWithPrevKey(const Slice& key_without_ts,
65 const Slice& /*prev_key_without_ts*/) = 0;
66
67 virtual bool IsEmpty() const = 0; // Empty == none added
68 // For reporting stats on how many entries the builder considered unique
69 virtual size_t EstimateEntriesAdded() = 0;
70
71 // When using AddWithPrevKey, this must be called before Finish(). (May also
72 // be called without AddWithPrevKey, but prev_key_without_ts must be
73 // accurate regardless.)
74 virtual void PrevKeyBeforeFinish(const Slice& /*prev_key_without_ts*/) {}
75
76 // Generate a filter block. Returns OK if finished, or Incomplete if more
77 // filters are needed (partitioned filter). In the latter case, subsequent
78 // calls require the BlockHandle of the most recently generated and written
79 // filter, in last_partition_block_handle.
80 //
81 // If filter_owner is not nullptr, Finish() may transfer ownership of
82 // underlying filter data to the caller, so that it can be freed as soon as
83 // possible. BlockBasedFilterBlock will ignore this parameter.
84 //
85 // For either OK or Incomplete, *filter is set to point to the next filter
86 // bytes, which survive until either this is destroyed, *filter_owner is
87 // destroyed, or next call to Finish.
88 virtual Status Finish(
89 const BlockHandle& last_partition_block_handle, Slice* filter,
90 std::unique_ptr<const char[]>* filter_owner = nullptr) = 0;
91
92 // This is called when finishes using the FilterBitsBuilder
93 // in order to release memory usage and cache charge
94 // associated with it timely
95 virtual void ResetFilterBitsBuilder() {}
96
97 // To optionally post-verify the filter returned from
98 // FilterBlockBuilder::Finish.
99 // Return Status::OK() if skipped.
100 virtual Status MaybePostVerifyFilter(const Slice& /* filter_content */) {
101 return Status::OK();
102 }
103
104#ifndef NDEBUG
105 Slice TEST_Finish() { // Generate Filter
106 const BlockHandle empty_handle;
107 Slice filter;
108 Status status = Finish(empty_handle, &filter);
109 assert(status.ok());
110 return filter;
111 }
112#endif // NDEBUG
113};

115// A FilterBlockReader is used to parse filter from SST table.
116// KeyMayMatch and PrefixMayMatch would trigger filter checking
117//
118// BlockBased/Full FilterBlock would be called in the same way.
119class FilterBlockReader {
120 public:
121 FilterBlockReader() = default;
122 virtual ~FilterBlockReader() = default;
123
124 FilterBlockReader(const FilterBlockReader&) = delete;
125 FilterBlockReader& operator=(const FilterBlockReader&) = delete;
126
133 virtual bool KeyMayMatch(const Slice& key, const Slice* const const_ikey_ptr,
134 GetContext* get_context,
135 BlockCacheLookupContext* lookup_context,
136 const ReadOptions& read_options) = 0;
137
138 virtual void KeysMayMatch(MultiGetRange* range,
139 BlockCacheLookupContext* lookup_context,
140 const ReadOptions& read_options) {
141 for (auto iter = range->begin(); iter != range->end(); ++iter) {
142 const Slice ukey_without_ts = iter->ukey_without_ts;
143 const Slice ikey = iter->ikey;
144 GetContext* const get_context = iter->get_context;
145 if (!KeyMayMatch(ukey_without_ts, &ikey, get_context, lookup_context,
146 read_options)) {
147 range->SkipKey(iter);
148 }
149 }
150 }
151
155 virtual bool PrefixMayMatch(const Slice& prefix,
156 const Slice* const const_ikey_ptr,
157 GetContext* get_context,
158 BlockCacheLookupContext* lookup_context,
159 const ReadOptions& read_options) = 0;
160
161 virtual void PrefixesMayMatch(MultiGetRange* range,
162 const SliceTransform* prefix_extractor,
163 BlockCacheLookupContext* lookup_context,
164 const ReadOptions& read_options) {
165 for (auto iter = range->begin(); iter != range->end(); ++iter) {
166 const Slice ukey_without_ts = iter->ukey_without_ts;
167 const Slice ikey = iter->ikey;
168 GetContext* const get_context = iter->get_context;
169 if (prefix_extractor->InDomain(ukey_without_ts) &&
170 !PrefixMayMatch(prefix_extractor->Transform(ukey_without_ts), &ikey,
171 get_context, lookup_context, read_options)) {
172 range->SkipKey(iter);
173 }
174 }
175 }
176
177 virtual size_t ApproximateMemoryUsage() const = 0;
178
179 // convert this object to a human readable form
180 virtual std::string ToString() const {
181 std::string error_msg("Unsupported filter \n");
182 return error_msg;
183 }
184
185 virtual Status CacheDependencies(
186 const ReadOptions& /*ro*/, bool /*pin*/,
187 FilePrefetchBuffer* /* tail_prefetch_buffer */) {
188 return Status::OK();
189 }
190
191 virtual void EraseFromCacheBeforeDestruction(
192 uint32_t /*uncache_aggressiveness*/) {}
193
194 virtual bool RangeMayExist(const Slice* /*iterate_upper_bound*/,
195 const Slice& user_key_without_ts,
196 const SliceTransform* prefix_extractor,
197 const Comparator* /*comparator*/,
198 const Slice* const const_ikey_ptr,
199 bool* filter_checked, bool need_upper_bound_check,
200 BlockCacheLookupContext* lookup_context,
201 const ReadOptions& read_options) = 0;
202};
