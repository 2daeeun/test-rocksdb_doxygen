36 CompactionServiceScheduleResponse Schedule(
37 const CompactionServiceJobInfo& info,
38 const std::string& compaction_service_input) override {
39 InstrumentedMutexLock l(&mutex_);
40 start_info_ = info;
41 assert(info.db_name == db_path_);
42 std::string unique_id = Env::Default()->GenerateUniqueId();
43 jobs_.emplace(unique_id, compaction_service_input);
44 infos_.emplace(unique_id, info);
45 CompactionServiceScheduleResponse response(
46 unique_id, is_override_start_status_
47 ? override_start_status_
48 : CompactionServiceJobStatus::kSuccess);
49 return response;
50 }

52 CompactionServiceJobStatus Wait(const std::string& scheduled_job_id,
53 std::string* result) override {
54 std::string compaction_input;
55 {
56 InstrumentedMutexLock l(&mutex_);
57 auto job_index = jobs_.find(scheduled_job_id);
58 if (job_index == jobs_.end()) {
59 return CompactionServiceJobStatus::kFailure;
60 }
61 compaction_input = std::move(job_index->second);
62 jobs_.erase(job_index);
63
64 auto info_index = infos_.find(scheduled_job_id);
65 if (info_index == infos_.end()) {
66 return CompactionServiceJobStatus::kFailure;
67 }
68 wait_info_ = std::move(info_index->second);
69 infos_.erase(info_index);
70 }
71 if (is_override_wait_status_) {
72 return override_wait_status_;
73 }
74 CompactionServiceOptionsOverride options_override;
75 options_override.env = options_.env;
76 options_override.file_checksum_gen_factory =
77 options_.file_checksum_gen_factory;
78 options_override.comparator = options_.comparator;
79 options_override.merge_operator = options_.merge_operator;
80 options_override.compaction_filter = options_.compaction_filter;
81 options_override.compaction_filter_factory =
82 options_.compaction_filter_factory;
83 options_override.prefix_extractor = options_.prefix_extractor;
84 options_override.table_factory = options_.table_factory;
85 options_override.sst_partitioner_factory = options_.sst_partitioner_factory;
86 options_override.statistics = statistics_;
87 if (!listeners_.empty()) {
88 options_override.listeners = listeners_;
89 }
90
91 if (!table_properties_collector_factories_.empty()) {
92 options_override.table_properties_collector_factories =
93 table_properties_collector_factories_;
94 }
95
96 OpenAndCompactOptions options;
97 options.canceled = &canceled_;
98
99 Status s =
100 DB::OpenAndCompact(options, db_path_, db_path_ + "/" + scheduled_job_id,
101 compaction_input, result, options_override);
102 {
103 InstrumentedMutexLock l(&mutex_);
104 if (is_override_wait_result_) {
105 *result = override_wait_result_;
106 }
107 result_ = *result;
108 }
109 compaction_num_.fetch_add(1);
110 if (s.ok()) {
111 return CompactionServiceJobStatus::kSuccess;
112 } else {
113 return CompactionServiceJobStatus::kFailure;
114 }
115 }

293 ASSERT_GE(primary_statistics->getTickerCount(REMOTE_COMPACT_READ_BYTES), 1);
294 ASSERT_GT(primary_statistics->getTickerCount(COMPACT_READ_BYTES),
295 primary_statistics->getTickerCount(REMOTE_COMPACT_READ_BYTES));
296 // compactor is already the remote side, which doesn't have remote

298 ASSERT_EQ(compactor_statistics->getTickerCount(REMOTE_COMPACT_WRITE_BYTES),

320 break;

324 break;

328 break;

370 ASSERT_TRUE(result.stats.is_manual_compaction);
371 ASSERT_FALSE(result.stats.is_full_compaction);

374}
