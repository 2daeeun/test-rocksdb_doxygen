22static int RegisterBuiltinAllocators(ObjectLibrary& library,
23 const std::string& /*arg*/) {
24 library.AddFactory<MemoryAllocator>(
25 DefaultMemoryAllocator::kClassName(),
26 [](const std::string& /*uri*/, std::unique_ptr<MemoryAllocator>* guard,
27 std::string* /*errmsg*/) {
28 guard->reset(new DefaultMemoryAllocator());
29 return guard->get();
30 });
31 library.AddFactory<MemoryAllocator>(
32 CountedMemoryAllocator::kClassName(),
33 [](const std::string& /*uri*/, std::unique_ptr<MemoryAllocator>* guard,
34 std::string* /*errmsg*/) {
35 guard->reset(new CountedMemoryAllocator(
36 std::make_shared<DefaultMemoryAllocator>()));
37 return guard->get();
38 });
39 library.AddFactory<MemoryAllocator>(
40 JemallocNodumpAllocator::kClassName(),
41 [](const std::string& /*uri*/, std::unique_ptr<MemoryAllocator>* guard,
42 std::string* errmsg) {
43 if (JemallocNodumpAllocator::IsSupported(errmsg)) {
44 JemallocAllocatorOptions options;
45 guard->reset(new JemallocNodumpAllocator(options));
46 }
47 return guard->get();
48 });
49 library.AddFactory<MemoryAllocator>(
50 MemkindKmemAllocator::kClassName(),
51 [](const std::string& /*uri*/, std::unique_ptr<MemoryAllocator>* guard,
52 std::string* errmsg) {
53 if (MemkindKmemAllocator::IsSupported(errmsg)) {
54 guard->reset(new MemkindKmemAllocator());
55 }
56 return guard->get();
57 });
58 size_t num_types;
59 return static_cast<int>(library.GetFactoryCount(&num_types));
60}
61} // namespace

69Status MemoryAllocator::CreateFromString(

71 std::shared_ptr<MemoryAllocator>* result) {
72 static std::once_flag once;
73 std::call_once(once, [&]() {
74 RegisterBuiltinAllocators(*(ObjectLibrary::Default().get()), "");
75 });
76 ConfigOptions copy = options;
77 copy.invoke_prepare_options = true;
78 return LoadManagedObject<MemoryAllocator>(copy, value, result);
