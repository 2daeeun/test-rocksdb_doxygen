23// These functions are used to create the various environments under which this
24// test can execute. These functions are used to allow the test cases to be
25// created without the Env being initialized, thereby eliminating a potential
26// static initialization fiasco/race condition when attempting to get a
27// custom/configured env prior to main being invoked.

29static Env* GetDefaultEnv() { return Env::Default(); }

31static Env* GetMockEnv() {
32 static std::unique_ptr<Env> mock_env(MockEnv::Create(Env::Default()));
33 return mock_env.get();

35static Env* NewTestEncryptedEnv(Env* base, const std::string& provider_id) {
36 ConfigOptions config_opts;
37 config_opts.invoke_prepare_options = false;
38
39 std::shared_ptr<EncryptionProvider> provider;
40 EXPECT_OK(EncryptionProvider::CreateFromString(config_opts, provider_id,
41 &provider));
42 return NewEncryptedEnv(base, provider);
43}

56static Env* GetTestEnv() {
57 static std::shared_ptr<Env> env_guard;
58 static Env* custom_env = nullptr;
59 if (custom_env == nullptr) {
60 const char* env_uri = getenv("TEST_ENV_URI");
61 if (env_uri != nullptr) {
62 EXPECT_OK(Env::CreateFromUri(ConfigOptions(), env_uri, /*fs_uri=*/"",
63 &custom_env, &env_guard));
64 }
65 }
66 EXPECT_NE(custom_env, nullptr);
67 return custom_env;
68}

149TEST_P(EnvBasicTestWithParam, Basics) {
150 uint64_t file_size;
151 std::unique_ptr<WritableFile> writable_file;
152 std::vector<std::string> children;
153
154 // Check that the directory is empty.
155 ASSERT_EQ(Status::NotFound(), env_->FileExists(test_dir_ + "/non_existent"));
156 ASSERT_TRUE(!env_->GetFileSize(test_dir_ + "/non_existent", &file_size).ok());
157 ASSERT_OK(env_->GetChildren(test_dir_, &children));
158 ASSERT_EQ(0U, children.size());
159
160 // Create a file.
161 ASSERT_OK(env_->NewWritableFile(test_dir_ + "/f", &writable_file, soptions_));
162 ASSERT_OK(writable_file->Close());
163 writable_file.reset();
164
165 // Check that the file exists.
166 ASSERT_OK(env_->FileExists(test_dir_ + "/f"));
167 ASSERT_OK(env_->GetFileSize(test_dir_ + "/f", &file_size));
168 ASSERT_EQ(0U, file_size);
169 ASSERT_OK(env_->GetChildren(test_dir_, &children));
170 ASSERT_EQ(1U, children.size());
171 ASSERT_EQ("f", children[0]);
172 ASSERT_OK(env_->DeleteFile(test_dir_ + "/f"));
173
174 // Write to the file.
175 ASSERT_OK(
176 env_->NewWritableFile(test_dir_ + "/f1", &writable_file, soptions_));
177 ASSERT_OK(writable_file->Append("abc"));
178 ASSERT_OK(writable_file->Close());
179 writable_file.reset();
180 ASSERT_OK(
181 env_->NewWritableFile(test_dir_ + "/f2", &writable_file, soptions_));
182 ASSERT_OK(writable_file->Close());
183 writable_file.reset();
184
185 // Check for expected size.
186 ASSERT_OK(env_->GetFileSize(test_dir_ + "/f1", &file_size));
187 ASSERT_EQ(3U, file_size);
188
189 // Check that renaming works.
190 ASSERT_TRUE(
191 !env_->RenameFile(test_dir_ + "/non_existent", test_dir_ + "/g").ok());
192 ASSERT_OK(env_->RenameFile(test_dir_ + "/f1", test_dir_ + "/g"));
193 ASSERT_EQ(Status::NotFound(), env_->FileExists(test_dir_ + "/f1"));
194 ASSERT_OK(env_->FileExists(test_dir_ + "/g"));
195 ASSERT_OK(env_->GetFileSize(test_dir_ + "/g", &file_size));
196 ASSERT_EQ(3U, file_size);
197
198 // Check that renaming overwriting works
199 ASSERT_OK(env_->RenameFile(test_dir_ + "/f2", test_dir_ + "/g"));
200 ASSERT_OK(env_->GetFileSize(test_dir_ + "/g", &file_size));
201 ASSERT_EQ(0U, file_size);
202
203 // Check that opening non-existent file fails.
204 std::unique_ptr<SequentialFile> seq_file;
205 std::unique_ptr<RandomAccessFile> rand_file;
206 ASSERT_TRUE(!env_->NewSequentialFile(test_dir_ + "/non_existent", &seq_file,
207 soptions_)
208 .ok());
209 ASSERT_TRUE(!seq_file);
210 ASSERT_NOK(env_->NewRandomAccessFile(test_dir_ + "/non_existent", &rand_file,
211 soptions_));
212 ASSERT_TRUE(!rand_file);
213
214 // Check that deleting works.
215 ASSERT_NOK(env_->DeleteFile(test_dir_ + "/non_existent"));
216 ASSERT_OK(env_->DeleteFile(test_dir_ + "/g"));
217 ASSERT_EQ(Status::NotFound(), env_->FileExists(test_dir_ + "/g"));
218 ASSERT_OK(env_->GetChildren(test_dir_, &children));
219 ASSERT_EQ(0U, children.size());
220 Status s = env_->GetChildren(test_dir_ + "/non_existent", &children);
221 ASSERT_TRUE(s.IsNotFound());
222}

224TEST_P(EnvBasicTestWithParam, ReadWrite) {
225 std::unique_ptr<WritableFile> writable_file;
226 std::unique_ptr<SequentialFile> seq_file;
227 std::unique_ptr<RandomAccessFile> rand_file;
228 Slice result;
229 char scratch[100];
230
231 ASSERT_OK(env_->NewWritableFile(test_dir_ + "/f", &writable_file, soptions_));
232 ASSERT_OK(writable_file->Append("hello "));
233 ASSERT_OK(writable_file->Append("world"));
234 ASSERT_OK(writable_file->Close());
235 writable_file.reset();
236
237 // Read sequentially.
238 ASSERT_OK(env_->NewSequentialFile(test_dir_ + "/f", &seq_file, soptions_));
239 ASSERT_OK(seq_file->Read(5, &result, scratch)); // Read "hello".
240 ASSERT_EQ(0, result.compare("hello"));
241 ASSERT_OK(seq_file->Skip(1));
242 ASSERT_OK(seq_file->Read(1000, &result, scratch)); // Read "world".
243 ASSERT_EQ(0, result.compare("world"));
244 ASSERT_OK(seq_file->Read(1000, &result, scratch)); // Try reading past EOF.
245 ASSERT_EQ(0U, result.size());
246 ASSERT_OK(seq_file->Skip(100)); // Try to skip past end of file.
247 ASSERT_OK(seq_file->Read(1000, &result, scratch));
248 ASSERT_EQ(0U, result.size());
249
250 // Random reads.
251 ASSERT_OK(env_->NewRandomAccessFile(test_dir_ + "/f", &rand_file, soptions_));
252 ASSERT_OK(rand_file->Read(6, 5, &result, scratch)); // Read "world".
253 ASSERT_EQ(0, result.compare("world"));
254 ASSERT_OK(rand_file->Read(0, 5, &result, scratch)); // Read "hello".
255 ASSERT_EQ(0, result.compare("hello"));
256 ASSERT_OK(rand_file->Read(10, 100, &result, scratch)); // Read "d".
257 ASSERT_EQ(0, result.compare("d"));
258
259 // Too high offset.
260 ASSERT_TRUE(rand_file->Read(1000, 5, &result, scratch).ok());
261}

274TEST_P(EnvBasicTestWithParam, LargeWrite) {
275 const size_t kWriteSize = 300 * 1024;
276 char* scratch = new char[kWriteSize * 2];
277
278 std::string write_data;
279 for (size_t i = 0; i < kWriteSize; ++i) {
280 write_data.append(1, static_cast<char>(i));
281 }
282
283 std::unique_ptr<WritableFile> writable_file;
284 ASSERT_OK(env_->NewWritableFile(test_dir_ + "/f", &writable_file, soptions_));
285 ASSERT_OK(writable_file->Append("foo"));
286 ASSERT_OK(writable_file->Append(write_data));
287 ASSERT_OK(writable_file->Close());
288 writable_file.reset();
289
290 std::unique_ptr<SequentialFile> seq_file;
291 Slice result;
292 ASSERT_OK(env_->NewSequentialFile(test_dir_ + "/f", &seq_file, soptions_));
293 ASSERT_OK(seq_file->Read(3, &result, scratch)); // Read "foo".
294 ASSERT_EQ(0, result.compare("foo"));
295
296 size_t read = 0;
297 std::string read_data;
298 while (read < kWriteSize) {
299 ASSERT_OK(seq_file->Read(kWriteSize - read, &result, scratch));
300 read_data.append(result.data(), result.size());
301 read += result.size();
302 }
303 ASSERT_TRUE(write_data == read_data);
304 delete[] scratch;
305}

327 // empty folder returns empty vector

329 std::vector<Env::FileAttributes> childAttr;
330 ASSERT_OK(env_->CreateDirIfMissing(test_dir_));
331 ASSERT_OK(env_->GetChildren(test_dir_, &children));
332 ASSERT_OK(env_->FileExists(test_dir_));
333 ASSERT_OK(env_->GetChildrenFileAttributes(test_dir_, &childAttr));
334 ASSERT_EQ(0U, children.size());
335 ASSERT_EQ(0U, childAttr.size());
336
337 // folder with contents returns relative path to test dir
338 ASSERT_OK(env_->CreateDirIfMissing(test_dir_ + "/niu"));
339 ASSERT_OK(env_->CreateDirIfMissing(test_dir_ + "/you"));
340 ASSERT_OK(env_->CreateDirIfMissing(test_dir_ + "/guo"));
341 ASSERT_OK(env_->GetChildren(test_dir_, &children));
342 ASSERT_OK(env_->GetChildrenFileAttributes(test_dir_, &childAttr));
343 ASSERT_EQ(3U, children.size());
344 ASSERT_EQ(3U, childAttr.size());
345 for (const auto& each : children) {
346 env_->DeleteDir(test_dir_ + "/" + each).PermitUncheckedError();
347 } // necessary for default POSIX env
348
349 // non-exist directory returns IOError
350 ASSERT_OK(env_->DeleteDir(test_dir_));
351 ASSERT_NOK(env_->FileExists(test_dir_));
352 ASSERT_NOK(env_->GetChildren(test_dir_, &children));
353 ASSERT_NOK(env_->GetChildrenFileAttributes(test_dir_, &childAttr));
354
355 // if dir is a file, returns IOError
356 ASSERT_OK(env_->CreateDir(test_dir_));
357 std::unique_ptr<WritableFile> writable_file;
358 ASSERT_OK(
359 env_->NewWritableFile(test_dir_ + "/file", &writable_file, soptions_));
360 ASSERT_OK(writable_file->Close());
361 writable_file.reset();
362 ASSERT_NOK(env_->GetChildren(test_dir_ + "/file", &children));
363 ASSERT_EQ(0U, children.size());
