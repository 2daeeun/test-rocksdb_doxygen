16// Iterator that iterates over partitioned index.
17// Some upper and lower bound tricks played in block based table iterators
18// could be played here, but it's too complicated to reason about index
19// keys with upper or lower bound, so we skip it for simplicity.
20class PartitionedIndexIterator : public InternalIteratorBase<IndexValue> {
21 // compaction_readahead_size: its value will only be used if for_compaction =
22 // true
23 public:
24 PartitionedIndexIterator(
25 const BlockBasedTable* table, const ReadOptions& read_options,
26 const InternalKeyComparator& icomp,
27 std::unique_ptr<InternalIteratorBase<IndexValue>>&& index_iter,
28 TableReaderCaller caller, size_t compaction_readahead_size = 0)
29 : index_iter_(std::move(index_iter)),
30 table_(table),
31 read_options_(read_options),
32#ifndef NDEBUG
33 icomp_(icomp),
34#endif
35 user_comparator_(icomp.user_comparator()),
36 block_iter_points_to_real_block_(false),
37 lookup_context_(caller),
38 block_prefetcher_(
39 compaction_readahead_size,
40 table_->get_rep()->table_options.initial_auto_readahead_size) {
41 }
42
43 ~PartitionedIndexIterator() override {}
44
45 void Seek(const Slice& target) override;
46 void SeekForPrev(const Slice&) override {
47 // Shouldn't be called.
48 assert(false);
49 }
50 void SeekToFirst() override;
51 void SeekToLast() override;
52 void Next() final override;
53 bool NextAndGetResult(IterateResult*) override {
54 assert(false);
55 return false;
56 }
57 void Prev() override;
58 bool Valid() const override {
59 return block_iter_points_to_real_block_ && block_iter_.Valid();
60 }
61 Slice key() const override {
62 assert(Valid());
63 return block_iter_.key();
64 }
65 Slice user_key() const override {
66 assert(Valid());
67 return block_iter_.user_key();
68 }
69 IndexValue value() const override {
70 assert(Valid());
71 return block_iter_.value();
72 }
73 Status status() const override {
74 // Prefix index set status to NotFound when the prefix does not exist
75 if (!index_iter_->status().ok() && !index_iter_->status().IsNotFound()) {
76 return index_iter_->status();
77 } else if (block_iter_points_to_real_block_) {
78 return block_iter_.status();
79 } else {
80 return Status::OK();
81 }
82 }
83 inline IterBoundCheck UpperBoundCheckResult() override {
84 // Shouldn't be called.
85 assert(false);
86 return IterBoundCheck::kUnknown;
87 }
88 void SetPinnedItersMgr(PinnedIteratorsManager*) override {
89 // Shouldn't be called.
90 assert(false);
91 }
92 bool IsKeyPinned() const override {
93 // Shouldn't be called.
94 assert(false);
95 return false;
96 }
97 bool IsValuePinned() const override {
98 // Shouldn't be called.
99 assert(false);
100 return false;
101 }
102
103 void ResetPartitionedIndexIter() {
104 if (block_iter_points_to_real_block_) {
105 block_iter_.Invalidate(Status::OK());
106 block_iter_points_to_real_block_ = false;
107 }
108 }
109
110 void SavePrevIndexValue() {
111 if (block_iter_points_to_real_block_) {
112 // Reseek. If they end up with the same data block, we shouldn't re-fetch
113 // the same data block.
114 prev_block_offset_ = index_iter_->value().handle.offset();
115 }
116 }
117
118 void GetReadaheadState(ReadaheadFileInfo* readahead_file_info) override {
119 if (block_prefetcher_.prefetch_buffer() != nullptr &&
120 read_options_.adaptive_readahead) {
121 block_prefetcher_.prefetch_buffer()->GetReadaheadState(
122 &(readahead_file_info->index_block_readahead_info));
123 }
124 }
125
126 void SetReadaheadState(ReadaheadFileInfo* readahead_file_info) override {
127 if (read_options_.adaptive_readahead) {
128 block_prefetcher_.SetReadaheadState(
129 &(readahead_file_info->index_block_readahead_info));
130 }
131 }
132
133 std::unique_ptr<InternalIteratorBase<IndexValue>> index_iter_;
134
135 private:
136 friend class BlockBasedTableReaderTestVerifyChecksum_ChecksumMismatch_Test;
137 const BlockBasedTable* table_;
138 const ReadOptions read_options_;
139#ifndef NDEBUG
140 const InternalKeyComparator& icomp_;
141#endif
142 UserComparatorWrapper user_comparator_;
143 IndexBlockIter block_iter_;
144
145 // True if block_iter_ is initialized and points to the same block
146 // as index iterator.
147 bool block_iter_points_to_real_block_;
148 uint64_t prev_block_offset_ = std::numeric_limits<uint64_t>::max();
149 BlockCacheLookupContext lookup_context_;
150 BlockPrefetcher block_prefetcher_;
151
152 // If `target` is null, seek to first.
153 void SeekImpl(const Slice* target);
154
155 void InitPartitionedIndexBlock();
156 void FindKeyForward();
157 void FindBlockForward();
158 void FindKeyBackward();
159};
160} // namespace ROCKSDB_NAMESPACE
