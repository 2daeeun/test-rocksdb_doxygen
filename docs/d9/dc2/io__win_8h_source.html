25std::string GetWindowsErrSz(DWORD err);
26
27inline IOStatus IOErrorFromWindowsError(const std::string& context, DWORD err) {
28 return ((err == ERROR_HANDLE_DISK_FULL) || (err == ERROR_DISK_FULL))
29 ? IOStatus::NoSpace(context, GetWindowsErrSz(err))
30 : ((err == ERROR_FILE_NOT_FOUND) || (err == ERROR_PATH_NOT_FOUND))
31 ? IOStatus::PathNotFound(context, GetWindowsErrSz(err))
32 : IOStatus::IOError(context, GetWindowsErrSz(err));
33}
34
35inline IOStatus IOErrorFromLastWindowsError(const std::string& context) {
36 return IOErrorFromWindowsError(context, GetLastError());
37}
38
39inline IOStatus IOError(const std::string& context, int err_number) {
40 return (err_number == ENOSPC)
41 ? IOStatus::NoSpace(context, errnoStr(err_number).c_str())
42 : (err_number == ENOENT)
43 ? IOStatus::PathNotFound(context, errnoStr(err_number).c_str())
44 : IOStatus::IOError(context, errnoStr(err_number).c_str());
45}

297// This is a sequential write class. It has been mimicked (as others) after
298// the original Posix class. We add support for unbuffered I/O on windows as
299// well
300// we utilize the original buffer as an alignment buffer to write directly to
301// file with no buffering.
302// No buffering requires that the provided buffer is aligned to the physical
303// sector size (SSD page size) and
304// that all SetFilePointer() operations to occur with such an alignment.
305// We thus always write in sector/page size increments to the drive and leave
306// the tail for the next write OR for Close() at which point we pad with zeros.
307// No padding is required for
308// buffered access.
309class WinWritableImpl {
310 protected:
311 WinFileData* file_data_;
312 const uint64_t alignment_;
313 uint64_t
314 next_write_offset_; // Needed because Windows does not support O_APPEND
315 uint64_t reservedsize_; // how far we have reserved space
316
317 virtual IOStatus PreallocateInternal(uint64_t spaceToReserve);
318
319 WinWritableImpl(WinFileData* file_data, size_t alignment);
320
321 ~WinWritableImpl() {}
322
323 uint64_t GetAlignment() const { return alignment_; }
324
325 IOStatus AppendImpl(const Slice& data);
326
327 // Requires that the data is aligned as specified by
328 // GetRequiredBufferAlignment()
329 IOStatus PositionedAppendImpl(const Slice& data, uint64_t offset);
330
331 IOStatus TruncateImpl(uint64_t size);
332
333 IOStatus CloseImpl();
334
335 IOStatus SyncImpl(const IOOptions& options, IODebugContext* dbg);
336
337 uint64_t GetFileNextWriteOffset() {
338 // Double accounting now here with WritableFileWriter
339 // and this size will be wrong when unbuffered access is used
340 // but tests implement their own writable files and do not use
341 // WritableFileWrapper
342 // so we need to squeeze a square peg through
343 // a round hole here.
344 return next_write_offset_;
345 }
346
347 IOStatus AllocateImpl(uint64_t offset, uint64_t len);
348
349 public:
350 WinWritableImpl(const WinWritableImpl&) = delete;
351 WinWritableImpl& operator=(const WinWritableImpl&) = delete;
352};
