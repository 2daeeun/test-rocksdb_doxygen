119int toku_keycompare(const void *key1, size_t key1len, const void *key2,
120 size_t key2len) {
121 size_t comparelen = key1len < key2len ? key1len : key2len;
122 int c = memcmp(key1, key2, comparelen);
123 if (__builtin_expect(c != 0, 1)) {
124 return c;
125 } else {
126 if (key1len < key2len) {
127 return -1;
128 } else if (key1len > key2len) {
129 return 1;
130 } else {
131 return 0;
132 }
133 }
134}
