21// A handle for lookup result. Immediately after SecondaryCache::Lookup() with
22// wait=false (and depending on the implementation), the handle could be in any
23// of the below states. It must not be destroyed while in the pending state.
24// * Pending state (IsReady() == false): result is not ready. Value() and Size()
25// must not be called.
26// * Ready + not found state (IsReady() == true, Value() == nullptr): the lookup
27// has completed, finding no match. Or an error occurred that prevented
28// normal completion of the Lookup.
29// * Ready + found state (IsReady() == false, Value() != nullptr): the lookup
30// has completed, finding an entry that has been loaded into an object that is
31// now owned by the caller.
32//
33// Wait() or SecondaryCache::WaitAll() may be skipped if IsReady() happens to
34// return true, but (depending on the implementation) IsReady() might never
35// return true without Wait() or SecondaryCache::WaitAll(). After the handle
36// is known ready, calling Value() is required to avoid a memory leak in case
37// of a cache hit.
38class SecondaryCacheResultHandle {
39 public:
40 virtual ~SecondaryCacheResultHandle() = default;
41
42 // Returns whether the handle is ready or not
43 virtual bool IsReady() = 0;
44
45 // Block until handle becomes ready
46 virtual void Wait() = 0;
47
48 // Return the cache entry object (also known as value). If nullptr, it means
49 // the lookup was unsuccessful.
50 virtual Cache::ObjectPtr Value() = 0;
51
52 // Return the out_charge from the helper->create_cb used to construct the
53 // object.
54 // WART: potentially confusing name
55 virtual size_t Size() = 0;
56};

58// SecondaryCache
59//
60// Cache interface for caching blocks on a secondary tier (which can include
61// non-volatile media, or alternate forms of caching such as compressed data)
62//
63// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
64// because RocksDB is not exception-safe. This could cause undefined behavior
65// including data loss, unreported corruption, deadlocks, and more.
66class SecondaryCache : public Customizable {
67 public:
68 ~SecondaryCache() override = default;
69
70 static const char* Type() { return "SecondaryCache"; }
71 static Status CreateFromString(const ConfigOptions& config_options,
72 const std::string& id,
73 std::shared_ptr<SecondaryCache>* result);
74
75 // Suggest inserting an entry into this cache. The caller retains ownership
76 // of `obj` (also called the "value"), so is only used directly by the
77 // SecondaryCache during Insert(). When the cache chooses to perform the
78 // suggested insertion, it uses the size_cb and saveto_cb provided by
79 // `helper` to extract the persistable data (typically an uncompressed block)
80 // and writes it to this cache tier. OK may be returned even if the insertion
81 // is not made.
82 virtual Status Insert(const Slice& key, Cache::ObjectPtr obj,
83 const Cache::CacheItemHelper* helper,
84 bool force_insert) = 0;
85
86 // Insert a value from its saved/persistable data (typically uncompressed
87 // block), as if generated by SaveToCallback/SizeCallback. The data can be
88 // compressed, in which case the type argument should specify the
89 // compression algorithm used. Additionally, the source argument should
90 // be set to the appropriate tier that will be responsible for
91 // uncompressing the data.
92 //
93 // This method can be used in "warming up" the cache from some auxiliary
94 // source, and like Insert() may or may not write it to cache depending on
95 // the admission control policy, even if the return status is success.
96 virtual Status InsertSaved(
97 const Slice& key, const Slice& saved,
98 CompressionType type = CompressionType::kNoCompression,
99 CacheTier source = CacheTier::kVolatileTier) = 0;
100
101 // Lookup the data for the given key in this cache. The create_cb
102 // will be used to create the object. The handle returned may not be
103 // ready yet, unless wait=true, in which case Lookup() will block until
104 // the handle is ready.
105 //
106 // advise_erase is a hint from the primary cache indicating that the handle
107 // will be cached there, so the secondary cache is advised to drop it from
108 // the cache as an optimization. To use this feature, SupportForceErase()
109 // needs to return true.
110 // This hint can also be safely ignored.
111 //
112 // kept_in_sec_cache is to indicate whether the entry will be kept in the
113 // secondary cache after the Lookup (rather than erased because of Lookup)
114 virtual std::unique_ptr<SecondaryCacheResultHandle> Lookup(
115 const Slice& key, const Cache::CacheItemHelper* helper,
116 Cache::CreateContext* create_context, bool wait, bool advise_erase,
117 Statistics* stats, bool& kept_in_sec_cache) = 0;
118
119 // Indicate whether a handle can be erased in this secondary cache.
120 [[nodiscard]] virtual bool SupportForceErase() const = 0;
121
122 // At the discretion of the implementation, erase the data associated
123 // with key.
124 virtual void Erase(const Slice& key) = 0;
125
126 // Wait for a collection of handles to become ready.
127 virtual void WaitAll(std::vector<SecondaryCacheResultHandle*> handles) = 0;
128
129 // Set the maximum configured capacity of the cache.
130 // When the new capacity is less than the old capacity and the existing usage
131 // is greater than new capacity, the implementation will do its best job to
132 // purge the released entries from the cache in order to lower the usage.
133 //
134 // The derived class can make this function no-op and return NotSupported().
135 virtual Status SetCapacity(size_t /* capacity */) {
136 return Status::NotSupported();
137 }
138
139 // The derived class can make this function no-op and return NotSupported().
140 virtual Status GetCapacity(size_t& /* capacity */) {
141 return Status::NotSupported();
142 }
143
144 // Temporarily decrease the cache capacity in RAM by the specified amount.
145 // The caller should call Inflate() to restore the cache capacity. This is
146 // intended to be lighter weight than SetCapacity(). The latter evenly
147 // distributes the new capacity across all shards and is meant for large
148 // changes in capacity, whereas the former is meant for relatively small
149 // changes and may be uneven by lowering capacity in a single shard.
150 virtual Status Deflate(size_t /*decrease*/) { return Status::NotSupported(); }
151
152 // Restore the capacity reduced by a prior call to Deflate().
153 virtual Status Inflate(size_t /*increase*/) { return Status::NotSupported(); }
154};

156// A wrapper around a SecondaryCache object. A derived class may selectively
157// override methods to implement a different behavior.
158class SecondaryCacheWrapper : public SecondaryCache {
159 public:
160 explicit SecondaryCacheWrapper(std::shared_ptr<SecondaryCache> target)
161 : target_(std::move(target)) {}
162
163 Status Insert(const Slice& key, Cache::ObjectPtr obj,
164 const Cache::CacheItemHelper* helper,
165 bool force_insert) override {
166 return target()->Insert(key, obj, helper, force_insert);
167 }
168
169 Status InsertSaved(const Slice& key, const Slice& saved,
170 CompressionType type = CompressionType::kNoCompression,
171 CacheTier source = CacheTier::kVolatileTier) override {
172 return target()->InsertSaved(key, saved, type, source);
173 }
174
175 std::unique_ptr<SecondaryCacheResultHandle> Lookup(
176 const Slice& key, const Cache::CacheItemHelper* helper,
177 Cache::CreateContext* create_context, bool wait, bool advise_erase,
178 Statistics* stats, bool& kept_in_sec_cache) override {
179 return target()->Lookup(key, helper, create_context, wait, advise_erase,
180 stats, kept_in_sec_cache);
181 }
182
183 bool SupportForceErase() const override {
184 return target()->SupportForceErase();
185 }
186
187 void Erase(const Slice& key) override { target()->Erase(key); }
188
189 void WaitAll(std::vector<SecondaryCacheResultHandle*> handles) override {
190 target()->WaitAll(handles);
191 }
192
193 Status SetCapacity(size_t capacity) override {
194 return target()->SetCapacity(capacity);
195 }
196
197 Status GetCapacity(size_t& capacity) override {
198 return target()->GetCapacity(capacity);
199 }
200
201 Status Deflate(size_t decrease) override {
202 return target()->Deflate(decrease);
203 }
204
205 Status Inflate(size_t increase) override {
206 return target()->Inflate(increase);
207 }
208
209 protected:
210 SecondaryCache* target() const { return target_.get(); }
211
212 private:
213 std::shared_ptr<SecondaryCache> target_;
214};
