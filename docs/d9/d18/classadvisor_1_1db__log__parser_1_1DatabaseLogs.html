99 def trigger_conditions_for_log(self, conditions, log):
100 # For a LogCondition object, trigger is:
101 # Dict[column_family_name, List[Log]]. This explains why the condition
102 # was triggered and for which column families.
103 for cond in conditions:
104 if re.search(cond.regex, log.get_message(), re.IGNORECASE):
105 trigger = cond.get_trigger()
106 if not trigger:
107 trigger = {}
108 if log.get_column_family() not in trigger:
109 trigger[log.get_column_family()] = []
110 trigger[log.get_column_family()].append(log)
111 cond.set_trigger(trigger)
