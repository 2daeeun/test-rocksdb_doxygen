31// Implementations of the DB interface
32Status DBImplReadOnly::GetImpl(const ReadOptions& read_options,
33 const Slice& key,
34 GetImplOptions& get_impl_options) {
35 assert(get_impl_options.value != nullptr ||
36 get_impl_options.columns != nullptr);
37 assert(get_impl_options.column_family);
38
39 Status s;
40
41 if (read_options.timestamp) {
42 s = FailIfTsMismatchCf(get_impl_options.column_family,
43 *(read_options.timestamp));
44 if (!s.ok()) {
45 return s;
46 }
47 } else {
48 s = FailIfCfHasTs(get_impl_options.column_family);
49 if (!s.ok()) {
50 return s;
51 }
52 }
53
54 // Clear the timestamps for returning results so that we can distinguish
55 // between tombstone or key that has never been written
56 if (get_impl_options.timestamp) {
57 get_impl_options.timestamp->clear();
58 }
59
60 PERF_CPU_TIMER_GUARD(get_cpu_nanos, immutable_db_options_.clock);
61 StopWatch sw(immutable_db_options_.clock, stats_, DB_GET);
62 PERF_TIMER_GUARD(get_snapshot_time);
63
64 const Comparator* ucmp = get_impl_options.column_family->GetComparator();
65 assert(ucmp);
66 std::string* ts =
67 ucmp->timestamp_size() > 0 ? get_impl_options.timestamp : nullptr;
68 SequenceNumber snapshot = versions_->LastSequence();
69 GetWithTimestampReadCallback read_cb(snapshot);
70 auto cfh = static_cast_with_check<ColumnFamilyHandleImpl>(
71 get_impl_options.column_family);
72 auto cfd = cfh->cfd();
73 if (tracer_) {
74 InstrumentedMutexLock lock(&trace_mutex_);
75 if (tracer_) {
76 tracer_->Get(get_impl_options.column_family, key);
77 }
78 }
79
80 // In read-only mode Get(), no super version operation is needed (i.e.
81 // GetAndRefSuperVersion and ReturnAndCleanupSuperVersion)
82 SuperVersion* super_version = cfd->GetSuperVersion();
83 if (read_options.timestamp && read_options.timestamp->size() > 0) {
84 s = FailIfReadCollapsedHistory(cfd, super_version,
85 *(read_options.timestamp));
86 if (!s.ok()) {
87 return s;
88 }
89 }
90 MergeContext merge_context;
91 SequenceNumber max_covering_tombstone_seq = 0;
92 LookupKey lkey(key, snapshot, read_options.timestamp);
93 PERF_TIMER_STOP(get_snapshot_time);
94
95 // Look up starts here
96 if (super_version->mem->Get(
97 lkey,
98 get_impl_options.value ? get_impl_options.value->GetSelf() : nullptr,
99 get_impl_options.columns, ts, &s, &merge_context,
100 &max_covering_tombstone_seq, read_options,
101 false /* immutable_memtable */, &read_cb)) {
102 if (get_impl_options.value) {
103 get_impl_options.value->PinSelf();
104 }
105 RecordTick(stats_, MEMTABLE_HIT);
106 } else {
107 PERF_TIMER_GUARD(get_from_output_files_time);
108 PinnedIteratorsManager pinned_iters_mgr;
109 super_version->current->Get(
110 read_options, lkey, get_impl_options.value, get_impl_options.columns,
111 ts, &s, &merge_context, &max_covering_tombstone_seq, &pinned_iters_mgr,
112 /*value_found*/ nullptr,
113 /*key_exists*/ nullptr, /*seq*/ nullptr, &read_cb,
114 /*is_blob*/ nullptr,
115 /*do_merge*/ true);
116 RecordTick(stats_, MEMTABLE_MISS);
117 }
118 {
119 RecordTick(stats_, NUMBER_KEYS_READ);
120 size_t size = 0;
121 if (get_impl_options.value) {
122 size = get_impl_options.value->size();
123 } else if (get_impl_options.columns) {
124 size = get_impl_options.columns->serialized_size();
125 }
126 RecordTick(stats_, BYTES_READ, size);
127 RecordInHistogram(stats_, BYTES_PER_READ, size);
128 PERF_COUNTER_ADD(get_read_bytes, size);
129 }
130 return s;
131}

133Iterator* DBImplReadOnly::NewIterator(const ReadOptions& _read_options,
134 ColumnFamilyHandle* column_family) {
135 if (_read_options.io_activity != Env::IOActivity::kUnknown &&
136 _read_options.io_activity != Env::IOActivity::kDBIterator) {
137 return NewErrorIterator(Status::InvalidArgument(
138 "Can only call NewIterator with `ReadOptions::io_activity` is "
139 "`Env::IOActivity::kUnknown` or `Env::IOActivity::kDBIterator`"));
140 }
141 ReadOptions read_options(_read_options);
142 if (read_options.io_activity == Env::IOActivity::kUnknown) {
143 read_options.io_activity = Env::IOActivity::kDBIterator;
144 }
145 assert(column_family);
146 if (read_options.timestamp) {
147 const Status s =
148 FailIfTsMismatchCf(column_family, *(read_options.timestamp));
149 if (!s.ok()) {
150 return NewErrorIterator(s);
151 }
152 } else {
153 const Status s = FailIfCfHasTs(column_family);
154 if (!s.ok()) {
155 return NewErrorIterator(s);
156 }
157 }
158 auto cfh = static_cast_with_check<ColumnFamilyHandleImpl>(column_family);
159 auto cfd = cfh->cfd();
160 SuperVersion* super_version = cfd->GetSuperVersion()->Ref();
161 if (read_options.timestamp && read_options.timestamp->size() > 0) {
162 const Status s = FailIfReadCollapsedHistory(cfd, super_version,
163 *(read_options.timestamp));
164 if (!s.ok()) {
165 cfd->GetSuperVersion()->Unref();
166 return NewErrorIterator(s);
167 }
168 }
169 SequenceNumber latest_snapshot = versions_->LastSequence();
170 SequenceNumber read_seq =
171 read_options.snapshot != nullptr
172 ? static_cast<const SnapshotImpl*>(read_options.snapshot)->number_
173 : latest_snapshot;
174 ReadCallback* read_callback = nullptr; // No read callback provided.
175 auto db_iter = NewArenaWrappedDbIterator(
176 env_, read_options, *cfd->ioptions(), super_version->mutable_cf_options,
177 super_version->current, read_seq,
178 super_version->mutable_cf_options.max_sequential_skip_in_iterations,
179 super_version->version_number, read_callback);
180 auto internal_iter = NewInternalIterator(
181 db_iter->GetReadOptions(), cfd, super_version, db_iter->GetArena(),
182 read_seq, /* allow_unprepared_value */ true, db_iter);
183 db_iter->SetIterUnderDBIter(internal_iter);
184 return db_iter;
185}

187Status DBImplReadOnly::NewIterators(
188 const ReadOptions& read_options,
189 const std::vector<ColumnFamilyHandle*>& column_families,
190 std::vector<Iterator*>* iterators) {
191 if (read_options.timestamp) {
192 for (auto* cf : column_families) {
193 assert(cf);
194 const Status s = FailIfTsMismatchCf(cf, *(read_options.timestamp));
195 if (!s.ok()) {
196 return s;
197 }
198 }
199 } else {
200 for (auto* cf : column_families) {
201 assert(cf);
202 const Status s = FailIfCfHasTs(cf);
203 if (!s.ok()) {
204 return s;
205 }
206 }
207 }
208
209 ReadCallback* read_callback = nullptr; // No read callback provided.
210 if (iterators == nullptr) {
211 return Status::InvalidArgument("iterators not allowed to be nullptr");
212 }
213 iterators->clear();
214 iterators->reserve(column_families.size());
215 SequenceNumber latest_snapshot = versions_->LastSequence();
216 SequenceNumber read_seq =
217 read_options.snapshot != nullptr
218 ? static_cast<const SnapshotImpl*>(read_options.snapshot)->number_
219 : latest_snapshot;
220
221 autovector<std::tuple<ColumnFamilyData*, SuperVersion*>> cfd_to_sv;
222
223 const bool check_read_ts =
224 read_options.timestamp && read_options.timestamp->size() > 0;
225 for (auto cfh : column_families) {
226 auto* cfd = static_cast_with_check<ColumnFamilyHandleImpl>(cfh)->cfd();
227 auto* sv = cfd->GetSuperVersion()->Ref();
228 cfd_to_sv.emplace_back(cfd, sv);
229 if (check_read_ts) {
230 const Status s =
231 FailIfReadCollapsedHistory(cfd, sv, *(read_options.timestamp));
232 if (!s.ok()) {
233 for (auto prev_entry : cfd_to_sv) {
234 std::get<1>(prev_entry)->Unref();
235 }
236 return s;
237 }
238 }
239 }
240 assert(cfd_to_sv.size() == column_families.size());
241 for (auto [cfd, sv] : cfd_to_sv) {
242 auto* db_iter = NewArenaWrappedDbIterator(
243 env_, read_options, *cfd->ioptions(), sv->mutable_cf_options,
244 sv->current, read_seq,
245 sv->mutable_cf_options.max_sequential_skip_in_iterations,
246 sv->version_number, read_callback);
247 auto* internal_iter = NewInternalIterator(
248 db_iter->GetReadOptions(), cfd, sv, db_iter->GetArena(), read_seq,
249 /* allow_unprepared_value */ true, db_iter);
250 db_iter->SetIterUnderDBIter(internal_iter);
251 iterators->push_back(db_iter);
252 }
253
254 return Status::OK();
255}

260Status OpenForReadOnlyCheckExistence(const DBOptions& db_options,
261 const std::string& dbname) {
262 Status s;
263 if (!db_options.create_if_missing) {
264 // Attempt to read "CURRENT" file
265 const std::shared_ptr<FileSystem>& fs = db_options.env->GetFileSystem();
266 std::string manifest_path;
267 uint64_t manifest_file_number;
268 s = GetCurrentManifestPath(dbname, fs.get(), /*is_retry=*/false,
269 &manifest_path, &manifest_file_number);
270 } else {
271 // Historic behavior that doesn't necessarily make sense
272 s = db_options.env->CreateDirIfMissing(dbname);
273 }
274 return s;
275}

278Status DB::OpenForReadOnly(const Options& options, const std::string& dbname,
279 DB** dbptr, bool /*error_if_wal_file_exists*/) {
280 Status s = OpenForReadOnlyCheckExistence(options, dbname);
281 if (!s.ok()) {
282 return s;
283 }
284
285 *dbptr = nullptr;
286
287 // Try to first open DB as fully compacted DB
288 s = CompactedDBImpl::Open(options, dbname, dbptr);
289 if (s.ok()) {
290 return s;
291 }
292
293 DBOptions db_options(options);
294 ColumnFamilyOptions cf_options(options);
295 std::vector<ColumnFamilyDescriptor> column_families;
296 column_families.emplace_back(kDefaultColumnFamilyName, cf_options);
297 std::vector<ColumnFamilyHandle*> handles;
298
299 s = DBImplReadOnly::OpenForReadOnlyWithoutCheck(
300 db_options, dbname, column_families, &handles, dbptr);
301 if (s.ok()) {
302 assert(handles.size() == 1);
303 // i can delete the handle since DBImpl is always holding a
304 // reference to default column family
305 delete handles[0];
306 }
307 return s;
308}

310Status DB::OpenForReadOnly(
311 const DBOptions& db_options, const std::string& dbname,
312 const std::vector<ColumnFamilyDescriptor>& column_families,
313 std::vector<ColumnFamilyHandle*>* handles, DB** dbptr,
314 bool error_if_wal_file_exists) {
315 // If dbname does not exist in the file system, should not do anything
316 Status s = OpenForReadOnlyCheckExistence(db_options, dbname);
317 if (!s.ok()) {
318 return s;
319 }
320
321 return DBImplReadOnly::OpenForReadOnlyWithoutCheck(
322 db_options, dbname, column_families, handles, dbptr,
323 error_if_wal_file_exists);

326Status DBImplReadOnly::OpenForReadOnlyWithoutCheck(
327 const DBOptions& db_options, const std::string& dbname,
328 const std::vector<ColumnFamilyDescriptor>& column_families,
329 std::vector<ColumnFamilyHandle*>* handles, DB** dbptr,
330 bool error_if_wal_file_exists) {
331 *dbptr = nullptr;
332 handles->clear();
333
334 SuperVersionContext sv_context(/* create_superversion */ true);
335 DBImplReadOnly* impl = new DBImplReadOnly(db_options, dbname);
336 impl->mutex_.Lock();
337 Status s = impl->Recover(column_families, true /* read only */,
338 error_if_wal_file_exists);
339 if (s.ok()) {
340 // set column family handles
341 for (const auto& cf : column_families) {
342 auto cfd =
343 impl->versions_->GetColumnFamilySet()->GetColumnFamily(cf.name);
344 if (cfd == nullptr) {
345 s = Status::InvalidArgument("Column family not found", cf.name);
346 break;
347 }
348 handles->push_back(new ColumnFamilyHandleImpl(cfd, impl, &impl->mutex_));
349 }
350 }
351 if (s.ok()) {
352 for (auto cfd : *impl->versions_->GetColumnFamilySet()) {
353 sv_context.NewSuperVersion();
354 cfd->InstallSuperVersion(&sv_context, &impl->mutex_);
355 }
356 }
357 impl->mutex_.Unlock();
358 sv_context.Clean();
359 if (s.ok()) {
360 *dbptr = impl;
361 for (auto* h : *handles) {
362 impl->NewThreadStatusCfInfo(
363 static_cast_with_check<ColumnFamilyHandleImpl>(h)->cfd());
364 }
365 } else {
366 for (auto h : *handles) {
367 delete h;
368 }
369 handles->clear();
370 delete impl;
371 }
372 return s;
373}
