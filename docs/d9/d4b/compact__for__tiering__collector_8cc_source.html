49Status CompactForTieringCollector::AddUserKey(const Slice& /*key*/,
50 const Slice& value,
51 EntryType type,
52 SequenceNumber seq,
53 uint64_t /*file_size*/) {
54 SequenceNumber seq_for_check = seq;
55 if (type == kEntryTimedPut) {
56 seq_for_check = ParsePackedValueForSeqno(value);
57 }
58 if (seq_for_check < last_level_inclusive_max_seqno_threshold_) {
59 last_level_eligible_entries_counter_++;
60 }
61 total_entries_counter_ += 1;
62 return Status::OK();
63}
64
65Status CompactForTieringCollector::Finish(UserCollectedProperties* properties) {
66 assert(!finish_called_);
67 assert(compaction_trigger_ratio_ > 0);
68 if (last_level_eligible_entries_counter_ >=
69 compaction_trigger_ratio_ * total_entries_counter_) {
70 assert(compaction_trigger_ratio_ <= 1);
71 need_compaction_ = true;
72 }
73 if (last_level_eligible_entries_counter_ > 0) {
74 *properties = UserCollectedProperties{
75 {kNumEligibleLastLevelEntriesPropertyName,
76 std::to_string(last_level_eligible_entries_counter_)},
77 };
78 }
79 finish_called_ = true;
80 return Status::OK();
81}

102TablePropertiesCollector*
103CompactForTieringCollectorFactory::CreateTablePropertiesCollector(
104 TablePropertiesCollectorFactory::Context context) {
105 double compaction_trigger_ratio = GetCompactionTriggerRatio();
106 if (compaction_trigger_ratio <= 0 ||
107 context.level_at_creation == context.num_levels - 1 ||
108 context.last_level_inclusive_max_seqno_threshold == kMaxSequenceNumber) {
109 return nullptr;
110 }
111 // TODO(yuzhangyu): pass actual value.
112 return new CompactForTieringCollector(
113 context.last_level_inclusive_max_seqno_threshold,
114 compaction_trigger_ratio,
115 /*collect_data_age_stats*/ false);
116}
