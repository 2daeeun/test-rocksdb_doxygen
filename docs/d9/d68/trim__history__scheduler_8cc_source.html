21ColumnFamilyData* TrimHistoryScheduler::TakeNextColumnFamily() {
22 std::lock_guard<std::mutex> lock(checking_mutex_);
23 while (true) {
24 if (cfds_.empty()) {
25 return nullptr;
26 }
27 ColumnFamilyData* cfd = cfds_.back();
28 cfds_.pop_back();
29 if (cfds_.empty()) {
30 is_empty_.store(true, std::memory_order_relaxed);
31 }
32
33 if (!cfd->IsDropped()) {
34 // success
35 return cfd;
36 }
37 cfd->UnrefAndTryDelete();
38 }
39}

41bool TrimHistoryScheduler::Empty() {

46void TrimHistoryScheduler::Clear() {
47 ColumnFamilyData* cfd;
48 while ((cfd = TakeNextColumnFamily()) != nullptr) {
49 cfd->UnrefAndTryDelete();
50 }
51 assert(Empty());
52}
