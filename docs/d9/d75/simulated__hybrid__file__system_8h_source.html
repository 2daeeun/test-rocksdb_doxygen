15// A FileSystem simulates hybrid file system by ingesting latency and limit
16// IOPs.
17// This class is only used for development purpose and should not be used
18// in production.
19// Right now we ingest 15ms latency and allow 100 requests per second when
20// the file is for warm temperature.
21// When the object is destroyed, the list of warm files are written to a
22// file, which can be used to reopen a FileSystem and still recover the
23// list. This is to allow the information to preserve between db_bench
24// runs.
25class SimulatedHybridFileSystem : public FileSystemWrapper {
26 public:
27 // metadata_file_name stores metadata of the files, so that it can be
28 // loaded after process restarts. If the file doesn't exist, create
29 // one. The file is written when the class is destroyed.
30 // throughput_multiplier: multiplier of throughput. For example, 1 is to
31 // simulate single disk spindle. 4 is to simualte 4 disk spindles.
32 // is_full_fs_warm: if true, all files are all included in slow I/O
33 // simulation.
34 SimulatedHybridFileSystem(const std::shared_ptr<FileSystem>& base,
35 const std::string& metadata_file_name,
36 int throughput_multiplier, bool is_full_fs_warm);
37
38 ~SimulatedHybridFileSystem() override;
39
40 public:
41 IOStatus NewRandomAccessFile(const std::string& fname,
42 const FileOptions& file_opts,
43 std::unique_ptr<FSRandomAccessFile>* result,
44 IODebugContext* dbg) override;
45 IOStatus NewWritableFile(const std::string& fname,
46 const FileOptions& file_opts,
47 std::unique_ptr<FSWritableFile>* result,
48 IODebugContext* dbg) override;
49 IOStatus DeleteFile(const std::string& fname, const IOOptions& options,
50 IODebugContext* dbg) override;
51
52 const char* Name() const override { return name_.c_str(); }
53
54 private:
55 // Limit 100 requests per second. Rate limiter is designed to byte but
56 // we use it as fixed bytes is one request.
57 std::shared_ptr<RateLimiter> rate_limiter_;
58 std::mutex mutex_;
59 std::unordered_set<std::string> warm_file_set_;
60 std::string metadata_file_name_;
61 std::string name_;
62 bool is_full_fs_warm_;
63};
