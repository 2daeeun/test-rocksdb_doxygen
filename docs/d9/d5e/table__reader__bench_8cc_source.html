38// Make a key that i determines the first 4 characters and j determines the
39// last 4 characters.
40static std::string MakeKey(int i, int j, bool through_db) {
41 char buf[100];
42 snprintf(buf, sizeof(buf), "%04d__key___%04d", i, j);
43 if (through_db) {
44 return std::string(buf);
45 }
46 // If we directly query table, which operates on internal keys
47 // instead of user keys, we need to add 8 bytes of internal
48 // information (row type etc) to user key to make an internal
49 // key.
50 InternalKey key(std::string(buf), 0, ValueType::kTypeValue);
51 return key.Encode().ToString();
52}

59// A very simple benchmark that.
60// Create a table with roughly numKey1 * numKey2 keys,
61// where there are numKey1 prefixes of the key, each has numKey2 number of
62// distinguished key, differing in the suffix part.
63// If if_query_empty_keys = false, query the existing keys numKey1 * numKey2
64// times randomly.
65// If if_query_empty_keys = true, query numKey1 * numKey2 random empty keys.
66// Print out the total time.
67// If through_db=true, a full DB will be created and queries will be against
68// it. Otherwise, operations will be directly through table level.
69//
70// If for_terator=true, instead of just query one key each time, it queries
71// a range sharing the same prefix.
72namespace {
73void TableReaderBenchmark(Options& opts, EnvOptions& env_options,
74 ReadOptions& read_options, int num_keys1,
75 int num_keys2, int num_iter, int /*prefix_len*/,
76 bool if_query_empty_keys, bool for_iterator,
77 bool through_db, bool measured_by_nanosecond) {
78 ROCKSDB_NAMESPACE::InternalKeyComparator ikc(opts.comparator);
79
80 std::string file_name =
81 test::PerThreadDBPath("rocksdb_table_reader_benchmark");
82 std::string dbname = test::PerThreadDBPath("rocksdb_table_reader_bench_db");
83 WriteOptions wo;
84 Env* env = Env::Default();
85 auto* clock = env->GetSystemClock().get();
86 TableBuilder* tb = nullptr;
87 DB* db = nullptr;
88 Status s;
89 const ImmutableOptions ioptions(opts);
90 const ColumnFamilyOptions cfo(opts);
91 const MutableCFOptions moptions(cfo);
92 std::unique_ptr<WritableFileWriter> file_writer;
93 if (!through_db) {
94 ASSERT_OK(WritableFileWriter::Create(env->GetFileSystem(), file_name,
95 FileOptions(env_options), &file_writer,
96 nullptr));
97
98 InternalTblPropCollFactories internal_tbl_prop_coll_factories;
99
100 int unknown_level = -1;
101 const WriteOptions write_options;
102 tb = opts.table_factory->NewTableBuilder(
103 TableBuilderOptions(
104 ioptions, moptions, read_options, write_options, ikc,
105 &internal_tbl_prop_coll_factories, CompressionType::kNoCompression,
106 CompressionOptions(), 0 /* column_family_id */,
107 kDefaultColumnFamilyName, unknown_level, kUnknownNewestKeyTime),
108 file_writer.get());
109 } else {
110 s = DB::Open(opts, dbname, &db);
111 ASSERT_OK(s);
112 ASSERT_TRUE(db != nullptr);
113 }
114 // Populate slightly more than 1M keys
115 for (int i = 0; i < num_keys1; i++) {
116 for (int j = 0; j < num_keys2; j++) {
117 std::string key = MakeKey(i * 2, j, through_db);
118 if (!through_db) {
119 tb->Add(key, key);
120 } else {
121 db->Put(wo, key, key);
122 }
123 }
124 }
125 if (!through_db) {
126 tb->Finish();
127 file_writer->Close(IOOptions());
128 } else {
129 db->Flush(FlushOptions());
130 }
131
132 std::unique_ptr<TableReader> table_reader;
133 if (!through_db) {
134 const auto& fs = env->GetFileSystem();
135 FileOptions fopts(env_options);
136
137 std::unique_ptr<FSRandomAccessFile> raf;
138 s = fs->NewRandomAccessFile(file_name, fopts, &raf, nullptr);
139 if (!s.ok()) {
140 fprintf(stderr, "Create File Error: %s\n", s.ToString().c_str());
141 exit(1);
142 }
143 uint64_t file_size;
144 fs->GetFileSize(file_name, fopts.io_options, &file_size, nullptr);
145 std::unique_ptr<RandomAccessFileReader> file_reader(
146 new RandomAccessFileReader(std::move(raf), file_name));
147 s = opts.table_factory->NewTableReader(
148 TableReaderOptions(ioptions, moptions.prefix_extractor, env_options,
149 ikc, 0 /* block_protection_bytes_per_key */),
150 std::move(file_reader), file_size, &table_reader);
151 if (!s.ok()) {
152 fprintf(stderr, "Open Table Error: %s\n", s.ToString().c_str());
153 exit(1);
154 }
155 }
156
157 Random rnd(301);
158 std::string result;
159 HistogramImpl hist;
160
161 for (int it = 0; it < num_iter; it++) {
162 for (int i = 0; i < num_keys1; i++) {
163 for (int j = 0; j < num_keys2; j++) {
164 int r1 = rnd.Uniform(num_keys1) * 2;
165 int r2 = rnd.Uniform(num_keys2);
166 if (if_query_empty_keys) {
167 r1++;
168 r2 = num_keys2 * 2 - r2;
169 }
170
171 if (!for_iterator) {
172 // Query one existing key;
173 std::string key = MakeKey(r1, r2, through_db);
174 uint64_t start_time = Now(clock, measured_by_nanosecond);
175 if (!through_db) {
176 PinnableSlice value;
177 MergeContext merge_context;
178 SequenceNumber max_covering_tombstone_seq = 0;
179 GetContext get_context(
180 ioptions.user_comparator, ioptions.merge_operator.get(),
181 ioptions.logger, ioptions.stats, GetContext::kNotFound,
182 Slice(key), &value, /*columns=*/nullptr, /*timestamp=*/nullptr,
183 &merge_context, true, &max_covering_tombstone_seq, clock);
184 s = table_reader->Get(read_options, key, &get_context, nullptr);
185 } else {
186 s = db->Get(read_options, key, &result);
187 }
188 hist.Add(Now(clock, measured_by_nanosecond) - start_time);
189 } else {
190 int r2_len;
191 if (if_query_empty_keys) {
192 r2_len = 0;
193 } else {
194 r2_len = rnd.Uniform(num_keys2) + 1;
195 if (r2_len + r2 > num_keys2) {
196 r2_len = num_keys2 - r2;
197 }
198 }
199 std::string start_key = MakeKey(r1, r2, through_db);
200 std::string end_key = MakeKey(r1, r2 + r2_len, through_db);
201 uint64_t total_time = 0;
202 uint64_t start_time = Now(clock, measured_by_nanosecond);
203 Iterator* iter = nullptr;
204 InternalIterator* iiter = nullptr;
205 if (!through_db) {
206 iiter = table_reader->NewIterator(
207 read_options, /*prefix_extractor=*/nullptr, /*arena=*/nullptr,
208 /*skip_filters=*/false, TableReaderCaller::kUncategorized);
209 } else {
210 iter = db->NewIterator(read_options);
211 }
212 int count = 0;
213 for (through_db ? iter->Seek(start_key) : iiter->Seek(start_key);
214 through_db ? iter->Valid() : iiter->Valid();
215 through_db ? iter->Next() : iiter->Next()) {
216 if (if_query_empty_keys) {
217 break;
218 }
219 // verify key;
220 total_time += Now(clock, measured_by_nanosecond) - start_time;
221 assert(Slice(MakeKey(r1, r2 + count, through_db)) ==
222 (through_db ? iter->key() : iiter->key()));
223 start_time = Now(clock, measured_by_nanosecond);
224 if (++count >= r2_len) {
225 break;
226 }
227 }
228 if (count != r2_len) {
229 fprintf(stderr,
230 "Iterator cannot iterate expected number of entries. "
231 "Expected %d but got %d\n",
232 r2_len, count);
233 assert(false);
234 }
235 delete iter;
236 total_time += Now(clock, measured_by_nanosecond) - start_time;
237 hist.Add(total_time);
238 }
239 }
240 }
241 }
242
243 fprintf(
244 stderr,
245 "==================================================="
246 "====================================================\n"
247 "InMemoryTableSimpleBenchmark: %20s num_key1: %5d "
248 "num_key2: %5d %10s\n"
249 "==================================================="
250 "===================================================="
251 "\nHistogram (unit: %s): \n%s",
252 opts.table_factory->Name(), num_keys1, num_keys2,
253 for_iterator ? "iterator" : (if_query_empty_keys ? "empty" : "non_empty"),
254 measured_by_nanosecond ? "nanosecond" : "microsecond",
255 hist.ToString().c_str());
256 if (!through_db) {
257 env->DeleteFile(file_name);
258 } else {
259 delete db;
260 db = nullptr;
261 DestroyDB(dbname, opts);
262 }
263}
264} // namespace
265} // namespace ROCKSDB_NAMESPACE

285int main(int argc, char** argv) {
286 SetUsageMessage(std::string("\nUSAGE:\n") + std::string(argv[0]) +
287 " [OPTIONS]...");
288 ParseCommandLineFlags(&argc, &argv, true);
289
290 std::shared_ptr<ROCKSDB_NAMESPACE::TableFactory> tf;
291 ROCKSDB_NAMESPACE::Options options;
292 if (FLAGS_prefix_len < 16) {
293 options.prefix_extractor.reset(
294 ROCKSDB_NAMESPACE::NewFixedPrefixTransform(FLAGS_prefix_len));
295 }
296 ROCKSDB_NAMESPACE::ReadOptions ro;
297 ROCKSDB_NAMESPACE::EnvOptions env_options;
298 options.create_if_missing = true;
299 options.compression = ROCKSDB_NAMESPACE::CompressionType::kNoCompression;
300
301 if (FLAGS_table_factory == "cuckoo_hash") {
302 options.allow_mmap_reads = FLAGS_mmap_read;
303 env_options.use_mmap_reads = FLAGS_mmap_read;
304 ROCKSDB_NAMESPACE::CuckooTableOptions table_options;
305 table_options.hash_table_ratio = 0.75;
306 tf.reset(ROCKSDB_NAMESPACE::NewCuckooTableFactory(table_options));
307 } else if (FLAGS_table_factory == "plain_table") {
308 options.allow_mmap_reads = FLAGS_mmap_read;
309 env_options.use_mmap_reads = FLAGS_mmap_read;
310
311 ROCKSDB_NAMESPACE::PlainTableOptions plain_table_options;
312 plain_table_options.user_key_len = 16;
313 plain_table_options.bloom_bits_per_key = (FLAGS_prefix_len == 16) ? 0 : 8;
314 plain_table_options.hash_table_ratio = 0.75;
315
316 tf.reset(new ROCKSDB_NAMESPACE::PlainTableFactory(plain_table_options));
317 options.prefix_extractor.reset(
318 ROCKSDB_NAMESPACE::NewFixedPrefixTransform(FLAGS_prefix_len));
319 } else if (FLAGS_table_factory == "block_based") {
320 tf.reset(new ROCKSDB_NAMESPACE::BlockBasedTableFactory());
321 } else {
322 fprintf(stderr, "Invalid table type %s\n", FLAGS_table_factory.c_str());
323 }
324
325 if (tf) {
326 // if user provides invalid options, just fall back to microsecond.
327 bool measured_by_nanosecond = FLAGS_time_unit == "nanosecond";
328
329 options.table_factory = tf;
330 ROCKSDB_NAMESPACE::TableReaderBenchmark(
331 options, env_options, ro, FLAGS_num_keys1, FLAGS_num_keys2, FLAGS_iter,
332 FLAGS_prefix_len, FLAGS_query_empty, FLAGS_iterator, FLAGS_through_db,
333 measured_by_nanosecond);
334 } else {
335 return 1;
336 }
337
338 return 0;
339}
