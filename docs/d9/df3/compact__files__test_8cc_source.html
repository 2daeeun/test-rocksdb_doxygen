34// A class which remembers the name of each flushed file.
35class FlushedFileCollector : public EventListener {
36 public:
37 FlushedFileCollector() = default;
38 ~FlushedFileCollector() override = default;
39
40 void OnFlushCompleted(DB* /*db*/, const FlushJobInfo& info) override {
41 std::lock_guard<std::mutex> lock(mutex_);
42 flushed_files_.push_back(info.file_path);
43 }
44
45 std::vector<std::string> GetFlushedFiles() {
46 std::lock_guard<std::mutex> lock(mutex_);
47 std::vector<std::string> result;
48 for (const auto& fname : flushed_files_) {
49 result.push_back(fname);
50 }
51 return result;
52 }
53 void ClearFlushedFiles() {
54 std::lock_guard<std::mutex> lock(mutex_);
55 flushed_files_.clear();
56 }
57
58 private:
59 std::vector<std::string> flushed_files_;
60 std::mutex mutex_;
61};

63TEST_F(CompactFilesTest, L0ConflictsFiles) {
64 Options options;
65 // to trigger compaction more easily
66 const int kWriteBufferSize = 10000;
67 const int kLevel0Trigger = 2;
68 options.create_if_missing = true;
69 options.level_compaction_dynamic_level_bytes = false;
70 options.compaction_style = kCompactionStyleLevel;
71 // Small slowdown and stop trigger for experimental purpose.
72 options.level0_slowdown_writes_trigger = 20;
73 options.level0_stop_writes_trigger = 20;
74 options.level0_stop_writes_trigger = 20;
75 options.write_buffer_size = kWriteBufferSize;
76 options.level0_file_num_compaction_trigger = kLevel0Trigger;
77 options.compression = kNoCompression;
78
79 DB* db = nullptr;
80 ASSERT_OK(DestroyDB(db_name_, options));
81 Status s = DB::Open(options, db_name_, &db);
82 assert(s.ok());
83 assert(db);
84
85 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->LoadDependency({
86 {"CompactFilesImpl:0", "BackgroundCallCompaction:0"},
87 {"BackgroundCallCompaction:1", "CompactFilesImpl:1"},
88 });
89 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
90
91 // create couple files
92 // Background compaction starts and waits in BackgroundCallCompaction:0
93 for (int i = 0; i < kLevel0Trigger * 4; ++i) {
94 ASSERT_OK(db->Put(WriteOptions(), std::to_string(i), ""));
95 ASSERT_OK(db->Put(WriteOptions(), std::to_string(100 - i), ""));
96 ASSERT_OK(db->Flush(FlushOptions()));
97 }
98
99 ROCKSDB_NAMESPACE::ColumnFamilyMetaData meta;
100 db->GetColumnFamilyMetaData(&meta);
101 std::string file1;
102 for (auto& file : meta.levels[0].files) {
103 ASSERT_EQ(0, meta.levels[0].level);
104 if (file1 == "") {
105 file1 = file.db_path + "/" + file.name;
106 } else {
107 std::string file2 = file.db_path + "/" + file.name;
108 // Another thread starts a compact files and creates an L0 compaction
109 // The background compaction then notices that there is an L0 compaction
110 // already in progress and doesn't do an L0 compaction
111 // Once the background compaction finishes, the compact files finishes
112 ASSERT_OK(db->CompactFiles(ROCKSDB_NAMESPACE::CompactionOptions(),
113 {file1, file2}, 0));
114 break;
115 }
116 }
117 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
118 delete db;
119}

121TEST_F(CompactFilesTest, MultipleLevel) {
122 Options options;
123 options.create_if_missing = true;
124 // Otherwise background compaction can happen to
125 // drain unnecessary level
126 options.level_compaction_dynamic_level_bytes = false;
127 options.num_levels = 6;
128 // Add listener
129 FlushedFileCollector* collector = new FlushedFileCollector();
130 options.listeners.emplace_back(collector);
131
132 DB* db = nullptr;
133 ASSERT_OK(DestroyDB(db_name_, options));
134 Status s = DB::Open(options, db_name_, &db);
135 ASSERT_OK(s);
136 ASSERT_NE(db, nullptr);
137
138 // create couple files in L0, L3, L4 and L5
139 for (int i = 5; i > 2; --i) {
140 collector->ClearFlushedFiles();
141 ASSERT_OK(db->Put(WriteOptions(), std::to_string(i), ""));
142 ASSERT_OK(db->Flush(FlushOptions()));
143 // Ensure background work is fully finished including listener callbacks
144 // before accessing listener state.
145 ASSERT_OK(static_cast_with_check<DBImpl>(db)->TEST_WaitForBackgroundWork());
146 auto l0_files = collector->GetFlushedFiles();
147 ASSERT_OK(db->CompactFiles(CompactionOptions(), l0_files, i));
148
149 std::string prop;
150 ASSERT_TRUE(db->GetProperty(
151 "rocksdb.num-files-at-level" + std::to_string(i), &prop));
152 ASSERT_EQ("1", prop);
153 }
154 ASSERT_OK(db->Put(WriteOptions(), std::to_string(0), ""));
155 ASSERT_OK(db->Flush(FlushOptions()));
156
157 ColumnFamilyMetaData meta;
158 db->GetColumnFamilyMetaData(&meta);
159 // Compact files except the file in L3
160 std::vector<std::string> files;
161 for (int i = 0; i < 6; ++i) {
162 if (i == 3) {
163 continue;
164 }
165 for (auto& file : meta.levels[i].files) {
166 files.push_back(file.db_path + "/" + file.name);
167 }
168 }
169
170 SyncPoint::GetInstance()->LoadDependency({
171 {"CompactionJob::Run():Start", "CompactFilesTest.MultipleLevel:0"},
172 {"CompactFilesTest.MultipleLevel:1", "CompactFilesImpl:3"},
173 });
174 SyncPoint::GetInstance()->EnableProcessing();
175
176 std::thread thread([&] {
177 TEST_SYNC_POINT("CompactFilesTest.MultipleLevel:0");
178 ASSERT_OK(db->Put(WriteOptions(), "bar", "v2"));
179 ASSERT_OK(db->Put(WriteOptions(), "foo", "v2"));
180 ASSERT_OK(db->Flush(FlushOptions()));
181 TEST_SYNC_POINT("CompactFilesTest.MultipleLevel:1");
182 });
183
184 // Compaction cannot move up the data to higher level
185 // here we have input file from level 5, so the output level has to be >= 5
186 for (int invalid_output_level = 0; invalid_output_level < 5;
187 invalid_output_level++) {
188 s = db->CompactFiles(CompactionOptions(), files, invalid_output_level);
189 ASSERT_TRUE(s.IsInvalidArgument());
190 }
191
192 ASSERT_OK(db->CompactFiles(CompactionOptions(), files, 5));
193 SyncPoint::GetInstance()->DisableProcessing();
194 thread.join();
195
196 delete db;
197}

392TEST_F(CompactFilesTest, SentinelCompressionType) {
393 if (!Zlib_Supported()) {
394 fprintf(stderr, "zlib compression not supported, skip this test\n");
395 return;
396 }
397 if (!Snappy_Supported()) {
398 fprintf(stderr, "snappy compression not supported, skip this test\n");
399 return;
400 }
401 // Check that passing `CompressionType::kDisableCompressionOption` to
402 // `CompactFiles` causes it to use the column family compression options.
403 for (auto compaction_style : {CompactionStyle::kCompactionStyleLevel,
404 CompactionStyle::kCompactionStyleUniversal,
405 CompactionStyle::kCompactionStyleNone}) {
406 ASSERT_OK(DestroyDB(db_name_, Options()));
407 Options options;
408 options.level_compaction_dynamic_level_bytes = false;
409 options.compaction_style = compaction_style;
410 // L0: Snappy, L1: ZSTD, L2: Snappy
411 options.compression_per_level = {CompressionType::kSnappyCompression,
412 CompressionType::kZlibCompression,
413 CompressionType::kSnappyCompression};
414 options.create_if_missing = true;
415 FlushedFileCollector* collector = new FlushedFileCollector();
416 options.listeners.emplace_back(collector);
417 DB* db = nullptr;
418 ASSERT_OK(DB::Open(options, db_name_, &db));
419
420 ASSERT_OK(db->Put(WriteOptions(), "key", "val"));
421 ASSERT_OK(db->Flush(FlushOptions()));
422
423 // Ensure background work is fully finished including listener callbacks
424 // before accessing listener state.
425 ASSERT_OK(static_cast_with_check<DBImpl>(db)->TEST_WaitForBackgroundWork());
426 auto l0_files = collector->GetFlushedFiles();
427 ASSERT_EQ(1, l0_files.size());
428
429 // L0->L1 compaction, so output should be ZSTD-compressed
430 CompactionOptions compaction_opts;
431 compaction_opts.compression = CompressionType::kDisableCompressionOption;
432 ASSERT_OK(db->CompactFiles(compaction_opts, l0_files, 1));
433
434 ROCKSDB_NAMESPACE::TablePropertiesCollection all_tables_props;
435 ASSERT_OK(db->GetPropertiesOfAllTables(&all_tables_props));
436 for (const auto& name_and_table_props : all_tables_props) {
437 ASSERT_EQ(CompressionTypeToString(CompressionType::kZlibCompression),
438 name_and_table_props.second->compression_name);
439 }
440 delete db;
441 }
