41static Status ParseCompressionOptions(const std::string& value,
42 const std::string& name,
43 CompressionOptions& compression_opts) {
44 const char kDelimiter = ':';
45 std::istringstream field_stream(value);
46 std::string field;
47
48 if (!std::getline(field_stream, field, kDelimiter)) {
49 return Status::InvalidArgument("unable to parse the specified CF option " +
50 name);
51 }
52 compression_opts.window_bits = ParseInt(field);
53
54 if (!std::getline(field_stream, field, kDelimiter)) {
55 return Status::InvalidArgument("unable to parse the specified CF option " +
56 name);
57 }
58 compression_opts.level = ParseInt(field);
59
60 if (!std::getline(field_stream, field, kDelimiter)) {
61 return Status::InvalidArgument("unable to parse the specified CF option " +
62 name);
63 }
64 compression_opts.strategy = ParseInt(field);
65
66 // max_dict_bytes is optional for backwards compatibility
67 if (!field_stream.eof()) {
68 if (!std::getline(field_stream, field, kDelimiter)) {
69 return Status::InvalidArgument(
70 "unable to parse the specified CF option " + name);
71 }
72 compression_opts.max_dict_bytes = ParseInt(field);
73 }
74
75 // zstd_max_train_bytes is optional for backwards compatibility
76 if (!field_stream.eof()) {
77 if (!std::getline(field_stream, field, kDelimiter)) {
78 return Status::InvalidArgument(
79 "unable to parse the specified CF option " + name);
80 }
81 compression_opts.zstd_max_train_bytes = ParseInt(field);
82 }
83
84 // parallel_threads is optional for backwards compatibility
85 if (!field_stream.eof()) {
86 if (!std::getline(field_stream, field, kDelimiter)) {
87 return Status::InvalidArgument(
88 "unable to parse the specified CF option " + name);
89 }
90 // Since parallel_threads comes before enabled but was added optionally
91 // later, we need to check if this is the final token (meaning it is the
92 // enabled bit), or if there are more tokens (meaning this one is
93 // parallel_threads).
94 if (!field_stream.eof()) {
95 compression_opts.parallel_threads = ParseInt(field);
96 } else {
97 // parallel_threads is not serialized with this format, but enabled is
98 compression_opts.enabled = ParseBoolean("", field);
99 }
100 }
101
102 // enabled is optional for backwards compatibility
103 if (!field_stream.eof()) {
104 if (!std::getline(field_stream, field, kDelimiter)) {
105 return Status::InvalidArgument(
106 "unable to parse the specified CF option " + name);
107 }
108 compression_opts.enabled = ParseBoolean("", field);
109 }
110
111 // max_dict_buffer_bytes is optional for backwards compatibility
112 if (!field_stream.eof()) {
113 if (!std::getline(field_stream, field, kDelimiter)) {
114 return Status::InvalidArgument(
115 "unable to parse the specified CF option " + name);
116 }
117 compression_opts.max_dict_buffer_bytes = ParseUint64(field);
118 }
119
120 // use_zstd_dict_trainer is optional for backwards compatibility
121 if (!field_stream.eof()) {
122 if (!std::getline(field_stream, field, kDelimiter)) {
123 return Status::InvalidArgument(
124 "unable to parse the specified CF option " + name);
125 }
126 compression_opts.use_zstd_dict_trainer = ParseBoolean("", field);
127 }
128
129 if (!field_stream.eof()) {
130 return Status::InvalidArgument("unable to parse the specified CF option " +
131 name);
132 }
133 return Status::OK();
134}

136static Status TableFactoryParseFn(const ConfigOptions& opts,
137 const std::string& name,
138 const std::string& value, void* addr) {
139 assert(addr);
140 auto table_factory = static_cast<std::shared_ptr<TableFactory>*>(addr);
141
142 // The general approach to mutating a table factory is to clone it, then
143 // mutate and save the clone. This avoids race conditions between SetOptions
144 // and consumers of table_factory/table options by leveraging
145 // MutableCFOptions infrastructure to track the table_factory pointer.
146
147 // However, in the atypical case of setting an option that is safely mutable
148 // under something pointed to by the table factory, we should avoid cloning.
149 // The simple way to detect that case is to try with "mutable_options_only"
150 // and see if it works. If it does, we are finished. If not, we proceed to
151 // cloning etc.
152 //
153 // The canonical example of what is handled here is
154 // table_factory.filter_policy.bloom_before_level for RibbonFilterPolicy.
155 if (table_factory->get() != nullptr && !EndsWith(name, "table_factory")) {
156 ConfigOptions opts_mutable_only{opts};
157 opts_mutable_only.mutable_options_only = true;
158 Status s =
159 table_factory->get()->ConfigureOption(opts_mutable_only, name, value);
160 if (s.ok()) {
161 return s;
162 }
163 s.PermitUncheckedError();
164 }
165
166 std::shared_ptr<TableFactory> new_factory;
167 Status s;
168 if (name == "block_based_table_factory") {
169 if (table_factory->get() != nullptr) {
170 std::string factory_name = table_factory->get()->Name();
171 if (factory_name == TableFactory::kBlockBasedTableName()) {
172 new_factory = table_factory->get()->Clone();
173 } else {
174 s = Status::InvalidArgument("Cannot modify " + factory_name + " as " +
175 name);
176 return s;
177 }
178 } else {
179 new_factory.reset(NewBlockBasedTableFactory());
180 }
181 // Passing an object string to configure/instantiate a table factory
182 s = new_factory->ConfigureFromString(opts, value);
183 } else if (name == "plain_table_factory") {
184 if (table_factory->get() != nullptr) {
185 std::string factory_name = table_factory->get()->Name();
186 if (factory_name == TableFactory::kPlainTableName()) {
187 new_factory = table_factory->get()->Clone();
188 } else {
189 s = Status::InvalidArgument("Cannot modify " + factory_name + " as " +
190 name);
191 return s;
192 }
193 } else {
194 new_factory.reset(NewPlainTableFactory());
195 }
196 // Passing an object string to configure/instantiate a table factory
197 s = new_factory->ConfigureFromString(opts, value);
198 } else if (name == "table_factory" || name == OptionTypeInfo::kIdPropName()) {
199 // Related to OptionTypeInfo::AsCustomSharedPtr
200 if (value.empty()) {
201 new_factory = nullptr;
202 } else {
203 s = TableFactory::CreateFromString(opts, value, &new_factory);
204 }
205 } else if (table_factory->get() != nullptr) {
206 new_factory = table_factory->get()->Clone();
207 // Presumably passing a value for a specific field of the table factory
208 s = new_factory->ConfigureOption(opts, name, value);
209 } else {
210 s = Status::NotFound("Unable to instantiate a table factory from option: ",
211 name);
212 return s;
213 }
214
215 // Only keep the modified clone if everything went OK
216 if (s.ok()) {
217 *table_factory = std::move(new_factory);
218 }
219 return s;
220}

917 bool OptionsAreEqual(const ConfigOptions& config_options,
918 const OptionTypeInfo& opt_info,
919 const std::string& opt_name, const void* const this_ptr,
920 const void* const that_ptr,
921 std::string* mismatch) const override {
922 bool equals = opt_info.AreEqual(config_options, opt_name, this_ptr,
923 that_ptr, mismatch);
924 if (!equals && opt_info.IsByName()) {
925 if (opt_map_ == nullptr) {
926 equals = true;
927 } else {
928 const auto& iter = opt_map_->find(opt_name);
929 if (iter == opt_map_->end()) {
930 equals = true;
931 } else {
932 equals = opt_info.AreEqualByName(config_options, opt_name, this_ptr,
933 iter->second);
934 }
935 }
936 if (equals) { // False alarm, clear mismatch
937 *mismatch = "";
938 }
939 }
940 if (equals && opt_info.IsConfigurable() && opt_map_ != nullptr) {
941 const auto* this_config = opt_info.AsRawPointer<Configurable>(this_ptr);
942 if (this_config == nullptr) {
943 const auto& iter = opt_map_->find(opt_name);
944 // If the name exists in the map and is not empty/null,
945 // then the this_config should be set.
946 if (iter != opt_map_->end() && !iter->second.empty() &&
947 iter->second != kNullptrString) {
948 *mismatch = opt_name;
949 equals = false;
950 }
951 }
952 }
953 return equals;
954 }

1032// Multiple two operands. If they overflow, return op1.
1033uint64_t MultiplyCheckOverflow(uint64_t op1, double op2) {
1034 if (op1 == 0 || op2 <= 0) {
1035 return 0;
1036 }
1037 if (std::numeric_limits<uint64_t>::max() / op1 < op2) {
1038 return op1;
1039 }
1040 return static_cast<uint64_t>(op1 * op2);
1041}

1043// when level_compaction_dynamic_level_bytes is true and leveled compaction
1044// is used, the base level is not always L1, so precomupted max_file_size can
1045// no longer be used. Recompute file_size_for_level from base level.
1046uint64_t MaxFileSizeForLevel(const MutableCFOptions& cf_options,
1047 int level, CompactionStyle compaction_style, int base_level,
1048 bool level_compaction_dynamic_level_bytes) {
1049 if (!level_compaction_dynamic_level_bytes || level < base_level ||
1050 compaction_style != kCompactionStyleLevel) {
1051 assert(level >= 0);
1052 assert(level < (int)cf_options.max_file_size.size());
1053 return cf_options.max_file_size[level];
1054 } else {
1055 assert(level >= 0 && base_level >= 0);
1056 assert(level - base_level < (int)cf_options.max_file_size.size());
1057 return cf_options.max_file_size[level - base_level];
1058 }

1061size_t MaxFileSizeForL0MetaPin(const MutableCFOptions& cf_options) {
1062 // We do not want to pin meta-blocks that almost certainly came from intra-L0
1063 // or a former larger `write_buffer_size` value to avoid surprising users with
1064 // pinned memory usage. We use a factor of 1.5 to account for overhead
1065 // introduced during flush in most cases.
1066 if (std::numeric_limits<size_t>::max() / 3 <
1067 cf_options.write_buffer_size / 2) {
1068 return std::numeric_limits<size_t>::max();
1069 }
1070 return cf_options.write_buffer_size / 2 * 3;
1071}

1073void MutableCFOptions::RefreshDerivedOptions(int num_levels,
1074 CompactionStyle compaction_style) {
1075 max_file_size.resize(num_levels);
1076 for (int i = 0; i < num_levels; ++i) {
1077 if (i == 0 && compaction_style == kCompactionStyleUniversal) {
1078 max_file_size[i] = ULLONG_MAX;
1079 } else if (i > 1) {
1080 max_file_size[i] = MultiplyCheckOverflow(max_file_size[i - 1],
1081 target_file_size_multiplier);
1082 } else {
1083 max_file_size[i] = target_file_size_base;
1084 }
1085 }
