165class DbStressFSWrapper : public FileSystemWrapper {
166 public:
167 explicit DbStressFSWrapper(const std::shared_ptr<FileSystem>& t)
168 : FileSystemWrapper(t) {}
169 static const char* kClassName() { return "DbStressFS"; }
170 const char* Name() const override { return kClassName(); }
171
172 IOStatus NewRandomAccessFile(const std::string& f,
173 const FileOptions& file_opts,
174 std::unique_ptr<FSRandomAccessFile>* r,
175 IODebugContext* dbg) override {
176 std::unique_ptr<FSRandomAccessFile> file;
177 IOStatus s = target()->NewRandomAccessFile(f, file_opts, &file, dbg);
178 if (s.ok()) {
179 r->reset(new DbStressRandomAccessFileWrapper(std::move(file)));
180 }
181 return s;
182 }
183
184 IOStatus NewWritableFile(const std::string& f, const FileOptions& file_opts,
185 std::unique_ptr<FSWritableFile>* r,
186 IODebugContext* dbg) override {
187 std::unique_ptr<FSWritableFile> file;
188 IOStatus s = target()->NewWritableFile(f, file_opts, &file, dbg);
189 if (s.ok()) {
190 r->reset(new DbStressWritableFileWrapper(std::move(file)));
191 }
192 return s;
193 }
194
195 IOStatus DeleteFile(const std::string& f, const IOOptions& opts,
196 IODebugContext* dbg) override {
197 // We determine whether it is a manifest file by searching a strong,
198 // so that there will be false positive if the directory path contains the
199 // keyword but it is unlikely.
200 // Checkpoint, backup, and restore directories needs to be exempted.
201 if (!if_preserve_all_manifests ||
202 f.find("MANIFEST-") == std::string::npos ||
203 f.find("checkpoint") != std::string::npos ||
204 f.find(".backup") != std::string::npos ||
205 f.find(".restore") != std::string::npos) {
206 return target()->DeleteFile(f, opts, dbg);
207 }
208 // Rename the file instead of deletion to keep the history, and
209 // at the same time it is not visible to RocksDB.
210 return target()->RenameFile(f, f + "_renamed_", opts, dbg);
211 }
212
213 // If true, all manifest files will not be delted in DeleteFile().
214 bool if_preserve_all_manifests = true;
215};
