28PartitionerResult SstPartitionerFixedPrefix::ShouldPartition(
29 const PartitionerRequest& request) {
30 Slice last_key_fixed(*request.prev_user_key);
31 if (last_key_fixed.size() > len_) {
32 last_key_fixed.size_ = len_;
33 }
34 Slice current_key_fixed(*request.current_user_key);
35 if (current_key_fixed.size() > len_) {
36 current_key_fixed.size_ = len_;
37 }
38 return last_key_fixed.compare(current_key_fixed) != 0 ? kRequired
39 : kNotRequired;
40}

42bool SstPartitionerFixedPrefix::CanDoTrivialMove(
43 const Slice& smallest_user_key, const Slice& largest_user_key) {
44 return ShouldPartition(PartitionerRequest(smallest_user_key, largest_user_key,
45 0)) == kNotRequired;
46}
