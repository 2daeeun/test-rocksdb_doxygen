63// construct a tiered RAM+Block cache
64std::unique_ptr<PersistentTieredCache> NewTieredCache(
65 const size_t mem_size, const PersistentCacheConfig& opt) {
66 std::unique_ptr<PersistentTieredCache> tcache(new PersistentTieredCache());
67 // create primary tier
68 assert(mem_size);
69 auto pcache = std::shared_ptr<PersistentCacheTier>(new VolatileCacheTier(
70 /*is_compressed*/ true, mem_size));
71 tcache->AddTier(pcache);
72 // create secondary tier
73 auto scache = std::shared_ptr<PersistentCacheTier>(new BlockCacheTier(opt));
74 tcache->AddTier(scache);
75
76 Status s = tcache->Open();
77 assert(s.ok());
78 return tcache;
79}

81// create block cache
82std::unique_ptr<PersistentCacheTier> NewBlockCache(
83 Env* env, const std::string& path,
84 const uint64_t max_size = std::numeric_limits<uint64_t>::max(),
85 const bool enable_direct_writes = false) {
86 const uint32_t max_file_size =
87 static_cast<uint32_t>(12 * 1024 * 1024 * kStressFactor);
88 auto log = std::make_shared<ConsoleLogger>();
89 PersistentCacheConfig opt(env, path, max_size, log);
90 opt.cache_file_size = max_file_size;
91 opt.max_write_pipeline_backlog_size = std::numeric_limits<uint64_t>::max();
92 opt.enable_direct_writes = enable_direct_writes;
93 std::unique_ptr<PersistentCacheTier> scache(new BlockCacheTier(opt));
94 Status s = scache->Open();
95 assert(s.ok());
96 return scache;
97}

99// create a new cache tier
100std::unique_ptr<PersistentTieredCache> NewTieredCache(
101 Env* env, const std::string& path, const uint64_t max_volatile_cache_size,
102 const uint64_t max_block_cache_size =
103 std::numeric_limits<uint64_t>::max()) {
104 const uint32_t max_file_size =
105 static_cast<uint32_t>(12 * 1024 * 1024 * kStressFactor);
106 auto log = std::make_shared<ConsoleLogger>();
107 auto opt = PersistentCacheConfig(env, path, max_block_cache_size, log);
108 opt.cache_file_size = max_file_size;
109 opt.max_write_pipeline_backlog_size = std::numeric_limits<uint64_t>::max();
110 // create tier out of the two caches
111 auto cache = NewTieredCache(max_volatile_cache_size, opt);
112 return cache;
113}

295// test template
296void PersistentCacheDBTest::RunTest(
297 const std::function<std::shared_ptr<PersistentCacheTier>(bool)>& new_pcache,
298 const size_t max_keys = 100 * 1024, const size_t max_usecase = 3) {
299 // number of insertion interations
300 int num_iter = static_cast<int>(max_keys * kStressFactor);
301
302 for (size_t iter = 0; iter < max_usecase; iter++) {
303 Options options;
304 options.write_buffer_size =
305 static_cast<size_t>(64 * 1024 * kStressFactor); // small write buffer
306 options.statistics = ROCKSDB_NAMESPACE::CreateDBStatistics();
307 options = CurrentOptions(options);
308
309 // setup page cache
310 std::shared_ptr<PersistentCacheTier> pcache;
311 BlockBasedTableOptions table_options;
312 table_options.cache_index_and_filter_blocks = true;
313
314 const size_t size_max = std::numeric_limits<size_t>::max();
315
316 switch (iter) {
317 case 0:
318 // page cache, block cache, no-compressed cache
319 pcache = new_pcache(/*is_compressed=*/true);
320 table_options.persistent_cache = pcache;
321 table_options.block_cache = NewLRUCache(size_max);
322 options.table_factory.reset(NewBlockBasedTableFactory(table_options));
323 break;
324 case 1:
325 // page cache, no block cache, no compressed cache
326 pcache = new_pcache(/*is_compressed=*/false);
327 table_options.persistent_cache = pcache;
328 table_options.block_cache = nullptr;
329 options.table_factory.reset(NewBlockBasedTableFactory(table_options));
330 break;
331 case 2:
332 // page cache, no block cache, no compressed cache
333 // Page cache caches compressed blocks
334 pcache = new_pcache(/*is_compressed=*/true);
335 table_options.persistent_cache = pcache;
336 table_options.block_cache = nullptr;
337 options.table_factory.reset(NewBlockBasedTableFactory(table_options));
338 break;
339 default:
340 FAIL();
341 }
342
343 std::vector<std::string> values;
344 // insert data
345 Insert(options, table_options, num_iter, &values);
346 // flush all data in cache to device
347 pcache->TEST_Flush();
348 // verify data
349 Verify(num_iter, values);
350
351 auto block_miss = TestGetTickerCount(options, BLOCK_CACHE_MISS);
352 auto page_hit = TestGetTickerCount(options, PERSISTENT_CACHE_HIT);
353 auto page_miss = TestGetTickerCount(options, PERSISTENT_CACHE_MISS);
354
355 // check that we triggered the appropriate code paths in the cache
356 switch (iter) {
357 case 0:
358 // page cache, block cache, no-compressed cache
359 ASSERT_GT(page_miss, 0);
360 ASSERT_GT(page_hit, 0);
361 ASSERT_GT(block_miss, 0);
362 break;
363 case 1:
364 case 2:
365 // page cache, no block cache, no compressed cache
366 ASSERT_GT(page_miss, 0);
367 ASSERT_GT(page_hit, 0);
368 break;
369 default:
370 FAIL();
371 }
372
373 options.create_if_missing = true;
374 DestroyAndReopen(options);
375
376 ASSERT_OK(pcache->Close());
377 }
378}
