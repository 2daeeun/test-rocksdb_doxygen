19CompactOnDeletionCollector::CompactOnDeletionCollector(
20 size_t sliding_window_size, size_t deletion_trigger, double deletion_ratio)
21 : bucket_size_((sliding_window_size + kNumBuckets - 1) / kNumBuckets),
22 current_bucket_(0),
23 num_keys_in_current_bucket_(0),
24 num_deletions_in_observation_window_(0),
25 deletion_trigger_(deletion_trigger),
26 deletion_ratio_(deletion_ratio),
27 deletion_ratio_enabled_(deletion_ratio > 0 && deletion_ratio <= 1),
28 need_compaction_(false),
29 finished_(false) {
30 memset(num_deletions_in_buckets_, 0, sizeof(size_t) * kNumBuckets);
31}
32
33// AddUserKey() will be called when a new key/value pair is inserted into the
34// table.
35// @params key the user key that is inserted into the table.
36// @params value the value that is inserted into the table.
37// @params file_size file size up to now
38Status CompactOnDeletionCollector::AddUserKey(const Slice& /*key*/,
39 const Slice& /*value*/,
40 EntryType type,
41 SequenceNumber /*seq*/,
42 uint64_t /*file_size*/) {
43 assert(!finished_);
44 if (!bucket_size_ && !deletion_ratio_enabled_) {
45 // This collector is effectively disabled
46 return Status::OK();
47 }
48
49 if (need_compaction_) {
50 // If the output file already needs to be compacted, skip the check.
51 return Status::OK();
52 }
53
54 if (deletion_ratio_enabled_) {
55 total_entries_++;
56 if (type == kEntryDelete) {
57 deletion_entries_++;
58 }
59 }
60
61 if (bucket_size_) {
62 if (num_keys_in_current_bucket_ == bucket_size_) {
63 // When the current bucket is full, advance the cursor of the
64 // ring buffer to the next bucket.
65 current_bucket_ = (current_bucket_ + 1) % kNumBuckets;
66
67 // Update the current count of observed deletion keys by excluding
68 // the number of deletion keys in the oldest bucket in the
69 // observation window.
70 assert(num_deletions_in_observation_window_ >=
71 num_deletions_in_buckets_[current_bucket_]);
72 num_deletions_in_observation_window_ -=
73 num_deletions_in_buckets_[current_bucket_];
74 num_deletions_in_buckets_[current_bucket_] = 0;
75 num_keys_in_current_bucket_ = 0;
76 }
77
78 num_keys_in_current_bucket_++;
79 if (type == kEntryDelete) {
80 num_deletions_in_observation_window_++;
81 num_deletions_in_buckets_[current_bucket_]++;
82 if (num_deletions_in_observation_window_ >= deletion_trigger_) {
83 need_compaction_ = true;
84 }
85 }
86 }
87
88 return Status::OK();
89}

91Status CompactOnDeletionCollector::Finish(
92 UserCollectedProperties* /*properties*/) {
93 if (!need_compaction_ && deletion_ratio_enabled_ && total_entries_ > 0) {
94 double ratio = static_cast<double>(deletion_entries_) / total_entries_;
95 need_compaction_ = ratio >= deletion_ratio_;
96 }
97 finished_ = true;
98 return Status::OK();
99}
