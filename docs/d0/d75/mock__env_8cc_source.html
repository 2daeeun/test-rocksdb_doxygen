83class MemFile {
84 public:
85 explicit MemFile(SystemClock* clock, const std::string& fn,
86 bool _is_lock_file = false)
87 : clock_(clock),
88 fn_(fn),
89 refs_(0),
90 is_lock_file_(_is_lock_file),
91 locked_(false),
92 size_(0),
93 modified_time_(Now()),
94 rnd_(Lower32of64(GetSliceNPHash64(fn))),
95 fsynced_bytes_(0) {}
96 // No copying allowed.
97 MemFile(const MemFile&) = delete;
98 void operator=(const MemFile&) = delete;
99
100 void Ref() {
101 MutexLock lock(&mutex_);
102 ++refs_;
103 }
104
105 bool is_lock_file() const { return is_lock_file_; }
106
107 bool Lock() {
108 assert(is_lock_file_);
109 MutexLock lock(&mutex_);
110 if (locked_) {
111 return false;
112 } else {
113 locked_ = true;
114 return true;
115 }
116 }
117
118 void Unlock() {
119 assert(is_lock_file_);
120 MutexLock lock(&mutex_);
121 locked_ = false;
122 }
123
124 void Unref() {
125 bool do_delete = false;
126 {
127 MutexLock lock(&mutex_);
128 --refs_;
129 assert(refs_ >= 0);
130 if (refs_ <= 0) {
131 do_delete = true;
132 }
133 }
134
135 if (do_delete) {
136 delete this;
137 }
138 }
139
140 uint64_t Size() const { return size_; }
141
142 void Truncate(size_t size, const IOOptions& /*options*/,
143 IODebugContext* /*dbg*/) {
144 MutexLock lock(&mutex_);
145 if (size < size_) {
146 data_.resize(size);
147 size_ = size;
148 }
149 }
150
151 void CorruptBuffer() {
152 if (fsynced_bytes_ >= size_) {
153 return;
154 }
155 uint64_t buffered_bytes = size_ - fsynced_bytes_;
156 uint64_t start =
157 fsynced_bytes_ + rnd_.Uniform(static_cast<int>(buffered_bytes));
158 uint64_t end = std::min(start + 512, size_.load());
159 MutexLock lock(&mutex_);
160 for (uint64_t pos = start; pos < end; ++pos) {
161 data_[static_cast<size_t>(pos)] = static_cast<char>(rnd_.Uniform(256));
162 }
163 }
164
165 IOStatus Read(uint64_t offset, size_t n, const IOOptions& /*options*/,
166 Slice* result, char* scratch, IODebugContext* /*dbg*/) const {
167 {
168 IOStatus s;
169 TEST_SYNC_POINT_CALLBACK("MemFile::Read:IOStatus", &s);
170 if (!s.ok()) {
171 // with sync point only
172 *result = Slice();
173 return s;
174 }
175 }
176 MutexLock lock(&mutex_);
177 const uint64_t available = Size() - std::min(Size(), offset);
178 size_t offset_ = static_cast<size_t>(offset);
179 if (n > available) {
180 n = static_cast<size_t>(available);
181 }
182 if (n == 0) {
183 *result = Slice();
184 return IOStatus::OK();
185 }
186 if (scratch) {
187 memcpy(scratch, &(data_[offset_]), n);
188 *result = Slice(scratch, n);
189 } else {
190 *result = Slice(&(data_[offset_]), n);
191 }
192 return IOStatus::OK();
193 }
194
195 IOStatus Write(uint64_t offset, const Slice& data,
196 const IOOptions& /*options*/, IODebugContext* /*dbg*/) {
197 MutexLock lock(&mutex_);
198 size_t offset_ = static_cast<size_t>(offset);
199 if (offset + data.size() > data_.size()) {
200 data_.resize(offset_ + data.size());
201 }
202 data_.replace(offset_, data.size(), data.data(), data.size());
203 size_ = data_.size();
204 modified_time_ = Now();
205 return IOStatus::OK();
206 }
207
208 IOStatus Append(const Slice& data, const IOOptions& /*options*/,
209 IODebugContext* /*dbg*/) {
210 MutexLock lock(&mutex_);
211 data_.append(data.data(), data.size());
212 size_ = data_.size();
213 modified_time_ = Now();
214 return IOStatus::OK();
215 }
216
217 IOStatus Fsync(const IOOptions& /*options*/, IODebugContext* /*dbg*/) {
218 fsynced_bytes_ = size_.load();
219 return IOStatus::OK();
220 }
221
222 uint64_t ModifiedTime() const { return modified_time_; }
223
224 private:
225 uint64_t Now() {
226 int64_t unix_time = 0;
227 auto s = clock_->GetCurrentTime(&unix_time);
228 assert(s.ok());
229 return static_cast<uint64_t>(unix_time);
230 }
231
232 // Private since only Unref() should be used to delete it.
233 ~MemFile() { assert(refs_ == 0); }
234
235 SystemClock* clock_;
236 const std::string fn_;
237 mutable port::Mutex mutex_;
238 int refs_;
239 bool is_lock_file_;
240 bool locked_;
241
242 // Data written into this file, all bytes before fsynced_bytes are
243 // persistent.
244 std::string data_;
245 std::atomic<uint64_t> size_;
246 std::atomic<uint64_t> modified_time_;
247
248 Random rnd_;
249 std::atomic<uint64_t> fsynced_bytes_;
250};

463class TestMemLogger : public Logger {
464 private:
465 std::unique_ptr<FSWritableFile> file_;
466 std::atomic_size_t log_size_;
467 static const uint64_t flush_every_seconds_ = 5;
468 std::atomic_uint_fast64_t last_flush_micros_;
469 SystemClock* clock_;
470 IOOptions options_;
471 IODebugContext* dbg_;
472 std::atomic<bool> flush_pending_;
473
474 public:
475 TestMemLogger(std::unique_ptr<FSWritableFile> f, SystemClock* clock,
476 const IOOptions& options, IODebugContext* dbg,
477 const InfoLogLevel log_level = InfoLogLevel::ERROR_LEVEL)
478 : Logger(log_level),
479 file_(std::move(f)),
480 log_size_(0),
481 last_flush_micros_(0),
482 clock_(clock),
483 options_(options),
484 dbg_(dbg),
485 flush_pending_(false) {}
486 ~TestMemLogger() override = default;
487
488 void Flush() override {
489 if (flush_pending_) {
490 flush_pending_ = false;
491 }
492 last_flush_micros_ = clock_->NowMicros();
493 }
494
495 using Logger::Logv;
496 void Logv(const char* format, va_list ap) override {
497 // We try twice: the first time with a fixed-size stack allocated buffer,
498 // and the second time with a much larger dynamically allocated buffer.
499 char buffer[500];
500 for (int iter = 0; iter < 2; iter++) {
501 char* base;
502 int bufsize;
503 if (iter == 0) {
504 bufsize = sizeof(buffer);
505 base = buffer;
506 } else {
507 bufsize = 30000;
508 base = new char[bufsize];
509 }
510 char* p = base;
511 char* limit = base + bufsize;
512
513 port::TimeVal now_tv;
514 port::GetTimeOfDay(&now_tv, nullptr);
515 const time_t seconds = now_tv.tv_sec;
516 struct tm t;
517 memset(&t, 0, sizeof(t));
518 struct tm* ret __attribute__((__unused__));
519 ret = port::LocalTimeR(&seconds, &t);
520 assert(ret);
521 p += snprintf(p, limit - p, "%04d/%02d/%02d-%02d:%02d:%02d.%06d ",
522 t.tm_year + 1900, t.tm_mon + 1, t.tm_mday, t.tm_hour,
523 t.tm_min, t.tm_sec, static_cast<int>(now_tv.tv_usec));
524
525 // Print the message
526 if (p < limit) {
527 va_list backup_ap;
528 va_copy(backup_ap, ap);
529 p += vsnprintf(p, limit - p, format, backup_ap);
530 va_end(backup_ap);
531 }
532
533 // Truncate to available space if necessary
534 if (p >= limit) {
535 if (iter == 0) {
536 continue; // Try again with larger buffer
537 } else {
538 p = limit - 1;
539 }
540 }
541
542 // Add newline if necessary
543 if (p == base || p[-1] != '\n') {
544 *p++ = '\n';
545 }
546
547 assert(p <= limit);
548 const size_t write_size = p - base;
549
550 Status s = file_->Append(Slice(base, write_size), options_, dbg_);
551 if (s.ok()) {
552 flush_pending_ = true;
553 log_size_ += write_size;
554 }
555 uint64_t now_micros =
556 static_cast<uint64_t>(now_tv.tv_sec) * 1000000 + now_tv.tv_usec;
557 if (now_micros - last_flush_micros_ >= flush_every_seconds_ * 1000000) {
558 flush_pending_ = false;
559 last_flush_micros_ = now_micros;
560 }
561 if (base != buffer) {
562 delete[] base;
563 }
564 break;
565 }
566 }
567 size_t GetLogFileSize() const override { return log_size_; }
568};

710IOStatus MockFileSystem::ReopenWritableFile(
711 const std::string& fname, const FileOptions& file_opts,
712 std::unique_ptr<FSWritableFile>* result, IODebugContext* /*dbg*/) {
713 auto fn = NormalizeMockPath(fname);
714 MutexLock lock(&mutex_);
715 MemFile* file = nullptr;
716 if (file_map_.find(fn) == file_map_.end()) {
717 file = new MemFile(clock_, fn, false);
718 // Only take a reference when we create the file objectt
719 file->Ref();
720 file_map_[fn] = file;
721 } else {
722 file = file_map_[fn];

741 const IOOptions& /*io_opts*/,
742 IODebugContext* /*dbg*/) {
743 auto fn = NormalizeMockPath(fname);
744 MutexLock lock(&mutex_);
745 if (file_map_.find(fn) != file_map_.end()) {
746 // File exists
747 return IOStatus::OK();
748 }
749 // Now also check if fn exists as a dir

760bool MockFileSystem::GetChildrenInternal(const std::string& dir,
761 std::vector<std::string>* result) {
762 auto d = NormalizeMockPath(dir);
763 bool found_dir = false;
764 result->clear();
765 for (const auto& iter : file_map_) {
766 const std::string& filename = iter.first;
767
768 if (filename == d) {
769 found_dir = true;
770 } else if (filename.size() >= d.size() + 1 && filename[d.size()] == '/' &&
771 Slice(filename).starts_with(Slice(d))) {
772 found_dir = true;
773 size_t next_slash = filename.find('/', d.size() + 1);
774 if (next_slash != std::string::npos) {
775 result->push_back(
776 filename.substr(d.size() + 1, next_slash - d.size() - 1));
777 } else {
778 result->push_back(filename.substr(d.size() + 1));
779 }
780 }
781 }
782 result->erase(std::unique(result->begin(), result->end()), result->end());
783 return found_dir;
784}
