248void CompactionJob::Prepare() {
249 AutoThreadOperationStageUpdater stage_updater(
250 ThreadStatus::STAGE_COMPACTION_PREPARE);
251
252 // Generate file_levels_ for compaction before making Iterator
253 auto* c = compact_->compaction;
254 [[maybe_unused]] ColumnFamilyData* cfd = c->column_family_data();
255 assert(cfd != nullptr);
256 const VersionStorageInfo* storage_info = c->input_version()->storage_info();
257 assert(storage_info);
258 assert(storage_info->NumLevelFiles(compact_->compaction->level()) > 0);
259
260 write_hint_ = storage_info->CalculateSSTWriteHint(c->output_level());
261 bottommost_level_ = c->bottommost_level();
262
263 if (c->ShouldFormSubcompactions()) {
264 StopWatch sw(db_options_.clock, stats_, SUBCOMPACTION_SETUP_TIME);
265 GenSubcompactionBoundaries();
266 }
267 if (boundaries_.size() >= 1) {
268 for (size_t i = 0; i <= boundaries_.size(); i++) {
269 compact_->sub_compact_states.emplace_back(
270 c, (i != 0) ? std::optional<Slice>(boundaries_[i - 1]) : std::nullopt,
271 (i != boundaries_.size()) ? std::optional<Slice>(boundaries_[i])
272 : std::nullopt,
273 static_cast<uint32_t>(i));
274 // assert to validate that boundaries don't have same user keys (without
275 // timestamp part).
276 assert(i == 0 || i == boundaries_.size() ||
277 cfd->user_comparator()->CompareWithoutTimestamp(
278 boundaries_[i - 1], boundaries_[i]) < 0);
279 }
280 RecordInHistogram(stats_, NUM_SUBCOMPACTIONS_SCHEDULED,
281 compact_->sub_compact_states.size());
282 } else {
283 compact_->sub_compact_states.emplace_back(c, std::nullopt, std::nullopt,
284 /*sub_job_id*/ 0);
285 }
286
287 // collect all seqno->time information from the input files which will be used
288 // to encode seqno->time to the output files.
289
290 uint64_t preserve_time_duration =
291 std::max(c->mutable_cf_options()->preserve_internal_time_seconds,
292 c->mutable_cf_options()->preclude_last_level_data_seconds);
293
294 if (preserve_time_duration > 0) {
295 const ReadOptions read_options(Env::IOActivity::kCompaction);
296 // Setup seqno_to_time_mapping_ with relevant time range.
297 seqno_to_time_mapping_.SetMaxTimeSpan(preserve_time_duration);
298 for (const auto& each_level : *c->inputs()) {
299 for (const auto& fmd : each_level.files) {
300 std::shared_ptr<const TableProperties> tp;
301 Status s = c->input_version()->GetTableProperties(read_options, &tp,
302 fmd, nullptr);
303 if (s.ok()) {
304 s = seqno_to_time_mapping_.DecodeFrom(tp->seqno_to_time_mapping);
305 }
306 if (!s.ok()) {
307 ROCKS_LOG_WARN(
308 db_options_.info_log,
309 "Problem reading or processing seqno-to-time mapping: %s",
310 s.ToString().c_str());
311 }
312 }
313 }
314
315 int64_t _current_time = 0;
316 Status s = db_options_.clock->GetCurrentTime(&_current_time);
317 if (!s.ok()) {
318 ROCKS_LOG_WARN(db_options_.info_log,
319 "Failed to get current time in compaction: Status: %s",
320 s.ToString().c_str());
321 // preserve all time information
322 preserve_time_min_seqno_ = 0;
323 preclude_last_level_min_seqno_ = 0;
324 seqno_to_time_mapping_.Enforce();
325 } else {
326 seqno_to_time_mapping_.Enforce(_current_time);
327 seqno_to_time_mapping_.GetCurrentTieringCutoffSeqnos(
328 static_cast<uint64_t>(_current_time),
329 c->mutable_cf_options()->preserve_internal_time_seconds,
330 c->mutable_cf_options()->preclude_last_level_data_seconds,
331 &preserve_time_min_seqno_, &preclude_last_level_min_seqno_);
332 }
333 // For accuracy of the GetProximalSeqnoBeforeTime queries above, we only
334 // limit the capacity after them.
335 // Here If we set capacity to the per-SST limit, we could be throwing away
336 // fidelity when a compaction output file has a narrower seqno range than
337 // all the inputs. If we only limit capacity for each compaction output, we
338 // could be doing a lot of unnecessary recomputation in a large compaction
339 // (up to quadratic in number of files). Thus, we do soemthing in the
340 // middle: enforce a resonably large constant size limit substantially
341 // larger than kMaxSeqnoTimePairsPerSST.
342 seqno_to_time_mapping_.SetCapacity(kMaxSeqnoToTimeEntries);
343 }
344}

353void CompactionJob::AcquireSubcompactionResources(
354 int num_extra_required_subcompactions) {
355 TEST_SYNC_POINT("CompactionJob::AcquireSubcompactionResources:0");
356 TEST_SYNC_POINT("CompactionJob::AcquireSubcompactionResources:1");
357 int max_db_compactions =
358 DBImpl::GetBGJobLimits(
359 mutable_db_options_copy_.max_background_flushes,
360 mutable_db_options_copy_.max_background_compactions,
361 mutable_db_options_copy_.max_background_jobs,
362 versions_->GetColumnFamilySet()
363 ->write_controller()
364 ->NeedSpeedupCompaction())
365 .max_compactions;
366 InstrumentedMutexLock l(db_mutex_);
367 // Apply min function first since We need to compute the extra subcompaction
368 // against compaction limits. And then try to reserve threads for extra
369 // subcompactions. The actual number of reserved threads could be less than
370 // the desired number.
371 int available_bg_compactions_against_db_limit =
372 std::max(max_db_compactions - *bg_compaction_scheduled_ -
373 *bg_bottom_compaction_scheduled_,
374 0);
375 // Reservation only supports backgrdoun threads of which the priority is
376 // between BOTTOM and HIGH. Need to degrade the priority to HIGH if the
377 // origin thread_pri_ is higher than that. Similar to ReleaseThreads().
378 extra_num_subcompaction_threads_reserved_ =
379 env_->ReserveThreads(std::min(num_extra_required_subcompactions,
380 available_bg_compactions_against_db_limit),
381 std::min(thread_pri_, Env::Priority::HIGH));
382
383 // Update bg_compaction_scheduled_ or bg_bottom_compaction_scheduled_
384 // depending on if this compaction has the bottommost priority
385 if (thread_pri_ == Env::Priority::BOTTOM) {
386 *bg_bottom_compaction_scheduled_ +=
387 extra_num_subcompaction_threads_reserved_;
388 } else {
389 *bg_compaction_scheduled_ += extra_num_subcompaction_threads_reserved_;
390 }
391}

393void CompactionJob::ShrinkSubcompactionResources(uint64_t num_extra_resources) {
394 // Do nothing when we have zero resources to shrink
395 if (num_extra_resources == 0) {
396 return;
397 }
398 db_mutex_->Lock();
399 // We cannot release threads more than what we reserved before
400 int extra_num_subcompaction_threads_released = env_->ReleaseThreads(
401 (int)num_extra_resources, std::min(thread_pri_, Env::Priority::HIGH));
402 // Update the number of reserved threads and the number of background
403 // scheduled compactions for this compaction job
404 extra_num_subcompaction_threads_reserved_ -=
405 extra_num_subcompaction_threads_released;
406 // TODO (zichen): design a test case with new subcompaction partitioning
407 // when the number of actual partitions is less than the number of planned
408 // partitions
409 assert(extra_num_subcompaction_threads_released == (int)num_extra_resources);
410 // Update bg_compaction_scheduled_ or bg_bottom_compaction_scheduled_
411 // depending on if this compaction has the bottommost priority
412 if (thread_pri_ == Env::Priority::BOTTOM) {
413 *bg_bottom_compaction_scheduled_ -=
414 extra_num_subcompaction_threads_released;
415 } else {
416 *bg_compaction_scheduled_ -= extra_num_subcompaction_threads_released;
417 }
418 db_mutex_->Unlock();
419 TEST_SYNC_POINT("CompactionJob::ShrinkSubcompactionResources:0");
420}

422void CompactionJob::ReleaseSubcompactionResources() {
423 if (extra_num_subcompaction_threads_reserved_ == 0) {
424 return;
425 }
426 {
427 InstrumentedMutexLock l(db_mutex_);
428 // The number of reserved threads becomes larger than 0 only if the
429 // compaction prioity is round robin and there is no sufficient
430 // sub-compactions available
431
432 // The scheduled compaction must be no less than 1 + extra number
433 // subcompactions using acquired resources since this compaction job has not
434 // finished yet
435 assert(*bg_bottom_compaction_scheduled_ >=
436 1 + extra_num_subcompaction_threads_reserved_ ||
437 *bg_compaction_scheduled_ >=
438 1 + extra_num_subcompaction_threads_reserved_);
439 }
440 ShrinkSubcompactionResources(extra_num_subcompaction_threads_reserved_);
441}

443void CompactionJob::GenSubcompactionBoundaries() {
444 // The goal is to find some boundary keys so that we can evenly partition
445 // the compaction input data into max_subcompactions ranges.
446 // For every input file, we ask TableReader to estimate 128 anchor points
447 // that evenly partition the input file into 128 ranges and the range
448 // sizes. This can be calculated by scanning index blocks of the file.
449 // Once we have the anchor points for all the input files, we merge them
450 // together and try to find keys dividing ranges evenly.
451 // For example, if we have two input files, and each returns following
452 // ranges:
453 // File1: (a1, 1000), (b1, 1200), (c1, 1100)
454 // File2: (a2, 1100), (b2, 1000), (c2, 1000)
455 // We total sort the keys to following:
456 // (a1, 1000), (a2, 1100), (b1, 1200), (b2, 1000), (c1, 1100), (c2, 1000)
457 // We calculate the total size by adding up all ranges' size, which is 6400.
458 // If we would like to partition into 2 subcompactions, the target of the
459 // range size is 3200. Based on the size, we take "b1" as the partition key
460 // since the first three ranges would hit 3200.
461 //
462 // Note that the ranges are actually overlapping. For example, in the example
463 // above, the range ending with "b1" is overlapping with the range ending with
464 // "b2". So the size 1000+1100+1200 is an underestimation of data size up to
465 // "b1". In extreme cases where we only compact N L0 files, a range can
466 // overlap with N-1 other ranges. Since we requested a relatively large number
467 // (128) of ranges from each input files, even N range overlapping would
468 // cause relatively small inaccuracy.
469 ReadOptions read_options(Env::IOActivity::kCompaction);
470 read_options.rate_limiter_priority = GetRateLimiterPriority();
471 auto* c = compact_->compaction;
472 if (c->mutable_cf_options()->table_factory->Name() ==
473 TableFactory::kPlainTableName()) {
474 return;
475 }
476
477 if (c->max_subcompactions() <= 1 &&
478 !(c->immutable_options()->compaction_pri == kRoundRobin &&
479 c->immutable_options()->compaction_style == kCompactionStyleLevel)) {
480 return;
481 }
482 auto* cfd = c->column_family_data();
483 const Comparator* cfd_comparator = cfd->user_comparator();
484 const InternalKeyComparator& icomp = cfd->internal_comparator();
485
486 auto* v = compact_->compaction->input_version();
487 int base_level = v->storage_info()->base_level();
488 InstrumentedMutexUnlock unlock_guard(db_mutex_);
489
490 uint64_t total_size = 0;
491 std::vector<TableReader::Anchor> all_anchors;
492 int start_lvl = c->start_level();
493 int out_lvl = c->output_level();
494
495 for (size_t lvl_idx = 0; lvl_idx < c->num_input_levels(); lvl_idx++) {
496 int lvl = c->level(lvl_idx);
497 if (lvl >= start_lvl && lvl <= out_lvl) {
498 const LevelFilesBrief* flevel = c->input_levels(lvl_idx);
499 size_t num_files = flevel->num_files;
500
501 if (num_files == 0) {
502 continue;
503 }
504
505 for (size_t i = 0; i < num_files; i++) {
506 FileMetaData* f = flevel->files[i].file_metadata;
507 std::vector<TableReader::Anchor> my_anchors;
508 Status s = cfd->table_cache()->ApproximateKeyAnchors(
509 read_options, icomp, *f, *c->mutable_cf_options(), my_anchors);
510 if (!s.ok() || my_anchors.empty()) {
511 my_anchors.emplace_back(f->largest.user_key(), f->fd.GetFileSize());
512 }
513 for (auto& ac : my_anchors) {
514 // Can be optimize to avoid this loop.
515 total_size += ac.range_size;
516 }
517
518 all_anchors.insert(all_anchors.end(), my_anchors.begin(),
519 my_anchors.end());
520 }
521 }
522 }
523 // Here we total sort all the anchor points across all files and go through
524 // them in the sorted order to find partitioning boundaries.
525 // Not the most efficient implementation. A much more efficient algorithm
526 // probably exists. But they are more complex. If performance turns out to
527 // be a problem, we can optimize.
528 std::sort(
529 all_anchors.begin(), all_anchors.end(),
530 [cfd_comparator](TableReader::Anchor& a, TableReader::Anchor& b) -> bool {
531 return cfd_comparator->CompareWithoutTimestamp(a.user_key, b.user_key) <
532 0;
533 });
534
535 // Remove duplicated entries from boundaries.
536 all_anchors.erase(
537 std::unique(all_anchors.begin(), all_anchors.end(),
538 [cfd_comparator](TableReader::Anchor& a,
539 TableReader::Anchor& b) -> bool {
540 return cfd_comparator->CompareWithoutTimestamp(
541 a.user_key, b.user_key) == 0;
542 }),
543 all_anchors.end());
544
545 // Get the number of planned subcompactions, may update reserve threads
546 // and update extra_num_subcompaction_threads_reserved_ for round-robin
547 uint64_t num_planned_subcompactions;
548 if (c->immutable_options()->compaction_pri == kRoundRobin &&
549 c->immutable_options()->compaction_style == kCompactionStyleLevel) {
550 // For round-robin compaction prioity, we need to employ more
551 // subcompactions (may exceed the max_subcompaction limit). The extra
552 // subcompactions will be executed using reserved threads and taken into
553 // account bg_compaction_scheduled or bg_bottom_compaction_scheduled.
554
555 // Initialized by the number of input files
556 num_planned_subcompactions = static_cast<uint64_t>(c->num_input_files(0));
557 uint64_t max_subcompactions_limit = GetSubcompactionsLimit();
558 if (max_subcompactions_limit < num_planned_subcompactions) {
559 // Assert two pointers are not empty so that we can use extra
560 // subcompactions against db compaction limits
561 assert(bg_bottom_compaction_scheduled_ != nullptr);
562 assert(bg_compaction_scheduled_ != nullptr);
563 // Reserve resources when max_subcompaction is not sufficient
564 AcquireSubcompactionResources(
565 (int)(num_planned_subcompactions - max_subcompactions_limit));
566 // Subcompactions limit changes after acquiring additional resources.
567 // Need to call GetSubcompactionsLimit() again to update the number
568 // of planned subcompactions
569 num_planned_subcompactions =
570 std::min(num_planned_subcompactions, GetSubcompactionsLimit());
571 } else {
572 num_planned_subcompactions = max_subcompactions_limit;
573 }
574 } else {
575 num_planned_subcompactions = GetSubcompactionsLimit();
576 }
577
578 TEST_SYNC_POINT_CALLBACK("CompactionJob::GenSubcompactionBoundaries:0",
579 &num_planned_subcompactions);
580 if (num_planned_subcompactions == 1) {
581 return;
582 }
583
584 // Group the ranges into subcompactions
585 uint64_t target_range_size = std::max(
586 total_size / num_planned_subcompactions,
587 MaxFileSizeForLevel(
588 *(c->mutable_cf_options()), out_lvl,
589 c->immutable_options()->compaction_style, base_level,
590 c->immutable_options()->level_compaction_dynamic_level_bytes));
591
592 if (target_range_size >= total_size) {
593 return;
594 }
595
596 uint64_t next_threshold = target_range_size;
597 uint64_t cumulative_size = 0;
598 uint64_t num_actual_subcompactions = 1U;
599 for (TableReader::Anchor& anchor : all_anchors) {
600 cumulative_size += anchor.range_size;
601 if (cumulative_size > next_threshold) {
602 next_threshold += target_range_size;
603 num_actual_subcompactions++;
604 boundaries_.push_back(anchor.user_key);
605 }
606 if (num_actual_subcompactions == num_planned_subcompactions) {
607 break;
608 }
609 }
610 TEST_SYNC_POINT_CALLBACK("CompactionJob::GenSubcompactionBoundaries:1",
611 &num_actual_subcompactions);
612 // Shrink extra subcompactions resources when extra resrouces are acquired
613 ShrinkSubcompactionResources(
614 std::min((int)(num_planned_subcompactions - num_actual_subcompactions),
615 extra_num_subcompaction_threads_reserved_));
616}

630 // Launch a thread for each of subcompactions 1...num_threads-1
631 std::vector<port::Thread> thread_pool;
632 thread_pool.reserve(num_threads - 1);

634 thread_pool.emplace_back(&CompactionJob::ProcessKeyValueCompaction, this,
635 &compact_->sub_compact_states[i]);
636 }

639 // others) in the current thread to be efficient with resources

641
642 // Wait for all other threads (if there are any) to finish execution
643 for (auto& thread : thread_pool) {
644 thread.join();
645 }

655 compaction_stats_.stats.micros);

657 compaction_stats_.stats.cpu_micros);
658
659 TEST_SYNC_POINT("CompactionJob::Run:BeforeVerify");
660
661 // Check if any thread encountered an error during execution
662 Status status;
663 IOStatus io_s;
664 bool wrote_new_blob_files = false;
665
666 for (const auto& state : compact_->sub_compact_states) {
667 if (!state.status.ok()) {
668 status = state.status;
669 io_s = state.io_status;
670 break;

672
673 if (state.Current().HasBlobFileAdditions()) {
674 wrote_new_blob_files = true;
675 }
676 }
677
678 if (io_status_.ok()) {

680 }
681 if (status.ok()) {

684 if (output_directory_) {

686 IOOptions(), dbg,
687 DirFsyncOptions(DirFsyncOptions::FsyncReason::kNewFileSynced));
688 }
689
690 if (io_s.ok() && wrote_new_blob_files && blob_output_directory_ &&
691 blob_output_directory_ != output_directory_) {

693 IOOptions(), dbg,
694 DirFsyncOptions(DirFsyncOptions::FsyncReason::kNewFileSynced));

696 }
697 if (io_status_.ok()) {
698 io_status_ = io_s;
699 }

702 }

721 // OptimizeForCompactionTableRead here because this is a special case
722 // after we finish the table building No matter whether
723 // use_direct_io_for_flush_and_compaction is true, we will regard this
724 // verification as user reads since the goal is to cache it here for
725 // further user reads

734 /*table_reader_ptr=*/nullptr,

736 compact_->compaction->output_level()),
737 TableReaderCaller::kCompactionRefill, /*arena=*/nullptr,
738 /*skip_filters=*/false, compact_->compaction->output_level(),

742 /*largest_compaction_key=*/nullptr,

777 for (auto& thread : thread_pool) {
778 thread.join();
779 }

784 break;
785 }
786 }

790 TEST_SYNC_POINT("CompactionJob::ReleaseSubcompactionResources:0");
791 TEST_SYNC_POINT("CompactionJob::ReleaseSubcompactionResources:1");
792
793 TablePropertiesCollection tp;
794 for (const auto& state : compact_->sub_compact_states) {
795 for (const auto& output : state.GetOutputs()) {
796 auto fn =
797 TableFileName(state.compaction->immutable_options()->cf_paths,
798 output.meta.fd.GetNumber(), output.meta.fd.GetPathId());
799 compact_->compaction->SetOutputTableProperties(fn,
800 output.table_properties);
801 }
802 }
803
804 // Before the compaction starts, is_remote_compaction was set to true if
805 // compaction_service is set. We now know whether each sub_compaction was
806 // done remotely or not. Reset is_remote_compaction back to false and allow
807 // AggregateCompactionStats() to set the right value.
808 compaction_job_stats_->is_remote_compaction = false;
809
810 // Finish up all bookkeeping to unify the subcompaction results.
811 compact_->AggregateCompactionStats(compaction_stats_, *compaction_job_stats_);
812 uint64_t num_input_range_del = 0;
813 bool ok = UpdateCompactionStats(&num_input_range_del);
814 // (Sub)compactions returned ok, do sanity check on the number of input keys.
815 if (status.ok() && ok && compaction_job_stats_->has_num_input_records) {
816 size_t ts_sz = compact_->compaction->column_family_data()
817 ->user_comparator()
818 ->timestamp_size();
819 // When trim_ts_ is non-empty, CompactionIterator takes
820 // HistoryTrimmingIterator as input iterator and sees a trimmed view of
821 // input keys. So the number of keys it processed is not suitable for
822 // verification here.
823 // TODO: support verification when trim_ts_ is non-empty.
824 if (!(ts_sz > 0 && !trim_ts_.empty())) {
825 assert(compaction_stats_.stats.num_input_records > 0);
826 // TODO: verify the number of range deletion entries.
827 uint64_t expected =
828 compaction_stats_.stats.num_input_records - num_input_range_del;
829 uint64_t actual = compaction_job_stats_->num_input_records;
830 if (expected != actual) {
831 char scratch[2345];
832 compact_->compaction->Summary(scratch, sizeof(scratch));
833 std::string msg =
834 "Compaction number of input keys does not match "
835 "number of keys processed. Expected " +
836 std::to_string(expected) + " but processed " +
837 std::to_string(actual) + ". Compaction summary: " + scratch;
838 ROCKS_LOG_WARN(
839 db_options_.info_log, "[%s] [JOB %d] Compaction with status: %s",
840 compact_->compaction->column_family_data()->GetName().c_str(),
841 job_context_->job_id, msg.c_str());
842 if (db_options_.compaction_verify_record_count) {
843 status = Status::Corruption(msg);
844 }
845 }
846 }
847 }
848 RecordCompactionIOStats();
849 LogFlush(db_options_.info_log);
850 TEST_SYNC_POINT("CompactionJob::Run():End");
851 compact_->status = status;
852 TEST_SYNC_POINT_CALLBACK("CompactionJob::Run():EndStatusSet", &status);
853 return status;
854}

1080void CompactionJob::ProcessKeyValueCompaction(SubcompactionState* sub_compact) {
1081 assert(sub_compact);
1082 assert(sub_compact->compaction);
1083 if (db_options_.compaction_service) {
1084 CompactionServiceJobStatus comp_status =
1085 ProcessKeyValueCompactionWithCompactionService(sub_compact);
1086 if (comp_status == CompactionServiceJobStatus::kSuccess ||
1087 comp_status == CompactionServiceJobStatus::kFailure) {
1088 return;
1089 }
1090 // fallback to local compaction
1091 assert(comp_status == CompactionServiceJobStatus::kUseLocal);
1092 sub_compact->compaction_job_stats.is_remote_compaction = false;
1093 }
1094
1095 uint64_t prev_cpu_micros = db_options_.clock->CPUMicros();
1096
1097 ColumnFamilyData* cfd = sub_compact->compaction->column_family_data();
1098
1099 // Create compaction filter and fail the compaction if
1100 // IgnoreSnapshots() = false because it is not supported anymore
1101 const CompactionFilter* compaction_filter =
1102 cfd->ioptions()->compaction_filter;
1103 std::unique_ptr<CompactionFilter> compaction_filter_from_factory = nullptr;
1104 if (compaction_filter == nullptr) {
1105 compaction_filter_from_factory =
1106 sub_compact->compaction->CreateCompactionFilter();
1107 compaction_filter = compaction_filter_from_factory.get();
1108 }
1109 if (compaction_filter != nullptr && !compaction_filter->IgnoreSnapshots()) {
1110 sub_compact->status = Status::NotSupported(
1111 "CompactionFilter::IgnoreSnapshots() = false is not supported "
1112 "anymore.");
1113 return;
1114 }
1115
1116 NotifyOnSubcompactionBegin(sub_compact);
1117
1118 auto range_del_agg = std::make_unique<CompactionRangeDelAggregator>(
1119 &cfd->internal_comparator(), existing_snapshots_, &full_history_ts_low_,
1120 &trim_ts_);
1121
1122 // TODO: since we already use C++17, should use
1123 // std::optional<const Slice> instead.
1124 const std::optional<Slice> start = sub_compact->start;
1125 const std::optional<Slice> end = sub_compact->end;
1126
1127 std::optional<Slice> start_without_ts;
1128 std::optional<Slice> end_without_ts;
1129
1130 ReadOptions read_options;
1131 read_options.verify_checksums = true;
1132 read_options.fill_cache = false;
1133 read_options.rate_limiter_priority = GetRateLimiterPriority();
1134 read_options.io_activity = Env::IOActivity::kCompaction;
1135 // Compaction iterators shouldn't be confined to a single prefix.
1136 // Compactions use Seek() for
1137 // (a) concurrent compactions,
1138 // (b) CompactionFilter::Decision::kRemoveAndSkipUntil.
1139 read_options.total_order_seek = true;
1140
1141 const WriteOptions write_options(Env::IOPriority::IO_LOW,
1142 Env::IOActivity::kCompaction);
1143
1144 // Remove the timestamps from boundaries because boundaries created in
1145 // GenSubcompactionBoundaries doesn't strip away the timestamp.
1146 size_t ts_sz = cfd->user_comparator()->timestamp_size();
1147 if (start.has_value()) {
1148 read_options.iterate_lower_bound = &(*start);
1149 if (ts_sz > 0) {
1150 start_without_ts = StripTimestampFromUserKey(*start, ts_sz);
1151 read_options.iterate_lower_bound = &(*start_without_ts);
1152 }
1153 }
1154 if (end.has_value()) {
1155 read_options.iterate_upper_bound = &(*end);
1156 if (ts_sz > 0) {
1157 end_without_ts = StripTimestampFromUserKey(*end, ts_sz);
1158 read_options.iterate_upper_bound = &(*end_without_ts);
1159 }
1160 }
1161
1162 // Although the v2 aggregator is what the level iterator(s) know about,
1163 // the AddTombstones calls will be propagated down to the v1 aggregator.
1164 std::unique_ptr<InternalIterator> raw_input(versions_->MakeInputIterator(
1165 read_options, sub_compact->compaction, range_del_agg.get(),
1166 file_options_for_read_, start, end));
1167 InternalIterator* input = raw_input.get();
1168
1169 IterKey start_ikey;
1170 IterKey end_ikey;
1171 Slice start_slice;
1172 Slice end_slice;
1173 Slice start_user_key{};
1174 Slice end_user_key{};
1175
1176 static constexpr char kMaxTs[] =
1177 "\xff\xff\xff\xff\xff\xff\xff\xff\xff\xff\xff\xff\xff\xff\xff\xff";
1178 Slice ts_slice;
1179 std::string max_ts;
1180 if (ts_sz > 0) {
1181 if (ts_sz <= strlen(kMaxTs)) {
1182 ts_slice = Slice(kMaxTs, ts_sz);
1183 } else {
1184 max_ts = std::string(ts_sz, '\xff');
1185 ts_slice = Slice(max_ts);
1186 }
1187 }
1188
1189 if (start.has_value()) {
1190 start_ikey.SetInternalKey(*start, kMaxSequenceNumber, kValueTypeForSeek);
1191 if (ts_sz > 0) {

1447 // only range-dels, no file was opened, to save the range-dels, it need to
1448 // create a new output file.
1449 status = sub_compact->CloseCompactionFiles(status, open_file_func,
1450 close_file_func);
1451
1452 if (blob_file_builder) {
1453 if (status.ok()) {
1454 status = blob_file_builder->Finish();
1455 } else {
1456 blob_file_builder->Abandon(status);
1457 }
1458 blob_file_builder.reset();
1459 sub_compact->Current().UpdateBlobStats();
1460 }

1462 uint64_t cur_cpu_micros = db_options_.clock->CPUMicros();

1464 cur_cpu_micros - prev_cpu_micros;
1465 RecordTick(stats_, COMPACTION_CPU_TOTAL_TIME,
1466 cur_cpu_micros - last_cpu_micros);

1469 sub_compact->compaction_job_stats.file_write_nanos +=
1470 IOSTATS(write_nanos) - prev_write_nanos;
1471 sub_compact->compaction_job_stats.file_fsync_nanos +=
1472 IOSTATS(fsync_nanos) - prev_fsync_nanos;
1473 sub_compact->compaction_job_stats.file_range_sync_nanos +=
1474 IOSTATS(range_sync_nanos) - prev_range_sync_nanos;
1475 sub_compact->compaction_job_stats.file_prepare_write_nanos +=
1476 IOSTATS(prepare_write_nanos) - prev_prepare_write_nanos;
1477 sub_compact->compaction_job_stats.cpu_micros -=
1478 (IOSTATS(cpu_write_nanos) - prev_cpu_write_nanos +
1479 IOSTATS(cpu_read_nanos) - prev_cpu_read_nanos) /
1480 1000;
1481 if (prev_perf_level != PerfLevel::kEnableTimeAndCPUTimeExceptForMutex) {
1482 SetPerfLevel(prev_perf_level);
1483 }
1484 }
1485#ifdef ROCKSDB_ASSERT_STATUS_CHECKED
1486 if (!status.ok()) {
1487 if (c_iter) {
1488 c_iter->status().PermitUncheckedError();
1489 }
1490 if (input) {
1491 input->status().PermitUncheckedError();
1492 }
1493 }
1494#endif // ROCKSDB_ASSERT_STATUS_CHECKED
1495
1496 blob_counter.reset();
1497 clip.reset();
1498 raw_input.reset();
1499 sub_compact->status = status;
1500 NotifyOnSubcompactionCompleted(sub_compact);
1501}

1544Status CompactionJob::FinishCompactionOutputFile(
1545 const Status& input_status, SubcompactionState* sub_compact,
1546 CompactionOutputs& outputs, const Slice& next_table_min_key,
1547 const Slice* comp_start_user_key, const Slice* comp_end_user_key) {
1548 AutoThreadOperationStageUpdater stage_updater(
1549 ThreadStatus::STAGE_COMPACTION_SYNC_FILE);
1550 assert(sub_compact != nullptr);
1551 assert(outputs.HasBuilder());
1552
1553 FileMetaData* meta = outputs.GetMetaData();
1554 uint64_t output_number = meta->fd.GetNumber();
1555 assert(output_number != 0);
1556
1557 ColumnFamilyData* cfd = sub_compact->compaction->column_family_data();
1558 std::string file_checksum = kUnknownFileChecksum;
1559 std::string file_checksum_func_name = kUnknownFileChecksumFuncName;
1560
1561 // Check for iterator errors
1562 Status s = input_status;
1563
1564 // Add range tombstones
1565 auto earliest_snapshot = kMaxSequenceNumber;
1566 if (existing_snapshots_.size() > 0) {
1567 earliest_snapshot = existing_snapshots_[0];
1568 }
1569 if (s.ok()) {
1570 CompactionIterationStats range_del_out_stats;
1571 // if the compaction supports per_key_placement, only output range dels to
1572 // the penultimate level.
1573 // Note: Use `bottommost_level_ = true` for both bottommost and
1574 // output_to_penultimate_level compaction here, as it's only used to decide
1575 // if range dels could be dropped.
1576 if (outputs.HasRangeDel()) {
1577 s = outputs.AddRangeDels(comp_start_user_key, comp_end_user_key,
1578 range_del_out_stats, bottommost_level_,
1579 cfd->internal_comparator(), earliest_snapshot,
1580 next_table_min_key, full_history_ts_low_);
1581 }
1582 RecordDroppedKeys(range_del_out_stats, &sub_compact->compaction_job_stats);
1583 TEST_SYNC_POINT("CompactionJob::FinishCompactionOutputFile1");
1584 }
1585
1586 const uint64_t current_entries = outputs.NumEntries();
1587
1588 s = outputs.Finish(s, seqno_to_time_mapping_);
1589 TEST_SYNC_POINT_CALLBACK(
1590 "CompactionJob::FinishCompactionOutputFile()::AfterFinish", &s);
1591
1592 if (s.ok()) {
1593 // With accurate smallest and largest key, we can get a slightly more
1594 // accurate oldest ancester time.
1595 // This makes oldest ancester time in manifest more accurate than in
1596 // table properties. Not sure how to resolve it.
1597 if (meta->smallest.size() > 0 && meta->largest.size() > 0) {
1598 uint64_t refined_oldest_ancester_time;
1599 Slice new_smallest = meta->smallest.user_key();
1600 Slice new_largest = meta->largest.user_key();
1601 if (!new_largest.empty() && !new_smallest.empty()) {
1602 refined_oldest_ancester_time =
1603 sub_compact->compaction->MinInputFileOldestAncesterTime(
1604 &(meta->smallest), &(meta->largest));
1605 if (refined_oldest_ancester_time !=
1606 std::numeric_limits<uint64_t>::max()) {
1607 meta->oldest_ancester_time = refined_oldest_ancester_time;
1608 }
1609 }
1610 }
1611 }
1612
1613 // Finish and check for file errors
1614 IOStatus io_s = outputs.WriterSyncClose(s, db_options_.clock, stats_,
1615 db_options_.use_fsync);
1616
1617 if (s.ok() && io_s.ok()) {
1618 file_checksum = meta->file_checksum;
1619 file_checksum_func_name = meta->file_checksum_func_name;
1620 }
1621
1622 if (s.ok()) {
1623 s = io_s;
1624 }
1625 if (sub_compact->io_status.ok()) {
1626 sub_compact->io_status = io_s;
1627 // Since this error is really a copy of the
1628 // "normal" status, it does not also need to be checked
1629 sub_compact->io_status.PermitUncheckedError();
1630 }
1631
1632 TableProperties tp;
1633 if (s.ok()) {
1634 tp = outputs.GetTableProperties();
1635 }
1636
1637 if (s.ok() && current_entries == 0 && tp.num_range_deletions == 0) {
1638 // If there is nothing to output, no necessary to generate a sst file.
1639 // This happens when the output level is bottom level, at the same time
1640 // the sub_compact output nothing.
1641 std::string fname =
1642 TableFileName(sub_compact->compaction->immutable_options()->cf_paths,
1643 meta->fd.GetNumber(), meta->fd.GetPathId());
1644
1645 // TODO(AR) it is not clear if there are any larger implications if
1646 // DeleteFile fails here
1647 Status ds = env_->DeleteFile(fname);
1648 if (!ds.ok()) {
1649 ROCKS_LOG_WARN(
1650 db_options_.info_log,
1651 "[%s] [JOB %d] Unable to remove SST file for table #%" PRIu64
1652 " at bottom level%s",
1653 cfd->GetName().c_str(), job_id_, output_number,
1654 meta->marked_for_compaction ? " (need compaction)" : "");
1655 }
1656
1657 // Also need to remove the file from outputs, or it will be added to the
1658 // VersionEdit.
1659 outputs.RemoveLastOutput();
1660 meta = nullptr;
1661 }
1662
1663 if (s.ok() && (current_entries > 0 || tp.num_range_deletions > 0)) {
1664 // Output to event logger and fire events.
1665 outputs.UpdateTableProperties();
1666 ROCKS_LOG_INFO(db_options_.info_log,
1667 "[%s] [JOB %d] Generated table #%" PRIu64 ": %" PRIu64
1668 " keys, %" PRIu64 " bytes%s, temperature: %s",
1669 cfd->GetName().c_str(), job_id_, output_number,
1670 current_entries, meta->fd.file_size,
1671 meta->marked_for_compaction ? " (need compaction)" : "",
1672 temperature_to_string[meta->temperature].c_str());
1673 }
1674 std::string fname;
1675 FileDescriptor output_fd;
1676 uint64_t oldest_blob_file_number = kInvalidBlobFileNumber;
1677 Status status_for_listener = s;
1678 if (meta != nullptr) {
1679 fname = GetTableFileName(meta->fd.GetNumber());
1680 output_fd = meta->fd;
1681 oldest_blob_file_number = meta->oldest_blob_file_number;
1682 } else {
1683 fname = "(nil)";
1684 if (s.ok()) {
1685 status_for_listener = Status::Aborted("Empty SST file not kept");
1686 }
1687 }
1688 EventHelpers::LogAndNotifyTableFileCreationFinished(
1689 event_logger_, cfd->ioptions()->listeners, dbname_, cfd->GetName(), fname,
1690 job_id_, output_fd, oldest_blob_file_number, tp,
1691 TableFileCreationReason::kCompaction, status_for_listener, file_checksum,
1692 file_checksum_func_name);
1693
1694 // Report new file to SstFileManagerImpl
1695 auto sfm =
1696 static_cast<SstFileManagerImpl*>(db_options_.sst_file_manager.get());
1697 if (sfm && meta != nullptr && meta->fd.GetPathId() == 0) {
1698 Status add_s = sfm->OnAddFile(fname);
1699 if (!add_s.ok() && s.ok()) {
1700 s = add_s;
1701 }
1702 if (sfm->IsMaxAllowedSpaceReached()) {
1703 // TODO(ajkr): should we return OK() if max space was reached by the final
1704 // compaction output file (similarly to how flush works when full)?
1705 s = Status::SpaceLimit("Max allowed space was reached");
1706 TEST_SYNC_POINT(
1707 "CompactionJob::FinishCompactionOutputFile:MaxAllowedSpaceReached");
1708 InstrumentedMutexLock l(db_mutex_);
1709 db_error_handler_->SetBGError(s, BackgroundErrorReason::kCompaction);
1710 }
1711 }
1712
1713 outputs.ResetBuilder();
1714 return s;
1715}
