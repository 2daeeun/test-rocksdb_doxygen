48 {
49 InstrumentedMutexLock l(&mu_);
50 closing_ = true;
51 cv_.SignalAll();
52 }

61Status DeleteScheduler::DeleteFile(const std::string& file_path,
62 const std::string& dir_to_sync,
63 const bool force_bg) {
64 uint64_t total_size = sst_file_manager_->GetTotalSize();
65 if (rate_bytes_per_sec_.load() <= 0 ||
66 (!force_bg &&
67 total_trash_size_.load() > total_size * max_trash_db_ratio_.load())) {
68 // Rate limiting is disabled or trash size makes up more than
69 // max_trash_db_ratio_ (default 25%) of the total DB size
70 Status s = DeleteFileImmediately(file_path, /*accounted=*/true);
71 if (s.ok()) {
72 ROCKS_LOG_INFO(info_log_,
73 "Deleted file %s immediately, rate_bytes_per_sec %" PRIi64
74 ", total_trash_size %" PRIu64 ", total_size %" PRIi64
75 ", max_trash_db_ratio %lf",
76 file_path.c_str(), rate_bytes_per_sec_.load(),
77 total_trash_size_.load(), total_size,
78 max_trash_db_ratio_.load());
79 }
80 return s;
81 }
82 return AddFileToDeletionQueue(file_path, dir_to_sync, /*bucket=*/std::nullopt,
83 /*accounted=*/true);
84}

86Status DeleteScheduler::DeleteUnaccountedFile(const std::string& file_path,
87 const std::string& dir_to_sync,
88 const bool force_bg,
89 std::optional<int32_t> bucket) {
90 uint64_t num_hard_links = 1;
91 fs_->NumFileLinks(file_path, IOOptions(), &num_hard_links, nullptr)
92 .PermitUncheckedError();
93
94 // We can tolerate rare races where we might immediately delete both links
95 // to a file.
96 if (rate_bytes_per_sec_.load() <= 0 || (!force_bg && num_hard_links > 1)) {
97 Status s = DeleteFileImmediately(file_path, /*accounted=*/false);
98 if (s.ok()) {
99 ROCKS_LOG_INFO(info_log_,
100 "Deleted file %s immediately, rate_bytes_per_sec %" PRIi64,
101 file_path.c_str(), rate_bytes_per_sec_.load());
102 }
103 return s;
104 }
105 return AddFileToDeletionQueue(file_path, dir_to_sync, bucket,
106 /*accounted=*/false);
107}

123Status DeleteScheduler::AddFileToDeletionQueue(const std::string& file_path,
124 const std::string& dir_to_sync,
125 std::optional<int32_t> bucket,
126 bool accounted) {
127 // Move file to trash
128 std::string trash_file;
129 Status s = MarkAsTrash(file_path, accounted, &trash_file);
130 ROCKS_LOG_INFO(info_log_, "Mark file: %s as trash -- %s", trash_file.c_str(),
131 s.ToString().c_str());
132
133 if (!s.ok()) {
134 ROCKS_LOG_ERROR(info_log_, "Failed to mark %s as trash -- %s",
135 file_path.c_str(), s.ToString().c_str());
136 s = fs_->DeleteFile(file_path, IOOptions(), nullptr);
137 if (s.ok()) {
138 s = OnDeleteFile(file_path, accounted);
139 ROCKS_LOG_INFO(info_log_, "Deleted file %s immediately",
140 trash_file.c_str());
141 InstrumentedMutexLock l(&mu_);
142 RecordTick(stats_.get(), FILES_DELETED_IMMEDIATELY);
143 }
144 return s;
145 }
146
147 // Update the total trash size
148 if (accounted) {
149 uint64_t trash_file_size = 0;
150 IOStatus io_s =
151 fs_->GetFileSize(trash_file, IOOptions(), &trash_file_size, nullptr);
152 if (io_s.ok()) {
153 total_trash_size_.fetch_add(trash_file_size);
154 }
155 }
156 //**TODO: What should we do if we failed to
157 // get the file size?
158
159 // Add file to delete queue
160 {
161 InstrumentedMutexLock l(&mu_);
162 RecordTick(stats_.get(), FILES_MARKED_TRASH);
163 queue_.emplace(trash_file, dir_to_sync, accounted, bucket);
164 pending_files_++;
165 if (bucket.has_value()) {
166 auto iter = pending_files_in_buckets_.find(bucket.value());
167 assert(iter != pending_files_in_buckets_.end());
168 if (iter != pending_files_in_buckets_.end()) {
169 iter->second++;
170 }
171 }
172 if (pending_files_ == 1) {
173 cv_.SignalAll();
174 }
175 }
176 return s;
177}

185bool DeleteScheduler::IsTrashFile(const std::string& file_path) {
186 return (file_path.size() >= kTrashExtension.size() &&
187 file_path.rfind(kTrashExtension) ==
188 file_path.size() - kTrashExtension.size());

191Status DeleteScheduler::CleanupDirectory(Env* env, SstFileManagerImpl* sfm,
192 const std::string& path) {
193 Status s;
194 // Check if there are any files marked as trash in this path
195 std::vector<std::string> files_in_path;
196 const auto& fs = env->GetFileSystem();
197 IOOptions io_opts;
198 io_opts.do_not_recurse = true;
199 s = fs->GetChildren(path, io_opts, &files_in_path,
200 /*IODebugContext*=*/nullptr);
201 if (!s.ok()) {
202 return s;
203 }
204 for (const std::string& current_file : files_in_path) {
205 if (!DeleteScheduler::IsTrashFile(current_file)) {
206 // not a trash file, skip
207 continue;
208 }
209
210 Status file_delete;
211 std::string trash_file = path + "/" + current_file;
212 if (sfm) {
213 // We have an SstFileManager that will schedule the file delete
214 s = sfm->OnAddFile(trash_file);
215 file_delete = sfm->ScheduleFileDeletion(trash_file, path);
216 } else {
217 // Delete the file immediately
218 file_delete = env->DeleteFile(trash_file);
219 }
220
221 if (s.ok() && !file_delete.ok()) {
222 s = file_delete;
223 }
224 }
225
226 return s;
227}

229Status DeleteScheduler::MarkAsTrash(const std::string& file_path,
230 bool accounted, std::string* trash_file) {
231 // Sanity check of the path
232 size_t idx = file_path.rfind('/');
233 if (idx == std::string::npos || idx == file_path.size() - 1) {
234 return Status::InvalidArgument("file_path is corrupted");
235 }
236
237 if (DeleteScheduler::IsTrashFile(file_path)) {
238 // This is already a trash file
239 *trash_file = file_path;
240 return Status::OK();
241 }
242
243 *trash_file = file_path + kTrashExtension;
244 // TODO(tec) : Implement Env::RenameFileIfNotExist and remove
245 // file_move_mu mutex.
246 int cnt = 0;
247 Status s;
248 InstrumentedMutexLock l(&file_move_mu_);
249 while (true) {
250 s = fs_->FileExists(*trash_file, IOOptions(), nullptr);
251 if (s.IsNotFound()) {
252 // We found a path for our file in trash
253 s = fs_->RenameFile(file_path, *trash_file, IOOptions(), nullptr);
254 break;
255 } else if (s.ok()) {
256 // Name conflict, generate new random suffix
257 *trash_file = file_path + std::to_string(cnt) + kTrashExtension;
258 } else {
259 // Error during FileExists call, we cannot continue
260 break;
261 }
262 cnt++;
263 }
264 if (s.ok() && accounted) {
265 s = sst_file_manager_->OnMoveFile(file_path, *trash_file);
266 }
267 return s;
268}

270void DeleteScheduler::BackgroundEmptyTrash() {
271 TEST_SYNC_POINT("DeleteScheduler::BackgroundEmptyTrash");
272
273 while (true) {
274 InstrumentedMutexLock l(&mu_);
275 while (queue_.empty() && !closing_) {
276 cv_.Wait();
277 }
278
279 if (closing_) {
280 return;
281 }
282
283 // Delete all files in queue_
284 uint64_t start_time = clock_->NowMicros();
285 uint64_t total_deleted_bytes = 0;
286 int64_t current_delete_rate = rate_bytes_per_sec_.load();
287 while (!queue_.empty() && !closing_) {
288 // Satisfy static analysis.
289 std::optional<int32_t> bucket = std::nullopt;
290 if (current_delete_rate != rate_bytes_per_sec_.load()) {
291 // User changed the delete rate
292 current_delete_rate = rate_bytes_per_sec_.load();
293 start_time = clock_->NowMicros();
294 total_deleted_bytes = 0;
295 ROCKS_LOG_INFO(info_log_, "rate_bytes_per_sec is changed to %" PRIi64,
296 current_delete_rate);
297 }
298
299 // Get new file to delete
300 const FileAndDir& fad = queue_.front();
301 std::string path_in_trash = fad.fname;
302 std::string dir_to_sync = fad.dir;
303 bool accounted = fad.accounted;
304 bucket = fad.bucket;
305
306 // We don't need to hold the lock while deleting the file
307 mu_.Unlock();
308 uint64_t deleted_bytes = 0;
309 bool is_complete = true;
310 // Delete file from trash and update total_penlty value
311 Status s = DeleteTrashFile(path_in_trash, dir_to_sync, accounted,
312 &deleted_bytes, &is_complete);
313 total_deleted_bytes += deleted_bytes;
314 mu_.Lock();
315 if (is_complete) {
316 RecordTick(stats_.get(), FILES_DELETED_FROM_TRASH_QUEUE);
317 queue_.pop();
318 }
319
320 if (!s.ok()) {
321 bg_errors_[path_in_trash] = s;
322 }
323
324 // Apply penalty if necessary
325 uint64_t total_penalty;
326 if (current_delete_rate > 0) {
327 // rate limiting is enabled
328 total_penalty =
329 ((total_deleted_bytes * kMicrosInSecond) / current_delete_rate);
330 ROCKS_LOG_INFO(info_log_,
331 "Rate limiting is enabled with penalty %" PRIu64
332 " after deleting file %s",
333 total_penalty, path_in_trash.c_str());
334 while (!closing_ && !cv_.TimedWait(start_time + total_penalty)) {
335 }
336 } else {
337 // rate limiting is disabled
338 total_penalty = 0;
339 ROCKS_LOG_INFO(info_log_,
340 "Rate limiting is disabled after deleting file %s",
341 path_in_trash.c_str());
342 }
343 TEST_SYNC_POINT_CALLBACK("DeleteScheduler::BackgroundEmptyTrash:Wait",
344 &total_penalty);
345
346 int32_t pending_files_in_bucket = std::numeric_limits<int32_t>::max();
347 if (is_complete) {
348 pending_files_--;
349 if (bucket.has_value()) {
350 auto iter = pending_files_in_buckets_.find(bucket.value());
351 assert(iter != pending_files_in_buckets_.end());
352 if (iter != pending_files_in_buckets_.end()) {
353 pending_files_in_bucket = iter->second--;
354 }
355 }
356 }
357 if (pending_files_ == 0 || pending_files_in_bucket == 0) {
358 // Unblock WaitForEmptyTrash or WaitForEmptyTrashBucket since there are
359 // no more files waiting to be deleted
360 cv_.SignalAll();
361 }
362 }
363 }
364}

366Status DeleteScheduler::DeleteTrashFile(const std::string& path_in_trash,
367 const std::string& dir_to_sync,
368 bool accounted, uint64_t* deleted_bytes,
369 bool* is_complete) {
370 uint64_t file_size;
371 Status s = fs_->GetFileSize(path_in_trash, IOOptions(), &file_size, nullptr);
372 *is_complete = true;
373 TEST_SYNC_POINT("DeleteScheduler::DeleteTrashFile:DeleteFile");
374 TEST_SYNC_POINT_CALLBACK("DeleteScheduler::DeleteTrashFile::cb",
375 const_cast<std::string*>(&path_in_trash));
376 if (s.ok()) {
377 bool need_full_delete = true;
378 if (bytes_max_delete_chunk_ != 0 && file_size > bytes_max_delete_chunk_) {
379 uint64_t num_hard_links = 2;
380 // We don't have to worry aobut data race between linking a new
381 // file after the number of file link check and ftruncte because
382 // the file is now in trash and no hardlink is supposed to create
383 // to trash files by RocksDB.
384 Status my_status = fs_->NumFileLinks(path_in_trash, IOOptions(),
385 &num_hard_links, nullptr);
386 if (my_status.ok()) {
387 if (num_hard_links == 1) {
388 std::unique_ptr<FSWritableFile> wf;
389 my_status = fs_->ReopenWritableFile(path_in_trash, FileOptions(), &wf,
390 nullptr);
391 if (my_status.ok()) {
392 my_status = wf->Truncate(file_size - bytes_max_delete_chunk_,
393 IOOptions(), nullptr);
394 if (my_status.ok()) {
395 TEST_SYNC_POINT("DeleteScheduler::DeleteTrashFile:Fsync");
396 my_status = wf->Fsync(IOOptions(), nullptr);
397 }
398 }
399 if (my_status.ok()) {
400 *deleted_bytes = bytes_max_delete_chunk_;
401 need_full_delete = false;
402 *is_complete = false;
403 } else {
404 ROCKS_LOG_WARN(info_log_,
405 "Failed to partially delete %s from trash -- %s",
406 path_in_trash.c_str(), my_status.ToString().c_str());
407 }
408 } else {
409 ROCKS_LOG_INFO(info_log_,
410 "Cannot delete %s slowly through ftruncate from trash "
411 "as it has other links",
412 path_in_trash.c_str());
413 }
414 } else if (!num_link_error_printed_) {
415 ROCKS_LOG_INFO(
416 info_log_,
417 "Cannot delete files slowly through ftruncate from trash "
418 "as Env::NumFileLinks() returns error: %s",
419 my_status.ToString().c_str());
420 num_link_error_printed_ = true;
421 }
422 }
423
424 if (need_full_delete) {
425 s = fs_->DeleteFile(path_in_trash, IOOptions(), nullptr);
426 if (!dir_to_sync.empty()) {
427 std::unique_ptr<FSDirectory> dir_obj;
428 if (s.ok()) {
429 s = fs_->NewDirectory(dir_to_sync, IOOptions(), &dir_obj, nullptr);
430 }
431 if (s.ok()) {
432 s = dir_obj->FsyncWithDirOptions(
433 IOOptions(), nullptr,
434 DirFsyncOptions(DirFsyncOptions::FsyncReason::kFileDeleted));
435 TEST_SYNC_POINT_CALLBACK(
436 "DeleteScheduler::DeleteTrashFile::AfterSyncDir",
437 static_cast<void*>(const_cast<std::string*>(&dir_to_sync)));
438 }
439 }
440 if (s.ok()) {
441 *deleted_bytes = file_size;
442 s = OnDeleteFile(path_in_trash, accounted);
443 }
444 }
445 }
446 if (!s.ok()) {
447 // Error while getting file size or while deleting
448 ROCKS_LOG_ERROR(info_log_, "Failed to delete %s from trash -- %s",
449 path_in_trash.c_str(), s.ToString().c_str());
450 *deleted_bytes = 0;
451 } else {
452 if (accounted) {
453 total_trash_size_.fetch_sub(*deleted_bytes);
454 }
455 }
456
457 return s;
458}

477std::optional<int32_t> DeleteScheduler::NewTrashBucket() {
478 if (rate_bytes_per_sec_.load() <= 0) {
479 return std::nullopt;
480 }
481 InstrumentedMutexLock l(&mu_);
482 int32_t bucket_number = next_trash_bucket_++;
483 pending_files_in_buckets_.emplace(bucket_number, 0);
484 return bucket_number;
485}
486
487void DeleteScheduler::WaitForEmptyTrashBucket(int32_t bucket) {
488 InstrumentedMutexLock l(&mu_);
489 if (bucket >= next_trash_bucket_) {
490 return;
491 }
492 auto iter = pending_files_in_buckets_.find(bucket);
493 while (iter != pending_files_in_buckets_.end() && iter->second > 0 &&
494 !closing_) {
495 cv_.Wait();
496 iter = pending_files_in_buckets_.find(bucket);
497 }
498 pending_files_in_buckets_.erase(bucket);
499}
