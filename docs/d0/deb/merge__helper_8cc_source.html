60template <typename Visitor>
61Status MergeHelper::TimedFullMergeCommonImpl(
62 const MergeOperator* merge_operator, const Slice& key,
63 MergeOperator::MergeOperationInputV3::ExistingValue&& existing_value,
64 const std::vector<Slice>& operands, Logger* logger, Statistics* statistics,
65 SystemClock* clock, bool update_num_ops_stats,
66 MergeOperator::OpFailureScope* op_failure_scope, Visitor&& visitor) {
67 assert(merge_operator);
68 assert(!operands.empty());
69
70 if (update_num_ops_stats) {
71 RecordInHistogram(statistics, READ_NUM_MERGE_OPERANDS,
72 static_cast<uint64_t>(operands.size()));
73 }
74
75 const MergeOperator::MergeOperationInputV3 merge_in(
76 key, std::move(existing_value), operands, logger);
77 MergeOperator::MergeOperationOutputV3 merge_out;
78
79 bool success = false;
80
81 {
82 StopWatchNano timer(clock, statistics != nullptr);
83 PERF_TIMER_GUARD(merge_operator_time_nanos);
84
85 success = merge_operator->FullMergeV3(merge_in, &merge_out);
86
87 RecordTick(statistics, MERGE_OPERATION_TOTAL_TIME,
88 statistics ? timer.ElapsedNanos() : 0);
89 }
90
91 if (!success) {
92 RecordTick(statistics, NUMBER_MERGE_FAILURES);
93
94 if (op_failure_scope) {
95 *op_failure_scope = merge_out.op_failure_scope;
96 // Apply default per merge_operator.h
97 if (*op_failure_scope == MergeOperator::OpFailureScope::kDefault) {
98 *op_failure_scope = MergeOperator::OpFailureScope::kTryMerge;
99 }
100 }
101
102 return Status::Corruption(Status::SubCode::kMergeOperatorFailed);
103 }
104
105 return std::visit(std::forward<Visitor>(visitor),
106 std::move(merge_out.new_value));
107}

109Status MergeHelper::TimedFullMergeImpl(
110 const MergeOperator* merge_operator, const Slice& key,
111 MergeOperator::MergeOperationInputV3::ExistingValue&& existing_value,
112 const std::vector<Slice>& operands, Logger* logger, Statistics* statistics,
113 SystemClock* clock, bool update_num_ops_stats,
114 MergeOperator::OpFailureScope* op_failure_scope, std::string* result,
115 Slice* result_operand, ValueType* result_type) {
116 assert(result);
117 assert(result_type);
118
119 auto visitor = overload{
120 [&](std::string&& new_value) -> Status {
121 *result_type = kTypeValue;
122
123 if (result_operand) {
124 *result_operand = Slice(nullptr, 0);
125 }
126
127 *result = std::move(new_value);
128
129 return Status::OK();
130 },
131 [&](MergeOperator::MergeOperationOutputV3::NewColumns&& new_columns)
132 -> Status {
133 *result_type = kTypeWideColumnEntity;
134
135 if (result_operand) {
136 *result_operand = Slice(nullptr, 0);
137 }
138
139 result->clear();
140
141 WideColumns sorted_columns;
142 sorted_columns.reserve(new_columns.size());
143
144 for (const auto& column : new_columns) {
145 sorted_columns.emplace_back(column.first, column.second);
146 }
147
148 WideColumnsHelper::SortColumns(sorted_columns);
149
150 return WideColumnSerialization::Serialize(sorted_columns, *result);
151 },
152 [&](Slice&& operand) -> Status {
153 *result_type = kTypeValue;
154
155 if (result_operand) {
156 *result_operand = operand;
157 result->clear();
158 } else {
159 result->assign(operand.data(), operand.size());
160 }
161
162 return Status::OK();
163 }};
164
165 return TimedFullMergeCommonImpl(merge_operator, key,
166 std::move(existing_value), operands, logger,
167 statistics, clock, update_num_ops_stats,
168 op_failure_scope, std::move(visitor));
169}

171Status MergeHelper::TimedFullMergeImpl(
172 const MergeOperator* merge_operator, const Slice& key,
173 MergeOperator::MergeOperationInputV3::ExistingValue&& existing_value,
174 const std::vector<Slice>& operands, Logger* logger, Statistics* statistics,
175 SystemClock* clock, bool update_num_ops_stats,
176 MergeOperator::OpFailureScope* op_failure_scope, std::string* result_value,
177 PinnableWideColumns* result_entity) {
178 assert(result_value || result_entity);
179 assert(!result_value || !result_entity);
180
181 auto visitor = overload{
182 [&](std::string&& new_value) -> Status {
183 if (result_value) {
184 *result_value = std::move(new_value);
185
186 return Status::OK();
187 }
188
189 assert(result_entity);
190 result_entity->SetPlainValue(std::move(new_value));
191
192 return Status::OK();
193 },
194 [&](MergeOperator::MergeOperationOutputV3::NewColumns&& new_columns)
195 -> Status {
196 if (result_value) {
197 if (!new_columns.empty() &&
198 new_columns.front().first == kDefaultWideColumnName) {
199 *result_value = std::move(new_columns.front().second);
200 } else {
201 result_value->clear();
202 }
203
204 return Status::OK();
205 }
206
207 assert(result_entity);
208
209 WideColumns sorted_columns;
210 sorted_columns.reserve(new_columns.size());
211
212 for (const auto& column : new_columns) {
213 sorted_columns.emplace_back(column.first, column.second);
214 }
215
216 WideColumnsHelper::SortColumns(sorted_columns);
217
218 std::string result;
219 const Status s =
220 WideColumnSerialization::Serialize(sorted_columns, result);
221 if (!s.ok()) {
222 result_entity->Reset();
223 return s;
224 }
225
226 return result_entity->SetWideColumnValue(std::move(result));
227 },
228 [&](Slice&& operand) -> Status {
229 if (result_value) {
230 result_value->assign(operand.data(), operand.size());
231
232 return Status::OK();
233 }
234
235 assert(result_entity);
236 result_entity->SetPlainValue(operand);
237
238 return Status::OK();
239 }};
240
241 return TimedFullMergeCommonImpl(merge_operator, key,
242 std::move(existing_value), operands, logger,
243 statistics, clock, update_num_ops_stats,
244 op_failure_scope, std::move(visitor));
245}

247// PRE: iter points to the first merge type entry
248// POST: iter points to the first entry beyond the merge process (or the end)
249// keys_, operands_ are updated to reflect the merge result.
250// keys_ stores the list of keys encountered while merging.
251// operands_ stores the list of merge operands encountered while merging.
252// keys_[i] corresponds to operands_[i] for each i.
253//
254// TODO: Avoid the snapshot stripe map lookup in CompactionRangeDelAggregator
255// and just pass the StripeRep corresponding to the stripe being merged.
256Status MergeHelper::MergeUntil(InternalIterator* iter,
257 CompactionRangeDelAggregator* range_del_agg,
258 const SequenceNumber stop_before,
259 const bool at_bottom,
260 const bool allow_data_in_errors,
261 const BlobFetcher* blob_fetcher,
262 const std::string* const full_history_ts_low,
263 PrefetchBufferCollection* prefetch_buffers,
264 CompactionIterationStats* c_iter_stats) {
265 // Get a copy of the internal key, before it's invalidated by iter->Next()
266 // Also maintain the list of merge operands seen.
267 assert(HasOperator());
268 keys_.clear();
269 merge_context_.Clear();
270 has_compaction_filter_skip_until_ = false;
271 assert(user_merge_operator_);
272 assert(user_comparator_);
273 const size_t ts_sz = user_comparator_->timestamp_size();
274 if (full_history_ts_low) {
275 assert(ts_sz > 0);
276 assert(ts_sz == full_history_ts_low->size());
277 }
278 bool first_key = true;
279
280 // We need to parse the internal key again as the parsed key is
281 // backed by the internal key!
282 // Assume no internal key corruption as it has been successfully parsed
283 // by the caller.
284 // original_key_is_iter variable is just caching the information:
285 // original_key_is_iter == (iter->key().ToString() == original_key)
286 bool original_key_is_iter = true;
287 std::string original_key = iter->key().ToString();
288 // Important:
289 // orig_ikey is backed by original_key if keys_.empty()
290 // orig_ikey is backed by keys_.back() if !keys_.empty()
291 ParsedInternalKey orig_ikey;
292
293 Status s = ParseInternalKey(original_key, &orig_ikey, allow_data_in_errors);
294 assert(s.ok());
295 if (!s.ok()) {
296 return s;
297 }
298
299 assert(kTypeMerge == orig_ikey.type);
300
301 bool hit_the_next_user_key = false;
302 int cmp_with_full_history_ts_low = 0;
303 for (; iter->Valid(); iter->Next(), original_key_is_iter = false) {
304 if (IsShuttingDown()) {
305 s = Status::ShutdownInProgress();
306 return s;
307 }
308 // Skip range tombstones emitted by the compaction iterator.
309 if (iter->IsDeleteRangeSentinelKey()) {
310 continue;
311 }
312
313 ParsedInternalKey ikey;
314 assert(keys_.size() == merge_context_.GetNumOperands());
315
316 Status pik_status =
317 ParseInternalKey(iter->key(), &ikey, allow_data_in_errors);
318 Slice ts;
319 if (pik_status.ok()) {
320 ts = ExtractTimestampFromUserKey(ikey.user_key, ts_sz);
321 if (full_history_ts_low) {
322 cmp_with_full_history_ts_low =
323 user_comparator_->CompareTimestamp(ts, *full_history_ts_low);
324 }
325 }
326 if (!pik_status.ok()) {
327 // stop at corrupted key
328 if (assert_valid_internal_key_) {
329 return pik_status;
330 }
331 break;
332 } else if (first_key) {
333 // If user-defined timestamp is enabled, we expect both user key and
334 // timestamps are equal, as a sanity check.
335 assert(user_comparator_->Equal(ikey.user_key, orig_ikey.user_key));
336 first_key = false;
337 } else if (!user_comparator_->EqualWithoutTimestamp(ikey.user_key,
338 orig_ikey.user_key) ||
339 (ts_sz > 0 &&
340 !user_comparator_->Equal(ikey.user_key, orig_ikey.user_key) &&
341 cmp_with_full_history_ts_low >= 0)) {
342 // 1) hit a different user key, or
343 // 2) user-defined timestamp is enabled, and hit a version of user key NOT
344 // eligible for GC, then stop right here.
345 hit_the_next_user_key = true;
346 break;
347 } else if (stop_before > 0 && ikey.sequence <= stop_before &&
348 LIKELY(snapshot_checker_ == nullptr ||
349 snapshot_checker_->CheckInSnapshot(ikey.sequence,
350 stop_before) !=
351 SnapshotCheckerResult::kNotInSnapshot)) {
352 // hit an entry that's possibly visible by the previous snapshot, can't
353 // touch that
354 break;
355 }
356
357 // At this point we are guaranteed that we need to process this key.
358
359 assert(IsValueType(ikey.type));
360 if (ikey.type != kTypeMerge) {
361 // hit a put/delete/single delete
362 // => merge the put value or a nullptr with operands_
363 // => store result in operands_.back() (and update keys_.back())
364 // => change the entry type for keys_.back()
365 // We are done! Success!
366
367 // If there are no operands, just return the Status::OK(). That will cause
368 // the compaction iterator to write out the key we're currently at, which
369 // is the put/delete we just encountered.
370 if (keys_.empty()) {
371 return s;
372 }
373
374 // TODO: if we're in compaction and it's a put, it would be nice to run
375 // compaction filter on it.
376 std::string merge_result;
377 ValueType merge_result_type;
378 MergeOperator::OpFailureScope op_failure_scope;
379
380 if (range_del_agg &&
381 range_del_agg->ShouldDelete(
382 ikey, RangeDelPositioningMode::kForwardTraversal)) {
383 s = TimedFullMerge(user_merge_operator_, ikey.user_key, kNoBaseValue,
384 merge_context_.GetOperands(), logger_, stats_,
385 clock_, /* update_num_ops_stats */ false,
386 &op_failure_scope, &merge_result,
387 /* result_operand */ nullptr, &merge_result_type);
388 } else if (ikey.type == kTypeValue) {
389 s = TimedFullMerge(user_merge_operator_, ikey.user_key, kPlainBaseValue,
390 iter->value(), merge_context_.GetOperands(), logger_,
391 stats_, clock_, /* update_num_ops_stats */ false,
392 &op_failure_scope, &merge_result,
393 /* result_operand */ nullptr, &merge_result_type);
394 } else if (ikey.type == kTypeValuePreferredSeqno) {
395 // When a TimedPut is merged with some merge operands, its original
396 // write time info is obsolete and removed, and the merge result is a
397 // kTypeValue.
398 Slice unpacked_value = ParsePackedValueForValue(iter->value());
399 s = TimedFullMerge(user_merge_operator_, ikey.user_key, kPlainBaseValue,
400 unpacked_value, merge_context_.GetOperands(),
401 logger_, stats_, clock_,
402 /* update_num_ops_stats */ false, &op_failure_scope,
403 &merge_result,
404 /* result_operand */ nullptr, &merge_result_type);
405
406 } else if (ikey.type == kTypeBlobIndex) {
407 BlobIndex blob_index;
408
409 s = blob_index.DecodeFrom(iter->value());
410 if (!s.ok()) {
411 return s;
412 }
413
414 FilePrefetchBuffer* prefetch_buffer =
415 prefetch_buffers ? prefetch_buffers->GetOrCreatePrefetchBuffer(
416 blob_index.file_number())
417 : nullptr;
418
419 uint64_t bytes_read = 0;
420
421 assert(blob_fetcher);
422
423 PinnableSlice blob_value;
424 s = blob_fetcher->FetchBlob(ikey.user_key, blob_index, prefetch_buffer,
425 &blob_value, &bytes_read);
426 if (!s.ok()) {
427 return s;
428 }
429
430 if (c_iter_stats) {
431 ++c_iter_stats->num_blobs_read;
432 c_iter_stats->total_blob_bytes_read += bytes_read;
433 }
434
435 s = TimedFullMerge(user_merge_operator_, ikey.user_key, kPlainBaseValue,
436 blob_value, merge_context_.GetOperands(), logger_,
437 stats_, clock_, /* update_num_ops_stats */ false,
438 &op_failure_scope, &merge_result,
439 /* result_operand */ nullptr, &merge_result_type);
440 } else if (ikey.type == kTypeWideColumnEntity) {
441 s = TimedFullMerge(user_merge_operator_, ikey.user_key, kWideBaseValue,
442 iter->value(), merge_context_.GetOperands(), logger_,
443 stats_, clock_, /* update_num_ops_stats */ false,
444 &op_failure_scope, &merge_result,
445 /* result_operand */ nullptr, &merge_result_type);
446 } else {
447 s = TimedFullMerge(user_merge_operator_, ikey.user_key, kNoBaseValue,
448 merge_context_.GetOperands(), logger_, stats_,
449 clock_, /* update_num_ops_stats */ false,
450 &op_failure_scope, &merge_result,
451 /* result_operand */ nullptr, &merge_result_type);
452 }
453
454 // We store the result in keys_.back() and operands_.back()
455 // if nothing went wrong (i.e.: no operand corruption on disk)
456 if (s.ok()) {
457 // The original key encountered
458 original_key = std::move(keys_.back());
459
460 assert(merge_result_type == kTypeValue ||
461 merge_result_type == kTypeWideColumnEntity);
462 orig_ikey.type = merge_result_type;
463 UpdateInternalKey(&original_key, orig_ikey.sequence, orig_ikey.type);
464
465 keys_.clear();
466 merge_context_.Clear();
467 keys_.emplace_front(std::move(original_key));
468 merge_context_.PushOperand(merge_result);
469
470 // move iter to the next entry
471 iter->Next();
472 } else if (op_failure_scope ==
473 MergeOperator::OpFailureScope::kMustMerge) {
474 // Change to `Status::MergeInProgress()` to denote output consists of
475 // merge operands only. Leave `iter` at the non-merge entry so it will
476 // be output after.
477 s = Status::MergeInProgress();
478 }
479 return s;
480 } else {
481 // hit a merge
482 // => if there is a compaction filter, apply it.
483 // => check for range tombstones covering the operand
484 // => merge the operand into the front of the operands_ list
485 // if not filtered
486 // => then continue because we haven't yet seen a Put/Delete.
487 //
488 // Keep queuing keys and operands until we either meet a put / delete
489 // request or later did a partial merge.
490
491 Slice value_slice = iter->value();
492 // add an operand to the list if:
493 // 1) it's included in one of the snapshots. in that case we *must* write
494 // it out, no matter what compaction filter says
495 // 2) it's not filtered by a compaction filter
496 CompactionFilter::Decision filter =
497 ikey.sequence <= latest_snapshot_
498 ? CompactionFilter::Decision::kKeep
499 : FilterMerge(orig_ikey.user_key, value_slice);
500 if (filter != CompactionFilter::Decision::kRemoveAndSkipUntil &&
501 range_del_agg != nullptr &&
502 range_del_agg->ShouldDelete(
503 iter->key(), RangeDelPositioningMode::kForwardTraversal)) {
504 filter = CompactionFilter::Decision::kRemove;
505 }
506 if (filter == CompactionFilter::Decision::kKeep ||
507 filter == CompactionFilter::Decision::kChangeValue) {
508 if (original_key_is_iter) {
509 // this is just an optimization that saves us one memcpy
510 keys_.emplace_front(original_key);
511 } else {
512 keys_.emplace_front(iter->key().ToString());
513 }
514 if (keys_.size() == 1) {
515 // we need to re-anchor the orig_ikey because it was anchored by
516 // original_key before
517 pik_status =
518 ParseInternalKey(keys_.back(), &orig_ikey, allow_data_in_errors);
519 pik_status.PermitUncheckedError();
520 assert(pik_status.ok());
521 }
522 if (filter == CompactionFilter::Decision::kKeep) {
523 merge_context_.PushOperand(
524 value_slice, iter->IsValuePinned() /* operand_pinned */);
525 } else {
526 assert(filter == CompactionFilter::Decision::kChangeValue);
527 // Compaction filter asked us to change the operand from value_slice
528 // to compaction_filter_value_.
529 merge_context_.PushOperand(compaction_filter_value_, false);
530 }
531 } else if (filter == CompactionFilter::Decision::kRemoveAndSkipUntil) {
532 // Compaction filter asked us to remove this key altogether
533 // (not just this operand), along with some keys following it.
534 keys_.clear();
535 merge_context_.Clear();
536 has_compaction_filter_skip_until_ = true;
537 return s;
538 }
539 }
540 }
541
542 if (cmp_with_full_history_ts_low >= 0) {
543 size_t num_merge_operands = merge_context_.GetNumOperands();
544 if (ts_sz && num_merge_operands > 1) {
545 // We do not merge merge operands with different timestamps if they are
546 // not eligible for GC.
547 ROCKS_LOG_ERROR(logger_, "ts_sz=%d, %d merge oprands",
548 static_cast<int>(ts_sz),
549 static_cast<int>(num_merge_operands));
550 assert(false);
551 }
552 }
553
554 if (merge_context_.GetNumOperands() == 0) {
555 // we filtered out all the merge operands
556 return s;
557 }
558
559 // We are sure we have seen this key's entire history if:
560 // at_bottom == true (this does not necessarily mean it is the bottommost
561 // layer, but rather that we are confident the key does not appear on any of
562 // the lower layers, at_bottom == false doesn't mean it does appear, just
563 // that we can't be sure, see Compaction::IsBottommostLevel for details)
564 // AND
565 // we have either encountered another key or end of key history on this
566 // layer.
567 // Note that if user-defined timestamp is enabled, we need some extra caution
568 // here: if full_history_ts_low is nullptr, or it's not null but the key's
569 // timestamp is greater than or equal to full_history_ts_low, it means this
570 // key cannot be dropped. We may not have seen the beginning of the key.
571 //
572 // When these conditions are true we are able to merge all the keys
573 // using full merge.
574 //
575 // For these cases we are not sure about, we simply miss the opportunity
576 // to combine the keys. Since VersionSet::SetupOtherInputs() always makes
577 // sure that all merge-operands on the same level get compacted together,
578 // this will simply lead to these merge operands moving to the next level.
579 bool surely_seen_the_beginning =
580 (hit_the_next_user_key || !iter->Valid()) && at_bottom &&
581 (ts_sz == 0 || cmp_with_full_history_ts_low < 0);
582 if (surely_seen_the_beginning) {
583 // do a final merge with nullptr as the existing value and say
584 // bye to the merge type (it's now converted to a Put)
585 assert(kTypeMerge == orig_ikey.type);
586 assert(merge_context_.GetNumOperands() >= 1);
587 assert(merge_context_.GetNumOperands() == keys_.size());
588 std::string merge_result;
589 ValueType merge_result_type;
590 MergeOperator::OpFailureScope op_failure_scope;
591 s = TimedFullMerge(user_merge_operator_, orig_ikey.user_key, kNoBaseValue,
592 merge_context_.GetOperands(), logger_, stats_, clock_,
593 /* update_num_ops_stats */ false, &op_failure_scope,
594 &merge_result,
595 /* result_operand */ nullptr, &merge_result_type);
596 if (s.ok()) {
597 // The original key encountered
598 // We are certain that keys_ is not empty here (see assertions couple of
599 // lines before).
600 original_key = std::move(keys_.back());
601
602 assert(merge_result_type == kTypeValue ||
603 merge_result_type == kTypeWideColumnEntity);
604 orig_ikey.type = merge_result_type;
605 UpdateInternalKey(&original_key, orig_ikey.sequence, orig_ikey.type);
606
607 keys_.clear();
608 merge_context_.Clear();
609 keys_.emplace_front(std::move(original_key));
610 merge_context_.PushOperand(merge_result);
611 } else if (op_failure_scope == MergeOperator::OpFailureScope::kMustMerge) {
612 // Change to `Status::MergeInProgress()` to denote output consists of
613 // merge operands only.
614 s = Status::MergeInProgress();
615 }
616 } else {
617 // We haven't seen the beginning of the key nor a Put/Delete.
618 // Attempt to use the user's associative merge function to
619 // merge the stacked merge operands into a single operand.
620 s = Status::MergeInProgress();
621 if (merge_context_.GetNumOperands() >= 2 ||
622 (allow_single_operand_ && merge_context_.GetNumOperands() == 1)) {
623 bool merge_success = false;
624 std::string merge_result;
625 {
626 StopWatchNano timer(clock_, stats_ != nullptr);
627 PERF_TIMER_GUARD(merge_operator_time_nanos);
628 merge_success = user_merge_operator_->PartialMergeMulti(
629 orig_ikey.user_key,
630 std::deque<Slice>(merge_context_.GetOperands().begin(),
631 merge_context_.GetOperands().end()),
632 &merge_result, logger_);
633 RecordTick(stats_, MERGE_OPERATION_TOTAL_TIME,
634 stats_ ? timer.ElapsedNanosSafe() : 0);
635 }
636 if (merge_success) {
637 // Merging of operands (associative merge) was successful.
638 // Replace operands with the merge result
639 merge_context_.Clear();
640 merge_context_.PushOperand(merge_result);
641 keys_.erase(keys_.begin(), keys_.end() - 1);
642 }
643 }
644 }
645
646 return s;
647}

649MergeOutputIterator::MergeOutputIterator(const MergeHelper* merge_helper)

651 it_keys_ = merge_helper_->keys().rend();

668CompactionFilter::Decision MergeHelper::FilterMerge(const Slice& user_key,
669 const Slice& value_slice) {
670 if (compaction_filter_ == nullptr) {
671 return CompactionFilter::Decision::kKeep;
672 }
673 if (stats_ != nullptr && ShouldReportDetailedTime(env_, stats_)) {
674 filter_timer_.Start();
675 }
676 compaction_filter_value_.clear();
677 compaction_filter_skip_until_.Clear();
678 auto ret = compaction_filter_->FilterV3(
679 level_, user_key, CompactionFilter::ValueType::kMergeOperand,
680 &value_slice, /* existing_columns */ nullptr, &compaction_filter_value_,
681 /* new_columns */ nullptr, compaction_filter_skip_until_.rep());
682 if (ret == CompactionFilter::Decision::kRemoveAndSkipUntil) {
683 if (user_comparator_->Compare(*compaction_filter_skip_until_.rep(),
684 user_key) <= 0) {
685 // Invalid skip_until returned from compaction filter.
686 // Keep the key as per FilterV2/FilterV3 documentation.
687 ret = CompactionFilter::Decision::kKeep;
688 } else {
689 compaction_filter_skip_until_.ConvertFromUserKey(kMaxSequenceNumber,
690 kValueTypeForSeek);
691 }
692 }
693 if (stats_ != nullptr && ShouldReportDetailedTime(env_, stats_)) {
694 total_filter_time_ += filter_timer_.ElapsedNanosSafe();
695 }
696 return ret;
697}
