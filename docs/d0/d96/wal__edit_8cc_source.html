14void WalAddition::EncodeTo(std::string* dst) const {
15 PutVarint64(dst, number_);
16
17 if (metadata_.HasSyncedSize()) {
18 PutVarint32(dst, static_cast<uint32_t>(WalAdditionTag::kSyncedSize));
19 PutVarint64(dst, metadata_.GetSyncedSizeInBytes());
20 }
21
22 PutVarint32(dst, static_cast<uint32_t>(WalAdditionTag::kTerminate));
23}

25Status WalAddition::DecodeFrom(Slice* src) {
26 constexpr char class_name[] = "WalAddition";
27
28 if (!GetVarint64(src, &number_)) {
29 return Status::Corruption(class_name, "Error decoding WAL log number");
30 }
31
32 while (true) {
33 uint32_t tag_value = 0;
34 if (!GetVarint32(src, &tag_value)) {
35 return Status::Corruption(class_name, "Error decoding tag");
36 }
37 WalAdditionTag tag = static_cast<WalAdditionTag>(tag_value);
38 switch (tag) {
39 case WalAdditionTag::kSyncedSize: {
40 uint64_t size = 0;
41 if (!GetVarint64(src, &size)) {
42 return Status::Corruption(class_name, "Error decoding WAL file size");
43 }
44 metadata_.SetSyncedSizeInBytes(size);
45 break;
46 }
47 // TODO: process future tags such as checksum.
48 case WalAdditionTag::kTerminate:
49 return Status::OK();
50 default: {
51 std::stringstream ss;
52 ss << "Unknown tag " << tag_value;
53 return Status::Corruption(class_name, ss.str());
54 }
55 }
56 }
57}

107Status WalSet::AddWal(const WalAddition& wal) {
108 if (wal.GetLogNumber() < min_wal_number_to_keep_) {
109 // The WAL has been obsolete, ignore it.
110 return Status::OK();
111 }
112
113 auto it = wals_.lower_bound(wal.GetLogNumber());
114 bool existing = it != wals_.end() && it->first == wal.GetLogNumber();
115
116 if (!existing) {
117 wals_.insert(it, {wal.GetLogNumber(), wal.GetMetadata()});
118 return Status::OK();
119 }
120
121 assert(existing);
122 if (!wal.GetMetadata().HasSyncedSize()) {
123 std::stringstream ss;
124 ss << "WAL " << wal.GetLogNumber() << " is created more than once";
125 return Status::Corruption("WalSet::AddWal", ss.str());
126 }
127
128 assert(wal.GetMetadata().HasSyncedSize());
129 if (it->second.HasSyncedSize() && wal.GetMetadata().GetSyncedSizeInBytes() <=
130 it->second.GetSyncedSizeInBytes()) {
131 // This is possible because version edits with different synced WAL sizes
132 // for the same WAL can be committed out-of-order. For example, thread
133 // 1 synces the first 10 bytes of 1.log, while thread 2 synces the first 20
134 // bytes of 1.log. It's possible that thread 1 calls LogAndApply() after
135 // thread 2.
136 // In this case, just return ok.
137 return Status::OK();
138 }
139
140 // Update synced size for the given WAL.
141 it->second.SetSyncedSizeInBytes(wal.GetMetadata().GetSyncedSizeInBytes());
142 return Status::OK();
143}

156Status WalSet::DeleteWalsBefore(WalNumber wal) {
157 if (wal > min_wal_number_to_keep_) {
158 min_wal_number_to_keep_ = wal;
159 wals_.erase(wals_.begin(), wals_.lower_bound(wal));
160 }
161 return Status::OK();

169Status WalSet::CheckWals(
170 Env* env,
171 const std::unordered_map<WalNumber, std::string>& logs_on_disk) const {
172 assert(env != nullptr);
173
174 Status s;
175 for (const auto& wal : wals_) {
176 const uint64_t log_number = wal.first;
177 const WalMetadata& wal_meta = wal.second;
178
179 if (!wal_meta.HasSyncedSize()) {
180 // The WAL and WAL directory is not even synced,
181 // so the WAL's inode may not be persisted,
182 // then the WAL might not show up when listing WAL directory.
183 continue;
184 }
185
186 if (logs_on_disk.find(log_number) == logs_on_disk.end()) {
187 std::stringstream ss;
188 ss << "Missing WAL with log number: " << log_number << ".";
189 s = Status::Corruption(ss.str());
190 break;
191 }
192
193 uint64_t log_file_size = 0;
194 s = env->GetFileSize(logs_on_disk.at(log_number), &log_file_size);
195 if (!s.ok()) {
196 break;
197 }
198 if (log_file_size < wal_meta.GetSyncedSizeInBytes()) {
199 std::stringstream ss;
200 ss << "Size mismatch: WAL (log number: " << log_number
201 << ") in MANIFEST is " << wal_meta.GetSyncedSizeInBytes()
202 << " bytes , but actually is " << log_file_size << " bytes on disk.";
203 s = Status::Corruption(ss.str());
204 break;
205 }
206 }
207
208 return s;
209}
