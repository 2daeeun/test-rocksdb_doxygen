127// Generic cache interface that shards cache by hash of keys. 2^num_shard_bits
128// shards will be created, with capacity split evenly to each of the shards.
129// Keys are typically sharded by the lowest num_shard_bits bits of hash value
130// so that the upper bits of the hash value can keep a stable ordering of
131// table entries even as the table grows (using more upper hash bits).
132// See CacheShardBase above for what is expected of the CacheShard parameter.
133template <class CacheShard>
134class ShardedCache : public ShardedCacheBase {
135 public:
136 using HashVal = typename CacheShard::HashVal;
137 using HashCref = typename CacheShard::HashCref;
138 using HandleImpl = typename CacheShard::HandleImpl;
139
140 explicit ShardedCache(const ShardedCacheOptions& opts)
141 : ShardedCacheBase(opts),
142 shards_(static_cast<CacheShard*>(port::cacheline_aligned_alloc(
143 sizeof(CacheShard) * GetNumShards()))),
144 destroy_shards_in_dtor_(false) {}
145
146 virtual ~ShardedCache() {
147 if (destroy_shards_in_dtor_) {
148 ForEachShard([](CacheShard* cs) { cs->~CacheShard(); });
149 }
150 port::cacheline_aligned_free(shards_);
151 }
152
153 CacheShard& GetShard(HashCref hash) {
154 return shards_[CacheShard::HashPieceForSharding(hash) & shard_mask_];
155 }
156
157 const CacheShard& GetShard(HashCref hash) const {
158 return shards_[CacheShard::HashPieceForSharding(hash) & shard_mask_];
159 }
160
161 void SetCapacity(size_t capacity) override {
162 MutexLock l(&config_mutex_);
163 capacity_ = capacity;
164 auto per_shard = ComputePerShardCapacity(capacity);
165 ForEachShard([=](CacheShard* cs) { cs->SetCapacity(per_shard); });
166 }
167
168 void SetStrictCapacityLimit(bool s_c_l) override {
169 MutexLock l(&config_mutex_);
170 strict_capacity_limit_ = s_c_l;
171 ForEachShard(
172 [s_c_l](CacheShard* cs) { cs->SetStrictCapacityLimit(s_c_l); });
173 }
174
175 Status Insert(
176 const Slice& key, ObjectPtr obj, const CacheItemHelper* helper,
177 size_t charge, Handle** handle = nullptr,
178 Priority priority = Priority::LOW,
179 const Slice& /*compressed_value*/ = Slice(),
180 CompressionType /*type*/ = CompressionType::kNoCompression) override {
181 assert(helper);
182 HashVal hash = CacheShard::ComputeHash(key, hash_seed_);
183 auto h_out = reinterpret_cast<HandleImpl**>(handle);
184 return GetShard(hash).Insert(key, hash, obj, helper, charge, h_out,
185 priority);
186 }
187
188 Handle* CreateStandalone(const Slice& key, ObjectPtr obj,
189 const CacheItemHelper* helper, size_t charge,
190 bool allow_uncharged) override {
191 assert(helper);
192 HashVal hash = CacheShard::ComputeHash(key, hash_seed_);
193 HandleImpl* result = GetShard(hash).CreateStandalone(
194 key, hash, obj, helper, charge, allow_uncharged);
195 return static_cast<Handle*>(result);
196 }
197
198 Handle* Lookup(const Slice& key, const CacheItemHelper* helper = nullptr,
199 CreateContext* create_context = nullptr,
200 Priority priority = Priority::LOW,
201 Statistics* stats = nullptr) override {
202 HashVal hash = CacheShard::ComputeHash(key, hash_seed_);
203 HandleImpl* result = GetShard(hash).Lookup(key, hash, helper,
204 create_context, priority, stats);
205 return static_cast<Handle*>(result);
206 }
207
208 void Erase(const Slice& key) override {
209 HashVal hash = CacheShard::ComputeHash(key, hash_seed_);
210 GetShard(hash).Erase(key, hash);
211 }
212
213 bool Release(Handle* handle, bool useful,
214 bool erase_if_last_ref = false) override {
215 auto h = static_cast<HandleImpl*>(handle);
216 return GetShard(h->GetHash()).Release(h, useful, erase_if_last_ref);
217 }
218 bool Ref(Handle* handle) override {
219 auto h = static_cast<HandleImpl*>(handle);
220 return GetShard(h->GetHash()).Ref(h);
221 }
222 bool Release(Handle* handle, bool erase_if_last_ref = false) override {
223 return Release(handle, true /*useful*/, erase_if_last_ref);
224 }
225 using ShardedCacheBase::GetUsage;
226 size_t GetUsage() const override {
227 return SumOverShards2(&CacheShard::GetUsage);
228 }
229 size_t GetPinnedUsage() const override {
230 return SumOverShards2(&CacheShard::GetPinnedUsage);
231 }
232 size_t GetOccupancyCount() const override {
233 return SumOverShards2(&CacheShard::GetOccupancyCount);
234 }
235 size_t GetTableAddressCount() const override {
236 return SumOverShards2(&CacheShard::GetTableAddressCount);
237 }
238 void ApplyToAllEntries(
239 const std::function<void(const Slice& key, ObjectPtr value, size_t charge,
240 const CacheItemHelper* helper)>& callback,
241 const ApplyToAllEntriesOptions& opts) override {
242 uint32_t num_shards = GetNumShards();
243 // Iterate over part of each shard, rotating between shards, to
244 // minimize impact on latency of concurrent operations.
245 std::unique_ptr<size_t[]> states(new size_t[num_shards]{});
246
247 size_t aepl = opts.average_entries_per_lock;
248 aepl = std::min(aepl, size_t{1});
249
250 bool remaining_work;
251 do {
252 remaining_work = false;
253 for (uint32_t i = 0; i < num_shards; i++) {
254 if (states[i] != SIZE_MAX) {
255 shards_[i].ApplyToSomeEntries(callback, aepl, &states[i]);
256 remaining_work |= states[i] != SIZE_MAX;
257 }
258 }
259 } while (remaining_work);
260 }
261
262 void EraseUnRefEntries() override {
263 ForEachShard([](CacheShard* cs) { cs->EraseUnRefEntries(); });
264 }
265
266 void DisownData() override {
267 // Leak data only if that won't generate an ASAN/valgrind warning.
268 if (!kMustFreeHeapAllocations) {
269 destroy_shards_in_dtor_ = false;
270 }
271 }
272
273 protected:
274 inline void ForEachShard(const std::function<void(CacheShard*)>& fn) {
275 uint32_t num_shards = GetNumShards();
276 for (uint32_t i = 0; i < num_shards; i++) {
277 fn(shards_ + i);
278 }
279 }
280
281 inline void ForEachShard(
282 const std::function<void(const CacheShard*)>& fn) const {
283 uint32_t num_shards = GetNumShards();
284 for (uint32_t i = 0; i < num_shards; i++) {
285 fn(shards_ + i);
286 }
287 }
288
289 inline size_t SumOverShards(
290 const std::function<size_t(CacheShard&)>& fn) const {
291 uint32_t num_shards = GetNumShards();
292 size_t result = 0;
293 for (uint32_t i = 0; i < num_shards; i++) {
294 result += fn(shards_[i]);
295 }
296 return result;
297 }
298
299 inline size_t SumOverShards2(size_t (CacheShard::*fn)() const) const {
300 return SumOverShards([fn](CacheShard& cs) { return (cs.*fn)(); });
301 }
302
303 // Must be called exactly once by derived class constructor
304 void InitShards(const std::function<void(CacheShard*)>& placement_new) {
305 ForEachShard(placement_new);
306 destroy_shards_in_dtor_ = true;
307 }
308
309 void AppendPrintableOptions(std::string& str) const override {
310 shards_[0].AppendPrintableOptions(str);
311 }
312
313 private:
314 CacheShard* const shards_;
315 bool destroy_shards_in_dtor_;
316};
