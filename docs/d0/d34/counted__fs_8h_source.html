18struct OpCounter {
19 std::atomic<int> ops;
20 std::atomic<uint64_t> bytes;
21
22 OpCounter() : ops(0), bytes(0) {}
23
24 void Reset() {
25 ops = 0;
26 bytes = 0;
27 }
28 void RecordOp(const IOStatus& io_s, size_t added_bytes) {
29 if (!io_s.IsNotSupported()) {
30 ops.fetch_add(1, std::memory_order_relaxed);
31 }
32 if (io_s.ok()) {
33 bytes.fetch_add(added_bytes, std::memory_order_relaxed);
34 }
35 }
36};
