34class CuckooBuilderTest : public testing::Test {
35 public:
36 CuckooBuilderTest() {
37 env_ = Env::Default();
38 Options options;
39 options.allow_mmap_reads = true;
40 file_options_ = FileOptions(options);
41 }
42
43 void CheckFileContents(const std::vector<std::string>& keys,
44 const std::vector<std::string>& values,
45 const std::vector<uint64_t>& expected_locations,
46 std::string expected_unused_bucket,
47 uint64_t expected_table_size,
48 uint32_t expected_num_hash_func,
49 bool expected_is_last_level,
50 uint32_t expected_cuckoo_block_size = 1) {
51 uint64_t num_deletions = 0;
52 for (const auto& key : keys) {
53 ParsedInternalKey parsed;
54 Status pik_status =
55 ParseInternalKey(key, &parsed, true /* log_err_key */);
56 if (pik_status.ok() && parsed.type == kTypeDeletion) {
57 num_deletions++;
58 }
59 }
60 // Read file
61 uint64_t read_file_size;
62 ASSERT_OK(env_->GetFileSize(fname, &read_file_size));
63 std::unique_ptr<RandomAccessFileReader> file_reader;
64 ASSERT_OK(RandomAccessFileReader::Create(
65 env_->GetFileSystem(), fname, file_options_, &file_reader, nullptr));
66
67 Options options;
68 options.allow_mmap_reads = true;
69 ImmutableOptions ioptions(options);
70
71 // Assert Table Properties.
72 std::unique_ptr<TableProperties> props;
73 const ReadOptions read_options;
74 ASSERT_OK(ReadTableProperties(file_reader.get(), read_file_size,
75 kCuckooTableMagicNumber, ioptions,
76 read_options, &props));
77 // Check unused bucket.
78 std::string unused_key =
79 props->user_collected_properties[CuckooTablePropertyNames::kEmptyKey];
80 ASSERT_EQ(expected_unused_bucket.substr(0, props->fixed_key_len),
81 unused_key);
82
83 uint64_t value_len_found = *reinterpret_cast<const uint64_t*>(
84 props->user_collected_properties[CuckooTablePropertyNames::kValueLength]
85 .data());
86 ASSERT_EQ(values.empty() ? 0 : values[0].size(), value_len_found);
87 ASSERT_EQ(props->raw_value_size, values.size() * value_len_found);
88 const uint64_t table_size = *reinterpret_cast<const uint64_t*>(
89 props
90 ->user_collected_properties
91 [CuckooTablePropertyNames::kHashTableSize]
92 .data());
93 ASSERT_EQ(expected_table_size, table_size);
94 const uint32_t num_hash_func_found = *reinterpret_cast<const uint32_t*>(
95 props->user_collected_properties[CuckooTablePropertyNames::kNumHashFunc]
96 .data());
97 ASSERT_EQ(expected_num_hash_func, num_hash_func_found);
98 const uint32_t cuckoo_block_size = *reinterpret_cast<const uint32_t*>(
99 props
100 ->user_collected_properties
101 [CuckooTablePropertyNames::kCuckooBlockSize]
102 .data());
103 ASSERT_EQ(expected_cuckoo_block_size, cuckoo_block_size);
104 const bool is_last_level_found = *reinterpret_cast<const bool*>(
105 props->user_collected_properties[CuckooTablePropertyNames::kIsLastLevel]
106 .data());
107 ASSERT_EQ(expected_is_last_level, is_last_level_found);
108
109 ASSERT_EQ(props->num_entries, keys.size());
110 ASSERT_EQ(props->num_deletions, num_deletions);
111 ASSERT_EQ(props->fixed_key_len, keys.empty() ? 0 : keys[0].size());
112 ASSERT_EQ(props->data_size,
113 expected_unused_bucket.size() *
114 (expected_table_size + expected_cuckoo_block_size - 1));
115 ASSERT_EQ(props->raw_key_size, keys.size() * props->fixed_key_len);
116 ASSERT_EQ(props->column_family_id, 0);
117 ASSERT_EQ(props->column_family_name, kDefaultColumnFamilyName);
118
119 // Check contents of the bucket.
120 std::vector<bool> keys_found(keys.size(), false);
121 size_t bucket_size = expected_unused_bucket.size();
122 for (uint32_t i = 0; i + 1 < table_size + cuckoo_block_size; ++i) {
123 Slice read_slice;
124 ASSERT_OK(file_reader->Read(IOOptions(), i * bucket_size, bucket_size,
125 &read_slice, nullptr, nullptr));
126 size_t key_idx =
127 std::find(expected_locations.begin(), expected_locations.end(), i) -
128 expected_locations.begin();
129 if (key_idx == keys.size()) {
130 // i is not one of the expected locations. Empty bucket.
131 if (read_slice.data() == nullptr) {
132 ASSERT_EQ(0, expected_unused_bucket.size());
133 } else {
134 ASSERT_EQ(read_slice.compare(expected_unused_bucket), 0);
135 }
136 } else {
137 keys_found[key_idx] = true;
138 ASSERT_EQ(read_slice.compare(keys[key_idx] + values[key_idx]), 0);
139 }
140 }
141 for (auto key_found : keys_found) {
142 // Check that all keys wereReader found.
143 ASSERT_TRUE(key_found);
144 }
145 }
146
147 std::string GetInternalKey(Slice user_key, bool zero_seqno,
148 ValueType type = kTypeValue) {
149 IterKey ikey;
150 ikey.SetInternalKey(user_key, zero_seqno ? 0 : 1000, type);
151 return ikey.GetInternalKey().ToString();
152 }
153
154 uint64_t NextPowOf2(uint64_t num) {
155 uint64_t n = 2;
156 while (n <= num) {
157 n *= 2;
158 }
159 return n;
160 }
161
162 uint64_t GetExpectedTableSize(uint64_t num) {
163 return NextPowOf2(static_cast<uint64_t>(num / kHashTableRatio));
164 }
165
166 Env* env_;
167 FileOptions file_options_;
168 std::string fname;
169 const double kHashTableRatio = 0.9;
170};

189TEST_F(CuckooBuilderTest, WriteSuccessNoCollisionFullKey) {
190 for (auto type : {kTypeValue, kTypeDeletion}) {
191 uint32_t num_hash_fun = 4;
192 std::vector<std::string> user_keys = {"key01", "key02", "key03", "key04"};
193 std::vector<std::string> values;
194 if (type == kTypeValue) {
195 values = {"v01", "v02", "v03", "v04"};
196 } else {
197 values = {"", "", "", ""};
198 }
199 // Need to have a temporary variable here as VS compiler does not currently
200 // support operator= with initializer_list as a parameter
201 std::unordered_map<std::string, std::vector<uint64_t>> hm = {
202 {user_keys[0], {0, 1, 2, 3}},
203 {user_keys[1], {1, 2, 3, 4}},
204 {user_keys[2], {2, 3, 4, 5}},
205 {user_keys[3], {3, 4, 5, 6}}};
206 hash_map = std::move(hm);
207
208 std::vector<uint64_t> expected_locations = {0, 1, 2, 3};
209 std::vector<std::string> keys;
210 for (auto& user_key : user_keys) {
211 keys.push_back(GetInternalKey(user_key, false, type));
212 }
213 uint64_t expected_table_size = GetExpectedTableSize(keys.size());
214
215 fname = test::PerThreadDBPath("NoCollisionFullKey");
216 std::unique_ptr<WritableFileWriter> file_writer;
217 ASSERT_OK(WritableFileWriter::Create(env_->GetFileSystem(), fname,
218 file_options_, &file_writer, nullptr));
219 CuckooTableBuilder builder(file_writer.get(), kHashTableRatio, num_hash_fun,
220 100, BytewiseComparator(), 1, false, false,
221 GetSliceHash, 0 /* column_family_id */,
222 kDefaultColumnFamilyName);
223 ASSERT_OK(builder.status());
224 for (uint32_t i = 0; i < user_keys.size(); i++) {
225 builder.Add(Slice(keys[i]), Slice(values[i]));
226 ASSERT_EQ(builder.NumEntries(), i + 1);
227 ASSERT_OK(builder.status());
228 }
229 size_t bucket_size = keys[0].size() + values[0].size();
230 ASSERT_EQ(expected_table_size * bucket_size - 1, builder.FileSize());
231 ASSERT_OK(builder.Finish());
232 ASSERT_OK(file_writer->Close(IOOptions()));
233 ASSERT_LE(expected_table_size * bucket_size, builder.FileSize());
234
235 std::string expected_unused_bucket = GetInternalKey("key00", true);
236 expected_unused_bucket += std::string(values[0].size(), 'a');
237 CheckFileContents(keys, values, expected_locations, expected_unused_bucket,
238 expected_table_size, 2, false);
239 }
240}
