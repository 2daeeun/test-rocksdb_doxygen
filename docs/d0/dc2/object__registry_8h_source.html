54 // Class for matching target strings to a pattern.
55 // Entries consist of a name that starts the pattern and attributes
56 // The following attributes can be added to the entry:
57 // -Suffix: Comparable to name(suffix)
58 // -Separator: Comparable to name(separator).+ or name(separator).*
59 // -Number: Comparable to name(separator).[0-9]+
60 // -AltName: Comparable to (name|alt)
61 // -Optional: Comparable to name(separator)?
62 // Multiple separators can be combined and cause multiple matches.
63 // For example, Pattern("A").AnotherName("B").AddSeparator("@").AddNumber("#")
64 // is roughly equivalent to "(A|B)@.+#.+"
65 //
66 // Note that though this class does provide some regex-style matching,
67 // it is not a full regex parser and has some key differences:
68 // - Separators are matched left-most. For example, an entry
69 // Name("Hello").AddSeparator(" ").AddSuffix("!") would match
70 // "Hello world!", but not "Hello world!!"
71 // - No backtracking is necessary, enabling reliably efficient matching
72 class PatternEntry : public Entry {
73 private:
74 enum Quantifier {
75 kMatchZeroOrMore, // [suffix].*
76 kMatchAtLeastOne, // [suffix].+
77 kMatchExact, // [suffix]
78 kMatchInteger, // [suffix][0-9]+
79 kMatchDecimal, // [suffix][0-9]+[.][0-9]+
80 };
81
82 public:
83 // Short-cut for creating an entry that matches to a
84 // Customizable::IndividualId
85 static PatternEntry AsIndividualId(const std::string& name) {
86 PatternEntry entry(name, true);
87 entry.AddSeparator("@");
88 entry.AddSeparator("#");
89 return entry;
90 }
91
92 // Creates a new PatternEntry for "name". If optional is true,
93 // Matches will also return true if name==target
94 explicit PatternEntry(const std::string& name, bool optional = true)
95 : name_(name), optional_(optional), slength_(0) {
96 nlength_ = name_.size();
97 }
98
99 // Adds a suffix (exact match of separator with no trailing characters) to
100 // the separator
101 PatternEntry& AddSuffix(const std::string& suffix) {
102 separators_.emplace_back(suffix, kMatchExact);
103 slength_ += suffix.size();
104 return *this;
105 }
106
107 // Adds a separator (exact match of separator with trailing characters) to
108 // the entry
109 // If at_least_one is true, the separator must be followed by at least
110 // one character (e.g. separator.+).
111 // If at_least_one is false, the separator may be followed by zero or
112 // more characters (e.g. separator.*).
113 PatternEntry& AddSeparator(const std::string& separator,
114 bool at_least_one = true) {
115 slength_ += separator.size();
116 if (at_least_one) {
117 separators_.emplace_back(separator, kMatchAtLeastOne);
118 ++slength_;
119 } else {
120 separators_.emplace_back(separator, kMatchZeroOrMore);
121 }
122 return *this;
123 }
124
125 // Adds a separator (exact match of separator with trailing numbers) to the
126 // entry
127 PatternEntry& AddNumber(const std::string& separator, bool is_int = true) {
128 separators_.emplace_back(separator,
129 (is_int) ? kMatchInteger : kMatchDecimal);
130 slength_ += separator.size() + 1;
131 return *this;
132 }
133
134 // Sets another name that this entry will match, similar to (name|alt)
135 PatternEntry& AnotherName(const std::string& alt) {
136 names_.emplace_back(alt);
137 return *this;
138 }
139
140 // Sets whether the separators are required -- similar to name(separator)?
141 // If optional is true, then name(separator)? would match
142 // If optional is false, then the separators must also match
143 PatternEntry& SetOptional(bool optional) {
144 optional_ = optional;
145 return *this;
146 }
147
148 // Checks to see if the target matches this entry
149 bool Matches(const std::string& target) const override;
150 const char* Name() const override { return name_.c_str(); }
151
152 private:
153 size_t MatchSeparatorAt(size_t start, Quantifier mode,
154 const std::string& target, size_t tlen,
155 const std::string& pattern) const;
156
157 bool MatchesTarget(const std::string& name, size_t nlen,
158 const std::string& target, size_t ylen) const;
159 std::string name_; // The base name for this entry
160 size_t nlength_; // The length of name_
161 std::vector<std::string> names_; // Alternative names for this entry
162 bool optional_; // Whether matching of separators is required
163 size_t slength_; // The minimum required length to match the separators
164 std::vector<std::pair<std::string, Quantifier>>
165 separators_; // What to match
166 }; // End class Entry

196 // Finds the factory function for the input target.
197 // @see PatternEntry for the matching rules to target
198 // @return If matched, the FactoryFunc for this target, else nullptr
199 template <typename T>
200 FactoryFunc<T> FindFactory(const std::string& target) const {
201 std::unique_lock<std::mutex> lock(mu_);
202 auto factories = factories_.find(T::Type());
203 if (factories != factories_.end()) {
204 for (const auto& e : factories->second) {
205 if (e->Matches(target)) {
206 const auto* fe =
207 static_cast<const ObjectLibrary::FactoryEntry<T>*>(e.get());
208 return fe->GetFactory();
209 }
210 }
211 }
212 return nullptr;
213 }

314 // Finds the factory for target and instantiates a new T.
315 // Returns NotSupported if no factory is found
316 // Returns InvalidArgument if a factory is found but the factory failed.
317 template <typename T>
318 Status NewObject(const std::string& target, T** object,
319 std::unique_ptr<T>* guard) {
320 assert(guard != nullptr);
321 guard->reset();
322 auto factory = FindFactory<T>(target);
323 if (factory != nullptr) {
324 std::string errmsg;
325 *object = factory(target, guard, &errmsg);
326 if (*object != nullptr) {
327 return Status::OK();
328 } else if (errmsg.empty()) {
329 return Status::InvalidArgument(
330 std::string("Could not load ") + T::Type(), target);
331 } else {
332 return Status::InvalidArgument(errmsg, target);
333 }
334 } else {
335 return Status::NotSupported(std::string("Could not load ") + T::Type(),
336 target);
337 }
338 }
339 // Creates a new unique T using the input factory functions.
340 // Returns OK if a new unique T was successfully created
341 // Returns NotSupported if the type/target could not be created
342 // Returns InvalidArgument if the factory return an unguarded object
343 // (meaning it cannot be managed by a unique ptr)
344 template <typename T>
345 Status NewUniqueObject(const std::string& target,
346 std::unique_ptr<T>* result) {
347 T* ptr = nullptr;
348 std::unique_ptr<T> guard;
349 Status s = NewObject(target, &ptr, &guard);
350 if (!s.ok()) {
351 return s;
352 } else if (guard) {
353 result->reset(guard.release());
354 return Status::OK();
355 } else {
356 return Status::InvalidArgument(std::string("Cannot make a unique ") +
357 T::Type() + " from unguarded one ",
358 target);
359 }
360 }

457 // Creates a new ManagedObject in the registry for the id if one does not
458 // currently exist. If an object with that ID already exists, the current
459 // object is returned.
460 //
461 // The ID is the identifier of the object to be returned/created and returned
462 // in result
463 // If a new object is created (using the object factories), the cfunc
464 // parameter will be invoked to configure the new object.
465 template <typename T>
466 Status GetOrCreateManagedObject(const std::string& id,
467 std::shared_ptr<T>* result,
468 const ConfigureFunc<T>& cfunc = nullptr) {
469 if (parent_ != nullptr) {
470 auto object = parent_->GetManagedObject(T::Type(), id);
471 if (object != nullptr) {
472 *result = std::static_pointer_cast<T>(object);
473 return Status::OK();
474 }
475 }
476 {
477 std::unique_lock<std::mutex> lock(objects_mutex_);
478 auto key = ToManagedObjectKey(T::Type(), id);
479 auto iter = managed_objects_.find(key);
480 if (iter != managed_objects_.end()) {
481 auto object = iter->second.lock();
482 if (object != nullptr) {
483 *result = std::static_pointer_cast<T>(object);
484 return Status::OK();
485 }
486 }
487 std::shared_ptr<T> object;
488 Status s = NewSharedObject(id, &object);
489 if (s.ok() && cfunc != nullptr) {
490 s = cfunc(object.get());
491 }
492 if (s.ok()) {
493 auto c = std::static_pointer_cast<Customizable>(object);
494 if (id != c->Name()) {
495 // If the ID is not the base name of the class, add the new
496 // object under the input ID
497 managed_objects_[key] = c;
498 }
499 if (id != c->GetId() && c->GetId() != c->Name()) {
500 // If the input and current ID do not match, and the
501 // current ID is not the base bame, add the new object under
502 // its new ID
503 key = ToManagedObjectKey(T::Type(), c->GetId());
504 managed_objects_[key] = c;
505 }
506 *result = object;
507 }
508 return s;
509 }
510 }

550 // Searches (from back to front) the libraries looking for the
551 // factory that matches this name.
552 // Returns the factory if it is found, and nullptr otherwise
553 template <typename T>
554 const FactoryFunc<T> FindFactory(const std::string& name) const {
555 {
556 std::unique_lock<std::mutex> lock(library_mutex_);
557 for (auto iter = libraries_.crbegin(); iter != libraries_.crend();
558 ++iter) {
559 const auto factory = iter->get()->FindFactory<T>(name);
560 if (factory != nullptr) {
561 return factory;
562 }
563 }
564 }
565 if (parent_ == nullptr) {
566 return nullptr;
567 } else {
568 return parent_->FindFactory<T>(name);
569 }
570 }
