20struct FragmentedRangeTombstoneList {
21 public:
22 // A compact representation of a "stack" of range tombstone fragments, which
23 // start and end at the same user keys but have different sequence numbers.
24 // The members seq_start_idx and seq_end_idx are intended to be parameters to
25 // seq_iter().
26 // If user-defined timestamp is enabled, `start` and `end` should be user keys
27 // with timestamp, and the timestamps are set to max timestamp to be returned
28 // by parsed_start_key()/parsed_end_key(). seq_start_idx and seq_end_idx will
29 // also be used as parameters to ts_iter().
30 struct RangeTombstoneStack {
31 RangeTombstoneStack(const Slice& start, const Slice& end, size_t start_idx,
32 size_t end_idx)
33 : start_key(start),
34 end_key(end),
35 seq_start_idx(start_idx),
36 seq_end_idx(end_idx) {}
37 Slice start_key;
38 Slice end_key;
39 size_t seq_start_idx;
40 size_t seq_end_idx;
41 };
42 // Assumes unfragmented_tombstones->key() and unfragmented_tombstones->value()
43 // both contain timestamp if enabled.
44 FragmentedRangeTombstoneList(
45 std::unique_ptr<InternalIterator> unfragmented_tombstones,
46 const InternalKeyComparator& icmp, bool for_compaction = false,
47 const std::vector<SequenceNumber>& snapshots = {},
48 const bool tombstone_end_include_ts = true);
49
50 std::vector<RangeTombstoneStack>::const_iterator begin() const {

90 private:
91 // Given an ordered range tombstone iterator unfragmented_tombstones,
92 // "fragment" the tombstones into non-overlapping pieces. Each
93 // "non-overlapping piece" is a RangeTombstoneStack in tombstones_, which
94 // contains start_key, end_key, and indices that points to sequence numbers
95 // (in tombstone_seqs_) and timestamps (in tombstone_timestamps_). If
96 // for_compaction is true, then `snapshots` should be provided. Range
97 // tombstone fragments are dropped if they are not visible in any snapshot and
98 // user-defined timestamp is not enabled. That is, for each snapshot stripe
99 // [lower, upper], the range tombstone fragment with largest seqno in [lower,
100 // upper] is preserved, and all the other range tombstones are dropped.
101 void FragmentTombstones(
102 std::unique_ptr<InternalIterator> unfragmented_tombstones,
103 const InternalKeyComparator& icmp, bool for_compaction,
104 const std::vector<SequenceNumber>& snapshots);

112 PinnedIteratorsManager pinned_iters_mgr_;

125// FragmentedRangeTombstoneIterator converts an InternalIterator of a range-del
126// meta block into an iterator over non-overlapping tombstone fragments. The
127// tombstone fragmentation process should be more efficient than the range
128// tombstone collapsing algorithm in RangeDelAggregator because this leverages
129// the internal key ordering already provided by the input iterator, if
130// applicable (when the iterator is unsorted, a new sorted iterator is created
131// before proceeding). If there are few overlaps, creating a
132// FragmentedRangeTombstoneIterator should be O(n), while the RangeDelAggregator
133// tombstone collapsing is always O(n log n).

334 // Check the current RangeTombstoneStack `pos_` against timestamp
335 // upper bound `ts_upper_bound_` and sequence number upper bound
336 // `upper_bound_`. Update the sequence number (and timestamp) pointer
337 // `seq_pos_` to the first valid position satisfying both bounds.
338 void SetMaxVisibleSeqAndTimestamp() {
339 seq_pos_ = std::lower_bound(tombstones_->seq_iter(pos_->seq_start_idx),
340 tombstones_->seq_iter(pos_->seq_end_idx),
341 upper_bound_, std::greater<SequenceNumber>());
342 if (ts_upper_bound_ && !ts_upper_bound_->empty()) {
343 auto ts_pos = std::lower_bound(
344 tombstones_->ts_iter(pos_->seq_start_idx),
345 tombstones_->ts_iter(pos_->seq_end_idx), *ts_upper_bound_,
346 [this](const Slice& s1, const Slice& s2) {
347 return ucmp_->CompareTimestamp(s1, s2) > 0;
348 });
349 auto ts_idx = ts_pos - tombstones_->ts_iter(pos_->seq_start_idx);
350 auto seq_idx = seq_pos_ - tombstones_->seq_iter(pos_->seq_start_idx);
351 if (seq_idx < ts_idx) {
352 // seq and ts are ordered in non-increasing order. Only updates seq_pos_
353 // to a larger index for smaller sequence number and timestamp.
354 seq_pos_ = tombstones_->seq_iter(pos_->seq_start_idx + ts_idx);
355 }
356 }
357 }
