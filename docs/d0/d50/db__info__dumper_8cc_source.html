19void DumpDBFileSummary(const ImmutableDBOptions& options,
20 const std::string& dbname,
21 const std::string& session_id) {
22 if (options.info_log == nullptr) {
23 return;
24 }
25
26 auto* env = options.env;
27 uint64_t number = 0;
28 FileType type = kInfoLogFile;
29
30 std::vector<std::string> files;
31 uint64_t file_num = 0;
32 uint64_t file_size;
33 std::string file_info, wal_info;
34
35 Header(options.info_log, "DB SUMMARY\n");
36 {
37 std::string hostname;
38 if (env->GetHostNameString(&hostname).ok()) {
39 Header(options.info_log, "Host name (Env): %s\n", hostname.c_str());
40 }
41 }
42 Header(options.info_log, "DB Session ID: %s\n", session_id.c_str());
43
44 Status s;
45 // Get files in dbname dir
46 s = env->GetChildren(dbname, &files);
47 if (!s.ok()) {
48 Error(options.info_log, "Error when reading %s dir %s\n", dbname.c_str(),
49 s.ToString().c_str());
50 }
51 std::sort(files.begin(), files.end());
52 for (const std::string& file : files) {
53 if (!ParseFileName(file, &number, &type)) {
54 continue;
55 }
56 switch (type) {
57 case kCurrentFile:
58 Header(options.info_log, "CURRENT file: %s\n", file.c_str());
59 break;
60 case kIdentityFile:
61 Header(options.info_log, "IDENTITY file: %s\n", file.c_str());
62 break;
63 case kDescriptorFile:
64 s = env->GetFileSize(dbname + "/" + file, &file_size);
65 if (s.ok()) {
66 Header(options.info_log,
67 "MANIFEST file: %s size: %" PRIu64 " Bytes\n", file.c_str(),
68 file_size);
69 } else {
70 Error(options.info_log,
71 "Error when reading MANIFEST file: %s/%s %s\n", dbname.c_str(),
72 file.c_str(), s.ToString().c_str());
73 }
74 break;
75 case kWalFile:
76 s = env->GetFileSize(dbname + "/" + file, &file_size);
77 if (s.ok()) {
78 wal_info.append(file)
79 .append(" size: ")
80 .append(std::to_string(file_size))
81 .append(" ; ");
82 } else {
83 Error(options.info_log, "Error when reading LOG file: %s/%s %s\n",
84 dbname.c_str(), file.c_str(), s.ToString().c_str());
85 }
86 break;
87 case kTableFile:
88 if (++file_num < 10) {
89 file_info.append(file).append(" ");
90 }
91 break;
92 default:
93 break;
94 }
95 }
96
97 // Get sst files in db_path dir
98 for (auto& db_path : options.db_paths) {
99 if (dbname.compare(db_path.path) != 0) {
100 s = env->GetChildren(db_path.path, &files);
101 if (s.IsNotFound() || s.IsPathNotFound()) {
102 Header(options.info_log,
103 "Directory from db_paths/cf_paths does not yet exist: %s\n",
104 db_path.path.c_str());
105 continue;
106 } else if (!s.ok()) {
107 Error(options.info_log, "Error when reading %s dir %s\n",
108 db_path.path.c_str(), s.ToString().c_str());
109 continue;
110 }
111 std::sort(files.begin(), files.end());
112 for (const std::string& file : files) {
113 if (ParseFileName(file, &number, &type)) {
114 if (type == kTableFile && ++file_num < 10) {
115 file_info.append(file).append(" ");
116 }
117 }
118 }
119 }
120 Header(options.info_log,
121 "SST files in %s dir, Total Num: %" PRIu64 ", files: %s\n",
122 db_path.path.c_str(), file_num, file_info.c_str());
123 file_num = 0;
124 file_info.clear();
125 }
126
127 // Get wal file in wal_dir
128 const auto& wal_dir = options.GetWalDir(dbname);
129 bool log_wal_info = true;
130 if (!options.IsWalDirSameAsDBPath(dbname)) {
131 s = env->GetChildren(wal_dir, &files);
132 if (s.IsNotFound() || s.IsPathNotFound()) {
133 Header(options.info_log,
134 "Write Ahead Log directory does not yet exist: %s\n",
135 wal_dir.c_str());
136 log_wal_info = false;
137 } else if (!s.ok()) {
138 Error(options.info_log, "Error when reading wal dir %s: %s\n",
139 wal_dir.c_str(), s.ToString().c_str());
140 log_wal_info = false;
141 }
142 wal_info.clear();
143 for (const std::string& file : files) {
144 if (ParseFileName(file, &number, &type)) {
145 if (type == kWalFile) {
146 s = env->GetFileSize(wal_dir + "/" + file, &file_size);
147 if (s.ok()) {
148 wal_info.append(file)
149 .append(" size: ")
150 .append(std::to_string(file_size))
151 .append(" ; ");
152 } else {
153 Error(options.info_log, "Error when reading LOG file %s/%s %s\n",
154 wal_dir.c_str(), file.c_str(), s.ToString().c_str());
155 }
156 }
157 }
158 }
159 }
160 if (log_wal_info) {
161 Header(options.info_log, "Write Ahead Log file in %s: %s\n",
162 wal_dir.c_str(), wal_info.c_str());
163 }
164}
165} // namespace ROCKSDB_NAMESPACE
