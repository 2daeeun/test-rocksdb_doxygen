41std::shared_ptr<DB> OpenDb(bool read_only = false) {
42 DB* db;
43 Options options;
44 options.create_if_missing = true;
45 options.max_open_files = -1;
46 options.write_buffer_size = FLAGS_write_buffer_size;
47 options.max_write_buffer_number = FLAGS_max_write_buffer_number;
48 options.min_write_buffer_number_to_merge =
49 FLAGS_min_write_buffer_number_to_merge;
50
51 if (FLAGS_use_set_based_memetable) {
52 options.prefix_extractor.reset(
53 ROCKSDB_NAMESPACE::NewFixedPrefixTransform(0));
54 options.memtable_factory.reset(NewHashSkipListRepFactory());
55 }
56
57 Status s;
58 if (!read_only) {
59 s = DB::Open(options, kDbName, &db);
60 } else {
61 s = DB::OpenForReadOnly(options, kDbName, &db);
62 }
63 EXPECT_OK(s);
64 return std::shared_ptr<DB>(db);
65}

512// make perf_context_test
513// export ROCKSDB_TESTS=PerfContextTest.SeekKeyComparison
514// For one memtable:
515// ./perf_context_test --write_buffer_size=500000 --total_keys=10000
516// For two memtables:
517// ./perf_context_test --write_buffer_size=250000 --total_keys=10000
518// Specify --random_key=1 to shuffle the key before insertion
519// Results show that, for sequential insertion, worst-case Seek Key comparison
520// is close to the total number of keys (linear), when there is only one
521// memtable. When there are two memtables, even the avg Seek Key comparison
522// starts to become linear to the input size.
523
524TEST_F(PerfContextTest, SeekKeyComparison) {
525 ASSERT_OK(DestroyDB(kDbName, Options()));
526 auto db = OpenDb();
527 WriteOptions write_options;
528 ReadOptions read_options;
529
530 if (FLAGS_verbose) {
531 std::cout << "Inserting " << FLAGS_total_keys << " key/value pairs\n...\n";
532 }
533
534 std::vector<int> keys;
535 for (int i = 0; i < FLAGS_total_keys; ++i) {
536 keys.push_back(i);
537 }
538
539 if (FLAGS_random_key) {
540 RandomShuffle(std::begin(keys), std::end(keys));
541 }
542
543 HistogramImpl hist_put_time;
544 HistogramImpl hist_wal_time;
545 HistogramImpl hist_time_diff;
546
547 SetPerfLevel(kEnableTime);
548 StopWatchNano timer(SystemClock::Default().get());
549 for (const int i : keys) {
550 std::string key = "k" + std::to_string(i);
551 std::string value = "v" + std::to_string(i);
552
553 get_perf_context()->Reset();
554 timer.Start();
555 ASSERT_OK(db->Put(write_options, key, value));
556 auto put_time = timer.ElapsedNanos();
557 hist_put_time.Add(put_time);
558 hist_wal_time.Add(get_perf_context()->write_wal_time);
559 hist_time_diff.Add(put_time - get_perf_context()->write_wal_time);
560 }
561
562 if (FLAGS_verbose) {
563 std::cout << "Put time:\n"
564 << hist_put_time.ToString() << "WAL time:\n"
565 << hist_wal_time.ToString() << "time diff:\n"
566 << hist_time_diff.ToString();
567 }
568
569 HistogramImpl hist_seek;
570 HistogramImpl hist_next;
571
572 for (int i = 0; i < FLAGS_total_keys; ++i) {
573 std::string key = "k" + std::to_string(i);
574 std::string value = "v" + std::to_string(i);
575
576 std::unique_ptr<Iterator> iter(db->NewIterator(read_options));
577 get_perf_context()->Reset();
578 iter->Seek(key);
579 ASSERT_TRUE(iter->Valid());
580 ASSERT_EQ(iter->value().ToString(), value);
581 hist_seek.Add(get_perf_context()->user_key_comparison_count);
582 }
583
584 std::unique_ptr<Iterator> iter(db->NewIterator(read_options));
585 for (iter->SeekToFirst(); iter->Valid();) {
586 get_perf_context()->Reset();
587 iter->Next();
588 hist_next.Add(get_perf_context()->user_key_comparison_count);
589 }
590 ASSERT_OK(iter->status());
591 if (FLAGS_verbose) {
592 std::cout << "Seek:\n"
593 << hist_seek.ToString() << "Next:\n"
594 << hist_next.ToString();
595 }
596}

598TEST_F(PerfContextTest, DBMutexLockCounter) {
599 int stats_code[] = {0, static_cast<int>(DB_MUTEX_WAIT_MICROS)};
600 for (PerfLevel perf_level_test :
601 {PerfLevel::kEnableTimeExceptForMutex, PerfLevel::kEnableTime}) {
602 for (int c = 0; c < 2; ++c) {
603 InstrumentedMutex mutex(nullptr, SystemClock::Default().get(),
604 stats_code[c]);
605 mutex.Lock();
606 ROCKSDB_NAMESPACE::port::Thread child_thread([&] {
607 SetPerfLevel(perf_level_test);
608 get_perf_context()->Reset();
609 ASSERT_EQ(get_perf_context()->db_mutex_lock_nanos, 0);
610 mutex.Lock();
611 mutex.Unlock();
612 if (perf_level_test == PerfLevel::kEnableTimeExceptForMutex ||
613 stats_code[c] != DB_MUTEX_WAIT_MICROS) {
614 ASSERT_EQ(get_perf_context()->db_mutex_lock_nanos, 0);
