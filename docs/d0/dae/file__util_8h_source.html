78inline IOStatus PrepareIOFromReadOptions(const ReadOptions& ro,
79 SystemClock* clock, IOOptions& opts) {
80 if (ro.deadline.count()) {
81 std::chrono::microseconds now =
82 std::chrono::microseconds(clock->NowMicros());
83 // Ensure there is atleast 1us available. We don't want to pass a value of
84 // 0 as that means no timeout
85 if (now >= ro.deadline) {
86 return IOStatus::TimedOut("Deadline exceeded");
87 }
88 opts.timeout = ro.deadline - now;
89 }
90
91 if (ro.io_timeout.count() &&
92 (!opts.timeout.count() || ro.io_timeout < opts.timeout)) {
93 opts.timeout = ro.io_timeout;
94 }
95
96 opts.rate_limiter_priority = ro.rate_limiter_priority;
97 opts.io_activity = ro.io_activity;
98
99 return IOStatus::OK();
100}

102inline IOStatus PrepareIOFromWriteOptions(const WriteOptions& wo,
103 IOOptions& opts) {
104 opts.rate_limiter_priority = wo.rate_limiter_priority;
105 opts.io_activity = wo.io_activity;
106
107 return IOStatus::OK();
108}
