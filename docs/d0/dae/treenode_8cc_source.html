67void treenode::init(const comparator *cmp) {
68 m_txnid = TXNID_NONE;
69 m_is_root = false;
70 m_is_empty = true;
71 m_cmp = cmp;
72
73 m_is_shared = false;
74 m_owners = nullptr;
75
76 // use an adaptive mutex at each node since we expect the time the
77 // lock is held to be relatively short compared to a context switch.
78 // indeed, this improves performance at high thread counts considerably.
79 memset(&m_mutex, 0, sizeof(toku_mutex_t));
80 toku_pthread_mutexattr_t attr;
81 toku_mutexattr_init(&attr);
82 toku_mutexattr_settype(&attr, TOKU_MUTEX_ADAPTIVE);
83 toku_mutex_init(treenode_mutex_key, &m_mutex, &attr);
84 toku_mutexattr_destroy(&attr);
85 m_left_child.set(nullptr);
86 m_right_child.set(nullptr);
87}

143bool treenode::add_shared_owner(TXNID txnid) {
144 assert(m_is_shared);
145 if (txnid == m_txnid)
146 return false; // acquiring a lock on the same range by the same trx
147
148 if (m_txnid != TXNID_SHARED) {
149 m_owners = new TxnidVector;
150 m_owners->insert(m_txnid);
151 m_txnid = TXNID_SHARED;
152 }
153 m_owners->insert(txnid);
154 return true;
155}

177uint32_t treenode::get_depth_estimate(void) const {
178 const uint32_t left_est = m_left_child.depth_est;
179 const uint32_t right_est = m_right_child.depth_est;
180 return (left_est > right_est ? left_est : right_est) + 1;

183treenode *treenode::find_node_with_overlapping_child(
184 const keyrange &range, const keyrange::comparison *cmp_hint) {
185 // determine which child to look at based on a comparison. if we were
186 // given a comparison hint, use that. otherwise, compare them now.
187 keyrange::comparison c =
188 cmp_hint ? *cmp_hint : range.compare(*m_cmp, m_range);
189
190 treenode *child;
191 if (c == keyrange::comparison::LESS_THAN) {
192 child = lock_and_rebalance_left();
193 } else {
194 // The caller (locked_keyrange::acquire) handles the case where
195 // the root of the locked_keyrange is the node that overlaps.
196 // range is guaranteed not to overlap this node.
197 invariant(c == keyrange::comparison::GREATER_THAN);
198 child = lock_and_rebalance_right();
199 }
200
201 // if the search would lead us to an empty subtree (child == nullptr),
202 // or the child overlaps, then we know this node is the parent we want.
203 // otherwise we need to recur into that child.
204 if (child == nullptr) {
205 return this;
206 } else {
207 c = range.compare(*m_cmp, child->m_range);
208 if (c == keyrange::comparison::EQUALS ||
209 c == keyrange::comparison::OVERLAPS) {
210 child->mutex_unlock();
211 return this;
212 } else {
213 // unlock this node before recurring into the locked child,
214 // passing in a comparison hint since we just comapred range
215 // to the child's range.
216 mutex_unlock();
217 return child->find_node_with_overlapping_child(range, &c);
218 }
219 }

222bool treenode::insert(const keyrange &range, TXNID txnid, bool is_shared) {
223 int rc = true;
224 // choose a child to check. if that child is null, then insert the new node
225 // there. otherwise recur down that child's subtree
226 keyrange::comparison c = range.compare(*m_cmp, m_range);
227 if (c == keyrange::comparison::LESS_THAN) {
228 treenode *left_child = lock_and_rebalance_left();
229 if (left_child == nullptr) {
230 left_child = treenode::alloc(m_cmp, range, txnid, is_shared);
231 m_left_child.set(left_child);
232 } else {
233 left_child->insert(range, txnid, is_shared);
234 left_child->mutex_unlock();
235 }
236 } else if (c == keyrange::comparison::GREATER_THAN) {
237 // invariant(c == keyrange::comparison::GREATER_THAN);
238 treenode *right_child = lock_and_rebalance_right();
239 if (right_child == nullptr) {
240 right_child = treenode::alloc(m_cmp, range, txnid, is_shared);
241 m_right_child.set(right_child);
242 } else {
243 right_child->insert(range, txnid, is_shared);
244 right_child->mutex_unlock();
245 }
246 } else if (c == keyrange::comparison::EQUALS) {
247 invariant(is_shared);
248 invariant(m_is_shared);
249 rc = add_shared_owner(txnid);
250 } else {
251 invariant(0);
252 }
253 return rc;
254}

256treenode *treenode::find_child_at_extreme(int direction, treenode **parent) {
257 treenode *child =
258 direction > 0 ? m_right_child.get_locked() : m_left_child.get_locked();
259
260 if (child) {
261 *parent = this;
262 treenode *child_extreme = child->find_child_at_extreme(direction, parent);
263 child->mutex_unlock();
264 return child_extreme;
265 } else {
266 return this;
267 }
268}

278treenode *treenode::remove_root_of_subtree() {
279 // if this node has no children, just free it and return null
280 if (m_left_child.ptr == nullptr && m_right_child.ptr == nullptr) {
281 // treenode::free requires that non-root nodes are unlocked
282 if (!is_root()) {
283 mutex_unlock();
284 }
285 treenode::free(this);
286 return nullptr;
287 }
288
289 // we have a child, so get either the in-order successor or
290 // predecessor of this node to be our replacement.
291 // replacement_parent is updated by the find functions as
292 // they recur down the tree, so initialize it to this.
293 treenode *child, *replacement;
294 treenode *replacement_parent = this;
295 if (m_left_child.ptr != nullptr) {
296 child = m_left_child.get_locked();
297 replacement = child->find_rightmost_child(&replacement_parent);
298 invariant(replacement == child || replacement_parent != this);
299
300 // detach the replacement from its parent
301 if (replacement_parent == this) {
302 m_left_child = replacement->m_left_child;
303 } else {
304 replacement_parent->m_right_child = replacement->m_left_child;
305 }
306 } else {
307 child = m_right_child.get_locked();
308 replacement = child->find_leftmost_child(&replacement_parent);
309 invariant(replacement == child || replacement_parent != this);
310
311 // detach the replacement from its parent
312 if (replacement_parent == this) {
313 m_right_child = replacement->m_right_child;
314 } else {
315 replacement_parent->m_left_child = replacement->m_right_child;
316 }
317 }
318 child->mutex_unlock();
319
320 // swap in place with the detached replacement, then destroy it
321 treenode::swap_in_place(replacement, this);
322 treenode::free(replacement);
323
324 return this;
325}

327void treenode::recursive_remove(void) {
328 treenode *left = m_left_child.ptr;
329 if (left) {
330 left->recursive_remove();
331 }
332 m_left_child.set(nullptr);
333
334 treenode *right = m_right_child.ptr;
335 if (right) {
336 right->recursive_remove();
337 }
338 m_right_child.set(nullptr);
339
340 // we do not take locks on the way down, so we know non-root nodes
341 // are unlocked here and the caller is required to pass a locked
342 // root, so this free is correct.
343 treenode::free(this);
344}

346void treenode::remove_shared_owner(TXNID txnid) {
347 assert(m_owners->size() > 1);
348 m_owners->erase(txnid);
349 assert(m_owners->size() > 0);
350 /* if there is just one owner left, move it to m_txnid */
351 if (m_owners->size() == 1) {
352 m_txnid = *m_owners->begin();
353 delete m_owners;
354 m_owners = nullptr;
355 }

358treenode *treenode::remove(const keyrange &range, TXNID txnid) {
359 treenode *child;
360 // if the range is equal to this node's range, then just remove
361 // the root of this subtree. otherwise search down the tree
362 // in either the left or right children.
363 keyrange::comparison c = range.compare(*m_cmp, m_range);
364 switch (c) {
365 case keyrange::comparison::EQUALS: {
366 // if we are the only owners, remove. Otherwise, just remove
367 // us from the owners list.
368 if (txnid != TXNID_ANY && has_multiple_owners()) {
369 remove_shared_owner(txnid);
370 return this;
371 } else {
372 return remove_root_of_subtree();
373 }
374 }
375 case keyrange::comparison::LESS_THAN:
376 child = m_left_child.get_locked();
377 invariant_notnull(child);
378 child = child->remove(range, txnid);
379
380 // unlock the child if there still is one.
381 // regardless, set the right child pointer
382 if (child) {
383 child->mutex_unlock();
384 }
385 m_left_child.set(child);
386 break;
387 case keyrange::comparison::GREATER_THAN:
388 child = m_right_child.get_locked();
389 invariant_notnull(child);
390 child = child->remove(range, txnid);
391
392 // unlock the child if there still is one.
393 // regardless, set the right child pointer
394 if (child) {
395 child->mutex_unlock();
396 }
397 m_right_child.set(child);
398 break;
399 case keyrange::comparison::OVERLAPS:
400 // shouldn't be overlapping, since the tree is
401 // non-overlapping and this range must exist
402 abort();
403 }
404
405 return this;
406}

408bool treenode::left_imbalanced(int threshold) const {
409 uint32_t left_depth = m_left_child.depth_est;
410 uint32_t right_depth = m_right_child.depth_est;
411 return m_left_child.ptr != nullptr && left_depth > threshold + right_depth;
412}

414bool treenode::right_imbalanced(int threshold) const {
415 uint32_t left_depth = m_left_child.depth_est;
416 uint32_t right_depth = m_right_child.depth_est;
417 return m_right_child.ptr != nullptr && right_depth > threshold + left_depth;
418}

420// effect: rebalances the subtree rooted at this node
421// using AVL style O(1) rotations. unlocks this
422// node if it is not the new root of the subtree.
423// requires: node is locked by this thread, children are not
424// returns: locked root node of the rebalanced tree
425treenode *treenode::maybe_rebalance(void) {
426 // if we end up not rotating at all, the new root is this
427 treenode *new_root = this;
428 treenode *child = nullptr;
429
430 if (left_imbalanced(IMBALANCE_THRESHOLD)) {
431 child = m_left_child.get_locked();
432 if (child->right_imbalanced(0)) {
433 treenode *grandchild = child->m_right_child.get_locked();
434
435 child->m_right_child = grandchild->m_left_child;
436 grandchild->m_left_child.set(child);
437
438 m_left_child = grandchild->m_right_child;
439 grandchild->m_right_child.set(this);
440
441 new_root = grandchild;
442 } else {
443 m_left_child = child->m_right_child;
444 child->m_right_child.set(this);
445 new_root = child;
446 }
447 } else if (right_imbalanced(IMBALANCE_THRESHOLD)) {
448 child = m_right_child.get_locked();
449 if (child->left_imbalanced(0)) {
450 treenode *grandchild = child->m_left_child.get_locked();
451
452 child->m_left_child = grandchild->m_right_child;
453 grandchild->m_right_child.set(child);
454
455 m_right_child = grandchild->m_left_child;
456 grandchild->m_left_child.set(this);
457
458 new_root = grandchild;
459 } else {
460 m_right_child = child->m_left_child;
461 child->m_left_child.set(this);
462 new_root = child;
463 }
464 }
465
466 // up to three nodes may be locked.
467 // - this
468 // - child
469 // - grandchild (but if it is locked, its the new root)
470 //
471 // one of them is the new root. we unlock everything except the new root.
472 if (child && child != new_root) {
473 TOKU_VALGRIND_RESET_MUTEX_ORDERING_INFO(&child->m_mutex);
474 child->mutex_unlock();
475 }
476 if (this != new_root) {
477 TOKU_VALGRIND_RESET_MUTEX_ORDERING_INFO(&m_mutex);
478 mutex_unlock();
479 }
480 TOKU_VALGRIND_RESET_MUTEX_ORDERING_INFO(&new_root->m_mutex);
481 return new_root;
482}

484treenode *treenode::lock_and_rebalance_left(void) {
485 treenode *child = m_left_child.get_locked();
486 if (child) {
487 treenode *new_root = child->maybe_rebalance();
488 m_left_child.set(new_root);
489 child = new_root;
490 }
491 return child;
492}
