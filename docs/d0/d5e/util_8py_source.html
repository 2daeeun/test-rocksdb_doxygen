67def run_shell_command(shell_cmd, cmd_dir=None):
68 """Run a single shell command.
69 @returns a tuple of shell command return code, stdout, stderr"""
70
71 if cmd_dir is not None and not os.path.exists(cmd_dir):
72 run_shell_command("mkdir -p %s" % cmd_dir)
73
74 start = time.time()
75 print("\t>>> Running: " + shell_cmd)
76 p = subprocess.Popen( # noqa
77 shell_cmd,
78 shell=True,
79 stdout=subprocess.PIPE,
80 stderr=subprocess.PIPE,
81 cwd=cmd_dir,
82 )
83 stdout, stderr = p.communicate()
84 end = time.time()
85
86 # Report time if we spent more than 5 minutes executing a command
87 execution_time = end - start
88 if execution_time > (60 * 5):
89 mins = execution_time / 60
90 secs = execution_time % 60
91 print("\t>time spent: %d minutes %d seconds" % (mins, secs))
92
93 return p.returncode, stdout, stderr

96def run_shell_commands(shell_cmds, cmd_dir=None, verbose=False):
97 """Execute a sequence of shell commands, which is equivalent to
98 running `cmd1 && cmd2 && cmd3`
99 @returns boolean indication if all commands succeeds.
100 """
101
102 if cmd_dir:
103 print("\t=== Set current working directory => %s" % cmd_dir)
104
105 for shell_cmd in shell_cmds:
106 ret_code, stdout, stderr = run_shell_command(shell_cmd, cmd_dir)
107 if stdout:
108 if verbose or ret_code != 0:
109 print(ColorString.info("stdout: \n"), stdout)
110 if stderr:
111 # contents in stderr is not necessarily to be error messages.
112 if verbose or ret_code != 0:
113 print(ColorString.error("stderr: \n"), stderr)
114 if ret_code != 0:
115 return False
116
117 return True
