57void AppendVarint64(IterKey* key, uint64_t v) {
58 char buf[10];
59 auto ptr = EncodeVarint64(buf, v);
60 key->TrimAppend(key->Size(), buf, ptr - buf);
61}

90Status TableCache::GetTableReader(
91 const ReadOptions& ro, const FileOptions& file_options,
92 const InternalKeyComparator& internal_comparator,
93 const FileMetaData& file_meta, bool sequential_mode,
94 HistogramImpl* file_read_hist, std::unique_ptr<TableReader>* table_reader,
95 const MutableCFOptions& mutable_cf_options, bool skip_filters, int level,
96 bool prefetch_index_and_filter_in_cache,
97 size_t max_file_size_for_l0_meta_pin, Temperature file_temperature) {
98 std::string fname = TableFileName(
99 ioptions_.cf_paths, file_meta.fd.GetNumber(), file_meta.fd.GetPathId());
100 std::unique_ptr<FSRandomAccessFile> file;
101 FileOptions fopts = file_options;
102 fopts.temperature = file_temperature;
103 Status s = PrepareIOFromReadOptions(ro, ioptions_.clock, fopts.io_options);
104 TEST_SYNC_POINT_CALLBACK("TableCache::GetTableReader:BeforeOpenFile",
105 const_cast<Status*>(&s));
106 if (s.ok()) {
107 s = ioptions_.fs->NewRandomAccessFile(fname, fopts, &file, nullptr);
108 }
109 if (s.ok()) {
110 RecordTick(ioptions_.stats, NO_FILE_OPENS);
111 } else if (s.IsPathNotFound()) {
112 fname = Rocks2LevelTableFileName(fname);
113 // If this file is also not found, we want to use the error message
114 // that contains the table file name which is less confusing.
115 Status temp_s =
116 PrepareIOFromReadOptions(ro, ioptions_.clock, fopts.io_options);
117 if (temp_s.ok()) {
118 temp_s = ioptions_.fs->NewRandomAccessFile(fname, file_options, &file,
119 nullptr);
120 }
121 if (temp_s.ok()) {
122 RecordTick(ioptions_.stats, NO_FILE_OPENS);
123 s = temp_s;
124 }
125 }
126
127 if (s.ok()) {
128 if (!sequential_mode && ioptions_.advise_random_on_open) {
129 file->Hint(FSRandomAccessFile::kRandom);
130 }
131 if (ioptions_.default_temperature != Temperature::kUnknown &&
132 file_temperature == Temperature::kUnknown) {
133 file_temperature = ioptions_.default_temperature;
134 }
135 StopWatch sw(ioptions_.clock, ioptions_.stats, TABLE_OPEN_IO_MICROS);
136 std::unique_ptr<RandomAccessFileReader> file_reader(
137 new RandomAccessFileReader(std::move(file), fname, ioptions_.clock,
138 io_tracer_, ioptions_.stats, SST_READ_MICROS,
139 file_read_hist, ioptions_.rate_limiter.get(),
140 ioptions_.listeners, file_temperature,
141 level == ioptions_.num_levels - 1));
142 UniqueId64x2 expected_unique_id;
143 if (ioptions_.verify_sst_unique_id_in_manifest) {
144 expected_unique_id = file_meta.unique_id;
145 } else {
146 expected_unique_id = kNullUniqueId64x2; // null ID == no verification
147 }
148 s = mutable_cf_options.table_factory->NewTableReader(
149 ro,
150 TableReaderOptions(
151 ioptions_, mutable_cf_options.prefix_extractor, file_options,
152 internal_comparator,
153 mutable_cf_options.block_protection_bytes_per_key, skip_filters,
154 immortal_tables_, false /* force_direct_prefetch */, level,
155 block_cache_tracer_, max_file_size_for_l0_meta_pin, db_session_id_,
156 file_meta.fd.GetNumber(), expected_unique_id,
157 file_meta.fd.largest_seqno, file_meta.tail_size,
158 file_meta.user_defined_timestamps_persisted),
159 std::move(file_reader), file_meta.fd.GetFileSize(), table_reader,
160 prefetch_index_and_filter_in_cache);
161 TEST_SYNC_POINT("TableCache::GetTableReader:0");
162 }
163 return s;
164}

172Status TableCache::FindTable(
173 const ReadOptions& ro, const FileOptions& file_options,
174 const InternalKeyComparator& internal_comparator,
175 const FileMetaData& file_meta, TypedHandle** handle,
176 const MutableCFOptions& mutable_cf_options, const bool no_io,
177 HistogramImpl* file_read_hist, bool skip_filters, int level,
178 bool prefetch_index_and_filter_in_cache,
179 size_t max_file_size_for_l0_meta_pin, Temperature file_temperature) {
180 PERF_TIMER_GUARD_WITH_CLOCK(find_table_nanos, ioptions_.clock);
181 uint64_t number = file_meta.fd.GetNumber();
182 // NOTE: sharing same Cache with BlobFileCache
183 Slice key = GetSliceForFileNumber(&number);
184 *handle = cache_.Lookup(key);
185 TEST_SYNC_POINT_CALLBACK("TableCache::FindTable:0",
186 const_cast<bool*>(&no_io));
187
188 if (*handle == nullptr) {
189 if (no_io) {
190 return Status::Incomplete("Table not found in table_cache, no_io is set");
191 }
192 MutexLock load_lock(&loader_mutex_.Get(key));
193 // We check the cache again under loading mutex
194 *handle = cache_.Lookup(key);
195 if (*handle != nullptr) {
196 return Status::OK();
197 }
198
199 std::unique_ptr<TableReader> table_reader;
200 Status s = GetTableReader(ro, file_options, internal_comparator, file_meta,
201 false /* sequential mode */, file_read_hist,
202 &table_reader, mutable_cf_options, skip_filters,
203 level, prefetch_index_and_filter_in_cache,
204 max_file_size_for_l0_meta_pin, file_temperature);
205 if (!s.ok()) {
206 assert(table_reader == nullptr);
207 RecordTick(ioptions_.stats, NO_FILE_ERRORS);
208 // We do not cache error results so that if the error is transient,
209 // or somebody repairs the file, we recover automatically.
210 } else {
211 s = cache_.Insert(key, table_reader.get(), 1, handle);
212 if (s.ok()) {
213 // Release ownership of table reader.
214 table_reader.release();
215 }
216 }
217 return s;
218 }
219 return Status::OK();
220}

222InternalIterator* TableCache::NewIterator(
223 const ReadOptions& options, const FileOptions& file_options,
224 const InternalKeyComparator& icomparator, const FileMetaData& file_meta,
225 RangeDelAggregator* range_del_agg,
226 const MutableCFOptions& mutable_cf_options, TableReader** table_reader_ptr,
227 HistogramImpl* file_read_hist, TableReaderCaller caller, Arena* arena,
228 bool skip_filters, int level, size_t max_file_size_for_l0_meta_pin,
229 const InternalKey* smallest_compaction_key,
230 const InternalKey* largest_compaction_key, bool allow_unprepared_value,
231 const SequenceNumber* read_seqno,
232 std::unique_ptr<TruncatedRangeDelIterator>* range_del_iter) {
233 PERF_TIMER_GUARD(new_table_iterator_nanos);
234
235 Status s;
236 TableReader* table_reader = nullptr;
237 TypedHandle* handle = nullptr;
238 if (table_reader_ptr != nullptr) {
239 *table_reader_ptr = nullptr;
240 }
241 bool for_compaction = caller == TableReaderCaller::kCompaction;
242 auto& fd = file_meta.fd;
243 table_reader = fd.table_reader;
244 if (table_reader == nullptr) {
245 s = FindTable(options, file_options, icomparator, file_meta, &handle,
246 mutable_cf_options,
247 options.read_tier == kBlockCacheTier /* no_io */,
248 file_read_hist, skip_filters, level,
249 true /* prefetch_index_and_filter_in_cache */,
250 max_file_size_for_l0_meta_pin, file_meta.temperature);
251 if (s.ok()) {
252 table_reader = cache_.Value(handle);
253 }
254 }
255 InternalIterator* result = nullptr;
256 if (s.ok()) {
257 if (options.table_filter &&
258 !options.table_filter(*table_reader->GetTableProperties())) {
259 result = NewEmptyInternalIterator<Slice>(arena);
260 } else {
261 result = table_reader->NewIterator(
262 options, mutable_cf_options.prefix_extractor.get(), arena,
263 skip_filters, caller, file_options.compaction_readahead_size,
264 allow_unprepared_value);
265 }
266 if (handle != nullptr) {
267 cache_.RegisterReleaseAsCleanup(handle, *result);
268 handle = nullptr; // prevent from releasing below
269 }
270
271 if (for_compaction) {
272 table_reader->SetupForCompaction();
273 }
274 if (table_reader_ptr != nullptr) {
275 *table_reader_ptr = table_reader;
276 }
277 }
278 if (s.ok() && !options.ignore_range_deletions) {
279 if (range_del_iter != nullptr) {
280 auto new_range_del_iter =
281 read_seqno ? table_reader->NewRangeTombstoneIterator(
282 *read_seqno, options.timestamp)
283 : table_reader->NewRangeTombstoneIterator(options);
284 if (new_range_del_iter == nullptr || new_range_del_iter->empty()) {
285 delete new_range_del_iter;
286 *range_del_iter = nullptr;
287 } else {
288 *range_del_iter = std::make_unique<TruncatedRangeDelIterator>(
289 std::unique_ptr<FragmentedRangeTombstoneIterator>(
290 new_range_del_iter),
291 &icomparator, &file_meta.smallest, &file_meta.largest);
292 }
293 }
294 if (range_del_agg != nullptr) {
295 if (range_del_agg->AddFile(fd.GetNumber())) {
296 std::unique_ptr<FragmentedRangeTombstoneIterator> new_range_del_iter(
297 static_cast<FragmentedRangeTombstoneIterator*>(
298 table_reader->NewRangeTombstoneIterator(options)));
299 if (new_range_del_iter != nullptr) {
300 s = new_range_del_iter->status();
301 }
302 if (s.ok()) {
303 const InternalKey* smallest = &file_meta.smallest;
304 const InternalKey* largest = &file_meta.largest;
305 if (smallest_compaction_key != nullptr) {
306 smallest = smallest_compaction_key;
307 }
308 if (largest_compaction_key != nullptr) {
309 largest = largest_compaction_key;
310 }
311 range_del_agg->AddTombstones(std::move(new_range_del_iter), smallest,
312 largest);
313 }
314 }
315 }
316 }
317
318 if (handle != nullptr) {
319 cache_.Release(handle);
320 }
321 if (!s.ok()) {
322 assert(result == nullptr);
323 result = NewErrorInternalIterator<Slice>(s, arena);
324 }
325 return result;
326}

328Status TableCache::GetRangeTombstoneIterator(
329 const ReadOptions& options,
330 const InternalKeyComparator& internal_comparator,
331 const FileMetaData& file_meta, const MutableCFOptions& mutable_cf_options,
332 std::unique_ptr<FragmentedRangeTombstoneIterator>* out_iter) {
333 assert(out_iter);
334 const FileDescriptor& fd = file_meta.fd;
335 Status s;
336 TableReader* t = fd.table_reader;
337 TypedHandle* handle = nullptr;
338 if (t == nullptr) {
339 s = FindTable(options, file_options_, internal_comparator, file_meta,
340 &handle, mutable_cf_options);
341 if (s.ok()) {
342 t = cache_.Value(handle);
343 }
344 }
345 if (s.ok()) {
346 // Note: NewRangeTombstoneIterator could return nullptr
347 out_iter->reset(t->NewRangeTombstoneIterator(options));
348 }
349 if (handle) {
350 if (*out_iter) {
351 cache_.RegisterReleaseAsCleanup(handle, **out_iter);
352 } else {
353 cache_.Release(handle);
354 }
355 }
356 return s;
357}

359uint64_t TableCache::CreateRowCacheKeyPrefix(const ReadOptions& options,
360 const FileDescriptor& fd,
361 const Slice& internal_key,
362 GetContext* get_context,
363 IterKey& row_cache_key) {
364 uint64_t fd_number = fd.GetNumber();
365 // We use the user key as cache key instead of the internal key,
366 // otherwise the whole cache would be invalidated every time the
367 // sequence key increases. However, to support caching snapshot
368 // reads, we append a sequence number (incremented by 1 to
369 // distinguish from 0) other than internal_key seq no
370 // to determine row cache entry visibility.
371 // If the snapshot is larger than the largest seqno in the file,
372 // all data should be exposed to the snapshot, so we treat it
373 // the same as there is no snapshot. The exception is that if
374 // a seq-checking callback is registered, some internal keys
375 // may still be filtered out.
376 uint64_t cache_entry_seq_no = 0;
377
378 // Maybe we can include the whole file ifsnapshot == fd.largest_seqno.
379 if (options.snapshot != nullptr &&
380 (get_context->has_callback() ||
381 static_cast_with_check<const SnapshotImpl>(options.snapshot)
382 ->GetSequenceNumber() <= fd.largest_seqno)) {
383 // We should consider to use options.snapshot->GetSequenceNumber()
384 // instead of GetInternalKeySeqno(k), which will make the code
385 // easier to understand.
386 cache_entry_seq_no = 1 + GetInternalKeySeqno(internal_key);
387 }
388
389 // Compute row cache key.
390 row_cache_key.TrimAppend(row_cache_key.Size(), row_cache_id_.data(),
391 row_cache_id_.size());
392 AppendVarint64(&row_cache_key, fd_number);
393 AppendVarint64(&row_cache_key, cache_entry_seq_no);
394
395 // Provide a sequence number for callback checking on cache hit.
396 // As cache_entry_seq_no starts at 1, decrease it's value by 1 to get
397 // a sequence number align with get context's logic.
398 return cache_entry_seq_no == 0 ? 0 : cache_entry_seq_no - 1;
399}

401bool TableCache::GetFromRowCache(const Slice& user_key, IterKey& row_cache_key,
402 size_t prefix_size, GetContext* get_context,
403 Status* read_status, SequenceNumber seq_no) {
404 bool found = false;
405
406 row_cache_key.TrimAppend(prefix_size, user_key.data(), user_key.size());
407 RowCacheInterface row_cache{ioptions_.row_cache.get()};
408 if (auto row_handle = row_cache.Lookup(row_cache_key.GetUserKey())) {
409 // Cleanable routine to release the cache entry
410 Cleanable value_pinner;
411 // If it comes here value is located on the cache.
412 // found_row_cache_entry points to the value on cache,
413 // and value_pinner has cleanup procedure for the cached entry.
414 // After replayGetContextLog() returns, get_context.pinnable_slice_
415 // will point to cache entry buffer (or a copy based on that) and
416 // cleanup routine under value_pinner will be delegated to
417 // get_context.pinnable_slice_. Cache entry is released when
418 // get_context.pinnable_slice_ is reset.
419 row_cache.RegisterReleaseAsCleanup(row_handle, value_pinner);
420 // If row cache hit, knowing cache key is the same to row_cache_key,
421 // can use row_cache_key's seq no to construct InternalKey.
422 *read_status = replayGetContextLog(*row_cache.Value(row_handle), user_key,
423 get_context, &value_pinner, seq_no);
424 RecordTick(ioptions_.stats, ROW_CACHE_HIT);
425 found = true;
426 } else {
427 RecordTick(ioptions_.stats, ROW_CACHE_MISS);
428 }
429 return found;
430}

432Status TableCache::Get(const ReadOptions& options,
433 const InternalKeyComparator& internal_comparator,
434 const FileMetaData& file_meta, const Slice& k,
435 GetContext* get_context,
436 const MutableCFOptions& mutable_cf_options,
437 HistogramImpl* file_read_hist, bool skip_filters,
438 int level, size_t max_file_size_for_l0_meta_pin) {
439 auto& fd = file_meta.fd;
440 std::string* row_cache_entry = nullptr;
441 bool done = false;
442 IterKey row_cache_key;
443 std::string row_cache_entry_buffer;
444
445 // Check row cache if enabled.
446 // Reuse row_cache_key sequence number when row cache hits.
447 Status s;
448 if (ioptions_.row_cache && !get_context->NeedToReadSequence()) {
449 auto user_key = ExtractUserKey(k);
450 uint64_t cache_entry_seq_no =
451 CreateRowCacheKeyPrefix(options, fd, k, get_context, row_cache_key);
452 done = GetFromRowCache(user_key, row_cache_key, row_cache_key.Size(),
453 get_context, &s, cache_entry_seq_no);
454 if (!done) {
455 row_cache_entry = &row_cache_entry_buffer;
456 }
457 }
458 TableReader* t = fd.table_reader;
459 TypedHandle* handle = nullptr;
460 if (s.ok() && !done) {
461 if (t == nullptr) {
462 s = FindTable(options, file_options_, internal_comparator, file_meta,
463 &handle, mutable_cf_options,
464 options.read_tier == kBlockCacheTier /* no_io */,
465 file_read_hist, skip_filters, level,
466 true /* prefetch_index_and_filter_in_cache */,
467 max_file_size_for_l0_meta_pin, file_meta.temperature);
468 if (s.ok()) {
469 t = cache_.Value(handle);
470 }
471 }
472 SequenceNumber* max_covering_tombstone_seq =
473 get_context->max_covering_tombstone_seq();
474 if (s.ok() && max_covering_tombstone_seq != nullptr &&
475 !options.ignore_range_deletions) {
476 std::unique_ptr<FragmentedRangeTombstoneIterator> range_del_iter(
477 t->NewRangeTombstoneIterator(options));
478 if (range_del_iter != nullptr) {
479 SequenceNumber seq =
480 range_del_iter->MaxCoveringTombstoneSeqnum(ExtractUserKey(k));
481 if (seq > *max_covering_tombstone_seq) {
482 *max_covering_tombstone_seq = seq;
483 if (get_context->NeedTimestamp()) {
484 get_context->SetTimestampFromRangeTombstone(
485 range_del_iter->timestamp());
486 }
487 }
488 }
489 }
490 if (s.ok()) {
491 get_context->SetReplayLog(row_cache_entry); // nullptr if no cache.
492 s = t->Get(options, k, get_context,
493 mutable_cf_options.prefix_extractor.get(), skip_filters);
494 get_context->SetReplayLog(nullptr);
495 } else if (options.read_tier == kBlockCacheTier && s.IsIncomplete()) {
496 // Couldn't find table in cache and couldn't open it because of no_io.
497 get_context->MarkKeyMayExist();
498 done = true;
499 }
500 }
501
502 // Put the replay log in row cache only if something was found.
503 if (!done && s.ok() && row_cache_entry && !row_cache_entry->empty()) {
504 RowCacheInterface row_cache{ioptions_.row_cache.get()};
505 size_t charge = row_cache_entry->capacity() + sizeof(std::string);
506 auto row_ptr = new std::string(std::move(*row_cache_entry));
507 Status rcs = row_cache.Insert(row_cache_key.GetUserKey(), row_ptr, charge);
508 if (!rcs.ok()) {
509 // If row cache is full, it's OK to continue, but we keep ownership of
510 // row_ptr.
511 delete row_ptr;
512 }
513 }
514
515 if (handle != nullptr) {
516 cache_.Release(handle);
517 }
518 return s;
519}

521void TableCache::UpdateRangeTombstoneSeqnums(
522 const ReadOptions& options, TableReader* t,
523 MultiGetContext::Range& table_range) {
524 std::unique_ptr<FragmentedRangeTombstoneIterator> range_del_iter(
525 t->NewRangeTombstoneIterator(options));
526 if (range_del_iter != nullptr) {
527 for (auto iter = table_range.begin(); iter != table_range.end(); ++iter) {
528 SequenceNumber* max_covering_tombstone_seq =
529 iter->get_context->max_covering_tombstone_seq();
530 SequenceNumber seq =
531 range_del_iter->MaxCoveringTombstoneSeqnum(iter->ukey_with_ts);
532 if (seq > *max_covering_tombstone_seq) {
533 *max_covering_tombstone_seq = seq;
534 if (iter->get_context->NeedTimestamp()) {
535 iter->get_context->SetTimestampFromRangeTombstone(
536 range_del_iter->timestamp());
537 }
538 }
539 }
540 }

543Status TableCache::MultiGetFilter(
544 const ReadOptions& options,
545 const InternalKeyComparator& internal_comparator,
546 const FileMetaData& file_meta, const MutableCFOptions& mutable_cf_options,
547 HistogramImpl* file_read_hist, int level,
548 MultiGetContext::Range* mget_range, TypedHandle** table_handle) {
549 auto& fd = file_meta.fd;
550 IterKey row_cache_key;
551 std::string row_cache_entry_buffer;
552
553 // Check if we need to use the row cache. If yes, then we cannot do the
554 // filtering here, since the filtering needs to happen after the row cache
555 // lookup.
556 KeyContext& first_key = *mget_range->begin();
557 if (ioptions_.row_cache && !first_key.get_context->NeedToReadSequence()) {
558 return Status::NotSupported();
559 }
560 Status s;
561 TableReader* t = fd.table_reader;
562 TypedHandle* handle = nullptr;
563 MultiGetContext::Range tombstone_range(*mget_range, mget_range->begin(),
564 mget_range->end());
565 if (t == nullptr) {
566 s = FindTable(options, file_options_, internal_comparator, file_meta,
567 &handle, mutable_cf_options,
568 options.read_tier == kBlockCacheTier /* no_io */,
569 file_read_hist,
570 /*skip_filters=*/false, level,
571 true /* prefetch_index_and_filter_in_cache */,
572 /*max_file_size_for_l0_meta_pin=*/0, file_meta.temperature);
573 if (s.ok()) {
574 t = cache_.Value(handle);
575 }
576 *table_handle = handle;
577 }
578 if (s.ok()) {
579 s = t->MultiGetFilter(options, mutable_cf_options.prefix_extractor.get(),
580 mget_range);
581 }
582 if (s.ok() && !options.ignore_range_deletions) {
583 // Update the range tombstone sequence numbers for the keys here
584 // as TableCache::MultiGet may or may not be called, and even if it
585 // is, it may be called with fewer keys in the rangedue to filtering.
586 UpdateRangeTombstoneSeqnums(options, t, tombstone_range);
587 }
588 if (mget_range->empty() && handle) {
589 cache_.Release(handle);
590 *table_handle = nullptr;
591 }
592
593 return s;
594}

596Status TableCache::GetTableProperties(
597 const FileOptions& file_options, const ReadOptions& read_options,
598 const InternalKeyComparator& internal_comparator,
599 const FileMetaData& file_meta,
600 std::shared_ptr<const TableProperties>* properties,
601 const MutableCFOptions& mutable_cf_options, bool no_io) {
602 auto table_reader = file_meta.fd.table_reader;
603 // table already been pre-loaded?
604 if (table_reader) {
