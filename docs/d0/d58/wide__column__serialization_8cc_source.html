19Status WideColumnSerialization::Serialize(const WideColumns& columns,
20 std::string& output) {
21 const size_t num_columns = columns.size();
22
23 if (num_columns > static_cast<size_t>(std::numeric_limits<uint32_t>::max())) {
24 return Status::InvalidArgument("Too many wide columns");
25 }
26
27 PutVarint32(&output, kCurrentVersion);
28
29 PutVarint32(&output, static_cast<uint32_t>(num_columns));
30
31 const Slice* prev_name = nullptr;
32
33 for (size_t i = 0; i < columns.size(); ++i) {
34 const WideColumn& column = columns[i];
35
36 const Slice& name = column.name();
37 if (name.size() >
38 static_cast<size_t>(std::numeric_limits<uint32_t>::max())) {
39 return Status::InvalidArgument("Wide column name too long");
40 }
41
42 if (prev_name && prev_name->compare(name) >= 0) {
43 return Status::Corruption("Wide columns out of order");
44 }
45
46 const Slice& value = column.value();
47 if (value.size() >
48 static_cast<size_t>(std::numeric_limits<uint32_t>::max())) {
49 return Status::InvalidArgument("Wide column value too long");
50 }
51
52 PutLengthPrefixedSlice(&output, name);
53 PutVarint32(&output, static_cast<uint32_t>(value.size()));
54
55 prev_name = &name;
56 }
57
58 for (const auto& column : columns) {
59 const Slice& value = column.value();
60
61 output.append(value.data(), value.size());
62 }
63
64 return Status::OK();
65}

67Status WideColumnSerialization::Deserialize(Slice& input,
68 WideColumns& columns) {
69 assert(columns.empty());
70
71 uint32_t version = 0;
72 if (!GetVarint32(&input, &version)) {
73 return Status::Corruption("Error decoding wide column version");
74 }
75
76 if (version > kCurrentVersion) {
77 return Status::NotSupported("Unsupported wide column version");
78 }
79
80 uint32_t num_columns = 0;
81 if (!GetVarint32(&input, &num_columns)) {
82 return Status::Corruption("Error decoding number of wide columns");
83 }
84
85 if (!num_columns) {
86 return Status::OK();
87 }
88
89 columns.reserve(num_columns);
90
91 autovector<uint32_t, 16> column_value_sizes;
92 column_value_sizes.reserve(num_columns);
93
94 for (uint32_t i = 0; i < num_columns; ++i) {
95 Slice name;
96 if (!GetLengthPrefixedSlice(&input, &name)) {
97 return Status::Corruption("Error decoding wide column name");
98 }
99
100 if (!columns.empty() && columns.back().name().compare(name) >= 0) {
101 return Status::Corruption("Wide columns out of order");
102 }
103
104 columns.emplace_back(name, Slice());
105
106 uint32_t value_size = 0;
107 if (!GetVarint32(&input, &value_size)) {
108 return Status::Corruption("Error decoding wide column value size");
109 }
110
111 column_value_sizes.emplace_back(value_size);
112 }
113
114 const Slice data(input);
115 size_t pos = 0;
116
117 for (uint32_t i = 0; i < num_columns; ++i) {
118 const uint32_t value_size = column_value_sizes[i];
119
120 if (pos + value_size > data.size()) {
121 return Status::Corruption("Error decoding wide column value payload");
122 }
123
124 columns[i].value() = Slice(data.data() + pos, value_size);
125
126 pos += value_size;
127 }
128
129 return Status::OK();
130}

132Status WideColumnSerialization::GetValueOfDefaultColumn(Slice& input,
133 Slice& value) {
134 WideColumns columns;
135
136 const Status s = Deserialize(input, columns);
137 if (!s.ok()) {
138 return s;
139 }
140
141 if (!WideColumnsHelper::HasDefaultColumn(columns)) {
142 value.clear();
143 return Status::OK();
144 }
145
146 value = WideColumnsHelper::GetDefaultColumn(columns);
147
148 return Status::OK();
149}
