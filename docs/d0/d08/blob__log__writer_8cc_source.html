51Status BlobLogWriter::WriteHeader(const WriteOptions& write_options,
52 BlobLogHeader& header) {
53 assert(block_offset_ == 0);
54 assert(last_elem_type_ == kEtNone);
55 std::string str;
56 header.EncodeTo(&str);
57
58 IOOptions opts;
59 Status s = WritableFileWriter::PrepareIOOptions(write_options, opts);
60 if (s.ok()) {
61 s = dest_->Append(opts, Slice(str));
62 }
63 if (s.ok()) {
64 block_offset_ += str.size();
65 if (do_flush_) {
66 s = dest_->Flush(opts);
67 }
68 }
69 last_elem_type_ = kEtFileHdr;
70 if (s.ok()) {
71 RecordTick(statistics_, BLOB_DB_BLOB_FILE_BYTES_WRITTEN,
72 BlobLogHeader::kSize);
73 }
74 return s;
75}

77Status BlobLogWriter::AppendFooter(const WriteOptions& write_options,
78 BlobLogFooter& footer,
79 std::string* checksum_method,
80 std::string* checksum_value) {
81 assert(block_offset_ != 0);
82 assert(last_elem_type_ == kEtFileHdr || last_elem_type_ == kEtRecord);
83
84 std::string str;
85 footer.EncodeTo(&str);
86
87 Status s;
88 if (dest_->seen_error()) {
89 s.PermitUncheckedError();
90 return Status::IOError("Seen Error. Skip closing.");
91 } else {
92 IOOptions opts;
93 s = WritableFileWriter::PrepareIOOptions(write_options, opts);
94 if (s.ok()) {
95 s = dest_->Append(opts, Slice(str));
96 }
97 if (s.ok()) {
98 block_offset_ += str.size();
99 s = Sync(write_options);
100
101 if (s.ok()) {
102 s = dest_->Close(opts);
103
104 if (s.ok()) {
105 assert(!!checksum_method == !!checksum_value);
106
107 if (checksum_method) {
108 assert(checksum_method->empty());
109
110 std::string method = dest_->GetFileChecksumFuncName();
111 if (method != kUnknownFileChecksumFuncName) {
112 *checksum_method = std::move(method);
113 }
114 }
115 if (checksum_value) {
116 assert(checksum_value->empty());
117
118 std::string value = dest_->GetFileChecksum();
119 if (value != kUnknownFileChecksum) {
120 *checksum_value = std::move(value);
121 }
122 }
123 }
124 }
125 }
126
127 dest_.reset();
128 }
129
130 last_elem_type_ = kEtFileFooter;
131 if (s.ok()) {
132 RecordTick(statistics_, BLOB_DB_BLOB_FILE_BYTES_WRITTEN,
133 BlobLogFooter::kSize);
134 }
135 return s;
136}

139 const Slice& key, const Slice& val,
140 uint64_t expiration, uint64_t* key_offset,

167void BlobLogWriter::ConstructBlobHeader(std::string* buf, const Slice& key,
168 const Slice& val, uint64_t expiration) {
169 BlobLogRecord record;
170 record.key = key;
171 record.value = val;
172 record.expiration = expiration;

176Status BlobLogWriter::EmitPhysicalRecord(const WriteOptions& write_options,
177 const std::string& headerbuf,
178 const Slice& key, const Slice& val,
179 uint64_t* key_offset,
180 uint64_t* blob_offset) {
181 IOOptions opts;
182 Status s = WritableFileWriter::PrepareIOOptions(write_options, opts);
183 if (s.ok()) {
184 s = dest_->Append(opts, Slice(headerbuf));
185 }
186 if (s.ok()) {
187 s = dest_->Append(opts, key);
188 }
189 if (s.ok()) {
190 s = dest_->Append(opts, val);
191 }
192 if (do_flush_ && s.ok()) {
193 s = dest_->Flush(opts);
194 }
195
196 *key_offset = block_offset_ + BlobLogRecord::kHeaderSize;
197 *blob_offset = *key_offset + key.size();
198 block_offset_ = *blob_offset + val.size();
199 last_elem_type_ = kEtRecord;
200 if (s.ok()) {
201 RecordTick(statistics_, BLOB_DB_BLOB_FILE_BYTES_WRITTEN,
202 BlobLogRecord::kHeaderSize + key.size() + val.size());
203 }
204 return s;
205}
