59class AtomicCounter {
60 public:
61 explicit AtomicCounter(Env* env = NULL)
62 : env_(env), cond_count_(&mu_), count_(0) {}
63
64 void Increment() {
65 MutexLock l(&mu_);
66 count_++;
67 cond_count_.SignalAll();
68 }
69
70 int Read() {
71 MutexLock l(&mu_);
72 return count_;
73 }
74
75 bool WaitFor(int count) {
76 MutexLock l(&mu_);
77
78 uint64_t start = env_->NowMicros();
79 while (count_ < count) {
80 uint64_t now = env_->NowMicros();
81 cond_count_.TimedWait(now + /*1s*/ 1 * 1000 * 1000);
82 if (env_->NowMicros() - start > /*10s*/ 10 * 1000 * 1000) {
83 return false;
84 }
85 if (count_ < count) {
86 GTEST_LOG_(WARNING) << "WaitFor is taking more time than usual";
87 }
88 }
89
90 return true;
91 }
92
93 void Reset() {
94 MutexLock l(&mu_);
95 count_ = 0;
96 cond_count_.SignalAll();
97 }
98
99 private:
100 Env* env_;
101 port::Mutex mu_;
102 port::CondVar cond_count_;
103 int count_;
104};
