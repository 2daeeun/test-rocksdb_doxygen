24Status Transaction::CommitAndTryCreateSnapshot(
25 std::shared_ptr<TransactionNotifier> notifier, TxnTimestamp ts,
26 std::shared_ptr<const Snapshot>* snapshot) {
27 if (snapshot) {
28 snapshot->reset();
29 }
30 TxnTimestamp commit_ts = GetCommitTimestamp();
31 if (commit_ts == kMaxTxnTimestamp) {
32 if (ts == kMaxTxnTimestamp) {
33 return Status::InvalidArgument("Commit timestamp unset");
34 } else {
35 const Status s = SetCommitTimestamp(ts);
36 if (!s.ok()) {
37 return s;
38 }
39 }
40 } else if (ts != kMaxTxnTimestamp) {
41 if (ts != commit_ts) {
42 // For now we treat this as error.
43 return Status::InvalidArgument("Different commit ts specified");
44 }
45 }
46 SetSnapshotOnNextOperation(notifier);
47 Status s = Commit();
48 if (!s.ok()) {
49 return s;
50 }
51 assert(s.ok());
52 // If we reach here, we must return ok status for this function.
53 std::shared_ptr<const Snapshot> new_snapshot = GetTimestampedSnapshot();
54
55 if (snapshot) {
56 *snapshot = new_snapshot;
57 }
58 return Status::OK();
59}

130void TransactionBaseImpl::SetSnapshotInternal(const Snapshot* snapshot) {
131 // Set a custom deleter for the snapshot_ SharedPtr as the snapshot needs to

133 snapshot_.reset(snapshot, std::bind(&TransactionBaseImpl::ReleaseSnapshot,
134 this, std::placeholders::_1, db_));
135 snapshot_needed_ = false;
136 snapshot_notifier_ = nullptr;

155Status TransactionBaseImpl::TryLock(ColumnFamilyHandle* column_family,
156 const SliceParts& key, bool read_only,
157 bool exclusive, const bool do_validate,
158 const bool assume_tracked) {
159 size_t key_size = 0;
160 for (int i = 0; i < key.num_parts; ++i) {
161 key_size += key.parts[i].size();
162 }
163
164 std::string str;
165 str.reserve(key_size);
166
167 for (int i = 0; i < key.num_parts; ++i) {
168 str.append(key.parts[i].data(), key.parts[i].size());
169 }
170
171 return TryLock(column_family, str, read_only, exclusive, do_validate,
172 assume_tracked);
173}

175void TransactionBaseImpl::SetSavePoint() {
176 if (save_points_ == nullptr) {
177 save_points_.reset(
178 new std::stack<TransactionBaseImpl::SavePoint,
179 autovector<TransactionBaseImpl::SavePoint>>());
180 }
181 save_points_->emplace(snapshot_, snapshot_needed_, snapshot_notifier_,
182 num_puts_, num_put_entities_, num_deletes_, num_merges_,
183 lock_tracker_factory_);
184 write_batch_.SetSavePoint();
185}

187Status TransactionBaseImpl::RollbackToSavePoint() {
188 if (save_points_ != nullptr && save_points_->size() > 0) {
189 // Restore saved SavePoint
190 TransactionBaseImpl::SavePoint& save_point = save_points_->top();
191 snapshot_ = save_point.snapshot_;
192 snapshot_needed_ = save_point.snapshot_needed_;
193 snapshot_notifier_ = save_point.snapshot_notifier_;
194 num_puts_ = save_point.num_puts_;
195 num_put_entities_ = save_point.num_put_entities_;
196 num_deletes_ = save_point.num_deletes_;
197 num_merges_ = save_point.num_merges_;
198
199 // Rollback batch
200 Status s = write_batch_.RollbackToSavePoint();
201 assert(s.ok());
202
203 // Rollback any keys that were tracked since the last savepoint
204 tracked_locks_->Subtract(*save_point.new_locks_);
205
206 save_points_->pop();
207
208 return s;
209 } else {
210 assert(write_batch_.RollbackToSavePoint().IsNotFound());
211 return Status::NotFound();
212 }
213}
214
215Status TransactionBaseImpl::PopSavePoint() {
216 if (save_points_ == nullptr || save_points_->empty()) {
217 // No SavePoint yet.
218 assert(write_batch_.PopSavePoint().IsNotFound());
219 return Status::NotFound();
220 }
221
222 assert(!save_points_->empty());
223 // If there is another savepoint A below the current savepoint B, then A needs
224 // to inherit tracked_keys in B so that if we rollback to savepoint A, we
225 // remember to unlock keys in B. If there is no other savepoint below, then we
226 // can safely discard savepoint info.
227 if (save_points_->size() == 1) {
228 save_points_->pop();
229 } else {
230 TransactionBaseImpl::SavePoint top(lock_tracker_factory_);
231 std::swap(top, save_points_->top());
232 save_points_->pop();
233
234 save_points_->top().new_locks_->Merge(*top.new_locks_);
235 }
236
237 return write_batch_.PopSavePoint();
238}

301Status TransactionBaseImpl::GetForUpdate(const ReadOptions& read_options,
302 ColumnFamilyHandle* column_family,
303 const Slice& key, std::string* value,
304 bool exclusive,
305 const bool do_validate) {
306 if (!do_validate && read_options.snapshot != nullptr) {
307 return Status::InvalidArgument(
308 "If do_validate is false then GetForUpdate with snapshot is not "
309 "defined.");
310 }
311 if (read_options.io_activity != Env::IOActivity::kUnknown) {
312 return Status::InvalidArgument(
313 "Cannot call GetForUpdate with `ReadOptions::io_activity` != "
314 "`Env::IOActivity::kUnknown`");
315 }
316 Status s =
317 TryLock(column_family, key, true /* read_only */, exclusive, do_validate);
318
319 if (s.ok() && value != nullptr) {
320 assert(value != nullptr);
321 PinnableSlice pinnable_val(value);
322 assert(!pinnable_val.IsPinned());
323 s = GetImpl(read_options, column_family, key, &pinnable_val);
324 if (s.ok() && pinnable_val.IsPinned()) {
325 value->assign(pinnable_val.data(), pinnable_val.size());
326 } // else value is already assigned
327 }
328 return s;
329}

374std::vector<Status> TransactionBaseImpl::MultiGet(
375 const ReadOptions& _read_options,
376 const std::vector<ColumnFamilyHandle*>& column_family,
377 const std::vector<Slice>& keys, std::vector<std::string>* values) {
378 size_t num_keys = keys.size();
379 std::vector<Status> stat_list(num_keys);
380 if (_read_options.io_activity != Env::IOActivity::kUnknown &&
381 _read_options.io_activity != Env::IOActivity::kMultiGet) {
382 Status s = Status::InvalidArgument(
383 "Can only call MultiGet with `ReadOptions::io_activity` is "
384 "`Env::IOActivity::kUnknown` or `Env::IOActivity::kMultiGet`");
385
386 for (size_t i = 0; i < num_keys; ++i) {
387 stat_list[i] = s;
388 }
389 return stat_list;
390 }
391 ReadOptions read_options(_read_options);
392 if (read_options.io_activity == Env::IOActivity::kUnknown) {
393 read_options.io_activity = Env::IOActivity::kMultiGet;
394 }
395
396 values->resize(num_keys);
397 for (size_t i = 0; i < num_keys; ++i) {
398 stat_list[i] =
399 GetImpl(read_options, column_family[i], keys[i], &(*values)[i]);
400 }
401
402 return stat_list;
403}

440std::vector<Status> TransactionBaseImpl::MultiGetForUpdate(
441 const ReadOptions& read_options,
442 const std::vector<ColumnFamilyHandle*>& column_family,
443 const std::vector<Slice>& keys, std::vector<std::string>* values) {
444 size_t num_keys = keys.size();
445 if (read_options.io_activity != Env::IOActivity::kUnknown) {
446 Status s = Status::InvalidArgument(
447 "Cannot call MultiGetForUpdate with `ReadOptions::io_activity` != "
448 "`Env::IOActivity::kUnknown`");
449 return std::vector<Status>(num_keys, s);
450 }
451 // Regardless of whether the MultiGet succeeded, track these keys.
452 values->resize(num_keys);
453
454 // Lock all keys
455 for (size_t i = 0; i < num_keys; ++i) {
456 Status s = TryLock(column_family[i], keys[i], true /* read_only */,
457 true /* exclusive */);
458 if (!s.ok()) {
459 // Fail entire multiget if we cannot lock all keys
460 return std::vector<Status>(num_keys, s);
461 }
462 }
463
464 // TODO(agiardullo): optimize multiget?
465 std::vector<Status> stat_list(num_keys);
466 for (size_t i = 0; i < num_keys; ++i) {
467 stat_list[i] =
468 GetImpl(read_options, column_family[i], keys[i], &(*values)[i]);
469 }
470
471 return stat_list;
472}

491template <typename IterType, typename ImplType, typename ErrorIteratorFuncType>
492std::unique_ptr<IterType> TransactionBaseImpl::NewMultiCfIterator(
493 const ReadOptions& read_options,
494 const std::vector<ColumnFamilyHandle*>& column_families,
495 ErrorIteratorFuncType error_iterator_func) {
496 if (column_families.empty()) {
497 return error_iterator_func(
498 Status::InvalidArgument("No Column Family was provided"));
499 }
500
501 const Comparator* const first_comparator =
502 column_families[0]->GetComparator();
503 assert(first_comparator);
504
505 for (size_t i = 1; i < column_families.size(); ++i) {
506 const Comparator* cf_comparator = column_families[i]->GetComparator();
507 assert(cf_comparator);
508
509 if (first_comparator != cf_comparator &&
510 first_comparator->GetId() != cf_comparator->GetId()) {
511 return error_iterator_func(Status::InvalidArgument(
512 "Different comparators are being used across CFs"));
513 }
514 }
515
516 std::vector<Iterator*> child_iterators;
517 const Status s =
518 db_->NewIterators(read_options, column_families, &child_iterators);
519 if (!s.ok()) {
520 return error_iterator_func(s);
521 }
522
523 assert(column_families.size() == child_iterators.size());
524
525 std::vector<std::pair<ColumnFamilyHandle*, std::unique_ptr<Iterator>>>
526 cfh_iter_pairs;
527 cfh_iter_pairs.reserve(column_families.size());
528 for (size_t i = 0; i < column_families.size(); ++i) {
529 cfh_iter_pairs.emplace_back(
530 column_families[i],
531 write_batch_.NewIteratorWithBase(column_families[i], child_iterators[i],
532 &read_options));
533 }
534
535 return std::make_unique<ImplType>(read_options,
536 column_families[0]->GetComparator(),
537 std::move(cfh_iter_pairs));
538}

558Status TransactionBaseImpl::PutEntityImpl(ColumnFamilyHandle* column_family,
559 const Slice& key,
560 const WideColumns& columns,
561 bool do_validate,
562 bool assume_tracked) {
563 {
564 constexpr bool read_only = false;
565 constexpr bool exclusive = true;
566 const Status s = TryLock(column_family, key, read_only, exclusive,
567 do_validate, assume_tracked);
568 if (!s.ok()) {
569 return s;

572

581 return Status::OK();

823void TransactionBaseImpl::TrackKey(uint32_t cfh_id, const std::string& key,
824 SequenceNumber seq, bool read_only,
825 bool exclusive) {
826 PointLockRequest r;
827 r.column_family_id = cfh_id;
828 r.key = key;
829 r.seq = seq;
830 r.read_only = read_only;
831 r.exclusive = exclusive;

834 tracked_locks_->Track(r);

866void TransactionBaseImpl::UndoGetForUpdate(ColumnFamilyHandle* column_family,
867 const Slice& key) {
868 PointLockRequest r;
869 r.column_family_id = GetColumnFamilyID(column_family);
870 r.key = key.ToString();
871 r.read_only = true;
872
873 bool can_untrack = false;
874 if (save_points_ != nullptr && !save_points_->empty()) {
875 // If there is no GetForUpdate of the key in this save point,
876 // then cannot untrack from the global lock tracker.
877 UntrackStatus s = save_points_->top().new_locks_->Untrack(r);
878 can_untrack = (s != UntrackStatus::NOT_TRACKED);
879 } else {
880 // No save point, so can untrack from the global lock tracker.
881 can_untrack = true;
882 }
883
884 if (can_untrack) {
885 // If erased from the global tracker, then can unlock the key.
886 UntrackStatus s = tracked_locks_->Untrack(r);
887 bool can_unlock = (s == UntrackStatus::REMOVED);
888 if (can_unlock) {
889 UnlockGetForUpdate(column_family, key);
890 }
891 }

938 // recovery. there should not be any meta markers in the batches
939 // we are processing.
940 Status MarkBeginPrepare(bool) override { return Status::InvalidArgument(); }

956 }
957 size_t GetTimestampSize(uint32_t cf_id) {
958 auto cfd = db_->versions_->GetColumnFamilySet()->GetColumnFamily(cf_id);
