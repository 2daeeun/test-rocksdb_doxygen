13Status MemoryUtil::GetApproximateMemoryUsageByType(
14 const std::vector<DB*>& dbs,
15 const std::unordered_set<const Cache*> cache_set,
16 std::map<MemoryUtil::UsageType, uint64_t>* usage_by_type) {
17 usage_by_type->clear();
18
19 // MemTable
20 for (auto* db : dbs) {
21 uint64_t usage = 0;
22 if (db->GetAggregatedIntProperty(DB::Properties::kSizeAllMemTables,
23 &usage)) {
24 (*usage_by_type)[MemoryUtil::kMemTableTotal] += usage;
25 }
26 if (db->GetAggregatedIntProperty(DB::Properties::kCurSizeAllMemTables,
27 &usage)) {
28 (*usage_by_type)[MemoryUtil::kMemTableUnFlushed] += usage;
29 }
30 }
31
32 // Table Readers
33 for (auto* db : dbs) {
34 uint64_t usage = 0;
35 if (db->GetAggregatedIntProperty(DB::Properties::kEstimateTableReadersMem,
36 &usage)) {
37 (*usage_by_type)[MemoryUtil::kTableReadersTotal] += usage;
38 }
39 }
40
41 // Cache
42 for (const auto* cache : cache_set) {
43 if (cache != nullptr) {
44 (*usage_by_type)[MemoryUtil::kCacheTotal] += cache->GetUsage();
45 }
46 }
47
48 return Status::OK();
49}
