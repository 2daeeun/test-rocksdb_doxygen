19// WriteUnprepared transactions needs to be able to read their own uncommitted
20// writes, and supporting this requires some careful consideration. Because
21// writes in the current transaction may be flushed to DB already, we cannot
22// rely on the contents of WriteBatchWithIndex to determine whether a key should
23// be visible or not, so we have to remember to check the DB for any uncommitted
24// keys that should be visible to us. First, we will need to change the seek to
25// snapshot logic, to seek to max_visible_seq = max(snap_seq, max_unprep_seq).
26// Any key greater than max_visible_seq should not be visible because they
27// cannot be unprepared by the current transaction and they are not in its
28// snapshot.
29//
30// When we seek to max_visible_seq, one of these cases will happen:
31// 1. We hit a unprepared key from the current transaction.
32// 2. We hit a unprepared key from the another transaction.
33// 3. We hit a committed key with snap_seq < seq < max_unprep_seq.
34// 4. We hit a committed key with seq <= snap_seq.
35//
36// IsVisibleFullCheck handles all cases correctly.
37//
38// Other notes:
39// Note that max_visible_seq is only calculated once at iterator construction
40// time, meaning if the same transaction is adding more unprep seqs through
41// writes during iteration, these newer writes may not be visible. This is not a
42// problem for MySQL though because it avoids modifying the index as it is
43// scanning through it to avoid the Halloween Problem. Instead, it scans the
44// index once up front, and modifies based on a temporary copy.
45//
46// In DBIter, there is a "reseek" optimization if the iterator skips over too
47// many keys. However, this assumes that the reseek seeks exactly to the
48// required key. In write unprepared, even after seeking directly to
49// max_visible_seq, some iteration may be required before hitting a visible key,
50// and special precautions must be taken to avoid performing another reseek,
51// leading to an infinite loop.
52//
53class WriteUnpreparedTxnReadCallback : public ReadCallback {
54 public:
55 WriteUnpreparedTxnReadCallback(
56 WritePreparedTxnDB* db, SequenceNumber snapshot,
57 SequenceNumber min_uncommitted,
58 const std::map<SequenceNumber, size_t>& unprep_seqs,
59 SnapshotBackup backed_by_snapshot)
60 // Pass our last uncommitted seq as the snapshot to the parent class to
61 // ensure that the parent will not prematurely filter out own writes. We
62 // will do the exact comparison against snapshots in IsVisibleFullCheck
63 // override.
64 : ReadCallback(CalcMaxVisibleSeq(unprep_seqs, snapshot), min_uncommitted),
65 db_(db),
66 unprep_seqs_(unprep_seqs),
67 wup_snapshot_(snapshot),
68 backed_by_snapshot_(backed_by_snapshot) {
69 (void)backed_by_snapshot_; // to silence unused private field warning
70 }
71
72 virtual ~WriteUnpreparedTxnReadCallback() {
73 // If it is not backed by snapshot, the caller must check validity
74 assert(valid_checked_ || backed_by_snapshot_ == kBackedByDBSnapshot);
75 }
76
77 bool IsVisibleFullCheck(SequenceNumber seq) override;
78
79 inline bool valid() {
80 valid_checked_ = true;
81 return snap_released_ == false;
82 }
83
84 void Refresh(SequenceNumber seq) override {
85 max_visible_seq_ = std::max(max_visible_seq_, seq);
86 wup_snapshot_ = seq;
87 }
88
89 static SequenceNumber CalcMaxVisibleSeq(
90 const std::map<SequenceNumber, size_t>& unprep_seqs,
91 SequenceNumber snapshot_seq) {
92 SequenceNumber max_unprepared = 0;
93 if (unprep_seqs.size()) {
94 max_unprepared =
95 unprep_seqs.rbegin()->first + unprep_seqs.rbegin()->second - 1;
96 }
97 return std::max(max_unprepared, snapshot_seq);
98 }
99
100 private:
101 WritePreparedTxnDB* db_;
102 const std::map<SequenceNumber, size_t>& unprep_seqs_;
103 SequenceNumber wup_snapshot_;
104 // Whether max_visible_seq_ is backed by a snapshot
105 const SnapshotBackup backed_by_snapshot_;
106 bool snap_released_ = false;
107 // Safety check to ensure that the caller has checked invalid statuses
108 bool valid_checked_ = false;
109};

278 // We have 3 data structures holding savepoint information:
279 // 1. TransactionBaseImpl::save_points_
280 // 2. WriteUnpreparedTxn::flushed_save_points_
281 // 3. WriteUnpreparecTxn::unflushed_save_points_
282 //
283 // TransactionBaseImpl::save_points_ holds information about all write
284 // batches, including the current in-memory write_batch_, or unprepared
285 // batches that have been written out. Its responsibility is just to track
286 // which keys have been modified in every savepoint.
287 //
288 // WriteUnpreparedTxn::flushed_save_points_ holds information about savepoints
289 // set on unprepared batches that have already flushed. It holds the snapshot
290 // and unprep_seqs at that savepoint, so that the rollback process can
291 // determine which keys were visible at that point in time.
292 //
293 // WriteUnpreparecTxn::unflushed_save_points_ holds information about
294 // savepoints on the current in-memory write_batch_. It simply records the
295 // size of the write batch at every savepoint.
296 //
297 // TODO(lth): Remove the redundancy between save_point_boundaries_ and
298 // write_batch_.save_points_.
299 //
300 // Based on this information, here are some invariants:
301 // size(unflushed_save_points_) = size(write_batch_.save_points_)
302 // size(flushed_save_points_) + size(unflushed_save_points_)
303 // = size(save_points_)
304 //
305 std::unique_ptr<autovector<WriteUnpreparedTxn::SavePoint>>
306 flushed_save_points_;
307 std::unique_ptr<autovector<size_t>> unflushed_save_points_;
308
309 // It is currently unsafe to flush a write batch if there are active iterators
310 // created from this transaction. This is because we use WriteBatchWithIndex
311 // to do merging reads from the DB and the write batch. If we flush the write
312 // batch, it is possible that the delta iterator on the iterator will point to
313 // invalid memory.
314 std::vector<Iterator*> active_iterators_;
315
316 // Untracked keys that we have to rollback.
317 //
318 // TODO(lth): Currently we we do not record untracked keys per-savepoint.
319 // This means that when rolling back to savepoints, we have to check all
320 // keys in the current transaction for rollback. Note that this is only
321 // inefficient, but still correct because we take a snapshot at every
322 // savepoint, and we will use that snapshot to construct the rollback batch.
323 // The rollback batch will then contain a reissue of the same marker.
324 //
325 // A more optimal solution would be to only check keys changed since the
326 // last savepoint. Also, it may make sense to merge this into tracked_keys_
327 // and differentiate between tracked but not locked keys to avoid having two
328 // very similar data structures.
329 using KeySet = std::unordered_map<uint32_t, std::vector<std::string>>;
330 KeySet untracked_keys_;
331};
