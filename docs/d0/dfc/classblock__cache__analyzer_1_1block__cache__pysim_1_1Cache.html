818 def _access_kv(self, trace_record, key, hash, value_size, no_insert):
819 # Sanity checks.
820 assert self.used_size <= self.cache_size
821 if self._lookup(trace_record, key, hash):
822 # A cache hit.
823 return True
824 if no_insert or value_size <= 0:
