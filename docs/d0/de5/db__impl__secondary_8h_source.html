17// A wrapper class to hold log reader, log reporter, log status.
18class LogReaderContainer {
19 public:
20 LogReaderContainer()
21 : reader_(nullptr), reporter_(nullptr), status_(nullptr) {}
22 LogReaderContainer(Env* env, std::shared_ptr<Logger> info_log,
23 std::string fname,
24 std::unique_ptr<SequentialFileReader>&& file_reader,
25 uint64_t log_number) {
26 LogReporter* reporter = new LogReporter();
27 status_ = new Status();
28 reporter->env = env;
29 reporter->info_log = info_log.get();
30 reporter->fname = std::move(fname);
31 reporter->status = status_;
32 reporter_ = reporter;
33 // We intentially make log::Reader do checksumming even if
34 // paranoid_checks==false so that corruptions cause entire commits
35 // to be skipped instead of propagating bad information (like overly
36 // large sequence numbers).
37 reader_ = new log::FragmentBufferedReader(info_log, std::move(file_reader),
38 reporter, true /*checksum*/,
39 log_number);
40 }
41 log::FragmentBufferedReader* reader_;
42 log::Reader::Reporter* reporter_;
43 Status* status_;
44 ~LogReaderContainer() {
45 delete reader_;
46 delete reporter_;
47 delete status_;
48 }
49
50 private:
51 struct LogReporter : public log::Reader::Reporter {
52 Env* env;
53 Logger* info_log;
54 std::string fname;
55 Status* status; // nullptr if immutable_db_options_.paranoid_checks==false
56 void Corruption(size_t bytes, const Status& s) override {
57 ROCKS_LOG_WARN(info_log, "%s%s: dropping %d bytes; %s",
58 (this->status == nullptr ? "(ignoring error) " : ""),
59 fname.c_str(), static_cast<int>(bytes),
60 s.ToString().c_str());
61 if (this->status != nullptr && this->status->ok()) {
62 *this->status = s;
63 }
64 }
65 };
66};
