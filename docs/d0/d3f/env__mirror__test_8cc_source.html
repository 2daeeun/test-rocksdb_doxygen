34TEST_F(EnvMirrorTest, Basics) {

114TEST_F(EnvMirrorTest, ReadWrite) {
115 std::unique_ptr<WritableFile> writable_file;
116 std::unique_ptr<SequentialFile> seq_file;
117 std::unique_ptr<RandomAccessFile> rand_file;
118 Slice result;
119 char scratch[100];
120
121 ASSERT_OK(env_->CreateDir("/dir"));
122
123 ASSERT_OK(env_->NewWritableFile("/dir/f", &writable_file, soptions_));
124 ASSERT_OK(writable_file->Append("hello "));
125 ASSERT_OK(writable_file->Append("world"));
126 writable_file.reset();
127
128 // Read sequentially.
129 ASSERT_OK(env_->NewSequentialFile("/dir/f", &seq_file, soptions_));
130 ASSERT_OK(seq_file->Read(5, &result, scratch)); // Read "hello".
131 ASSERT_EQ(0, result.compare("hello"));
132 ASSERT_OK(seq_file->Skip(1));
133 ASSERT_OK(seq_file->Read(1000, &result, scratch)); // Read "world".
134 ASSERT_EQ(0, result.compare("world"));
135 ASSERT_OK(seq_file->Read(1000, &result, scratch)); // Try reading past EOF.
136 ASSERT_EQ(0U, result.size());
137 ASSERT_OK(seq_file->Skip(100)); // Try to skip past end of file.
138 ASSERT_OK(seq_file->Read(1000, &result, scratch));
139 ASSERT_EQ(0U, result.size());
140
141 // Random reads.
142 ASSERT_OK(env_->NewRandomAccessFile("/dir/f", &rand_file, soptions_));
143 ASSERT_OK(rand_file->Read(6, 5, &result, scratch)); // Read "world".
144 ASSERT_EQ(0, result.compare("world"));
145 ASSERT_OK(rand_file->Read(0, 5, &result, scratch)); // Read "hello".
146 ASSERT_EQ(0, result.compare("hello"));
147 ASSERT_OK(rand_file->Read(10, 100, &result, scratch)); // Read "d".
148 ASSERT_EQ(0, result.compare("d"));
149
150 // Too high offset.
151 ASSERT_TRUE(!rand_file->Read(1000, 5, &result, scratch).ok());
152}

177TEST_F(EnvMirrorTest, LargeWrite) {
178 const size_t kWriteSize = 300 * 1024;
179 char* scratch = new char[kWriteSize * 2];
180
181 std::string write_data;
182 for (size_t i = 0; i < kWriteSize; ++i) {
183 write_data.append(1, static_cast<char>(i));
184 }
185
186 std::unique_ptr<WritableFile> writable_file;
187 ASSERT_OK(env_->NewWritableFile("/dir/f", &writable_file, soptions_));
188 ASSERT_OK(writable_file->Append("foo"));
189 ASSERT_OK(writable_file->Append(write_data));
190 writable_file.reset();
191
192 std::unique_ptr<SequentialFile> seq_file;
193 Slice result;
194 ASSERT_OK(env_->NewSequentialFile("/dir/f", &seq_file, soptions_));
195 ASSERT_OK(seq_file->Read(3, &result, scratch)); // Read "foo".
196 ASSERT_EQ(0, result.compare("foo"));
197
198 size_t read = 0;
199 std::string read_data;
200 while (read < kWriteSize) {
201 ASSERT_OK(seq_file->Read(kWriteSize - read, &result, scratch));
202 read_data.append(result.data(), result.size());
203 read += result.size();
204 }
205 ASSERT_TRUE(write_data == read_data);
206 delete[] scratch;
207}
