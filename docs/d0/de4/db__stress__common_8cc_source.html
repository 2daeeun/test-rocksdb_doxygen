37// Zipfian distribution is generated based on a pre-calculated array.
38// It should be used before start the stress test.
39// First, the probability distribution function (PDF) of this Zipfian follows
40// power low. P(x) = 1/(x^alpha).
41// So we calculate the PDF when x is from 0 to zipf_sum_size in first for loop
42// and add the PDF value togetger as c. So we get the total probability in c.
43// Next, we calculate inverse CDF of Zipfian and store the value of each in
44// an array (sum_probs). The rank is from 0 to zipf_sum_size. For example, for
45// integer k, its Zipfian CDF value is sum_probs[k].
46// Third, when we need to get an integer whose probability follows Zipfian
47// distribution, we use a rand_seed [0,1] which follows uniform distribution
48// as a seed and search it in the sum_probs via binary search. When we find
49// the closest sum_probs[i] of rand_seed, i is the integer that in
50// [0, zipf_sum_size] following Zipfian distribution with parameter alpha.
51// Finally, we can scale i to [0, max_key] scale.
52// In order to avoid that hot keys are close to each other and skew towards 0,
53// we use Rando64 to shuffle it.
54void InitializeHotKeyGenerator(double alpha) {
55 double c = 0;
56 for (int64_t i = 1; i <= zipf_sum_size; i++) {
57 c = c + (1.0 / std::pow(static_cast<double>(i), alpha));
58 }
59 c = 1.0 / c;
60
61 sum_probs[0] = 0;
62 for (int64_t i = 1; i <= zipf_sum_size; i++) {
63 sum_probs[i] =
64 sum_probs[i - 1] + c / std::pow(static_cast<double>(i), alpha);
65 }
66}
67
68// Generate one key that follows the Zipfian distribution. The skewness
69// is decided by the parameter alpha. Input is the rand_seed [0,1] and
70// the max of the key to be generated. If we directly return tmp_zipf_seed,
71// the closer to 0, the higher probability will be. To randomly distribute
72// the hot keys in [0, max_key], we use Random64 to shuffle it.
73int64_t GetOneHotKeyID(double rand_seed, int64_t max_key) {
74 int64_t low = 1, mid, high = zipf_sum_size, zipf = 0;
75 while (low <= high) {
76 mid = (low + high) / 2;
77 if (sum_probs[mid] >= rand_seed && sum_probs[mid - 1] < rand_seed) {
78 zipf = mid;
79 break;
80 } else if (sum_probs[mid] >= rand_seed) {
81 high = mid - 1;
82 } else {
83 low = mid + 1;
84 }
85 }
86 int64_t tmp_zipf_seed = zipf * max_key / zipf_sum_size;
87 Random64 rand_local(tmp_zipf_seed);
88 return rand_local.Next() % max_key;
89}

91void PoolSizeChangeThread(void* v) {
92 assert(FLAGS_compaction_thread_pool_adjust_interval > 0);
93 ThreadState* thread = static_cast<ThreadState*>(v);
94 SharedState* shared = thread->shared;
95
96 while (true) {
97 {
98 MutexLock l(shared->GetMutex());
99 if (shared->ShouldStopBgThread()) {
100 shared->IncBgThreadsFinished();
101 if (shared->BgThreadsFinished()) {
102 shared->GetCondVar()->SignalAll();
103 }
104 return;
105 }
106 }
107
108 auto thread_pool_size_base = FLAGS_max_background_compactions;
109 auto thread_pool_size_var = FLAGS_compaction_thread_pool_variations;
110 int new_thread_pool_size =
111 thread_pool_size_base - thread_pool_size_var +
112 thread->rand.Next() % (thread_pool_size_var * 2 + 1);
113 if (new_thread_pool_size < 1) {
114 new_thread_pool_size = 1;
115 }
116 db_stress_env->SetBackgroundThreads(new_thread_pool_size,
117 ROCKSDB_NAMESPACE::Env::Priority::LOW);
118 // Sleep up to 3 seconds
119 db_stress_env->SleepForMicroseconds(
120 thread->rand.Next() % FLAGS_compaction_thread_pool_adjust_interval *
121 1000 +
122 1);
123 }
124}

152void CompressedCacheSetCapacityThread(void* v) {
153 assert(FLAGS_compressed_secondary_cache_size > 0 ||
154 FLAGS_compressed_secondary_cache_ratio > 0.0);
155 auto* thread = static_cast<ThreadState*>(v);
156 SharedState* shared = thread->shared;
157 while (true) {
158 {
159 MutexLock l(shared->GetMutex());
160 if (shared->ShouldStopBgThread()) {
161 shared->IncBgThreadsFinished();
162 if (shared->BgThreadsFinished()) {
163 shared->GetCondVar()->SignalAll();
164 }
165 return;
166 }
167 }
168 db_stress_env->SleepForMicroseconds(FLAGS_secondary_cache_update_interval);
169 if (FLAGS_compressed_secondary_cache_size > 0) {
170 Status s = compressed_secondary_cache->SetCapacity(0);
171 size_t capacity;
172 if (s.ok()) {
173 s = compressed_secondary_cache->GetCapacity(capacity);
174 assert(capacity == 0);
175 }
176 db_stress_env->SleepForMicroseconds(10 * 1000 * 1000);
177 if (s.ok()) {
178 s = compressed_secondary_cache->SetCapacity(
179 FLAGS_compressed_secondary_cache_size);
180 }
181 if (s.ok()) {
182 s = compressed_secondary_cache->GetCapacity(capacity);
183 assert(capacity == FLAGS_compressed_secondary_cache_size);
184 }
185 if (!s.ok()) {
186 fprintf(stderr, "Compressed cache Set/GetCapacity returned error: %s\n",
187 s.ToString().c_str());
188 }
189 } else if (FLAGS_compressed_secondary_cache_ratio > 0.0) {
190 if (thread->rand.OneIn(2)) { // if (thread->rand.OneIn(2)) {
191 size_t capacity = block_cache->GetCapacity();
192 size_t adjustment;
193 if (FLAGS_use_write_buffer_manager && FLAGS_db_write_buffer_size > 0) {
194 adjustment = (capacity - FLAGS_db_write_buffer_size);
195 } else {
196 adjustment = capacity;
197 }
198 // Lower by upto 50% of usable block cache capacity
199 adjustment = (adjustment * thread->rand.Uniform(50)) / 100;
200 block_cache->SetCapacity(capacity - adjustment);
201 fprintf(stdout, "New cache capacity = %zu\n",
202 block_cache->GetCapacity());
203 db_stress_env->SleepForMicroseconds(10 * 1000 * 1000);
204 block_cache->SetCapacity(capacity);
205 } else {
206 Status s;
207 double new_comp_cache_ratio =
208 (double)thread->rand.Uniform(
209 FLAGS_compressed_secondary_cache_ratio * 100) /
210 100;
211 fprintf(stdout, "New comp cache ratio = %f\n", new_comp_cache_ratio);
212
213 s = UpdateTieredCache(block_cache, /*capacity*/ -1,
214 new_comp_cache_ratio);
215 if (s.ok()) {
216 db_stress_env->SleepForMicroseconds(10 * 1000 * 1000);
217 }
218 if (s.ok()) {
219 s = UpdateTieredCache(block_cache, /*capacity*/ -1,
220 FLAGS_compressed_secondary_cache_ratio);
221 }
222 if (!s.ok()) {
223 fprintf(stderr, "UpdateTieredCache returned error: %s\n",
224 s.ToString().c_str());
225 }
226 }
227 }

248// Note that if hot_key_alpha != 0, it generates the key based on Zipfian
249// distribution. Keys are randomly scattered to [0, FLAGS_max_key]. It does
250// not ensure the order of the keys being generated and the keys does not have
251// the active range which is related to FLAGS_active_width.
252int64_t GenerateOneKey(ThreadState* thread, uint64_t iteration) {
253 const double completed_ratio =
254 static_cast<double>(iteration) / FLAGS_ops_per_thread;
255 const int64_t base_key = static_cast<int64_t>(
256 completed_ratio * (FLAGS_max_key - FLAGS_active_width));
257 int64_t rand_seed = base_key + thread->rand.Next() % FLAGS_active_width;
258 int64_t cur_key = rand_seed;
259 if (FLAGS_hot_key_alpha != 0) {
260 // If set the Zipfian distribution Alpha to non 0, use Zipfian
261 double float_rand =
262 (static_cast<double>(thread->rand.Next() % FLAGS_max_key)) /
263 FLAGS_max_key;
264 cur_key = GetOneHotKeyID(float_rand, FLAGS_max_key);
265 }
266 return cur_key;
267}
268
269// Note that if hot_key_alpha != 0, it generates the key based on Zipfian
270// distribution. Keys being generated are in random order.
271// If user want to generate keys based on uniform distribution, user needs to
272// set hot_key_alpha == 0. It will generate the random keys in increasing
273// order in the key array (ensure key[i] >= key[i+1]) and constrained in a
274// range related to FLAGS_active_width.
275std::vector<int64_t> GenerateNKeys(ThreadState* thread, int num_keys,
276 uint64_t iteration) {
277 const double completed_ratio =
278 static_cast<double>(iteration) / FLAGS_ops_per_thread;
279 const int64_t base_key = static_cast<int64_t>(
280 completed_ratio * (FLAGS_max_key - FLAGS_active_width));
281 std::vector<int64_t> keys;
282 keys.reserve(num_keys);
283 int64_t next_key = base_key + thread->rand.Next() % FLAGS_active_width;
284 keys.push_back(next_key);
285 for (int i = 1; i < num_keys; ++i) {
286 // Generate the key follows zipfian distribution
287 if (FLAGS_hot_key_alpha != 0) {
288 double float_rand =
289 (static_cast<double>(thread->rand.Next() % FLAGS_max_key)) /
290 FLAGS_max_key;
291 next_key = GetOneHotKeyID(float_rand, FLAGS_max_key);
292 } else {
293 // This may result in some duplicate keys
294 next_key = next_key + thread->rand.Next() %
295 (FLAGS_active_width - (next_key - base_key));
296 }
297 keys.push_back(next_key);
298 }
299 return keys;
300}

302size_t GenerateValue(uint32_t rand, char* v, size_t max_sz) {
303 size_t value_sz =
304 ((rand % kRandomValueMaxFactor) + 1) * FLAGS_value_size_mult;
305 assert(value_sz <= max_sz && value_sz >= sizeof(uint32_t));
306 (void)max_sz;
307 PutUnaligned(reinterpret_cast<uint32_t*>(v), rand);
308 for (size_t i = sizeof(uint32_t); i < value_sz; i++) {
309 v[i] = (char)(rand ^ i);
310 }
311 v[value_sz] = '\0';
312 return value_sz; // the size of the value set.
313}

333WideColumns GenerateWideColumns(uint32_t value_base, const Slice& slice) {
334 WideColumns columns;
335
336 constexpr size_t max_columns = 4;
337 const size_t num_columns = (value_base % max_columns) + 1;
338
339 columns.reserve(num_columns);
340
341 assert(slice.size() >= num_columns);
342
343 columns.emplace_back(kDefaultWideColumnName, slice);
344
345 for (size_t i = 1; i < num_columns; ++i) {
346 const Slice name(slice.data(), i);
347 const Slice value(slice.data() + i, slice.size() - i);
348
349 columns.emplace_back(name, value);
350 }
351
352 return columns;
353}

369bool VerifyWideColumns(const Slice& value, const WideColumns& columns) {
370 if (value.size() < sizeof(uint32_t)) {
371 return false;
372 }
373
374 const uint32_t value_base = GetValueBase(value);
375
376 const WideColumns expected_columns =
377 GenerateExpectedWideColumns(value_base, value);
378
379 if (columns != expected_columns) {
380 return false;
381 }
382
383 return true;
384}

386bool VerifyWideColumns(const WideColumns& columns) {
387 if (!WideColumnsHelper::HasDefaultColumn(columns)) {
388 return false;
389 }
390
391 const Slice& value_of_default = WideColumnsHelper::GetDefaultColumn(columns);
392
393 return VerifyWideColumns(value_of_default, columns);
394}

413uint64_t GetWriteUnixTime(ThreadState* thread) {
414 static uint64_t kPreserveSeconds =
415 std::max(FLAGS_preserve_internal_time_seconds,
416 FLAGS_preclude_last_level_data_seconds);
417 static uint64_t kFallbackTime = std::numeric_limits<uint64_t>::max();
418 int64_t write_time = 0;
419 Status s = db_stress_env->GetCurrentTime(&write_time);
420 uint32_t write_time_mode = thread->rand.Uniform(3);
421 if (write_time_mode == 0 || !s.ok()) {
422 return kFallbackTime;
423 } else if (write_time_mode == 1) {
424 uint64_t delta = kPreserveSeconds > 0
425 ? static_cast<uint64_t>(thread->rand.Uniform(
426 static_cast<int>(kPreserveSeconds)))
427 : 0;
428 return static_cast<uint64_t>(write_time) - delta;
429 } else {
430 return static_cast<uint64_t>(write_time) - kPreserveSeconds;
431 }
432}
