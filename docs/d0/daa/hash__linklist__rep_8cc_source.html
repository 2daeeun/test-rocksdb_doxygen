29// A data structure used as the header of a link list of a hash bucket.
30struct BucketHeader {
31 Pointer next;
32 std::atomic<uint32_t> num_entries;
33
34 explicit BucketHeader(void* n, uint32_t count)
35 : next(n), num_entries(count) {}
36
37 bool IsSkipListBucket() {
38 return next.load(std::memory_order_relaxed) == this;
39 }
40
41 uint32_t GetNumEntries() const {
42 return num_entries.load(std::memory_order_relaxed);
43 }
44
45 // REQUIRES: called from single-threaded Insert()
46 void IncNumEntries() {
47 // Only one thread can do write at one time. No need to do atomic
48 // incremental. Update it with relaxed load and store.
49 num_entries.store(GetNumEntries() + 1, std::memory_order_relaxed);
50 }
51};

53// A data structure used as the header of a skip list of a hash bucket.
54struct SkipListBucketHeader {
55 BucketHeader Counting_header;
56 MemtableSkipList skip_list;
57
58 explicit SkipListBucketHeader(const MemTableRep::KeyComparator& cmp,
59 Allocator* allocator, uint32_t count)
60 : Counting_header(this, // Pointing to itself to indicate header type.
61 count),
62 skip_list(cmp, allocator) {}

65struct Node {
66 // Accessors/mutators for links. Wrapped in methods so we can
67 // add the appropriate barriers as necessary.
68 Node* Next() {
69 // Use an 'acquire load' so that we observe a fully initialized
70 // version of the returned Node.
71 return next_.load(std::memory_order_acquire);
72 }
73 void SetNext(Node* x) {
74 // Use a 'release store' so that anybody who reads through this
75 // pointer observes a fully initialized version of the inserted node.
76 next_.store(x, std::memory_order_release);
77 }
78 // No-barrier variants that can be safely used in a few locations.
79 Node* NoBarrier_Next() { return next_.load(std::memory_order_relaxed); }
80
81 void NoBarrier_SetNext(Node* x) { next_.store(x, std::memory_order_relaxed); }
82
83 // Needed for placement new below which is fine
84 Node() = default;
85
86 private:
87 std::atomic<Node*> next_;
88
89 // Prohibit copying due to the below
90 Node(const Node&) = delete;
91 Node& operator=(const Node&) = delete;
92
93 public:
94 char key[1];
95};

97// Memory structure of the mem table:
98// It is a hash table, each bucket points to one entry, a linked list or a
99// skip list. In order to track total number of records in a bucket to determine
100// whether should switch to skip list, a header is added just to indicate
101// number of entries in the bucket.
102//
103//
104// +-----> NULL Case 1. Empty bucket
105// |
106// |
107// | +---> +-------+
108// | | | Next +--> NULL
109// | | +-------+
110// +-----+ | | | | Case 2. One Entry in bucket.
111// | +-+ | | Data | next pointer points to
112// +-----+ | | | NULL. All other cases
113// | | | | | next pointer is not NULL.
114// +-----+ | +-------+
115// | +---+
116// +-----+ +-> +-------+ +> +-------+ +-> +-------+
117// | | | | Next +--+ | Next +--+ | Next +-->NULL
118// +-----+ | +-------+ +-------+ +-------+
119// | +-----+ | Count | | | | |
120// +-----+ +-------+ | Data | | Data |
121// | | | | | |
122// +-----+ Case 3. | | | |
123// | | A header +-------+ +-------+
124// +-----+ points to
125// | | a linked list. Count indicates total number
126// +-----+ of rows in this bucket.
127// | |
128// +-----+ +-> +-------+ <--+
129// | | | | Next +----+
130// +-----+ | +-------+ Case 4. A header points to a skip
131// | +----+ | Count | list and next pointer points to
132// +-----+ +-------+ itself, to distinguish case 3 or 4.
133// | | | | Count still is kept to indicates total
134// +-----+ | Skip +--> of entries in the bucket for debugging
135// | | | List | Data purpose.
136// | | | +-->
137// +-----+ | |
138// | | +-------+
139// +-----+
140//
141// We don't have data race when changing cases because:
142// (1) When changing from case 2->3, we create a new bucket header, put the
143// single node there first without changing the original node, and do a
144// release store when changing the bucket pointer. In that case, a reader
145// who sees a stale value of the bucket pointer will read this node, while
146// a reader sees the correct value because of the release store.
147// (2) When changing case 3->4, a new header is created with skip list points
148// to the data, before doing an acquire store to change the bucket pointer.
149// The old header and nodes are never changed, so any reader sees any
150// of those existing pointers will guarantee to be able to iterate to the
151// end of the linked list.
152// (3) Header's next pointer in case 3 might change, but they are never equal
153// to itself, so no matter a reader sees any stale or newer value, it will
154// be able to correctly distinguish case 3 and 4.
155//
156// The reason that we use case 2 is we want to make the format to be efficient
157// when the utilization of buckets is relatively low. If we use case 3 for
158// single entry bucket, we will need to waste 12 bytes for every entry,
159// which can be significant decrease of memory utilization.

408 class DynamicIterator : public HashLinkListRep::LinkListIterator {
409 public:
410 explicit DynamicIterator(HashLinkListRep& memtable_rep)
411 : HashLinkListRep::LinkListIterator(&memtable_rep, nullptr),
412 memtable_rep_(memtable_rep) {}
413
414 // Advance to the first entry with a key >= target
415 void Seek(const Slice& k, const char* memtable_key) override {
416 auto transformed = memtable_rep_.GetPrefix(k);
417 Pointer& bucket = memtable_rep_.GetBucket(transformed);
418
419 if (memtable_rep_.IsEmptyBucket(bucket)) {
420 skip_list_iter_.reset();
421 Reset(nullptr);
422 } else {
423 Node* first_linked_list_node =
424 memtable_rep_.GetLinkListFirstNode(bucket);
425 if (first_linked_list_node != nullptr) {
426 // The bucket is organized as a linked list
427 skip_list_iter_.reset();
428 Reset(first_linked_list_node);
429 HashLinkListRep::LinkListIterator::Seek(k, memtable_key);
430
431 } else {
432 SkipListBucketHeader* skip_list_header =
433 memtable_rep_.GetSkipListBucketHeader(bucket);
434 assert(skip_list_header != nullptr);
435 // The bucket is organized as a skip list
436 if (!skip_list_iter_) {
437 skip_list_iter_.reset(
438 new MemtableSkipList::Iterator(&skip_list_header->skip_list));
439 } else {
440 skip_list_iter_->SetList(&skip_list_header->skip_list);
441 }
442 if (memtable_key != nullptr) {
443 skip_list_iter_->Seek(memtable_key);
444 } else {
445 IterKey encoded_key;
446 encoded_key.EncodeLengthPrefixedKey(k);
447 skip_list_iter_->Seek(encoded_key.GetUserKey().data());
448 }
449 }
450 }
451 }
452
453 bool Valid() const override {
454 if (skip_list_iter_) {
455 return skip_list_iter_->Valid();
456 }
457 return HashLinkListRep::LinkListIterator::Valid();
458 }
459
460 const char* key() const override {
461 if (skip_list_iter_) {
462 return skip_list_iter_->key();
463 }
464 return HashLinkListRep::LinkListIterator::key();
465 }
466
467 void Next() override {
468 if (skip_list_iter_) {
469 skip_list_iter_->Next();
470 } else {
471 HashLinkListRep::LinkListIterator::Next();
472 }
473 }
474
475 private:
476 // the underlying memtable
477 const HashLinkListRep& memtable_rep_;
478 std::unique_ptr<MemtableSkipList::Iterator> skip_list_iter_;
479 };

504HashLinkListRep::HashLinkListRep(
505 const MemTableRep::KeyComparator& compare, Allocator* allocator,
506 const SliceTransform* transform, size_t bucket_size,
507 uint32_t threshold_use_skiplist, size_t huge_page_tlb_size, Logger* logger,
508 int bucket_entries_logging_threshold, bool if_log_bucket_dist_when_flash)
509 : MemTableRep(allocator),

514 transform_(transform),

518 if_log_bucket_dist_when_flash_(if_log_bucket_dist_when_flash) {
519 char* mem = allocator_->AllocateAligned(sizeof(Pointer) * bucket_size,
520 huge_page_tlb_size, logger);
521
522 buckets_ = new (mem) Pointer[bucket_size];

524 for (size_t i = 0; i < bucket_size_; ++i) {
525 buckets_[i].store(nullptr, std::memory_order_relaxed);

538SkipListBucketHeader* HashLinkListRep::GetSkipListBucketHeader(
539 Pointer& bucket_pointer) const {
540 Pointer* first_next_pointer =
541 static_cast<Pointer*>(bucket_pointer.load(std::memory_order_acquire));
542 assert(first_next_pointer != nullptr);
543 assert(first_next_pointer->load(std::memory_order_relaxed) != nullptr);
544
545 // Counting header
546 BucketHeader* header = reinterpret_cast<BucketHeader*>(first_next_pointer);
547 assert(header->IsSkipListBucket());
548 assert(header->GetNumEntries() > threshold_use_skiplist_);
549 auto* skip_list_bucket_header =
550 reinterpret_cast<SkipListBucketHeader*>(header);
551 assert(skip_list_bucket_header->Counting_header.next.load(
552 std::memory_order_relaxed) == header);
553 return skip_list_bucket_header;
554}
555
556Node* HashLinkListRep::GetLinkListFirstNode(Pointer& bucket_pointer) const {
557 Pointer* first_next_pointer =
558 static_cast<Pointer*>(bucket_pointer.load(std::memory_order_acquire));
559 assert(first_next_pointer != nullptr);
560 if (first_next_pointer->load(std::memory_order_relaxed) == nullptr) {
561 // Single entry bucket
562 return reinterpret_cast<Node*>(first_next_pointer);
563 }
564
565 // It is possible that after we fetch first_next_pointer it is modified
566 // and the next is not null anymore. In this case, the bucket should have been
567 // modified to a counting header, so we should reload the first_next_pointer
568 // to make sure we see the update.
569 first_next_pointer =
570 static_cast<Pointer*>(bucket_pointer.load(std::memory_order_acquire));
571 // Counting header
572 BucketHeader* header = reinterpret_cast<BucketHeader*>(first_next_pointer);
573 if (!header->IsSkipListBucket()) {
574 assert(header->GetNumEntries() <= threshold_use_skiplist_);
575 return reinterpret_cast<Node*>(
576 header->next.load(std::memory_order_acquire));
577 }
578 assert(header->GetNumEntries() > threshold_use_skiplist_);
579 return nullptr;
580}

582void HashLinkListRep::Insert(KeyHandle handle) {
583 Node* x = static_cast<Node*>(handle);
584 assert(!Contains(x->key));
585 Slice internal_key = GetLengthPrefixedSlice(x->key);
586 auto transformed = GetPrefix(internal_key);
587 auto& bucket = buckets_[GetHash(transformed)];
588 Pointer* first_next_pointer =
589 static_cast<Pointer*>(bucket.load(std::memory_order_relaxed));
590
591 if (first_next_pointer == nullptr) {
592 // Case 1. empty bucket
593 // NoBarrier_SetNext() suffices since we will add a barrier when
594 // we publish a pointer to "x" in prev[i].
595 x->NoBarrier_SetNext(nullptr);
596 bucket.store(x, std::memory_order_release);
597 return;
598 }
599
600 BucketHeader* header = nullptr;
601 if (first_next_pointer->load(std::memory_order_relaxed) == nullptr) {
602 // Case 2. only one entry in the bucket
603 // Need to convert to a Counting bucket and turn to case 4.
604 Node* first = reinterpret_cast<Node*>(first_next_pointer);
605 // Need to add a bucket header.
606 // We have to first convert it to a bucket with header before inserting
607 // the new node. Otherwise, we might need to change next pointer of first.
608 // In that case, a reader might sees the next pointer is NULL and wrongly
609 // think the node is a bucket header.
610 auto* mem = allocator_->AllocateAligned(sizeof(BucketHeader));
611 header = new (mem) BucketHeader(first, 1);
612 bucket.store(header, std::memory_order_release);
613 } else {
614 header = reinterpret_cast<BucketHeader*>(first_next_pointer);
615 if (header->IsSkipListBucket()) {
616 // Case 4. Bucket is already a skip list
617 assert(header->GetNumEntries() > threshold_use_skiplist_);
618 auto* skip_list_bucket_header =
619 reinterpret_cast<SkipListBucketHeader*>(header);
620 // Only one thread can execute Insert() at one time. No need to do atomic
621 // incremental.
622 skip_list_bucket_header->Counting_header.IncNumEntries();
623 skip_list_bucket_header->skip_list.Insert(x->key);
624 return;
625 }
626 }
627
628 if (bucket_entries_logging_threshold_ > 0 &&
629 header->GetNumEntries() ==
630 static_cast<uint32_t>(bucket_entries_logging_threshold_)) {
631 Info(logger_,
632 "HashLinkedList bucket %" ROCKSDB_PRIszt
633 " has more than %d "
634 "entries. Key to insert: %s",
635 GetHash(transformed), header->GetNumEntries(),
636 GetLengthPrefixedSlice(x->key).ToString(true).c_str());
637 }
638
639 if (header->GetNumEntries() == threshold_use_skiplist_) {
640 // Case 3. number of entries reaches the threshold so need to convert to
641 // skip list.
642 LinkListIterator bucket_iter(
643 this, reinterpret_cast<Node*>(
644 first_next_pointer->load(std::memory_order_relaxed)));
645 auto mem = allocator_->AllocateAligned(sizeof(SkipListBucketHeader));
646 SkipListBucketHeader* new_skip_list_header = new (mem)
647 SkipListBucketHeader(compare_, allocator_, header->GetNumEntries() + 1);
648 auto& skip_list = new_skip_list_header->skip_list;
649
650 // Add all current entries to the skip list
651 for (bucket_iter.SeekToHead(); bucket_iter.Valid(); bucket_iter.Next()) {
652 skip_list.Insert(bucket_iter.key());
653 }
654
655 // insert the new entry
656 skip_list.Insert(x->key);
657 // Set the bucket
658 bucket.store(new_skip_list_header, std::memory_order_release);
659 } else {
660 // Case 5. Need to insert to the sorted linked list without changing the
661 // header.
662 Node* first =
663 reinterpret_cast<Node*>(header->next.load(std::memory_order_relaxed));
664 assert(first != nullptr);
665 // Advance counter unless the bucket needs to be advanced to skip list.
666 // In that case, we need to make sure the previous count never exceeds
667 // threshold_use_skiplist_ to avoid readers to cast to wrong format.
668 header->IncNumEntries();
669
670 Node* cur = first;
671 Node* prev = nullptr;
672 while (true) {
673 if (cur == nullptr) {
674 break;
675 }
676 Node* next = cur->Next();
677 // Make sure the lists are sorted.
678 // If x points to head_ or next points nullptr, it is trivially satisfied.
679 assert((cur == first) || (next == nullptr) ||
680 KeyIsAfterNode(next->key, cur));
681 if (KeyIsAfterNode(internal_key, cur)) {
682 // Keep searching in this list
683 prev = cur;
684 cur = next;
685 } else {
686 break;
687 }
688 }
689
690 // Our data structure does not allow duplicate insertion
691 assert(cur == nullptr || !Equal(x->key, cur->key));
692
693 // NoBarrier_SetNext() suffices since we will add a barrier when
694 // we publish a pointer to "x" in prev[i].
695 x->NoBarrier_SetNext(cur);

705bool HashLinkListRep::Contains(const char* key) const {
706 Slice internal_key = GetLengthPrefixedSlice(key);
707
708 auto transformed = GetPrefix(internal_key);
709 Pointer& bucket = GetBucket(transformed);
710 if (IsEmptyBucket(bucket)) {
711 return false;
712 }
713
714 Node* linked_list_node = GetLinkListFirstNode(bucket);
715 if (linked_list_node != nullptr) {
716 return LinkListContains(linked_list_node, internal_key);
717 }
718
719 SkipListBucketHeader* skip_list_header = GetSkipListBucketHeader(bucket);
720 if (skip_list_header != nullptr) {
721 return skip_list_header->skip_list.Contains(key);
722 }
723 return false;

760MemTableRep::Iterator* HashLinkListRep::GetIterator(Arena* alloc_arena) {
761 // allocate a new arena of similar size to the one currently in use
762 Arena* new_arena = new Arena(allocator_->BlockSize());
763 auto list = new MemtableSkipList(compare_, new_arena);
764 HistogramImpl keys_per_bucket_hist;
765
766 for (size_t i = 0; i < bucket_size_; ++i) {
767 int count = 0;
768 Pointer& bucket = GetBucket(i);
769 if (!IsEmptyBucket(bucket)) {
770 auto* link_list_head = GetLinkListFirstNode(bucket);
771 if (link_list_head != nullptr) {
772 LinkListIterator itr(this, link_list_head);
773 for (itr.SeekToHead(); itr.Valid(); itr.Next()) {
774 list->Insert(itr.key());
775 count++;
776 }
777 } else {
778 auto* skip_list_header = GetSkipListBucketHeader(bucket);
779 assert(skip_list_header != nullptr);
780 // Is a skip list
781 MemtableSkipList::Iterator itr(&skip_list_header->skip_list);
782 for (itr.SeekToFirst(); itr.Valid(); itr.Next()) {
783 list->Insert(itr.key());
784 count++;
785 }
786 }
787 }
788 if (if_log_bucket_dist_when_flash_) {
789 keys_per_bucket_hist.Add(count);
790 }
791 }
792 if (if_log_bucket_dist_when_flash_ && logger_ != nullptr) {
793 Info(logger_, "hashLinkedList Entry distribution among buckets: %s",
794 keys_per_bucket_hist.ToString().c_str());
795 }
796
797 if (alloc_arena == nullptr) {
798 return new FullListIterator(list, new_arena);
799 } else {
800 auto mem = alloc_arena->AllocateAligned(sizeof(FullListIterator));
801 return new (mem) FullListIterator(list, new_arena);
802 }
803}

821Node* HashLinkListRep::FindGreaterOrEqualInBucket(Node* head,
822 const Slice& key) const {
823 Node* x = head;
824 while (true) {
825 if (x == nullptr) {
826 return x;
827 }
828 Node* next = x->Next();
829 // Make sure the lists are sorted.
830 // If x points to head_ or next points nullptr, it is trivially satisfied.
831 assert((x == head) || (next == nullptr) || KeyIsAfterNode(next->key, x));
832 if (KeyIsAfterNode(key, x)) {
833 // Keep searching in this list
834 x = next;
835 } else {
836 break;
837 }
838 }
839 return x;
840}
