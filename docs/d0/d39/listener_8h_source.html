113enum class CompactionReason : int {
114 kUnknown = 0,
115 // [Level] number of L0 files > level0_file_num_compaction_trigger
116 kLevelL0FilesNum,
117 // [Level] total size of level > MaxBytesForLevel()
118 kLevelMaxLevelSize,
119 // [Universal] Compacting for size amplification
120 kUniversalSizeAmplification,
121 // [Universal] Compacting for size ratio
122 kUniversalSizeRatio,
123 // [Universal] number of sorted runs > level0_file_num_compaction_trigger
124 kUniversalSortedRunNum,
125 // [FIFO] total size > max_table_files_size
126 kFIFOMaxSize,
127 // [FIFO] reduce number of files.
128 kFIFOReduceNumFiles,
129 // [FIFO] files with creation time < (current_time - interval)
130 kFIFOTtl,
131 // Manual compaction
132 kManualCompaction,
133 // DB::SuggestCompactRange() marked files for compaction
134 kFilesMarkedForCompaction,
135 // [Level] Automatic compaction within bottommost level to cleanup duplicate
136 // versions of same user key, usually due to a released snapshot.
137 kBottommostFiles,
138 // Compaction based on TTL
139 kTtl,
140 // According to the comments in flush_job.cc, RocksDB treats flush as
141 // a level 0 compaction in internal stats.
142 kFlush,
143 // [InternalOnly] External sst file ingestion treated as a compaction
144 // with placeholder input level L0 as file ingestion
145 // technically does not have an input level like other compactions.
146 // Used only for internal stats and conflict checking with other compactions
147 kExternalSstIngestion,
148 // Compaction due to SST file being too old
149 kPeriodicCompaction,
150 // Compaction in order to move files to temperature
151 kChangeTemperature,
152 // Compaction scheduled to force garbage collection of blob files
153 kForcedBlobGC,
154 // A special TTL compaction for RoundRobin policy, which basically the same as
155 // kLevelMaxLevelSize, but the goal is to compact TTLed files.
156 kRoundRobinTtl,
157 // [InternalOnly] DBImpl::ReFitLevel treated as a compaction,
158 // Used only for internal conflict checking with other compactions
159 kRefitLevel,
160 // total number of compaction reasons, new reasons must be added above this.
161 kNumOfReasons,
162};

549// EventListener class contains a set of callback functions that will
550// be called when specific RocksDB event happens such as flush. It can
551// be used as a building block for developing custom features such as
552// stats-collector or external compaction algorithm.
553//
554// IMPORTANT
555// Because compaction is needed to resolve a "writes stopped" condition,
556// calling or waiting for any blocking DB write function (no_slowdown=false)
557// from a compaction-related listener callback can hang RocksDB. For DB
558// writes from a callback we recommend a WriteBatch and no_slowdown=true,
559// because the WriteBatch can accumulate writes for later in case DB::Write
560// returns Status::Incomplete. Similarly, calling CompactRange or similar
561// could hang by waiting for a background worker that is occupied until the
562// callback returns.
563//
564// Otherwise, callback functions should not run for an extended period of
565// time before the function returns, because this will slow RocksDB.
566//
567// [Threading] All EventListener callback will be called using the
568// actual thread that involves in that specific event. For example, it
569// is the RocksDB background flush thread that does the actual flush to
570// call EventListener::OnFlushCompleted().
571//
572// [Locking] All EventListener callbacks are designed to be called without
573// the current thread holding any DB mutex. This is to prevent potential
574// deadlock and performance issue when using EventListener callback
575// in a complex way.
576//
577// [Exceptions] Exceptions MUST NOT propagate out of overridden functions into
578// RocksDB, because RocksDB is not exception-safe. This could cause undefined
579// behavior including data loss, unreported corruption, deadlocks, and more.
580class EventListener : public Customizable {
581 public:
582 static const char* Type() { return "EventListener"; }
583 static Status CreateFromString(const ConfigOptions& options,
584 const std::string& id,
585 std::shared_ptr<EventListener>* result);
586 const char* Name() const override {
587 // Since EventListeners did not have a name previously, we will assume
588 // an empty name. Instances should override this method.
589 return "";
590 }
591 // A callback function to RocksDB which will be called whenever a
592 // registered RocksDB flushes a file. The default implementation is
593 // no-op.
594 //
595 // Note that the this function must be implemented in a way such that
596 // it should not run for an extended period of time before the function
597 // returns. Otherwise, RocksDB may be blocked.
598 virtual void OnFlushCompleted(DB* /*db*/,
599 const FlushJobInfo& /*flush_job_info*/) {}
600
601 // A callback function to RocksDB which will be called before a
602 // RocksDB starts to flush memtables. The default implementation is
603 // no-op.
604 //
605 // Note that the this function must be implemented in a way such that
606 // it should not run for an extended period of time before the function
607 // returns. Otherwise, RocksDB may be blocked.
608 virtual void OnFlushBegin(DB* /*db*/,
609 const FlushJobInfo& /*flush_job_info*/) {}
610
611 // A callback function to RocksDB which will be called after a manual flush
612 // is scheduled. The default implementation is no-op.
613 // The size of the `manual_flush_info` vector should only be bigger than 1 if
614 // the DB enables atomic flush and has more than 1 column families. Its size
615 // should be 1 in all other cases.
616 virtual void OnManualFlushScheduled(
617 DB* /*db*/, const std::vector<ManualFlushInfo>& /*manual_flush_info*/) {}
618
619 // A callback function for RocksDB which will be called whenever
620 // a SST file is deleted. Different from OnCompactionCompleted and
621 // OnFlushCompleted, this callback is designed for external logging
622 // service and thus only provide string parameters instead
623 // of a pointer to DB. Applications that build logic basic based
624 // on file creations and deletions is suggested to implement
625 // OnFlushCompleted and OnCompactionCompleted.
626 //
627 // Note that if applications would like to use the passed reference
628 // outside this function call, they should make copies from the
629 // returned value.
630 virtual void OnTableFileDeleted(const TableFileDeletionInfo& /*info*/) {}
631
632 // A callback function to RocksDB which will be called before a
633 // RocksDB starts to compact. The default implementation is
634 // no-op.
635 //
636 // Note that the this function must be implemented in a way such that
637 // it should not run for an extended period of time before the function
638 // returns. Otherwise, RocksDB may be blocked.
639 virtual void OnCompactionBegin(DB* /*db*/, const CompactionJobInfo& /*ci*/) {}
640
641 // A callback function for RocksDB which will be called whenever
642 // a registered RocksDB compacts a file. The default implementation
643 // is a no-op.
644 //
645 // Note that this function must be implemented in a way such that
646 // it should not run for an extended period of time before the function
647 // returns. Otherwise, RocksDB may be blocked.
648 //
649 // @param db a pointer to the rocksdb instance which just compacted
650 // a file.
651 // @param ci a reference to a CompactionJobInfo struct. 'ci' is released
652 // after this function is returned, and must be copied if it is needed
653 // outside of this function.
654 virtual void OnCompactionCompleted(DB* /*db*/,
655 const CompactionJobInfo& /*ci*/) {}
656
657 // A callback function to RocksDB which will be called before a sub-compaction
658 // begins. If a compaction is split to 2 sub-compactions, it will trigger one
659 // `OnCompactionBegin()` first, then two `OnSubcompactionBegin()`.
660 // If compaction is not split, it will still trigger one
661 // `OnSubcompactionBegin()`, as internally, compaction is always handled by
662 // sub-compaction. The default implementation is a no-op.
663 //
664 // Note that this function must be implemented in a way such that
665 // it should not run for an extended period of time before the function
666 // returns. Otherwise, RocksDB may be blocked.
667 //
668 // @param ci a reference to a CompactionJobInfo struct, it contains a
669 // `sub_job_id` which is only unique within the specified compaction (which
670 // can be identified by `job_id`). 'ci' is released after this function is
671 // returned, and must be copied if it's needed outside this function.
672 // Note: `table_properties` is not set for sub-compaction, the information
673 // could be got from `OnCompactionBegin()`.
674 virtual void OnSubcompactionBegin(const SubcompactionJobInfo& /*si*/) {}
675
676 // A callback function to RocksDB which will be called whenever a
677 // sub-compaction completed. The same as `OnSubcompactionBegin()`, if a
678 // compaction is split to 2 sub-compactions, it will be triggered twice. If
