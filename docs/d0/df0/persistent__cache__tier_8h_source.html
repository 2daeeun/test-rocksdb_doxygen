21// Persistent Cache
22//
23// Persistent cache is tiered key-value cache that can use persistent medium. It
24// is a generic design and can leverage any storage medium -- disk/SSD/NVM/RAM.
25// The code has been kept generic but significant benchmark/design/development
26// time has been spent to make sure the cache performs appropriately for
27// respective storage medium.
28// The file defines
29// PersistentCacheTier : Implementation that handles individual cache tier
30// PersistentTieresCache : Implementation that handles all tiers as a logical
31// unit
32//
33// PersistentTieredCache architecture:
34// +--------------------------+ PersistentCacheTier that handles multiple tiers
35// | +----------------+ |
36// | | RAM | PersistentCacheTier that handles RAM (VolatileCacheImpl)
37// | +----------------+ |
38// | | next |
39// | v |
40// | +----------------+ |
41// | | NVM | PersistentCacheTier implementation that handles NVM
42// | +----------------+ (BlockCacheImpl)
43// | | next |
44// | V |
45// | +----------------+ |
46// | | LE-SSD | PersistentCacheTier implementation that handles LE-SSD
47// | +----------------+ (BlockCacheImpl)
48// | | |
49// | V |
50// | null |
51// +--------------------------+
52// |
53// V
54// null

57// Persistent Cache Config
58//
59// This struct captures all the options that are used to configure persistent
60// cache. Some of the terminologies used in naming the options are
61//
62// dispatch size :
63// This is the size in which IO is dispatched to the device
64//
65// write buffer size :
66// This is the size of an individual write buffer size. Write buffers are
67// grouped to form buffered file.
68//
69// cache size :
70// This is the logical maximum for the cache size
71//
72// qdepth :
73// This is the max number of IOs that can issues to the device in parallel
74//
75// pepeling :
76// The writer code path follows pipelined architecture, which means the
77// operations are handed off from one stage to another
78//
79// pipelining backlog size :
80// With the pipelined architecture, there can always be backlogging of ops in
81// pipeline queues. This is the maximum backlog size after which ops are dropped
82// from queue
83struct PersistentCacheConfig {
84 explicit PersistentCacheConfig(
85 Env* const _env, const std::string& _path, const uint64_t _cache_size,
86 const std::shared_ptr<Logger>& _log,
87 const uint32_t _write_buffer_size = 1 * 1024 * 1024 /*1MB*/) {
88 env = _env;
89 clock = (env != nullptr) ? env->GetSystemClock().get()
90 : SystemClock::Default().get();
91 path = _path;
92 log = _log;
93 cache_size = _cache_size;
94 writer_dispatch_size = write_buffer_size = _write_buffer_size;
95 }
96
97 //
98 // Validate the settings. Our intentions are to catch erroneous settings ahead
99 // of time instead going violating invariants or causing dead locks.
100 //
101 Status ValidateSettings() const {
102 // (1) check pre-conditions for variables
103 if (!env || path.empty()) {
104 return Status::InvalidArgument("empty or null args");
105 }
106
107 // (2) assert size related invariants
108 // - cache size cannot be less than cache file size
109 // - individual write buffer size cannot be greater than cache file size
110 // - total write buffer size cannot be less than 2X cache file size
111 if (cache_size < cache_file_size || write_buffer_size >= cache_file_size ||
112 write_buffer_size * write_buffer_count() < 2 * cache_file_size) {
113 return Status::InvalidArgument("invalid cache size");
114 }
115
116 // (2) check writer settings
117 // - Queue depth cannot be 0
118 // - writer_dispatch_size cannot be greater than writer_buffer_size
119 // - dispatch size and buffer size need to be aligned
120 if (!writer_qdepth || writer_dispatch_size > write_buffer_size ||
121 write_buffer_size % writer_dispatch_size) {
122 return Status::InvalidArgument("invalid writer settings");
123 }
124
125 return Status::OK();
126 }
127
128 //
129 // Env abstraction to use for system level operations
130 //
131 Env* env;
132 SystemClock* clock;
133 //
134 // Path for the block cache where blocks are persisted
135 //
136 std::string path;
137
138 //
139 // Log handle for logging messages
140 //
141 std::shared_ptr<Logger> log;
142
143 //
144 // Enable direct IO for reading
145 //
146 bool enable_direct_reads = true;
147
148 //
149 // Enable direct IO for writing
150 //
151 bool enable_direct_writes = false;
152
153 //
154 // Logical cache size
155 //
156 uint64_t cache_size = std::numeric_limits<uint64_t>::max();
157
158 // cache-file-size
159 //
160 // Cache consists of multiples of small files. This parameter defines the
161 // size of an individual cache file
162 //
163 // default: 1M
164 uint32_t cache_file_size = 100ULL * 1024 * 1024;
165
166 // writer-qdepth
167 //
168 // The writers can issues IO to the devices in parallel. This parameter
169 // controls the max number if IOs that can issues in parallel to the block
170 // device
171 //
172 // default :1
173 uint32_t writer_qdepth = 1;
174
175 // pipeline-writes
176 //
177 // The write optionally follow pipelined architecture. This helps
178 // avoid regression in the eviction code path of the primary tier. This
179 // parameter defines if pipelining is enabled or disabled
180 //
181 // default: true
182 bool pipeline_writes = true;
183
184 // max-write-pipeline-backlog-size
185 //
186 // Max pipeline buffer size. This is the maximum backlog we can accumulate
187 // while waiting for writes. After the limit, new ops will be dropped.
188 //
189 // Default: 1GiB
190 uint64_t max_write_pipeline_backlog_size = 1ULL * 1024 * 1024 * 1024;
191
192 // write-buffer-size
193 //
194 // This is the size in which buffer slabs are allocated.
195 //
196 // Default: 1M
197 uint32_t write_buffer_size = 1ULL * 1024 * 1024;
198
199 // write-buffer-count
200 //
201 // This is the total number of buffer slabs. This is calculated as a factor of
202 // file size in order to avoid dead lock.
203 size_t write_buffer_count() const {
204 assert(write_buffer_size);
205 return static_cast<size_t>((writer_qdepth + 1.2) * cache_file_size /
206 write_buffer_size);
207 }
208
209 // writer-dispatch-size
210 //
211 // The writer thread will dispatch the IO at the specified IO size
212 //
213 // default: 1M
214 uint64_t writer_dispatch_size = 1ULL * 1024 * 1024;
215
216 // is_compressed
217 //
218 // This option determines if the cache will run in compressed mode or
219 // uncompressed mode
220 bool is_compressed = true;
221
222 PersistentCacheConfig MakePersistentCacheConfig(
223 const std::string& path, const uint64_t size,
224 const std::shared_ptr<Logger>& log);
225
226 std::string ToString() const;
227};

229// Persistent Cache Tier
230//
231// This a logical abstraction that defines a tier of the persistent cache. Tiers
232// can be stacked over one another. PersistentCahe provides the basic definition
233// for accessing/storing in the cache. PersistentCacheTier extends the interface
234// to enable management and stacking of tiers.
235class PersistentCacheTier : public PersistentCache {
236 public:
237 using Tier = std::shared_ptr<PersistentCacheTier>;
238
239 virtual ~PersistentCacheTier() {}
240
241 // Open the persistent cache tier
242 virtual Status Open();
243
244 // Close the persistent cache tier
245 virtual Status Close();
246
247 // Reserve space up to 'size' bytes
248 virtual bool Reserve(const size_t size);
249
250 // Erase a key from the cache
251 virtual bool Erase(const Slice& key);
252
253 // Print stats to string recursively
254 virtual std::string PrintStats();
255
256 PersistentCache::StatsType Stats() override;
257
258 // Insert to page cache
259 Status Insert(const Slice& page_key, const char* data,
260 const size_t size) override = 0;
261
262 // Lookup page cache by page identifier
263 Status Lookup(const Slice& page_key, std::unique_ptr<char[]>* data,
264 size_t* size) override = 0;
265
266 // Does it store compressed data ?
267 bool IsCompressed() override = 0;
268
269 std::string GetPrintableOptions() const override = 0;
270
271 uint64_t NewId() override;
272
273 // Return a reference to next tier
274 virtual Tier& next_tier() { return next_tier_; }
275
276 // Set the value for next tier
277 virtual void set_next_tier(const Tier& tier) {
278 assert(!next_tier_);
279 next_tier_ = tier;
280 }
281
282 virtual void TEST_Flush() {
283 if (next_tier_) {
284 next_tier_->TEST_Flush();
285 }
286 }
287
288 private:
289 Tier next_tier_; // next tier
290 std::atomic<uint64_t> last_id_{1};
291};

293// PersistentTieredCache
294//
295// Abstraction that helps you construct a tiers of persistent caches as a
296// unified cache. The tier(s) of cache will act a single tier for management
297// ease and support PersistentCache methods for accessing data.
298class PersistentTieredCache : public PersistentCacheTier {
299 public:
300 virtual ~PersistentTieredCache();
301
302 Status Open() override;
303 Status Close() override;
304 bool Erase(const Slice& key) override;
305 std::string PrintStats() override;
306 PersistentCache::StatsType Stats() override;
307 Status Insert(const Slice& page_key, const char* data,
308 const size_t size) override;
309 Status Lookup(const Slice& page_key, std::unique_ptr<char[]>* data,
310 size_t* size) override;
311 bool IsCompressed() override;
312
313 std::string GetPrintableOptions() const override {
314 return "PersistentTieredCache";
315 }
316
317 void AddTier(const Tier& tier);
318
319 Tier& next_tier() override {
320 auto it = tiers_.end();
321 return (*it)->next_tier();
322 }
323
324 void set_next_tier(const Tier& tier) override {
325 auto it = tiers_.end();
326 (*it)->set_next_tier(tier);
327 }
328
329 void TEST_Flush() override {
330 assert(!tiers_.empty());
331 tiers_.front()->TEST_Flush();
332 PersistentCacheTier::TEST_Flush();
333 }
334
335 protected:
336 std::list<Tier> tiers_; // list of tiers top-down
337};
