425static bool ParseOptionHelper(void* opt_address, const OptionType& opt_type,
426 const std::string& value) {
427 switch (opt_type) {
428 case OptionType::kBoolean:
429 *static_cast<bool*>(opt_address) = ParseBoolean("", value);
430 break;
431 case OptionType::kInt:
432 *static_cast<int*>(opt_address) = ParseInt(value);
433 break;
434 case OptionType::kInt32T:
435 *static_cast<int32_t*>(opt_address) = ParseInt32(value);
436 break;
437 case OptionType::kInt64T:
438 PutUnaligned(static_cast<int64_t*>(opt_address), ParseInt64(value));
439 break;
440 case OptionType::kUInt:
441 *static_cast<unsigned int*>(opt_address) = ParseUint32(value);
442 break;
443 case OptionType::kUInt8T:
444 *static_cast<uint8_t*>(opt_address) = ParseUint8(value);
445 break;
446 case OptionType::kUInt32T:
447 *static_cast<uint32_t*>(opt_address) = ParseUint32(value);
448 break;
449 case OptionType::kUInt64T:
450 PutUnaligned(static_cast<uint64_t*>(opt_address), ParseUint64(value));
451 break;
452 case OptionType::kSizeT:
453 PutUnaligned(static_cast<size_t*>(opt_address), ParseSizeT(value));
454 break;
455 case OptionType::kAtomicInt:
456 static_cast<std::atomic<int>*>(opt_address)
457 ->store(ParseInt(value), std::memory_order_release);
458 break;
459 case OptionType::kString:
460 *static_cast<std::string*>(opt_address) = value;
461 break;
462 case OptionType::kDouble:
463 *static_cast<double*>(opt_address) = ParseDouble(value);
464 break;
465 case OptionType::kCompactionStyle:
466 return ParseEnum<CompactionStyle>(
467 compaction_style_string_map, value,
468 static_cast<CompactionStyle*>(opt_address));
469 case OptionType::kCompactionPri:
470 return ParseEnum<CompactionPri>(compaction_pri_string_map, value,
471 static_cast<CompactionPri*>(opt_address));
472 case OptionType::kCompressionType:
473 return ParseEnum<CompressionType>(

610Status StringToMap(const std::string& opts_str,
611 std::unordered_map<std::string, std::string>* opts_map) {
612 assert(opts_map);
613 // Example:
614 // opts_str = "write_buffer_size=1024;max_write_buffer_number=2;"
615 // "nested_opt={opt1=1;opt2=2};max_bytes_for_level_base=100"
616 size_t pos = 0;
617 std::string opts = trim(opts_str);
618 // If the input string starts and ends with "{...}", strip off the brackets
619 while (opts.size() > 2 && opts[0] == '{' && opts[opts.size() - 1] == '}') {
620 opts = trim(opts.substr(1, opts.size() - 2));
621 }
622
623 while (pos < opts.size()) {
624 size_t eq_pos = opts.find_first_of("={};", pos);
625 if (eq_pos == std::string::npos) {
626 return Status::InvalidArgument("Mismatched key value pair, '=' expected");
627 } else if (opts[eq_pos] != '=') {
628 return Status::InvalidArgument("Unexpected char in key");
629 }
630
631 std::string key = trim(opts.substr(pos, eq_pos - pos));
632 if (key.empty()) {
633 return Status::InvalidArgument("Empty key found");
634 }
635
636 std::string value;
637 Status s = OptionTypeInfo::NextToken(opts, ';', eq_pos + 1, &pos, &value);
638 if (!s.ok()) {
639 return s;
640 } else {
641 (*opts_map)[key] = value;
642 if (pos == std::string::npos) {
643 break;
644 } else {
645 pos++;
646 }
647 }
648 }
649
650 return Status::OK();
651}

849Status OptionTypeInfo::NextToken(const std::string& opts, char delimiter,
850 size_t pos, size_t* end, std::string* token) {
851 while (pos < opts.size() && isspace(opts[pos])) {
852 ++pos;
853 }
854 // Empty value at the end
855 if (pos >= opts.size()) {
856 *token = "";
857 *end = std::string::npos;
858 return Status::OK();
859 } else if (opts[pos] == '{') {
860 int count = 1;
861 size_t brace_pos = pos + 1;
862 while (brace_pos < opts.size()) {
863 if (opts[brace_pos] == '{') {
864 ++count;
865 } else if (opts[brace_pos] == '}') {
866 --count;
867 if (count == 0) {
868 break;
869 }
870 }
871 ++brace_pos;
872 }
873 // found the matching closing brace
874 if (count == 0) {
875 *token = trim(opts.substr(pos + 1, brace_pos - pos - 1));
876 // skip all whitespace and move to the next delimiter
877 // brace_pos points to the next position after the matching '}'
878 pos = brace_pos + 1;
879 while (pos < opts.size() && isspace(opts[pos])) {
880 ++pos;
881 }
882 if (pos < opts.size() && opts[pos] != delimiter) {
883 return Status::InvalidArgument("Unexpected chars after nested options");
884 }
885 *end = pos;
886 } else {
887 return Status::InvalidArgument(
888 "Mismatched curly braces for nested options");
889 }
890 } else {
891 *end = opts.find(delimiter, pos);
892 if (*end == std::string::npos) {
893 // It either ends with a trailing semi-colon or the last key-value pair
894 *token = trim(opts.substr(pos));
895 } else {
896 *token = trim(opts.substr(pos, *end - pos));
897 }
898 }
899 return Status::OK();
900}

902Status OptionTypeInfo::Parse(const ConfigOptions& config_options,
903 const std::string& opt_name,
904 const std::string& value, void* opt_ptr) const {
905 if (IsDeprecated()) {
906 return Status::OK();
907 }
908 try {
909 const std::string& opt_value = config_options.input_strings_escaped
910 ? UnescapeOptionString(value)
911 : value;
912
913 if (opt_ptr == nullptr) {
914 return Status::NotFound("Could not find option", opt_name);
915 } else if (parse_func_ != nullptr) {
916 ConfigOptions copy = config_options;
917 copy.invoke_prepare_options = false;
918 void* opt_addr = GetOffset(opt_ptr);
919 return parse_func_(copy, opt_name, opt_value, opt_addr);
920 } else if (ParseOptionHelper(GetOffset(opt_ptr), type_, opt_value)) {
921 return Status::OK();
922 } else if (IsConfigurable()) {
923 // The option is <config>.<name>
924 Configurable* config = AsRawPointer<Configurable>(opt_ptr);
925 if (opt_value.empty()) {
926 return Status::OK();
927 } else if (config == nullptr) {
928 return Status::NotFound("Could not find configurable: ", opt_name);
929 } else {
930 ConfigOptions copy = config_options;
931 copy.ignore_unknown_options = false;
932 copy.invoke_prepare_options = false;
933 if (opt_value.find('=') != std::string::npos) {
934 return config->ConfigureFromString(copy, opt_value);
935 } else {
936 return config->ConfigureOption(copy, opt_name, opt_value);
937 }
938 }
939 } else if (IsByName()) {
940 return Status::NotSupported("Deserializing the option " + opt_name +
941 " is not supported");
942 } else {
943 return Status::InvalidArgument("Error parsing:", opt_name);
944 }
945 } catch (std::exception& e) {
946 return Status::InvalidArgument("Error parsing " + opt_name + ":" +
947 std::string(e.what()));
948 }
949}

964Status OptionTypeInfo::ParseType(
965 const ConfigOptions& config_options,
966 const std::unordered_map<std::string, std::string>& opts_map,
967 const std::unordered_map<std::string, OptionTypeInfo>& type_map,
968 void* opt_addr, std::unordered_map<std::string, std::string>* unused) {
969 for (const auto& opts_iter : opts_map) {
970 std::string opt_name;
971 const auto* opt_info = Find(opts_iter.first, type_map, &opt_name);
972 if (opt_info != nullptr) {
973 Status status =
974 opt_info->Parse(config_options, opt_name, opts_iter.second, opt_addr);
975 if (!status.ok()) {
976 return status;
977 }
978 } else if (unused != nullptr) {
979 (*unused)[opts_iter.first] = opts_iter.second;
980 } else if (!config_options.ignore_unknown_options) {
981 return Status::NotFound("Unrecognized option", opts_iter.first);
982 }
983 }
984 return Status::OK();
985}

987Status OptionTypeInfo::ParseStruct(
988 const ConfigOptions& config_options, const std::string& struct_name,
989 const std::unordered_map<std::string, OptionTypeInfo>* struct_map,
990 const std::string& opt_name, const std::string& opt_value, void* opt_addr) {
991 assert(struct_map);
992 Status status;
993 if (opt_name == struct_name || EndsWith(opt_name, "." + struct_name)) {
994 // This option represents the entire struct
995 std::unordered_map<std::string, std::string> unused;
996 status =
997 ParseType(config_options, opt_value, *struct_map, opt_addr, &unused);
998 if (status.ok() && !unused.empty() &&
999 !config_options.ignore_unknown_options) {
1000 status = Status::InvalidArgument(
1001 "Unrecognized option", struct_name + "." + unused.begin()->first);
1002 }
1003 } else if (StartsWith(opt_name, struct_name + ".")) {
1004 // This option represents a nested field in the struct (e.g, struct.field)
1005 std::string elem_name;
1006 const auto opt_info =
1007 Find(opt_name.substr(struct_name.size() + 1), *struct_map, &elem_name);
1008 if (opt_info != nullptr) {
1009 status = opt_info->Parse(config_options, elem_name, opt_value, opt_addr);
1010 } else if (!config_options.ignore_unknown_options) {
1011 status = Status::InvalidArgument("Unrecognized option", opt_name);
1012 }
1013 } else {
1014 // This option represents a field in the struct (e.g. field)
1015 std::string elem_name;
1016 const auto opt_info = Find(opt_name, *struct_map, &elem_name);
1017 if (opt_info != nullptr) {
1018 status = opt_info->Parse(config_options, elem_name, opt_value, opt_addr);
1019 } else if (!config_options.ignore_unknown_options) {
1020 status = Status::InvalidArgument("Unrecognized option",
1021 struct_name + "." + opt_name);
1022 }
1023 }
1024 return status;
1025}

1027Status OptionTypeInfo::Serialize(const ConfigOptions& config_options,
1028 const std::string& opt_name,
1029 const void* const opt_ptr,
1030 std::string* opt_value) const {
1031 // If the option is no longer used in rocksdb and marked as deprecated,
1032 // we skip it in the serialization.
1033 if (opt_ptr == nullptr || IsDeprecated()) {
1034 return Status::OK();
1035 } else if (IsEnabled(OptionTypeFlags::kDontSerialize)) {
1036 return Status::NotSupported("Cannot serialize option: ", opt_name);
1037 } else if (serialize_func_ != nullptr) {
1038 const void* opt_addr = GetOffset(opt_ptr);
1039 return serialize_func_(config_options, opt_name, opt_addr, opt_value);
1040 } else if (IsCustomizable()) {
1041 const Customizable* custom = AsRawPointer<Customizable>(opt_ptr);
1042 opt_value->clear();
1043 if (custom == nullptr) {
1044 // We do not have a custom object to serialize.
1045 // If the option is not mutable and we are doing only mutable options,
1046 // we return an empty string (which will cause the option not to be
1047 // printed). Otherwise, we return the "nullptr" string, which will result
1048 // in "option=nullptr" being printed.
1049 if (IsMutable() || !config_options.mutable_options_only) {
1050 *opt_value = kNullptrString;
1051 } else {
1052 *opt_value = "";
1053 }
1054 } else if (IsEnabled(OptionTypeFlags::kStringNameOnly) &&
1055 !config_options.IsDetailed()) {
1056 if (!config_options.mutable_options_only || IsMutable()) {
1057 *opt_value = custom->GetId();
1058 }
1059 } else {
1060 ConfigOptions embedded = config_options;
1061 embedded.delimiter = ";";
1062 // If this option is mutable, everything inside it should be considered
1063 // mutable
1064 if (IsMutable()) {
1065 embedded.mutable_options_only = false;
1066 }
1067 std::string value = custom->ToString(embedded);
1068 if (!embedded.mutable_options_only ||
1069 value.find('=') != std::string::npos) {
1070 *opt_value = value;
1071 } else {
1072 *opt_value = "";
1073 }
1074 }
1075 return Status::OK();
1076 } else if (IsConfigurable()) {
1077 const Configurable* config = AsRawPointer<Configurable>(opt_ptr);
1078 if (config != nullptr) {
1079 ConfigOptions embedded = config_options;
1080 embedded.delimiter = ";";
1081 *opt_value = config->ToString(embedded);
1082 }
1083 return Status::OK();
1084 } else if (config_options.mutable_options_only && !IsMutable()) {
1085 return Status::OK();
1086 } else if (SerializeSingleOptionHelper(GetOffset(opt_ptr), type_,
1087 opt_value)) {
1088 return Status::OK();
1089 } else {
1090 return Status::InvalidArgument("Cannot serialize option: ", opt_name);
1091 }
1092}

1094Status OptionTypeInfo::SerializeType(
1095 const ConfigOptions& config_options,
1096 const std::unordered_map<std::string, OptionTypeInfo>& type_map,
1097 const void* opt_addr, std::string* result) {
1098 Status status;
1099 for (const auto& iter : type_map) {
1100 std::string single;
1101 const auto& opt_info = iter.second;
1102 if (opt_info.ShouldSerialize()) {
1103 status =
1104 opt_info.Serialize(config_options, iter.first, opt_addr, &single);
1105 if (!status.ok()) {
1106 return status;
1107 } else {
1108 result->append(iter.first + "=" + single + config_options.delimiter);
1109 }
1110 }
1111 }
1112 return status;
1113}

1115Status OptionTypeInfo::SerializeStruct(
1116 const ConfigOptions& config_options, const std::string& struct_name,
1117 const std::unordered_map<std::string, OptionTypeInfo>* struct_map,
1118 const std::string& opt_name, const void* opt_addr, std::string* value) {
1119 assert(struct_map);
1120 Status status;
1121 if (EndsWith(opt_name, struct_name)) {
1122 // We are going to write the struct as "{ prop1=value1; prop2=value2;}.
1123 // Set the delimiter to ";" so that the everything will be on one line.

1230 const std::string& opt_name,
1231 const void* const this_ptr,
1232 const void* const that_ptr,
1233 std::string* mismatch) const {
1234 auto level = GetSanityLevel();
1235 if (!config_options.IsCheckEnabled(level)) {
1236 return true; // If the sanity level is not being checked, skip it
1237 }
1238 if (this_ptr == nullptr || that_ptr == nullptr) {
1239 if (this_ptr == that_ptr) {
1240 return true;
1241 }
1242 } else if (equals_func_ != nullptr) {

1244 const void* that_addr = GetOffset(that_ptr);
1245 if (equals_func_(config_options, opt_name, this_addr, that_addr,
1246 mismatch)) {
1247 return true;
1248 }
1249 } else {
1250 const void* this_addr = GetOffset(this_ptr);
1251 const void* that_addr = GetOffset(that_ptr);
1252 if (AreOptionsEqual(type_, this_addr, that_addr)) {
1253 return true;
1254 } else if (IsConfigurable()) {
1255 const auto* this_config = AsRawPointer<Configurable>(this_ptr);
1256 const auto* that_config = AsRawPointer<Configurable>(that_ptr);
1257 if (this_config == that_config) {
1258 return true;
1259 } else if (this_config != nullptr && that_config != nullptr) {
1260 std::string bad_name;
1261 bool matches;
1262 if (level < config_options.sanity_level) {
1263 ConfigOptions copy = config_options;
1264 copy.sanity_level = level;
1265 matches = this_config->AreEquivalent(copy, that_config, &bad_name);
1266 } else {
1267 matches = this_config->AreEquivalent(config_options, that_config,
1268 &bad_name);
1269 }
1270 if (!matches) {
1271 *mismatch = opt_name + "." + bad_name;
1272 }
1273 return matches;
1274 }
1275 }
1276 }
1277 if (mismatch->empty()) {
1278 *mismatch = opt_name;
1279 }
1280 return false;
