22class CacheActivityLogger {
23 public:
24 CacheActivityLogger()
25 : activity_logging_enabled_(false), max_logging_size_(0) {}
26
27 ~CacheActivityLogger() {
28 MutexLock l(&mutex_);
29
30 StopLoggingInternal();
31 bg_status_.PermitUncheckedError();
32 }
33
34 Status StartLogging(const std::string& activity_log_file, Env* env,
35 uint64_t max_logging_size = 0) {
36 assert(activity_log_file != "");
37 assert(env != nullptr);
38
39 Status status;
40 FileOptions file_opts;
41
42 MutexLock l(&mutex_);
43
44 // Stop existing logging if any
45 StopLoggingInternal();
46
47 // Open log file
48 status = WritableFileWriter::Create(env->GetFileSystem(), activity_log_file,
49 file_opts, &file_writer_, nullptr);
50 if (!status.ok()) {
51 return status;
52 }
53
54 max_logging_size_ = max_logging_size;
55 activity_logging_enabled_.store(true);
56
57 return status;
58 }
59
60 void StopLogging() {
61 MutexLock l(&mutex_);
62
63 StopLoggingInternal();
64 }
65
66 void ReportLookup(const Slice& key) {
67 if (activity_logging_enabled_.load() == false) {
68 return;
69 }
70
71 std::ostringstream oss;
72 // line format: "LOOKUP - <KEY>"
73 oss << "LOOKUP - " << key.ToString(true) << std::endl;
74
75 MutexLock l(&mutex_);
76 Status s = file_writer_->Append(IOOptions(), oss.str());
77 if (!s.ok() && bg_status_.ok()) {
78 bg_status_ = s;
79 }
80 if (MaxLoggingSizeReached() || !bg_status_.ok()) {
81 // Stop logging if we have reached the max file size or
82 // encountered an error
83 StopLoggingInternal();
84 }
85 }
86
87 void ReportAdd(const Slice& key, size_t size) {
88 if (activity_logging_enabled_.load() == false) {
89 return;
90 }
91
92 std::ostringstream oss;
93 // line format: "ADD - <KEY> - <KEY-SIZE>"
94 oss << "ADD - " << key.ToString(true) << " - " << size << std::endl;
95 MutexLock l(&mutex_);
96 Status s = file_writer_->Append(IOOptions(), oss.str());
97 if (!s.ok() && bg_status_.ok()) {
98 bg_status_ = s;
99 }
100
101 if (MaxLoggingSizeReached() || !bg_status_.ok()) {
102 // Stop logging if we have reached the max file size or
103 // encountered an error
104 StopLoggingInternal();
105 }
106 }
107
108 Status& bg_status() {
109 MutexLock l(&mutex_);
110 return bg_status_;
111 }
112
113 private:
114 bool MaxLoggingSizeReached() {
115 mutex_.AssertHeld();
116
117 return (max_logging_size_ > 0 &&
118 file_writer_->GetFileSize() >= max_logging_size_);
119 }
120
121 void StopLoggingInternal() {
122 mutex_.AssertHeld();
123
124 if (!activity_logging_enabled_) {
125 return;
126 }
127
128 activity_logging_enabled_.store(false);
129 Status s = file_writer_->Close(IOOptions());
130 if (!s.ok() && bg_status_.ok()) {
131 bg_status_ = s;
132 }
133 }
134
135 // Mutex to sync writes to file_writer, and all following
136 // class data members
137 port::Mutex mutex_;
138 // Indicates if logging is currently enabled
139 // atomic to allow reads without mutex
140 std::atomic<bool> activity_logging_enabled_;
141 // When reached, we will stop logging and close the file
142 // Value of 0 means unlimited
143 uint64_t max_logging_size_;
144 std::unique_ptr<WritableFileWriter> file_writer_;
145 Status bg_status_;
146};

149class SimCacheImpl : public SimCache {
150 public:
151 // capacity for real cache (ShardedLRUCache)
152 // test_capacity for key only cache
153 SimCacheImpl(std::shared_ptr<Cache> sim_cache, std::shared_ptr<Cache> cache)
154 : SimCache(cache),
155 key_only_cache_(sim_cache),
156 miss_times_(0),
157 hit_times_(0),
158 stats_(nullptr) {}
159
160 ~SimCacheImpl() override = default;
161
162 const char* Name() const override { return "SimCache"; }
163
164 void SetCapacity(size_t capacity) override { target_->SetCapacity(capacity); }
165
166 void SetStrictCapacityLimit(bool strict_capacity_limit) override {
167 target_->SetStrictCapacityLimit(strict_capacity_limit);
168 }
169
170 Status Insert(const Slice& key, Cache::ObjectPtr value,
171 const CacheItemHelper* helper, size_t charge, Handle** handle,
172 Priority priority, const Slice& compressed = {},
173 CompressionType type = kNoCompression) override {
174 // The handle and value passed in are for real cache, so we pass nullptr
175 // to key_only_cache_ for both instead. Also, the deleter function pointer
176 // will be called by user to perform some external operation which should
177 // be applied only once. Thus key_only_cache accepts an empty function.
178 // *Lambda function without capture can be assgined to a function pointer
179 Handle* h = key_only_cache_->Lookup(key);
180 if (h == nullptr) {
181 // TODO: Check for error here?
182 auto s =
183 key_only_cache_->Insert(key, nullptr, &kNoopCacheItemHelper, charge,
184 nullptr, priority, compressed, type);
185 s.PermitUncheckedError();
186 } else {
187 key_only_cache_->Release(h);
188 }
189
190 cache_activity_logger_.ReportAdd(key, charge);
191 if (!target_) {
192 return Status::OK();
193 }
194 return target_->Insert(key, value, helper, charge, handle, priority,
195 compressed, type);
196 }
197
198 Handle* Lookup(const Slice& key, const CacheItemHelper* helper,
199 CreateContext* create_context,
200 Priority priority = Priority::LOW,
201 Statistics* stats = nullptr) override {
202 HandleLookup(key, stats);
203 if (!target_) {
204 return nullptr;
205 }
206 return target_->Lookup(key, helper, create_context, priority, stats);
207 }
208
209 void StartAsyncLookup(AsyncLookupHandle& async_handle) override {
210 HandleLookup(async_handle.key, async_handle.stats);
211 if (target_) {
212 target_->StartAsyncLookup(async_handle);
213 }
214 }
215
216 bool Ref(Handle* handle) override { return target_->Ref(handle); }

219 bool Release(Handle* handle, bool erase_if_last_ref = false) override {

281 uint64_t get_miss_counter() const override {
282 return miss_times_.load(std::memory_order_relaxed);
283 }
284
285 uint64_t get_hit_counter() const override {
286 return hit_times_.load(std::memory_order_relaxed);
287 }
288
289 void reset_counter() override {
290 miss_times_.store(0, std::memory_order_relaxed);
291 hit_times_.store(0, std::memory_order_relaxed);
292 SetTickerCount(stats_, SIM_BLOCK_CACHE_HIT, 0);
293 SetTickerCount(stats_, SIM_BLOCK_CACHE_MISS, 0);
294 }
295
296 std::string ToString() const override {
297 std::ostringstream oss;
298 oss << "SimCache MISSes: " << get_miss_counter() << std::endl;
299 oss << "SimCache HITs: " << get_hit_counter() << std::endl;
300 auto lookups = get_miss_counter() + get_hit_counter();
301 oss << "SimCache HITRATE: " << std::fixed << std::setprecision(2)
302 << (lookups == 0 ? 0 : get_hit_counter() * 100.0f / lookups)
303 << std::endl;
304 return oss.str();
305 }

307 std::string GetPrintableOptions() const override {
308 std::ostringstream oss;
309 oss << " cache_options:" << std::endl;
310 oss << target_->GetPrintableOptions();
311 oss << " sim_cache_options:" << std::endl;
312 oss << key_only_cache_->GetPrintableOptions();
313 return oss.str();
314 }

316 Status StartActivityLogging(const std::string& activity_log_file, Env* env,
317 uint64_t max_logging_size = 0) override {
318 return cache_activity_logger_.StartLogging(activity_log_file, env,
319 max_logging_size);
320 }
321
322 void StopActivityLogging() override { cache_activity_logger_.StopLogging(); }
323
324 Status GetActivityLoggingStatus() override {
325 return cache_activity_logger_.bg_status();
326 }
327
328 private:
329 std::shared_ptr<Cache> key_only_cache_;
330 std::atomic<uint64_t> miss_times_;
331 std::atomic<uint64_t> hit_times_;
332 Statistics* stats_;
333 CacheActivityLogger cache_activity_logger_;
334
335 void inc_miss_counter() {
336 miss_times_.fetch_add(1, std::memory_order_relaxed);
337 }
338 void inc_hit_counter() { hit_times_.fetch_add(1, std::memory_order_relaxed); }
339
340 void HandleLookup(const Slice& key, Statistics* stats) {
341 Handle* h = key_only_cache_->Lookup(key);
342 if (h != nullptr) {
343 key_only_cache_->Release(h);
344 inc_hit_counter();
345 RecordTick(stats, SIM_BLOCK_CACHE_HIT);
346 } else {
347 inc_miss_counter();
348 RecordTick(stats, SIM_BLOCK_CACHE_MISS);
349 }
350 cache_activity_logger_.ReportLookup(key);
351 }
352};

356// For instrumentation purpose, use NewSimCache instead
357std::shared_ptr<SimCache> NewSimCache(std::shared_ptr<Cache> cache,
358 size_t sim_capacity, int num_shard_bits) {
359 LRUCacheOptions co;
360 co.capacity = sim_capacity;
361 co.num_shard_bits = num_shard_bits;
362 co.metadata_charge_policy = kDontChargeCacheMetadata;
363 return NewSimCache(NewLRUCache(co), cache, num_shard_bits);
364}

366std::shared_ptr<SimCache> NewSimCache(std::shared_ptr<Cache> sim_cache,
367 std::shared_ptr<Cache> cache,
368 int num_shard_bits) {
369 if (num_shard_bits >= 20) {
370 return nullptr; // the cache cannot be sharded into too many fine pieces
371 }
372 return std::make_shared<SimCacheImpl>(sim_cache, cache);
373}
