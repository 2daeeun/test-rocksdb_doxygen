62// The MultiGetContext class is a container for the sorted list of keys that
63// we need to lookup in a batch. Its main purpose is to make batch execution
64// easier by allowing various stages of the MultiGet lookups to operate on
65// subsets of keys, potentially non-contiguous. In order to accomplish this,
66// it defines the following classes -
67//
68// MultiGetContext::Range
69// MultiGetContext::Range::Iterator
70// MultiGetContext::Range::IteratorWrapper
71//
72// Here is an example of how this can be used -
73//
74// {
75// MultiGetContext ctx(...);
76// MultiGetContext::Range range = ctx.GetMultiGetRange();
77//
78// // Iterate to determine some subset of the keys
79// MultiGetContext::Range::Iterator start = range.begin();
80// MultiGetContext::Range::Iterator end = ...;
81//
82// // Make a new range with a subset of keys
83// MultiGetContext::Range subrange(range, start, end);
84//
85// // Define an auxillary vector, if needed, to hold additional data for
86// // each key
87// std::array<Foo, MultiGetContext::MAX_BATCH_SIZE> aux;
88//
89// // Iterate over the subrange and the auxillary vector simultaneously
90// MultiGetContext::Range::Iterator iter = subrange.begin();
91// for (; iter != subrange.end(); ++iter) {
92// KeyContext& key = *iter;
93// Foo& aux_key = aux_iter[iter.index()];
94// ...
95// }
96// }
97class MultiGetContext {
98 public:
99 // Limit the number of keys in a batch to this number. Benchmarks show that
100 // there is negligible benefit for batches exceeding this. Keeping this < 32
101 // simplifies iteration, as well as reduces the amount of stack allocations
102 // that need to be performed
103 static const int MAX_BATCH_SIZE = 32;
104
105 // A bitmask of at least MAX_BATCH_SIZE - 1 bits, so that
106 // Mask{1} << MAX_BATCH_SIZE is well defined
107 using Mask = uint64_t;
108 static_assert(MAX_BATCH_SIZE < sizeof(Mask) * 8);
109
110 MultiGetContext(autovector<KeyContext*, MAX_BATCH_SIZE>* sorted_keys,
111 size_t begin, size_t num_keys, SequenceNumber snapshot,
112 const ReadOptions& read_opts, FileSystem* fs,
113 Statistics* stats)
114 : num_keys_(num_keys),
115 value_mask_(0),
116 value_size_(0),
117 lookup_key_ptr_(reinterpret_cast<LookupKey*>(lookup_key_stack_buf))
118#if USE_COROUTINES
119 ,
120 reader_(fs, stats),
121 executor_(reader_)
122#endif // USE_COROUTINES
123 {
124 (void)fs;
125 (void)stats;
126 assert(num_keys <= MAX_BATCH_SIZE);
127 if (num_keys > MAX_LOOKUP_KEYS_ON_STACK) {
128 lookup_key_heap_buf.reset(new char[sizeof(LookupKey) * num_keys]);
129 lookup_key_ptr_ = reinterpret_cast<LookupKey*>(lookup_key_heap_buf.get());
130 }
131
132 for (size_t iter = 0; iter != num_keys_; ++iter) {
133 // autovector may not be contiguous storage, so make a copy
134 sorted_keys_[iter] = (*sorted_keys)[begin + iter];
135 sorted_keys_[iter]->lkey = new (&lookup_key_ptr_[iter])
136 LookupKey(*sorted_keys_[iter]->key, snapshot, read_opts.timestamp);
137 sorted_keys_[iter]->ukey_with_ts = sorted_keys_[iter]->lkey->user_key();
138 sorted_keys_[iter]->ukey_without_ts = StripTimestampFromUserKey(
139 sorted_keys_[iter]->lkey->user_key(),
140 read_opts.timestamp == nullptr ? 0 : read_opts.timestamp->size());
141 sorted_keys_[iter]->ikey = sorted_keys_[iter]->lkey->internal_key();
142 sorted_keys_[iter]->timestamp = (*sorted_keys)[begin + iter]->timestamp;
143 sorted_keys_[iter]->get_context =
144 (*sorted_keys)[begin + iter]->get_context;
145 }
146 }
147
148 ~MultiGetContext() {
149 for (size_t i = 0; i < num_keys_; ++i) {
150 lookup_key_ptr_[i].~LookupKey();
151 }
152 }

154#if USE_COROUTINES
155 SingleThreadExecutor& executor() { return executor_; }
156
157 AsyncFileReader& reader() { return reader_; }
158#endif // USE_COROUTINES

160 private:
161 static const int MAX_LOOKUP_KEYS_ON_STACK = 16;
162 alignas(
163 alignof(LookupKey)) char lookup_key_stack_buf[sizeof(LookupKey) *
164 MAX_LOOKUP_KEYS_ON_STACK];
165 std::array<KeyContext*, MAX_BATCH_SIZE> sorted_keys_;
166 size_t num_keys_;
167 Mask value_mask_;
168 uint64_t value_size_;
169 std::unique_ptr<char[]> lookup_key_heap_buf;
170 LookupKey* lookup_key_ptr_;
171#if USE_COROUTINES
172 AsyncFileReader reader_;
173 SingleThreadExecutor executor_;
174#endif // USE_COROUTINES
175
176 public:
177 // MultiGetContext::Range - Specifies a range of keys, by start and end index,
178 // from the parent MultiGetContext. Each range contains a bit vector that
179 // indicates whether the corresponding keys need to be processed or skipped.
180 // A Range object can be copy constructed, and the new object inherits the
181 // original Range's bit vector. This is useful for progressively skipping
182 // keys as the lookup goes through various stages. For example, when looking
183 // up keys in the same SST file, a Range is created excluding keys not
184 // belonging to that file. A new Range is then copy constructed and individual
185 // keys are skipped based on bloom filter lookup.
186 class Range {
187 public:
188 // MultiGetContext::Range::Iterator - A forward iterator that iterates over
189 // non-skippable keys in a Range, as well as keys whose final value has been
190 // found. The latter is tracked by MultiGetContext::value_mask_
191 class Iterator {
192 public:
193 // -- iterator traits
194 using self_type = Iterator;
195 using value_type = KeyContext;
196 using reference = KeyContext&;
197 using pointer = KeyContext*;
198 using difference_type = int;
199 using iterator_category = std::forward_iterator_tag;
200
201 Iterator(const Range* range, size_t idx)
202 : range_(range), ctx_(range->ctx_), index_(idx) {
203 while (index_ < range_->end_ &&
204 (Mask{1} << index_) &
205 (range_->ctx_->value_mask_ | range_->skip_mask_ |
206 range_->invalid_mask_))
207 index_++;
208 }
209
210 Iterator(const Iterator&) = default;
211
212 Iterator(const Iterator& other, const Range* range)
213 : range_(range), ctx_(other.ctx_), index_(other.index_) {
214 assert(range->ctx_ == other.ctx_);
215 }
216 Iterator& operator=(const Iterator&) = default;
217
218 Iterator& operator++() {
219 while (++index_ < range_->end_ &&
220 (Mask{1} << index_) &
221 (range_->ctx_->value_mask_ | range_->skip_mask_ |
222 range_->invalid_mask_))
223 ;
224 return *this;
225 }
226
227 bool operator==(Iterator other) const {
228 assert(range_->ctx_ == other.range_->ctx_);
229 return index_ == other.index_;
230 }
231
232 bool operator!=(Iterator other) const {
233 assert(range_->ctx_ == other.range_->ctx_);
234 return index_ != other.index_;
235 }
236
237 KeyContext& operator*() {
238 assert(index_ < range_->end_ && index_ >= range_->start_);
239 return *(ctx_->sorted_keys_[index_]);
240 }
241
242 KeyContext* operator->() {
243 assert(index_ < range_->end_ && index_ >= range_->start_);
244 return ctx_->sorted_keys_[index_];
245 }
246
247 size_t index() { return index_; }
248
249 private:
250 friend Range;
251 const Range* range_;
252 const MultiGetContext* ctx_;
253 size_t index_;
254 };

256 Range(const Range& mget_range, const Iterator& first,
257 const Iterator& last) {
258 ctx_ = mget_range.ctx_;
259 if (first == last) {
260 // This means create an empty range based on mget_range. So just
261 // set start_ and and_ to the same value
262 start_ = mget_range.start_;
263 end_ = start_;
264 } else {
265 start_ = first.index_;
266 end_ = last.index_;
267 }
268 skip_mask_ = mget_range.skip_mask_;
269 invalid_mask_ = mget_range.invalid_mask_;
270 assert(start_ < 64);
271 assert(end_ < 64);
272 }
273
274 Range() = default;
275
276 Iterator begin() const { return Iterator(this, start_); }
277
278 Iterator end() const { return Iterator(this, end_); }
279
280 bool empty() const { return RemainingMask() == 0; }
281
282 void SkipIndex(size_t index) { skip_mask_ |= Mask{1} << index; }
283
284 void SkipKey(const Iterator& iter) { SkipIndex(iter.index_); }
285
286 bool IsKeySkipped(const Iterator& iter) const {
287 return skip_mask_ & (Mask{1} << iter.index_);
288 }
289
290 // Update the value_mask_ in MultiGetContext so its
291 // immediately reflected in all the Range Iterators
292 void MarkKeyDone(Iterator& iter) {
293 ctx_->value_mask_ |= (Mask{1} << iter.index_);
294 }
295
296 bool CheckKeyDone(Iterator& iter) const {
297 return ctx_->value_mask_ & (Mask{1} << iter.index_);
298 }
299
300 uint64_t KeysLeft() const { return BitsSetToOne(RemainingMask()); }
301
302 void AddSkipsFrom(const Range& other) {
303 assert(ctx_ == other.ctx_);
304 skip_mask_ |= other.skip_mask_;
305 }
306
307 uint64_t GetValueSize() { return ctx_->value_size_; }
308
309 void AddValueSize(uint64_t value_size) { ctx_->value_size_ += value_size; }

311 MultiGetContext* context() const { return ctx_; }

313 Range Suffix(const Range& other) const {
314 size_t other_last = other.FindLastRemaining();
315 size_t my_last = FindLastRemaining();
316
317 if (my_last > other_last) {
318 return Range(*this, Iterator(this, other_last),
319 Iterator(this, my_last));
320 } else {
321 return Range(*this, begin(), begin());
322 }
323 }
324
325 // The += operator expands the number of keys in this range. The expansion
326 // is always to the right, i.e start of the additional range >= end of
327 // current range. There should be no overlap. Any skipped keys in rhs are
328 // marked as invalid in the invalid_mask_.
329 Range& operator+=(const Range& rhs) {
330 assert(rhs.start_ >= end_);
331 // Check for non-overlapping ranges and adjust invalid_mask_ accordingly
332 if (end_ < rhs.start_) {
333 invalid_mask_ |= RangeMask(end_, rhs.start_);
334 skip_mask_ |= RangeMask(end_, rhs.start_);
335 }
336 start_ = std::min<size_t>(start_, rhs.start_);
337 end_ = std::max<size_t>(end_, rhs.end_);
338 skip_mask_ |= rhs.skip_mask_ & RangeMask(rhs.start_, rhs.end_);
339 invalid_mask_ |= (rhs.invalid_mask_ | rhs.skip_mask_) &
340 RangeMask(rhs.start_, rhs.end_);
341 assert(start_ < 64);
342 assert(end_ < 64);
343 return *this;
344 }
345
346 // The -= operator removes keys from this range. The removed keys should
347 // come from a range completely overlapping the current range. The removed
348 // keys are marked invalid in the invalid_mask_.
349 Range& operator-=(const Range& rhs) {
350 assert(start_ <= rhs.start_ && end_ >= rhs.end_);
351 skip_mask_ |= (~rhs.skip_mask_ | rhs.invalid_mask_) &
352 RangeMask(rhs.start_, rhs.end_);
353 invalid_mask_ |= (~rhs.skip_mask_ | rhs.invalid_mask_) &
354 RangeMask(rhs.start_, rhs.end_);
355 return *this;
356 }
357
358 // Return a complement of the current range
359 Range operator~() {
360 Range res = *this;
361 res.skip_mask_ = ~skip_mask_ & RangeMask(start_, end_);
362 return res;
363 }
364
365 private:
366 friend MultiGetContext;
367 MultiGetContext* ctx_;
368 size_t start_;
369 size_t end_;
370 Mask skip_mask_;
371 Mask invalid_mask_;
372
373 Range(MultiGetContext* ctx, size_t num_keys)
374 : ctx_(ctx),
375 start_(0),
376 end_(num_keys),
377 skip_mask_(0),
378 invalid_mask_(0) {
379 assert(num_keys < 64);
380 }
381
382 static Mask RangeMask(size_t start, size_t end) {
383 return (((Mask{1} << (end - start)) - 1) << start);
384 }
385
386 Mask RemainingMask() const {
387 return (((Mask{1} << end_) - 1) & ~((Mask{1} << start_) - 1) &
388 ~(ctx_->value_mask_ | skip_mask_));
389 }
390
391 size_t FindLastRemaining() const {
392 Mask mask = RemainingMask();
393 size_t index = (mask >>= start_) ? start_ : 0;
394 while (mask >>= 1) {
395 index++;
396 }
397 return index;
398 }
399 };
400
401 // Return the initial range that encompasses all the keys in the batch
402 Range GetMultiGetRange() { return Range(this, num_keys_); }
403};
