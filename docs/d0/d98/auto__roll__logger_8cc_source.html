21AutoRollLogger::AutoRollLogger(const std::shared_ptr<FileSystem>& fs,
22 const std::shared_ptr<SystemClock>& clock,
23 const std::string& dbname,
24 const std::string& db_log_dir,
25 size_t log_max_size,
26 size_t log_file_time_to_roll,
27 size_t keep_log_file_num,
28 const InfoLogLevel log_level)
29 : Logger(log_level),
30 dbname_(dbname),
31 db_log_dir_(db_log_dir),
32 fs_(fs),
33 clock_(clock),
34 status_(Status::OK()),
35 kMaxLogFileSize(log_max_size),
36 kLogFileTimeToRoll(log_file_time_to_roll),
37 kKeepLogFileNum(keep_log_file_num),
38 cached_now(static_cast<uint64_t>(clock_->NowMicros() * 1e-6)),
39 ctime_(cached_now),
40 cached_now_access_count(0),
41 call_NowMicros_every_N_records_(100),
42 mutex_() {
43 Status s = fs->GetAbsolutePath(dbname, io_options_, &db_absolute_path_,
44 &io_context_);
45 if (s.IsNotSupported()) {
46 db_absolute_path_ = dbname;
47 } else {
48 status_ = s;
49 }
50 log_fname_ = InfoLogFileName(dbname_, db_absolute_path_, db_log_dir_);
51 if (fs_->FileExists(log_fname_, io_options_, &io_context_).ok()) {
52 RollLogFile();
53 }
54 GetExistingFiles();
55 s = ResetLogger();
56 if (s.ok() && status_.ok()) {
57 status_ = TrimOldLogFiles();
58 }
59}

61Status AutoRollLogger::ResetLogger() {
62 TEST_SYNC_POINT("AutoRollLogger::ResetLogger:BeforeNewLogger");
63 status_ = fs_->NewLogger(log_fname_, io_options_, &logger_, &io_context_);
64 TEST_SYNC_POINT("AutoRollLogger::ResetLogger:AfterNewLogger");
65
66 if (!status_.ok()) {
67 return status_;
68 }
69 assert(logger_);
70 logger_->SetInfoLogLevel(Logger::GetInfoLogLevel());
71
72 if (logger_->GetLogFileSize() == Logger::kDoNotSupportGetLogFileSize) {
73 status_ = Status::NotSupported(
74 "The underlying logger doesn't support GetLogFileSize()");
75 }
76 if (status_.ok()) {
77 cached_now = static_cast<uint64_t>(clock_->NowMicros() * 1e-6);
78 ctime_ = cached_now;
79 cached_now_access_count = 0;
80 }
81
82 return status_;
83}

85void AutoRollLogger::RollLogFile() {
86 // This function is called when log is rotating. Two rotations
87 // can happen quickly (NowMicro returns same value). To not overwrite
88 // previous log file we increment by one micro second and try again.
89 uint64_t now = clock_->NowMicros();
90 std::string old_fname;
91 do {
92 old_fname =
93 OldInfoLogFileName(dbname_, now, db_absolute_path_, db_log_dir_);
94 now++;
95 } while (fs_->FileExists(old_fname, io_options_, &io_context_).ok());
96 // Wait for logger_ reference count to turn to 1 as it might be pinned by
97 // Flush. Pinned Logger can't be closed till Flush is completed on that
98 // Logger.
99 while (logger_.use_count() > 1) {
100 }
101 // Close the existing logger first to release the existing handle
102 // before renaming the file using the file system. If this call
103 // fails there is nothing much we can do and we will continue with the
104 // rename and hence ignoring the result status.
105 if (logger_) {
106 logger_->Close().PermitUncheckedError();
107 }
108 Status s = fs_->RenameFile(log_fname_, old_fname, io_options_, &io_context_);
109 if (!s.ok()) {
110 // What should we do on error?
111 }
112 old_log_files_.push(old_fname);
113}

115void AutoRollLogger::GetExistingFiles() {
116 {
117 // Empty the queue to avoid duplicated entries in the queue.
118 std::queue<std::string> empty;
119 std::swap(old_log_files_, empty);
120 }
121
122 std::string parent_dir;
123 std::vector<std::string> info_log_files;
124 Status s =
125 GetInfoLogFiles(fs_, db_log_dir_, dbname_, &parent_dir, &info_log_files);
126 if (status_.ok()) {
127 status_ = s;
128 }
129 // We need to sort the file before enqueing it so that when we
130 // delete file from the front, it is the oldest file.
131 std::sort(info_log_files.begin(), info_log_files.end());
132
133 for (const std::string& f : info_log_files) {
134 old_log_files_.push(parent_dir + "/" + f);
135 }
136}

138Status AutoRollLogger::TrimOldLogFiles() {
139 // Here we directly list info files and delete them through FileSystem.
140 // The deletion isn't going through DB, so there are shortcomes:
141 // 1. the deletion is not rate limited by SstFileManager
142 // 2. there is a chance that an I/O will be issued here
143 // Since it's going to be complicated to pass DB object down to
144 // here, we take a simple approach to keep the code easier to
145 // maintain.
146
147 // old_log_files_.empty() is helpful for the corner case that
148 // kKeepLogFileNum == 0. We can instead check kKeepLogFileNum != 0 but
149 // it's essentially the same thing, and checking empty before accessing
150 // the queue feels safer.
151 while (!old_log_files_.empty() && old_log_files_.size() >= kKeepLogFileNum) {
152 Status s =
153 fs_->DeleteFile(old_log_files_.front(), io_options_, &io_context_);
154 // Remove the file from the tracking anyway. It's possible that
155 // DB cleaned up the old log file, or people cleaned it up manually.
156 old_log_files_.pop();
157 // To make the file really go away, we should sync parent directory.
158 // Since there isn't any consistency issue involved here, skipping
159 // this part to avoid one I/O here.
160 if (!s.ok()) {
161 return s;
162 }
163 }
164 return Status::OK();
165}

167std::string AutoRollLogger::ValistToString(const char* format,
168 va_list args) const {
169 // Any log messages longer than 1024 will get truncated.
170 // The user is responsible for chopping longer messages into multi line log
171 static const int MAXBUFFERSIZE = 1024;
172 char buffer[MAXBUFFERSIZE];
173
174 int count = vsnprintf(buffer, MAXBUFFERSIZE, format, args);
175 (void)count;
176 assert(count >= 0);
177
178 return buffer;
179}

181void AutoRollLogger::LogInternal(const char* format, ...) {
182 mutex_.AssertHeld();
183
184 if (!logger_) {
185 return;

188 va_list args;

190 logger_->Logv(format, args);
191 va_end(args);

194void AutoRollLogger::Logv(const char* format, va_list ap) {
195 std::shared_ptr<Logger> logger;
196 {
197 MutexLock l(&mutex_);
198 assert(GetStatus().ok());
199 if (!logger_) {
200 return;
201 }
202 if ((kLogFileTimeToRoll > 0 && LogExpired()) ||
203 (kMaxLogFileSize > 0 && logger_->GetLogFileSize() >= kMaxLogFileSize)) {
204 RollLogFile();
205 Status s = ResetLogger();
206 Status s2 = TrimOldLogFiles();
207
208 if (!s.ok()) {
209 // can't really log the error if creating a new LOG file failed
210 return;
211 }
212
213 WriteHeaderInfo();
214
215 if (!s2.ok()) {
216 ROCKS_LOG_WARN(logger.get(), "Fail to trim old info log file: %s",
217 s2.ToString().c_str());
218 }
219 }
220
221 // pin down the current logger_ instance before releasing the mutex.
222 logger = logger_;
223 }
224
225 // Another thread could have put a new Logger instance into logger_ by now.
226 // However, since logger is still hanging on to the previous instance
227 // (reference count is not zero), we don't have to worry about it being
228 // deleted while we are accessing it.
229 // Note that logv itself is not mutex protected to allow maximum concurrency,
230 // as thread safety should have been handled by the underlying logger.
231 logger->Logv(format, ap);
232}

234void AutoRollLogger::WriteHeaderInfo() {
235 mutex_.AssertHeld();
236 for (auto& header : headers_) {
237 LogInternal("%s", header.c_str());

241void AutoRollLogger::LogHeader(const char* format, va_list args) {
242 // header message are to be retained in memory. Since we cannot make any
243 // assumptions about the data contained in va_list, we will retain them as
244 // strings
245 va_list tmp;
246 va_copy(tmp, args);
247 std::string data = ValistToString(format, tmp);
248 va_end(tmp);
249
250 MutexLock l(&mutex_);
251 if (!logger_) {
252 return;
253 }
254 headers_.push_back(data);
255
256 // Log the original message to the current log
257 logger_->Logv(format, args);
258}

260bool AutoRollLogger::LogExpired() {
261 if (cached_now_access_count >= call_NowMicros_every_N_records_) {
262 cached_now = static_cast<uint64_t>(clock_->NowMicros() * 1e-6);
263 cached_now_access_count = 0;
264 }
265
266 ++cached_now_access_count;
267 return cached_now >= ctime_ + kLogFileTimeToRoll;
268}

270Status CreateLoggerFromOptions(const std::string& dbname,
271 const DBOptions& options,
272 std::shared_ptr<Logger>* logger) {
273 if (options.info_log) {
274 *logger = options.info_log;
275 return Status::OK();
276 }
277
278 Env* env = options.env;
279 std::string db_absolute_path;
280 Status s = env->GetAbsolutePath(dbname, &db_absolute_path);
281 TEST_SYNC_POINT_CALLBACK("rocksdb::CreateLoggerFromOptions:AfterGetPath", &s);
282 if (!s.ok()) {
283 return s;
284 }
285 std::string fname =
286 InfoLogFileName(dbname, db_absolute_path, options.db_log_dir);
287
288 const auto& clock = env->GetSystemClock();
289 // In case it does not exist.
290 s = env->CreateDirIfMissing(dbname);
291 if (!s.ok()) {
292 if (options.db_log_dir.empty()) {
293 return s;
294 } else {
295 // Ignore the error returned during creation of dbname because dbname and
296 // db_log_dir can be on different filesystems in which case dbname will
297 // not exist and error should be ignored. db_log_dir creation will handle
298 // the error in case there is any error in the creation of dbname on same
299 // filesystem.
300 s = Status::OK();
301 }
302 }
303 assert(s.ok());
304
305 if (!options.db_log_dir.empty()) {
306 s = env->CreateDirIfMissing(options.db_log_dir);
307 if (!s.ok()) {
308 return s;
309 }
310 }
311 // Currently we only support roll by time-to-roll and log size
312 if (options.log_file_time_to_roll > 0 || options.max_log_file_size > 0) {
313 AutoRollLogger* result = new AutoRollLogger(
314 env->GetFileSystem(), clock, dbname, options.db_log_dir,
315 options.max_log_file_size, options.log_file_time_to_roll,
316 options.keep_log_file_num, options.info_log_level);
317 s = result->GetStatus();
318 if (!s.ok()) {
319 delete result;
320 } else {
321 logger->reset(result);
322 }
323 return s;
324 }
325 // Open a log file in the same directory as the db
326 s = env->FileExists(fname);
327 if (s.ok()) {
328 s = env->RenameFile(
329 fname, OldInfoLogFileName(dbname, clock->NowMicros(), db_absolute_path,
330 options.db_log_dir));
331
332 // The operation sequence of "FileExists -> Rename" is not atomic. It's
333 // possible that FileExists returns OK but file gets deleted before Rename.
334 // This can cause Rename to return IOError with subcode PathNotFound.
335 // Although it may be a rare case and applications should be discouraged
336 // to not concurrently modifying the contents of the directories accessed
337 // by the database instance, it is still helpful if we can perform some
338 // simple handling of this case. Therefore, we do the following:
339 // 1. if Rename() returns IOError with PathNotFound subcode, then we check
340 // whether the source file, i.e. LOG, exists.
341 // 2. if LOG exists, it means Rename() failed due to something else. Then
342 // we report error.
343 // 3. if LOG does not exist, it means it may have been removed/renamed by
344 // someone else. Since it does not exist, we can reset Status to OK so
345 // that this caller can try creating a new LOG file. If this succeeds,
346 // we should still allow it.
347 if (s.IsPathNotFound()) {
348 s = env->FileExists(fname);
349 if (s.IsNotFound()) {
350 s = Status::OK();
351 }
352 }
353 } else if (s.IsNotFound()) {
354 // "LOG" is not required to exist since this could be a new DB.
355 s = Status::OK();
356 }
357 if (s.ok()) {
358 s = env->NewLogger(fname, logger);
359 }
360 if (s.ok() && logger->get() != nullptr) {
361 (*logger)->SetInfoLogLevel(options.info_log_level);
362 }
363 return s;
364}
