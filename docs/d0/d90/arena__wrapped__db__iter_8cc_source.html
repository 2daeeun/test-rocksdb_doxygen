42void ArenaWrappedDBIter::Init(
43 Env* env, const ReadOptions& read_options, const ImmutableOptions& ioptions,
44 const MutableCFOptions& mutable_cf_options, const Version* version,
45 const SequenceNumber& sequence, uint64_t max_sequential_skip_in_iteration,
46 uint64_t version_number, ReadCallback* read_callback,
47 ColumnFamilyHandleImpl* cfh, bool expose_blob_index, bool allow_refresh) {
48 read_options_ = read_options;
49 if (!CheckFSFeatureSupport(env->GetFileSystem().get(),
50 FSSupportedOps::kAsyncIO)) {
51 read_options_.async_io = false;
52 }
53 read_options_.total_order_seek |= ioptions.prefix_seek_opt_in_only;
54
55 auto mem = arena_.AllocateAligned(sizeof(DBIter));
56 db_iter_ = new (mem) DBIter(env, read_options_, ioptions, mutable_cf_options,
57 ioptions.user_comparator,
58 /* iter */ nullptr, version, sequence, true,
59 max_sequential_skip_in_iteration, read_callback,
60 cfh, expose_blob_index);
61
62 sv_number_ = version_number;
63 allow_refresh_ = allow_refresh;
64 memtable_range_tombstone_iter_ = nullptr;
65}

69Status ArenaWrappedDBIter::Refresh(const Snapshot* snapshot) {
70 if (cfh_ == nullptr || !allow_refresh_) {
71 return Status::NotSupported("Creating renew iterator is not allowed.");
72 }
73 assert(db_iter_ != nullptr);
74 auto cfd = cfh_->cfd();
75 auto db_impl = cfh_->db();
76
77 // TODO(yiwu): For last_seq_same_as_publish_seq_==false, this is not the
78 // correct behavior. Will be corrected automatically when we take a snapshot
79 // here for the case of WritePreparedTxnDB.
80 uint64_t cur_sv_number = cfd->GetSuperVersionNumber();
81 // If we recreate a new internal iterator below (NewInternalIterator()),
82 // we will pass in read_options_. We need to make sure it
83 // has the right snapshot.
84 read_options_.snapshot = snapshot;
85 TEST_SYNC_POINT("ArenaWrappedDBIter::Refresh:1");
86 TEST_SYNC_POINT("ArenaWrappedDBIter::Refresh:2");
87
88 auto reinit_internal_iter = [&]() {
89 Env* env = db_iter_->env();
90 db_iter_->~DBIter();
91 arena_.~Arena();
92 new (&arena_) Arena();
93
94 SuperVersion* sv = cfd->GetReferencedSuperVersion(db_impl);
95 assert(sv->version_number >= cur_sv_number);
96 SequenceNumber read_seq = GetSeqNum(db_impl, snapshot);
97 if (read_callback_) {
98 read_callback_->Refresh(read_seq);
99 }
100 Init(env, read_options_, *(cfd->ioptions()), sv->mutable_cf_options,
101 sv->current, read_seq,
102 sv->mutable_cf_options.max_sequential_skip_in_iterations,
103 sv->version_number, read_callback_, cfh_, expose_blob_index_,
104 allow_refresh_);
105
106 InternalIterator* internal_iter = db_impl->NewInternalIterator(
107 read_options_, cfd, sv, &arena_, read_seq,
108 /* allow_unprepared_value */ true, /* db_iter */ this);
109 SetIterUnderDBIter(internal_iter);
110 };
111 while (true) {
112 if (sv_number_ != cur_sv_number) {
113 reinit_internal_iter();
114 break;
115 } else {
116 SequenceNumber read_seq = GetSeqNum(db_impl, snapshot);
117 // Refresh range-tombstones in MemTable
118 if (!read_options_.ignore_range_deletions) {
119 SuperVersion* sv = cfd->GetThreadLocalSuperVersion(db_impl);
120 TEST_SYNC_POINT_CALLBACK("ArenaWrappedDBIter::Refresh:SV", nullptr);
121 auto t = sv->mem->NewRangeTombstoneIterator(
122 read_options_, read_seq, false /* immutable_memtable */);
123 if (!t || t->empty()) {
124 // If memtable_range_tombstone_iter_ points to a non-empty tombstone
125 // iterator, then it means sv->mem is not the memtable that
126 // memtable_range_tombstone_iter_ points to, so SV must have changed
127 // after the sv_number_ != cur_sv_number check above. We will fall
128 // back to re-init the InternalIterator, and the tombstone iterator
129 // will be freed during db_iter destruction there.
130 if (memtable_range_tombstone_iter_) {
131 assert(!*memtable_range_tombstone_iter_ ||
132 sv_number_ != cfd->GetSuperVersionNumber());
133 }
134 delete t;
135 } else { // current mutable memtable has range tombstones
136 if (!memtable_range_tombstone_iter_) {
137 delete t;
138 db_impl->ReturnAndCleanupSuperVersion(cfd, sv);
139 // The memtable under DBIter did not have range tombstone before
140 // refresh.
141 reinit_internal_iter();
142 break;
143 } else {
144 *memtable_range_tombstone_iter_ =
145 std::make_unique<TruncatedRangeDelIterator>(
146 std::unique_ptr<FragmentedRangeTombstoneIterator>(t),
147 &cfd->internal_comparator(), nullptr, nullptr);
148 }
149 }
150 db_impl->ReturnAndCleanupSuperVersion(cfd, sv);
151 }
152 // Check again if the latest super version number is changed
153 uint64_t latest_sv_number = cfd->GetSuperVersionNumber();
154 if (latest_sv_number != cur_sv_number) {
155 // If the super version number is changed after refreshing,
156 // fallback to Re-Init the InternalIterator
157 cur_sv_number = latest_sv_number;
158 continue;
159 }
160 db_iter_->set_sequence(read_seq);
161 db_iter_->set_valid(false);
162 break;
163 }
164 }
165 return Status::OK();
166}

168ArenaWrappedDBIter* NewArenaWrappedDbIterator(
169 Env* env, const ReadOptions& read_options, const ImmutableOptions& ioptions,
170 const MutableCFOptions& mutable_cf_options, const Version* version,
171 const SequenceNumber& sequence, uint64_t max_sequential_skip_in_iterations,
172 uint64_t version_number, ReadCallback* read_callback,
173 ColumnFamilyHandleImpl* cfh, bool expose_blob_index, bool allow_refresh) {
174 ArenaWrappedDBIter* iter = new ArenaWrappedDBIter();
175 iter->Init(env, read_options, ioptions, mutable_cf_options, version, sequence,
176 max_sequential_skip_in_iterations, version_number, read_callback,
177 cfh, expose_blob_index, allow_refresh);
178 if (cfh != nullptr && allow_refresh) {
179 iter->StoreRefreshInfo(cfh, read_callback, expose_blob_index);
180 }
181
182 return iter;
183}
