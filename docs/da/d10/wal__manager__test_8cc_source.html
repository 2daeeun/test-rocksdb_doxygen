69 // NOT thread safe
70 void Put(const std::string& key, const std::string& value) {
71 assert(current_log_writer_.get() != nullptr);
72 uint64_t seq = versions_->LastSequence() + 1;
73 WriteBatch batch;
74 ASSERT_OK(batch.Put(key, value));
75 WriteBatchInternal::SetSequence(&batch, seq);
76 ASSERT_OK(current_log_writer_->AddRecord(
77 WriteOptions(), WriteBatchInternal::Contents(&batch)));
78 versions_->SetLastAllocatedSequence(seq);
79 versions_->SetLastPublishedSequence(seq);
80 versions_->SetLastSequence(seq);
81 }

88 std::unique_ptr<WritableFileWriter> file_writer;
89 ASSERT_OK(WritableFileWriter::Create(fs, fname, env_options_, &file_writer,

91 current_log_writer_.reset(
92 new log::Writer(std::move(file_writer), 0, false));
93 }

172uint64_t GetLogDirSize(std::string dir_path, Env* env) {
173 uint64_t dir_size = 0;
174 std::vector<std::string> files;
175 EXPECT_OK(env->GetChildren(dir_path, &files));
176 for (auto& f : files) {
177 uint64_t number;
178 FileType type;
179 if (ParseFileName(f, &number, &type) && type == kWalFile) {
180 std::string const file_path = dir_path + "/" + f;
181 uint64_t file_size;
182 EXPECT_OK(env->GetFileSize(file_path, &file_size));
183 dir_size += file_size;
184 }
185 }
186 return dir_size;
187}
188std::vector<std::uint64_t> ListSpecificFiles(
189 Env* env, const std::string& path, const FileType expected_file_type) {
190 std::vector<std::string> files;
191 std::vector<uint64_t> file_numbers;
192 uint64_t number;
193 FileType type;
194 EXPECT_OK(env->GetChildren(path, &files));
195 for (size_t i = 0; i < files.size(); ++i) {
196 if (ParseFileName(files[i], &number, &type)) {
197 if (type == expected_file_type) {
198 file_numbers.push_back(number);
199 }
200 }
201 }
202 return file_numbers;
203}

205int CountRecords(TransactionLogIterator* iter) {
206 int count = 0;
207 SequenceNumber lastSequence = 0;
208 BatchResult res;
209 while (iter->Valid()) {
210 res = iter->GetBatch();
211 EXPECT_TRUE(res.sequence > lastSequence);
212 ++count;
213 lastSequence = res.sequence;
214 EXPECT_OK(iter->status());
215 iter->Next();
216 }
217 EXPECT_OK(iter->status());
218 return count;
219}

222TEST_F(WalManagerTest, WALArchivalSizeLimit) {
223 db_options_.WAL_ttl_seconds = 0;
224 db_options_.WAL_size_limit_MB = 1000;
225 Init();
226
227 // TEST : Create WalManager with huge size limit and no ttl.
228 // Create some archived files and call PurgeObsoleteWALFiles().
229 // Count the archived log files that survived.
230 // Assert that all of them did.
231 // Change size limit. Re-open WalManager.
232 // Assert that archive is not greater than WAL_size_limit_MB after
233 // PurgeObsoleteWALFiles()
234 // Set ttl and time_to_check_ to small values. Re-open db.
235 // Assert that there are no archived logs left.
236
237 std::string archive_dir = ArchivalDirectory(dbname_);
238 CreateArchiveLogs(20, 5000);
239
240 std::vector<std::uint64_t> log_files =
241 ListSpecificFiles(env_.get(), archive_dir, kWalFile);
242 ASSERT_EQ(log_files.size(), 20U);
243
244 db_options_.WAL_size_limit_MB = 8;
245 Reopen();
246 wal_manager_->PurgeObsoleteWALFiles();
247
248 uint64_t archive_size = GetLogDirSize(archive_dir, env_.get());
249 ASSERT_TRUE(archive_size <= db_options_.WAL_size_limit_MB * 1024 * 1024);
250
251 db_options_.WAL_ttl_seconds = 1;
252 env_->SleepForMicroseconds(2 * 1000 * 1000);
253 Reopen();
254 wal_manager_->PurgeObsoleteWALFiles();
255
256 log_files = ListSpecificFiles(env_.get(), archive_dir, kWalFile);
257 ASSERT_TRUE(log_files.empty());
258}

308TEST_F(WalManagerTest, TransactionLogIteratorNewFileWhileScanning) {
309 Init();
310 CreateArchiveLogs(2, 100);
311 auto iter = OpenTransactionLogIter(0);
312 CreateArchiveLogs(1, 100);
313 int i = 0;

315 i++;
316 }
317 ASSERT_EQ(i, 200);
318 // A new log file was added after the iterator was created.
319 // TryAgain indicates a new iterator is needed to fetch the new data

322 iter = OpenTransactionLogIter(0);
323 i = 0;
324 for (; iter->Valid(); iter->Next()) {
325 i++;
