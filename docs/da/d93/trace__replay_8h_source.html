50// The data structure that defines a single trace.
51struct Trace {
52 uint64_t ts; // timestamp
53 TraceType type;
54 // Each bit in payload_map stores which corresponding struct member added in
55 // the payload. Each TraceType has its corresponding payload struct. For
56 // example, if bit at position 0 is set in write payload, then the write batch
57 // will be addedd.
58 uint64_t payload_map = 0;
59 // Each trace type has its own payload_struct, which will be serialized in the
60 // payload.
61 std::string payload;
62
63 void reset() {
64 ts = 0;
65 type = kTraceMax;
66 payload_map = 0;
67 payload.clear();
68 }
69};
70
71enum TracePayloadType : char {
72 // Each member of all query payload structs should have a corresponding flag
73 // here. Make sure to add them sequentially in the order of it is added.
74 kEmptyPayload = 0,
75 kWriteBatchData = 1,
76 kGetCFID = 2,
77 kGetKey = 3,
78 kIterCFID = 4,
79 kIterKey = 5,
80 kIterLowerBound = 6,
81 kIterUpperBound = 7,
82 kMultiGetSize = 8,
83 kMultiGetCFIDs = 9,
84 kMultiGetKeys = 10,
85};

117// Tracer captures all RocksDB operations using a user-provided TraceWriter.
118// Every RocksDB operation is written as a single trace. Each trace will have a
119// timestamp and type, followed by the trace payload.
120class Tracer {
121 public:
122 Tracer(SystemClock* clock, const TraceOptions& trace_options,
123 std::unique_ptr<TraceWriter>&& trace_writer);
124 ~Tracer();
125
126 // Trace all write operations -- Put, Merge, Delete, SingleDelete, Write
127 Status Write(WriteBatch* write_batch);
128
129 // Trace Get operations.
130 Status Get(ColumnFamilyHandle* cfname, const Slice& key);
131
132 // Trace Iterators.
133 Status IteratorSeek(const uint32_t& cf_id, const Slice& key,
134 const Slice& lower_bound, const Slice upper_bound);
135 Status IteratorSeekForPrev(const uint32_t& cf_id, const Slice& key,
136 const Slice& lower_bound, const Slice upper_bound);
137
138 // Trace MultiGet
139
140 Status MultiGet(const size_t num_keys, ColumnFamilyHandle** column_families,
141 const Slice* keys);
142
143 Status MultiGet(const size_t num_keys, ColumnFamilyHandle* column_family,
144 const Slice* keys);
145
146 Status MultiGet(const std::vector<ColumnFamilyHandle*>& column_family,
147 const std::vector<Slice>& keys);
148
149 // Returns true if the trace is over the configured max trace file limit.
150 // False otherwise.
151 bool IsTraceFileOverMax();
152
153 // Returns true if the order of write trace records must match the order of
154 // the corresponding records logged to WAL and applied to the DB.
155 bool IsWriteOrderPreserved() { return trace_options_.preserve_write_order; }
156
157 // Writes a trace footer at the end of the tracing
158 Status Close();
159
160 private:
161 // Write a trace header at the beginning, typically on initiating a trace,
162 // with some metadata like a magic number, trace version, RocksDB version, and
163 // trace format.
164 Status WriteHeader();
165
166 // Write a trace footer, typically on ending a trace, with some metadata.
167 Status WriteFooter();
168
169 // Write a single trace using the provided TraceWriter to the underlying
170 // system, say, a filesystem or a streaming service.
171 Status WriteTrace(const Trace& trace);
172
173 // Helps in filtering and sampling of traces.
174 // Returns true if a trace should be skipped, false otherwise.
175 bool ShouldSkipTrace(const TraceType& type);
176
177 SystemClock* clock_;
178 TraceOptions trace_options_;
179 std::unique_ptr<TraceWriter> trace_writer_;
180 uint64_t trace_request_count_;
181 Status trace_write_status_;
182};
