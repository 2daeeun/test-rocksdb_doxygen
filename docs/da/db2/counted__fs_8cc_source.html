211class CountedDirectory : public FSDirectoryWrapper {
212 private:
213 mutable CountedFileSystem* fs_;
214 bool closed_ = false;
215
216 public:
217 CountedDirectory(std::unique_ptr<FSDirectory>&& f, CountedFileSystem* fs)
218 : FSDirectoryWrapper(std::move(f)), fs_(fs) {}
219
220 IOStatus Fsync(const IOOptions& options, IODebugContext* dbg) override {
221 IOStatus rv = FSDirectoryWrapper::Fsync(options, dbg);
222 if (rv.ok()) {
223 fs_->counters()->dsyncs++;
224 }
225 return rv;
226 }
227
228 IOStatus Close(const IOOptions& options, IODebugContext* dbg) override {
229 IOStatus rv = FSDirectoryWrapper::Close(options, dbg);
230 if (rv.ok()) {
231 fs_->counters()->closes++;
232 fs_->counters()->dir_closes++;
233 closed_ = true;
234 }
235 return rv;
236 }
237
238 IOStatus FsyncWithDirOptions(const IOOptions& options, IODebugContext* dbg,
239 const DirFsyncOptions& dir_options) override {
240 IOStatus rv =
241 FSDirectoryWrapper::FsyncWithDirOptions(options, dbg, dir_options);
242 if (rv.ok()) {
243 fs_->counters()->dsyncs++;
244 }
245 return rv;
246 }
247
248 ~CountedDirectory() {
249 if (!closed_) {
250 // TODO: fix DB+CF code to use explicit Close, not rely on destructor
251 fs_->counters()->closes++;
252 fs_->counters()->dir_closes++;
253 }
254 }

258std::string FileOpCounters::PrintCounters() const {
259 std::stringstream ss;
260 ss << "Num files opened: " << opens.load(std::memory_order_relaxed)
261 << std::endl;
262 ss << "Num files deleted: " << deletes.load(std::memory_order_relaxed)
263 << std::endl;
264 ss << "Num files renamed: " << renames.load(std::memory_order_relaxed)
265 << std::endl;
266 ss << "Num Flush(): " << flushes.load(std::memory_order_relaxed) << std::endl;
267 ss << "Num Sync(): " << syncs.load(std::memory_order_relaxed) << std::endl;
268 ss << "Num Fsync(): " << fsyncs.load(std::memory_order_relaxed) << std::endl;
269 ss << "Num Dir Fsync(): " << dsyncs.load(std::memory_order_relaxed)
270 << std::endl;
271 ss << "Num Close(): " << closes.load(std::memory_order_relaxed) << std::endl;
272 ss << "Num Dir Open(): " << dir_opens.load(std::memory_order_relaxed)
273 << std::endl;
274 ss << "Num Dir Close(): " << dir_closes.load(std::memory_order_relaxed)
275 << std::endl;
276 ss << "Num Read(): " << reads.ops.load(std::memory_order_relaxed)
277 << std::endl;
278 ss << "Num Append(): " << writes.ops.load(std::memory_order_relaxed)
279 << std::endl;
280 ss << "Num bytes read: " << reads.bytes.load(std::memory_order_relaxed)
281 << std::endl;
282 ss << "Num bytes written: " << writes.bytes.load(std::memory_order_relaxed)
283 << std::endl;
284 return ss.str();
285}
