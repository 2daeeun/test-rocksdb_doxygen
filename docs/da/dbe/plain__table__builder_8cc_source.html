35// a utility that helps writing block content to the file
36// @offset will advance if @block_contents was successfully written.
37// @block_handle the block handle this particular block.
38IOStatus WriteBlock(const Slice& block_contents, WritableFileWriter* file,
39 uint64_t* offset, BlockHandle* block_handle) {
40 block_handle->set_offset(*offset);
41 block_handle->set_size(block_contents.size());
42 IOStatus io_s = file->Append(IOOptions(), block_contents);
43
44 if (io_s.ok()) {
45 *offset += block_contents.size();
46 }
47 return io_s;
48}

58PlainTableBuilder::PlainTableBuilder(
59 const ImmutableOptions& ioptions, const MutableCFOptions& moptions,
60 const InternalTblPropCollFactories* internal_tbl_prop_coll_factories,
61 uint32_t column_family_id, int level_at_creation, WritableFileWriter* file,
62 uint32_t user_key_len, EncodingType encoding_type, size_t index_sparseness,
63 uint32_t bloom_bits_per_key, const std::string& column_family_name,
64 uint32_t num_probes, size_t huge_page_tlb_size, double hash_table_ratio,
65 bool store_index_in_file, const std::string& db_id,
66 const std::string& db_session_id, uint64_t file_number)
67 : ioptions_(ioptions),
68 moptions_(moptions),
69 bloom_block_(num_probes),
70 file_(file),
71 bloom_bits_per_key_(bloom_bits_per_key),
72 huge_page_tlb_size_(huge_page_tlb_size),
73 encoder_(encoding_type, user_key_len, moptions.prefix_extractor.get(),
74 index_sparseness),
75 store_index_in_file_(store_index_in_file),
76 prefix_extractor_(moptions.prefix_extractor.get()) {
77 // Build index block and save it in the file if hash_table_ratio > 0
78 if (store_index_in_file_) {
79 assert(hash_table_ratio > 0 || IsTotalOrderMode());
80 index_builder_.reset(new PlainTableIndexBuilder(
81 &arena_, ioptions, moptions.prefix_extractor.get(), index_sparseness,
82 hash_table_ratio, huge_page_tlb_size_));
83 properties_
84 .user_collected_properties[PlainTablePropertyNames::kBloomVersion] =
85 "1"; // For future use
86 }
87
88 properties_.fixed_key_len = user_key_len;
89
90 // for plain table, we put all the data in a big chuck.
91 properties_.num_data_blocks = 1;
92 // Fill it later if store_index_in_file_ == true
93 properties_.index_size = 0;
94 properties_.filter_size = 0;
95 // To support roll-back to previous version, now still use version 0 for
96 // plain encoding.
97 properties_.format_version = (encoding_type == kPlain) ? 0 : 1;
98 properties_.column_family_id = column_family_id;
99 properties_.column_family_name = column_family_name;
100 properties_.db_id = db_id;
101 properties_.db_session_id = db_session_id;
102 properties_.db_host_id = ioptions.db_host_id;
103 if (!ReifyDbHostIdProperty(ioptions_.env, &properties_.db_host_id).ok()) {
104 ROCKS_LOG_INFO(ioptions_.logger, "db_host_id property will not be set");
105 }
106 properties_.orig_file_number = file_number;
107 properties_.prefix_extractor_name =
108 moptions_.prefix_extractor != nullptr
109 ? moptions_.prefix_extractor->AsString()
110 : "nullptr";
111
112 std::string val;
113 PutFixed32(&val, static_cast<uint32_t>(encoder_.GetEncodingType()));
114 properties_
115 .user_collected_properties[PlainTablePropertyNames::kEncodingType] = val;
116
117 assert(internal_tbl_prop_coll_factories);
118 for (auto& factory : *internal_tbl_prop_coll_factories) {
119 assert(factory);
120
121 std::unique_ptr<InternalTblPropColl> collector{
122 factory->CreateInternalTblPropColl(column_family_id, level_at_creation,
123 ioptions.num_levels)};
124 if (collector) {
125 table_properties_collectors_.emplace_back(std::move(collector));
126 }
127 }

137void PlainTableBuilder::Add(const Slice& key, const Slice& value) {
138 // temp buffer for metadata bytes between key and value.
139 char meta_bytes_buf[6];
140 size_t meta_bytes_buf_size = 0;
141 const IOOptions opts;
142
143 ParsedInternalKey internal_key;
144 if (!ParseInternalKey(key, &internal_key, false /* log_err_key */)
145 .ok()) { // TODO
146 assert(false);
147 return;
148 }
149 if (internal_key.type == kTypeRangeDeletion) {
150 status_ = Status::NotSupported("Range deletion unsupported");
151 return;
152 }
153
154 // Store key hash
155 if (store_index_in_file_) {
156 if (moptions_.prefix_extractor == nullptr) {
157 keys_or_prefixes_hashes_.push_back(GetSliceHash(internal_key.user_key));
158 } else {
159 Slice prefix =
160 moptions_.prefix_extractor->Transform(internal_key.user_key);
161 keys_or_prefixes_hashes_.push_back(GetSliceHash(prefix));
162 }
163 }
164
165 // Write value
166 assert(offset_ <= std::numeric_limits<uint32_t>::max());
167 auto prev_offset = static_cast<uint32_t>(offset_);
168 // Write out the key
169 io_status_ = encoder_.AppendKey(key, file_, &offset_, meta_bytes_buf,
170 &meta_bytes_buf_size);
171 if (SaveIndexInFile()) {
172 index_builder_->AddKeyPrefix(GetPrefix(internal_key), prev_offset);
173 }
174
175 // Write value length
176 uint32_t value_size = static_cast<uint32_t>(value.size());
177 if (io_status_.ok()) {
178 char* end_ptr =
179 EncodeVarint32(meta_bytes_buf + meta_bytes_buf_size, value_size);
180 assert(end_ptr <= meta_bytes_buf + sizeof(meta_bytes_buf));
181 meta_bytes_buf_size = end_ptr - meta_bytes_buf;
182 io_status_ =
183 file_->Append(opts, Slice(meta_bytes_buf, meta_bytes_buf_size));
184 }
185
186 // Write value
187 if (io_status_.ok()) {
188 io_status_ = file_->Append(opts, value);
189 offset_ += value_size + meta_bytes_buf_size;
190 }
191
192 if (io_status_.ok()) {
193 properties_.num_entries++;
194 properties_.raw_key_size += key.size();
195 properties_.raw_value_size += value.size();
196 if (internal_key.type == kTypeDeletion ||
197 internal_key.type == kTypeSingleDeletion) {
198 properties_.num_deletions++;
199 } else if (internal_key.type == kTypeMerge) {
200 properties_.num_merge_operands++;
201 }
202 }
203
204 // notify property collectors
205 NotifyCollectTableCollectorsOnAdd(
206 key, value, offset_, table_properties_collectors_, ioptions_.logger);
207 status_ = io_status_;
208}

210Status PlainTableBuilder::Finish() {
211 assert(!closed_);
212 closed_ = true;
213
214 properties_.data_size = offset_;
215
216 // Write the following blocks
217 // 1. [meta block: bloom] - optional
218 // 2. [meta block: index] - optional
219 // 3. [meta block: properties]
220 // 4. [metaindex block]
221 // 5. [footer]
222
223 MetaIndexBuilder meta_index_builer;
224
225 if (store_index_in_file_ && (properties_.num_entries > 0)) {
226 assert(properties_.num_entries <= std::numeric_limits<uint32_t>::max());
227 BlockHandle bloom_block_handle;
228 if (bloom_bits_per_key_ > 0) {
229 bloom_block_.SetTotalBits(
230 &arena_,
231 static_cast<uint32_t>(properties_.num_entries) * bloom_bits_per_key_,
232 ioptions_.bloom_locality, huge_page_tlb_size_, ioptions_.logger);
233
234 PutVarint32(&properties_.user_collected_properties
235 [PlainTablePropertyNames::kNumBloomBlocks],
236 bloom_block_.GetNumBlocks());
237
238 bloom_block_.AddKeysHashes(keys_or_prefixes_hashes_);
239
240 Slice bloom_finish_result = bloom_block_.Finish();
241
242 properties_.filter_size = bloom_finish_result.size();
243 io_status_ =
244 WriteBlock(bloom_finish_result, file_, &offset_, &bloom_block_handle);
245
246 if (!io_status_.ok()) {
247 status_ = io_status_;
248 return status_;
249 }
250 meta_index_builer.Add(BloomBlockBuilder::kBloomBlock, bloom_block_handle);
251 }
252 BlockHandle index_block_handle;
253 Slice index_finish_result = index_builder_->Finish();
254
255 properties_.index_size = index_finish_result.size();
256 io_status_ =
257 WriteBlock(index_finish_result, file_, &offset_, &index_block_handle);
258
259 if (!io_status_.ok()) {
260 status_ = io_status_;
261 return status_;
262 }
263
264 meta_index_builer.Add(PlainTableIndexBuilder::kPlainTableIndexBlock,
265 index_block_handle);
266 }
267
268 // Calculate bloom block size and index block size
269 PropertyBlockBuilder property_block_builder;
270 // -- Add basic properties
271 property_block_builder.AddTableProperty(properties_);
272 // -- Add eixsting user collected properties
273 property_block_builder.Add(properties_.user_collected_properties);
274 // -- Add more user collected properties
275 UserCollectedProperties more_user_collected_properties;
276 NotifyCollectTableCollectorsOnFinish(
277 table_properties_collectors_, ioptions_.logger, &property_block_builder,
278 more_user_collected_properties, properties_.readable_properties);
279 properties_.user_collected_properties.insert(
280 more_user_collected_properties.begin(),
281 more_user_collected_properties.end());
282
283 // -- Write property block
284 BlockHandle property_block_handle;
285 io_status_ = WriteBlock(property_block_builder.Finish(), file_, &offset_,
286 &property_block_handle);
287 if (!io_status_.ok()) {
288 status_ = io_status_;
289 return status_;
290 }
291 meta_index_builer.Add(kPropertiesBlockName, property_block_handle);
292
293 // -- write metaindex block
294 BlockHandle metaindex_block_handle;
295 io_status_ = WriteBlock(meta_index_builer.Finish(), file_, &offset_,
296 &metaindex_block_handle);
297 if (!io_status_.ok()) {
298 status_ = io_status_;
299 return status_;
300 }
301
302 // Write Footer
303 // no need to write out new footer if we're using default checksum
304 FooterBuilder footer;
305 Status s = footer.Build(kPlainTableMagicNumber, /* format_version */ 0,
306 offset_, kNoChecksum, metaindex_block_handle);
307 if (!s.ok()) {
308 status_ = s;
309 return status_;
310 }
311 io_status_ = file_->Append(IOOptions(), footer.GetSlice());
312 if (io_status_.ok()) {
313 offset_ += footer.GetSlice().size();
314 }
315 status_ = io_status_;
316 return status_;
317}
