22enum WalFileType {
23 /* Indicates that WAL file is in archive directory. WAL files are moved from
24 * the main db directory to archive directory once they are not live and stay
25 * there until cleaned up. Files are cleaned depending on archive size
26 * (Options::WAL_size_limit_MB) and time since last cleaning
27 * (Options::WAL_ttl_seconds).
28 */

89// A TransactionLogIterator is used to iterate over the transactions in a db.
90// One run of the iterator is continuous, i.e. the iterator will stop at the
91// beginning of any gap in sequences
92class TransactionLogIterator {
93 public:
94 TransactionLogIterator() {}
95 virtual ~TransactionLogIterator() {}
96
97 // An iterator is either positioned at a WriteBatch or not valid.
98 // This method returns true if the iterator is valid.
99 // Can read data from a valid iterator.
100 virtual bool Valid() = 0;
101
102 // Moves the iterator to the next WriteBatch.
103 // REQUIRES: Valid() to be true.
104 virtual void Next() = 0;
105
106 // Returns ok if the iterator is valid.
107 // Returns the Error when something has gone wrong.
108 virtual Status status() = 0;
109
110 // If valid return's the current write_batch and the sequence number of the
111 // earliest transaction contained in the batch.
112 // ONLY use if Valid() is true and status() is OK.
113 virtual BatchResult GetBatch() = 0;
114
115 // The read options for TransactionLogIterator.
116 struct ReadOptions {
117 // If true, all data read from underlying storage will be
118 // verified against corresponding checksums.
119 // Default: true
120 bool verify_checksums_;
121
122 ReadOptions() : verify_checksums_(true) {}
123
124 explicit ReadOptions(bool verify_checksums)
125 : verify_checksums_(verify_checksums) {}
126 };
127};
