25void FilePrefetchBuffer::PrepareBufferForRead(
26 BufferInfo* buf, size_t alignment, uint64_t offset, size_t roundup_len,
27 bool refit_tail, bool use_fs_buffer, uint64_t& aligned_useful_len) {
28 uint64_t aligned_useful_offset_in_buf = 0;
29 bool copy_data_to_new_buffer = false;
30 // Check if requested bytes are in the existing buffer_.
31 // If only a few bytes exist -- reuse them & read only what is really needed.
32 // This is typically the case of incremental reading of data.
33 // If no bytes exist in buffer -- full pread.
34 if (buf->DoesBufferContainData() && buf->IsOffsetInBuffer(offset)) {
35 // Only a few requested bytes are in the buffer. memmove those chunk of
36 // bytes to the beginning, and memcpy them back into the new buffer if a
37 // new buffer is created.
38 aligned_useful_offset_in_buf =
39 Rounddown(static_cast<size_t>(offset - buf->offset_), alignment);
40 // aligned_useful_len is passed by reference and used to calculate how much
41 // data needs to be read, so it is needed regardless of whether
42 // use_fs_buffer is true
43 aligned_useful_len = static_cast<uint64_t>(buf->CurrentSize()) -
44 aligned_useful_offset_in_buf;
45 assert(aligned_useful_offset_in_buf % alignment == 0);
46 assert(aligned_useful_len % alignment == 0);
47 assert(aligned_useful_offset_in_buf + aligned_useful_len <=
48 buf->offset_ + buf->CurrentSize());
49 if (aligned_useful_len > 0) {
50 copy_data_to_new_buffer = true;
51 } else {
52 // this reset is not necessary, but just to be safe.
53 aligned_useful_offset_in_buf = 0;
54 }
55 }
56
57 // The later buffer allocation / tail refitting does not apply when
58 // use_fs_buffer is true. If we allocate a new buffer, we end up throwing it
59 // away later when we reuse the file system allocated buffer. If we refit
60 // the tail in the main buffer, we don't have a place to put the next chunk of
61 // data provided by the file system (without performing another copy, which we
62 // are trying to avoid in the first place)
63 if (use_fs_buffer) {
64 return;
65 }
66
67 // Create a new buffer only if current capacity is not sufficient, and memcopy
68 // bytes from old buffer if needed (i.e., if aligned_useful_len is greater
69 // than 0).
70 if (buf->buffer_.Capacity() < roundup_len) {
71 buf->buffer_.Alignment(alignment);
72 buf->buffer_.AllocateNewBuffer(
73 static_cast<size_t>(roundup_len), copy_data_to_new_buffer,
74 aligned_useful_offset_in_buf, static_cast<size_t>(aligned_useful_len));
75 } else if (aligned_useful_len > 0 && refit_tail) {
76 // New buffer not needed. But memmove bytes from tail to the beginning
77 // since aligned_useful_len is greater than 0.
78 buf->buffer_.RefitTail(static_cast<size_t>(aligned_useful_offset_in_buf),
79 static_cast<size_t>(aligned_useful_len));
80 } else if (aligned_useful_len > 0) {
81 // For async prefetching, it doesn't call RefitTail with aligned_useful_len
82 // > 0. Allocate new buffer if needed because aligned buffer calculate
83 // remaining buffer as capacity - cursize which might not be the case in
84 // this as it's not refitting.
85 // TODO: Use refit_tail for async prefetching too.
86 buf->buffer_.Alignment(alignment);
87 buf->buffer_.AllocateNewBuffer(
88 static_cast<size_t>(roundup_len), copy_data_to_new_buffer,
89 aligned_useful_offset_in_buf, static_cast<size_t>(aligned_useful_len));
90 }
91}

93Status FilePrefetchBuffer::Read(BufferInfo* buf, const IOOptions& opts,
94 RandomAccessFileReader* reader,
95 uint64_t read_len, uint64_t aligned_useful_len,
96 uint64_t start_offset, bool use_fs_buffer) {
97 Slice result;
98 Status s;
99 char* to_buf = nullptr;
100 if (use_fs_buffer) {
101 s = FSBufferDirectRead(reader, buf, opts, start_offset + aligned_useful_len,
102 read_len, result);
103 } else {
104 to_buf = buf->buffer_.BufferStart() + aligned_useful_len;
105 s = reader->Read(opts, start_offset + aligned_useful_len, read_len, &result,
106 to_buf, /*aligned_buf=*/nullptr);
107 }
108
109#ifndef NDEBUG
110 if (result.size() < read_len) {
111 // Fake an IO error to force db_stress fault injection to ignore
112 // truncated read errors
113 IGNORE_STATUS_IF_ERROR(Status::IOError());
114 }
115#endif
116 if (!s.ok()) {
117 return s;
118 }
119 if (!use_fs_buffer && result.data() != to_buf) {
120 // If the read is coming from some other buffer already in memory (such as
121 // mmap) then it would be inefficient to create another copy in this
122 // FilePrefetchBuffer. The caller is expected to exclude this case.
123 assert(false);
124 return Status::Corruption("File read didn't populate our buffer");
125 }
126
127 if (usage_ == FilePrefetchBufferUsage::kUserScanPrefetch) {
128 RecordTick(stats_, PREFETCH_BYTES, read_len);
129 }
130 if (!use_fs_buffer) {
131 // Update the buffer size.
132 // We already explicitly set the buffer size when we reuse the FS buffer
133 buf->buffer_.Size(static_cast<size_t>(aligned_useful_len) + result.size());
134 }
135 return s;
136}

138Status FilePrefetchBuffer::ReadAsync(BufferInfo* buf, const IOOptions& opts,
139 RandomAccessFileReader* reader,
140 uint64_t read_len, uint64_t start_offset) {
141 TEST_SYNC_POINT("FilePrefetchBuffer::ReadAsync");
142 // callback for async read request.
143 auto fp = std::bind(&FilePrefetchBuffer::PrefetchAsyncCallback, this,
144 std::placeholders::_1, std::placeholders::_2);
145 FSReadRequest req;
146 Slice result;
147 req.len = read_len;
148 req.offset = start_offset;
149 req.result = result;
150 req.scratch = buf->buffer_.BufferStart();
151 buf->async_req_len_ = req.len;
152
153 Status s = reader->ReadAsync(req, opts, fp, buf, &(buf->io_handle_),
154 &(buf->del_fn_), /*aligned_buf =*/nullptr);
155 req.status.PermitUncheckedError();
156 if (s.ok()) {
157 RecordTick(stats_, PREFETCH_BYTES, read_len);
158 buf->async_read_in_progress_ = true;
159 }
160 return s;
161}

163Status FilePrefetchBuffer::Prefetch(const IOOptions& opts,
164 RandomAccessFileReader* reader,
165 uint64_t offset, size_t n) {
166 if (!enable_ || reader == nullptr) {
167 return Status::OK();
168 }
169
170 assert(num_buffers_ == 1);
171
172 AllocateBufferIfEmpty();
173 BufferInfo* buf = GetFirstBuffer();
174
175 TEST_SYNC_POINT("FilePrefetchBuffer::Prefetch:Start");
176
177 if (offset + n <= buf->offset_ + buf->CurrentSize()) {
178 // All requested bytes are already in the buffer. So no need to Read again.
179 return Status::OK();
180 }
181
182 size_t alignment = GetRequiredBufferAlignment(reader);
183 uint64_t rounddown_offset = offset, roundup_end = 0, aligned_useful_len = 0;
184 size_t read_len = 0;
185 // TODO: Enable file system buffer reuse optimization. Need to incorporate
186 // overlap buffer logic here (similar to what is done in PrefetchInternal).
187 // Currently, if we attempt to use the optimization, it results in an
188 // unsigned integer overflow because the returned buffer's offset ends up
189 // higher than the requested offset.
190 bool use_fs_buffer = false;
191
192 ReadAheadSizeTuning(buf, /*read_curr_block=*/true,
193 /*refit_tail=*/true, use_fs_buffer, rounddown_offset,
194 alignment, 0, n, rounddown_offset, roundup_end, read_len,
195 aligned_useful_len);
196
197 Status s;
198 if (read_len > 0) {
199 s = Read(buf, opts, reader, read_len, aligned_useful_len, rounddown_offset,
200 use_fs_buffer);
201 }
202
203 if (usage_ == FilePrefetchBufferUsage::kTableOpenPrefetchTail && s.ok()) {
204 RecordInHistogram(stats_, TABLE_OPEN_PREFETCH_TAIL_READ_BYTES, read_len);
205 }
206 assert(buf->offset_ <= offset);
207 return s;
208}

210// Copy data from src to overlap_buf_.
211void FilePrefetchBuffer::CopyDataToOverlapBuffer(BufferInfo* src,
212 uint64_t& offset,
213 size_t& length) {
214 if (length == 0) {
215 return;
216 }
217 assert(src->IsOffsetInBuffer(offset));
218 uint64_t copy_offset = (offset - src->offset_);
219 size_t copy_len = 0;
220 if (src->IsDataBlockInBuffer(offset, length)) {
221 // All the bytes are in src.
222 copy_len = length;
223 } else {
224 copy_len = src->CurrentSize() - copy_offset;
225 }
226
227 BufferInfo* dst = overlap_buf_;
228 assert(copy_len <= dst->buffer_.Capacity() - dst->buffer_.CurrentSize());
229 dst->buffer_.Append(src->buffer_.BufferStart() + copy_offset, copy_len);
230
231 // Update offset and length.
232 offset += copy_len;
233 length -= copy_len;
234
235 // length > 0 indicates it has consumed all data from the src buffer and it
236 // still needs to read more other buffer.
237 if (length > 0) {
238 FreeFrontBuffer();
239 }
240 TEST_SYNC_POINT("FilePrefetchBuffer::CopyDataToOverlapBuffer:Complete");
241}

243// Clear the buffers if it contains outdated data. Outdated data can be because
244// previous sequential reads were read from the cache instead of these buffer.
245// In that case outdated IOs should be aborted.
246void FilePrefetchBuffer::AbortOutdatedIO(uint64_t offset) {
247 std::vector<void*> handles;
248 std::vector<BufferInfo*> tmp_buf;
249 for (auto& buf : bufs_) {
250 if (buf->IsBufferOutdatedWithAsyncProgress(offset)) {
251 handles.emplace_back(buf->io_handle_);
252 tmp_buf.emplace_back(buf);
253 }
254 }
255
256 if (!handles.empty()) {
257 StopWatch sw(clock_, stats_, ASYNC_PREFETCH_ABORT_MICROS);
258 Status s = fs_->AbortIO(handles);
259 assert(s.ok());
260 }
261
262 for (auto& buf : tmp_buf) {
263 if (buf->async_read_in_progress_) {
264 DestroyAndClearIOHandle(buf);
265 buf->async_read_in_progress_ = false;
266 }
267 buf->ClearBuffer();
268 }
269}

271void FilePrefetchBuffer::AbortAllIOs() {
272 std::vector<void*> handles;
273 for (auto& buf : bufs_) {
274 if (buf->async_read_in_progress_ && buf->io_handle_ != nullptr) {
275 handles.emplace_back(buf->io_handle_);
276 }
277 }
278 if (!handles.empty()) {
279 StopWatch sw(clock_, stats_, ASYNC_PREFETCH_ABORT_MICROS);
280 Status s = fs_->AbortIO(handles);
281 assert(s.ok());
282 }
283
284 for (auto& buf : bufs_) {
285 if (buf->io_handle_ != nullptr && buf->del_fn_ != nullptr) {
286 DestroyAndClearIOHandle(buf);
287 }
288 buf->async_read_in_progress_ = false;
289 }
290}

292// Clear the buffers if it contains outdated data wrt offset. Outdated data can
293// be because previous sequential reads were read from the cache instead of
294// these buffer or there is IOError while filling the buffers.
295//
296// offset - the offset requested to be read. This API makes sure that the
297// front/first buffer in bufs_ should contain this offset, otherwise, all
298// buffers will be freed.
299void FilePrefetchBuffer::ClearOutdatedData(uint64_t offset, size_t length) {
300 while (!IsBufferQueueEmpty()) {
301 BufferInfo* buf = GetFirstBuffer();
302 // Offset is greater than this buffer's end offset.
303 if (buf->IsBufferOutdated(offset)) {
304 FreeFrontBuffer();
305 } else {
306 break;
307 }
308 }
309
310 if (IsBufferQueueEmpty() || NumBuffersAllocated() == 1) {
311 return;
312 }
313
314 BufferInfo* buf = GetFirstBuffer();
315
316 if (buf->async_read_in_progress_) {
317 FreeEmptyBuffers();
318 return;
319 }
320
321 // Below handles the case for Overlapping buffers (NumBuffersAllocated > 1).
322 bool abort_io = false;
323
324 if (buf->DoesBufferContainData() && buf->IsOffsetInBuffer(offset)) {
325 BufferInfo* next_buf = bufs_[1];
326 if (/* next buffer doesn't align with first buffer and requested data
327 overlaps with next buffer */
328 ((buf->offset_ + buf->CurrentSize() != next_buf->offset_) &&
329 (offset + length > buf->offset_ + buf->CurrentSize()))) {
330 abort_io = true;
331 }
332 } else {
333 // buffer with offset doesn't contain data or offset doesn't lie in this
334 // buffer.
335 buf->ClearBuffer();
336 abort_io = true;
337 }
338
339 if (abort_io) {
340 AbortAllIOs();
341 // Clear all buffers after first.
342 for (size_t i = 1; i < bufs_.size(); ++i) {
343 bufs_[i]->ClearBuffer();
344 }
345 }
346 FreeEmptyBuffers();
347 assert(IsBufferQueueEmpty() || buf->IsOffsetInBuffer(offset));
348}

350void FilePrefetchBuffer::PollIfNeeded(uint64_t offset, size_t length) {
351 BufferInfo* buf = GetFirstBuffer();
352
353 if (buf->async_read_in_progress_ && fs_ != nullptr) {
354 if (buf->io_handle_ != nullptr) {
355 // Wait for prefetch data to complete.
356 // No mutex is needed as async_read_in_progress behaves as mutex and is
357 // updated by main thread only.
358 std::vector<void*> handles;
359 handles.emplace_back(buf->io_handle_);
360 StopWatch sw(clock_, stats_, POLL_WAIT_MICROS);
361 fs_->Poll(handles, 1).PermitUncheckedError();
362 }
363
364 // Reset and Release io_handle after the Poll API as request has been
365 // completed.
366 DestroyAndClearIOHandle(buf);
367 }
368
369 // Always call outdated data after Poll as Buffers might be out of sync w.r.t
370 // offset and length.
371 ClearOutdatedData(offset, length);
372}

374// ReadAheadSizeTuning API calls readaheadsize_cb_
375// (BlockBasedTableIterator::BlockCacheLookupForReadAheadSize) to lookup in the
376// cache and tune the start and end offsets based on cache hits/misses.
377//
378// Arguments -
379// read_curr_block : True if this call was due to miss in the cache and
380// FilePrefetchBuffer wants to read that block
381// synchronously.
382// False if current call is to prefetch additional data in
383// extra buffers through ReadAsync API.
384// prev_buf_end_offset : End offset of the previous buffer. It's used in case
385// of ReadAsync to make sure it doesn't read anything from
386// previous buffer which is already prefetched.
387void FilePrefetchBuffer::ReadAheadSizeTuning(
388 BufferInfo* buf, bool read_curr_block, bool refit_tail, bool use_fs_buffer,
389 uint64_t prev_buf_end_offset, size_t alignment, size_t length,
390 size_t readahead_size, uint64_t& start_offset, uint64_t& end_offset,
391 size_t& read_len, uint64_t& aligned_useful_len) {
392 uint64_t updated_start_offset = Rounddown(start_offset, alignment);
393 uint64_t updated_end_offset =
394 Roundup(start_offset + length + readahead_size, alignment);
395 uint64_t initial_end_offset = updated_end_offset;
396 uint64_t initial_start_offset = updated_start_offset;
397
398 // Callback to tune the start and end offsets.
399 if (readaheadsize_cb_ != nullptr && readahead_size > 0) {
400 readaheadsize_cb_(read_curr_block, updated_start_offset,
401 updated_end_offset);
402 }
403
404 // read_len will be 0 and there is nothing to read/prefetch.
405 if (updated_start_offset == updated_end_offset) {
406 start_offset = end_offset = updated_start_offset;
407 UpdateReadAheadTrimmedStat((initial_end_offset - initial_start_offset),
408 (updated_end_offset - updated_start_offset));
409 return;
410 }
411
412 assert(updated_start_offset < updated_end_offset);
413
414 if (!read_curr_block) {
415 // Handle the case when callback added block handles which are already
416 // prefetched and nothing new needs to be prefetched. In that case end
417 // offset updated by callback will be less than prev_buf_end_offset which
418 // means data has been already prefetched.
419 if (updated_end_offset <= prev_buf_end_offset) {
420 start_offset = end_offset = prev_buf_end_offset;
421 UpdateReadAheadTrimmedStat((initial_end_offset - initial_start_offset),
422 (end_offset - start_offset));
423 return;
424 }
425 }
426
427 // Realign if start and end offsets are not aligned after tuning.
428 start_offset = Rounddown(updated_start_offset, alignment);
429 end_offset = Roundup(updated_end_offset, alignment);
430
431 if (!read_curr_block && start_offset < prev_buf_end_offset) {
432 // Previous buffer already contains the data till prev_buf_end_offset
433 // because of alignment. Update the start offset after that to avoid
434 // prefetching it again.
435 start_offset = prev_buf_end_offset;
436 }
437
438 uint64_t roundup_len = end_offset - start_offset;
439
440 PrepareBufferForRead(buf, alignment, start_offset, roundup_len, refit_tail,
441 use_fs_buffer, aligned_useful_len);
442 assert(roundup_len >= aligned_useful_len);
443
444 // Update the buffer offset.
445 buf->offset_ = start_offset;
446 // Update the initial end offset of this buffer which will be the starting
447 // offset of next prefetch.
448 buf->initial_end_offset_ = initial_end_offset;
449 read_len = static_cast<size_t>(roundup_len - aligned_useful_len);
450
451 UpdateReadAheadTrimmedStat((initial_end_offset - initial_start_offset),
452 (end_offset - start_offset));
453}

455// This is for when num_buffers_ = 1.
456// If we are reusing the file system allocated buffer, and only some of the
457// requested data is in the buffer, we copy the relevant data to overlap_buf_
458void FilePrefetchBuffer::HandleOverlappingSyncData(uint64_t offset,
459 size_t length,
460 uint64_t& tmp_offset,
461 size_t& tmp_length,
462 bool& use_overlap_buffer) {
463 if (IsBufferQueueEmpty()) {
464 return;
465 }
466 BufferInfo* buf = GetFirstBuffer();
467 // We should only be calling this when num_buffers_ = 1, so there should
468 // not be any async reads.
469 assert(!buf->async_read_in_progress_);
470
471 if (!buf->async_read_in_progress_ && buf->DoesBufferContainData() &&
472 buf->IsOffsetInBuffer(offset) &&
473 buf->offset_ + buf->CurrentSize() < offset + length) {
474 // Allocated overlap_buf_ is just enough to hold the result for the user
475 // Alignment does not matter here
476 use_overlap_buffer = true;
477 overlap_buf_->ClearBuffer();
478 overlap_buf_->buffer_.Alignment(1);
479 overlap_buf_->buffer_.AllocateNewBuffer(length);
480 overlap_buf_->offset_ = offset;
481 CopyDataToOverlapBuffer(buf, tmp_offset, tmp_length);
482 UpdateStats(/*found_in_buffer=*/false, overlap_buf_->CurrentSize());
483 }
484}

486// This is for when num_buffers_ > 1.
487// If data is overlapping between two buffers then during this call:
488// - data from first buffer is copied into overlapping buffer,
489// - first is removed from bufs_ and freed so that it can be used for async
490// prefetching of further data.
491Status FilePrefetchBuffer::HandleOverlappingAsyncData(
492 const IOOptions& opts, RandomAccessFileReader* reader, uint64_t offset,
493 size_t length, size_t readahead_size, bool& copy_to_overlap_buffer,
494 uint64_t& tmp_offset, size_t& tmp_length) {
495 // No Overlapping of data between 2 buffers.
496 if (IsBufferQueueEmpty() || NumBuffersAllocated() == 1) {
497 return Status::OK();
498 }
499
500 Status s;
501 size_t alignment = GetRequiredBufferAlignment(reader);
502
503 BufferInfo* buf = GetFirstBuffer();
504
505 // Check if the first buffer has the required offset and the async read is
506 // still in progress. This should only happen if a prefetch was initiated
507 // by Seek, but the next access is at another offset.
508 if (buf->async_read_in_progress_ &&
509 buf->IsOffsetInBufferWithAsyncProgress(offset)) {
510 PollIfNeeded(offset, length);
511 }
512
513 if (IsBufferQueueEmpty() || NumBuffersAllocated() == 1) {
514 return Status::OK();
515 }
516
517 BufferInfo* next_buf = bufs_[1];
518
519 // If data is overlapping over two buffers, copy the data from front and
520 // call ReadAsync on freed buffer.
521 if (!buf->async_read_in_progress_ && buf->DoesBufferContainData() &&
522 buf->IsOffsetInBuffer(offset) &&
523 (/*Data extends over two buffers and second buffer either has data or in
524 process of population=*/
525 (offset + length > next_buf->offset_) &&
526 (next_buf->async_read_in_progress_ ||
527 next_buf->DoesBufferContainData()))) {
528 // Allocate new buffer to overlap_buf_.
529 overlap_buf_->ClearBuffer();
530 overlap_buf_->buffer_.Alignment(alignment);
531 overlap_buf_->buffer_.AllocateNewBuffer(length);
532 overlap_buf_->offset_ = offset;
533 copy_to_overlap_buffer = true;
534
535 CopyDataToOverlapBuffer(buf, tmp_offset, tmp_length);
536 UpdateStats(/*found_in_buffer=*/false, overlap_buf_->CurrentSize());
537
538 // Call async prefetching on freed buffer since data has been consumed
539 // only if requested data lies within next buffer.
540 size_t second_size = next_buf->async_read_in_progress_
541 ? next_buf->async_req_len_
542 : next_buf->CurrentSize();
543 uint64_t start_offset = next_buf->initial_end_offset_;
544
545 // If requested bytes - tmp_offset + tmp_length are in next buffer, freed
546 // buffer can go for further prefetching.
547 // If requested bytes are not in next buffer, next buffer has to go for sync
548 // call to get remaining requested bytes. In that case it shouldn't go for
549 // async prefetching as async prefetching calculates offset based on
550 // previous buffer end offset and previous buffer has to go for sync
551 // prefetching.
552
553 if (tmp_offset + tmp_length <= next_buf->offset_ + second_size) {
554 AllocateBuffer();
555 BufferInfo* new_buf = GetLastBuffer();
556 size_t read_len = 0;
557 uint64_t end_offset = start_offset, aligned_useful_len = 0;
558
559 ReadAheadSizeTuning(new_buf, /*read_curr_block=*/false,
560 /*refit_tail=*/false, /*use_fs_buffer=*/false,
561 next_buf->offset_ + second_size, alignment,
562 /*length=*/0, readahead_size, start_offset,
563 end_offset, read_len, aligned_useful_len);
564 if (read_len > 0) {
565 s = ReadAsync(new_buf, opts, reader, read_len, start_offset);
566 if (!s.ok()) {
567 DestroyAndClearIOHandle(new_buf);
568 FreeLastBuffer();
569 return s;
570 }
571 }
572 }
573 }
574 return s;
575}

577// When data is outdated, we clear the first buffer and free it as the
578// data has been consumed because of sequential reads.
579//
580// Scenarios for prefetching asynchronously:
581// Case1: If all buffers are in free_bufs_, prefetch n + readahead_size_/2 bytes
582// synchronously in first buffer and prefetch readahead_size_/2 async in
583// remaining buffers (num_buffers_ -1 ).
584// Case2: If first buffer has partial data, prefetch readahead_size_/2 async in
585// remaining buffers. In case of partial data, prefetch remaining bytes
586// from size n synchronously to fulfill the requested bytes request.
587// Case5: (Special case) If data is overlapping in two buffers, copy requested
588// data from first, free that buffer to send for async request, wait for
589// poll to fill next buffer (if any), and copy remaining data from that
590// buffer to overlap buffer.
591Status FilePrefetchBuffer::PrefetchInternal(const IOOptions& opts,
592 RandomAccessFileReader* reader,
593 uint64_t offset, size_t length,
594 size_t readahead_size,
595 bool& copy_to_overlap_buffer) {
596 if (!enable_) {
597 return Status::OK();
598 }
599
600 TEST_SYNC_POINT("FilePrefetchBuffer::Prefetch:Start");
601
602 size_t alignment = GetRequiredBufferAlignment(reader);
603 Status s;
604 uint64_t tmp_offset = offset;
605 size_t tmp_length = length;
606 size_t original_length = length;
607
608 // Abort outdated IO.
609 if (!explicit_prefetch_submitted_) {
610 AbortOutdatedIO(offset);
611 FreeEmptyBuffers();
612 }
613 ClearOutdatedData(offset, length);
614
615 // Handle overlapping data over two buffers (async prefetching case).
616 s = HandleOverlappingAsyncData(opts, reader, offset, length, readahead_size,
617 copy_to_overlap_buffer, tmp_offset,
618 tmp_length);
619 if (!s.ok()) {
620 return s;
621 }
622 // Handle partially available data when reusing the file system buffer
623 // and num_buffers_ = 1 (sync prefetching case)
624 bool use_fs_buffer = UseFSBuffer(reader);
625 if (!copy_to_overlap_buffer && use_fs_buffer) {
626 HandleOverlappingSyncData(offset, length, tmp_offset, tmp_length,
627 copy_to_overlap_buffer);
628 }
629
630 AllocateBufferIfEmpty();
631 BufferInfo* buf = GetFirstBuffer();
632
633 // Call Poll only if data is needed for the second buffer.
634 // - Return if whole data is in first and second buffer is in progress or
635 // already full.
636 // - If second buffer is empty, it will go for ReadAsync for second buffer.
637 if (!buf->async_read_in_progress_ && buf->DoesBufferContainData() &&
638 buf->IsDataBlockInBuffer(offset, length)) {
639 // Whole data is in buffer.
640 if (!IsEligibleForFurtherPrefetching()) {
641 UpdateStats(/*found_in_buffer=*/true, original_length);
642 return s;
643 }
644 } else {
645 PollIfNeeded(tmp_offset, tmp_length);
646 }
647
648 AllocateBufferIfEmpty();
649 buf = GetFirstBuffer();
650 offset = tmp_offset;
651 length = tmp_length;
652
653 // After polling, if all the requested bytes are in first buffer, it will only
654 // go for async prefetching.
655 if (buf->DoesBufferContainData()) {
656 if (copy_to_overlap_buffer) {
657 // Data is overlapping i.e. some of the data has been copied to overlap
658 // buffer and remaining will be updated below.
659 // Note: why do we not end up performing a duplicate copy when we already
660 // copy to the overlap buffer in HandleOverlappingAsyncData /
661 // HandleOverlappingSyncData? The reason is that when we call
662 // CopyDataToOverlapBuffer, if the buffer is only a "partial hit", then we
663 // clear it out since it does not have any more useful data once we copy
664 // to the overlap buffer. Once we reallocate a fresh buffer, that buffer
665 // will have no data, and it will be the "first" buffer when num_buffers_
666 // = 1. When num_buffers_ > 1, we call ClearOutdatedData() so we know
667 // that, if we get to this point in the control flow, the "front" buffer
668 // has to have the data we need.
669 size_t initial_buf_size = overlap_buf_->CurrentSize();
670 CopyDataToOverlapBuffer(buf, offset, length);
671 UpdateStats(
672 /*found_in_buffer=*/false,
673 overlap_buf_->CurrentSize() - initial_buf_size);
674
675 // Length == 0: All the requested data has been copied to overlap buffer
676 // and it has already gone for async prefetching. It can return without
677 // doing anything further.
678 // Length > 0: More data needs to be consumed so it will continue async
679 // and sync prefetching and copy the remaining data to overlap buffer in
680 // the end.
681 if (length == 0) {
682 UpdateStats(/*found_in_buffer=*/true, length);
683 return s;
684 }
685 } else {
686 if (buf->IsDataBlockInBuffer(offset, length)) {
687 offset += length;
688 length = 0;
689 // Since async request was submitted directly by calling PrefetchAsync
690 // in last call, we don't need to prefetch further as this call is to
691 // poll the data submitted in previous call.
692 if (explicit_prefetch_submitted_) {
693 return s;
694 }
695 if (!IsEligibleForFurtherPrefetching()) {
696 UpdateStats(/*found_in_buffer=*/true, original_length);
697 return s;
698 }
699 }
700 }
701 }
702
703 AllocateBufferIfEmpty();
704 buf = GetFirstBuffer();
705
706 assert(!buf->async_read_in_progress_);
707
708 // Go for ReadAsync and Read (if needed).
709 // offset and size alignment for first buffer with synchronous prefetching
710 uint64_t start_offset1 = offset, end_offset1 = 0, aligned_useful_len1 = 0;
711 size_t read_len1 = 0;
712
713 // For length == 0, skip the synchronous prefetching. read_len1 will be 0.
714 if (length > 0) {
715 if (buf->IsOffsetInBuffer(offset)) {
716 UpdateStats(/*found_in_buffer=*/false,
717 (buf->offset_ + buf->CurrentSize() - offset));
718 }
719 ReadAheadSizeTuning(buf, /*read_curr_block=*/true, /*refit_tail=*/
720 true, /*use_fs_buffer=*/use_fs_buffer, start_offset1,
721 alignment, length, readahead_size, start_offset1,
722 end_offset1, read_len1, aligned_useful_len1);
723 } else {
724 UpdateStats(/*found_in_buffer=*/true, original_length);
725 }
726
727 // Prefetch in remaining buffer only if readahead_size > 0.
728 if (readahead_size > 0) {
729 s = PrefetchRemBuffers(opts, reader, end_offset1, alignment,
730 readahead_size);
731 if (!s.ok()) {
732 return s;
733 }
734 }
735
736 if (read_len1 > 0) {
737 s = Read(buf, opts, reader, read_len1, aligned_useful_len1, start_offset1,
738 use_fs_buffer);
739 if (!s.ok()) {
740 AbortAllIOs();
741 FreeAllBuffers();
742 return s;
743 }
744 }
745
746 // Copy remaining requested bytes to overlap_buf_. No need to
747 // update stats as data is prefetched during this call.
748 if (copy_to_overlap_buffer && length > 0) {
749 CopyDataToOverlapBuffer(buf, offset, length);
750 }
751 return s;
752}

771bool FilePrefetchBuffer::TryReadFromCacheUntracked(
772 const IOOptions& opts, RandomAccessFileReader* reader, uint64_t offset,
773 size_t n, Slice* result, Status* status, bool for_compaction) {
774 if (track_min_offset_ && offset < min_offset_read_) {
775 min_offset_read_ = static_cast<size_t>(offset);
776 }
777
778 if (!enable_) {
779 return false;
780 }
781
782 if (explicit_prefetch_submitted_) {
783 // explicit_prefetch_submitted_ is special case where it expects request
784 // submitted in PrefetchAsync should match with this request. Otherwise
785 // buffers will be outdated.
786 // Random offset called. So abort the IOs.
787 if (prev_offset_ != offset) {
788 AbortAllIOs();
789 FreeAllBuffers();
790 explicit_prefetch_submitted_ = false;
791 return false;
792 }
793 }
794
795 AllocateBufferIfEmpty();
796 BufferInfo* buf = GetFirstBuffer();
797
798 if (!explicit_prefetch_submitted_ && offset < buf->offset_) {
799 return false;
800 }
801
802 bool prefetched = false;
803 bool copy_to_overlap_buffer = false;
804 // If the buffer contains only a few of the requested bytes:
805 // If readahead is enabled: prefetch the remaining bytes + readahead
806 // bytes
807 // and satisfy the request.
808 // If readahead is not enabled: return false.
809 TEST_SYNC_POINT_CALLBACK("FilePrefetchBuffer::TryReadFromCache",
810 &readahead_size_);
811
812 if (explicit_prefetch_submitted_ ||
813 (buf->async_read_in_progress_ ||
814 offset + n > buf->offset_ + buf->CurrentSize())) {
815 // In case readahead_size is trimmed (=0), we still want to poll the data
816 // submitted with explicit_prefetch_submitted_=true.
817 if (readahead_size_ > 0 || explicit_prefetch_submitted_) {
818 Status s;
819 assert(reader != nullptr);
820 assert(max_readahead_size_ >= readahead_size_);
821
822 if (for_compaction) {
823 s = Prefetch(opts, reader, offset, std::max(n, readahead_size_));
824 } else {
825 if (implicit_auto_readahead_) {
826 if (!IsEligibleForPrefetch(offset, n)) {
827 // Ignore status as Prefetch is not called.
828 s.PermitUncheckedError();
829 return false;
830 }
831 }
832
833 // Prefetch n + readahead_size_/2 synchronously as remaining
834 // readahead_size_/2 will be prefetched asynchronously if num_buffers_
835 // > 1.
836 s = PrefetchInternal(
837 opts, reader, offset, n,
838 (num_buffers_ > 1 ? readahead_size_ / 2 : readahead_size_),
839 copy_to_overlap_buffer);
840 explicit_prefetch_submitted_ = false;
841 }
842
843 if (!s.ok()) {
844 if (status) {
845 *status = s;
846 }
847#ifndef NDEBUG
848 IGNORE_STATUS_IF_ERROR(s);
849#endif
850 return false;
851 }
852 prefetched = explicit_prefetch_submitted_ ? false : true;
853 } else {
854 return false;
855 }
856 } else if (!for_compaction) {
857 UpdateStats(/*found_in_buffer=*/true, n);
858 }
859
860 UpdateReadPattern(offset, n, /*decrease_readaheadsize=*/false);
861
862 buf = GetFirstBuffer();
863 if (copy_to_overlap_buffer) {
864 buf = overlap_buf_;
865 }
866 assert(buf->offset_ <= offset);
867 uint64_t offset_in_buffer = offset - buf->offset_;
868 *result = Slice(buf->buffer_.BufferStart() + offset_in_buffer, n);
869 if (prefetched) {
870 readahead_size_ = std::min(max_readahead_size_, readahead_size_ * 2);
871 }
872 return true;
873}

875void FilePrefetchBuffer::PrefetchAsyncCallback(FSReadRequest& req,
876 void* cb_arg) {
877 BufferInfo* buf = static_cast<BufferInfo*>(cb_arg);
878
879#ifndef NDEBUG
880 if (req.result.size() < req.len) {
881 // Fake an IO error to force db_stress fault injection to ignore
882 // truncated read errors
883 IGNORE_STATUS_IF_ERROR(Status::IOError());
884 }
885 IGNORE_STATUS_IF_ERROR(req.status);
886#endif
887
888 if (req.status.ok()) {
889 if (req.offset + req.result.size() <= buf->offset_ + buf->CurrentSize()) {
890 // All requested bytes are already in the buffer or no data is read
891 // because of EOF. So no need to update.
892 return;
893 }
894 if (req.offset < buf->offset_) {
895 // Next block to be read has changed (Recent read was not a sequential
896 // read). So ignore this read.
897 return;
898 }
899 size_t current_size = buf->CurrentSize();
900 buf->buffer_.Size(current_size + req.result.size());
901 }
902}

904Status FilePrefetchBuffer::PrefetchAsync(const IOOptions& opts,
905 RandomAccessFileReader* reader,
906 uint64_t offset, size_t n,
907 Slice* result) {
908 assert(reader != nullptr);
909 if (!enable_) {
910 return Status::NotSupported();
911 }
912
913 TEST_SYNC_POINT("FilePrefetchBuffer::PrefetchAsync:Start");
914
915 num_file_reads_ = 0;
916 explicit_prefetch_submitted_ = false;
917 bool is_eligible_for_prefetching = false;
918
919 if (readahead_size_ > 0 &&
920 (!implicit_auto_readahead_ ||
921 num_file_reads_ >= num_file_reads_for_auto_readahead_)) {
922 is_eligible_for_prefetching = true;
923 }
924
925 // Cancel any pending async read to make code simpler as buffers can be out
926 // of sync.
927 AbortAllIOs();
928 // Free empty buffers after aborting IOs.
929 FreeEmptyBuffers();
930 ClearOutdatedData(offset, n);
931
932 // - Since PrefetchAsync can be called on non sequential reads. So offset can
933 // be less than first buffers' offset. In that case it clears all
934 // buffers.
935 // - In case of tuning of readahead_size, on Reseek, we have to clear all
936 // buffers otherwise, we may end up with inconsistent BlockHandles in queue
937 // and data in buffer.
938 if (!IsBufferQueueEmpty()) {
939 BufferInfo* buf = GetFirstBuffer();
940 if (readaheadsize_cb_ != nullptr || !buf->IsOffsetInBuffer(offset)) {
941 FreeAllBuffers();
942 }
943 }
944
945 UpdateReadPattern(offset, n, /*decrease_readaheadsize=*/false);
946
947 bool data_found = false;
948
949 // If first buffer has full data.
950 if (!IsBufferQueueEmpty()) {
951 BufferInfo* buf = GetFirstBuffer();
952 if (buf->DoesBufferContainData() && buf->IsDataBlockInBuffer(offset, n)) {
953 uint64_t offset_in_buffer = offset - buf->offset_;
954 *result = Slice(buf->buffer_.BufferStart() + offset_in_buffer, n);
955 data_found = true;
956 UpdateStats(/*found_in_buffer=*/true, n);
957
958 // Update num_file_reads_ as TryReadFromCacheAsync won't be called for
959 // poll and update num_file_reads_ if data is found.
960 num_file_reads_++;
961
962 // If next buffer contains some data or is not eligible for prefetching,
963 // return.
964 if (!is_eligible_for_prefetching || NumBuffersAllocated() > 1) {
965 return Status::OK();
966 }
967 } else {
968 // Partial data in first buffer. Clear it to return continous data in one
969 // buffer.
970 FreeAllBuffers();
971 }
972 }
973
974 std::string msg;
975
976 Status s;
977 size_t alignment = GetRequiredBufferAlignment(reader);
978 size_t readahead_size = is_eligible_for_prefetching ? readahead_size_ / 2 : 0;
979 size_t offset_to_read = static_cast<size_t>(offset);
980 uint64_t start_offset1 = offset, end_offset1 = 0, aligned_useful_len1 = 0;
981 size_t read_len1 = 0;
982
983 AllocateBufferIfEmpty();
984 BufferInfo* buf = GetFirstBuffer();
985
986 // - If first buffer is empty.
987 // - Call async read for full data + readahead_size on first buffer.
988 // - Call async read for readahead_size on all remaining buffers if
989 // eligible.
990 // - If first buffer contains data,
991 // - Call async read for readahead_size on all remaining buffers if
992 // eligible.
993
994 // Calculate length and offsets for reading.
995 if (!buf->DoesBufferContainData()) {
996 uint64_t roundup_len1;
997 // Prefetch full data + readahead_size in the first buffer.
998 if (is_eligible_for_prefetching || reader->use_direct_io()) {
999 ReadAheadSizeTuning(buf, /*read_curr_block=*/true, /*refit_tail=*/false,
1000 /*use_fs_buffer=*/false,
1001 /*prev_buf_end_offset=*/start_offset1, alignment, n,
1002 readahead_size, start_offset1, end_offset1, read_len1,
1003 aligned_useful_len1);
1004 } else {
1005 // No alignment or extra prefetching.
1006 start_offset1 = offset_to_read;
1007 end_offset1 = offset_to_read + n;
1008 roundup_len1 = end_offset1 - start_offset1;
1009 PrepareBufferForRead(buf, alignment, start_offset1, roundup_len1,
1010 /*refit_tail=*/false, /*use_fs_buffer=*/false,
1011 aligned_useful_len1);
1012 assert(aligned_useful_len1 == 0);
1013 assert(roundup_len1 >= aligned_useful_len1);
1014 read_len1 = static_cast<size_t>(roundup_len1);
1015 buf->offset_ = start_offset1;
1016 }
1017
1018 if (read_len1 > 0) {
1019 s = ReadAsync(buf, opts, reader, read_len1, start_offset1);
1020 if (!s.ok()) {
1021 DestroyAndClearIOHandle(buf);
1022 FreeLastBuffer();
1023 return s;
1024 }
1025 explicit_prefetch_submitted_ = true;
1026 prev_len_ = 0;
1027 }
1028 }
1029
1030 if (is_eligible_for_prefetching) {
1031 s = PrefetchRemBuffers(opts, reader, end_offset1, alignment,
1032 readahead_size);
1033 if (!s.ok()) {
1034 return s;
1035 }
1036 readahead_size_ = std::min(max_readahead_size_, readahead_size_ * 2);
1037 }
1038 return (data_found ? Status::OK() : Status::TryAgain());
1039}

1041Status FilePrefetchBuffer::PrefetchRemBuffers(const IOOptions& opts,
1042 RandomAccessFileReader* reader,
1043 uint64_t end_offset1,
1044 size_t alignment,
1045 size_t readahead_size) {
1046 Status s;
1047 while (NumBuffersAllocated() < num_buffers_) {
1048 BufferInfo* prev_buf = GetLastBuffer();
1049 uint64_t start_offset2 = prev_buf->initial_end_offset_;
1050
1051 AllocateBuffer();
1052 BufferInfo* new_buf = GetLastBuffer();
1053
1054 uint64_t end_offset2 = start_offset2, aligned_useful_len2 = 0;
1055 size_t read_len2 = 0;
1056 ReadAheadSizeTuning(new_buf, /*read_curr_block=*/false,
1057 /*refit_tail=*/false, /*use_fs_buffer=*/false,
1058 /*prev_buf_end_offset=*/end_offset1, alignment,
1059 /*length=*/0, readahead_size, start_offset2,
1060 end_offset2, read_len2, aligned_useful_len2);
1061
1062 if (read_len2 > 0) {
1063 TEST_SYNC_POINT("FilePrefetchBuffer::PrefetchAsync:ExtraPrefetching");
1064 s = ReadAsync(new_buf, opts, reader, read_len2, start_offset2);
1065 if (!s.ok()) {
1066 DestroyAndClearIOHandle(new_buf);
1067 FreeLastBuffer();
1068 return s;
1069 }
1070 }
1071 end_offset1 = end_offset2;
1072 }
1073 return s;
1074}
