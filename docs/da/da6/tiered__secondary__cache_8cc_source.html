12// Creation callback for use in the lookup path. It calls the upper layer
13// create_cb to create the object, and optionally calls the compressed
14// secondary cache InsertSaved to save the compressed block. If
15// advise_erase is set, it means the primary cache wants the block to be
16// erased in the secondary cache, so we skip calling InsertSaved.
17//
18// For the time being, we assume that all blocks in the nvm tier belong to
19// the primary block cache (i.e CacheTier::kVolatileTier). That can be changed
20// if we implement demotion from the compressed secondary cache to the nvm
21// cache in the future.
22Status TieredSecondaryCache::MaybeInsertAndCreate(
23 const Slice& data, CompressionType type, CacheTier source,
24 Cache::CreateContext* ctx, MemoryAllocator* allocator,
25 Cache::ObjectPtr* out_obj, size_t* out_charge) {
26 TieredSecondaryCache::CreateContext* context =
27 static_cast<TieredSecondaryCache::CreateContext*>(ctx);
28 assert(source == CacheTier::kVolatileTier);
29 if (!context->advise_erase && type != kNoCompression) {
30 // Attempt to insert into compressed secondary cache
31 // TODO: Don't hardcode the source
32 context->comp_sec_cache->InsertSaved(*context->key, data, type, source)
33 .PermitUncheckedError();
34 RecordTick(context->stats, COMPRESSED_SECONDARY_CACHE_PROMOTIONS);
35 } else {
36 RecordTick(context->stats, COMPRESSED_SECONDARY_CACHE_PROMOTION_SKIPS);
37 }
38 // Primary cache will accept the object, so call its helper to create
39 // the object
40 return context->helper->create_cb(data, type, source, context->inner_ctx,
41 allocator, out_obj, out_charge);
42}

44// The lookup first looks up in the compressed secondary cache. If its a miss,
45// then the nvm cache lookup is called. The cache item helper and create
46// context are wrapped in order to intercept the creation callback to make
47// the decision on promoting to the compressed secondary cache.
48std::unique_ptr<SecondaryCacheResultHandle> TieredSecondaryCache::Lookup(
49 const Slice& key, const Cache::CacheItemHelper* helper,
50 Cache::CreateContext* create_context, bool wait, bool advise_erase,
51 Statistics* stats, bool& kept_in_sec_cache) {
52 bool dummy = false;
53 std::unique_ptr<SecondaryCacheResultHandle> result =
54 target()->Lookup(key, helper, create_context, wait, advise_erase, stats,
55 /*kept_in_sec_cache=*/dummy);
56 // We never want the item to spill back into the secondary cache
57 kept_in_sec_cache = true;
58 if (result) {
59 assert(result->IsReady());
60 return result;
61 }
62
63 // If wait is true, then we can be a bit more efficient and avoid a memory
64 // allocation for the CReateContext.
65 const Cache::CacheItemHelper* outer_helper =
66 TieredSecondaryCache::GetHelper();
67 if (wait) {
68 TieredSecondaryCache::CreateContext ctx;
69 ctx.key = &key;
70 ctx.advise_erase = advise_erase;
71 ctx.helper = helper;
72 ctx.inner_ctx = create_context;
73 ctx.comp_sec_cache = target();
74 ctx.stats = stats;
75
76 return nvm_sec_cache_->Lookup(key, outer_helper, &ctx, wait, advise_erase,
77 stats, kept_in_sec_cache);
78 }
79
80 // If wait is false, i.e its an async lookup, we have to allocate a result
81 // handle for tracking purposes. Embed the CreateContext inside the handle
82 // so we need only allocate memory once instead of twice.
83 std::unique_ptr<ResultHandle> handle(new ResultHandle());
84 handle->ctx()->key = &key;
85 handle->ctx()->advise_erase = advise_erase;
86 handle->ctx()->helper = helper;
87 handle->ctx()->inner_ctx = create_context;
88 handle->ctx()->comp_sec_cache = target();
89 handle->ctx()->stats = stats;
90 handle->SetInnerHandle(
91 nvm_sec_cache_->Lookup(key, outer_helper, handle->ctx(), wait,
92 advise_erase, stats, kept_in_sec_cache));
93 if (!handle->inner_handle()) {
94 handle.reset();
95 } else {
96 result.reset(handle.release());
97 }
98
99 return result;
100}

102// Call the nvm cache WaitAll to complete the lookups
103void TieredSecondaryCache::WaitAll(
104 std::vector<SecondaryCacheResultHandle*> handles) {
105 std::vector<SecondaryCacheResultHandle*> nvm_handles;
106 std::vector<ResultHandle*> my_handles;
107 nvm_handles.reserve(handles.size());
108 for (auto handle : handles) {
109 // The handle could belong to the compressed secondary cache. Skip if
110 // that's the case.
111 if (handle->IsReady()) {
112 continue;
113 }
114 ResultHandle* hdl = static_cast<ResultHandle*>(handle);
115 nvm_handles.push_back(hdl->inner_handle());
116 my_handles.push_back(hdl);
117 }
118 nvm_sec_cache_->WaitAll(nvm_handles);
119 for (auto handle : my_handles) {
120 assert(handle->inner_handle()->IsReady());
121 handle->Complete();
122 }
123}
