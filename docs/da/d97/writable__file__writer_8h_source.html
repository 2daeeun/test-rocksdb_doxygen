33// WritableFileWriter is a wrapper on top of Env::WritableFile. It provides
34// facilities to:
35// - Handle Buffered and Direct writes.
36// - Rate limit writes.
37// - Flush and Sync the data to the underlying filesystem.
38// - Notify any interested listeners on the completion of a write.
39// - Update IO stats.
40class WritableFileWriter {
41 private:
42 void NotifyOnFileWriteFinish(
43 uint64_t offset, size_t length,
44 const FileOperationInfo::StartTimePoint& start_ts,
45 const FileOperationInfo::FinishTimePoint& finish_ts,
46 const IOStatus& io_status) {
47 FileOperationInfo info(FileOperationType::kWrite, file_name_, start_ts,
48 finish_ts, io_status, temperature_);
49 info.offset = offset;
50 info.length = length;
51
52 for (auto& listener : listeners_) {
53 listener->OnFileWriteFinish(info);
54 }
55 info.status.PermitUncheckedError();
56 }
57 void NotifyOnFileFlushFinish(
58 const FileOperationInfo::StartTimePoint& start_ts,
59 const FileOperationInfo::FinishTimePoint& finish_ts,
60 const IOStatus& io_status) {

62 finish_ts, io_status, temperature_);
63

66 }

68 }

122 void NotifyOnIOError(const IOStatus& io_status, FileOperationType operation,
123 const std::string& file_path, size_t length = 0,
124 uint64_t offset = 0) {
125 if (listeners_.empty()) {
126 return;
127 }
128 IOErrorInfo io_error_info(io_status, operation, file_path, length, offset);
129 for (auto& listener : listeners_) {
130 listener->OnIOError(io_error_info);
131 }
132 io_error_info.io_status.PermitUncheckedError();
133 }

172 WritableFileWriter(
173 std::unique_ptr<FSWritableFile>&& file, const std::string& _file_name,
174 const FileOptions& options, SystemClock* clock = nullptr,
175 const std::shared_ptr<IOTracer>& io_tracer = nullptr,
176 Statistics* stats = nullptr,
177 Histograms hist_type = Histograms::HISTOGRAM_ENUM_MAX,
178 const std::vector<std::shared_ptr<EventListener>>& listeners = {},
179 FileChecksumGenFactory* file_checksum_gen_factory = nullptr,
180 bool perform_data_verification = false,
181 bool buffered_data_with_checksum = false)
182 : file_name_(_file_name),
183 writable_file_(std::move(file), io_tracer, _file_name),
184 clock_(clock),
185 buf_(),
186 max_buffer_size_(options.writable_file_max_buffer_size),
187 filesize_(0),
188 flushed_size_(0),
189 next_write_offset_(0),
190 pending_sync_(false),
191 seen_error_(false),
192#ifndef NDEBUG
193 seen_injected_error_(false),
194#endif // NDEBUG
195 last_sync_size_(0),
196 bytes_per_sync_(options.bytes_per_sync),
197 rate_limiter_(options.rate_limiter),
198 stats_(stats),
199 hist_type_(hist_type),
200 listeners_(),
201 checksum_generator_(nullptr),
202 checksum_finalized_(false),
203 perform_data_verification_(perform_data_verification),
204 buffered_data_crc32c_checksum_(0),
205 buffered_data_with_checksum_(buffered_data_with_checksum) {
206 temperature_ = options.temperature;
207 assert(!use_direct_io() || max_buffer_size_ > 0);
208 TEST_SYNC_POINT_CALLBACK("WritableFileWriter::WritableFileWriter:0",
209 reinterpret_cast<void*>(max_buffer_size_));
210 buf_.Alignment(writable_file_->GetRequiredBufferAlignment());
211 buf_.AllocateNewBuffer(std::min((size_t)65536, max_buffer_size_));
212 std::for_each(listeners.begin(), listeners.end(),
213 [this](const std::shared_ptr<EventListener>& e) {
214 if (e->ShouldBeNotifiedOnFileIO()) {
215 listeners_.emplace_back(e);
216 }
217 });
218 if (file_checksum_gen_factory != nullptr) {
219 FileChecksumGenContext checksum_gen_context;
220 checksum_gen_context.file_name = _file_name;
221 checksum_generator_ =
222 file_checksum_gen_factory->CreateFileChecksumGenerator(
223 checksum_gen_context);
224 }
225 }

320 (void)s;
321#ifndef NDEBUG

323 seen_injected_error_.store(true, std::memory_order_relaxed);
324 }
325#endif // NDEBUG
326 }

344 }
