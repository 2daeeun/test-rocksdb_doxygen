51Status::Status(Code _code, SubCode _subcode, const Slice& msg,
52 const Slice& msg2, Severity sev)
53 : code_(_code),
54 subcode_(_subcode),
55 sev_(sev),
56 retryable_(false),
57 data_loss_(false),
58 scope_(0) {
59 assert(subcode_ != kMaxSubCode);
60 const size_t len1 = msg.size();
61 const size_t len2 = msg2.size();
62 const size_t size = len1 + (len2 ? (2 + len2) : 0);
63 char* const result = new char[size + 1]; // +1 for null terminator
64 memcpy(result, msg.data(), len1);
65 if (len2) {
66 result[len1] = ':';
67 result[len1 + 1] = ' ';
68 memcpy(result + len1 + 2, msg2.data(), len2);
69 }
70 result[size] = '\0'; // null terminator for C style string
71 state_.reset(result);
72}

81std::string Status::ToString() const {
82#ifdef ROCKSDB_ASSERT_STATUS_CHECKED
83 checked_ = true;
84#endif // ROCKSDB_ASSERT_STATUS_CHECKED
85 const char* type = nullptr;
86 switch (code_) {
87 case kOk:
88 return "OK";
89 case kNotFound:
90 type = "NotFound: ";
91 break;
92 case kCorruption:
93 type = "Corruption: ";
94 break;
95 case kNotSupported:
96 type = "Not implemented: ";
97 break;
98 case kInvalidArgument:
99 type = "Invalid argument: ";
100 break;
101 case kIOError:
102 type = "IO error: ";
103 break;
104 case kMergeInProgress:
105 type = "Merge in progress: ";
106 break;
107 case kIncomplete:
108 type = "Result incomplete: ";
109 break;
110 case kShutdownInProgress:
111 type = "Shutdown in progress: ";
112 break;
113 case kTimedOut:
114 type = "Operation timed out: ";
115 break;
116 case kAborted:
117 type = "Operation aborted: ";
118 break;
119 case kBusy:
120 type = "Resource busy: ";
121 break;
122 case kExpired:
123 type = "Operation expired: ";
124 break;
125 case kTryAgain:
126 type = "Operation failed. Try again.: ";
127 break;
128 case kCompactionTooLarge:
129 type = "Compaction too large: ";
130 break;
131 case kColumnFamilyDropped:
132 type = "Column family dropped: ";
133 break;
134 case kMaxCode:
135 assert(false);
136 break;
137 }
138 char tmp[30];
139 if (type == nullptr) {
140 // This should not happen since `code_` should be a valid non-`kMaxCode`
141 // member of the `Code` enum. The above switch-statement should have had a
142 // case assigning `type` to a corresponding string.
143 assert(false);
144 snprintf(tmp, sizeof(tmp), "Unknown code(%d): ", static_cast<int>(code()));
145 type = tmp;
146 }
147 std::string result(type);
148 if (subcode_ != kNone) {
149 uint32_t index = static_cast<int32_t>(subcode_);
150 assert(sizeof(msgs) / sizeof(msgs[0]) > index);
151 result.append(msgs[index]);
152 }
153
154 if (state_ != nullptr) {
155 if (subcode_ != kNone) {
156 result.append(": ");
157 }
158 result.append(state_.get());
159 }
160 return result;
161}
