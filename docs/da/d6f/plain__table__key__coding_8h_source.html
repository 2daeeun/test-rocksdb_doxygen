62// The class does raw file reads for PlainTableReader.
63// It hides whether it is a mmap-read, or a non-mmap read.
64// The class is implemented in a way to favor the performance of mmap case.
65// The class is used by PlainTableReader.
66class PlainTableFileReader {
67 public:
68 explicit PlainTableFileReader(const PlainTableReaderFileInfo* _file_info)
69 : file_info_(_file_info), num_buf_(0) {}
70
71 ~PlainTableFileReader() {
72 // Should fix.
73 status_.PermitUncheckedError();
74 }
75
76 // In mmaped mode, the results point to mmaped area of the file, which
77 // means it is always valid before closing the file.
78 // In non-mmap mode, the results point to an internal buffer. If the caller
79 // makes another read call, the results may not be valid. So callers should
80 // make a copy when needed.
81 // In order to save read calls to files, we keep two internal buffers:
82 // the first read and the most recent read. This is efficient because it
83 // columns these two common use cases:
84 // (1) hash index only identify one location, we read the key to verify
85 // the location, and read key and value if it is the right location.
86 // (2) after hash index checking, we identify two locations (because of
87 // hash bucket conflicts), we binary search the two location to see
88 // which one is what we need and start to read from the location.
89 // These two most common use cases will be covered by the two buffers
90 // so that we don't need to re-read the same location.
91 // Currently we keep a fixed size buffer. If a read doesn't exactly fit
92 // the buffer, we replace the second buffer with the location user reads.
93 //
94 // If return false, status code is stored in status_.
95 bool Read(uint32_t file_offset, uint32_t len, Slice* out) {
96 if (file_info_->is_mmap_mode) {
97 assert(file_offset + len <= file_info_->data_end_offset);
98 *out = Slice(file_info_->file_data.data() + file_offset, len);
99 return true;
100 } else {
101 return ReadNonMmap(file_offset, len, out);
102 }
103 }
104
105 // If return false, status code is stored in status_.
106 bool ReadNonMmap(uint32_t file_offset, uint32_t len, Slice* output);
107
108 // *bytes_read = 0 means eof. false means failure and status is saved
109 // in status_. Not directly returning Status to save copying status
110 // object to map previous performance of mmap mode.
111 inline bool ReadVarint32(uint32_t offset, uint32_t* output,
112 uint32_t* bytes_read);
113
114 bool ReadVarint32NonMmap(uint32_t offset, uint32_t* output,
115 uint32_t* bytes_read);
116
117 Status status() const { return status_; }
118
119 const PlainTableReaderFileInfo* file_info() { return file_info_; }
120
121 private:
122 const PlainTableReaderFileInfo* file_info_;
