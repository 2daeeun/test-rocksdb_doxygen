29struct FileState {
30 std::string filename_;
31 ssize_t pos_;
32 ssize_t pos_at_last_sync_;
33 ssize_t pos_at_last_flush_;
34
35 explicit FileState(const std::string& filename)
36 : filename_(filename),
37 pos_(-1),
38 pos_at_last_sync_(-1),
39 pos_at_last_flush_(-1) {}
40
41 FileState() : pos_(-1), pos_at_last_sync_(-1), pos_at_last_flush_(-1) {}
42
43 bool IsFullySynced() const { return pos_ <= 0 || pos_ == pos_at_last_sync_; }
44
45 Status DropUnsyncedData(Env* env) const;
46
47 Status DropRandomUnsyncedData(Env* env, Random* rand) const;
48};

215 // Setting the filesystem to inactive is the test equivalent to simulating a
216 // system reset. Setting to inactive will freeze our saved filesystem state so
217 // that it will stop being recorded. It can then be reset back to the state at
218 // the time of the reset.
219 bool IsFilesystemActive() {
220 MutexLock l(&mutex_);
221 return filesystem_active_;
222 }
223 void SetFilesystemActiveNoLock(
224 bool active, Status error = Status::Corruption("Not active")) {
225 error.PermitUncheckedError();
226 filesystem_active_ = active;
227 if (!active) {
228 error_ = error;
229 }
230 error.PermitUncheckedError();
231 }
232 void SetFilesystemActive(bool active,
233 Status error = Status::Corruption("Not active")) {
234 error.PermitUncheckedError();
235 MutexLock l(&mutex_);
236 SetFilesystemActiveNoLock(active, error);
237 error.PermitUncheckedError();
238 }
239 void AssertNoOpenFile() { assert(open_managed_files_.empty()); }
240 Status GetError() { return error_; }
