53 void AddKeys(int numkeys, int startkey = 0) {
54 WriteOptions options;
55 options.sync = false;
56 ReadOptions roptions;
57 for (int i = startkey; i < (numkeys + startkey); i++) {
58 std::string temp = std::to_string(i);
59 Slice key(temp);
60 Slice value(temp);
61 ASSERT_OK(db_->Put(options, key, value));
62 }
63 }

65 int numKeysInLevels(std::vector<LiveFileMetaData>& metadata,
66 std::vector<int>* keysperlevel = nullptr) {
67 if (keysperlevel != nullptr) {
68 keysperlevel->resize(numlevels_);
69 }
70
71 int numKeys = 0;
72 for (size_t i = 0; i < metadata.size(); i++) {
73 int startkey = atoi(metadata[i].smallestkey.c_str());
74 int endkey = atoi(metadata[i].largestkey.c_str());
75 int numkeysinfile = (endkey - startkey + 1);
76 numKeys += numkeysinfile;
77 if (keysperlevel != nullptr) {
78 (*keysperlevel)[(int)metadata[i].level] += numkeysinfile;
79 }
80 fprintf(stderr, "level %d name %s smallest %s largest %s\n",
81 metadata[i].level, metadata[i].name.c_str(),
82 metadata[i].smallestkey.c_str(), metadata[i].largestkey.c_str());
83 }
84 return numKeys;
85 }

101 void CheckFileTypeCounts(const std::string& dir, int required_log,
102 int required_sst, int required_manifest) {
103 std::vector<std::string> filenames;
104 ASSERT_OK(env_->GetChildren(dir, &filenames));
105
106 int log_cnt = 0, sst_cnt = 0, manifest_cnt = 0;
107 for (const auto& file : filenames) {
108 uint64_t number;
109 FileType type;
110 if (ParseFileName(file, &number, &type)) {
111 log_cnt += (type == kWalFile);
112 sst_cnt += (type == kTableFile);
113 manifest_cnt += (type == kDescriptorFile);
114 }
115 }
116 if (required_log >= 0) {
117 ASSERT_EQ(required_log, log_cnt);
118 }
119 if (required_sst >= 0) {
120 ASSERT_EQ(required_sst, sst_cnt);
121 }
122 if (required_manifest >= 0) {
123 ASSERT_EQ(required_manifest, manifest_cnt);
124 }
125 }

548 // Should not succeed because live logs are not allowed to be deleted
549 std::unique_ptr<LogFile> alive_log = std::move(logfiles.back());
550 ASSERT_EQ(alive_log->Type(), kAliveLogFile);

555 ASSERT_OK(env_->FileExists(wal_dir_ + "/" + alive_log->PathName()));
556 logfiles.clear();
557

559 // Call Flush again to flush out memtable and move alive log to archived log
560 // and try to delete the archived log file
561 FlushOptions fopts;
562 ASSERT_OK(db_->Flush(fopts));
563 AddKeys(10, 0);
564 ASSERT_OK(db_->Flush(fopts));

566 ASSERT_GT(logfiles.size(), 0UL);
567 std::unique_ptr<LogFile> archived_log = std::move(logfiles.front());
568 ASSERT_EQ(archived_log->Type(), kArchivedLogFile);
569 ASSERT_OK(env_->FileExists(wal_dir_ + "/" + archived_log->PathName()));
570 fprintf(stdout, "Deleting archived log file %s\n",
571 archived_log->PathName().c_str());
572 ASSERT_OK(db_->DeleteFile(archived_log->PathName()));
573 ASSERT_TRUE(
574 env_->FileExists(wal_dir_ + "/" + archived_log->PathName()).IsNotFound());
