27// When CacheWithSecondaryAdapter is constructed with the distribute_cache_res
28// parameter set to true, it manages the entire memory budget across the
29// primary and secondary cache. The secondary cache is assumed to be in
30// memory, such as the CompressedSecondaryCache. When a placeholder entry
31// is inserted by a CacheReservationManager instance to reserve memory,
32// the CacheWithSecondaryAdapter ensures that the reservation is distributed
33// proportionally across the primary/secondary caches.
34//
35// The primary block cache is initially sized to the sum of the primary cache
36// budget + teh secondary cache budget, as follows -
37// |--------- Primary Cache Configured Capacity -----------|
38// |---Secondary Cache Budget----|----Primary Cache Budget-----|
39//
40// A ConcurrentCacheReservationManager member in the CacheWithSecondaryAdapter,
41// pri_cache_res_,
42// is used to help with tracking the distribution of memory reservations.
43// Initially, it accounts for the entire secondary cache budget as a
44// reservation against the primary cache. This shrinks the usable capacity of
45// the primary cache to the budget that the user originally desired.
46//
47// |--Reservation for Sec Cache--|-Pri Cache Usable Capacity---|
48//
49// When a reservation placeholder is inserted into the adapter, it is inserted
50// directly into the primary cache. This means the entire charge of the
51// placeholder is counted against the primary cache. To compensate and count
52// a portion of it against the secondary cache, the secondary cache Deflate()
53// method is called to shrink it. Since the Deflate() causes the secondary
54// actual usage to shrink, it is refelcted here by releasing an equal amount
55// from the pri_cache_res_ reservation. The Deflate() in the secondary cache
56// can be, but is not required to be, implemented using its own cache
57// reservation manager.
58//
59// For example, if the pri/sec ratio is 70/30, and the combined capacity is
60// 100MB, the intermediate and final state after inserting a reservation
61// placeholder for 10MB would be as follows -
62//
63// |-Reservation for Sec Cache-|-Pri Cache Usable Capacity-|---R---|
64// 1. After inserting the placeholder in primary
65// |------- 30MB -------------|------- 60MB -------------|-10MB--|
66// 2. After deflating the secondary and adjusting the reservation for
67// secondary against the primary
68// |------- 27MB -------------|------- 63MB -------------|-10MB--|
69//
70// Likewise, when the user inserted placeholder is released, the secondary
71// cache Inflate() method is called to grow it, and the pri_cache_res_
72// reservation is increased by an equal amount.
73//
74// Another way of implementing this would have been to simply split the user
75// reservation into primary and seconary components. However, this would
76// require allocating a structure to track the associated secondary cache
77// reservation, which adds some complexity and overhead.
78//
79CacheWithSecondaryAdapter::CacheWithSecondaryAdapter(
80 std::shared_ptr<Cache> target,
81 std::shared_ptr<SecondaryCache> secondary_cache,
82 TieredAdmissionPolicy adm_policy, bool distribute_cache_res)
83 : CacheWrapper(std::move(target)),
84 secondary_cache_(std::move(secondary_cache)),
85 adm_policy_(adm_policy),
86 distribute_cache_res_(distribute_cache_res),
87 placeholder_usage_(0),
88 reserved_usage_(0),
89 sec_reserved_(0) {
90 target_->SetEvictionCallback(
91 [this](const Slice& key, Handle* handle, bool was_hit) {
92 return EvictionHandler(key, handle, was_hit);
93 });
94 if (distribute_cache_res_) {
95 size_t sec_capacity = 0;
96 pri_cache_res_ = std::make_shared<ConcurrentCacheReservationManager>(
97 std::make_shared<CacheReservationManagerImpl<CacheEntryRole::kMisc>>(
98 target_));
99 Status s = secondary_cache_->GetCapacity(sec_capacity);
100 assert(s.ok());
101 // Initially, the primary cache is sized to uncompressed cache budget plsu
102 // compressed secondary cache budget. The secondary cache budget is then
103 // taken away from the primary cache through cache reservations. Later,
104 // when a placeholder entry is inserted by the caller, its inserted
105 // into the primary cache and the portion that should be assigned to the
106 // secondary cache is freed from the reservation.
107 s = pri_cache_res_->UpdateCacheReservation(sec_capacity);
108 assert(s.ok());
109 sec_cache_res_ratio_ = (double)sec_capacity / target_->GetCapacity();
110 }
111}

113CacheWithSecondaryAdapter::~CacheWithSecondaryAdapter() {
114 // `*this` will be destroyed before `*target_`, so we have to prevent
115 // use after free
116 target_->SetEvictionCallback({});
117#ifndef NDEBUG
118 if (distribute_cache_res_) {
119 size_t sec_capacity = 0;
120 Status s = secondary_cache_->GetCapacity(sec_capacity);
121 assert(s.ok());
122 assert(placeholder_usage_ == 0);
123 assert(reserved_usage_ == 0);

125 }
126#endif // NDEBUG

129bool CacheWithSecondaryAdapter::EvictionHandler(const Slice& key,
130 Handle* handle, bool was_hit) {
131 auto helper = GetCacheItemHelper(handle);
132 if (helper->IsSecondaryCacheCompatible() &&
133 adm_policy_ != TieredAdmissionPolicy::kAdmPolicyThreeQueue) {
134 auto obj = target_->Value(handle);
135 // Ignore dummy entry
136 if (obj != kDummyObj) {
137 bool force = false;
138 if (adm_policy_ == TieredAdmissionPolicy::kAdmPolicyAllowCacheHits) {
139 force = was_hit;
140 } else if (adm_policy_ == TieredAdmissionPolicy::kAdmPolicyAllowAll) {
141 force = true;
142 }
143 // Spill into secondary cache.
144 secondary_cache_->Insert(key, obj, helper, force).PermitUncheckedError();
145 }
146 }
147 // Never takes ownership of obj
148 return false;
149}

151bool CacheWithSecondaryAdapter::ProcessDummyResult(Cache::Handle** handle,
152 bool erase) {
153 if (*handle && target_->Value(*handle) == kDummyObj) {
154 target_->Release(*handle, erase);
155 *handle = nullptr;
156 return true;
157 } else {
158 return false;

169Cache::Handle* CacheWithSecondaryAdapter::Promote(
170 std::unique_ptr<SecondaryCacheResultHandle>&& secondary_handle,
171 const Slice& key, const CacheItemHelper* helper, Priority priority,
172 Statistics* stats, bool found_dummy_entry, bool kept_in_sec_cache) {
173 assert(secondary_handle->IsReady());
174
175 ObjectPtr obj = secondary_handle->Value();
176 if (!obj) {
177 // Nothing found.
178 return nullptr;
179 }
180 // Found something.
181 switch (helper->role) {
182 case CacheEntryRole::kFilterBlock:
183 RecordTick(stats, SECONDARY_CACHE_FILTER_HITS);
184 break;
185 case CacheEntryRole::kIndexBlock:
186 RecordTick(stats, SECONDARY_CACHE_INDEX_HITS);
187 break;
188 case CacheEntryRole::kDataBlock:
189 RecordTick(stats, SECONDARY_CACHE_DATA_HITS);
190 break;
191 default:
192 break;
193 }
194 PERF_COUNTER_ADD(secondary_cache_hit_count, 1);
195 RecordTick(stats, SECONDARY_CACHE_HITS);
196
197 // Note: SecondaryCache::Size() is really charge (from the CreateCallback)
198 size_t charge = secondary_handle->Size();
199 Handle* result = nullptr;
200 // Insert into primary cache, possibly as a standalone+dummy entries.
201 if (secondary_cache_->SupportForceErase() && !found_dummy_entry) {
202 // Create standalone and insert dummy
203 // Allow standalone to be created even if cache is full, to avoid
204 // reading the entry from storage.
205 result =
206 CreateStandalone(key, obj, helper, charge, /*allow_uncharged*/ true);
207 assert(result);
208 PERF_COUNTER_ADD(block_cache_standalone_handle_count, 1);
209
210 // Insert dummy to record recent use
211 // TODO: try to avoid case where inserting this dummy could overwrite a
212 // regular entry
213 Status s = Insert(key, kDummyObj, &kNoopCacheItemHelper, /*charge=*/0,
214 /*handle=*/nullptr, priority);
215 s.PermitUncheckedError();
216 // Nothing to do or clean up on dummy insertion failure
217 } else {
218 // Insert regular entry into primary cache.
219 // Don't allow it to spill into secondary cache again if it was kept there.
220 Status s = Insert(
221 key, obj, kept_in_sec_cache ? helper->without_secondary_compat : helper,
222 charge, &result, priority);
223 if (s.ok()) {
224 assert(result);
225 PERF_COUNTER_ADD(block_cache_real_handle_count, 1);
226 } else {
227 // Create standalone result instead, even if cache is full, to avoid
228 // reading the entry from storage.
229 result =
230 CreateStandalone(key, obj, helper, charge, /*allow_uncharged*/ true);
231 assert(result);
232 PERF_COUNTER_ADD(block_cache_standalone_handle_count, 1);
233 }
234 }
235 return result;
236}

238Status CacheWithSecondaryAdapter::Insert(const Slice& key, ObjectPtr value,
239 const CacheItemHelper* helper,
240 size_t charge, Handle** handle,
241 Priority priority,
242 const Slice& compressed_value,
243 CompressionType type) {
244 Status s = target_->Insert(key, value, helper, charge, handle, priority);
245 if (s.ok() && value == nullptr && distribute_cache_res_ && handle) {
246 charge = target_->GetCharge(*handle);
247
248 MutexLock l(&cache_res_mutex_);
249 placeholder_usage_ += charge;
250 // Check if total placeholder reservation is more than the overall
251 // cache capacity. If it is, then we don't try to charge the
252 // secondary cache because we don't want to overcharge it (beyond
253 // its capacity).
254 // In order to make this a bit more lightweight, we also check if
255 // the difference between placeholder_usage_ and reserved_usage_ is
256 // atleast kReservationChunkSize and avoid any adjustments if not.
257 if ((placeholder_usage_ <= target_->GetCapacity()) &&
258 ((placeholder_usage_ - reserved_usage_) >= kReservationChunkSize)) {
259 reserved_usage_ = placeholder_usage_ & ~(kReservationChunkSize - 1);
260 size_t new_sec_reserved =
261 static_cast<size_t>(reserved_usage_ * sec_cache_res_ratio_);
262 size_t sec_charge = new_sec_reserved - sec_reserved_;
263 s = secondary_cache_->Deflate(sec_charge);
264 assert(s.ok());
265 s = pri_cache_res_->UpdateCacheReservation(sec_charge,
266 /*increase=*/false);
267 assert(s.ok());
268 sec_reserved_ += sec_charge;
269 }
270 }
271 // Warm up the secondary cache with the compressed block. The secondary
272 // cache may choose to ignore it based on the admission policy.
273 if (value != nullptr && !compressed_value.empty() &&
274 adm_policy_ == TieredAdmissionPolicy::kAdmPolicyThreeQueue &&
275 helper->IsSecondaryCacheCompatible()) {
276 Status status = secondary_cache_->InsertSaved(key, compressed_value, type);
277 assert(status.ok() || status.IsNotSupported());
278 }
279
280 return s;
281}

283Cache::Handle* CacheWithSecondaryAdapter::Lookup(const Slice& key,
284 const CacheItemHelper* helper,
285 CreateContext* create_context,
286 Priority priority,
287 Statistics* stats) {
288 // NOTE: we could just StartAsyncLookup() and Wait(), but this should be a bit
289 // more efficient
290 Handle* result =
291 target_->Lookup(key, helper, create_context, priority, stats);
292 bool secondary_compatible = helper && helper->IsSecondaryCacheCompatible();
293 bool found_dummy_entry =
294 ProcessDummyResult(&result, /*erase=*/secondary_compatible);
295 if (!result && secondary_compatible) {
296 // Try our secondary cache
297 bool kept_in_sec_cache = false;
298 std::unique_ptr<SecondaryCacheResultHandle> secondary_handle =
299 secondary_cache_->Lookup(key, helper, create_context, /*wait*/ true,
300 found_dummy_entry, stats,
301 /*out*/ kept_in_sec_cache);
302 if (secondary_handle) {
303 result = Promote(std::move(secondary_handle), key, helper, priority,
304 stats, found_dummy_entry, kept_in_sec_cache);
305 }
306 }
307 return result;
308}

310bool CacheWithSecondaryAdapter::Release(Handle* handle,
311 bool erase_if_last_ref) {
312 if (erase_if_last_ref) {
313 ObjectPtr v = target_->Value(handle);
314 if (v == nullptr && distribute_cache_res_) {
315 size_t charge = target_->GetCharge(handle);
316
317 MutexLock l(&cache_res_mutex_);
318 placeholder_usage_ -= charge;
319 // Check if total placeholder reservation is more than the overall
320 // cache capacity. If it is, then we do nothing as reserved_usage_ must
321 // be already maxed out
322 if ((placeholder_usage_ <= target_->GetCapacity()) &&
323 (placeholder_usage_ < reserved_usage_)) {
324 // Adjust reserved_usage_ in chunks of kReservationChunkSize, so
325 // we don't hit this slow path too often.
326 reserved_usage_ = placeholder_usage_ & ~(kReservationChunkSize - 1);
327 size_t new_sec_reserved =
328 static_cast<size_t>(reserved_usage_ * sec_cache_res_ratio_);
329 size_t sec_charge = sec_reserved_ - new_sec_reserved;
330 Status s = secondary_cache_->Inflate(sec_charge);
331 assert(s.ok());
332 s = pri_cache_res_->UpdateCacheReservation(sec_charge,
333 /*increase=*/true);
334 assert(s.ok());
335 sec_reserved_ -= sec_charge;
336 }
337 }
338 }
339 return target_->Release(handle, erase_if_last_ref);
340}

366void CacheWithSecondaryAdapter::StartAsyncLookup(

368 target_->StartAsyncLookup(async_handle);
369 if (!async_handle.IsPending()) {

383void CacheWithSecondaryAdapter::WaitAll(AsyncLookupHandle* async_handles,
384 size_t count) {
385 if (count == 0) {
386 // Nothing to do
387 return;
388 }
389 // Requests that are pending on *my* secondary cache, at the start of this
390 // function
391 std::vector<AsyncLookupHandle*> my_pending;
392 // Requests that are pending on an "inner" secondary cache (managed somewhere
393 // under target_), as of the start of this function
394 std::vector<AsyncLookupHandle*> inner_pending;
395
396 // Initial accounting of pending handles, excluding those already handled
397 // by "outer" secondary caches. (See cur->pending_cache = nullptr.)
398 for (size_t i = 0; i < count; ++i) {
399 AsyncLookupHandle* cur = async_handles + i;
400 if (cur->pending_cache) {
401 assert(cur->IsPending());
402 assert(cur->helper);
403 assert(cur->helper->IsSecondaryCacheCompatible());
404 if (cur->pending_cache == secondary_cache_.get()) {
405 my_pending.push_back(cur);
406 // Mark as "to be handled by this caller"
407 cur->pending_cache = nullptr;
408 } else {
409 // Remember as potentially needing a lookup in my secondary
410 inner_pending.push_back(cur);
411 }
412 }
413 }
414
415 // Wait on inner-most cache lookups first
416 // TODO with stacked secondaries: because we are not using proper
417 // async/await constructs here yet, there is a false synchronization point
418 // here where all the results at one level are needed before initiating
419 // any lookups at the next level. Probably not a big deal, but worth noting.
420 if (!inner_pending.empty()) {
421 target_->WaitAll(async_handles, count);
422 }
423
424 // For those that failed to find something, convert to lookup in my
425 // secondary cache.
426 for (AsyncLookupHandle* cur : inner_pending) {
427 if (cur->Result() == nullptr) {
428 // Not found, try my secondary
429 StartAsyncLookupOnMySecondary(*cur);
430 if (cur->IsPending()) {
431 assert(cur->pending_cache == secondary_cache_.get());
432 my_pending.push_back(cur);
433 // Mark as "to be handled by this caller"
434 cur->pending_cache = nullptr;
435 }
436 }
437 }
438
439 // Wait on all lookups on my secondary cache
440 {
441 std::vector<SecondaryCacheResultHandle*> my_secondary_handles;
442 for (AsyncLookupHandle* cur : my_pending) {
443 my_secondary_handles.push_back(cur->pending_handle);
444 }
445 secondary_cache_->WaitAll(std::move(my_secondary_handles));
446 }
447
448 // Process results
449 for (AsyncLookupHandle* cur : my_pending) {
450 std::unique_ptr<SecondaryCacheResultHandle> secondary_handle(
451 cur->pending_handle);
452 cur->pending_handle = nullptr;
453 cur->result_handle = Promote(
454 std::move(secondary_handle), cur->key, cur->helper, cur->priority,
455 cur->stats, cur->found_dummy_entry, cur->kept_in_sec_cache);
456 assert(cur->pending_cache == nullptr);
457 }
458}

467const char* CacheWithSecondaryAdapter::Name() const {
468 if (distribute_cache_res_) {
469 return kTieredCacheName;
470 } else {
471 // To the user, at least for now, configure the underlying cache with
472 // a secondary cache. So we pretend to be that cache
473 return target_->Name();
474 }
475}

477// Update the total cache capacity. If we're distributing cache reservations
478// to both primary and secondary, then update the pri_cache_res_reservation
479// as well. At the moment, we don't have a good way of handling the case
480// where the new capacity < total cache reservations.
481void CacheWithSecondaryAdapter::SetCapacity(size_t capacity) {
482 size_t sec_capacity = static_cast<size_t>(
483 capacity * (distribute_cache_res_ ? sec_cache_res_ratio_ : 0.0));
484 size_t old_sec_capacity = 0;
485
486 if (distribute_cache_res_) {
487 MutexLock m(&cache_res_mutex_);
488
489 Status s = secondary_cache_->GetCapacity(old_sec_capacity);
490 if (!s.ok()) {
491 return;
492 }
493 if (old_sec_capacity > sec_capacity) {
494 // We're shrinking the cache. We do things in the following order to
495 // avoid a temporary spike in usage over the configured capacity -
496 // 1. Lower the secondary cache capacity
497 // 2. Credit an equal amount (by decreasing pri_cache_res_) to the
498 // primary cache
499 // 3. Decrease the primary cache capacity to the total budget
500 s = secondary_cache_->SetCapacity(sec_capacity);
501 if (s.ok()) {
502 if (placeholder_usage_ > capacity) {
503 // Adjust reserved_usage_ down
504 reserved_usage_ = capacity & ~(kReservationChunkSize - 1);
505 }
506 size_t new_sec_reserved =
507 static_cast<size_t>(reserved_usage_ * sec_cache_res_ratio_);
508 s = pri_cache_res_->UpdateCacheReservation(
509 (old_sec_capacity - sec_capacity) -
510 (sec_reserved_ - new_sec_reserved),
511 /*increase=*/false);
512 sec_reserved_ = new_sec_reserved;
513 assert(s.ok());
514 target_->SetCapacity(capacity);
515 }
516 } else {
517 // We're expanding the cache. Do it in the following order to avoid
518 // unnecessary evictions -
519 // 1. Increase the primary cache capacity to total budget
520 // 2. Reserve additional memory in primary on behalf of secondary (by
521 // increasing pri_cache_res_ reservation)
522 // 3. Increase secondary cache capacity
523 target_->SetCapacity(capacity);
524 s = pri_cache_res_->UpdateCacheReservation(
525 sec_capacity - old_sec_capacity,
526 /*increase=*/true);
527 assert(s.ok());
528 s = secondary_cache_->SetCapacity(sec_capacity);
529 assert(s.ok());
530 }
531 } else {
532 // No cache reservation distribution. Just set the primary cache capacity.
533 target_->SetCapacity(capacity);
534 }
535}

542Status CacheWithSecondaryAdapter::GetSecondaryCachePinnedUsage(
543 size_t& size) const {
544 Status s;
545 if (distribute_cache_res_) {
546 MutexLock m(&cache_res_mutex_);
547 size_t capacity = 0;
548 s = secondary_cache_->GetCapacity(capacity);
549 if (s.ok()) {
550 size = capacity - pri_cache_res_->GetTotalMemoryUsed();
551 } else {
552 size = 0;
553 }
554 } else {
555 size = 0;
556 }
557 return s;
558}

560// Update the secondary/primary allocation ratio (remember, the primary
561// capacity is the total memory budget when distribute_cache_res_ is true).
562// When the ratio changes, we may accumulate some error in the calculations
563// for secondary cache inflate/deflate and pri_cache_res_ reservations.
564// This is due to the rounding of the reservation amount.
565//
566// We rely on the current pri_cache_res_ total memory used to estimate the
567// new secondary cache reservation after the ratio change. For this reason,
568// once the ratio is lowered to 0.0 (effectively disabling the secondary
569// cache and pri_cache_res_ total mem used going down to 0), we cannot
570// increase the ratio and re-enable it, We might remove this limitation
571// in the future.
572Status CacheWithSecondaryAdapter::UpdateCacheReservationRatio(
573 double compressed_secondary_ratio) {
574 if (!distribute_cache_res_) {
575 return Status::NotSupported();
576 }
577
578 MutexLock m(&cache_res_mutex_);
579 size_t pri_capacity = target_->GetCapacity();
580 size_t sec_capacity =
581 static_cast<size_t>(pri_capacity * compressed_secondary_ratio);
582 size_t old_sec_capacity;
583 Status s = secondary_cache_->GetCapacity(old_sec_capacity);
584 if (!s.ok()) {
585 return s;
586 }
587
588 // Calculate the new secondary cache reservation
589 // reserved_usage_ will never be > the cache capacity, so we don't
590 // have to worry about adjusting it here.
591 sec_cache_res_ratio_ = compressed_secondary_ratio;
592 size_t new_sec_reserved =
593 static_cast<size_t>(reserved_usage_ * sec_cache_res_ratio_);
594 if (sec_capacity > old_sec_capacity) {
595 // We're increasing the ratio, thus ending up with a larger secondary
596 // cache and a smaller usable primary cache capacity. Similar to
597 // SetCapacity(), we try to avoid a temporary increase in total usage
598 // beyond the configured capacity -
599 // 1. A higher secondary cache ratio means it gets a higher share of
600 // cache reservations. So first account for that by deflating the
601 // secondary cache
602 // 2. Increase pri_cache_res_ reservation to reflect the new secondary
603 // cache utilization (increase in capacity - increase in share of cache
604 // reservation)
605 // 3. Increase secondary cache capacity
606 s = secondary_cache_->Deflate(new_sec_reserved - sec_reserved_);
607 assert(s.ok());
608 s = pri_cache_res_->UpdateCacheReservation(
609 (sec_capacity - old_sec_capacity) - (new_sec_reserved - sec_reserved_),
610 /*increase=*/true);
611 assert(s.ok());
612 sec_reserved_ = new_sec_reserved;
613 s = secondary_cache_->SetCapacity(sec_capacity);
614 assert(s.ok());
615 } else {
616 // We're shrinking the ratio. Try to avoid unnecessary evictions -
617 // 1. Lower the secondary cache capacity
618 // 2. Decrease pri_cache_res_ reservation to relect lower secondary
619 // cache utilization (decrease in capacity - decrease in share of cache
620 // reservations)
621 // 3. Inflate the secondary cache to give it back the reduction in its
622 // share of cache reservations
623 s = secondary_cache_->SetCapacity(sec_capacity);
624 if (s.ok()) {
625 s = pri_cache_res_->UpdateCacheReservation(
626 (old_sec_capacity - sec_capacity) -
627 (sec_reserved_ - new_sec_reserved),
628 /*increase=*/false);
629 assert(s.ok());
630 s = secondary_cache_->Inflate(sec_reserved_ - new_sec_reserved);
631 assert(s.ok());
632 sec_reserved_ = new_sec_reserved;
633 }
634 }
635
636 return s;
637}

645std::shared_ptr<Cache> NewTieredCache(const TieredCacheOptions& _opts) {
646 if (!_opts.cache_opts) {
647 return nullptr;
648 }
649
650 TieredCacheOptions opts = _opts;
651 {
652 bool valid_adm_policy = true;
653
654 switch (_opts.adm_policy) {
655 case TieredAdmissionPolicy::kAdmPolicyAuto:
656 // Select an appropriate default policy
657 if (opts.adm_policy == TieredAdmissionPolicy::kAdmPolicyAuto) {
658 if (opts.nvm_sec_cache) {
659 opts.adm_policy = TieredAdmissionPolicy::kAdmPolicyThreeQueue;
660 } else {
661 opts.adm_policy = TieredAdmissionPolicy::kAdmPolicyPlaceholder;
662 }
663 }
664 break;
665 case TieredAdmissionPolicy::kAdmPolicyPlaceholder:
666 case TieredAdmissionPolicy::kAdmPolicyAllowCacheHits:
667 case TieredAdmissionPolicy::kAdmPolicyAllowAll:
668 if (opts.nvm_sec_cache) {
669 valid_adm_policy = false;
670 }
671 break;
672 case TieredAdmissionPolicy::kAdmPolicyThreeQueue:
673 if (!opts.nvm_sec_cache) {
674 valid_adm_policy = false;
675 }
676 break;
677 default:
678 valid_adm_policy = false;
679 }
680 if (!valid_adm_policy) {
681 return nullptr;
682 }
683 }
684
685 std::shared_ptr<Cache> cache;
686 if (opts.cache_type == PrimaryCacheType::kCacheTypeLRU) {
687 LRUCacheOptions cache_opts =
688 *(static_cast_with_check<LRUCacheOptions, ShardedCacheOptions>(
689 opts.cache_opts));
690 cache_opts.capacity = opts.total_capacity;
691 cache_opts.secondary_cache = nullptr;
692 cache = cache_opts.MakeSharedCache();
693 } else if (opts.cache_type == PrimaryCacheType::kCacheTypeHCC) {
694 HyperClockCacheOptions cache_opts =
695 *(static_cast_with_check<HyperClockCacheOptions, ShardedCacheOptions>(
696 opts.cache_opts));
697 cache_opts.capacity = opts.total_capacity;
698 cache_opts.secondary_cache = nullptr;
699 cache = cache_opts.MakeSharedCache();
700 } else {
701 return nullptr;
702 }
703 std::shared_ptr<SecondaryCache> sec_cache;
704 opts.comp_cache_opts.capacity = static_cast<size_t>(
705 opts.total_capacity * opts.compressed_secondary_ratio);
706 sec_cache = NewCompressedSecondaryCache(opts.comp_cache_opts);
707
708 if (opts.nvm_sec_cache) {
709 if (opts.adm_policy == TieredAdmissionPolicy::kAdmPolicyThreeQueue) {
710 sec_cache = std::make_shared<TieredSecondaryCache>(
711 sec_cache, opts.nvm_sec_cache,
712 TieredAdmissionPolicy::kAdmPolicyThreeQueue);
713 } else {
714 return nullptr;
715 }
716 }
717
718 return std::make_shared<CacheWithSecondaryAdapter>(
719 cache, sec_cache, opts.adm_policy, /*distribute_cache_res=*/true);
720}
721
722Status UpdateTieredCache(const std::shared_ptr<Cache>& cache,
723 int64_t total_capacity,
724 double compressed_secondary_ratio,
725 TieredAdmissionPolicy adm_policy) {
726 if (!cache || strcmp(cache->Name(), kTieredCacheName)) {
727 return Status::InvalidArgument();
728 }
729 CacheWithSecondaryAdapter* tiered_cache =
730 static_cast<CacheWithSecondaryAdapter*>(cache.get());
731
732 Status s;
733 if (total_capacity > 0) {
734 tiered_cache->SetCapacity(total_capacity);
735 }
736 if (compressed_secondary_ratio >= 0.0 && compressed_secondary_ratio <= 1.0) {
737 s = tiered_cache->UpdateCacheReservationRatio(compressed_secondary_ratio);
738 }
739 if (adm_policy < TieredAdmissionPolicy::kAdmPolicyMax) {
740 s = tiered_cache->UpdateAdmissionPolicy(adm_policy);
741 }
742 return s;
743}
