98void DBSecondaryTestBase::CheckFileTypeCounts(const std::string& dir,
99 int expected_log,
100 int expected_sst,
101 int expected_manifest) const {
102 std::vector<std::string> filenames;
103 ASSERT_OK(env_->GetChildren(dir, &filenames));
104
105 int log_cnt = 0, sst_cnt = 0, manifest_cnt = 0;
106 for (const auto& file : filenames) {
107 uint64_t number;
108 FileType type;
109 if (ParseFileName(file, &number, &type)) {
110 log_cnt += (type == kWalFile);
111 sst_cnt += (type == kTableFile);
112 manifest_cnt += (type == kDescriptorFile);
113 }
114 }
115 ASSERT_EQ(expected_log, log_cnt);
116 ASSERT_EQ(expected_sst, sst_cnt);
117 ASSERT_EQ(expected_manifest, manifest_cnt);
118}

505 return s;
506 }

509 return files_closed_.load(std::memory_order_relaxed);
510 }
