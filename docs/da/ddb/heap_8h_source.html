17// Binary heap implementation optimized for use in multi-way merge sort.
18// Comparison to std::priority_queue:
19// - In libstdc++, std::priority_queue::pop() usually performs just over logN
20// comparisons but never fewer.
21// - std::priority_queue does not have a replace-top operation, requiring a
22// pop+push. If the replacement element is the new top, this requires
23// around 2logN comparisons.
24// - This heap's pop() uses a "schoolbook" downheap which requires up to ~2logN
25// comparisons.
26// - This heap provides a replace_top() operation which requires [1, 2logN]
27// comparisons. When the replacement element is also the new top, this
28// takes just 1 or 2 comparisons.
29//
30// The last property can yield an order-of-magnitude performance improvement
31// when merge-sorting real-world non-random data. If the merge operation is
32// likely to take chunks of elements from the same input stream, only 1
33// comparison per element is needed. In RocksDB-land, this happens when
34// compacting a database where keys are not randomly distributed across L0
35// files but nearby keys are likely to be in the same L0 file.
36//
37// The container uses the same counterintuitive ordering as
38// std::priority_queue: the comparison operator is expected to provide the
39// less-than relation, but top() will return the maximum.
40
41template <typename T, typename Compare = std::less<T>>
42class BinaryHeap {
43 public:
44 BinaryHeap() {}
45 explicit BinaryHeap(Compare cmp) : cmp_(std::move(cmp)) {}
46
47 void push(const T& value) {
48 data_.push_back(value);
49 upheap(data_.size() - 1);
50 }
51
52 void push(T&& value) {
53 data_.push_back(std::move(value));
54 upheap(data_.size() - 1);
55 }
56
57 const T& top() const {
58 assert(!empty());
59 return data_.front();
60 }
61
62 void replace_top(const T& value) {
63 assert(!empty());
64 data_.front() = value;
65 downheap(get_root());
66 }
67
68 void replace_top(T&& value) {
69 assert(!empty());
70 data_.front() = std::move(value);
71 downheap(get_root());
72 }
73
74 void pop() {
75 assert(!empty());
76 if (data_.size() > 1) {
77 // Avoid self-move-assign, because it could cause problems with
78 // classes which are not prepared for this and it trips up the
79 // STL debugger when activated.
80 data_.front() = std::move(data_.back());
81 }
82 data_.pop_back();
83 if (!empty()) {
84 downheap(get_root());
85 } else {
86 reset_root_cmp_cache();
87 }
88 }
89
90 void swap(BinaryHeap& other) {
91 std::swap(cmp_, other.cmp_);
92 data_.swap(other.data_);
93 std::swap(root_cmp_cache_, other.root_cmp_cache_);
94 }
95
96 void clear() {
97 data_.clear();
98 reset_root_cmp_cache();
99 }
100
101 bool empty() const { return data_.empty(); }
102
103 size_t size() const { return data_.size(); }
104
105 void reset_root_cmp_cache() {
106 root_cmp_cache_ = std::numeric_limits<size_t>::max();
107 }
108
109 private:
110 static inline size_t get_root() { return 0; }
111 static inline size_t get_parent(size_t index) { return (index - 1) / 2; }
112 static inline size_t get_left(size_t index) { return 2 * index + 1; }
113 static inline size_t get_right(size_t index) { return 2 * index + 2; }
114
115 void upheap(size_t index) {
116 T v = std::move(data_[index]);
117 while (index > get_root()) {
118 const size_t parent = get_parent(index);
119 if (!cmp_(data_[parent], v)) {
120 break;
121 }
122 data_[index] = std::move(data_[parent]);
123 index = parent;
124 }
125 data_[index] = std::move(v);
126 reset_root_cmp_cache();
127 }
128
129 void downheap(size_t index) {
130 T v = std::move(data_[index]);
131
132 size_t picked_child = std::numeric_limits<size_t>::max();
133 while (1) {
134 const size_t left_child = get_left(index);
135 if (get_left(index) >= data_.size()) {
136 break;
137 }
138 const size_t right_child = left_child + 1;
139 assert(right_child == get_right(index));
140 picked_child = left_child;
141 if (index == 0 && root_cmp_cache_ < data_.size()) {
142 picked_child = root_cmp_cache_;
143 } else if (right_child < data_.size() &&
144 cmp_(data_[left_child], data_[right_child])) {
145 picked_child = right_child;
146 }
147 if (!cmp_(v, data_[picked_child])) {
148 break;
149 }
150 data_[index] = std::move(data_[picked_child]);
151 index = picked_child;
152 }
153
154 if (index == 0) {
155 // We did not change anything in the tree except for the value
156 // of the root node, left and right child did not change, we can
157 // cache that `picked_child` is the smallest child
158 // so next time we compare againist it directly
159 root_cmp_cache_ = picked_child;
160 } else {
161 // the tree changed, reset cache
162 reset_root_cmp_cache();
163 }
164
165 data_[index] = std::move(v);
166 }
167
168 Compare cmp_;
169 autovector<T> data_;
170 // Used to reduce number of cmp_ calls in downheap()
171 size_t root_cmp_cache_ = std::numeric_limits<size_t>::max();
172};
