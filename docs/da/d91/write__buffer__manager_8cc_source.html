21WriteBufferManager::WriteBufferManager(size_t _buffer_size,
22 std::shared_ptr<Cache> cache,
23 bool allow_stall)
24 : buffer_size_(_buffer_size),
25 mutable_limit_(buffer_size_ * 7 / 8),
26 memory_used_(0),
27 memory_active_(0),
28 cache_res_mgr_(nullptr),
29 allow_stall_(allow_stall),
30 stall_active_(false) {
31 if (cache) {
32 // Memtable's memory usage tends to fluctuate frequently
33 // therefore we set delayed_decrease = true to save some dummy entry
34 // insertion on memory increase right after memory decrease
35 cache_res_mgr_ = std::make_shared<
36 CacheReservationManagerImpl<CacheEntryRole::kWriteBuffer>>(
37 cache, true /* delayed_decrease */);
38 }
39}

56void WriteBufferManager::ReserveMem(size_t mem) {
57 if (cache_res_mgr_ != nullptr) {
58 ReserveMemWithCache(mem);
59 } else if (enabled()) {
60 memory_used_.fetch_add(mem, std::memory_order_relaxed);
61 }
62 if (enabled()) {
63 memory_active_.fetch_add(mem, std::memory_order_relaxed);
64 }
65}

67// Should only be called from write thread
68void WriteBufferManager::ReserveMemWithCache(size_t mem) {
69 assert(cache_res_mgr_ != nullptr);
70 // Use a mutex to protect various data structures. Can be optimized to a
71 // lock-free solution if it ends up with a performance bottleneck.
72 std::lock_guard<std::mutex> lock(cache_res_mgr_mu_);
73
74 size_t new_mem_used = memory_used_.load(std::memory_order_relaxed) + mem;
75 memory_used_.store(new_mem_used, std::memory_order_relaxed);
76 Status s = cache_res_mgr_->UpdateCacheReservation(new_mem_used);
77
78 // We absorb the error since WriteBufferManager is not able to handle
79 // this failure properly. Ideallly we should prevent this allocation
80 // from happening if this cache charging fails.
81 // [TODO] We'll need to improve it in the future and figure out what to do on
82 // error
83 s.PermitUncheckedError();
84}

92void WriteBufferManager::FreeMem(size_t mem) {
93 if (cache_res_mgr_ != nullptr) {
94 FreeMemWithCache(mem);
95 } else if (enabled()) {
96 memory_used_.fetch_sub(mem, std::memory_order_relaxed);
97 }
98 // Check if stall is active and can be ended.
99 MaybeEndWriteStall();
100}

102void WriteBufferManager::FreeMemWithCache(size_t mem) {
103 assert(cache_res_mgr_ != nullptr);
104 // Use a mutex to protect various data structures. Can be optimized to a
105 // lock-free solution if it ends up with a performance bottleneck.
106 std::lock_guard<std::mutex> lock(cache_res_mgr_mu_);
107 size_t new_mem_used = memory_used_.load(std::memory_order_relaxed) - mem;
108 memory_used_.store(new_mem_used, std::memory_order_relaxed);
109 Status s = cache_res_mgr_->UpdateCacheReservation(new_mem_used);
110

112 // this failure properly.
113 // [TODO] We'll need to improve it in the future and figure out what to do on
114 // error
115 s.PermitUncheckedError();

118void WriteBufferManager::BeginWriteStall(StallInterface* wbm_stall) {
119 assert(wbm_stall != nullptr);
120
121 // Allocate outside of the lock.
122 std::list<StallInterface*> new_node = {wbm_stall};
123
124 {
125 std::unique_lock<std::mutex> lock(mu_);
126 // Verify if the stall conditions are stil active.
127 if (ShouldStall()) {
128 stall_active_.store(true, std::memory_order_relaxed);
129 queue_.splice(queue_.end(), std::move(new_node));
130 }
131 }
132
133 // If the node was not consumed, the stall has ended already and we can signal
134 // the caller.
135 if (!new_node.empty()) {
136 new_node.front()->Signal();
137 }
138}

140// Called when memory is freed in FreeMem or the buffer size has changed.
141void WriteBufferManager::MaybeEndWriteStall() {
142 // Stall conditions have not been resolved.
143 if (allow_stall_.load(std::memory_order_relaxed) &&
144 IsStallThresholdExceeded()) {
145 return;
146 }
147
148 // Perform all deallocations outside of the lock.
149 std::list<StallInterface*> cleanup;
150
151 std::unique_lock<std::mutex> lock(mu_);
152 if (!stall_active_.load(std::memory_order_relaxed)) {
153 return; // Nothing to do.
154 }
155
156 // Unblock new writers.
157 stall_active_.store(false, std::memory_order_relaxed);
158
159 // Unblock the writers in the queue.
160 for (StallInterface* wbm_stall : queue_) {
161 wbm_stall->Signal();
162 }
163 cleanup = std::move(queue_);
164}

166void WriteBufferManager::RemoveDBFromQueue(StallInterface* wbm_stall) {
167 assert(wbm_stall != nullptr);
168
169 // Deallocate the removed nodes outside of the lock.
170 std::list<StallInterface*> cleanup;
171
172 if (enabled() && allow_stall_.load(std::memory_order_relaxed)) {
173 std::unique_lock<std::mutex> lock(mu_);
174 for (auto it = queue_.begin(); it != queue_.end();) {
175 auto next = std::next(it);
176 if (*it == wbm_stall) {
177 cleanup.splice(cleanup.end(), queue_, std::move(it));
178 }
179 it = next;
180 }
181 }
182 wbm_stall->Signal();
183}
