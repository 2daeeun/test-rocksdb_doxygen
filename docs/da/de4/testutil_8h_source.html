95// A test comparator which compare two strings in this way:
96// (1) first compare prefix of 8 bytes in alphabet order,
97// (2) if two strings share the same prefix, sort the other part of the string
98// in the reverse alphabet order.
99// This helps simulate the case of compounded key of [entity][timestamp] and
100// latest timestamp first.
101class SimpleSuffixReverseComparator : public Comparator {
102 public:
103 SimpleSuffixReverseComparator() {}
104 static const char* kClassName() { return "SimpleSuffixReverseComparator"; }
105 const char* Name() const override { return kClassName(); }
106
107 int Compare(const Slice& a, const Slice& b) const override {
108 Slice prefix_a = Slice(a.data(), 8);
109 Slice prefix_b = Slice(b.data(), 8);
110 int prefix_comp = prefix_a.compare(prefix_b);
111 if (prefix_comp != 0) {
112 return prefix_comp;
113 } else {
114 Slice suffix_a = Slice(a.data() + 8, a.size() - 8);
115 Slice suffix_b = Slice(b.data() + 8, b.size() - 8);
116 return -(suffix_a.compare(suffix_b));
117 }
118 }
119 void FindShortestSeparator(std::string* /*start*/,
120 const Slice& /*limit*/) const override {}
121
122 void FindShortSuccessor(std::string* /*key*/) const override {}
123};

142 explicit StringSink(Slice* reader_contents = nullptr)
143 : FSWritableFile(),
144 contents_(""),
145 reader_contents_(reader_contents),
146 last_flush_(0) {
147 if (reader_contents_ != nullptr) {
148 *reader_contents_ = Slice(contents_.data(), 0);
149 }
150 }
151
152 const std::string& contents() const { return contents_; }
153
154 IOStatus Truncate(uint64_t size, const IOOptions& /*opts*/,
155 IODebugContext* /*dbg*/) override {
156 contents_.resize(static_cast<size_t>(size));
157 return IOStatus::OK();
158 }
159 IOStatus Close(const IOOptions& /*opts*/, IODebugContext* /*dbg*/) override {
160 return IOStatus::OK();
161 }
162 IOStatus Flush(const IOOptions& /*opts*/, IODebugContext* /*dbg*/) override {
163 if (reader_contents_ != nullptr) {
164 assert(reader_contents_->size() <= last_flush_);
165 size_t offset = last_flush_ - reader_contents_->size();
166 *reader_contents_ =
167 Slice(contents_.data() + offset, contents_.size() - offset);
168 last_flush_ = contents_.size();
169 }
170
171 return IOStatus::OK();
172 }
173 IOStatus Sync(const IOOptions& /*opts*/, IODebugContext* /*dbg*/) override {
174 return IOStatus::OK();
175 }
176
177 using FSWritableFile::Append;
178 IOStatus Append(const Slice& slice, const IOOptions& /*opts*/,
179 IODebugContext* /*dbg*/) override {
180 contents_.append(slice.data(), slice.size());
181 return IOStatus::OK();
182 }
183 void Drop(size_t bytes) {
184 if (reader_contents_ != nullptr) {
185 contents_.resize(contents_.size() - bytes);
186 *reader_contents_ =
187 Slice(reader_contents_->data(), reader_contents_->size() - bytes);
188 last_flush_ = contents_.size();
189 }
190 }
191
192 uint64_t GetFileSize(const IOOptions& /*options*/,
193 IODebugContext* /*dbg*/) override {
194 return contents_.size();
195 }
196
197 private:
198 Slice* reader_contents_;
199 size_t last_flush_;
200};

304class StringSource : public FSRandomAccessFile {
305 public:
306 explicit StringSource(const Slice& contents, uint64_t uniq_id = 0,
307 bool mmap = false)
308 : contents_(contents.data(), contents.size()),
309 uniq_id_(uniq_id),
310 mmap_(mmap),
311 total_reads_(0) {}
312
313 virtual ~StringSource() {}
314
315 uint64_t Size() const { return contents_.size(); }
316
317 IOStatus Prefetch(uint64_t /*offset*/, size_t /*n*/,
318 const IOOptions& /*options*/,
319 IODebugContext* /*dbg*/) override {
320 // If we are using mmap_, it is equivalent to performing a prefetch
321 if (mmap_) {
322 return IOStatus::OK();
323 } else {
324 return IOStatus::NotSupported("Prefetch not supported");
325 }
326 }
327
328 IOStatus Read(uint64_t offset, size_t n, const IOOptions& /*opts*/,
329 Slice* result, char* scratch,
330 IODebugContext* /*dbg*/) const override {
331 total_reads_++;
332 if (offset > contents_.size()) {
333 return IOStatus::InvalidArgument("invalid Read offset");
334 }
335 if (offset + n > contents_.size()) {
336 n = contents_.size() - static_cast<size_t>(offset);
337 }
338 if (!mmap_) {
339 memcpy(scratch, &contents_[static_cast<size_t>(offset)], n);
340 *result = Slice(scratch, n);
341 } else {
342 *result = Slice(&contents_[static_cast<size_t>(offset)], n);
343 }
344 return IOStatus::OK();
345 }
346
347 size_t GetUniqueId(char* id, size_t max_size) const override {
348 if (max_size < 20) {
349 return 0;
350 }
351
352 char* rid = id;
353 rid = EncodeVarint64(rid, uniq_id_);
354 rid = EncodeVarint64(rid, 0);
355 return static_cast<size_t>(rid - id);
356 }
357
358 int total_reads() const { return total_reads_; }
359
360 void set_total_reads(int tr) { total_reads_ = tr; }
361
362 private:
363 std::string contents_;
364 uint64_t uniq_id_;
365 bool mmap_;
366 mutable int total_reads_;
367};

386class SleepingBackgroundTask {
387 public:
388 SleepingBackgroundTask()
389 : bg_cv_(&mutex_),
390 should_sleep_(true),
391 done_with_sleep_(false),
392 sleeping_(false) {}
393
394 ~SleepingBackgroundTask() {
395 MutexLock l(&mutex_);
396 should_sleep_ = false;
397 while (sleeping_) {
398 assert(!should_sleep_);
399 bg_cv_.SignalAll();
400 bg_cv_.Wait();
401 }
402 }
403
404 bool IsSleeping() {
405 MutexLock l(&mutex_);
406 return sleeping_;
407 }
408 void DoSleep() {
409 MutexLock l(&mutex_);
410 sleeping_ = true;
411 bg_cv_.SignalAll();
412 while (should_sleep_) {
413 bg_cv_.Wait();
414 }
415 sleeping_ = false;
416 done_with_sleep_ = true;
417 bg_cv_.SignalAll();
418 }
419 void WaitUntilSleeping() {
420 MutexLock l(&mutex_);
421 while (!sleeping_ || !should_sleep_) {
422 bg_cv_.Wait();
423 }
424 }
425 // Waits for the status to change to sleeping,
426 // otherwise times out.
427 // wait_time is in microseconds.
428 // Returns true when times out, false otherwise.
429 bool TimedWaitUntilSleeping(uint64_t wait_time);
430
431 void WakeUp() {
432 MutexLock l(&mutex_);
433 should_sleep_ = false;
434 bg_cv_.SignalAll();
435 }
436 void WaitUntilDone() {
437 MutexLock l(&mutex_);
438 while (!done_with_sleep_) {
439 bg_cv_.Wait();
440 }
441 }
442 // Similar to TimedWaitUntilSleeping.
443 // Waits until the task is done.
444 bool TimedWaitUntilDone(uint64_t wait_time);
445
446 bool WokenUp() {
447 MutexLock l(&mutex_);
448 return should_sleep_ == false;
449 }
450
451 void Reset() {
452 MutexLock l(&mutex_);
453 should_sleep_ = true;
454 done_with_sleep_ = false;
455 }
456
457 static void DoSleepTask(void* arg) {
458 static_cast<SleepingBackgroundTask*>(arg)->DoSleep();
459 }
460
461 private:
462 port::Mutex mutex_;
463 port::CondVar bg_cv_; // Signalled when background work finishes
464 bool should_sleep_;
465 bool done_with_sleep_;
466 bool sleeping_;
467};

508class SeqStringSource : public FSSequentialFile {
509 public:
510 SeqStringSource(const std::string& data, std::atomic<int>* read_count)
511 : data_(data), offset_(0), read_count_(read_count) {}
512 ~SeqStringSource() override {}
513 IOStatus Read(size_t n, const IOOptions& /*opts*/, Slice* result,
514 char* scratch, IODebugContext* /*dbg*/) override {
515 std::string output;
516 if (offset_ < data_.size()) {
517 n = std::min(data_.size() - offset_, n);
518 memcpy(scratch, data_.data() + offset_, n);
519 offset_ += n;
520 *result = Slice(scratch, n);
521 } else {
522 return IOStatus::InvalidArgument(
523 "Attempt to read when it already reached eof.");
524 }
525 (*read_count_)++;
526 return IOStatus::OK();
527 }
528
529 IOStatus Skip(uint64_t n) override {
530 if (offset_ >= data_.size()) {
531 return IOStatus::InvalidArgument(
532 "Attempt to read when it already reached eof.");
533 }
534 // TODO(yhchiang): Currently doesn't handle the overflow case.
535 offset_ += static_cast<size_t>(n);
536 return IOStatus::OK();
537 }
538
539 private:
540 std::string data_;
541 size_t offset_;
542 std::atomic<int>* read_count_;
543};
