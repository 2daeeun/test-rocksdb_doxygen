19// The classes and functions in this header file is used for dumping out the
20// blocks in a block cache, storing or transfering the blocks to another
21// destination host, and load these blocks to the secondary cache at destination
22// host.
23// NOTE that: The classes, functions, and data structures are EXPERIMENTAL! They
24// my be changed in the future when the development continues.
25
26// The major and minor version number of the data format to be stored/trandfered
27// via CacheDumpWriter and read out via CacheDumpReader
28static const int kCacheDumpMajorVersion = 0;
29static const int kCacheDumpMinorVersion = 1;

31// NOTE that: this class is EXPERIMENTAL! May be changed in the future!
32// This is an abstract class to write or transfer the data that is created by
33// CacheDumper. We pack one block with its block type, dump time, block key in
34// the block cache, block len, block crc32c checksum and block itself as a unit
35// and it is stored via WritePacket. Before we call WritePacket, we must call
36// WriteMetadata once, which stores the sequence number, block unit checksum,
37// and block unit size.
38// We provide file based CacheDumpWriter to store the metadata and its package
39// sequentially in a file as the defualt implementation. Users can implement
40// their own CacheDumpWriter to store/transfer the data. For example, user can
41// create a subclass which transfer the metadata and package on the fly.
42class CacheDumpWriter {
43 public:
44 virtual ~CacheDumpWriter() = default;
45
46 // Called ONCE before the calls to WritePacket
47 virtual IOStatus WriteMetadata(const Slice& metadata) = 0;
48 virtual IOStatus WritePacket(const Slice& data) = 0;
49 virtual IOStatus Close() = 0;
50};
51
52// NOTE that: this class is EXPERIMENTAL! May be changed in the future!
53// This is an abstract class to read or receive the data that is stored
54// or transfered by CacheDumpWriter. Note that, ReadMetadata must be called
55// once before we call a ReadPacket.
56class CacheDumpReader {
57 public:
58 virtual ~CacheDumpReader() = default;
59 // Called ONCE before the calls to ReadPacket
60 virtual IOStatus ReadMetadata(std::string* metadata) = 0;
61 // Sets data to empty string on EOF
62 virtual IOStatus ReadPacket(std::string* data) = 0;
63 // (Close not needed)
64};

66// CacheDumpOptions is the option for CacheDumper and CacheDumpedLoader. Any
67// dump or load process related control variables can be added here.
68struct CacheDumpOptions {
69 SystemClock* clock;
70 // Deadline for dumper or loader in microseconds
71 std::chrono::microseconds deadline = std::chrono::microseconds::zero();
72 // Max size bytes for dumper or loader
73 uint64_t max_size_bytes = 0;
74};

76// NOTE that: this class is EXPERIMENTAL! May be changed in the future!
77// This the class to dump out the block in the block cache, store/transfer them
78// via CacheDumpWriter. In order to dump out the blocks belonging to a certain
79// DB or a list of DB (block cache can be shared by many DB), user needs to call
80// SetDumpFilter to specify a list of DB to filter out the blocks that do not
81// belong to those DB.
82// A typical use case is: when we migrate a DB instance from host A to host B.
83// We need to reopen the DB at host B after all the files are copied to host B.
84// At this moment, the block cache at host B does not have any block from this
85// migrated DB. Therefore, the read performance can be low due to cache warm up.
86// By using CacheDumper before we shut down the DB at host A and using
87// CacheDumpedLoader at host B before we reopen the DB, we can warmup the cache
88// ahead. This function can be used in other use cases also.
89class CacheDumper {
90 public:
91 virtual ~CacheDumper() = default;
92 // Only dump the blocks in the block cache that belong to the DBs in this list
93 virtual Status SetDumpFilter(std::vector<DB*> db_list) {
94 (void)db_list;
95 return Status::NotSupported("SetDumpFilter is not supported");
96 }
97 // The main function to dump out all the blocks that satisfy the filter
98 // condition from block cache to a certain CacheDumpWriter in one shot. This
99 // process may take some time.
100 virtual IOStatus DumpCacheEntriesToWriter() {
101 return IOStatus::NotSupported("DumpCacheEntriesToWriter is not supported");
102 }
103};

105// NOTE that: this class is EXPERIMENTAL! May be changed in the future!
106// This is the class to load the dumped blocks to the destination cache. For now
107// we only load the blocks to the SecondaryCache. In the future, we may plan to
108// support loading to the block cache.
109class CacheDumpedLoader {
110 public:
111 virtual ~CacheDumpedLoader() = default;
112 virtual IOStatus RestoreCacheEntriesToSecondaryCache() {
113 return IOStatus::NotSupported(
114 "RestoreCacheEntriesToSecondaryCache is not supported");
115 }
116};
