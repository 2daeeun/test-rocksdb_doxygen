14// TransactionDBMutex and TransactionDBCondVar APIs allows applications to
15// implement custom mutexes and condition variables to be used by a
16// TransactionDB when locking keys.
17//
18// To open a TransactionDB with a custom TransactionDBMutexFactory, set
19// TransactionDBOptions.custom_mutex_factory.
20class TransactionDBMutex {
21 public:
22 virtual ~TransactionDBMutex() {}
23
24 // Attempt to acquire lock. Return OK on success, or other Status on failure.
25 // If returned status is OK, TransactionDB will eventually call UnLock().
26 virtual Status Lock() = 0;
27
28 // Attempt to acquire lock. If timeout is non-negative, operation may be
29 // failed after this many microseconds.
30 // Returns OK on success,
31 // TimedOut if timed out,
32 // or other Status on failure.
33 // If returned status is OK, TransactionDB will eventually call UnLock().
34 virtual Status TryLockFor(int64_t timeout_time) = 0;
35
36 // Unlock Mutex that was successfully locked by Lock() or TryLockUntil()
37 virtual void UnLock() = 0;
38};
39
40class TransactionDBCondVar {
41 public:
42 virtual ~TransactionDBCondVar() {}
43
44 // Block current thread until condition variable is notified by a call to
45 // Notify() or NotifyAll(). Wait() will be called with mutex locked.
46 // Returns OK if notified.
47 // Returns non-OK if TransactionDB should stop waiting and fail the operation.
48 // May return OK spuriously even if not notified.
49 virtual Status Wait(std::shared_ptr<TransactionDBMutex> mutex) = 0;
50
51 // Block current thread until condition variable is notified by a call to
52 // Notify() or NotifyAll(), or if the timeout is reached.
53 // Wait() will be called with mutex locked.
54 //
55 // If timeout is non-negative, operation should be failed after this many
56 // microseconds.
57 // If implementing a custom version of this class, the implementation may
58 // choose to ignore the timeout.
59 //
60 // Returns OK if notified.
61 // Returns TimedOut if timeout is reached.
62 // Returns other status if TransactionDB should otherwise stop waiting and
63 // fail the operation.
64 // May return OK spuriously even if not notified.
65 virtual Status WaitFor(std::shared_ptr<TransactionDBMutex> mutex,
66 int64_t timeout_time) = 0;
67
68 // If any threads are waiting on *this, unblock at least one of the
69 // waiting threads.
70 virtual void Notify() = 0;
71
72 // Unblocks all threads waiting on *this.
73 virtual void NotifyAll() = 0;
74};
