142int LDBCommandRunner::RunCommand(
143 int argc, char const* const* argv, Options options,
144 const LDBOptions& ldb_options,
145 const std::vector<ColumnFamilyDescriptor>* column_families) {
146 if (argc <= 2) {
147 if (argc <= 1) {
148 PrintHelp(ldb_options, argv[0], /*to_stderr*/ true);
149 return 1;
150 } else if (std::string(argv[1]) == "--version") {
151 printf("ldb from RocksDB %d.%d.%d\n", ROCKSDB_MAJOR, ROCKSDB_MINOR,
152 ROCKSDB_PATCH);
153 return 0;
154 } else if (std::string(argv[1]) == "--help") {
155 PrintHelp(ldb_options, argv[0], /*to_stderr*/ false);
156 return 0;
157 } else {
158 PrintHelp(ldb_options, argv[0], /*to_stderr*/ true);
159 return 1;
160 }
161 }
162
163 LDBCommand* cmdObj = LDBCommand::InitFromCmdLineArgs(
164 argc, argv, options, ldb_options, column_families);
165 if (cmdObj == nullptr) {
166 fprintf(stderr, "Unknown command\n");
167 PrintHelp(ldb_options, argv[0], /*to_stderr*/ true);
168 return 1;
169 }
170
171 if (!cmdObj->ValidateCmdLineOptions()) {
172 return 1;
173 }
174
175 cmdObj->Run();
176 LDBCommandExecuteResult ret = cmdObj->GetExecuteState();
177 if (!ret.ToString().empty()) {
178 fprintf(stderr, "%s\n", ret.ToString().c_str());
179 }
180 delete cmdObj;
181
182 return ret.IsFailed() ? 1 : 0;
183}

189 ldb_options, column_families);
190 exit(error_code);
191}
