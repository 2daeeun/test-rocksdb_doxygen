44void PointLockTracker::Track(const PointLockRequest& r) {
45 auto& keys = tracked_keys_[r.column_family_id];
46 auto result = keys.try_emplace(r.key, r.seq);
47 auto it = result.first;
48 if (!result.second && r.seq < it->second.seq) {
49 // Now tracking this key with an earlier sequence number
50 it->second.seq = r.seq;
51 }
52 // else we do not update the seq. The smaller the tracked seq, the stronger it
53 // the guarantee since it implies from the seq onward there has not been a
54 // concurrent update to the key. So we update the seq if it implies stronger
55 // guarantees, i.e., if it is smaller than the existing tracked seq.
56
57 if (r.read_only) {
58 it->second.num_reads++;
59 } else {
60 it->second.num_writes++;
61 }
62
63 it->second.exclusive = it->second.exclusive || r.exclusive;
64}

66UntrackStatus PointLockTracker::Untrack(const PointLockRequest& r) {
67 auto cf_keys = tracked_keys_.find(r.column_family_id);
68 if (cf_keys == tracked_keys_.end()) {
69 return UntrackStatus::NOT_TRACKED;
70 }
71
72 auto& keys = cf_keys->second;
73 auto it = keys.find(r.key);
74 if (it == keys.end()) {
75 return UntrackStatus::NOT_TRACKED;
76 }
77
78 bool untracked = false;
79 auto& info = it->second;
80 if (r.read_only) {
81 if (info.num_reads > 0) {
82 info.num_reads--;
83 untracked = true;
84 }
85 } else {
86 if (info.num_writes > 0) {
87 info.num_writes--;
88 untracked = true;
89 }
90 }
91
92 bool removed = false;
93 if (info.num_reads == 0 && info.num_writes == 0) {
94 keys.erase(it);
95 if (keys.empty()) {
96 tracked_keys_.erase(cf_keys);
97 }
98 removed = true;
99 }
100
101 if (removed) {
102 return UntrackStatus::REMOVED;
103 }
104 if (untracked) {
105 return UntrackStatus::UNTRACKED;
106 }
107 return UntrackStatus::NOT_TRACKED;
108}

110void PointLockTracker::Merge(const LockTracker& tracker) {
111 const PointLockTracker& t = static_cast<const PointLockTracker&>(tracker);
112 for (const auto& cf_keys : t.tracked_keys_) {
113 ColumnFamilyId cf = cf_keys.first;
114 const auto& keys = cf_keys.second;
115
116 auto current_cf_keys = tracked_keys_.find(cf);
117 if (current_cf_keys == tracked_keys_.end()) {
118 tracked_keys_.emplace(cf_keys);
119 } else {
120 auto& current_keys = current_cf_keys->second;
121 for (const auto& key_info : keys) {
122 const std::string& key = key_info.first;
123 const TrackedKeyInfo& info = key_info.second;
124 // If key was not previously tracked, just copy the whole struct over.
125 // Otherwise, some merging needs to occur.
126 auto current_info = current_keys.find(key);
127 if (current_info == current_keys.end()) {
128 current_keys.emplace(key_info);
129 } else {
130 current_info->second.Merge(info);
131 }
132 }
133 }
134 }

137void PointLockTracker::Subtract(const LockTracker& tracker) {
138 const PointLockTracker& t = static_cast<const PointLockTracker&>(tracker);
139 for (const auto& cf_keys : t.tracked_keys_) {
140 ColumnFamilyId cf = cf_keys.first;
141 const auto& keys = cf_keys.second;
142
143 auto& current_keys = tracked_keys_.at(cf);
144 for (const auto& key_info : keys) {
145 const std::string& key = key_info.first;
146 const TrackedKeyInfo& info = key_info.second;
147 uint32_t num_reads = info.num_reads;
148 uint32_t num_writes = info.num_writes;
149
150 auto current_key_info = current_keys.find(key);
151 assert(current_key_info != current_keys.end());
152
153 // Decrement the total reads/writes of this key by the number of
154 // reads/writes done since the last SavePoint.
155 if (num_reads > 0) {
156 assert(current_key_info->second.num_reads >= num_reads);
157 current_key_info->second.num_reads -= num_reads;
158 }
159 if (num_writes > 0) {
160 assert(current_key_info->second.num_writes >= num_writes);
161 current_key_info->second.num_writes -= num_writes;
162 }
163 if (current_key_info->second.num_reads == 0 &&
164 current_key_info->second.num_writes == 0) {
165 current_keys.erase(current_key_info);
166 }
167 }
168 }

171LockTracker* PointLockTracker::GetTrackedLocksSinceSavePoint(
172 const LockTracker& save_point_tracker) const {
173 // Examine the number of reads/writes performed on all keys written
174 // since the last SavePoint and compare to the total number of reads/writes
175 // for each key.
176 LockTracker* t = new PointLockTracker();
177 const PointLockTracker& save_point_t =
178 static_cast<const PointLockTracker&>(save_point_tracker);
179 for (const auto& cf_keys : save_point_t.tracked_keys_) {
180 ColumnFamilyId cf = cf_keys.first;
181 const auto& keys = cf_keys.second;
182
183 auto& current_keys = tracked_keys_.at(cf);
184 for (const auto& key_info : keys) {
185 const std::string& key = key_info.first;
186 const TrackedKeyInfo& info = key_info.second;
187 uint32_t num_reads = info.num_reads;
188 uint32_t num_writes = info.num_writes;
189
190 auto current_key_info = current_keys.find(key);
191 assert(current_key_info != current_keys.end());
192 assert(current_key_info->second.num_reads >= num_reads);
193 assert(current_key_info->second.num_writes >= num_writes);
194
195 if (current_key_info->second.num_reads == num_reads &&
196 current_key_info->second.num_writes == num_writes) {
197 // All the reads/writes to this key were done in the last savepoint.
198 PointLockRequest r;
199 r.column_family_id = cf;
200 r.key = key;
201 r.seq = info.seq;
202 r.read_only = (num_writes == 0);
203 r.exclusive = info.exclusive;
204 t->Track(r);
205 }
206 }
207 }
208 return t;
209}

211PointLockStatus PointLockTracker::GetPointLockStatus(
212 ColumnFamilyId column_family_id, const std::string& key) const {
213 assert(IsPointLockSupported());
214 PointLockStatus status;
215 auto it = tracked_keys_.find(column_family_id);
216 if (it == tracked_keys_.end()) {
217 return status;
218 }
219
220 const auto& keys = it->second;
221 auto key_it = keys.find(key);
222 if (key_it == keys.end()) {
223 return status;
224 }
225
226 const TrackedKeyInfo& key_info = key_it->second;
227 status.locked = true;
228 status.exclusive = key_info.exclusive;
229 status.seq = key_info.seq;
230 return status;
231}
