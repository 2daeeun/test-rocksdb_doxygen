227// This class stores references to all the immutable memtables.
228// The memtables are flushed to L0 as soon as possible and in
229// any order. If there are more than one immutable memtable, their
230// flushes can occur concurrently. However, they are 'committed'
231// to the manifest in FIFO order to maintain correctness and
232// recoverability from a crash.
233//
234//
235// Other than imm_flush_needed and imm_trim_needed, this class is not
236// thread-safe and requires external synchronization (such as holding the db
237// mutex or being on the write thread.)

294 // Reset status of the given memtable list back to pending state so that
295 // they can get picked up again on the next round of flush.
296 //
297 // @param rollback_succeeding_memtables If true, will rollback adjacent
298 // younger memtables whose flush is completed. Specifically, suppose the
299 // current immutable memtables are M_0,M_1...M_N ordered from youngest to
300 // oldest. Suppose that the youngest memtable in `mems` is M_K. We will try to
301 // rollback M_K-1, M_K-2... until the first memtable whose flush is
302 // not completed. These are the memtables that would have been installed
303 // by this flush job if it were to succeed. This flag is currently used
304 // by non atomic_flush rollback.
305 // Note that we also do rollback in `write_manifest_cb` by calling
306 // `RemoveMemTablesOrRestoreFlags()`. There we rollback the entire batch so
307 // it is similar to what we do here with rollback_succeeding_memtables=true.

358 // Request a flush of all existing memtables to storage. This will
359 // cause future calls to IsFlushPending() to return true if this list is
360 // non-empty (regardless of the min_write_buffer_number_to_merge
361 // parameter). This flush request will persist until the next time
362 // PickMemtablesToFlush() is called.
363 void FlushRequested() {
364 flush_requested_ = true;
365 // If there are some memtables stored in imm() that don't trigger
366 // flush (eg: mempurge output memtable), then update imm_flush_needed.
367 // Note: if race condition and imm_flush_needed is set to true
368 // when there is num_flush_not_started_==0, then there is no
369 // impact whatsoever. Imm_flush_needed is only used in an assert
370 // in IsFlushPending().
371 if (num_flush_not_started_ > 0) {
372 imm_flush_needed.store(true, std::memory_order_release);
373 }

414 uint64_t GetLatestMemTableID(bool for_atomic_flush) const {
415 auto& memlist = current_->memlist_;
416 if (memlist.empty()) {
417 return 0;
418 }
419 if (for_atomic_flush) {
420 // Scan the memtable list from new to old
421 for (auto it = memlist.begin(); it != memlist.end(); ++it) {
422 ReadOnlyMemTable* m = *it;
423 if (m->atomic_flush_seqno_ != kMaxSequenceNumber) {
424 return m->GetID();
425 }
426 }
427 return 0;
428 }
429 return memlist.front()->GetID();
430 }

432 // DB mutex held.
433 // Gets the newest user-defined timestamp for the Memtables in ascending ID
434 // order, up to the `max_memtable_id`. Used by background flush job
435 // to check Memtables' eligibility for flush w.r.t retaining UDTs.
436 std::vector<Slice> GetTablesNewestUDT(uint64_t max_memtable_id) {
437 std::vector<Slice> newest_udts;
438 auto& memlist = current_->memlist_;
439 // Iterating through the memlist starting at the end, the vector<MemTable*>
440 // ret is filled with memtables already sorted in increasing MemTable ID.
441 for (auto it = memlist.rbegin(); it != memlist.rend(); ++it) {
442 ReadOnlyMemTable* m = *it;
443 if (m->GetID() > max_memtable_id) {
444 break;
445 }
446 newest_udts.push_back(m->GetNewestUDT());
447 }
448 return newest_udts;
449 }

451 void AssignAtomicFlushSeq(const SequenceNumber& seq) {
452 const auto& memlist = current_->memlist_;
453 // Scan the memtable list from new to old
454 for (auto it = memlist.begin(); it != memlist.end(); ++it) {
455 ReadOnlyMemTable* mem = *it;
456 if (mem->atomic_flush_seqno_ == kMaxSequenceNumber) {
457 mem->atomic_flush_seqno_ = seq;
458 } else {
459 // Earlier memtables must have been assigned a atomic flush seq, no
460 // need to continue scan.
461 break;
462 }
463 }
464 }
