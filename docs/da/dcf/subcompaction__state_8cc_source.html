31void SubcompactionState::Cleanup(Cache* cache) {
32 penultimate_level_outputs_.Cleanup();
33 compaction_outputs_.Cleanup();
34
35 if (!status.ok()) {
36 for (const auto& out : GetOutputs()) {
37 // If this file was inserted into the table cache then remove it here
38 // because this compaction was not committed. This is not strictly
39 // required because of a backstop TableCache::Evict() in
40 // PurgeObsoleteFiles() but is our opportunity to apply
41 // uncache_aggressiveness. TODO: instead, put these files into the
42 // VersionSet::obsolete_files_ pipeline so that they don't have to
43 // be picked up by scanning the DB directory.
44 TableCache::ReleaseObsolete(
45 cache, out.meta.fd.GetNumber(), nullptr /*handle*/,
46 compaction->mutable_cf_options()->uncache_aggressiveness);
47 }
48 }
49 // TODO: sub_compact.io_status is not checked like status. Not sure if thats
50 // intentional. So ignoring the io_status as of now.
51 io_status.PermitUncheckedError();
52}

54Slice SubcompactionState::SmallestUserKey() const {
55 if (has_penultimate_level_outputs_) {
56 Slice a = compaction_outputs_.SmallestUserKey();
57 Slice b = penultimate_level_outputs_.SmallestUserKey();
58 if (a.empty()) {
59 return b;
60 }
61 if (b.empty()) {
62 return a;
63 }
64 const Comparator* user_cmp =
65 compaction->column_family_data()->user_comparator();
66 if (user_cmp->Compare(a, b) > 0) {
67 return b;
68 } else {
69 return a;
70 }
71 } else {
72 return compaction_outputs_.SmallestUserKey();
73 }
74}

98Status SubcompactionState::AddToOutput(
99 const CompactionIterator& iter,
100 const CompactionFileOpenFunc& open_file_func,
101 const CompactionFileCloseFunc& close_file_func) {
102 // update target output first
103 is_current_penultimate_level_ = iter.output_to_penultimate_level();
104 current_outputs_ = is_current_penultimate_level_ ? &penultimate_level_outputs_
105 : &compaction_outputs_;
106 if (is_current_penultimate_level_) {
107 has_penultimate_level_outputs_ = true;
108 }
109
110 return Current().AddToOutput(iter, open_file_func, close_file_func);
111}
