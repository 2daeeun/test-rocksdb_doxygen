33IOStatus EncryptedSequentialFile::Read(size_t n, const IOOptions& options,
34 Slice* result, char* scratch,
35 IODebugContext* dbg) {
36 assert(scratch);
37 IOStatus io_s = file_->Read(n, options, result, scratch, dbg);
38 if (!io_s.ok()) {
39 return io_s;
40 }
41 {
42 PERF_TIMER_GUARD(decrypt_data_nanos);
43 io_s = status_to_io_status(
44 stream_->Decrypt(offset_, (char*)result->data(), result->size()));
45 }
46 if (io_s.ok()) {
47 offset_ += result->size(); // We've already read data from disk, so update
48 // offset_ even if decryption fails.
49 }
50 return io_s;
51}

139IOStatus EncryptedWritableFile::Append(const Slice& data,
140 const IOOptions& options,
141 IODebugContext* dbg) {
142 AlignedBuffer buf;
143 Slice dataToAppend(data);
144 if (data.size() > 0) {
145 auto offset = file_->GetFileSize(options, dbg); // size including prefix
146 // Encrypt in cloned buffer
147 buf.Alignment(GetRequiredBufferAlignment());
148 buf.AllocateNewBuffer(data.size());
149 // TODO (sagar0): Modify AlignedBuffer.Append to allow doing a memmove
150 // so that the next two lines can be replaced with buf.Append().
151 memmove(buf.BufferStart(), data.data(), data.size());
152 buf.Size(data.size());
153 IOStatus io_s;
154 {
155 PERF_TIMER_GUARD(encrypt_data_nanos);
156 io_s = status_to_io_status(
157 stream_->Encrypt(offset, buf.BufferStart(), buf.CurrentSize()));
158 }
159 if (!io_s.ok()) {
160 return io_s;
161 }
162 dataToAppend = Slice(buf.BufferStart(), buf.CurrentSize());
163 }
164 return file_->Append(dataToAppend, options, dbg);
165}

377 // Creates a CipherStream for the underlying file/name using the options
378 // If a writable provider is found and encryption is enabled, uses
379 // this provider to create a cipher stream.
380 // @param fname Name of the writable file
381 // @param underlying The underlying "raw" file
382 // @param options Options for creating the file/cipher
383 // @param prefix_length Returns the length of the encryption prefix used for
384 // this file
385 // @param stream Returns the cipher stream to use for this file if it
386 // should be encrypted
387 // @return OK on success, non-OK on failure.
388 template <class TypeFile>
389 IOStatus CreateWritableCipherStream(
390 const std::string& fname, const std::unique_ptr<TypeFile>& underlying,
391 const FileOptions& options, size_t* prefix_length,
392 std::unique_ptr<BlockAccessCipherStream>* stream, IODebugContext* dbg) {
393 EncryptionProvider* provider = nullptr;
394 *prefix_length = 0;
395 IOStatus status = GetWritableProvider(fname, &provider);
396 if (!status.ok()) {
397 return status;
398 } else if (provider != nullptr) {
399 // Initialize & write prefix (if needed)
400 AlignedBuffer buffer;
401 Slice prefix;
402 *prefix_length = provider->GetPrefixLength();
403 if (*prefix_length > 0) {
404 // Initialize prefix
405 buffer.Alignment(underlying->GetRequiredBufferAlignment());
406 buffer.AllocateNewBuffer(*prefix_length);
407 status = status_to_io_status(provider->CreateNewPrefix(
408 fname, buffer.BufferStart(), *prefix_length));
409 if (status.ok()) {
410 buffer.Size(*prefix_length);
411 prefix = Slice(buffer.BufferStart(), buffer.CurrentSize());
412 // Write prefix
413 status = underlying->Append(prefix, options.io_options, dbg);
414 }
415 if (!status.ok()) {
416 return status;
417 }
418 }
419 // Create cipher stream
420 status = status_to_io_status(
421 provider->CreateCipherStream(fname, options, prefix, stream));
422 }
423 return status;
424 }

427 IOStatus CreateWritableEncryptedFile(const std::string& fname,
428 std::unique_ptr<TypeFile>& underlying,
429 const FileOptions& options,
430 std::unique_ptr<TypeFile>* result,
431 IODebugContext* dbg) {

444 }

497 // Creates a CipherStream for the underlying file/name using the options
498 // If a readable provider is found and the file is encrypted, uses
499 // this provider to create a cipher stream.
500 // @param fname Name of the writable file
501 // @param underlying The underlying "raw" file
502 // @param options Options for creating the file/cipher
503 // @param prefix_length Returns the length of the encryption prefix used for
504 // this file
505 // @param stream Returns the cipher stream to use for this file if it
506 // is encrypted
507 // @return OK on success, non-OK on failure.

510 const std::string& fname, const std::unique_ptr<TypeFile>& underlying,

580 IOStatus NewSequentialFile(const std::string& fname,
581 const FileOptions& options,
582 std::unique_ptr<FSSequentialFile>* result,
583 IODebugContext* dbg) override {
584 result->reset();
585 if (options.use_mmap_reads) {
586 return IOStatus::InvalidArgument();
587 }
588 // Open file using underlying Env implementation
589 std::unique_ptr<FSSequentialFile> underlying;
590 auto status =
591 FileSystemWrapper::NewSequentialFile(fname, options, &underlying, dbg);
592 if (!status.ok()) {
593 return status;
594 }
595 uint64_t file_size;
596 status = FileSystemWrapper::GetFileSize(fname, options.io_options,
597 &file_size, dbg);
598 if (!status.ok()) {
599 return status;
600 }
601 if (!file_size) {
602 *result = std::move(underlying);
603 return status;
604 }
605 // Create cipher stream
606 std::unique_ptr<BlockAccessCipherStream> stream;
607 size_t prefix_length;
608 status = CreateSequentialCipherStream(fname, underlying, options,
609 &prefix_length, &stream, dbg);
610 if (status.ok()) {
611 result->reset(new EncryptedSequentialFile(
612 std::move(underlying), std::move(stream), prefix_length));
613 }
614 return status;
615 }

818Status BlockAccessCipherStream::Encrypt(uint64_t fileOffset, char* data,
819 size_t dataSize) {
820 // Calculate block index
821 auto blockSize = BlockSize();
822 uint64_t blockIndex = fileOffset / blockSize;
823 size_t blockOffset = fileOffset % blockSize;
824 std::unique_ptr<char[]> blockBuffer;
825
826 std::string scratch;
827 AllocateScratch(scratch);
828
829 // Encrypt individual blocks.
830 while (true) {
831 char* block = data;
832 size_t n = std::min(dataSize, blockSize - blockOffset);
833 if (n != blockSize) {
834 // We're not encrypting a full block.
835 // Copy data to blockBuffer
836 if (!blockBuffer) {
837 // Allocate buffer
838 blockBuffer = std::unique_ptr<char[]>(new char[blockSize]);
839 }
840 block = blockBuffer.get();
841 // Copy plain data to block buffer
842 memmove(block + blockOffset, data, n);
843 }
844 auto status = EncryptBlock(blockIndex, block, (char*)scratch.data());
845 if (!status.ok()) {
846 return status;
847 }
848 if (block != data) {
849 // Copy encrypted data back to `data`.
850 memmove(data, block + blockOffset, n);
851 }
852 dataSize -= n;
853 if (dataSize == 0) {
854 return Status::OK();
855 }
856 data += n;
857 blockOffset = 0;
858 blockIndex++;
859 }
860}
861
862Status BlockAccessCipherStream::Decrypt(uint64_t fileOffset, char* data,
863 size_t dataSize) {
864 // Calculate block index
865 auto blockSize = BlockSize();
866 uint64_t blockIndex = fileOffset / blockSize;
867 size_t blockOffset = fileOffset % blockSize;
868 std::unique_ptr<char[]> blockBuffer;
869
870 std::string scratch;
871 AllocateScratch(scratch);
872
873 // Decrypt individual blocks.
874 while (true) {
875 char* block = data;
876 size_t n = std::min(dataSize, blockSize - blockOffset);
877 if (n != blockSize) {
878 // We're not decrypting a full block.
879 // Copy data to blockBuffer
880 if (!blockBuffer) {
881 // Allocate buffer
882 blockBuffer = std::unique_ptr<char[]>(new char[blockSize]);
883 }
884 block = blockBuffer.get();
885 // Copy encrypted data to block buffer
886 memmove(block + blockOffset, data, n);
887 }
888 auto status = DecryptBlock(blockIndex, block, (char*)scratch.data());
889 if (!status.ok()) {
890 return status;
891 }
892 if (block != data) {
893 // Copy decrypted data back to `data`.
894 memmove(data, block + blockOffset, n);
895 }
896
897 // Simply decrementing dataSize by n could cause it to underflow,
898 // which will very likely make it read over the original bounds later
899 assert(dataSize >= n);
900 if (dataSize < n) {
901 return Status::Corruption("Cannot decrypt data at given offset");
902 }
903
904 dataSize -= n;
905 if (dataSize == 0) {
906 return Status::OK();
907 }
908 data += n;
909 blockOffset = 0;
910 blockIndex++;
911 }

963Status CTRCipherStream::EncryptBlock(uint64_t blockIndex, char* data,
964 char* scratch) {
965 // Create nonce + counter
966 auto blockSize = cipher_->BlockSize();
967 memmove(scratch, iv_.data(), blockSize);
968 EncodeFixed64(scratch, blockIndex + initialCounter_);
969
970 // Encrypt nonce + counter
971 auto status = cipher_->Encrypt(scratch);
972 if (!status.ok()) {
973 return status;
974 }
975
976 // XOR data with ciphertext.
977 for (size_t i = 0; i < blockSize; i++) {
978 data[i] = data[i] ^ scratch[i];
979 }
980 return Status::OK();
981}

984 char* scratch) {

986 return EncryptBlock(blockIndex, data, scratch);

1022// decodeCTRParameters decodes the initial counter & IV from the given
1023// (plain text) prefix.
1024static void decodeCTRParameters(const char* prefix, size_t blockSize,
1025 uint64_t& initialCounter, Slice& iv) {
1026 // First block contains 64-bit initial counter
1027 initialCounter = DecodeFixed64(prefix);
1028 // Second block contains IV
1029 iv = Slice(prefix + blockSize, blockSize);

1032Status CTREncryptionProvider::CreateNewPrefix(const std::string& /*fname*/,
1033 char* prefix,
1034 size_t prefixLength) const {
1035 if (!cipher_) {
1036 return Status::InvalidArgument("Encryption Cipher is missing");
1037 }
1038 // Create & seed rnd.
1039 Random rnd((uint32_t)SystemClock::Default()->NowMicros());
1040 // Fill entire prefix block with random values.
1041 for (size_t i = 0; i < prefixLength; i++) {
1042 prefix[i] = rnd.Uniform(256) & 0xFF;
1043 }
1044 // Take random data to extract initial counter & IV
1045 auto blockSize = cipher_->BlockSize();
1046 uint64_t initialCounter;
1047 Slice prefixIV;
1048 decodeCTRParameters(prefix, blockSize, initialCounter, prefixIV);
1049
1050 // Now populate the rest of the prefix, starting from the third block.
1051 PopulateSecretPrefixPart(prefix + (2 * blockSize),
1052 prefixLength - (2 * blockSize), blockSize);
1053
1054 // Encrypt the prefix, starting from block 2 (leave block 0, 1 with initial
1055 // counter & IV unencrypted)
1056 CTRCipherStream cipherStream(cipher_, prefixIV.data(), initialCounter);
1057 Status status;
1058 {
1059 PERF_TIMER_GUARD(encrypt_data_nanos);
1060 status = cipherStream.Encrypt(0, prefix + (2 * blockSize),
1061 prefixLength - (2 * blockSize));
1062 }
1063
1064 return status;
1065}

1077Status CTREncryptionProvider::CreateCipherStream(
1078 const std::string& fname, const EnvOptions& options, Slice& prefix,
1079 std::unique_ptr<BlockAccessCipherStream>* result) {
1080 if (!cipher_) {
1081 return Status::InvalidArgument("Encryption Cipher is missing");
1082 }
1083 // Read plain text part of prefix.
1084 auto blockSize = cipher_->BlockSize();
1085 uint64_t initialCounter;
1086 Slice iv;
1087 decodeCTRParameters(prefix.data(), blockSize, initialCounter, iv);
1088
1089 // If the prefix is smaller than twice the block size, we would below read a
1090 // very large chunk of the file (and very likely read over the bounds)
1091 assert(prefix.size() >= 2 * blockSize);
1092 if (prefix.size() < 2 * blockSize) {
1093 return Status::Corruption("Unable to read from file " + fname +
1094 ": read attempt would read beyond file bounds");
1095 }
1096
1097 // Decrypt the encrypted part of the prefix, starting from block 2 (block 0, 1
1098 // with initial counter & IV are unencrypted)
1099 CTRCipherStream cipherStream(cipher_, iv.data(), initialCounter);
1100 Status status;
1101 {
1102 PERF_TIMER_GUARD(decrypt_data_nanos);
1103 status = cipherStream.Decrypt(0, (char*)prefix.data() + (2 * blockSize),
1104 prefix.size() - (2 * blockSize));
1105 }
1106 if (!status.ok()) {
1107 return status;
1108 }
1109
1110 // Create cipher stream
1111 return CreateCipherStreamFromPrefix(fname, options, initialCounter, iv,
1112 prefix, result);
1113}
