42// Convert an index iterator value (i.e., an encoded BlockHandle)
43// into an iterator over the contents of the corresponding block.
44// If input_iter is null, new a iterator
45// If input_iter is not null, update this iter and return it
46template <typename TBlockIter>
47TBlockIter* BlockBasedTable::NewDataBlockIterator(
48 const ReadOptions& ro, const BlockHandle& handle, TBlockIter* input_iter,
49 BlockType block_type, GetContext* get_context,
50 BlockCacheLookupContext* lookup_context,
51 FilePrefetchBuffer* prefetch_buffer, bool for_compaction, bool async_read,
52 Status& s, bool use_block_cache_for_lookup) const {
53 using IterBlocklike = typename IterTraits<TBlockIter>::IterBlocklike;
54 PERF_TIMER_GUARD(new_table_block_iter_nanos);
55
56 TBlockIter* iter = input_iter != nullptr ? input_iter : new TBlockIter;
57 if (!s.ok()) {
58 iter->Invalidate(s);
59 return iter;
60 }
61
62 CachableEntry<Block> block;
63 if (rep_->uncompression_dict_reader && block_type == BlockType::kData) {
64 CachableEntry<UncompressionDict> uncompression_dict;
65 // For async scans, don't use the prefetch buffer since an async prefetch
66 // might already be under way and this would invalidate it. Also, the
67 // uncompression dict is typically at the end of the file and would
68 // most likely break the sequentiality of the access pattern.
69 // Same is with auto_readahead_size. It iterates over index to lookup for
70 // data blocks. And this could break the the sequentiality of the access
71 // pattern.
72 s = rep_->uncompression_dict_reader->GetOrReadUncompressionDictionary(
73 ((ro.async_io || ro.auto_readahead_size) ? nullptr : prefetch_buffer),
74 ro, get_context, lookup_context, &uncompression_dict);
75 if (!s.ok()) {
76 iter->Invalidate(s);
77 return iter;
78 }
79 const UncompressionDict& dict = uncompression_dict.GetValue()
80 ? *uncompression_dict.GetValue()
81 : UncompressionDict::GetEmptyDict();
82 s = RetrieveBlock(
83 prefetch_buffer, ro, handle, dict, &block.As<IterBlocklike>(),
84 get_context, lookup_context, for_compaction,
85 /* use_cache */ true, async_read, use_block_cache_for_lookup);
86 } else {
87 s = RetrieveBlock(
88 prefetch_buffer, ro, handle, UncompressionDict::GetEmptyDict(),
89 &block.As<IterBlocklike>(), get_context, lookup_context, for_compaction,
90 /* use_cache */ true, async_read, use_block_cache_for_lookup);
91 }
92
93 if (s.IsTryAgain() && async_read) {
94 return iter;
95 }
96
97 if (!s.ok()) {
98 assert(block.IsEmpty());
99 iter->Invalidate(s);
100 return iter;
101 }
102
103 assert(block.GetValue() != nullptr);
104
105 // Block contents are pinned and it is still pinned after the iterator
106 // is destroyed as long as cleanup functions are moved to another object,
107 // when:
108 // 1. block cache handle is set to be released in cleanup function, or
109 // 2. it's pointing to immortal source. If own_bytes is true then we are
110 // not reading data from the original source, whether immortal or not.
111 // Otherwise, the block is pinned iff the source is immortal.
112 const bool block_contents_pinned =
113 block.IsCached() ||
114 (!block.GetValue()->own_bytes() && rep_->immortal_table);
115 iter = InitBlockIterator<TBlockIter>(rep_, block.GetValue(), block_type, iter,
116 block_contents_pinned);
117
118 if (!block.IsCached()) {
119 if (!ro.fill_cache) {
120 IterPlaceholderCacheInterface block_cache{
121 rep_->table_options.block_cache.get()};
122 if (block_cache) {
123 // insert a dummy record to block cache to track the memory usage
124 Cache::Handle* cache_handle = nullptr;
125 CacheKey key =
126 CacheKey::CreateUniqueForCacheLifetime(block_cache.get());
127 s = block_cache.Insert(key.AsSlice(),
128 block.GetValue()->ApproximateMemoryUsage(),
129 &cache_handle);
130
131 if (s.ok()) {
132 assert(cache_handle != nullptr);
133 iter->RegisterCleanup(&ForceReleaseCachedEntry, block_cache.get(),
134 cache_handle);
135 }
136 }
137 }
138 } else {
139 iter->SetCacheHandle(block.GetCacheHandle());
140 }
141
142 block.TransferTo(iter);
143
144 return iter;
145}
