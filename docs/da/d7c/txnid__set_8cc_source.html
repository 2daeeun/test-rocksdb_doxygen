61int find_by_txnid(const TXNID &txnid_a, const TXNID &txnid_b);
62int find_by_txnid(const TXNID &txnid_a, const TXNID &txnid_b) {
63 if (txnid_a < txnid_b) {
64 return -1;
65 } else if (txnid_a == txnid_b) {
66 return 0;
67 } else {
68 return 1;
69 }
70}

80// Return true if the given transaction id is a member of the set.
81// Otherwise, return false.
82bool txnid_set::contains(TXNID txnid) const {
83 TXNID find_txnid;
84 int r = m_txnids.find_zero<TXNID, find_by_txnid>(txnid, &find_txnid, nullptr);
85 return r == 0 ? true : false;
86}
87
88// Add a given txnid to the set
89void txnid_set::add(TXNID txnid) {
90 int r = m_txnids.insert<TXNID, find_by_txnid>(txnid, txnid, nullptr);
91 invariant(r == 0 || r == DB_KEYEXIST);
92}
93
94// Delete a given txnid from the set.
95void txnid_set::remove(TXNID txnid) {
96 uint32_t idx;
97 int r = m_txnids.find_zero<TXNID, find_by_txnid>(txnid, nullptr, &idx);
98 if (r == 0) {
99 r = m_txnids.delete_at(idx);
100 invariant_zero(r);
101 }
102}
103
104// Return the size of the set
105uint32_t txnid_set::size(void) const { return m_txnids.size(); }
106
107// Get the ith id in the set, assuming that the set is sorted.
108TXNID txnid_set::get(uint32_t i) const {
109 TXNID txnid;
110 int r = m_txnids.fetch(i, &txnid);
111 if (r == EINVAL) /* Shouldn't happen, avoid compiler warning */
112 return TXNID_NONE;
113 invariant_zero(r);
114 return txnid;
115}
