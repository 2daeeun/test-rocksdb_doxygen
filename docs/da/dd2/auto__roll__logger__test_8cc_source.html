137void AutoRollLoggerTest::RollLogFileBySizeTest(AutoRollLogger* logger,
138 size_t log_max_size,
139 const std::string& log_message) {
140 logger->SetInfoLogLevel(InfoLogLevel::INFO_LEVEL);
141 ASSERT_EQ(InfoLogLevel::INFO_LEVEL, logger->GetInfoLogLevel());
142 ASSERT_EQ(InfoLogLevel::INFO_LEVEL,
143 logger->TEST_inner_logger()->GetInfoLogLevel());
144 // measure the size of each message, which is supposed
145 // to be equal or greater than log_message.size()
146 LogMessage(logger, log_message.c_str());
147 size_t message_size = logger->GetLogFileSize();
148 size_t current_log_size = message_size;
149
150 // Test the cases when the log file will not be rolled.
151 while (current_log_size + message_size < log_max_size) {
152 LogMessage(logger, log_message.c_str());
153 current_log_size += message_size;
154 ASSERT_EQ(current_log_size, logger->GetLogFileSize());
155 }
156
157 // Now the log file will be rolled
158 LogMessage(logger, log_message.c_str());
159 // Since rotation is checked before actual logging, we need to
160 // trigger the rotation by logging another message.
161 LogMessage(logger, log_message.c_str());
162
163 ASSERT_TRUE(message_size == logger->GetLogFileSize());
164}

166void AutoRollLoggerTest::RollLogFileByTimeTest(
167 const std::shared_ptr<FileSystem>& fs,
168 const std::shared_ptr<SystemClock>& sc, AutoRollLogger* logger, size_t time,
169 const std::string& log_message) {
170 uint64_t expected_ctime;
171 uint64_t actual_ctime;
172
173 uint64_t total_log_size;
174 EXPECT_OK(fs->GetFileSize(kLogFile, IOOptions(), &total_log_size, nullptr));
175 expected_ctime = logger->TEST_ctime();
176 logger->SetCallNowMicrosEveryNRecords(0);
177
178 // -- Write to the log for several times, which is supposed
179 // to be finished before time.
180 for (int i = 0; i < 10; ++i) {
181 sc->SleepForMicroseconds(50000);
182 LogMessage(logger, log_message.c_str());
183 EXPECT_OK(logger->GetStatus());
184 // Make sure we always write to the same log file (by
185 // checking the create time);
186
187 actual_ctime = logger->TEST_ctime();
188
189 // Also make sure the log size is increasing.
190 EXPECT_EQ(expected_ctime, actual_ctime);
191 EXPECT_GT(logger->GetLogFileSize(), total_log_size);
192 total_log_size = logger->GetLogFileSize();
193 }
194
195 // -- Make the log file expire
196 sc->SleepForMicroseconds(static_cast<int>(time * 1000000));
197 LogMessage(logger, log_message.c_str());
198
199 // At this time, the new log file should be created.
200 actual_ctime = logger->TEST_ctime();
201 EXPECT_LT(expected_ctime, actual_ctime);
202 EXPECT_LT(logger->GetLogFileSize(), total_log_size);
203}

452TEST_F(AutoRollLoggerTest, LogFlushWhileRolling) {
453 DBOptions options;
454 std::shared_ptr<Logger> logger;
455
456 InitTestDb();
457 options.max_log_file_size = 1024 * 5;
458 ASSERT_OK(CreateLoggerFromOptions(kTestDir, options, &logger));
459 AutoRollLogger* auto_roll_logger =
460 dynamic_cast<AutoRollLogger*>(logger.get());
461 ASSERT_TRUE(auto_roll_logger);
462 ROCKSDB_NAMESPACE::port::Thread flush_thread;
463
464 // Notes:
465 // (1) Need to pin the old logger before beginning the roll, as rolling grabs
466 // the mutex, which would prevent us from accessing the old logger. This
467 // also marks flush_thread with AutoRollLogger::Flush:PinnedLogger.
468 // (2) New logger will be cut in AutoRollLogger::RollLogFile only when flush
469 // is completed and reference to pinned logger is released.
470 // (3) EnvLogger::Flush() happens in both threads but its SyncPoints only
471 // are enabled in flush_thread (the one pinning the old logger).
472 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->LoadDependencyAndMarkers(
473 {{"AutoRollLogger::Flush:PinnedLogger",
474 "AutoRollLoggerTest::LogFlushWhileRolling:PreRollAndPostThreadInit"}},
475 {{"AutoRollLogger::Flush:PinnedLogger", "EnvLogger::Flush:Begin1"},
476 {"AutoRollLogger::Flush:PinnedLogger", "EnvLogger::Flush:Begin2"}});
477 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
478
479 flush_thread = port::Thread([&]() { auto_roll_logger->Flush(); });
480 TEST_SYNC_POINT(
481 "AutoRollLoggerTest::LogFlushWhileRolling:PreRollAndPostThreadInit");
482 RollLogFileBySizeTest(auto_roll_logger, options.max_log_file_size,
483 kSampleMessage + ":LogFlushWhileRolling");
484 flush_thread.join();
485 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();
486}
