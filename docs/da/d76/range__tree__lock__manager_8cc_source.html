40static const char SUFFIX_SUPREMUM = 0x1;

42// Convert Endpoint into an internal format used for storing it in locktree
43// (DBT structure is used for passing endpoints to locktree and getting back)
44void serialize_endpoint(const Endpoint& endp, std::string* buf) {
45 buf->push_back(endp.inf_suffix ? SUFFIX_SUPREMUM : SUFFIX_INFIMUM);
46 buf->append(endp.slice.data(), endp.slice.size());
47}
48
49// Decode the endpoint from the format it is stored in the locktree (DBT) to
50// the one used outside: either Endpoint or EndpointWithString
51template <typename EndpointStruct>
52void deserialize_endpoint(const DBT* dbt, EndpointStruct* endp) {
53 assert(dbt->size >= 1);
54 const char* dbt_data = (const char*)dbt->data;
55 char suffix = dbt_data[0];
56 assert(suffix == SUFFIX_INFIMUM || suffix == SUFFIX_SUPREMUM);
57 endp->inf_suffix = (suffix == SUFFIX_SUPREMUM);
58 endp->slice = decltype(EndpointStruct::slice)(dbt_data + 1, dbt->size - 1);
59}

61// Get a range lock on [start_key; end_key] range
62Status RangeTreeLockManager::TryLock(PessimisticTransaction* txn,
63 uint32_t column_family_id,
64 const Endpoint& start_endp,
65 const Endpoint& end_endp, Env*,
66 bool exclusive) {
67 toku::lock_request request;
68 request.create(mutex_factory_);
69 DBT start_key_dbt, end_key_dbt;
70
71 TEST_SYNC_POINT("RangeTreeLockManager::TryRangeLock:enter");
72 std::string start_key;
73 std::string end_key;
74 serialize_endpoint(start_endp, &start_key);
75 serialize_endpoint(end_endp, &end_key);
76
77 toku_fill_dbt(&start_key_dbt, start_key.data(), start_key.size());
78 toku_fill_dbt(&end_key_dbt, end_key.data(), end_key.size());
79
80 auto lt = GetLockTreeForCF(column_family_id);
81
82 // Put the key waited on into request's m_extra. See
83 // wait_callback_for_locktree for details.
84 std::string wait_key(start_endp.slice.data(), start_endp.slice.size());
85
86 request.set(lt.get(), (TXNID)txn, &start_key_dbt, &end_key_dbt,
87 exclusive ? toku::lock_request::WRITE : toku::lock_request::READ,
88 false /* not a big txn */, &wait_key);
89
90 // This is for "periodically wake up and check if the wait is killed" feature
91 // which we are not using.
92 uint64_t killed_time_msec = 0;
93 uint64_t wait_time_msec = txn->GetLockTimeout();
94
95 if (wait_time_msec == static_cast<uint64_t>(-1)) {
96 // The transaction has no wait timeout. lock_request::wait doesn't support
97 // this, it needs a number of milliseconds to wait. Pass it one year to
98 // be safe.
99 wait_time_msec = uint64_t(1000) * 60 * 60 * 24 * 365;
100 } else {
101 // convert microseconds to milliseconds
102 wait_time_msec = (wait_time_msec + 500) / 1000;
103 }
104
105 std::vector<RangeDeadlockInfo> di_path;
106 request.m_deadlock_cb = [&](TXNID txnid, bool is_exclusive,
107 const DBT* start_dbt, const DBT* end_dbt) {
108 EndpointWithString start;
109 EndpointWithString end;
110 deserialize_endpoint(start_dbt, &start);
111 deserialize_endpoint(end_dbt, &end);
112
113 di_path.push_back({txnid, column_family_id, is_exclusive, std::move(start),
114 std::move(end)});
115 };
116
117 request.start();
118
119 const int r = request.wait(wait_time_msec, killed_time_msec,
120 nullptr, // killed_callback
121 wait_callback_for_locktree, nullptr);
122
123 // Inform the txn that we are no longer waiting:
124 txn->ClearWaitingTxn();
125
126 request.destroy();
127 switch (r) {
128 case 0:
129 break; // fall through
130 case DB_LOCK_NOTGRANTED:
131 return Status::TimedOut(Status::SubCode::kLockTimeout);
132 case TOKUDB_OUT_OF_LOCKS:
133 return Status::Busy(Status::SubCode::kLockLimit);
134 case DB_LOCK_DEADLOCK: {
135 std::reverse(di_path.begin(), di_path.end());
136 dlock_buffer_.AddNewPath(
137 RangeDeadlockPath(di_path, request.get_start_time()));
138 return Status::Busy(Status::SubCode::kDeadlock);
139 }
140 default:
141 assert(0);
142 return Status::Busy(Status::SubCode::kLockLimit);
143 }
144
145 return Status::OK();
146}

148// Wait callback that locktree library will call to inform us about
149// the lock waits that are in progress.
150void wait_callback_for_locktree(void*, toku::lock_wait_infos* infos) {
151 TEST_SYNC_POINT("RangeTreeLockManager::TryRangeLock:EnterWaitingTxn");
152 for (const auto& wait_info : *infos) {
153 // As long as we hold the lock on the locktree's pending request queue
154 // this should be safe.
155 auto txn = (PessimisticTransaction*)wait_info.waiter;
156 auto cf_id = (ColumnFamilyId)wait_info.ltree->get_dict_id().dictid;
157
158 autovector<TransactionID> waitee_ids;
159 for (auto waitee : wait_info.waitees) {
160 waitee_ids.push_back(waitee);
161 }
162 txn->SetWaitingTxn(waitee_ids, cf_id, (std::string*)wait_info.m_extra);
163 }
164
165 // Here we can assume that the locktree code will now wait for some lock
166 TEST_SYNC_POINT("RangeTreeLockManager::TryRangeLock:WaitingTxn");
167}

169void RangeTreeLockManager::UnLock(PessimisticTransaction* txn,
170 ColumnFamilyId column_family_id,
171 const std::string& key, Env*) {
172 auto locktree = GetLockTreeForCF(column_family_id);
173 std::string endp_image;
174 serialize_endpoint({key.data(), key.size(), false}, &endp_image);
175
176 DBT key_dbt;
177 toku_fill_dbt(&key_dbt, endp_image.data(), endp_image.size());
178
179 toku::range_buffer range_buf;
180 range_buf.create();
181 range_buf.append(&key_dbt, &key_dbt);
182
183 locktree->release_locks((TXNID)txn, &range_buf);
184 range_buf.destroy();
185
186 toku::lock_request::retry_all_lock_requests(
187 locktree.get(), wait_callback_for_locktree, nullptr);
188}

204int RangeTreeLockManager::CompareDbtEndpoints(void* arg, const DBT* a_key,
205 const DBT* b_key) {
206 const char* a = (const char*)a_key->data;
207 const char* b = (const char*)b_key->data;
208
209 size_t a_len = a_key->size;
210 size_t b_len = b_key->size;
211
212 size_t min_len = std::min(a_len, b_len);
213
214 // Compare the values. The first byte encodes the endpoint type, its value
215 // is either SUFFIX_INFIMUM or SUFFIX_SUPREMUM.
216 Comparator* cmp = (Comparator*)arg;
217 int res = cmp->Compare(Slice(a + 1, min_len - 1), Slice(b + 1, min_len - 1));
218 if (!res) {
219 if (b_len > min_len) {
220 // a is shorter;
221 if (a[0] == SUFFIX_INFIMUM) {
222 return -1; //"a is smaller"
223 } else {
224 // a is considered padded with 0xFF:FF:FF:FF...
225 return 1; // "a" is bigger
226 }
227 } else if (a_len > min_len) {
228 // the opposite of the above: b is shorter.
229 if (b[0] == SUFFIX_INFIMUM) {
230 return 1; //"b is smaller"
231 } else {
232 // b is considered padded with 0xFF:FF:FF:FF...
233 return -1; // "b" is bigger
234 }
235 } else {
236 // the lengths are equal (and the key values, too)
237 if (a[0] < b[0]) {
238 return -1;
239 } else if (a[0] > b[0]) {
240 return 1;
241 } else {
242 return 0;
243 }
244 }
245 } else {
246 return res;
247 }
248}

297std::vector<DeadlockPath> RangeTreeLockManager::GetDeadlockInfoBuffer() {
298 std::vector<DeadlockPath> res;
299 std::vector<RangeDeadlockPath> data = GetRangeDeadlockInfoBuffer();
300 // report left endpoints
301 for (auto it = data.begin(); it != data.end(); ++it) {
302 std::vector<DeadlockInfo> path;
303
304 for (auto it2 = it->path.begin(); it2 != it->path.end(); ++it2) {
305 path.push_back(
306 {it2->m_txn_id, it2->m_cf_id, it2->m_exclusive, it2->m_start.slice});
307 }
308 res.emplace_back(path, it->deadlock_time);
309 }
310 return res;

369void RangeTreeLockManager::AddColumnFamily(const ColumnFamilyHandle* cfh) {
370 uint32_t column_family_id = cfh->GetID();
371
372 InstrumentedMutexLock l(&ltree_map_mutex_);
373 if (ltree_map_.find(column_family_id) == ltree_map_.end()) {
374 DICTIONARY_ID dict_id = {.dictid = column_family_id};
375 toku::comparator cmp;
376 cmp.create(CompareDbtEndpoints, (void*)cfh->GetComparator());
377 toku::locktree* ltree =
378 ltm_.get_lt(dict_id, cmp,
379 /* on_create_extra*/ static_cast<void*>(this));
380 // This is ok to because get_lt has copied the comparator:
381 cmp.destroy();
382
383 ltree_map_.insert({column_family_id, MakeLockTreePtr(ltree)});
384 }

387void RangeTreeLockManager::RemoveColumnFamily(const ColumnFamilyHandle* cfh) {
388 uint32_t column_family_id = cfh->GetID();
389 // Remove lock_map for this column family. Since the lock map is stored
390 // as a shared ptr, concurrent transactions can still keep using it
391 // until they release their references to it.
392
393 // TODO what if one drops a column family while transaction(s) still have
394 // locks in it?
395 // locktree uses column family'c Comparator* as the criteria to do tree
396 // ordering. If the comparator is gone, we won't even be able to remove the
397 // elements from the locktree.
398 // A possible solution might be to remove everything right now:
399 // - wait until everyone traversing the locktree are gone
400 // - remove everything from the locktree.
401 // - some transactions may have acquired locks in their LockTracker objects.
402 // Arrange something so we don't blow up when they try to release them.
403 // - ...
404 // This use case (drop column family while somebody is using it) doesn't seem
405 // the priority, though.
406
407 {
408 InstrumentedMutexLock l(&ltree_map_mutex_);
409
410 auto lock_maps_iter = ltree_map_.find(column_family_id);
411 assert(lock_maps_iter != ltree_map_.end());
412 ltree_map_.erase(lock_maps_iter);
413 } // lock_map_mutex_
414
415 autovector<void*> local_caches;
416 ltree_lookup_cache_->Scrape(&local_caches, nullptr);
417 for (auto cache : local_caches) {
418 delete static_cast<LockTreeMap*>(cache);
419 }
420}

422std::shared_ptr<toku::locktree> RangeTreeLockManager::GetLockTreeForCF(
423 ColumnFamilyId column_family_id) {
424 // First check thread-local cache
425 if (ltree_lookup_cache_->Get() == nullptr) {
426 ltree_lookup_cache_->Reset(new LockTreeMap());
427 }
428
429 auto ltree_map_cache = static_cast<LockTreeMap*>(ltree_lookup_cache_->Get());
430
431 auto it = ltree_map_cache->find(column_family_id);
432 if (it != ltree_map_cache->end()) {
433 // Found lock map for this column family.
434 return it->second;
435 }
436
437 // Not found in local cache, grab mutex and check shared LockMaps
438 InstrumentedMutexLock l(&ltree_map_mutex_);
439
440 it = ltree_map_.find(column_family_id);
441 if (it == ltree_map_.end()) {
442 return nullptr;
443 } else {
444 // Found lock map. Store in thread-local cache and return.
445 ltree_map_cache->insert({column_family_id, it->second});
446 return it->second;
447 }

467static void push_into_lock_status_data(void* param, const DBT* left,
468 const DBT* right, TXNID txnid_arg,
469 bool is_shared, TxnidVector* owners) {
470 struct LOCK_PRINT_CONTEXT* ctx = (LOCK_PRINT_CONTEXT*)param;
471 struct RangeLockInfo info;
472
473 info.exclusive = !is_shared;
474
475 deserialize_endpoint(left, &info.start);
476 deserialize_endpoint(right, &info.end);
477
478 if (txnid_arg != TXNID_SHARED) {
479 info.ids.push_back(txnid_arg);
480 } else {
481 for (auto it : *owners) {
482 info.ids.push_back(it);
483 }
484 }
485 ctx->data->insert({ctx->cfh_id, info});
486}

488LockManager::RangeLockStatus RangeTreeLockManager::GetRangeLockStatus() {
489 LockManager::RangeLockStatus data;
490 {
491 InstrumentedMutexLock l(&ltree_map_mutex_);
492 for (const auto& it : ltree_map_) {
493 LOCK_PRINT_CONTEXT ctx = {&data, it.first};
494 it.second->dump_locks((void*)&ctx, push_into_lock_status_data);

496 }
