23static int RegisterBuiltinMergeOperators(ObjectLibrary& library,
24 const std::string& /*arg*/) {
25 size_t num_types;
26 library.AddFactory<MergeOperator>(
27 ObjectLibrary::PatternEntry(StringAppendOperator::kClassName())
28 .AnotherName(StringAppendOperator::kNickName()),
29 [](const std::string& /*uri*/, std::unique_ptr<MergeOperator>* guard,
30 std::string* /*errmsg*/) {
31 guard->reset(new StringAppendOperator(","));
32 return guard->get();
33 });
34 library.AddFactory<MergeOperator>(
35 ObjectLibrary::PatternEntry(StringAppendTESTOperator::kClassName())
36 .AnotherName(StringAppendTESTOperator::kNickName()),
37 [](const std::string& /*uri*/, std::unique_ptr<MergeOperator>* guard,
38 std::string* /*errmsg*/) {
39 guard->reset(new StringAppendTESTOperator(","));
40 return guard->get();
41 });

46 std::string* /*errmsg*/) {

93Status MergeOperator::CreateFromString(const ConfigOptions& config_options,
94 const std::string& value,
95 std::shared_ptr<MergeOperator>* result) {
96 static std::once_flag once;
97 std::call_once(once, [&]() {
98 RegisterBuiltinMergeOperators(*(ObjectLibrary::Default().get()), "");
99 });
100 return LoadSharedObject<MergeOperator>(config_options, value, result);
101}

103std::shared_ptr<MergeOperator> MergeOperators::CreateFromStringId(
104 const std::string& id) {
105 std::shared_ptr<MergeOperator> result;
106 Status s = MergeOperator::CreateFromString(ConfigOptions(), id, &result);
107 if (s.ok()) {
108 return result;
109 } else {
110 // Empty or unknown, just return nullptr
111 return nullptr;
112 }
113}
