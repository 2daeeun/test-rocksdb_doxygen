6"""Filter for error messages in test output:
7 - Receives merged stdout/stderr from test on stdin
8 - Finds patterns of known error messages for test name (first argument)
9 - Prints those error messages to stdout
10"""

17class ErrorParserBase:
18 def parse_error(self, line):
19 """Parses a line of test output. If it contains an error, returns a
20 formatted message describing the error; otherwise, returns None.
21 Subclasses must override this method.
22 """
23 raise NotImplementedError

26class GTestErrorParser(ErrorParserBase):
27 """A parser that remembers the last test that began running so it can print
28 that test's name upon detecting failure.
29 """
30
31 _GTEST_NAME_PATTERN = re.compile(r"\[ RUN \] (\S+)$")
32 # format: '<filename or "unknown file">:<line #>: Failure'
33 _GTEST_FAIL_PATTERN = re.compile(r"(unknown file|\S+:\d+): Failure$")
34
35 def __init__(self):
36 self._last_gtest_name = "Unknown test"
37
38 def parse_error(self, line):
39 gtest_name_match = self._GTEST_NAME_PATTERN.match(line)
40 if gtest_name_match:
41 self._last_gtest_name = gtest_name_match.group(1)
42 return None
43 gtest_fail_match = self._GTEST_FAIL_PATTERN.match(line)
44 if gtest_fail_match:
45 return "{} failed: {}".format(self._last_gtest_name, gtest_fail_match.group(1))
46 return None

49class MatchErrorParser(ErrorParserBase):
50 """A simple parser that returns the whole line if it matches the pattern."""
51
52 def __init__(self, pattern):
53 self._pattern = re.compile(pattern)
54
55 def parse_error(self, line):
56 if self._pattern.match(line):
57 return line
58 return None

160def main():
161 if len(sys.argv) != 2:
162 return "Usage: %s <test name>" % sys.argv[0]
163 test_name = sys.argv[1]
164 if test_name not in _TEST_NAME_TO_PARSERS:
165 return "Unknown test name: %s" % test_name
166
167 error_parsers = []
168 for parser_cls in _TEST_NAME_TO_PARSERS[test_name]:
169 error_parsers.append(parser_cls())
170
171 for line in sys.stdin:
172 line = line.strip()
173 for error_parser in error_parsers:
174 error_msg = error_parser.parse_error(line)
175 if error_msg is not None:
176 print(error_msg)
177
178
179if __name__ == "__main__":
180 sys.exit(main())
