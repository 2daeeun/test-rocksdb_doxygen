31// A helper function that ensures the table properties returned in
32// `GetPropertiesOfAllTablesTest` is correct.
33// This test assumes entries size is different for each of the tables.

36void VerifyTableProperties(DB* db, uint64_t expected_entries_size) {
37 TablePropertiesCollection props;
38 ASSERT_OK(db->GetPropertiesOfAllTables(&props));
39
40 ASSERT_EQ(4U, props.size());
41 std::unordered_set<uint64_t> unique_entries;
42
43 // Indirect test
44 uint64_t sum = 0;
45 for (const auto& item : props) {
46 unique_entries.insert(item.second->num_entries);
47 sum += item.second->num_entries;
48 }
49
50 ASSERT_EQ(props.size(), unique_entries.size());
51 ASSERT_EQ(expected_entries_size, sum);
52
53 VerifySstUniqueIds(props);
54}
55} // anonymous namespace

64TEST_F(DBTablePropertiesTest, GetPropertiesOfAllTablesTest) {
65 Options options = CurrentOptions();
66 options.level0_file_num_compaction_trigger = 8;
67 // Part of strategy to prevent pinning table files
68 options.max_open_files = 42;
69 Reopen(options);
70
71 // Create 4 tables
72 for (int table = 0; table < 4; ++table) {
73 // Use old meta name for table properties for one file
74 if (table == 3) {
75 SyncPoint::GetInstance()->SetCallBack(
76 "BlockBasedTableBuilder::WritePropertiesBlock:Meta", [&](void* meta) {
77 *static_cast<const std::string**>(meta) = &kPropertiesBlockOldName;
78 });
79 SyncPoint::GetInstance()->EnableProcessing();
80 }
81 // Build file
82 for (int i = 0; i < 10 + table; ++i) {
83 ASSERT_OK(
84 db_->Put(WriteOptions(), std::to_string(table * 100 + i), "val"));
85 }
86 ASSERT_OK(db_->Flush(FlushOptions()));
87 }
88 SyncPoint::GetInstance()->DisableProcessing();
89 std::string original_session_id;
90 ASSERT_OK(db_->GetDbSessionId(original_session_id));
91
92 // Part of strategy to prevent pinning table files
93 SyncPoint::GetInstance()->SetCallBack(
94 "VersionEditHandler::LoadTables:skip_load_table_files",
95 [&](void* skip_load) { *static_cast<bool*>(skip_load) = true; });
96 SyncPoint::GetInstance()->EnableProcessing();
97
98 // 1. Read table properties directly from file
99 Reopen(options);
100 // Clear out auto-opened files
101 dbfull()->TEST_table_cache()->EraseUnRefEntries();
102 ASSERT_EQ(dbfull()->TEST_table_cache()->GetUsage(), 0U);
103 VerifyTableProperties(db_, 10 + 11 + 12 + 13);
104
105 // 2. Put two tables to table cache and
106 Reopen(options);
107 // Clear out auto-opened files
108 dbfull()->TEST_table_cache()->EraseUnRefEntries();
109 ASSERT_EQ(dbfull()->TEST_table_cache()->GetUsage(), 0U);
110 // fetch key from 1st and 2nd table, which will internally place that table to
111 // the table cache.
112 for (int i = 0; i < 2; ++i) {
113 Get(std::to_string(i * 100 + 0));
114 }
115
116 VerifyTableProperties(db_, 10 + 11 + 12 + 13);
117
118 // 3. Put all tables to table cache
119 Reopen(options);
120 // fetch key from all tables, which will place them in table cache.
121 for (int i = 0; i < 4; ++i) {
122 Get(std::to_string(i * 100 + 0));
123 }
124 VerifyTableProperties(db_, 10 + 11 + 12 + 13);
125
126 // 4. Try to read CORRUPT properties (a) directly from file, and (b)
127 // through reader on Get
128
129 // It's not practical to prevent table file read on Open, so we
130 // corrupt after open and after purging table cache.
131 for (bool direct : {true, false}) {
132 Reopen(options);
133 // Clear out auto-opened files
134 dbfull()->TEST_table_cache()->EraseUnRefEntries();
135 ASSERT_EQ(dbfull()->TEST_table_cache()->GetUsage(), 0U);
136
137 TablePropertiesCollection props;
138 ASSERT_OK(db_->GetPropertiesOfAllTables(&props));
139 std::string sst_file = props.begin()->first;
140
141 // Corrupt the file's TableProperties using session id
142 std::string contents;
143 ASSERT_OK(
144 ReadFileToString(env_->GetFileSystem().get(), sst_file, &contents));
145 size_t pos = contents.find(original_session_id);
146 ASSERT_NE(pos, std::string::npos);
147 ASSERT_OK(test::CorruptFile(env_, sst_file, static_cast<int>(pos), 1,
148 /*verify checksum fails*/ false));
149
150 // Try to read CORRUPT properties
151 if (direct) {
152 ASSERT_TRUE(db_->GetPropertiesOfAllTables(&props).IsCorruption());
153 } else {
154 bool found_corruption = false;
155 for (int i = 0; i < 4; ++i) {
156 std::string result = Get(std::to_string(i * 100 + 0));
157 if (result.find_first_of("Corruption: block checksum mismatch") !=
158 std::string::npos) {
159 found_corruption = true;
160 }
161 }
162 ASSERT_TRUE(found_corruption);
163 }
164
165 // UN-corrupt file for next iteration
166 ASSERT_OK(test::CorruptFile(env_, sst_file, static_cast<int>(pos), 1,
167 /*verify checksum fails*/ false));
168 }
169
170 SyncPoint::GetInstance()->DisableProcessing();
171}

274 TablePropertiesCollection TestGetPropertiesOfTablesInRange(
275 std::vector<Range> ranges, std::size_t* num_properties = nullptr,
276 std::size_t* num_files = nullptr) {
277 // Since we deref zero element in the vector it can not be empty
278 // otherwise we pass an address to some random memory
279 EXPECT_GT(ranges.size(), 0U);
280 // run the query
281 TablePropertiesCollection props;
282 ColumnFamilyHandle* default_cf = db_->DefaultColumnFamily();
283 EXPECT_OK(db_->GetPropertiesOfTablesInRange(default_cf, ranges.data(),
284 ranges.size(), &props));
285
286 const Comparator* ucmp = default_cf->GetComparator();
287 EXPECT_NE(ucmp, nullptr);
288 const size_t ts_sz = ucmp->timestamp_size();
289 const size_t range_size = ranges.size();
290 autovector<UserKeyRange> ukey_ranges;
291 std::vector<std::string> keys;
292 ukey_ranges.reserve(range_size);
293 keys.reserve(range_size * 2);
294 for (auto& r : ranges) {
295 auto [start, limit] = MaybeAddTimestampsToRange(
296 &r.start, &r.limit, ts_sz, &keys.emplace_back(), &keys.emplace_back(),
297 /*exclusive_end=*/false);
298 EXPECT_TRUE(start.has_value());
299 EXPECT_TRUE(limit.has_value());
300 ukey_ranges.emplace_back(start.value(), limit.value());
301 }
302 // Make sure that we've received properties for those and for those files
303 // only which fall within requested ranges
304 std::vector<LiveFileMetaData> vmd;
305 db_->GetLiveFilesMetaData(&vmd);
306 for (auto& md : vmd) {
307 std::string fn = md.db_path + md.name;
308 bool in_range = false;
309 for (auto& r : ukey_ranges) {
310 if (ucmp->Compare(r.start, md.largestkey) <= 0 &&
311 ucmp->Compare(r.limit, md.smallestkey) >= 0) {
312 in_range = true;
313 EXPECT_GT(props.count(fn), 0);
314 }
315 }
316 if (!in_range) {
317 EXPECT_EQ(props.count(fn), 0);
318 }
319 }
320
321 if (num_properties) {
322 *num_properties = props.size();
323 }
324
325 if (num_files) {
326 *num_files = vmd.size();
327 }
328 return props;
329 }
330
331 bool enable_udt_ = false;
332 Slice min_ts_ = MinU64Ts();
333};

335TEST_P(DBTablePropertiesInRangeTest, GetPropertiesOfTablesInRange) {
336 // Fixed random sead
337 Random rnd(301);
338
339 Options options;
340 options.level_compaction_dynamic_level_bytes = false;
341 options.create_if_missing = true;
342 options.write_buffer_size = 4096;
343 options.max_write_buffer_number = 2;
344 options.level0_file_num_compaction_trigger = 2;
345 options.level0_slowdown_writes_trigger = 2;
346 options.level0_stop_writes_trigger = 2;
347 options.target_file_size_base = 2048;
348 options.max_bytes_for_level_base = 40960;
349 options.max_bytes_for_level_multiplier = 4;
350 options.hard_pending_compaction_bytes_limit = 16 * 1024;
351 options.num_levels = 8;
352 options.env = env_;
353 bool udt_enabled = GetParam();
354 if (udt_enabled) {
355 options.comparator = test::BytewiseComparatorWithU64TsWrapper();
356 }
357
358 DestroyAndReopen(options);
359
360 // build a decent LSM
361 for (int i = 0; i < 10000; i++) {
362 PutKeyValue(test::RandomKey(&rnd, 5), rnd.RandomString(102));
363 }
364 ASSERT_OK(Flush());
365 ASSERT_OK(dbfull()->TEST_WaitForCompact());
366 if (NumTableFilesAtLevel(0) == 0) {
367 PutKeyValue(test::RandomKey(&rnd, 5), rnd.RandomString(102));
368 ASSERT_OK(Flush());
369 }
370
371 ASSERT_OK(db_->PauseBackgroundWork());
372
373 // Ensure that we have at least L0, L1 and L2
374 ASSERT_GT(NumTableFilesAtLevel(0), 0);
375 ASSERT_GT(NumTableFilesAtLevel(1), 0);
376 ASSERT_GT(NumTableFilesAtLevel(2), 0);
377
378 // Query the largest range
379 std::size_t num_properties, num_files;
380 TestGetPropertiesOfTablesInRange(
381 {Range(test::RandomKey(&rnd, 5, test::RandomKeyType::SMALLEST),
382 test::RandomKey(&rnd, 5, test::RandomKeyType::LARGEST))},
383 &num_properties, &num_files);
384 ASSERT_EQ(num_properties, num_files);
385
386 // Query the empty range
387 TestGetPropertiesOfTablesInRange(
388 {Range(test::RandomKey(&rnd, 5, test::RandomKeyType::LARGEST),
389 test::RandomKey(&rnd, 5, test::RandomKeyType::SMALLEST))},
390 &num_properties, &num_files);
391 ASSERT_GT(num_files, 0);
392 ASSERT_EQ(num_properties, 0);
393
394 // Query the middle rangee
395 TestGetPropertiesOfTablesInRange(
396 {Range(test::RandomKey(&rnd, 5, test::RandomKeyType::MIDDLE),
397 test::RandomKey(&rnd, 5, test::RandomKeyType::LARGEST))},
398 &num_properties, &num_files);
399 ASSERT_GT(num_files, 0);
400 ASSERT_GT(num_files, num_properties);
401 ASSERT_GT(num_properties, 0);
402
403 // Query a bunch of random ranges
404 for (int j = 0; j < 100; j++) {
405 // create a bunch of ranges
406 std::vector<std::string> random_keys;
407 // Random returns numbers with zero included
408 // when we pass empty ranges TestGetPropertiesOfTablesInRange()
409 // derefs random memory in the empty ranges[0]
410 // so want to be greater than zero and even since
411 // the below loop requires that random_keys.size() to be even.
412 auto n = 2 * (rnd.Uniform(50) + 1);
413
414 for (uint32_t i = 0; i < n; ++i) {
415 random_keys.push_back(test::RandomKey(&rnd, 5));
416 }
417
418 ASSERT_GT(random_keys.size(), 0U);
419 ASSERT_EQ((random_keys.size() % 2), 0U);
420
421 std::vector<Range> ranges;
422 auto it = random_keys.begin();
423 while (it != random_keys.end()) {
424 ranges.emplace_back(*it, *(it + 1));
425 it += 2;
426 }
427
428 TestGetPropertiesOfTablesInRange(std::move(ranges));
429 }
430}

484 struct JunkTablePropertiesCollector : public TablePropertiesCollector {
485 const char* Name() const override { return "JunkTablePropertiesCollector"; }
486 Status Finish(UserCollectedProperties* properties) override {
487 properties->insert({"Junk", "Junk"});
488 return Status::OK();
489 }
490 UserCollectedProperties GetReadableProperties() const override {
491 return {};
492 }
493 };
494
495 // Alternates between putting a "Junk" property and using `nullptr` to
496 // opt out.
497 static RelaxedAtomic<int> count{0};

499 : public TablePropertiesCollectorFactory {
500 const char* Name() const override {
501 return "SometimesTablePropertiesCollectorFactory";
502 }
503 TablePropertiesCollector* CreateTablePropertiesCollector(
504 TablePropertiesCollectorFactory::Context /*context*/) override {
505 if (count.FetchAddRelaxed(1) & 1) {
506 return nullptr;
507 } else {
508 return new JunkTablePropertiesCollector();
509 }
510 }
511 };
512
513 Options options = CurrentOptions();
514 options.table_properties_collector_factories.emplace_back(
515 std::make_shared<SometimesTablePropertiesCollectorFactory>());
516 // For plain table
517 options.prefix_extractor.reset(NewFixedPrefixTransform(4));
518 for (const std::shared_ptr<TableFactory>& tf :
519 {options.table_factory,
520 std::shared_ptr<TableFactory>(NewPlainTableFactory({}))}) {
521 SCOPED_TRACE("Table factory = " + std::string(tf->Name()));
522 options.table_factory = tf;
523
524 DestroyAndReopen(options);
525
526 ASSERT_OK(Put("key0", "value1"));
527 ASSERT_OK(Flush());
528 ASSERT_OK(Put("key0", "value2"));
529 ASSERT_OK(Flush());
530
531 TablePropertiesCollection props;
532 ASSERT_OK(db_->GetPropertiesOfAllTables(&props));
533 int no_junk_count = 0;
534 int junk_count = 0;
535 for (const auto& item : props) {
536 if (item.second->user_collected_properties.find("Junk") !=
537 item.second->user_collected_properties.end()) {
538 junk_count++;
539 } else {
540 no_junk_count++;
541 }
542 }
543 EXPECT_EQ(1, no_junk_count);
544 EXPECT_EQ(1, junk_count);

610TEST_P(DBTablePropertiesTest, DeletionTriggeredCompactionMarking) {
611 int kNumKeys = 1000;
612 int kWindowSize = 100;
613 int kNumDelsTrigger = 90;
614 std::shared_ptr<TablePropertiesCollectorFactory> compact_on_del =
615 NewCompactOnDeletionCollectorFactory(kWindowSize, kNumDelsTrigger);
616
617 Options opts = CurrentOptions();
618 opts.statistics = ROCKSDB_NAMESPACE::CreateDBStatistics();
619 opts.table_properties_collector_factories.emplace_back(compact_on_del);
620
621 if (GetParam() == "kCompactionStyleUniversal") {
622 opts.compaction_style = kCompactionStyleUniversal;
623 }
624 Reopen(opts);
625
626 // add an L1 file to prevent tombstones from dropping due to obsolescence
627 // during flush
628 ASSERT_OK(Put(Key(0), "val"));
629 ASSERT_OK(Flush());
630 MoveFilesToLevel(1);
631
632 DeletionTriggeredCompactionTestListener* listener =
633 new DeletionTriggeredCompactionTestListener();
634 opts.listeners.emplace_back(listener);
635 Reopen(opts);
636
637 for (int i = 0; i < kNumKeys; ++i) {
638 if (i >= kNumKeys - kWindowSize &&
639 i < kNumKeys - kWindowSize + kNumDelsTrigger) {
640 ASSERT_OK(Delete(Key(i)));
641 } else {
642 ASSERT_OK(Put(Key(i), "val"));
643 }
644 }
645 ASSERT_OK(Flush());
646
647 ASSERT_OK(dbfull()->TEST_WaitForCompact());
648 ASSERT_EQ(0, NumTableFilesAtLevel(0));
649
650 // Change the window size and deletion trigger and ensure new values take
651 // effect
652 kWindowSize = 50;
653 kNumDelsTrigger = 40;
654 static_cast<CompactOnDeletionCollectorFactory*>(compact_on_del.get())
655 ->SetWindowSize(kWindowSize);
