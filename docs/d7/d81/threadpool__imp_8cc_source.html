99 int ReserveThreads(int threads_to_be_reserved) {
100 std::unique_lock<std::mutex> lock(mu_);
101 // We can reserve at most num_waiting_threads_ in total so the number of
102 // threads that can be reserved might be fewer than the desired one. In
103 // rare cases, num_waiting_threads_ could be less than reserved_threads
104 // due to SetBackgroundThreadInternal or last excessive threads. If that
105 // happens, we cannot reserve any other threads.
106 int reserved_threads_in_success =
107 std::min(std::max(num_waiting_threads_ - reserved_threads_, 0),
108 threads_to_be_reserved);
109 reserved_threads_ += reserved_threads_in_success;
110 return reserved_threads_in_success;
111 }
112
113 int ReleaseThreads(int threads_to_be_released) {
114 std::unique_lock<std::mutex> lock(mu_);
115 // We cannot release more than reserved_threads_
116 int released_threads_in_success =
117 std::min(reserved_threads_, threads_to_be_released);
118 reserved_threads_ -= released_threads_in_success;
119 WakeUpAllThreads();
120 return released_threads_in_success;
121 }

179void ThreadPoolImpl::Impl::JoinThreads(bool wait_for_jobs_to_complete) {
180 std::unique_lock<std::mutex> lock(mu_);
181 assert(!exit_all_threads_);
182
183 wait_for_jobs_to_complete_ = wait_for_jobs_to_complete;
184 exit_all_threads_ = true;
185 // prevent threads from being recreated right after they're joined, in case
186 // the user is concurrently submitting jobs.
187 total_threads_limit_ = 0;
188 reserved_threads_ = 0;
189 num_waiting_threads_ = 0;
190
191 lock.unlock();
192
193 bgsignal_.notify_all();
194
195 for (auto& th : bgthreads_) {
196 th.join();
197 }
198
199 bgthreads_.clear();
200
201 exit_all_threads_ = false;
202 wait_for_jobs_to_complete_ = false;
203}

215void ThreadPoolImpl::Impl::BGThread(size_t thread_id) {
216 bool low_io_priority = false;
217 CpuPriority current_cpu_priority = CpuPriority::kNormal;
218
219 while (true) {
220 // Wait until there is an item that is ready to run
221 std::unique_lock<std::mutex> lock(mu_);
222 // Stop waiting if the thread needs to do work or needs to terminate.
223 // Increase num_waiting_threads_ once this task has started waiting
224 num_waiting_threads_++;
225
226 TEST_SYNC_POINT("ThreadPoolImpl::BGThread::WaitingThreadsInc");
227 TEST_IDX_SYNC_POINT("ThreadPoolImpl::BGThread::Start:th", thread_id);
228 // When not exist_all_threads and the current thread id is not the last
229 // excessive thread, it may be blocked due to 3 reasons: 1) queue is empty
230 // 2) it is the excessive thread (not the last one)
231 // 3) the number of waiting threads is not greater than reserved threads
232 // (i.e, no available threads due to full reservation")
233 while (!exit_all_threads_ && !IsLastExcessiveThread(thread_id) &&
234 (queue_.empty() || IsExcessiveThread(thread_id) ||
235 num_waiting_threads_ <= reserved_threads_)) {
236 bgsignal_.wait(lock);
237 }
238 // Decrease num_waiting_threads_ once the thread is not waiting
239 num_waiting_threads_--;
240
241 if (exit_all_threads_) { // mechanism to let BG threads exit safely
242
243 if (!wait_for_jobs_to_complete_ || queue_.empty()) {
244 break;
245 }
246 } else if (IsLastExcessiveThread(thread_id)) {
247 // Current thread is the last generated one and is excessive.
248 // We always terminate excessive thread in the reverse order of
249 // generation time. But not when `exit_all_threads_ == true`,
250 // otherwise `JoinThreads()` could try to `join()` a `detach()`ed
251 // thread.
252 auto& terminating_thread = bgthreads_.back();
253 terminating_thread.detach();
254 bgthreads_.pop_back();
255 if (HasExcessiveThread()) {
256 // There is still at least more excessive thread to terminate.
257 WakeUpAllThreads();
258 }
259 TEST_IDX_SYNC_POINT("ThreadPoolImpl::BGThread::Termination:th",
260 thread_id);
261 TEST_SYNC_POINT("ThreadPoolImpl::BGThread::Termination");
262 break;
263 }
264
265 auto func = std::move(queue_.front().function);
266 queue_.pop_front();
267
268 queue_len_.store(static_cast<unsigned int>(queue_.size()),
269 std::memory_order_relaxed);
270
271 bool decrease_io_priority = (low_io_priority != low_io_priority_);
272 CpuPriority cpu_priority = cpu_priority_;
273 lock.unlock();
274
275 if (cpu_priority < current_cpu_priority) {
276 TEST_SYNC_POINT_CALLBACK("ThreadPoolImpl::BGThread::BeforeSetCpuPriority",
277 &current_cpu_priority);
278 // 0 means current thread.
279 port::SetCpuPriority(0, cpu_priority);
280 current_cpu_priority = cpu_priority;
281 TEST_SYNC_POINT_CALLBACK("ThreadPoolImpl::BGThread::AfterSetCpuPriority",
282 &current_cpu_priority);
283 }
284
285#ifdef OS_LINUX
286 if (decrease_io_priority) {
287#define IOPRIO_CLASS_SHIFT (13)
288#define IOPRIO_PRIO_VALUE(class, data) (((class) << IOPRIO_CLASS_SHIFT) | data)
289 // Put schedule into IOPRIO_CLASS_IDLE class (lowest)
290 // These system calls only have an effect when used in conjunction
291 // with an I/O scheduler that supports I/O priorities. As at
292 // kernel 2.6.17 the only such scheduler is the Completely
293 // Fair Queuing (CFQ) I/O scheduler.
294 // To change scheduler:
295 // echo cfq > /sys/block/<device_name>/queue/schedule
296 // Tunables to consider:
297 // /sys/block/<device_name>/queue/slice_idle
298 // /sys/block/<device_name>/queue/slice_sync
299 syscall(SYS_ioprio_set, 1, // IOPRIO_WHO_PROCESS
300 0, // current thread
301 IOPRIO_PRIO_VALUE(3, 0));
302 low_io_priority = true;
303 }
304#else
305 (void)decrease_io_priority; // avoid 'unused variable' error
306#endif
307
308 TEST_SYNC_POINT_CALLBACK("ThreadPoolImpl::Impl::BGThread:BeforeRun",
309 &priority_);
310
311 func();
312 }

323void ThreadPoolImpl::Impl::BGThreadWrapper(void* arg) {
324 BGThreadMetadata* meta = static_cast<BGThreadMetadata*>(arg);
325 size_t thread_id = meta->thread_id_;
326 ThreadPoolImpl::Impl* tp = meta->thread_pool_;
327#ifdef ROCKSDB_USING_THREAD_STATUS
328 // initialize it because compiler isn't good enough to see we don't use it
329 // uninitialized
330 ThreadStatus::ThreadType thread_type = ThreadStatus::NUM_THREAD_TYPES;
331 switch (tp->GetThreadPriority()) {
332 case Env::Priority::HIGH:
333 thread_type = ThreadStatus::HIGH_PRIORITY;
334 break;
335 case Env::Priority::LOW:
336 thread_type = ThreadStatus::LOW_PRIORITY;
337 break;
338 case Env::Priority::BOTTOM:
339 thread_type = ThreadStatus::BOTTOM_PRIORITY;
340 break;
341 case Env::Priority::USER:
342 thread_type = ThreadStatus::USER;
343 break;
344 case Env::Priority::TOTAL:
345 assert(false);
346 return;
347 }
348 assert(thread_type != ThreadStatus::NUM_THREAD_TYPES);
349 ThreadStatusUtil::RegisterThread(tp->GetHostEnv(), thread_type);
350#endif
351 delete meta;
352 tp->BGThread(thread_id);
353#ifdef ROCKSDB_USING_THREAD_STATUS
354 ThreadStatusUtil::UnregisterThread();
355#endif
356 return;
357}

359void ThreadPoolImpl::Impl::SetBackgroundThreadsInternal(int num,
360 bool allow_reduce) {
361 std::lock_guard<std::mutex> lock(mu_);
362 if (exit_all_threads_) {
363 return;
364 }
365 if (num > total_threads_limit_ ||
366 (num < total_threads_limit_ && allow_reduce)) {
367 total_threads_limit_ = std::max(0, num);
368 WakeUpAllThreads();
369 StartBGThreads();
370 }
371}

374 std::unique_lock<std::mutex> lock(mu_);
375 return total_threads_limit_;

378void ThreadPoolImpl::Impl::StartBGThreads() {
379 // Start background thread if necessary
380 while ((int)bgthreads_.size() < total_threads_limit_) {
381 port::Thread p_t(&BGThreadWrapper,
382 new BGThreadMetadata(this, bgthreads_.size()));
383
384// Set the thread name to aid debugging
385#if defined(_GNU_SOURCE) && defined(__GLIBC_PREREQ)
386#if __GLIBC_PREREQ(2, 12)
387 auto th_handle = p_t.native_handle();
388 std::string thread_priority = Env::PriorityToString(GetThreadPriority());
389 std::ostringstream thread_name_stream;
390 thread_name_stream << "rocksdb:";
391 for (char c : thread_priority) {
392 thread_name_stream << static_cast<char>(tolower(c));
393 }
394 pthread_setname_np(th_handle, thread_name_stream.str().c_str());
395#endif
396#endif
397 bgthreads_.push_back(std::move(p_t));
398 }

401void ThreadPoolImpl::Impl::Submit(std::function<void()>&& schedule,
402 std::function<void()>&& unschedule,
403 void* tag) {
404 std::lock_guard<std::mutex> lock(mu_);
405
406 if (exit_all_threads_) {
407 return;
408 }
409
410 StartBGThreads();
411
412 // Add to priority queue
413 queue_.push_back(BGItem());
414 TEST_SYNC_POINT("ThreadPoolImpl::Submit::Enqueue");
415 auto& item = queue_.back();
416 item.tag = tag;
417 item.function = std::move(schedule);
418 item.unschedFunction = std::move(unschedule);
419
420 queue_len_.store(static_cast<unsigned int>(queue_.size()),
421 std::memory_order_relaxed);
422
423 if (!HasExcessiveThread()) {
424 // Wake up at least one waiting thread.
425 bgsignal_.notify_one();
426 } else {
427 // Need to wake up all threads to make sure the one woken
428 // up is not the one to terminate.
429 WakeUpAllThreads();
430 }
431}

433int ThreadPoolImpl::Impl::UnSchedule(void* arg) {
434 int count = 0;
435
436 std::vector<std::function<void()>> candidates;
437 {
438 std::lock_guard<std::mutex> lock(mu_);
439
440 // Remove from priority queue
441 BGQueue::iterator it = queue_.begin();
442 while (it != queue_.end()) {
443 if (arg == (*it).tag) {
444 if (it->unschedFunction) {
445 candidates.push_back(std::move(it->unschedFunction));
446 }
447 it = queue_.erase(it);
448 count++;
449 } else {
450 ++it;
451 }
452 }
453 queue_len_.store(static_cast<unsigned int>(queue_.size()),
454 std::memory_order_relaxed);
455 }
456
457 // Run unschedule functions outside the mutex
458 for (auto& f : candidates) {
459 f();
460 }
461
462 return count;
463}
