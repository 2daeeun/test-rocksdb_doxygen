39// A structure that describes the current status of a thread.
40// The status of active threads can be fetched using
41// ROCKSDB_NAMESPACE::GetThreadList().
42struct ThreadStatus {
43 // The type of a thread.
44 enum ThreadType : int {
45 HIGH_PRIORITY = 0, // RocksDB BG thread in high-pri thread pool
46 LOW_PRIORITY, // RocksDB BG thread in low-pri thread pool
47 USER, // User thread (Non-RocksDB BG thread)
48 BOTTOM_PRIORITY, // RocksDB BG thread in bottom-pri thread pool
49 NUM_THREAD_TYPES
50 };
51
52 // The type used to refer to a thread operation.
53 // A thread operation describes high-level action of a thread.
54 // Examples include compaction and flush.
55 enum OperationType : int {
56 OP_UNKNOWN = 0,
57 OP_COMPACTION,
58 OP_FLUSH,
59 OP_DBOPEN,
60 OP_GET,
61 OP_MULTIGET,
62 OP_DBITERATOR,
63 OP_VERIFY_DB_CHECKSUM,
64 OP_VERIFY_FILE_CHECKSUMS,
65 OP_GETENTITY,
66 OP_MULTIGETENTITY,
67 OP_READ_MANIFEST,
68 NUM_OP_TYPES
69 };
70
71 enum OperationStage : int {
72 STAGE_UNKNOWN = 0,
73 STAGE_FLUSH_RUN,
74 STAGE_FLUSH_WRITE_L0,
75 STAGE_COMPACTION_PREPARE,
76 STAGE_COMPACTION_RUN,
77 STAGE_COMPACTION_PROCESS_KV,
78 STAGE_COMPACTION_INSTALL,
79 STAGE_COMPACTION_SYNC_FILE,
80 STAGE_PICK_MEMTABLES_TO_FLUSH,
81 STAGE_MEMTABLE_ROLLBACK,
82 STAGE_MEMTABLE_INSTALL_FLUSH_RESULTS,
83 NUM_OP_STAGES
84 };
85
86 enum CompactionPropertyType : int {
87 COMPACTION_JOB_ID = 0,
88 COMPACTION_INPUT_OUTPUT_LEVEL,
89 COMPACTION_PROP_FLAGS,
90 COMPACTION_TOTAL_INPUT_BYTES,
91 COMPACTION_BYTES_READ,
92 COMPACTION_BYTES_WRITTEN,
93 NUM_COMPACTION_PROPERTIES
94 };
95
96 enum FlushPropertyType : int {
97 FLUSH_JOB_ID = 0,
98 FLUSH_BYTES_MEMTABLES,
99 FLUSH_BYTES_WRITTEN,
100 NUM_FLUSH_PROPERTIES
101 };
102
103 // The maximum number of properties of an operation.
104 // This number should be set to the biggest NUM_XXX_PROPERTIES.
105 static const int kNumOperationProperties =
106 constexpr_max<NUM_COMPACTION_PROPERTIES, NUM_FLUSH_PROPERTIES>::result;
107
108 // The type used to refer to a thread state.
109 // A state describes lower-level action of a thread
110 // such as reading / writing a file or waiting for a mutex.
111 enum StateType : int {
112 STATE_UNKNOWN = 0,
113 STATE_MUTEX_WAIT = 1,
114 NUM_STATE_TYPES
115 };
116
117 ThreadStatus(const uint64_t _id, const ThreadType _thread_type,
118 const std::string& _db_name, const std::string& _cf_name,
119 const OperationType _operation_type,
120 const uint64_t _op_elapsed_micros,
121 const OperationStage _operation_stage,
122 const uint64_t _op_props[], const StateType _state_type)
123 : thread_id(_id),
124 thread_type(_thread_type),
125 db_name(_db_name),
126 cf_name(_cf_name),
127 operation_type(_operation_type),
128 op_elapsed_micros(_op_elapsed_micros),
129 operation_stage(_operation_stage),
130 state_type(_state_type) {
131 for (int i = 0; i < kNumOperationProperties; ++i) {
132 op_properties[i] = _op_props[i];
133 }
134 }
135
136 // An unique ID for the thread.
137 const uint64_t thread_id;
138
139 // The type of the thread, it could be HIGH_PRIORITY,
140 // LOW_PRIORITY, and USER
141 const ThreadType thread_type;
142
143 // The name of the DB instance where the thread is currently
144 // involved with. It would be set to empty string if the thread
145 // does not involve in any DB operation.
146 const std::string db_name;
147
148 // The name of the column family where the thread is currently
149 // It would be set to empty string if the thread does not involve
150 // in any column family.
151 const std::string cf_name;
152
153 // The operation (high-level action) that the current thread is involved.
154 const OperationType operation_type;
155
156 // The elapsed time of the current thread operation in microseconds.
157 const uint64_t op_elapsed_micros;
158
159 // An integer showing the current stage where the thread is involved
160 // in the current operation.
161 const OperationStage operation_stage;
162
163 // A list of properties that describe some details about the current
164 // operation. Same field in op_properties[] might have different
165 // meanings for different operations.
166 uint64_t op_properties[kNumOperationProperties];
167
168 // The state (lower-level action) that the current thread is involved.
169 const StateType state_type;
170
171 // The followings are a set of utility functions for interpreting
172 // the information of ThreadStatus
173
174 static std::string GetThreadTypeName(ThreadType thread_type);
175
176 // Obtain the name of an operation given its type.
177 static const std::string& GetOperationName(OperationType op_type);
178
179 static const std::string MicrosToString(uint64_t op_elapsed_time);
180
181 // Obtain a human-readable string describing the specified operation stage.
182 static const std::string& GetOperationStageName(OperationStage stage);
183
184 // Obtain the name of the "i"th operation property of the
185 // specified operation.
186 static const std::string& GetOperationPropertyName(OperationType op_type,
187 int i);
188
189 // Translate the "i"th property of the specified operation given
190 // a property value.
191 static std::map<std::string, uint64_t> InterpretOperationProperties(
192 OperationType op_type, const uint64_t* op_properties);
193
194 // Obtain the name of a state given its type.
195 static const std::string& GetStateName(StateType state_type);
196};

198} // namespace ROCKSDB_NAMESPACE
