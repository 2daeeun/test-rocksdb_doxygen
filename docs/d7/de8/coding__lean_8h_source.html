24inline void EncodeFixed16(char* buf, uint16_t value) {
25 if (port::kLittleEndian) {
26 memcpy(buf, &value, sizeof(value));
27 } else {
28 buf[0] = value & 0xff;
29 buf[1] = (value >> 8) & 0xff;
30 }
31}

62inline uint16_t DecodeFixed16(const char* ptr) {
63 if (port::kLittleEndian) {
64 // Load the raw bytes
65 uint16_t result;
66 memcpy(&result, ptr, sizeof(result)); // gcc optimizes this to a plain load
67 return result;
68 } else {
69 return ((static_cast<uint16_t>(static_cast<unsigned char>(ptr[0]))) |
70 (static_cast<uint16_t>(static_cast<unsigned char>(ptr[1])) << 8));
71 }
72}

74inline uint32_t DecodeFixed32(const char* ptr) {
75 if (port::kLittleEndian) {
76 // Load the raw bytes
77 uint32_t result;
78 memcpy(&result, ptr, sizeof(result)); // gcc optimizes this to a plain load
79 return result;
80 } else {
81 return ((static_cast<uint32_t>(static_cast<unsigned char>(ptr[0]))) |
82 (static_cast<uint32_t>(static_cast<unsigned char>(ptr[1])) << 8) |
83 (static_cast<uint32_t>(static_cast<unsigned char>(ptr[2])) << 16) |
84 (static_cast<uint32_t>(static_cast<unsigned char>(ptr[3])) << 24));
85 }
86}

88inline uint64_t DecodeFixed64(const char* ptr) {
89 if (port::kLittleEndian) {
90 // Load the raw bytes
91 uint64_t result;
92 memcpy(&result, ptr, sizeof(result)); // gcc optimizes this to a plain load
93 return result;
94 } else {
95 uint64_t lo = DecodeFixed32(ptr);
96 uint64_t hi = DecodeFixed32(ptr + 4);
97 return (hi << 32) | lo;
98 }
99}
