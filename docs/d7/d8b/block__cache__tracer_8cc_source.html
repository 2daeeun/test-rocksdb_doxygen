24bool ShouldTrace(const Slice& block_key,
25 const BlockCacheTraceOptions& trace_options) {
26 if (trace_options.sampling_frequency == 0 ||
27 trace_options.sampling_frequency == 1) {
28 return true;
29 }
30 // We use spatial downsampling so that we have a complete access history for a
31 // block.
32 return 0 == GetSliceRangedNPHash(block_key, trace_options.sampling_frequency);
33}

63std::string BlockCacheTraceHelper::ComputeRowKey(
64 const BlockCacheTraceRecord& access) {
65 if (!IsGetOrMultiGet(access.caller)) {
66 return "";
67 }
68 Slice key = ExtractUserKey(access.referenced_key);
69 return std::to_string(access.sst_fd_number) + "_" + key.ToString();

72uint64_t BlockCacheTraceHelper::GetTableId(
73 const BlockCacheTraceRecord& access) {
74 if (!IsGetOrMultiGet(access.caller) || access.referenced_key.size() < 4) {
75 return 0;
76 }
77 return static_cast<uint64_t>(DecodeFixed32(access.referenced_key.data())) + 1;
78}

80uint64_t BlockCacheTraceHelper::GetSequenceNumber(
81 const BlockCacheTraceRecord& access) {
82 if (!IsGetOrMultiGet(access.caller)) {
83 return 0;
84 }
85 if (access.caller == TableReaderCaller::kUserMultiGet &&
86 access.referenced_key.size() < 4) {
87 return 0;
88 }
89 return access.get_from_user_specified_snapshot
90 ? 1 + GetInternalKeySeqno(access.referenced_key)
91 : 0;
92}

94uint64_t BlockCacheTraceHelper::GetBlockOffsetInFile(
95 const BlockCacheTraceRecord& access) {
96 Slice input(access.block_key);
97 uint64_t offset = 0;
98 while (true) {
99 uint64_t tmp = 0;
100 if (GetVarint64(&input, &tmp)) {
101 offset = tmp;
102 } else {
103 break;
104 }
105 }
106 return offset;
107}

116Status BlockCacheTraceWriterImpl::WriteBlockAccess(
117 const BlockCacheTraceRecord& record, const Slice& block_key,
118 const Slice& cf_name, const Slice& referenced_key) {
119 uint64_t trace_file_size = trace_writer_->GetFileSize();
120 if (trace_file_size > trace_options_.max_trace_file_size) {
121 return Status::OK();
122 }
123 Trace trace;
124 trace.ts = record.access_timestamp;
125 trace.type = record.block_type;
126 PutLengthPrefixedSlice(&trace.payload, block_key);
127 PutFixed64(&trace.payload, record.block_size);
128 PutFixed64(&trace.payload, record.cf_id);
129 PutLengthPrefixedSlice(&trace.payload, cf_name);
130 PutFixed32(&trace.payload, record.level);
131 PutFixed64(&trace.payload, record.sst_fd_number);
132 trace.payload.push_back(record.caller);
133 trace.payload.push_back(record.is_cache_hit);
134 trace.payload.push_back(record.no_insert);
135 if (BlockCacheTraceHelper::IsGetOrMultiGet(record.caller)) {
136 PutFixed64(&trace.payload, record.get_id);
137 trace.payload.push_back(record.get_from_user_specified_snapshot);
138 PutLengthPrefixedSlice(&trace.payload, referenced_key);
139 }
140 if (BlockCacheTraceHelper::IsGetOrMultiGetOnDataBlock(record.block_type,
141 record.caller)) {
142 PutFixed64(&trace.payload, record.referenced_data_size);
143 PutFixed64(&trace.payload, record.num_keys_in_block);
144 trace.payload.push_back(record.referenced_key_exist_in_block);
145 }
146 std::string encoded_trace;
147 TracerHelper::EncodeTrace(trace, &encoded_trace);
148 return trace_writer_->Write(encoded_trace);
149}

151Status BlockCacheTraceWriterImpl::WriteHeader() {
152 Trace trace;
153 trace.ts = clock_->NowMicros();
154 trace.type = TraceType::kTraceBegin;
155 PutLengthPrefixedSlice(&trace.payload, kTraceMagic);
156 PutFixed32(&trace.payload, kMajorVersion);
157 PutFixed32(&trace.payload, kMinorVersion);
158 std::string encoded_trace;
159 TracerHelper::EncodeTrace(trace, &encoded_trace);
160 return trace_writer_->Write(encoded_trace);
161}

167Status BlockCacheTraceReader::ReadHeader(BlockCacheTraceHeader* header) {
168 assert(header != nullptr);
169 std::string encoded_trace;
170 Status s = trace_reader_->Read(&encoded_trace);
171 if (!s.ok()) {
172 return s;
173 }
174 Trace trace;
175 s = TracerHelper::DecodeTrace(encoded_trace, &trace);
176 if (!s.ok()) {
177 return s;
178 }
179 header->start_time = trace.ts;
180 Slice enc_slice = Slice(trace.payload);
181 Slice magnic_number;
182 if (!GetLengthPrefixedSlice(&enc_slice, &magnic_number)) {
183 return Status::Corruption(
184 "Corrupted header in the trace file: Failed to read the magic number.");
185 }
186 if (magnic_number.ToString() != kTraceMagic) {
187 return Status::Corruption(
188 "Corrupted header in the trace file: Magic number does not match.");
189 }
190 if (!GetFixed32(&enc_slice, &header->rocksdb_major_version)) {
191 return Status::Corruption(
192 "Corrupted header in the trace file: Failed to read rocksdb major "
193 "version number.");
194 }
195 if (!GetFixed32(&enc_slice, &header->rocksdb_minor_version)) {
196 return Status::Corruption(
197 "Corrupted header in the trace file: Failed to read rocksdb minor "
198 "version number.");
199 }
200 // We should have retrieved all information in the header.
201 if (!enc_slice.empty()) {
202 return Status::Corruption(
203 "Corrupted header in the trace file: The length of header is too "
204 "long.");
205 }
206 return Status::OK();
207}

209Status BlockCacheTraceReader::ReadAccess(BlockCacheTraceRecord* record) {
210 assert(record);
211 std::string encoded_trace;
212 Status s = trace_reader_->Read(&encoded_trace);
213 if (!s.ok()) {
214 return s;
215 }
216 Trace trace;
217 s = TracerHelper::DecodeTrace(encoded_trace, &trace);
218 if (!s.ok()) {
219 return s;
220 }
221 record->access_timestamp = trace.ts;
222 record->block_type = trace.type;
223 Slice enc_slice = Slice(trace.payload);
224
225 const unsigned int kCharSize = 1;
226
227 Slice block_key;
228 if (!GetLengthPrefixedSlice(&enc_slice, &block_key)) {
229 return Status::Incomplete(
230 "Incomplete access record: Failed to read block key.");
231 }
232 record->block_key = block_key.ToString();
233 if (!GetFixed64(&enc_slice, &record->block_size)) {
234 return Status::Incomplete(
235 "Incomplete access record: Failed to read block size.");
236 }
237 if (!GetFixed64(&enc_slice, &record->cf_id)) {
238 return Status::Incomplete(
239 "Incomplete access record: Failed to read column family ID.");
240 }
241 Slice cf_name;
242 if (!GetLengthPrefixedSlice(&enc_slice, &cf_name)) {
243 return Status::Incomplete(
244 "Incomplete access record: Failed to read column family name.");
245 }
246 record->cf_name = cf_name.ToString();
247 if (!GetFixed32(&enc_slice, &record->level)) {
248 return Status::Incomplete(
249 "Incomplete access record: Failed to read level.");
250 }
251 if (!GetFixed64(&enc_slice, &record->sst_fd_number)) {
252 return Status::Incomplete(
253 "Incomplete access record: Failed to read SST file number.");
254 }
255 if (enc_slice.empty()) {
256 return Status::Incomplete(
257 "Incomplete access record: Failed to read caller.");
258 }
259 record->caller = static_cast<TableReaderCaller>(enc_slice[0]);
260 enc_slice.remove_prefix(kCharSize);
261 if (enc_slice.empty()) {
262 return Status::Incomplete(
263 "Incomplete access record: Failed to read is_cache_hit.");
264 }
265 record->is_cache_hit = static_cast<char>(enc_slice[0]);
266 enc_slice.remove_prefix(kCharSize);
267 if (enc_slice.empty()) {
268 return Status::Incomplete(
269 "Incomplete access record: Failed to read no_insert.");
270 }
271 record->no_insert = static_cast<char>(enc_slice[0]);
272 enc_slice.remove_prefix(kCharSize);
273 if (BlockCacheTraceHelper::IsGetOrMultiGet(record->caller)) {
274 if (!GetFixed64(&enc_slice, &record->get_id)) {
275 return Status::Incomplete(
276 "Incomplete access record: Failed to read the get id.");
277 }
278 if (enc_slice.empty()) {
279 return Status::Incomplete(
280 "Incomplete access record: Failed to read "
281 "get_from_user_specified_snapshot.");
282 }
283 record->get_from_user_specified_snapshot = static_cast<char>(enc_slice[0]);
284 enc_slice.remove_prefix(kCharSize);
285 Slice referenced_key;
286 if (!GetLengthPrefixedSlice(&enc_slice, &referenced_key)) {
287 return Status::Incomplete(
288 "Incomplete access record: Failed to read the referenced key.");
289 }
290 record->referenced_key = referenced_key.ToString();
291 }
292 if (BlockCacheTraceHelper::IsGetOrMultiGetOnDataBlock(record->block_type,
293 record->caller)) {
294 if (!GetFixed64(&enc_slice, &record->referenced_data_size)) {
295 return Status::Incomplete(
296 "Incomplete access record: Failed to read the referenced data size.");
297 }
298 if (!GetFixed64(&enc_slice, &record->num_keys_in_block)) {
299 return Status::Incomplete(
300 "Incomplete access record: Failed to read the number of keys in the "
301 "block.");
302 }
303 if (enc_slice.empty()) {
304 return Status::Incomplete(
305 "Incomplete access record: Failed to read "
306 "referenced_key_exist_in_block.");
307 }
308 record->referenced_key_exist_in_block = static_cast<char>(enc_slice[0]);
309 }
310 return Status::OK();
311}

331Status BlockCacheHumanReadableTraceWriter::WriteHumanReadableTraceRecord(
332 const BlockCacheTraceRecord& access, uint64_t block_id,
333 uint64_t get_key_id) {
334 if (!human_readable_trace_file_writer_) {
335 return Status::OK();
336 }
337 int ret = snprintf(
338 trace_record_buffer_, sizeof(trace_record_buffer_),
339 "%" PRIu64 ",%" PRIu64 ",%u,%" PRIu64 ",%" PRIu64 ",%s,%" PRIu32
340 ",%" PRIu64 ",%u,%u,%" PRIu64 ",%" PRIu64 ",%" PRIu64 ",%u,%u,%" PRIu64
341 ",%" PRIu64 ",%" PRIu64 ",%" PRIu64 ",%" PRIu64 ",%" PRIu64 "\n",
342 access.access_timestamp, block_id, access.block_type, access.block_size,
343 access.cf_id, access.cf_name.c_str(), access.level, access.sst_fd_number,
344 access.caller, access.no_insert, access.get_id, get_key_id,
345 access.referenced_data_size, access.is_cache_hit,
346 access.referenced_key_exist_in_block, access.num_keys_in_block,
347 BlockCacheTraceHelper::GetTableId(access),
348 BlockCacheTraceHelper::GetSequenceNumber(access),
349 static_cast<uint64_t>(access.block_key.size()),
350 static_cast<uint64_t>(access.referenced_key.size()),
351 BlockCacheTraceHelper::GetBlockOffsetInFile(access));
352 if (ret < 0) {
353 return Status::IOError("failed to format the output");
354 }
355 std::string printout(trace_record_buffer_);
356 return human_readable_trace_file_writer_->Append(printout);
357}

374Status BlockCacheHumanReadableTraceReader::ReadAccess(
375 BlockCacheTraceRecord* record) {
376 std::string line;
377 if (!std::getline(human_readable_trace_reader_, line)) {
378 return Status::Incomplete("No more records to read.");
379 }
380 std::stringstream ss(line);
381 std::vector<std::string> record_strs;
382 while (ss.good()) {
383 std::string substr;
384 getline(ss, substr, ',');
385 record_strs.push_back(substr);
386 }
387 if (record_strs.size() != 21) {
388 return Status::Incomplete("Records format is wrong.");
389 }
390
391 record->access_timestamp = ParseUint64(record_strs[0]);
392 uint64_t block_key = ParseUint64(record_strs[1]);
393 record->block_type = static_cast<TraceType>(ParseUint64(record_strs[2]));
394 record->block_size = ParseUint64(record_strs[3]);
395 record->cf_id = ParseUint64(record_strs[4]);
396 record->cf_name = record_strs[5];
397 record->level = static_cast<uint32_t>(ParseUint64(record_strs[6]));
398 record->sst_fd_number = ParseUint64(record_strs[7]);
399 record->caller = static_cast<TableReaderCaller>(ParseUint64(record_strs[8]));
400 record->no_insert = static_cast<char>(ParseUint64(record_strs[9]));
401 record->get_id = ParseUint64(record_strs[10]);
402 uint64_t get_key_id = ParseUint64(record_strs[11]);
403
404 record->referenced_data_size = ParseUint64(record_strs[12]);
405 record->is_cache_hit = static_cast<char>(ParseUint64(record_strs[13]));
406 record->referenced_key_exist_in_block =
407 static_cast<char>(ParseUint64(record_strs[14]));
408 record->num_keys_in_block = ParseUint64(record_strs[15]);
409 uint64_t table_id = ParseUint64(record_strs[16]);
410 if (table_id > 0) {
411 // Decrement since valid table id in the trace file equals traced table id
412 // + 1.
413 table_id -= 1;
414 }
415 uint64_t get_sequence_number = ParseUint64(record_strs[17]);
416 if (get_sequence_number > 0) {
417 record->get_from_user_specified_snapshot = true;
418 // Decrement since valid seq number in the trace file equals traced seq
419 // number + 1.
420 get_sequence_number -= 1;
421 }
422 uint64_t block_key_size = ParseUint64(record_strs[18]);
423 uint64_t get_key_size = ParseUint64(record_strs[19]);
424 uint64_t block_offset = ParseUint64(record_strs[20]);
425
426 std::string tmp_block_key;
427 PutVarint64(&tmp_block_key, block_key);
428 PutVarint64(&tmp_block_key, block_offset);
429 // Append 1 until the size is the same as traced block key size.
430 while (record->block_key.size() < block_key_size - tmp_block_key.size()) {
431 record->block_key += "1";
432 }
433 record->block_key += tmp_block_key;
434
435 if (get_key_id != 0) {
436 std::string tmp_get_key;
437 PutFixed64(&tmp_get_key, get_key_id);
438 PutFixed64(&tmp_get_key, get_sequence_number << 8);
439 PutFixed32(&record->referenced_key, static_cast<uint32_t>(table_id));
440 // Append 1 until the size is the same as traced key size.
441 while (record->referenced_key.size() < get_key_size - tmp_get_key.size()) {
442 record->referenced_key += "1";
443 }
444 record->referenced_key += tmp_get_key;
445 }
446 return Status::OK();
447}

453Status BlockCacheTracer::StartTrace(
454 const BlockCacheTraceOptions& trace_options,
455 std::unique_ptr<BlockCacheTraceWriter>&& trace_writer) {
456 InstrumentedMutexLock lock_guard(&trace_writer_mutex_);
457 if (writer_.load()) {
458 return Status::Busy();
459 }
460 get_id_counter_.store(1);
461 trace_options_ = trace_options;
462 writer_.store(trace_writer.release());
463 return writer_.load()->WriteHeader();
464}

466void BlockCacheTracer::EndTrace() {
467 InstrumentedMutexLock lock_guard(&trace_writer_mutex_);
468 if (!writer_.load()) {
469 return;
470 }
471 delete writer_.load();
472 writer_.store(nullptr);
473}

490uint64_t BlockCacheTracer::NextGetId() {
491 if (!writer_.load(std::memory_order_relaxed)) {
492 return BlockCacheTraceHelper::kReservedGetId;
493 }
494 uint64_t prev_value = get_id_counter_.fetch_add(1);
495 if (prev_value == BlockCacheTraceHelper::kReservedGetId) {
496 // fetch and add again.
497 return get_id_counter_.fetch_add(1);
498 }
499 return prev_value;
500}
