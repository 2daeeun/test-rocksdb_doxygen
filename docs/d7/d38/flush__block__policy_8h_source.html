20// FlushBlockPolicy provides a configurable way to determine when to flush a
21// block in the block based tables.
22//
23// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
24// because RocksDB is not exception-safe. This could cause undefined behavior
25// including data loss, unreported corruption, deadlocks, and more.
26class FlushBlockPolicy {
27 public:
28 // Keep track of the key/value sequences and return the boolean value to
29 // determine if table builder should flush current data block.
30 virtual bool Update(const Slice& key, const Slice& value) = 0;
31
32 virtual ~FlushBlockPolicy() {}
33};

35class FlushBlockPolicyFactory : public Customizable {
36 public:
37 static const char* Type() { return "FlushBlockPolicyFactory"; }
38
39 // Creates a FlushBlockPolicyFactory based on the input value.
40 // By default, this method can create EveryKey or BySize PolicyFactory,
41 // which take now config_options.
42 static Status CreateFromString(
43 const ConfigOptions& config_options, const std::string& value,
44 std::shared_ptr<FlushBlockPolicyFactory>* result);
45
46 // Return a new block flush policy that flushes data blocks by data size.
47 // FlushBlockPolicy may need to access the metadata of the data block
48 // builder to determine when to flush the blocks.
49 //
50 // Callers must delete the result after any database that is using the
51 // result has been closed.
52 virtual FlushBlockPolicy* NewFlushBlockPolicy(
53 const BlockBasedTableOptions& table_options,
54 const BlockBuilder& data_block_builder) const = 0;
55
56 virtual ~FlushBlockPolicyFactory() {}
57};
