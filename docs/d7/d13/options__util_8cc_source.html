16Status LoadOptionsFromFile(const ConfigOptions& config_options,
17 const std::string& file_name, DBOptions* db_options,
18 std::vector<ColumnFamilyDescriptor>* cf_descs,
19 std::shared_ptr<Cache>* cache) {
20 RocksDBOptionsParser parser;
21 const auto& fs = config_options.env->GetFileSystem();
22 Status s = parser.Parse(config_options, file_name, fs.get());
23 if (!s.ok()) {
24 return s;
25 }
26 *db_options = *parser.db_opt();
27 const std::vector<std::string>& cf_names = *parser.cf_names();
28 const std::vector<ColumnFamilyOptions>& cf_opts = *parser.cf_opts();
29 cf_descs->clear();
30 for (size_t i = 0; i < cf_opts.size(); ++i) {
31 cf_descs->push_back({cf_names[i], cf_opts[i]});
32 if (cache != nullptr) {
33 TableFactory* tf = cf_opts[i].table_factory.get();
34 if (tf != nullptr) {
35 auto* opts = tf->GetOptions<BlockBasedTableOptions>();
36 if (opts != nullptr) {
37 opts->block_cache = *cache;
38 }
39 }
40 }
41 }
42 return Status::OK();
43}

45Status GetLatestOptionsFileName(const std::string& dbpath, Env* env,
46 std::string* options_file_name) {
47 Status s;
48 std::string latest_file_name;
49 uint64_t latest_time_stamp = 0;
50 std::vector<std::string> file_names;
51 s = env->GetChildren(dbpath, &file_names);
52 if (s.IsNotFound()) {
53 return Status::NotFound(Status::kPathNotFound,
54 "No options files found in the DB directory.",
55 dbpath);
56 } else if (!s.ok()) {
57 return s;
58 }
59 for (auto& file_name : file_names) {
60 uint64_t time_stamp;
61 FileType type;
62 if (ParseFileName(file_name, &time_stamp, &type) && type == kOptionsFile) {
63 if (time_stamp > latest_time_stamp) {
64 latest_time_stamp = time_stamp;
65 latest_file_name = file_name;
66 }
67 }
68 }
69 if (latest_file_name.size() == 0) {
70 return Status::NotFound(Status::kPathNotFound,
71 "No options files found in the DB directory.",
72 dbpath);
73 }
74 *options_file_name = latest_file_name;
75 return Status::OK();
76}

78Status LoadLatestOptions(const ConfigOptions& config_options,
79 const std::string& dbpath, DBOptions* db_options,
80 std::vector<ColumnFamilyDescriptor>* cf_descs,
81 std::shared_ptr<Cache>* cache) {
82 std::string options_file_name;
83 Status s =
84 GetLatestOptionsFileName(dbpath, config_options.env, &options_file_name);
85 if (!s.ok()) {
86 return s;
87 }
88 return LoadOptionsFromFile(config_options, dbpath + "/" + options_file_name,
89 db_options, cf_descs, cache);
90}

92Status CheckOptionsCompatibility(
93 const ConfigOptions& config_options, const std::string& dbpath,
94 const DBOptions& db_options,
95 const std::vector<ColumnFamilyDescriptor>& cf_descs) {
96 std::string options_file_name;
97 Status s =
98 GetLatestOptionsFileName(dbpath, config_options.env, &options_file_name);
99 if (!s.ok()) {
100 return s;
101 }
102
103 std::vector<std::string> cf_names;
104 std::vector<ColumnFamilyOptions> cf_opts;
105 for (const auto& cf_desc : cf_descs) {
106 cf_names.push_back(cf_desc.name);
107 cf_opts.push_back(cf_desc.options);
108 }
109
110 const auto& fs = config_options.env->GetFileSystem();
111
112 return RocksDBOptionsParser::VerifyRocksDBOptionsFromFile(
113 config_options, db_options, cf_names, cf_opts,
114 dbpath + "/" + options_file_name, fs.get());
115}
