36CuckooTableReader::CuckooTableReader(
37 const ImmutableOptions& ioptions,
38 std::unique_ptr<RandomAccessFileReader>&& file, uint64_t file_size,
39 const Comparator* comparator,
40 uint64_t (*get_slice_hash)(const Slice&, uint32_t, uint64_t))
41 : file_(std::move(file)),
42 is_last_level_(false),
43 identity_as_first_hash_(false),
44 use_module_hash_(false),
45 num_hash_func_(0),
46
47 key_length_(0),
48 user_key_length_(0),
49 value_length_(0),
50 bucket_length_(0),
51 cuckoo_block_size_(0),
52 cuckoo_block_bytes_minus_one_(0),
53 table_size_(0),
54 ucomp_(comparator),
55 get_slice_hash_(get_slice_hash) {
56 if (!ioptions.allow_mmap_reads) {
57 status_ = Status::InvalidArgument("File is not mmaped");
58 return;
59 }
60 {
61 std::unique_ptr<TableProperties> props;
62 // TODO: plumb Env::IOActivity, Env::IOPriority
63 const ReadOptions read_options;
64 status_ =
65 ReadTableProperties(file_.get(), file_size, kCuckooTableMagicNumber,
66 ioptions, read_options, &props);
67 if (!status_.ok()) {
68 return;
69 }
70 table_props_ = std::move(props);
71 }
72 auto& user_props = table_props_->user_collected_properties;
73 auto hash_funs = user_props.find(CuckooTablePropertyNames::kNumHashFunc);
74 if (hash_funs == user_props.end()) {
75 status_ = Status::Corruption("Number of hash functions not found");
76 return;
77 }
78 num_hash_func_ = *reinterpret_cast<const uint32_t*>(hash_funs->second.data());
79 auto unused_key = user_props.find(CuckooTablePropertyNames::kEmptyKey);
80 if (unused_key == user_props.end()) {
81 status_ = Status::Corruption("Empty bucket value not found");
82 return;
83 }
84 unused_key_ = unused_key->second;
85
86 key_length_ = static_cast<uint32_t>(table_props_->fixed_key_len);
87 auto user_key_len = user_props.find(CuckooTablePropertyNames::kUserKeyLength);
88 if (user_key_len == user_props.end()) {
89 status_ = Status::Corruption("User key length not found");
90 return;
91 }
92 user_key_length_ =
93 *reinterpret_cast<const uint32_t*>(user_key_len->second.data());
94
95 auto value_length = user_props.find(CuckooTablePropertyNames::kValueLength);
96 if (value_length == user_props.end()) {
97 status_ = Status::Corruption("Value length not found");
98 return;
99 }
100 value_length_ =
101 *reinterpret_cast<const uint32_t*>(value_length->second.data());
102 bucket_length_ = key_length_ + value_length_;
103
104 auto hash_table_size =
105 user_props.find(CuckooTablePropertyNames::kHashTableSize);
106 if (hash_table_size == user_props.end()) {
107 status_ = Status::Corruption("Hash table size not found");
108 return;
109 }
110 table_size_ =
111 *reinterpret_cast<const uint64_t*>(hash_table_size->second.data());
112
113 auto is_last_level = user_props.find(CuckooTablePropertyNames::kIsLastLevel);
114 if (is_last_level == user_props.end()) {
115 status_ = Status::Corruption("Is last level not found");
116 return;
117 }
118 is_last_level_ = *reinterpret_cast<const bool*>(is_last_level->second.data());
119
120 auto identity_as_first_hash =
121 user_props.find(CuckooTablePropertyNames::kIdentityAsFirstHash);
122 if (identity_as_first_hash == user_props.end()) {
123 status_ = Status::Corruption("identity as first hash not found");
124 return;
125 }
126 identity_as_first_hash_ =
127 *reinterpret_cast<const bool*>(identity_as_first_hash->second.data());
128
129 auto use_module_hash =
130 user_props.find(CuckooTablePropertyNames::kUseModuleHash);
131 if (use_module_hash == user_props.end()) {
132 status_ = Status::Corruption("hash type is not found");
133 return;
134 }
135 use_module_hash_ =
136 *reinterpret_cast<const bool*>(use_module_hash->second.data());
137 auto cuckoo_block_size =
138 user_props.find(CuckooTablePropertyNames::kCuckooBlockSize);
139 if (cuckoo_block_size == user_props.end()) {
140 status_ = Status::Corruption("Cuckoo block size not found");
141 return;
142 }
143 cuckoo_block_size_ =
144 *reinterpret_cast<const uint32_t*>(cuckoo_block_size->second.data());
145 cuckoo_block_bytes_minus_one_ = cuckoo_block_size_ * bucket_length_ - 1;
146 // TODO: rate limit reads of whole cuckoo tables.
147 status_ = file_->Read(IOOptions(), 0, static_cast<size_t>(file_size),
148 &file_data_, nullptr, nullptr);
149}

151Status CuckooTableReader::Get(const ReadOptions& /*readOptions*/,
152 const Slice& key, GetContext* get_context,
153 const SliceTransform* /* prefix_extractor */,
154 bool /*skip_filters*/) {
155 assert(key.size() == key_length_ + (is_last_level_ ? 8 : 0));
156 Slice user_key = ExtractUserKey(key);
157 for (uint32_t hash_cnt = 0; hash_cnt < num_hash_func_; ++hash_cnt) {
158 uint64_t offset =
159 bucket_length_ * CuckooHash(user_key, hash_cnt, use_module_hash_,
160 table_size_, identity_as_first_hash_,
161 get_slice_hash_);
162 const char* bucket = &file_data_.data()[offset];
163 for (uint32_t block_idx = 0; block_idx < cuckoo_block_size_;
164 ++block_idx, bucket += bucket_length_) {
165 if (ucomp_->Equal(Slice(unused_key_.data(), user_key.size()),
166 Slice(bucket, user_key.size()))) {
167 return Status::OK();
168 }
169 // Here, we compare only the user key part as we support only one entry
170 // per user key and we don't support snapshot.
171 if (ucomp_->Equal(user_key, Slice(bucket, user_key.size()))) {
172 Slice value(bucket + key_length_, value_length_);
173 if (is_last_level_) {
174 // Sequence number is not stored at the last level, so we will use
175 // kMaxSequenceNumber since it is unknown. This could cause some
176 // transactions to fail to lock a key due to known sequence number.
177 // However, it is expected for anyone to use a CuckooTable in a
178 // TransactionDB.
179 get_context->SaveValue(value, kMaxSequenceNumber);
180 } else {
181 Slice full_key(bucket, key_length_);
182 ParsedInternalKey found_ikey;
183 Status s = ParseInternalKey(full_key, &found_ikey,
184 false /* log_err_key */); // TODO
185 if (!s.ok()) {
186 return s;
187 }
188 bool dont_care __attribute__((__unused__));
189 get_context->SaveValue(found_ikey, value, &dont_care, &s);
190 if (!s.ok()) {
191 return s;
192 }
193 }
194 // We don't support merge operations. So, we return here.
195 return Status::OK();
196 }
197 }
198 }
199 return Status::OK();
200}

202void CuckooTableReader::Prepare(const Slice& key) {
203 // Prefetch the first Cuckoo Block.
204 Slice user_key = ExtractUserKey(key);
205 uint64_t addr =
206 reinterpret_cast<uint64_t>(file_data_.data()) +
207 bucket_length_ * CuckooHash(user_key, 0, use_module_hash_, table_size_,
208 identity_as_first_hash_, nullptr);
209 uint64_t end_addr = addr + cuckoo_block_bytes_minus_one_;
210 for (addr &= CACHE_LINE_MASK; addr < end_addr; addr += CACHE_LINE_SIZE) {
211 PREFETCH(reinterpret_cast<const char*>(addr), 0, 3);
212 }
213}

215class CuckooTableIterator : public InternalIterator {
216 public:
217 explicit CuckooTableIterator(CuckooTableReader* reader);
218 // No copying allowed
219 CuckooTableIterator(const CuckooTableIterator&) = delete;
220 void operator=(const Iterator&) = delete;
221 ~CuckooTableIterator() override = default;
222 bool Valid() const override;
223 void SeekToFirst() override;
224 void SeekToLast() override;
225 void Seek(const Slice& target) override;
226 void SeekForPrev(const Slice& target) override;
227 void Next() override;
228 void Prev() override;
229 Slice key() const override;
230 Slice value() const override;
231 Status status() const override { return Status::OK(); }
232 void InitIfNeeded();
233
234 private:
235 struct BucketComparator {
236 BucketComparator(const Slice& file_data, const Comparator* ucomp,
237 uint32_t bucket_len, uint32_t user_key_len,
238 const Slice& target = Slice())
239 : file_data_(file_data),
240 ucomp_(ucomp),
241 bucket_len_(bucket_len),
242 user_key_len_(user_key_len),
243 target_(target) {}
244 bool operator()(const uint32_t first, const uint32_t second) const {
245 const char* first_bucket = (first == kInvalidIndex)
246 ? target_.data()
247 : &file_data_.data()[first * bucket_len_];
248 const char* second_bucket =
249 (second == kInvalidIndex) ? target_.data()
250 : &file_data_.data()[second * bucket_len_];
251 return ucomp_->Compare(Slice(first_bucket, user_key_len_),
252 Slice(second_bucket, user_key_len_)) < 0;
253 }
254
255 private:
256 const Slice file_data_;
257 const Comparator* ucomp_;
258 const uint32_t bucket_len_;
259 const uint32_t user_key_len_;
260 const Slice target_;
261 };
262
263 const BucketComparator bucket_comparator_;
264 void PrepareKVAtCurrIdx();
265 CuckooTableReader* reader_;
266 bool initialized_;
267 // Contains a map of keys to bucket_id sorted in key order.
268 std::vector<uint32_t> sorted_bucket_ids_;
269 // We assume that the number of items can be stored in uint32 (4 Billion).
270 uint32_t curr_key_idx_;
271 Slice curr_value_;
272 IterKey curr_key_;
273};

286void CuckooTableIterator::InitIfNeeded() {
287 if (initialized_) {
288 return;
289 }
290 sorted_bucket_ids_.reserve(
291 static_cast<size_t>(reader_->GetTableProperties()->num_entries));
292 uint64_t num_buckets = reader_->table_size_ + reader_->cuckoo_block_size_ - 1;
293 assert(num_buckets < kInvalidIndex);
294 const char* bucket = reader_->file_data_.data();
295 for (uint32_t bucket_id = 0; bucket_id < num_buckets; ++bucket_id) {
296 if (Slice(bucket, reader_->key_length_) != Slice(reader_->unused_key_)) {
297 sorted_bucket_ids_.push_back(bucket_id);
298 }
299 bucket += reader_->bucket_length_;
300 }
301 assert(sorted_bucket_ids_.size() ==
302 reader_->GetTableProperties()->num_entries);
303 std::sort(sorted_bucket_ids_.begin(), sorted_bucket_ids_.end(),
304 bucket_comparator_);
305 curr_key_idx_ = kInvalidIndex;
306 initialized_ = true;
307}

309void CuckooTableIterator::SeekToFirst() {
310 InitIfNeeded();

312 PrepareKVAtCurrIdx();
313}

315void CuckooTableIterator::SeekToLast() {
316 InitIfNeeded();
317 curr_key_idx_ = static_cast<uint32_t>(sorted_bucket_ids_.size()) - 1;
318 PrepareKVAtCurrIdx();

321void CuckooTableIterator::Seek(const Slice& target) {
322 InitIfNeeded();
323 const BucketComparator seek_comparator(
324 reader_->file_data_, reader_->ucomp_, reader_->bucket_length_,
325 reader_->user_key_length_, ExtractUserKey(target));
326 auto seek_it =
327 std::lower_bound(sorted_bucket_ids_.begin(), sorted_bucket_ids_.end(),
328 kInvalidIndex, seek_comparator);
329 curr_key_idx_ =
330 static_cast<uint32_t>(std::distance(sorted_bucket_ids_.begin(), seek_it));
331 PrepareKVAtCurrIdx();
332}

339bool CuckooTableIterator::Valid() const {
340 return curr_key_idx_ < sorted_bucket_ids_.size();
341}

343void CuckooTableIterator::PrepareKVAtCurrIdx() {
344 if (!Valid()) {
345 curr_value_.clear();
346 curr_key_.Clear();
347 return;
348 }
349 uint32_t id = sorted_bucket_ids_[curr_key_idx_];
350 const char* offset =
351 reader_->file_data_.data() + id * reader_->bucket_length_;
352 if (reader_->is_last_level_) {
353 // Always return internal key.
354 curr_key_.SetInternalKey(Slice(offset, reader_->user_key_length_), 0,
355 kTypeValue);
356 } else {
357 curr_key_.SetInternalKey(Slice(offset, reader_->key_length_));
358 }
359 curr_value_ = Slice(offset + reader_->key_length_, reader_->value_length_);
360}
361
362void CuckooTableIterator::Next() {
363 if (!Valid()) {
364 curr_value_.clear();
365 curr_key_.Clear();
366 return;
367 }
368 ++curr_key_idx_;
369 PrepareKVAtCurrIdx();
370}
371
372void CuckooTableIterator::Prev() {
373 if (curr_key_idx_ == 0) {
374 curr_key_idx_ = static_cast<uint32_t>(sorted_bucket_ids_.size());
375 }
376 if (!Valid()) {
377 curr_value_.clear();
378 curr_key_.Clear();
379 return;
380 }
381 --curr_key_idx_;
382 PrepareKVAtCurrIdx();
383}

391 assert(Valid());

395InternalIterator* CuckooTableReader::NewIterator(
396 const ReadOptions& /*read_options*/,
397 const SliceTransform* /* prefix_extractor */, Arena* arena,
398 bool /*skip_filters*/, TableReaderCaller /*caller*/,
399 size_t /*compaction_readahead_size*/, bool /* allow_unprepared_value */) {
400 if (!status().ok()) {
401 return NewErrorInternalIterator<Slice>(
402 Status::Corruption("CuckooTableReader status is not okay."), arena);
403 }
404 CuckooTableIterator* iter;
405 if (arena == nullptr) {
406 iter = new CuckooTableIterator(this);
407 } else {
408 auto iter_mem = arena->AllocateAligned(sizeof(CuckooTableIterator));
409 iter = new (iter_mem) CuckooTableIterator(this);
410 }
411 return iter;
412}
