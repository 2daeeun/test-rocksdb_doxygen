19// Utility class for stress testing transactions. Can be used to write many
20// transactions in parallel and then validate that the data written is logically
21// consistent. This class assumes the input DB is initially empty.
22//
23// Each call to TransactionDBInsert()/OptimisticTransactionDBInsert() will
24// increment the value of a key in #num_sets sets of keys. Regardless of
25// whether the transaction succeeds, the total sum of values of keys in each
26// set is an invariant that should remain equal.
27//
28// After calling TransactionDBInsert()/OptimisticTransactionDBInsert() many
29// times, Verify() can be called to validate that the invariant holds.
30//
31// To test writing Transaction in parallel, multiple threads can create a
32// RandomTransactionInserter with similar arguments using the same DB.
33class RandomTransactionInserter {
34 public:
35 static bool RollbackDeletionTypeCallback(const Slice& key) {
36 // These are hard-coded atm. See how RandomTransactionInserter::DoInsert()
37 // determines whether to use SingleDelete or Delete for a key.
38 assert(key.size() >= 4);
39 const char* ptr = key.data();
40 assert(ptr);
41 while (ptr && ptr < key.data() + 4 && *ptr == '0') {
42 ++ptr;
43 }
44 std::string prefix(ptr, 4 - (ptr - key.data()));
45 unsigned long set_i = std::stoul(prefix);
46 assert(set_i > 0);
47 assert(set_i <= 9999);
48 --set_i;
49 return ((set_i % 4) != 0);
50 }
51
52 // num_keys is the number of keys in each set.
53 // num_sets is the number of sets of keys.
54 // cmt_delay_ms is the delay between prepare (if there is any) and commit
55 // first_id is the id of the first transaction
56 explicit RandomTransactionInserter(
57 Random64* rand, const WriteOptions& write_options = WriteOptions(),
58 const ReadOptions& read_options = ReadOptions(), uint64_t num_keys = 1000,
59 uint16_t num_sets = 3, const uint64_t cmt_delay_ms = 0,
60 const uint64_t first_id = 0);

62 ~RandomTransactionInserter();
