78Status BlobFile::WriteFooterAndCloseLocked(const WriteOptions& write_options,
79 SequenceNumber sequence) {
80 BlobLogFooter footer;
81 footer.blob_count = blob_count_;
82 if (HasTTL()) {
83 footer.expiration_range = expiration_range_;
84 }
85
86 // this will close the file and reset the Writable File Pointer.
87 Status s = log_writer_->AppendFooter(write_options, footer,
88 /* checksum_method */ nullptr,
89 /* checksum_value */ nullptr);
90 if (s.ok()) {
91 closed_ = true;
92 immutable_sequence_ = sequence;
93 file_size_ += BlobLogFooter::kSize;
94 }
95 // delete the sequential writer
96 log_writer_.reset();
97 return s;
98}

100Status BlobFile::ReadFooter(BlobLogFooter* bf) {
101 if (file_size_ < (BlobLogHeader::kSize + BlobLogFooter::kSize)) {
102 return Status::IOError("File does not have footer", PathName());
103 }
104
105 uint64_t footer_offset = file_size_ - BlobLogFooter::kSize;
106 // assume that ra_file_reader_ is valid before we enter this
107 assert(ra_file_reader_);
108
109 Slice result;
110 std::string buf;
111 AlignedBuf aligned_buf;
112 Status s;
113 // TODO: rate limit reading footers from blob files.
114 if (ra_file_reader_->use_direct_io()) {
115 s = ra_file_reader_->Read(IOOptions(), footer_offset, BlobLogFooter::kSize,
116 &result, nullptr, &aligned_buf);
117 } else {
118 buf.reserve(BlobLogFooter::kSize + 10);
119 s = ra_file_reader_->Read(IOOptions(), footer_offset, BlobLogFooter::kSize,
120 &result, buf.data(), nullptr);
121 }
122 if (!s.ok()) {
123 return s;
124 }
125 if (result.size() != BlobLogFooter::kSize) {
126 // should not happen
127 return Status::IOError("EOF reached before footer");
128 }
129
130 s = bf->DecodeFrom(result);
131 return s;
132}

154Status BlobFile::GetReader(Env* env, const FileOptions& file_options,
155 std::shared_ptr<RandomAccessFileReader>* reader,
156 bool* fresh_open) {
157 assert(reader != nullptr);
158 assert(fresh_open != nullptr);
159 *fresh_open = false;
160 int64_t current_time = 0;
161 if (env->GetCurrentTime(&current_time).ok()) {
162 last_access_.store(current_time);
163 }
164 Status s;
165
166 {
167 ReadLock lockbfile_r(&mutex_);
168 if (ra_file_reader_) {
169 *reader = ra_file_reader_;
170 return s;
171 }
172 }
173
174 WriteLock lockbfile_w(&mutex_);
175 // Double check.
176 if (ra_file_reader_) {
177 *reader = ra_file_reader_;
178 return s;
179 }
180
181 std::unique_ptr<FSRandomAccessFile> rfile;
182 s = env->GetFileSystem()->NewRandomAccessFile(PathName(), file_options,
183 &rfile, nullptr);
184 if (!s.ok()) {
185 ROCKS_LOG_ERROR(info_log_,
186 "Failed to open blob file for random-read: %s status: '%s'"
187 " exists: '%s'",
188 PathName().c_str(), s.ToString().c_str(),
189 env->FileExists(PathName()).ToString().c_str());
190 return s;
191 }
192
193 ra_file_reader_ =
194 std::make_shared<RandomAccessFileReader>(std::move(rfile), PathName());
195 *reader = ra_file_reader_;
196 *fresh_open = true;
197 return s;
198}

200Status BlobFile::ReadMetadata(const std::shared_ptr<FileSystem>& fs,
201 const FileOptions& file_options) {
202 assert(Immutable());
203 // Get file size.
204 uint64_t file_size = 0;
205 Status s =
206 fs->GetFileSize(PathName(), file_options.io_options, &file_size, nullptr);
207 if (s.ok()) {
208 file_size_ = file_size;
209 } else {
210 ROCKS_LOG_ERROR(info_log_,
211 "Failed to get size of blob file %" PRIu64 ", status: %s",
212 file_number_, s.ToString().c_str());
213 return s;
214 }
215 if (file_size < BlobLogHeader::kSize) {
216 ROCKS_LOG_ERROR(
217 info_log_, "Incomplete blob file blob file %" PRIu64 ", size: %" PRIu64,
218 file_number_, file_size);
219 return Status::Corruption("Incomplete blob file header.");
220 }
221
222 // Create file reader.
223 std::unique_ptr<RandomAccessFileReader> file_reader;
224 s = RandomAccessFileReader::Create(fs, PathName(), file_options, &file_reader,
225 nullptr);
226 if (!s.ok()) {
227 ROCKS_LOG_ERROR(info_log_,
228 "Failed to open blob file %" PRIu64 ", status: %s",
229 file_number_, s.ToString().c_str());
230 return s;
231 }
232
233 // Read file header.
234 std::string header_buf;
235 AlignedBuf aligned_buf;
236 Slice header_slice;
237 // TODO: rate limit reading headers from blob files.
238 if (file_reader->use_direct_io()) {
239 s = file_reader->Read(IOOptions(), 0, BlobLogHeader::kSize, &header_slice,
240 nullptr, &aligned_buf);
241 } else {
242 header_buf.reserve(BlobLogHeader::kSize);
243 s = file_reader->Read(IOOptions(), 0, BlobLogHeader::kSize, &header_slice,
244 header_buf.data(), nullptr);
245 }
246 if (!s.ok()) {
247 ROCKS_LOG_ERROR(
248 info_log_, "Failed to read header of blob file %" PRIu64 ", status: %s",
249 file_number_, s.ToString().c_str());
250 return s;
251 }
252 BlobLogHeader header;
253 s = header.DecodeFrom(header_slice);
254 if (!s.ok()) {
255 ROCKS_LOG_ERROR(info_log_,
256 "Failed to decode header of blob file %" PRIu64
257 ", status: %s",
258 file_number_, s.ToString().c_str());
259 return s;
260 }
261 column_family_id_ = header.column_family_id;
262 compression_ = header.compression;
263 has_ttl_ = header.has_ttl;
264 if (has_ttl_) {
265 expiration_range_ = header.expiration_range;
266 }
267 header_valid_ = true;
268
269 // Read file footer.
270 if (file_size_ < BlobLogHeader::kSize + BlobLogFooter::kSize) {
271 // OK not to have footer.
272 assert(!footer_valid_);
273 return Status::OK();
274 }
275 std::string footer_buf;
276 Slice footer_slice;
277 // TODO: rate limit reading footers from blob files.
278 if (file_reader->use_direct_io()) {
279 s = file_reader->Read(IOOptions(), file_size - BlobLogFooter::kSize,
280 BlobLogFooter::kSize, &footer_slice, nullptr,
281 &aligned_buf);
282 } else {
283 footer_buf.reserve(BlobLogFooter::kSize);
284 s = file_reader->Read(IOOptions(), file_size - BlobLogFooter::kSize,
285 BlobLogFooter::kSize, &footer_slice,
286 footer_buf.data(), nullptr);
287 }
288 if (!s.ok()) {
289 ROCKS_LOG_ERROR(
290 info_log_, "Failed to read footer of blob file %" PRIu64 ", status: %s",
291 file_number_, s.ToString().c_str());
292 return s;
293 }
294 BlobLogFooter footer;
295 s = footer.DecodeFrom(footer_slice);
296 if (!s.ok()) {
297 // OK not to have footer.
298 assert(!footer_valid_);
299 return Status::OK();
300 }
301 blob_count_ = footer.blob_count;
302 if (has_ttl_) {
303 assert(header.expiration_range.first <= footer.expiration_range.first);
304 assert(header.expiration_range.second >= footer.expiration_range.second);
305 expiration_range_ = footer.expiration_range;
306 }
307 footer_valid_ = true;
308 return Status::OK();
309}
