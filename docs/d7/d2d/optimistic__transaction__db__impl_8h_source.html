44class OptimisticTransactionDBImpl : public OptimisticTransactionDB {
45 public:
46 explicit OptimisticTransactionDBImpl(
47 DB* db, const OptimisticTransactionDBOptions& occ_options,
48 bool take_ownership = true)
49 : OptimisticTransactionDB(db),
50 db_owner_(take_ownership),
51 validate_policy_(occ_options.validate_policy) {
52 if (validate_policy_ == OccValidationPolicy::kValidateParallel) {
53 auto bucketed_locks = occ_options.shared_lock_buckets;
54 if (!bucketed_locks) {
55 uint32_t bucket_count = std::max(16u, occ_options.occ_lock_buckets);
56 bucketed_locks = MakeSharedOccLockBuckets(bucket_count);
57 }
58 bucketed_locks_ = static_cast_with_check<OccLockBucketsImplBase>(
59 std::move(bucketed_locks));
60 }
61 }
62
63 ~OptimisticTransactionDBImpl() {
64 // Prevent this stackable from destroying
65 // base db
66 if (!db_owner_) {
67 db_ = nullptr;
68 }
69 }
70
71 Transaction* BeginTransaction(const WriteOptions& write_options,
72 const OptimisticTransactionOptions& txn_options,
73 Transaction* old_txn) override;
74
75 // Transactional `DeleteRange()` is not yet supported.
76 using StackableDB::DeleteRange;
77 Status DeleteRange(const WriteOptions&, ColumnFamilyHandle*, const Slice&,
78 const Slice&) override {
79 return Status::NotSupported();
80 }
81
82 // Range deletions also must not be snuck into `WriteBatch`es as they are
83 // incompatible with `OptimisticTransactionDB`.
84 Status Write(const WriteOptions& write_opts, WriteBatch* batch) override {
85 if (batch->HasDeleteRange()) {
86 return Status::NotSupported();
87 }
88 return OptimisticTransactionDB::Write(write_opts, batch);
89 }
90
91 OccValidationPolicy GetValidatePolicy() const { return validate_policy_; }
92
93 port::Mutex& GetLockBucket(const Slice& key, uint64_t seed) {
94 return bucketed_locks_->GetLockBucket(key, seed);
95 }
96
97 private:
98 std::shared_ptr<OccLockBucketsImplBase> bucketed_locks_;
99
100 bool db_owner_;
101
102 const OccValidationPolicy validate_policy_;
103
104 void ReinitializeTransaction(Transaction* txn,
105 const WriteOptions& write_options,
106 const OptimisticTransactionOptions& txn_options =
107 OptimisticTransactionOptions());
108};
