99Status DBIter::GetProperty(std::string prop_name, std::string* prop) {
100 if (prop == nullptr) {
101 return Status::InvalidArgument("prop is nullptr");
102 }
103 if (prop_name == "rocksdb.iterator.super-version-number") {
104 // First try to pass the value returned from inner iterator.
105 return iter_.iter()->GetProperty(prop_name, prop);
106 } else if (prop_name == "rocksdb.iterator.is-key-pinned") {
107 if (valid_) {
108 *prop = (pin_thru_lifetime_ && saved_key_.IsKeyPinned()) ? "1" : "0";
109 } else {
110 *prop = "Iterator is not valid.";
111 }
112 return Status::OK();
113 } else if (prop_name == "rocksdb.iterator.is-value-pinned") {
114 if (valid_) {
115 *prop = (pin_thru_lifetime_ && iter_.Valid() &&
116 iter_.value().data() == value_.data())
117 ? "1"
118 : "0";
119 } else {
120 *prop = "Iterator is not valid.";
121 }
122 return Status::OK();
123 } else if (prop_name == "rocksdb.iterator.internal-key") {
124 *prop = saved_key_.GetUserKey().ToString();
125 return Status::OK();
126 } else if (prop_name == "rocksdb.iterator.write-time") {
127 PutFixed64(prop, saved_write_unix_time_);
128 return Status::OK();
129 }
130 return Status::InvalidArgument("Unidentified property.");
131}

145void DBIter::Next() {
146 assert(valid_);
147 assert(status_.ok());
148
149 PERF_COUNTER_ADD(iter_next_count, 1);
150 PERF_CPU_TIMER_GUARD(iter_next_cpu_nanos, clock_);
151 // Release temporarily pinned blocks from last operation
152 ReleaseTempPinnedData();
153 ResetBlobData();
154 ResetValueAndColumns();
155 local_stats_.skip_count_ += num_internal_keys_skipped_;
156 local_stats_.skip_count_--;
157 num_internal_keys_skipped_ = 0;
158 bool ok = true;
159 if (direction_ == kReverse) {
160 is_key_seqnum_zero_ = false;
161 if (!ReverseToForward()) {
162 ok = false;
163 }
164 } else if (!current_entry_is_merged_) {
165 // If the current value is not a merge, the iter position is the
166 // current key, which is already returned. We can safely issue a
167 // Next() without checking the current key.
168 // If the current key is a merge, very likely iter already points
169 // to the next internal position.
170 assert(iter_.Valid());
171 iter_.Next();
172 PERF_COUNTER_ADD(internal_key_skipped_count, 1);
173 }
174
175 local_stats_.next_count_++;
176 if (ok && iter_.Valid()) {
177 ClearSavedValue();
178
179 if (prefix_same_as_start_) {
180 assert(prefix_extractor_ != nullptr);
181 const Slice prefix = prefix_.GetUserKey();
182 FindNextUserEntry(true /* skipping the current user key */, &prefix);
183 } else {
184 FindNextUserEntry(true /* skipping the current user key */, nullptr);
185 }
186 } else {
187 is_key_seqnum_zero_ = false;
188 valid_ = false;
189 }
190 if (statistics_ != nullptr && valid_) {
191 local_stats_.next_found_count_++;
192 local_stats_.bytes_read_ += (key().size() + value().size());
193 }
194}

326// PRE: saved_key_ has the current user key if skipping_saved_key
327// POST: saved_key_ should have the next user key if valid_,
328// if the current entry is a result of merge
329// current_entry_is_merged_ => true
330// saved_value_ => the merged value
331//
332// NOTE: In between, saved_key_ can point to a user key that has
333// a delete marker or a sequence number higher than sequence_
334// saved_key_ MUST have a proper user_key before calling this function
335//
336// The prefix parameter, if not null, indicates that we need to iterate
337// within the prefix, and the iterator needs to be made invalid, if no
338// more entry for the prefix can be found.
339bool DBIter::FindNextUserEntry(bool skipping_saved_key, const Slice* prefix) {
340 PERF_TIMER_GUARD(find_next_user_entry_time);
341 return FindNextUserEntryInternal(skipping_saved_key, prefix);
342}

344// Actual implementation of DBIter::FindNextUserEntry()
345bool DBIter::FindNextUserEntryInternal(bool skipping_saved_key,
346 const Slice* prefix) {
347 // Loop until we hit an acceptable entry to yield
348 assert(iter_.Valid());
349 assert(status_.ok());
350 assert(direction_ == kForward);
351 current_entry_is_merged_ = false;
352
353 // How many times in a row we have skipped an entry with user key less than
354 // or equal to saved_key_. We could skip these entries either because
355 // sequence numbers were too high or because skipping_saved_key = true.
356 // What saved_key_ contains throughout this method:
357 // - if skipping_saved_key : saved_key_ contains the key that we need
358 // to skip, and we haven't seen any keys greater
359 // than that,
360 // - if num_skipped > 0 : saved_key_ contains the key that we have skipped
361 // num_skipped times, and we haven't seen any keys
362 // greater than that,
363 // - none of the above : saved_key_ can contain anything, it doesn't
364 // matter.
365 uint64_t num_skipped = 0;
366 // For write unprepared, the target sequence number in reseek could be larger
367 // than the snapshot, and thus needs to be skipped again. This could result in
368 // an infinite loop of reseeks. To avoid that, we limit the number of reseeks
369 // to one.
370 bool reseek_done = false;
371
372 do {
373 // Will update is_key_seqnum_zero_ as soon as we parsed the current key
374 // but we need to save the previous value to be used in the loop.
375 bool is_prev_key_seqnum_zero = is_key_seqnum_zero_;
376 if (!ParseKey(&ikey_)) {
377 is_key_seqnum_zero_ = false;
378 return false;
379 }
380 Slice user_key_without_ts =
381 StripTimestampFromUserKey(ikey_.user_key, timestamp_size_);
382
383 is_key_seqnum_zero_ = (ikey_.sequence == 0);
384
385 assert(iterate_upper_bound_ == nullptr ||
386 iter_.UpperBoundCheckResult() != IterBoundCheck::kInbound ||
387 user_comparator_.CompareWithoutTimestamp(
388 user_key_without_ts, /*a_has_ts=*/false, *iterate_upper_bound_,
389 /*b_has_ts=*/false) < 0);
390 if (iterate_upper_bound_ != nullptr &&
391 iter_.UpperBoundCheckResult() != IterBoundCheck::kInbound &&
392 user_comparator_.CompareWithoutTimestamp(
393 user_key_without_ts, /*a_has_ts=*/false, *iterate_upper_bound_,
394 /*b_has_ts=*/false) >= 0) {
395 break;
396 }
397
398 assert(prefix == nullptr || prefix_extractor_ != nullptr);
399 if (prefix != nullptr &&
400 prefix_extractor_->Transform(user_key_without_ts).compare(*prefix) !=
401 0) {
402 assert(prefix_same_as_start_);
403 break;
404 }
405
406 if (TooManyInternalKeysSkipped()) {
407 return false;
408 }
409
410 assert(ikey_.user_key.size() >= timestamp_size_);
411 Slice ts = timestamp_size_ > 0 ? ExtractTimestampFromUserKey(
412 ikey_.user_key, timestamp_size_)
413 : Slice();
414 bool more_recent = false;
415 if (IsVisible(ikey_.sequence, ts, &more_recent)) {
416 // If the previous entry is of seqnum 0, the current entry will not
417 // possibly be skipped. This condition can potentially be relaxed to
418 // prev_key.seq <= ikey_.sequence. We are cautious because it will be more
419 // prone to bugs causing the same user key with the same sequence number.
420 // Note that with current timestamp implementation, the same user key can
421 // have different timestamps and zero sequence number on the bottommost
422 // level. This may change in the future.
423 if ((!is_prev_key_seqnum_zero || timestamp_size_ > 0) &&
424 skipping_saved_key &&
425 CompareKeyForSkip(ikey_.user_key, saved_key_.GetUserKey()) <= 0) {
426 num_skipped++; // skip this entry
427 PERF_COUNTER_ADD(internal_key_skipped_count, 1);
428 } else {
429 assert(!skipping_saved_key ||
430 CompareKeyForSkip(ikey_.user_key, saved_key_.GetUserKey()) > 0);
431 num_skipped = 0;
432 reseek_done = false;
433 switch (ikey_.type) {
434 case kTypeDeletion:
435 case kTypeDeletionWithTimestamp:
436 case kTypeSingleDeletion:
437 // Arrange to skip all upcoming entries for this key since
438 // they are hidden by this deletion.
439 if (timestamp_lb_) {
440 saved_key_.SetInternalKey(ikey_);
441 valid_ = true;
442 return true;
443 } else {
444 saved_key_.SetUserKey(
445 ikey_.user_key, !pin_thru_lifetime_ ||
446 !iter_.iter()->IsKeyPinned() /* copy */);
447 skipping_saved_key = true;
448 PERF_COUNTER_ADD(internal_delete_skipped_count, 1);
449 }
450 break;
451 case kTypeValue:
452 case kTypeValuePreferredSeqno:
453 case kTypeBlobIndex:
454 case kTypeWideColumnEntity:
455 if (!PrepareValueInternal()) {
456 return false;
457 }
458 if (timestamp_lb_) {
459 saved_key_.SetInternalKey(ikey_);
460 } else {
461 saved_key_.SetUserKey(
462 ikey_.user_key, !pin_thru_lifetime_ ||
463 !iter_.iter()->IsKeyPinned() /* copy */);
464 }
465
466 if (ikey_.type == kTypeBlobIndex) {
467 if (!SetValueAndColumnsFromBlob(ikey_.user_key, iter_.value())) {
468 return false;
469 }
470 } else if (ikey_.type == kTypeWideColumnEntity) {
471 if (!SetValueAndColumnsFromEntity(iter_.value())) {
472 return false;
473 }
474 } else {
475 assert(ikey_.type == kTypeValue ||
476 ikey_.type == kTypeValuePreferredSeqno);
477 Slice value = iter_.value();
478 saved_write_unix_time_ = iter_.write_unix_time();
479 if (ikey_.type == kTypeValuePreferredSeqno) {
480 value = ParsePackedValueForValue(value);
481 }
482 SetValueAndColumnsFromPlain(value);
483 }
484
485 valid_ = true;
486 return true;
487 break;
488 case kTypeMerge:
489 if (!PrepareValueInternal()) {
490 return false;
491 }
492 saved_key_.SetUserKey(
493 ikey_.user_key,
494 !pin_thru_lifetime_ || !iter_.iter()->IsKeyPinned() /* copy */);
495 // By now, we are sure the current ikey is going to yield a value
496 current_entry_is_merged_ = true;
497 valid_ = true;
498 return MergeValuesNewToOld(); // Go to a different state machine
499 break;
500 default:
501 valid_ = false;
502 status_ = Status::Corruption(
503 "Unknown value type: " +
504 std::to_string(static_cast<unsigned int>(ikey_.type)));
505 return false;
506 }
507 }
508 } else {
509 if (more_recent) {
510 PERF_COUNTER_ADD(internal_recent_skipped_count, 1);
511 }
512
513 // This key was inserted after our snapshot was taken or skipped by
514 // timestamp range. If this happens too many times in a row for the same
515 // user key, we want to seek to the target sequence number.
516 int cmp = user_comparator_.CompareWithoutTimestamp(
517 ikey_.user_key, saved_key_.GetUserKey());
518 if (cmp == 0 || (skipping_saved_key && cmp < 0)) {
519 num_skipped++;
520 } else {
521 saved_key_.SetUserKey(
522 ikey_.user_key,
523 !iter_.iter()->IsKeyPinned() || !pin_thru_lifetime_ /* copy */);
524 skipping_saved_key = false;
525 num_skipped = 0;
526 reseek_done = false;
527 }
528 }
529
530 // If we have sequentially iterated via numerous equal keys, then it's
531 // better to seek so that we can avoid too many key comparisons.
532 //
533 // To avoid infinite loops, do not reseek if we have already attempted to
534 // reseek previously.
535 //
536 // TODO(lth): If we reseek to sequence number greater than ikey_.sequence,
537 // then it does not make sense to reseek as we would actually land further
538 // away from the desired key. There is opportunity for optimization here.
539 if (num_skipped > max_skip_ && !reseek_done) {
540 is_key_seqnum_zero_ = false;
541 num_skipped = 0;
542 reseek_done = true;
543 std::string last_key;
544 if (skipping_saved_key) {
545 // We're looking for the next user-key but all we see are the same
546 // user-key with decreasing sequence numbers. Fast forward to
547 // sequence number 0 and type deletion (the smallest type).
548 if (timestamp_size_ == 0) {
549 AppendInternalKey(
550 &last_key,
551 ParsedInternalKey(saved_key_.GetUserKey(), 0, kTypeDeletion));
552 } else {
553 const std::string kTsMin(timestamp_size_, '\0');
554 AppendInternalKeyWithDifferentTimestamp(
555 &last_key,
556 ParsedInternalKey(saved_key_.GetUserKey(), 0, kTypeDeletion),
557 kTsMin);
558 }
559 // Don't set skipping_saved_key = false because we may still see more
560 // user-keys equal to saved_key_.
561 } else {
562 // We saw multiple entries with this user key and sequence numbers
563 // higher than sequence_. Fast forward to sequence_.
564 // Note that this only covers a case when a higher key was overwritten
565 // many times since our snapshot was taken, not the case when a lot of
566 // different keys were inserted after our snapshot was taken.
567 if (timestamp_size_ == 0) {
568 AppendInternalKey(
569 &last_key, ParsedInternalKey(saved_key_.GetUserKey(), sequence_,
570 kValueTypeForSeek));
571 } else {
572 AppendInternalKeyWithDifferentTimestamp(
573 &last_key,
574 ParsedInternalKey(saved_key_.GetUserKey(), sequence_,
575 kValueTypeForSeek),
576 *timestamp_ub_);
577 }
578 }
579 iter_.Seek(last_key);
580 RecordTick(statistics_, NUMBER_OF_RESEEKS_IN_ITERATION);
581 } else {
582 iter_.Next();
583 }
584
585 // This could be a long-running operation due to tombstones, etc.
586 bool aborted = ROCKSDB_THREAD_YIELD_CHECK_ABORT();
587 if (aborted) {
588 valid_ = false;
589 status_ = Status::Aborted("Query abort.");
590 return false;
591 }
592 } while (iter_.Valid());
593
594 valid_ = false;
595 return iter_.status().ok();
596}
597
598// Merge values of the same user key starting from the current iter_ position
599// Scan from the newer entries to older entries.
600// PRE: iter_.key() points to the first merge type entry
601// saved_key_ stores the user key
602// iter_.PrepareValue() has been called
603// POST: saved_value_ has the merged value for the user key
604// iter_ points to the next entry (or invalid)
605bool DBIter::MergeValuesNewToOld() {
606 if (!merge_operator_) {
607 ROCKS_LOG_ERROR(logger_, "Options::merge_operator is null.");
608 status_ = Status::InvalidArgument("merge_operator_ must be set.");
609 valid_ = false;
610 return false;
611 }
612
613 // Temporarily pin the blocks that hold merge operands
614 TempPinData();
615 merge_context_.Clear();
616 // Start the merge process by pushing the first operand
617 merge_context_.PushOperand(
618 iter_.value(), iter_.iter()->IsValuePinned() /* operand_pinned */);
619 PERF_COUNTER_ADD(internal_merge_count, 1);
620
621 TEST_SYNC_POINT("DBIter::MergeValuesNewToOld:PushedFirstOperand");
622
623 ParsedInternalKey ikey;
624 for (iter_.Next(); iter_.Valid(); iter_.Next()) {
625 TEST_SYNC_POINT("DBIter::MergeValuesNewToOld:SteppedToNextOperand");
626 if (!ParseKey(&ikey)) {
627 return false;
628 }
629
630 if (!user_comparator_.EqualWithoutTimestamp(ikey.user_key,
631 saved_key_.GetUserKey())) {
632 // hit the next user key, stop right here
633 break;
634 }
635 if (kTypeDeletion == ikey.type || kTypeSingleDeletion == ikey.type ||
636 kTypeDeletionWithTimestamp == ikey.type) {
637 // hit a delete with the same user key, stop right here
638 // iter_ is positioned after delete
639 iter_.Next();
640 break;
641 }
642 if (!PrepareValueInternal()) {
643 return false;
644 }
645
646 if (kTypeValue == ikey.type || kTypeValuePreferredSeqno == ikey.type) {
647 Slice value = iter_.value();
648 saved_write_unix_time_ = iter_.write_unix_time();
649 if (kTypeValuePreferredSeqno == ikey.type) {
650 value = ParsePackedValueForValue(value);
651 }
652 // hit a put or put equivalent, merge the put value with operands and
653 // store the final result in saved_value_. We are done!
654 if (!MergeWithPlainBaseValue(value, ikey.user_key)) {
655 return false;
656 }
657 // iter_ is positioned after put
658 iter_.Next();
659 if (!iter_.status().ok()) {
660 valid_ = false;
661 return false;
662 }
663 return true;
664 } else if (kTypeMerge == ikey.type) {
665 // hit a merge, add the value as an operand and run associative merge.
666 // when complete, add result to operands and continue.
667 merge_context_.PushOperand(
668 iter_.value(), iter_.iter()->IsValuePinned() /* operand_pinned */);
669 PERF_COUNTER_ADD(internal_merge_count, 1);
670 } else if (kTypeBlobIndex == ikey.type) {
671 if (!MergeWithBlobBaseValue(iter_.value(), ikey.user_key)) {
672 return false;
673 }
674
675 // iter_ is positioned after put
676 iter_.Next();
677 if (!iter_.status().ok()) {
678 valid_ = false;
679 return false;
680 }
681
682 return true;
683 } else if (kTypeWideColumnEntity == ikey.type) {
684 if (!MergeWithWideColumnBaseValue(iter_.value(), ikey.user_key)) {
685 return false;
686 }
687
688 // iter_ is positioned after put
689 iter_.Next();
690 if (!iter_.status().ok()) {
691 valid_ = false;
692 return false;
693 }
694
695 return true;
696 } else {
697 valid_ = false;
698 status_ = Status::Corruption(
699 "Unrecognized value type: " +
700 std::to_string(static_cast<unsigned int>(ikey.type)));
701 return false;
702 }
703 }
704
705 if (!iter_.status().ok()) {
706 valid_ = false;
707 return false;
708 }
709
710 // we either exhausted all internal keys under this user key, or hit
711 // a deletion marker.
712 // feed null as the existing value to the merge operator, such that
713 // client can differentiate this scenario and do things accordingly.
714 if (!MergeWithNoBaseValue(saved_key_.GetUserKey())) {
715 return false;
716 }
717 assert(status_.ok());
718 return true;
719}

757bool DBIter::ReverseToForward() {
758 assert(iter_.status().ok());
759
760 // When moving backwards, iter_ is positioned on _previous_ key, which may
761 // not exist or may have different prefix than the current key().
762 // If that's the case, seek iter_ to current key.
763 if (!expect_total_order_inner_iter() || !iter_.Valid()) {
764 std::string last_key;
765 if (timestamp_size_ == 0) {
766 AppendInternalKey(
767 &last_key, ParsedInternalKey(saved_key_.GetUserKey(),
768 kMaxSequenceNumber, kValueTypeForSeek));
769 } else {
770 // TODO: pre-create kTsMax.
771 const std::string kTsMax(timestamp_size_, '\xff');
772 AppendInternalKeyWithDifferentTimestamp(
773 &last_key,
774 ParsedInternalKey(saved_key_.GetUserKey(), kMaxSequenceNumber,
775 kValueTypeForSeek),
776 kTsMax);
777 }
778 iter_.Seek(last_key);
779 RecordTick(statistics_, NUMBER_OF_RESEEKS_IN_ITERATION);
780 }
781
782 direction_ = kForward;
783 // Skip keys less than the current key() (a.k.a. saved_key_).
784 while (iter_.Valid()) {
785 ParsedInternalKey ikey;
786 if (!ParseKey(&ikey)) {
787 return false;
788 }
789 if (user_comparator_.Compare(ikey.user_key, saved_key_.GetUserKey()) >= 0) {
790 return true;
791 }
792 iter_.Next();
793 }
794
795 if (!iter_.status().ok()) {
796 valid_ = false;
797 return false;
798 }
799
800 return true;
801}

803// Move iter_ to the key before saved_key_.
804bool DBIter::ReverseToBackward() {
805 assert(iter_.status().ok());
806
807 // When current_entry_is_merged_ is true, iter_ may be positioned on the next
808 // key, which may not exist or may have prefix different from current.
809 // If that's the case, seek to saved_key_.
810 if (current_entry_is_merged_ &&
811 (!expect_total_order_inner_iter() || !iter_.Valid())) {
812 IterKey last_key;
813 // Using kMaxSequenceNumber and kValueTypeForSeek
814 // (not kValueTypeForSeekForPrev) to seek to a key strictly smaller
815 // than saved_key_.
816 last_key.SetInternalKey(ParsedInternalKey(
817 saved_key_.GetUserKey(), kMaxSequenceNumber, kValueTypeForSeek));
818 if (!expect_total_order_inner_iter()) {
819 iter_.SeekForPrev(last_key.GetInternalKey());
820 } else {
821 // Some iterators may not support SeekForPrev(), so we avoid using it
822 // when prefix seek mode is disabled. This is somewhat expensive
823 // (an extra Prev(), as well as an extra change of direction of iter_),
824 // so we may need to reconsider it later.
825 iter_.Seek(last_key.GetInternalKey());
826 if (!iter_.Valid() && iter_.status().ok()) {
827 iter_.SeekToLast();
828 }
829 }
830 RecordTick(statistics_, NUMBER_OF_RESEEKS_IN_ITERATION);
831 }
832
833 direction_ = kReverse;
834 return FindUserKeyBeforeSavedKey();
835}

837void DBIter::PrevInternal(const Slice* prefix) {
838 while (iter_.Valid()) {
839 saved_key_.SetUserKey(
840 ExtractUserKey(iter_.key()),
841 !iter_.iter()->IsKeyPinned() || !pin_thru_lifetime_ /* copy */);
842
843 assert(prefix == nullptr || prefix_extractor_ != nullptr);
844 if (prefix != nullptr &&
845 prefix_extractor_
846 ->Transform(StripTimestampFromUserKey(saved_key_.GetUserKey(),
847 timestamp_size_))
848 .compare(*prefix) != 0) {
849 assert(prefix_same_as_start_);
850 // Current key does not have the same prefix as start
851 valid_ = false;
852 return;
853 }
854
855 assert(iterate_lower_bound_ == nullptr || iter_.MayBeOutOfLowerBound() ||
856 user_comparator_.CompareWithoutTimestamp(
857 saved_key_.GetUserKey(), /*a_has_ts=*/true,
858 *iterate_lower_bound_, /*b_has_ts=*/false) >= 0);
859 if (iterate_lower_bound_ != nullptr && iter_.MayBeOutOfLowerBound() &&
860 user_comparator_.CompareWithoutTimestamp(
861 saved_key_.GetUserKey(), /*a_has_ts=*/true, *iterate_lower_bound_,
862 /*b_has_ts=*/false) < 0) {
863 // We've iterated earlier than the user-specified lower bound.
864 valid_ = false;
865 return;
866 }
867
868 if (!FindValueForCurrentKey()) { // assigns valid_
869 return;
870 }
871
872 // Whether or not we found a value for current key, we need iter_ to end up
873 // on a smaller key.
874 if (!FindUserKeyBeforeSavedKey()) {
875 return;
876 }
877
878 if (valid_) {
879 // Found the value.
880 return;
881 }
882
883 if (TooManyInternalKeysSkipped(false)) {
884 return;
885 }
886 }
887
888 // We haven't found any key - iterator is not valid
889 valid_ = false;
890}

892// Used for backwards iteration.
893// Looks at the entries with user key saved_key_ and finds the most up-to-date
894// value for it, or executes a merge, or determines that the value was deleted.
895// Sets valid_ to true if the value is found and is ready to be presented to
896// the user through value().
897// Sets valid_ to false if the value was deleted, and we should try another key.
898// Returns false if an error occurred, and !status().ok() and !valid_.
899//
900// PRE: iter_ is positioned on the last entry with user key equal to saved_key_.
901// POST: iter_ is positioned on one of the entries equal to saved_key_, or on
902// the entry just before them, or on the entry just after them.
903bool DBIter::FindValueForCurrentKey() {
904 assert(iter_.Valid());
905 merge_context_.Clear();
906 current_entry_is_merged_ = false;
907 // last entry before merge (could be kTypeDeletion,
908 // kTypeDeletionWithTimestamp, kTypeSingleDeletion, kTypeValue
909 // kTypeBlobIndex, kTypeWideColumnEntity or kTypeValuePreferredSeqno)
910 ValueType last_not_merge_type = kTypeDeletion;
911 ValueType last_key_entry_type = kTypeDeletion;
912
913 // If false, it indicates that we have not seen any valid entry, even though
914 // last_key_entry_type is initialized to kTypeDeletion.
915 bool valid_entry_seen = false;
916
917 // Temporarily pin blocks that hold (merge operands / the value)
918 ReleaseTempPinnedData();
919 TempPinData();
920 size_t num_skipped = 0;
921 while (iter_.Valid()) {
922 ParsedInternalKey ikey;
923 if (!ParseKey(&ikey)) {
924 return false;
925 }
926
927 if (!user_comparator_.EqualWithoutTimestamp(ikey.user_key,
928 saved_key_.GetUserKey())) {
929 // Found a smaller user key, thus we are done with current user key.
930 break;
931 }
932
933 assert(ikey.user_key.size() >= timestamp_size_);
934 Slice ts;
935 if (timestamp_size_ > 0) {
936 ts = Slice(ikey.user_key.data() + ikey.user_key.size() - timestamp_size_,
937 timestamp_size_);
938 }
939
940 bool visible = IsVisible(ikey.sequence, ts);
941 if (!visible &&
942 (timestamp_lb_ == nullptr ||
943 user_comparator_.CompareTimestamp(ts, *timestamp_ub_) > 0)) {
944 // Found an invisible version of the current user key, and it must have
945 // a higher sequence number or timestamp. Therefore, we are done with the
946 // current user key.
947 break;
948 }
949
950 if (!ts.empty()) {
951 saved_timestamp_.assign(ts.data(), ts.size());
952 }
953
954 if (TooManyInternalKeysSkipped()) {
955 return false;
956 }
957
958 // This user key has lots of entries.
959 // We're going from old to new, and it's taking too long. Let's do a Seek()
960 // and go from new to old. This helps when a key was overwritten many times.
961 if (num_skipped >= max_skip_) {
962 return FindValueForCurrentKeyUsingSeek();
963 }
964
965 if (!PrepareValueInternal()) {
966 return false;
967 }
968
969 if (timestamp_lb_ != nullptr) {
970 // Only needed when timestamp_lb_ is not null
971 [[maybe_unused]] const bool ret = ParseKey(&ikey_);
972 // Since the preceding ParseKey(&ikey) succeeds, so must this.
973 assert(ret);
974 saved_key_.SetInternalKey(ikey);
975 } else if (user_comparator_.Compare(ikey.user_key,
976 saved_key_.GetUserKey()) < 0) {
977 saved_key_.SetUserKey(
978 ikey.user_key,
979 !pin_thru_lifetime_ || !iter_.iter()->IsKeyPinned() /* copy */);
980 }
981
982 valid_entry_seen = true;
983 last_key_entry_type = ikey.type;
984 switch (last_key_entry_type) {
985 case kTypeValue:
986 case kTypeValuePreferredSeqno:
987 case kTypeBlobIndex:
988 case kTypeWideColumnEntity:
989 if (iter_.iter()->IsValuePinned()) {
990 saved_write_unix_time_ = iter_.write_unix_time();
991 if (last_key_entry_type == kTypeValuePreferredSeqno) {
992 pinned_value_ = ParsePackedValueForValue(iter_.value());
993 } else {
994 pinned_value_ = iter_.value();
995 }
996 } else {
997 valid_ = false;
998 status_ = Status::NotSupported(
999 "Backward iteration not supported if underlying iterator's value "
1000 "cannot be pinned.");
1001 }
1002 merge_context_.Clear();
1003 last_not_merge_type = last_key_entry_type;
1004 if (!status_.ok()) {
1005 return false;
1006 }
1007 break;
1008 case kTypeDeletion:
1009 case kTypeDeletionWithTimestamp:
1010 case kTypeSingleDeletion:
1011 merge_context_.Clear();
1012 last_not_merge_type = last_key_entry_type;
1013 PERF_COUNTER_ADD(internal_delete_skipped_count, 1);
1014 break;
1015 case kTypeMerge: {
1016 assert(merge_operator_ != nullptr);
1017 merge_context_.PushOperandBack(
1018 iter_.value(), iter_.iter()->IsValuePinned() /* operand_pinned */);
1019 PERF_COUNTER_ADD(internal_merge_count, 1);
1020 } break;
1021 default:
1022 valid_ = false;
1023 status_ = Status::Corruption(
1024 "Unknown value type: " +
1025 std::to_string(static_cast<unsigned int>(last_key_entry_type)));
1026 return false;
1027 }
1028
1029 PERF_COUNTER_ADD(internal_key_skipped_count, 1);
1030 iter_.Prev();
1031 ++num_skipped;
1032
1033 if (visible && timestamp_lb_ != nullptr) {
1034 // If timestamp_lb_ is not nullptr, we do not have to look further for
1035 // another internal key. We can return this current internal key. Yet we
1036 // still keep the invariant that iter_ is positioned before the returned
1037 // key.
1038 break;
1039 }
1040 }
1041
1042 if (!iter_.status().ok()) {
1043 valid_ = false;
1044 return false;
1045 }
1046
1047 if (!valid_entry_seen) {
1048 // Since we haven't seen any valid entry, last_key_entry_type remains
1049 // unchanged and the same as its initial value.
1050 assert(last_key_entry_type == kTypeDeletion);
1051 assert(last_not_merge_type == kTypeDeletion);
1052 valid_ = false;
1053 return true;
1054 }
1055
1056 if (timestamp_lb_ != nullptr) {
1057 assert(last_key_entry_type == ikey_.type);
1058 }
1059
1060 switch (last_key_entry_type) {
1061 case kTypeDeletion:
1062 case kTypeDeletionWithTimestamp:
1063 case kTypeSingleDeletion:
1064 if (timestamp_lb_ == nullptr) {
1065 valid_ = false;
1066 } else {
1067 valid_ = true;
1068 }
1069 return true;
1070 case kTypeMerge:
1071 current_entry_is_merged_ = true;
1072 if (last_not_merge_type == kTypeDeletion ||
1073 last_not_merge_type == kTypeSingleDeletion ||
1074 last_not_merge_type == kTypeDeletionWithTimestamp) {
1075 if (!MergeWithNoBaseValue(saved_key_.GetUserKey())) {
1076 return false;
1077 }
1078 return true;
1079 } else if (last_not_merge_type == kTypeBlobIndex) {
1080 if (!MergeWithBlobBaseValue(pinned_value_, saved_key_.GetUserKey())) {
1081 return false;
1082 }
1083
1084 return true;
1085 } else if (last_not_merge_type == kTypeWideColumnEntity) {
1086 if (!MergeWithWideColumnBaseValue(pinned_value_,
1087 saved_key_.GetUserKey())) {
1088 return false;
1089 }
1090
1091 return true;
1092 } else {
1093 assert(last_not_merge_type == kTypeValue ||
1094 last_not_merge_type == kTypeValuePreferredSeqno);
1095 if (!MergeWithPlainBaseValue(pinned_value_, saved_key_.GetUserKey())) {
1096 return false;
1097 }
1098 return true;
1099 }
1100 break;
1101 case kTypeValue:
1102 case kTypeValuePreferredSeqno:
1103 SetValueAndColumnsFromPlain(pinned_value_);
1104
1105 break;
1106 case kTypeBlobIndex:
1107 if (!SetValueAndColumnsFromBlob(saved_key_.GetUserKey(), pinned_value_)) {
1108 return false;
1109 }
1110 break;
1111 case kTypeWideColumnEntity:
1112 if (!SetValueAndColumnsFromEntity(pinned_value_)) {
1113 return false;
1114 }
1115 break;
1116 default:
1117 valid_ = false;
1118 status_ = Status::Corruption(
1119 "Unknown value type: " +
1120 std::to_string(static_cast<unsigned int>(last_key_entry_type)));
1121 return false;
1122 }
1123 valid_ = true;
1124 return true;
1125}
1126
1127// This function is used in FindValueForCurrentKey.
1128// We use Seek() function instead of Prev() to find necessary value
1129// TODO: This is very similar to FindNextUserEntry() and MergeValuesNewToOld().
1130// Would be nice to reuse some code.
1131bool DBIter::FindValueForCurrentKeyUsingSeek() {
1132 // FindValueForCurrentKey will enable pinning before calling
1133 // FindValueForCurrentKeyUsingSeek()
1134 assert(pinned_iters_mgr_.PinningEnabled());
1135 std::string last_key;
1136 if (0 == timestamp_size_) {
1137 AppendInternalKey(&last_key,
1138 ParsedInternalKey(saved_key_.GetUserKey(), sequence_,
1139 kValueTypeForSeek));
1140 } else {
1141 AppendInternalKeyWithDifferentTimestamp(
1142 &last_key,
1143 ParsedInternalKey(saved_key_.GetUserKey(), sequence_,
1144 kValueTypeForSeek),
1145 timestamp_lb_ == nullptr ? *timestamp_ub_ : *timestamp_lb_);
1146 }
1147 iter_.Seek(last_key);
1148 RecordTick(statistics_, NUMBER_OF_RESEEKS_IN_ITERATION);
1149
1150 // In case read_callback presents, the value we seek to may not be visible.
1151 // Find the next value that's visible.
1152 ParsedInternalKey ikey;
1153
1154 while (true) {
1155 if (!iter_.Valid()) {
1156 valid_ = false;
1157 return iter_.status().ok();
1158 }
1159
1160 if (!ParseKey(&ikey)) {
1161 return false;
1162 }
1163 assert(ikey.user_key.size() >= timestamp_size_);
1164 Slice ts;
1165 if (timestamp_size_ > 0) {
1166 ts = Slice(ikey.user_key.data() + ikey.user_key.size() - timestamp_size_,
1167 timestamp_size_);
1168 }
1169
1170 if (!user_comparator_.EqualWithoutTimestamp(ikey.user_key,
1171 saved_key_.GetUserKey())) {
1172 // No visible values for this key, even though FindValueForCurrentKey()
1173 // has seen some. This is possible if we're using a tailing iterator, and
1174 // the entries were discarded in a compaction.
1175 valid_ = false;
1176 return true;
1177 }
1178
1179 if (IsVisible(ikey.sequence, ts)) {
1180 break;
1181 }
1182
1183 iter_.Next();
1184 }
1185
1186 if (ikey.type == kTypeDeletion || ikey.type == kTypeSingleDeletion ||
1187 kTypeDeletionWithTimestamp == ikey.type) {
1188 if (timestamp_lb_ == nullptr) {
1189 valid_ = false;
1190 } else {
1191 valid_ = true;
1192 saved_key_.SetInternalKey(ikey);
1193 }
1194 return true;
1195 }
1196 if (!PrepareValueInternal()) {
1197 return false;
1198 }
1199 if (timestamp_size_ > 0) {
1200 Slice ts = ExtractTimestampFromUserKey(ikey.user_key, timestamp_size_);
1201 saved_timestamp_.assign(ts.data(), ts.size());
1202 }
1203 if (ikey.type == kTypeValue || ikey.type == kTypeValuePreferredSeqno ||
1204 ikey.type == kTypeBlobIndex || ikey.type == kTypeWideColumnEntity) {
1205 assert(iter_.iter()->IsValuePinned());
1206 saved_write_unix_time_ = iter_.write_unix_time();
1207 if (ikey.type == kTypeValuePreferredSeqno) {
1208 pinned_value_ = ParsePackedValueForValue(iter_.value());
1209 } else {
1210 pinned_value_ = iter_.value();
1211 }
1212 if (ikey.type == kTypeBlobIndex) {
1213 if (!SetValueAndColumnsFromBlob(ikey.user_key, pinned_value_)) {
1214 return false;
1215 }
1216 } else if (ikey.type == kTypeWideColumnEntity) {
1217 if (!SetValueAndColumnsFromEntity(pinned_value_)) {
1218 return false;
1219 }
1220 } else {
1221 assert(ikey.type == kTypeValue || ikey.type == kTypeValuePreferredSeqno);
1222 SetValueAndColumnsFromPlain(pinned_value_);
1223 }
1224
1225 if (timestamp_lb_ != nullptr) {
1226 saved_key_.SetInternalKey(ikey);
1227 }
1228
1229 valid_ = true;
1230 return true;
1231 }
1232
1233 // kTypeMerge. We need to collect all kTypeMerge values and save them
1234 // in operands
1235 assert(ikey.type == kTypeMerge);
1236 current_entry_is_merged_ = true;
1237 merge_context_.Clear();
1238 merge_context_.PushOperand(
1239 iter_.value(), iter_.iter()->IsValuePinned() /* operand_pinned */);
1240 PERF_COUNTER_ADD(internal_merge_count, 1);
1241
1242 while (true) {
1243 iter_.Next();
1244
1245 if (!iter_.Valid()) {
1246 if (!iter_.status().ok()) {
1247 valid_ = false;
1248 return false;
1249 }
1250 break;
1251 }
1252 if (!ParseKey(&ikey)) {
1253 return false;
1254 }
1255 if (!user_comparator_.EqualWithoutTimestamp(ikey.user_key,
1256 saved_key_.GetUserKey())) {
1257 break;
1258 }
1259 if (ikey.type == kTypeDeletion || ikey.type == kTypeSingleDeletion ||
1260 ikey.type == kTypeDeletionWithTimestamp) {
1261 break;
1262 }
1263 if (!PrepareValueInternal()) {
1264 return false;
1265 }
1266
1267 if (ikey.type == kTypeValue || ikey.type == kTypeValuePreferredSeqno) {
1268 Slice value = iter_.value();
1269 if (ikey.type == kTypeValuePreferredSeqno) {
1270 value = ParsePackedValueForValue(value);
1271 }
1272 if (!MergeWithPlainBaseValue(value, saved_key_.GetUserKey())) {
1273 return false;
1274 }
1275 return true;
1276 } else if (ikey.type == kTypeMerge) {
1277 merge_context_.PushOperand(
1278 iter_.value(), iter_.iter()->IsValuePinned() /* operand_pinned */);
1279 PERF_COUNTER_ADD(internal_merge_count, 1);
1280 } else if (ikey.type == kTypeBlobIndex) {
1281 if (!MergeWithBlobBaseValue(iter_.value(), saved_key_.GetUserKey())) {
1282 return false;
1283 }
1284
1285 return true;
1286 } else if (ikey.type == kTypeWideColumnEntity) {
1287 if (!MergeWithWideColumnBaseValue(iter_.value(),
1288 saved_key_.GetUserKey())) {
1289 return false;
1290 }
1291
1292 return true;
1293 } else {
1294 valid_ = false;
1295 status_ = Status::Corruption(
1296 "Unknown value type: " +
1297 std::to_string(static_cast<unsigned int>(ikey.type)));
1298 return false;
1299 }
1300 }
1301
1302 if (!MergeWithNoBaseValue(saved_key_.GetUserKey())) {
1303 return false;
1304 }
1305
1306 // Make sure we leave iter_ in a good state. If it's valid and we don't care
1307 // about prefixes, that's already good enough. Otherwise it needs to be
1308 // seeked to the current key.
1309 if (!expect_total_order_inner_iter() || !iter_.Valid()) {
1310 if (!expect_total_order_inner_iter()) {
1311 iter_.SeekForPrev(last_key);
1312 } else {
1313 iter_.Seek(last_key);
1314 if (!iter_.Valid() && iter_.status().ok()) {
1315 iter_.SeekToLast();
1316 }
1317 }
1318 RecordTick(statistics_, NUMBER_OF_RESEEKS_IN_ITERATION);
1319 }
1320
1321 valid_ = true;
1322 return true;
1323}

1392// Move backwards until the key smaller than saved_key_.
1393// Changes valid_ only if return value is false.
1394bool DBIter::FindUserKeyBeforeSavedKey() {
1395 assert(status_.ok());
1396 size_t num_skipped = 0;
1397 while (iter_.Valid()) {
1398 ParsedInternalKey ikey;
1399 if (!ParseKey(&ikey)) {
1400 return false;
1401 }
1402
1403 if (CompareKeyForSkip(ikey.user_key, saved_key_.GetUserKey()) < 0) {
1404 return true;
1405 }
1406
1407 if (TooManyInternalKeysSkipped()) {
1408 return false;
1409 }
1410
1411 assert(ikey.sequence != kMaxSequenceNumber);
1412 assert(ikey.user_key.size() >= timestamp_size_);
1413 Slice ts;
1414 if (timestamp_size_ > 0) {
1415 ts = Slice(ikey.user_key.data() + ikey.user_key.size() - timestamp_size_,
1416 timestamp_size_);
1417 }
1418 if (!IsVisible(ikey.sequence, ts)) {
1419 PERF_COUNTER_ADD(internal_recent_skipped_count, 1);
1420 } else {
1421 PERF_COUNTER_ADD(internal_key_skipped_count, 1);
1422 }
1423
1424 if (num_skipped >= max_skip_) {
1425 num_skipped = 0;
1426 std::string last_key;
1427 if (timestamp_size_ == 0) {
1428 AppendInternalKey(&last_key, ParsedInternalKey(saved_key_.GetUserKey(),
1429 kMaxSequenceNumber,
1430 kValueTypeForSeek));
1431 } else {
1432 // TODO: pre-create kTsMax.
1433 const std::string kTsMax(timestamp_size_, '\xff');
1434 AppendInternalKeyWithDifferentTimestamp(
1435 &last_key,
1436 ParsedInternalKey(saved_key_.GetUserKey(), kMaxSequenceNumber,
1437 kValueTypeForSeek),
1438 kTsMax);
1439 }
1440 // It would be more efficient to use SeekForPrev() here, but some
1441 // iterators may not support it.
1442 iter_.Seek(last_key);
1443 RecordTick(statistics_, NUMBER_OF_RESEEKS_IN_ITERATION);
1444 if (!iter_.Valid()) {
1445 break;
1446 }
1447 } else {
1448 ++num_skipped;
1449 }
1450
1451 iter_.Prev();
1452 }
1453
1454 if (!iter_.status().ok()) {
1455 valid_ = false;
1456 return false;
1457 }
1458
1459 return true;
1460}

1542void DBIter::Seek(const Slice& target) {
1543 PERF_COUNTER_ADD(iter_seek_count, 1);
1544 PERF_CPU_TIMER_GUARD(iter_seek_cpu_nanos, clock_);
1545 StopWatch sw(clock_, statistics_, DB_SEEK);
1546
1547 if (cfh_ != nullptr) {
1548 // TODO: What do we do if this returns an error?
1549 Slice lower_bound, upper_bound;
1550 if (iterate_lower_bound_ != nullptr) {
1551 lower_bound = *iterate_lower_bound_;
1552 } else {
1553 lower_bound = Slice("");
1554 }
1555 if (iterate_upper_bound_ != nullptr) {
1556 upper_bound = *iterate_upper_bound_;
1557 } else {
1558 upper_bound = Slice("");
1559 }
1560 cfh_->db()
1561 ->TraceIteratorSeek(cfh_->cfd()->GetID(), target, lower_bound,
1562 upper_bound)
1563 .PermitUncheckedError();
1564 }
1565
1566 status_ = Status::OK();
1567 ReleaseTempPinnedData();
1568 ResetBlobData();
1569 ResetValueAndColumns();
1570 ResetInternalKeysSkippedCounter();
1571
1572 // Seek the inner iterator based on the target key.
1573 {
1574 PERF_TIMER_GUARD(seek_internal_seek_time);
1575
1576 SetSavedKeyToSeekTarget(target);
1577 iter_.Seek(saved_key_.GetInternalKey());
1578
1579 RecordTick(statistics_, NUMBER_DB_SEEK);
1580 }
1581 if (!iter_.Valid()) {
1582 valid_ = false;
1583 return;
1584 }
1585 direction_ = kForward;
1586
1587 // Now the inner iterator is placed to the target position. From there,
1588 // we need to find out the next key that is visible to the user.
1589 ClearSavedValue();
1590 if (prefix_same_as_start_) {
1591 // The case where the iterator needs to be invalidated if it has exhausted
1592 // keys within the same prefix of the seek key.
1593 assert(prefix_extractor_ != nullptr);
1594 Slice target_prefix = prefix_extractor_->Transform(target);
1595 FindNextUserEntry(false /* not skipping saved_key */,
1596 &target_prefix /* prefix */);
1597 if (valid_) {
1598 // Remember the prefix of the seek key for the future Next() call to
1599 // check.
1600 prefix_.SetUserKey(target_prefix);
1601 }
1602 } else {
1603 FindNextUserEntry(false /* not skipping saved_key */, nullptr);
1604 }
1605 if (!valid_) {
1606 return;
1607 }
1608
1609 // Updating stats and perf context counters.
1610 if (statistics_ != nullptr) {
1611 // Decrement since we don't want to count this key as skipped
1612 RecordTick(statistics_, NUMBER_DB_SEEK_FOUND);
1613 RecordTick(statistics_, ITER_BYTES_READ, key().size() + value().size());
1614 }
1615 PERF_COUNTER_ADD(iter_read_bytes, key().size() + value().size());
1616}
