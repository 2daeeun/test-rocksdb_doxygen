252#ifndef __SSE4_2__
253 uint32_t c = static_cast<uint32_t>(*l ^ LE_LOAD32(*p));
254 *p += 4;
255 *l = table3_[c & 0xff] ^ table2_[(c >> 8) & 0xff] ^
256 table1_[(c >> 16) & 0xff] ^ table0_[c >> 24];
257 // DO it twice.
258 c = static_cast<uint32_t>(*l ^ LE_LOAD32(*p));
259 *p += 4;
260 *l = table3_[c & 0xff] ^ table2_[(c >> 8) & 0xff] ^
261 table1_[(c >> 16) & 0xff] ^ table0_[c >> 24];
262#elif defined(__LP64__) || defined(_WIN64)
263 *l = _mm_crc32_u64(*l, DecodeFixed64(reinterpret_cast<const char*>(*p)));
264 *p += 8;
265#else

267 *p += 4;
268 *l = _mm_crc32_u32(static_cast<unsigned int>(*l), LE_LOAD32(*p));

273template <void (*CRC32)(uint64_t*, uint8_t const**)>
274uint32_t ExtendImpl(uint32_t crc, const char* buf, size_t size) {
275 const uint8_t* p = reinterpret_cast<const uint8_t*>(buf);
276 const uint8_t* e = p + size;
277 uint64_t l = crc ^ 0xffffffffu;
278
279// Align n to (1 << m) byte boundary
280#define ALIGN(n, m) ((n + ((1 << m) - 1)) & ~((1 << m) - 1))
281
282#define STEP1 \
283 do { \
284 int c = (l & 0xff) ^ *p++; \
285 l = table0_[c] ^ (l >> 8); \
286 } while (0)
287
288 // Point x at first 16-byte aligned byte in string. This might be
289 // just past the end of the string.
290 const uintptr_t pval = reinterpret_cast<uintptr_t>(p);
291 const uint8_t* x = reinterpret_cast<const uint8_t*>(ALIGN(pval, 4));
292 if (x <= e) {
293 // Process bytes until finished or p is 16-byte aligned
294 while (p != x) {
295 STEP1;
296 }
297 }
298 // Process bytes 16 at a time
299 while ((e - p) >= 16) {
300 CRC32(&l, &p);
301 CRC32(&l, &p);
302 }
303 // Process bytes 8 at a time
304 while ((e - p) >= 8) {
305 CRC32(&l, &p);
306 }
307 // Process the last few bytes
308 while (p != e) {
309 STEP1;
310 }
311#undef STEP1
312#undef ALIGN
313 return static_cast<uint32_t>(l ^ 0xffffffffu);
314}

531 next += sizeof(uint16_t);

534 crc32bit = _mm_crc32_u8(crc32bit, *(next));
535 next++;
536 }
537 crc0 = crc32bit;

541// CombineCRC performs pclmulqdq multiplication of 2 partial CRC's and a well
542// chosen constant and xor's these with the remaining CRC.
543//
544inline uint64_t CombineCRC(
545 size_t block_size,
546 uint64_t crc0,
547 uint64_t crc1,
548 uint64_t crc2,
549 const uint64_t* next2) {
550 const auto multiplier =
551 *(reinterpret_cast<const __m128i*>(clmul_constants) + block_size - 1);
552 const auto crc0_xmm = _mm_set_epi64x(0, crc0);
553 const auto res0 = _mm_clmulepi64_si128(crc0_xmm, multiplier, 0x00);
554 const auto crc1_xmm = _mm_set_epi64x(0, crc1);
555 const auto res1 = _mm_clmulepi64_si128(crc1_xmm, multiplier, 0x10);
556 const auto res = _mm_xor_si128(res0, res1);
557 crc0 = _mm_cvtsi128_si64(res);
558 crc0 = crc0 ^ *((uint64_t*)next2 - 1);
559 crc2 = _mm_crc32_u64(crc2, crc0);
560 return crc2;
561}

571uint32_t crc32c_3way(uint32_t crc, const char* buf, size_t len) {
572 const unsigned char* next = (const unsigned char*)buf;
573 uint64_t count;
574 uint64_t crc0, crc1, crc2;

576
577
578 if (len >= 8) {
579 // if len > 216 then align and use triplets
580 if (len > 216) {
581 {
582 // Work on the bytes (< 8) before the first 8-byte alignment addr starts
583 uint64_t align_bytes = (8 - (uintptr_t)next) & 7;
584 len -= align_bytes;
585 align_to_8(align_bytes, crc0, next);
586 }
587
588 // Now work on the remaining blocks
589 count = len / 24; // number of triplets
590 len %= 24; // bytes remaining
591 uint64_t n = count >> 7; // #blocks = first block + full blocks
592 uint64_t block_size = count & 127;
593 if (block_size == 0) {
594 block_size = 128;
595 } else {
596 n++;
597 }
598 // points to the first byte of the next block
599 const uint64_t* next0 = (uint64_t*)next + block_size;
600 const uint64_t* next1 = next0 + block_size;
601 const uint64_t* next2 = next1 + block_size;
602
603 crc1 = crc2 = 0;
604 // Use Duff's device, a for() loop inside a switch()
605 // statement. This needs to execute at least once, round len
606 // down to nearest triplet multiple
607 switch (block_size) {
608 case 128:
609 do {
610 // jumps here for a full block of len 128
611 CRCtriplet(crc, next, -128);
612 FALLTHROUGH_INTENDED;
613 case 127:
614 // jumps here or below for the first block smaller
615 CRCtriplet(crc, next, -127);
616 FALLTHROUGH_INTENDED;
617 case 126:
618 CRCtriplet(crc, next, -126); // than 128
619 FALLTHROUGH_INTENDED;
620 case 125:
621 CRCtriplet(crc, next, -125);
622 FALLTHROUGH_INTENDED;
623 case 124:
624 CRCtriplet(crc, next, -124);
625 FALLTHROUGH_INTENDED;
626 case 123:
627 CRCtriplet(crc, next, -123);
628 FALLTHROUGH_INTENDED;
629 case 122:
630 CRCtriplet(crc, next, -122);
631 FALLTHROUGH_INTENDED;
632 case 121:
633 CRCtriplet(crc, next, -121);
634 FALLTHROUGH_INTENDED;
635 case 120:
636 CRCtriplet(crc, next, -120);
637 FALLTHROUGH_INTENDED;
638 case 119:
639 CRCtriplet(crc, next, -119);
640 FALLTHROUGH_INTENDED;
641 case 118:
642 CRCtriplet(crc, next, -118);
643 FALLTHROUGH_INTENDED;
644 case 117:
645 CRCtriplet(crc, next, -117);
646 FALLTHROUGH_INTENDED;
647 case 116:
648 CRCtriplet(crc, next, -116);
649 FALLTHROUGH_INTENDED;
650 case 115:
651 CRCtriplet(crc, next, -115);
652 FALLTHROUGH_INTENDED;
653 case 114:
654 CRCtriplet(crc, next, -114);
655 FALLTHROUGH_INTENDED;
656 case 113:
657 CRCtriplet(crc, next, -113);
658 FALLTHROUGH_INTENDED;

1234// Background:
1235// RocksDB uses two kinds of crc32c values: masked and unmasked. Neither is
1236// a "pure" CRC because a pure CRC satisfies (^ for xor)
1237// crc(a ^ b) = crc(a) ^ crc(b)
1238// The unmasked is closest, and this function takes unmasked crc32c values.
1239// The unmasked values are impure in two ways:
1240// * The initial setting at the start of CRC computation is all 1 bits
1241// (like -1) instead of zero.
1242// * The result has all bits invered.
1243// Note that together, these result in the empty string having a crc32c of
1244// zero. See
1245// https://en.wikipedia.org/wiki/Computation_of_cyclic_redundancy_checks#CRC_variants
1246//
1247// Simplified version of strategy, using xor through pure CRCs (+ for concat):
1248//
1249// pure_crc(str1 + str2) = pure_crc(str1 + zeros(len(str2))) ^
1250// pure_crc(zeros(len(str1)) + str2)
1251//
1252// because the xor of these two zero-padded strings is str1 + str2. For pure
1253// CRC, leading zeros don't affect the result, so we only need
1254//
1255// pure_crc(str1 + str2) = pure_crc(str1 + zeros(len(str2))) ^
1256// pure_crc(str2)
1257//
1258// Considering we aren't working with pure CRCs, what is actually in the input?
1259//
1260// crc1 = PureToInverted(PureExtendCrc32c(-1, zeros, crc1len) ^
1261// PureCrc32c(str1, crc1len))
1262// crc2 = PureToInverted(PureExtendCrc32c(-1, zeros, crc2len) ^
1263// PureCrc32c(str2, crc2len))
1264//
1265// The result we want to compute is
1266// combined = PureToInverted(PureExtendCrc32c(PureExtendCrc32c(-1, zeros,
1267// crc1len) ^
1268// PureCrc32c(str1, crc1len),
1269// zeros, crc2len) ^
1270// PureCrc32c(str2, crc2len))
1271//
1272// Thus, in addition to extending crc1 over the length of str2 in (virtual)
1273// zeros, we need to cancel out the -1 initializer that was used in computing
1274// crc2. To cancel it out, we also need to extend it over crc2len in zeros.
1275// To simplify, since the end of str1 and that -1 initializer for crc2 are at
1276// the same logical position, we can combine them before we extend over the
1277// zeros.
1278uint32_t Crc32cCombine(uint32_t crc1, uint32_t crc2, size_t crc2len) {
1279 uint32_t pure_crc1_with_init = InvertedToPure(crc1);
1280 uint32_t pure_crc2_with_init = InvertedToPure(crc2);
1281 uint32_t pure_crc2_init = static_cast<uint32_t>(-1);
1282
1283 // Append up to 32 bits of zeroes in the normal way
