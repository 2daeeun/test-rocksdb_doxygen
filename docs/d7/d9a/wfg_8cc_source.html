69// Destroy the internals of the lock request graph
70void wfg::destroy(void) {
71 uint32_t n_nodes = m_nodes.size();
72 for (uint32_t i = 0; i < n_nodes; i++) {
73 node *n;
74 int r = m_nodes.fetch(i, &n);
75 invariant_zero(r);
76 invariant_notnull(n);
77 if (r) continue; // Get rid of "may be used uninitialized" warning
78 node::free(n);
79 }
80 m_nodes.destroy();
81}

97bool wfg::cycle_exists_from_node(node *target, node *head,
98 std::function<void(TXNID)> reporter) {
99 bool cycle_found = false;
100 head->visited = true;
101 uint32_t n_edges = head->edges.size();
102 for (uint32_t i = 0; i < n_edges && !cycle_found; i++) {
103 TXNID edge_id = head->edges.get(i);
104 if (target->txnid == edge_id) {
105 cycle_found = true;
106 if (reporter) reporter(edge_id);
107 } else {
108 node *new_head = find_node(edge_id);
109 if (new_head && !new_head->visited) {
110 cycle_found = cycle_exists_from_node(target, new_head, reporter);
111 if (cycle_found && reporter) reporter(edge_id);
112 }
113 }
114 }
115 head->visited = false;
116 return cycle_found;
117}

119// Return true if there exists a cycle from a given transaction id in the graph.
120// Return false otherwise.
121bool wfg::cycle_exists_from_txnid(TXNID txnid,
122 std::function<void(TXNID)> reporter) {
123 node *a_node = find_node(txnid);
124 bool cycles_found = false;
125 if (a_node) {
126 cycles_found = cycle_exists_from_node(a_node, a_node, reporter);
127 }
128 return cycles_found;
129}

131// Apply a given function f to all of the nodes in the graph. The apply
132// function returns when the function f is called for all of the nodes in the
133// graph, or the function f returns non-zero.
134void wfg::apply_nodes(int (*fn)(TXNID id, void *extra), void *extra) {
135 int r = 0;
136 uint32_t n_nodes = m_nodes.size();
137 for (uint32_t i = 0; i < n_nodes && r == 0; i++) {
138 node *n;
139 r = m_nodes.fetch(i, &n);
140 invariant_zero(r);
141 if (r) continue; // Get rid of "may be used uninitialized" warning
142 r = fn(n->txnid, extra);
143 }
144}

146// Apply a given function f to all of the edges whose origin is a given node id.
147// The apply function returns when the function f is called for all edges in the
148// graph rooted at node id, or the function f returns non-zero.
149void wfg::apply_edges(TXNID txnid,
150 int (*fn)(TXNID txnid, TXNID edge_txnid, void *extra),
151 void *extra) {
152 node *n = find_node(txnid);
153 if (n) {
154 int r = 0;
155 uint32_t n_edges = n->edges.size();
156 for (uint32_t i = 0; i < n_edges && r == 0; i++) {
157 r = fn(txnid, n->edges.get(i), extra);
158 }
159 }
160}

162// find node by id
163wfg::node *wfg::find_node(TXNID txnid) {
164 node *n = nullptr;
165 int r = m_nodes.find_zero<TXNID, find_by_txnid>(txnid, &n, nullptr);
166 invariant(r == 0 || r == DB_NOTFOUND);
167 return n;
168}

170// this is the omt comparison function
171// nodes are compared by their txnid.
172int wfg::find_by_txnid(node *const &node_a, const TXNID &txnid_b) {
173 TXNID txnid_a = node_a->txnid;
174 if (txnid_a < txnid_b) {
175 return -1;
176 } else if (txnid_a == txnid_b) {
177 return 0;
178 } else {
179 return 1;
180 }
181}

183// insert a new node
184wfg::node *wfg::find_create_node(TXNID txnid) {
185 node *n;
186 uint32_t idx;
187 int r = m_nodes.find_zero<TXNID, find_by_txnid>(txnid, &n, &idx);
188 if (r == DB_NOTFOUND) {
189 n = node::alloc(txnid);
190 r = m_nodes.insert_at(n, idx);
191 invariant_zero(r);
192 }
193 invariant_notnull(n);
194 return n;
195}

197wfg::node *wfg::node::alloc(TXNID txnid) {
198 node *XCALLOC(n);
199 n->txnid = txnid;
200 n->visited = false;
201 n->edges.create();
202 return n;
203}
