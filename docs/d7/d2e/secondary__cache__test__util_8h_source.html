59 std::shared_ptr<Cache> NewCache(
60 size_t capacity,
61 std::function<void(ShardedCacheOptions&)> modify_opts_fn = {}) {
62 const auto& type = Type();
63 if (type == kLRU) {
64 LRUCacheOptions lru_opts;
65 lru_opts.capacity = capacity;
66 lru_opts.hash_seed = 0; // deterministic tests
67 if (modify_opts_fn) {
68 modify_opts_fn(lru_opts);
69 }
70 return lru_opts.MakeSharedCache();
71 }
72 if (IsHyperClock(type)) {
73 HyperClockCacheOptions hc_opts{
74 capacity, type == kFixedHyperClock ? estimated_value_size_ : 0};
75 hc_opts.min_avg_entry_charge =
76 std::max(size_t{1}, estimated_value_size_ / 2);
77 hc_opts.hash_seed = 0; // deterministic tests
78 if (modify_opts_fn) {
79 modify_opts_fn(hc_opts);
80 }
81 return hc_opts.MakeSharedCache();
82 }
83 assert(false);
84 return nullptr;
85 }
