21// CachableEntry is a handle to an object that may or may not be in the block
22// cache. It is used in a variety of ways:
23//
24// 1) It may refer to an object in the block cache. In this case, cache_ and
25// cache_handle_ are not nullptr, and the cache handle has to be released when
26// the CachableEntry is destroyed (the lifecycle of the cached object, on the
27// other hand, is managed by the cache itself).
28// 2) It may uniquely own the (non-cached) object it refers to (examples include
29// a block read directly from file, or uncompressed blocks when there is a
30// compressed block cache but no uncompressed block cache). In such cases, the
31// object has to be destroyed when the CachableEntry is destroyed.
32// 3) It may point to an object (cached or not) without owning it. In this case,
33// no action is needed when the CachableEntry is destroyed.
34// 4) Sometimes, management of a cached or owned object (see #1 and #2 above)
35// is transferred to some other object. This is used for instance with iterators
36// (where cleanup is performed using a chain of cleanup functions,
37// see Cleanable).
38//
39// Because of #1 and #2 above, copying a CachableEntry is not safe (and thus not
40// allowed); hence, this is a move-only type, where a move transfers the
41// management responsibilities, and leaves the source object in an empty state.
42
43template <class T>
44class CachableEntry {
45 public:
46 CachableEntry() = default;
47
48 CachableEntry(T* value, Cache* cache, Cache::Handle* cache_handle,
49 bool own_value)
50 : value_(value),
51 cache_(cache),
52 cache_handle_(cache_handle),
53 own_value_(own_value) {
54 assert(value_ != nullptr ||
55 (cache_ == nullptr && cache_handle_ == nullptr && !own_value_));
56 assert(!!cache_ == !!cache_handle_);
57 assert(!cache_handle_ || !own_value_);
58 }
59
60 CachableEntry(const CachableEntry&) = delete;
61 CachableEntry& operator=(const CachableEntry&) = delete;
62
63 CachableEntry(CachableEntry&& rhs) noexcept
64 : value_(rhs.value_),
65 cache_(rhs.cache_),
66 cache_handle_(rhs.cache_handle_),
67 own_value_(rhs.own_value_) {
68 assert(value_ != nullptr ||
69 (cache_ == nullptr && cache_handle_ == nullptr && !own_value_));
70 assert(!!cache_ == !!cache_handle_);
71 assert(!cache_handle_ || !own_value_);
72
73 rhs.ResetFields();
74 }
75
76 CachableEntry& operator=(CachableEntry&& rhs) noexcept {
77 if (UNLIKELY(this == &rhs)) {
78 return *this;
79 }
80
81 ReleaseResource(/*erase_if_last_ref=*/false);
82
83 value_ = rhs.value_;
84 cache_ = rhs.cache_;
85 cache_handle_ = rhs.cache_handle_;
86 own_value_ = rhs.own_value_;
87
88 assert(value_ != nullptr ||
89 (cache_ == nullptr && cache_handle_ == nullptr && !own_value_));
90 assert(!!cache_ == !!cache_handle_);
91 assert(!cache_handle_ || !own_value_);
92
93 rhs.ResetFields();
94
95 return *this;
96 }
97
98 ~CachableEntry() { ReleaseResource(/*erase_if_last_ref=*/false); }
99
100 bool IsEmpty() const {
101 return value_ == nullptr && cache_ == nullptr && cache_handle_ == nullptr &&
102 !own_value_;
103 }
104
105 bool IsCached() const {
106 assert(!!cache_ == !!cache_handle_);
107
108 return cache_handle_ != nullptr;
109 }
110
111 T* GetValue() const { return value_; }
112 Cache* GetCache() const { return cache_; }
113 Cache::Handle* GetCacheHandle() const { return cache_handle_; }
114 bool GetOwnValue() const { return own_value_; }
115
116 void Reset() {
117 ReleaseResource(/*erase_if_last_ref=*/false);
118 ResetFields();
119 }
120
121 void ResetEraseIfLastRef() {
122 ReleaseResource(/*erase_if_last_ref=*/true);
123 ResetFields();
124 }
125
126 void TransferTo(Cleanable* cleanable) {
127 if (cleanable) {
128 if (cache_handle_ != nullptr) {
129 assert(cache_ != nullptr);
130 cleanable->RegisterCleanup(&ReleaseCacheHandle, cache_, cache_handle_);
131 } else if (own_value_) {
132 cleanable->RegisterCleanup(&DeleteValue, value_, nullptr);
133 }
134 }
135
136 ResetFields();
137 }
138
139 void SetOwnedValue(std::unique_ptr<T>&& value) {
140 assert(value.get() != nullptr);
141
142 if (UNLIKELY(value_ == value.get() && own_value_)) {
143 assert(cache_ == nullptr && cache_handle_ == nullptr);
144 return;
145 }
146
147 Reset();
148
149 value_ = value.release();
150 own_value_ = true;
151 }
152
153 void SetUnownedValue(T* value) {
154 assert(value != nullptr);
155
156 if (UNLIKELY(value_ == value && cache_ == nullptr &&
157 cache_handle_ == nullptr && !own_value_)) {
158 return;
159 }
160
161 Reset();
162
163 value_ = value;
164 assert(!own_value_);
165 }
166
167 void SetCachedValue(T* value, Cache* cache, Cache::Handle* cache_handle) {
168 assert(cache != nullptr);
169 assert(cache_handle != nullptr);
170
171 if (UNLIKELY(value_ == value && cache_ == cache &&
172 cache_handle_ == cache_handle && !own_value_)) {
173 return;
174 }
175
176 Reset();
177
178 value_ = value;
179 cache_ = cache;
180 cache_handle_ = cache_handle;
181 assert(!own_value_);
182 }
183
184 // Since this class is essentially an elaborate pointer, it's sometimes
185 // useful to be able to upcast or downcast the base type of the pointer,
186 // especially when interacting with typed_cache.h.
187 template <class TWrapper>
188 std::enable_if_t<sizeof(TWrapper) == sizeof(T) &&
189 (std::is_base_of_v<TWrapper, T> ||
190 std::is_base_of_v<T, TWrapper>),
191 /* Actual return type */
192 CachableEntry<TWrapper>&>
193 As() {
194 CachableEntry<TWrapper>* result_ptr =
195 reinterpret_cast<CachableEntry<TWrapper>*>(this);
196 // Ensure no weirdness in template instantiations
197 assert(static_cast<void*>(&this->value_) ==
198 static_cast<void*>(&result_ptr->value_));
199 assert(&this->cache_handle_ == &result_ptr->cache_handle_);
200 // This function depends on no arithmetic involved in the pointer
201 // conversion, which is not statically checkable.
202 assert(static_cast<void*>(this->value_) ==
203 static_cast<void*>(result_ptr->value_));
204 return *result_ptr;
205 }
206
207 private:
208 void ReleaseResource(bool erase_if_last_ref) noexcept {
209 if (LIKELY(cache_handle_ != nullptr)) {
210 assert(cache_ != nullptr);
211 cache_->Release(cache_handle_, erase_if_last_ref);
212 } else if (own_value_) {
213 delete value_;
214 }
215 }
216
217 void ResetFields() noexcept {
218 value_ = nullptr;
219 cache_ = nullptr;
220 cache_handle_ = nullptr;
221 own_value_ = false;
222 }
223
224 static void ReleaseCacheHandle(void* arg1, void* arg2) {
225 Cache* const cache = static_cast<Cache*>(arg1);
226 assert(cache);
227
228 Cache::Handle* const cache_handle = static_cast<Cache::Handle*>(arg2);
229 assert(cache_handle);
230
231 cache->Release(cache_handle);
232 }
233
234 static void DeleteValue(void* arg1, void* /* arg2 */) {
235 delete static_cast<T*>(arg1);
236 }
237
238 private:
239 // Have to be your own best friend
240 template <class TT>
241 friend class CachableEntry;
242
243 T* value_ = nullptr;
244 Cache* cache_ = nullptr;
245 Cache::Handle* cache_handle_ = nullptr;
246 bool own_value_ = false;
247};
