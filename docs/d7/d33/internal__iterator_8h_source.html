22enum class IterBoundCheck : char {
23 kUnknown = 0,
24 kOutOfBound,
25 kInbound,
26};

28struct IterateResult {
29 Slice key;
30 IterBoundCheck bound_check_result = IterBoundCheck::kUnknown;
31 // If false, PrepareValue() needs to be called before value().
32 bool value_prepared = true;
33};

35template <class TValue>
36class InternalIteratorBase : public Cleanable {
37 public:
38 InternalIteratorBase() {}
39
40 // No copying allowed
41 InternalIteratorBase(const InternalIteratorBase&) = delete;
42 InternalIteratorBase& operator=(const InternalIteratorBase&) = delete;
43
44 virtual ~InternalIteratorBase() {}
45
46 // This iterator will only process range tombstones with sequence
47 // number <= `read_seqno`.
48 // Noop for most child classes.
49 // For range tombstone iterators (TruncatedRangeDelIterator,
50 // FragmentedRangeTombstoneIterator), will only return range tombstones with
51 // sequence number <= `read_seqno`. For LevelIterator, it may open new table
52 // files and create new range tombstone iterators during scanning. It will use
53 // `read_seqno` as the sequence number for creating new range tombstone
54 // iterators.
55 virtual void SetRangeDelReadSeqno(SequenceNumber /* read_seqno */) {}
56
57 // An iterator is either positioned at a key/value pair, or
58 // not valid. This method returns true iff the iterator is valid.
59 // Always returns false if !status().ok().
60 virtual bool Valid() const = 0;
61
62 // Position at the first key in the source. The iterator is Valid()
63 // after this call iff the source is not empty.
64 virtual void SeekToFirst() = 0;
65
66 // Position at the last key in the source. The iterator is
67 // Valid() after this call iff the source is not empty.
68 virtual void SeekToLast() = 0;
69
70 // Position at the first key in the source that at or past target
71 // The iterator is Valid() after this call iff the source contains
72 // an entry that comes at or past target.
73 // All Seek*() methods clear any error status() that the iterator had prior to
74 // the call; after the seek, status() indicates only the error (if any) that
75 // happened during the seek, not any past errors.
76 // 'target' contains user timestamp if timestamp is enabled.
77 virtual void Seek(const Slice& target) = 0;
78
79 // Position at the first key in the source that at or before target
80 // The iterator is Valid() after this call iff the source contains
81 // an entry that comes at or before target.
82 virtual void SeekForPrev(const Slice& target) = 0;
83
84 // Moves to the next entry in the source. After this call, Valid() is
85 // true iff the iterator was not positioned at the last entry in the source.
86 // REQUIRES: Valid()
87 virtual void Next() = 0;
88
89 // Moves to the next entry in the source, and return result. Iterator
90 // implementation should override this method to help methods inline better,
91 // or when UpperBoundCheckResult() is non-trivial.
92 // REQUIRES: Valid()
93 virtual bool NextAndGetResult(IterateResult* result) {
94 Next();
95 bool is_valid = Valid();
96 if (is_valid) {
97 result->key = key();
98 // Default may_be_out_of_upper_bound to true to avoid unnecessary virtual
99 // call. If an implementation has non-trivial UpperBoundCheckResult(),
100 // it should also override NextAndGetResult().
101 result->bound_check_result = IterBoundCheck::kUnknown;
102 result->value_prepared = false;
103 assert(UpperBoundCheckResult() != IterBoundCheck::kOutOfBound);
104 }
105 return is_valid;
106 }
107
108 // Moves to the previous entry in the source. After this call, Valid() is
109 // true iff the iterator was not positioned at the first entry in source.
110 // REQUIRES: Valid()
111 virtual void Prev() = 0;
112
113 // Return the key for the current entry. The underlying storage for
114 // the returned slice is valid only until the next modification of
115 // the iterator.
116 // REQUIRES: Valid()
117 virtual Slice key() const = 0;
118
119 // Returns the approximate write time of this entry, which is deduced from
120 // sequence number if sequence number to time mapping is available.
121 // The default implementation returns maximum uint64_t and that indicates the
122 // write time is unknown.
123 virtual uint64_t write_unix_time() const {
124 return std::numeric_limits<uint64_t>::max();
125 }
126
127 // Return user key for the current entry.
128 // REQUIRES: Valid()
129 virtual Slice user_key() const { return ExtractUserKey(key()); }
130
131 // Return the value for the current entry. The underlying storage for
132 // the returned slice is valid only until the next modification of
133 // the iterator.
134 // REQUIRES: Valid()
135 // REQUIRES: PrepareValue() has been called if needed (see PrepareValue()).
136 virtual TValue value() const = 0;
137
138 // If an error has occurred, return it. Else return an ok status.
139 // If non-blocking IO is requested and this operation cannot be
140 // satisfied without doing some IO, then this returns Status::Incomplete().
141 virtual Status status() const = 0;
142
143 // For some types of iterators, sometimes Seek()/Next()/SeekForPrev()/etc may
144 // load key but not value (to avoid the IO cost of reading the value from disk
145 // if it won't be not needed). This method loads the value in such situation.
146 //
147 // Needs to be called before value() at least once after each iterator
148 // movement (except if IterateResult::value_prepared = true), for iterators
149 // created with allow_unprepared_value = true.
150 //
151 // Returns false if an error occurred; in this case Valid() is also changed
152 // to false, and status() is changed to non-ok.
153 // REQUIRES: Valid()
154 virtual bool PrepareValue() { return true; }
155
156 // Keys return from this iterator can be smaller than iterate_lower_bound.
157 virtual bool MayBeOutOfLowerBound() { return true; }
158
159 // If the iterator has checked the key against iterate_upper_bound, returns
160 // the result here. The function can be used by user of the iterator to skip
161 // their own checks. If Valid() = true, IterBoundCheck::kUnknown is always
162 // a valid value. If Valid() = false, IterBoundCheck::kOutOfBound indicates
163 // that the iterator is filtered out by upper bound checks.
164 virtual IterBoundCheck UpperBoundCheckResult() {
165 return IterBoundCheck::kUnknown;
166 }
167
168 // Pass the PinnedIteratorsManager to the Iterator, most Iterators don't
169 // communicate with PinnedIteratorsManager so default implementation is no-op
170 // but for Iterators that need to communicate with PinnedIteratorsManager
171 // they will implement this function and use the passed pointer to communicate
172 // with PinnedIteratorsManager.
173 virtual void SetPinnedItersMgr(PinnedIteratorsManager* /*pinned_iters_mgr*/) {
174 }
175
176 // If true, this means that the Slice returned by key() is valid as long as
177 // PinnedIteratorsManager::ReleasePinnedData is not called and the
178 // Iterator is not deleted.
179 //
180 // IsKeyPinned() is guaranteed to always return true if
181 // - Iterator is created with ReadOptions::pin_data = true
182 // - DB tables were created with BlockBasedTableOptions::use_delta_encoding
183 // set to false.
184 virtual bool IsKeyPinned() const { return false; }
185
186 // If true, this means that the Slice returned by value() is valid as long as
187 // PinnedIteratorsManager::ReleasePinnedData is not called and the
188 // Iterator is not deleted.
189 // REQUIRES: Same as for value().
190 virtual bool IsValuePinned() const { return false; }
191
192 virtual Status GetProperty(std::string /*prop_name*/, std::string* /*prop*/) {
193 return Status::NotSupported("");
194 }
195
196 // When iterator moves from one file to another file at same level, new file's
197 // readahead state (details of last block read) is updated with previous
198 // file's readahead state. This way internal readahead_size of Prefetch Buffer
199 // doesn't start from scratch and can fall back to 8KB with no prefetch if
200 // reads are not sequential.
201 //
202 // Default implementation is no-op and its implemented by iterators.
203 virtual void GetReadaheadState(ReadaheadFileInfo* /*readahead_file_info*/) {}
204
205 // Default implementation is no-op and its implemented by iterators.
206 virtual void SetReadaheadState(ReadaheadFileInfo* /*readahead_file_info*/) {}
207
208 // When used under merging iterator, LevelIterator treats file boundaries
209 // as sentinel keys to prevent it from moving to next SST file before range
210 // tombstones in the current SST file are no longer needed. This method makes
211 // it cheap to check if the current key is a sentinel key. This should only be
212 // used by MergingIterator and LevelIterator for now.
213 virtual bool IsDeleteRangeSentinelKey() const { return false; }
214
215 protected:
216 void SeekForPrevImpl(const Slice& target, const CompareInterface* cmp) {
217 Seek(target);
218 if (!Valid()) {
219 SeekToLast();
220 }
221 while (Valid() && cmp->Compare(target, key()) < 0) {
222 Prev();
223 }
224 }
225};
