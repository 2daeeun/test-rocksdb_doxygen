97 uint64_t write_unix_time() const override {
98 assert(Valid());
99 ParsedInternalKey pikey;
100 SequenceNumber seqno;
101 const SeqnoToTimeMapping& seqno_to_time_mapping =
102 table_->GetSeqnoToTimeMapping();
103 Status s = ParseInternalKey(key(), &pikey, /*log_err_key=*/false);
104 if (!s.ok()) {
105 return std::numeric_limits<uint64_t>::max();
106 } else if (kUnknownSeqnoBeforeAll == pikey.sequence) {
107 return kUnknownTimeBeforeAll;
108 } else if (seqno_to_time_mapping.Empty()) {
109 return std::numeric_limits<uint64_t>::max();
110 } else if (kTypeValuePreferredSeqno == pikey.type) {
111 seqno = ParsePackedValueForSeqno(value());
112 } else {
113 seqno = pikey.sequence;
114 }
115 return seqno_to_time_mapping.GetProximalTimeBeforeSeqno(seqno);
116 }

118 Slice value() const override {
119 // PrepareValue() must have been called.
120 assert(!is_at_first_key_from_index_);
121 assert(Valid());
122
123 if (seek_stat_state_ & kReportOnUseful) {
124 bool filter_used = (seek_stat_state_ & kFilterUsed) != 0;
125 RecordTick(
126 table_->GetStatistics(),
127 filter_used
128 ? (is_last_level_ ? LAST_LEVEL_SEEK_DATA_USEFUL_FILTER_MATCH
129 : NON_LAST_LEVEL_SEEK_DATA_USEFUL_FILTER_MATCH)
130 : (is_last_level_ ? LAST_LEVEL_SEEK_DATA_USEFUL_NO_FILTER
131 : NON_LAST_LEVEL_SEEK_DATA_USEFUL_NO_FILTER));
132 seek_stat_state_ = kDataBlockReadSinceLastSeek;
133 }
134
135 return block_iter_.value();
136 }
137 Status status() const override {
138 // In case of block cache readahead lookup, it won't add the block to
139 // block_handles if it's index is invalid. So index_iter_->status check can
140 // be skipped.
141 // Prefix index set status to NotFound when the prefix does not exist.
142 if (IsIndexAtCurr() && !index_iter_->status().ok() &&
143 !index_iter_->status().IsNotFound()) {
144 return index_iter_->status();

146 return block_iter_.status();
147 } else if (async_read_in_progress_) {
148 return Status::TryAgain("Async read in progress");
149 } else {
150 return Status::OK();
151 }

232 enum class IterDirection {
233 kForward,
234 kBackward,
235 };
236 // This enum indicates whether the upper bound falls into current block
237 // or beyond.
238 // +-------------+
239 // | cur block | <-- (1)
240 // +-------------+
241 // <-- (2)
242 // --- <boundary key> ---
243 // <-- (3)
244 // +-------------+
245 // | next block | <-- (4)
246 // ......
247 //
248 // When the block is smaller than <boundary key>, kUpperBoundInCurBlock
249 // is the value to use. The examples are (1) or (2) in the graph. It means
250 // all keys in the next block or beyond will be out of bound. Keys within
251 // the current block may or may not be out of bound.
252 // When the block is larger or equal to <boundary key>,
253 // kUpperBoundBeyondCurBlock is to be used. The examples are (3) and (4)
254 // in the graph. It means that all keys in the current block is within the
255 // upper bound and keys in the next block may or may not be within the uppder
256 // bound.
257 // If the boundary key hasn't been checked against the upper bound,
258 // kUnknown can be used.
259 enum class BlockUpperBound : uint8_t {
260 kUpperBoundInCurBlock,
261 kUpperBoundBeyondCurBlock,
262 kUnknown,
263 };
264
265 // State bits for collecting stats on seeks and whether they returned useful
266 // results.
267 enum SeekStatState : uint8_t {
268 kNone = 0,
269 // Most recent seek checked prefix filter (or similar future feature)
270 kFilterUsed = 1 << 0,
271 // Already recorded that a data block was accessed since the last seek.
272 kDataBlockReadSinceLastSeek = 1 << 1,
273 // Have not yet recorded that a value() was accessed.
274 kReportOnUseful = 1 << 2,
275 };

380 bool CheckPrefixMayMatch(const Slice& ikey, IterDirection direction,
381 bool* filter_checked) {
382 if (need_upper_bound_check_ && direction == IterDirection::kBackward) {
383 // Upper bound check isn't sufficient for backward direction to
384 // guarantee the same result as total order, so disable prefix
385 // check.
386 return true;
387 }
388 if (check_filter_ &&
389 !table_->PrefixRangeMayMatch(ikey, read_options_, prefix_extractor_,
390 need_upper_bound_check_, &lookup_context_,
391 filter_checked)) {
392 // TODO remember the iterator is invalidated because of prefix
393 // match. This can avoid the upper level file iterator to falsely
394 // believe the position is the end of the SST file and move to
395 // the first key of the next file.
396 ResetDataIter();
397 return false;
398 }
399 return true;

416 bool IsNextBlockOutOfReadaheadBound() {
417 const Slice& index_iter_user_key = index_iter_->user_key();
418 // If curr block's index key >= iterate_upper_bound, it means all the keys
419 // in next block or above are out of bound.
420 bool out_of_upper_bound =
421 read_options_.iterate_upper_bound != nullptr &&
422 (user_comparator_.CompareWithoutTimestamp(
423 index_iter_user_key,
424 /*a_has_ts=*/true, *read_options_.iterate_upper_bound,
425 /*b_has_ts=*/false) >= 0
426 ? true
427 : false);
428 if (out_of_upper_bound) {
429 return true;
430 }
431
432 // If curr block's index key has a different prefix from the seek key's, it
433 // means all the keys in next block or above has a different prefix from the
434 // seek key's.
435 bool out_of_prefix_bound =
436 (read_options_.prefix_same_as_start &&
437 !seek_key_prefix_for_readahead_trimming_.empty() &&
438 (prefix_extractor_->InDomain(index_iter_user_key)
439 ? (prefix_extractor_->Transform(index_iter_user_key)
440 .compare(seek_key_prefix_for_readahead_trimming_) != 0)
441 : user_comparator_.CompareWithoutTimestamp(
442 index_iter_user_key,
443 /*a_has_ts=*/true, seek_key_prefix_for_readahead_trimming_,
444 /*b_has_ts=*/false) > 0));
445
446 if (out_of_prefix_bound) {
447 return true;
448 }
449
450 return false;
451 }
