35// Implementation for the merge operation (merges two Cassandra values)
36bool CassandraValueMergeOperator::FullMergeV2(
37 const MergeOperationInput& merge_in,
38 MergeOperationOutput* merge_out) const {
39 // Clear the *new_value for writing.
40 merge_out->new_value.clear();
41 std::vector<RowValue> row_values;
42 if (merge_in.existing_value) {
43 row_values.push_back(RowValue::Deserialize(
44 merge_in.existing_value->data(), merge_in.existing_value->size()));
45 }
46
47 for (auto& operand : merge_in.operand_list) {
48 row_values.push_back(RowValue::Deserialize(operand.data(), operand.size()));
49 }
50
51 RowValue merged = RowValue::Merge(std::move(row_values));
52 merged = merged.RemoveTombstones(options_.gc_grace_period_in_seconds);
53 merge_out->new_value.reserve(merged.Size());
54 merged.Serialize(&(merge_out->new_value));
55
56 return true;
57}

59bool CassandraValueMergeOperator::PartialMergeMulti(
60 const Slice& /*key*/, const std::deque<Slice>& operand_list,
61 std::string* new_value, Logger* /*logger*/) const {
62 // Clear the *new_value for writing.
63 assert(new_value);
64 new_value->clear();
65
66 std::vector<RowValue> row_values;
67 for (auto& operand : operand_list) {
68 row_values.push_back(RowValue::Deserialize(operand.data(), operand.size()));
69 }
70 RowValue merged = RowValue::Merge(std::move(row_values));
71 new_value->reserve(merged.Size());
72 merged.Serialize(new_value);
73 return true;
74}
