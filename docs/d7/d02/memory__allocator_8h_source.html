15// MemoryAllocator is an interface that a client can implement to supply custom
16// memory allocation and deallocation methods. See rocksdb/cache.h for more
17// information.
18// All methods should be thread-safe.
19class MemoryAllocator : public Customizable {
20 public:
21 static const char* Type() { return "MemoryAllocator"; }
22 static Status CreateFromString(const ConfigOptions& options,
23 const std::string& value,
24 std::shared_ptr<MemoryAllocator>* result);
25
26 // Allocate a block of at least size. Has to be thread-safe.
27 virtual void* Allocate(size_t size) = 0;
28
29 // Deallocate previously allocated block. Has to be thread-safe.
30 virtual void Deallocate(void* p) = 0;
31
32 // Returns the memory size of the block allocated at p. The default
33 // implementation that just returns the original allocation_size is fine.
34 virtual size_t UsableSize(void* /*p*/, size_t allocation_size) const {
35 // default implementation just returns the allocation size
36 return allocation_size;
37 }
38
39 std::string GetId() const override { return GenerateIndividualId(); }
40};

42struct JemallocAllocatorOptions {
43 static const char* kName() { return "JemallocAllocatorOptions"; }
44 // Jemalloc tcache cache allocations by size class. For each size class,
45 // it caches between 20 (for large size classes) to 200 (for small size
46 // classes). To reduce tcache memory usage in case the allocator is access
47 // by large number of threads, we can control whether to cache an allocation
48 // by its size.
49 bool limit_tcache_size = false;
50
51 // Lower bound of allocation size to use tcache, if limit_tcache_size=true.
52 // When used with block cache, it is recommended to set it to block_size/4.
53 size_t tcache_size_lower_bound = 1024;
54
55 // Upper bound of allocation size to use tcache, if limit_tcache_size=true.
56 // When used with block cache, it is recommended to set it to block_size.
57 size_t tcache_size_upper_bound = 16 * 1024;
58
59 // Number of arenas across which we spread allocation requests. Increasing
60 // this setting can mitigate arena mutex contention. The value must be
61 // positive.
62 size_t num_arenas = 1;
63};

65// Generate memory allocator which allocates through Jemalloc and utilize
66// MADV_DONTDUMP through madvise to exclude cache items from core dump.
67// Applications can use the allocator with block cache to exclude block cache
68// usage from core dump.
69//
70// Implementation details:
71// The JemallocNodumpAllocator creates a dedicated jemalloc arena, and all
72// allocations of the JemallocNodumpAllocator are through the same arena.
73// The memory allocator hooks memory allocation of the arena, and calls
74// madvise() with MADV_DONTDUMP flag to exclude the piece of memory from
75// core dump. Side benefit of using single arena would be reduction of jemalloc
76// metadata for some workloads.
77//
78// To mitigate mutex contention for using one single arena (see also
79// `JemallocAllocatorOptions::num_arenas` above), jemalloc tcache
80// (thread-local cache) is enabled to cache unused allocations for future use.
81// The tcache normally incurs 0.5M extra memory usage per-thread. The usage
82// can be reduced by limiting allocation sizes to cache.
83Status NewJemallocNodumpAllocator(
84 const JemallocAllocatorOptions& options,
85 std::shared_ptr<MemoryAllocator>* memory_allocator);
