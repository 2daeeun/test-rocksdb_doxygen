27Status CompactionFilter::CreateFromString(const ConfigOptions& config_options,
28 const std::string& value,
29 const CompactionFilter** result) {
30 static std::once_flag once;
31 std::call_once(once, [&]() {
32 RegisterBuiltinCompactionFilters(*(ObjectLibrary::Default().get()), "");
33 });
34 CompactionFilter* filter = const_cast<CompactionFilter*>(*result);
35 Status status =
36 LoadStaticObject<CompactionFilter>(config_options, value, &filter);
37 if (status.ok()) {
38 *result = const_cast<CompactionFilter*>(filter);
39 }
40 return status;
41}
