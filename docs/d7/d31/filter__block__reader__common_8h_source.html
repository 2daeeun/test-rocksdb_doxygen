19// Encapsulates common functionality for the various filter block reader
20// implementations. Provides access to the filter block regardless of whether
21// it is owned by the reader or stored in the cache, or whether it is pinned
22// in the cache or not.
23template <typename TBlocklike>
24class FilterBlockReaderCommon : public FilterBlockReader {
25 public:
26 FilterBlockReaderCommon(const BlockBasedTable* t,
27 CachableEntry<TBlocklike>&& filter_block)
28 : table_(t), filter_block_(std::move(filter_block)) {
29 assert(table_);
30 const SliceTransform* const prefix_extractor = table_prefix_extractor();
31 if (prefix_extractor) {
32 full_length_enabled_ =
33 prefix_extractor->FullLengthEnabled(&prefix_extractor_full_length_);
34 }
35 }
36
37 bool RangeMayExist(const Slice* iterate_upper_bound, const Slice& user_key,
38 const SliceTransform* prefix_extractor,
39 const Comparator* comparator,
40 const Slice* const const_ikey_ptr, bool* filter_checked,
41 bool need_upper_bound_check,
42 BlockCacheLookupContext* lookup_context,
43 const ReadOptions& read_options) override;
44
45 void EraseFromCacheBeforeDestruction(
46 uint32_t /*uncache_aggressiveness*/) override;
47
48 protected:
49 static Status ReadFilterBlock(const BlockBasedTable* table,
50 FilePrefetchBuffer* prefetch_buffer,
51 const ReadOptions& read_options, bool use_cache,
52 GetContext* get_context,
53 BlockCacheLookupContext* lookup_context,
54 CachableEntry<TBlocklike>* filter_block);
55
56 const BlockBasedTable* table() const { return table_; }
57 const SliceTransform* table_prefix_extractor() const;
58 bool whole_key_filtering() const;
59 bool cache_filter_blocks() const;
60

62 BlockCacheLookupContext* lookup_context,
63 CachableEntry<TBlocklike>* filter_block,
64 const ReadOptions& read_options) const;
65
66 size_t ApproximateFilterBlockMemoryUsage() const;
67
68 private:
69 bool IsFilterCompatible(const Slice* iterate_upper_bound, const Slice& prefix,
70 const Comparator* comparator) const;
71
72 private:
73 const BlockBasedTable* table_;
74 CachableEntry<TBlocklike> filter_block_;
75 size_t prefix_extractor_full_length_ = 0;
76 bool full_length_enabled_ = false;
77};
