145 : uint32_t {
146 // Backup blob filenames are <file_number>_<crc32c>_<file_size>.blob and
147 // backup SST filenames are <file_number>_<crc32c>_<file_size>.sst
148 // where <crc32c> is an unsigned decimal integer. This is the
149 // original/legacy naming scheme for share_files_with_checksum,
150 // with two problems:
151 // * At massive scale, collisions on this triple with different file
152 // contents is plausible.
153 // * Determining the name to use requires computing the checksum,
154 // so generally requires reading the whole file even if the file
155 // is already backed up.
156 //
157 // ** ONLY RECOMMENDED FOR PRESERVING OLD BEHAVIOR **
158 kLegacyCrc32cAndFileSize = 1U,
159
160 // Backup SST filenames are <file_number>_s<db_session_id>.sst. This
161 // pair of values should be very strongly unique for a given SST file
162 // and easily determined before computing a checksum. The 's' indicates
163 // the value is a DB session id, not a checksum.
164 //
165 // Exceptions:
166 // * For blob files, kLegacyCrc32cAndFileSize is used as currently
167 // db_session_id is not supported by the blob file format.
168 // * For old SST files without a DB session id, kLegacyCrc32cAndFileSize
169 // will be used instead, matching the names assigned by RocksDB versions
170 // not supporting the newer naming scheme.
171 // * See also flags below.
172 kUseDbSessionId = 2U,
173
174 kMaskNoNamingFlags = 0xffffU,
175
176 // If not already part of the naming scheme, insert
177 // _<file_size>
178 // before .sst and .blob in the name. In case of user code actually parsing
179 // the last _<whatever> before the .sst and .blob as the file size, this
180 // preserves that feature of kLegacyCrc32cAndFileSize. In other words, this
181 // option makes official that unofficial feature of the backup metadata.
182 //
183 // We do not consider SST and blob file sizes to have sufficient entropy to
184 // contribute significantly to naming uniqueness.
185 kFlagIncludeFileSize = 1U << 31,
186
187 kMaskNamingFlags = ~kMaskNoNamingFlags,
188 };
