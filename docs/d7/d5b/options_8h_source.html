248 // If non-nullptr, use the specified function to put keys in contiguous
249 // groups called "prefixes". These prefixes are used to place one
250 // representative entry for the group into the Bloom filter
251 // rather than an entry for each key (see whole_key_filtering).
252 // Under certain conditions, this enables optimizing some range queries
253 // (Iterators) in addition to some point lookups (Get/MultiGet).
254 //
255 // Together `prefix_extractor` and `comparator` must satisfy one essential
256 // property for valid prefix filtering of range queries:
257 // If Compare(k1, k2) <= 0 and Compare(k2, k3) <= 0 and
258 // InDomain(k1) and InDomain(k3) and prefix(k1) == prefix(k3),
259 // Then InDomain(k2) and prefix(k2) == prefix(k1)
260 //
261 // In other words, all keys with the same prefix must be in a contiguous
262 // group by comparator order, and cannot be interrupted by keys with no
263 // prefix ("out of domain"). (This makes it valid to conclude that no
264 // entries within some bounds are present if the upper and lower bounds
265 // have a common prefix and no entries with that same prefix are present.)
266 //
267 // Some other properties are recommended but not strictly required. Under
268 // most sensible comparators, the following will need to hold true to
269 // satisfy the essential property above:
270 // * "Prefix is a prefix": key.starts_with(prefix(key))
271 // * "Prefixes preserve ordering": If Compare(k1, k2) <= 0, then
272 // Compare(prefix(k1), prefix(k2)) <= 0
273 //
274 // The next two properties ensure that seeking to a prefix allows
275 // enumerating all entries with that prefix:
276 // * "Prefix starts the group": Compare(prefix(key), key) <= 0
277 // * "Prefix idempotent": prefix(prefix(key)) == prefix(key)
278 //
279 // Default: nullptr

353 // EXPERIMENTAL
354 // When > 0, RocksDB attempts to erase some block cache entries for files
355 // that have become obsolete, which means they are about to be deleted.
356 // To avoid excessive tracking, this "uncaching" process is iterative and
357 // speculative, meaning it could incur extra background CPU effort if the
358 // file's blocks are generally not cached. A larger number indicates more
359 // willingness to spend CPU time to maximize block cache hit rates by
360 // erasing known-obsolete entries.
361 //
362 // When uncache_aggressiveness=1, block cache entries for an obsolete file
363 // are only erased until any attempted erase operation fails because the
364 // block is not cached. Then no further attempts are made to erase cached
365 // blocks for that file.
366 //
367 // For larger values, erasure is attempted until evidence incidates that the
368 // chance of success is < 0.99^(a-1), where a = uncache_aggressiveness. For
369 // example:
370 // 2 -> Attempt only while expecting >= 99% successful/useful erasure
371 // 11 -> 90%
372 // 69 -> 50%
373 // 110 -> 33%
374 // 230 -> 10%
375 // 460 -> 1%
376 // 690 -> 0.1%
377 // 1000 -> 1 in 23000
378 // 10000 -> Always (for all practical purposes)
379 // NOTE: UINT32_MAX and nearby values could take additional special meanings
380 // in the future.
381 //
382 // Pinned cache entries (guaranteed present) are always erased if
383 // uncache_aggressiveness > 0, but are not used in predicting the chances of
384 // successful erasure of non-pinned entries.
385 //
386 // NOTE: In the case of copied DBs (such as Checkpoints) sharing a block
387 // cache, it is possible that a file becoming obsolete doesn't mean its
388 // block cache entries (shared among copies) are obsolete. Such a scenerio
389 // is the best case for uncache_aggressiveness = 0.
390 //
391 // When using allow_mmap_reads=true, this option is ignored (no un-caching).
392 //
393 // Once validated in production, the default will likely change to something
394 // around 300.
395 uint32_t uncache_aggressiveness = 0;

405enum class WALRecoveryMode : char {
406 // Original levelDB recovery
407 //
408 // We tolerate the last record in any log to be incomplete due to a crash
409 // while writing it. Zeroed bytes from preallocation are also tolerated in the
410 // trailing data of any log.
411 //
412 // Use case: Applications for which updates, once applied, must not be rolled
413 // back even after a crash-recovery. In this recovery mode, RocksDB guarantees
414 // this as long as `WritableFile::Append()` writes are durable. In case the
415 // user needs the guarantee in more situations (e.g., when
416 // `WritableFile::Append()` writes to page cache, but the user desires this
417 // guarantee in face of power-loss crash-recovery), RocksDB offers various
418 // mechanisms to additionally invoke `WritableFile::Sync()` in order to
419 // strengthen the guarantee.
420 //
421 // This differs from `kPointInTimeRecovery` in that, in case a corruption is
422 // detected during recovery, this mode will refuse to open the DB. Whereas,
423 // `kPointInTimeRecovery` will stop recovery just before the corruption since
424 // that is a valid point-in-time to which to recover.
425 kTolerateCorruptedTailRecords = 0x00,
426 // Recover from clean shutdown
427 // We don't expect to find any corruption in the WAL
428 // Use case : This is ideal for unit tests and rare applications that
429 // can require high consistency guarantee
430 kAbsoluteConsistency = 0x01,
431 // Recover to point-in-time consistency (default)
432 // We stop the WAL playback on discovering WAL inconsistency
433 // Use case : Ideal for systems that have disk controller cache like
434 // hard disk, SSD without super capacitor that store related data
435 kPointInTimeRecovery = 0x02,
436 // Recovery after a disaster
437 // We ignore any corruption in the WAL and try to salvage as much data as
438 // possible
439 // Use case : Ideal for last ditch effort to recover data or systems that
440 // operate with low grade unrelated data
441 kSkipAnyCorruptedRecords = 0x03,
442};
