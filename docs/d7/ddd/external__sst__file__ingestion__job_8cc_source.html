28Status ExternalSstFileIngestionJob::Prepare(
29 const std::vector<std::string>& external_files_paths,
30 const std::vector<std::string>& files_checksums,
31 const std::vector<std::string>& files_checksum_func_names,
32 const Temperature& file_temperature, uint64_t next_file_number,
33 SuperVersion* sv) {
34 Status status;
35
36 // Read the information of files we are ingesting
37 for (const std::string& file_path : external_files_paths) {
38 IngestedFileInfo file_to_ingest;
39 // For temperature, first assume it matches provided hint
40 file_to_ingest.file_temperature = file_temperature;
41 status =
42 GetIngestedFileInfo(file_path, next_file_number++, &file_to_ingest, sv);
43 if (!status.ok()) {
44 return status;
45 }
46
47 // Files generated in another DB or CF may have a different column family
48 // ID, so we let it pass here.
49 if (file_to_ingest.cf_id !=
50 TablePropertiesCollectorFactory::Context::kUnknownColumnFamily &&
51 file_to_ingest.cf_id != cfd_->GetID() &&
52 !ingestion_options_.allow_db_generated_files) {
53 return Status::InvalidArgument(
54 "External file column family id don't match");
55 }
56
57 if (file_to_ingest.num_entries == 0 &&
58 file_to_ingest.num_range_deletions == 0) {
59 return Status::InvalidArgument("File contain no entries");
60 }
61
62 if (!file_to_ingest.smallest_internal_key.Valid() ||
63 !file_to_ingest.largest_internal_key.Valid()) {
64 return Status::Corruption("Generated table have corrupted keys");
65 }
66
67 files_to_ingest_.emplace_back(std::move(file_to_ingest));
68 }
69
70 auto num_files = files_to_ingest_.size();
71 if (num_files == 0) {
72 return Status::InvalidArgument("The list of files is empty");
73 } else if (num_files > 1) {
74 // Verify that passed files don't have overlapping ranges
75 autovector<const IngestedFileInfo*> sorted_files;
76 for (size_t i = 0; i < num_files; i++) {
77 sorted_files.push_back(&files_to_ingest_[i]);
78 }
79
80 std::sort(sorted_files.begin(), sorted_files.end(), file_range_checker_);
81
82 for (size_t i = 0; i + 1 < num_files; i++) {
83 if (file_range_checker_.OverlapsWithPrev(sorted_files[i],
84 sorted_files[i + 1],
85 /* ranges_sorted= */ true)) {
86 files_overlap_ = true;
87 break;
88 }
89 }
90 }
91
92 if (ingestion_options_.ingest_behind && files_overlap_) {
93 return Status::NotSupported(
94 "Files with overlapping ranges cannot be ingested with ingestion "
95 "behind mode.");
96 }
97
98 // Overlapping files need at least two different sequence numbers. If settings
99 // disables global seqno, ingestion will fail anyway, so fail fast in prepare.
100 if (!ingestion_options_.allow_global_seqno && files_overlap_) {
101 return Status::InvalidArgument(
102 "Global seqno is required, but disabled (because external files key "
103 "range overlaps).");
104 }
105
106 if (ucmp_->timestamp_size() > 0 && files_overlap_) {
107 return Status::NotSupported(
108 "Files with overlapping ranges cannot be ingested to column "
109 "family with user-defined timestamp enabled.");
110 }
111
112 // Copy/Move external files into DB
113 std::unordered_set<size_t> ingestion_path_ids;
114 for (IngestedFileInfo& f : files_to_ingest_) {
115 f.copy_file = false;
116 const std::string path_outside_db = f.external_file_path;
117 const std::string path_inside_db = TableFileName(
118 cfd_->ioptions()->cf_paths, f.fd.GetNumber(), f.fd.GetPathId());
119 if (ingestion_options_.move_files || ingestion_options_.link_files) {
120 status =
121 fs_->LinkFile(path_outside_db, path_inside_db, IOOptions(), nullptr);
122 if (status.ok()) {
123 // It is unsafe to assume application had sync the file and file
124 // directory before ingest the file. For integrity of RocksDB we need
125 // to sync the file.
126 std::unique_ptr<FSWritableFile> file_to_sync;
127 Status s = fs_->ReopenWritableFile(path_inside_db, env_options_,
128 &file_to_sync, nullptr);
129 TEST_SYNC_POINT_CALLBACK("ExternalSstFileIngestionJob::Prepare:Reopen",
130 &s);
131 // Some file systems (especially remote/distributed) don't support
132 // reopening a file for writing and don't require reopening and
133 // syncing the file. Ignore the NotSupported error in that case.
134 if (!s.IsNotSupported()) {
135 status = s;
136 if (status.ok()) {
137 TEST_SYNC_POINT(
138 "ExternalSstFileIngestionJob::BeforeSyncIngestedFile");
139 status = SyncIngestedFile(file_to_sync.get());
140 TEST_SYNC_POINT(
141 "ExternalSstFileIngestionJob::AfterSyncIngestedFile");
142 if (!status.ok()) {
143 ROCKS_LOG_WARN(db_options_.info_log,
144 "Failed to sync ingested file %s: %s",
145 path_inside_db.c_str(), status.ToString().c_str());
146 }
147 }
148 }
149 } else if (status.IsNotSupported() &&
150 ingestion_options_.failed_move_fall_back_to_copy) {
151 // Original file is on a different FS, use copy instead of hard linking.
152 f.copy_file = true;
153 ROCKS_LOG_INFO(db_options_.info_log,
154 "Tried to link file %s but it's not supported : %s",
155 path_outside_db.c_str(), status.ToString().c_str());
156 }
157 } else {
158 f.copy_file = true;
159 }
160
161 if (f.copy_file) {
162 TEST_SYNC_POINT_CALLBACK("ExternalSstFileIngestionJob::Prepare:CopyFile",
163 nullptr);
164 // Always determining the destination temperature from the ingested-to
165 // level would be difficult because in general we only find out the level
166 // ingested to later, during Run().
167 // However, we can guarantee "last level" temperature for when the user
168 // requires ingestion to the last level.
169 Temperature dst_temp =
170 (ingestion_options_.ingest_behind ||
171 ingestion_options_.fail_if_not_bottommost_level)
172 ? sv->mutable_cf_options.last_level_temperature
173 : sv->mutable_cf_options.default_write_temperature;
174 // Note: CopyFile also syncs the new file.
175 status = CopyFile(fs_.get(), path_outside_db, f.file_temperature,
176 path_inside_db, dst_temp, 0, db_options_.use_fsync,
177 io_tracer_);
178 // The destination of the copy will be ingested
179 f.file_temperature = dst_temp;
180 } else {
181 // Note: we currently assume that linking files does not cross
182 // temperatures, so no need to change f.file_temperature
183 }
184 TEST_SYNC_POINT("ExternalSstFileIngestionJob::Prepare:FileAdded");
185 if (!status.ok()) {
186 break;
187 }
188 f.internal_file_path = path_inside_db;
189 // Initialize the checksum information of ingested files.
190 f.file_checksum = kUnknownFileChecksum;
191 f.file_checksum_func_name = kUnknownFileChecksumFuncName;
192 ingestion_path_ids.insert(f.fd.GetPathId());
193 }
194
195 TEST_SYNC_POINT("ExternalSstFileIngestionJob::BeforeSyncDir");
196 if (status.ok()) {
197 for (auto path_id : ingestion_path_ids) {
198 status = directories_->GetDataDir(path_id)->FsyncWithDirOptions(
199 IOOptions(), nullptr,
200 DirFsyncOptions(DirFsyncOptions::FsyncReason::kNewFileSynced));
201 if (!status.ok()) {
202 ROCKS_LOG_WARN(db_options_.info_log,
203 "Failed to sync directory %" ROCKSDB_PRIszt
204 " while ingest file: %s",
205 path_id, status.ToString().c_str());
206 break;
207 }
208 }
209 }
210 TEST_SYNC_POINT("ExternalSstFileIngestionJob::AfterSyncDir");
211
212 // Generate and check the sst file checksum. Note that, if
213 // IngestExternalFileOptions::write_global_seqno is true, we will not update
214 // the checksum information in the files_to_ingests_ here, since the file is
215 // updated with the new global_seqno. After global_seqno is updated, DB will
216 // generate the new checksum and store it in the Manifest. In all other cases
217 // if ingestion_options_.write_global_seqno == true and
218 // verify_file_checksum is false, we only check the checksum function name.
219 if (status.ok() && db_options_.file_checksum_gen_factory != nullptr) {
220 if (ingestion_options_.verify_file_checksum == false &&
221 files_checksums.size() == files_to_ingest_.size() &&
222 files_checksum_func_names.size() == files_to_ingest_.size()) {
223 // Only when verify_file_checksum == false and the checksum for ingested
224 // files are provided, DB will use the provided checksum and does not
225 // generate the checksum for ingested files.
226 need_generate_file_checksum_ = false;
227 } else {
228 need_generate_file_checksum_ = true;
229 }
230 FileChecksumGenContext gen_context;
231 std::unique_ptr<FileChecksumGenerator> file_checksum_gen =
232 db_options_.file_checksum_gen_factory->CreateFileChecksumGenerator(
233 gen_context);
234 std::vector<std::string> generated_checksums;
235 std::vector<std::string> generated_checksum_func_names;
236 // Step 1: generate the checksum for ingested sst file.
237 if (need_generate_file_checksum_) {
238 for (size_t i = 0; i < files_to_ingest_.size(); i++) {
239 std::string generated_checksum;
240 std::string generated_checksum_func_name;
241 std::string requested_checksum_func_name;
242 // TODO: rate limit file reads for checksum calculation during file
243 // ingestion.
244 // TODO: plumb Env::IOActivity
245 ReadOptions ro;
246 IOStatus io_s = GenerateOneFileChecksum(
247 fs_.get(), files_to_ingest_[i].internal_file_path,
248 db_options_.file_checksum_gen_factory.get(),
249 requested_checksum_func_name, &generated_checksum,
250 &generated_checksum_func_name,
251 ingestion_options_.verify_checksums_readahead_size,
252 db_options_.allow_mmap_reads, io_tracer_,
253 db_options_.rate_limiter.get(), ro, db_options_.stats,
254 db_options_.clock);
255 if (!io_s.ok()) {
256 status = io_s;
257 ROCKS_LOG_WARN(db_options_.info_log,
258 "Sst file checksum generation of file: %s failed: %s",
259 files_to_ingest_[i].internal_file_path.c_str(),
260 status.ToString().c_str());
261 break;
262 }
263 if (ingestion_options_.write_global_seqno == false) {
264 files_to_ingest_[i].file_checksum = generated_checksum;
265 files_to_ingest_[i].file_checksum_func_name =
266 generated_checksum_func_name;
267 }
268 generated_checksums.push_back(generated_checksum);
269 generated_checksum_func_names.push_back(generated_checksum_func_name);
270 }
271 }
272
273 // Step 2: based on the verify_file_checksum and ingested checksum
274 // information, do the verification.
275 if (status.ok()) {
276 if (files_checksums.size() == files_to_ingest_.size() &&
277 files_checksum_func_names.size() == files_to_ingest_.size()) {
278 // Verify the checksum and checksum function name.
279 if (ingestion_options_.verify_file_checksum) {
280 for (size_t i = 0; i < files_to_ingest_.size(); i++) {
281 if (files_checksum_func_names[i] !=
282 generated_checksum_func_names[i]) {
283 status = Status::InvalidArgument(
284 "Checksum function name does not match with the checksum "
285 "function name of this DB");
286 ROCKS_LOG_WARN(
287 db_options_.info_log,
288 "Sst file checksum verification of file: %s failed: %s",
289 external_files_paths[i].c_str(), status.ToString().c_str());
290 break;
291 }
292 if (files_checksums[i] != generated_checksums[i]) {
293 status = Status::Corruption(
294 "Ingested checksum does not match with the generated "
295 "checksum");
296 ROCKS_LOG_WARN(
297 db_options_.info_log,
298 "Sst file checksum verification of file: %s failed: %s",
299 files_to_ingest_[i].internal_file_path.c_str(),
300 status.ToString().c_str());
301 break;
302 }
303 }
304 } else {
305 // If verify_file_checksum is not enabled, we only verify the
306 // checksum function name. If it does not match, fail the ingestion.
307 // If matches, we trust the ingested checksum information and store
308 // in the Manifest.
309 for (size_t i = 0; i < files_to_ingest_.size(); i++) {
310 if (files_checksum_func_names[i] != file_checksum_gen->Name()) {
311 status = Status::InvalidArgument(
312 "Checksum function name does not match with the checksum "
313 "function name of this DB");
314 ROCKS_LOG_WARN(
315 db_options_.info_log,
316 "Sst file checksum verification of file: %s failed: %s",
317 external_files_paths[i].c_str(), status.ToString().c_str());
318 break;
319 }
320 files_to_ingest_[i].file_checksum = files_checksums[i];
321 files_to_ingest_[i].file_checksum_func_name =
322 files_checksum_func_names[i];
323 }
324 }
325 } else if (files_checksums.size() != files_checksum_func_names.size() ||
326 files_checksums.size() != 0) {
327 // The checksum or checksum function name vector are not both empty
328 // and they are incomplete.
329 status = Status::InvalidArgument(
330 "The checksum information of ingested sst files are nonempty and "
331 "the size of checksums or the size of the checksum function "
332 "names "
333 "does not match with the number of ingested sst files");
334 ROCKS_LOG_WARN(
335 db_options_.info_log,
336 "The ingested sst files checksum information is incomplete: %s",
337 status.ToString().c_str());
338 }
339 }
340 }
341
342 if (status.ok()) {
343 DivideInputFilesIntoBatches();
344 }
345
346 return status;
347}

349void ExternalSstFileIngestionJob::DivideInputFilesIntoBatches() {
350 if (!files_overlap_) {
351 // No overlap, treat as one batch without the need of tracking overall batch
352 // range.
353 file_batches_to_ingest_.emplace_back(/* _track_batch_range= */ false);
354 for (auto& file : files_to_ingest_) {
355 file_batches_to_ingest_.back().AddFile(&file, file_range_checker_);
356 }
357 return;
358 }
359
360 file_batches_to_ingest_.emplace_back(/* _track_batch_range= */ true);
361 for (auto& file : files_to_ingest_) {
362 if (file_range_checker_.OverlapsWithPrev(&file_batches_to_ingest_.back(),
363 &file,
364 /* ranges_sorted= */ false)) {
365 file_batches_to_ingest_.emplace_back(/* _track_batch_range= */ true);
366 }
367 file_batches_to_ingest_.back().AddFile(&file, file_range_checker_);
368 }

371Status ExternalSstFileIngestionJob::NeedsFlush(bool* flush_needed,
372 SuperVersion* super_version) {
373 size_t n = files_to_ingest_.size();
374 autovector<UserKeyRange> ranges;
375 ranges.reserve(n);
376 for (const IngestedFileInfo& file_to_ingest : files_to_ingest_) {
377 ranges.emplace_back(file_to_ingest.start_ukey, file_to_ingest.limit_ukey);
378 }
379 Status status = cfd_->RangesOverlapWithMemtables(
380 ranges, super_version, db_options_.allow_data_in_errors, flush_needed);
381 if (status.ok() && *flush_needed) {
382 if (!ingestion_options_.allow_blocking_flush) {
383 status = Status::InvalidArgument("External file requires flush");
384 }
385 if (ucmp_->timestamp_size() > 0) {
386 status = Status::InvalidArgument(
387 "Column family enables user-defined timestamps, please make "
388 "sure the key range (without timestamp) of external file does not "
389 "overlap with key range in the memtables.");
390 }
391 }
392 return status;
393}

395// REQUIRES: we have become the only writer by entering both write_thread_ and
396// nonmem_write_thread_
397Status ExternalSstFileIngestionJob::Run() {
398 SuperVersion* super_version = cfd_->GetSuperVersion();
399 // If column family is flushed after Prepare and before Run, we should have a
400 // specific state of Memtables. The mutable Memtable should be empty, and the
401 // immutable Memtable list should be empty.
402 if (flushed_before_run_ && (super_version->imm->NumNotFlushed() != 0 ||
403 !super_version->mem->IsEmpty())) {
404 return Status::TryAgain(
405 "Inconsistent memtable state detected when flushed before run.");
406 }
407 Status status;
408#ifndef NDEBUG
409 // We should never run the job with a memtable that is overlapping
410 // with the files we are ingesting
411 bool need_flush = false;
412 status = NeedsFlush(&need_flush, super_version);
413 if (!status.ok()) {
414 return status;
415 }
416 if (need_flush) {
417 return Status::TryAgain("need_flush");
418 }
419 assert(status.ok() && need_flush == false);
420#endif
421
422 bool force_global_seqno = false;
423
424 if (ingestion_options_.snapshot_consistency && !db_snapshots_->empty()) {
425 // We need to assign a global sequence number to all the files even
426 // if the don't overlap with any ranges since we have snapshots
427 force_global_seqno = true;
428 }
429 // It is safe to use this instead of LastAllocatedSequence since we are
430 // the only active writer, and hence they are equal
431 SequenceNumber last_seqno = versions_->LastSequence();
432 edit_.SetColumnFamily(cfd_->GetID());
433 // The levels that the files will be ingested into
434
435 std::optional<int> prev_batch_uppermost_level;
436 for (auto& batch : file_batches_to_ingest_) {
437 int batch_uppermost_level = 0;
438 status = AssignLevelsForOneBatch(batch, super_version, force_global_seqno,
439 &last_seqno, &batch_uppermost_level,
440 prev_batch_uppermost_level);
441 if (!status.ok()) {
442 return status;
443 }
444
445 prev_batch_uppermost_level = batch_uppermost_level;
446 }
447
448 CreateEquivalentFileIngestingCompactions();
449 return status;
450}

452Status ExternalSstFileIngestionJob::AssignLevelsForOneBatch(
453 FileBatchInfo& batch, SuperVersion* super_version, bool force_global_seqno,
454 SequenceNumber* last_seqno, int* batch_uppermost_level,
455 std::optional<int> prev_batch_uppermost_level) {
456 Status status;
457 assert(batch_uppermost_level);
458 *batch_uppermost_level = std::numeric_limits<int>::max();
459 for (IngestedFileInfo* file : batch.files) {
460 assert(file);
461 SequenceNumber assigned_seqno = 0;
462 if (ingestion_options_.ingest_behind) {
463 status = CheckLevelForIngestedBehindFile(file);
464 } else {
465 status = AssignLevelAndSeqnoForIngestedFile(
466 super_version, force_global_seqno, cfd_->ioptions()->compaction_style,
467 *last_seqno, file, &assigned_seqno, prev_batch_uppermost_level);
468 }
469
470 // Modify the smallest/largest internal key to include the sequence number
471 // that we just learned. Only overwrite sequence number zero. There could
472 // be a nonzero sequence number already to indicate a range tombstone's
473 // exclusive endpoint.
474 ParsedInternalKey smallest_parsed, largest_parsed;
475 if (status.ok()) {
476 status = ParseInternalKey(*(file->smallest_internal_key.rep()),
477 &smallest_parsed, false /* log_err_key */);
478 }
479 if (status.ok()) {
480 status = ParseInternalKey(*(file->largest_internal_key.rep()),
481 &largest_parsed, false /* log_err_key */);
482 }
483 if (!status.ok()) {
484 return status;
485 }
486 if (smallest_parsed.sequence == 0 && assigned_seqno != 0) {
487 UpdateInternalKey(file->smallest_internal_key.rep(), assigned_seqno,
488 smallest_parsed.type);
489 }
490 if (largest_parsed.sequence == 0 && assigned_seqno != 0) {
491 UpdateInternalKey(file->largest_internal_key.rep(), assigned_seqno,
492 largest_parsed.type);
493 }
494
495 status = AssignGlobalSeqnoForIngestedFile(file, assigned_seqno);
496 if (!status.ok()) {
497 return status;
498 }
499 TEST_SYNC_POINT_CALLBACK("ExternalSstFileIngestionJob::Run",
500 &assigned_seqno);
501 assert(assigned_seqno == 0 || assigned_seqno == *last_seqno + 1);
502 if (assigned_seqno > *last_seqno) {
503 *last_seqno = assigned_seqno;
504 ++consumed_seqno_count_;
505 }
506
507 status = GenerateChecksumForIngestedFile(file);
508 if (!status.ok()) {
509 return status;
510 }
511
512 // We use the import time as the ancester time. This is the time the data
513 // is written to the database.
514 int64_t temp_current_time = 0;
515 uint64_t current_time = kUnknownFileCreationTime;
516 uint64_t oldest_ancester_time = kUnknownOldestAncesterTime;
517 if (clock_->GetCurrentTime(&temp_current_time).ok()) {
518 current_time = oldest_ancester_time =
519 static_cast<uint64_t>(temp_current_time);
520 }
521 uint64_t tail_size = 0;
522 bool contain_no_data_blocks = file->table_properties.num_entries > 0 &&
523 (file->table_properties.num_entries ==
524 file->table_properties.num_range_deletions);
525 if (file->table_properties.tail_start_offset > 0 ||
526 contain_no_data_blocks) {
527 uint64_t file_size = file->fd.GetFileSize();
528 assert(file->table_properties.tail_start_offset <= file_size);
529 tail_size = file_size - file->table_properties.tail_start_offset;
530 }
531
532 bool marked_for_compaction =
533 file->table_properties.num_range_deletions == 1 &&
534 (file->table_properties.num_entries ==
535 file->table_properties.num_range_deletions);
536 FileMetaData f_metadata(
537 file->fd.GetNumber(), file->fd.GetPathId(), file->fd.GetFileSize(),
538 file->smallest_internal_key, file->largest_internal_key,
539 file->assigned_seqno, file->assigned_seqno, false,
540 file->file_temperature, kInvalidBlobFileNumber, oldest_ancester_time,
541 current_time,
542 ingestion_options_.ingest_behind
543 ? kReservedEpochNumberForFileIngestedBehind
544 : cfd_->NewEpochNumber(),
545 file->file_checksum, file->file_checksum_func_name, file->unique_id, 0,
546 tail_size, file->user_defined_timestamps_persisted);
547 f_metadata.temperature = file->file_temperature;
548 f_metadata.marked_for_compaction = marked_for_compaction;
549 edit_.AddFile(file->picked_level, f_metadata);
550
551 *batch_uppermost_level =
552 std::min(*batch_uppermost_level, file->picked_level);
553 }
554
555 return Status::OK();
556}

558void ExternalSstFileIngestionJob::CreateEquivalentFileIngestingCompactions() {
559 // A map from output level to input of compactions equivalent to this
560 // ingestion job.
561 // TODO: simplify below logic to creating compaction per ingested file
562 // instead of per output level, once we figure out how to treat ingested files
563 // with adjacent range deletion tombstones to same output level in the same
564 // job as non-overlapping compactions.
565 std::map<int, CompactionInputFiles>
566 output_level_to_file_ingesting_compaction_input;
567
568 for (const auto& pair : edit_.GetNewFiles()) {
569 int output_level = pair.first;
570 const FileMetaData& f_metadata = pair.second;
571
572 CompactionInputFiles& input =
573 output_level_to_file_ingesting_compaction_input[output_level];
574 if (input.files.empty()) {
575 // Treat the source level of ingested files to be level 0
576 input.level = 0;
577 }
578
579 compaction_input_metdatas_.push_back(new FileMetaData(f_metadata));
580 input.files.push_back(compaction_input_metdatas_.back());
581 }
582
583 for (const auto& pair : output_level_to_file_ingesting_compaction_input) {
584 int output_level = pair.first;
585 const CompactionInputFiles& input = pair.second;
586
587 const auto& mutable_cf_options = *(cfd_->GetLatestMutableCFOptions());
588 file_ingesting_compactions_.push_back(new Compaction(
589 cfd_->current()->storage_info(), *cfd_->ioptions(), mutable_cf_options,
590 mutable_db_options_, {input}, output_level,
591 /* output file size limit not applicable */
592 MaxFileSizeForLevel(mutable_cf_options, output_level,
593 cfd_->ioptions()->compaction_style),
594 LLONG_MAX /* max compaction bytes, not applicable */,
595 0 /* output path ID, not applicable */, mutable_cf_options.compression,
596 mutable_cf_options.compression_opts,

769Status ExternalSstFileIngestionJob::SanityCheckTableProperties(
770 const std::string& external_file, uint64_t new_file_number,
771 SuperVersion* sv, IngestedFileInfo* file_to_ingest,
772 std::unique_ptr<TableReader>* table_reader) {
773 // Get the external file properties
774 auto props = table_reader->get()->GetTableProperties();
775 assert(props.get());
776 const auto& uprops = props->user_collected_properties;
777
778 // Get table version
779 auto version_iter = uprops.find(ExternalSstFilePropertyNames::kVersion);
780 if (version_iter == uprops.end()) {
781 assert(!SstFileWriter::CreatedBySstFileWriter(*props));
782 if (!ingestion_options_.allow_db_generated_files) {
783 return Status::Corruption("External file version not found");
784 } else {
785 // 0 is special version for when a file from live DB does not have the
786 // version table property
787 file_to_ingest->version = 0;
788 }
789 } else {
790 assert(SstFileWriter::CreatedBySstFileWriter(*props));
791 file_to_ingest->version = DecodeFixed32(version_iter->second.c_str());
792 }
793
794 auto seqno_iter = uprops.find(ExternalSstFilePropertyNames::kGlobalSeqno);
795 if (file_to_ingest->version == 2) {
796 // version 2 imply that we have global sequence number
797 if (seqno_iter == uprops.end()) {
798 return Status::Corruption(
799 "External file global sequence number not found");
800 }
801
802 // Set the global sequence number
803 file_to_ingest->original_seqno = DecodeFixed64(seqno_iter->second.c_str());
804 if (props->external_sst_file_global_seqno_offset == 0) {
805 file_to_ingest->global_seqno_offset = 0;
806 return Status::Corruption("Was not able to find file global seqno field");
807 }
808 file_to_ingest->global_seqno_offset =
809 static_cast<size_t>(props->external_sst_file_global_seqno_offset);
810 } else if (file_to_ingest->version == 1) {
811 // SST file V1 should not have global seqno field
812 assert(seqno_iter == uprops.end());
813 file_to_ingest->original_seqno = 0;
814 if (ingestion_options_.allow_blocking_flush ||
815 ingestion_options_.allow_global_seqno) {
816 return Status::InvalidArgument(
817 "External SST file V1 does not support global seqno");
818 }
819 } else if (file_to_ingest->version == 0) {
820 // allow_db_generated_files is true
821 assert(seqno_iter == uprops.end());
822 file_to_ingest->original_seqno = 0;
823 file_to_ingest->global_seqno_offset = 0;
824 } else {
825 return Status::InvalidArgument("External file version " +
826 std::to_string(file_to_ingest->version) +
827 " is not supported");
828 }
829
830 file_to_ingest->cf_id = static_cast<uint32_t>(props->column_family_id);
831 // This assignment works fine even though `table_reader` may later be reset,
832 // since that will not affect how table properties are parsed, and this
833 // assignment is making a copy.
834 file_to_ingest->table_properties = *props;
835
836 // Get number of entries in table
837 file_to_ingest->num_entries = props->num_entries;
838 file_to_ingest->num_range_deletions = props->num_range_deletions;
839
840 // Validate table properties related to comparator name and user defined
841 // timestamps persisted flag.
842 file_to_ingest->user_defined_timestamps_persisted =
843 static_cast<bool>(props->user_defined_timestamps_persisted);
844 bool mark_sst_file_has_no_udt = false;
845 Status s = ValidateUserDefinedTimestampsOptions(
846 cfd_->user_comparator(), props->comparator_name,
847 cfd_->ioptions()->persist_user_defined_timestamps,
848 file_to_ingest->user_defined_timestamps_persisted,
849 &mark_sst_file_has_no_udt);
850 if (s.ok() && mark_sst_file_has_no_udt) {
851 // A column family that enables user-defined timestamps in Memtable only
852 // feature can also ingest external files created by a setting that disables
853 // user-defined timestamps. In that case, we need to re-mark the
854 // user_defined_timestamps_persisted flag for the file.
855 file_to_ingest->user_defined_timestamps_persisted = false;
856 } else if (!s.ok()) {
857 return s;
858 }
859
860 // `TableReader` is initialized with `user_defined_timestamps_persisted` flag
861 // to be true. If its value changed to false after this sanity check, we
862 // need to reset the `TableReader`.
863 if (ucmp_->timestamp_size() > 0 &&
864 !file_to_ingest->user_defined_timestamps_persisted) {
865 s = ResetTableReader(external_file, new_file_number,
866 file_to_ingest->user_defined_timestamps_persisted, sv,
867 file_to_ingest, table_reader);
868 }
869 return s;
870}

872Status ExternalSstFileIngestionJob::GetIngestedFileInfo(
873 const std::string& external_file, uint64_t new_file_number,
874 IngestedFileInfo* file_to_ingest, SuperVersion* sv) {
875 file_to_ingest->external_file_path = external_file;
876
877 // Get external file size
878 Status status = fs_->GetFileSize(external_file, IOOptions(),
879 &file_to_ingest->file_size, nullptr);
880 if (!status.ok()) {
881 return status;
882 }
883
884 // Assign FD with number
885 file_to_ingest->fd =
886 FileDescriptor(new_file_number, 0, file_to_ingest->file_size);
887
888 // Create TableReader for external file
889 std::unique_ptr<TableReader> table_reader;
890 // Initially create the `TableReader` with flag
891 // `user_defined_timestamps_persisted` to be true since that's the most common
892 // case
893 status = ResetTableReader(external_file, new_file_number,
894 /*user_defined_timestamps_persisted=*/true, sv,
895 file_to_ingest, &table_reader);
896 if (!status.ok()) {
897 return status;
898 }
899
900 status = SanityCheckTableProperties(external_file, new_file_number, sv,
901 file_to_ingest, &table_reader);
902 if (!status.ok()) {
903 return status;
904 }
905
906 if (ingestion_options_.verify_checksums_before_ingest) {
907 // If customized readahead size is needed, we can pass a user option
908 // all the way to here. Right now we just rely on the default readahead
909 // to keep things simple.
910 // TODO: plumb Env::IOActivity, Env::IOPriority
911 ReadOptions ro;
912 ro.readahead_size = ingestion_options_.verify_checksums_readahead_size;
913 ro.fill_cache = ingestion_options_.fill_cache;
914 status = table_reader->VerifyChecksum(
915 ro, TableReaderCaller::kExternalSSTIngestion);
916 if (!status.ok()) {
917 return status;
918 }
919 }
920
921 ParsedInternalKey key;
922 // TODO: plumb Env::IOActivity, Env::IOPriority
923 ReadOptions ro;
924 ro.fill_cache = ingestion_options_.fill_cache;
925 std::unique_ptr<InternalIterator> iter(table_reader->NewIterator(
926 ro, sv->mutable_cf_options.prefix_extractor.get(), /*arena=*/nullptr,
927 /*skip_filters=*/false, TableReaderCaller::kExternalSSTIngestion));
928
929 // Get first (smallest) and last (largest) key from file.
930 bool allow_data_in_errors = db_options_.allow_data_in_errors;
931 iter->SeekToFirst();
932 if (iter->Valid()) {
933 Status pik_status =
934 ParseInternalKey(iter->key(), &key, allow_data_in_errors);
935 if (!pik_status.ok()) {
936 return Status::Corruption("Corrupted key in external file. ",
937 pik_status.getState());
938 }
939 if (key.sequence != 0) {
940 return Status::Corruption("External file has non zero sequence number");
941 }
942 file_to_ingest->smallest_internal_key.SetFrom(key);
943
944 Slice largest;
945 if (strcmp(sv->mutable_cf_options.table_factory->Name(), "PlainTable") ==
946 0) {
947 // PlainTable iterator does not support SeekToLast().
948 largest = iter->key();
949 for (; iter->Valid(); iter->Next()) {
950 if (cfd_->internal_comparator().Compare(iter->key(), largest) > 0) {
951 largest = iter->key();
952 }
953 }
954 if (!iter->status().ok()) {
955 return iter->status();
956 }
957 } else {
958 iter->SeekToLast();
959 if (!iter->Valid()) {
960 if (iter->status().ok()) {
961 // The file contains at least 1 key since iter is valid after
962 // SeekToFirst().
963 return Status::Corruption("Can not find largest key in sst file");
964 } else {
965 return iter->status();
966 }
967 }
968 largest = iter->key();
969 }
970
971 pik_status = ParseInternalKey(largest, &key, allow_data_in_errors);
972 if (!pik_status.ok()) {
973 return Status::Corruption("Corrupted key in external file. ",
974 pik_status.getState());
975 }
976 if (key.sequence != 0) {
977 return Status::Corruption("External file has non zero sequence number");
978 }
979 file_to_ingest->largest_internal_key.SetFrom(key);
980 } else if (!iter->status().ok()) {
981 return iter->status();
982 }
983 SequenceNumber largest_seqno =
984 table_reader.get()->GetTableProperties()->key_largest_seqno;
985 // UINT64_MAX means unknown and the file is generated before table property
986 // `key_largest_seqno` is introduced.
987 if (largest_seqno != UINT64_MAX && largest_seqno > 0) {
988 return Status::Corruption(
989 "External file has non zero largest sequence number " +
990 std::to_string(largest_seqno));
991 }
992 if (ingestion_options_.allow_db_generated_files &&
993 largest_seqno == UINT64_MAX) {
994 // Need to verify that all keys have seqno zero.
995 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
996 Status pik_status =
997 ParseInternalKey(iter->key(), &key, allow_data_in_errors);
998 if (!pik_status.ok()) {
999 return Status::Corruption("Corrupted key in external file. ",
1000 pik_status.getState());
1001 }
1002 if (key.sequence != 0) {
1003 return Status::NotSupported(
1004 "External file has a key with non zero sequence number.");
1005 }
1006 }
1007 if (!iter->status().ok()) {
1008 return iter->status();
1009 }
1010 }
1011
1012 std::unique_ptr<InternalIterator> range_del_iter(
1013 table_reader->NewRangeTombstoneIterator(ro));
1014 // We may need to adjust these key bounds, depending on whether any range
1015 // deletion tombstones extend past them.
1016 if (range_del_iter != nullptr) {
1017 for (range_del_iter->SeekToFirst(); range_del_iter->Valid();
1018 range_del_iter->Next()) {
1019 Status pik_status =
1020 ParseInternalKey(range_del_iter->key(), &key, allow_data_in_errors);
1021 if (!pik_status.ok()) {
1022 return Status::Corruption("Corrupted key in external file. ",
1023 pik_status.getState());
1024 }
1025 if (key.sequence != 0) {
1026 return Status::Corruption(
1027 "External file has a range deletion with non zero sequence "
1028 "number.");
1029 }
1030 RangeTombstone tombstone(key, range_del_iter->value());
1031 file_range_checker_.MaybeUpdateRange(tombstone.SerializeKey(),
1032 tombstone.SerializeEndKey(),
1033 file_to_ingest);
1034 }
1035 }
1036
1037 const size_t ts_sz = ucmp_->timestamp_size();
1038 Slice smallest = file_to_ingest->smallest_internal_key.user_key();
1039 Slice largest = file_to_ingest->largest_internal_key.user_key();
1040 if (ts_sz > 0) {
1041 AppendUserKeyWithMaxTimestamp(&file_to_ingest->start_ukey, smallest, ts_sz);
1042 AppendUserKeyWithMinTimestamp(&file_to_ingest->limit_ukey, largest, ts_sz);
1043 } else {
1044 file_to_ingest->start_ukey.assign(smallest.data(), smallest.size());
1045 file_to_ingest->limit_ukey.assign(largest.data(), largest.size());
1046 }
1047
1048 auto s =
1049 GetSstInternalUniqueId(file_to_ingest->table_properties.db_id,
1050 file_to_ingest->table_properties.db_session_id,
1051 file_to_ingest->table_properties.orig_file_number,
1052 &(file_to_ingest->unique_id));
1053 if (!s.ok()) {
1054 ROCKS_LOG_WARN(db_options_.info_log,
1055 "Failed to get SST unique id for file %s",
1056 file_to_ingest->internal_file_path.c_str());
1057 file_to_ingest->unique_id = kNullUniqueId64x2;
1058 }
1059
1060 return status;
1061}

1063Status ExternalSstFileIngestionJob::AssignLevelAndSeqnoForIngestedFile(
1064 SuperVersion* sv, bool force_global_seqno, CompactionStyle compaction_style,
1065 SequenceNumber last_seqno, IngestedFileInfo* file_to_ingest,
1066 SequenceNumber* assigned_seqno,
1067 std::optional<int> prev_batch_uppermost_level) {
1068 Status status;
1069 *assigned_seqno = 0;
1070 const size_t ts_sz = ucmp_->timestamp_size();
1071 assert(!prev_batch_uppermost_level.has_value() ||
1072 prev_batch_uppermost_level.value() < cfd_->NumberLevels());
1073 bool must_assign_to_l0 = prev_batch_uppermost_level.has_value() &&
1074 prev_batch_uppermost_level.value() == 0;
1075 if (force_global_seqno || files_overlap_ ||
1076 compaction_style == kCompactionStyleFIFO || must_assign_to_l0) {
1077 *assigned_seqno = last_seqno + 1;
1078 if (compaction_style == kCompactionStyleFIFO || must_assign_to_l0) {
1079 assert(ts_sz == 0);
1080 file_to_ingest->picked_level = 0;
1081 if (ingestion_options_.fail_if_not_bottommost_level &&
1082 cfd_->NumberLevels() > 1) {
1083 status = Status::TryAgain(
1084 "Files cannot be ingested to Lmax. Please make sure key range of "
1085 "Lmax does not overlap with files to ingest.");
1086 }
1087 return status;
1088 }
1089 }
1090
1091 bool overlap_with_db = false;
1092 Arena arena;
1093 // TODO: plumb Env::IOActivity, Env::IOPriority
1094 ReadOptions ro;
1095 ro.fill_cache = ingestion_options_.fill_cache;
1096 ro.total_order_seek = true;
1097 int target_level = 0;
1098 auto* vstorage = cfd_->current()->storage_info();
1099 assert(!must_assign_to_l0);
1100 int exclusive_end_level = prev_batch_uppermost_level.has_value()
1101 ? prev_batch_uppermost_level.value()
1102 : cfd_->NumberLevels();
1103
1104 for (int lvl = 0; lvl < exclusive_end_level; lvl++) {
1105 if (lvl > 0 && lvl < vstorage->base_level()) {
1106 continue;
1107 }
1108 if (cfd_->RangeOverlapWithCompaction(file_to_ingest->start_ukey,
1109 file_to_ingest->limit_ukey, lvl)) {
1110 // We must use L0 or any level higher than `lvl` to be able to overwrite
1111 // the compaction output keys that we overlap with in this level, We also
1112 // need to assign this file a seqno to overwrite the compaction output
1113 // keys in level `lvl`
1114 overlap_with_db = true;
1115 break;
1116 } else if (vstorage->NumLevelFiles(lvl) > 0) {
1117 bool overlap_with_level = false;
1118 status = sv->current->OverlapWithLevelIterator(
1119 ro, env_options_, file_to_ingest->start_ukey,
1120 file_to_ingest->limit_ukey, lvl, &overlap_with_level);
1121 if (!status.ok()) {
1122 return status;
1123 }
1124 if (overlap_with_level) {
1125 // We must use L0 or any level higher than `lvl` to be able to overwrite
1126 // the keys that we overlap with in this level, We also need to assign
1127 // this file a seqno to overwrite the existing keys in level `lvl`
1128 overlap_with_db = true;
1129 break;
1130 }
1131 }
1132
1133 // We don't overlap with any keys in this level, but we still need to check
1134 // if our file can fit in it
1135 if (IngestedFileFitInLevel(file_to_ingest, lvl)) {
1136 target_level = lvl;
1137 }
1138 }
1139
1140 if (ingestion_options_.fail_if_not_bottommost_level &&
1141 target_level < cfd_->NumberLevels() - 1) {
1142 status = Status::TryAgain(
1143 "Files cannot be ingested to Lmax. Please make sure key range of Lmax "
1144 "and ongoing compaction's output to Lmax"
1145 "does not overlap with files to ingest.");
1146 return status;
1147 }
1148
1149 TEST_SYNC_POINT_CALLBACK(
1150 "ExternalSstFileIngestionJob::AssignLevelAndSeqnoForIngestedFile",
1151 &overlap_with_db);
1152 file_to_ingest->picked_level = target_level;
1153 if (overlap_with_db) {
1154 if (ts_sz > 0) {
1155 status = Status::InvalidArgument(
1156 "Column family enables user-defined timestamps, please make sure the "
1157 "key range (without timestamp) of external file does not overlap "
1158 "with key range (without timestamp) in the db");
1159 return status;
1160 }
1161 if (*assigned_seqno == 0) {
1162 *assigned_seqno = last_seqno + 1;
1163 }
1164 }
1165
1166 if (ingestion_options_.allow_db_generated_files && *assigned_seqno != 0) {
1167 return Status::InvalidArgument(
1168 "An ingested file is assigned to a non-zero sequence number, which is "
1169 "incompatible with ingestion option allow_db_generated_files.");
1170 }
1171 return status;
1172}

1174Status ExternalSstFileIngestionJob::CheckLevelForIngestedBehindFile(
1175 IngestedFileInfo* file_to_ingest) {
1176 auto* vstorage = cfd_->current()->storage_info();
1177 // First, check if new files fit in the last level
1178 int last_lvl = cfd_->NumberLevels() - 1;
1179 if (!IngestedFileFitInLevel(file_to_ingest, last_lvl)) {
1180 return Status::InvalidArgument(
1181 "Can't ingest_behind file as it doesn't fit "

1192 "there are files with 0 seqno in database at upper levels!");

1201Status ExternalSstFileIngestionJob::AssignGlobalSeqnoForIngestedFile(
1202 IngestedFileInfo* file_to_ingest, SequenceNumber seqno) {
1203 if (file_to_ingest->original_seqno == seqno) {
1204 // This file already have the correct global seqno
1205 return Status::OK();
1206 } else if (!ingestion_options_.allow_global_seqno) {
1207 return Status::InvalidArgument("Global seqno is required, but disabled");
1208 } else if (ingestion_options_.write_global_seqno &&
1209 file_to_ingest->global_seqno_offset == 0) {
1210 return Status::InvalidArgument(
1211 "Trying to set global seqno for a file that don't have a global seqno "
1212 "field");
1213 }
1214
1215 if (ingestion_options_.write_global_seqno) {
1216 // Determine if we can write global_seqno to a given offset of file.
1217 // If the file system does not support random write, then we should not.
1218 // Otherwise we should.
1219 std::unique_ptr<FSRandomRWFile> rwfile;
1220 Status status = fs_->NewRandomRWFile(file_to_ingest->internal_file_path,
1221 env_options_, &rwfile, nullptr);
1222 TEST_SYNC_POINT_CALLBACK("ExternalSstFileIngestionJob::NewRandomRWFile",
1223 &status);
1224 if (status.ok()) {
1225 FSRandomRWFilePtr fsptr(std::move(rwfile), io_tracer_,
1226 file_to_ingest->internal_file_path);
1227 std::string seqno_val;
1228 PutFixed64(&seqno_val, seqno);
1229 status = fsptr->Write(file_to_ingest->global_seqno_offset, seqno_val,
1230 IOOptions(), nullptr);
1231 if (status.ok()) {
1232 TEST_SYNC_POINT("ExternalSstFileIngestionJob::BeforeSyncGlobalSeqno");
1233 status = SyncIngestedFile(fsptr.get());
1234 TEST_SYNC_POINT("ExternalSstFileIngestionJob::AfterSyncGlobalSeqno");
1235 if (!status.ok()) {
1236 ROCKS_LOG_WARN(db_options_.info_log,
1237 "Failed to sync ingested file %s after writing global "
1238 "sequence number: %s",
1239 file_to_ingest->internal_file_path.c_str(),
1240 status.ToString().c_str());
1241 }
1242 }
1243 if (!status.ok()) {
1244 return status;
1245 }
1246 } else if (!status.IsNotSupported()) {
1247 return status;
1248 }
1249 }
1250
1251 file_to_ingest->assigned_seqno = seqno;
1252 return Status::OK();
1253}
