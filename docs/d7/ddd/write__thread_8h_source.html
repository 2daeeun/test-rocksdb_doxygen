32class WriteThread {
33 public:
34 enum State : uint8_t {
35 // The initial state of a writer. This is a Writer that is
36 // waiting in JoinBatchGroup. This state can be left when another
37 // thread informs the waiter that it has become a group leader
38 // (-> STATE_GROUP_LEADER), when a leader that has chosen to be
39 // non-parallel informs a follower that its writes have been committed
40 // (-> STATE_COMPLETED), or when a leader that has chosen to perform
41 // updates in parallel and needs this Writer to apply its batch (->
42 // STATE_PARALLEL_MEMTABLE_WRITER).
43 STATE_INIT = 1,
44
45 // The state used to inform a waiting Writer that it has become the
46 // leader, and it should now build a write batch group. Tricky:
47 // this state is not used if newest_writer_ is empty when a writer
48 // enqueues itself, because there is no need to wait (or even to
49 // create the mutex and condvar used to wait) in that case. This is
50 // a terminal state unless the leader chooses to make this a parallel
51 // batch, in which case the last parallel worker to finish will move
52 // the leader to STATE_COMPLETED.
53 STATE_GROUP_LEADER = 2,
54
55 // The state used to inform a waiting writer that it has become the
56 // leader of memtable writer group. The leader will either write
57 // memtable for the whole group, or launch a parallel group write
58 // to memtable by calling LaunchParallelMemTableWrite.
59 STATE_MEMTABLE_WRITER_LEADER = 4,
60
61 // The state used to inform a waiting writer that it has become a
62 // parallel memtable writer. It can be the group leader who launch the
63 // parallel writer group, or one of the followers. The writer should then
64 // apply its batch to the memtable concurrently and call
65 // CompleteParallelMemTableWriter.
66 STATE_PARALLEL_MEMTABLE_WRITER = 8,
67
68 // A follower whose writes have been applied, or a parallel leader
69 // whose followers have all finished their work. This is a terminal
70 // state.
71 STATE_COMPLETED = 16,
72
73 // A state indicating that the thread may be waiting using StateMutex()
74 // and StateCondVar()
75 STATE_LOCKED_WAITING = 32,
76
77 // The state used to inform a waiting writer that it has become a
78 // caller to call some other waiting writers to write to memtable
79 // by calling SetMemWritersEachStride. After doing
80 // this, it will also write to memtable.
81 STATE_PARALLEL_MEMTABLE_CALLER = 64,
82 };

95 struct Iterator {
96 Writer* writer;
97 Writer* last_writer;
98
99 explicit Iterator(Writer* w, Writer* last)
100 : writer(w), last_writer(last) {}
101
102 Writer* operator*() const { return writer; }
103
104 Iterator& operator++() {
105 assert(writer != nullptr);
106 if (writer == last_writer) {
107 writer = nullptr;
108 } else {
109 writer = writer->link_newer;
110 }
111 return *this;
112 }
113
114 bool operator!=(const Iterator& other) const {
115 return writer != other.writer;
116 }
117 };
118
119 Iterator begin() const { return Iterator(leader, last_writer); }
120 Iterator end() const { return Iterator(nullptr, nullptr); }
121 };

233 void CreateMutex() {
234 if (!made_waitable) {
235 // Note that made_waitable is tracked separately from state
236 // transitions, because we can't atomically create the mutex and
237 // link into the list.
238 made_waitable = true;
239 new (&state_mutex_bytes) std::mutex;
240 new (&state_cv_bytes) std::condition_variable;
241 }
242 }
243
244 // returns the aggregate status of this Writer
245 Status FinalStatus() {
246 if (!status.ok()) {
247 // a non-ok memtable write status takes presidence
248 assert(callback == nullptr || callback_status.ok());
249 return status;
250 } else if (!callback_status.ok()) {
251 // if the callback failed then that is the status we want
252 // because a memtable insert should not have been attempted
253 assert(callback != nullptr);
254 assert(status.ok());
255 return callback_status;
256 } else {
257 // if there is no callback then we only care about
258 // the memtable insert status
259 assert(callback == nullptr || callback_status.ok());
260 return status;
261 }
262 }
263
264 bool CallbackFailed() {
265 return (callback != nullptr) && !callback_status.ok();
266 }
267
268 bool ShouldWriteToMemtable() {
269 return status.ok() && !CallbackFailed() && !disable_memtable;
270 }
271
272 bool ShouldWriteToWAL() {
273 return status.ok() && !CallbackFailed() && !disable_wal;
274 }
275
276 // No other mutexes may be acquired while holding StateMutex(), it is
277 // always last in the order
278 std::mutex& StateMutex() {
279 assert(made_waitable);
280 return *static_cast<std::mutex*>(static_cast<void*>(&state_mutex_bytes));
281 }
282
283 std::condition_variable& StateCV() {
284 assert(made_waitable);
285 return *static_cast<std::condition_variable*>(
286 static_cast<void*>(&state_cv_bytes));
287 }

301 // IMPORTANT: None of the methods in this class rely on the db mutex
302 // for correctness. All of the methods except JoinBatchGroup and
303 // EnterUnbatched may be called either with or without the db mutex held.
304 // Correctness is maintained by ensuring that only a single thread is
305 // a leader at a time.
306
307 // Registers w as ready to become part of a batch group, waits until the
308 // caller should perform some work, and returns the current state of the
309 // writer. If w has become the leader of a write batch group, returns
310 // STATE_GROUP_LEADER. If w has been made part of a sequential batch
311 // group and the leader has performed the write, returns STATE_DONE.
312 // If w has been made part of a parallel batch group and is responsible
313 // for updating the memtable, returns STATE_PARALLEL_MEMTABLE_WRITER.
314 //
315 // The db mutex SHOULD NOT be held when calling this function, because
316 // it will block.
317 //
318 // Writer* w: Writer to be executed as part of a batch group
319 void JoinBatchGroup(Writer* w);
320
321 // Constructs a write batch group led by leader, which should be a
322 // Writer passed to JoinBatchGroup on the current thread.
323 //
324 // Writer* leader: Writer that is STATE_GROUP_LEADER
325 // WriteGroup* write_group: Out-param of group members
326 // returns: Total batch group byte size
327 size_t EnterAsBatchGroupLeader(Writer* leader, WriteGroup* write_group);
328
329 // Unlinks the Writer-s in a batch group, wakes up the non-leaders,
330 // and wakes up the next leader (if any).
331 //
332 // WriteGroup* write_group: the write group

334 void ExitAsBatchGroupLeader(WriteGroup& write_group, Status& status);
335
336 // Exit batch group on behalf of batch group leader.
337 void ExitAsBatchGroupFollower(Writer* w);
338
339 // Constructs a write batch group led by leader from newest_memtable_writers_
340 // list. The leader should either write memtable for the whole group and
341 // call ExitAsMemTableWriter, or launch parallel memtable write through
342 // LaunchParallelMemTableWriters.

345 // Memtable writer group leader, or the last finished writer in a parallel
346 // write group, exit from the newest_memtable_writers_ list, and wake up
347 // the next leader if needed.
348 void ExitAsMemTableWriter(Writer* self, WriteGroup& write_group);
349
350 // Causes JoinBatchGroup to return STATE_PARALLEL_MEMTABLE_WRITER for all of
351 // the non-leader members of this write batch group. Sets Writer::sequence
352 // before waking them up.
353 // If the size of write_group n is not small, the leader will call n^0.5
354 // members to be PARALLEL_MEMTABLE_CALLER in the write_group to help to set
355 // other's status parallel. This ensures that the cost to call SetState
356 // sequentially does not exceed 2(n^0.5).

358 // WriteGroup* write_group: Extra state used to coordinate the parallel add

360
361 // One of the every stride=N number writer in the WriteGroup are set to the
362 // MemTableWriters, where N is equal to square of the total number of this
363 // write_group, and all of these MemTableWriters will write to memtable.
364 void SetMemWritersEachStride(Writer* w);
365
366 // Reports the completion of w's batch to the parallel group leader, and
367 // waits for the rest of the parallel batch to complete. Returns true
368 // if this thread is the last to complete, and hence should advance
369 // the sequence number and then call EarlyExitParallelGroup, false if
370 // someone else has already taken responsibility for that.

373 // Waits for all preceding writers (unlocking mu while waiting), then
