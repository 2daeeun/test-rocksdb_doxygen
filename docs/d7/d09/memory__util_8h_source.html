19// Returns the current memory usage of the specified DB instances.
20class MemoryUtil {
21 public:
22 enum UsageType : int {
23 // Memory usage of all the mem-tables.
24 kMemTableTotal = 0,
25 // Memory usage of those un-flushed mem-tables.
26 kMemTableUnFlushed = 1,
27 // Memory usage of all the table readers.
28 kTableReadersTotal = 2,
29 // Memory usage by Cache.
30 kCacheTotal = 3,
31 kNumUsageTypes = 4
32 };
33
34 // Returns the approximate memory usage of different types in the input
35 // list of DBs and Cache set. For instance, in the output map
36 // usage_by_type, usage_by_type[kMemTableTotal] will store the memory
37 // usage of all the mem-tables from all the input rocksdb instances.
38 //
39 // Note that for memory usage inside Cache class, we will
40 // only report the usage of the input "cache_set" without
41 // including those Cache usage inside the input list "dbs"
42 // of DBs.
43 static Status GetApproximateMemoryUsageByType(
44 const std::vector<DB*>& dbs,
45 const std::unordered_set<const Cache*> cache_set,
46 std::map<MemoryUtil::UsageType, uint64_t>* usage_by_type);
47};
48} // namespace ROCKSDB_NAMESPACE
