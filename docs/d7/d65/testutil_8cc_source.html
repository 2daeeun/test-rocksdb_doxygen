51std::string RandomKey(Random* rnd, int len, RandomKeyType type) {
52 // Make sure to generate a wide variety of characters so we
53 // test the boundary conditions for short-key optimizations.
54 static const char kTestChars[] = {'\0', '\1', 'a', 'b', 'c',
55 'd', 'e', '\xfd', '\xfe', '\xff'};
56 std::string result;
57 for (int i = 0; i < len; i++) {
58 std::size_t indx = 0;
59 switch (type) {
60 case RandomKeyType::RANDOM:
61 indx = rnd->Uniform(sizeof(kTestChars));
62 break;
63 case RandomKeyType::LARGEST:
64 indx = sizeof(kTestChars) - 1;
65 break;
66 case RandomKeyType::MIDDLE:
67 indx = sizeof(kTestChars) / 2;
68 break;
69 case RandomKeyType::SMALLEST:
70 indx = 0;
71 break;
72 }
73 result += kTestChars[indx];
74 }
75 return result;
76}

94Slice CompressibleString(Random* rnd, double compressed_fraction, int len,
95 std::string* dst) {
96 int raw = static_cast<int>(len * compressed_fraction);
97 if (raw < 1) {
98 raw = 1;
99 }
100 std::string raw_data = rnd->RandomBinaryString(raw);
101
102 // Duplicate the random data until we have filled "len" bytes
103 dst->clear();
104 while (dst->size() < (unsigned int)len) {
105 dst->append(raw_data);
106 }
107 dst->resize(len);
108 return Slice(*dst);
109}

112class Uint64ComparatorImpl : public Comparator {
113 public:
114 Uint64ComparatorImpl() = default;
115
116 const char* Name() const override { return "rocksdb.Uint64Comparator"; }
117
118 int Compare(const Slice& a, const Slice& b) const override {
119 assert(a.size() == sizeof(uint64_t) && b.size() == sizeof(uint64_t));
120 const uint64_t* left = reinterpret_cast<const uint64_t*>(a.data());
121 const uint64_t* right = reinterpret_cast<const uint64_t*>(b.data());
122 uint64_t leftValue;
123 uint64_t rightValue;
124 GetUnaligned(left, &leftValue);
125 GetUnaligned(right, &rightValue);
126 if (leftValue == rightValue) {
127 return 0;
128 } else if (leftValue < rightValue) {
129 return -1;
130 } else {
131 return 1;
132 }
133 }
134
135 void FindShortestSeparator(std::string* /*start*/,
136 const Slice& /*limit*/) const override {}
137
138 void FindShortSuccessor(std::string* /*key*/) const override {}
139};

191bool SleepingBackgroundTask::TimedWaitUntilSleeping(uint64_t wait_time) {
192 auto abs_time = SystemClock::Default()->NowMicros() + wait_time;
193 MutexLock l(&mutex_);
194 while (!sleeping_ || !should_sleep_) {
195 if (bg_cv_.TimedWait(abs_time)) {
196 return true;
197 }
198 }
199 return false;

209 }

498Status CorruptFile(Env* env, const std::string& fname, int offset,
499 int bytes_to_corrupt, bool verify_checksum /*=true*/) {
500 uint64_t size;
501 Status s = env->GetFileSize(fname, &size);
502 if (!s.ok()) {
503 return s;
504 } else if (offset < 0) {
505 // Relative to end of file; make it absolute
506 if (-offset > static_cast<int>(size)) {
507 offset = 0;
508 } else {
509 offset = static_cast<int>(size + offset);
510 }
511 }
512 if (offset > static_cast<int>(size)) {
513 offset = static_cast<int>(size);
514 }
515 if (offset + bytes_to_corrupt > static_cast<int>(size)) {
516 bytes_to_corrupt = static_cast<int>(size - offset);
517 }
518
519 // Do it
520 std::string contents;
521 s = ReadFileToString(env, fname, &contents);
522 if (s.ok()) {
523 for (int i = 0; i < bytes_to_corrupt; i++) {
524 contents[i + offset] ^= 0x80;
525 }
526 s = WriteStringToFile(env, contents, fname, false /* should_sync */);
527 }
528 if (s.ok() && verify_checksum) {
529 Options options;
530 options.env = env;
531 EnvOptions env_options;
532 Status v = VerifySstFileChecksum(options, env_options, fname);
533 assert(!v.ok());
534 }
535 return s;
536}

538Status TruncateFile(Env* env, const std::string& fname, uint64_t new_length) {
539 uint64_t old_length;
540 Status s = env->GetFileSize(fname, &old_length);
541 if (!s.ok() || new_length == old_length) {
542 return s;
543 }
544 // Do it
545 std::string contents;
546 s = ReadFileToString(env, fname, &contents);
547 if (s.ok()) {
548 contents.resize(static_cast<size_t>(new_length), 'b');
549 s = WriteStringToFile(env, contents, fname, false /* should_sync */);
550 }
551 return s;
552}

617 num_entries_(0) {}

620 return memtable_->Allocate(len, buf);
621 }
622
623 // Insert key into the list.
624 // REQUIRES: nothing that compares equal to key is currently in the list.
625 void Insert(KeyHandle handle) override {
626 num_entries_++;
627 memtable_->Insert(handle);
628 }

631 num_entries_++;
632 memtable_->Insert(handle);
633 }

638 }

640 size_t ApproximateMemoryUsage() override {

642 // to trigger a flush.

644 }

654 }

657 return memtable_->GetIterator(arena);
658 }
659
660 ~SpecialMemTableRep() override = default;
661
662 private:
663 std::unique_ptr<MemTableRep> memtable_;
664 int num_entries_flush_;
665 int num_entries_;
666};
