18bool MatchesInteger(const std::string &target, size_t start, size_t pos) {
19 // If it is numeric, everything up to the match must be a number
20 int digits = 0;
21 if (target[start] == '-') {
22 start++; // Allow negative numbers
23 }
24 while (start < pos) {
25 if (!isdigit(target[start++])) {
26 return false;
27 } else {
28 digits++;
29 }
30 }
31 return (digits > 0);
32}
33
34bool MatchesDecimal(const std::string &target, size_t start, size_t pos) {
35 int digits = 0;
36 if (target[start] == '-') {
37 start++; // Allow negative numbers
38 }
39 for (bool point = false; start < pos; start++) {
40 if (target[start] == '.') {
41 if (point) {
42 return false;
43 } else {
44 point = true;
45 }
46 } else if (!isdigit(target[start])) {
47 return false;
48 } else {
49 digits++;
50 }
51 }
52 return (digits > 0);
53}

56size_t ObjectLibrary::PatternEntry::MatchSeparatorAt(
57 size_t start, Quantifier mode, const std::string &target, size_t tlen,
58 const std::string &separator) const {
59 size_t slen = separator.size();
60 // See if there is enough space. If so, find the separator
61 if (tlen < start + slen) {
62 return std::string::npos; // not enough space left
63 } else if (mode == kMatchExact) {
64 // Exact mode means the next thing we are looking for is the separator
65 if (target.compare(start, slen, separator) != 0) {
66 return std::string::npos;
67 } else {
68 return start + slen; // Found the separator, return where we found it
69 }
70 } else {
71 auto pos = start + 1;
72 if (!separator.empty()) {
73 pos = target.find(separator, pos);
74 }
75 if (pos == std::string::npos) {
76 return pos;
77 } else if (mode == kMatchInteger) {
78 if (!MatchesInteger(target, start, pos)) {
79 return std::string::npos;
80 }
81 } else if (mode == kMatchDecimal) {
82 if (!MatchesDecimal(target, start, pos)) {
83 return std::string::npos;
84 }
85 }
86 return pos + slen;
87 }
88}

90bool ObjectLibrary::PatternEntry::MatchesTarget(const std::string &name,
91 size_t nlen,
92 const std::string &target,
93 size_t tlen) const {
94 if (separators_.empty()) {
95 assert(optional_); // If there are no separators, it must be only a name
96 return nlen == tlen && name == target;
97 } else if (nlen == tlen) { // The lengths are the same
98 return optional_ && name == target;
99 } else if (tlen < nlen + slength_) {
100 // The target is not long enough
101 return false;
102 } else if (target.compare(0, nlen, name) != 0) {
103 return false; // Target does not start with name
104 } else {
105 // Loop through all of the separators one at a time matching them.
106 // Note that we first match the separator and then its quantifiers.
107 // Since we expect the separator first, we start with an exact match
108 // Subsequent matches will use the quantifier of the previous separator
109 size_t start = nlen;
110 auto mode = kMatchExact;
111 for (size_t idx = 0; idx < separators_.size(); ++idx) {
112 const auto &separator = separators_[idx];
113 start = MatchSeparatorAt(start, mode, target, tlen, separator.first);
114 if (start == std::string::npos) {
115 return false;
116 } else {
117 mode = separator.second;
118 }
119 }
120 // We have matched all of the separators. Now check that what is left
121 // unmatched in the target is acceptable.
122 if (mode == kMatchExact) {
123 return (start == tlen);
124 } else if (start > tlen || (start == tlen && mode != kMatchZeroOrMore)) {
125 return false;
126 } else if (mode == kMatchInteger) {
127 return MatchesInteger(target, start, tlen);
128 } else if (mode == kMatchDecimal) {
129 return MatchesDecimal(target, start, tlen);
130 }
131 }
132 return true;
133}

135bool ObjectLibrary::PatternEntry::Matches(const std::string &target) const {
136 auto tlen = target.size();
137 if (MatchesTarget(name_, nlength_, target, tlen)) {
138 return true;
139 } else if (!names_.empty()) {
140 for (const auto &alt : names_) {
141 if (MatchesTarget(alt, alt.size(), target, tlen)) {
142 return true;
143 }
144 }
145 }
146 return false;
147}

240Status ObjectRegistry::SetManagedObject(
241 const std::string &type, const std::string &id,
242 const std::shared_ptr<Customizable> &object) {
243 std::string object_key = ToManagedObjectKey(type, id);
244 std::shared_ptr<Customizable> curr;
245 if (parent_ != nullptr) {
246 curr = parent_->GetManagedObject(type, id);
247 }
248 if (curr == nullptr) {
249 // We did not find the object in any parent. Update in the current
250 std::unique_lock<std::mutex> lock(objects_mutex_);
251 auto iter = managed_objects_.find(object_key);
252 if (iter != managed_objects_.end()) { // The object exists
253 curr = iter->second.lock();
254 if (curr != nullptr && curr != object) {
255 return Status::InvalidArgument("Object already exists: ", object_key);
256 } else {
257 iter->second = object;
258 }
259 } else {
260 // The object does not exist. Add it
261 managed_objects_[object_key] = object;
262 }
263 } else if (curr != object) {
264 return Status::InvalidArgument("Object already exists: ", object_key);
265 }
266 return Status::OK();
267}

285Status ObjectRegistry::ListManagedObjects(
286 const std::string &type, const std::string &name,
287 std::vector<std::shared_ptr<Customizable>> *results) const {
288 {
289 std::string key = ToManagedObjectKey(type, name);
290 std::unique_lock<std::mutex> lock(objects_mutex_);
291 for (auto iter = managed_objects_.lower_bound(key);
292 iter != managed_objects_.end() && StartsWith(iter->first, key);
293 ++iter) {
294 auto shared = iter->second.lock();
295 if (shared != nullptr) {
296 if (name.empty() || shared->IsInstanceOf(name)) {
297 results->emplace_back(shared);
298 }
299 }
300 }
301 }
302 if (parent_ != nullptr) {
303 return parent_->ListManagedObjects(type, name, results);
304 } else {
305 return Status::OK();
306 }
