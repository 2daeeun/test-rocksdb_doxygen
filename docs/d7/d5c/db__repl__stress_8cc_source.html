52struct DataPumpThread {
53 DB* db; // Assumption DB is Open'ed already.
54};

56static void DataPumpThreadBody(void* arg) {
57 DataPumpThread* t = static_cast<DataPumpThread*>(arg);
58 DB* db = t->db;
59 Random rnd(301);
60 uint64_t i = 0;
61 while (i++ < FLAGS_num_inserts) {
62 if (!db->Put(WriteOptions(), Slice(rnd.RandomString(500)),
63 Slice(rnd.RandomString(500)))
64 .ok()) {
65 fprintf(stderr, "Error in put\n");
66 exit(1);
67 }
68 }
69}

71int main(int argc, const char** argv) {
72 SetUsageMessage(
73 std::string("\nUSAGE:\n") + std::string(argv[0]) +
74 " --num_inserts=<num_inserts> --wal_ttl_seconds=<WAL_ttl_seconds>" +
75 " --wal_size_limit_MB=<WAL_size_limit_MB>");
76 ParseCommandLineFlags(&argc, const_cast<char***>(&argv), true);
77
78 Env* env = Env::Default();
79 std::string default_db_path;
80 env->GetTestDirectory(&default_db_path);
81 default_db_path += "db_repl_stress";
82 Options options;
83 options.create_if_missing = true;
84 options.WAL_ttl_seconds = FLAGS_wal_ttl_seconds;
85 options.WAL_size_limit_MB = FLAGS_wal_size_limit_MB;
86 DB* db;
87 DestroyDB(default_db_path, options);
88
89 Status s = DB::Open(options, default_db_path, &db);
90
91 if (!s.ok()) {
92 fprintf(stderr, "Could not open DB due to %s\n", s.ToString().c_str());
93 exit(1);
94 }
95
96 DataPumpThread dataPump;
97 dataPump.db = db;
98 env->StartThread(DataPumpThreadBody, &dataPump);
99
100 std::unique_ptr<TransactionLogIterator> iter;
101 SequenceNumber currentSeqNum = 1;
102 uint64_t num_read = 0;
103 for (;;) {
104 iter.reset();
105 // Continue to probe a bit more after all received
106 size_t probes = 0;
107 while (!db->GetUpdatesSince(currentSeqNum, &iter).ok()) {
108 probes++;
109 if (probes > 100 && num_read >= FLAGS_num_inserts) {
110 if (num_read > FLAGS_num_inserts) {
111 fprintf(stderr, "Too many updates read: %ld expected: %ld\n",
112 (long)num_read, (long)FLAGS_num_inserts);
113 exit(1);
114 }
115 fprintf(stderr, "Successful!\n");
116 return 0;
117 }
118 }
119 fprintf(stderr, "Refreshing iterator\n");
120 for (; iter->Valid(); iter->Next(), num_read++, currentSeqNum++) {
121 BatchResult res = iter->GetBatch();
122 if (res.sequence != currentSeqNum) {
123 fprintf(stderr, "Missed a seq no. b/w %ld and %ld\n",
124 (long)currentSeqNum, (long)res.sequence);
125 exit(1);
126 }
127 }
128 }
129}
