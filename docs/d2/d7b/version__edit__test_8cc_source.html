23static void TestEncodeDecode(const VersionEdit& edit) {
24 // Encoding one `VersionEdit` and decoding it again should result in the
25 // exact same `VersionEdit`. However, a special handling is applied to file
26 // boundaries: `FileMetaData.smallest`, `FileMetaData.largest` when
27 // user-defined timestamps should not be persisted. In that scenario, this
28 // invariant does not hold. We disable this scenario in this util method to
29 // enable all other test cases continue to verify this invariant, while the
30 // special case is separately covered in test
31 // `EncodeDecodeNewFile4HandleFileBoundary`.
32 std::string encoded, encoded2;
33 edit.EncodeTo(&encoded, 0 /* ts_sz */);
34 VersionEdit parsed;
35 Status s = parsed.DecodeFrom(encoded);
36 ASSERT_TRUE(s.ok()) << s.ToString();
37 parsed.EncodeTo(&encoded2, 0 /* ts_sz */);
38 ASSERT_EQ(encoded, encoded2);
39}

133TEST_F(VersionEditTest, EncodeDecodeNewFile4HandleFileBoundary) {
134 static const uint64_t kBig = 1ull << 50;
135 size_t ts_sz = 16;
136 static std::string min_ts(ts_sz, static_cast<unsigned char>(0));
137 VersionEdit edit;
138 std::string smallest = "foo";
139 std::string largest = "zoo";
140 // In real manifest writing scenarios, one `VersionEdit` should not contain
141 // files with different `user_defined_timestamps_persisted` flag value.
142 // This is just for testing file boundaries handling w.r.t persisting user
143 // defined timestamps during `VersionEdit` encoding.
144 edit.AddFile(
145 3, 300, 3, 100, InternalKey(smallest + min_ts, kBig + 500, kTypeValue),
146 InternalKey(largest + min_ts, kBig + 600, kTypeDeletion), kBig + 500,
147 kBig + 600, true, Temperature::kUnknown, kInvalidBlobFileNumber,
148 kUnknownOldestAncesterTime, kUnknownFileCreationTime,
149 300 /* epoch_number */, kUnknownFileChecksum,
150 kUnknownFileChecksumFuncName, kNullUniqueId64x2,
151 0 /* compensated_range_deletion_size */, 0 /* tail_size */,
152 false /* user_defined_timestamps_persisted */);
153 edit.AddFile(3, 300, 3, 100,
154 InternalKey(smallest + min_ts, kBig + 500, kTypeValue),
155 InternalKey(largest + min_ts, kBig + 600, kTypeDeletion),
156 kBig + 500, kBig + 600, true, Temperature::kUnknown,
157 kInvalidBlobFileNumber, kUnknownOldestAncesterTime,
158 kUnknownFileCreationTime, 300 /* epoch_number */,
159 kUnknownFileChecksum, kUnknownFileChecksumFuncName,
160 kNullUniqueId64x2, 0 /* compensated_range_deletion_size */,
161 0 /* tail_size */, true /* user_defined_timestamps_persisted */);
162
163 std::string encoded;
164 edit.EncodeTo(&encoded, ts_sz);
165 VersionEdit parsed;
166 Status s = parsed.DecodeFrom(encoded);
167 ASSERT_TRUE(s.ok()) << s.ToString();
168 auto& new_files = parsed.GetNewFiles();
169 ASSERT_TRUE(new_files.size() == 2);
170 ASSERT_FALSE(new_files[0].second.user_defined_timestamps_persisted);
171 // First file's boundaries do not contain user-defined timestamps.
172 ASSERT_EQ(InternalKey(smallest, kBig + 500, kTypeValue).Encode(),
173 new_files[0].second.smallest.Encode());
174 ASSERT_EQ(InternalKey(largest, kBig + 600, kTypeDeletion).Encode(),
175 new_files[0].second.largest.Encode());
176 ASSERT_TRUE(new_files[1].second.user_defined_timestamps_persisted);
177 // Second file's boundaries contain user-defined timestamps.
178 ASSERT_EQ(InternalKey(smallest + min_ts, kBig + 500, kTypeValue).Encode(),
179 new_files[1].second.smallest.Encode());
180 ASSERT_EQ(InternalKey(largest + min_ts, kBig + 600, kTypeDeletion).Encode(),
181 new_files[1].second.largest.Encode());
182}

198 edit.DeleteFile(4, 700);

200 edit.SetComparatorName("foo");

207
208 // Call back function to add extra customized builds.
209 bool first = true;
210 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->SetCallBack(
211 "VersionEdit::EncodeTo:NewFile4:CustomizeFields", [&](void* arg) {
212 std::string* str = reinterpret_cast<std::string*>(arg);
213 PutVarint32(str, 33);
214 const std::string str1 = "random_string";
215 PutLengthPrefixedSlice(str, str1);
216 if (first) {
217 first = false;
218 PutVarint32(str, 22);
219 const std::string str2 = "s";
220 PutLengthPrefixedSlice(str, str2);
221 }
222 });
223 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->EnableProcessing();
224 edit.EncodeTo(&encoded, 0 /* ts_sz */);
225 ROCKSDB_NAMESPACE::SyncPoint::GetInstance()->DisableProcessing();

228 Status s = parsed.DecodeFrom(encoded);
229 ASSERT_TRUE(s.ok()) << s.ToString();
230 ASSERT_TRUE(!first);
231 auto& new_files = parsed.GetNewFiles();
232 ASSERT_TRUE(new_files[0].second.marked_for_compaction);
233 ASSERT_TRUE(!new_files[1].second.marked_for_compaction);
234 ASSERT_EQ(3u, new_files[0].second.fd.GetPathId());
235 ASSERT_EQ(3u, new_files[1].second.fd.GetPathId());
236 ASSERT_EQ(1u, parsed.GetDeletedFiles().size());
237 ASSERT_TRUE(parsed.GetPersistUserDefinedTimestamps());

314TEST_F(VersionEditTest, IgnorableField) {
315 VersionEdit ve;
316 std::string encoded;
317
318 // Size of ignorable field is too large
319 PutVarint32Varint64(&encoded, 2 /* kLogNumber */, 66);
320 // This is a customized ignorable tag
321 PutVarint32Varint64(&encoded,
322 0x2710 /* A field with kTagSafeIgnoreMask set */,
323 5 /* fieldlength 5 */);
324 encoded += "abc"; // Only fills 3 bytes,
325 ASSERT_NOK(ve.DecodeFrom(encoded));
326
327 encoded.clear();
328 // Error when seeing unidentified tag that is not ignorable
329 PutVarint32Varint64(&encoded, 2 /* kLogNumber */, 66);
330 // This is a customized ignorable tag
331 PutVarint32Varint64(&encoded, 666 /* A field with kTagSafeIgnoreMask unset */,
332 3 /* fieldlength 3 */);
333 encoded += "abc"; // Fill 3 bytes
334 PutVarint32Varint64(&encoded, 3 /* next file number */, 88);
335 ASSERT_NOK(ve.DecodeFrom(encoded));
336
337 // Safely ignore an identified but safely ignorable entry
338 encoded.clear();
339 PutVarint32Varint64(&encoded, 2 /* kLogNumber */, 66);
340 // This is a customized ignorable tag
341 PutVarint32Varint64(&encoded,
342 0x2710 /* A field with kTagSafeIgnoreMask set */,
343 3 /* fieldlength 3 */);
344 encoded += "abc"; // Fill 3 bytes
345 PutVarint32Varint64(&encoded, 3 /* kNextFileNumber */, 88);
346
347 ASSERT_OK(ve.DecodeFrom(encoded));
348
349 ASSERT_TRUE(ve.HasLogNumber());
350 ASSERT_TRUE(ve.HasNextFile());
351 ASSERT_EQ(66, ve.GetLogNumber());
352 ASSERT_EQ(88, ve.GetNextFile());
353}
