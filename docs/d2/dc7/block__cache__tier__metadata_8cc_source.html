46bool BlockCacheTierMetadata::Lookup(const Slice& key, LBA* lba) {
47 BlockInfo lookup_key(key);
48 BlockInfo* block;
49 port::RWMutex* rlock = nullptr;
50 if (!block_index_.Find(&lookup_key, &block, &rlock)) {
51 return false;
52 }
53
54 ReadUnlock _(rlock);
55 assert(block->key_ == key.ToString());
56 if (lba) {
57 *lba = block->lba_;
58 }
59 return true;
60}

71void BlockCacheTierMetadata::RemoveAllKeys(BlockCacheFile* f) {
72 for (BlockInfo* binfo : f->block_infos()) {
73 BlockInfo* tmp = nullptr;
74 bool status = block_index_.Erase(binfo, &tmp);
75 (void)status;
76 assert(status);
77 assert(tmp == binfo);
78 delete binfo;
79 }
80 f->block_infos().clear();
81}
