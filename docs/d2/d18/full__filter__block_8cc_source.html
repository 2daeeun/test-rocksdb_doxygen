38void FullFilterBlockBuilder::Add(const Slice& key_without_ts) {
39 if (prefix_extractor_ && prefix_extractor_->InDomain(key_without_ts)) {
40 Slice prefix = prefix_extractor_->Transform(key_without_ts);
41 if (whole_key_filtering_) {
42 filter_bits_builder_->AddKeyAndAlt(key_without_ts, prefix);
43 } else {
44 filter_bits_builder_->AddKey(prefix);
45 }
46 } else if (whole_key_filtering_) {
47 filter_bits_builder_->AddKey(key_without_ts);
48 }
49}

76std::unique_ptr<FilterBlockReader> FullFilterBlockReader::Create(
77 const BlockBasedTable* table, const ReadOptions& ro,
78 FilePrefetchBuffer* prefetch_buffer, bool use_cache, bool prefetch,
79 bool pin, BlockCacheLookupContext* lookup_context) {
80 assert(table);
81 assert(table->get_rep());
82 assert(!pin || prefetch);
83
84 CachableEntry<ParsedFullFilterBlock> filter_block;
85 if (prefetch || !use_cache) {
86 const Status s = ReadFilterBlock(table, prefetch_buffer, ro, use_cache,
87 nullptr /* get_context */, lookup_context,
88 &filter_block);
89 if (!s.ok()) {
90 IGNORE_STATUS_IF_ERROR(s);
91 return std::unique_ptr<FilterBlockReader>();
92 }
93
94 if (use_cache && !pin) {
95 filter_block.Reset();
96 }
97 }
98
99 return std::unique_ptr<FilterBlockReader>(
100 new FullFilterBlockReader(table, std::move(filter_block)));
101}

110bool FullFilterBlockReader::MayMatch(const Slice& entry,
111 GetContext* get_context,
112 BlockCacheLookupContext* lookup_context,
113 const ReadOptions& read_options) const {
114 CachableEntry<ParsedFullFilterBlock> filter_block;
115
116 const Status s = GetOrReadFilterBlock(get_context, lookup_context,
117 &filter_block, read_options);
118 if (!s.ok()) {
119 IGNORE_STATUS_IF_ERROR(s);
120 return true;
121 }
122
123 assert(filter_block.GetValue());
124
125 FilterBitsReader* const filter_bits_reader =
126 filter_block.GetValue()->filter_bits_reader();
127
128 if (filter_bits_reader) {
129 if (filter_bits_reader->MayMatch(entry)) {
130 PERF_COUNTER_ADD(bloom_sst_hit_count, 1);
131 return true;
132 } else {
133 PERF_COUNTER_ADD(bloom_sst_miss_count, 1);
134 return false;
135 }
136 }
137 return true;
138}

157void FullFilterBlockReader::MayMatch(MultiGetRange* range,
158 const SliceTransform* prefix_extractor,
159 BlockCacheLookupContext* lookup_context,
160 const ReadOptions& read_options) const {
161 CachableEntry<ParsedFullFilterBlock> filter_block;
162
163 const Status s = GetOrReadFilterBlock(
164 range->begin()->get_context, lookup_context, &filter_block, read_options);
165 if (!s.ok()) {
166 IGNORE_STATUS_IF_ERROR(s);
167 return;
168 }
169
170 assert(filter_block.GetValue());
171
172 FilterBitsReader* const filter_bits_reader =
173 filter_block.GetValue()->filter_bits_reader();
174
175 if (!filter_bits_reader) {
176 return;
177 }
178
179 // We need to use an array instead of autovector for may_match since
180 // &may_match[0] doesn't work for autovector<bool> (compiler error). So
181 // declare both keys and may_match as arrays, which is also slightly less
182 // expensive compared to autovector
183 std::array<Slice*, MultiGetContext::MAX_BATCH_SIZE> keys;
184 std::array<bool, MultiGetContext::MAX_BATCH_SIZE> may_match = {{true}};
185 autovector<Slice, MultiGetContext::MAX_BATCH_SIZE> prefixes;
186 int num_keys = 0;
187 MultiGetRange filter_range(*range, range->begin(), range->end());
188 for (auto iter = filter_range.begin(); iter != filter_range.end(); ++iter) {
189 if (!prefix_extractor) {
190 keys[num_keys++] = &iter->ukey_without_ts;
191 } else if (prefix_extractor->InDomain(iter->ukey_without_ts)) {
192 prefixes.emplace_back(prefix_extractor->Transform(iter->ukey_without_ts));
193 keys[num_keys++] = &prefixes.back();
194 } else {
195 filter_range.SkipKey(iter);
196 }
197 }
198
199 filter_bits_reader->MayMatch(num_keys, keys.data(), may_match.data());
200
201 int i = 0;
202 for (auto iter = filter_range.begin(); iter != filter_range.end(); ++iter) {
203 if (!may_match[i]) {
204 // Update original MultiGet range to skip this key. The filter_range
205 // was temporarily used just to skip keys not in prefix_extractor domain
206 range->SkipKey(iter);
207 PERF_COUNTER_ADD(bloom_sst_miss_count, 1);
208 } else {
209 // PERF_COUNTER_ADD(bloom_sst_hit_count, 1);
210 PerfContext* perf_ctx = get_perf_context();
211 perf_ctx->bloom_sst_hit_count++;
212 }
213 ++i;
214 }
215}
