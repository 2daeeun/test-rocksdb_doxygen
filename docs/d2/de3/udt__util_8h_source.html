24// Dummy record in WAL logs signaling user-defined timestamp sizes for
25// subsequent records.
26class UserDefinedTimestampSizeRecord {
27 public:
28 UserDefinedTimestampSizeRecord() {}
29 explicit UserDefinedTimestampSizeRecord(
30 std::vector<std::pair<uint32_t, size_t>>&& cf_to_ts_sz)
31 : cf_to_ts_sz_(std::move(cf_to_ts_sz)) {}
32
33 const std::vector<std::pair<uint32_t, size_t>>& GetUserDefinedTimestampSize()
34 const {
35 return cf_to_ts_sz_;
36 }
37
38 inline void EncodeTo(std::string* dst) const {
39 assert(dst != nullptr);
40 for (const auto& [cf_id, ts_sz] : cf_to_ts_sz_) {
41 assert(ts_sz != 0);
42 PutFixed32(dst, cf_id);
43 PutFixed16(dst, static_cast<uint16_t>(ts_sz));
44 }
45 }
46
47 inline Status DecodeFrom(Slice* src) {
48 const size_t total_size = src->size();
49 if ((total_size % kSizePerColumnFamily) != 0) {
50 std::ostringstream oss;
51 oss << "User-defined timestamp size record length: " << total_size
52 << " is not a multiple of " << kSizePerColumnFamily << std::endl;
53 return Status::Corruption(oss.str());
54 }
55 int num_of_entries = static_cast<int>(total_size / kSizePerColumnFamily);
56 for (int i = 0; i < num_of_entries; i++) {
57 uint32_t cf_id = 0;
58 uint16_t ts_sz = 0;
59 if (!GetFixed32(src, &cf_id) || !GetFixed16(src, &ts_sz)) {
60 return Status::Corruption(
61 "Error decoding user-defined timestamp size record entry");
62 }
63 cf_to_ts_sz_.emplace_back(cf_id, static_cast<size_t>(ts_sz));
64 }
65 return Status::OK();
66 }
67
68 inline std::string DebugString() const {
69 std::ostringstream oss;
70
71 for (const auto& [cf_id, ts_sz] : cf_to_ts_sz_) {
72 oss << "Column family: " << cf_id
73 << ", user-defined timestamp size: " << ts_sz << std::endl;
74 }
75 return oss.str();
76 }
77
78 private:
79 // 4 bytes for column family id, 2 bytes for user-defined timestamp size.
80 static constexpr size_t kSizePerColumnFamily = 4 + 2;
81
82 std::vector<std::pair<uint32_t, size_t>> cf_to_ts_sz_;
83};

85// This handler is used to recover a WriteBatch read from WAL logs during
86// recovery. It does a best-effort recovery if the column families contained in
87// the WriteBatch have inconsistency between the recorded timestamp size and the
88// running timestamp size. And creates a new WriteBatch that are consistent with
89// the running timestamp size with entries from the original WriteBatch.
90//
91// Note that for a WriteBatch with no inconsistency, a new WriteBatch is created
92// nonetheless, and it should be exactly the same as the original WriteBatch.
93//
94// To access the new WriteBatch, invoke `TransferNewBatch` after calling
95// `Iterate`. The handler becomes invalid afterwards.
96//
97// For the user key in each entry, the best effort recovery means:
98// 1) If recorded timestamp size is 0, running timestamp size is > 0, a min
99// timestamp of length running timestamp size is padded to the user key.
100// 2) If recorded timestamp size is > 0, running timestamp size is 0, the last
101// bytes of length recorded timestamp size is stripped from user key.
102// 3) If recorded timestamp size is the same as running timestamp size, no-op.
103// 4) If recorded timestamp size and running timestamp size are both non-zero
104// but not equal, return Status::InvalidArgument.
105class TimestampRecoveryHandler : public WriteBatch::Handler {
106 public:
107 TimestampRecoveryHandler(const UnorderedMap<uint32_t, size_t>& running_ts_sz,
108 const UnorderedMap<uint32_t, size_t>& record_ts_sz,
109 bool seq_per_batch, bool batch_per_txn);

194// Mode for checking and handling timestamp size inconsistency encountered in a
195// WriteBatch read from WAL log.
196enum class TimestampSizeConsistencyMode {
197 // Verified that the recorded user-defined timestamp size is consistent with
198 // the running one for all the column families involved in a WriteBatch.
199 // Column families referred to in the WriteBatch but are dropped are ignored.
200 kVerifyConsistency,
201 // Verified that if any inconsistency exists in a WriteBatch, it's all
202 // tolerable by a best-effort reconciliation. And optionally creates a new
203 // WriteBatch from the original WriteBatch that is consistent with the running
204 // timestamp size. Column families referred to in the WriteBatch but are
205 // dropped are ignored. If a new WriteBatch is created, such entries are
206 // copied over as is.
207 kReconcileInconsistency,
208};

210// Handles the inconsistency between recorded timestamp sizes and running
211// timestamp sizes for a WriteBatch. A non-OK `status` indicates there are
212// intolerable inconsistency with the specified `check_mode`.
213//
214// If `check_mode` is `kVerifyConsistency`, intolerable inconsistency means any
215// running column family has an inconsistent user-defined timestamp size.
216//
217// If `check_mode` is `kReconcileInconsistency`, intolerable inconsistency means
218// any running column family has an inconsistent user-defined timestamp size
219// that cannot be reconciled with a best-effort recovery. Check
220// `TimestampRecoveryHandler` for what a best-effort recovery is capable of. In
221// this mode, output argument `new_batch` should be set, a new WriteBatch is
222// created on the heap and transferred to `new_batch` if there is tolerable
223// inconsistency.
224//
225// An invariant that WAL logging ensures is that all timestamp size info
226// is logged prior to a WriteBatch that needed this info. And zero timestamp
227// size is skipped. So `record_ts_sz` only contains column family with non-zero
228// timestamp size and a column family id absent from `record_ts_sz` will be
229// interpreted as that column family has zero timestamp size. On the other hand,
230// `running_ts_sz` should contain the timestamp size for all running column
231// families including the ones with zero timestamp size.
232Status HandleWriteBatchTimestampSizeDifference(
233 const WriteBatch* batch,
234 const UnorderedMap<uint32_t, size_t>& running_ts_sz,
235 const UnorderedMap<uint32_t, size_t>& record_ts_sz,
236 TimestampSizeConsistencyMode check_mode, bool seq_per_batch,
237 bool batch_per_txn, std::unique_ptr<WriteBatch>* new_batch = nullptr);

239// This util function is used when opening an existing column family and
240// processing its VersionEdit. It does a sanity check for the column family's
241// old user comparator and the persist_user_defined_timestamps flag as recorded
242// in the VersionEdit, against its new settings from the column family's
243// ImmutableCFOptions.
244//
245// Valid settings change include:
246// 1) no user comparator change and no effective persist_user_defined_timestamp
247// flag change.
248// 2) switch user comparator to enable user-defined timestamps feature provided
249// the immediately effective persist_user_defined_timestamps flag is false.
250// 3) switch user comparator to disable user-defined timestamps feature provided
251// that the before-change persist_user_defined_timestamps is already false.
252//
253// Switch user comparator to disable/enable UDT is only sanity checked by a user
254// comparator name comparison. The full check includes enforcing the new user
255// comparator ranks user keys exactly the same as the old user comparator and
256// only add / remove the user-defined timestamp comparison. We don't have ways
257// to strictly enforce this so currently only the RocksDB builtin comparator
258// wrapper `ComparatorWithU64TsImpl` is supported to enable / disable
259// user-defined timestamps. It formats user-defined timestamps as uint64_t.
260//
261// When the settings indicate a legit change to enable user-defined timestamps
262// feature on a column family, `mark_sst_files_has_no_udt` will be set to true
263// to indicate marking all existing SST files has no user-defined timestamps
264// when re-writing the manifest.
265Status ValidateUserDefinedTimestampsOptions(
266 const Comparator* new_comparator, const std::string& old_comparator_name,
267 bool new_persist_udt, bool old_persist_udt,
268 bool* mark_sst_files_has_no_udt);
