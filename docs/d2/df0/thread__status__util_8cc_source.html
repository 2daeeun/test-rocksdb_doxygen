19void ThreadStatusUtil::RegisterThread(const Env* env,
20 ThreadStatus::ThreadType thread_type) {
21 if (!MaybeInitThreadLocalUpdater(env)) {
22 return;
23 }
24 assert(thread_updater_local_cache_);
25 thread_updater_local_cache_->RegisterThread(thread_type, env->GetThreadID());
26}

28void ThreadStatusUtil::UnregisterThread() {
29 thread_updater_initialized_ = false;
30 if (thread_updater_local_cache_ != nullptr) {
31 thread_updater_local_cache_->UnregisterThread();
32 thread_updater_local_cache_ = nullptr;
33 }
34}

51void ThreadStatusUtil::SetThreadOperation(ThreadStatus::OperationType op) {
52 if (thread_updater_local_cache_ == nullptr) {
53 return;
54 }
55
56 if (op != ThreadStatus::OP_UNKNOWN) {
57 uint64_t current_time = SystemClock::Default()->NowMicros();
58 thread_updater_local_cache_->SetOperationStartTime(current_time);
59 } else {
60 // TDOO(yhchiang): we could report the time when we set operation to
61 // OP_UNKNOWN once the whole instrumentation has been done.
62 thread_updater_local_cache_->SetOperationStartTime(0);
63 }
64 thread_updater_local_cache_->SetThreadOperation(op);
65}

152bool ThreadStatusUtil::MaybeInitThreadLocalUpdater(const Env* env) {
153 if (!thread_updater_initialized_ && env != nullptr) {
154 thread_updater_initialized_ = true;
155 thread_updater_local_cache_ = env->GetThreadStatusUpdater();
156 }
157 return (thread_updater_local_cache_ != nullptr);
158}
