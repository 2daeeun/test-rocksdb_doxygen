6// APIs for accessing Cache in a type-safe and convenient way. Cache is kept
7// at a low, thin level of abstraction so that different implementations can
8// be plugged in, but these wrappers provide clean, convenient access to the
9// most common operations.
10//
11// A number of template classes are needed for sharing common structure. The
12// key classes are these:
13//
14// * PlaceholderCacheInterface - Used for making cache reservations, with
15// entries that have a charge but no value.
16// * BasicTypedCacheInterface<TValue> - Used for primary cache storage of
17// objects of type TValue.
18// * FullTypedCacheHelper<TValue, TCreateContext> - Used for secondary cache
19// compatible storage of objects of type TValue.
20// * For each of these, there's a "Shared" version
21// (e.g. FullTypedSharedCacheInterface) that holds a shared_ptr to the Cache,
22// rather than assuming external ownership by holding only a raw `Cache*`.
23

37// For future consideration:
38// * Pass in value to Insert with std::unique_ptr& to simplify ownership
39// transfer logic in callers
40// * Make key type a template parameter (e.g. useful for table cache)
41// * Closer integration with CacheHandleGuard (opt-in, so not always
42// paying the extra overhead)
43
44#define CACHE_TYPE_DEFS() \
45 using Priority = Cache::Priority; \
46 using Handle = Cache::Handle; \
47 using ObjectPtr = Cache::ObjectPtr; \
48 using CreateContext = Cache::CreateContext; \
49 using CacheItemHelper = Cache::CacheItemHelper /* caller ; */
50
51template <typename CachePtr>
52class BaseCacheInterface {
53 public:
54 CACHE_TYPE_DEFS();
55
56 /*implicit*/ BaseCacheInterface(CachePtr cache) : cache_(std::move(cache)) {}
57
58 inline void Release(Handle* handle) { cache_->Release(handle); }
59
60 inline void ReleaseAndEraseIfLastRef(Handle* handle) {
61 cache_->Release(handle, /*erase_if_last_ref*/ true);
62 }
63
64 inline void RegisterReleaseAsCleanup(Handle* handle, Cleanable& cleanable) {
65 cleanable.RegisterCleanup(&ReleaseCacheHandleCleanup, get(), handle);
66 }
67
68 inline Cache* get() const { return &*cache_; }
69
70 explicit inline operator bool() const noexcept { return cache_ != nullptr; }
71
72 protected:
73 CachePtr cache_;
74};

76// PlaceholderCacheInterface - Used for making cache reservations, with
77// entries that have a charge but no value. CacheEntryRole is required as
78// a template parameter.
79template <CacheEntryRole kRole, typename CachePtr = Cache*>
80class PlaceholderCacheInterface : public BaseCacheInterface<CachePtr> {
81 public:
82 CACHE_TYPE_DEFS();

93 }

100template <class TValue>
101class BasicTypedCacheHelperFns {
102 public:
103 CACHE_TYPE_DEFS();
104 // E.g. char* for char[]
105 using TValuePtr = std::remove_extent_t<TValue>*;
106
107 protected:
108 inline static ObjectPtr UpCastValue(TValuePtr value) { return value; }
109 inline static TValuePtr DownCastValue(ObjectPtr value) {
110 return static_cast<TValuePtr>(value);
111 }
112
113 static void Delete(ObjectPtr value, MemoryAllocator* allocator) {
114 // FIXME: Currently, no callers actually allocate the ObjectPtr objects
115 // using the custom allocator, just subobjects that keep a reference to
116 // the allocator themselves (with CacheAllocationPtr).
117 if (/*DISABLED*/ false && allocator) {
118 if constexpr (std::is_destructible_v<TValue>) {
119 DownCastValue(value)->~TValue();
120 }
121 allocator->Deallocate(value);
122 } else {
123 // Like delete but properly handles TValue=char[] etc.
124 std::default_delete<TValue>{}(DownCastValue(value));
125 }
126 }
127};

137 return &kHelper;
138 }

141// BasicTypedCacheInterface - Used for primary cache storage of objects of
142// type TValue, which can be cleaned up with std::default_delete<TValue>. The
143// role is provided by TValue::kCacheEntryRole or given in an optional
144// template parameter.
145template <class TValue, CacheEntryRole kRole = TValue::kCacheEntryRole,
146 typename CachePtr = Cache*>
147class BasicTypedCacheInterface : public BaseCacheInterface<CachePtr>,
148 public BasicTypedCacheHelper<TValue, kRole> {
149 public:
150 CACHE_TYPE_DEFS();
151 using typename BasicTypedCacheHelperFns<TValue>::TValuePtr;
152 struct TypedHandle : public Handle {};
153 using BasicTypedCacheHelper<TValue, kRole>::GetBasicHelper;
154 // ctor
155 using BaseCacheInterface<CachePtr>::BaseCacheInterface;
156 struct TypedAsyncLookupHandle : public Cache::AsyncLookupHandle {
157 TypedHandle* Result() {
158 return static_cast<TypedHandle*>(Cache::AsyncLookupHandle::Result());
159 }
160 };
161
162 inline Status Insert(const Slice& key, TValuePtr value, size_t charge,
163 TypedHandle** handle = nullptr,
164 Priority priority = Priority::LOW) {
165 auto untyped_handle = reinterpret_cast<Handle**>(handle);
166 return this->cache_->Insert(
167 key, BasicTypedCacheHelperFns<TValue>::UpCastValue(value),
168 GetBasicHelper(), charge, untyped_handle, priority);
169 }
170
171 inline TypedHandle* Lookup(const Slice& key, Statistics* stats = nullptr) {
172 return static_cast<TypedHandle*>(this->cache_->BasicLookup(key, stats));
173 }
174
175 inline void StartAsyncLookup(TypedAsyncLookupHandle& async_handle) {
176 assert(async_handle.helper == nullptr);
177 this->cache_->StartAsyncLookup(async_handle);
178 }
179
180 inline CacheHandleGuard<TValue> Guard(TypedHandle* handle) {
181 if (handle) {
182 return CacheHandleGuard<TValue>(&*this->cache_, handle);
183 } else {
184 return {};
185 }
186 }
187
188 inline std::shared_ptr<TValue> SharedGuard(TypedHandle* handle) {
189 if (handle) {
190 return MakeSharedCacheHandleGuard<TValue>(&*this->cache_, handle);
191 } else {
192 return {};
193 }
194 }
195
196 inline TValuePtr Value(TypedHandle* handle) {
197 return BasicTypedCacheHelperFns<TValue>::DownCastValue(
198 this->cache_->Value(handle));
199 }
200};

208// TValue must implement ContentSlice() and ~TValue
209// TCreateContext must implement Create(std::unique_ptr<TValue>*, ...)
210template <class TValue, class TCreateContext>
211class FullTypedCacheHelperFns : public BasicTypedCacheHelperFns<TValue> {
212 public:
213 CACHE_TYPE_DEFS();
214
215 protected:
216 using typename BasicTypedCacheHelperFns<TValue>::TValuePtr;
217 using BasicTypedCacheHelperFns<TValue>::DownCastValue;
218 using BasicTypedCacheHelperFns<TValue>::UpCastValue;
219
220 static size_t Size(ObjectPtr v) {
221 TValuePtr value = DownCastValue(v);
222 auto slice = value->ContentSlice();
223 return slice.size();
224 }
225
226 static Status SaveTo(ObjectPtr v, size_t from_offset, size_t length,
227 char* out) {
228 TValuePtr value = DownCastValue(v);
229 auto slice = value->ContentSlice();
230 assert(from_offset < slice.size());
231 assert(from_offset + length <= slice.size());
232 std::copy_n(slice.data() + from_offset, length, out);
233 return Status::OK();
234 }
235
236 static Status Create(const Slice& data, CompressionType type,
237 CacheTier source, CreateContext* context,
238 MemoryAllocator* allocator, ObjectPtr* out_obj,
239 size_t* out_charge) {
240 std::unique_ptr<TValue> value = nullptr;
241 if (source != CacheTier::kVolatileTier) {
242 return Status::InvalidArgument();
243 }
244 if constexpr (sizeof(TCreateContext) > 0) {
245 TCreateContext* tcontext = static_cast<TCreateContext*>(context);
246 tcontext->Create(&value, out_charge, data, type, allocator);
247 } else {
248 TCreateContext::Create(&value, out_charge, data, type, allocator);
249 }
250 *out_obj = UpCastValue(value.release());
251 return Status::OK();
252 }
253};

273// FullTypedCacheHelper - Used for secondary cache compatible storage of
274// objects of type TValue. In addition to BasicTypedCacheInterface constraints,
275// we require TValue::ContentSlice() to return persistable data. This
276// simplifies usage for the normal case of simple secondary cache compatibility
277// (can give you a Slice to the data already in memory). In addition to
278// TCreateContext performing the role of Cache::CreateContext, it is also
279// expected to provide a function Create(std::unique_ptr<TValue>* value,
280// size_t* out_charge, const Slice& data, MemoryAllocator* allocator) for
281// creating new TValue.
282template <class TValue, class TCreateContext,
283 CacheEntryRole kRole = TValue::kCacheEntryRole,
284 typename CachePtr = Cache*>
285class FullTypedCacheInterface
286 : public BasicTypedCacheInterface<TValue, kRole, CachePtr>,
287 public FullTypedCacheHelper<TValue, TCreateContext, kRole> {
288 public:
289 CACHE_TYPE_DEFS();
290 using typename BasicTypedCacheInterface<TValue, kRole, CachePtr>::TypedHandle;
291 using typename BasicTypedCacheInterface<TValue, kRole,
292 CachePtr>::TypedAsyncLookupHandle;
293 using typename BasicTypedCacheHelperFns<TValue>::TValuePtr;
294 using BasicTypedCacheHelper<TValue, kRole>::GetBasicHelper;
295 using FullTypedCacheHelper<TValue, TCreateContext, kRole>::GetFullHelper;
296 using BasicTypedCacheHelperFns<TValue>::UpCastValue;
297 using BasicTypedCacheHelperFns<TValue>::DownCastValue;
298 // ctor
299 using BasicTypedCacheInterface<TValue, kRole,
300 CachePtr>::BasicTypedCacheInterface;
301
302 // Insert with SecondaryCache compatibility (subject to CacheTier).
303 // (Basic Insert() also inherited.)
304 inline Status InsertFull(
305 const Slice& key, TValuePtr value, size_t charge,
306 TypedHandle** handle = nullptr, Priority priority = Priority::LOW,
307 CacheTier lowest_used_cache_tier = CacheTier::kNonVolatileBlockTier,
308 const Slice& compressed = Slice(),
309 CompressionType type = CompressionType::kNoCompression) {
310 auto untyped_handle = reinterpret_cast<Handle**>(handle);
311 auto helper = lowest_used_cache_tier > CacheTier::kVolatileTier
312 ? GetFullHelper()
313 : GetBasicHelper();
314 return this->cache_->Insert(key, UpCastValue(value), helper, charge,
315 untyped_handle, priority, compressed, type);
316 }
317
318 // Like SecondaryCache::InsertSaved, with SecondaryCache compatibility
319 // (subject to CacheTier).
320 inline Status InsertSaved(
321 const Slice& key, const Slice& data, TCreateContext* create_context,
322 Priority priority = Priority::LOW,
323 CacheTier lowest_used_cache_tier = CacheTier::kNonVolatileBlockTier,
324 size_t* out_charge = nullptr) {
325 ObjectPtr value;
326 size_t charge;
327 Status st = GetFullHelper()->create_cb(
328 data, kNoCompression, CacheTier::kVolatileTier, create_context,
329 this->cache_->memory_allocator(), &value, &charge);
330 if (out_charge) {
331 *out_charge = charge;
332 }
333 if (st.ok()) {
334 st = InsertFull(key, DownCastValue(value), charge, nullptr /*handle*/,
335 priority, lowest_used_cache_tier);
336 } else {
337 GetFullHelper()->del_cb(value, this->cache_->memory_allocator());
338 }
339 return st;
340 }
341
342 // Lookup with SecondaryCache support (subject to CacheTier).
343 // (Basic Lookup() also inherited.)
344 inline TypedHandle* LookupFull(
345 const Slice& key, TCreateContext* create_context = nullptr,
346 Priority priority = Priority::LOW, Statistics* stats = nullptr,
347 CacheTier lowest_used_cache_tier = CacheTier::kNonVolatileBlockTier) {
348 if (lowest_used_cache_tier > CacheTier::kVolatileTier) {
349 return static_cast<TypedHandle*>(this->cache_->Lookup(
350 key, GetFullHelper(), create_context, priority, stats));
351 } else {
352 return BasicTypedCacheInterface<TValue, kRole, CachePtr>::Lookup(key,
353 stats);
354 }
355 }
356
357 inline void StartAsyncLookupFull(
358 TypedAsyncLookupHandle& async_handle,
359 CacheTier lowest_used_cache_tier = CacheTier::kNonVolatileBlockTier) {
360 if (lowest_used_cache_tier > CacheTier::kVolatileTier) {
361 async_handle.helper = GetFullHelper();
362 this->cache_->StartAsyncLookup(async_handle);
363 } else {
364 BasicTypedCacheInterface<TValue, kRole, CachePtr>::StartAsyncLookup(
365 async_handle);
366 }
367 }
368};
