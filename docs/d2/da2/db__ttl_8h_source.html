16// Database with TTL support.
17//
18// USE-CASES:
19// This API should be used to open the db when key-values inserted are
20// meant to be removed from the db in a non-strict 'ttl' amount of time
21// Therefore, this guarantees that key-values inserted will remain in the
22// db for >= ttl amount of time and the db will make efforts to remove the
23// key-values as soon as possible after ttl seconds of their insertion.
24//
25// BEHAVIOUR:
26// TTL is accepted in seconds
27// (int32_t)Timestamp(creation) is suffixed to values in Put internally
28// Expired TTL values deleted in compaction only:(Timestamp+ttl<time_now)
29// Get/Iterator may return expired entries(compaction not run on them yet)
30// Different TTL may be used during different Opens
31// Example: Open1 at t=0 with ttl=4 and insert k1,k2, close at t=2
32// Open2 at t=3 with ttl=5. Now k1,k2 should be deleted at t>=5
33// read_only=true opens in the usual read-only mode. Compactions will not be
34// triggered(neither manual nor automatic), so no expired entries removed
35//
36// CONSTRAINTS:
37// Not specifying/passing or non-positive TTL behaves like TTL = infinity
38//
39// !!!WARNING!!!:
40// Calling DB::Open directly to re-open a db created by this API will get
41// corrupt values(timestamp suffixed) and no ttl effect will be there
42// during the second Open, so use this API consistently to open the db
43// Be careful when passing ttl with a small positive value because the
44// whole database may be deleted in a small amount of time
45
46class DBWithTTL : public StackableDB {
47 public:
48 virtual Status CreateColumnFamilyWithTtl(
49 const ColumnFamilyOptions& options, const std::string& column_family_name,
50 ColumnFamilyHandle** handle, int ttl) = 0;
51
52 static Status Open(const Options& options, const std::string& dbname,
53 DBWithTTL** dbptr, int32_t ttl = 0,
54 bool read_only = false);
55
56 static Status Open(const DBOptions& db_options, const std::string& dbname,
57 const std::vector<ColumnFamilyDescriptor>& column_families,
58 std::vector<ColumnFamilyHandle*>* handles,
59 DBWithTTL** dbptr, const std::vector<int32_t>& ttls,
60 bool read_only = false);
61
62 virtual void SetTtl(int32_t ttl) = 0;
63
64 virtual void SetTtl(ColumnFamilyHandle* h, int32_t ttl) = 0;
65
66 protected:
67 explicit DBWithTTL(DB* db) : StackableDB(db) {}
68};
69
70} // namespace ROCKSDB_NAMESPACE
