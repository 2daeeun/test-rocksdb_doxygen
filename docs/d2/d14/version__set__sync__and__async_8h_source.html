14// Lookup a batch of keys in a single SST file
15DEFINE_SYNC_AND_ASYNC(Status, Version::MultiGetFromSST)
16(const ReadOptions& read_options, MultiGetRange file_range, int hit_file_level,
17 bool skip_filters, bool skip_range_deletions, FdWithKeyRange* f,
18 std::unordered_map<uint64_t, BlobReadContexts>& blob_ctxs,
19 TableCache::TypedHandle* table_handle, uint64_t& num_filter_read,
20 uint64_t& num_index_read, uint64_t& num_sst_read) {
21 bool timer_enabled = GetPerfLevel() >= PerfLevel::kEnableTimeExceptForMutex &&
22 get_perf_context()->per_level_perf_context_enabled;
23
24 Status s;
25 StopWatchNano timer(clock_, timer_enabled /* auto_start */);
26 s = CO_AWAIT(table_cache_->MultiGet)(
27 read_options, *internal_comparator(), *f->file_metadata, &file_range,
28 mutable_cf_options_,
29 cfd_->internal_stats()->GetFileReadHist(hit_file_level), skip_filters,
30 skip_range_deletions, hit_file_level, table_handle);
31 // TODO: examine the behavior for corrupted key
32 if (timer_enabled) {
33 PERF_COUNTER_BY_LEVEL_ADD(get_from_table_nanos, timer.ElapsedNanos(),
34 hit_file_level);
35 }
36 if (!s.ok()) {
37 // TODO: Set status for individual keys appropriately
38 for (auto iter = file_range.begin(); iter != file_range.end(); ++iter) {
39 *iter->s = s;
40 file_range.MarkKeyDone(iter);
41 }
42 CO_RETURN s;
43 }
44 uint64_t batch_size = 0;
45 for (auto iter = file_range.begin(); s.ok() && iter != file_range.end();
46 ++iter) {
47 GetContext& get_context = *iter->get_context;
48 Status* status = iter->s;
49 // The Status in the KeyContext takes precedence over GetContext state
50 // Status may be an error if there were any IO errors in the table
51 // reader. We never expect Status to be NotFound(), as that is
52 // determined by get_context
53 assert(!status->IsNotFound());
54 if (!status->ok()) {
55 file_range.MarkKeyDone(iter);
56 continue;
57 }
58
59 if (get_context.sample()) {
60 sample_file_read_inc(f->file_metadata);
61 }
62 batch_size++;
63 num_index_read += get_context.get_context_stats_.num_index_read;
64 num_filter_read += get_context.get_context_stats_.num_filter_read;
65 num_sst_read += get_context.get_context_stats_.num_sst_read;
66 // Reset these stats since they're specific to a level
67 get_context.get_context_stats_.num_index_read = 0;
68 get_context.get_context_stats_.num_filter_read = 0;
69 get_context.get_context_stats_.num_sst_read = 0;
70
71 // report the counters before returning
72 if (get_context.State() != GetContext::kNotFound &&
73 get_context.State() != GetContext::kMerge &&
74 db_statistics_ != nullptr) {
75 get_context.ReportCounters();
76 } else {
77 if (iter->max_covering_tombstone_seq > 0) {
78 // The remaining files we look at will only contain covered keys, so
79 // we stop here for this key
80 file_range.SkipKey(iter);
81 }
82 }
83 switch (get_context.State()) {
84 case GetContext::kNotFound:
85 // Keep searching in other files
86 break;
87 case GetContext::kMerge:
88 // TODO: update per-level perfcontext user_key_return_count for kMerge
89 break;
90 case GetContext::kFound:
91 if (hit_file_level == 0) {
92 RecordTick(db_statistics_, GET_HIT_L0);
93 } else if (hit_file_level == 1) {
94 RecordTick(db_statistics_, GET_HIT_L1);
95 } else if (hit_file_level >= 2) {
96 RecordTick(db_statistics_, GET_HIT_L2_AND_UP);
97 }
98
99 PERF_COUNTER_BY_LEVEL_ADD(user_key_return_count, 1, hit_file_level);
100
101 file_range.MarkKeyDone(iter);
102
103 if (iter->is_blob_index) {
104 BlobIndex blob_index;
105 Status tmp_s;
106
107 if (iter->value) {
108 TEST_SYNC_POINT_CALLBACK("Version::MultiGet::TamperWithBlobIndex",
109 &(*iter));
110
111 tmp_s = blob_index.DecodeFrom(*(iter->value));
112
113 } else {
114 assert(iter->columns);
115
116 tmp_s = blob_index.DecodeFrom(
117 WideColumnsHelper::GetDefaultColumn(iter->columns->columns()));
118 }
119
120 if (tmp_s.ok()) {
121 const uint64_t blob_file_num = blob_index.file_number();
122 blob_ctxs[blob_file_num].emplace_back(blob_index, &*iter);
123 } else {
124 *(iter->s) = tmp_s;
125 }
126 } else {
127 if (iter->value) {
128 file_range.AddValueSize(iter->value->size());
129 } else {
130 assert(iter->columns);
131 file_range.AddValueSize(iter->columns->serialized_size());
132 }
133
134 if (file_range.GetValueSize() > read_options.value_size_soft_limit) {
135 s = Status::Aborted();
136 break;
137 }
138 }
139 continue;
140 case GetContext::kDeleted:
141 // Use empty error message for speed
142 *status = Status::NotFound();
143 file_range.MarkKeyDone(iter);
144 continue;
145 case GetContext::kCorrupt:
146 *status =
147 Status::Corruption("corrupted key for ", iter->lkey->user_key());
148 file_range.MarkKeyDone(iter);
149 continue;
150 case GetContext::kUnexpectedBlobIndex:
151 ROCKS_LOG_ERROR(info_log_, "Encounter unexpected blob index.");
152 *status = Status::NotSupported(
153 "Encounter unexpected blob index. Please open DB with "
154 "ROCKSDB_NAMESPACE::blob_db::BlobDB instead.");
155 file_range.MarkKeyDone(iter);
156 continue;
157 case GetContext::kMergeOperatorFailed:
158 *status = Status::Corruption(Status::SubCode::kMergeOperatorFailed);
159 file_range.MarkKeyDone(iter);
160 continue;
161 }
162 }
163
164 RecordInHistogram(db_statistics_, SST_BATCH_SIZE, batch_size);
165 CO_RETURN s;
166}
167} // namespace ROCKSDB_NAMESPACE
