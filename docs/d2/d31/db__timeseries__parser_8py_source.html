47 def fetch_burst_epochs(
48 self,
49 entities: str,
50 statistic: int,
51 window_sec: float,
52 threshold: bool,
53 percent: bool,
54 ) -> dict[str, dict[int, float]]:
55 # this method calculates the (percent) rate change in the 'statistic'
56 # for each entity (over 'window_sec' seconds) and returns the epochs
57 # where this rate change is greater than or equal to the 'threshold'
58 # value
59 if self.stats_freq_sec == 0:
60 # not time series data, cannot check for bursty behavior
61 return
62 if window_sec < self.stats_freq_sec:
63 window_sec = self.stats_freq_sec
64 # 'window_samples' is the number of windows to go back to
65 # compare the current window with, while calculating rate change.
66 window_samples = math.ceil(window_sec / self.stats_freq_sec)
67 burst_epochs = {}
68 # if percent = False:
69 # curr_val = value at window for which rate change is being calculated
70 # prev_val = value at window that is window_samples behind curr_window
71 # Then rate_without_percent =
72 # ((curr_val-prev_val)*duration_sec)/(curr_timestamp-prev_timestamp)
73 # if percent = True:
74 # rate_with_percent = (rate_without_percent * 100) / prev_val
75 # These calculations are in line with the rate() transform supported
76 # by ODS
77 for entity in entities:
78 if statistic not in self.keys_ts[entity]:
79 continue
80 timestamps = sorted(list(self.keys_ts[entity][statistic].keys()))
81 for ix in range(window_samples, len(timestamps), 1):
82 first_ts = timestamps[ix - window_samples]
83 last_ts = timestamps[ix]
84 first_val = self.keys_ts[entity][statistic][first_ts]
85 last_val = self.keys_ts[entity][statistic][last_ts]
86 diff = last_val - first_val
87 if percent:
88 diff = diff * 100 / first_val
89 rate = (diff * self.duration_sec) / (last_ts - first_ts)
90 # if the rate change is greater than the provided threshold,
91 # then the condition is triggered for entity at time 'last_ts'
92 if rate >= threshold:
93 if entity not in burst_epochs:
94 burst_epochs[entity] = {}
95 burst_epochs[entity][last_ts] = rate
96 return burst_epochs

98 def fetch_aggregated_values(self, entity, statistics, aggregation_op):
99 # this method performs the aggregation specified by 'aggregation_op'
100 # on the timeseries of 'statistics' for 'entity' and returns:
101 # Dict[statistic, aggregated_value]
102 result = {}
103 for stat in statistics:
104 if stat not in self.keys_ts[entity]:
105 continue
106 agg_val = None
107 if aggregation_op is self.AggregationOperator.latest:
108 latest_timestamp = max(list(self.keys_ts[entity][stat].keys()))
109 agg_val = self.keys_ts[entity][stat][latest_timestamp]
110 elif aggregation_op is self.AggregationOperator.oldest:
111 oldest_timestamp = min(list(self.keys_ts[entity][stat].keys()))
112 agg_val = self.keys_ts[entity][stat][oldest_timestamp]
113 elif aggregation_op is self.AggregationOperator.max:
114 agg_val = max(list(self.keys_ts[entity][stat].values()))
115 elif aggregation_op is self.AggregationOperator.min:
116 agg_val = min(list(self.keys_ts[entity][stat].values()))
117 elif aggregation_op is self.AggregationOperator.avg:
118 values = list(self.keys_ts[entity][stat].values())
119 agg_val = sum(values) / len(values)
120 result[stat] = agg_val
121 return result

123 def check_and_trigger_conditions(self, conditions):
124 # get the list of statistics that need to be fetched
125 reqd_keys = self.get_keys_from_conditions(conditions)
126 # fetch the required statistics and populate the map 'keys_ts'
127 self.fetch_timeseries(reqd_keys)
128 # Trigger the appropriate conditions
129 for cond in conditions:
130 complete_keys = self.get_keys_from_conditions([cond])
131 # Get the entities that have all statistics required by 'cond':
132 # an entity is checked for a given condition only if we possess all
133 # of the condition's 'keys' for that entity
134 entities_with_stats = []
135 for entity in self.keys_ts:
136 stat_missing = False
137 for stat in complete_keys:
138 if stat not in self.keys_ts[entity]:
139 stat_missing = True
140 break
141 if not stat_missing:
142 entities_with_stats.append(entity)
143 if not entities_with_stats:
144 continue
145 if cond.behavior is self.Behavior.bursty:
146 # for a condition that checks for bursty behavior, only one key
147 # should be present in the condition's 'keys' field
148 result = self.fetch_burst_epochs(
149 entities_with_stats,
150 complete_keys[0], # there should be only one key
151 cond.window_sec,
152 cond.rate_threshold,
153 True,
154 )
155 # Trigger in this case is:
156 # Dict[entity_name, Dict[timestamp, rate_change]]
157 # where the inner dictionary contains rate_change values when
158 # the rate_change >= threshold provided, with the
159 # corresponding timestamps
160 if result:
161 cond.set_trigger(result)
162 elif cond.behavior is self.Behavior.evaluate_expression:
163 self.handle_evaluate_expression(
164 cond, complete_keys, entities_with_stats
165 )

167 def handle_evaluate_expression(self, condition, statistics, entities):
168 trigger = {}
169 # check 'condition' for each of these entities
170 for entity in entities:
171 if hasattr(condition, "aggregation_op"):
172 # in this case, the aggregation operation is performed on each
173 # of the condition's 'keys' and then with aggregated values
174 # condition's 'expression' is evaluated; if it evaluates to
175 # True, then list of the keys values is added to the
176 # condition's trigger: Dict[entity_name, List[stats]]
177 result = self.fetch_aggregated_values(
178 entity, statistics, condition.aggregation_op
179 )
180 keys = [result[key] for key in statistics]
181 try:
182 if eval(condition.expression):
183 trigger[entity] = keys
184 except Exception as e:
185 print("WARNING(TimeSeriesData) check_and_trigger: " + str(e))
186 else:
187 # assumption: all stats have same series of timestamps
188 # this is similar to the above but 'expression' is evaluated at
189 # each timestamp, since there is no aggregation, and all the
190 # epochs are added to the trigger when the condition's
191 # 'expression' evaluated to true; so trigger is:
192 # Dict[entity, Dict[timestamp, List[stats]]]
193 for epoch in self.keys_ts[entity][statistics[0]].keys():
194 keys = [self.keys_ts[entity][key][epoch] for key in statistics]
195 try:
196 if eval(condition.expression):
197 if entity not in trigger:
198 trigger[entity] = {}
199 trigger[entity][epoch] = keys
200 except Exception as e:
201 print("WARNING(TimeSeriesData) check_and_trigger: " + str(e))
202 if trigger:
203 condition.set_trigger(trigger)
