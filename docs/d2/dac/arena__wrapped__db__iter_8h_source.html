29// A wrapper iterator which wraps DB Iterator and the arena, with which the DB
30// iterator is supposed to be allocated. This class is used as an entry point of
31// a iterator hierarchy whose memory can be allocated inline. In that way,
32// accessing the iterator tree can be more cache friendly. It is also faster
33// to allocate.
34// When using the class's Iterator interface, the behavior is exactly
35// the same as the inner DBIter.
36class ArenaWrappedDBIter : public Iterator {
37 public:
38 ~ArenaWrappedDBIter() override {
39 if (db_iter_ != nullptr) {
40 db_iter_->~DBIter();
41 } else {
42 assert(false);
43 }
44 }
45
46 // Get the arena to be used to allocate memory for DBIter to be wrapped,
47 // as well as child iterators in it.
48 virtual Arena* GetArena() { return &arena_; }
49
50 const ReadOptions& GetReadOptions() { return read_options_; }
51
52 // Set the internal iterator wrapped inside the DB Iterator. Usually it is
53 // a merging iterator.
54 virtual void SetIterUnderDBIter(InternalIterator* iter) {
55 db_iter_->SetIter(iter);
56 }
57
58 void SetMemtableRangetombstoneIter(
59 std::unique_ptr<TruncatedRangeDelIterator>* iter) {
60 memtable_range_tombstone_iter_ = iter;
61 }
62
63 bool Valid() const override { return db_iter_->Valid(); }
64 void SeekToFirst() override { db_iter_->SeekToFirst(); }
65 void SeekToLast() override { db_iter_->SeekToLast(); }
66 // 'target' does not contain timestamp, even if user timestamp feature is
67 // enabled.
68 void Seek(const Slice& target) override { db_iter_->Seek(target); }
69 void SeekForPrev(const Slice& target) override {
70 db_iter_->SeekForPrev(target);
71 }
72 void Next() override { db_iter_->Next(); }
73 void Prev() override { db_iter_->Prev(); }
74 Slice key() const override { return db_iter_->key(); }
75 Slice value() const override { return db_iter_->value(); }
76 const WideColumns& columns() const override { return db_iter_->columns(); }
77 Status status() const override { return db_iter_->status(); }
78 Slice timestamp() const override { return db_iter_->timestamp(); }
79 bool IsBlob() const { return db_iter_->IsBlob(); }
80
81 Status GetProperty(std::string prop_name, std::string* prop) override;
82
83 Status Refresh() override;
84 Status Refresh(const Snapshot*) override;
85
86 bool PrepareValue() override { return db_iter_->PrepareValue(); }
87
88 void Init(Env* env, const ReadOptions& read_options,
89 const ImmutableOptions& ioptions,
90 const MutableCFOptions& mutable_cf_options, const Version* version,
91 const SequenceNumber& sequence,
92 uint64_t max_sequential_skip_in_iterations, uint64_t version_number,
93 ReadCallback* read_callback, ColumnFamilyHandleImpl* cfh,
94 bool expose_blob_index, bool allow_refresh);
95
96 // Store some parameters so we can refresh the iterator at a later point
97 // with these same params
98 void StoreRefreshInfo(ColumnFamilyHandleImpl* cfh,
99 ReadCallback* read_callback, bool expose_blob_index) {
100 cfh_ = cfh;
101 read_callback_ = read_callback;
102 expose_blob_index_ = expose_blob_index;
103 }
104
105 private:
106 DBIter* db_iter_ = nullptr;
107 Arena arena_;
108 uint64_t sv_number_;
109 ColumnFamilyHandleImpl* cfh_ = nullptr;
110 ReadOptions read_options_;
111 ReadCallback* read_callback_;
112 bool expose_blob_index_ = false;
113 bool allow_refresh_ = true;
114 // If this is nullptr, it means the mutable memtable does not contain range
115 // tombstone when added under this DBIter.
116 std::unique_ptr<TruncatedRangeDelIterator>* memtable_range_tombstone_iter_ =
117 nullptr;
118};
