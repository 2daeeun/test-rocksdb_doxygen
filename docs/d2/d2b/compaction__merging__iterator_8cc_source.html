10 public:
11 CompactionMergingIterator(
12 const InternalKeyComparator* comparator, InternalIterator** children,
13 int n, bool is_arena_mode,
14 std::vector<std::pair<std::unique_ptr<TruncatedRangeDelIterator>,
15 std::unique_ptr<TruncatedRangeDelIterator>**>>&
16 range_tombstones)
17 : is_arena_mode_(is_arena_mode),
18 comparator_(comparator),
19 current_(nullptr),
20 minHeap_(CompactionHeapItemComparator(comparator_)),
21 pinned_iters_mgr_(nullptr) {
22 children_.resize(n);
23 for (int i = 0; i < n; i++) {
24 children_[i].level = i;
25 children_[i].iter.Set(children[i]);
26 assert(children_[i].type == HeapItem::ITERATOR);
27 }
28 assert(range_tombstones.size() == static_cast<size_t>(n));
29 for (auto& p : range_tombstones) {
30 range_tombstone_iters_.push_back(std::move(p.first));
31 }
32 pinned_heap_item_.resize(n);
33 for (int i = 0; i < n; ++i) {
34 if (range_tombstones[i].second) {
35 // for LevelIterator
36 *range_tombstones[i].second = &range_tombstone_iters_[i];
37 }
38 pinned_heap_item_[i].level = i;
39 pinned_heap_item_[i].type = HeapItem::DELETE_RANGE_START;
40 }
41 }

162 class CompactionHeapItemComparator {
163 public:
164 explicit CompactionHeapItemComparator(
165 const InternalKeyComparator* comparator)
166 : comparator_(comparator) {}
167
168 bool operator()(HeapItem* a, HeapItem* b) const {
169 int r = comparator_->Compare(a->key(), b->key());
170 // For each file, we assume all range tombstone start keys come before
171 // its file boundary sentinel key (file's meta.largest key).
172 // In the case when meta.smallest = meta.largest and range tombstone start
173 // key is truncated at meta.smallest, the start key will have op_type =
174 // kMaxValid to make it smaller (see TruncatedRangeDelIterator
175 // constructor). The following assertion validates this assumption.
176 assert(a->type == b->type || r != 0);
177 return r > 0;
178 }
179
180 private:
181 const InternalKeyComparator* comparator_;
182 };
183
184 using CompactionMinHeap = BinaryHeap<HeapItem*, CompactionHeapItemComparator>;
185 bool is_arena_mode_;
186 const InternalKeyComparator* comparator_;
187 // HeapItem for all child point iterators.
188 std::vector<HeapItem> children_;
189 // HeapItem for range tombstones. pinned_heap_item_[i] corresponds to the
190 // current range tombstone from range_tombstone_iters_[i].
191 std::vector<HeapItem> pinned_heap_item_;
192 // range_tombstone_iters_[i] contains range tombstones in the sorted run that
193 // corresponds to children_[i]. range_tombstone_iters_[i] ==
194 // nullptr means the sorted run of children_[i] does not have range
195 // tombstones (or the current SSTable does not have range tombstones in the
196 // case of LevelIterator).
197 std::vector<std::unique_ptr<TruncatedRangeDelIterator>>
198 range_tombstone_iters_;
199 // Used as value for range tombstone keys
200 std::string dummy_tombstone_val{};
201
202 // Skip file boundary sentinel keys.
203 void FindNextVisibleKey();
204
205 // top of minHeap_
206 HeapItem* current_;
207 // If any of the children have non-ok status, this is one of them.
208 Status status_;
209 CompactionMinHeap minHeap_;
210 PinnedIteratorsManager* pinned_iters_mgr_;
211 // Process a child that is not in the min heap.
212 // If valid, add to the min heap. Otherwise, check status.
213 void AddToMinHeapOrCheckStatus(HeapItem*);
214
215 HeapItem* CurrentForward() const {
216 return !minHeap_.empty() ? minHeap_.top() : nullptr;
217 }
218
219 void InsertRangeTombstoneAtLevel(size_t level) {
220 if (range_tombstone_iters_[level]->Valid()) {
221 pinned_heap_item_[level].SetTombstoneForCompaction(
222 range_tombstone_iters_[level]->start_key());
223 minHeap_.push(&pinned_heap_item_[level]);
224 }
225 }
226};

228void CompactionMergingIterator::SeekToFirst() {
229 minHeap_.clear();
230 status_ = Status::OK();
231 for (auto& child : children_) {
232 child.iter.SeekToFirst();
233 AddToMinHeapOrCheckStatus(&child);
234 }
235
236 for (size_t i = 0; i < range_tombstone_iters_.size(); ++i) {
237 if (range_tombstone_iters_[i]) {
238 range_tombstone_iters_[i]->SeekToFirst();
239 InsertRangeTombstoneAtLevel(i);
240 }
241 }
242
243 FindNextVisibleKey();
244 current_ = CurrentForward();
245}

247void CompactionMergingIterator::Seek(const Slice& target) {
248 minHeap_.clear();
249 status_ = Status::OK();
250 for (auto& child : children_) {
251 child.iter.Seek(target);
252 AddToMinHeapOrCheckStatus(&child);
253 }
254
255 ParsedInternalKey pik;
256 ParseInternalKey(target, &pik, false /* log_err_key */)
257 .PermitUncheckedError();
258 for (size_t i = 0; i < range_tombstone_iters_.size(); ++i) {
259 if (range_tombstone_iters_[i]) {
260 range_tombstone_iters_[i]->Seek(pik.user_key);
261 // For compaction, output keys should all be after seek target.
262 while (range_tombstone_iters_[i]->Valid() &&
263 comparator_->Compare(range_tombstone_iters_[i]->start_key(), pik) <
264 0) {
265 range_tombstone_iters_[i]->Next();
266 }
267 InsertRangeTombstoneAtLevel(i);
268 }
269 }
270
271 FindNextVisibleKey();
272 current_ = CurrentForward();
273}

275void CompactionMergingIterator::Next() {
276 assert(Valid());
277 // For the heap modifications below to be correct, current_ must be the
278 // current top of the heap.
279 assert(current_ == CurrentForward());
280 // as the current points to the current record. move the iterator forward.
281 if (current_->type == HeapItem::ITERATOR) {
282 current_->iter.Next();
283 if (current_->iter.Valid()) {
284 // current is still valid after the Next() call above. Call
285 // replace_top() to restore the heap property. When the same child
286 // iterator yields a sequence of keys, this is cheap.
287 assert(current_->iter.status().ok());
288 minHeap_.replace_top(current_);
289 } else {
290 // current stopped being valid, remove it from the heap.
291 considerStatus(current_->iter.status());
292 minHeap_.pop();
293 }
294 } else {
295 assert(current_->type == HeapItem::DELETE_RANGE_START);
296 size_t level = current_->level;
297 assert(range_tombstone_iters_[level]);
298 range_tombstone_iters_[level]->Next();
299 if (range_tombstone_iters_[level]->Valid()) {
300 pinned_heap_item_[level].SetTombstoneForCompaction(
301 range_tombstone_iters_[level]->start_key());
302 minHeap_.replace_top(&pinned_heap_item_[level]);
303 } else {
304 minHeap_.pop();
305 }
306 }
307 FindNextVisibleKey();
308 current_ = CurrentForward();
309}

311void CompactionMergingIterator::FindNextVisibleKey() {
312 while (!minHeap_.empty()) {
313 HeapItem* current = minHeap_.top();
314 // IsDeleteRangeSentinelKey() here means file boundary sentinel keys.
315 if (current->type != HeapItem::ITERATOR ||
316 !current->iter.IsDeleteRangeSentinelKey()) {
317 return;
318 }
319 // range tombstone start keys from the same SSTable should have been
320 // exhausted
321 assert(!range_tombstone_iters_[current->level] ||
322 !range_tombstone_iters_[current->level]->Valid());
323 // current->iter is a LevelIterator, and it enters a new SST file in the
324 // Next() call here.
325 current->iter.Next();
326 if (current->iter.Valid()) {
327 assert(current->iter.status().ok());
328 minHeap_.replace_top(current);
329 } else {
330 considerStatus(current->iter.status());
331 minHeap_.pop();
332 }
333 if (range_tombstone_iters_[current->level]) {
334 InsertRangeTombstoneAtLevel(current->level);
335 }
336 }
337}
338
339void CompactionMergingIterator::AddToMinHeapOrCheckStatus(HeapItem* child) {
340 if (child->iter.Valid()) {
341 assert(child->iter.status().ok());
342 minHeap_.push(child);
343 } else {
344 considerStatus(child->iter.status());
345 }
346}

348InternalIterator* NewCompactionMergingIterator(
349 const InternalKeyComparator* comparator, InternalIterator** children, int n,
350 std::vector<std::pair<std::unique_ptr<TruncatedRangeDelIterator>,
351 std::unique_ptr<TruncatedRangeDelIterator>**>>&
352 range_tombstone_iters,
353 Arena* arena) {
354 assert(n >= 0);
355 if (n == 0) {
356 return NewEmptyInternalIterator<Slice>(arena);
357 } else {
358 if (arena == nullptr) {
359 return new CompactionMergingIterator(comparator, children, n,
360 false /* is_arena_mode */,
361 range_tombstone_iters);
362 } else {
363 auto mem = arena->AllocateAligned(sizeof(CompactionMergingIterator));
364 return new (mem) CompactionMergingIterator(comparator, children, n,
365 true /* is_arena_mode */,
366 range_tombstone_iters);
367 }
368 }
