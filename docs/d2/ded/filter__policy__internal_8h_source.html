21// A class that takes a bunch of keys, then generates filter
22class FilterBitsBuilder {
23 public:
24 virtual ~FilterBitsBuilder() {}
25
26 // Add a key (or prefix) to the filter. Typically, a builder will keep
27 // a set of 64-bit key hashes and only build the filter in Finish
28 // when the final number of keys is known. Keys are added in sorted order
29 // and duplicated keys are possible, so typically, the builder will
30 // only add this key if its hash is different from the most recently
31 // added.
32 virtual void AddKey(const Slice& key) = 0;
33
34 // Add two entries to the filter, typically a key and, as the alternate,
35 // its prefix. This differs from AddKey(key); AddKey(alt); in that there
36 // is extra state for de-duplicating successive `alt` entries, as well
37 // as successive `key` entries. And there is de-duplication between `key`
38 // and `alt` entries, even in adjacent calls, because a whole key might
39 // be its own prefix. More specifically,
40 // AddKey(k1);
41 // AddKeyAndAlt(k2, a2); // de-dup k2<>k1, k2<>a2, a2<>k1
42 // AddKeyAndAlt(k3, a3); // de-dup k3<>k2, a3<>a2, k3<>a2, a3<>k2
43 // AddKey(k4); // de-dup k4<>k3 BUT NOT k4<>a3
44 virtual void AddKeyAndAlt(const Slice& key, const Slice& alt) = 0;
45
46 // Called by RocksDB before Finish to populate
47 // TableProperties::num_filter_entries, so should represent the
48 // number of unique keys (and/or prefixes) added. MUST return 0
49 // if and only if none have been added, but otherwise can be estimated.
50 virtual size_t EstimateEntriesAdded() = 0;
51
52 // Generate the filter using the keys that are added
53 // The return value of this function would be the filter bits,
54 // The ownership of actual data is set to buf
55 virtual Slice Finish(std::unique_ptr<const char[]>* buf) = 0;
56
57 // Similar to Finish(std::unique_ptr<const char[]>* buf), except that
58 // for a non-null status pointer argument, it will point to
59 // Status::Corruption() when there is any corruption during filter
60 // construction or Status::OK() otherwise.
61 //
62 // WARNING: do not use a filter resulted from a corrupted construction
63 // TODO: refactor this to have a better signature, consolidate
64 virtual Slice Finish(std::unique_ptr<const char[]>* buf,
65 Status* /* status */) {
66 return Finish(buf);
67 }
68
69 // Verify the filter returned from calling FilterBitsBuilder::Finish.
70 // The function returns Status::Corruption() if there is any corruption in the
71 // constructed filter or Status::OK() otherwise.
72 //
73 // Implementations should normally consult
74 // FilterBuildingContext::table_options.detect_filter_construct_corruption
75 // to determine whether to perform verification or to skip by returning
76 // Status::OK(). The decision is left to the FilterBitsBuilder so that
77 // verification prerequisites before PostVerify can be skipped when not
78 // configured.
79 //
80 // RocksDB internal will always call MaybePostVerify() on the filter after
81 // it is returned from calling FilterBitsBuilder::Finish
82 // except for FilterBitsBuilder::Finish resulting a corruption
83 // status, which indicates the filter is already in a corrupted state and
84 // there is no need to post-verify
85 virtual Status MaybePostVerify(const Slice& /* filter_content */) {
86 return Status::OK();
87 }
88
89 // Approximate the number of keys that can be added and generate a filter
90 // <= the specified number of bytes. Callers (including RocksDB) should
91 // only use this result for optimizing performance and not as a guarantee.
92 virtual size_t ApproximateNumEntries(size_t bytes) = 0;
93};

95// A class that checks if a key can be in filter
96// It should be initialized by Slice generated by BitsBuilder
97class FilterBitsReader {
98 public:
99 virtual ~FilterBitsReader() {}
100
101 // Check if the entry match the bits in filter
102 virtual bool MayMatch(const Slice& entry) = 0;
103
104 // Check if an array of entries match the bits in filter
105 virtual void MayMatch(int num_keys, Slice** keys, bool* may_match) {
106 for (int i = 0; i < num_keys; ++i) {
107 may_match[i] = MayMatch(*keys[i]);
108 }
109 }
110};

112// Exposes any extra information needed for testing built-in

114class BuiltinFilterBitsBuilder : public FilterBitsBuilder {

116 // Calculate number of bytes needed for a new filter, including
117 // metadata. Passing the result to ApproximateNumEntries should
118 // (ideally, usually) return >= the num_entry passed in.
119 // When optimize_filters_for_memory is enabled, this function

138// Base class for RocksDB built-in filter policies. This provides the
139// ability to read all kinds of built-in filters (so that old filters can
140// be used even when you change between built-in policies).
141class BuiltinFilterPolicy : public FilterPolicy {
142 public: // overrides
143 // Read metadata to determine what kind of FilterBitsReader is needed
144 // and return a new one. This must successfully process any filter data
145 // generated by a built-in FilterBitsBuilder, regardless of the impl
146 // chosen for this BloomFilterPolicy.
147 FilterBitsReader* GetFilterBitsReader(const Slice& contents) const override;
148 static const char* kClassName();
149 bool IsInstanceOf(const std::string& id) const override;
150 // All variants of BuiltinFilterPolicy can read each others filters.
151 const char* CompatibilityName() const override;
152 static const char* kCompatibilityName();
153
154 public: // new
155 // An internal function for the implementation of
156 // BuiltinFilterBitsReader::GetFilterBitsReader without requiring an instance
157 // or working around potential virtual overrides.
158 static BuiltinFilterBitsReader* GetBuiltinFilterBitsReader(
159 const Slice& contents);
160
161 // Returns a new FilterBitsBuilder from the filter_policy in
162 // table_options of a context, or nullptr if not applicable.
163 // (An internal convenience function to save boilerplate.)
164 static FilterBitsBuilder* GetBuilderFromContext(const FilterBuildingContext&);
165
166 private:
167 // For Bloom filter implementation(s)
168 static BuiltinFilterBitsReader* GetBloomBitsReader(const Slice& contents);
169
170 // For Ribbon filter implementation(s)
171 static BuiltinFilterBitsReader* GetRibbonBitsReader(const Slice& contents);
172};

174// A "read only" filter policy used for backward compatibility with old
175// OPTIONS files, which did not specifying a Bloom configuration, just
176// "rocksdb.BuiltinBloomFilter". Although this can read existing filters,
177// this policy does not build new filters, so new SST files generated
178// under the policy will get no filters (like nullptr FilterPolicy).
179// This class is considered internal API and subject to change.
180class ReadOnlyBuiltinFilterPolicy : public BuiltinFilterPolicy {
181 public:
182 const char* Name() const override { return kClassName(); }
183 static const char* kClassName();
184
185 // Does not write filters.
186 FilterBitsBuilder* GetBuilderWithContext(
187 const FilterBuildingContext&) const override {
188 return nullptr;
189 }
190};

192// RocksDB built-in filter policy for Bloom or Bloom-like filters including
193// Ribbon filters.
194// This class is considered internal API and subject to change.
195// See NewBloomFilterPolicy and NewRibbonFilterPolicy.
196class BloomLikeFilterPolicy : public BuiltinFilterPolicy {
197 public:
198 explicit BloomLikeFilterPolicy(double bits_per_key);
199
200 ~BloomLikeFilterPolicy() override;
201 static const char* kClassName();
202 bool IsInstanceOf(const std::string& id) const override;
203
204 std::string GetId() const override;
205
206 // Essentially for testing only: configured millibits/key
207 int GetMillibitsPerKey() const { return millibits_per_key_; }
208 // Essentially for testing only: legacy whole bits/key
209 int GetWholeBitsPerKey() const { return whole_bits_per_key_; }
210
211 // All the different underlying implementations that a BloomLikeFilterPolicy
212 // might use, as a configuration string name for a testing mode for
213 // "always use this implementation." Only appropriate for unit tests.
214 static const std::vector<std::string>& GetAllFixedImpls();
215
216 // Convenience function for creating by name for fixed impls
217 static std::shared_ptr<const FilterPolicy> Create(const std::string& name,
218 double bits_per_key);
219
220 protected:
221 // Some implementations used by aggregating policies
222 FilterBitsBuilder* GetLegacyBloomBuilderWithContext(
223 const FilterBuildingContext& context) const;
224 FilterBitsBuilder* GetFastLocalBloomBuilderWithContext(
225 const FilterBuildingContext& context) const;
226 FilterBitsBuilder* GetStandard128RibbonBuilderWithContext(
227 const FilterBuildingContext& context) const;
228
229 std::string GetBitsPerKeySuffix() const;
230
231 private:
232 // Bits per key settings are for configuring Bloom filters.
233
234 // Newer filters support fractional bits per key. For predictable behavior
235 // of 0.001-precision values across floating point implementations, we
236 // round to thousandths of a bit (on average) per key.
237 int millibits_per_key_;
238
239 // Older filters round to whole number bits per key. (There *should* be no
240 // compatibility issue with fractional bits per key, but preserving old
241 // behavior with format_version < 5 just in case.)
242 int whole_bits_per_key_;
243
244 // For configuring Ribbon filter: a desired value for 1/fp_rate. For
245 // example, 100 -> 1% fp rate.
246 double desired_one_in_fp_rate_;
247
248 // Whether relevant warnings have been logged already. (Remember so we
249 // only report once per BloomFilterPolicy instance, to keep the noise down.)
250 mutable std::atomic<bool> warned_;
251
252 // State for implementing optimize_filters_for_memory. Essentially, this
253 // tracks a surplus or deficit in total FP rate of filters generated by
254 // builders under this policy vs. what would have been generated without
255 // optimize_filters_for_memory.
256 //
257 // To avoid floating point weirdness, the actual value is
258 // Sum over all generated filters f:
259 // (predicted_fp_rate(f) - predicted_fp_rate(f|o_f_f_m=false)) * 2^32
260 mutable std::atomic<int64_t> aggregate_rounding_balance_;
261};
