59class DestroyAllCompactionFilter : public CompactionFilter {
60 public:
61 DestroyAllCompactionFilter() = default;
62
63 bool Filter(int /*level*/, const Slice& /*key*/, const Slice& existing_value,
64 std::string* /*new_value*/,
65 bool* /*value_changed*/) const override {
66 return existing_value.ToString() == "destroy";
67 }
68
69 const char* Name() const override { return "DestroyAllCompactionFilter"; }

72class LogCompactionFilter : public CompactionFilter {
73 public:
74 const char* Name() const override { return "LogCompactionFilter"; }
75
76 bool Filter(int level, const Slice& key, const Slice& /*existing_value*/,
77 std::string* /*new_value*/,
78 bool* /*value_changed*/) const override {
79 key_level_[key.ToString()] = level;
80 return false;
81 }
82
83 void Reset() { key_level_.clear(); }
84
85 size_t NumKeys() const { return key_level_.size(); }
86
87 int KeyLevel(const Slice& key) {
88 auto it = key_level_.find(key.ToString());
89 if (it == key_level_.end()) {
90 return -1;
91 }
92 return it->second;
93 }
94
95 private:
96 mutable std::map<std::string, int> key_level_;
97};

99TEST_F(ManualCompactionTest, CompactTouchesAllKeys) {
100 for (int iter = 0; iter < 2; ++iter) {
101 DB* db;
102 Options options;
103 if (iter == 0) { // level compaction
104 options.num_levels = 3;
105 options.compaction_style = CompactionStyle::kCompactionStyleLevel;
106 } else { // universal compaction
107 options.compaction_style = CompactionStyle::kCompactionStyleUniversal;
108 }
109 options.create_if_missing = true;
110 options.compression = CompressionType::kNoCompression;
111 options.compaction_filter = new DestroyAllCompactionFilter();
112 ASSERT_OK(DB::Open(options, dbname_, &db));
113
114 ASSERT_OK(db->Put(WriteOptions(), Slice("key1"), Slice("destroy")));
115 ASSERT_OK(db->Put(WriteOptions(), Slice("key2"), Slice("destroy")));
116 ASSERT_OK(db->Put(WriteOptions(), Slice("key3"), Slice("value3")));
117 ASSERT_OK(db->Put(WriteOptions(), Slice("key4"), Slice("destroy")));
118
119 Slice key4("key4");
120 ASSERT_OK(db->CompactRange(CompactRangeOptions(), nullptr, &key4));
121 Iterator* itr = db->NewIterator(ReadOptions());
122 itr->SeekToFirst();
123 ASSERT_TRUE(itr->Valid());
124 ASSERT_EQ("key3", itr->key().ToString());
125 itr->Next();
126 ASSERT_TRUE(!itr->Valid());
127 ASSERT_OK(itr->status());
128 delete itr;
129
130 delete options.compaction_filter;
131 delete db;
132 ASSERT_OK(DestroyDB(dbname_, options));
133 }
134}

136TEST_F(ManualCompactionTest, Test) {
137 // Open database. Disable compression since it affects the creation
138 // of layers and the code below is trying to test against a very
139 // specific scenario.
140 DB* db;
141 Options db_options;
142 db_options.write_buffer_size = 1024;
143 db_options.create_if_missing = true;
144 db_options.compression = CompressionType::kNoCompression;
145 ASSERT_OK(DB::Open(db_options, dbname_, &db));
146
147 // create first key range
148 WriteBatch batch;
149 for (int i = 0; i < kNumKeys; i++) {
150 ASSERT_OK(batch.Put(Key1(i), "value for range 1 key"));
151 }
152 ASSERT_OK(db->Write(WriteOptions(), &batch));
153
154 // create second key range
155 batch.Clear();
156 for (int i = 0; i < kNumKeys; i++) {
157 ASSERT_OK(batch.Put(Key2(i), "value for range 2 key"));
158 }
159 ASSERT_OK(db->Write(WriteOptions(), &batch));
160
161 // delete second key range
162 batch.Clear();
163 for (int i = 0; i < kNumKeys; i++) {
164 ASSERT_OK(batch.Delete(Key2(i)));
165 }
166 ASSERT_OK(db->Write(WriteOptions(), &batch));
167
168 // compact database
169 std::string start_key = Key1(0);
170 std::string end_key = Key1(kNumKeys - 1);
171 Slice least(start_key.data(), start_key.size());
172 Slice greatest(end_key.data(), end_key.size());
173
174 // commenting out the line below causes the example to work correctly
175 ASSERT_OK(db->CompactRange(CompactRangeOptions(), &least, &greatest));
176
177 // count the keys
178 Iterator* iter = db->NewIterator(ReadOptions());
179 int num_keys = 0;
180 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
181 num_keys++;
182 }
183 ASSERT_OK(iter->status());
184 delete iter;
185 ASSERT_EQ(kNumKeys, num_keys) << "Bad number of keys";
186
187 // close database
188 delete db;
189 ASSERT_OK(DestroyDB(dbname_, Options()));
190}

192TEST_F(ManualCompactionTest, SkipLevel) {
193 DB* db;
194 Options options;
195 options.level_compaction_dynamic_level_bytes = false;
196 options.num_levels = 3;
197 // Initially, flushed L0 files won't exceed 100.
198 options.level0_file_num_compaction_trigger = 100;
199 options.compaction_style = CompactionStyle::kCompactionStyleLevel;
200 options.create_if_missing = true;
201 options.compression = CompressionType::kNoCompression;
202 LogCompactionFilter* filter = new LogCompactionFilter();
203 options.compaction_filter = filter;
204 ASSERT_OK(DB::Open(options, dbname_, &db));
205
206 WriteOptions wo;
207 FlushOptions fo;
208 ASSERT_OK(db->Put(wo, "1", ""));
209 ASSERT_OK(db->Flush(fo));
210 ASSERT_OK(db->Put(wo, "2", ""));
211 ASSERT_OK(db->Flush(fo));
212 ASSERT_OK(db->Put(wo, "4", ""));
213 ASSERT_OK(db->Put(wo, "8", ""));
214 ASSERT_OK(db->Flush(fo));
215
216 {
217 // L0: 1, 2, [4, 8]
218 // no file has keys in range [5, 7]
219 Slice start("5");
220 Slice end("7");
221 filter->Reset();
222 ASSERT_OK(db->CompactRange(CompactRangeOptions(), &start, &end));
223 ASSERT_EQ(0, filter->NumKeys());
224 }
225
226 {
227 // L0: 1, 2, [4, 8]
228 // [3, 7] overlaps with 4 in L0
229 Slice start("3");
230 Slice end("7");
231 filter->Reset();
232 ASSERT_OK(db->CompactRange(CompactRangeOptions(), &start, &end));
233 ASSERT_EQ(2, filter->NumKeys());
234 ASSERT_EQ(0, filter->KeyLevel("4"));
235 ASSERT_EQ(0, filter->KeyLevel("8"));
236 }
237
238 {
239 // L0: 1, 2
240 // L1: [4, 8]
241 // no file has keys in range (-inf, 0]
242 Slice end("0");
243 filter->Reset();
244 ASSERT_OK(db->CompactRange(CompactRangeOptions(), nullptr, &end));
245 ASSERT_EQ(0, filter->NumKeys());
246 }
247
248 {
249 // L0: 1, 2
250 // L1: [4, 8]
251 // no file has keys in range [9, inf)
252 Slice start("9");
253 filter->Reset();
254 ASSERT_OK(db->CompactRange(CompactRangeOptions(), &start, nullptr));
255 ASSERT_EQ(0, filter->NumKeys());
256 }
257
258 {
259 // L0: 1, 2
260 // L1: [4, 8]
261 // [2, 2] overlaps with 2 in L0
262 Slice start("2");
263 Slice end("2");
264 filter->Reset();
265 ASSERT_OK(db->CompactRange(CompactRangeOptions(), &start, &end));
266 ASSERT_EQ(1, filter->NumKeys());
267 ASSERT_EQ(0, filter->KeyLevel("2"));
268 }
269
270 {
271 // L0: 1
272 // L1: 2, [4, 8]
273 // [2, 5] overlaps with 2 and [4, 8) in L1, skip L0
274 Slice start("2");
275 Slice end("5");
276 filter->Reset();
277 ASSERT_OK(db->CompactRange(CompactRangeOptions(), &start, &end));
278 ASSERT_EQ(3, filter->NumKeys());
279 ASSERT_EQ(1, filter->KeyLevel("2"));
280 ASSERT_EQ(1, filter->KeyLevel("4"));
281 ASSERT_EQ(1, filter->KeyLevel("8"));
282 }
283
284 {
285 // L0: 1
286 // L1: [2, 4, 8]
287 // [0, inf) overlaps all files
288 Slice start("0");
289 filter->Reset();
290 ASSERT_OK(db->CompactRange(CompactRangeOptions(), &start, nullptr));
291 ASSERT_EQ(4, filter->NumKeys());
292 // 1 is first compacted from L0 to L1, and then L1 intra level compaction
293 // compacts [2, 4, 8] only.
294 ASSERT_EQ(0, filter->KeyLevel("1"));
295 ASSERT_EQ(1, filter->KeyLevel("2"));
296 ASSERT_EQ(1, filter->KeyLevel("4"));
297 ASSERT_EQ(1, filter->KeyLevel("8"));
298 }
299
300 delete filter;
301 delete db;
302 ASSERT_OK(DestroyDB(dbname_, options));
303}
