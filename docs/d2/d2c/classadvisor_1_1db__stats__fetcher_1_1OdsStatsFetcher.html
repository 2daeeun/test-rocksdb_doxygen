213 def parse_rapido_output(self):
214 # Output looks like the following:
215 # <entity_name>\t<key_name>\t[[ts, value], [ts, value], ...]
216 # ts = timestamp; value = value of key_name in entity_name at time ts
217 self.keys_ts = {}
218 with open(self.OUTPUT_FILE) as fp:
219 for line in fp:
220 token_list = line.strip().split("\t")
221 entity = token_list[0]
222 key = token_list[1]
223 if entity not in self.keys_ts:
224 self.keys_ts[entity] = {}
225 if key not in self.keys_ts[entity]:
226 self.keys_ts[entity][key] = {}
227 list_of_lists = [
228 self._get_time_value_pair(pair_string)
229 for pair_string in token_list[2].split("],")
230 ]
231 value = {pair[0]: pair[1] for pair in list_of_lists}
232 self.keys_ts[entity][key] = value
