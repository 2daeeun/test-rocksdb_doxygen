24class MemoryAllocatorTest
25 : public testing::Test,
26 public ::testing::WithParamInterface<std::tuple<std::string, bool>> {
27 public:
28 MemoryAllocatorTest() {
29 std::tie(id_, supported_) = GetParam();
30 Status s =
31 MemoryAllocator::CreateFromString(ConfigOptions(), id_, &allocator_);
32 EXPECT_EQ(supported_, s.ok());
33 }
34 bool IsSupported() { return supported_; }
35
36 std::shared_ptr<MemoryAllocator> allocator_;
37 std::string id_;
38
39 private:
40 bool supported_;
41};

43TEST_P(MemoryAllocatorTest, Allocate) {
44 if (!IsSupported()) {
45 return;
46 }
47 void* p = allocator_->Allocate(1024);
48 ASSERT_NE(p, nullptr);
49 size_t size = allocator_->UsableSize(p, 1024);
50 ASSERT_GE(size, 1024);
51 allocator_->Deallocate(p);
52}

54TEST_P(MemoryAllocatorTest, CreateAllocator) {
55 ConfigOptions config_options;
56 config_options.ignore_unknown_options = false;
57 config_options.ignore_unsupported_options = false;
58 std::shared_ptr<MemoryAllocator> orig, copy;
59 Status s = MemoryAllocator::CreateFromString(config_options, id_, &orig);
60 if (!IsSupported()) {
61 ASSERT_TRUE(s.IsNotSupported());
62 } else {
63 ASSERT_OK(s);
64 ASSERT_NE(orig, nullptr);
65 std::string str = orig->ToString(config_options);
66 ASSERT_OK(MemoryAllocator::CreateFromString(config_options, str, &copy));
67 ASSERT_EQ(orig, copy);
68 }
69}

71TEST_P(MemoryAllocatorTest, DatabaseBlockCache) {
72 if (!IsSupported()) {
73 // Check if a memory node is available for allocation
74 }
75
76 // Create database with block cache using the MemoryAllocator
77 Options options;
78 std::string dbname = test::PerThreadDBPath("allocator_test");
79 ASSERT_OK(DestroyDB(dbname, options));
80
81 options.create_if_missing = true;
82 BlockBasedTableOptions table_options;
83 auto cache = NewLRUCache(1024 * 1024, 6, false, 0.0, allocator_);
84 table_options.block_cache = cache;
85 options.table_factory.reset(NewBlockBasedTableFactory(table_options));
86 DB* db = nullptr;
87 Status s = DB::Open(options, dbname, &db);
88 ASSERT_OK(s);
89 ASSERT_NE(db, nullptr);
90 ASSERT_LE(cache->GetUsage(), 104); // Cache will contain stats
91
92 // Write 2kB (200 values, each 10 bytes)
93 int num_keys = 200;
94 WriteOptions wo;
95 std::string val = "0123456789";
96 for (int i = 0; i < num_keys; i++) {
97 std::string key = std::to_string(i);
98 s = db->Put(wo, Slice(key), Slice(val));
99 ASSERT_OK(s);
100 }
101 ASSERT_OK(db->Flush(FlushOptions())); // Flush all data from memtable so that
102 // reads are from block cache
103
104 // Read and check block cache usage
105 ReadOptions ro;
106 std::string result;
107 for (int i = 0; i < num_keys; i++) {
108 std::string key = std::to_string(i);
109 s = db->Get(ro, key, &result);
110 ASSERT_OK(s);
111 ASSERT_EQ(result, val);
112 }
113 ASSERT_GT(cache->GetUsage(), 2000);
114
115 // Close database
116 s = db->Close();
117 ASSERT_OK(s);
118 delete db;
119 ASSERT_OK(DestroyDB(dbname, options));
120}

131TEST_F(CreateMemoryAllocatorTest, JemallocOptionsTest) {
132 std::shared_ptr<MemoryAllocator> allocator;
133 std::string id = std::string("id=") + JemallocNodumpAllocator::kClassName();
134 Status s = MemoryAllocator::CreateFromString(config_options_, id, &allocator);
135 if (!JemallocNodumpAllocator::IsSupported()) {
136 ASSERT_NOK(s);
137 ROCKSDB_GTEST_BYPASS("JEMALLOC not supported");
138 return;
139 }
140 ASSERT_OK(s);
141 ASSERT_NE(allocator, nullptr);
142 JemallocAllocatorOptions jopts;
143 auto opts = allocator->GetOptions<JemallocAllocatorOptions>();
144 ASSERT_NE(opts, nullptr);
145 ASSERT_EQ(opts->limit_tcache_size, jopts.limit_tcache_size);
146 ASSERT_EQ(opts->tcache_size_lower_bound, jopts.tcache_size_lower_bound);
147 ASSERT_EQ(opts->tcache_size_upper_bound, jopts.tcache_size_upper_bound);
148
149 ASSERT_NOK(MemoryAllocator::CreateFromString(
150 config_options_,
151 id + "; limit_tcache_size=true; tcache_size_lower_bound=4096; "
152 "tcache_size_upper_bound=1024",
153 &allocator));
154 ASSERT_OK(MemoryAllocator::CreateFromString(
155 config_options_,
156 id + "; limit_tcache_size=false; tcache_size_lower_bound=4096; "
157 "tcache_size_upper_bound=1024",
158 &allocator));
159 opts = allocator->GetOptions<JemallocAllocatorOptions>();
160 ASSERT_NE(opts, nullptr);
161 ASSERT_EQ(opts->limit_tcache_size, false);
162 ASSERT_EQ(opts->tcache_size_lower_bound, 4096U);
163 ASSERT_EQ(opts->tcache_size_upper_bound, 1024U);
164 ASSERT_OK(MemoryAllocator::CreateFromString(
165 config_options_,
166 id + "; limit_tcache_size=true; tcache_size_upper_bound=4096; "
167 "tcache_size_lower_bound=1024",
168 &allocator));
169 opts = allocator->GetOptions<JemallocAllocatorOptions>();
170 ASSERT_NE(opts, nullptr);
171 ASSERT_EQ(opts->limit_tcache_size, true);
172 ASSERT_EQ(opts->tcache_size_lower_bound, 1024U);
173 ASSERT_EQ(opts->tcache_size_upper_bound, 4096U);
174}

176TEST_F(CreateMemoryAllocatorTest, NewJemallocNodumpAllocator) {
177 JemallocAllocatorOptions jopts;
178 std::shared_ptr<MemoryAllocator> allocator;
179
180 jopts.limit_tcache_size = true;
181 jopts.tcache_size_lower_bound = 2 * 1024;
182 jopts.tcache_size_upper_bound = 1024;
183
184 ASSERT_NOK(NewJemallocNodumpAllocator(jopts, nullptr));
185 Status s = NewJemallocNodumpAllocator(jopts, &allocator);
186 std::string msg;
187 if (!JemallocNodumpAllocator::IsSupported(&msg)) {
188 ASSERT_NOK(s);
189 ROCKSDB_GTEST_BYPASS("JEMALLOC not supported");
190 return;
191 }
192 ASSERT_NOK(s); // Invalid options
193 ASSERT_EQ(allocator, nullptr);
194
195 jopts.tcache_size_upper_bound = 4 * 1024;
196 ASSERT_OK(NewJemallocNodumpAllocator(jopts, &allocator));
197 ASSERT_NE(allocator, nullptr);
198 auto opts = allocator->GetOptions<JemallocAllocatorOptions>();
199 ASSERT_EQ(opts->tcache_size_upper_bound, jopts.tcache_size_upper_bound);
200 ASSERT_EQ(opts->tcache_size_lower_bound, jopts.tcache_size_lower_bound);
201 ASSERT_EQ(opts->limit_tcache_size, jopts.limit_tcache_size);
202
203 jopts.limit_tcache_size = false;
204 ASSERT_OK(NewJemallocNodumpAllocator(jopts, &allocator));
205 ASSERT_NE(allocator, nullptr);
206 opts = allocator->GetOptions<JemallocAllocatorOptions>();
207 ASSERT_EQ(opts->tcache_size_upper_bound, jopts.tcache_size_upper_bound);
208 ASSERT_EQ(opts->tcache_size_lower_bound, jopts.tcache_size_lower_bound);
209 ASSERT_EQ(opts->limit_tcache_size, jopts.limit_tcache_size);
210}
