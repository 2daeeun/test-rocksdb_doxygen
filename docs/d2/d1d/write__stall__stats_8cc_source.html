14const std::string& WriteStallCauseToHyphenString(WriteStallCause cause) {
15 static const std::string kMemtableLimit = "memtable-limit";
16 static const std::string kL0FileCountLimit = "l0-file-count-limit";
17 static const std::string kPendingCompactionBytes = "pending-compaction-bytes";
18 static const std::string kWriteBufferManagerLimit =
19 "write-buffer-manager-limit";
20 switch (cause) {
21 case WriteStallCause::kMemtableLimit:
22 return kMemtableLimit;
23 case WriteStallCause::kL0FileCountLimit:
24 return kL0FileCountLimit;
25 case WriteStallCause::kPendingCompactionBytes:
26 return kPendingCompactionBytes;
27 case WriteStallCause::kWriteBufferManagerLimit:
28 return kWriteBufferManagerLimit;
29 default:
30 break;
31 }
32 return InvalidWriteStallHyphenString();
33}

110bool isCFScopeWriteStallCause(WriteStallCause cause) {

156std::string WriteStallStatsMapKeys::CauseConditionCount(
157 WriteStallCause cause, WriteStallCondition condition) {
158 std::string cause_condition_count_name;
159
160 std::string cause_name;
161 if (isCFScopeWriteStallCause(cause) || isDBScopeWriteStallCause(cause)) {
162 cause_name = WriteStallCauseToHyphenString(cause);
163 } else {
164 assert(false);
165 return "";
166 }
167
168 const std::string& condition_name =
169 WriteStallConditionToHyphenString(condition);
170
171 cause_condition_count_name.reserve(cause_name.size() + 1 +
172 condition_name.size());
173 cause_condition_count_name.append(cause_name);
174 cause_condition_count_name.append("-");
175 cause_condition_count_name.append(condition_name);
176
177 return cause_condition_count_name;
178}
