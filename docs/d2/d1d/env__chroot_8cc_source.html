31Status ChrootFileSystem::PrepareOptions(const ConfigOptions& options) {
32 Status s = FileSystemWrapper::PrepareOptions(options);
33 if (!s.ok()) {
34 return s;
35 } else if (chroot_dir_.empty()) {
36 s = Status::InvalidArgument("ChRootFileSystem requires a chroot dir");
37 } else {
38 s = target_->FileExists(chroot_dir_, IOOptions(), nullptr);
39 }
40 if (s.ok()) {
41#if defined(OS_AIX)
42 char resolvedName[PATH_MAX];
43 char* real_chroot_dir = realpath(chroot_dir_.c_str(), resolvedName);
44#else
45 char* real_chroot_dir = realpath(chroot_dir_.c_str(), nullptr);
46#endif
47 // chroot_dir must exist so realpath() returns non-nullptr.
48 assert(real_chroot_dir != nullptr);
49 chroot_dir_ = real_chroot_dir;
50#if !defined(OS_AIX)
51 free(real_chroot_dir);
52#endif
53 }
54 return s;
55}

70// Returns status and expanded absolute path including the chroot directory.
71// Checks whether the provided path breaks out of the chroot. If it returns
72// non-OK status, the returned path should not be used.
73std::pair<IOStatus, std::string> ChrootFileSystem::EncodePath(
74 const std::string& path) {
75 if (path.empty() || path[0] != '/') {
76 return {IOStatus::InvalidArgument(path, "Not an absolute path"), ""};
77 }
78 std::pair<IOStatus, std::string> res;
79 res.second = chroot_dir_ + path;
80#if defined(OS_AIX)
81 char resolvedName[PATH_MAX];
82 char* normalized_path = realpath(res.second.c_str(), resolvedName);
83#else
84 char* normalized_path = realpath(res.second.c_str(), nullptr);
85#endif
86 if (normalized_path == nullptr) {
87 res.first = IOStatus::NotFound(res.second, errnoStr(errno).c_str());
88 } else if (strlen(normalized_path) < chroot_dir_.size() ||
89 strncmp(normalized_path, chroot_dir_.c_str(),
90 chroot_dir_.size()) != 0) {
91 res.first = IOStatus::IOError(res.second,
92 "Attempted to access path outside chroot");
93 } else {
94 res.first = IOStatus::OK();
95 }
96#if !defined(OS_AIX)
97 free(normalized_path);
98#endif
99 return res;
100}

102// Similar to EncodePath() except assumes the basename in the path hasn't been
103// created yet.
104std::pair<IOStatus, std::string> ChrootFileSystem::EncodePathWithNewBasename(
105 const std::string& path) {
106 if (path.empty() || path[0] != '/') {
107 return {IOStatus::InvalidArgument(path, "Not an absolute path"), ""};
108 }
109 // Basename may be followed by trailing slashes
110 size_t final_idx = path.find_last_not_of('/');
111 if (final_idx == std::string::npos) {
112 // It's only slashes so no basename to extract
113 return EncodePath(path);
114 }
115
116 // Pull off the basename temporarily since realname(3) (used by
117 // EncodePath()) requires a path that exists
118 size_t base_sep = path.rfind('/', final_idx);
119 auto status_and_enc_path = EncodePath(path.substr(0, base_sep + 1));
120 status_and_enc_path.second.append(path.substr(base_sep + 1));
121 return status_and_enc_path;
122}

135Env* NewChrootEnv(Env* base_env, const std::string& chroot_dir) {
136 if (!base_env->FileExists(chroot_dir).ok()) {
137 return nullptr;
138 }
139 auto chroot_fs = NewChrootFileSystem(base_env->GetFileSystem(), chroot_dir);
140 if (chroot_fs != nullptr) {
141 return new CompositeEnvWrapper(base_env, chroot_fs);
142 } else {
143 return nullptr;
144 }
