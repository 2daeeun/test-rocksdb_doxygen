28void Multiplier(void* arg1, void* arg2) {
29 int* res = static_cast<int*>(arg1);
30 int* num = static_cast<int*>(arg2);
31 *res *= *num;

60TEST_F(PinnableSliceTest, Move) {
61 int n2 = 2;
62 int res = 1;
63 const std::string const_str1 = "123";
64 const std::string const_str2 = "ABC";
65 Slice slice1(const_str1);
66 Slice slice2(const_str2);
67
68 {
69 // Test move constructor on a pinned slice.
70 res = 1;
71 PinnableSlice v1;
72 v1.PinSlice(slice1, Multiplier, &res, &n2);
73 PinnableSlice v2(std::move(v1));
74
75 // Since v1's Cleanable has been moved to v2,
76 // no cleanup should happen in Reset.
77 v1.Reset();
78 ASSERT_EQ(1, res);
79
80 AssertSameData(const_str1, v2);
81 }
82 // v2 is cleaned up.
83 ASSERT_EQ(2, res);
84
85 {
86 // Test move constructor on an unpinned slice.
87 PinnableSlice v1;
88 v1.PinSelf(slice1);
89 PinnableSlice v2(std::move(v1));
90
91 AssertSameData(const_str1, v2);
92 }
93
94 {
95 // Test move assignment from a pinned slice to
96 // another pinned slice.
97 res = 1;
98 PinnableSlice v1;
99 v1.PinSlice(slice1, Multiplier, &res, &n2);
100 PinnableSlice v2;
101 v2.PinSlice(slice2, Multiplier, &res, &n2);
102 v2 = std::move(v1);
103
104 // v2's Cleanable will be Reset before moving
105 // anything from v1.
106 ASSERT_EQ(2, res);
107 // Since v1's Cleanable has been moved to v2,
108 // no cleanup should happen in Reset.
109 v1.Reset();
110 ASSERT_EQ(2, res);
111
112 AssertSameData(const_str1, v2);
113 }
114 // The Cleanable moved from v1 to v2 will be Reset.
115 ASSERT_EQ(4, res);
116
117 {
118 // Test move assignment from a pinned slice to
119 // an unpinned slice.
120 res = 1;
121 PinnableSlice v1;
122 v1.PinSlice(slice1, Multiplier, &res, &n2);
123 PinnableSlice v2;
124 v2.PinSelf(slice2);
125 v2 = std::move(v1);
126
127 // Since v1's Cleanable has been moved to v2,
128 // no cleanup should happen in Reset.
129 v1.Reset();
130 ASSERT_EQ(1, res);
131
132 AssertSameData(const_str1, v2);
133 }
134 // The Cleanable moved from v1 to v2 will be Reset.
135 ASSERT_EQ(2, res);
136
137 {
138 // Test move assignment from an upinned slice to
139 // another unpinned slice.
140 PinnableSlice v1;
141 v1.PinSelf(slice1);
142 PinnableSlice v2;
143 v2.PinSelf(slice2);
144 v2 = std::move(v1);
145
146 AssertSameData(const_str1, v2);
147 }
148
149 {
150 // Test move assignment from an upinned slice to
151 // a pinned slice.
152 res = 1;
153 PinnableSlice v1;
154 v1.PinSelf(slice1);
155 PinnableSlice v2;
156 v2.PinSlice(slice2, Multiplier, &res, &n2);
157 v2 = std::move(v1);
158
159 // v2's Cleanable will be Reset before moving
160 // anything from v1.
161 ASSERT_EQ(2, res);
162
163 AssertSameData(const_str1, v2);
164 }

166 ASSERT_EQ(2, res);
