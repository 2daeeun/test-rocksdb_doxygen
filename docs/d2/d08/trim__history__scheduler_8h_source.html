19// Similar to FlushScheduler, TrimHistoryScheduler is a FIFO queue that keeps
20// track of column families whose flushed immutable memtables may need to be
21// removed (aka trimmed). The actual trimming may be slightly delayed. Due to
22// the use of the mutex and atomic variable, ScheduleWork,
23// TakeNextColumnFamily, and, Empty can be called concurrently.
24class TrimHistoryScheduler {
25 public:
26 TrimHistoryScheduler() : is_empty_(true) {}
27
28 // When a column family needs history trimming, add cfd to the FIFO queue
29 void ScheduleWork(ColumnFamilyData* cfd);
30
31 // Remove the column family from the queue, the caller is responsible for
32 // calling `MemtableList::TrimHistory`
33 ColumnFamilyData* TakeNextColumnFamily();
34
35 bool Empty();
36
37 void Clear();
38
39 // Not on critical path, use mutex to ensure thread safety
40 private:
41 std::atomic<bool> is_empty_;
42 autovector<ColumnFamilyData*> cfds_;
43 std::mutex checking_mutex_;
44};
