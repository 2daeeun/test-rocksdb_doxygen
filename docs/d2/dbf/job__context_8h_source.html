25struct SuperVersionContext {
26 struct WriteStallNotification {
27 WriteStallInfo write_stall_info;
28 const ImmutableOptions* immutable_options;
29 };
30
31 autovector<SuperVersion*> superversions_to_free;
32#ifndef ROCKSDB_DISABLE_STALL_NOTIFICATION
33 autovector<WriteStallNotification> write_stall_notifications;
34#endif
35 std::unique_ptr<SuperVersion>
36 new_superversion; // if nullptr no new superversion
37
38 // If not nullptr, a new seqno to time mapping is available to be installed.
39 // Otherwise, make a shared copy of the one in the existing SuperVersion and
40 // carry it over to the new SuperVersion. This is moved to the SuperVersion
41 // during installation.
42 std::shared_ptr<const SeqnoToTimeMapping> new_seqno_to_time_mapping{nullptr};
43
44 explicit SuperVersionContext(bool create_superversion = false)
45 : new_superversion(create_superversion ? new SuperVersion() : nullptr) {}
46
47 explicit SuperVersionContext(SuperVersionContext&& other) noexcept
48 : superversions_to_free(std::move(other.superversions_to_free)),
49#ifndef ROCKSDB_DISABLE_STALL_NOTIFICATION
50 write_stall_notifications(std::move(other.write_stall_notifications)),
51#endif
52 new_superversion(std::move(other.new_superversion)) {
53 }
54 // No copies
55 SuperVersionContext(const SuperVersionContext& other) = delete;
56 void operator=(const SuperVersionContext& other) = delete;
57
58 void NewSuperVersion() {
59 new_superversion = std::unique_ptr<SuperVersion>(new SuperVersion());
60 }
61
62 inline bool HaveSomethingToDelete() const {
63#ifndef ROCKSDB_DISABLE_STALL_NOTIFICATION
64 return !superversions_to_free.empty() || !write_stall_notifications.empty();
65#else
66 return !superversions_to_free.empty();
67#endif
68 }
69
70 void PushWriteStallNotification(WriteStallCondition old_cond,
71 WriteStallCondition new_cond,
72 const std::string& name,
73 const ImmutableOptions* ioptions) {
74#if !defined(ROCKSDB_DISABLE_STALL_NOTIFICATION)
75 WriteStallNotification notif;
76 notif.write_stall_info.cf_name = name;
77 notif.write_stall_info.condition.prev = old_cond;
78 notif.write_stall_info.condition.cur = new_cond;
79 notif.immutable_options = ioptions;
80 write_stall_notifications.push_back(notif);
81#else
82 (void)old_cond;
83 (void)new_cond;
84 (void)name;
85 (void)ioptions;
86#endif // !defined(ROCKSDB_DISABLE_STALL_NOTIFICATION)
87 }
88
89 void Clean() {
90#if !defined(ROCKSDB_DISABLE_STALL_NOTIFICATION)
91 // notify listeners on changed write stall conditions
92 for (auto& notif : write_stall_notifications) {
93 for (auto& listener : notif.immutable_options->listeners) {
94 listener->OnStallConditionsChanged(notif.write_stall_info);
95 }
96 }
97 write_stall_notifications.clear();
98#endif
99 // free superversions
100 for (auto s : superversions_to_free) {
101 delete s;
102 }
103 superversions_to_free.clear();
104 }
105
106 ~SuperVersionContext() {
107#ifndef ROCKSDB_DISABLE_STALL_NOTIFICATION
108 assert(write_stall_notifications.empty());
109#endif
110 assert(superversions_to_free.empty());
111 }
112};

114struct JobContext {
115 inline bool HaveSomethingToDelete() const {
116 return !(full_scan_candidate_files.empty() && sst_delete_files.empty() &&
117 blob_delete_files.empty() && log_delete_files.empty() &&
118 manifest_delete_files.empty());
119 }
120
121 inline bool HaveSomethingToClean() const {
122 bool sv_have_sth = false;
123 for (const auto& sv_ctx : superversion_contexts) {
124 if (sv_ctx.HaveSomethingToDelete()) {
125 sv_have_sth = true;
126 break;
127 }
128 }
129 return memtables_to_free.size() > 0 || logs_to_free.size() > 0 ||
130 job_snapshot != nullptr || sv_have_sth;
131 }
132
133 SequenceNumber GetJobSnapshotSequence() const {
134 if (job_snapshot) {
135 assert(job_snapshot->snapshot());
136 return job_snapshot->snapshot()->GetSequenceNumber();
137 }
138 return kMaxSequenceNumber;
139 }

141 // Structure to store information for candidate files to delete.
142 struct CandidateFileInfo {
143 std::string file_name;
144 std::string file_path;
145 CandidateFileInfo(std::string name, std::string path)
146 : file_name(std::move(name)), file_path(std::move(path)) {}
147 bool operator==(const CandidateFileInfo& other) const {
148 return file_name == other.file_name && file_path == other.file_path;
149 }
150 };
151
152 // Unique job id
153 int job_id;
154
155 // a list of all files that we'll consider deleting
156 // (every once in a while this is filled up with all files
157 // in the DB directory)
158 // (filled only if we're doing full scan)
159 std::vector<CandidateFileInfo> full_scan_candidate_files;
160
161 // the list of all live sst files that cannot be deleted
162 std::vector<uint64_t> sst_live;
163
164 // the list of sst files that we need to delete
165 std::vector<ObsoleteFileInfo> sst_delete_files;
166
167 // the list of all live blob files that cannot be deleted
168 std::vector<uint64_t> blob_live;
169
170 // the list of blob files that we need to delete
171 std::vector<ObsoleteBlobFileInfo> blob_delete_files;
172
173 // a list of log files that we need to delete
174 std::vector<uint64_t> log_delete_files;
175
176 // a list of log files that we need to preserve during full purge since they
177 // will be reused later
178 std::vector<uint64_t> log_recycle_files;
179
180 // Files quarantined from deletion. This list contains file numbers for files
181 // that are in an ambiguous states. This includes newly generated SST files
182 // and blob files from flush and compaction job whose VersionEdits' persist
183 // state in Manifest are unclear. An old manifest file whose immediately
184 // following new manifest file's CURRENT file creation is in an unclear state.
185 // WAL logs don't have this premature deletion risk since
186 // min_log_number_to_keep is only updated after successful manifest commits.
187 // So this data structure doesn't track log files.
188 autovector<uint64_t> files_to_quarantine;
189
190 // a list of manifest files that we need to delete
191 std::vector<std::string> manifest_delete_files;
192
193 // a list of memtables to be free
194 autovector<ReadOnlyMemTable*> memtables_to_free;
195
196 // contexts for installing superversions for multiple column families
197 std::vector<SuperVersionContext> superversion_contexts;
198
199 autovector<log::Writer*> logs_to_free;
200
201 // the current manifest_file_number, log_number and prev_log_number
202 // that corresponds to the set of files in 'live'.
203 uint64_t manifest_file_number;
204 uint64_t pending_manifest_file_number;
205
206 // Used for remote compaction. To prevent OPTIONS files from getting
207 // purged by PurgeObsoleteFiles() of the primary host
208 uint64_t min_options_file_number;
209 uint64_t log_number;
210 uint64_t prev_log_number;
211
212 uint64_t min_pending_output = 0;
213 uint64_t prev_total_log_size = 0;
214 size_t num_alive_log_files = 0;
215 uint64_t size_log_to_delete = 0;
216
217 // Snapshot taken before flush/compaction job.
218 std::unique_ptr<ManagedSnapshot> job_snapshot;
219
220 explicit JobContext(int _job_id, bool create_superversion = false) {
221 job_id = _job_id;
222 manifest_file_number = 0;
223 pending_manifest_file_number = 0;
224 log_number = 0;
225 prev_log_number = 0;
226 superversion_contexts.emplace_back(
227 SuperVersionContext(create_superversion));
228 }
229
230 // For non-empty JobContext Clean() has to be called at least once before
231 // before destruction (see asserts in ~JobContext()). Should be called with
232 // unlocked DB mutex. Destructor doesn't call Clean() to avoid accidentally
233 // doing potentially slow Clean() with locked DB mutex.
234 void Clean() {
235 // free superversions
236 for (auto& sv_context : superversion_contexts) {
237 sv_context.Clean();
238 }
239 // free pending memtables
240 for (auto m : memtables_to_free) {
241 delete m;
242 }
243 for (auto l : logs_to_free) {
244 delete l;
245 }
246
247 memtables_to_free.clear();
248 logs_to_free.clear();
249 job_snapshot.reset();
250 }
251
252 ~JobContext() {
253 assert(memtables_to_free.size() == 0);
254 assert(logs_to_free.size() == 0);
255 }
256};
