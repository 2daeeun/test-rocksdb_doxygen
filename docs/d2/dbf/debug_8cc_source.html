67Status GetAllKeyVersions(DB* db, ColumnFamilyHandle* cfh, Slice begin_key,
68 Slice end_key, size_t max_num_ikeys,
69 std::vector<KeyVersion>* key_versions) {
70 if (nullptr == db) {
71 return Status::InvalidArgument("db cannot be null.");
72 }
73 if (nullptr == cfh) {
74 return Status::InvalidArgument("Column family handle cannot be null.");
75 }
76 if (nullptr == key_versions) {
77 return Status::InvalidArgument("key_versions cannot be null.");
78 }
79 key_versions->clear();
80
81 DBImpl* idb = static_cast<DBImpl*>(db->GetRootDB());
82 auto icmp = InternalKeyComparator(idb->GetOptions(cfh).comparator);
83 ReadOptions read_options;
84 Arena arena;
85 ScopedArenaPtr<InternalIterator> iter(
86 idb->NewInternalIterator(read_options, &arena, kMaxSequenceNumber, cfh));
87
88 const Comparator* ucmp = icmp.user_comparator();
89 size_t ts_sz = ucmp->timestamp_size();
90
91 Slice from_slice = begin_key;
92 bool has_begin = !begin_key.empty();
93 Slice end_slice = end_key;
94 bool has_end = !end_key.empty();
95 std::string begin_key_buf, end_key_buf;
96 auto [from, end] = MaybeAddTimestampsToRange(
97 has_begin ? &from_slice : nullptr, has_end ? &end_slice : nullptr, ts_sz,
98 &begin_key_buf, &end_key_buf);
99 if (has_begin) {
100 assert(from.has_value());
101 InternalKey ikey;
102 ikey.SetMinPossibleForUserKey(from.value());
103 iter->Seek(ikey.Encode());
104 } else {
105 iter->SeekToFirst();
106 }
107
108 size_t num_keys = 0;
109 for (; iter->Valid(); iter->Next()) {
110 ParsedInternalKey ikey;
111 Status pik_status =
112 ParseInternalKey(iter->key(), &ikey, true /* log_err_key */); // TODO
113 if (!pik_status.ok()) {
114 return pik_status;
115 }
116
117 if (has_end && end.has_value() &&
118 icmp.user_comparator()->Compare(ikey.user_key, end.value()) > 0) {
119 break;
120 }
121
122 key_versions->emplace_back(ikey.user_key.ToString() /* _user_key */,
123 iter->value().ToString() /* _value */,
124 ikey.sequence /* _sequence */,
125 static_cast<int>(ikey.type) /* _type */);
126 if (++num_keys >= max_num_ikeys) {
127 break;
128 }
129 }
130 return Status::OK();
131}
