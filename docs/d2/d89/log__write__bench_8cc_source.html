25// A simple benchmark to simulate transactional logs

27DEFINE_int32(num_records, 6000, "Number of records.");
28DEFINE_int32(record_size, 249, "Size of each record.");
29DEFINE_int32(record_interval, 10000, "Interval between records (microSec)");

31DEFINE_bool(enable_sync, false, "sync after each write.");

33namespace ROCKSDB_NAMESPACE {
34void RunBenchmark() {
35 std::string file_name = test::PerThreadDBPath("log_write_benchmark.log");
36 DBOptions options;
37 Env* env = Env::Default();
38 const auto& clock = env->GetSystemClock();
39 EnvOptions env_options = env->OptimizeForLogWrite(EnvOptions(), options);
40 env_options.bytes_per_sync = FLAGS_bytes_per_sync;
41 std::unique_ptr<WritableFile> file;
42 env->NewWritableFile(file_name, &file, env_options);
43 std::unique_ptr<WritableFileWriter> writer;
44 writer.reset(new WritableFileWriter(
45 std::move(file), file_name, env_options, clock, nullptr /* stats */,
46 Histograms::HISTOGRAM_ENUM_MAX /* hist_type */, options.listeners));
47
48 std::string record;
49 record.assign(FLAGS_record_size, 'X');
50
51 HistogramImpl hist;
52
53 uint64_t start_time = clock->NowMicros();
54 for (int i = 0; i < FLAGS_num_records; i++) {
55 uint64_t start_nanos = clock->NowNanos();
56 writer->Append(record);
57 writer->Flush();
58 if (FLAGS_enable_sync) {
59 writer->Sync(false);
60 }
61 hist.Add(clock->NowNanos() - start_nanos);
62
63 if (i % 1000 == 1) {
64 fprintf(stderr, "Wrote %d records...\n", i);
65 }
66
67 int time_to_sleep =
68 (i + 1) * FLAGS_record_interval - (clock->NowMicros() - start_time);
69 if (time_to_sleep > 0) {
70 clock->SleepForMicroseconds(time_to_sleep);
71 }
72 }
73
74 fprintf(stderr, "Distribution of latency of append+flush: \n%s",
75 hist.ToString().c_str());
76}
77} // namespace ROCKSDB_NAMESPACE

81 " [OPTIONS]...");
82 ParseCommandLineFlags(&argc, &argv, true);
83
84 ROCKSDB_NAMESPACE::RunBenchmark();
85 return 0;
