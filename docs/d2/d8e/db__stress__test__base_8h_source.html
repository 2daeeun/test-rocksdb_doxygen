71 void UpdateIfInitialWriteFails(Env* db_stress_env, const Status& write_s,
72 Status* initial_write_s,
73 bool* initial_wal_write_may_succeed,
74 uint64_t* wait_for_recover_start_time,
75 bool commit_bypass_memtable = false) {
76 assert(db_stress_env && initial_write_s && initial_wal_write_may_succeed &&
77 wait_for_recover_start_time);
78 // Only update `initial_write_s`, `initial_wal_write_may_succeed` when the
79 // first write fails
80 if (!write_s.ok() && (*initial_write_s).ok()) {
81 *initial_write_s = write_s;
82 // With commit_bypass_memtable, we create a new WAL after WAL write
83 // succeeds, that wal creation may fail due to injected error. So the
84 // initial wal write may succeed even if status is failed to write to wal
85 *initial_wal_write_may_succeed =
86 commit_bypass_memtable ||
87 !FaultInjectionTestFS::IsFailedToWriteToWALError(*initial_write_s);
88 *wait_for_recover_start_time = db_stress_env->NowMicros();
89 }
90 }

92 void PrintWriteRecoveryWaitTimeIfNeeded(Env* db_stress_env,
93 const Status& initial_write_s,
94 bool initial_wal_write_may_succeed,
95 uint64_t wait_for_recover_start_time,
96 const std::string& thread_name) {
97 assert(db_stress_env);
98 bool waited_for_recovery = !initial_write_s.ok() &&
99 IsErrorInjectedAndRetryable(initial_write_s) &&
100 initial_wal_write_may_succeed;
101 if (waited_for_recovery) {
102 uint64_t elapsed_sec =
103 (db_stress_env->NowMicros() - wait_for_recover_start_time) / 1000000;
104 if (elapsed_sec > 10) {
105 fprintf(stdout,
106 "%s thread slept to wait for write recovery for "
107 "%" PRIu64 " seconds\n",
108 thread_name.c_str(), elapsed_sec);
109 }
110 }

432// Initialize `options` on which `InitializeOptionsFromFile()` and
433// `InitializeOptionsFromFlags()` have both been called already.
434// There are two cases.
435// Case 1: OPTIONS file is not specified. Command line arguments have been used
436// to initialize `options`. InitializeOptionsGeneral() will use
437// `cache` and `filter_policy` to initialize
438// corresponding fields of `options`. InitializeOptionsGeneral() will
439// also set up other fields of `options` so that stress test can run.
440// Examples include `create_if_missing` and
441// `create_missing_column_families`, etc.
442// Case 2: OPTIONS file is specified. It is possible that, after loading from
443// the given OPTIONS files, some shared object fields are still not
444// initialized because they are not set in the OPTIONS file. In this
445// case, if command line arguments indicate that the user wants to set
446// up such shared objects, e.g. block cache, compressed block cache,
447// row cache, filter policy, then InitializeOptionsGeneral() will honor
448// the user's choice, thus passing `cache`,
449// `filter_policy` as input arguments.
450//
451// InitializeOptionsGeneral() must not overwrite fields of `options` loaded
452// from OPTIONS file.
453void InitializeOptionsGeneral(
454 const std::shared_ptr<Cache>& cache,
455 const std::shared_ptr<const FilterPolicy>& filter_policy,
456 const std::shared_ptr<SstQueryFilterConfigsManager::Factory>& sqfc_factory,
457 Options& options);
