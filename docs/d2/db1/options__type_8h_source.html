83// A set of modifier flags used to alter how an option is evaluated or
84// processed. These flags can be combined together (e.g. kMutable | kShared).
85// The kCompare flags can be used to control if/when options are compared.
86// If kCompareNever is set, two related options would never be compared (always
87// equal) If kCompareExact is set, the options will only be compared if the
88// sanity mode
89// is exact
90// kMutable means the option can be changed after it is prepared
91// kShared means the option is contained in a std::shared_ptr
92// kUnique means the option is contained in a std::uniqued_ptr
93// kRawPointer means the option is a raw pointer value.
94// kAllowNull means that an option is allowed to be null for verification
95// purposes.
96// kDontSerialize means this option should not be serialized and included in
97// the string representation.
98// kDontPrepare means do not call PrepareOptions for this pointer value.
99enum class OptionTypeFlags : uint32_t {
100 kNone = 0x00, // No flags
101 kCompareDefault = 0x0,
102 kCompareNever = ConfigOptions::kSanityLevelNone,
103 kCompareLoose = ConfigOptions::kSanityLevelLooselyCompatible,
104 kCompareExact = ConfigOptions::kSanityLevelExactMatch,
105
106 kMutable = 0x0100, // Option is mutable
107 kRawPointer = 0x0200, // The option is stored as a raw pointer
108 kShared = 0x0400, // The option is stored as a shared_ptr
109 kUnique = 0x0800, // The option is stored as a unique_ptr
110 kAllowNull = 0x1000, // The option can be null
111 kDontSerialize = 0x2000, // Don't serialize the option
112 kDontPrepare = 0x4000, // Don't prepare or sanitize this option
113 kStringNameOnly = 0x8000, // The option serializes to a name only
114};

128// Converts an string into its enumerated value.
129// @param type_map Mapping between strings and enum values
130// @param type The string representation of the enum
131// @param value Returns the enum value represented by the string
132// @return true if the string was found in the enum map, false otherwise.
133template <typename T>
134bool ParseEnum(const std::unordered_map<std::string, T>& type_map,
135 const std::string& type, T* value) {
136 auto iter = type_map.find(type);
137 if (iter != type_map.end()) {
138 *value = iter->second;
139 return true;
140 }
141 return false;
142}

144// Converts an enum into its string representation.
145// @param type_map Mapping between strings and enum values
146// @param type The enum
147// @param value Returned as the string representation of the enum
148// @return true if the enum was found in the enum map, false otherwise.
149template <typename T>
150bool SerializeEnum(const std::unordered_map<std::string, T>& type_map,
151 const T& type, std::string* value) {
152 for (const auto& pair : type_map) {
153 if (pair.second == type) {
154 *value = pair.first;
155 return true;
156 }
157 }
158 return false;
159}

260 // Creates an OptionTypeInfo for an enum type. Enums use an additional
261 // map to convert the enums to/from their string representation.
262 // To create an OptionTypeInfo that is an Enum, one should:
263 // - Create a static map of string values to the corresponding enum value
264 // - Call this method passing the static map in as a parameter.
265 // Note that it is not necessary to add a new OptionType or make any
266 // other changes -- the returned object handles parsing, serialization, and
267 // comparisons.
268 //
269 // @param offset The offset in the option object for this enum
270 // @param map The string to enum mapping for this enum
271 template <typename T>
272 static OptionTypeInfo Enum(
273 int offset, const std::unordered_map<std::string, T>* const map,
274 OptionTypeFlags flags = OptionTypeFlags::kNone,
275 OptionVerificationType verification = OptionVerificationType::kNormal) {
276 OptionTypeInfo info(offset, OptionType::kEnum, verification, flags);
277 info.SetParseFunc(
278 // Uses the map argument to convert the input string into
279 // its corresponding enum value. If value is found in the map,
280 // addr is updated to the corresponding map entry.
281 // @return OK if the value is found in the map
282 // @return InvalidArgument if the value is not found in the map
283 [map](const ConfigOptions&, const std::string& name,
284 const std::string& value, void* addr) {
285 if (map == nullptr) {
286 return Status::NotSupported("No enum mapping ", name);
287 } else if (ParseEnum<T>(*map, value, static_cast<T*>(addr))) {
288 return Status::OK();
289 } else {
290 return Status::InvalidArgument("No mapping for enum ", name);
291 }
292 });
293 info.SetSerializeFunc(
294 // Uses the map argument to convert the input enum into
295 // its corresponding string value. If enum value is found in the map,
296 // value is updated to the corresponding string value in the map.
297 // @return OK if the enum is found in the map
298 // @return InvalidArgument if the enum is not found in the map
299 [map](const ConfigOptions&, const std::string& name, const void* addr,
300 std::string* value) {
301 if (map == nullptr) {
302 return Status::NotSupported("No enum mapping ", name);
303 } else if (SerializeEnum<T>(*map, (*static_cast<const T*>(addr)),
304 value)) {
305 return Status::OK();
306 } else {
307 return Status::InvalidArgument("No mapping for enum ", name);
308 }
309 });
310 info.SetEqualsFunc(
311 // Casts addr1 and addr2 to the enum type and returns true if
312 // they are equal, false otherwise.
313 [](const ConfigOptions&, const std::string&, const void* addr1,
314 const void* addr2, std::string*) {
315 return (*static_cast<const T*>(addr1) ==
316 *static_cast<const T*>(addr2));
317 });
318 return info;
319 } // End OptionTypeInfo::Enum

321 // Creates an OptionTypeInfo for a Struct type. Structs have a
322 // map of string-OptionTypeInfo associated with them that describes how
323 // to process the object for parsing, serializing, and matching.
324 // Structs also have a struct_name, which is the name of the object
325 // as registered in the parent map.
326 // When processing a struct, the option name can be specified as:
327 // - <struct_name> Meaning to process the entire struct.
328 // - <struct_name.field> Meaning to process the single field
329 // - <field> Process the single fields
330 // The CompactionOptionsFIFO, CompactionOptionsUniversal, and LRUCacheOptions
331 // are all examples of Struct options.
332 //
333 // To create an OptionTypeInfo that is a Struct, one should:
334 // - Create a static map of string-OptionTypeInfo corresponding to the
335 // properties of the object that can be set via the options.
336 // - Call this method passing the name and map in as parameters.
337 // Note that it is not necessary to add a new OptionType or make any
338 // other changes -- the returned object handles parsing, serialization, and
339 // comparisons.
340 //
341 // @param offset The offset in the option object for this enum
342 // @param map The string to enum mapping for this enum
343 static OptionTypeInfo Struct(
344 const std::string& struct_name,
345 const std::unordered_map<std::string, OptionTypeInfo>* struct_map,
346 int offset, OptionVerificationType verification, OptionTypeFlags flags) {
347 OptionTypeInfo info(offset, OptionType::kStruct, verification, flags);
348 info.SetParseFunc(
349 // Parses the struct and updates the fields at addr
350 [struct_name, struct_map](const ConfigOptions& opts,
351 const std::string& name,
352 const std::string& value, void* addr) {
353 return ParseStruct(opts, struct_name, struct_map, name, value, addr);
354 });
355 info.SetSerializeFunc(
356 // Serializes the struct options into value
357 [struct_name, struct_map](const ConfigOptions& opts,
358 const std::string& name, const void* addr,
359 std::string* value) {
360 return SerializeStruct(opts, struct_name, struct_map, name, addr,
361 value);
362 });
363 info.SetEqualsFunc(
364 // Compares the struct fields of addr1 and addr2 for equality
365 [struct_name, struct_map](const ConfigOptions& opts,
366 const std::string& name, const void* addr1,
367 const void* addr2, std::string* mismatch) {
368 return StructsAreEqual(opts, struct_name, struct_map, name, addr1,
369 addr2, mismatch);
370 });
371 return info;
372 }
373 static OptionTypeInfo Struct(
374 const std::string& struct_name,
375 const std::unordered_map<std::string, OptionTypeInfo>* struct_map,
376 int offset, OptionVerificationType verification, OptionTypeFlags flags,
377 const ParseFunc& parse_func) {
378 OptionTypeInfo info(
379 Struct(struct_name, struct_map, offset, verification, flags));
380 return info.SetParseFunc(parse_func);
381 }

445 static OptionTypeInfo StringMap(int _offset,
446 OptionVerificationType _verification,
447 OptionTypeFlags _flags,
448 char kv_separator = '=',
449 char item_separator = ';') {
450 OptionTypeInfo info(_offset, OptionType::kStringMap, _verification, _flags);
451 info.SetParseFunc(
452 [kv_separator, item_separator](const ConfigOptions&, const std::string&,
453 const std::string& value, void* addr) {
454 std::map<std::string, std::string> map;
455 Status s;
456 for (size_t start = 0, end = 0;
457 s.ok() && start < value.size() && end != std::string::npos;
458 start = end + 1) {
459 std::string token;
460 s = OptionTypeInfo::NextToken(value, item_separator, start, &end,
461 &token);
462 if (s.ok() && !token.empty()) {
463 size_t pos = token.find(kv_separator);
464 assert(pos != std::string::npos);
465 std::string k = token.substr(0, pos);
466 std::string v = token.substr(pos + 1);
467 std::string decoded_key;
468 std::string decoded_value;
469 (Slice(k)).DecodeHex(&decoded_key);
470 (Slice(v)).DecodeHex(&decoded_value);
471 map.emplace(std::move(decoded_key), std::move(decoded_value));
472 }
473 }
474 if (s.ok()) {
475 *(static_cast<std::map<std::string, std::string>*>(addr)) = map;
476 }
477 return s;
478 });
479 info.SetSerializeFunc(
480 [kv_separator, item_separator](const ConfigOptions&, const std::string&,
481 const void* addr, std::string* value) {
482 const auto map =
483 static_cast<const std::map<std::string, std::string>*>(addr);
484 value->append("{");
485 for (const auto& entry : *map) {
486 value->append(Slice(entry.first).ToString(true));
487 *value += kv_separator;
488 value->append(Slice(entry.second).ToString(true));
489 *value += item_separator;
490 }
491 value->append("}");
492 return Status::OK();
493 });
494 info.SetEqualsFunc([](const ConfigOptions&, const std::string&,
495 const void* addr1, const void* addr2, std::string*) {
496 return (*static_cast<const std::map<std::string, std::string>*>(addr1) ==
497 *static_cast<const std::map<std::string, std::string>*>(addr2));
498 });
499 return info;
500 }

503 // This function will call the T::CreateFromString method to create a new
504 // std::shared_ptr<T> object.

506 // @param offset The offset for the Customizable from the base pointer
507 // @param ovt How to verify this option

513 int offset, OptionVerificationType ovt,

516 const EqualsFunc& equals_func = {}) {

519 auto* shared = static_cast<std::shared_ptr<T>*>(addr);
520 if (name == kIdPropName() && value.empty()) {
521 shared->reset();
522 return Status::OK();
523 } else {
524 return T::CreateFromString(opts, value, shared);
525 }
526 };
527 return OptionTypeInfo(offset, OptionType::kCustomizable, ovt,
528 flags | OptionTypeFlags::kShared, parse_func,
529 serialize_func, equals_func);

977// @param elem_info Controls how individual tokens in value are parsed
978// @param separator Character separating tokens in values (':' in the above
979// example)
980// @param name The name associated with this array option
981// @param value The input string to parse into tokens
982// @param result Returns the results of parsing value into its elements.
983// @return OK if the value was successfully parse
984// @return InvalidArgument if the value is improperly formed or element number

986// or if the token could not be parsed
987// @return NotFound If the tokenized value contains unknown options for

991 const OptionTypeInfo& elem_info, char separator,

993 std::array<T, kSize>* result) {
994 Status status;
995
996 ConfigOptions copy = config_options;
997 copy.ignore_unsupported_options = false;
998 size_t i = 0, start = 0, end = 0;
999 for (; status.ok() && i < kSize && start < value.size() &&
1000 end != std::string::npos;
1001 i++, start = end + 1) {
1002 std::string token;
1003 status = OptionTypeInfo::NextToken(value, separator, start, &end, &token);

1006 if (config_options.ignore_unsupported_options &&
1007 status.IsNotSupported()) {
1008 // If we were ignoring unsupported options and this one should be
1009 // ignored, ignore it by setting the status to OK
1010 status = Status::OK();
1011 }
1012 }
1013 }
1014 if (!status.ok()) {
1015 return status;
1016 }
1017 // make sure the element number matches the array size
1018 if (i < kSize) {
1019 return Status::InvalidArgument(
1020 "Serialized value has less elements than array size", name);
1021 }
1022 if (start < value.size() && end != std::string::npos) {
1023 return Status::InvalidArgument(
