24 return Hash(prefix) % num_buckets;

27// The prefix block index is simply a bucket array, with each entry pointing to
28// the blocks that span the prefixes hashed to this bucket.
29//
30// To reduce memory footprint, if there is only one block per bucket, the entry
31// stores the block id directly. If there are more than one blocks per bucket,
32// because of hash collision or a single prefix spanning multiple blocks,
33// the entry points to an array of block ids. The block array is an array of
34// uint32_t's. The first uint32_t indicates the total number of blocks, followed
35// by the block ids.
36//
37// To differentiate the two cases, the high order bit of the entry indicates
38// whether it is a 'pointer' into a separate block array.
39// 0x7FFFFFFF is reserved for empty bucket.
40
41const uint32_t kNoneBlock = 0x7FFFFFFF;
42const uint32_t kBlockArrayMask = 0x80000000;
43
44inline bool IsNone(uint32_t block_id) { return block_id == kNoneBlock; }
45
46inline bool IsBlockId(uint32_t block_id) {
47 return (block_id & kBlockArrayMask) == 0;
48}
49
50inline uint32_t DecodeIndex(uint32_t block_id) {
51 uint32_t index = block_id ^ kBlockArrayMask;
52 assert(index < kBlockArrayMask);
53 return index;
54}
55
56inline uint32_t EncodeIndex(uint32_t index) {
57 assert(index < kBlockArrayMask);
58 return index | kBlockArrayMask;
59}

61// temporary storage for prefix information during index building
62struct PrefixRecord {
63 Slice prefix;
64 uint32_t start_block;
65 uint32_t end_block;
66 uint32_t num_blocks;
67 PrefixRecord* next;
68};

70class BlockPrefixIndex::Builder {
71 public:
72 void Add(const Slice& key_prefix, uint32_t start_block, uint32_t num_blocks) {
73 PrefixRecord* record = reinterpret_cast<PrefixRecord*>(
74 arena_.AllocateAligned(sizeof(PrefixRecord)));
75 record->prefix = key_prefix;
76 record->start_block = start_block;
77 record->end_block = start_block + num_blocks - 1;
78 record->num_blocks = num_blocks;
79 prefixes_.push_back(record);
80 }
81
82 BlockPrefixIndex* Finish(const SliceTransform* prefix_extractor) {
83 // For now, use roughly 1:1 prefix to bucket ratio.
84 uint32_t num_buckets = static_cast<uint32_t>(prefixes_.size()) + 1;
85
86 // Collect prefix records that hash to the same bucket, into a single
87 // linklist.
88 std::vector<PrefixRecord*> prefixes_per_bucket(num_buckets, nullptr);
89 std::vector<uint32_t> num_blocks_per_bucket(num_buckets, 0);
90 for (PrefixRecord* current : prefixes_) {
91 uint32_t bucket = PrefixToBucket(current->prefix, num_buckets);
92 // merge the prefix block span if the first block of this prefix is
93 // connected to the last block of the previous prefix.
94 PrefixRecord* prev = prefixes_per_bucket[bucket];
95 if (prev) {
96 assert(current->start_block >= prev->end_block);
97 auto distance = current->start_block - prev->end_block;
98 if (distance <= 1) {
99 prev->end_block = current->end_block;
100 prev->num_blocks = prev->end_block - prev->start_block + 1;
101 num_blocks_per_bucket[bucket] += (current->num_blocks + distance - 1);
102 continue;
103 }
104 }
105 current->next = prev;
106 prefixes_per_bucket[bucket] = current;
107 num_blocks_per_bucket[bucket] += current->num_blocks;
108 }
109
110 // Calculate the block array buffer size
111 uint32_t total_block_array_entries = 0;
112 for (uint32_t i = 0; i < num_buckets; i++) {
113 uint32_t num_blocks = num_blocks_per_bucket[i];
114 if (num_blocks > 1) {
115 total_block_array_entries += (num_blocks + 1);
116 }
117 }
118
119 // Populate the final prefix block index
120 uint32_t* block_array_buffer = new uint32_t[total_block_array_entries];
121 uint32_t* buckets = new uint32_t[num_buckets];
122 uint32_t offset = 0;
123 for (uint32_t i = 0; i < num_buckets; i++) {
124 uint32_t num_blocks = num_blocks_per_bucket[i];
125 if (num_blocks == 0) {
126 assert(prefixes_per_bucket[i] == nullptr);
127 buckets[i] = kNoneBlock;
128 } else if (num_blocks == 1) {
129 assert(prefixes_per_bucket[i] != nullptr);
130 assert(prefixes_per_bucket[i]->next == nullptr);
131 buckets[i] = prefixes_per_bucket[i]->start_block;
132 } else {
133 assert(total_block_array_entries > 0);
134 assert(prefixes_per_bucket[i] != nullptr);
135 buckets[i] = EncodeIndex(offset);
136 block_array_buffer[offset] = num_blocks;
137 uint32_t* last_block = &block_array_buffer[offset + num_blocks];
138 auto current = prefixes_per_bucket[i];
139 // populate block ids from largest to smallest
140 while (current != nullptr) {
141 for (uint32_t iter = 0; iter < current->num_blocks; iter++) {
142 *last_block = current->end_block - iter;
143 last_block--;
144 }
145 current = current->next;
146 }
147 assert(last_block == &block_array_buffer[offset]);
148 offset += (num_blocks + 1);
149 }
150 }
151
152 assert(offset == total_block_array_entries);
153
154 return new BlockPrefixIndex(prefix_extractor, num_buckets, buckets,
155 total_block_array_entries, block_array_buffer);
156 }
157
158 private:
159 std::vector<PrefixRecord*> prefixes_;
160 Arena arena_;
161};

163Status BlockPrefixIndex::Create(const SliceTransform* prefix_extractor,
164 const Slice& prefixes, const Slice& prefix_meta,
165 BlockPrefixIndex** prefix_index) {
166 uint64_t pos = 0;
167 auto meta_pos = prefix_meta;
168 Status s;
169 Builder builder;
170
171 while (!meta_pos.empty()) {
172 uint32_t prefix_size = 0;
173 uint32_t entry_index = 0;
174 uint32_t num_blocks = 0;
175 if (!GetVarint32(&meta_pos, &prefix_size) ||
176 !GetVarint32(&meta_pos, &entry_index) ||
177 !GetVarint32(&meta_pos, &num_blocks)) {
178 s = Status::Corruption(
179 "Corrupted prefix meta block: unable to read from it.");
180 break;
181 }
182 if (pos + prefix_size > prefixes.size()) {
183 s = Status::Corruption(
184 "Corrupted prefix meta block: size inconsistency.");
185 break;
186 }
187 Slice prefix(prefixes.data() + pos, prefix_size);
188 builder.Add(prefix, entry_index, num_blocks);
189
190 pos += prefix_size;
191 }
192
193 if (s.ok() && pos != prefixes.size()) {
194 s = Status::Corruption("Corrupted prefix meta block");
195 }
196
197 if (s.ok()) {
198 *prefix_index = builder.Finish(prefix_extractor);
199 }
200
201 return s;
202}

204uint32_t BlockPrefixIndex::GetBlocks(const Slice& key, uint32_t** blocks) {
205 Slice prefix = internal_prefix_extractor_.Transform(key);
206
207 uint32_t bucket = PrefixToBucket(prefix, num_buckets_);
208 uint32_t block_id = buckets_[bucket];
209
210 if (IsNone(block_id)) {
211 return 0;
212 } else if (IsBlockId(block_id)) {
213 *blocks = &buckets_[bucket];
214 return 1;
215 } else {
216 uint32_t index = DecodeIndex(block_id);
217 assert(index < num_block_array_buffer_entries_);
218 *blocks = &block_array_buffer_[index + 1];
219 uint32_t num_blocks = block_array_buffer_[index];
220 assert(num_blocks > 1);
221 assert(index + num_blocks < num_block_array_buffer_entries_);
222 return num_blocks;
223 }
224}
