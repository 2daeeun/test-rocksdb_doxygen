61// create a keyrange by borrowing the left and right dbt
62// pointers. no memory is copied. no checks for infinity needed.
63void keyrange::create(const DBT *left, const DBT *right) {
64 init_empty();
65 m_left_key = left;
66 m_right_key = right;
67}

69// destroy the key copies. if they were never set, then destroy does nothing.
70void keyrange::destroy(void) {
71 toku_destroy_dbt(&m_left_key_copy);
72 toku_destroy_dbt(&m_right_key_copy);
73}

75// create a keyrange by copying the keys from the given range.
76void keyrange::create_copy(const keyrange &range) {
77 // start with an initialized, empty range
78 init_empty();
79
80 // optimize the case where the left and right keys are the same.
81 // we'd like to only have one copy of the data.
82 if (toku_dbt_equals(range.get_left_key(), range.get_right_key())) {
83 set_both_keys(range.get_left_key());
84 } else {
85 // replace our empty left and right keys with
86 // copies of the range's left and right keys
87 replace_left_key(range.get_left_key());
88 replace_right_key(range.get_right_key());
89 }
90}

92// extend this keyrange by choosing the leftmost and rightmost
93// endpoints between this range and the given. replaced keys
94// in this range are freed and inherited keys are copied.
95void keyrange::extend(const comparator &cmp, const keyrange &range) {
96 const DBT *range_left = range.get_left_key();
97 const DBT *range_right = range.get_right_key();
98 if (cmp(range_left, get_left_key()) < 0) {
99 replace_left_key(range_left);
100 }
101 if (cmp(range_right, get_right_key()) > 0) {
102 replace_right_key(range_right);
103 }
104}
105
106// how much memory does this keyrange take?
107// - the size of the left and right keys
108// --- ignore the fact that we may have optimized the point case.
109// it complicates things for little gain.
110// - the size of the keyrange class itself
111uint64_t keyrange::get_memory_size(void) const {
112 const DBT *left_key = get_left_key();
113 const DBT *right_key = get_right_key();
114 return left_key->size + right_key->size + sizeof(keyrange);
115}

117// compare ranges.
118keyrange::comparison keyrange::compare(const comparator &cmp,
119 const keyrange &range) const {
120 if (cmp(get_right_key(), range.get_left_key()) < 0) {
121 return comparison::LESS_THAN;
122 } else if (cmp(get_left_key(), range.get_right_key()) > 0) {
123 return comparison::GREATER_THAN;
124 } else if (cmp(get_left_key(), range.get_left_key()) == 0 &&
125 cmp(get_right_key(), range.get_right_key()) == 0) {
126 return comparison::EQUALS;
127 } else {
128 return comparison::OVERLAPS;
129 }
130}
131
132bool keyrange::overlaps(const comparator &cmp, const keyrange &range) const {
133 // equality is a stronger form of overlapping.
134 // so two ranges "overlap" if they're either equal or just overlapping.
135 comparison c = compare(cmp, range);
136 return c == comparison::EQUALS || c == comparison::OVERLAPS;
137}

169// copy the given once and set both the left and right pointers.
170// optimization for point ranges, so the left and right ranges
171// are not copied twice.
172void keyrange::set_both_keys(const DBT *key) {
173 if (toku_dbt_is_infinite(key)) {
174 m_left_key = key;
175 m_right_key = key;
176 } else {
177 toku_clone_dbt(&m_left_key_copy, *key);
178 toku_copyref_dbt(&m_right_key_copy, m_left_key_copy);
179 }
180 m_point_range = true;
181}
182
183// destroy the current left key. set and possibly copy the new one
184void keyrange::replace_left_key(const DBT *key) {
185 // a little magic:
186 //
187 // if this is a point range, then the left and right keys share
188 // one copy of the data, and it lives in the left key copy. so
189 // if we're replacing the left key, move the real data to the
190 // right key copy instead of destroying it. now, the memory is
191 // owned by the right key and the left key may be replaced.
192 if (m_point_range) {
193 m_right_key_copy = m_left_key_copy;
194 } else {
195 toku_destroy_dbt(&m_left_key_copy);
196 }
197
198 if (toku_dbt_is_infinite(key)) {
199 m_left_key = key;
200 } else {
201 toku_clone_dbt(&m_left_key_copy, *key);
202 m_left_key = nullptr;
203 }
204 m_point_range = false;
205}
206
207// destroy the current right key. set and possibly copy the new one
208void keyrange::replace_right_key(const DBT *key) {
209 toku_destroy_dbt(&m_right_key_copy);
210 if (toku_dbt_is_infinite(key)) {
211 m_right_key = key;
212 } else {
213 toku_clone_dbt(&m_right_key_copy, *key);
214 m_right_key = nullptr;
215 }
216 m_point_range = false;
217}
