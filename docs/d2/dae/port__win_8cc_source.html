55void GetTimeOfDay(TimeVal* tv, struct timezone* /* tz */) {
56 std::chrono::microseconds usNow(
57 std::chrono::duration_cast<std::chrono::microseconds>(
58 std::chrono::system_clock::now().time_since_epoch()));
59
60 std::chrono::seconds secNow(
61 std::chrono::duration_cast<std::chrono::seconds>(usNow));
62
63 tv->tv_sec = static_cast<long>(secNow.count());
64 tv->tv_usec = static_cast<long>(
65 usNow.count() -
66 std::chrono::duration_cast<std::chrono::microseconds>(secNow).count());
67}

73void CondVar::Wait() {
74 // Caller must ensure that mutex is held prior to calling this method
75 std::unique_lock<std::mutex> lk(mu_->getLock(), std::adopt_lock);
76#ifndef NDEBUG
77 mu_->locked_ = false;
78#endif
79 cv_.wait(lk);
80#ifndef NDEBUG
81 mu_->locked_ = true;
82#endif
83 // Release ownership of the lock as we don't want it to be unlocked when
84 // it goes out of scope (as we adopted the lock and didn't lock it ourselves)
85 lk.release();
86}

88bool CondVar::TimedWait(uint64_t abs_time_us) {
89 // MSVC++ library implements wait_until in terms of wait_for so
90 // we need to convert absolute wait into relative wait.
91 std::chrono::microseconds usAbsTime(abs_time_us);
92
93 std::chrono::microseconds usNow(
94 std::chrono::duration_cast<std::chrono::microseconds>(
95 std::chrono::system_clock::now().time_since_epoch()));
96 std::chrono::microseconds relTimeUs = (usAbsTime > usNow)
97 ? (usAbsTime - usNow)
98 : std::chrono::microseconds::zero();
99
100 // Caller must ensure that mutex is held prior to calling this method
101 std::unique_lock<std::mutex> lk(mu_->getLock(), std::adopt_lock);
102
103 // Work around https://github.com/microsoft/STL/issues/369
104 // std::condition_variable_any::wait_for had a fix, but
105 // std::condition_variable still doesn't have a fix in STL yet
106#if defined(_MSC_VER)
107 if (relTimeUs == std::chrono::microseconds::zero()) {
108 lk.unlock();
109 lk.lock();
110 }
111#endif
112#ifndef NDEBUG
113 mu_->locked_ = false;
114#endif
115 std::cv_status cvStatus = cv_.wait_for(lk, relTimeUs);
116#ifndef NDEBUG
117 mu_->locked_ = true;
118#endif
119 // Release ownership of the lock as we don't want it to be unlocked when
120 // it goes out of scope (as we adopted the lock and didn't lock it ourselves)
121 lk.release();
122
123 if (cvStatus == std::cv_status::timeout) {
124 return true;
125 }
126
127 return false;
128}

141struct DIR {
142 HANDLE handle_;
143 bool firstread_;
144 RX_WIN32_FIND_DATA data_;

147 DIR() : handle_(INVALID_HANDLE_VALUE), firstread_(true) {}
148
149 DIR(const DIR&) = delete;
150 DIR& operator=(const DIR&) = delete;
151
152 ~DIR() {
153 if (INVALID_HANDLE_VALUE != handle_) {
154 ::FindClose(handle_);
155 }
156 }
157};

159DIR* opendir(const char* name) {
160 if (!name || *name == 0) {
161 errno = ENOENT;
162 return nullptr;
163 }
164
165 std::string pattern(name);
166 pattern.append("\\").append("*");
167
168 std::unique_ptr<DIR> dir(new DIR);
169
170 dir->handle_ =
171 RX_FindFirstFileEx(RX_FN(pattern).c_str(),
172 FindExInfoBasic, // Do not want alternative name
173 &dir->data_, FindExSearchNameMatch,
174 NULL, // lpSearchFilter
175 0);
176
177 if (dir->handle_ == INVALID_HANDLE_VALUE) {
178 return nullptr;
179 }
180
181 RX_FILESTRING x(dir->data_.cFileName, RX_FNLEN(dir->data_.cFileName));
182 strcpy_s(dir->entry_.d_name, sizeof(dir->entry_.d_name), FN_TO_RX(x).c_str());
183
184 return dir.release();
185}

187struct dirent* readdir(DIR* dirp) {
188 if (!dirp || dirp->handle_ == INVALID_HANDLE_VALUE) {
189 errno = EBADF;
190 return nullptr;
191 }
192
193 if (dirp->firstread_) {
194 dirp->firstread_ = false;
195 return &dirp->entry_;
196 }
197
198 auto ret = RX_FindNextFile(dirp->handle_, &dirp->data_);
199
200 if (ret == 0) {
201 return nullptr;
202 }
203
204 RX_FILESTRING x(dirp->data_.cFileName, RX_FNLEN(dirp->data_.cFileName));
205 strcpy_s(dirp->entry_.d_name, sizeof(dirp->entry_.d_name),
206 FN_TO_RX(x).c_str());
207
208 return &dirp->entry_;
209}

224int Truncate(std::string path, int64_t len) {
225 if (len < 0) {
226 errno = EINVAL;
227 return -1;
228 }
229
230 HANDLE hFile =
231 RX_CreateFile(RX_FN(path).c_str(), GENERIC_READ | GENERIC_WRITE,
232 FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
233 NULL, // Security attrs
234 OPEN_EXISTING, // Truncate existing file only
235 FILE_ATTRIBUTE_NORMAL, NULL);
236
237 if (INVALID_HANDLE_VALUE == hFile) {
238 auto lastError = GetLastError();
239 if (lastError == ERROR_FILE_NOT_FOUND) {
240 errno = ENOENT;
241 } else if (lastError == ERROR_ACCESS_DENIED) {
242 errno = EACCES;
243 } else {
244 errno = EIO;
245 }
246 return -1;
247 }
248
249 int result = 0;
250 FILE_END_OF_FILE_INFO end_of_file;
251 end_of_file.EndOfFile.QuadPart = len;
252
253 if (!SetFileInformationByHandle(hFile, FileEndOfFileInfo, &end_of_file,
254 sizeof(FILE_END_OF_FILE_INFO))) {
255 errno = EIO;
256 result = -1;
257 }
258
259 CloseHandle(hFile);
260 return result;
261}

282bool GenerateRfcUuid(std::string* output) {
283 UUID uuid;
284 UuidCreateSequential(&uuid);
285
286 RPC_CSTR rpc_str;
287 auto status = UuidToStringA(&uuid, &rpc_str);
288 if (status != RPC_S_OK) {
289 return false;
290 }
291
292 // rpc_str is nul-terminated.
293 // reinterpret_cast for possible change between signed/unsigned char.
294 *output = reinterpret_cast<char*>(rpc_str);
295
296 status = RpcStringFreeA(&rpc_str);
297 assert(status == RPC_S_OK);
298
299 return true;
300}
