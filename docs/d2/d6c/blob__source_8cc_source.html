46Status BlobSource::GetBlobFromCache(
47 const Slice& cache_key, CacheHandleGuard<BlobContents>* cached_blob) const {
48 assert(blob_cache_);
49 assert(!cache_key.empty());
50 assert(cached_blob);
51 assert(cached_blob->IsEmpty());
52
53 Cache::Handle* cache_handle = nullptr;
54 cache_handle = GetEntryFromCache(cache_key);
55 if (cache_handle != nullptr) {
56 *cached_blob =
57 CacheHandleGuard<BlobContents>(blob_cache_.get(), cache_handle);
58
59 assert(cached_blob->GetValue());
60
61 PERF_COUNTER_ADD(blob_cache_hit_count, 1);
62 RecordTick(statistics_, BLOB_DB_CACHE_HIT);
63 RecordTick(statistics_, BLOB_DB_CACHE_BYTES_READ,
64 cached_blob->GetValue()->size());
65
66 return Status::OK();
67 }
68
69 RecordTick(statistics_, BLOB_DB_CACHE_MISS);
70
71 return Status::NotFound("Blob not found in cache");
72}

74Status BlobSource::PutBlobIntoCache(
75 const Slice& cache_key, std::unique_ptr<BlobContents>* blob,
76 CacheHandleGuard<BlobContents>* cached_blob) const {
77 assert(blob_cache_);
78 assert(!cache_key.empty());
79 assert(blob);
80 assert(*blob);
81 assert(cached_blob);
82 assert(cached_blob->IsEmpty());
83
84 TypedHandle* cache_handle = nullptr;
85 const Status s = InsertEntryIntoCache(cache_key, blob->get(),
86 &cache_handle, Cache::Priority::BOTTOM);
87 if (s.ok()) {
88 blob->release();
89
90 assert(cache_handle != nullptr);
91 *cached_blob =
92 CacheHandleGuard<BlobContents>(blob_cache_.get(), cache_handle);
93
94 assert(cached_blob->GetValue());
95
96 RecordTick(statistics_, BLOB_DB_CACHE_ADD);
97 RecordTick(statistics_, BLOB_DB_CACHE_BYTES_WRITE,
98 cached_blob->GetValue()->size());
99
100 } else {
101 RecordTick(statistics_, BLOB_DB_CACHE_ADD_FAILURES);
102 }
103
104 return s;
105}

113void BlobSource::PinCachedBlob(CacheHandleGuard<BlobContents>* cached_blob,
114 PinnableSlice* value) {
115 assert(cached_blob);
116 assert(cached_blob->GetValue());
117 assert(value);
118
119 // To avoid copying the cached blob into the buffer provided by the
120 // application, we can simply transfer ownership of the cache handle to
121 // the target PinnableSlice. This has the potential to save a lot of
122 // CPU, especially with large blob values.
123
124 value->Reset();
125
126 constexpr Cleanable* cleanable = nullptr;
127 value->PinSlice(cached_blob->GetValue()->data(), cleanable);
128
129 cached_blob->TransferTo(value);
130}

132void BlobSource::PinOwnedBlob(std::unique_ptr<BlobContents>* owned_blob,
133 PinnableSlice* value) {
134 assert(owned_blob);
135 assert(*owned_blob);
136 assert(value);
137
138 BlobContents* const blob = owned_blob->release();
139 assert(blob);
140
141 value->Reset();
142 value->PinSlice(
143 blob->data(),
144 [](void* arg1, void* /* arg2 */) {
145 delete static_cast<BlobContents*>(arg1);
146 },
147 blob, nullptr);
148}

158Status BlobSource::GetBlob(const ReadOptions& read_options,
159 const Slice& user_key, uint64_t file_number,
160 uint64_t offset, uint64_t file_size,
161 uint64_t value_size,
162 CompressionType compression_type,
163 FilePrefetchBuffer* prefetch_buffer,
164 PinnableSlice* value, uint64_t* bytes_read) {
165 assert(value);
166
167 Status s;
168
169 const CacheKey cache_key = GetCacheKey(file_number, file_size, offset);
170
171 CacheHandleGuard<BlobContents> blob_handle;
172
173 // First, try to get the blob from the cache
174 //
175 // If blob cache is enabled, we'll try to read from it.
176 if (blob_cache_) {
177 Slice key = cache_key.AsSlice();
178 s = GetBlobFromCache(key, &blob_handle);
179 if (s.ok()) {
180 PinCachedBlob(&blob_handle, value);
181
182 // For consistency, the size of on-disk (possibly compressed) blob record
183 // is assigned to bytes_read.
184 uint64_t adjustment =
185 read_options.verify_checksums
186 ? BlobLogRecord::CalculateAdjustmentForRecordHeader(
187 user_key.size())
188 : 0;
189 assert(offset >= adjustment);
190
191 uint64_t record_size = value_size + adjustment;
192 if (bytes_read) {
193 *bytes_read = record_size;
194 }
195 return s;
196 }
197 }
198
199 assert(blob_handle.IsEmpty());
200
201 const bool no_io = read_options.read_tier == kBlockCacheTier;
202 if (no_io) {
203 s = Status::Incomplete("Cannot read blob(s): no disk I/O allowed");
204 return s;
205 }
206
207 // Can't find the blob from the cache. Since I/O is allowed, read from the
208 // file.
209 std::unique_ptr<BlobContents> blob_contents;
210
211 {
212 CacheHandleGuard<BlobFileReader> blob_file_reader;
213 s = blob_file_cache_->GetBlobFileReader(read_options, file_number,
214 &blob_file_reader);
215 if (!s.ok()) {
216 return s;
217 }
218
219 assert(blob_file_reader.GetValue());
220
221 if (compression_type != blob_file_reader.GetValue()->GetCompressionType()) {
222 return Status::Corruption("Compression type mismatch when reading blob");
223 }
224
225 MemoryAllocator* const allocator =
226 (blob_cache_ && read_options.fill_cache)
227 ? blob_cache_.get()->memory_allocator()
228 : nullptr;
229
230 uint64_t read_size = 0;
231 s = blob_file_reader.GetValue()->GetBlob(
232 read_options, user_key, offset, value_size, compression_type,
233 prefetch_buffer, allocator, &blob_contents, &read_size);
234 if (!s.ok()) {
235 return s;
236 }
237 if (bytes_read) {
238 *bytes_read = read_size;
239 }
240 }
241
242 if (blob_cache_ && read_options.fill_cache) {
243 // If filling cache is allowed and a cache is configured, try to put the
244 // blob to the cache.
245 Slice key = cache_key.AsSlice();
246 s = PutBlobIntoCache(key, &blob_contents, &blob_handle);
247 if (!s.ok()) {
248 return s;
249 }
250
251 PinCachedBlob(&blob_handle, value);
252 } else {
253 PinOwnedBlob(&blob_contents, value);
254 }
255
256 assert(s.ok());
257 return s;
258}

260void BlobSource::MultiGetBlob(const ReadOptions& read_options,
261 autovector<BlobFileReadRequests>& blob_reqs,
262 uint64_t* bytes_read) {
263 assert(blob_reqs.size() > 0);
264
265 uint64_t total_bytes_read = 0;
266 uint64_t bytes_read_in_file = 0;
267
268 for (auto& [file_number, file_size, blob_reqs_in_file] : blob_reqs) {
269 // sort blob_reqs_in_file by file offset.
270 std::sort(
271 blob_reqs_in_file.begin(), blob_reqs_in_file.end(),
272 [](const BlobReadRequest& lhs, const BlobReadRequest& rhs) -> bool {
273 return lhs.offset < rhs.offset;
274 });
275
276 MultiGetBlobFromOneFile(read_options, file_number, file_size,
277 blob_reqs_in_file, &bytes_read_in_file);
278
279 total_bytes_read += bytes_read_in_file;
280 }
281
282 if (bytes_read) {
283 *bytes_read = total_bytes_read;
284 }
285}

287void BlobSource::MultiGetBlobFromOneFile(const ReadOptions& read_options,
288 uint64_t file_number,
289 uint64_t /*file_size*/,
290 autovector<BlobReadRequest>& blob_reqs,
291 uint64_t* bytes_read) {
292 const size_t num_blobs = blob_reqs.size();
293 assert(num_blobs > 0);
294 assert(num_blobs <= MultiGetContext::MAX_BATCH_SIZE);
295
296#ifndef NDEBUG
297 for (size_t i = 0; i < num_blobs - 1; ++i) {
298 assert(blob_reqs[i].offset <= blob_reqs[i + 1].offset);
299 }
300#endif // !NDEBUG
301
302 using Mask = uint64_t;
303 Mask cache_hit_mask = 0;
304
305 uint64_t total_bytes = 0;
306 const OffsetableCacheKey base_cache_key(db_id_, db_session_id_, file_number);
307
308 if (blob_cache_) {
309 size_t cached_blob_count = 0;
310 for (size_t i = 0; i < num_blobs; ++i) {
311 auto& req = blob_reqs[i];
312
313 CacheHandleGuard<BlobContents> blob_handle;
314 const CacheKey cache_key = base_cache_key.WithOffset(req.offset);
315 const Slice key = cache_key.AsSlice();
316
317 const Status s = GetBlobFromCache(key, &blob_handle);
318
319 if (s.ok()) {
320 assert(req.status);
321 *req.status = s;
322
323 PinCachedBlob(&blob_handle, req.result);
324
325 // Update the counter for the number of valid blobs read from the cache.
326 ++cached_blob_count;
327
328 // For consistency, the size of each on-disk (possibly compressed) blob
329 // record is accumulated to total_bytes.
330 uint64_t adjustment =
331 read_options.verify_checksums
332 ? BlobLogRecord::CalculateAdjustmentForRecordHeader(
333 req.user_key->size())
334 : 0;
335 assert(req.offset >= adjustment);
336 total_bytes += req.len + adjustment;
337 cache_hit_mask |= (Mask{1} << i); // cache hit
338 }
339 }
340
341 // All blobs were read from the cache.
342 if (cached_blob_count == num_blobs) {
343 if (bytes_read) {
344 *bytes_read = total_bytes;
345 }
346 return;
347 }
348 }
349
350 const bool no_io = read_options.read_tier == kBlockCacheTier;
351 if (no_io) {
352 for (size_t i = 0; i < num_blobs; ++i) {
353 if (!(cache_hit_mask & (Mask{1} << i))) {
354 BlobReadRequest& req = blob_reqs[i];
355 assert(req.status);
356
357 *req.status =
358 Status::Incomplete("Cannot read blob(s): no disk I/O allowed");
359 }
360 }
361 return;
362 }
363
364 {
365 // Find the rest of blobs from the file since I/O is allowed.
366 autovector<std::pair<BlobReadRequest*, std::unique_ptr<BlobContents>>>
367 _blob_reqs;
368 uint64_t _bytes_read = 0;
369
370 for (size_t i = 0; i < num_blobs; ++i) {
371 if (!(cache_hit_mask & (Mask{1} << i))) {
372 _blob_reqs.emplace_back(&blob_reqs[i], std::unique_ptr<BlobContents>());
373 }
374 }
375
376 CacheHandleGuard<BlobFileReader> blob_file_reader;
377 Status s = blob_file_cache_->GetBlobFileReader(read_options, file_number,
378 &blob_file_reader);
379 if (!s.ok()) {
380 for (size_t i = 0; i < _blob_reqs.size(); ++i) {
381 BlobReadRequest* const req = _blob_reqs[i].first;
382 assert(req);
383 assert(req->status);
384
385 *req->status = s;
386 }
387 return;
388 }
389
390 assert(blob_file_reader.GetValue());
391
392 MemoryAllocator* const allocator =
393 (blob_cache_ && read_options.fill_cache)
394 ? blob_cache_.get()->memory_allocator()
395 : nullptr;
396
397 blob_file_reader.GetValue()->MultiGetBlob(read_options, allocator,
398 _blob_reqs, &_bytes_read);
399
400 if (blob_cache_ && read_options.fill_cache) {
401 // If filling cache is allowed and a cache is configured, try to put
402 // the blob(s) to the cache.
403 for (auto& [req, blob_contents] : _blob_reqs) {
404 assert(req);
405
406 if (req->status->ok()) {
407 CacheHandleGuard<BlobContents> blob_handle;
408 const CacheKey cache_key = base_cache_key.WithOffset(req->offset);
409 const Slice key = cache_key.AsSlice();
410 s = PutBlobIntoCache(key, &blob_contents, &blob_handle);
411 if (!s.ok()) {
412 *req->status = s;
413 } else {
414 PinCachedBlob(&blob_handle, req->result);
415 }
416 }
417 }
418 } else {
419 for (auto& [req, blob_contents] : _blob_reqs) {
420 assert(req);
421
422 if (req->status->ok()) {
423 PinOwnedBlob(&blob_contents, req->result);
424 }
425 }
426 }
427
428 total_bytes += _bytes_read;
429 if (bytes_read) {
430 *bytes_read = total_bytes;
431 }
432 }
433}

435bool BlobSource::TEST_BlobInCache(uint64_t file_number, uint64_t file_size,
436 uint64_t offset, size_t* charge) const {
437 const CacheKey cache_key = GetCacheKey(file_number, file_size, offset);
438 const Slice key = cache_key.AsSlice();
439
440 CacheHandleGuard<BlobContents> blob_handle;
441 const Status s = GetBlobFromCache(key, &blob_handle);
442
443 if (s.ok() && blob_handle.GetValue() != nullptr) {
444 if (charge) {
445 const Cache* const cache = blob_handle.GetCache();
446 assert(cache);
447
448 Cache::Handle* const handle = blob_handle.GetCacheHandle();
449 assert(handle);
450
451 *charge = cache->GetUsage(handle);
452 }
453
454 return true;
455 }
456
457 return false;
458}
