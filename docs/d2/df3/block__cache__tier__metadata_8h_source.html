20//
21// Block Cache Tier Metadata
22//
23// The BlockCacheTierMetadata holds all the metadata associated with block
24// cache. It
25// fundamentally contains 2 indexes and an LRU.
26//
27// Block Cache Index
28//
29// This is a forward index that maps a given key to a LBA (Logical Block
30// Address). LBA is a disk pointer that points to a record on the cache.
31//
32// LBA = { cache-id, offset, size }
33//
34// Cache File Index
35//
36// This is a forward index that maps a given cache-id to a cache file object.
37// Typically you would lookup using LBA and use the object to read or write
38struct BlockInfo {
39 explicit BlockInfo(const Slice& key, const LBA& lba = LBA())
40 : key_(key.ToString()), lba_(lba) {}
41
42 std::string key_;
43 LBA lba_;
44};

113 };
