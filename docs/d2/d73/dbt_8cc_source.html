78void toku_destroy_dbt(DBT *dbt) {
79 switch (dbt->flags) {
80 case DB_DBT_MALLOC:
81 case DB_DBT_REALLOC:
82 toku_free(dbt->data);
83 toku_init_dbt(dbt);
84 break;
85 }
86}

131bool toku_dbt_is_infinite(const DBT *dbt) {
132 return dbt == toku_dbt_positive_infinity() ||
133 dbt == toku_dbt_negative_infinity();
134}
135
136bool toku_dbt_is_empty(const DBT *dbt) {
137 // can't have a null data field with a non-zero size
138 paranoid_invariant(dbt->data != nullptr || dbt->size == 0);
139 return dbt->data == nullptr;
140}
141
142int toku_dbt_infinite_compare(const DBT *a, const DBT *b) {
143 if (a == b) {
144 return 0;
145 } else if (a == toku_dbt_positive_infinity()) {
146 return 1;
147 } else if (b == toku_dbt_positive_infinity()) {
148 return -1;
149 } else if (a == toku_dbt_negative_infinity()) {
150 return -1;
151 } else {
152 invariant(b == toku_dbt_negative_infinity());
153 return 1;
154 }
155}
156
157bool toku_dbt_equals(const DBT *a, const DBT *b) {
158 if (!toku_dbt_is_infinite(a) && !toku_dbt_is_infinite(b)) {
159 return a->data == b->data && a->size == b->size;
160 } else {
161 // a or b is infinite, so they're equal if they are the same infinite
162 return a == b ? true : false;
163 }
164}
