23size_t Arena::OptimizeBlockSize(size_t block_size) {
24 // Make sure block_size is in optimal range
25 block_size = std::max(Arena::kMinBlockSize, block_size);
26 block_size = std::min(Arena::kMaxBlockSize, block_size);
27
28 // make sure block_size is the multiple of kAlignUnit
29 if (block_size % kAlignUnit != 0) {
30 block_size = (1 + block_size / kAlignUnit) * kAlignUnit;
31 }
32
33 return block_size;
34}

36Arena::Arena(size_t block_size, AllocTracker* tracker, size_t huge_page_size)
37 : kBlockSize(OptimizeBlockSize(block_size)), tracker_(tracker) {
38 assert(kBlockSize >= kMinBlockSize && kBlockSize <= kMaxBlockSize &&
39 kBlockSize % kAlignUnit == 0);
40 TEST_SYNC_POINT_CALLBACK("Arena::Arena:0", const_cast<size_t*>(&kBlockSize));
41 alloc_bytes_remaining_ = sizeof(inline_block_);
42 blocks_memory_ += alloc_bytes_remaining_;
43 aligned_alloc_ptr_ = inline_block_;
44 unaligned_alloc_ptr_ = inline_block_ + alloc_bytes_remaining_;
45 if (MemMapping::kHugePageSupported) {
46 hugetlb_size_ = huge_page_size;
47 if (hugetlb_size_ && kBlockSize > hugetlb_size_) {
48 hugetlb_size_ = ((kBlockSize - 1U) / hugetlb_size_ + 1U) * hugetlb_size_;
49 }
50 }
51 if (tracker_ != nullptr) {
52 tracker_->Allocate(kInlineSize);
53 }
54}

63char* Arena::AllocateFallback(size_t bytes, bool aligned) {
64 if (bytes > kBlockSize / 4) {
65 ++irregular_block_num;
66 // Object is more than a quarter of our block size. Allocate it separately
67 // to avoid wasting too much space in leftover bytes.
68 return AllocateNewBlock(bytes);
69 }
70
71 // We waste the remaining space in the current block.
72 size_t size = 0;
73 char* block_head = nullptr;
74 if (MemMapping::kHugePageSupported && hugetlb_size_ > 0) {
75 size = hugetlb_size_;
76 block_head = AllocateFromHugePage(size);
77 }
78 if (!block_head) {
79 size = kBlockSize;
80 block_head = AllocateNewBlock(size);
81 }
82 alloc_bytes_remaining_ = size - bytes;
83
84 if (aligned) {
85 aligned_alloc_ptr_ = block_head + bytes;
86 unaligned_alloc_ptr_ = block_head + size;
87 return block_head;
88 } else {
89 aligned_alloc_ptr_ = block_head;
90 unaligned_alloc_ptr_ = block_head + size - bytes;
91 return unaligned_alloc_ptr_;
92 }
93}

95char* Arena::AllocateFromHugePage(size_t bytes) {
96 MemMapping mm = MemMapping::AllocateHuge(bytes);
97 auto addr = static_cast<char*>(mm.Get());
98 if (addr) {
99 huge_blocks_.push_back(std::move(mm));
100 blocks_memory_ += bytes;
101 if (tracker_ != nullptr) {
102 tracker_->Allocate(bytes);
103 }
104 }
105 return addr;
106}

108char* Arena::AllocateAligned(size_t bytes, size_t huge_page_size,
109 Logger* logger) {
110 if (MemMapping::kHugePageSupported && hugetlb_size_ > 0 &&
111 huge_page_size > 0 && bytes > 0) {
112 // Allocate from a huge page TLB table.
113 size_t reserved_size =
114 ((bytes - 1U) / huge_page_size + 1U) * huge_page_size;
115 assert(reserved_size >= bytes);
116
117 char* addr = AllocateFromHugePage(reserved_size);
118 if (addr == nullptr) {
119 ROCKS_LOG_WARN(logger,
120 "AllocateAligned fail to allocate huge TLB pages: %s",
121 errnoStr(errno).c_str());
122 // fail back to malloc
123 } else {
124 return addr;
125 }
126 }
127
128 size_t current_mod =
129 reinterpret_cast<uintptr_t>(aligned_alloc_ptr_) & (kAlignUnit - 1);
130 size_t slop = (current_mod == 0 ? 0 : kAlignUnit - current_mod);
131 size_t needed = bytes + slop;
132 char* result;
133 if (needed <= alloc_bytes_remaining_) {
134 result = aligned_alloc_ptr_ + slop;
135 aligned_alloc_ptr_ += needed;
136 alloc_bytes_remaining_ -= needed;
137 } else {
138 // AllocateFallback always returns aligned memory
139 result = AllocateFallback(bytes, true /* aligned */);
140 }
141 assert((reinterpret_cast<uintptr_t>(result) & (kAlignUnit - 1)) == 0);
142 return result;
143}

145char* Arena::AllocateNewBlock(size_t block_bytes) {
146 // NOTE: std::make_unique zero-initializes the block so is not appropriate
147 // here
148 char* block = new char[block_bytes];
149 blocks_.push_back(std::unique_ptr<char[]>(block));
150
151 size_t allocated_size;
152#ifdef ROCKSDB_MALLOC_USABLE_SIZE
153 allocated_size = malloc_usable_size(block);
154#ifndef NDEBUG
155 // It's hard to predict what malloc_usable_size() returns.
156 // A callback can allow users to change the costed size.
157 std::pair<size_t*, size_t*> pair(&allocated_size, &block_bytes);
158 TEST_SYNC_POINT_CALLBACK("Arena::AllocateNewBlock:0", &pair);
159#endif // NDEBUG
160#else
161 allocated_size = block_bytes;
162#endif // ROCKSDB_MALLOC_USABLE_SIZE
163 blocks_memory_ += allocated_size;
164 if (tracker_ != nullptr) {
165 tracker_->Allocate(allocated_size);
166 }
167 return block;
168}
