20// A class that can be used to compute the amount of additional garbage
21// generated by a compaction. It parses the keys and blob references in the
22// input and output of a compaction, and aggregates the "inflow" and "outflow"
23// on a per-blob file basis. The amount of additional garbage for any given blob
24// file can then be computed by subtracting the outflow from the inflow.
25class BlobGarbageMeter {
26 public:
27 // A class to store the number and total size of blobs on a per-blob file
28 // basis.
29 class BlobStats {
30 public:
31 void Add(uint64_t bytes) {
32 ++count_;
33 bytes_ += bytes;
34 }
35 void Add(uint64_t count, uint64_t bytes) {
36 count_ += count;
37 bytes_ += bytes;
38 }
39
40 uint64_t GetCount() const { return count_; }
41 uint64_t GetBytes() const { return bytes_; }
42
43 private:
44 uint64_t count_ = 0;
45 uint64_t bytes_ = 0;
46 };
47
48 // A class to keep track of the "inflow" and the "outflow" and to compute the
49 // amount of additional garbage for a given blob file.
50 class BlobInOutFlow {
51 public:
52 void AddInFlow(uint64_t bytes) {
53 in_flow_.Add(bytes);
54 assert(IsValid());
55 }
56 void AddOutFlow(uint64_t bytes) {
57 out_flow_.Add(bytes);
58 assert(IsValid());
59 }
60
61 const BlobStats& GetInFlow() const { return in_flow_; }
62 const BlobStats& GetOutFlow() const { return out_flow_; }
63
64 bool IsValid() const {
65 return in_flow_.GetCount() >= out_flow_.GetCount() &&
66 in_flow_.GetBytes() >= out_flow_.GetBytes();
67 }
68 bool HasGarbage() const {
69 assert(IsValid());
70 return in_flow_.GetCount() > out_flow_.GetCount();
71 }
72 uint64_t GetGarbageCount() const {
73 assert(IsValid());
74 assert(HasGarbage());
75 return in_flow_.GetCount() - out_flow_.GetCount();
76 }
77 uint64_t GetGarbageBytes() const {
78 assert(IsValid());
79 assert(HasGarbage());
80 return in_flow_.GetBytes() - out_flow_.GetBytes();
81 }
82
83 private:
84 BlobStats in_flow_;
85 BlobStats out_flow_;
86 };
87
88 Status ProcessInFlow(const Slice& key, const Slice& value);
89 Status ProcessOutFlow(const Slice& key, const Slice& value);
90
91 const std::unordered_map<uint64_t, BlobInOutFlow>& flows() const {
92 return flows_;
93 }
94
95 private:
96 static Status Parse(const Slice& key, const Slice& value,
97 uint64_t* blob_file_number, uint64_t* bytes);
98
99 std::unordered_map<uint64_t, BlobInOutFlow> flows_;
100};
