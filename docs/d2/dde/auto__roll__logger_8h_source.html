37 // Write a header entry to the log. All header information will be written
38 // again every time the log rolls over.
39 void LogHeader(const char* format, va_list ap) override;
40
41 // check if the logger has encountered any problem.
42 Status GetStatus() { return status_; }
43
44 size_t GetLogFileSize() const override {
45 std::shared_ptr<Logger> logger;
46 {
47 MutexLock l(&mutex_);
48 if (!logger_) {
49 return 0;
50 }
51 // pin down the current logger_ instance before releasing the mutex.
52 logger = logger_;
53 }
54 return logger->GetLogFileSize();
55 }
56
57 void Flush() override {
58 std::shared_ptr<Logger> logger;
59 {
60 MutexLock l(&mutex_);
61 // pin down the current logger_ instance before releasing the mutex.
62 logger = logger_;
63 }
64 TEST_SYNC_POINT("AutoRollLogger::Flush:PinnedLogger");
65 if (logger) {
66 logger->Flush();
67 }
68 }
69
70 virtual ~AutoRollLogger() {
71 if (logger_ && !closed_) {
72 logger_->Close().PermitUncheckedError();
73 }
74 status_.PermitUncheckedError();
75 }
76
77 using Logger::GetInfoLogLevel;
78 InfoLogLevel GetInfoLogLevel() const override {
79 MutexLock l(&mutex_);
80 if (!logger_) {
81 return Logger::GetInfoLogLevel();
82 }
83 return logger_->GetInfoLogLevel();
84 }
85
86 using Logger::SetInfoLogLevel;
87 void SetInfoLogLevel(const InfoLogLevel log_level) override {
88 MutexLock lock(&mutex_);
89 Logger::SetInfoLogLevel(log_level);
90 if (logger_) {
91 logger_->SetInfoLogLevel(log_level);
92 }
93 }
94
95 void SetCallNowMicrosEveryNRecords(uint64_t call_NowMicros_every_N_records) {
96 call_NowMicros_every_N_records_ = call_NowMicros_every_N_records;
97 }

103
104 Logger* TEST_inner_logger() const { return logger_.get(); }
105
106 protected:
107 // Implementation of Close()
108 Status CloseImpl() override {
109 if (logger_) {
110 return logger_->Close();
111 } else {
112 return Status::OK();
113 }
114 }

145 // List of all existing info log files. Used for enforcing number of
