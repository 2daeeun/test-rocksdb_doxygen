24class TestFilterBitsBuilder : public FilterBitsBuilder {
25 public:
26 explicit TestFilterBitsBuilder() = default;
27
28 // Add Key to filter
29 void AddKey(const Slice& key) override {
30 hash_entries_.push_back(Hash(key.data(), key.size(), 1));
31 }
32 void AddKeyAndAlt(const Slice& key, const Slice& alt) override {
33 AddKey(key);
34 AddKey(alt);
35 }
36
37 using FilterBitsBuilder::Finish;
38
39 // Generate the filter using the keys that are added
40 Slice Finish(std::unique_ptr<const char[]>* buf) override {
41 uint32_t len = static_cast<uint32_t>(hash_entries_.size()) * 4;
42 char* data = new char[len];
43 for (size_t i = 0; i < hash_entries_.size(); i++) {
44 EncodeFixed32(data + i * 4, hash_entries_[i]);
45 }
46 const char* const_data = data;
47 buf->reset(const_data);
48 return Slice(data, len);
49 }
50
51 size_t EstimateEntriesAdded() override { return hash_entries_.size(); }
52
53 size_t ApproximateNumEntries(size_t bytes) override { return bytes / 4; }
54
55 private:
56 std::vector<uint32_t> hash_entries_;
57};

65class TestFilterBitsReader : public FilterBitsReader {
66 public:
67 explicit TestFilterBitsReader(const Slice& contents)
68 : data_(contents.data()), len_(static_cast<uint32_t>(contents.size())) {}
69
70 // Silence compiler warning about overloaded virtual
71 using FilterBitsReader::MayMatch;
72 bool MayMatch(const Slice& entry) override {
73 uint32_t h = Hash(entry.data(), entry.size(), 1);
74 for (size_t i = 0; i + 4 <= len_; i += 4) {
75 if (h == DecodeFixed32(data_ + i)) {
76 return true;
77 }
78 }
79 return false;
80 }
81
82 private:
83 const char* data_;
84 uint32_t len_;
85};

197class CountUniqueFilterBitsBuilderWrapper : public FilterBitsBuilder {
198 std::unique_ptr<FilterBitsBuilder> b_;
199 std::set<std::string> uniq_;
200
201 public:
202 explicit CountUniqueFilterBitsBuilderWrapper(FilterBitsBuilder* b) : b_(b) {}
203
204 ~CountUniqueFilterBitsBuilderWrapper() override = default;
205
206 void AddKey(const Slice& key) override {
207 b_->AddKey(key);
208 uniq_.insert(key.ToString());
209 }
210 void AddKeyAndAlt(const Slice& key, const Slice& alt) override {
211 b_->AddKeyAndAlt(key, alt);
212 uniq_.insert(key.ToString());
213 uniq_.insert(alt.ToString());
214 }
215
216 using FilterBitsBuilder::Finish;
217
218 Slice Finish(std::unique_ptr<const char[]>* buf) override {
219 Slice rv = b_->Finish(buf);
220 Status s_dont_care = b_->MaybePostVerify(rv);
221 s_dont_care.PermitUncheckedError();
222 uniq_.clear();
223 return rv;
224 }
225
226 size_t EstimateEntriesAdded() override { return b_->EstimateEntriesAdded(); }
227
228 size_t ApproximateNumEntries(size_t bytes) override {
229 return b_->ApproximateNumEntries(bytes);
230 }
231
232 size_t CountUnique() { return uniq_.size(); }
233};

235TEST_F(FullFilterBlockTest, DuplicateEntries) {
236 { // empty prefixes
237 std::unique_ptr<const SliceTransform> prefix_extractor(
238 NewFixedPrefixTransform(0));
239 auto bits_builder = new CountUniqueFilterBitsBuilderWrapper(GetBuilder());
240 const bool WHOLE_KEY = true;
241 FullFilterBlockBuilder builder(prefix_extractor.get(), WHOLE_KEY,
242 bits_builder);
243 ASSERT_EQ(0, bits_builder->CountUnique());
244 // adds key and empty prefix; both abstractions count them
245 builder.Add("key1");
246 ASSERT_EQ(2, bits_builder->CountUnique());
247 // Add different key (unique) and also empty prefix (not unique).
248 // From here in this test, it's immaterial whether the block builder
249 // can count unique keys.
250 builder.Add("key2");
251 ASSERT_EQ(3, bits_builder->CountUnique());
252 // Empty key -> nothing unique
253 builder.Add("");
254 ASSERT_EQ(3, bits_builder->CountUnique());
255 }
256
257 // mix of empty and non-empty
258 std::unique_ptr<const SliceTransform> prefix_extractor(
259 NewFixedPrefixTransform(7));
260 auto bits_builder = new CountUniqueFilterBitsBuilderWrapper(GetBuilder());
261 const bool WHOLE_KEY = true;
262 FullFilterBlockBuilder builder(prefix_extractor.get(), WHOLE_KEY,
263 bits_builder);
264 builder.Add(""); // test with empty key too
265 builder.Add("prefix1key1");
266 builder.Add("prefix1key1");
267 builder.Add("prefix1key2");
268 builder.Add("prefix1key3");
269 builder.Add("prefix2key4");
270 // 1 empty, 2 non-empty prefixes, and 4 non-empty keys
271 ASSERT_EQ(1 + 2 + 4, bits_builder->CountUnique());
272}
