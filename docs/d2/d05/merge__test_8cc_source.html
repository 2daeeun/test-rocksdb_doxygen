37class CountMergeOperator : public AssociativeMergeOperator {
38 public:
39 CountMergeOperator() {
40 mergeOperator_ = MergeOperators::CreateUInt64AddOperator();
41 }
42
43 bool Merge(const Slice& key, const Slice* existing_value, const Slice& value,
44 std::string* new_value, Logger* logger) const override {
45 assert(new_value->empty());
46 ++num_merge_operator_calls;
47 if (existing_value == nullptr) {
48 new_value->assign(value.data(), value.size());
49 return true;
50 }
51
52 return mergeOperator_->PartialMerge(key, *existing_value, value, new_value,
53 logger);
54 }
55
56 bool PartialMergeMulti(const Slice& key,
57 const std::deque<Slice>& operand_list,
58 std::string* new_value,
59 Logger* logger) const override {
60 assert(new_value->empty());
61 ++num_partial_merge_calls;
62 return mergeOperator_->PartialMergeMulti(key, operand_list, new_value,
63 logger);
64 }
65
66 const char* Name() const override { return "UInt64AddOperator"; }
67
68 private:
69 std::shared_ptr<MergeOperator> mergeOperator_;
70};

99std::shared_ptr<DB> OpenDb(const std::string& dbname, const bool ttl = false,
100 const size_t max_successive_merges = 0) {
101 DB* db;
102 Options options;
103 options.create_if_missing = true;
104 options.merge_operator = std::make_shared<CountMergeOperator>();
105 options.max_successive_merges = max_successive_merges;
106 options.env = EnvMergeTest::GetInstance();
107 EXPECT_OK(DestroyDB(dbname, Options()));
108 Status s;
109 if (ttl) {
110 DBWithTTL* db_with_ttl;
111 s = DBWithTTL::Open(options, dbname, &db_with_ttl);
112 db = db_with_ttl;
113 } else {
114 s = DB::Open(options, dbname, &db);
115 }
116 EXPECT_OK(s);
117 assert(s.ok());
118 // Allowed to call NowNanos during DB creation (in GenerateRawUniqueId() for
119 // session ID)
120 EnvMergeTest::now_nanos_count_ = 0;
121 return std::shared_ptr<DB>(db);
122}

124// Imagine we are maintaining a set of uint64 counters.
125// Each counter has a distinct name. And we would like
126// to support four high level operations:
127// set, add, get and remove
128// This is a quick implementation without a Merge operation.
129class Counters {
130 protected:
131 std::shared_ptr<DB> db_;
132
133 WriteOptions put_option_;
134 ReadOptions get_option_;
135 WriteOptions delete_option_;
136
137 uint64_t default_;
138
139 public:
140 explicit Counters(std::shared_ptr<DB> db, uint64_t defaultCount = 0)
141 : db_(db),
142 put_option_(),
143 get_option_(),
144 delete_option_(),
145 default_(defaultCount) {
146 assert(db_);
147 }
148
149 virtual ~Counters() = default;
150
151 // public interface of Counters.
152 // All four functions return false
153 // if the underlying level db operation failed.
154
155 // mapped to a levedb Put
156 bool set(const std::string& key, uint64_t value) {
157 // just treat the internal rep of int64 as the string
158 char buf[sizeof(value)];
159 EncodeFixed64(buf, value);
160 Slice slice(buf, sizeof(value));
161 auto s = db_->Put(put_option_, key, slice);
162
163 if (s.ok()) {
164 return true;
165 } else {
166 std::cerr << s.ToString() << std::endl;
167 return false;
168 }
169 }
170
171 // mapped to a rocksdb Delete
172 bool remove(const std::string& key) {
173 auto s = db_->Delete(delete_option_, key);
174
175 if (s.ok()) {
176 return true;
177 } else {
178 std::cerr << s.ToString() << std::endl;
179 return false;
180 }
181 }
182
183 // mapped to a rocksdb Get
184 bool get(const std::string& key, uint64_t* value) {
185 std::string str;
186 auto s = db_->Get(get_option_, key, &str);
187
188 if (s.IsNotFound()) {
189 // return default value if not found;
190 *value = default_;
191 return true;
192 } else if (s.ok()) {
193 // deserialization
194 if (str.size() != sizeof(uint64_t)) {
195 std::cerr << "value corruption\n";
196 return false;
197 }
198 *value = DecodeFixed64(str.data());
199 return true;
200 } else {
201 std::cerr << s.ToString() << std::endl;
202 return false;
203 }
204 }
205
206 // 'add' is implemented as get -> modify -> set
207 // An alternative is a single merge operation, see MergeBasedCounters
208 virtual bool add(const std::string& key, uint64_t value) {
209 uint64_t base = default_;
210 return get(key, &base) && set(key, base + value);
211 }
212
213 // convenience functions for testing
214 void assert_set(const std::string& key, uint64_t value) {
215 assert(set(key, value));
216 }
217
218 void assert_remove(const std::string& key) { assert(remove(key)); }
219
220 uint64_t assert_get(const std::string& key) {
221 uint64_t value = default_;
222 int result = get(key, &value);
223 assert(result);
224 if (result == 0) {
225 exit(1); // Disable unused variable warning.
226 }
227 return value;
228 }
229
230 void assert_add(const std::string& key, uint64_t value) {
231 int result = add(key, value);
232 assert(result);
233 if (result == 0) {
234 exit(1); // Disable unused variable warning.
235 }
236 }
237};

259 return false;
260 }

273void testCounters(Counters& counters, DB* db, bool test_compaction) {
274 FlushOptions o;
275 o.wait = true;
276
277 counters.assert_set("a", 1);
278
279 if (test_compaction) {
280 ASSERT_OK(db->Flush(o));
281 }
282
283 ASSERT_EQ(counters.assert_get("a"), 1);
284
285 counters.assert_remove("b");
286
287 // defaut value is 0 if non-existent
288 ASSERT_EQ(counters.assert_get("b"), 0);
289
290 counters.assert_add("a", 2);
291
292 if (test_compaction) {
293 ASSERT_OK(db->Flush(o));
294 }
295
296 // 1+2 = 3
297 ASSERT_EQ(counters.assert_get("a"), 3);
298
299 dumpDb(db);
300
301 // 1+...+49 = ?
302 uint64_t sum = 0;
303 for (int i = 1; i < 50; i++) {
304 counters.assert_add("b", i);
305 sum += i;
306 }
307 ASSERT_EQ(counters.assert_get("b"), sum);
308
309 dumpDb(db);
310
311 if (test_compaction) {
312 ASSERT_OK(db->Flush(o));
313
314 ASSERT_OK(db->CompactRange(CompactRangeOptions(), nullptr, nullptr));
315
316 dumpDb(db);
317
318 ASSERT_EQ(counters.assert_get("a"), 3);
319 ASSERT_EQ(counters.assert_get("b"), sum);
320 }

413void testSuccessiveMerge(Counters& counters, size_t max_num_merges,
414 size_t num_merges) {
415 counters.assert_remove("z");
416 uint64_t sum = 0;
417
418 for (size_t i = 1; i <= num_merges; ++i) {
419 resetNumMergeOperatorCalls();
420 counters.assert_add("z", i);
421 sum += i;
422
423 if (i % (max_num_merges + 1) == 0) {
424 ASSERT_EQ(num_merge_operator_calls, max_num_merges + 1);
425 } else {
426 ASSERT_EQ(num_merge_operator_calls, 0);
427 }
428
429 resetNumMergeOperatorCalls();
430 ASSERT_EQ(counters.assert_get("z"), sum);
431 ASSERT_EQ(num_merge_operator_calls, i % (max_num_merges + 1));
432 }
433}

435void testPartialMerge(Counters* counters, DB* db, size_t max_merge,
436 size_t min_merge, size_t count) {
437 FlushOptions o;
438 o.wait = true;
439
440 // Test case 1: partial merge should be called when the number of merge
441 // operands exceeds the threshold.
442 uint64_t tmp_sum = 0;
443 resetNumPartialMergeCalls();
444 for (size_t i = 1; i <= count; i++) {
445 counters->assert_add("b", i);
446 tmp_sum += i;
447 }
448 ASSERT_OK(db->Flush(o));
449 ASSERT_OK(db->CompactRange(CompactRangeOptions(), nullptr, nullptr));
450 ASSERT_EQ(tmp_sum, counters->assert_get("b"));
451 if (count > max_merge) {
452 // in this case, FullMerge should be called instead.
453 ASSERT_EQ(num_partial_merge_calls, 0U);
454 } else {
455 // if count >= min_merge, then partial merge should be called once.
456 ASSERT_EQ((count >= min_merge), (num_partial_merge_calls == 1));
457 }
458
459 // Test case 2: partial merge should not be called when a put is found.
460 resetNumPartialMergeCalls();
461 tmp_sum = 0;
462 ASSERT_OK(db->Put(ROCKSDB_NAMESPACE::WriteOptions(), "c", "10"));
463 for (size_t i = 1; i <= count; i++) {
464 counters->assert_add("c", i);
465 tmp_sum += i;
466 }
467 ASSERT_OK(db->Flush(o));
468 ASSERT_OK(db->CompactRange(CompactRangeOptions(), nullptr, nullptr));
469 ASSERT_EQ(tmp_sum, counters->assert_get("c"));
470 ASSERT_EQ(num_partial_merge_calls, 0U);
471 // NowNanos was previously called in MergeHelper::FilterMerge(), which
472 // harmed performance.
473 ASSERT_EQ(EnvMergeTest::now_nanos_count_, 0U);
474}

476void testSingleBatchSuccessiveMerge(DB* db, size_t max_num_merges,
477 size_t num_merges) {
478 ASSERT_GT(num_merges, max_num_merges);
479
480 Slice key("BatchSuccessiveMerge");
481 uint64_t merge_value = 1;
482 char buf[sizeof(merge_value)];
483 EncodeFixed64(buf, merge_value);
484 Slice merge_value_slice(buf, sizeof(merge_value));
485
486 // Create the batch
487 WriteBatch batch;
488 for (size_t i = 0; i < num_merges; ++i) {
489 ASSERT_OK(batch.Merge(key, merge_value_slice));
490 }
491
492 // Apply to memtable and count the number of merges
493 resetNumMergeOperatorCalls();
494 ASSERT_OK(db->Write(WriteOptions(), &batch));
495 ASSERT_EQ(
496 num_merge_operator_calls,
497 static_cast<size_t>(num_merges - (num_merges % (max_num_merges + 1))));
498
499 // Get the value
500 resetNumMergeOperatorCalls();
501 std::string get_value_str;
502 ASSERT_OK(db->Get(ReadOptions(), key, &get_value_str));
503 assert(get_value_str.size() == sizeof(uint64_t));
504 uint64_t get_value = DecodeFixed64(get_value_str.data());
505 ASSERT_EQ(get_value, num_merges * merge_value);
506 ASSERT_EQ(num_merge_operator_calls,
507 static_cast<size_t>((num_merges % (max_num_merges + 1))));
508}

687 const WideColumns entity{{"one", "1"}, {"two", "2"}};
688 MergeOperator::MergeOperationInputV3::ExistingValue existing_value(entity);
689 const MergeOperator::MergeOperationInputV3 merge_in(
690 key, std::move(existing_value), operands, logger);
691
692 MergeOperator::MergeOperationOutputV3 merge_out;
693
694 ASSERT_TRUE(append_operator->FullMergeV3(merge_in, &merge_out));
