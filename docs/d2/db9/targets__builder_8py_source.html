11class LiteralValue:
12 def __init__(self, value):
13 self.value = value
14
15 def __str__(self):
16 return str(self.value)

18def smart_quote_value(val):
19 if isinstance(val, LiteralValue):
20 return str(val)
21 return '"%s"' % val

23def pretty_list(lst, indent=8):
24 if lst is None or len(lst) == 0:
25 return ""
26
27 if len(lst) == 1:
28 return smart_quote_value(lst[0])
29
30 separator = ',\n%s' % (" " * indent)
31 res = separator.join(sorted(map(smart_quote_value, lst)))
32 res = "\n" + (" " * indent) + res + ',\n' + (" " * (indent - 4))
33 return res
