39Status BlobFileCache::GetBlobFileReader(
40 const ReadOptions& read_options, uint64_t blob_file_number,
41 CacheHandleGuard<BlobFileReader>* blob_file_reader) {
42 assert(blob_file_reader);
43 assert(blob_file_reader->IsEmpty());
44
45 // NOTE: sharing same Cache with table_cache
46 const Slice key = GetSliceForKey(&blob_file_number);
47
48 assert(cache_);
49
50 TypedHandle* handle = cache_.Lookup(key);
51 if (handle) {
52 *blob_file_reader = cache_.Guard(handle);
53 return Status::OK();
54 }
55
56 TEST_SYNC_POINT("BlobFileCache::GetBlobFileReader:DoubleCheck");
57
58 // Check again while holding mutex
59 MutexLock lock(&mutex_.Get(key));
60
61 handle = cache_.Lookup(key);
62 if (handle) {
63 *blob_file_reader = cache_.Guard(handle);
64 return Status::OK();
65 }
66
67 assert(immutable_options_);
68 Statistics* const statistics = immutable_options_->stats;
69
70 RecordTick(statistics, NO_FILE_OPENS);
71
72 std::unique_ptr<BlobFileReader> reader;
73
74 {
75 assert(file_options_);
76 const Status s = BlobFileReader::Create(
77 *immutable_options_, read_options, *file_options_, column_family_id_,
78 blob_file_read_hist_, blob_file_number, io_tracer_, &reader);
79 if (!s.ok()) {
80 RecordTick(statistics, NO_FILE_ERRORS);
81 return s;
82 }
83 }
84
85 {
86 constexpr size_t charge = 1;
87
88 const Status s = cache_.Insert(key, reader.get(), charge, &handle);
89 if (!s.ok()) {
90 RecordTick(statistics, NO_FILE_ERRORS);
91 return s;
92 }
93 }
94
95 reader.release();
96
97 *blob_file_reader = cache_.Guard(handle);
98
99 return Status::OK();
100}
