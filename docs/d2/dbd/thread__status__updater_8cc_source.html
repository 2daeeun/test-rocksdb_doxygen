22void ThreadStatusUpdater::RegisterThread(ThreadStatus::ThreadType ttype,
23 uint64_t thread_id) {
24 if (UNLIKELY(thread_status_data_ == nullptr)) {
25 thread_status_data_ = new ThreadStatusData();
26 thread_status_data_->thread_type = ttype;
27 thread_status_data_->thread_id = thread_id;
28 std::lock_guard<std::mutex> lck(thread_list_mutex_);
29 thread_data_set_.insert(thread_status_data_);
30 }
31
32 ClearThreadOperationProperties();
33}

35void ThreadStatusUpdater::UnregisterThread() {
36 if (thread_status_data_ != nullptr) {
37 std::lock_guard<std::mutex> lck(thread_list_mutex_);
38 thread_data_set_.erase(thread_status_data_);
39 delete thread_status_data_;
40 thread_status_data_ = nullptr;
41 }

74void ThreadStatusUpdater::SetThreadOperation(
75 const ThreadStatus::OperationType type) {
76 auto* data = GetLocalThreadStatus();
77 if (data == nullptr) {
78 return;
79 }
80 // NOTE: Our practice here is to set all the thread operation properties
81 // and stage before we set thread operation, and thread operation
82 // will be set in std::memory_order_release. This is to ensure
83 // whenever a thread operation is not OP_UNKNOWN, we will always
84 // have a consistent information on its properties.
85 data->operation_type.store(type, std::memory_order_release);
86 if (type == ThreadStatus::OP_UNKNOWN) {
87 data->operation_stage.store(ThreadStatus::STAGE_UNKNOWN,
88 std::memory_order_relaxed);
89 ClearThreadOperationProperties();
90 }
91}

174Status ThreadStatusUpdater::GetThreadList(
175 std::vector<ThreadStatus>* thread_list) {
176 thread_list->clear();
177 std::vector<std::shared_ptr<ThreadStatusData>> valid_list;
178 uint64_t now_micros = SystemClock::Default()->NowMicros();
179
180 std::lock_guard<std::mutex> lck(thread_list_mutex_);
181 for (auto* thread_data : thread_data_set_) {
182 assert(thread_data);
183 auto thread_id = thread_data->thread_id.load(std::memory_order_relaxed);
184 auto thread_type = thread_data->thread_type.load(std::memory_order_relaxed);
185 // Since any change to cf_info_map requires thread_list_mutex,
186 // which is currently held by GetThreadList(), here we can safely
187 // use "memory_order_relaxed" to load the cf_key.
188 auto cf_key = thread_data->cf_key.load(std::memory_order_relaxed);
189
190 ThreadStatus::OperationType op_type = ThreadStatus::OP_UNKNOWN;
191 ThreadStatus::OperationStage op_stage = ThreadStatus::STAGE_UNKNOWN;
192 ThreadStatus::StateType state_type = ThreadStatus::STATE_UNKNOWN;
193 uint64_t op_elapsed_micros = 0;
194 uint64_t op_props[ThreadStatus::kNumOperationProperties] = {0};
195
196 auto iter = cf_info_map_.find(cf_key);
197 if (iter != cf_info_map_.end()) {
198 op_type = thread_data->operation_type.load(std::memory_order_acquire);
199 // display lower-level info only when higher-level info is available.
200 if (op_type != ThreadStatus::OP_UNKNOWN) {
201 op_elapsed_micros = now_micros - thread_data->op_start_time.load(
202 std::memory_order_relaxed);
203 op_stage = thread_data->operation_stage.load(std::memory_order_relaxed);
204 state_type = thread_data->state_type.load(std::memory_order_relaxed);
205 for (int i = 0; i < ThreadStatus::kNumOperationProperties; ++i) {
206 op_props[i] =
207 thread_data->op_properties[i].load(std::memory_order_relaxed);
208 }
209 }
210 }
211
212 thread_list->emplace_back(
213 thread_id, thread_type,
214 iter != cf_info_map_.end() ? iter->second.db_name : "",
215 iter != cf_info_map_.end() ? iter->second.cf_name : "", op_type,
216 op_elapsed_micros, op_stage, op_props, state_type);
217 }
218
219 return Status::OK();
220}

232void ThreadStatusUpdater::NewColumnFamilyInfo(const void* db_key,
233 const std::string& db_name,
234 const void* cf_key,
235 const std::string& cf_name) {
236 // Acquiring same lock as GetThreadList() to guarantee
237 // a consistent view of global column family table (cf_info_map).
238 std::lock_guard<std::mutex> lck(thread_list_mutex_);
239
240 cf_info_map_.emplace(std::piecewise_construct, std::make_tuple(cf_key),
241 std::make_tuple(db_key, db_name, cf_name));
242 db_key_map_[db_key].insert(cf_key);
243}

245void ThreadStatusUpdater::EraseColumnFamilyInfo(const void* cf_key) {
246 // Acquiring same lock as GetThreadList() to guarantee
247 // a consistent view of global column family table (cf_info_map).
248 std::lock_guard<std::mutex> lck(thread_list_mutex_);
249
250 auto cf_pair = cf_info_map_.find(cf_key);
251 if (cf_pair != cf_info_map_.end()) {
252 // Remove its entry from db_key_map_ by the following steps:
253 // 1. Obtain the entry in db_key_map_ whose set contains cf_key
254 // 2. Remove it from the set.
255 ConstantColumnFamilyInfo& cf_info = cf_pair->second;
256 auto db_pair = db_key_map_.find(cf_info.db_key);
257 assert(db_pair != db_key_map_.end());
258 size_t result __attribute__((__unused__));
259 result = db_pair->second.erase(cf_key);
260 assert(result);
261 cf_info_map_.erase(cf_pair);
262 }
263}

265void ThreadStatusUpdater::EraseDatabaseInfo(const void* db_key) {
266 // Acquiring same lock as GetThreadList() to guarantee
267 // a consistent view of global column family table (cf_info_map).
268 std::lock_guard<std::mutex> lck(thread_list_mutex_);
269 auto db_pair = db_key_map_.find(db_key);
270 if (UNLIKELY(db_pair == db_key_map_.end())) {
271 // In some occasional cases such as DB::Open fails, we won't
272 // register ColumnFamilyInfo for a db.
273 return;
274 }
275
276 for (auto cf_key : db_pair->second) {
277 auto cf_pair = cf_info_map_.find(cf_key);
278 if (cf_pair != cf_info_map_.end()) {
279 cf_info_map_.erase(cf_pair);
280 }
281 }
282 db_key_map_.erase(db_key);
283}
