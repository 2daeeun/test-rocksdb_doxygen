17def main(args):
18 # initialise the RulesSpec parser
19 rule_spec_parser = RulesSpec(args.rules_spec)
20 # initialise the benchmark runner
21 bench_runner_module = __import__(
22 args.benchrunner_module, fromlist=[args.benchrunner_class]
23 )
24 bench_runner_class = getattr(bench_runner_module, args.benchrunner_class)
25 ods_args = {}
26 if args.ods_client and args.ods_entity:
27 ods_args["client_script"] = args.ods_client
28 ods_args["entity"] = args.ods_entity
29 if args.ods_key_prefix:
30 ods_args["key_prefix"] = args.ods_key_prefix
31 db_bench_runner = bench_runner_class(args.benchrunner_pos_args, ods_args)
32 # initialise the database configuration
33 db_options = DatabaseOptions(args.rocksdb_options, args.misc_options)
34 # set the frequency at which stats are dumped in the LOG file and the
35 # location of the LOG file.
36 db_log_dump_settings = {
37 "DBOptions.stats_dump_period_sec": {NO_COL_FAMILY: args.stats_dump_period_sec}
38 }
39 db_options.update_options(db_log_dump_settings)
40 # initialise the configuration optimizer
41 config_optimizer = ConfigOptimizer(
42 db_bench_runner, db_options, rule_spec_parser, args.base_db_path
43 )
44 # run the optimiser to improve the database configuration for given
45 # benchmarks, with the help of expert-specified rules
46 final_db_options = config_optimizer.run()
47 # generate the final rocksdb options file
48 print(
49 "Final configuration in: " + final_db_options.generate_options_config("final")
50 )
51 print("Final miscellaneous options: " + repr(final_db_options.get_misc_options()))
