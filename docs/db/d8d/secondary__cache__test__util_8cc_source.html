51// If helpers without_secondary are provided, returns helpers with secondary
52// support. If not provided, returns helpers without secondary support.
53auto GenerateHelpersByRole(
54 const std::array<Cache::CacheItemHelper, kNumCacheEntryRoles>*
55 without_secondary,
56 bool fail) {
57 std::array<Cache::CacheItemHelper, kNumCacheEntryRoles> a;
58 for (uint32_t i = 0; i < kNumCacheEntryRoles; ++i) {
59 if (without_secondary) {
60 a[i] =
61 Cache::CacheItemHelper{static_cast<CacheEntryRole>(i),
62 &DeletionCallback,
63 &SizeCallback,
64 fail ? &SaveToCallbackFail : &SaveToCallback,
65 &CreateCallback,
66 &(*without_secondary)[i]};
67 } else {
68 a[i] = Cache::CacheItemHelper{static_cast<CacheEntryRole>(i),
69 &DeletionCallback};
70 }
71 }
72 return a;
73}

76const Cache::CacheItemHelper* WithCacheType::GetHelper(
77 CacheEntryRole r, bool secondary_compatible, bool fail) {
78 static const std::array<Cache::CacheItemHelper, kNumCacheEntryRoles>
79 without_secondary = GenerateHelpersByRole(nullptr, false);
80 static const std::array<Cache::CacheItemHelper, kNumCacheEntryRoles>
81 with_secondary = GenerateHelpersByRole(&without_secondary, false);
82 static const std::array<Cache::CacheItemHelper, kNumCacheEntryRoles>
83 with_secondary_fail = GenerateHelpersByRole(&without_secondary, true);
84 return &(fail ? with_secondary_fail
85 : secondary_compatible ? with_secondary
86 : without_secondary)[static_cast<int>(r)];
87}

89const Cache::CacheItemHelper* WithCacheType::GetHelperFail(CacheEntryRole r) {
90 return GetHelper(r, true, true);
