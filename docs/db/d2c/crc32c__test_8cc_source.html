66TEST(CRC, StandardResults) {
67 // Original Fast_CRC32 tests.
68 // From rfc3720 section B.4.
69 char buf[32];
70
71 memset(buf, 0, sizeof(buf));
72 ASSERT_EQ(0x8a9136aaU, Value(buf, sizeof(buf)));
73
74 memset(buf, 0xff, sizeof(buf));
75 ASSERT_EQ(0x62a8ab43U, Value(buf, sizeof(buf)));
76
77 for (int i = 0; i < 32; i++) {
78 buf[i] = static_cast<char>(i);
79 }
80 ASSERT_EQ(0x46dd794eU, Value(buf, sizeof(buf)));
81
82 for (int i = 0; i < 32; i++) {
83 buf[i] = static_cast<char>(31 - i);

85 ASSERT_EQ(0x113fdb5cU, Value(buf, sizeof(buf)));

95 // 3-Way Crc32c tests ported from folly.
96 // Test 1: single computation
97 for (auto expected : expectedResults) {
98 uint32_t result = Value(buffer + expected.offset, expected.length);
99 EXPECT_EQ(~expected.crc32c, result);
100 }
101
102 // Test 2: stitching two computations
103 for (auto expected : expectedResults) {
104 size_t partialLength = expected.length / 2;
105 uint32_t partialChecksum = Value(buffer + expected.offset, partialLength);
106 uint32_t result =
107 Extend(partialChecksum, buffer + expected.offset + partialLength,
108 expected.length - partialLength);
109 EXPECT_EQ(~expected.crc32c, result);
110 }
111}

174// copied from folly
175const uint64_t FNV_64_HASH_START = 14695981039346656037ULL;
176inline uint64_t fnv64_buf(const void* buf, size_t n,
177 uint64_t hash = FNV_64_HASH_START) {
178 // forcing signed char, since other platforms can use unsigned
179 const signed char* char_buf = reinterpret_cast<const signed char*>(buf);
180
181 for (size_t i = 0; i < n; ++i) {
182 hash += (hash << 1) + (hash << 4) + (hash << 5) + (hash << 7) +
183 (hash << 8) + (hash << 40);
184 hash ^= char_buf[i];
185 }
186 return hash;
187}

189int main(int argc, char** argv) {
190 ROCKSDB_NAMESPACE::port::InstallStackTraceHandler();
191 ::testing::InitGoogleTest(&argc, argv);
192
193 // Populate a buffer with a deterministic pattern
194 // on which to compute checksums
195
196 const uint8_t* src = (uint8_t*)ROCKSDB_NAMESPACE::crc32c::buffer;
197 uint64_t* dst = (uint64_t*)ROCKSDB_NAMESPACE::crc32c::buffer;
198 const uint64_t* end =
199 (const uint64_t*)(ROCKSDB_NAMESPACE::crc32c::buffer +
200 ROCKSDB_NAMESPACE::crc32c::BUFFER_SIZE);
201 *dst++ = 0;
202 while (dst < end) {
203 ROCKSDB_NAMESPACE::EncodeFixed64(
204 reinterpret_cast<char*>(dst),
205 fnv64_buf((const char*)src, sizeof(uint64_t)));
206 dst++;
207 src += sizeof(uint64_t);
208 }
209
210 return RUN_ALL_TESTS();
211}
