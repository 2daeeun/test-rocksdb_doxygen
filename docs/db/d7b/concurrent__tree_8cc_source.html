74void concurrent_tree::locked_keyrange::prepare(concurrent_tree *tree) {
75 // the first step in acquiring a locked keyrange is locking the root
76 treenode *const root = &tree->m_root;
77 m_tree = tree;
78 m_subtree = root;
79 m_range = keyrange::get_infinite_range();
80 root->mutex_lock();
81}
82
83void concurrent_tree::locked_keyrange::acquire(const keyrange &range) {
84 treenode *const root = &m_tree->m_root;
85
86 treenode *subtree;
87 if (root->is_empty() || root->range_overlaps(range)) {
88 subtree = root;
89 } else {
90 // we do not have a precomputed comparison hint, so pass null
91 const keyrange::comparison *cmp_hint = nullptr;
92 subtree = root->find_node_with_overlapping_child(range, cmp_hint);
93 }
94
95 // subtree is locked. it will be unlocked when this is release()'d
96 invariant_notnull(subtree);
97 m_range = range;
98 m_subtree = subtree;
99}

101bool concurrent_tree::locked_keyrange::add_shared_owner(const keyrange &range,
102 TXNID new_owner) {
103 return m_subtree->insert(range, new_owner, /*is_shared*/ true);
104}

106void concurrent_tree::locked_keyrange::release(void) {
107 m_subtree->mutex_unlock();
108}
109
110void concurrent_tree::locked_keyrange::insert(const keyrange &range,
111 TXNID txnid, bool is_shared) {
112 // empty means no children, and only the root should ever be empty
113 if (m_subtree->is_empty()) {
114 m_subtree->set_range_and_txnid(range, txnid, is_shared);
115 } else {
116 m_subtree->insert(range, txnid, is_shared);
117 }
118}
119
120void concurrent_tree::locked_keyrange::remove(const keyrange &range,
121 TXNID txnid) {
122 invariant(!m_subtree->is_empty());
123 treenode *new_subtree = m_subtree->remove(range, txnid);
124 // if removing range changed the root of the subtree,
125 // then the subtree must be the root of the entire tree.
126 if (new_subtree == nullptr) {
127 invariant(m_subtree->is_root());
128 invariant(m_subtree->is_empty());
129 }
130}
131
132void concurrent_tree::locked_keyrange::remove_all(void) {
133 m_subtree->recursive_remove();
134}
