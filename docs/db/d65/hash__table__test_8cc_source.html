82TEST_F(HashTableTest, TestInsert) {
83 const uint64_t max_keys = 1024 * 1024;
84
85 // insert
86 for (uint64_t k = 0; k < max_keys; ++k) {
87 map_.Insert(Node(k, std::string(1000, k % 255)));
88 }
89
90 // verify
91 for (uint64_t k = 0; k < max_keys; ++k) {
92 Node val;
93 port::RWMutex* rlock = nullptr;
94 assert(map_.Find(Node(k), &val, &rlock));
95 rlock->ReadUnlock();
96 assert(val.val_ == std::string(1000, k % 255));

100TEST_F(HashTableTest, TestErase) {
101 const uint64_t max_keys = 1024 * 1024;
102 // insert
103 for (uint64_t k = 0; k < max_keys; ++k) {
104 map_.Insert(Node(k, std::string(1000, k % 255)));
105 }
106
107 auto rand = Random64(time(nullptr));
108 // erase a few keys randomly
109 std::set<uint64_t> erased;
110 for (int i = 0; i < 1024; ++i) {
111 uint64_t k = rand.Next() % max_keys;
112 if (erased.find(k) != erased.end()) {
113 continue;
114 }
115 assert(map_.Erase(Node(k), /*ret=*/nullptr));
116 erased.insert(k);
117 }
118
119 // verify
120 for (uint64_t k = 0; k < max_keys; ++k) {
121 Node val;
122 port::RWMutex* rlock = nullptr;
123 bool status = map_.Find(Node(k), &val, &rlock);
124 if (erased.find(k) == erased.end()) {
125 assert(status);
126 rlock->ReadUnlock();
127 assert(val.val_ == std::string(1000, k % 255));
128 } else {
129 assert(!status);
130 }
131 }
132}

144 Node* val = map_.Evict();

148 assert(val->val_ == std::string(1000, val->key_ % 255));
