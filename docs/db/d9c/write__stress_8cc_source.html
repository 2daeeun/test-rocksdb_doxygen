7// The goal of this tool is to be a simple stress test with focus on catching:
8// * bugs in compaction/flush processes, especially the ones that cause
9// assertion errors
10// * bugs in the code that deletes obsolete files
11//
12// There are two parts of the test:
13// * write_stress, a binary that writes to the database
14// * write_stress_runner.py, a script that invokes and kills write_stress
15//
16// Here are some interesting parts of write_stress:
17// * Runs with very high concurrency of compactions and flushes (32 threads
18// total) and tries to create a huge amount of small files
19// * The keys written to the database are not uniformly distributed -- there is
20// a 3-character prefix that mutates occasionally (in prefix mutator thread), in
21// such a way that the first character mutates slower than second, which mutates
22// slower than third character. That way, the compaction stress tests some
23// interesting compaction features like trivial moves and bottommost level
24// calculation
25// * There is a thread that creates an iterator, holds it for couple of seconds
26// and then iterates over all keys. This is supposed to test RocksDB's abilities
27// to keep the files alive when there are references to them.
28// * Some writes trigger WAL sync. This is stress testing our WAL sync code.
29// * At the end of the run, we make sure that we didn't leak any of the sst
30// files
31//
32// write_stress_runner.py changes the mode in which we run write_stress and also
33// kills and restarts it. There are some interesting characteristics:
34// * At the beginning we divide the full test runtime into smaller parts --
35// shorter runtimes (couple of seconds) and longer runtimes (100, 1000) seconds
36// * The first time we run write_stress, we destroy the old DB. Every next time
37// during the test, we use the same DB.
38// * We can run in kill mode or clean-restart mode. Kill mode kills the
39// write_stress violently.
40// * We can run in mode where delete_obsolete_files_with_fullscan is true or
41// false
42// * We can run with low_open_files mode turned on or off. When it's turned on,
43// we configure table cache to only hold a couple of files -- that way we need
44// to reopen files every time we access them.
45//
46// Another goal was to create a stress test without a lot of parameters. So
47// tools/write_stress_runner.py should only take one parameter -- runtime_sec
48// and it should figure out everything else on its own.

108namespace ROCKSDB_NAMESPACE {
109
110static const int kPrefixSize = 3;
111
112class WriteStress {
113 public:
114 WriteStress() : stop_(false) {
115 // initialize key_prefix
116 for (int i = 0; i < kPrefixSize; ++i) {
117 key_prefix_[i].store('a');
118 }
119
120 // Choose a location for the test database if none given with --db=<path>
121 if (FLAGS_db.empty()) {
122 std::string default_db_path;
123 Env::Default()->GetTestDirectory(&default_db_path);
124 default_db_path += "/write_stress";
125 FLAGS_db = default_db_path;
126 }
127
128 Options options;
129 if (FLAGS_destroy_db) {
130 DestroyDB(FLAGS_db, options); // ignore
131 }
132
133 // make the LSM tree deep, so that we have many concurrent flushes and
134 // compactions
135 options.create_if_missing = true;
136 options.write_buffer_size = 256 * 1024; // 256k
137 options.max_bytes_for_level_base = 1 * 1024 * 1024; // 1MB
138 options.target_file_size_base = 100 * 1024; // 100k
139 options.max_write_buffer_number = 16;
140 options.max_background_compactions = 16;
141 options.max_background_flushes = 16;
142 options.max_open_files = FLAGS_low_open_files_mode ? 20 : -1;
143 if (FLAGS_delete_obsolete_files_with_fullscan) {
144 options.delete_obsolete_files_period_micros = 0;
145 }
146
147 // open DB
148 DB* db;
149 Status s = DB::Open(options, FLAGS_db, &db);
150 if (!s.ok()) {
151 fprintf(stderr, "Can't open database: %s\n", s.ToString().c_str());
152 std::abort();
153 }
154 db_.reset(db);
155 }
156
157 void WriteThread() {
158 std::mt19937 rng(static_cast<unsigned int>(FLAGS_seed));
159 std::uniform_real_distribution<double> dist(0, 1);
160
161 auto random_string = [](std::mt19937& r, int len) {
162 std::uniform_int_distribution<int> char_dist('a', 'z');
163 std::string ret;
164 for (int i = 0; i < len; ++i) {
165 ret += static_cast<char>(char_dist(r));
166 }
167 return ret;
168 };
169
170 while (!stop_.load(std::memory_order_relaxed)) {
171 std::string prefix;
172 prefix.resize(kPrefixSize);
173 for (int i = 0; i < kPrefixSize; ++i) {
174 prefix[i] = key_prefix_[i].load(std::memory_order_relaxed);
175 }
176 auto key = prefix + random_string(rng, FLAGS_key_size - kPrefixSize);
177 auto value = random_string(rng, FLAGS_value_size);
178 WriteOptions woptions;
179 woptions.sync = dist(rng) < FLAGS_sync_probability;
180 auto s = db_->Put(woptions, key, value);
181 if (!s.ok()) {
182 fprintf(stderr, "Write to DB failed: %s\n", s.ToString().c_str());
183 std::abort();
184 }
185 }
186 }
187
188 void IteratorHoldThread() {
189 while (!stop_.load(std::memory_order_relaxed)) {
190 std::unique_ptr<Iterator> iterator(db_->NewIterator(ReadOptions()));
191 SystemClock::Default()->SleepForMicroseconds(FLAGS_iterator_hold_sec *
192 1000 * 1000LL);
193 for (iterator->SeekToFirst(); iterator->Valid(); iterator->Next()) {
194 }
195 if (!iterator->status().ok()) {
196 fprintf(stderr, "Iterator statuts not OK: %s\n",
197 iterator->status().ToString().c_str());
198 std::abort();
199 }
200 }
201 }
202
203 void PrefixMutatorThread() {
204 std::mt19937 rng(static_cast<unsigned int>(FLAGS_seed));
205 std::uniform_real_distribution<double> dist(0, 1);
206 std::uniform_int_distribution<int> char_dist('a', 'z');
207 while (!stop_.load(std::memory_order_relaxed)) {
208 SystemClock::Default()->SleepForMicroseconds(
209 static_cast<int>(FLAGS_prefix_mutate_period_sec * 1000 * 1000LL));
210 if (dist(rng) < FLAGS_first_char_mutate_probability) {
211 key_prefix_[0].store(static_cast<char>(char_dist(rng)),
212 std::memory_order_relaxed);
213 }
214 if (dist(rng) < FLAGS_second_char_mutate_probability) {
215 key_prefix_[1].store(static_cast<char>(char_dist(rng)),
216 std::memory_order_relaxed);
217 }
218 if (dist(rng) < FLAGS_third_char_mutate_probability) {
219 key_prefix_[2].store(static_cast<char>(char_dist(rng)),
220 std::memory_order_relaxed);
221 }
222 }
223 }
224
225 int Run() {
226 threads_.emplace_back([&]() { WriteThread(); });
227 threads_.emplace_back([&]() { PrefixMutatorThread(); });
228 threads_.emplace_back([&]() { IteratorHoldThread(); });
229
230 if (FLAGS_runtime_sec == -1) {
231 // infinite runtime, until we get killed
232 while (true) {
233 SystemClock::Default()->SleepForMicroseconds(1000 * 1000);
234 }
235 }
236
237 SystemClock::Default()->SleepForMicroseconds(FLAGS_runtime_sec * 1000 *
238 1000);
239
240 stop_.store(true, std::memory_order_relaxed);
241 for (auto& t : threads_) {
242 t.join();
243 }
244 threads_.clear();
245
246 // let's see if we leaked some files
247 db_->PauseBackgroundWork();
248 std::vector<LiveFileMetaData> metadata;
249 db_->GetLiveFilesMetaData(&metadata);
250 std::set<uint64_t> sst_file_numbers;
251 for (const auto& file : metadata) {
252 uint64_t number;
253 FileType type;
254 if (!ParseFileName(file.name, &number, "LOG", &type)) {
255 continue;
256 }
257 if (type == kTableFile) {
258 sst_file_numbers.insert(number);
259 }
260 }
261
262 std::vector<std::string> children;
263 Env::Default()->GetChildren(FLAGS_db, &children);
264 for (const auto& child : children) {
265 uint64_t number;
266 FileType type;
267 if (!ParseFileName(child, &number, "LOG", &type)) {
268 continue;
269 }
270 if (type == kTableFile) {
271 if (sst_file_numbers.find(number) == sst_file_numbers.end()) {
272 fprintf(stderr,
273 "Found a table file in DB path that should have been "
274 "deleted: %s\n",
275 child.c_str());
276 std::abort();
277 }
278 }
279 }
280 db_->ContinueBackgroundWork();
281
282 return 0;
283 }
284
285 private:
286 // each key is prepended with this prefix. we occasionally change it. third
287 // letter is changed more frequently than second, which is changed more
288 // frequently than the first one.
289 std::atomic<char> key_prefix_[kPrefixSize];
290 std::atomic<bool> stop_;
291 std::vector<port::Thread> threads_;
292 std::unique_ptr<DB> db_;
293};
