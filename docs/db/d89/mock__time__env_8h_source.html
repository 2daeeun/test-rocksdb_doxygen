19// NOTE: SpecialEnv offers most of this functionality, along with hooks
20// for safe DB behavior under a mock time environment, so should be used
21// instead of MockSystemClock for DB tests.
22class MockSystemClock : public SystemClockWrapper {
23 public:
24 explicit MockSystemClock(const std::shared_ptr<SystemClock>& base)
25 : SystemClockWrapper(base) {}
26
27 static const char* kClassName() { return "MockSystemClock"; }
28 const char* Name() const override { return kClassName(); }
29 Status GetCurrentTime(int64_t* time_sec) override {
30 assert(time_sec != nullptr);
31 *time_sec = static_cast<int64_t>(current_time_us_ / kMicrosInSecond);
32 return Status::OK();
33 }
34
35 virtual uint64_t NowSeconds() { return current_time_us_ / kMicrosInSecond; }
36
37 uint64_t NowMicros() override { return current_time_us_; }
38
39 uint64_t NowNanos() override {
40 assert(current_time_us_ <= std::numeric_limits<uint64_t>::max() / 1000);
41 return current_time_us_ * 1000;
42 }
43
44 uint64_t RealNowMicros() { return target_->NowMicros(); }
45
46 void SetCurrentTime(uint64_t time_sec) {
47 assert(time_sec < std::numeric_limits<uint64_t>::max() / kMicrosInSecond);
48 assert(time_sec * kMicrosInSecond >= current_time_us_);
49 current_time_us_ = time_sec * kMicrosInSecond;
50 }

52 // It's a fake sleep that just updates the Env current time, which is similar
53 // to `NoSleepEnv.SleepForMicroseconds()` and
54 // `SpecialEnv.MockSleepForMicroseconds()`.
55 // It's also similar to `set_current_time()`, which takes an absolute time in
56 // seconds, vs. this one takes the sleep in microseconds.
57 // Note: Not thread safe.
58 void SleepForMicroseconds(int micros) override {
59 assert(micros >= 0);
60 assert(current_time_us_ + static_cast<uint64_t>(micros) >=
61 current_time_us_);
62 current_time_us_.fetch_add(micros);
63 }
64
65 void MockSleepForSeconds(int seconds) {
66 assert(seconds >= 0);
67 uint64_t micros = static_cast<uint64_t>(seconds) * kMicrosInSecond;
68 assert(current_time_us_ + micros >= current_time_us_);
69 current_time_us_.fetch_add(micros);
70 }
71
72 bool TimedWait(port::CondVar* cv,
73 std::chrono::microseconds deadline) override {
74 uint64_t now_micros = NowMicros();
75 uint64_t deadline_micros = static_cast<uint64_t>(deadline.count());
76 uint64_t delay_micros;
77 if (deadline_micros > now_micros) {
78 delay_micros = deadline_micros - now_micros;
79 } else {
80 delay_micros = 0;
81 }
82 // To prevent slowdown, this `TimedWait()` is completely synthetic. First,
83 // it yields to coerce other threads to run while the lock is released.
84 // Second, it randomly selects between mocking an immediate wakeup and a
85 // timeout.
86 cv->GetMutex()->Unlock();
87 std::this_thread::yield();
88 bool mock_timeout = Random::GetTLSInstance()->OneIn(2);
89 if (mock_timeout) {
90 TEST_SYNC_POINT("MockSystemClock::TimedWait:UnlockedPreSleep");
91 current_time_us_.fetch_add(delay_micros);
92 TEST_SYNC_POINT("MockSystemClock::TimedWait:UnlockedPostSleep1");
93 TEST_SYNC_POINT("MockSystemClock::TimedWait:UnlockedPostSleep2");
94 }
95 cv->GetMutex()->Lock();
96 return mock_timeout;
97 }
98
99 // TODO: this is a workaround for the different behavior on different platform
100 // for timedwait timeout. Ideally timedwait API should be moved to env.
101 // details: PR #7101.
102 void InstallTimedWaitFixCallback();
103
104 private:
105 std::atomic<uint64_t> current_time_us_{0};
106 static constexpr uint64_t kMicrosInSecond = 1000U * 1000U;
107};
