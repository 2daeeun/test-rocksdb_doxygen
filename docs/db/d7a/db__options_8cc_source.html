608 bool OptionsAreEqual(const ConfigOptions& config_options,
609 const OptionTypeInfo& opt_info,
610 const std::string& opt_name, const void* const this_ptr,
611 const void* const that_ptr,
612 std::string* mismatch) const override {
613 bool equals = opt_info.AreEqual(config_options, opt_name, this_ptr,
614 that_ptr, mismatch);
615 if (!equals && opt_info.IsByName()) {
616 if (opt_map_ == nullptr) {
617 equals = true;
618 } else {
619 const auto& iter = opt_map_->find(opt_name);
620 if (iter == opt_map_->end()) {
621 equals = true;
622 } else {
623 equals = opt_info.AreEqualByName(config_options, opt_name, this_ptr,
624 iter->second);
625 }
626 }
627 if (equals) { // False alarm, clear mismatch
628 *mismatch = "";
629 }
630 }
631 if (equals && opt_info.IsConfigurable() && opt_map_ != nullptr) {
632 const auto* this_config = opt_info.AsRawPointer<Configurable>(this_ptr);
633 if (this_config == nullptr) {
634 const auto& iter = opt_map_->find(opt_name);
635 // If the name exists in the map and is not empty/null,
636 // then the this_config should be set.
637 if (iter != opt_map_->end() && !iter->second.empty() &&
638 iter->second != kNullptrString) {
639 *mismatch = opt_name;
640 equals = false;
641 }
642 }
643 }
644 return equals;
645 }
646
647 protected:
648 MutableDBOptions mutable_;
649 const std::unordered_map<std::string, std::string>* opt_map_;
650};
