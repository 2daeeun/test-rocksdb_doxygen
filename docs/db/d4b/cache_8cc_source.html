104Status SecondaryCache::CreateFromString(
105 const ConfigOptions& config_options, const std::string& value,
106 std::shared_ptr<SecondaryCache>* result) {
107 if (value.find("compressed_secondary_cache://") == 0) {
108 std::string args = value;
109 args.erase(0, std::strlen("compressed_secondary_cache://"));
110 Status status;
111 std::shared_ptr<SecondaryCache> sec_cache;
112
113 CompressedSecondaryCacheOptions sec_cache_opts;
114 status = OptionTypeInfo::ParseStruct(config_options, "",
115 &comp_sec_cache_options_type_info, "",
116 args, &sec_cache_opts);
117 if (status.ok()) {
118 sec_cache = NewCompressedSecondaryCache(sec_cache_opts);
119 }
120
121
122 if (status.ok()) {
123 result->swap(sec_cache);
124 }
125 return status;
126 } else {
127 return LoadSharedObject<SecondaryCache>(config_options, value, result);
128 }
129}

131Status Cache::CreateFromString(const ConfigOptions& config_options,
132 const std::string& value,
133 std::shared_ptr<Cache>* result) {
134 Status status;
135 std::shared_ptr<Cache> cache;
136 if (StartsWith(value, "null")) {
137 cache = nullptr;
138 } else if (value.find("://") == std::string::npos) {
139 if (value.find('=') == std::string::npos) {
140 cache = NewLRUCache(ParseSizeT(value));
141 } else {
142 LRUCacheOptions cache_opts;
143 status = OptionTypeInfo::ParseStruct(config_options, "",
144 &lru_cache_options_type_info, "",
145 value, &cache_opts);
146 if (status.ok()) {
147 cache = NewLRUCache(cache_opts);
148 }
149 }
150 if (status.ok()) {
151 result->swap(cache);
152 }
153 } else {
154 status = LoadSharedObject<Cache>(config_options, value, result);
155 }
156 return status;
157}

170void Cache::StartAsyncLookup(AsyncLookupHandle& async_handle) {
171 async_handle.found_dummy_entry = false; // in case re-used
172 assert(!async_handle.IsPending());
173 async_handle.result_handle =
174 Lookup(async_handle.key, async_handle.helper, async_handle.create_context,
175 async_handle.priority, async_handle.stats);
176}

178Cache::Handle* Cache::Wait(AsyncLookupHandle& async_handle) {
179 WaitAll(&async_handle, 1);
180 return async_handle.Result();
181}
182
183void Cache::WaitAll(AsyncLookupHandle* async_handles, size_t count) {
184 for (size_t i = 0; i < count; ++i) {
185 if (async_handles[i].IsPending()) {
186 // If a pending handle gets here, it should be marked at "to be handled
187 // by a caller" by that caller erasing the pending_cache on it.
188 assert(async_handles[i].pending_cache == nullptr);
189 }
190 }
191}
