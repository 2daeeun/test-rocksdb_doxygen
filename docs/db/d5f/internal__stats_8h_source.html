476 // Compaction stats, for per_key_placement compaction, it includes 2 levels
477 // stats: the last level and the penultimate level.
478 struct CompactionStatsFull {
479 // the stats for the target primary output level
480 CompactionStats stats;
481
482 // stats for penultimate level output if exist
483 bool has_penultimate_level_output = false;
484 CompactionStats penultimate_level_stats;
485
486 explicit CompactionStatsFull() : stats(), penultimate_level_stats() {}
487
488 explicit CompactionStatsFull(CompactionReason reason, int c)
489 : stats(reason, c), penultimate_level_stats(reason, c) {}
490
491 uint64_t TotalBytesWritten() const {
492 uint64_t bytes_written = stats.bytes_written + stats.bytes_written_blob;
493 if (has_penultimate_level_output) {
494 bytes_written += penultimate_level_stats.bytes_written +
495 penultimate_level_stats.bytes_written_blob;
496 }
497 return bytes_written;
498 }
499
500 uint64_t DroppedRecords() {
501 uint64_t output_records = stats.num_output_records;
502 if (has_penultimate_level_output) {
503 output_records += penultimate_level_stats.num_output_records;
504 }
505 if (stats.num_input_records > output_records) {
506 return stats.num_input_records - output_records;
507 }
508 return 0;
509 }

512 stats.micros = val;
513 penultimate_level_stats.micros = val;
514 }
515
516 void AddCpuMicros(uint64_t val) {
517 stats.cpu_micros += val;
518 penultimate_level_stats.cpu_micros += val;
519 }
520 };
