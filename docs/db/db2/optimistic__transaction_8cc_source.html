61Status OptimisticTransaction::Commit() {
62 auto txn_db_impl = static_cast_with_check<OptimisticTransactionDBImpl,
63 OptimisticTransactionDB>(txn_db_);
64 assert(txn_db_impl);
65 switch (txn_db_impl->GetValidatePolicy()) {
66 case OccValidationPolicy::kValidateParallel:
67 return CommitWithParallelValidate();
68 case OccValidationPolicy::kValidateSerial:
69 return CommitWithSerialValidate();
70 default:
71 assert(0);
72 }
73 // unreachable, just void compiler complain
74 return Status::OK();
75}

77Status OptimisticTransaction::CommitWithSerialValidate() {
78 // Set up callback which will call CheckTransactionForConflicts() to
79 // check whether this transaction is safe to be committed.
80 OptimisticTransactionCallback callback(this);
81
82 DBImpl* db_impl = static_cast_with_check<DBImpl>(db_->GetRootDB());
83
84 Status s = db_impl->WriteWithCallback(
85 write_options_, GetWriteBatch()->GetWriteBatch(), &callback);
86
87 if (s.ok()) {
88 Clear();
89 }
90
91 return s;
92}

94Status OptimisticTransaction::CommitWithParallelValidate() {
95 auto txn_db_impl = static_cast_with_check<OptimisticTransactionDBImpl,
96 OptimisticTransactionDB>(txn_db_);
97 assert(txn_db_impl);
98 DBImpl* db_impl = static_cast_with_check<DBImpl>(db_->GetRootDB());
99 assert(db_impl);
100 std::set<port::Mutex*> lk_ptrs;
101 std::unique_ptr<LockTracker::ColumnFamilyIterator> cf_it(
102 tracked_locks_->GetColumnFamilyIterator());
103 assert(cf_it != nullptr);
104 while (cf_it->HasNext()) {
105 ColumnFamilyId cf = cf_it->Next();
106
107 // To avoid the same key(s) contending across CFs or DBs, seed the
108 // hash independently.
109 uint64_t seed = reinterpret_cast<uintptr_t>(db_impl) +
110 uint64_t{0xb83c07fbc6ced699} /*random prime*/ * cf;
111
112 std::unique_ptr<LockTracker::KeyIterator> key_it(
113 tracked_locks_->GetKeyIterator(cf));
114 assert(key_it != nullptr);
115 while (key_it->HasNext()) {
116 auto lock_bucket_ptr = &txn_db_impl->GetLockBucket(key_it->Next(), seed);
117 TEST_SYNC_POINT_CALLBACK(
118 "OptimisticTransaction::CommitWithParallelValidate::lock_bucket_ptr",
119 lock_bucket_ptr);
120 lk_ptrs.insert(lock_bucket_ptr);
121 }
122 }
123 // NOTE: in a single txn, all bucket-locks are taken in ascending order.
124 // In this way, txns from different threads all obey this rule so that
125 // deadlock can be avoided.
126 for (auto v : lk_ptrs) {
127 // WART: if an exception is thrown during a Lock(), previously locked will
128 // not be Unlock()ed. But a vector of MutexLock is likely inefficient.
129 v->Lock();
130 }
131 Defer unlocks([&]() {
132 for (auto v : lk_ptrs) {
133 v->Unlock();
134 }
135 });
136
137 Status s = TransactionUtil::CheckKeysForConflicts(db_impl, *tracked_locks_,
138 true /* cache_only */);
139 if (!s.ok()) {
140 return s;
141 }
142
143 s = db_impl->Write(write_options_, GetWriteBatch()->GetWriteBatch());
144 if (s.ok()) {
145 Clear();
146 }
147
148 return s;
149}

156// Record this key so that we can check it for conflicts at commit time.
157//
158// 'exclusive' is unused for OptimisticTransaction.
159Status OptimisticTransaction::TryLock(ColumnFamilyHandle* column_family,
160 const Slice& key, bool read_only,
161 bool exclusive, const bool do_validate,
162 const bool assume_tracked) {
163 assert(!assume_tracked); // not supported
164 (void)assume_tracked;
165 if (!do_validate) {
166 return Status::OK();
167 }
168 uint32_t cfh_id = GetColumnFamilyID(column_family);
169
170 SetSnapshotIfNeeded();
171
172 SequenceNumber seq;
173 if (snapshot_) {
174 seq = snapshot_->GetSequenceNumber();
175 } else {
176 seq = db_->GetLatestSequenceNumber();
177 }
178
179 std::string key_str = key.ToString();
180
181 TrackKey(cfh_id, key_str, seq, read_only, exclusive);
182
183 // Always return OK. Confilct checking will happen at commit time.
184 return Status::OK();
185}
186
187// Returns OK if it is safe to commit this transaction. Returns Status::Busy
188// if there are read or write conflicts that would prevent us from committing OR
189// if we can not determine whether there would be any such conflicts.
190//
191// Should only be called on writer thread in order to avoid any race conditions
192// in detecting write conflicts.
193Status OptimisticTransaction::CheckTransactionForConflicts(DB* db) {
194 auto db_impl = static_cast_with_check<DBImpl>(db);
195
196 // Since we are on the write thread and do not want to block other writers,
197 // we will do a cache-only conflict check. This can result in TryAgain
198 // getting returned if there is not sufficient memtable history to check
199 // for conflicts.
200 return TransactionUtil::CheckKeysForConflicts(db_impl, *tracked_locks_,
201 true /* cache_only */);
202}
