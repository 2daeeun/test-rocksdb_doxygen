172Slice PropertyBlockBuilder::Finish() {
173 for (const auto& prop : props_) {
174 assert(last_prop_added_to_block_.empty() ||
175 comparator_->Compare(prop.first, last_prop_added_to_block_) > 0);
176 properties_block_->Add(prop.first, prop.second);
177#ifndef NDEBUG
178 last_prop_added_to_block_ = prop.first;
179#endif /* !NDEBUG */
180 }
181
182 return properties_block_->Finish();
183}

222bool NotifyCollectTableCollectorsOnFinish(
223 const std::vector<std::unique_ptr<InternalTblPropColl>>& collectors,
224 Logger* info_log, PropertyBlockBuilder* builder,
225 UserCollectedProperties& user_collected_properties,
226 UserCollectedProperties& readable_properties) {
227 bool all_succeeded = true;
228 for (auto& collector : collectors) {
229 UserCollectedProperties user_properties;
230 Status s = collector->Finish(&user_properties);
231 if (s.ok()) {
232 for (const auto& prop : collector->GetReadableProperties()) {
233 readable_properties.insert(prop);
234 }
235#ifndef NDEBUG
236 // Check different user properties collectors are not adding properties of
237 // the same name.
238 for (const auto& pair : user_properties) {
239 assert(user_collected_properties.find(pair.first) ==
240 user_collected_properties.end());
241 }
242#endif /* !NDEBUG */
243 user_collected_properties.merge(user_properties);
244 } else {
245 LogPropertiesCollectionError(info_log, "Finish" /* method */,
246 collector->Name());
247 if (all_succeeded) {
248 all_succeeded = false;
249 }
250 }
251 }
252 builder->Add(user_collected_properties);
253 return all_succeeded;
254}

256// FIXME: should be a parameter for reading table properties to use persistent
257// cache?
258Status ReadTablePropertiesHelper(
259 const ReadOptions& ro, const BlockHandle& handle,
260 RandomAccessFileReader* file, FilePrefetchBuffer* prefetch_buffer,
261 const Footer& footer, const ImmutableOptions& ioptions,
262 std::unique_ptr<TableProperties>* table_properties,
263 MemoryAllocator* memory_allocator) {
264 assert(table_properties);
265
266 Status s;
267 bool retry = false;
268 while (true) {
269 BlockContents block_contents;
270 size_t len = handle.size() + footer.GetBlockTrailerSize();
271 // If this is an external SST file ingested with write_global_seqno set to
272 // true, then we expect the checksum mismatch because checksum was written
273 // by SstFileWriter, but its global seqno in the properties block may have
274 // been changed during ingestion. For this reason, we initially read
275 // and process without checksum verification, then later try checksum
276 // verification so that if it fails, we can copy to a temporary buffer with
277 // global seqno set to its original value, i.e. 0, and attempt checksum
278 // verification again.
279 if (!retry) {
280 ReadOptions modified_ro = ro;
281 modified_ro.verify_checksums = false;
282 BlockFetcher block_fetcher(
283 file, prefetch_buffer, footer, modified_ro, handle, &block_contents,
284 ioptions, false /* decompress */, false /*maybe_compressed*/,
285 BlockType::kProperties, UncompressionDict::GetEmptyDict(),
286 PersistentCacheOptions::kEmpty, memory_allocator);
287 s = block_fetcher.ReadBlockContents();
288 if (!s.ok()) {
289 return s;
290 }
291 assert(block_fetcher.GetBlockSizeWithTrailer() == len);
292 TEST_SYNC_POINT_CALLBACK("ReadTablePropertiesHelper:0",
293 &block_contents.data);
294 } else {
295 assert(s.IsCorruption());
296 // If retrying, use a stronger file system read to check and correct
297 // data corruption
298 IOOptions opts;
299 if (PrepareIOFromReadOptions(ro, ioptions.clock, opts) !=
300 IOStatus::OK()) {
301 return s;
302 }
303 opts.verify_and_reconstruct_read = true;
304 std::unique_ptr<char[]> data(new char[len]);
305 Slice result;
306 IOStatus io_s =
307 file->Read(opts, handle.offset(), len, &result, data.get(), nullptr);
308 RecordTick(ioptions.stats, FILE_READ_CORRUPTION_RETRY_COUNT);
309 if (!io_s.ok()) {
310 ROCKS_LOG_INFO(ioptions.info_log,
311 "Reading properties block failed - %s",
312 io_s.ToString().c_str());
313 // Return the original corruption error as that's more serious
314 return s;
315 }
316 if (result.size() < len) {
317 return Status::Corruption("Reading properties block failed - " +
318 std::to_string(result.size()) +
319 " bytes read");
320 }
321 RecordTick(ioptions.stats, FILE_READ_CORRUPTION_RETRY_SUCCESS_COUNT);
322 block_contents = BlockContents(std::move(data), handle.size());
323 }
324
325 uint64_t block_size = block_contents.data.size();
326 Block properties_block(std::move(block_contents));
327 // Unfortunately, Block::size() might not equal block_contents.data.size(),
328 // and Block hides block_contents
329 std::unique_ptr<MetaBlockIter> iter(properties_block.NewMetaIterator());
330
331 std::unique_ptr<TableProperties> new_table_properties{new TableProperties};
332 // All pre-defined properties of type uint64_t
333 std::unordered_map<std::string, uint64_t*> predefined_uint64_properties = {
334 {TablePropertiesNames::kOriginalFileNumber,
335 &new_table_properties->orig_file_number},
336 {TablePropertiesNames::kDataSize, &new_table_properties->data_size},
337 {TablePropertiesNames::kIndexSize, &new_table_properties->index_size},
338 {TablePropertiesNames::kIndexPartitions,
339 &new_table_properties->index_partitions},
340 {TablePropertiesNames::kTopLevelIndexSize,
341 &new_table_properties->top_level_index_size},
342 {TablePropertiesNames::kIndexKeyIsUserKey,
343 &new_table_properties->index_key_is_user_key},
344 {TablePropertiesNames::kIndexValueIsDeltaEncoded,
345 &new_table_properties->index_value_is_delta_encoded},
346 {TablePropertiesNames::kFilterSize, &new_table_properties->filter_size},
347 {TablePropertiesNames::kRawKeySize,
348 &new_table_properties->raw_key_size},
349 {TablePropertiesNames::kRawValueSize,
350 &new_table_properties->raw_value_size},
351 {TablePropertiesNames::kNumDataBlocks,
352 &new_table_properties->num_data_blocks},
353 {TablePropertiesNames::kNumEntries, &new_table_properties->num_entries},
354 {TablePropertiesNames::kNumFilterEntries,
355 &new_table_properties->num_filter_entries},
356 {TablePropertiesNames::kDeletedKeys,
357 &new_table_properties->num_deletions},
358 {TablePropertiesNames::kMergeOperands,
359 &new_table_properties->num_merge_operands},
360 {TablePropertiesNames::kNumRangeDeletions,
361 &new_table_properties->num_range_deletions},
362 {TablePropertiesNames::kFormatVersion,
363 &new_table_properties->format_version},
364 {TablePropertiesNames::kFixedKeyLen,
365 &new_table_properties->fixed_key_len},
366 {TablePropertiesNames::kColumnFamilyId,
367 &new_table_properties->column_family_id},
368 {TablePropertiesNames::kCreationTime,
369 &new_table_properties->creation_time},
370 {TablePropertiesNames::kOldestKeyTime,
371 &new_table_properties->oldest_key_time},
372 {TablePropertiesNames::kNewestKeyTime,
373 &new_table_properties->newest_key_time},
374 {TablePropertiesNames::kFileCreationTime,
375 &new_table_properties->file_creation_time},
376 {TablePropertiesNames::kSlowCompressionEstimatedDataSize,
377 &new_table_properties->slow_compression_estimated_data_size},
378 {TablePropertiesNames::kFastCompressionEstimatedDataSize,
379 &new_table_properties->fast_compression_estimated_data_size},
380 {TablePropertiesNames::kTailStartOffset,
381 &new_table_properties->tail_start_offset},
382 {TablePropertiesNames::kUserDefinedTimestampsPersisted,
383 &new_table_properties->user_defined_timestamps_persisted},
384 {TablePropertiesNames::kKeyLargestSeqno,
385 &new_table_properties->key_largest_seqno},
386 };
387
388 std::string last_key;
389 for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
390 s = iter->status();
391 if (!s.ok()) {
392 break;
393 }
394
395 auto key = iter->key().ToString();
396 // properties block should be strictly sorted with no duplicate key.
397 if (!last_key.empty() &&
398 BytewiseComparator()->Compare(key, last_key) <= 0) {
399 s = Status::Corruption("properties unsorted");
400 break;
401 }
402 last_key = key;
403
404 auto raw_val = iter->value();
405 auto pos = predefined_uint64_properties.find(key);
406
407 if (key == ExternalSstFilePropertyNames::kGlobalSeqno) {
408 new_table_properties->external_sst_file_global_seqno_offset =
409 handle.offset() + iter->ValueOffset();
410 }
411
412 if (pos != predefined_uint64_properties.end()) {
413 if (key == TablePropertiesNames::kDeletedKeys ||
414 key == TablePropertiesNames::kMergeOperands) {
415 // Insert in user-collected properties for API backwards compatibility
416 new_table_properties->user_collected_properties.insert(
417 {key, raw_val.ToString()});
418 }
419 // handle predefined rocksdb properties
420 uint64_t val;
421 if (!GetVarint64(&raw_val, &val)) {
422 // skip malformed value
423 auto error_msg =
424 "Detect malformed value in properties meta-block:"
425 "\tkey: " +
426 key + "\tval: " + raw_val.ToString();
427 ROCKS_LOG_ERROR(ioptions.logger, "%s", error_msg.c_str());
428 continue;
429 }
430 *(pos->second) = val;
431 } else if (key == TablePropertiesNames::kDbId) {
432 new_table_properties->db_id = raw_val.ToString();
433 } else if (key == TablePropertiesNames::kDbSessionId) {
434 new_table_properties->db_session_id = raw_val.ToString();
435 } else if (key == TablePropertiesNames::kDbHostId) {
436 new_table_properties->db_host_id = raw_val.ToString();
437 } else if (key == TablePropertiesNames::kFilterPolicy) {
438 new_table_properties->filter_policy_name = raw_val.ToString();
439 } else if (key == TablePropertiesNames::kColumnFamilyName) {
440 new_table_properties->column_family_name = raw_val.ToString();
441 } else if (key == TablePropertiesNames::kComparator) {
442 new_table_properties->comparator_name = raw_val.ToString();
443 } else if (key == TablePropertiesNames::kMergeOperator) {
444 new_table_properties->merge_operator_name = raw_val.ToString();
445 } else if (key == TablePropertiesNames::kPrefixExtractorName) {
446 new_table_properties->prefix_extractor_name = raw_val.ToString();
447 } else if (key == TablePropertiesNames::kPropertyCollectors) {
448 new_table_properties->property_collectors_names = raw_val.ToString();
449 } else if (key == TablePropertiesNames::kCompression) {
450 new_table_properties->compression_name = raw_val.ToString();
451 } else if (key == TablePropertiesNames::kCompressionOptions) {
452 new_table_properties->compression_options = raw_val.ToString();
453 } else if (key == TablePropertiesNames::kSequenceNumberTimeMapping) {
454 new_table_properties->seqno_to_time_mapping = raw_val.ToString();
455 } else {
456 // handle user-collected properties
457 new_table_properties->user_collected_properties.insert(
458 {key, raw_val.ToString()});
459 }
460 }
461
462 // Modified version of BlockFetcher checksum verification
463 // (See write_global_seqno comment above)
464 if (s.ok() && footer.GetBlockTrailerSize() > 0) {
465 s = VerifyBlockChecksum(footer, properties_block.data(), block_size,
466 file->file_name(), handle.offset());
467 if (s.IsCorruption()) {
468 if (new_table_properties->external_sst_file_global_seqno_offset != 0) {
469 std::string tmp_buf(properties_block.data(), len);
470 uint64_t global_seqno_offset =
471 new_table_properties->external_sst_file_global_seqno_offset -
472 handle.offset();
473 EncodeFixed64(&tmp_buf[static_cast<size_t>(global_seqno_offset)], 0);
474 s = VerifyBlockChecksum(footer, tmp_buf.data(), block_size,
475 file->file_name(), handle.offset());
476 }
477 }
478 }
479
480 // If we detected a corruption and the file system supports verification
481 // and reconstruction, retry the read
482 if (s.IsCorruption() && !retry &&
483 CheckFSFeatureSupport(ioptions.fs.get(),
484 FSSupportedOps::kVerifyAndReconstructRead)) {
485 retry = true;
486 } else {
487 if (s.ok()) {
488 *table_properties = std::move(new_table_properties);
489 }
490 break;
491 }
492 }
493
494 return s;
495}

497Status ReadTableProperties(RandomAccessFileReader* file, uint64_t file_size,
498 uint64_t table_magic_number,
499 const ImmutableOptions& ioptions,
500 const ReadOptions& read_options,
501 std::unique_ptr<TableProperties>* properties,

518 } else {
519 s = Status::NotFound();
520 }
521 return s;
522}

524Status FindOptionalMetaBlock(InternalIterator* meta_index_iter,
525 const std::string& meta_block_name,
526 BlockHandle* block_handle) {
527 assert(block_handle != nullptr);
528 meta_index_iter->Seek(meta_block_name);
529 if (meta_index_iter->status().ok()) {
530 if (meta_index_iter->Valid() && meta_index_iter->key() == meta_block_name) {
531 Slice v = meta_index_iter->value();
532 return block_handle->DecodeFrom(&v);
533 } else if (meta_block_name == kPropertiesBlockName) {
534 // Have to try old name for compatibility
535 meta_index_iter->Seek(kPropertiesBlockOldName);
536 if (meta_index_iter->status().ok() && meta_index_iter->Valid() &&
537 meta_index_iter->key() == kPropertiesBlockOldName) {
538 Slice v = meta_index_iter->value();
539 return block_handle->DecodeFrom(&v);
540 }
541 }
542 }
543 // else
544 *block_handle = BlockHandle::NullBlockHandle();
545 return meta_index_iter->status();
546}

561 uint64_t file_size, uint64_t table_magic_number,

593Status FindMetaBlockInFile(
594 RandomAccessFileReader* file, uint64_t file_size,
595 uint64_t table_magic_number, const ImmutableOptions& ioptions,
596 const ReadOptions& read_options, const std::string& meta_block_name,
597 BlockHandle* block_handle, MemoryAllocator* memory_allocator,
598 FilePrefetchBuffer* prefetch_buffer, Footer* footer_out) {
599 BlockContents metaindex_contents;
600 auto s = ReadMetaIndexBlockInFile(
601 file, file_size, table_magic_number, ioptions, read_options,
602 &metaindex_contents, memory_allocator, prefetch_buffer, footer_out);
603 if (!s.ok()) {
604 return s;
605 }
606 // meta blocks are never compressed. Need to add uncompress logic if we are to
607 // compress it.
608 Block metaindex_block(std::move(metaindex_contents));
609
610 std::unique_ptr<InternalIterator> meta_iter;
611 meta_iter.reset(metaindex_block.NewMetaIterator());
