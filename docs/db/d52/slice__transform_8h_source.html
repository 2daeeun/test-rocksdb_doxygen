28// A SliceTransform is a generic pluggable way of transforming one string
29// to another. Its primary use-case is in configuring RocksDB prefix Bloom
30// filters, by setting prefix_extractor in ColumnFamilyOptions.
31//
32// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
33// because RocksDB is not exception-safe. This could cause undefined behavior
34// including data loss, unreported corruption, deadlocks, and more.
35class SliceTransform : public Customizable {
36 public:
37 virtual ~SliceTransform(){}
38
39 // Return the name of this transformation.
40 const char* Name() const override = 0;
41 static const char* Type() { return "SliceTransform"; }
42
43 // Creates and configures a new SliceTransform from the input options and id.
44 static Status CreateFromString(const ConfigOptions& config_options,
45 const std::string& id,
46 std::shared_ptr<const SliceTransform>* result);
47
48 // Returns a string representation of this SliceTransform, representing the ID
49 // and any additional properties.
50 std::string AsString() const;
51
52 // Extract a prefix from a specified key, partial key, iterator upper bound,
53 // etc. This is normally used for building and checking prefix Bloom filters
54 // but should accept any string for which InDomain() returns true.
55 // See ColumnFamilyOptions::prefix_extractor for specific properties that
56 // must be satisfied by prefix extractors.
57 virtual Slice Transform(const Slice& key) const = 0;
58
59 // Determine whether the specified key is compatible with the logic
60 // specified in the Transform method. Keys for which InDomain returns
61 // false will not be added to or queried against prefix Bloom filters.
62 //
63 // For example, if the Transform method returns a fixed length
64 // prefix of size 4, then an invocation to InDomain("abc") returns
65 // false because the specified key length(3) is shorter than the
66 // prefix size of 4.
67 //
68 // Wiki documentation here:
69 // https://github.com/facebook/rocksdb/wiki/Prefix-Seek
70 //
71 virtual bool InDomain(const Slice& key) const = 0;
72
73 // DEPRECATED: This is currently not used and remains here for backward
74 // compatibility.
75 virtual bool InRange(const Slice& /*dst*/) const { return false; }
76
77 // Returns information on maximum prefix length, if there is one.
78 // If Transform(x).size() == n for some keys and otherwise < n,
79 // should return true and set *len = n. Returning false is safe but
80 // currently disables some auto_prefix_mode filtering.
81 // Specifically, if the iterate_upper_bound is the immediate successor (see
82 // Comparator::IsSameLengthImmediateSuccessor) of the seek key's prefix,
83 // we require this function return true and iterate_upper_bound.size() == n
84 // to recognize and optimize the prefix seek.
85 // Otherwise (including FullLengthEnabled returns false, or prefix length is
86 // less than maximum), Seek with auto_prefix_mode is only optimized if the
87 // iterate_upper_bound and seek key have the same prefix.
88 // BUG: Despite all these conditions and even with the extra condition on
89 // IsSameLengthImmediateSuccessor (see it's "BUG" section), it is not
90 // sufficient to ensure auto_prefix_mode returns all entries that
91 // total_order_seek would return. See auto_prefix_mode "BUG" section.
92 virtual bool FullLengthEnabled(size_t* /*len*/) const { return false; }
93
94 // Transform(s)=Transform(`prefix`) for any s with `prefix` as a prefix.
95 //
96 // This function is not used by RocksDB, but for users. If users pass
97 // Options by string to RocksDB, they might not know what prefix extractor
98 // they are using. This function is to help users can determine:
99 // if they want to iterate all keys prefixing `prefix`, whether it is
100 // safe to use prefix bloom filter and seek to key `prefix`.
101 // If this function returns true, this means a user can Seek() to a prefix
102 // using the bloom filter. Otherwise, user needs to skip the bloom filter
103 // by setting ReadOptions.total_order_seek = true.
104 //
105 // Here is an example: Suppose we implement a slice transform that returns
106 // the first part of the string up to and including first ",":
107 // 1. SameResultWhenAppended("abc,") should return true. If applying prefix
108 // bloom filter using it, all slices matching "abc,.*" will be extracted
109 // to "abc,", so any SST file or memtable containing any of those key
110 // will not be filtered out.
111 // 2. SameResultWhenAppended("abc") should return false. A user will not be
112 // guaranteed to see all the keys matching "abc.*" if a user prefix
113 // seeks to "abc" against a DB with the same setting. If one SST file
114 // only contains "abcd,e", the file can be filtered out and the key will
115 // be invisible, because the prefix according to the configured extractor
116 // is "abcd,".
117 //
118 // i.e., an implementation always returning false is safe.
119 virtual bool SameResultWhenAppended(const Slice& /*prefix*/) const {
120 return false;
121 }
122};
123
124// The prefix is the first `prefix_len` bytes of the key, and keys shorter
125// then `prefix_len` are not InDomain.
126const SliceTransform* NewFixedPrefixTransform(size_t prefix_len);
127
128// The prefix is the first min(length(key),`cap_len`) bytes of the key, and
129// all keys are InDomain.
130const SliceTransform* NewCappedPrefixTransform(size_t cap_len);
131
132// Prefix is equal to key. All keys are InDomain.
133const SliceTransform* NewNoopTransform();
