138BlockBasedTable::~BlockBasedTable() {
139 auto ua = rep_->uncache_aggressiveness.LoadRelaxed();
140 // NOTE: there is an undiagnosed incompatibility with mmap reads,
141 // where attempting to read the index below can result in bus error.
142 // In theory the mmap should remain in place until destruction of
143 // rep_, so even a page fault should be satisfiable. But also, combining
144 // mmap reads with block cache is weird, so it's not a concerning loss.
145 if (ua > 0 && rep_->table_options.block_cache &&
146 !rep_->ioptions.allow_mmap_reads) {
147 if (rep_->filter) {
148 rep_->filter->EraseFromCacheBeforeDestruction(ua);
149 }
150 if (rep_->index_reader) {
151 {
152 // TODO: Also uncache data blocks known after any gaps in partitioned
153 // index. Right now the iterator errors out as soon as there's an
154 // index partition not in cache.
155 IndexBlockIter iiter_on_stack;
156 ReadOptions ropts;
157 ropts.read_tier = kBlockCacheTier; // No I/O
158 auto iiter = NewIndexIterator(
159 ropts, /*disable_prefix_seek=*/false, &iiter_on_stack,
160 /*get_context=*/nullptr, /*lookup_context=*/nullptr);
161 std::unique_ptr<InternalIteratorBase<IndexValue>> iiter_unique_ptr;
162 if (iiter != &iiter_on_stack) {
163 iiter_unique_ptr.reset(iiter);
164 }
165 // Un-cache the data blocks the index iterator with tell us about
166 // without I/O. (NOTE: It's extremely unlikely that a data block
167 // will be in block cache without the index block pointing to it
168 // also in block cache.)
169 UncacheAggressivenessAdvisor advisor(ua);
170 for (iiter->SeekToFirst(); iiter->Valid() && advisor.ShouldContinue();
171 iiter->Next()) {
172 bool erased = EraseFromCache(iiter->value().handle);
173 advisor.Report(erased);
174 }
175 iiter->status().PermitUncheckedError();
176 }
177
178 // Un-cache the index block(s)
179 rep_->index_reader->EraseFromCacheBeforeDestruction(ua);
180 }
181 }
182 delete rep_;
183}

186// Read the block identified by "handle" from "file".
187// The only relevant option is options.verify_checksums for now.
188// On failure return non-OK.
189// On success fill *result and return OK - caller owns *result
190// @param uncompression_dict Data for presetting the compression library's
191// dictionary.
192template <typename TBlocklike>
193Status ReadAndParseBlockFromFile(
194 RandomAccessFileReader* file, FilePrefetchBuffer* prefetch_buffer,
195 const Footer& footer, const ReadOptions& options, const BlockHandle& handle,
196 std::unique_ptr<TBlocklike>* result, const ImmutableOptions& ioptions,
197 BlockCreateContext& create_context, bool maybe_compressed,
198 const UncompressionDict& uncompression_dict,
199 const PersistentCacheOptions& cache_options,
200 MemoryAllocator* memory_allocator, bool for_compaction, bool async_read) {
201 assert(result);
202
203 BlockContents contents;
204 BlockFetcher block_fetcher(
205 file, prefetch_buffer, footer, options, handle, &contents, ioptions,
206 /*do_uncompress*/ maybe_compressed, maybe_compressed,
207 TBlocklike::kBlockType, uncompression_dict, cache_options,
208 memory_allocator, nullptr, for_compaction);
209 Status s;
210 // If prefetch_buffer is not allocated, it will fallback to synchronous
211 // reading of block contents.
212 if (async_read && prefetch_buffer != nullptr) {
213 s = block_fetcher.ReadAsyncBlockContents();
214 if (!s.ok()) {
215 return s;
216 }
217 } else {
218 s = block_fetcher.ReadBlockContents();
219 }
220 if (s.ok()) {
221 create_context.Create(result, std::move(contents));
222 }
223 return s;
224}

241 return true;
242 } else {
243 return false;

259void BlockBasedTable::UpdateCacheHitMetrics(BlockType block_type,
260 GetContext* get_context,
261 size_t usage) const {
262 Statistics* const statistics = rep_->ioptions.stats;
263
264 PERF_COUNTER_ADD(block_cache_hit_count, 1);
265 PERF_COUNTER_ADD(block_cache_read_byte, usage);
266 PERF_COUNTER_BY_LEVEL_ADD(block_cache_hit_count, 1,
267 static_cast<uint32_t>(rep_->level));
268
269 if (get_context) {
270 ++get_context->get_context_stats_.num_cache_hit;
271 get_context->get_context_stats_.num_cache_bytes_read += usage;
272 } else {
273 RecordTick(statistics, BLOCK_CACHE_HIT);
274 RecordTick(statistics, BLOCK_CACHE_BYTES_READ, usage);
275 }
276
277 switch (block_type) {
278 case BlockType::kFilter:
279 case BlockType::kFilterPartitionIndex:
280 PERF_COUNTER_ADD(block_cache_filter_hit_count, 1);
281 PERF_COUNTER_ADD(block_cache_filter_read_byte, usage);
282
283 if (get_context) {
284 ++get_context->get_context_stats_.num_cache_filter_hit;
285 } else {
286 RecordTick(statistics, BLOCK_CACHE_FILTER_HIT);
287 }
288 break;
289
290 case BlockType::kCompressionDictionary:
291 // TODO: introduce perf counter for compression dictionary hit count
292 PERF_COUNTER_ADD(block_cache_compression_dict_read_byte, usage);
293 if (get_context) {
294 ++get_context->get_context_stats_.num_cache_compression_dict_hit;
295 } else {
296 RecordTick(statistics, BLOCK_CACHE_COMPRESSION_DICT_HIT);
297 }
298 break;
299
300 case BlockType::kIndex:
301 PERF_COUNTER_ADD(block_cache_index_hit_count, 1);
302 PERF_COUNTER_ADD(block_cache_index_read_byte, usage);
303
304 if (get_context) {
305 ++get_context->get_context_stats_.num_cache_index_hit;
306 } else {
307 RecordTick(statistics, BLOCK_CACHE_INDEX_HIT);
308 }
309 break;
310
311 default:
312 // TODO: introduce dedicated tickers/statistics/counters
313 // for range tombstones
314 if (get_context) {
315 ++get_context->get_context_stats_.num_cache_data_hit;
316 } else {
317 RecordTick(statistics, BLOCK_CACHE_DATA_HIT);
318 }
319 break;
320 }

468// Return True if table_properties has `user_prop_name` has a `true` value
469// or it doesn't contain this property (for backward compatible).
470bool IsFeatureSupported(const TableProperties& table_properties,
471 const std::string& user_prop_name, Logger* info_log) {
472 auto& props = table_properties.user_collected_properties;
473 auto pos = props.find(user_prop_name);
474 // Older version doesn't have this value set. Skip this check.
475 if (pos != props.end()) {
476 if (pos->second == kPropFalse) {
477 return false;
478 } else if (pos->second != kPropTrue) {
479 ROCKS_LOG_WARN(info_log, "Property %s has invalidate value %s",
480 user_prop_name.c_str(), pos->second.c_str());
481 }
482 }
483 return true;

486// Caller has to ensure seqno is not nullptr.
487// Set *seqno to the global sequence number for reading this file.
488Status GetGlobalSequenceNumber(const TableProperties& table_properties,
489 SequenceNumber largest_seqno,
490 SequenceNumber* seqno) {
491 const auto& props = table_properties.user_collected_properties;
492 const auto version_pos = props.find(ExternalSstFilePropertyNames::kVersion);
493 const auto seqno_pos = props.find(ExternalSstFilePropertyNames::kGlobalSeqno);
494
495 *seqno = kDisableGlobalSequenceNumber;
496 if (version_pos == props.end()) {
497 if (seqno_pos != props.end()) {
498 std::array<char, 200> msg_buf;
499 // This is not an external sst file, global_seqno is not supported.
500 snprintf(
501 msg_buf.data(), msg_buf.max_size(),
502 "A non-external sst file have global seqno property with value %s",
503 seqno_pos->second.c_str());
504 return Status::Corruption(msg_buf.data());
505 }
506 return Status::OK();
507 }
508
509 uint32_t version = DecodeFixed32(version_pos->second.c_str());
510 if (version != 2) {
511 std::array<char, 200> msg_buf;
512 if (version != 1) {
513 snprintf(msg_buf.data(), msg_buf.max_size(),
514 "An external sst file has corrupted version %u.", version);
515 return Status::Corruption(msg_buf.data());
516 }
517 if (seqno_pos != props.end()) {
518 // This is a v1 external sst file, global_seqno is not supported.
519 snprintf(msg_buf.data(), msg_buf.max_size(),
520 "An external sst file with version %u has global seqno "
521 "property with value %s",
522 version, seqno_pos->second.c_str());
523 return Status::Corruption(msg_buf.data());
524 }
525 return Status::OK();
526 }
527
528 // Since we have a plan to deprecate global_seqno, we do not return failure
529 // if seqno_pos == props.end(). We rely on version_pos to detect whether the
530 // SST is external.
531 SequenceNumber global_seqno(0);
532 if (seqno_pos != props.end()) {
533 global_seqno = DecodeFixed64(seqno_pos->second.c_str());
534 }
535 // SstTableReader open table reader with kMaxSequenceNumber as largest_seqno
536 // to denote it is unknown.
537 if (largest_seqno < kMaxSequenceNumber) {
538 if (global_seqno == 0) {
539 global_seqno = largest_seqno;
540 }
541 if (global_seqno != largest_seqno) {
542 std::array<char, 200> msg_buf;
543 snprintf(
544 msg_buf.data(), msg_buf.max_size(),
545 "An external sst file with version %u have global seqno property "
546 "with value %s, while largest seqno in the file is %llu",
547 version, seqno_pos->second.c_str(),
548 static_cast<unsigned long long>(largest_seqno));
549 return Status::Corruption(msg_buf.data());
550 }
551 }
552 *seqno = global_seqno;
553
554 if (global_seqno > kMaxSequenceNumber) {
555 std::array<char, 200> msg_buf;
556 snprintf(msg_buf.data(), msg_buf.max_size(),
557 "An external sst file with version %u have global seqno property "
558 "with value %llu, which is greater than kMaxSequenceNumber",
559 version, static_cast<unsigned long long>(global_seqno));
560 return Status::Corruption(msg_buf.data());
561 }
562
563 return Status::OK();
564}

567void BlockBasedTable::SetupBaseCacheKey(const TableProperties* properties,
568 const std::string& cur_db_session_id,
569 uint64_t cur_file_number,
570 OffsetableCacheKey* out_base_cache_key,
571 bool* out_is_stable) {
572 // Use a stable cache key if sufficient data is in table properties
573 std::string db_session_id;
574 uint64_t file_num;
575 std::string db_id;
576 if (properties && !properties->db_session_id.empty() &&
577 properties->orig_file_number > 0) {
578 // (Newer SST file case)
579 // We must have both properties to get a stable unique id because
580 // CreateColumnFamilyWithImport or IngestExternalFiles can change the
581 // file numbers on a file.
582 db_session_id = properties->db_session_id;
583 file_num = properties->orig_file_number;
584 // Less critical, populated in earlier release than above
585 db_id = properties->db_id;
586 if (out_is_stable) {
587 *out_is_stable = true;
588 }
589 } else {
590 // (Old SST file case)
591 // We use (unique) cache keys based on current identifiers. These are at
592 // least stable across table file close and re-open, but not across
593 // different DBs nor DB close and re-open.
594 db_session_id = cur_db_session_id;
595 file_num = cur_file_number;
596 // Plumbing through the DB ID to here would be annoying, and of limited
597 // value because of the case of VersionSet::Recover opening some table
598 // files and later setting the DB ID. So we just rely on uniqueness
599 // level provided by session ID.
600 db_id = "unknown";
601 if (out_is_stable) {
602 *out_is_stable = false;
603 }
604 }
605
606 // Too many tests to update to get these working
607 // assert(file_num > 0);
608 // assert(!db_session_id.empty());
609 // assert(!db_id.empty());
610
611 // Minimum block size is 5 bytes; therefore we can trim off two lower bits
612 // from offsets. See GetCacheKey.
613 *out_base_cache_key = OffsetableCacheKey(db_id, db_session_id, file_num);
614}

616CacheKey BlockBasedTable::GetCacheKey(const OffsetableCacheKey& base_cache_key,
617 const BlockHandle& handle) {
618 // Minimum block size is 5 bytes; therefore we can trim off two lower bits
619 // from offet.
620 return base_cache_key.WithOffset(handle.offset() >> 2);

626 const InternalKeyComparator& internal_comparator,
627 std::unique_ptr<RandomAccessFileReader>&& file, uint64_t file_size,
628 uint8_t block_protection_bytes_per_key,
629 std::unique_ptr<TableReader>* table_reader, uint64_t tail_size,
630 std::shared_ptr<CacheReservationManager> table_reader_cache_res_mgr,
631 const std::shared_ptr<const SliceTransform>& prefix_extractor,
632 const bool prefetch_index_and_filter_in_cache, const bool skip_filters,
633 const int level, const bool immortal_table,
634 const SequenceNumber largest_seqno, const bool force_direct_prefetch,
635 TailPrefetchStats* tail_prefetch_stats,
636 BlockCacheTracer* const block_cache_tracer,
637 size_t max_file_size_for_l0_meta_pin, const std::string& cur_db_session_id,
638 uint64_t cur_file_num, UniqueId64x2 expected_unique_id,
639 const bool user_defined_timestamps_persisted) {
640 table_reader->reset();
641
642 Status s;
643 Footer footer;
644 std::unique_ptr<FilePrefetchBuffer> prefetch_buffer;
645
646 // From read_options, retain deadline, io_timeout, rate_limiter_priority, and
647 // verify_checksums. In future, we may retain more options.
648 // TODO: audit more ReadOptions and do this in a way that brings attention
649 // on new ReadOptions?
650 ReadOptions ro;
651 ro.deadline = read_options.deadline;
652 ro.io_timeout = read_options.io_timeout;
653 ro.rate_limiter_priority = read_options.rate_limiter_priority;
654 ro.verify_checksums = read_options.verify_checksums;
655 ro.io_activity = read_options.io_activity;
656 ro.fill_cache = read_options.fill_cache;
657
658 // prefetch both index and filters, down to all partitions
659 const bool prefetch_all = prefetch_index_and_filter_in_cache || level == 0;
660 const bool preload_all = !table_options.cache_index_and_filter_blocks;
661
662 if (!ioptions.allow_mmap_reads && !env_options.use_mmap_reads) {
663 s = PrefetchTail(ro, ioptions, file.get(), file_size, force_direct_prefetch,
664 tail_prefetch_stats, prefetch_all, preload_all,
665 &prefetch_buffer, ioptions.stats, tail_size,
666 ioptions.logger);
667 // Return error in prefetch path to users.
668 if (!s.ok()) {
669 return s;
670 }
671 } else {
672 // Should not prefetch for mmap mode.
673 prefetch_buffer.reset(new FilePrefetchBuffer(
674 ReadaheadParams(), false /* enable */, true /* track_min_offset */));
675 }
676
677 // Read in the following order:
678 // 1. Footer
679 // 2. [metaindex block]
680 // 3. [meta block: properties]
681 // 4. [meta block: range deletion tombstone]
682 // 5. [meta block: compression dictionary]
683 // 6. [meta block: index]
684 // 7. [meta block: filter]
685 IOOptions opts;
686 s = file->PrepareIOOptions(ro, opts);
687 if (s.ok()) {
688 s = ReadFooterFromFile(opts, file.get(), *ioptions.fs,
689 prefetch_buffer.get(), file_size, &footer,
690 kBlockBasedTableMagicNumber, ioptions.stats);
691 }
692 if (!s.ok()) {
693 if (s.IsCorruption()) {
694 RecordTick(ioptions.statistics.get(), SST_FOOTER_CORRUPTION_COUNT);
695 }
696 return s;
697 }
698 if (!IsSupportedFormatVersion(footer.format_version())) {
699 return Status::Corruption(
700 "Unknown Footer version. Maybe this file was created with newer "
701 "version of RocksDB?");
702 }
703
704 BlockCacheLookupContext lookup_context{TableReaderCaller::kPrefetch};
705 Rep* rep = new BlockBasedTable::Rep(
706 ioptions, env_options, table_options, internal_comparator, skip_filters,
707 file_size, level, immortal_table, user_defined_timestamps_persisted);
708 rep->file = std::move(file);
709 rep->footer = footer;
710
711 // For fully portable/stable cache keys, we need to read the properties
712 // block before setting up cache keys. TODO: consider setting up a bootstrap
713 // cache key for PersistentCache to use for metaindex and properties blocks.
714 rep->persistent_cache_options = PersistentCacheOptions();
715
716 // Meta-blocks are not dictionary compressed. Explicitly set the dictionary
717 // handle to null, otherwise it may be seen as uninitialized during the below
718 // meta-block reads.
719 rep->compression_dict_handle = BlockHandle::NullBlockHandle();
720
721 rep->create_context.protection_bytes_per_key = block_protection_bytes_per_key;
722 // Read metaindex
723 std::unique_ptr<BlockBasedTable> new_table(
724 new BlockBasedTable(rep, block_cache_tracer));
725 std::unique_ptr<Block> metaindex;
726 std::unique_ptr<InternalIterator> metaindex_iter;
727 s = new_table->ReadMetaIndexBlock(ro, prefetch_buffer.get(), &metaindex,
728 &metaindex_iter);
729 if (!s.ok()) {
730 return s;
731 }
732
733 // Populates table_properties and some fields that depend on it,
734 // such as index_type.
735 s = new_table->ReadPropertiesBlock(ro, prefetch_buffer.get(),
736 metaindex_iter.get(), largest_seqno);
737 if (!s.ok()) {
738 return s;
739 }
740
741 // Populate BlockCreateContext
742 bool blocks_definitely_zstd_compressed =
743 rep->table_properties &&
744 (rep->table_properties->compression_name ==
745 CompressionTypeToString(kZSTD) ||
746 rep->table_properties->compression_name ==
747 CompressionTypeToString(kZSTDNotFinalCompression));
748 rep->create_context = BlockCreateContext(
749 &rep->table_options, &rep->ioptions, rep->ioptions.stats,
750 blocks_definitely_zstd_compressed, block_protection_bytes_per_key,
751 rep->internal_comparator.user_comparator(), rep->index_value_is_full,
752 rep->index_has_first_key);
753
754 // Check expected unique id if provided
755 if (expected_unique_id != kNullUniqueId64x2) {
756 auto props = rep->table_properties;
757 if (!props) {
758 return Status::Corruption("Missing table properties on file " +
759 std::to_string(cur_file_num) +
760 " with known unique ID");
761 }
762 UniqueId64x2 actual_unique_id{};
763 s = GetSstInternalUniqueId(props->db_id, props->db_session_id,
764 props->orig_file_number, &actual_unique_id,
765 /*force*/ true);
766 assert(s.ok()); // because force=true
767 if (expected_unique_id != actual_unique_id) {
768 return Status::Corruption(
769 "Mismatch in unique ID on table file " +
770 std::to_string(cur_file_num) +
771 ". Expected: " + InternalUniqueIdToHumanString(&expected_unique_id) +
772 " Actual: " + InternalUniqueIdToHumanString(&actual_unique_id));
773 }
774 TEST_SYNC_POINT_CALLBACK("BlockBasedTable::Open::PassedVerifyUniqueId",
775 &actual_unique_id);
776 } else {
777 TEST_SYNC_POINT_CALLBACK("BlockBasedTable::Open::SkippedVerifyUniqueId",
778 nullptr);
779 if (ioptions.verify_sst_unique_id_in_manifest && ioptions.logger) {
780 // A crude but isolated way of reporting unverified files. This should not
781 // be an ongoing concern so doesn't deserve a place in Statistics IMHO.
782 static std::atomic<uint64_t> unverified_count{0};
783 auto prev_count =
784 unverified_count.fetch_add(1, std::memory_order_relaxed);
785 if (prev_count == 0) {
786 ROCKS_LOG_WARN(
787 ioptions.logger,
788 "At least one SST file opened without unique ID to verify: %" PRIu64
789 ".sst",
790 cur_file_num);
791 } else if (prev_count % 1000 == 0) {
792 ROCKS_LOG_WARN(
793 ioptions.logger,
794 "Another ~1000 SST files opened without unique ID to verify");
795 }
796 }
797 }
798
799 // Set up prefix extracto as needed
800 bool force_null_table_prefix_extractor = false;
801 TEST_SYNC_POINT_CALLBACK(
802 "BlockBasedTable::Open::ForceNullTablePrefixExtractor",
803 &force_null_table_prefix_extractor);
804 if (force_null_table_prefix_extractor) {
805 assert(!rep->table_prefix_extractor);
806 } else if (!PrefixExtractorChangedHelper(rep->table_properties.get(),
807 prefix_extractor.get())) {
808 // Establish fast path for unchanged prefix_extractor
809 rep->table_prefix_extractor = prefix_extractor;
810 } else {
811 // Current prefix_extractor doesn't match table
812 if (rep->table_properties) {
813 //**TODO: If/When the DBOptions has a registry in it, the ConfigOptions
814 // will need to use it
815 ConfigOptions config_options;
816 Status st = SliceTransform::CreateFromString(
817 config_options, rep->table_properties->prefix_extractor_name,
818 &(rep->table_prefix_extractor));
819 if (!st.ok()) {
820 //**TODO: Should this be error be returned or swallowed?
821 ROCKS_LOG_ERROR(rep->ioptions.logger,
822 "Failed to create prefix extractor[%s]: %s",
823 rep->table_properties->prefix_extractor_name.c_str(),
824 st.ToString().c_str());
825 }
826 }
827 }
828
829 // With properties loaded, we can set up portable/stable cache keys
830 SetupBaseCacheKey(rep->table_properties.get(), cur_db_session_id,
831 cur_file_num, &rep->base_cache_key);
832
833 rep->persistent_cache_options =
834 PersistentCacheOptions(rep->table_options.persistent_cache,
835 rep->base_cache_key, rep->ioptions.stats);
836
837 s = new_table->ReadRangeDelBlock(ro, prefetch_buffer.get(),
838 metaindex_iter.get(), internal_comparator,
839 &lookup_context);
840 if (!s.ok()) {
841 return s;
842 }
843 rep->verify_checksum_set_on_open = ro.verify_checksums;
844 s = new_table->PrefetchIndexAndFilterBlocks(
845 ro, prefetch_buffer.get(), metaindex_iter.get(), new_table.get(),
846 prefetch_all, table_options, level, file_size,
847 max_file_size_for_l0_meta_pin, &lookup_context);
848
849 if (s.ok()) {
850 // Update tail prefetch stats
851 assert(prefetch_buffer.get() != nullptr);
852 if (tail_prefetch_stats != nullptr) {
853 assert(prefetch_buffer->min_offset_read() < file_size);
854 tail_prefetch_stats->RecordEffectiveSize(
855 static_cast<size_t>(file_size) - prefetch_buffer->min_offset_read());
856 }
857 }
858
859 if (s.ok() && table_reader_cache_res_mgr) {
860 std::size_t mem_usage = new_table->ApproximateMemoryUsage();
861 s = table_reader_cache_res_mgr->MakeCacheReservation(
862 mem_usage, &(rep->table_reader_cache_res_handle));
863 if (s.IsMemoryLimit()) {
864 s = Status::MemoryLimit(
865 "Can't allocate " +
866 kCacheEntryRoleToCamelString[static_cast<std::uint32_t>(
867 CacheEntryRole::kBlockBasedTableReader)] +
868 " due to memory limit based on "
869 "cache capacity for memory allocation");
870 }
871 }
872
873 if (s.ok()) {
874 *table_reader = std::move(new_table);
875 }
876 return s;
877}

879Status BlockBasedTable::PrefetchTail(
880 const ReadOptions& ro, const ImmutableOptions& ioptions,
881 RandomAccessFileReader* file, uint64_t file_size,
882 bool force_direct_prefetch, TailPrefetchStats* tail_prefetch_stats,
883 const bool prefetch_all, const bool preload_all,
884 std::unique_ptr<FilePrefetchBuffer>* prefetch_buffer, Statistics* stats,
885 uint64_t tail_size, Logger* const logger) {
886 assert(tail_size <= file_size);
887
888 size_t tail_prefetch_size = 0;
889 if (tail_size != 0) {
890 tail_prefetch_size = tail_size;
891 } else {
892 // Fallback for SST files, for which tail size is not recorded in the
893 // manifest. Eventually, this fallback might be removed, so it's
894 // better to make sure that such SST files get compacted.
895 // See https://github.com/facebook/rocksdb/issues/12664
896 if (tail_prefetch_stats != nullptr) {
897 // Multiple threads may get a 0 (no history) when running in parallel,
898 // but it will get cleared after the first of them finishes.
899 tail_prefetch_size = tail_prefetch_stats->GetSuggestedPrefetchSize();
900 }
901 if (tail_prefetch_size == 0) {
902 // Before read footer, readahead backwards to prefetch data. Do more
903 // readahead if we're going to read index/filter.
904 // TODO: This may incorrectly select small readahead in case partitioned
905 // index/filter is enabled and top-level partition pinning is enabled.
906 // That's because we need to issue readahead before we read the
907 // properties, at which point we don't yet know the index type.
908 tail_prefetch_size = prefetch_all || preload_all ? 512 * 1024 : 4 * 1024;
909
910 ROCKS_LOG_WARN(
911 logger,
912 "[%s] Tail prefetch size %zu is calculated based on heuristics.",
913 file->file_name().c_str(), tail_prefetch_size);
914 } else {
915 ROCKS_LOG_WARN(logger,
916 "[%s] Tail prefetch size %zu is calculated based on "
917 "TailPrefetchStats.",
918 file->file_name().c_str(), tail_prefetch_size);
919 }
920 }
921 size_t prefetch_off;
922 size_t prefetch_len;
923 if (file_size < tail_prefetch_size) {
924 prefetch_off = 0;
925 prefetch_len = static_cast<size_t>(file_size);
926 } else {
927 prefetch_off = static_cast<size_t>(file_size - tail_prefetch_size);
928 prefetch_len = tail_prefetch_size;
929 }
930
931#ifndef NDEBUG
932 std::pair<size_t*, size_t*> prefetch_off_len_pair = {&prefetch_off,
933 &prefetch_len};
934 TEST_SYNC_POINT_CALLBACK("BlockBasedTable::Open::TailPrefetchLen",
935 &prefetch_off_len_pair);
936#endif // NDEBUG
937
938 IOOptions opts;
939 Status s = file->PrepareIOOptions(ro, opts);
940 // Try file system prefetch
941 if (s.ok() && !file->use_direct_io() && !force_direct_prefetch) {
942 if (!file->Prefetch(opts, prefetch_off, prefetch_len).IsNotSupported()) {
943 prefetch_buffer->reset(new FilePrefetchBuffer(
944 ReadaheadParams(), false /* enable */, true /* track_min_offset */));
945 return Status::OK();
946 }
947 }
948
949 // Use `FilePrefetchBuffer`
950 prefetch_buffer->reset(new FilePrefetchBuffer(
951 ReadaheadParams(), true /* enable */, true /* track_min_offset */,
952 ioptions.fs.get() /* fs */, nullptr /* clock */, stats,
953 /* readahead_cb */ nullptr,
954 FilePrefetchBufferUsage::kTableOpenPrefetchTail));
955
956 if (s.ok()) {
957 s = (*prefetch_buffer)->Prefetch(opts, file, prefetch_off, prefetch_len);
958 }
959 return s;
960}

962Status BlockBasedTable::ReadPropertiesBlock(
963 const ReadOptions& ro, FilePrefetchBuffer* prefetch_buffer,
964 InternalIterator* meta_iter, const SequenceNumber largest_seqno) {
965 Status s;
966 BlockHandle handle;
967 s = FindOptionalMetaBlock(meta_iter, kPropertiesBlockName, &handle);
968
969 if (!s.ok()) {
970 ROCKS_LOG_WARN(rep_->ioptions.logger,
971 "Error when seeking to properties block from file: %s",
972 s.ToString().c_str());
973 } else if (!handle.IsNull()) {
974 s = meta_iter->status();
975 std::unique_ptr<TableProperties> table_properties;
976 if (s.ok()) {
977 s = ReadTablePropertiesHelper(
978 ro, handle, rep_->file.get(), prefetch_buffer, rep_->footer,
979 rep_->ioptions, &table_properties, nullptr /* memory_allocator */);
980 }
981 IGNORE_STATUS_IF_ERROR(s);
982
983 if (!s.ok()) {
984 ROCKS_LOG_WARN(rep_->ioptions.logger,
985 "Encountered error while reading data from properties "
986 "block %s",
987 s.ToString().c_str());
988 } else {
989 assert(table_properties != nullptr);
990 rep_->table_properties = std::move(table_properties);
991
992 if (s.ok()) {
993 s = rep_->seqno_to_time_mapping.DecodeFrom(
994 rep_->table_properties->seqno_to_time_mapping);
995 }
996 if (!s.ok()) {
997 ROCKS_LOG_WARN(
998 rep_->ioptions.logger,
999 "Problem reading or processing seqno-to-time mapping: %s",
1000 s.ToString().c_str());
1001 }
1002 rep_->blocks_maybe_compressed =
1003 rep_->table_properties->compression_name !=
1004 CompressionTypeToString(kNoCompression);
1005 }
1006 } else {
1007 ROCKS_LOG_ERROR(rep_->ioptions.logger,
1008 "Cannot find Properties block from file.");
1009 }
1010
1011 // Read the table properties, if provided.
1012 if (rep_->table_properties) {
1013 rep_->whole_key_filtering &=
1014 IsFeatureSupported(*(rep_->table_properties),
1015 BlockBasedTablePropertyNames::kWholeKeyFiltering,
1016 rep_->ioptions.logger);
1017 rep_->prefix_filtering &= IsFeatureSupported(
1018 *(rep_->table_properties),
1019 BlockBasedTablePropertyNames::kPrefixFiltering, rep_->ioptions.logger);
1020
1021 rep_->index_key_includes_seq =
1022 rep_->table_properties->index_key_is_user_key == 0;
1023 rep_->index_value_is_full =
1024 rep_->table_properties->index_value_is_delta_encoded == 0;
1025
1026 // Update index_type with the true type.
1027 // If table properties don't contain index type, we assume that the table
1028 // is in very old format and has kBinarySearch index type.
1029 auto& props = rep_->table_properties->user_collected_properties;
1030 auto index_type_pos = props.find(BlockBasedTablePropertyNames::kIndexType);
1031 if (index_type_pos != props.end()) {
1032 rep_->index_type = static_cast<BlockBasedTableOptions::IndexType>(
1033 DecodeFixed32(index_type_pos->second.c_str()));
1034 }
1035 auto min_ts_pos = props.find("rocksdb.timestamp_min");
1036 if (min_ts_pos != props.end()) {
1037 rep_->min_timestamp = Slice(min_ts_pos->second);
1038 }
1039 auto max_ts_pos = props.find("rocksdb.timestamp_max");
1040 if (max_ts_pos != props.end()) {
1041 rep_->max_timestamp = Slice(max_ts_pos->second);
1042 }
1043
1044 rep_->index_has_first_key =
1045 rep_->index_type == BlockBasedTableOptions::kBinarySearchWithFirstKey;
1046
1047 s = GetGlobalSequenceNumber(*(rep_->table_properties), largest_seqno,
1048 &(rep_->global_seqno));
1049 if (!s.ok()) {
1050 ROCKS_LOG_ERROR(rep_->ioptions.logger, "%s", s.ToString().c_str());
1051 }
1052 }
1053 return s;
1054}

1056Status BlockBasedTable::ReadRangeDelBlock(
1057 const ReadOptions& read_options, FilePrefetchBuffer* prefetch_buffer,
1058 InternalIterator* meta_iter,
1059 const InternalKeyComparator& internal_comparator,
1060 BlockCacheLookupContext* lookup_context) {
1061 Status s;
1062 BlockHandle range_del_handle;
1063 s = FindOptionalMetaBlock(meta_iter, kRangeDelBlockName, &range_del_handle);
1064 if (!s.ok()) {
1065 ROCKS_LOG_WARN(
1066 rep_->ioptions.logger,
1067 "Error when seeking to range delete tombstones block from file: %s",
1068 s.ToString().c_str());
1069 } else if (!range_del_handle.IsNull()) {
1070 Status tmp_status;
1071 std::unique_ptr<InternalIterator> iter(NewDataBlockIterator<DataBlockIter>(
1072 read_options, range_del_handle,
1073 /*input_iter=*/nullptr, BlockType::kRangeDeletion,
1074 /*get_context=*/nullptr, lookup_context, prefetch_buffer,
1075 /*for_compaction= */ false, /*async_read= */ false, tmp_status,
1076 /*use_block_cache_for_lookup=*/true));
1077 assert(iter != nullptr);
1078 s = iter->status();
1079 if (!s.ok()) {
1080 ROCKS_LOG_WARN(
1081 rep_->ioptions.logger,
1082 "Encountered error while reading data from range del block %s",
1083 s.ToString().c_str());
1084 IGNORE_STATUS_IF_ERROR(s);
1085 } else {
1086 std::vector<SequenceNumber> snapshots;
1087 // When user defined timestamps are not persisted, the range tombstone end
1088 // key read from the data block doesn't include user timestamp.
1089 // The range tombstone start key should already include user timestamp as
1090 // it's handled at block parsing level in the same way as the other data
1091 // blocks.
1092 rep_->fragmented_range_dels =
1093 std::make_shared<FragmentedRangeTombstoneList>(
1094 std::move(iter), internal_comparator, false /*for_compaction=*/,
1095 snapshots, rep_->user_defined_timestamps_persisted);
1096 }
1097 }
1098 return s;
1099}
1100
1101Status BlockBasedTable::PrefetchIndexAndFilterBlocks(
1102 const ReadOptions& ro, FilePrefetchBuffer* prefetch_buffer,
1103 InternalIterator* meta_iter, BlockBasedTable* new_table, bool prefetch_all,
1104 const BlockBasedTableOptions& table_options, const int level,
1105 size_t file_size, size_t max_file_size_for_l0_meta_pin,
1106 BlockCacheLookupContext* lookup_context) {
1107 // Find filter handle and filter type
1108 if (rep_->filter_policy) {
1109 auto name = rep_->filter_policy->CompatibilityName();
1110 bool builtin_compatible =
1111 strcmp(name, BuiltinFilterPolicy::kCompatibilityName()) == 0;
1112
1113 for (const auto& [filter_type, prefix] :
1114 {std::make_pair(Rep::FilterType::kFullFilter, kFullFilterBlockPrefix),
1115 std::make_pair(Rep::FilterType::kPartitionedFilter,
1116 kPartitionedFilterBlockPrefix),
1117 std::make_pair(Rep::FilterType::kNoFilter,
1118 kObsoleteFilterBlockPrefix)}) {
1119 if (builtin_compatible) {
1120 // This code is only here to deal with a hiccup in early 7.0.x where
1121 // there was an unintentional name change in the SST files metadata.
1122 // It should be OK to remove this in the future (late 2022) and just
1123 // have the 'else' code.
1124 // NOTE: the test:: names below are likely not needed but included
1125 // out of caution
1126 static const std::unordered_set<std::string> kBuiltinNameAndAliases = {
1127 BuiltinFilterPolicy::kCompatibilityName(),
1128 test::LegacyBloomFilterPolicy::kClassName(),
1129 test::FastLocalBloomFilterPolicy::kClassName(),
1130 test::Standard128RibbonFilterPolicy::kClassName(),
1131 "rocksdb.internal.DeprecatedBlockBasedBloomFilter",
1132 BloomFilterPolicy::kClassName(),
1133 RibbonFilterPolicy::kClassName(),
1134 };
1135
1136 // For efficiency, do a prefix seek and see if the first match is
1137 // good.
1138 meta_iter->Seek(prefix);
1139 if (meta_iter->status().ok() && meta_iter->Valid()) {
1140 Slice key = meta_iter->key();
1141 if (key.starts_with(prefix)) {
1142 key.remove_prefix(prefix.size());
1143 if (kBuiltinNameAndAliases.find(key.ToString()) !=
1144 kBuiltinNameAndAliases.end()) {
1145 Slice v = meta_iter->value();
1146 Status s = rep_->filter_handle.DecodeFrom(&v);
1147 if (s.ok()) {
1148 rep_->filter_type = filter_type;
1149 if (filter_type == Rep::FilterType::kNoFilter) {
1150 ROCKS_LOG_WARN(rep_->ioptions.logger,
1151 "Detected obsolete filter type in %s. Read "
1152 "performance might suffer until DB is fully "
1153 "re-compacted.",
1154 rep_->file->file_name().c_str());
1155 }
1156 break;
1157 }
1158 }
1159 }
1160 }
1161 } else {
1162 std::string filter_block_key = prefix + name;
1163 if (FindMetaBlock(meta_iter, filter_block_key, &rep_->filter_handle)
1164 .ok()) {
1165 rep_->filter_type = filter_type;
1166 if (filter_type == Rep::FilterType::kNoFilter) {
1167 ROCKS_LOG_WARN(
1168 rep_->ioptions.logger,
1169 "Detected obsolete filter type in %s. Read performance might "
1170 "suffer until DB is fully re-compacted.",
1171 rep_->file->file_name().c_str());
1172 }
1173 break;
1174 }
1175 }
1176 }
1177 }
1178 // Partition filters cannot be enabled without partition indexes
1179 assert(rep_->filter_type != Rep::FilterType::kPartitionedFilter ||
1180 rep_->index_type == BlockBasedTableOptions::kTwoLevelIndexSearch);
1181
1182 // Find compression dictionary handle
1183 Status s = FindOptionalMetaBlock(meta_iter, kCompressionDictBlockName,
1184 &rep_->compression_dict_handle);
1185 if (!s.ok()) {
1186 return s;
1187 }
1188
1189 BlockBasedTableOptions::IndexType index_type = rep_->index_type;
1190
1191 const bool use_cache = table_options.cache_index_and_filter_blocks;
1192
1193 const bool maybe_flushed =
1194 level == 0 && file_size <= max_file_size_for_l0_meta_pin;
1195 std::function<bool(PinningTier, PinningTier)> is_pinned =
1196 [maybe_flushed, &is_pinned](PinningTier pinning_tier,
1197 PinningTier fallback_pinning_tier) {
1198 // Fallback to fallback would lead to infinite recursion. Disallow it.
1199 assert(fallback_pinning_tier != PinningTier::kFallback);
1200
1201 switch (pinning_tier) {
1202 case PinningTier::kFallback:
1203 return is_pinned(fallback_pinning_tier,
1204 PinningTier::kNone /* fallback_pinning_tier */);
1205 case PinningTier::kNone:
1206 return false;
1207 case PinningTier::kFlushedAndSimilar:
1208 return maybe_flushed;
1209 case PinningTier::kAll:
1210 return true;
1211 };
1212
1213 // In GCC, this is needed to suppress `control reaches end of non-void
1214 // function [-Werror=return-type]`.
1215 assert(false);
1216 return false;
1217 };
1218 const bool pin_top_level_index = is_pinned(
1219 table_options.metadata_cache_options.top_level_index_pinning,
1220 table_options.pin_top_level_index_and_filter ? PinningTier::kAll
1221 : PinningTier::kNone);
1222 const bool pin_partition =
1223 is_pinned(table_options.metadata_cache_options.partition_pinning,
1224 table_options.pin_l0_filter_and_index_blocks_in_cache
1225 ? PinningTier::kFlushedAndSimilar
1226 : PinningTier::kNone);
1227 const bool pin_unpartitioned =
1228 is_pinned(table_options.metadata_cache_options.unpartitioned_pinning,
1229 table_options.pin_l0_filter_and_index_blocks_in_cache
1230 ? PinningTier::kFlushedAndSimilar
1231 : PinningTier::kNone);
1232
1233 // pin the first level of index
1234 const bool pin_index =
1235 index_type == BlockBasedTableOptions::kTwoLevelIndexSearch
1236 ? pin_top_level_index
1237 : pin_unpartitioned;
1238 // prefetch the first level of index
1239 // WART: this might be redundant (unnecessary cache hit) if !pin_index,
1240 // depending on prepopulate_block_cache option
1241 const bool prefetch_index = prefetch_all || pin_index;
1242
1243 std::unique_ptr<IndexReader> index_reader;
1244 s = new_table->CreateIndexReader(ro, prefetch_buffer, meta_iter, use_cache,
1245 prefetch_index, pin_index, lookup_context,
1246 &index_reader);
1247 if (!s.ok()) {
1248 return s;
1249 }
1250
1251 rep_->index_reader = std::move(index_reader);
1252
1253 // The partitions of partitioned index are always stored in cache. They
1254 // are hence follow the configuration for pin and prefetch regardless of
1255 // the value of cache_index_and_filter_blocks
1256 if (prefetch_all || pin_partition) {
1257 s = rep_->index_reader->CacheDependencies(ro, pin_partition,
1258 prefetch_buffer);
1259 }
1260 if (!s.ok()) {
1261 return s;
1262 }
1263
1264 // pin the first level of filter
1265 const bool pin_filter =
1266 rep_->filter_type == Rep::FilterType::kPartitionedFilter
1267 ? pin_top_level_index
1268 : pin_unpartitioned;
1269 // prefetch the first level of filter
1270 // WART: this might be redundant (unnecessary cache hit) if !pin_filter,
1271 // depending on prepopulate_block_cache option
1272 const bool prefetch_filter = prefetch_all || pin_filter;
1273
1274 if (rep_->filter_policy) {
1275 auto filter = new_table->CreateFilterBlockReader(
1276 ro, prefetch_buffer, use_cache, prefetch_filter, pin_filter,
1277 lookup_context);
1278
1279 if (filter) {
1280 // Refer to the comment above about paritioned indexes always being cached
1281 if (prefetch_all || pin_partition) {
1282 s = filter->CacheDependencies(ro, pin_partition, prefetch_buffer);
1283 if (!s.ok()) {
1284 return s;
1285 }
1286 }
1287 rep_->filter = std::move(filter);
1288 }
1289 }
1290
1291 if (!rep_->compression_dict_handle.IsNull()) {
1292 std::unique_ptr<UncompressionDictReader> uncompression_dict_reader;
1293 s = UncompressionDictReader::Create(
1294 this, ro, prefetch_buffer, use_cache, prefetch_all || pin_unpartitioned,
1295 pin_unpartitioned, lookup_context, &uncompression_dict_reader);
1296 if (!s.ok()) {
1297 return s;
1298 }
1299
1300 rep_->uncompression_dict_reader = std::move(uncompression_dict_reader);
1301 }
1302
1303 assert(s.ok());
1304 return s;

1372template <typename TBlocklike>
1373Cache::Priority BlockBasedTable::GetCachePriority() const {
1374 // Here we treat the legacy name "...index_and_filter_blocks..." to mean all
1375 // metadata blocks that might go into block cache, EXCEPT only those needed
1376 // for the read path (Get, etc.). TableProperties should not be needed on the
1377 // read path (prefix extractor setting is an O(1) size special case that we

1382 return Cache::Priority::LOW;
1383 } else if (rep_->table_options
1384 .cache_index_and_filter_blocks_with_high_priority) {
1385 return Cache::Priority::HIGH;
1386 } else {
1387 return Cache::Priority::LOW;
1388 }

1391template <typename TBlocklike>
1392WithBlocklikeCheck<Status, TBlocklike> BlockBasedTable::GetDataBlockFromCache(
1393 const Slice& cache_key, BlockCacheInterface<TBlocklike> block_cache,
1394 CachableEntry<TBlocklike>* out_parsed_block, GetContext* get_context,
1395 const UncompressionDict* dict) const {
1396 assert(out_parsed_block);
1397 assert(out_parsed_block->IsEmpty());
1398
1399 Status s;
1400 Statistics* statistics = rep_->ioptions.statistics.get();
1401
1402 // Lookup uncompressed cache first
1403 if (block_cache) {
1404 BlockCreateContext create_ctx = rep_->create_context;
1405 create_ctx.dict = dict;
1406 assert(!cache_key.empty());
1407 auto cache_handle = block_cache.LookupFull(
1408 cache_key, &create_ctx, GetCachePriority<TBlocklike>(), statistics,
1409 rep_->ioptions.lowest_used_cache_tier);
1410
1411 // Avoid updating metrics here if the handle is not complete yet. This
1412 // happens with MultiGet and secondary cache. So update the metrics only
1413 // if its a miss, or a hit and value is ready
1414 if (!cache_handle) {
1415 UpdateCacheMissMetrics(TBlocklike::kBlockType, get_context);
1416 } else {
1417 TBlocklike* value = block_cache.Value(cache_handle);
1418 if (value) {
1419 UpdateCacheHitMetrics(TBlocklike::kBlockType, get_context,
1420 block_cache.get()->GetUsage(cache_handle));
1421 }
1422 out_parsed_block->SetCachedValue(value, block_cache.get(), cache_handle);
1423 return s;
1424 }
1425 }
1426
1427 // If not found, search from the compressed block cache.
1428 assert(out_parsed_block->IsEmpty());
1429
1430 return s;
1431}

1433template <typename TBlocklike>
1434WithBlocklikeCheck<Status, TBlocklike> BlockBasedTable::PutDataBlockToCache(
1435 const Slice& cache_key, BlockCacheInterface<TBlocklike> block_cache,
1436 CachableEntry<TBlocklike>* out_parsed_block,
1437 BlockContents&& uncompressed_block_contents,
1438 BlockContents&& compressed_block_contents, CompressionType block_comp_type,
1439 const UncompressionDict& uncompression_dict,
1440 MemoryAllocator* memory_allocator, GetContext* get_context) const {
1441 const ImmutableOptions& ioptions = rep_->ioptions;
1442 const uint32_t format_version = rep_->table_options.format_version;
1443 assert(out_parsed_block);
1444 assert(out_parsed_block->IsEmpty());
1445
1446 Status s;
1447 Statistics* statistics = ioptions.stats;
1448
1449 std::unique_ptr<TBlocklike> block_holder;
1450 if (block_comp_type != kNoCompression &&
1451 uncompressed_block_contents.data.empty()) {
1452 assert(compressed_block_contents.data.data());
1453 // Retrieve the uncompressed contents into a new buffer
1454 UncompressionContext context(block_comp_type);
1455 UncompressionInfo info(context, uncompression_dict, block_comp_type);
1456 s = UncompressBlockData(info, compressed_block_contents.data.data(),
1457 compressed_block_contents.data.size(),
1458 &uncompressed_block_contents, format_version,
1459 ioptions, memory_allocator);
1460 if (!s.ok()) {
1461 return s;
1462 }
1463 }
1464 rep_->create_context.Create(&block_holder,
1465 std::move(uncompressed_block_contents));
1466
1467 // insert into uncompressed block cache
1468 if (block_cache && block_holder->own_bytes()) {
1469 size_t charge = block_holder->ApproximateMemoryUsage();
1470 BlockCacheTypedHandle<TBlocklike>* cache_handle = nullptr;
1471 s = block_cache.InsertFull(cache_key, block_holder.get(), charge,
1472 &cache_handle, GetCachePriority<TBlocklike>(),
1473 rep_->ioptions.lowest_used_cache_tier,
1474 compressed_block_contents.data, block_comp_type);
1475
1476 if (s.ok()) {
1477 assert(cache_handle != nullptr);
1478 out_parsed_block->SetCachedValue(block_holder.release(),
1479 block_cache.get(), cache_handle);
1480
1481 UpdateCacheInsertionMetrics(TBlocklike::kBlockType, get_context, charge,
1482 s.IsOkOverwritten(), rep_->ioptions.stats);
1483 } else {
1484 RecordTick(statistics, BLOCK_CACHE_ADD_FAILURES);
1485 }
1486 } else {
1487 out_parsed_block->SetOwnedValue(std::move(block_holder));
1488 }
1489
1490 return s;
1491}

1561// Right now only called for Data blocks.
1562template <typename TBlocklike>
1563Status BlockBasedTable::LookupAndPinBlocksInCache(
1564 const ReadOptions& ro, const BlockHandle& handle,
1565 CachableEntry<TBlocklike>* out_parsed_block) const {
1566 BlockCacheInterface<TBlocklike> block_cache{
1567 rep_->table_options.block_cache.get()};
1568
1569 assert(block_cache);
1570
1571 Status s;
1572 CachableEntry<UncompressionDict> uncompression_dict;
1573 if (rep_->uncompression_dict_reader) {
1574 s = rep_->uncompression_dict_reader->GetOrReadUncompressionDictionary(
1575 /* prefetch_buffer= */ nullptr, ro,
1576 /* get_context= */ nullptr, /* lookup_context= */ nullptr,
1577 &uncompression_dict);
1578 if (!s.ok()) {
1579 return s;
1580 }
1581 }
1582
1583 // Do the lookup.
1584 CacheKey key_data = GetCacheKey(rep_->base_cache_key, handle);
1585 const Slice key = key_data.AsSlice();
1586
1587 Statistics* statistics = rep_->ioptions.statistics.get();
1588
1589 BlockCreateContext create_ctx = rep_->create_context;
1590 create_ctx.dict = uncompression_dict.GetValue()
1591 ? uncompression_dict.GetValue()
1592 : &UncompressionDict::GetEmptyDict();
1593
1594 auto cache_handle =
1595 block_cache.LookupFull(key, &create_ctx, GetCachePriority<TBlocklike>(),
1596 statistics, rep_->ioptions.lowest_used_cache_tier);
1597
1598 if (!cache_handle) {
1599 UpdateCacheMissMetrics(TBlocklike::kBlockType, /* get_context = */ nullptr);
1600 return s;
1601 }
1602
1603 // Found in Cache.
1604 TBlocklike* value = block_cache.Value(cache_handle);
1605 if (value) {
1606 UpdateCacheHitMetrics(TBlocklike::kBlockType, /* get_context = */ nullptr,
1607 block_cache.get()->GetUsage(cache_handle));
1608 }
1609 out_parsed_block->SetCachedValue(value, block_cache.get(), cache_handle);
1610
1611 assert(!out_parsed_block->IsEmpty());
1612
1613 return s;
1614}

1616// If contents is nullptr, this function looks up the block caches for the
1617// data block referenced by handle, and read the block from disk if necessary.
1618// If contents is non-null, it skips the cache lookup and disk read, since
1619// the caller has already read it. In both cases, if ro.fill_cache is true,
1620// it inserts the block into the block cache.
1621template <typename TBlocklike>
1622WithBlocklikeCheck<Status, TBlocklike>
1623BlockBasedTable::MaybeReadBlockAndLoadToCache(
1624 FilePrefetchBuffer* prefetch_buffer, const ReadOptions& ro,
1625 const BlockHandle& handle, const UncompressionDict& uncompression_dict,
1626 bool for_compaction, CachableEntry<TBlocklike>* out_parsed_block,
1627 GetContext* get_context, BlockCacheLookupContext* lookup_context,
1628 BlockContents* contents, bool async_read,
1629 bool use_block_cache_for_lookup) const {
1630 assert(out_parsed_block != nullptr);
1631 const bool no_io = (ro.read_tier == kBlockCacheTier);
1632 BlockCacheInterface<TBlocklike> block_cache{
1633 rep_->table_options.block_cache.get()};
1634 // First, try to get the block from the cache
1635 //
1636 // If either block cache is enabled, we'll try to read from it.
1637 Status s;
1638 CacheKey key_data;
1639 Slice key;
1640 bool is_cache_hit = false;
1641 if (block_cache) {
1642 // create key for block cache
1643 key_data = GetCacheKey(rep_->base_cache_key, handle);
1644 key = key_data.AsSlice();
1645
1646 if (!contents) {
1647 if (use_block_cache_for_lookup) {
1648 s = GetDataBlockFromCache(key, block_cache, out_parsed_block,
1649 get_context, &uncompression_dict);
1650 // Value could still be null at this point, so check the cache handle
1651 // and update the read pattern for prefetching
1652 if (out_parsed_block->GetValue() ||
1653 out_parsed_block->GetCacheHandle()) {
1654 // TODO(haoyu): Differentiate cache hit on uncompressed block cache
1655 // and compressed block cache.
1656 is_cache_hit = true;
1657 if (prefetch_buffer) {
1658 // Update the block details so that PrefetchBuffer can use the read
1659 // pattern to determine if reads are sequential or not for
1660 // prefetching. It should also take in account blocks read from
1661 // cache.
1662 prefetch_buffer->UpdateReadPattern(
1663 handle.offset(), BlockSizeWithTrailer(handle),
1664 ro.adaptive_readahead /*decrease_readahead_size*/);
1665 }
1666 }
1667 }
1668 }
1669
1670 // Can't find the block from the cache. If I/O is allowed, read from the
1671 // file.
1672 if (out_parsed_block->GetValue() == nullptr &&
1673 out_parsed_block->GetCacheHandle() == nullptr && !no_io &&
1674 ro.fill_cache) {
1675 Statistics* statistics = rep_->ioptions.stats;
1676 const bool maybe_compressed =
1677 TBlocklike::kBlockType != BlockType::kFilter &&
1678 TBlocklike::kBlockType != BlockType::kCompressionDictionary &&
1679 rep_->blocks_maybe_compressed;
1680 // This flag, if true, tells BlockFetcher to return the uncompressed
1681 // block when ReadBlockContents() is called.
1682 const bool do_uncompress = maybe_compressed;
1683 CompressionType contents_comp_type;
1684 // Maybe serialized or uncompressed
1685 BlockContents tmp_contents;
1686 BlockContents uncomp_contents;
1687 BlockContents comp_contents;
1688 if (!contents) {
1689 Histograms histogram = for_compaction ? READ_BLOCK_COMPACTION_MICROS
1690 : READ_BLOCK_GET_MICROS;
1691 StopWatch sw(rep_->ioptions.clock, statistics, histogram);
1692 // Setting do_uncompress to false may cause an extra mempcy in the
1693 // following cases -
1694 // 1. Compression is enabled, but block is not actually compressed
1695 // 2. Compressed block is in the prefetch buffer
1696 // 3. Direct IO
1697 //
1698 // It would also cause a memory allocation to be used rather than
1699 // stack if the compressed block size is < 5KB
1700 BlockFetcher block_fetcher(
1701 rep_->file.get(), prefetch_buffer, rep_->footer, ro, handle,
1702 &tmp_contents, rep_->ioptions, do_uncompress, maybe_compressed,
1703 TBlocklike::kBlockType, uncompression_dict,
1704 rep_->persistent_cache_options,
1705 GetMemoryAllocator(rep_->table_options),
1706 /*allocator=*/nullptr);
1707
1708 // If prefetch_buffer is not allocated, it will fallback to synchronous
1709 // reading of block contents.
1710 if (async_read && prefetch_buffer != nullptr) {
1711 s = block_fetcher.ReadAsyncBlockContents();
1712 if (!s.ok()) {
1713 return s;
1714 }
1715 } else {
1716 s = block_fetcher.ReadBlockContents();
1717 }
1718
1719 contents_comp_type = block_fetcher.get_compression_type();
1720 if (get_context) {
1721 switch (TBlocklike::kBlockType) {
1722 case BlockType::kIndex:
1723 ++get_context->get_context_stats_.num_index_read;
1724 break;
1725 case BlockType::kFilter:
1726 case BlockType::kFilterPartitionIndex:
1727 ++get_context->get_context_stats_.num_filter_read;
1728 break;
1729 default:
1730 break;
1731 }
1732 }
1733 if (s.ok()) {
1734 if (do_uncompress && contents_comp_type != kNoCompression) {
1735 comp_contents = BlockContents(block_fetcher.GetCompressedBlock());
1736 uncomp_contents = std::move(tmp_contents);
1737 } else if (contents_comp_type != kNoCompression) {
1738 // do_uncompress must be false, so output of BlockFetcher is
1739 // compressed
1740 comp_contents = std::move(tmp_contents);
1741 } else {
1742 uncomp_contents = std::move(tmp_contents);
1743 }
1744
1745 // If filling cache is allowed and a cache is configured, try to put
1746 // the block to the cache. Do this here while block_fetcher is in
1747 // scope, since comp_contents will be a reference to the compressed
1748 // block in block_fetcher
1749 s = PutDataBlockToCache(
1750 key, block_cache, out_parsed_block, std::move(uncomp_contents),
1751 std::move(comp_contents), contents_comp_type, uncompression_dict,
1752 GetMemoryAllocator(rep_->table_options), get_context);
1753 }
1754 } else {
1755 contents_comp_type = GetBlockCompressionType(*contents);
1756 if (contents_comp_type != kNoCompression) {
1757 comp_contents = std::move(*contents);
1758 } else {
1759 uncomp_contents = std::move(*contents);
1760 }
1761
1762 if (s.ok()) {
1763 // If filling cache is allowed and a cache is configured, try to put
1764 // the block to the cache.
1765 s = PutDataBlockToCache(
1766 key, block_cache, out_parsed_block, std::move(uncomp_contents),
1767 std::move(comp_contents), contents_comp_type, uncompression_dict,
1768 GetMemoryAllocator(rep_->table_options), get_context);
1769 }
1770 }
1771 }
1772 }
1773
1774 // TODO: optimize so that lookup_context != nullptr implies the others
1775 if (block_cache_tracer_ && block_cache_tracer_->is_tracing_enabled() &&
1776 lookup_context) {
1777 SaveLookupContextOrTraceRecord(
1778 key, is_cache_hit, ro, out_parsed_block->GetValue(), lookup_context);
1779 }
1780
1781 assert(s.ok() || out_parsed_block->GetValue() == nullptr);
1782 return s;
1783}

1785template <typename TBlocklike>
1786WithBlocklikeCheck<void, TBlocklike>
1787BlockBasedTable::SaveLookupContextOrTraceRecord(
1788 const Slice& block_key, bool is_cache_hit, const ReadOptions& ro,
1789 const TBlocklike* parsed_block_value,
1790 BlockCacheLookupContext* lookup_context) const {
1791 assert(lookup_context);
1792 size_t usage = 0;
1793 uint64_t nkeys = 0;
1794 if (parsed_block_value) {
1795 // Approximate the number of keys in the block using restarts.
1796 int interval = rep_->table_options.block_restart_interval;
1797 nkeys = interval * GetBlockNumRestarts(*parsed_block_value);
1798 // On average, the last restart should be just over half utilized.
1799 // Specifically, 1..N should be N/2 + 0.5. For example, 7 -> 4, 8 -> 4.5.
1800 // Use the get_id to alternate between rounding up vs. down.
1801 if (nkeys > 0) {
1802 bool rounding = static_cast<int>(lookup_context->get_id) & 1;
1803 nkeys -= (interval - rounding) / 2;
1804 }
1805 usage = parsed_block_value->ApproximateMemoryUsage();
1806 }
1807 TraceType trace_block_type = TraceType::kTraceMax;
1808 switch (TBlocklike::kBlockType) {
1809 case BlockType::kData:
1810 trace_block_type = TraceType::kBlockTraceDataBlock;
1811 break;
1812 case BlockType::kFilter:
1813 case BlockType::kFilterPartitionIndex:
1814 trace_block_type = TraceType::kBlockTraceFilterBlock;
1815 break;
1816 case BlockType::kCompressionDictionary:
1817 trace_block_type = TraceType::kBlockTraceUncompressionDictBlock;
1818 break;

2207Status BlockBasedTable::ApproximateKeyAnchors(const ReadOptions& read_options,
2208 std::vector<Anchor>& anchors) {
2209 // We iterator the whole index block here. More efficient implementation
2210 // is possible if we push this operation into IndexReader. For example, we
2211 // can directly sample from restart block entries in the index block and
2212 // only read keys needed. Here we take a simple solution. Performance is
2213 // likely not to be a problem. We are compacting the whole file, so all
2214 // keys will be read out anyway. An extra read to index block might be
2215 // a small share of the overhead. We can try to optimize if needed.
2216 //
2217 // `CacheDependencies()` brings all the blocks into cache using one I/O. That
2218 // way the full index scan usually finds the index data it is looking for in
2219 // cache rather than doing an I/O for each "dependency" (partition).
2220 Status s = rep_->index_reader->CacheDependencies(
2221 read_options, false /* pin */, nullptr /* prefetch_buffer */);
2222 if (!s.ok()) {
2223 return s;
2224 }
2225
2226 IndexBlockIter iiter_on_stack;
2227 auto iiter = NewIndexIterator(
2228 read_options, /*disable_prefix_seek=*/false, &iiter_on_stack,
2229 /*get_context=*/nullptr, /*lookup_context=*/nullptr);
2230 std::unique_ptr<InternalIteratorBase<IndexValue>> iiter_unique_ptr;
2231 if (iiter != &iiter_on_stack) {
2232 iiter_unique_ptr.reset(iiter);
2233 }
2234
2235 // If needed the threshold could be more adaptive. For example, it can be
2236 // based on size, so that a larger will be sampled to more partitions than a
2237 // smaller file. The size might also need to be passed in by the caller based
2238 // on total compaction size.
2239 const uint64_t kMaxNumAnchors = uint64_t{128};
2240 uint64_t num_blocks = this->GetTableProperties()->num_data_blocks;
2241 uint64_t num_blocks_per_anchor = num_blocks / kMaxNumAnchors;
2242 if (num_blocks_per_anchor == 0) {
2243 num_blocks_per_anchor = 1;
2244 }
2245
2246 uint64_t count = 0;
2247 std::string last_key;
2248 uint64_t range_size = 0;
2249 uint64_t prev_offset = 0;
2250 for (iiter->SeekToFirst(); iiter->Valid(); iiter->Next()) {
2251 const BlockHandle& bh = iiter->value().handle;
2252 range_size += bh.offset() + bh.size() - prev_offset;
2253 prev_offset = bh.offset() + bh.size();
2254 if (++count % num_blocks_per_anchor == 0) {
2255 count = 0;
2256 anchors.emplace_back(iiter->user_key(), range_size);
2257 range_size = 0;
2258 } else {
2259 last_key = iiter->user_key().ToString();
2260 }
2261 }
2262 if (count != 0) {
2263 anchors.emplace_back(last_key, range_size);
2264 }
2265 return Status::OK();
2266}

2281Status BlockBasedTable::Get(const ReadOptions& read_options, const Slice& key,
2282 GetContext* get_context,
2283 const SliceTransform* prefix_extractor,
2284 bool skip_filters) {
2285 // Similar to Bloom filter !may_match
2286 // If timestamp is beyond the range of the table, skip
2287 if (!TimestampMayMatch(read_options)) {
2288 return Status::OK();
2289 }
2290 assert(key.size() >= 8); // key must be internal key
2291 assert(get_context != nullptr);
2292 Status s;
2293
2294 FilterBlockReader* const filter =
2295 !skip_filters ? rep_->filter.get() : nullptr;
2296
2297 // First check the full filter
2298 // If full filter not useful, Then go into each block
2299 uint64_t tracing_get_id = get_context->get_tracing_get_id();
2300 BlockCacheLookupContext lookup_context{
2301 TableReaderCaller::kUserGet, tracing_get_id,
2302 /*get_from_user_specified_snapshot=*/read_options.snapshot != nullptr};
2303 if (block_cache_tracer_ && block_cache_tracer_->is_tracing_enabled()) {
2304 // Trace the key since it contains both user key and sequence number.
2305 lookup_context.referenced_key = key.ToString();
2306 lookup_context.get_from_user_specified_snapshot =
2307 read_options.snapshot != nullptr;
2308 }
2309 TEST_SYNC_POINT("BlockBasedTable::Get:BeforeFilterMatch");
2310 const bool may_match =
2311 FullFilterKeyMayMatch(filter, key, prefix_extractor, get_context,
2312 &lookup_context, read_options);
2313 TEST_SYNC_POINT("BlockBasedTable::Get:AfterFilterMatch");
2314 if (may_match) {
2315 IndexBlockIter iiter_on_stack;
2316 // if prefix_extractor found in block differs from options, disable
2317 // BlockPrefixIndex. Only do this check when index_type is kHashSearch.
2318 bool need_upper_bound_check = false;
2319 if (rep_->index_type == BlockBasedTableOptions::kHashSearch) {
2320 need_upper_bound_check = PrefixExtractorChanged(prefix_extractor);
2321 }
2322 auto iiter =
2323 NewIndexIterator(read_options, need_upper_bound_check, &iiter_on_stack,
2324 get_context, &lookup_context);
2325 std::unique_ptr<InternalIteratorBase<IndexValue>> iiter_unique_ptr;
2326 if (iiter != &iiter_on_stack) {
2327 iiter_unique_ptr.reset(iiter);
2328 }
2329
2330 size_t ts_sz =
2331 rep_->internal_comparator.user_comparator()->timestamp_size();
2332 bool matched = false; // if such user key matched a key in SST
2333 bool done = false;
2334 for (iiter->Seek(key); iiter->Valid() && !done; iiter->Next()) {
2335 IndexValue v = iiter->value();
2336
2337 if (!v.first_internal_key.empty() && !skip_filters &&
2338 UserComparatorWrapper(rep_->internal_comparator.user_comparator())
2339 .CompareWithoutTimestamp(
2340 ExtractUserKey(key),
2341 ExtractUserKey(v.first_internal_key)) < 0) {
2342 // The requested key falls between highest key in previous block and
2343 // lowest key in current block.
2344 break;
2345 }
2346
2347 BlockCacheLookupContext lookup_data_block_context{
2348 TableReaderCaller::kUserGet, tracing_get_id,
2349 /*get_from_user_specified_snapshot=*/read_options.snapshot !=
2350 nullptr};
2351 bool does_referenced_key_exist = false;
2352 DataBlockIter biter;
2353 uint64_t referenced_data_size = 0;
2354 Status tmp_status;
2355 NewDataBlockIterator<DataBlockIter>(
2356 read_options, v.handle, &biter, BlockType::kData, get_context,
2357 &lookup_data_block_context, /*prefetch_buffer=*/nullptr,
2358 /*for_compaction=*/false, /*async_read=*/false, tmp_status,
2359 /*use_block_cache_for_lookup=*/true);
2360
2361 if (read_options.read_tier == kBlockCacheTier &&
2362 biter.status().IsIncomplete()) {
2363 // couldn't get block from block_cache
2364 // Update Saver.state to Found because we are only looking for
2365 // whether we can guarantee the key is not there when "no_io" is set
2366 get_context->MarkKeyMayExist();
2367 s = biter.status();
2368 break;
2369 }
2370 if (!biter.status().ok()) {
2371 s = biter.status();
2372 break;
2373 }
2374
2375 bool may_exist = biter.SeekForGet(key);
2376 // If user-specified timestamp is supported, we cannot end the search
2377 // just because hash index lookup indicates the key+ts does not exist.
2378 if (!may_exist && ts_sz == 0) {
2379 // HashSeek cannot find the key this block and the the iter is not
2380 // the end of the block, i.e. cannot be in the following blocks
2381 // either. In this case, the seek_key cannot be found, so we break
2382 // from the top level for-loop.
2383 done = true;
2384 } else {
2385 // Call the *saver function on each entry/block until it returns false
2386 for (; biter.Valid(); biter.Next()) {
2387 ParsedInternalKey parsed_key;
2388 Status pik_status = ParseInternalKey(
2389 biter.key(), &parsed_key, false /* log_err_key */); // TODO
2390 if (!pik_status.ok()) {
2391 s = pik_status;
2392 break;
2393 }
2394
2395 Status read_status;
2396 bool ret = get_context->SaveValue(
2397 parsed_key, biter.value(), &matched, &read_status,
2398 biter.IsValuePinned() ? &biter : nullptr);
2399 if (!read_status.ok()) {
2400 s = read_status;
2401 break;
2402 }
2403 if (!ret) {
2404 if (get_context->State() == GetContext::GetState::kFound) {
2405 does_referenced_key_exist = true;
2406 referenced_data_size = biter.key().size() + biter.value().size();
2407 }
2408 done = true;
2409 break;
2410 }
2411 }
2412 if (s.ok()) {
2413 s = biter.status();
2414 }
2415 if (!s.ok()) {
2416 break;
2417 }
2418 }
2419 // Write the block cache access record.
2420 if (block_cache_tracer_ && block_cache_tracer_->is_tracing_enabled()) {
2421 // Avoid making copy of block_key, cf_name, and referenced_key when
2422 // constructing the access record.
2423 Slice referenced_key;
2424 if (does_referenced_key_exist) {
2425 referenced_key = biter.key();
2426 } else {
2427 referenced_key = key;
2428 }
2429 FinishTraceRecord(lookup_data_block_context,
2430 lookup_data_block_context.block_key, referenced_key,
2431 does_referenced_key_exist, referenced_data_size);
2432 }
2433
2434 if (done) {
2435 // Avoid the extra Next which is expensive in two-level indexes
2436 break;
2437 }
2438 }
2439 if (matched && filter != nullptr) {
2440 if (rep_->whole_key_filtering) {
2441 RecordTick(rep_->ioptions.stats, BLOOM_FILTER_FULL_TRUE_POSITIVE);
2442 } else {
2443 RecordTick(rep_->ioptions.stats, BLOOM_FILTER_PREFIX_TRUE_POSITIVE);
2444 }
2445 // Includes prefix stats
2446 PERF_COUNTER_BY_LEVEL_ADD(bloom_filter_full_true_positive, 1,
2447 rep_->level);
2448 }
2449
2450 if (s.ok() && !iiter->status().IsNotFound()) {
2451 s = iiter->status();
2452 }
2453 }
2454
2455 return s;
2456}

2487Status BlockBasedTable::Prefetch(const ReadOptions& read_options,
2488 const Slice* const begin,
2489 const Slice* const end) {
2490 auto& comparator = rep_->internal_comparator;
2491 UserComparatorWrapper user_comparator(comparator.user_comparator());
2492 // pre-condition
2493 if (begin && end && comparator.Compare(*begin, *end) > 0) {
2494 return Status::InvalidArgument(*begin, *end);
2495 }
2496 BlockCacheLookupContext lookup_context{TableReaderCaller::kPrefetch};
2497 IndexBlockIter iiter_on_stack;
2498 auto iiter = NewIndexIterator(read_options, /*need_upper_bound_check=*/false,
2499 &iiter_on_stack, /*get_context=*/nullptr,
2500 &lookup_context);
2501 std::unique_ptr<InternalIteratorBase<IndexValue>> iiter_unique_ptr;
2502 if (iiter != &iiter_on_stack) {
2503 iiter_unique_ptr = std::unique_ptr<InternalIteratorBase<IndexValue>>(iiter);
2504 }
2505
2506 if (!iiter->status().ok()) {
2507 // error opening index iterator
2508 return iiter->status();
2509 }
2510
2511 // indicates if we are on the last page that need to be pre-fetched
2512 bool prefetching_boundary_page = false;
2513
2514 for (begin ? iiter->Seek(*begin) : iiter->SeekToFirst(); iiter->Valid();
2515 iiter->Next()) {
2516 BlockHandle block_handle = iiter->value().handle;
2517 const bool is_user_key = !rep_->index_key_includes_seq;
2518 if (end &&
2519 ((!is_user_key && comparator.Compare(iiter->key(), *end) >= 0) ||
2520 (is_user_key &&
2521 user_comparator.Compare(iiter->key(), ExtractUserKey(*end)) >= 0))) {
2522 if (prefetching_boundary_page) {
2523 break;
2524 }
2525
2526 // The index entry represents the last key in the data block.
2527 // We should load this page into memory as well, but no more
2528 prefetching_boundary_page = true;
2529 }
2530
2531 // Load the block specified by the block_handle into the block cache
2532 DataBlockIter biter;
2533 Status tmp_status;
2534 NewDataBlockIterator<DataBlockIter>(
2535 read_options, block_handle, &biter, /*type=*/BlockType::kData,
2536 /*get_context=*/nullptr, &lookup_context,
2537 /*prefetch_buffer=*/nullptr, /*for_compaction=*/false,
2538 /*async_read=*/false, tmp_status, /*use_block_cache_for_lookup=*/true);
2539
2540 if (!biter.status().ok()) {
2541 // there was an unexpected error while pre-fetching
2542 return biter.status();
2543 }
2544 }
2545
2546 return Status::OK();
2547}

2583Status BlockBasedTable::VerifyChecksumInBlocks(
2584 const ReadOptions& read_options,
2585 InternalIteratorBase<IndexValue>* index_iter) {
2586 Status s;
2587 // We are scanning the whole file, so no need to do exponential
2588 // increasing of the buffer size.
2589 size_t readahead_size = (read_options.readahead_size != 0)
2590 ? read_options.readahead_size
2591 : rep_->table_options.max_auto_readahead_size;
2592 // FilePrefetchBuffer doesn't work in mmap mode and readahead is not
2593 // needed there.
2594 ReadaheadParams readahead_params;
2595 readahead_params.initial_readahead_size = readahead_size;
2596 readahead_params.max_readahead_size = readahead_size;
2597 FilePrefetchBuffer prefetch_buffer(
2598 readahead_params, !rep_->ioptions.allow_mmap_reads /* enable */);
2599
2600 for (index_iter->SeekToFirst(); index_iter->Valid(); index_iter->Next()) {
2601 s = index_iter->status();
2602 if (!s.ok()) {
2603 break;
2604 }
2605 BlockHandle handle = index_iter->value().handle;
2606 BlockContents contents;
2607 BlockFetcher block_fetcher(
2608 rep_->file.get(), &prefetch_buffer, rep_->footer, read_options, handle,
2609 &contents, rep_->ioptions, false /* decompress */,
2610 false /*maybe_compressed*/, BlockType::kData,
2611 UncompressionDict::GetEmptyDict(), rep_->persistent_cache_options);
2612 s = block_fetcher.ReadBlockContents();
2613 if (!s.ok()) {
2614 break;
2615 }
2616 }
2617 if (s.ok()) {
2618 // In the case of two level indexes, we would have exited the above loop
2619 // by checking index_iter->Valid(), but Valid() might have returned false
2620 // due to an IO error. So check the index_iter status
2621 s = index_iter->status();
2622 }
2623 return s;
