11namespace ROCKSDB_NAMESPACE {

13class PerfStepTimer {
14 public:
15 explicit PerfStepTimer(
16 uint64_t* metric, SystemClock* clock = nullptr, bool use_cpu_time = false,
17 PerfLevel enable_level = PerfLevel::kEnableTimeExceptForMutex,
18 Statistics* statistics = nullptr, uint32_t ticker_type = 0)
19 : perf_counter_enabled_(perf_level >= enable_level),
20 use_cpu_time_(use_cpu_time),
21 ticker_type_(ticker_type),
22 clock_((perf_counter_enabled_ || statistics != nullptr)
23 ? (clock ? clock : SystemClock::Default().get())
24 : nullptr),
25 start_(0),
26 metric_(metric),
27 statistics_(statistics) {}
28
29 ~PerfStepTimer() { Stop(); }
30
31 void Start() {
32 if (perf_counter_enabled_ || statistics_ != nullptr) {
33 start_ = time_now();
34 }
35 }
36
37 void Measure() {
38 if (start_) {
39 uint64_t now = time_now();
40 *metric_ += now - start_;
41 start_ = now;
42 }
43 }
44
45 void Stop() {
46 if (start_) {
47 uint64_t duration = time_now() - start_;
48 if (perf_counter_enabled_) {
49 *metric_ += duration;
50 }
51
52 if (statistics_ != nullptr) {
53 RecordTick(statistics_, ticker_type_, duration);
54 }
55 start_ = 0;
56 }
57 }
58
59 private:
60 uint64_t time_now() {
61 if (!use_cpu_time_) {
62 return clock_->NowNanos();
63 } else {
64 return clock_->CPUNanos();
65 }
66 }
67
68 const bool perf_counter_enabled_;
69 const bool use_cpu_time_;
70 uint32_t ticker_type_;
71 SystemClock* const clock_;
72 uint64_t start_;
73 uint64_t* metric_;
74 Statistics* statistics_;
75};
