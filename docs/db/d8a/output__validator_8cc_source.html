13 if (enable_hash_) {

15 paranoid_hash_ = NPHash64(key.data(), key.size(), paranoid_hash_);
16 paranoid_hash_ = NPHash64(value.data(), value.size(), paranoid_hash_);
17 }
18 if (key.size() < kNumInternalBytes) {

20 "Compaction tries to write a key without internal bytes.");
21 }
22 // prev_key_ starts with empty.
23 if (!prev_key_.empty() && icmp_.Compare(key, prev_key_) < 0) {
24 return Status::Corruption("Compaction sees out-of-order keys.");
25 }

28}
