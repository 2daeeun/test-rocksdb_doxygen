22// PersistentCache
23//
24// Persistent cache interface for caching IO pages on a persistent medium. The
25// cache interface is specifically designed for persistent read cache.
26class PersistentCache {
27 public:
28 using StatsType = std::vector<std::map<std::string, double>>;
29
30 virtual ~PersistentCache() {}
31
32 // Insert to page cache
33 //
34 // page_key Identifier to identify a page uniquely across restarts
35 // data Page data to copy (caller retains ownership)
36 // size Size of the page
37 virtual Status Insert(const Slice& key, const char* data,
38 const size_t size) = 0;
39
40 // Lookup page cache by page identifier
41 //
42 // page_key Page identifier
43 // buf Buffer where the data should be copied
44 // size Size of the page
45 virtual Status Lookup(const Slice& key, std::unique_ptr<char[]>* data,
46 size_t* size) = 0;
47
48 // True if the cache is configured to store serialized blocks, which are
49 // potentially compressed and include a trailer (when SST format calls for
50 // one). False if the cache stores uncompressed blocks (no trailer).
51 virtual bool IsCompressed() = 0;
52
53 // Return stats as map of {string, double} per-tier
54 //
55 // Persistent cache can be initialized as a tier of caches. The stats are per
56 // tire top-down
57 virtual StatsType Stats() = 0;
58
59 virtual std::string GetPrintableOptions() const = 0;
60
61 // Return a new numeric id. May be used by multiple clients who are
62 // sharding the same persistent cache to partition the key space. Typically
63 // the client will allocate a new id at startup and prepend the id to its
64 // cache keys.
65 virtual uint64_t NewId() = 0;
66};

74} // namespace ROCKSDB_NAMESPACE
