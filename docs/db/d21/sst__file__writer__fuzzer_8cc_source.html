42// Keys in SST file writer operations must be unique and in ascending order.
43// For each DBOperation generated by the fuzzer, this function is called on
44// it to deduplicate and sort the keys in the DBOperations.
45protobuf_mutator::libfuzzer::PostProcessorRegistration<DBOperations> reg = {
46 [](DBOperations* input, unsigned int /* seed */) {
47 const Comparator* comparator = BytewiseComparator();
48 auto ops = input->mutable_operations();
49
50 // Make sure begin <= end for DELETE_RANGE.
51 for (DBOperation& op : *ops) {
52 if (op.type() == OpType::DELETE_RANGE) {
53 auto begin = op.key();
54 auto end = op.value();
55 if (comparator->Compare(begin, end) > 0) {
56 std::swap(begin, end);
57 op.set_key(begin);
58 op.set_value(end);
59 }
60 }
61 }
62
63 std::sort(ops->begin(), ops->end(),
64 [&comparator](const DBOperation& a, const DBOperation& b) {
65 return comparator->Compare(a.key(), b.key()) < 0;
66 });
67
68 auto last = std::unique(
69 ops->begin(), ops->end(),
70 [&comparator](const DBOperation& a, const DBOperation& b) {
71 return comparator->Compare(a.key(), b.key()) == 0;
72 });
73 ops->erase(last, ops->end());
74 }};

76TableReader* NewTableReader(const std::string& sst_file_path,
77 const Options& options,
78 const EnvOptions& env_options,
79 const ImmutableCFOptions& cf_ioptions) {
80 // This code block is similar to SstFileReader::Open.
81
82 uint64_t file_size = 0;
83 std::unique_ptr<RandomAccessFileReader> file_reader;
84 std::unique_ptr<TableReader> table_reader;
85 const auto& fs = options.env->GetFileSystem();
86 FileOptions fopts(env_options);
87 Status s = options.env->GetFileSize(sst_file_path, &file_size);
88 if (s.ok()) {
89 s = RandomAccessFileReader::Create(fs, sst_file_path, fopts, &file_reader,
90 nullptr);
91 }
92 if (s.ok()) {
93 ImmutableOptions iopts(options, cf_ioptions);
94 TableReaderOptions t_opt(iopts, /*prefix_extractor=*/nullptr, env_options,
95 cf_ioptions.internal_comparator,
96 0 /* block_protection_bytes_per_key */);
97 t_opt.largest_seqno = kMaxSequenceNumber;
98 s = options.table_factory->NewTableReader(t_opt, std::move(file_reader),
99 file_size, &table_reader,
100 /*prefetch=*/false);
101 }
102 if (!s.ok()) {
103 std::cerr << "Failed to create TableReader for " << sst_file_path << ": "
104 << s.ToString() << std::endl;
105 abort();
106 }
107 return table_reader.release();
108}

110ValueType ToValueType(OpType op_type) {
111 switch (op_type) {
112 case OpType::PUT:
113 return ValueType::kTypeValue;
114 case OpType::MERGE:
115 return ValueType::kTypeMerge;
116 case OpType::DELETE:
117 return ValueType::kTypeDeletion;
118 case OpType::DELETE_RANGE:
119 return ValueType::kTypeRangeDeletion;
120 default:
121 std::cerr << "Unknown operation type " << static_cast<int>(op_type)
122 << std::endl;
123 abort();
124 }
125}

127// Fuzzes DB operations as input, let SstFileWriter generate a SST file
128// according to the operations, then let TableReader read and check all the
129// key-value pairs from the generated SST file.
130DEFINE_PROTO_FUZZER(DBOperations& input) {
131 if (input.operations().empty()) {
132 return;
133 }
134
135 std::string sstfile;
136 {
137 auto fs = FileSystem::Default();
138 std::string dir;
139 IOOptions opt;
140 CHECK_OK(fs->GetTestDirectory(opt, &dir, nullptr));
141 sstfile = dir + "/SstFileWriterFuzzer.sst";
142 }
143
144 Options options;
145 EnvOptions env_options(options);
146 ImmutableCFOptions cf_ioptions(options);
147
148 // Generate sst file.
149 SstFileWriter writer(env_options, options);
150 CHECK_OK(writer.Open(sstfile));
151 for (const DBOperation& op : input.operations()) {
152 switch (op.type()) {
153 case OpType::PUT: {
154 CHECK_OK(writer.Put(op.key(), op.value()));
155 break;
156 }
157 case OpType::MERGE: {
158 CHECK_OK(writer.Merge(op.key(), op.value()));
159 break;
160 }
161 case OpType::DELETE: {
162 CHECK_OK(writer.Delete(op.key()));
163 break;
164 }
165 case OpType::DELETE_RANGE: {
166 CHECK_OK(writer.DeleteRange(op.key(), op.value()));
167 break;
168 }
169 default: {
170 std::cerr << "Unsupported operation" << static_cast<int>(op.type())
171 << std::endl;
172 abort();
173 }
174 }
175 }
176 ExternalSstFileInfo info;
177 CHECK_OK(writer.Finish(&info));
178
179 // Iterate and verify key-value pairs.
180 std::unique_ptr<TableReader> table_reader(
181 ::NewTableReader(sstfile, options, env_options, cf_ioptions));
182 ReadOptions roptions;
183 CHECK_OK(table_reader->VerifyChecksum(roptions,
184 TableReaderCaller::kUncategorized));
185 std::unique_ptr<InternalIterator> it(
186 table_reader->NewIterator(roptions, /*prefix_extractor=*/nullptr,
187 /*arena=*/nullptr, /*skip_filters=*/true,
188 TableReaderCaller::kUncategorized));
189 it->SeekToFirst();
190 for (const DBOperation& op : input.operations()) {
191 if (op.type() == OpType::DELETE_RANGE) {
192 // InternalIterator cannot iterate over DELETE_RANGE entries.
193 continue;
194 }
195 CHECK_TRUE(it->Valid());
196 ParsedInternalKey ikey;
197 CHECK_OK(ParseInternalKey(it->key(), &ikey, /*log_err_key=*/true));
198 CHECK_EQ(ikey.user_key.ToString(), op.key());
199 CHECK_EQ(ikey.sequence, 0);
200 CHECK_EQ(ikey.type, ToValueType(op.type()));
201 if (op.type() != OpType::DELETE) {
202 CHECK_EQ(op.value(), it->value().ToString());
203 }
204 it->Next();
205 }
206 CHECK_TRUE(!it->Valid());
207
208 // Delete sst file.
209 remove(sstfile.c_str());
210}
