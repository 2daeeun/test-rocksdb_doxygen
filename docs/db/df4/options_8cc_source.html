483//
484// The goal of this method is to create a configuration that
485// allows an application to write all files into L0 and
486// then do a single compaction to output all files into L1.
487Options*
488Options::PrepareForBulkLoad()
489{
490 // never slowdown ingest.
491 level0_file_num_compaction_trigger = (1<<30);
492 level0_slowdown_writes_trigger = (1<<30);
493 level0_stop_writes_trigger = (1<<30);
494 soft_pending_compaction_bytes_limit = 0;
495 hard_pending_compaction_bytes_limit = 0;
496
497 // no auto compactions please. The application should issue a
498 // manual compaction after all data is loaded into L0.
499 disable_auto_compactions = true;
500 // A manual compaction run should pick all files in L0 in
501 // a single compaction run.
502 max_compaction_bytes = (static_cast<uint64_t>(1) << 60);
503
504 // It is better to have only 2 levels, otherwise a manual
505 // compaction would compact at every possible level, thereby
506 // increasing the total time needed for compactions.
507 num_levels = 2;
508
509 // Need to allow more write buffers to allow more parallism
510 // of flushes.
511 max_write_buffer_number = 6;
512 min_write_buffer_number_to_merge = 1;
513
514 // When compaction is disabled, more parallel flush threads can
515 // help with write throughput.
516 max_background_flushes = 4;
517
518 // Prevent a memtable flush to automatically promote files
519 // to L1. This is helpful so that all files that are
520 // input to the manual compaction are all at L0.
521 max_background_compactions = 2;
522
523 // The compaction would create large files in L1.

648ColumnFamilyOptions* ColumnFamilyOptions::OptimizeLevelStyleCompaction(
649 uint64_t memtable_memory_budget) {
650 write_buffer_size = static_cast<size_t>(memtable_memory_budget / 4);
651 // merge two memtables when flushing to L0
652 min_write_buffer_number_to_merge = 2;
653 // this means we'll use 50% extra memory in the worst case, but will reduce
654 // write stalls.
655 max_write_buffer_number = 6;
656 // start flushing L0->L1 as soon as possible. each file on level0 is
657 // (memtable_memory_budget / 2). This will flush level 0 when it's bigger than
658 // memtable_memory_budget.
659 level0_file_num_compaction_trigger = 2;
660 // doesn't really matter much, but we don't want to create too many files
661 target_file_size_base = memtable_memory_budget / 8;
662 // make Level1 size equal to Level0 size, so that L0->L1 compactions are fast
663 max_bytes_for_level_base = memtable_memory_budget;
664
665 // level style compaction
666 compaction_style = kCompactionStyleLevel;
667
668 // only compress levels >= 2
669 compression_per_level.resize(num_levels);
670 for (int i = 0; i < num_levels; ++i) {
671 if (i < 2) {
672 compression_per_level[i] = kNoCompression;
673 } else {
674 compression_per_level[i] =
675 LZ4_Supported()
676 ? kLZ4Compression
677 : (Snappy_Supported() ? kSnappyCompression : kNoCompression);
678 }
679 }
680 return this;
681}
