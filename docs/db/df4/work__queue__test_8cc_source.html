34struct Popper {
35 WorkQueue<int>* queue;
36 int* results;
37 std::mutex* mutex;
38
39 void operator()() {
40 int result;
41 while (queue->pop(result)) {
42 std::lock_guard<std::mutex> lock(*mutex);
43 results[result] = result;
44 }
45 }
46};

48TEST(WorkQueue, SingleThreaded) {
49 WorkQueue<int> queue;
50 int result;
51
52 queue.push(5);
53 EXPECT_TRUE(queue.pop(result));
54 EXPECT_EQ(5, result);
55
56 queue.push(1);

58 EXPECT_TRUE(queue.pop(result));
59 EXPECT_EQ(1, result);

64 queue.push(2);
65 queue.finish();
66 EXPECT_TRUE(queue.pop(result));
67 EXPECT_EQ(1, result);
68 EXPECT_TRUE(queue.pop(result));
69 EXPECT_EQ(2, result);
70 EXPECT_FALSE(queue.pop(result));
71

73}
