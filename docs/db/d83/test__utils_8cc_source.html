16std::shared_ptr<ColumnBase> CreateTestColumn(int8_t mask, int8_t index,
17 int64_t timestamp) {
18 if ((mask & ColumnTypeMask::DELETION_MASK) != 0) {
19 return std::shared_ptr<Tombstone>(
20 new Tombstone(mask, index, ToSeconds(timestamp), timestamp));
21 } else if ((mask & ColumnTypeMask::EXPIRATION_MASK) != 0) {
22 return std::shared_ptr<ExpiringColumn>(new ExpiringColumn(
23 mask, index, timestamp, sizeof(kExpiringData), kExpiringData, kTtl));
24 } else {
25 return std::shared_ptr<Column>(
26 new Column(mask, index, timestamp, sizeof(kData), kData));
27 }
28}

36RowValue CreateTestRowValue(
37 std::vector<std::tuple<int8_t, int8_t, int64_t>> column_specs) {
38 std::vector<std::shared_ptr<ColumnBase>> columns;
39 int64_t last_modified_time = 0;
40 for (auto spec : column_specs) {
41 auto c = CreateTestColumn(std::get<0>(spec), std::get<1>(spec),
42 std::get<2>(spec));
43 last_modified_time = std::max(last_modified_time, c->Timestamp());
44 columns.push_back(std::move(c));
45 }
46 return RowValue(std::move(columns), last_modified_time);
47}
