136inline void UpdateResult(struct io_uring_cqe* cqe, const std::string& file_name,
137 size_t len, size_t iov_len, bool async_read,
138 bool use_direct_io, size_t alignment,
139 size_t& finished_len, FSReadRequest* req,
140 size_t& bytes_read, bool& read_again) {
141 read_again = false;
142 if (cqe->res < 0) {
143 req->result = Slice(req->scratch, 0);
144 req->status = IOError("Req failed", file_name, cqe->res);
145 } else {
146 bytes_read = static_cast<size_t>(cqe->res);
147 TEST_SYNC_POINT_CALLBACK("UpdateResults::io_uring_result", &bytes_read);
148 if (bytes_read == iov_len) {
149 req->result = Slice(req->scratch, req->len);
150 req->status = IOStatus::OK();
151 } else if (bytes_read == 0) {
153 // comment
154 // https://github.com/facebook/rocksdb/pull/6441#issuecomment-589843435
155 // Fall back to pread in this case.
156 if (use_direct_io && !IsSectorAligned(finished_len, alignment)) {
157 // Bytes reads don't fill sectors. Should only happen at the end
158 // of the file.
159 req->result = Slice(req->scratch, finished_len);
160 req->status = IOStatus::OK();
161 } else {
162 if (async_read) {
163 // No bytes read. It can means EOF. In case of partial results, it's
164 // caller responsibility to call read/readasync again.
165 req->result = Slice(req->scratch, 0);
166 req->status = IOStatus::OK();
167 } else {
168 read_again = true;
169 }
170 }
171 } else if (bytes_read < iov_len) {
172 assert(bytes_read > 0);
173 if (async_read) {
174 req->result = Slice(req->scratch, bytes_read);
175 req->status = IOStatus::OK();
176 } else {
177 assert(bytes_read + finished_len < len);
178 finished_len += bytes_read;
179 }
180 } else {
181 req->result = Slice(req->scratch, 0);
182 req->status = IOError("Req returned more bytes than requested", file_name,
183 cqe->res);
184 }
185 }
186#ifdef NDEBUG
187 (void)len;
188#endif
189}

193// Files under a specific directory have the same logical block size.
194// This class caches the logical block size for the specified directories to
195// save the CPU cost of computing the size.
196// Safe for concurrent access from multiple threads without any external
197// synchronization.
198class LogicalBlockSizeCache {
199 public:
200 LogicalBlockSizeCache(
201 std::function<size_t(int)> get_logical_block_size_of_fd =
202 PosixHelper::GetLogicalBlockSizeOfFd,
203 std::function<Status(const std::string&, size_t*)>
204 get_logical_block_size_of_directory =
205 PosixHelper::GetLogicalBlockSizeOfDirectory)
206 : get_logical_block_size_of_fd_(get_logical_block_size_of_fd),
207 get_logical_block_size_of_directory_(
208 get_logical_block_size_of_directory) {}
209
210 // Takes the following actions:
211 // 1. Increases reference count of the directories;
212 // 2. If the directory's logical block size is not cached,
213 // compute the buffer size and cache the result.
214 Status RefAndCacheLogicalBlockSize(
215 const std::vector<std::string>& directories);
216
217 // Takes the following actions:
218 // 1. Decreases reference count of the directories;
219 // 2. If the reference count of a directory reaches 0, remove the directory
220 // from the cache.
221 void UnrefAndTryRemoveCachedLogicalBlockSize(
222 const std::vector<std::string>& directories);
223
224 // Returns the logical block size for the file.
225 //
226 // If the file is under a cached directory, return the cached size.
227 // Otherwise, the size is computed.
228 size_t GetLogicalBlockSize(const std::string& fname, int fd);
229
230 int GetRefCount(const std::string& dir) {
231 ReadLock lock(&cache_mutex_);
232 auto it = cache_.find(dir);
233 if (it == cache_.end()) {
234 return 0;
235 }
236 return it->second.ref;
237 }
238
239 size_t Size() const { return cache_.size(); }
240
241 bool Contains(const std::string& dir) {
242 ReadLock lock(&cache_mutex_);
243 return cache_.find(dir) != cache_.end();
244 }
245
246 private:
247 struct CacheValue {
248 CacheValue() : size(0), ref(0) {}
249
250 // Logical block size of the directory.
251 size_t size;
252 // Reference count of the directory.
253 int ref;
254 };
255
256 std::function<size_t(int)> get_logical_block_size_of_fd_;
257 std::function<Status(const std::string&, size_t*)>
258 get_logical_block_size_of_directory_;
259
260 std::map<std::string, CacheValue> cache_;
261 port::RWMutex cache_mutex_;
262};
