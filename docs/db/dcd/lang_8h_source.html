42// Coding guidelines say to avoid static objects with non-trivial destructors,
43// because it's easy to cause trouble (UB) in static destruction. This
44// macro makes it easier to define static objects that are normally never
45// destructed, except are destructed when running under ASAN. This should
46// avoid unexpected, unnecessary destruction behavior in production.
47// Note that constructor arguments can be provided as in
48// STATIC_AVOID_DESTRUCTION(Foo, foo)(arg1, arg2);
49#ifdef MUST_FREE_HEAP_ALLOCATIONS
50#define STATIC_AVOID_DESTRUCTION(Type, name) static Type name
51constexpr bool kMustFreeHeapAllocations = true;
52#else
53#define STATIC_AVOID_DESTRUCTION(Type, name) static Type& name = *new Type
54constexpr bool kMustFreeHeapAllocations = false;
55#endif
