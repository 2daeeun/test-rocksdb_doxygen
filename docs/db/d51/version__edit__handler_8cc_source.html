24void VersionEditHandlerBase::Iterate(log::Reader& reader,
25 Status* log_read_status) {
26 Slice record;
27 std::string scratch;
28 assert(log_read_status);
29 assert(log_read_status->ok());
30
31 [[maybe_unused]] size_t recovered_edits = 0;
32 Status s = Initialize();
33 while (reader.LastRecordEnd() < max_manifest_read_size_ && s.ok() &&
34 reader.ReadRecord(&record, &scratch) && log_read_status->ok()) {
35 VersionEdit edit;
36 s = edit.DecodeFrom(record);
37 if (s.ok()) {
38 s = read_buffer_.AddEdit(&edit);
39 }
40 if (s.ok()) {
41 ColumnFamilyData* cfd = nullptr;
42 if (edit.IsInAtomicGroup()) {
43 if (read_buffer_.IsFull()) {
44 s = OnAtomicGroupReplayBegin();
45 for (size_t i = 0; s.ok() && i < read_buffer_.replay_buffer().size();
46 i++) {
47 auto& e = read_buffer_.replay_buffer()[i];
48 s = ApplyVersionEdit(e, &cfd);
49 if (s.ok()) {
50 recovered_edits++;
51 }
52 }
53 if (s.ok()) {
54 read_buffer_.Clear();
55 s = OnAtomicGroupReplayEnd();
56 }
57 }
58 } else {
59 s = ApplyVersionEdit(edit, &cfd);
60 if (s.ok()) {
61 recovered_edits++;
62 }
63 }
64 }
65 }
66 if (s.ok() && !log_read_status->ok()) {
67 s = *log_read_status;
68 }
69
70 CheckIterationResult(reader, &s);
71
72 if (!s.ok()) {
73 if (s.IsCorruption()) {
74 // when we find a Corruption error, something is
75 // wrong with the underlying file. in this case we
76 // want to report the filename, so in here we append
77 // the filename to the Corruption message
78 assert(reader.file());
79
80 // build a new error message
81 std::stringstream message;
82 // append previous dynamic state message
83 const char* state = s.getState();
84 if (state != nullptr) {
85 message << state;
86 message << ' ';
87 }
88 // append the filename to the corruption message
89 message << " The file " << reader.file()->file_name()
90 << " may be corrupted.";
91 // overwrite the status with the extended status
92 s = Status(s.code(), s.subcode(), s.severity(), message.str());
93 }
94 status_ = s;
95 }
96 TEST_SYNC_POINT_CALLBACK("VersionEditHandlerBase::Iterate:Finish",
97 &recovered_edits);
98}

100Status ListColumnFamiliesHandler::ApplyVersionEdit(
101 VersionEdit& edit, ColumnFamilyData** /*unused*/) {
102 Status s;
103 uint32_t cf_id = edit.GetColumnFamily();
104 if (edit.IsColumnFamilyAdd()) {
105 if (column_family_names_.find(cf_id) != column_family_names_.end()) {
106 s = Status::Corruption("Manifest adding the same column family twice");
107 } else {
108 column_family_names_.insert({cf_id, edit.GetColumnFamilyName()});
109 }
110 } else if (edit.IsColumnFamilyDrop()) {
111 if (column_family_names_.find(cf_id) == column_family_names_.end()) {
112 s = Status::Corruption("Manifest - dropping non-existing column family");
113 } else {
114 column_family_names_.erase(cf_id);
115 }
116 }
117 return s;
118}

120Status FileChecksumRetriever::ApplyVersionEdit(VersionEdit& edit,
121 ColumnFamilyData** /*unused*/) {
122 for (const auto& deleted_file : edit.GetDeletedFiles()) {
123 Status s = file_checksum_list_.RemoveOneFileChecksum(deleted_file.second);
124 if (!s.ok()) {
125 return s;
126 }
127 }
128 for (const auto& new_file : edit.GetNewFiles()) {
129 Status s = file_checksum_list_.InsertOneFileChecksum(
130 new_file.second.fd.GetNumber(), new_file.second.file_checksum,
131 new_file.second.file_checksum_func_name);
132 if (!s.ok()) {
133 return s;
134 }
135 }
136 for (const auto& new_blob_file : edit.GetBlobFileAdditions()) {
137 std::string checksum_value = new_blob_file.GetChecksumValue();
138 std::string checksum_method = new_blob_file.GetChecksumMethod();
139 assert(checksum_value.empty() == checksum_method.empty());
140 if (checksum_method.empty()) {
141 checksum_value = kUnknownFileChecksum;
142 checksum_method = kUnknownFileChecksumFuncName;
143 }
144 Status s = file_checksum_list_.InsertOneFileChecksum(
145 new_blob_file.GetBlobFileNumber(), checksum_value, checksum_method);
146 if (!s.ok()) {
147 return s;
148 }
149 }
150 return Status::OK();
151}

195 }
196 }
197 return s;

200Status VersionEditHandler::ApplyVersionEdit(VersionEdit& edit,
201 ColumnFamilyData** cfd) {
202 Status s;
203 if (edit.IsColumnFamilyAdd()) {
204 s = OnColumnFamilyAdd(edit, cfd);
205 } else if (edit.IsColumnFamilyDrop()) {
206 s = OnColumnFamilyDrop(edit, cfd);
207 } else if (edit.IsWalAddition()) {
208 s = OnWalAddition(edit);
209 } else if (edit.IsWalDeletion()) {
210 s = OnWalDeletion(edit);
211 } else {
212 s = OnNonCfOperation(edit, cfd);
213 }
214 if (s.ok()) {
215 assert(cfd != nullptr);
216 s = ExtractInfoFromVersionEdit(*cfd, edit);
217 }
218 return s;
219}

221Status VersionEditHandler::OnColumnFamilyAdd(VersionEdit& edit,
222 ColumnFamilyData** cfd) {
223 bool do_not_open_cf = false;
224 bool cf_in_builders = false;
225 CheckColumnFamilyId(edit, &do_not_open_cf, &cf_in_builders);
226
227 assert(cfd != nullptr);
228 *cfd = nullptr;
229 const std::string& cf_name = edit.GetColumnFamilyName();
230 Status s;
231 if (cf_in_builders || do_not_open_cf) {
232 s = Status::Corruption("MANIFEST adding the same column family twice: " +
233 cf_name);
234 }
235 if (s.ok()) {
236 auto cf_options = name_to_options_.find(cf_name);
237 // implicitly add persistent_stats column family without requiring user
238 // to specify
239 ColumnFamilyData* tmp_cfd = nullptr;
240 bool is_persistent_stats_column_family =
241 cf_name.compare(kPersistentStatsColumnFamilyName) == 0;
242 if (cf_options == name_to_options_.end() &&
243 !is_persistent_stats_column_family) {
244 do_not_open_column_families_.emplace(edit.GetColumnFamily(), cf_name);
245 } else {
246 if (is_persistent_stats_column_family) {
247 ColumnFamilyOptions cfo;
248 OptimizeForPersistentStats(&cfo);
249 tmp_cfd = CreateCfAndInit(cfo, edit);
250 } else {
251 tmp_cfd = CreateCfAndInit(cf_options->second, edit);
252 }
253 *cfd = tmp_cfd;
254 }
255 }
256 return s;
257}

259Status VersionEditHandler::OnColumnFamilyDrop(VersionEdit& edit,
260 ColumnFamilyData** cfd) {
261 bool do_not_open_cf = false;
262 bool cf_in_builders = false;
263 CheckColumnFamilyId(edit, &do_not_open_cf, &cf_in_builders);
264
265 assert(cfd != nullptr);
266 *cfd = nullptr;
267 ColumnFamilyData* tmp_cfd = nullptr;
268 Status s;
269 if (cf_in_builders) {
270 tmp_cfd = DestroyCfAndCleanup(edit);
271 } else if (do_not_open_cf) {
272 do_not_open_column_families_.erase(edit.GetColumnFamily());
273 } else {
274 s = Status::Corruption("MANIFEST - dropping non-existing column family");
275 }
276 *cfd = tmp_cfd;
277 return s;
278}

291Status VersionEditHandler::OnNonCfOperation(VersionEdit& edit,
292 ColumnFamilyData** cfd) {
293 bool do_not_open_cf = false;
294 bool cf_in_builders = false;
295 CheckColumnFamilyId(edit, &do_not_open_cf, &cf_in_builders);
296
297 assert(cfd != nullptr);
298 *cfd = nullptr;
299 Status s;
300 if (!do_not_open_cf) {
301 if (!cf_in_builders) {
302 s = Status::Corruption(
303 "MANIFEST record referencing unknown column family");
304 }
305 ColumnFamilyData* tmp_cfd = nullptr;
306 if (s.ok()) {
307 tmp_cfd = version_set_->GetColumnFamilySet()->GetColumnFamily(
308 edit.GetColumnFamily());
309 assert(tmp_cfd != nullptr);
310 // It's important to handle file boundaries before `MaybeCreateVersion`
311 // because `VersionEditHandlerPointInTime::MaybeCreateVersion` does
312 // `FileMetaData` verification that involves the file boundaries.
313 // All `VersionEditHandlerBase` subclasses that need to deal with
314 // `FileMetaData` for new files are also subclasses of
315 // `VersionEditHandler`, so it's sufficient to do the file boundaries
316 // handling in this method.
317 s = MaybeHandleFileBoundariesForNewFiles(edit, tmp_cfd);
318 if (!s.ok()) {
319 return s;
320 }
321 s = MaybeCreateVersionBeforeApplyEdit(edit, tmp_cfd,
322 /*force_create_version=*/false);
323 }
324 *cfd = tmp_cfd;
325 }
326 return s;
327}

329void VersionEditHandler::CheckColumnFamilyId(const VersionEdit& edit,
330 bool* do_not_open_cf,
331 bool* cf_in_builders) const {
332 assert(do_not_open_cf != nullptr);
333 assert(cf_in_builders != nullptr);
334 // Not found means that user didn't supply that column
335 // family option AND we encountered column family add
336 // record. Once we encounter column family drop record,
337 // we will delete the column family from
338 // do_not_open_column_families_.
339 uint32_t cf_id = edit.GetColumnFamily();
340 bool in_do_not_open = do_not_open_column_families_.find(cf_id) !=
341 do_not_open_column_families_.end();
342 // in builders means that user supplied that column family
343 // option AND that we encountered column family add record
344 bool in_builders = builders_.find(cf_id) != builders_.end();
345 // They cannot both be true
346 assert(!(in_do_not_open && in_builders));
347 *do_not_open_cf = in_do_not_open;
348 *cf_in_builders = in_builders;
349}

351void VersionEditHandler::CheckIterationResult(const log::Reader& reader,
352 Status* s) {
353 assert(s != nullptr);
354 if (!s->ok()) {
355 // Do nothing here.
356 } else if (!version_edit_params_.HasLogNumber() ||
357 !version_edit_params_.HasNextFile() ||
358 !version_edit_params_.HasLastSequence()) {
359 std::string msg("no ");
360 if (!version_edit_params_.HasLogNumber()) {
361 msg.append("log_file_number, ");
362 }
363 if (!version_edit_params_.HasNextFile()) {
364 msg.append("next_file_number, ");
365 }
366 if (!version_edit_params_.HasLastSequence()) {
367 msg.append("last_sequence, ");
368 }
369 msg = msg.substr(0, msg.size() - 2);
370 msg.append(" entry in MANIFEST");
371 *s = Status::Corruption(msg);
372 }
373 // There were some column families in the MANIFEST that weren't specified
374 // in the argument. This is OK in read_only mode
375 if (s->ok() && MustOpenAllColumnFamilies() &&
376 !do_not_open_column_families_.empty()) {
377 std::string msg;
378 for (const auto& cf : do_not_open_column_families_) {
379 msg.append(", ");
380 msg.append(cf.second);
381 }
382 msg = msg.substr(2);
383 *s = Status::InvalidArgument("Column families not opened: " + msg);
384 }
385 if (s->ok()) {
386 version_set_->GetColumnFamilySet()->UpdateMaxColumnFamily(
387 version_edit_params_.GetMaxColumnFamily());
388 version_set_->MarkMinLogNumberToKeep(
389 version_edit_params_.GetMinLogNumberToKeep());
390 version_set_->MarkFileNumberUsed(version_edit_params_.GetPrevLogNumber());
391 version_set_->MarkFileNumberUsed(version_edit_params_.GetLogNumber());
392 for (auto* cfd : *(version_set_->GetColumnFamilySet())) {
393 if (cfd->IsDropped()) {
394 continue;
395 }
396 auto builder_iter = builders_.find(cfd->GetID());
397 assert(builder_iter != builders_.end());
398 auto* builder = builder_iter->second->version_builder();
399 if (!builder->CheckConsistencyForNumLevels()) {
400 *s = Status::InvalidArgument(
401 "db has more levels than options.num_levels");
402 break;
403 }
404 }
405 }
406 if (s->ok()) {
407 for (auto* cfd : *(version_set_->GetColumnFamilySet())) {
408 if (cfd->IsDropped()) {
409 continue;
410 }
411 if (read_only_) {
412 cfd->table_cache()->SetTablesAreImmortal();
413 }
414 *s = LoadTables(cfd, /*prefetch_index_and_filter_in_cache=*/false,
415 /*is_initial_load=*/true);
416 if (!s->ok()) {
417 // If s is IOError::PathNotFound, then we mark the db as corrupted.
418 if (s->IsPathNotFound()) {
419 *s = Status::Corruption("Corruption: " + s->ToString());
420 }
421 break;
422 }
423 }
424 }
425
426 if (s->ok()) {
427 for (auto* cfd : *(version_set_->column_family_set_)) {
428 if (cfd->IsDropped()) {
429 continue;
430 }
431 assert(cfd->initialized());
432 VersionEdit edit;
433 *s = MaybeCreateVersionBeforeApplyEdit(edit, cfd,
434 /*force_create_version=*/true);
435 if (!s->ok()) {
436 break;
437 }
438 }
439 }
440 if (s->ok()) {
441 version_set_->manifest_file_size_ = reader.GetReadOffset();
442 assert(version_set_->manifest_file_size_ > 0);
443 version_set_->next_file_number_.store(version_edit_params_.GetNextFile() +
444 1);
445 SequenceNumber last_seq = version_edit_params_.GetLastSequence();
446 assert(last_seq != kMaxSequenceNumber);
447 if (last_seq != kMaxSequenceNumber &&
448 last_seq > version_set_->last_allocated_sequence_.load()) {
449 version_set_->last_allocated_sequence_.store(last_seq);
450 }
451 if (last_seq != kMaxSequenceNumber &&
452 last_seq > version_set_->last_published_sequence_.load()) {
453 version_set_->last_published_sequence_.store(last_seq);
454 }
455 if (last_seq != kMaxSequenceNumber &&
456 last_seq > version_set_->last_sequence_.load()) {
457 version_set_->last_sequence_.store(last_seq);
458 }
459 if (last_seq != kMaxSequenceNumber &&
460 last_seq > version_set_->descriptor_last_sequence_) {
461 // This is the maximum last sequence of all `VersionEdit`s iterated. It
462 // may be greater than the maximum `largest_seqno` of all files in case
463 // the newest data referred to by the MANIFEST has been dropped or had its
464 // sequence number zeroed through compaction.
465 version_set_->descriptor_last_sequence_ = last_seq;
466 }
467 version_set_->prev_log_number_ = version_edit_params_.GetPrevLogNumber();
468 }
469}

472 const ColumnFamilyOptions& cf_options, const VersionEdit& edit) {

486 const VersionEdit& edit) {
487 uint32_t cf_id = edit.GetColumnFamily();

496 ret = nullptr;

500Status VersionEditHandler::MaybeCreateVersionBeforeApplyEdit(
501 const VersionEdit& edit, ColumnFamilyData* cfd, bool force_create_version) {
502 assert(cfd->initialized());
503 Status s;
504 auto builder_iter = builders_.find(cfd->GetID());
505 assert(builder_iter != builders_.end());
506 auto* builder = builder_iter->second->version_builder();
507 if (force_create_version) {
508 auto* v = new Version(cfd, version_set_, version_set_->file_options_,
509 *cfd->GetLatestMutableCFOptions(), io_tracer_,
510 version_set_->current_version_number_++,
511 epoch_number_requirement_);
512 s = builder->SaveTo(v->storage_info());
513 if (s.ok()) {
514 // Install new version
515 v->PrepareAppend(
516 *cfd->GetLatestMutableCFOptions(), read_options_,
517 !(version_set_->db_options_->skip_stats_update_on_db_open));
518 version_set_->AppendVersion(cfd, v);
519 } else {
520 delete v;
521 }
522 }
523 s = builder->Apply(&edit);
524 return s;
525}

528 bool prefetch_index_and_filter_in_cache,
529 bool is_initial_load) {
530 bool skip_load_table_files = skip_load_table_files_;
531 TEST_SYNC_POINT_CALLBACK(

535 return Status::OK();
536 }
537 assert(cfd != nullptr);
538 assert(!cfd->IsDropped());
539 auto builder_iter = builders_.find(cfd->GetID());
540 assert(builder_iter != builders_.end());
541 assert(builder_iter->second != nullptr);

543 assert(builder);

545 Status s = builder->LoadTableHandlers(

547 version_set_->db_options_->max_file_opening_threads,

551 s = Status::OK();
552 }
553 if (!s.ok() && !version_set_->db_options_->paranoid_checks) {
554 s = Status::OK();
555 }
556 return s;

559Status VersionEditHandler::ExtractInfoFromVersionEdit(ColumnFamilyData* cfd,
560 const VersionEdit& edit) {
561 Status s;
562 if (edit.HasDbId()) {
563 version_set_->db_id_ = edit.GetDbId();
564 version_edit_params_.SetDBId(edit.GetDbId());
565 }
566 if (cfd != nullptr) {
567 if (edit.HasLogNumber()) {
568 if (cfd->GetLogNumber() > edit.GetLogNumber()) {
569 ROCKS_LOG_WARN(
570 version_set_->db_options()->info_log,
571 "MANIFEST corruption detected, but ignored - Log numbers in "
572 "records NOT monotonically increasing");
573 } else {
574 cfd->SetLogNumber(edit.GetLogNumber());
575 version_edit_params_.SetLogNumber(edit.GetLogNumber());
576 }
577 }
578 if (edit.HasComparatorName()) {
579 bool mark_sst_files_has_no_udt = false;
580 // If `persist_user_defined_timestamps` flag is recorded in manifest, it
581 // is guaranteed to be in the same VersionEdit as comparator. Otherwise,
582 // it's not recorded and it should have default value true.
583 s = ValidateUserDefinedTimestampsOptions(
584 cfd->user_comparator(), edit.GetComparatorName(),
585 cfd->ioptions()->persist_user_defined_timestamps,
586 edit.GetPersistUserDefinedTimestamps(), &mark_sst_files_has_no_udt);
587 if (!s.ok() && cf_to_cmp_names_) {
588 cf_to_cmp_names_->emplace(cfd->GetID(), edit.GetComparatorName());
589 }
590 if (mark_sst_files_has_no_udt) {
591 cfds_to_mark_no_udt_.insert(cfd->GetID());
592 }
593 }
594 if (edit.HasFullHistoryTsLow()) {
595 const std::string& new_ts = edit.GetFullHistoryTsLow();
596 cfd->SetFullHistoryTsLow(new_ts);
597 }
598 }
599
600 if (s.ok()) {
601 if (edit.HasPrevLogNumber()) {
602 version_edit_params_.SetPrevLogNumber(edit.GetPrevLogNumber());
603 }
604 if (edit.HasNextFile()) {
605 version_edit_params_.SetNextFile(edit.GetNextFile());
606 }
607 if (edit.HasMaxColumnFamily()) {
608 version_edit_params_.SetMaxColumnFamily(edit.GetMaxColumnFamily());
609 }
610 if (edit.HasMinLogNumberToKeep()) {
611 version_edit_params_.SetMinLogNumberToKeep(
612 std::max(version_edit_params_.GetMinLogNumberToKeep(),
613 edit.GetMinLogNumberToKeep()));
614 }
615 if (edit.HasLastSequence()) {
616 // `VersionEdit::last_sequence_`s are assumed to be non-decreasing. This
617 // is legacy behavior that cannot change without breaking downgrade
618 // compatibility.
619 assert(!version_edit_params_.HasLastSequence() ||
620 version_edit_params_.GetLastSequence() <= edit.GetLastSequence());
621 version_edit_params_.SetLastSequence(edit.GetLastSequence());
622 }
623 if (!version_edit_params_.HasPrevLogNumber()) {
624 version_edit_params_.SetPrevLogNumber(0);
625 }
626 }
627 return s;
628}

630Status VersionEditHandler::MaybeHandleFileBoundariesForNewFiles(
631 VersionEdit& edit, const ColumnFamilyData* cfd) {
632 if (edit.GetNewFiles().empty()) {
633 return Status::OK();
634 }
635 auto ucmp = cfd->user_comparator();
636 assert(ucmp);
637 size_t ts_sz = ucmp->timestamp_size();
638 if (ts_sz == 0) {
639 return Status::OK();
640 }
641
642 VersionEdit::NewFiles& new_files = edit.GetMutableNewFiles();
643 assert(!new_files.empty());
644 // If true, enabling user-defined timestamp is detected for this column
645 // family. All its existing SST files need to have the file boundaries handled
646 // and their `persist_user_defined_timestamps` flag set to false regardless of
647 // its existing value.
648 bool mark_existing_ssts_with_no_udt =
649 cfds_to_mark_no_udt_.find(cfd->GetID()) != cfds_to_mark_no_udt_.end();
650 bool file_boundaries_need_handling = false;
651 for (auto& new_file : new_files) {
652 FileMetaData& meta = new_file.second;
653 if (meta.user_defined_timestamps_persisted &&
654 !mark_existing_ssts_with_no_udt) {
655 // `FileMetaData.user_defined_timestamps_persisted` field is the value of
656 // the flag `AdvancedColumnFamilyOptions.persist_user_defined_timestamps`
657 // at the time when the SST file was created. As a result, all added SST
658 // files in one `VersionEdit` should have the same value for it.
659 if (file_boundaries_need_handling) {
660 return Status::Corruption(
661 "New files in one VersionEdit has different "
662 "user_defined_timestamps_persisted value.");
663 }
664 break;
665 }
666 file_boundaries_need_handling = true;
667 assert(!meta.user_defined_timestamps_persisted ||
668 mark_existing_ssts_with_no_udt);
669 if (mark_existing_ssts_with_no_udt) {
670 meta.user_defined_timestamps_persisted = false;
671 }
672 std::string smallest_buf;
673 std::string largest_buf;
674 Slice largest_slice = meta.largest.Encode();
675 PadInternalKeyWithMinTimestamp(&smallest_buf, meta.smallest.Encode(),
676 ts_sz);
677 auto largest_footer = ExtractInternalKeyFooter(largest_slice);
678 if (largest_footer == kRangeTombstoneSentinel) {
679 PadInternalKeyWithMaxTimestamp(&largest_buf, largest_slice, ts_sz);
680 } else {
681 PadInternalKeyWithMinTimestamp(&largest_buf, largest_slice, ts_sz);
682 }
683 meta.smallest.DecodeFrom(smallest_buf);
684 meta.largest.DecodeFrom(largest_buf);
685 }
686 return Status::OK();
687}

710Status VersionEditHandlerPointInTime::OnAtomicGroupReplayBegin() {
711 if (in_atomic_group_) {
712 return Status::Corruption("unexpected AtomicGroup start");
713 }
714
715 // The AtomicGroup that is about to begin may block column families in a valid
716 // state from saving any more updates. So we should save any valid states
717 // before proceeding.
718 for (const auto& cfid_and_builder : builders_) {
719 ColumnFamilyData* cfd = version_set_->GetColumnFamilySet()->GetColumnFamily(
720 cfid_and_builder.first);
721 assert(!cfd->IsDropped());
722 assert(cfd->initialized());
723 VersionEdit edit;
724 Status s = MaybeCreateVersionBeforeApplyEdit(
725 edit, cfd, true /* force_create_version */);
726 if (!s.ok()) {
727 return s;
728 }
729 }
730
731 // An old AtomicGroup is incomplete. Throw away the versions that failed to
732 // complete it. They must not be used for completing the upcoming
733 // AtomicGroup since they are too old.
734 for (auto& cfid_and_version : atomic_update_versions_) {
735 delete cfid_and_version.second;
736 }
737
738 in_atomic_group_ = true;
739 // We lazily assume the column families that exist at this point are all
740 // involved in the AtomicGroup. Overestimating the scope of the AtomicGroup
741 // will sometimes cause less data to be recovered, which is fine for
742 // best-effort recovery.
743 atomic_update_versions_.clear();
744 for (const auto& cfid_and_builder : builders_) {
745 atomic_update_versions_[cfid_and_builder.first] = nullptr;
746 }
747 atomic_update_versions_missing_ = atomic_update_versions_.size();
748 return Status::OK();
749}
750
751Status VersionEditHandlerPointInTime::OnAtomicGroupReplayEnd() {
752 if (!in_atomic_group_) {
753 return Status::Corruption("unexpected AtomicGroup end");
754 }
755 in_atomic_group_ = false;
756
757 // The AtomicGroup must not have changed the column families. We don't support
758 // CF adds or drops in an AtomicGroup.
759 if (builders_.size() != atomic_update_versions_.size()) {
760 return Status::Corruption("unexpected CF change in AtomicGroup");
761 }
762 for (const auto& cfid_and_builder : builders_) {
763 if (atomic_update_versions_.find(cfid_and_builder.first) ==
764 atomic_update_versions_.end()) {
765 return Status::Corruption("unexpected CF add in AtomicGroup");
766 }
767 }
768 for (const auto& cfid_and_version : atomic_update_versions_) {
769 if (builders_.find(cfid_and_version.first) == builders_.end()) {
770 return Status::Corruption("unexpected CF drop in AtomicGroup");
771 }
772 }
773 return Status::OK();
774}

826Status VersionEditHandlerPointInTime::MaybeCreateVersionBeforeApplyEdit(
827 const VersionEdit& edit, ColumnFamilyData* cfd, bool force_create_version) {
828 TEST_SYNC_POINT(
829 "VersionEditHandlerPointInTime::MaybeCreateVersionBeforeApplyEdit:"
830 "Begin1");
831 TEST_SYNC_POINT(
832 "VersionEditHandlerPointInTime::MaybeCreateVersionBeforeApplyEdit:"
833 "Begin2");
834 assert(cfd != nullptr);
835 if (!force_create_version) {
836 assert(edit.GetColumnFamily() == cfd->GetID());
837 }
838
839 bool missing_info = !version_edit_params_.HasLogNumber() ||
840 !version_edit_params_.HasNextFile() ||
841 !version_edit_params_.HasLastSequence();
842
843 Status s;
844 auto builder_iter = builders_.find(cfd->GetID());
845 assert(builder_iter != builders_.end());
846 VersionBuilder* builder = builder_iter->second->version_builder();
847 const bool valid_pit_before_edit = builder->ValidVersionAvailable();
848 builder->CreateOrReplaceSavePoint();
849 s = builder->Apply(&edit);
850 const bool valid_pit_after_edit = builder->ValidVersionAvailable();
851
852 // A new version will be created if:
853 // 1) no error has occurred so far, and
854 // 2) log_number_, next_file_number_ and last_sequence_ are known, and
855 // 3) not in an AtomicGroup
856 // 4) any of the following:
857 // a) a valid Version is available before applying the edit
858 // and a valid Version is not available after the edit.
859 // b) a valid Version is available after the edit and the
860 // caller explicitly request that a new version be created.
861 if (s.ok() && !missing_info && !in_atomic_group_ &&
862 ((!valid_pit_after_edit && valid_pit_before_edit) ||
863 (valid_pit_after_edit && force_create_version))) {
864 const MutableCFOptions* cf_opts_ptr = cfd->GetLatestMutableCFOptions();
865 auto* version = new Version(cfd, version_set_, version_set_->file_options_,
866 *cf_opts_ptr, io_tracer_,
867 version_set_->current_version_number_++,
868 epoch_number_requirement_);
869 s = builder->LoadSavePointTableHandlers(
870 cfd->internal_stats(),
871 version_set_->db_options_->max_file_opening_threads, false, true,
872 *cf_opts_ptr, MaxFileSizeForL0MetaPin(*cf_opts_ptr), read_options_);
873 if (!s.ok()) {
874 delete version;
875 if (s.IsCorruption()) {
876 s = Status::OK();
877 }
878 return s;
879 }
880 s = builder->SaveSavePointTo(version->storage_info());
881 if (s.ok()) {
882 if (AtomicUpdateVersionsContains(cfd->GetID())) {
883 AtomicUpdateVersionsPut(version);
884 if (AtomicUpdateVersionsCompleted()) {
885 AtomicUpdateVersionsApply();
886 }
887 } else {
888 version->PrepareAppend(
889 *cfd->GetLatestMutableCFOptions(), read_options_,
890 !version_set_->db_options_->skip_stats_update_on_db_open);
891 auto v_iter = versions_.find(cfd->GetID());
892 if (v_iter != versions_.end()) {
893 delete v_iter->second;
894 v_iter->second = version;
895 } else {
896 versions_.emplace(cfd->GetID(), version);
897 }
898 }
899 } else {
900 delete version;
901 }
902 }
903
904 builder->ClearSavePoint();
905 return s;
906}
