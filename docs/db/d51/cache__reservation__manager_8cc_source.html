57template <CacheEntryRole R>
58Status CacheReservationManagerImpl<R>::UpdateCacheReservation(
59 std::size_t new_mem_used) {
60 memory_used_ = new_mem_used;
61 std::size_t cur_cache_allocated_size =
62 cache_allocated_size_.load(std::memory_order_relaxed);
63 if (new_mem_used == cur_cache_allocated_size) {
64 return Status::OK();
65 } else if (new_mem_used > cur_cache_allocated_size) {
66 Status s = IncreaseCacheReservation(new_mem_used);
67 return s;
68 } else {
69 // In delayed decrease mode, we don't decrease cache reservation
70 // untill the memory usage is less than 3/4 of what we reserve
71 // in the cache.
72 // We do this because
73 // (1) Dummy entry insertion is expensive in block cache
74 // (2) Delayed releasing previously inserted dummy entries can save such
75 // expensive dummy entry insertion on memory increase in the near future,
76 // which is likely to happen when the memory usage is greater than or equal
77 // to 3/4 of what we reserve
78 if (delayed_decrease_ && new_mem_used >= cur_cache_allocated_size / 4 * 3) {
79 return Status::OK();
80 } else {
81 Status s = DecreaseCacheReservation(new_mem_used);
82 return s;
83 }
84 }
85}

87template <CacheEntryRole R>
88Status CacheReservationManagerImpl<R>::MakeCacheReservation(
89 std::size_t incremental_memory_used,
90 std::unique_ptr<CacheReservationManager::CacheReservationHandle>* handle) {
91 assert(handle);
92 Status s =
93 UpdateCacheReservation(GetTotalMemoryUsed() + incremental_memory_used);
94 (*handle).reset(new CacheReservationManagerImpl::CacheReservationHandle(
95 incremental_memory_used,
96 std::enable_shared_from_this<
97 CacheReservationManagerImpl<R>>::shared_from_this()));
98 return s;
99}

101template <CacheEntryRole R>
102Status CacheReservationManagerImpl<R>::ReleaseCacheReservation(
103 std::size_t incremental_memory_used) {
104 assert(GetTotalMemoryUsed() >= incremental_memory_used);
105 std::size_t updated_total_mem_used =
106 GetTotalMemoryUsed() - incremental_memory_used;
107 Status s = UpdateCacheReservation(updated_total_mem_used);
108 return s;
109}

111template <CacheEntryRole R>
112Status CacheReservationManagerImpl<R>::IncreaseCacheReservation(
113 std::size_t new_mem_used) {
114 Status return_status = Status::OK();
115 while (new_mem_used > cache_allocated_size_.load(std::memory_order_relaxed)) {
116 Cache::Handle* handle = nullptr;
117 return_status = cache_.Insert(GetNextCacheKey(), kSizeDummyEntry, &handle);
118
119 if (return_status != Status::OK()) {
120 return return_status;
121 }
122
123 dummy_handles_.push_back(handle);
124 cache_allocated_size_ += kSizeDummyEntry;
125 }
126 return return_status;
127}

129template <CacheEntryRole R>
130Status CacheReservationManagerImpl<R>::DecreaseCacheReservation(
131 std::size_t new_mem_used) {
132 Status return_status = Status::OK();
133
134 // Decrease to the smallest multiple of kSizeDummyEntry that is greater than
135 // or equal to new_mem_used We do addition instead of new_mem_used <=
136 // cache_allocated_size_.load(std::memory_order_relaxed) - kSizeDummyEntry to
137 // avoid underflow of size_t when cache_allocated_size_ = 0
138 while (new_mem_used + kSizeDummyEntry <=
139 cache_allocated_size_.load(std::memory_order_relaxed)) {
140 assert(!dummy_handles_.empty());
141 auto* handle = dummy_handles_.back();
142 cache_.ReleaseAndEraseIfLastRef(handle);
143 dummy_handles_.pop_back();
144 cache_allocated_size_ -= kSizeDummyEntry;
145 }
146 return return_status;
147}

159template <CacheEntryRole R>
160Slice CacheReservationManagerImpl<R>::GetNextCacheKey() {
161 // Calling this function will have the side-effect of changing the
162 // underlying cache_key_ that is shared among other keys generated from this
163 // fucntion. Therefore please make sure the previous keys are saved/copied
164 // before calling this function.
165 cache_key_ = CacheKey::CreateUniqueForCacheLifetime(cache_.get());
166 return cache_key_.AsSlice();
167}
