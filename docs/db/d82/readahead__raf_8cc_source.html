22class ReadaheadRandomAccessFile : public FSRandomAccessFile {
23 public:
24 ReadaheadRandomAccessFile(std::unique_ptr<FSRandomAccessFile>&& file,
25 size_t readahead_size)
26 : file_(std::move(file)),
27 alignment_(file_->GetRequiredBufferAlignment()),
28 readahead_size_(Roundup(readahead_size, alignment_)),
29 buffer_(),
30 buffer_offset_(0) {
31 buffer_.Alignment(alignment_);
32 buffer_.AllocateNewBuffer(readahead_size_);
33 }
34
35 ReadaheadRandomAccessFile(const ReadaheadRandomAccessFile&) = delete;
36
37 ReadaheadRandomAccessFile& operator=(const ReadaheadRandomAccessFile&) =
38 delete;
39
40 IOStatus Read(uint64_t offset, size_t n, const IOOptions& options,
41 Slice* result, char* scratch,
42 IODebugContext* dbg) const override {
43 // Read-ahead only make sense if we have some slack left after reading
44 if (n + alignment_ >= readahead_size_) {
45 return file_->Read(offset, n, options, result, scratch, dbg);
46 }
47
48 std::unique_lock<std::mutex> lk(lock_);
49
50 size_t cached_len = 0;
51 // Check if there is a cache hit, meaning that [offset, offset + n) is
52 // either completely or partially in the buffer. If it's completely cached,
53 // including end of file case when offset + n is greater than EOF, then
54 // return.
55 if (TryReadFromCache(offset, n, &cached_len, scratch) &&
56 (cached_len == n || buffer_.CurrentSize() < readahead_size_)) {
57 // We read exactly what we needed, or we hit end of file - return.
58 *result = Slice(scratch, cached_len);
59 return IOStatus::OK();
60 }
61 size_t advanced_offset = static_cast<size_t>(offset + cached_len);
62 // In the case of cache hit advanced_offset is already aligned, means that
63 // chunk_offset equals to advanced_offset
64 size_t chunk_offset = TruncateToPageBoundary(alignment_, advanced_offset);
65
66 IOStatus s = ReadIntoBuffer(chunk_offset, readahead_size_, options, dbg);
67 if (s.ok()) {
68 // The data we need is now in cache, so we can safely read it
69 size_t remaining_len;
70 TryReadFromCache(advanced_offset, n - cached_len, &remaining_len,
71 scratch + cached_len);
72 *result = Slice(scratch, cached_len + remaining_len);
73 }
74 return s;
75 }
76
77 IOStatus Prefetch(uint64_t offset, size_t n, const IOOptions& options,
78 IODebugContext* dbg) override {
79 if (n < readahead_size_) {
80 // Don't allow smaller prefetches than the configured `readahead_size_`.
81 // `Read()` assumes a smaller prefetch buffer indicates EOF was reached.
82 return IOStatus::OK();
83 }
84
85 std::unique_lock<std::mutex> lk(lock_);
86
87 size_t offset_ = static_cast<size_t>(offset);
88 size_t prefetch_offset = TruncateToPageBoundary(alignment_, offset_);
89 if (prefetch_offset == buffer_offset_) {
90 return IOStatus::OK();
91 }
92 return ReadIntoBuffer(prefetch_offset,
93 Roundup(offset_ + n, alignment_) - prefetch_offset,
94 options, dbg);
95 }
96
97 size_t GetUniqueId(char* id, size_t max_size) const override {
98 return file_->GetUniqueId(id, max_size);
99 }
100
101 void Hint(AccessPattern pattern) override { file_->Hint(pattern); }
102
103 IOStatus InvalidateCache(size_t offset, size_t length) override {
104 std::unique_lock<std::mutex> lk(lock_);
105 buffer_.Clear();
106 return file_->InvalidateCache(offset, length);
107 }
108
109 bool use_direct_io() const override { return file_->use_direct_io(); }
110
111 private:
112 // Tries to read from buffer_ n bytes starting at offset. If anything was read
113 // from the cache, it sets cached_len to the number of bytes actually read,
114 // copies these number of bytes to scratch and returns true.
115 // If nothing was read sets cached_len to 0 and returns false.
116 bool TryReadFromCache(uint64_t offset, size_t n, size_t* cached_len,
117 char* scratch) const {
118 if (offset < buffer_offset_ ||
119 offset >= buffer_offset_ + buffer_.CurrentSize()) {
120 *cached_len = 0;
121 return false;
122 }
123 uint64_t offset_in_buffer = offset - buffer_offset_;
124 *cached_len = std::min(
125 buffer_.CurrentSize() - static_cast<size_t>(offset_in_buffer), n);
126 memcpy(scratch, buffer_.BufferStart() + offset_in_buffer, *cached_len);
127 return true;
128 }
129
130 // Reads into buffer_ the next n bytes from file_ starting at offset.
131 // Can actually read less if EOF was reached.
132 // Returns the status of the read operastion on the file.
133 IOStatus ReadIntoBuffer(uint64_t offset, size_t n, const IOOptions& options,
134 IODebugContext* dbg) const {
135 if (n > buffer_.Capacity()) {
136 n = buffer_.Capacity();
137 }
138 assert(IsFileSectorAligned(offset, alignment_));
139 assert(IsFileSectorAligned(n, alignment_));
140 Slice result;
141 IOStatus s =
142 file_->Read(offset, n, options, &result, buffer_.BufferStart(), dbg);
143 if (s.ok()) {
144 buffer_offset_ = offset;
145 buffer_.Size(result.size());
146 assert(result.size() == 0 || buffer_.BufferStart() == result.data());
147 }
148 return s;
149 }
150
151 const std::unique_ptr<FSRandomAccessFile> file_;
152 const size_t alignment_;
153 const size_t readahead_size_;
154
155 mutable std::mutex lock_;
156 // The buffer storing the prefetched data
157 mutable AlignedBuffer buffer_;
158 // The offset in file_, corresponding to data stored in buffer_
159 mutable uint64_t buffer_offset_;
160};
161} // namespace
