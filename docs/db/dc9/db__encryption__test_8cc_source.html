29TEST_F(DBEncryptionTest, CheckEncrypted) {
30 ASSERT_OK(Put("foo567", "v1.fetdq"));
31 ASSERT_OK(Put("bar123", "v2.dfgkjdfghsd"));
32 Close();
33
34 // Open all files and look for the values we've put in there.
35 // They should not be found if encrypted, otherwise
36 // they should be found.
37 std::vector<std::string> fileNames;
38 auto status = env_->GetChildren(dbname_, &fileNames);
39 ASSERT_OK(status);
40
41 Env* target = GetTargetEnv();
42 int hits = 0;
43 for (auto it = fileNames.begin(); it != fileNames.end(); ++it) {
44 if (*it == "LOCK") {
45 continue;
46 }
47 auto filePath = dbname_ + "/" + *it;
48 std::unique_ptr<SequentialFile> seqFile;
49 auto envOptions = EnvOptions(CurrentOptions());
50 status = target->NewSequentialFile(filePath, &seqFile, envOptions);
51 ASSERT_OK(status);
52
53 uint64_t fileSize;
54 status = target->GetFileSize(filePath, &fileSize);
55 ASSERT_OK(status);
56
57 std::string scratch;
58 scratch.reserve(fileSize);
59 Slice data;
60 status = seqFile->Read(fileSize, &data, (char*)scratch.data());
61 ASSERT_OK(status);
62
63 if (data.ToString().find("foo567") != std::string::npos) {
64 hits++;
65 // std::cout << "Hit in " << filePath << "\n";
66 }
67 if (data.ToString().find("v1.fetdq") != std::string::npos) {
68 hits++;
69 // std::cout << "Hit in " << filePath << "\n";
70 }
71 if (data.ToString().find("bar123") != std::string::npos) {
72 hits++;
73 // std::cout << "Hit in " << filePath << "\n";
74 }
75 if (data.ToString().find("v2.dfgkjdfghsd") != std::string::npos) {
76 hits++;
77 // std::cout << "Hit in " << filePath << "\n";
78 }
79 if (data.ToString().find("dfgk") != std::string::npos) {
80 hits++;
81 // std::cout << "Hit in " << filePath << "\n";
82 }
83 }
84 if (encrypted_env_) {
85 ASSERT_EQ(hits, 0);
86 } else {
87 ASSERT_GE(hits, 4);
88 }
89}

91TEST_F(DBEncryptionTest, ReadEmptyFile) {
92 auto defaultEnv = GetTargetEnv();
93
94 // create empty file for reading it back in later
95 auto envOptions = EnvOptions(CurrentOptions());
96 auto filePath = dbname_ + "/empty.empty";
97
98 Status status;
99 {
100 std::unique_ptr<WritableFile> writableFile;
101 status = defaultEnv->NewWritableFile(filePath, &writableFile, envOptions);
102 ASSERT_OK(status);
103 }
104
105 std::unique_ptr<SequentialFile> seqFile;
106 status = defaultEnv->NewSequentialFile(filePath, &seqFile, envOptions);
107 ASSERT_OK(status);
108
109 std::string scratch;
110 Slice data;
111 // reading back 16 bytes from the empty file shouldn't trigger an assertion.
112 // it should just work and return an empty string
113 status = seqFile->Read(16, &data, (char*)scratch.data());
114 ASSERT_OK(status);
115
116 ASSERT_TRUE(data.empty());
117}
