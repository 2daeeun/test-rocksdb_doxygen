50 std::string& output) {

77class AggMergeOperator::Accumulator {
78 public:
79 bool Add(const Slice& op, bool is_partial_aggregation) {
80 if (ignore_operands_) {
81 return true;
82 }
83 Slice my_func;
84 Slice my_value;
85 bool ret = ExtractAggFuncAndValue(op, my_func, my_value);
86 if (!ret) {
87 ignore_operands_ = true;
88 return true;
89 }
90
91 // Determine whether we need to do partial merge.
92 if (is_partial_aggregation && !my_func.empty()) {
93 auto f = func_map.find(my_func.ToString());
94 if (f == func_map.end() || !f->second->DoPartialAggregate()) {
95 return false;
96 }
97 }
98
99 if (!func_valid_) {
100 if (my_func != kUnnamedFuncName) {
101 func_ = my_func;
102 func_valid_ = true;
103 }
104 } else if (func_ != my_func) {
105 // User switched aggregation function. Need to aggregate the older
106 // one first.
107
108 // Previous aggreagion can't be done in partial merge
109 if (is_partial_aggregation) {
110 func_valid_ = false;
111 ignore_operands_ = true;
112 return false;
113 }
114
115 // We could consider stashing an iterator into the hash of aggregators
116 // to avoid repeated lookups when the aggregator doesn't change.
117 auto f = func_map.find(func_.ToString());
118 if (f == func_map.end() || !f->second->Aggregate(values_, scratch_)) {
119 func_valid_ = false;
120 ignore_operands_ = true;
121 return true;
122 }
123 std::swap(scratch_, aggregated_);
124 values_.clear();
125 values_.emplace_back(aggregated_);
126 func_ = my_func;
127 }
128 values_.push_back(my_value);
129 return true;
130 }
131
132 // Return false if aggregation fails.
133 // One possible reason
134 bool GetResult(std::string& result) {
135 if (!func_valid_) {
136 return false;
137 }
138 auto f = func_map.find(func_.ToString());
139 if (f == func_map.end()) {
140 return false;
141 }
142 if (!f->second->Aggregate(values_, scratch_)) {
143 return false;
144 }
145 result = EncodeAggFuncAndPayloadNoCheck(func_, scratch_);
146 return true;
147 }
148
149 void Clear() {
150 func_.clear();
151 values_.clear();
152 aggregated_.clear();
153 scratch_.clear();
154 ignore_operands_ = false;
155 func_valid_ = false;
156 }
157
158 private:
159 Slice func_;
160 std::vector<Slice> values_;
161 std::string aggregated_;
162 std::string scratch_;
163 bool ignore_operands_ = false;
164 bool func_valid_ = false;
165};

167// Creating and using a new Accumulator might invoke multiple malloc and is
168// expensive if it needs to be done when processing each merge operation.
169// AggMergeOperator's merge operators can be invoked concurrently by multiple
170// threads so we cannot simply create one Aggregator and reuse.
171// We use thread local instances instead.
172AggMergeOperator::Accumulator& AggMergeOperator::GetTLSAccumulator() {
173 static thread_local Accumulator tls_acc;
174 tls_acc.Clear();
175 return tls_acc;

178void AggMergeOperator::PackAllMergeOperands(const MergeOperationInput& merge_in,
179 MergeOperationOutput& merge_out) {
180 merge_out.new_value = "";
181 PutLengthPrefixedSlice(&merge_out.new_value, kErrorFuncName);
182 if (merge_in.existing_value != nullptr) {
183 PutLengthPrefixedSlice(&merge_out.new_value, *merge_in.existing_value);
184 }
185 for (const Slice& op : merge_in.operand_list) {
186 PutLengthPrefixedSlice(&merge_out.new_value, op);
187 }
188}

190bool AggMergeOperator::FullMergeV2(const MergeOperationInput& merge_in,
191 MergeOperationOutput* merge_out) const {
192 Accumulator& agg = GetTLSAccumulator();
193 if (merge_in.existing_value != nullptr) {
194 agg.Add(*merge_in.existing_value, /*is_partial_aggregation=*/false);
195 }
196 for (const Slice& e : merge_in.operand_list) {
197 agg.Add(e, /*is_partial_aggregation=*/false);
198 }
199
200 bool succ = agg.GetResult(merge_out->new_value);
201 if (!succ) {
202 // If aggregation can't happen, pack all merge operands. In contrast to
203 // merge operator, we don't want to fail the DB. If users insert wrong
204 // format or call unregistered an aggregation function, we still hope
205 // the DB can continue functioning with other keys.
206 PackAllMergeOperands(merge_in, *merge_out);
207 }
208 agg.Clear();
209 return true;
210}

212bool AggMergeOperator::PartialMergeMulti(const Slice& /*key*/,
213 const std::deque<Slice>& operand_list,
214 std::string* new_value,
215 Logger* /*logger*/) const {
216 Accumulator& agg = GetTLSAccumulator();
217 bool do_aggregation = true;
218 for (const Slice& item : operand_list) {
219 do_aggregation = agg.Add(item, /*is_partial_aggregation=*/true);
220 if (!do_aggregation) {
221 break;
222 }
223 }
224 if (do_aggregation) {
225 do_aggregation = agg.GetResult(*new_value);
226 }
227 agg.Clear();
228 return do_aggregation;
229}
