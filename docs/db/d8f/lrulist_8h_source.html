15// LRU element definition
16//
17// Any object that needs to be part of the LRU algorithm should extend this
18// class
19template <class T>
20struct LRUElement {
21 explicit LRUElement() : next_(nullptr), prev_(nullptr), refs_(0) {}
22
23 virtual ~LRUElement() { assert(!refs_); }
24
25 T* next_;
26 T* prev_;
27 std::atomic<size_t> refs_;
28};
29
30// LRU implementation
31//
32// In place LRU implementation. There is no copy or allocation involved when
33// inserting or removing an element. This makes the data structure slim
34template <class T>
35class LRUList {
36 public:
37 virtual ~LRUList() {
38 MutexLock _(&lock_);
39 assert(!head_);
40 assert(!tail_);
41 }
42
43 // Push element into the LRU at the cold end
44 inline void Push(T* const t) {
45 assert(t);
46 assert(!t->next_);
47 assert(!t->prev_);
48
49 MutexLock _(&lock_);
50
51 assert((!head_ && !tail_) || (head_ && tail_));
52 assert(!head_ || !head_->prev_);
53 assert(!tail_ || !tail_->next_);
54
55 t->next_ = head_;
56 if (head_) {
57 head_->prev_ = t;
58 }
59
60 head_ = t;
61 if (!tail_) {
62 tail_ = t;
63 }
64 }
65
66 // Unlink the element from the LRU
67 inline void Unlink(T* const t) {
68 MutexLock _(&lock_);
69 UnlinkImpl(t);
70 }
71
72 // Evict an element from the LRU
73 inline T* Pop() {
74 MutexLock _(&lock_);
75
76 assert(tail_ && head_);
77 assert(!tail_->next_);
78 assert(!head_->prev_);
79
80 T* t = head_;
81 while (t && t->refs_) {
82 t = t->next_;
83 }
84
85 if (!t) {
86 // nothing can be evicted
87 return nullptr;
88 }
89
90 assert(!t->refs_);
91
92 // unlike the element
93 UnlinkImpl(t);
94 return t;
95 }
96
97 // Move the element from the front of the list to the back of the list
98 inline void Touch(T* const t) {
99 MutexLock _(&lock_);
100 UnlinkImpl(t);
101 PushBackImpl(t);
102 }
103
104 // Check if the LRU is empty
105 inline bool IsEmpty() const {
106 MutexLock _(&lock_);
107 return !head_ && !tail_;
108 }
109
110 private:
111 // Unlink an element from the LRU
112 void UnlinkImpl(T* const t) {
113 assert(t);
114
115 lock_.AssertHeld();
116
117 assert(head_ && tail_);
118 assert(t->prev_ || head_ == t);
119 assert(t->next_ || tail_ == t);
120
121 if (t->prev_) {
122 t->prev_->next_ = t->next_;
123 }
124 if (t->next_) {
125 t->next_->prev_ = t->prev_;
126 }
127
128 if (tail_ == t) {
129 tail_ = tail_->prev_;
130 }
131 if (head_ == t) {
132 head_ = head_->next_;
133 }
134
135 t->next_ = t->prev_ = nullptr;
136 }
137
138 // Insert an element at the hot end
139 inline void PushBack(T* const t) {
140 MutexLock _(&lock_);
141 PushBackImpl(t);
142 }
143
144 inline void PushBackImpl(T* const t) {
145 assert(t);
146 assert(!t->next_);
147 assert(!t->prev_);
148
149 lock_.AssertHeld();
150
151 assert((!head_ && !tail_) || (head_ && tail_));
152 assert(!head_ || !head_->prev_);
153 assert(!tail_ || !tail_->next_);
154
155 t->prev_ = tail_;
156 if (tail_) {
157 tail_->next_ = t;
158 }
159
160 tail_ = t;
161 if (!head_) {
162 head_ = tail_;
163 }
164 }
165
166 mutable port::Mutex lock_; // synchronization primitive
167 T* head_ = nullptr; // front (cold)
168 T* tail_ = nullptr; // back (hot)
169};
