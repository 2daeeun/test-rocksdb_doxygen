25Status BlobLogSequentialReader::ReadSlice(uint64_t size, Slice* slice,
26 char* buf) {
27 assert(slice);
28 assert(file_);
29
30 StopWatch read_sw(clock_, statistics_, BLOB_DB_BLOB_FILE_READ_MICROS);
31 // TODO: rate limit `BlobLogSequentialReader` reads (it appears unused?)
32 Status s = file_->Read(IOOptions(), next_byte_, static_cast<size_t>(size),
33 slice, buf, nullptr);
34 next_byte_ += size;
35 if (!s.ok()) {
36 return s;
37 }
38 RecordTick(statistics_, BLOB_DB_BLOB_FILE_BYTES_READ, slice->size());
39 if (slice->size() != size) {
40 return Status::Corruption("EOF reached while reading record");
41 }
42 return s;
43}

45Status BlobLogSequentialReader::ReadHeader(BlobLogHeader* header) {
46 assert(header);
47 assert(next_byte_ == 0);
48
49 static_assert(BlobLogHeader::kSize <= sizeof(header_buf_),
50 "Buffer is smaller than BlobLogHeader::kSize");
51
52 Status s = ReadSlice(BlobLogHeader::kSize, &buffer_, header_buf_);
53 if (!s.ok()) {
54 return s;
55 }
56
57 if (buffer_.size() != BlobLogHeader::kSize) {
58 return Status::Corruption("EOF reached before file header");
59 }
60
61 return header->DecodeFrom(buffer_);
62}

64Status BlobLogSequentialReader::ReadRecord(BlobLogRecord* record,
65 ReadLevel level,
66 uint64_t* blob_offset) {
67 assert(record);
68 static_assert(BlobLogRecord::kHeaderSize <= sizeof(header_buf_),
69 "Buffer is smaller than BlobLogRecord::kHeaderSize");
70
71 Status s = ReadSlice(BlobLogRecord::kHeaderSize, &buffer_, header_buf_);
72 if (!s.ok()) {
73 return s;
74 }
75 if (buffer_.size() != BlobLogRecord::kHeaderSize) {
76 return Status::Corruption("EOF reached before record header");
77 }
78
79 s = record->DecodeHeaderFrom(buffer_);
80 if (!s.ok()) {
81 return s;
82 }
83
84 uint64_t kb_size = record->key_size + record->value_size;
85 if (blob_offset != nullptr) {
86 *blob_offset = next_byte_ + record->key_size;
87 }
88
89 switch (level) {
90 case kReadHeader:
91 next_byte_ += kb_size;
92 break;
93
94 case kReadHeaderKey:
95 record->key_buf.reset(new char[record->key_size]);
96 s = ReadSlice(record->key_size, &record->key, record->key_buf.get());
97 next_byte_ += record->value_size;
98 break;
99
100 case kReadHeaderKeyBlob:
101 record->key_buf.reset(new char[record->key_size]);
102 s = ReadSlice(record->key_size, &record->key, record->key_buf.get());
103 if (s.ok()) {
104 record->value_buf.reset(new char[record->value_size]);
105 s = ReadSlice(record->value_size, &record->value,
106 record->value_buf.get());
107 }
108 if (s.ok()) {
109 s = record->CheckBlobCRC();
110 }
111 break;
112 }
113 return s;
114}
