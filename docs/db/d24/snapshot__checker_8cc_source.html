19SnapshotCheckerResult WritePreparedSnapshotChecker::CheckInSnapshot(
20 SequenceNumber sequence, SequenceNumber snapshot_sequence) const {
21 bool snapshot_released = false;
22 // TODO(myabandeh): set min_uncommitted
23 bool in_snapshot = txn_db_->IsInSnapshot(
24 sequence, snapshot_sequence, kMinUnCommittedSeq, &snapshot_released);
25 if (snapshot_released) {
26 return SnapshotCheckerResult::kSnapshotReleased;
27 }
28 return in_snapshot ? SnapshotCheckerResult::kInSnapshot
29 : SnapshotCheckerResult::kNotInSnapshot;
30}

38bool DataIsDefinitelyInSnapshot(SequenceNumber seqno, SequenceNumber snapshot,
39 const SnapshotChecker* snapshot_checker) {
40 return ((seqno) <= (snapshot) &&
41 (snapshot_checker == nullptr ||
42 LIKELY(snapshot_checker->CheckInSnapshot((seqno), (snapshot)) ==
43 SnapshotCheckerResult::kInSnapshot)));
44}
45
46bool DataIsDefinitelyNotInSnapshot(SequenceNumber seqno,
47 SequenceNumber snapshot,
48 const SnapshotChecker* snapshot_checker) {
49 return ((seqno) > (snapshot) ||
50 (snapshot_checker != nullptr &&
51 UNLIKELY(snapshot_checker->CheckInSnapshot((seqno), (snapshot)) ==
52 SnapshotCheckerResult::kNotInSnapshot)));
53}
