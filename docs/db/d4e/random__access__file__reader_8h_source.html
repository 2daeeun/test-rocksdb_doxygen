41// RandomAccessFileReader is a wrapper on top of FSRandomAccessFile. It is
42// responsible for:
43// - Handling Buffered and Direct reads appropriately.
44// - Rate limiting compaction reads.
45// - Notifying any interested listeners on the completion of a read.
46// - Updating IO stats.
47class RandomAccessFileReader {
48 private:
49 void NotifyOnFileReadFinish(
50 uint64_t offset, size_t length,
51 const FileOperationInfo::StartTimePoint& start_ts,
52 const FileOperationInfo::FinishTimePoint& finish_ts,
53 const Status& status) const {
54 FileOperationInfo info(FileOperationType::kRead, file_name_, start_ts,
55 finish_ts, status, file_temperature_);
56 info.offset = offset;
57 info.length = length;
58
59 for (auto& listener : listeners_) {
60 listener->OnFileReadFinish(info);
61 }
62 info.status.PermitUncheckedError();
63 }
64
65 void NotifyOnIOError(const IOStatus& io_status, FileOperationType operation,
66 const std::string& file_path, size_t length,
67 uint64_t offset) const {
68 if (listeners_.empty()) {
69 return;
70 }
71 IOErrorInfo io_error_info(io_status, operation, file_path, length, offset);
72
73 for (auto& listener : listeners_) {
74 listener->OnIOError(io_error_info);
75 }
76 io_status.PermitUncheckedError();
77 }
78
79
80 bool ShouldNotifyListeners() const { return !listeners_.empty(); }

121 explicit RandomAccessFileReader(
122 std::unique_ptr<FSRandomAccessFile>&& raf, const std::string& _file_name,
123 SystemClock* clock = nullptr,
124 const std::shared_ptr<IOTracer>& io_tracer = nullptr,
125 Statistics* stats = nullptr,
126 uint32_t hist_type = Histograms::HISTOGRAM_ENUM_MAX,
127 HistogramImpl* file_read_hist = nullptr,
128 RateLimiter* rate_limiter = nullptr,
129 const std::vector<std::shared_ptr<EventListener>>& listeners = {},
130 Temperature file_temperature = Temperature::kUnknown,
131 bool is_last_level = false)
132 : file_(std::move(raf), io_tracer, _file_name),
133 file_name_(std::move(_file_name)),
134 clock_(clock),
135 stats_(stats),
136 hist_type_(hist_type),
137 file_read_hist_(file_read_hist),
138 rate_limiter_(rate_limiter),
139 listeners_(),
140 file_temperature_(file_temperature),
141 is_last_level_(is_last_level) {
142 std::for_each(listeners.begin(), listeners.end(),
143 [this](const std::shared_ptr<EventListener>& e) {
144 if (e->ShouldBeNotifiedOnFileIO()) {
145 listeners_.emplace_back(e);
146 }
147 });
148 }

150 static IOStatus Create(const std::shared_ptr<FileSystem>& fs,
151 const std::string& fname, const FileOptions& file_opts,
152 std::unique_ptr<RandomAccessFileReader>* reader,
153 IODebugContext* dbg);

155 RandomAccessFileReader& operator=(const RandomAccessFileReader&) = delete;

157 // In non-direct IO mode,
158 // 1. if using mmap, result is stored in a buffer other than scratch;
159 // 2. if not using mmap, result is stored in the buffer starting from scratch.
160 //
161 // In direct IO mode, an aligned buffer is allocated internally.
162 // 1. If aligned_buf is null, then results are copied to the buffer
163 // starting from scratch;
164 // 2. Otherwise, scratch is not used and can be null, the aligned_buf owns
165 // the internally allocated buffer on return, and the result refers to a
166 // region in aligned_buf.
167 IOStatus Read(const IOOptions& opts, uint64_t offset, size_t n, Slice* result,
168 char* scratch, AlignedBuf* aligned_buf) const;
169
170 // REQUIRES:
171 // num_reqs > 0, reqs do not overlap, and offsets in reqs are increasing.
172 // In non-direct IO mode, aligned_buf should be null;
173 // In direct IO mode, aligned_buf stores the aligned buffer allocated inside
174 // MultiRead, the result Slices in reqs refer to aligned_buf.
175 IOStatus MultiRead(const IOOptions& opts, FSReadRequest* reqs,
176 size_t num_reqs, AlignedBuf* aligned_buf) const;
177
178 IOStatus Prefetch(const IOOptions& opts, uint64_t offset, size_t n) const {
179 return file_->Prefetch(offset, n, opts, nullptr);
180 }
