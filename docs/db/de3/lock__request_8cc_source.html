119// get rid of any stored left and right key copies and
120// replace them with copies of the given left and right key
121void lock_request::copy_keys() {
122 if (!toku_dbt_is_infinite(m_left_key)) {
123 toku_clone_dbt(&m_left_key_copy, *m_left_key);
124 m_left_key = &m_left_key_copy;
125 }

140// build a wait-for-graph for this lock request and the given conflict set
141// for each transaction B that blocks A's lock request
142// if B is blocked then
143// add (A,T) to the WFG and if B is new, fill in the WFG from B
144void lock_request::build_wait_graph(wfg *wait_graph,
145 const txnid_set &conflicts) {
146 uint32_t num_conflicts = conflicts.size();
147 for (uint32_t i = 0; i < num_conflicts; i++) {
148 TXNID conflicting_txnid = conflicts.get(i);
149 lock_request *conflicting_request = find_lock_request(conflicting_txnid);
150 invariant(conflicting_txnid != m_txnid);
151 invariant(conflicting_request != this);
152 if (conflicting_request) {
153 bool already_exists = wait_graph->node_exists(conflicting_txnid);
154 wait_graph->add_edge(m_txnid, conflicting_txnid);
155 if (!already_exists) {
156 // recursively build the wait for graph rooted at the conflicting
157 // request, given its set of lock conflicts.
158 txnid_set other_conflicts;
159 other_conflicts.create();
160 conflicting_request->get_conflicts(&other_conflicts);
161 conflicting_request->build_wait_graph(wait_graph, other_conflicts);
162 other_conflicts.destroy();
163 }
164 }
165 }
166}

168// returns: true if the current set of lock requests contains
169// a deadlock, false otherwise.
170bool lock_request::deadlock_exists(const txnid_set &conflicts) {
171 wfg wait_graph;
172 wait_graph.create();
173
174 build_wait_graph(&wait_graph, conflicts);
175
176 std::function<void(TXNID)> reporter;
177 if (m_deadlock_cb) {
178 reporter = [this](TXNID a) {
179 lock_request *req = find_lock_request(a);
180 if (req) {
181 m_deadlock_cb(req->m_txnid, (req->m_type == lock_request::WRITE),
182 req->m_left_key, req->m_right_key);
183 }
184 };
185 }
186
187 bool deadlock = wait_graph.cycle_exists_from_txnid(m_txnid, reporter);
188 wait_graph.destroy();
189 return deadlock;
190}

192// try to acquire a lock described by this lock request.
193int lock_request::start(void) {
194 int r;
195
196 txnid_set conflicts;
197 conflicts.create();
198 if (m_type == type::WRITE) {
199 r = m_lt->acquire_write_lock(m_txnid, m_left_key, m_right_key, &conflicts,
200 m_big_txn);
201 } else {
202 invariant(m_type == type::READ);
203 r = m_lt->acquire_read_lock(m_txnid, m_left_key, m_right_key, &conflicts,
204 m_big_txn);
205 }
206
207 // if the lock is not granted, save it to the set of lock requests
208 // and check for a deadlock. if there is one, complete it as failed
209 if (r == DB_LOCK_NOTGRANTED) {
210 copy_keys();
211 m_state = state::PENDING;
212 m_start_time = toku_current_time_microsec() / 1000;
213 m_conflicting_txnid = conflicts.get(0);
214 if (m_start_before_pending_test_callback)
215 m_start_before_pending_test_callback();
216 toku_external_mutex_lock(&m_info->mutex);
217 insert_into_lock_requests();
218 if (deadlock_exists(conflicts)) {
219 remove_from_lock_requests();
220 r = DB_LOCK_DEADLOCK;
221 }
222 toku_external_mutex_unlock(&m_info->mutex);
223 if (m_start_test_callback) m_start_test_callback(); // test callback
224 }
225
226 if (r != DB_LOCK_NOTGRANTED) {
227 complete(r);
228 }
229
230 conflicts.destroy();
231 return r;
232}

240int lock_request::wait(uint64_t wait_time_ms, uint64_t killed_time_ms,
241 int (*killed_callback)(void),
242 void (*lock_wait_callback)(void *, lock_wait_infos *),
243 void *callback_arg) {
244 uint64_t t_now = toku_current_time_microsec();
245 uint64_t t_start = t_now;
246 uint64_t t_end = t_start + wait_time_ms * 1000;
247
248 toku_external_mutex_lock(&m_info->mutex);
249
250 // check again, this time locking out other retry calls
251 if (m_state == state::PENDING) {
252 lock_wait_infos conflicts_collector;
253 retry(&conflicts_collector);
254 if (m_state == state::PENDING) {
255 report_waits(&conflicts_collector, lock_wait_callback, callback_arg);
256 }
257 }
258
259 while (m_state == state::PENDING) {
260 // check if this thread is killed
261 if (killed_callback && killed_callback()) {
262 remove_from_lock_requests();
263 complete(DB_LOCK_NOTGRANTED);
264 continue;
265 }
266
267 // compute the time until we should wait
268 uint64_t t_wait;
269 if (killed_time_ms == 0) {
270 t_wait = t_end;
271 } else {
272 t_wait = t_now + killed_time_ms * 1000;
273 if (t_wait > t_end) t_wait = t_end;
274 }
275
276 int r = toku_external_cond_timedwait(&m_wait_cond, &m_info->mutex,
277 (int64_t)(t_wait - t_now));
278 invariant(r == 0 || r == ETIMEDOUT);
279
280 t_now = toku_current_time_microsec();
281 if (m_state == state::PENDING && (t_now >= t_end)) {
282 m_info->counters.timeout_count += 1;
283
284 // if we're still pending and we timed out, then remove our
285 // request from the set of lock requests and fail.
286 remove_from_lock_requests();
287
288 // complete sets m_state to COMPLETE, breaking us out of the loop
289 complete(DB_LOCK_NOTGRANTED);
290 }
291 }
292
293 uint64_t t_real_end = toku_current_time_microsec();
294 uint64_t duration = t_real_end - t_start;
295 m_info->counters.wait_count += 1;
296 m_info->counters.wait_time += duration;
297 if (duration >= 1000000) {
298 m_info->counters.long_wait_count += 1;
299 m_info->counters.long_wait_time += duration;
300 }
301 toku_external_mutex_unlock(&m_info->mutex);
302
303 invariant(m_state == state::COMPLETE);
304 return m_complete_r;
305}

325int lock_request::retry(lock_wait_infos *conflicts_collector) {
326 invariant(m_state == state::PENDING);
327 int r;
328 txnid_set conflicts;
329 conflicts.create();
330
331 if (m_type == type::WRITE) {
332 r = m_lt->acquire_write_lock(m_txnid, m_left_key, m_right_key, &conflicts,
333 m_big_txn);
334 } else {
335 r = m_lt->acquire_read_lock(m_txnid, m_left_key, m_right_key, &conflicts,
336 m_big_txn);
337 }
338
339 // if the acquisition succeeded then remove ourselves from the
340 // set of lock requests, complete, and signal the waiting thread.
341 if (r == 0) {
342 remove_from_lock_requests();
343 complete(r);
344 if (m_retry_test_callback) m_retry_test_callback(); // test callback
345 toku_external_cond_broadcast(&m_wait_cond);
346 } else {
347 m_conflicting_txnid = conflicts.get(0);
348 add_conflicts_to_waits(&conflicts, conflicts_collector);
349 }
350 conflicts.destroy();
351
352 return r;
353}

355void lock_request::retry_all_lock_requests(
356 locktree *lt, void (*lock_wait_callback)(void *, lock_wait_infos *),
357 void *callback_arg, void (*after_retry_all_test_callback)(void)) {
358 lt_lock_request_info *info = lt->get_lock_request_info();
359
360 // if there are no pending lock requests than there is nothing to do
361 // the unlocked data race on pending_is_empty is OK since lock requests
362 // are retried after added to the pending set.
363 if (info->pending_is_empty) return;
364
365 // get my retry generation (post increment of retry_want)
366 unsigned long long my_retry_want = (info->retry_want += 1);
367
368 toku_mutex_lock(&info->retry_mutex);
369
370 // here is the group retry algorithm.
371 // get the latest retry_want count and use it as the generation number of
372 // this retry operation. if this retry generation is > the last retry
373 // generation, then do the lock retries. otherwise, no lock retries
374 // are needed.
375 if ((my_retry_want - 1) == info->retry_done) {
376 for (;;) {
377 if (!info->running_retry) {
378 info->running_retry = true;
379 info->retry_done = info->retry_want;
380 toku_mutex_unlock(&info->retry_mutex);
381 retry_all_lock_requests_info(info, lock_wait_callback, callback_arg);
382 if (after_retry_all_test_callback) after_retry_all_test_callback();
383 toku_mutex_lock(&info->retry_mutex);
384 info->running_retry = false;
385 toku_cond_broadcast(&info->retry_cv);
386 break;
387 } else {
388 toku_cond_wait(&info->retry_cv, &info->retry_mutex);
389 }
390 }
391 }
392 toku_mutex_unlock(&info->retry_mutex);
393}

395void lock_request::retry_all_lock_requests_info(
396 lt_lock_request_info *info,
397 void (*lock_wait_callback)(void *, lock_wait_infos *), void *callback_arg) {
398 toku_external_mutex_lock(&info->mutex);
399 // retry all of the pending lock requests.
400 lock_wait_infos conflicts_collector;
401 for (uint32_t i = 0; i < info->pending_lock_requests.size();) {
402 lock_request *request;
403 int r = info->pending_lock_requests.fetch(i, &request);
404 invariant_zero(r);
405
406 // retry the lock request. if it didn't succeed,
407 // move on to the next lock request. otherwise
408 // the request is gone from the list so we may
409 // read the i'th entry for the next one.
410 r = request->retry(&conflicts_collector);
411 if (r != 0) {
412 i++;
413 }
414 }
415
416 // call report_waits while holding the pending queue lock since
417 // the waiter object is still valid while it's in the queue
418 report_waits(&conflicts_collector, lock_wait_callback, callback_arg);
419
420 // future threads should only retry lock requests if some still exist
421 info->should_retry_lock_requests = info->pending_lock_requests.size() > 0;
422 toku_external_mutex_unlock(&info->mutex);
423}

449void lock_request::kill_waiter(locktree *lt, void *extra) {
450 lt_lock_request_info *info = lt->get_lock_request_info();
451 toku_external_mutex_lock(&info->mutex);
452 for (uint32_t i = 0; i < info->pending_lock_requests.size(); i++) {
453 lock_request *request;
454 int r = info->pending_lock_requests.fetch(i, &request);
455 if (r == 0 && request->get_extra() == extra) {
456 request->kill_waiter();
457 break;
458 }
459 }
460 toku_external_mutex_unlock(&info->mutex);
461}

463// find another lock request by txnid. must hold the mutex.

474// insert this lock request into the locktree's set. must hold the mutex.
475void lock_request::insert_into_lock_requests(void) {
476 uint32_t idx;
477 lock_request *request;
478 int r = m_info->pending_lock_requests.find_zero<TXNID, find_by_txnid>(
479 m_txnid, &request, &idx);
480 invariant(r == DB_NOTFOUND);
481 r = m_info->pending_lock_requests.insert_at(this, idx);
482 invariant_zero(r);
483 m_info->pending_is_empty = false;
484}
485
486// remove this lock request from the locktree's set. must hold the mutex.
487void lock_request::remove_from_lock_requests(void) {
488 uint32_t idx;
489 lock_request *request;
490 int r = m_info->pending_lock_requests.find_zero<TXNID, find_by_txnid>(
491 m_txnid, &request, &idx);
492 invariant_zero(r);
493 invariant(request == this);
494 r = m_info->pending_lock_requests.delete_at(idx);
495 invariant_zero(r);
496 if (m_info->pending_lock_requests.size() == 0)
497 m_info->pending_is_empty = true;
498}

500int lock_request::find_by_txnid(lock_request *const &request,
501 const TXNID &txnid) {
502 TXNID request_txnid = request->m_txnid;
503 if (request_txnid < txnid) {
504 return -1;
505 } else if (request_txnid == txnid) {
506 return 0;
507 } else {
508 return 1;
509 }
