18AllocTracker::AllocTracker(WriteBufferManager* write_buffer_manager)
19 : write_buffer_manager_(write_buffer_manager),

26void AllocTracker::Allocate(size_t bytes) {
27 assert(write_buffer_manager_ != nullptr);
28 if (write_buffer_manager_->enabled() ||
29 write_buffer_manager_->cost_to_cache()) {
30 bytes_allocated_.fetch_add(bytes, std::memory_order_relaxed);
31 write_buffer_manager_->ReserveMem(bytes);
32 }
33}
34
35void AllocTracker::DoneAllocating() {
36 if (write_buffer_manager_ != nullptr && !done_allocating_) {
37 if (write_buffer_manager_->enabled() ||
38 write_buffer_manager_->cost_to_cache()) {
39 write_buffer_manager_->ScheduleFreeMem(
40 bytes_allocated_.load(std::memory_order_relaxed));
41 } else {
42 assert(bytes_allocated_.load(std::memory_order_relaxed) == 0);
43 }
44 done_allocating_ = true;
45 }
46}
47
48void AllocTracker::FreeMem() {
49 if (!done_allocating_) {
50 DoneAllocating();
51 }
52 if (write_buffer_manager_ != nullptr && !freed_) {
53 if (write_buffer_manager_->enabled() ||
54 write_buffer_manager_->cost_to_cache()) {
55 write_buffer_manager_->FreeMem(
56 bytes_allocated_.load(std::memory_order_relaxed));
57 } else {
58 assert(bytes_allocated_.load(std::memory_order_relaxed) == 0);
59 }
60 freed_ = true;
61 }
62}
