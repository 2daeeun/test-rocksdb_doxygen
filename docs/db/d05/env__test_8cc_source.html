78struct Deleter {
79 explicit Deleter(void (*fn)(void*)) : fn_(fn) {}
80
81 void operator()(void* ptr) {
82 assert(fn_);
83 assert(ptr);
84 (*fn_)(ptr);
85 }
86
87 void (*fn_)(void*);
88};

92std::unique_ptr<char, Deleter> NewAligned(const size_t size, const char ch) {
93 char* ptr = nullptr;
94#ifdef OS_WIN
95 if (nullptr ==
96 (ptr = reinterpret_cast<char*>(_aligned_malloc(size, kPageSize)))) {
97 return std::unique_ptr<char, Deleter>(nullptr, Deleter(_aligned_free));
98 }
99 std::unique_ptr<char, Deleter> uptr(ptr, Deleter(_aligned_free));
100#else
101 if (posix_memalign(reinterpret_cast<void**>(&ptr), kPageSize, size) != 0) {
102 return std::unique_ptr<char, Deleter>(nullptr, Deleter(free));
103 }
104 std::unique_ptr<char, Deleter> uptr(ptr, Deleter(free));
105#endif
106 memset(uptr.get(), ch, size);
107 return uptr;
108}

485 State* s = static_cast<State*>(arg);
486 s->mu.Lock();

512TEST_P(EnvPosixTestWithParam, TwoPools) {
513 // Data structures to signal tasks to run.
514 port::Mutex mutex;
515 port::CondVar cv(&mutex);
516 bool should_start = false;
517
518 class CB {
519 public:
520 CB(const std::string& pool_name, int pool_size, port::Mutex* trigger_mu,
521 port::CondVar* trigger_cv, bool* _should_start)
522 : mu_(),
523 num_running_(0),
524 num_finished_(0),
525 pool_size_(pool_size),
526 pool_name_(pool_name),
527 trigger_mu_(trigger_mu),
528 trigger_cv_(trigger_cv),
529 should_start_(_should_start) {}
530
531 static void Run(void* v) {
532 CB* cb = static_cast<CB*>(v);
533 cb->Run();
534 }
535
536 void Run() {
537 {
538 MutexLock l(&mu_);
539 num_running_++;
540 // make sure we don't have more than pool_size_ jobs running.
541 ASSERT_LE(num_running_, pool_size_.load());
542 }
543
544 {
545 MutexLock l(trigger_mu_);
546 while (!(*should_start_)) {
547 trigger_cv_->Wait();
548 }
549 }
550
551 {
552 MutexLock l(&mu_);
553 num_running_--;
554 num_finished_++;
555 }
556 }
557
558 int NumFinished() {
559 MutexLock l(&mu_);
560 return num_finished_;
561 }
562
563 void Reset(int pool_size) {
564 pool_size_.store(pool_size);
565 num_finished_ = 0;
566 }
567
568 private:
569 port::Mutex mu_;
570 int num_running_;
571 int num_finished_;
572 std::atomic<int> pool_size_;
573 std::string pool_name_;
574 port::Mutex* trigger_mu_;
575 port::CondVar* trigger_cv_;
576 bool* should_start_;
577 };
578
579 const int kLowPoolSize = 2;
580 const int kHighPoolSize = 4;
581 const int kJobs = 8;
582
583 CB low_pool_job("low", kLowPoolSize, &mutex, &cv, &should_start);
584 CB high_pool_job("high", kHighPoolSize, &mutex, &cv, &should_start);
585
586 env_->SetBackgroundThreads(kLowPoolSize);
587 env_->SetBackgroundThreads(kHighPoolSize, Env::Priority::HIGH);
588
589 ASSERT_EQ(0U, env_->GetThreadPoolQueueLen(Env::Priority::LOW));
590 ASSERT_EQ(0U, env_->GetThreadPoolQueueLen(Env::Priority::HIGH));
591
592 // schedule same number of jobs in each pool
593 for (int i = 0; i < kJobs; i++) {
594 env_->Schedule(&CB::Run, &low_pool_job);
595 env_->Schedule(&CB::Run, &high_pool_job, Env::Priority::HIGH);
596 }
597 // Wait a short while for the jobs to be dispatched.
598 int sleep_count = 0;
599 while ((unsigned int)(kJobs - kLowPoolSize) !=
600 env_->GetThreadPoolQueueLen(Env::Priority::LOW) ||
601 (unsigned int)(kJobs - kHighPoolSize) !=
602 env_->GetThreadPoolQueueLen(Env::Priority::HIGH)) {
603 env_->SleepForMicroseconds(kDelayMicros);
604 if (++sleep_count > 100) {
605 break;
606 }
607 }
608
609 ASSERT_EQ((unsigned int)(kJobs - kLowPoolSize),
610 env_->GetThreadPoolQueueLen());
611 ASSERT_EQ((unsigned int)(kJobs - kLowPoolSize),
612 env_->GetThreadPoolQueueLen(Env::Priority::LOW));
613 ASSERT_EQ((unsigned int)(kJobs - kHighPoolSize),
614 env_->GetThreadPoolQueueLen(Env::Priority::HIGH));
615
616 // Trigger jobs to run.
617 {
618 MutexLock l(&mutex);
619 should_start = true;
620 cv.SignalAll();
621 }
622
623 // wait for all jobs to finish
624 while (low_pool_job.NumFinished() < kJobs ||
625 high_pool_job.NumFinished() < kJobs) {
626 env_->SleepForMicroseconds(kDelayMicros);
627 }
628
629 ASSERT_EQ(0U, env_->GetThreadPoolQueueLen(Env::Priority::LOW));
630 ASSERT_EQ(0U, env_->GetThreadPoolQueueLen(Env::Priority::HIGH));
631
632 // Hold jobs to schedule;
633 should_start = false;
634
635 // call IncBackgroundThreadsIfNeeded to two pools. One increasing and
636 // the other decreasing
637 env_->IncBackgroundThreadsIfNeeded(kLowPoolSize - 1, Env::Priority::LOW);
638 env_->IncBackgroundThreadsIfNeeded(kHighPoolSize + 1, Env::Priority::HIGH);
639 high_pool_job.Reset(kHighPoolSize + 1);
640 low_pool_job.Reset(kLowPoolSize);
641
642 // schedule same number of jobs in each pool
643 for (int i = 0; i < kJobs; i++) {
644 env_->Schedule(&CB::Run, &low_pool_job);
645 env_->Schedule(&CB::Run, &high_pool_job, Env::Priority::HIGH);
646 }
647 // Wait a short while for the jobs to be dispatched.
648 sleep_count = 0;
649 while ((unsigned int)(kJobs - kLowPoolSize) !=
650 env_->GetThreadPoolQueueLen(Env::Priority::LOW) ||
651 (unsigned int)(kJobs - (kHighPoolSize + 1)) !=
652 env_->GetThreadPoolQueueLen(Env::Priority::HIGH)) {
653 env_->SleepForMicroseconds(kDelayMicros);
654 if (++sleep_count > 100) {
655 break;
656 }
657 }
658 ASSERT_EQ((unsigned int)(kJobs - kLowPoolSize),
659 env_->GetThreadPoolQueueLen());
660 ASSERT_EQ((unsigned int)(kJobs - kLowPoolSize),
661 env_->GetThreadPoolQueueLen(Env::Priority::LOW));
662 ASSERT_EQ((unsigned int)(kJobs - (kHighPoolSize + 1)),
663 env_->GetThreadPoolQueueLen(Env::Priority::HIGH));
664
665 // Trigger jobs to run.
666 {
667 MutexLock l(&mutex);
668 should_start = true;
669 cv.SignalAll();
670 }
671
672 // wait for all jobs to finish
673 while (low_pool_job.NumFinished() < kJobs ||
674 high_pool_job.NumFinished() < kJobs) {
675 env_->SleepForMicroseconds(kDelayMicros);
676 }
677
678 env_->SetBackgroundThreads(kHighPoolSize, Env::Priority::HIGH);
679 WaitThreadPoolsEmpty();
680}

992// Determine whether we can use the FS_IOC_GETVERSION ioctl
993// on a file in directory DIR. Create a temporary file therein,
994// try to apply the ioctl (save that result), cleanup and
995// return the result. Return true if it is supported, and
996// false if anything fails.
997// Note that this function "knows" that dir has just been created
998// and is empty, so we create a simply-named test file: "f".
999bool ioctl_support__FS_IOC_GETVERSION(const std::string& dir) {
1000#ifdef OS_WIN
1001 return true;
1002#else
1003 const std::string file = dir + "/f";
1004 int fd;
1005 do {
1006 fd = open(file.c_str(), O_CREAT | O_RDWR | O_TRUNC, 0644);
1007 } while (fd < 0 && errno == EINTR);
1008 long int version;
1009 bool ok = (fd >= 0 && ioctl(fd, FS_IOC_GETVERSION, &version) >= 0);
1010
1011 close(fd);
1012 unlink(file.c_str());
1013
1014 return ok;
1015#endif

1018// To ensure that Env::GetUniqueId-related tests work correctly, the files
1019// should be stored in regular storage like "hard disk" or "flash device",
1020// and not on a tmpfs file system (like /dev/shm and /tmp on some systems).
1021// Otherwise we cannot get the correct id.
1022//
1023// This function serves as the replacement for test::TmpDir(), which may be
1024// customized to be on a file system that doesn't work with GetUniqueId().
1025
1026class IoctlFriendlyTmpdir {
1027 public:
1028 explicit IoctlFriendlyTmpdir() {
1029 char dir_buf[100];
1030
1031 const char* fmt = "%s/rocksdb.XXXXXX";
1032 const char* tmp = getenv("TEST_IOCTL_FRIENDLY_TMPDIR");
1033
1034#ifdef OS_WIN
1035#define rmdir _rmdir
1036 if (tmp == nullptr) {
1037 tmp = getenv("TMP");
1038 }
1039
1040 snprintf(dir_buf, sizeof dir_buf, fmt, tmp);
1041 auto result = _mktemp(dir_buf);
1042 assert(result != nullptr);
1043 BOOL ret = CreateDirectory(dir_buf, NULL);
1044 assert(ret == TRUE);
1045 dir_ = dir_buf;
1046#else
1047 std::list<std::string> candidate_dir_list = {"/var/tmp", "/tmp"};
1048
1049 // If $TEST_IOCTL_FRIENDLY_TMPDIR/rocksdb.XXXXXX fits, use
1050 // $TEST_IOCTL_FRIENDLY_TMPDIR; subtract 2 for the "%s", and
1051 // add 1 for the trailing NUL byte.
1052 if (tmp && strlen(tmp) + strlen(fmt) - 2 + 1 <= sizeof dir_buf) {
1053 // use $TEST_IOCTL_FRIENDLY_TMPDIR value
1054 candidate_dir_list.push_front(tmp);
1055 }
1056
1057 for (const std::string& d : candidate_dir_list) {
1058 snprintf(dir_buf, sizeof dir_buf, fmt, d.c_str());
1059 if (mkdtemp(dir_buf)) {
1060 if (ioctl_support__FS_IOC_GETVERSION(dir_buf)) {
1061 dir_ = dir_buf;
1062 return;
1063 } else {
1064 // Diagnose ioctl-related failure only if this is the
1065 // directory specified via that envvar.
1066 if (tmp && tmp == d) {
1067 fprintf(stderr,
1068 "TEST_IOCTL_FRIENDLY_TMPDIR-specified directory is "
1069 "not suitable: %s\n",
1070 d.c_str());
1071 }
1072 rmdir(dir_buf); // ignore failure
1073 }
1074 } else {
1075 // mkdtemp failed: diagnose it, but don't give up.
1076 fprintf(stderr, "mkdtemp(%s/...) failed: %s\n", d.c_str(),
1077 errnoStr(errno).c_str());
1078 }
1079 }
1080
1081 // check if it's running test within a docker container, in which case, the
1082 // file system inside `overlayfs` may not support FS_IOC_GETVERSION
1083 // skip the tests
1084 struct stat buffer;
1085 if (stat("/.dockerenv", &buffer) == 0) {
1086 is_supported_ = false;
1087 return;
1088 }
1089
1090 fprintf(stderr,
1091 "failed to find an ioctl-friendly temporary directory;"
1092 " specify one via the TEST_IOCTL_FRIENDLY_TMPDIR envvar\n");
1093 std::abort();
1094#endif
1095 }

1097 ~IoctlFriendlyTmpdir() { rmdir(dir_.c_str()); }

1099 const std::string& name() const { return dir_; }
1100
1101 bool is_supported() const { return is_supported_; }
1102
1103 private:
1104 std::string dir_;
1105
1106 bool is_supported_ = true;
1107};
