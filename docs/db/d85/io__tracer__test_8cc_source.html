38 std::string GetFileOperation(uint64_t id) {
39 id = id % 4;
40 switch (id) {
41 case 0:
42 return "CreateDir";
43 case 1:
44 return "GetChildren";
45 case 2:
46 return "FileSize";
47 case 3:
48 return "DeleteDir";
49 default:
50 assert(false);
51 }
52 return "";
53 }

55 void WriteIOOp(IOTraceWriter* writer, uint64_t nrecords) {
56 assert(writer);
57 for (uint64_t i = 0; i < nrecords; i++) {
58 IOTraceRecord record;
59 record.io_op_data = 0;
60 record.trace_type = TraceType::kIOTracer;
61 record.io_op_data |= (1 << IOTraceOp::kIOLen);
62 record.io_op_data |= (1 << IOTraceOp::kIOOffset);
63 record.file_operation = GetFileOperation(i);
64 record.io_status = IOStatus::OK().ToString();
65 record.file_name = kDummyFile + std::to_string(i);
66 record.len = i;
67 record.offset = i + 20;
68 EXPECT_OK(writer->WriteIOOp(record, nullptr));
69 }
70 }
71
72 void VerifyIOOp(IOTraceReader* reader, uint32_t nrecords) {
73 assert(reader);
74 for (uint32_t i = 0; i < nrecords; i++) {
75 IOTraceRecord record;
76 ASSERT_OK(reader->ReadIOOp(&record));
77 ASSERT_EQ(record.file_operation, GetFileOperation(i));
78 ASSERT_EQ(record.io_status, IOStatus::OK().ToString());
79 ASSERT_EQ(record.len, i);
80 ASSERT_EQ(record.offset, i + 20);
81 }
82 }

91TEST_F(IOTracerTest, MultipleRecordsWithDifferentIOOpOptions) {
92 std::string file_name = kDummyFile + std::to_string(5);
93 {
94 TraceOptions trace_opt;
95 std::unique_ptr<TraceWriter> trace_writer;
96
97 ASSERT_OK(NewFileTraceWriter(env_, env_options_, trace_file_path_,
98 &trace_writer));
99 IOTracer writer;
100 ASSERT_OK(writer.StartIOTrace(clock_, trace_opt, std::move(trace_writer)));
101
102 // Write general record.
103 IOTraceRecord record0(0, TraceType::kIOTracer, 0 /*io_op_data*/,
104 GetFileOperation(0), 155 /*latency*/,
105 IOStatus::OK().ToString(), file_name);
106 writer.WriteIOOp(record0, nullptr);
107
108 // Write record with FileSize.
109 uint64_t io_op_data = 0;
110 io_op_data |= (1 << IOTraceOp::kIOFileSize);
111 IOTraceRecord record1(0, TraceType::kIOTracer, io_op_data,
112 GetFileOperation(1), 10 /*latency*/,
113 IOStatus::OK().ToString(), file_name,
114 256 /*file_size*/);
115 writer.WriteIOOp(record1, nullptr);
116
117 // Write record with Length.
118 io_op_data = 0;
119 io_op_data |= (1 << IOTraceOp::kIOLen);
120 IOTraceRecord record2(0, TraceType::kIOTracer, io_op_data,
121 GetFileOperation(2), 10 /*latency*/,
122 IOStatus::OK().ToString(), file_name, 100 /*length*/,
123 200 /*offset*/);
124 writer.WriteIOOp(record2, nullptr);
125
126 // Write record with Length and offset.
127 io_op_data = 0;
128 io_op_data |= (1 << IOTraceOp::kIOLen);
129 io_op_data |= (1 << IOTraceOp::kIOOffset);
130 IOTraceRecord record3(0, TraceType::kIOTracer, io_op_data,
131 GetFileOperation(3), 10 /*latency*/,
132 IOStatus::OK().ToString(), file_name, 120 /*length*/,
133 17 /*offset*/);
134 writer.WriteIOOp(record3, nullptr);
135
136 // Write record with offset.
137 io_op_data = 0;
138 io_op_data |= (1 << IOTraceOp::kIOOffset);
139 IOTraceRecord record4(0, TraceType::kIOTracer, io_op_data,
140 GetFileOperation(4), 10 /*latency*/,
141 IOStatus::OK().ToString(), file_name, 13 /*length*/,
142 50 /*offset*/);
143 writer.WriteIOOp(record4, nullptr);
144
145 // Write record with IODebugContext.
146 io_op_data = 0;
147 IODebugContext dbg;
148 dbg.SetRequestId("request_id_1");
149 IOTraceRecord record5(0, TraceType::kIOTracer, io_op_data,
150 GetFileOperation(5), 10 /*latency*/,
151 IOStatus::OK().ToString(), file_name);
152 writer.WriteIOOp(record5, &dbg);
153
154 ASSERT_OK(env_->FileExists(trace_file_path_));
155 }
156 {
157 // Verify trace file is generated correctly.
158 std::unique_ptr<TraceReader> trace_reader;
159 ASSERT_OK(NewFileTraceReader(env_, env_options_, trace_file_path_,
160 &trace_reader));
161 IOTraceReader reader(std::move(trace_reader));
162 IOTraceHeader header;
163 ASSERT_OK(reader.ReadHeader(&header));
164 ASSERT_EQ(kMajorVersion, static_cast<int>(header.rocksdb_major_version));
165 ASSERT_EQ(kMinorVersion, static_cast<int>(header.rocksdb_minor_version));
166
167 // Read general record.
168 IOTraceRecord record0;
169 ASSERT_OK(reader.ReadIOOp(&record0));
170 ASSERT_EQ(record0.file_operation, GetFileOperation(0));
171 ASSERT_EQ(record0.latency, 155);
172 ASSERT_EQ(record0.file_name, file_name);
173
174 // Read record with FileSize.
175 IOTraceRecord record1;
176 ASSERT_OK(reader.ReadIOOp(&record1));
177 ASSERT_EQ(record1.file_size, 256);
178 ASSERT_EQ(record1.len, 0);
179 ASSERT_EQ(record1.offset, 0);
180
181 // Read record with Length.
182 IOTraceRecord record2;
183 ASSERT_OK(reader.ReadIOOp(&record2));
184 ASSERT_EQ(record2.len, 100);
185 ASSERT_EQ(record2.file_size, 0);
186 ASSERT_EQ(record2.offset, 0);
187
188 // Read record with Length and offset.
189 IOTraceRecord record3;
190 ASSERT_OK(reader.ReadIOOp(&record3));
191 ASSERT_EQ(record3.len, 120);
192 ASSERT_EQ(record3.file_size, 0);
193 ASSERT_EQ(record3.offset, 17);
194
195 // Read record with offset.
196 IOTraceRecord record4;
197 ASSERT_OK(reader.ReadIOOp(&record4));
198 ASSERT_EQ(record4.len, 0);
199 ASSERT_EQ(record4.file_size, 0);
200 ASSERT_EQ(record4.offset, 50);
201
202 IOTraceRecord record5;
203 ASSERT_OK(reader.ReadIOOp(&record5));
204 ASSERT_EQ(record5.len, 0);
205 ASSERT_EQ(record5.file_size, 0);
206 ASSERT_EQ(record5.offset, 0);
207 ASSERT_EQ(record5.request_id, "request_id_1");
208 // Read one more record and it should report error.
209 IOTraceRecord record6;
210 ASSERT_NOK(reader.ReadIOOp(&record6));
211 }
212}
