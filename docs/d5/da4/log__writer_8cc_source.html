55 if (dest_->seen_error()) {
56#ifndef NDEBUG

63#endif // NDEBUG
64 return IOStatus::IOError("Seen error. Skip writing buffer.");
65 }
66 IOOptions opts;
67 IOStatus s = WritableFileWriter::PrepareIOOptions(write_options, opts);
68 if (!s.ok()) {
69 return s;
70 }
71 return dest_->Flush(opts);

94IOStatus Writer::AddRecord(const WriteOptions& write_options,
95 const Slice& slice) {
96 if (dest_->seen_error()) {
97#ifndef NDEBUG
98 if (dest_->seen_injected_error()) {
99 std::stringstream msg;
100 msg << "Seen " << FaultInjectionTestFS::kInjected
101 << " error. Skip writing buffer.";
102 return IOStatus::IOError(msg.str());
103 }
104#endif // NDEBUG
105 return IOStatus::IOError("Seen error. Skip writing buffer.");
106 }
107 const char* ptr = slice.data();
108 size_t left = slice.size();
109
110 // Fragment the record if necessary and emit it. Note that if slice
111 // is empty, we still want to iterate once to emit a single
112 // zero-length record
113 bool begin = true;
114 int compress_remaining = 0;
115 bool compress_start = false;
116 if (compress_) {
117 compress_->Reset();
118 compress_start = true;
119 }
120
121 IOStatus s;
122 IOOptions opts;
123 s = WritableFileWriter::PrepareIOOptions(write_options, opts);
124 if (s.ok()) {
125 do {
126 const int64_t leftover = kBlockSize - block_offset_;
127 assert(leftover >= 0);
128 if (leftover < header_size_) {
129 // Switch to a new block
130 if (leftover > 0) {
131 // Fill the trailer (literal below relies on kHeaderSize and
132 // kRecyclableHeaderSize being <= 11)
133 assert(header_size_ <= 11);
134 s = dest_->Append(opts,
135 Slice("\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00",
136 static_cast<size_t>(leftover)),
137 0 /* crc32c_checksum */);
138 if (!s.ok()) {
139 break;
140 }
141 }
142 block_offset_ = 0;
143 }
144
145 // Invariant: we never leave < header_size bytes in a block.
146 assert(static_cast<int64_t>(kBlockSize - block_offset_) >= header_size_);
147
148 const size_t avail = kBlockSize - block_offset_ - header_size_;
149
150 // Compress the record if compression is enabled.
151 // Compress() is called at least once (compress_start=true) and after the
152 // previous generated compressed chunk is written out as one or more
153 // physical records (left=0).
154 if (compress_ && (compress_start || left == 0)) {
155 compress_remaining = compress_->Compress(
156 slice.data(), slice.size(), compressed_buffer_.get(), &left);
157
158 if (compress_remaining < 0) {
159 // Set failure status
160 s = IOStatus::IOError("Unexpected WAL compression error");
161 s.SetDataLoss(true);
162 break;
163 } else if (left == 0) {
164 // Nothing left to compress
165 if (!compress_start) {
166 break;
167 }
168 }
169 compress_start = false;
170 ptr = compressed_buffer_.get();
171 }
172
173 const size_t fragment_length = (left < avail) ? left : avail;
174
175 RecordType type;
176 const bool end = (left == fragment_length && compress_remaining == 0);
177 if (begin && end) {
178 type = recycle_log_files_ ? kRecyclableFullType : kFullType;
179 } else if (begin) {
180 type = recycle_log_files_ ? kRecyclableFirstType : kFirstType;
181 } else if (end) {
182 type = recycle_log_files_ ? kRecyclableLastType : kLastType;
183 } else {
184 type = recycle_log_files_ ? kRecyclableMiddleType : kMiddleType;
185 }
186
187 s = EmitPhysicalRecord(write_options, type, ptr, fragment_length);
188 ptr += fragment_length;
189 left -= fragment_length;
190 begin = false;
191 } while (s.ok() && (left > 0 || compress_remaining > 0));
192 }
193 if (s.ok()) {
194 if (!manual_flush_) {
195 s = dest_->Flush(opts);
196 }
197 }
198
199 return s;
200}

202IOStatus Writer::AddCompressionTypeRecord(const WriteOptions& write_options) {
203 // Should be the first record
204 assert(block_offset_ == 0);
205
206 if (compression_type_ == kNoCompression) {
207 // No need to add a record
208 return IOStatus::OK();
209 }
210
211 if (dest_->seen_error()) {
212#ifndef NDEBUG
213 if (dest_->seen_injected_error()) {
214 std::stringstream msg;
215 msg << "Seen " << FaultInjectionTestFS::kInjected
216 << " error. Skip writing buffer.";
217 return IOStatus::IOError(msg.str());
218 }
219#endif // NDEBUG
220 return IOStatus::IOError("Seen error. Skip writing buffer.");
221 }
222
223 CompressionTypeRecord record(compression_type_);
224 std::string encode;
225 record.EncodeTo(&encode);
226 IOStatus s = EmitPhysicalRecord(write_options, kSetCompressionType,
227 encode.data(), encode.size());
228 if (s.ok()) {
229 if (!manual_flush_) {
230 IOOptions io_opts;
231 s = WritableFileWriter::PrepareIOOptions(write_options, io_opts);
232 if (s.ok()) {
233 s = dest_->Flush(io_opts);
234 }
235 }
236 // Initialize fields required for compression
237 const size_t max_output_buffer_len = kBlockSize - header_size_;
238 CompressionOptions opts;
239 constexpr uint32_t compression_format_version = 2;
240 compress_ = StreamingCompress::Create(compression_type_, opts,
241 compression_format_version,
242 max_output_buffer_len);
243 assert(compress_ != nullptr);
244 compressed_buffer_ =
245 std::unique_ptr<char[]>(new char[max_output_buffer_len]);
246 assert(compressed_buffer_);
247 } else {
248 // Disable compression if the record could not be added.
249 compression_type_ = kNoCompression;
250 }
251 return s;
252}

254IOStatus Writer::MaybeAddUserDefinedTimestampSizeRecord(
255 const WriteOptions& write_options,
256 const UnorderedMap<uint32_t, size_t>& cf_to_ts_sz) {
257 std::vector<std::pair<uint32_t, size_t>> ts_sz_to_record;
258 for (const auto& [cf_id, ts_sz] : cf_to_ts_sz) {
259 if (recorded_cf_to_ts_sz_.count(cf_id) != 0) {
260 // A column family's user-defined timestamp size should not be
261 // updated while DB is running.
262 assert(recorded_cf_to_ts_sz_[cf_id] == ts_sz);
263 } else if (ts_sz != 0) {
264 ts_sz_to_record.emplace_back(cf_id, ts_sz);
265 recorded_cf_to_ts_sz_.insert(std::make_pair(cf_id, ts_sz));
266 }
267 }
268 if (ts_sz_to_record.empty()) {
269 return IOStatus::OK();
270 }
271
272 UserDefinedTimestampSizeRecord record(std::move(ts_sz_to_record));
273 std::string encoded;
274 record.EncodeTo(&encoded);
275 RecordType type = recycle_log_files_ ? kRecyclableUserDefinedTimestampSizeType
276 : kUserDefinedTimestampSizeType;
277
278 // If there's not enough space for this record, switch to a new block.
279 const int64_t leftover = kBlockSize - block_offset_;
280 if (leftover < header_size_ + (int)encoded.size()) {
281 IOOptions opts;
282 IOStatus s = WritableFileWriter::PrepareIOOptions(write_options, opts);
283 if (!s.ok()) {
284 return s;
285 }
286
287 std::vector<char> trailer(leftover, '\x00');
288 s = dest_->Append(opts, Slice(trailer.data(), trailer.size()));
289 if (!s.ok()) {
290 return s;
291 }
292
293 block_offset_ = 0;
294 }
295
296 return EmitPhysicalRecord(write_options, type, encoded.data(),
297 encoded.size());
298}

302IOStatus Writer::EmitPhysicalRecord(const WriteOptions& write_options,
303 RecordType t, const char* ptr, size_t n) {
304 assert(n <= 0xffff); // Must fit in two bytes
305
306 size_t header_size;
307 char buf[kRecyclableHeaderSize];
308
309 // Format the header
310 buf[4] = static_cast<char>(n & 0xff);
311 buf[5] = static_cast<char>(n >> 8);
312 buf[6] = static_cast<char>(t);
313
314 uint32_t crc = type_crc_[t];
315 if (t < kRecyclableFullType || t == kSetCompressionType ||
316 t == kUserDefinedTimestampSizeType) {
317 // Legacy record format
318 assert(block_offset_ + kHeaderSize + n <= kBlockSize);
319 header_size = kHeaderSize;
320 } else {
321 // Recyclable record format
322 assert(block_offset_ + kRecyclableHeaderSize + n <= kBlockSize);
323 header_size = kRecyclableHeaderSize;
324
325 // Only encode low 32-bits of the 64-bit log number. This means
326 // we will fail to detect an old record if we recycled a log from
327 // ~4 billion logs ago, but that is effectively impossible, and
328 // even if it were we'dbe far more likely to see a false positive
329 // on the 32-bit CRC.
330 EncodeFixed32(buf + 7, static_cast<uint32_t>(log_number_));
331 crc = crc32c::Extend(crc, buf + 7, 4);
332 }
333
334 // Compute the crc of the record type and the payload.
335 uint32_t payload_crc = crc32c::Value(ptr, n);
336 crc = crc32c::Crc32cCombine(crc, payload_crc, n);
337 crc = crc32c::Mask(crc); // Adjust for storage
338 TEST_SYNC_POINT_CALLBACK("LogWriter::EmitPhysicalRecord:BeforeEncodeChecksum",
339 &crc);
340 EncodeFixed32(buf, crc);
341
342 // Write the header and the payload
343 IOOptions opts;
344 IOStatus s = WritableFileWriter::PrepareIOOptions(write_options, opts);
345 if (s.ok()) {
346 s = dest_->Append(opts, Slice(buf, header_size), 0 /* crc32c_checksum */);
347 }
348 if (s.ok()) {
349 s = dest_->Append(opts, Slice(ptr, n), payload_crc);
350 }
351 block_offset_ += header_size + n;
352 return s;
353}
