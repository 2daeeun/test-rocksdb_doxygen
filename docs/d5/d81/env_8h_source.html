83 // If true, then use mmap to read data.
84 // Not recommended for 32-bit OS.
85 bool use_mmap_reads = false;
86
87 // If true, then use mmap to write data
88 bool use_mmap_writes = true;
89
90 // If true, then use O_DIRECT for reading data
91 bool use_direct_reads = false;
92
93 // If true, then use O_DIRECT for writing data
94 bool use_direct_writes = false;
95
96 // If false, fallocate() calls are bypassed
97 bool allow_fallocate = true;
98
99 // If true, set the FD_CLOEXEC on open fd.
100 bool set_fd_cloexec = true;
101
102 // Allows OS to incrementally sync files to disk while they are being
103 // written, in the background. Issue one request for every bytes_per_sync
104 // written. 0 turns it off.
105 // Default: 0
106 uint64_t bytes_per_sync = 0;
107
108 // When true, guarantees the file has at most `bytes_per_sync` bytes submitted
109 // for writeback at any given time.
110 //
111 // - If `sync_file_range` is supported it achieves this by waiting for any
112 // prior `sync_file_range`s to finish before proceeding. In this way,
113 // processing (compression, etc.) can proceed uninhibited in the gap
114 // between `sync_file_range`s, and we block only when I/O falls behind.
115 // - Otherwise the `WritableFile::Sync` method is used. Note this mechanism
116 // always blocks, thus preventing the interleaving of I/O and processing.
117 //
118 // Note: Enabling this option does not provide any additional persistence
119 // guarantees, as it may use `sync_file_range`, which does not write out
120 // metadata.
121 //
122 // Default: false
123 bool strict_bytes_per_sync = false;
124
125 // If true, we will preallocate the file with FALLOC_FL_KEEP_SIZE flag, which
126 // means that file size won't change as part of preallocation.
127 // If false, preallocation will also change the file size. This option will
128 // improve the performance in workloads where you sync the data on every
129 // write. By default, we set it to true for MANIFEST writes and false for
130 // WAL writes
131 bool fallocate_with_keep_size = true;
132
133 // See DBOptions doc
134 size_t compaction_readahead_size = 0;
135
136 // See DBOptions doc
137 size_t random_access_max_buffer_size = 0;
138
139 // See DBOptions doc
140 size_t writable_file_max_buffer_size = 1024 * 1024;
141
142 // If not nullptr, write rate limiting is enabled for flush and compaction
143 RateLimiter* rate_limiter = nullptr;
144};

487 // Start a new thread, invoking "function(args...)" within the new thread.
488 // When "function(args...)" returns, the thread will be destroyed.
489 template <typename FunctionT, typename... Args>
490 void StartThreadTyped(FunctionT function, Args&&... args) {
491 using FWType = FunctorWrapper<Args...>;
492 StartThread(
493 [](void* arg) {
494 auto* functor = static_cast<FWType*>(arg);
495 functor->invoke();
496 delete functor;
497 },
