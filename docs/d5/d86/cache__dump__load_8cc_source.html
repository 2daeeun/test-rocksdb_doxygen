19IOStatus NewToFileCacheDumpWriter(const std::shared_ptr<FileSystem>& fs,
20 const FileOptions& file_opts,
21 const std::string& file_name,
22 std::unique_ptr<CacheDumpWriter>* writer) {
23 std::unique_ptr<WritableFileWriter> file_writer;
24 IOStatus io_s = WritableFileWriter::Create(fs, file_name, file_opts,
25 &file_writer, nullptr);
26 if (!io_s.ok()) {
27 return io_s;
28 }
29 writer->reset(new ToFileCacheDumpWriter(std::move(file_writer)));
30 return io_s;
