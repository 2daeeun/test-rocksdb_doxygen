28// Usage:
29// ForwardLevelIterator iter;
30// iter.SetFileIndex(file_index);
31// iter.Seek(target); // or iter.SeekToFirst();
32// iter.Next()
33class ForwardLevelIterator : public InternalIterator {
34 public:
35 ForwardLevelIterator(const ColumnFamilyData* const cfd,
36 const ReadOptions& read_options,
37 const std::vector<FileMetaData*>& files,
38 const MutableCFOptions& mutable_cf_options,
39 bool allow_unprepared_value)
40 : cfd_(cfd),
41 read_options_(read_options),
42 files_(files),
43 valid_(false),
44 file_index_(std::numeric_limits<uint32_t>::max()),
45 file_iter_(nullptr),
46 pinned_iters_mgr_(nullptr),
47 mutable_cf_options_(mutable_cf_options),
48 allow_unprepared_value_(allow_unprepared_value) {
49 status_.PermitUncheckedError(); // Allow uninitialized status through
50 }
51
52 ~ForwardLevelIterator() override {
53 // Reset current pointer
54 if (pinned_iters_mgr_ && pinned_iters_mgr_->PinningEnabled()) {
55 pinned_iters_mgr_->PinIterator(file_iter_);
56 } else {
57 delete file_iter_;
58 }
59 }
60
61 void SetFileIndex(uint32_t file_index) {
62 assert(file_index < files_.size());
63 status_ = Status::OK();
64 if (file_index != file_index_) {
65 file_index_ = file_index;
66 Reset();
67 }
68 }
69 void Reset() {
70 assert(file_index_ < files_.size());
71
72 // Reset current pointer
73 if (pinned_iters_mgr_ && pinned_iters_mgr_->PinningEnabled()) {
74 pinned_iters_mgr_->PinIterator(file_iter_);
75 } else {
76 delete file_iter_;
77 }
78
79 ReadRangeDelAggregator range_del_agg(&cfd_->internal_comparator(),
80 kMaxSequenceNumber /* upper_bound */);
81 file_iter_ = cfd_->table_cache()->NewIterator(
82 read_options_, *(cfd_->soptions()), cfd_->internal_comparator(),
83 *files_[file_index_],
84 read_options_.ignore_range_deletions ? nullptr : &range_del_agg,
85 mutable_cf_options_, /*table_reader_ptr=*/nullptr,
86 /*file_read_hist=*/nullptr, TableReaderCaller::kUserIterator,
87 /*arena=*/nullptr, /*skip_filters=*/false, /*level=*/-1,
88 /*max_file_size_for_l0_meta_pin=*/0,
89 /*smallest_compaction_key=*/nullptr,
90 /*largest_compaction_key=*/nullptr, allow_unprepared_value_);
91 file_iter_->SetPinnedItersMgr(pinned_iters_mgr_);
92 valid_ = false;
93 if (!range_del_agg.IsEmpty()) {
94 status_ = Status::NotSupported(
95 "Range tombstones unsupported with ForwardIterator");
96 }
97 }
98 void SeekToLast() override {
99 status_ = Status::NotSupported("ForwardLevelIterator::SeekToLast()");
100 valid_ = false;
101 }
102 void Prev() override {
103 status_ = Status::NotSupported("ForwardLevelIterator::Prev()");
104 valid_ = false;
105 }
106 bool Valid() const override { return valid_; }
107 void SeekToFirst() override {
108 assert(file_iter_ != nullptr);
109 if (!status_.ok()) {
110 assert(!valid_);
111 return;
112 }
113 file_iter_->SeekToFirst();
114 valid_ = file_iter_->Valid();
115 }
116 void Seek(const Slice& internal_key) override {
117 assert(file_iter_ != nullptr);
118
119 // This deviates from the usual convention for InternalIterator::Seek() in
120 // that it doesn't discard pre-existing error status. That's because this
121 // Seek() is only supposed to be called immediately after SetFileIndex()
122 // (which discards pre-existing error status), and SetFileIndex() may set
123 // an error status, which we shouldn't discard.
124 if (!status_.ok()) {
125 assert(!valid_);
126 return;
127 }
128
129 file_iter_->Seek(internal_key);
130 valid_ = file_iter_->Valid();
131 }
132 void SeekForPrev(const Slice& /*internal_key*/) override {
133 status_ = Status::NotSupported("ForwardLevelIterator::SeekForPrev()");
134 valid_ = false;
135 }
136 void Next() override {
137 assert(valid_);
138 file_iter_->Next();
139 for (;;) {
140 valid_ = file_iter_->Valid();
141 if (!file_iter_->status().ok()) {
142 assert(!valid_);
143 return;
144 }
145 if (valid_) {
146 return;
147 }
148 if (file_index_ + 1 >= files_.size()) {
149 valid_ = false;
150 return;
151 }
152 SetFileIndex(file_index_ + 1);
153 if (!status_.ok()) {
154 assert(!valid_);
155 return;
156 }
157 file_iter_->SeekToFirst();
158 }
159 }
160 Slice key() const override {
161 assert(valid_);
162 return file_iter_->key();
163 }
164 Slice value() const override {
165 assert(valid_);
166 return file_iter_->value();
167 }
168 uint64_t write_unix_time() const override {
169 assert(valid_);
170 return file_iter_->write_unix_time();
171 }
172 Status status() const override {
173 if (!status_.ok()) {
174 return status_;
175 } else if (file_iter_) {
176 return file_iter_->status();
177 }
178 return Status::OK();
179 }
180 bool PrepareValue() override {
181 assert(valid_);
182 if (file_iter_->PrepareValue()) {
183 return true;
184 }
185
186 assert(!file_iter_->Valid());
187 valid_ = false;
188 return false;
189 }
190 bool IsKeyPinned() const override {
191 return pinned_iters_mgr_ && pinned_iters_mgr_->PinningEnabled() &&
192 file_iter_->IsKeyPinned();
193 }
194 bool IsValuePinned() const override {
195 return pinned_iters_mgr_ && pinned_iters_mgr_->PinningEnabled() &&
196 file_iter_->IsValuePinned();
197 }
198 void SetPinnedItersMgr(PinnedIteratorsManager* pinned_iters_mgr) override {
199 pinned_iters_mgr_ = pinned_iters_mgr;
200 if (file_iter_) {
201 file_iter_->SetPinnedItersMgr(pinned_iters_mgr_);
202 }
203 }
204
205 private:
206 const ColumnFamilyData* const cfd_;
207 const ReadOptions& read_options_;
208 const std::vector<FileMetaData*>& files_;
209
210 bool valid_;
211 uint32_t file_index_;
212 Status status_;
213 InternalIterator* file_iter_;
214 PinnedIteratorsManager* pinned_iters_mgr_;
215 const MutableCFOptions& mutable_cf_options_;
216
217 const bool allow_unprepared_value_;
218};

258void ForwardIterator::SVCleanup(DBImpl* db, SuperVersion* sv,
259 bool background_purge_on_iterator_cleanup) {
260 if (sv->Unref()) {
261 // Job id == 0 means that this is not our background process, but rather
262 // user thread
263 JobContext job_context(0);
264 db->mutex_.Lock();
265 sv->Cleanup();
266 db->FindObsoleteFiles(&job_context, false, true);
267 if (background_purge_on_iterator_cleanup) {
268 db->ScheduleBgLogWriterClose(&job_context);
269 db->AddSuperVersionsToFreeQueue(sv);
270 db->SchedulePurge();
271 }
272 db->mutex_.Unlock();
273 if (!background_purge_on_iterator_cleanup) {
274 delete sv;
275 }
276 if (job_context.HaveSomethingToDelete()) {
277 db->PurgeObsoleteFiles(job_context, background_purge_on_iterator_cleanup);
278 }
279 job_context.Clean();
280 }
281}

299void ForwardIterator::SVCleanup() {
300 if (sv_ == nullptr) {
301 return;
302 }
303 bool background_purge =
304 read_options_.background_purge_on_iterator_cleanup ||
305 db_->immutable_db_options().avoid_unnecessary_blocking_io;
306 if (pinned_iters_mgr_ && pinned_iters_mgr_->PinningEnabled()) {
307 // pinned_iters_mgr_ tells us to make sure that all visited key-value slices
308 // are alive until pinned_iters_mgr_->ReleasePinnedData() is called.
309 // The slices may point into some memtables owned by sv_, so we need to keep
310 // sv_ referenced until pinned_iters_mgr_ unpins everything.
311 auto p = new SVCleanupParams{db_, sv_, background_purge};
312 pinned_iters_mgr_->PinPtr(p, &ForwardIterator::DeferredSVCleanup);
313 } else {
314 SVCleanup(db_, sv_, background_purge);
315 }
316}

318void ForwardIterator::Cleanup(bool release_sv) {
319 if (mutable_iter_ != nullptr) {
320 DeleteIterator(mutable_iter_, true /* is_arena */);
321 }
322
323 for (auto* m : imm_iters_) {
324 DeleteIterator(m, true /* is_arena */);
325 }
326 imm_iters_.clear();
327
328 for (auto* f : l0_iters_) {
329 DeleteIterator(f);
330 }
331 l0_iters_.clear();
332
333 for (auto* l : level_iters_) {
334 DeleteIterator(l);
335 }
336 level_iters_.clear();
337
338 if (release_sv) {
339 SVCleanup();
340 }
341}

366void ForwardIterator::Seek(const Slice& internal_key) {
367 if (sv_ == nullptr) {
368 RebuildIterators(true);
369 } else if (sv_->version_number != cfd_->GetSuperVersionNumber()) {
370 RenewIterators();
371 } else if (immutable_status_.IsIncomplete()) {
372 ResetIncompleteIterators();

381// In case of async_io, SeekInternal is called twice with seek_after_async_io
382// enabled in second call which only does seeking part to retrieve the blocks.
383void ForwardIterator::SeekInternal(const Slice& internal_key,
384 bool seek_to_first,
385 bool seek_after_async_io) {
386 assert(mutable_iter_);
387 // mutable
388 if (!seek_after_async_io) {
389 seek_to_first ? mutable_iter_->SeekToFirst()
390 : mutable_iter_->Seek(internal_key);
391 }
392
393 // immutable
394 // TODO(ljin): NeedToSeekImmutable has negative impact on performance
395 // if it turns to need to seek immutable often. We probably want to have
396 // an option to turn it off.
397 if (seek_to_first || seek_after_async_io ||
398 NeedToSeekImmutable(internal_key)) {
399 if (!seek_after_async_io) {
400 immutable_status_ = Status::OK();
401 if (has_iter_trimmed_for_upper_bound_ &&
402 (
403 // prev_ is not set yet
404 is_prev_set_ == false ||
405 // We are doing SeekToFirst() and internal_key.size() = 0
406 seek_to_first ||
407 // prev_key_ > internal_key
408 cfd_->internal_comparator().InternalKeyComparator::Compare(
409 prev_key_.GetInternalKey(), internal_key) > 0)) {
410 // Some iterators are trimmed. Need to rebuild.
411 RebuildIterators(true);
412 // Already seeked mutable iter, so seek again
413 seek_to_first ? mutable_iter_->SeekToFirst()
414 : mutable_iter_->Seek(internal_key);
415 }
416 {
417 auto tmp = MinIterHeap(MinIterComparator(&cfd_->internal_comparator()));
418 immutable_min_heap_.swap(tmp);
419 }
420 for (size_t i = 0; i < imm_iters_.size(); i++) {
421 auto* m = imm_iters_[i];
422 seek_to_first ? m->SeekToFirst() : m->Seek(internal_key);
423 if (!m->status().ok()) {
424 immutable_status_ = m->status();
425 } else if (m->Valid()) {
426 immutable_min_heap_.push(m);
427 }
428 }
429 }
430
431 Slice target_user_key;
432 if (!seek_to_first) {
433 target_user_key = ExtractUserKey(internal_key);
434 }
435 const VersionStorageInfo* vstorage = sv_->current->storage_info();
436 const std::vector<FileMetaData*>& l0 = vstorage->LevelFiles(0);
437 for (size_t i = 0; i < l0.size(); ++i) {
438 if (!l0_iters_[i]) {
439 continue;
440 }
441 if (seek_after_async_io) {
442 if (!l0_iters_[i]->status().IsTryAgain()) {
443 continue;
444 }
445 }
446
447 if (seek_to_first) {
448 l0_iters_[i]->SeekToFirst();
449 } else {
450 // If the target key passes over the largest key, we are sure Next()
451 // won't go over this file.
452 if (seek_after_async_io == false &&
453 user_comparator_->Compare(target_user_key,
454 l0[i]->largest.user_key()) > 0) {
455 if (read_options_.iterate_upper_bound != nullptr) {
456 has_iter_trimmed_for_upper_bound_ = true;
457 DeleteIterator(l0_iters_[i]);
458 l0_iters_[i] = nullptr;
459 }
460 continue;
461 }
462 l0_iters_[i]->Seek(internal_key);
463 }
464
465 if (l0_iters_[i]->status().IsTryAgain()) {
466 assert(!seek_after_async_io);
467 continue;
468 } else if (!l0_iters_[i]->status().ok()) {
469 immutable_status_ = l0_iters_[i]->status();
470 } else if (l0_iters_[i]->Valid() &&
471 !IsOverUpperBound(l0_iters_[i]->key())) {
472 immutable_min_heap_.push(l0_iters_[i]);
473 } else {
474 has_iter_trimmed_for_upper_bound_ = true;
475 DeleteIterator(l0_iters_[i]);
476 l0_iters_[i] = nullptr;
477 }
478 }
479
480 for (int32_t level = 1; level < vstorage->num_levels(); ++level) {
481 const std::vector<FileMetaData*>& level_files =
482 vstorage->LevelFiles(level);
483 if (level_files.empty()) {
484 continue;
485 }
486 if (level_iters_[level - 1] == nullptr) {
487 continue;
488 }
489
490 if (seek_after_async_io) {
491 if (!level_iters_[level - 1]->status().IsTryAgain()) {
492 continue;
493 }
494 }
495 uint32_t f_idx = 0;
496 if (!seek_to_first && !seek_after_async_io) {
497 f_idx = FindFileInRange(level_files, internal_key, 0,
498 static_cast<uint32_t>(level_files.size()));
499 }
500
501 // Seek
502 if (seek_after_async_io || f_idx < level_files.size()) {
503 if (!seek_after_async_io) {
504 level_iters_[level - 1]->SetFileIndex(f_idx);
505 }
506 seek_to_first ? level_iters_[level - 1]->SeekToFirst()
507 : level_iters_[level - 1]->Seek(internal_key);
508
509 if (level_iters_[level - 1]->status().IsTryAgain()) {
510 assert(!seek_after_async_io);
511 continue;
512 } else if (!level_iters_[level - 1]->status().ok()) {
513 immutable_status_ = level_iters_[level - 1]->status();
514 } else if (level_iters_[level - 1]->Valid() &&
515 !IsOverUpperBound(level_iters_[level - 1]->key())) {
516 immutable_min_heap_.push(level_iters_[level - 1]);
517 } else {
518 // Nothing in this level is interesting. Remove.
519 has_iter_trimmed_for_upper_bound_ = true;
520 DeleteIterator(level_iters_[level - 1]);
521 level_iters_[level - 1] = nullptr;
522 }
523 }
524 }
525
526 if (seek_to_first) {
527 is_prev_set_ = false;
528 } else {
529 prev_key_.SetInternalKey(internal_key);
530 is_prev_set_ = true;
531 is_prev_inclusive_ = true;
532 }
533
534 TEST_SYNC_POINT_CALLBACK("ForwardIterator::SeekInternal:Immutable", this);
535 } else if (current_ && current_ != mutable_iter_) {
536 // current_ is one of immutable iterators, push it back to the heap
537 immutable_min_heap_.push(current_);
538 }
539
540 // For async_io, it should be updated when seek_after_async_io is true (in
541 // second call).
542 if (seek_to_first || !read_options_.async_io || seek_after_async_io) {
543 UpdateCurrent();
544 }
545 TEST_SYNC_POINT_CALLBACK("ForwardIterator::SeekInternal:Return", this);
546}

548void ForwardIterator::Next() {
549 assert(valid_);
550 bool update_prev_key = false;
551
552 if (sv_ == nullptr || sv_->version_number != cfd_->GetSuperVersionNumber()) {
553 std::string current_key = key().ToString();
554 Slice old_key(current_key.data(), current_key.size());
555
556 if (sv_ == nullptr) {
557 RebuildIterators(true);
558 } else {
559 RenewIterators();
560 }
561
562 SeekInternal(old_key, false, false);
563 if (read_options_.async_io) {
564 SeekInternal(old_key, false, true);
565 }
566
567 if (!valid_ || key().compare(old_key) != 0) {
568 return;
569 }
570 } else if (current_ != mutable_iter_) {
571 // It is going to advance immutable iterator
572
573 if (is_prev_set_ && prefix_extractor_) {
574 // advance prev_key_ to current_ only if they share the same prefix
575 update_prev_key =
576 prefix_extractor_->Transform(prev_key_.GetUserKey())
577 .compare(prefix_extractor_->Transform(current_->key())) == 0;
578 } else {
579 update_prev_key = true;
580 }
581
582 if (update_prev_key) {
583 prev_key_.SetInternalKey(current_->key());
584 is_prev_set_ = true;
585 is_prev_inclusive_ = false;
586 }
587 }
588
589 current_->Next();
590 if (current_ != mutable_iter_) {
591 if (!current_->status().ok()) {
592 immutable_status_ = current_->status();
593 } else if ((current_->Valid()) && (!IsOverUpperBound(current_->key()))) {
594 immutable_min_heap_.push(current_);
595 } else {
596 if ((current_->Valid()) && (IsOverUpperBound(current_->key()))) {
597 // remove the current iterator
598 DeleteCurrentIter();
599 current_ = nullptr;
600 }
601 if (update_prev_key) {
602 mutable_iter_->Seek(prev_key_.GetInternalKey());
603 }
604 }
605 }
606 UpdateCurrent();
607 TEST_SYNC_POINT_CALLBACK("ForwardIterator::Next:Return", this);
608}

666void ForwardIterator::UpdateChildrenPinnedItersMgr() {
667 // Set PinnedIteratorsManager for mutable memtable iterator.
668 if (mutable_iter_) {
669 mutable_iter_->SetPinnedItersMgr(pinned_iters_mgr_);
670 }
671
672 // Set PinnedIteratorsManager for immutable memtable iterators.
673 for (InternalIterator* child_iter : imm_iters_) {
674 if (child_iter) {
675 child_iter->SetPinnedItersMgr(pinned_iters_mgr_);

680 for (InternalIterator* child_iter : l0_iters_) {

682 child_iter->SetPinnedItersMgr(pinned_iters_mgr_);
683 }
684 }
685
686 // Set PinnedIteratorsManager for L1+ levels iterators.
687 for (ForwardLevelIterator* child_iter : level_iters_) {
688 if (child_iter) {
689 child_iter->SetPinnedItersMgr(pinned_iters_mgr_);
690 }
691 }

704void ForwardIterator::RebuildIterators(bool refresh_sv) {
705 // Clean up
706 Cleanup(refresh_sv);
707 if (refresh_sv) {
708 // New
709 sv_ = cfd_->GetReferencedSuperVersion(db_);
710 }
711 ReadRangeDelAggregator range_del_agg(&cfd_->internal_comparator(),
712 kMaxSequenceNumber /* upper_bound */);
713 UnownedPtr<const SeqnoToTimeMapping> seqno_to_time_mapping =
714 sv_->GetSeqnoToTimeMapping();
715 mutable_iter_ =
716 sv_->mem->NewIterator(read_options_, seqno_to_time_mapping, &arena_,
717 sv_->mutable_cf_options.prefix_extractor.get(),
718 /*for_flush=*/false);

771void ForwardIterator::RenewIterators() {
772 SuperVersion* svnew;
773 assert(sv_);
774 svnew = cfd_->GetReferencedSuperVersion(db_);
775
776 if (mutable_iter_ != nullptr) {
777 DeleteIterator(mutable_iter_, true /* is_arena */);
778 }
779 for (auto* m : imm_iters_) {
780 DeleteIterator(m, true /* is_arena */);
781 }
782 imm_iters_.clear();
783
784 UnownedPtr<const SeqnoToTimeMapping> seqno_to_time_mapping =
785 svnew->GetSeqnoToTimeMapping();
786 mutable_iter_ =
787 svnew->mem->NewIterator(read_options_, seqno_to_time_mapping, &arena_,
788 svnew->mutable_cf_options.prefix_extractor.get(),
789 /*for_flush=*/false);
790 svnew->imm->AddIterators(read_options_, seqno_to_time_mapping,
791 svnew->mutable_cf_options.prefix_extractor.get(),
792 &imm_iters_, &arena_);
793 ReadRangeDelAggregator range_del_agg(&cfd_->internal_comparator(),
794 kMaxSequenceNumber /* upper_bound */);
795 if (!read_options_.ignore_range_deletions) {
796 std::unique_ptr<FragmentedRangeTombstoneIterator> range_del_iter(
797 svnew->mem->NewRangeTombstoneIterator(
798 read_options_, sv_->current->version_set()->LastSequence(),
799 false /* immutable_memtable */));
800 range_del_agg.AddTombstones(std::move(range_del_iter));
801 // Always return Status::OK().
802 Status temp_s = svnew->imm->AddRangeTombstoneIterators(
803 read_options_, &arena_, &range_del_agg);
804 assert(temp_s.ok());
805 }
806
807 const auto* vstorage = sv_->current->storage_info();
808 const auto& l0_files = vstorage->LevelFiles(0);
809 const auto* vstorage_new = svnew->current->storage_info();
810 const auto& l0_files_new = vstorage_new->LevelFiles(0);
811 size_t iold, inew;
812 bool found;
813 std::vector<InternalIterator*> l0_iters_new;
814 l0_iters_new.reserve(l0_files_new.size());
815
816 for (inew = 0; inew < l0_files_new.size(); inew++) {
817 found = false;
818 for (iold = 0; iold < l0_files.size(); iold++) {
819 if (l0_files[iold] == l0_files_new[inew]) {
820 found = true;
821 break;
822 }
823 }
824 if (found) {
825 if (l0_iters_[iold] == nullptr) {
826 l0_iters_new.push_back(nullptr);
827 TEST_SYNC_POINT_CALLBACK("ForwardIterator::RenewIterators:Null", this);
828 } else {
829 l0_iters_new.push_back(l0_iters_[iold]);
830 l0_iters_[iold] = nullptr;
831 TEST_SYNC_POINT_CALLBACK("ForwardIterator::RenewIterators:Copy", this);
832 }
833 continue;
834 }
835 l0_iters_new.push_back(cfd_->table_cache()->NewIterator(
836 read_options_, *cfd_->soptions(), cfd_->internal_comparator(),
837 *l0_files_new[inew],
838 read_options_.ignore_range_deletions ? nullptr : &range_del_agg,
839 svnew->mutable_cf_options,
840 /*table_reader_ptr=*/nullptr, /*file_read_hist=*/nullptr,
841 TableReaderCaller::kUserIterator, /*arena=*/nullptr,
842 /*skip_filters=*/false, /*level=*/-1,
843 MaxFileSizeForL0MetaPin(svnew->mutable_cf_options),
844 /*smallest_compaction_key=*/nullptr,
845 /*largest_compaction_key=*/nullptr, allow_unprepared_value_));
846 }
847
848 for (auto* f : l0_iters_) {
849 DeleteIterator(f);
850 }
851 l0_iters_.clear();
852 l0_iters_ = l0_iters_new;
853
854 for (auto* l : level_iters_) {
855 DeleteIterator(l);
856 }
857 level_iters_.clear();
858 BuildLevelIterators(vstorage_new, svnew);
859 current_ = nullptr;
860 is_prev_set_ = false;
861 SVCleanup();
862 sv_ = svnew;
863
864 UpdateChildrenPinnedItersMgr();
865 if (!range_del_agg.IsEmpty()) {
866 status_ = Status::NotSupported(
867 "Range tombstones unsupported with ForwardIterator");
868 valid_ = false;
869 }
870}

924void ForwardIterator::UpdateCurrent() {
925 if (immutable_min_heap_.empty() && !mutable_iter_->Valid()) {
926 current_ = nullptr;
927 } else if (immutable_min_heap_.empty()) {
928 current_ = mutable_iter_;
929 } else if (!mutable_iter_->Valid()) {
930 current_ = immutable_min_heap_.top();
931 immutable_min_heap_.pop();
932 } else {
933 current_ = immutable_min_heap_.top();
934 assert(current_ != nullptr);
935 assert(current_->Valid());
936 int cmp = cfd_->internal_comparator().InternalKeyComparator::Compare(
937 mutable_iter_->key(), current_->key());
938 assert(cmp != 0);
939 if (cmp > 0) {
940 immutable_min_heap_.pop();
941 } else {
942 current_ = mutable_iter_;
943 }
944 }
945 valid_ = current_ != nullptr && immutable_status_.ok();
946 if (!status_.ok()) {
947 status_ = Status::OK();
948 }
949
950 // Upper bound doesn't apply to the memtable iterator. We want Valid() to
951 // return false when all iterators are over iterate_upper_bound, but can't
952 // just set valid_ to false, as that would effectively disable the tailing
953 // optimization (Seek() would be called on all immutable iterators regardless
954 // of whether the target key is greater than prev_key_).
955 current_over_upper_bound_ = valid_ && IsOverUpperBound(current_->key());
956}

958bool ForwardIterator::NeedToSeekImmutable(const Slice& target) {
959 // We maintain the interval (prev_key_, immutable_min_heap_.top()->key())
960 // such that there are no records with keys within that range in
961 // immutable_min_heap_. Since immutable structures (SST files and immutable
962 // memtables) can't change in this version, we don't need to do a seek if
963 // 'target' belongs to that interval (immutable_min_heap_.top() is already
964 // at the correct position).
965
966 if (!valid_ || !current_ || !is_prev_set_ || !immutable_status_.ok()) {
967 return true;
968 }
969 Slice prev_key = prev_key_.GetInternalKey();
970 if (prefix_extractor_ && prefix_extractor_->Transform(target).compare(
971 prefix_extractor_->Transform(prev_key)) != 0) {
972 return true;
973 }
974 if (cfd_->internal_comparator().InternalKeyComparator::Compare(
975 prev_key, target) >= (is_prev_inclusive_ ? 1 : 0)) {
976 return true;
977 }
978
979 if (immutable_min_heap_.empty() && current_ == mutable_iter_) {
980 // Nothing to seek on.
981 return false;
982 }
983 if (cfd_->internal_comparator().InternalKeyComparator::Compare(
984 target, current_ == mutable_iter_ ? immutable_min_heap_.top()->key()
985 : current_->key()) > 0) {
986 return true;
987 }
988 return false;
989}

991void ForwardIterator::DeleteCurrentIter() {
992 const VersionStorageInfo* vstorage = sv_->current->storage_info();
993 const std::vector<FileMetaData*>& l0 = vstorage->LevelFiles(0);
994 for (size_t i = 0; i < l0.size(); ++i) {
995 if (!l0_iters_[i]) {
996 continue;
997 }

999 has_iter_trimmed_for_upper_bound_ = true;
1000 DeleteIterator(l0_iters_[i]);
1001 l0_iters_[i] = nullptr;
1002 return;
1003 }
1004 }
1005
1006 for (int32_t level = 1; level < vstorage->num_levels(); ++level) {
1007 if (level_iters_[level - 1] == nullptr) {
1008 continue;
1009 }
1010 if (level_iters_[level - 1] == current_) {
1011 has_iter_trimmed_for_upper_bound_ = true;
1012 DeleteIterator(level_iters_[level - 1]);
1013 level_iters_[level - 1] = nullptr;
1014 }
1015 }

1018bool ForwardIterator::TEST_CheckDeletedIters(int* pdeleted_iters,
1019 int* pnum_iters) {
1020 bool retval = false;
1021 int deleted_iters = 0;
1022 int num_iters = 0;
1023
1024 const VersionStorageInfo* vstorage = sv_->current->storage_info();

1028 retval = true;
1029 deleted_iters++;
1030 } else {
1031 num_iters++;
1032 }
1033 }
1034
1035 for (int32_t level = 1; level < vstorage->num_levels(); ++level) {
1036 if ((level_iters_[level - 1] == nullptr) &&
1037 (!vstorage->LevelFiles(level).empty())) {
1038 retval = true;
1039 deleted_iters++;
1040 } else if (!vstorage->LevelFiles(level).empty()) {
1041 num_iters++;
1042 }
1043 }
1044 if ((!retval) && num_iters <= 1) {
1045 retval = true;
1046 }
1047 if (pdeleted_iters) {
1048 *pdeleted_iters = deleted_iters;
1049 }
1050 if (pnum_iters) {
1051 *pnum_iters = num_iters;
1052 }
1053 return retval;
