14namespace ROCKSDB_NAMESPACE {
15// During recovery if the memtable is flushed we cannot rely on its help on
16// duplicate key detection and as key insert will not be attempted. This class
17// will be used as a emulator of memtable to tell if insertion of a key/seq
18// would have resulted in duplication.
19class DuplicateDetector {
20 public:
21 explicit DuplicateDetector(DBImpl* db) : db_(db) {}
22 bool IsDuplicateKeySeq(uint32_t cf, const Slice& key, SequenceNumber seq) {
23 assert(seq >= batch_seq_);
24 if (batch_seq_ != seq) { // it is a new batch
25 keys_.clear();
26 }
27 batch_seq_ = seq;
28 CFKeys& cf_keys = keys_[cf];
29 if (cf_keys.size() == 0) { // just inserted
30 InitWithComp(cf);
31 }
32 auto it = cf_keys.insert(key);
33 if (it.second == false) { // second is false if a element already existed.
34 keys_.clear();
35 InitWithComp(cf);
36 keys_[cf].insert(key);
37 return true;
38 }
39 return false;
40 }
41
42 private:
43 SequenceNumber batch_seq_ = 0;
44 DBImpl* db_;
45 using CFKeys = std::set<Slice, SetComparator>;
46 std::map<uint32_t, CFKeys> keys_;
47 void InitWithComp(const uint32_t cf) {
48 auto h = db_->GetColumnFamilyHandle(cf);
49 if (!h) {
50 // TODO(myabandeh): This is not a concern in MyRocks as drop cf is not
51 // implemented yet. When it does, we should return proper error instead
52 // of throwing exception.
53 ROCKS_LOG_FATAL(
54 db_->immutable_db_options().info_log,
55 "Recovering an entry from the dropped column family %" PRIu32
56 ". WAL must must have been emptied before dropping the column "
57 "family",
58 cf);
59 throw std::runtime_error(
60 "Recovering an entry from a dropped column family. "
61 "WAL must must have been flushed before dropping the column "
62 "family");
63 return;
64 }
65 auto cmp = h->GetComparator();
66 keys_[cf] = CFKeys(SetComparator(cmp));
67 }
68};
69} // namespace ROCKSDB_NAMESPACE
