41class BlockCacheTier : public PersistentCacheTier {
42 public:
43 explicit BlockCacheTier(const PersistentCacheConfig& opt)
44 : opt_(opt),
45 insert_ops_(static_cast<size_t>(opt_.max_write_pipeline_backlog_size)),
46 buffer_allocator_(opt.write_buffer_size, opt.write_buffer_count()),
47 writer_(this, opt_.writer_qdepth,
48 static_cast<size_t>(opt_.writer_dispatch_size)) {
49 Info(opt_.log, "Initializing allocator. size=%d B count=%" ROCKSDB_PRIszt,
50 opt_.write_buffer_size, opt_.write_buffer_count());
51 }
52
53 virtual ~BlockCacheTier() {
54 // Close is re-entrant so we can call close even if it is already closed
55 Close().PermitUncheckedError();
56 assert(!insert_th_.joinable());
57 }

71 PersistentCache::StatsType Stats() override;

73 void TEST_Flush() override {
74 while (insert_ops_.Size()) {
75 /* sleep override */
76 SystemClock::Default()->SleepForMicroseconds(1000000);
77 }
78 }

86 // Pipelined operation
87 struct InsertOp {

93 InsertOp() = delete;

100 std::string key_;

103 };

127 std::atomic<uint64_t> insert_dropped_{0};
128
129 double CacheHitPct() const {
130 const auto lookups = cache_hits_ + cache_misses_;
131 return lookups ? 100 * cache_hits_ / static_cast<double>(lookups) : 0.0;
132 }
133
134 double CacheMissPct() const {
135 const auto lookups = cache_hits_ + cache_misses_;
136 return lookups ? 100 * cache_misses_ / static_cast<double>(lookups) : 0.0;
137 }
