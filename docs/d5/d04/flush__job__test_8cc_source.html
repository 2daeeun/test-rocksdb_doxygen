207TEST_F(FlushJobTest, NonEmpty) {
208 JobContext job_context(0);
209 auto cfd = versions_->GetColumnFamilySet()->GetDefault();
210 auto new_mem = cfd->ConstructNewMemtable(*cfd->GetLatestMutableCFOptions(),
211 kMaxSequenceNumber);
212 new_mem->Ref();
213 auto inserted_keys = mock::MakeMockFile();
214 // Test data:
215 // seqno [ 1, 2 ... 8998, 8999, 9000, 9001, 9002 ... 9999 ]
216 // key [ 1001, 1002 ... 9998, 9999, 0, 1, 2 ... 999 ]
217 // range-delete "9995" -> "9999" at seqno 10000
218 // blob references with seqnos 10001..10006
219 for (int i = 1; i < 10000; ++i) {
220 std::string key(std::to_string((i + 1000) % 10000));
221 std::string value("value" + key);
222 ASSERT_OK(new_mem->Add(SequenceNumber(i), kTypeValue, key, value,
223 nullptr /* kv_prot_info */));
224 if ((i + 1000) % 10000 < 9995) {
225 InternalKey internal_key(key, SequenceNumber(i), kTypeValue);
226 inserted_keys.push_back({internal_key.Encode().ToString(), value});
227 }
228 }
229
230 {
231 ASSERT_OK(new_mem->Add(SequenceNumber(10000), kTypeRangeDeletion, "9995",
232 "9999a", nullptr /* kv_prot_info */));
233 InternalKey internal_key("9995", SequenceNumber(10000), kTypeRangeDeletion);
234 inserted_keys.push_back({internal_key.Encode().ToString(), "9999a"});
235 }
236
237 // Note: the first two blob references will not be considered when resolving
238 // the oldest blob file referenced (the first one is inlined TTL, while the
239 // second one is TTL and thus points to a TTL blob file).
240 constexpr std::array<uint64_t, 6> blob_file_numbers{
241 {kInvalidBlobFileNumber, 5, 103, 17, 102, 101}};
242 for (size_t i = 0; i < blob_file_numbers.size(); ++i) {
243 std::string key(std::to_string(i + 10001));
244 std::string blob_index;
245 if (i == 0) {
246 BlobIndex::EncodeInlinedTTL(&blob_index, /* expiration */ 1234567890ULL,
247 "foo");
248 } else if (i == 1) {
249 BlobIndex::EncodeBlobTTL(&blob_index, /* expiration */ 1234567890ULL,
250 blob_file_numbers[i], /* offset */ i << 10,
251 /* size */ i << 20, kNoCompression);
252 } else {
253 BlobIndex::EncodeBlob(&blob_index, blob_file_numbers[i],
254 /* offset */ i << 10, /* size */ i << 20,
255 kNoCompression);
256 }
257
258 const SequenceNumber seq(i + 10001);
259 ASSERT_OK(new_mem->Add(seq, kTypeBlobIndex, key, blob_index,
260 nullptr /* kv_prot_info */));
261
262 InternalKey internal_key(key, seq, kTypeBlobIndex);
263 inserted_keys.push_back({internal_key.Encode().ToString(), blob_index});
264 }
265 mock::SortKVVector(&inserted_keys);
266
267 autovector<ReadOnlyMemTable*> to_delete;
268 new_mem->ConstructFragmentedRangeTombstones();
269 cfd->imm()->Add(new_mem, &to_delete);
270 for (auto& m : to_delete) {
271 delete m;
272 }
273
274 EventLogger event_logger(db_options_.info_log.get());
275 SnapshotChecker* snapshot_checker = nullptr; // not relavant
276 FlushJob flush_job(
277 dbname_, versions_->GetColumnFamilySet()->GetDefault(), db_options_,
278 *cfd->GetLatestMutableCFOptions(),
279 std::numeric_limits<uint64_t>::max() /* memtable_id */, env_options_,
280 versions_.get(), &mutex_, &shutting_down_, {}, kMaxSequenceNumber,
281 snapshot_checker, &job_context, FlushReason::kTest, nullptr, nullptr,
282 nullptr, kNoCompression, db_options_.statistics.get(), &event_logger,
283 true, true /* sync_output_directory */, true /* write_manifest */,
284 Env::Priority::USER, nullptr /*IOTracer*/, empty_seqno_to_time_mapping_);
285
286 HistogramData hist;
287 FileMetaData file_meta;
288 mutex_.Lock();
289 flush_job.PickMemTable();
290 ASSERT_OK(flush_job.Run(nullptr, &file_meta));
291 mutex_.Unlock();
292 db_options_.statistics->histogramData(FLUSH_TIME, &hist);
293 ASSERT_GT(hist.average, 0.0);
294
295 ASSERT_EQ(std::to_string(0), file_meta.smallest.user_key().ToString());
296 ASSERT_EQ("9999a", file_meta.largest.user_key().ToString());
297 ASSERT_EQ(1, file_meta.fd.smallest_seqno);
298 ASSERT_EQ(10006, file_meta.fd.largest_seqno);
299 ASSERT_EQ(17, file_meta.oldest_blob_file_number);
300 mock_table_factory_->AssertSingleFile(inserted_keys);
301 job_context.Clean();
302}

371TEST_F(FlushJobTest, FlushMemtablesMultipleColumnFamilies) {
372 autovector<ColumnFamilyData*> all_cfds;
373 for (auto cfd : *versions_->GetColumnFamilySet()) {
374 all_cfds.push_back(cfd);
375 }
376 const std::vector<size_t> num_memtables = {2, 1, 3};
377 assert(num_memtables.size() == column_family_names_.size());
378 const size_t num_keys_per_memtable = 1000;
379 JobContext job_context(0);
380 std::vector<uint64_t> memtable_ids;
381 std::vector<SequenceNumber> smallest_seqs;
382 std::vector<SequenceNumber> largest_seqs;
383 autovector<ReadOnlyMemTable*> to_delete;
384 SequenceNumber curr_seqno = 0;
385 size_t k = 0;
386 for (auto cfd : all_cfds) {
387 smallest_seqs.push_back(curr_seqno);
388 for (size_t i = 0; i != num_memtables[k]; ++i) {
389 MemTable* mem = cfd->ConstructNewMemtable(
390 *cfd->GetLatestMutableCFOptions(), kMaxSequenceNumber);
391 mem->SetID(i);
392 mem->Ref();
393
394 for (size_t j = 0; j != num_keys_per_memtable; ++j) {
395 std::string key(std::to_string(j + i * num_keys_per_memtable));
396 std::string value("value" + key);
397 ASSERT_OK(mem->Add(curr_seqno++, kTypeValue, key, value,
398 nullptr /* kv_prot_info */));
399 }
400 mem->ConstructFragmentedRangeTombstones();
401 cfd->imm()->Add(mem, &to_delete);
402 }
403 largest_seqs.push_back(curr_seqno - 1);
404 memtable_ids.push_back(num_memtables[k++] - 1);
405 }
406
407 EventLogger event_logger(db_options_.info_log.get());
408 SnapshotChecker* snapshot_checker = nullptr; // not relevant
409 std::vector<std::unique_ptr<FlushJob>> flush_jobs;
410 k = 0;
411 for (auto cfd : all_cfds) {
412 std::vector<SequenceNumber> snapshot_seqs;
413 flush_jobs.emplace_back(new FlushJob(
414 dbname_, cfd, db_options_, *cfd->GetLatestMutableCFOptions(),
415 memtable_ids[k], env_options_, versions_.get(), &mutex_,
416 &shutting_down_, snapshot_seqs, kMaxSequenceNumber, snapshot_checker,

420 Env::Priority::USER, nullptr /*IOTracer*/,
421 empty_seqno_to_time_mapping_));
422 k++;
423 }
424 HistogramData hist;
425 std::vector<FileMetaData> file_metas;
426 // Call reserve to avoid auto-resizing
427 file_metas.reserve(flush_jobs.size());
428 mutex_.Lock();
429 for (auto& job : flush_jobs) {
430 job->PickMemTable();
431 }
432 for (auto& job : flush_jobs) {
433 FileMetaData meta;
434 // Run will release and re-acquire mutex
435 ASSERT_OK(job->Run(nullptr /**/, &meta));
436 file_metas.emplace_back(meta);
437 }
438 autovector<FileMetaData*> file_meta_ptrs;
439 for (auto& meta : file_metas) {
440 file_meta_ptrs.push_back(&meta);
441 }
442 autovector<const autovector<ReadOnlyMemTable*>*> mems_list;
443 for (size_t i = 0; i != all_cfds.size(); ++i) {

446 }

492 auto new_mem = cfd->ConstructNewMemtable(*cfd->GetLatestMutableCFOptions(),
493 kMaxSequenceNumber);
494
495 std::set<SequenceNumber> snapshots_set;
496 int keys = 10000;
497 int max_inserts_per_keys = 8;
498
499 Random rnd(301);
500 for (int i = 0; i < keys / 2; ++i) {
501 snapshots_set.insert(rnd.Uniform(keys * (max_inserts_per_keys / 2)) + 1);
502 }
503 // set has already removed the duplicate snapshots
504 std::vector<SequenceNumber> snapshots(snapshots_set.begin(),
505 snapshots_set.end());
506
507 new_mem->Ref();
508 SequenceNumber current_seqno = 0;
509 auto inserted_keys = mock::MakeMockFile();
510 for (int i = 1; i < keys; ++i) {
511 std::string key(std::to_string(i));
512 int insertions = rnd.Uniform(max_inserts_per_keys);
513 for (int j = 0; j < insertions; ++j) {
514 std::string value(rnd.HumanReadableString(10));
515 auto seqno = ++current_seqno;
516 ASSERT_OK(new_mem->Add(SequenceNumber(seqno), kTypeValue, key, value,
517 nullptr /* kv_prot_info */));
518 // a key is visible only if:
519 // 1. it's the last one written (j == insertions - 1)
520 // 2. there's a snapshot pointing at it
521 bool visible = (j == insertions - 1) ||
522 (snapshots_set.find(seqno) != snapshots_set.end());
523 if (visible) {
524 InternalKey internal_key(key, seqno, kTypeValue);
525 inserted_keys.push_back({internal_key.Encode().ToString(), value});
526 }
527 }
528 }
529 mock::SortKVVector(&inserted_keys);
530
531 autovector<ReadOnlyMemTable*> to_delete;
532 new_mem->ConstructFragmentedRangeTombstones();
533 cfd->imm()->Add(new_mem, &to_delete);
534 for (auto& m : to_delete) {
535 delete m;
536 }
537
538 EventLogger event_logger(db_options_.info_log.get());
539 SnapshotChecker* snapshot_checker = nullptr; // not relavant
540 FlushJob flush_job(
541 dbname_, versions_->GetColumnFamilySet()->GetDefault(), db_options_,
542 *cfd->GetLatestMutableCFOptions(),
543 std::numeric_limits<uint64_t>::max() /* memtable_id */, env_options_,
544 versions_.get(), &mutex_, &shutting_down_, snapshots, kMaxSequenceNumber,
545 snapshot_checker, &job_context, FlushReason::kTest, nullptr, nullptr,
546 nullptr, kNoCompression, db_options_.statistics.get(), &event_logger,
547 true, true /* sync_output_directory */, true /* write_manifest */,
548 Env::Priority::USER, nullptr /*IOTracer*/, empty_seqno_to_time_mapping_);
549 mutex_.Lock();
550 flush_job.PickMemTable();
551 ASSERT_OK(flush_job.Run());
552 mutex_.Unlock();
553 mock_table_factory_->AssertSingleFile(inserted_keys);
554 HistogramData hist;
555 db_options_.statistics->histogramData(FLUSH_TIME, &hist);
556 ASSERT_GT(hist.average, 0.0);
557 job_context.Clean();
