16// Iterator over a vector of keys/values
17class VectorIterator : public InternalIterator {
18 public:
19 VectorIterator(std::vector<std::string> keys, std::vector<std::string> values,
20 const CompareInterface* icmp = nullptr)
21 : keys_(std::move(keys)),
22 values_(std::move(values)),
23 current_(keys_.size()),
24 indexed_cmp_(icmp, &keys_) {
25 assert(keys_.size() == values_.size());
26
27 indices_.reserve(keys_.size());
28 for (size_t i = 0; i < keys_.size(); i++) {
29 indices_.push_back(i);
30 }
31 if (icmp != nullptr) {
32 std::sort(indices_.begin(), indices_.end(), indexed_cmp_);
33 }
34 }
35
36 bool Valid() const override {
37 return !indices_.empty() && current_ < indices_.size();
38 }
39
40 void SeekToFirst() override { current_ = 0; }
41 void SeekToLast() override { current_ = indices_.size() - 1; }
42
43 void Seek(const Slice& target) override {
44 if (indexed_cmp_.cmp != nullptr) {
45 current_ = std::lower_bound(indices_.begin(), indices_.end(), target,
46 indexed_cmp_) -
47 indices_.begin();
48 } else {
49 current_ =
50 std::lower_bound(keys_.begin(), keys_.end(), target.ToString()) -
51 keys_.begin();
52 }
53 }
54
55 void SeekForPrev(const Slice& target) override {
56 if (indexed_cmp_.cmp != nullptr) {
57 current_ = std::upper_bound(indices_.begin(), indices_.end(), target,
58 indexed_cmp_) -
59 indices_.begin();
60 } else {
61 current_ =
62 std::upper_bound(keys_.begin(), keys_.end(), target.ToString()) -
63 keys_.begin();
64 }
65 if (!Valid()) {
66 SeekToLast();
67 } else {
68 Prev();
69 }
70 }
71
72 void Next() override { current_++; }
73 void Prev() override { current_--; }
74
75 Slice key() const override { return Slice(keys_[indices_[current_]]); }
76 Slice value() const override { return Slice(values_[indices_[current_]]); }
77
78 Status status() const override { return Status::OK(); }
79
80 bool IsKeyPinned() const override { return true; }
81 bool IsValuePinned() const override { return true; }
82
83 protected:
84 std::vector<std::string> keys_;
85 std::vector<std::string> values_;
86 size_t current_;
87
88 private:
89 struct IndexedKeyComparator {
90 IndexedKeyComparator(const CompareInterface* c,
91 const std::vector<std::string>* ks)
92 : cmp(c), keys(ks) {}
93
94 bool operator()(size_t a, size_t b) const {

96 }
97

100 }
101
102 bool operator()(const Slice& a, size_t b) const {
103 return cmp->Compare(a, (*keys)[b]) < 0;
104 }
105
106 const CompareInterface* cmp;
107 const std::vector<std::string>* keys;
108 };
