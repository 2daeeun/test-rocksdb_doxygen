21// Snapshots are kept in a doubly-linked list in the DB.
22// Each SnapshotImpl corresponds to a particular sequence number.
23class SnapshotImpl : public Snapshot {
24 public:
25 SequenceNumber number_; // const after creation
26 // It indicates the smallest uncommitted data at the time the snapshot was
27 // taken. This is currently used by WritePrepared transactions to limit the
28 // scope of queries to IsInSnapshot.
29 SequenceNumber min_uncommitted_ = kMinUnCommittedSeq;
30
31 SequenceNumber GetSequenceNumber() const override { return number_; }
32
33 int64_t GetUnixTime() const override { return unix_time_; }
34
35 uint64_t GetTimestamp() const override { return timestamp_; }
36
37 private:
38 friend class SnapshotList;
39
40 // SnapshotImpl is kept in a doubly-linked circular list
41 SnapshotImpl* prev_;
42 SnapshotImpl* next_;
43
44 SnapshotList* list_; // just for sanity checks
45
46 int64_t unix_time_;
47
48 uint64_t timestamp_;
49
50 // Will this snapshot be used by a Transaction to do write-conflict checking?
51 bool is_write_conflict_boundary_;
52};

54class SnapshotList {
55 public:
56 SnapshotList() {
57 list_.prev_ = &list_;
58 list_.next_ = &list_;
59 list_.number_ = 0xFFFFFFFFL; // placeholder marker, for debugging
60 // Set all the variables to make UBSAN happy.
61 list_.list_ = nullptr;
62 list_.unix_time_ = 0;
63 list_.timestamp_ = 0;
64 list_.is_write_conflict_boundary_ = false;
65 count_ = 0;
66 }
67
68 // No copy-construct.
69 SnapshotList(const SnapshotList&) = delete;
70
71 bool empty() const {
72 assert(list_.next_ != &list_ || 0 == count_);
73 return list_.next_ == &list_;
74 }
75 SnapshotImpl* oldest() const {
76 assert(!empty());
77 return list_.next_;
78 }
79 SnapshotImpl* newest() const {
80 assert(!empty());
81 return list_.prev_;
82 }
83
84 SnapshotImpl* New(SnapshotImpl* s, SequenceNumber seq, uint64_t unix_time,
85 bool is_write_conflict_boundary,
86 uint64_t ts = std::numeric_limits<uint64_t>::max()) {
87 s->number_ = seq;
88 s->unix_time_ = unix_time;
89 s->timestamp_ = ts;
90 s->is_write_conflict_boundary_ = is_write_conflict_boundary;
91 s->list_ = this;
92 s->next_ = &list_;
93 s->prev_ = list_.prev_;
94 s->prev_->next_ = s;
95 s->next_->prev_ = s;
96 count_++;
97 return s;
98 }
99
100 // Do not responsible to free the object.
101 void Delete(const SnapshotImpl* s) {
102 assert(s->list_ == this);
103 s->prev_->next_ = s->next_;
104 s->next_->prev_ = s->prev_;
105 count_--;
106 }

108 // retrieve all snapshot numbers up until max_seq. They are sorted in
109 // ascending order (with no duplicates).
110 std::vector<SequenceNumber> GetAll(
111 SequenceNumber* oldest_write_conflict_snapshot = nullptr,
112 const SequenceNumber& max_seq = kMaxSequenceNumber) const {
113 std::vector<SequenceNumber> ret;
114 GetAll(&ret, oldest_write_conflict_snapshot, max_seq);
115 return ret;
116 }
117
118 void GetAll(std::vector<SequenceNumber>* snap_vector,
119 SequenceNumber* oldest_write_conflict_snapshot = nullptr,
120 const SequenceNumber& max_seq = kMaxSequenceNumber) const {
121 std::vector<SequenceNumber>& ret = *snap_vector;
122 // So far we have no use case that would pass a non-empty vector
123 assert(ret.size() == 0);
124
125 if (oldest_write_conflict_snapshot != nullptr) {
126 *oldest_write_conflict_snapshot = kMaxSequenceNumber;
127 }
128
129 if (empty()) {
130 return;
131 }
132 const SnapshotImpl* s = &list_;
133 while (s->next_ != &list_) {
134 if (s->next_->number_ > max_seq) {
135 break;
136 }
137 // Avoid duplicates
138 if (ret.empty() || ret.back() != s->next_->number_) {
139 ret.push_back(s->next_->number_);
140 }
141
142 if (oldest_write_conflict_snapshot != nullptr &&
143 *oldest_write_conflict_snapshot == kMaxSequenceNumber &&
144 s->next_->is_write_conflict_boundary_) {
145 // If this is the first write-conflict boundary snapshot in the list,
146 // it is the oldest
147 *oldest_write_conflict_snapshot = s->next_->number_;
148 }
149
150 s = s->next_;
151 }
152 return;
153 }
154
155 // get the sequence number of the most recent snapshot
156 SequenceNumber GetNewest() {
157 if (empty()) {
158 return 0;
159 }
160 return newest()->number_;
161 }
162
163 int64_t GetOldestSnapshotTime() const {
164 if (empty()) {
165 return 0;
166 } else {
167 return oldest()->unix_time_;
168 }
169 }
170
171 int64_t GetOldestSnapshotSequence() const {
172 if (empty()) {
173 return 0;
174 } else {
175 return oldest()->GetSequenceNumber();
176 }
177 }
178
179 uint64_t count() const { return count_; }
180
181 private:
182 // Dummy head of doubly-linked list of snapshots
183 SnapshotImpl list_;
184 uint64_t count_;
185};

187// All operations on TimestampedSnapshotList must be protected by db mutex.
188class TimestampedSnapshotList {
189 public:
190 explicit TimestampedSnapshotList() = default;
191
192 std::shared_ptr<const SnapshotImpl> GetSnapshot(uint64_t ts) const {
193 if (ts == std::numeric_limits<uint64_t>::max() && !snapshots_.empty()) {
194 auto it = snapshots_.rbegin();
195 assert(it != snapshots_.rend());
196 return it->second;
197 }
198 auto it = snapshots_.find(ts);
199 if (it == snapshots_.end()) {
200 return std::shared_ptr<const SnapshotImpl>();
201 }
202 return it->second;
203 }
204
205 void GetSnapshots(
206 uint64_t ts_lb, uint64_t ts_ub,
207 std::vector<std::shared_ptr<const Snapshot>>& snapshots) const {
208 assert(ts_lb < ts_ub);
209 auto it_low = snapshots_.lower_bound(ts_lb);
210 auto it_high = snapshots_.lower_bound(ts_ub);
211 for (auto it = it_low; it != it_high; ++it) {
212 snapshots.emplace_back(it->second);
213 }
214 }
215
216 void AddSnapshot(const std::shared_ptr<const SnapshotImpl>& snapshot) {
217 assert(snapshot);
218 snapshots_.try_emplace(snapshot->GetTimestamp(), snapshot);
219 }
220
221 // snapshots_to_release: the container to where the timestamped snapshots will
222 // be moved so that it retains the last reference to the snapshots and the
223 // snapshots won't be actually released which requires db mutex. The
224 // snapshots will be released by caller of ReleaseSnapshotsOlderThan().
225 void ReleaseSnapshotsOlderThan(
226 uint64_t ts,
227 autovector<std::shared_ptr<const SnapshotImpl>>& snapshots_to_release) {
228 auto ub = snapshots_.lower_bound(ts);
229 for (auto it = snapshots_.begin(); it != ub; ++it) {
230 snapshots_to_release.emplace_back(it->second);
231 }
232 snapshots_.erase(snapshots_.begin(), ub);
233 }
234
235 private:
236 std::map<uint64_t, std::shared_ptr<const SnapshotImpl>> snapshots_;
237};
