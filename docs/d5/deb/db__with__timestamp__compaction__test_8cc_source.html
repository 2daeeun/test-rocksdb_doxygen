53TEST_F(TimestampCompatibleCompactionTest, UserKeyCrossFileBoundary) {
54 Options options = CurrentOptions();
55 options.env = env_;
56 options.compaction_style = kCompactionStyleLevel;
57 options.comparator = test::BytewiseComparatorWithU64TsWrapper();
58 options.level0_file_num_compaction_trigger = 3;
59 constexpr size_t kNumKeysPerFile = 101;
60 options.memtable_factory.reset(
61 test::NewSpecialSkipListFactory(kNumKeysPerFile));
62 DestroyAndReopen(options);
63 SyncPoint::GetInstance()->DisableProcessing();
64 SyncPoint::GetInstance()->ClearAllCallBacks();
65 SyncPoint::GetInstance()->SetCallBack(
66 "LevelCompactionPicker::PickCompaction:Return", [&](void* arg) {
67 const auto* compaction = static_cast<Compaction*>(arg);
68 ASSERT_NE(nullptr, compaction);
69 ASSERT_EQ(0, compaction->start_level());
70 ASSERT_EQ(1, compaction->num_input_levels());
71 // Check that all 3 L0 ssts are picked for level compaction.
72 ASSERT_EQ(3, compaction->num_input_files(0));
73 });
74 SyncPoint::GetInstance()->EnableProcessing();
75 // Write a L0 with keys 0, 1, ..., 99 with ts from 100 to 199.
76 uint64_t ts = 100;
77 uint64_t key = 0;
78 WriteOptions write_opts;
79 for (; key < kNumKeysPerFile - 1; ++key, ++ts) {
80 std::string ts_str = Timestamp(ts);
81 ASSERT_OK(
82 db_->Put(write_opts, Key1(key), ts_str, "foo_" + std::to_string(key)));
83 }
84 // Write another L0 with keys 99 with newer ts.
85 ASSERT_OK(Flush());
86 uint64_t saved_read_ts1 = ts++;
87 key = 99;
88 for (int i = 0; i < 4; ++i, ++ts) {
89 std::string ts_str = Timestamp(ts);
90 ASSERT_OK(
91 db_->Put(write_opts, Key1(key), ts_str, "bar_" + std::to_string(key)));
92 }
93 ASSERT_OK(Flush());
94 uint64_t saved_read_ts2 = ts++;
95 // Write another L0 with keys 99, 100, 101, ..., 150
96 for (; key <= 150; ++key, ++ts) {
97 std::string ts_str = Timestamp(ts);
98 ASSERT_OK(
99 db_->Put(write_opts, Key1(key), ts_str, "foo1_" + std::to_string(key)));
100 }
101 ASSERT_OK(Flush());
102 // Wait for compaction to finish
103 ASSERT_OK(dbfull()->TEST_WaitForCompact());
104 uint64_t read_ts = ts;
105 ASSERT_EQ("foo_99", Get(Key1(99), saved_read_ts1));
106 ASSERT_EQ("bar_99", Get(Key1(99), saved_read_ts2));
107 ASSERT_EQ("foo1_99", Get(Key1(99), read_ts));
108 SyncPoint::GetInstance()->ClearAllCallBacks();
109 SyncPoint::GetInstance()->DisableProcessing();
110}

112TEST_F(TimestampCompatibleCompactionTest, MultipleSubCompactions) {
113 Options options = CurrentOptions();
114 options.env = env_;
115 options.compaction_style = kCompactionStyleUniversal;
116 options.comparator = test::BytewiseComparatorWithU64TsWrapper();
117 options.level0_file_num_compaction_trigger = 3;
118 options.max_subcompactions = 3;
119 options.target_file_size_base = 1024;
120 options.statistics = CreateDBStatistics();
121 DestroyAndReopen(options);
122
123 uint64_t ts = 100;
124 uint64_t key = 0;
125 WriteOptions write_opts;
126
127 // Write keys 0, 1, ..., 499 with ts from 100 to 599.
128 {
129 for (; key <= 499; ++key, ++ts) {
130 std::string ts_str = Timestamp(ts);
131 ASSERT_OK(db_->Put(write_opts, Key1(key), ts_str,
132 "foo_" + std::to_string(key)));
133 }
134 }
135
136 // Write keys 500, ..., 999 with ts from 600 to 1099.
137 {
138 for (; key <= 999; ++key, ++ts) {
139 std::string ts_str = Timestamp(ts);
140 ASSERT_OK(db_->Put(write_opts, Key1(key), ts_str,
141 "foo_" + std::to_string(key)));
142 }
143 ASSERT_OK(Flush());
144 }
145
146 // Wait for compaction to finish
147 {
148 ASSERT_OK(dbfull()->RunManualCompaction(
149 static_cast_with_check<ColumnFamilyHandleImpl>(
150 db_->DefaultColumnFamily())
151 ->cfd(),
152 0 /* input_level */, 1 /* output_level */, CompactRangeOptions(),
153 nullptr /* begin */, nullptr /* end */, true /* exclusive */,
154 true /* disallow_trivial_move */,
155 std::numeric_limits<uint64_t>::max() /* max_file_num_to_ignore */,
156 "" /*trim_ts*/));
157 }
158
159 // Check stats to make sure multiple subcompactions were scheduled for
160 // boundaries not to be nullptr.
161 {
162 HistogramData num_sub_compactions;
163 options.statistics->histogramData(NUM_SUBCOMPACTIONS_SCHEDULED,
164 &num_sub_compactions);
165 ASSERT_GT(num_sub_compactions.sum, 1);
166 }
167
168 for (key = 0; key <= 999; ++key) {
169 ASSERT_EQ("foo_" + std::to_string(key), Get(Key1(key), ts));
170 }
171}

201TEST_F(TimestampCompatibleCompactionTest, CompactFilesRangeCheckL0) {
202 Options options = CurrentOptions();
203 options.env = env_;
204 options.sst_partitioner_factory =
205 std::make_shared<TestFilePartitionerFactory>();
206 options.comparator = test::BytewiseComparatorWithU64TsWrapper();
207 options.disable_auto_compactions = true;
208 DestroyAndReopen(options);
209
210 constexpr int kNumFiles = 10;
211 constexpr int kKeysPerFile = 2;
212 const std::string user_key = "foo";
213 constexpr uint64_t start_ts = 10000;
214
215 uint64_t cur_ts = start_ts;
216 for (int k = 0; k < kNumFiles; ++k) {
217 for (int i = 0; i < kKeysPerFile; ++i) {
218 ASSERT_OK(db_->Put(WriteOptions(), user_key, Timestamp(cur_ts),
219 "v" + std::to_string(i)));
220 ++cur_ts;
221 }
222 ASSERT_OK(db_->Flush(FlushOptions()));
223 }
224
225 std::vector<std::string> input_files{};
226 {
227 std::vector<std::string> files;
228 ASSERT_OK(env_->GetChildren(dbname_, &files));
229 for (const auto& f : files) {
230 uint64_t file_num = 0;
231 FileType file_type = FileType::kWalFile;
232 if (!ParseFileName(f, &file_num, &file_type) ||
233 file_type != FileType::kTableFile) {
234 continue;
235 }
236 input_files.emplace_back(f);
237 }
238 // sorting here by name, which also happens to sort by generation date.
239 std::sort(input_files.begin(), input_files.end());
240 assert(kNumFiles == input_files.size());
241 std::vector<std::string> tmp;
242 tmp.emplace_back(input_files[input_files.size() / 2]);
243 input_files.swap(tmp);
244 }
245
246 {
247 std::vector<std::string> output_file_names;
248 CompactionJobInfo compaction_job_info;
249 ASSERT_OK(db_->CompactFiles(CompactionOptions(), input_files,
250 /*output_level=*/1, /*output_path_id=*/-1,
251 &output_file_names, &compaction_job_info));
252 // We expect the L0 files older than the original provided input were all
253 // included in the compaction.
254 ASSERT_EQ(static_cast<size_t>(kNumFiles / 2 + 1),
255 compaction_job_info.input_files.size());
256 }
257}

259TEST_F(TimestampCompatibleCompactionTest, CompactFilesRangeCheckL1) {
260 Options options = CurrentOptions();
261 options.env = env_;
262 options.sst_partitioner_factory =
263 std::make_shared<TestFilePartitionerFactory>();
264 options.comparator = test::BytewiseComparatorWithU64TsWrapper();
265
266 constexpr int kNumFiles = 4;
267 options.level0_file_num_compaction_trigger = kNumFiles;
268
269 DestroyAndReopen(options);
270
271 constexpr int kKeysPerFile = 2;
272 const std::string user_key = "foo";
273 constexpr uint64_t start_ts = 10000;
274
275 uint64_t cur_ts = start_ts;
276 // Generate some initial files in both L0 and L1.
277 for (int k = 0; k < kNumFiles; ++k) {
278 for (int i = 0; i < kKeysPerFile; ++i) {
279 ASSERT_OK(db_->Put(WriteOptions(), user_key, Timestamp(cur_ts),
280 "v" + std::to_string(i)));
281 ++cur_ts;
282 }
283 ASSERT_OK(db_->Flush(FlushOptions()));
284 }
285 ASSERT_OK(dbfull()->TEST_WaitForCompact());
286
287 ASSERT_EQ(0, NumTableFilesAtLevel(/*level=*/0, /*cf=*/0));
288 ASSERT_EQ(kNumFiles * kKeysPerFile,
289 NumTableFilesAtLevel(/*level=*/1, /*cf=*/0));
290
291 constexpr int additional_l0s = 2;
292 for (int i = 0; i < additional_l0s; ++i, ++cur_ts) {
293 ASSERT_OK(db_->Put(WriteOptions(), user_key, Timestamp(cur_ts), "v"));
294 ASSERT_OK(db_->Flush(FlushOptions()));
295 }
296 ASSERT_EQ(additional_l0s, NumTableFilesAtLevel(/*level=*/0, /*cf=*/0));
297
298 std::vector<std::string> inputs;
299 {
300 std::vector<LiveFileMetaData> fmetas;
301 db_->GetLiveFilesMetaData(&fmetas);
302 bool included_one_l1 = false;
303 for (const auto& meta : fmetas) {
304 if (meta.level == 0) {
305 inputs.emplace_back(meta.relative_filename);
306 } else if (!included_one_l1) {
307 inputs.emplace_back(meta.relative_filename);
308 included_one_l1 = true;
309 }
310 }
311 }
312 ASSERT_EQ(static_cast<size_t>(3), inputs.size());
313 {
314 std::vector<std::string> output_file_names;
315 CompactionJobInfo compaction_job_info;
316
317 ASSERT_OK(db_->CompactFiles(CompactionOptions(), inputs, /*output_level=*/1,
318 /*output_path_id=*/-1, &output_file_names,
319 &compaction_job_info));
320 ASSERT_EQ(kNumFiles * kKeysPerFile + 2, output_file_names.size());
321 ASSERT_EQ(kNumFiles * kKeysPerFile + 2,
322 static_cast<int>(compaction_job_info.input_files.size()));
323 }
324}
