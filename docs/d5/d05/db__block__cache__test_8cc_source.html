1807// Deconstruct cache key, based on knowledge of implementation details.
1808void DeconstructNonemptyCacheKey(const CacheKey& key, uint64_t* file_num_etc64,
1809 uint64_t* offset_etc64) {
1810 *file_num_etc64 = *reinterpret_cast<const uint64_t*>(key.AsSlice().data());
1811 *offset_etc64 = *reinterpret_cast<const uint64_t*>(key.AsSlice().data() + 8);
1812 assert(*file_num_etc64 != 0);
1813 if (*offset_etc64 == 0) {
1814 std::swap(*file_num_etc64, *offset_etc64);
1815 }
1816 assert(*offset_etc64 != 0);
1817}
1818
1819// Make a bit mask of 0 to 64 bits
1820uint64_t MakeMask64(int bits) {
1821 if (bits >= 64) {
1822 return uint64_t{0} - 1;
1823 } else {
1824 return (uint64_t{1} << bits) - 1;
1825 }
1826}
1827
1828// See CacheKeyTest::Encodings
1829struct CacheKeyDecoder {
1830 // Inputs
1831 uint64_t base_file_num_etc64, base_offset_etc64;
1832 int session_counter_bits, file_number_bits, offset_bits;
1833
1834 // Derived
1835 uint64_t session_counter_mask, file_number_mask, offset_mask;
1836
1837 // Outputs
1838 uint64_t decoded_session_counter, decoded_file_num, decoded_offset;
1839
1840 void SetBaseCacheKey(const CacheKey& base) {
1841 DeconstructNonemptyCacheKey(base, &base_file_num_etc64, &base_offset_etc64);
1842 }
1843
1844 void SetRanges(int _session_counter_bits, int _file_number_bits,
1845 int _offset_bits) {
1846 session_counter_bits = _session_counter_bits;
1847 session_counter_mask = MakeMask64(session_counter_bits);
1848 file_number_bits = _file_number_bits;
1849 file_number_mask = MakeMask64(file_number_bits);
1850 offset_bits = _offset_bits;
1851 offset_mask = MakeMask64(offset_bits);
1852 }
1853
1854 void Decode(const CacheKey& key) {
1855 uint64_t file_num_etc64, offset_etc64;
1856 DeconstructNonemptyCacheKey(key, &file_num_etc64, &offset_etc64);
1857
1858 // First decode session counter
1859 if (offset_bits + session_counter_bits <= 64) {
1860 // fully recoverable from offset_etc64
1861 decoded_session_counter =
1862 ReverseBits((offset_etc64 ^ base_offset_etc64)) &
1863 session_counter_mask;
1864 } else if (file_number_bits + session_counter_bits <= 64) {
1865 // fully recoverable from file_num_etc64
1866 decoded_session_counter = DownwardInvolution(
1867 (file_num_etc64 ^ base_file_num_etc64) & session_counter_mask);
1868 } else {
1869 // Need to combine parts from each word.
1870 // Piece1 will contain some correct prefix of the bottom bits of
1871 // session counter.
1872 uint64_t piece1 =
1873 ReverseBits((offset_etc64 ^ base_offset_etc64) & ~offset_mask);
1874 int piece1_bits = 64 - offset_bits;
1875 // Piece2 will contain involuded bits that we can combine with piece1
1876 // to infer rest of session counter
1877 int piece2_bits = std::min(64 - file_number_bits, 64 - piece1_bits);
1878 ASSERT_LT(piece2_bits, 64);
1879 uint64_t piece2_mask = MakeMask64(piece2_bits);
1880 uint64_t piece2 = (file_num_etc64 ^ base_file_num_etc64) & piece2_mask;
1881
1882 // Cancel out the part of piece2 that we can infer from piece1
1883 // (DownwardInvolution distributes over xor)
1884 piece2 ^= DownwardInvolution(piece1) & piece2_mask;
1885
1886 // Now we need to solve for the unknown original bits in higher
1887 // positions than piece1 provides. We use Gaussian elimination
1888 // because we know that a piece2_bits X piece2_bits submatrix of
1889 // the matrix underlying DownwardInvolution times the vector of
1890 // unknown original bits equals piece2.
1891 //
1892 // Build an augmented row matrix for that submatrix, built column by
1893 // column.
1894 std::array<uint64_t, 64> aug_rows{};
1895 for (int i = 0; i < piece2_bits; ++i) { // over columns
1896 uint64_t col_i = DownwardInvolution(uint64_t{1} << piece1_bits << i);
1897 ASSERT_NE(col_i & 1U, 0);
1898 for (int j = 0; j < piece2_bits; ++j) { // over rows
1899 aug_rows[j] |= (col_i & 1U) << i;
1900 col_i >>= 1;
1901 }
1902 }
1903 // Augment with right hand side
1904 for (int j = 0; j < piece2_bits; ++j) { // over rows
1905 aug_rows[j] |= (piece2 & 1U) << piece2_bits;
1906 piece2 >>= 1;
1907 }
1908 // Run Gaussian elimination
1909 for (int i = 0; i < piece2_bits; ++i) { // over columns
1910 // Find a row that can be used to cancel others
1911 uint64_t canceller = 0;
1912 // Note: Rows 0 through i-1 contain 1s in columns already eliminated
1913 for (int j = i; j < piece2_bits; ++j) { // over rows
1914 if (aug_rows[j] & (uint64_t{1} << i)) {
1915 // Swap into appropriate row
1916 std::swap(aug_rows[i], aug_rows[j]);
1917 // Keep a handy copy for row reductions
1918 canceller = aug_rows[i];
1919 break;
1920 }
1921 }
1922 ASSERT_NE(canceller, 0);
1923 for (int j = 0; j < piece2_bits; ++j) { // over rows
1924 if (i != j && ((aug_rows[j] >> i) & 1) != 0) {
1925 // Row reduction
1926 aug_rows[j] ^= canceller;
1927 }
1928 }
1929 }
1930 // Extract result
1931 decoded_session_counter = piece1;
1932 for (int j = 0; j < piece2_bits; ++j) { // over rows
1933 ASSERT_EQ(aug_rows[j] & piece2_mask, uint64_t{1} << j);
1934 decoded_session_counter |= aug_rows[j] >> piece2_bits << piece1_bits
1935 << j;
1936 }
1937 }
1938
1939 decoded_offset =
1940 offset_etc64 ^ base_offset_etc64 ^ ReverseBits(decoded_session_counter);
1941
1942 decoded_file_num = ReverseBits(file_num_etc64 ^ base_file_num_etc64 ^
1943 DownwardInvolution(decoded_session_counter));
1944 }
1945};
