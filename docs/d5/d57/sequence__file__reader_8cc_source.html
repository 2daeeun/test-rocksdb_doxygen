26IOStatus SequentialFileReader::Create(
27 const std::shared_ptr<FileSystem>& fs, const std::string& fname,
28 const FileOptions& file_opts, std::unique_ptr<SequentialFileReader>* reader,
29 IODebugContext* dbg, RateLimiter* rate_limiter) {

33 reader->reset(new SequentialFileReader(std::move(file), fname, nullptr, {},

35 }

39IOStatus SequentialFileReader::Read(size_t n, Slice* result, char* scratch,
40 Env::IOPriority rate_limiter_priority) {
41 IOStatus io_s;
42 IOOptions io_opts;
43 io_opts.rate_limiter_priority = rate_limiter_priority;
44 io_opts.verify_and_reconstruct_read = verify_and_reconstruct_read_;
45 if (use_direct_io()) {
46 //
47 // |-offset_advance-|---bytes returned--|
48 // |----------------------buf size-------------------------|
49 // | | | |
50 // aligned offset offset + n Roundup(offset + n,
51 // offset alignment)
52 //
53 size_t offset = offset_.fetch_add(n);
54 size_t alignment = file_->GetRequiredBufferAlignment();
55 size_t aligned_offset = TruncateToPageBoundary(alignment, offset);
56 size_t offset_advance = offset - aligned_offset;
57 size_t size = Roundup(offset + n, alignment) - aligned_offset;
58 size_t r = 0;
59 AlignedBuffer buf;
60 buf.Alignment(alignment);
61 buf.AllocateNewBuffer(size);
62
63 while (buf.CurrentSize() < size) {
64 size_t allowed;
65 if (rate_limiter_priority != Env::IO_TOTAL && rate_limiter_ != nullptr) {
66 allowed = rate_limiter_->RequestToken(
67 buf.Capacity() - buf.CurrentSize(), buf.Alignment(),
68 rate_limiter_priority, nullptr /* stats */,
69 RateLimiter::OpType::kRead);
70 } else {
71 assert(buf.CurrentSize() == 0);
72 allowed = size;
73 }
74
75 Slice tmp;
76 uint64_t orig_offset = 0;
77 FileOperationInfo::StartTimePoint start_ts;
78 if (ShouldNotifyListeners()) {
79 orig_offset = aligned_offset + buf.CurrentSize();
80 start_ts = FileOperationInfo::StartNow();
81 }
82 io_s = file_->PositionedRead(aligned_offset + buf.CurrentSize(), allowed,
83 io_opts, &tmp, buf.Destination(),
84 nullptr /* dbg */);
85 if (ShouldNotifyListeners()) {
86 auto finish_ts = FileOperationInfo::FinishNow();
87 NotifyOnFileReadFinish(orig_offset, tmp.size(), start_ts, finish_ts,
88 io_s);
89 }
90 buf.Size(buf.CurrentSize() + tmp.size());
91 if (!io_s.ok() || tmp.size() < allowed) {
92 break;
93 }
94 }
95
96 if (io_s.ok() && offset_advance < buf.CurrentSize()) {
97 r = buf.Read(scratch, offset_advance,
98 std::min(buf.CurrentSize() - offset_advance, n));
99 }
100 *result = Slice(scratch, r);
101 } else {
102 // To be paranoid, modify scratch a little bit, so in case underlying
103 // FileSystem doesn't fill the buffer but return success and `scratch`
104 // returns contains a previous block, returned value will not pass
105 // checksum.
106 // It's hard to find useful byte for direct I/O case, so we skip it.
107 if (n > 0 && scratch != nullptr) {
108 scratch[0]++;
109 }
110
111 size_t read = 0;
112 while (read < n) {
113 size_t allowed;
114 if (rate_limiter_priority != Env::IO_TOTAL && rate_limiter_ != nullptr) {
115 allowed = rate_limiter_->RequestToken(
116 n - read, 0 /* alignment */, rate_limiter_priority,
117 nullptr /* stats */, RateLimiter::OpType::kRead);
118 } else {
119 allowed = n;
120 }
121 FileOperationInfo::StartTimePoint start_ts;
122 if (ShouldNotifyListeners()) {
123 start_ts = FileOperationInfo::StartNow();
124 }
125 Slice tmp;
126 io_s = file_->Read(allowed, io_opts, &tmp, scratch + read,
127 nullptr /* dbg */);
128 if (ShouldNotifyListeners()) {
129 auto finish_ts = FileOperationInfo::FinishNow();
130 size_t offset = offset_.fetch_add(tmp.size());
131 NotifyOnFileReadFinish(offset, tmp.size(), start_ts, finish_ts, io_s);
132 }
133 read += tmp.size();
134 if (!io_s.ok() || tmp.size() < allowed) {
135 break;
136 }
137 }
138 *result = Slice(scratch, read);
139 }
140 IOSTATS_ADD(bytes_read, result->size());
141 return io_s;
142}

144IOStatus SequentialFileReader::Skip(uint64_t n) {
145 if (use_direct_io()) {
146 offset_ += static_cast<size_t>(n);
147 return IOStatus::OK();
148 }
149 return file_->Skip(n);

152namespace {
153// This class wraps a SequentialFile, exposing same API, with the differenece
154// of being able to prefetch up to readahead_size bytes and then serve them
155// from memory, avoiding the entire round-trip if, for example, the data for the
156// file is actually remote.
157class ReadaheadSequentialFile : public FSSequentialFile {
158 public:
159 ReadaheadSequentialFile(std::unique_ptr<FSSequentialFile>&& file,
160 size_t readahead_size)
161 : file_(std::move(file)),
162 alignment_(file_->GetRequiredBufferAlignment()),
163 readahead_size_(Roundup(readahead_size, alignment_)),
164 buffer_(),
165 buffer_offset_(0),
166 read_offset_(0) {
167 buffer_.Alignment(alignment_);
168 buffer_.AllocateNewBuffer(readahead_size_);
169 }
170
171 ReadaheadSequentialFile(const ReadaheadSequentialFile&) = delete;
172
173 ReadaheadSequentialFile& operator=(const ReadaheadSequentialFile&) = delete;
174
175 IOStatus Read(size_t n, const IOOptions& opts, Slice* result, char* scratch,
176 IODebugContext* dbg) override {
177 std::unique_lock<std::mutex> lk(lock_);
178
179 size_t cached_len = 0;
180 // Check if there is a cache hit, meaning that [offset, offset + n) is
181 // either completely or partially in the buffer. If it's completely cached,
182 // including end of file case when offset + n is greater than EOF, then
183 // return.
184 if (TryReadFromCache(n, &cached_len, scratch) &&
185 (cached_len == n || buffer_.CurrentSize() < readahead_size_)) {
186 // We read exactly what we needed, or we hit end of file - return.
187 *result = Slice(scratch, cached_len);
188 return IOStatus::OK();
189 }
190 n -= cached_len;
191
192 IOStatus s;
193 // Read-ahead only make sense if we have some slack left after reading
194 if (n + alignment_ >= readahead_size_) {
195 s = file_->Read(n, opts, result, scratch + cached_len, dbg);
196 if (s.ok()) {
197 read_offset_ += result->size();
198 *result = Slice(scratch, cached_len + result->size());
199 }
200 buffer_.Clear();
201 return s;
202 }
203
204 s = ReadIntoBuffer(readahead_size_, opts, dbg);
205 if (s.ok()) {
206 // The data we need is now in cache, so we can safely read it
207 size_t remaining_len;
208 TryReadFromCache(n, &remaining_len, scratch + cached_len);
209 *result = Slice(scratch, cached_len + remaining_len);
210 }
211 return s;
212 }
213
214 IOStatus Skip(uint64_t n) override {
215 std::unique_lock<std::mutex> lk(lock_);
216 IOStatus s = IOStatus::OK();
217 // First check if we need to skip already cached data
218 if (buffer_.CurrentSize() > 0) {
219 // Do we need to skip beyond cached data?
220 if (read_offset_ + n >= buffer_offset_ + buffer_.CurrentSize()) {
221 // Yes. Skip whaterver is in memory and adjust offset accordingly
222 n -= buffer_offset_ + buffer_.CurrentSize() - read_offset_;
223 read_offset_ = buffer_offset_ + buffer_.CurrentSize();
224 } else {
225 // No. The entire section to be skipped is entirely i cache.
226 read_offset_ += n;
227 n = 0;
228 }
229 }
230 if (n > 0) {
231 // We still need to skip more, so call the file API for skipping
232 s = file_->Skip(n);
233 if (s.ok()) {
234 read_offset_ += n;
235 }
236 buffer_.Clear();
237 }
238 return s;
239 }
240
241 IOStatus PositionedRead(uint64_t offset, size_t n, const IOOptions& opts,
242 Slice* result, char* scratch,
243 IODebugContext* dbg) override {
244 return file_->PositionedRead(offset, n, opts, result, scratch, dbg);
245 }
246
247 IOStatus InvalidateCache(size_t offset, size_t length) override {
248 std::unique_lock<std::mutex> lk(lock_);
249 buffer_.Clear();
250 return file_->InvalidateCache(offset, length);
251 }
252
253 bool use_direct_io() const override { return file_->use_direct_io(); }
254
255 private:
256 // Tries to read from buffer_ n bytes. If anything was read from the cache, it
257 // sets cached_len to the number of bytes actually read, copies these number
258 // of bytes to scratch and returns true.
259 // If nothing was read sets cached_len to 0 and returns false.
260 bool TryReadFromCache(size_t n, size_t* cached_len, char* scratch) {
261 if (read_offset_ < buffer_offset_ ||
262 read_offset_ >= buffer_offset_ + buffer_.CurrentSize()) {
263 *cached_len = 0;
264 return false;
265 }
266 uint64_t offset_in_buffer = read_offset_ - buffer_offset_;
267 *cached_len = std::min(
268 buffer_.CurrentSize() - static_cast<size_t>(offset_in_buffer), n);
269 memcpy(scratch, buffer_.BufferStart() + offset_in_buffer, *cached_len);
270 read_offset_ += *cached_len;
271 return true;
272 }
273
274 // Reads into buffer_ the next n bytes from file_.
275 // Can actually read less if EOF was reached.
276 // Returns the status of the read operastion on the file.
277 IOStatus ReadIntoBuffer(size_t n, const IOOptions& opts,
278 IODebugContext* dbg) {
279 if (n > buffer_.Capacity()) {
280 n = buffer_.Capacity();
281 }
282 assert(IsFileSectorAligned(n, alignment_));
283 Slice result;
284 IOStatus s = file_->Read(n, opts, &result, buffer_.BufferStart(), dbg);
285 if (s.ok()) {
286 buffer_offset_ = read_offset_;
287 buffer_.Size(result.size());
288 assert(result.size() == 0 || buffer_.BufferStart() == result.data());
289 }
290 return s;
291 }
292
293 const std::unique_ptr<FSSequentialFile> file_;
294 const size_t alignment_;
295 const size_t readahead_size_;
296
297 std::mutex lock_;
298 // The buffer storing the prefetched data
299 AlignedBuffer buffer_;
300 // The offset in file_, corresponding to data stored in buffer_
301 uint64_t buffer_offset_;
302 // The offset up to which data was read from file_. In fact, it can be larger
303 // than the actual file size, since the file_->Skip(n) call doesn't return the
304 // actual number of bytes that were skipped, which can be less than n.
305 // This is not a problemm since read_offset_ is monotonically increasing and
306 // its only use is to figure out if next piece of data should be read from
307 // buffer_ or file_ directly.
308 uint64_t read_offset_;
309};
310} // namespace

312std::unique_ptr<FSSequentialFile>
313SequentialFileReader::NewReadaheadSequentialFile(
314 std::unique_ptr<FSSequentialFile>&& file, size_t readahead_size) {
315 if (file->GetRequiredBufferAlignment() >= readahead_size) {
316 // Short-circuit and return the original file if readahead_size is
317 // too small and hence doesn't make sense to be used for prefetching.
318 return std::move(file);
319 }
320 std::unique_ptr<FSSequentialFile> result(
321 new ReadaheadSequentialFile(std::move(file), readahead_size));
322 return result;
323}
