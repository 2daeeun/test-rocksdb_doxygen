74enum TestConfigMode {
75 kEmptyMode = 0x0, // Don't register anything
76 kMutableMode = 0x01, // Configuration is mutable
77 kSimpleMode = 0x02, // Use the simple options
78 kEnumMode = 0x04, // Use the enum options
79 kDefaultMode = kSimpleMode, // Use no inner nested configurations
80 kSharedMode = 0x10, // Use shared configuration
81 kUniqueMode = 0x20, // Use unique configuration
82 kRawPtrMode = 0x40, // Use pointer configuration
83 kNestedMode = (kSharedMode | kUniqueMode | kRawPtrMode),
84 kAllOptMode = (kNestedMode | kEnumMode | kSimpleMode),
85};

99 TestConfigurable(const std::string& name, int mode,

101 &simple_option_info)
102 : name_(name), pointer_(nullptr) {
103 prefix_ = "test." + name + ".";
104 if ((mode & TestConfigMode::kSimpleMode) != 0) {
105 RegisterOptions(name_, &options_, map);
106 }
107 if ((mode & TestConfigMode::kEnumMode) != 0) {
108 RegisterOptions(name_ + "Enum", &options_, &enum_option_info);
109 }
110 }
111
112 ~TestConfigurable() override { delete pointer_; }
113};
