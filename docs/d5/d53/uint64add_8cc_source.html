19bool UInt64AddOperator::Merge(const Slice& /*key*/, const Slice* existing_value,
20 const Slice& value, std::string* new_value,
21 Logger* logger) const {
22 uint64_t orig_value = 0;
23 if (existing_value) {
24 orig_value = DecodeInteger(*existing_value, logger);
25 }
26 uint64_t operand = DecodeInteger(value, logger);
27
28 assert(new_value);
29 new_value->clear();
30 PutFixed64(new_value, orig_value + operand);
31
32 return true; // Return true always since corruption will be treated as 0
33}

35uint64_t UInt64AddOperator::DecodeInteger(const Slice& value,
36 Logger* logger) const {
37 uint64_t result = 0;
38
39 if (value.size() == sizeof(uint64_t)) {
40 result = DecodeFixed64(value.data());
41 } else if (logger != nullptr) {
42 // If value is corrupted, treat it as 0
43 ROCKS_LOG_ERROR(logger,
44 "uint64 value corruption, size: %" ROCKSDB_PRIszt
45 " > %" ROCKSDB_PRIszt,
46 value.size(), sizeof(uint64_t));
47 }
48
49 return result;
50}
