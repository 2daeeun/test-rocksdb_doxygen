23// Set the dump filter with a list of DBs. Block cache may be shared by multipe
24// DBs and we may only want to dump out the blocks belonging to certain DB(s).
25// Therefore, a filter is need to decide if the key of the block satisfy the
26// requirement.
27Status CacheDumperImpl::SetDumpFilter(std::vector<DB*> db_list) {
28 Status s = Status::OK();
29 dump_all_keys_ = false;
30 for (size_t i = 0; i < db_list.size(); i++) {
31 assert(i < db_list.size());
32 TablePropertiesCollection ptc;
33 assert(db_list[i] != nullptr);
34 s = db_list[i]->GetPropertiesOfAllTables(&ptc);
35 if (!s.ok()) {
36 return s;
37 }
38 for (auto id = ptc.begin(); id != ptc.end(); id++) {
39 OffsetableCacheKey base;
40 // We only want to save cache entries that are portable to another
41 // DB::Open, so only save entries with stable keys.
42 bool is_stable;
43 BlockBasedTable::SetupBaseCacheKey(id->second.get(),
44 /*cur_db_session_id*/ "",
45 /*cur_file_num*/ 0, &base, &is_stable);
46 if (is_stable) {
47 Slice prefix_slice = base.CommonPrefixSlice();
48 assert(prefix_slice.size() == OffsetableCacheKey::kCommonPrefixSize);
49 prefix_filter_.insert(prefix_slice.ToString());
50 }
51 }
52 }
53 return s;
54}

56// This is the main function to dump out the cache block entries to the writer.
57// The writer may create a file or write to other systems. Currently, we will
58// iterate the whole block cache, get the blocks, and write them to the writer
59IOStatus CacheDumperImpl::DumpCacheEntriesToWriter() {
60 // Prepare stage, check the parameters.
61 if (cache_ == nullptr) {
62 return IOStatus::InvalidArgument("Cache is null");
63 }
64 if (writer_ == nullptr) {
65 return IOStatus::InvalidArgument("CacheDumpWriter is null");
66 }
67 // Set the system clock
68 if (options_.clock == nullptr) {
69 return IOStatus::InvalidArgument("System clock is null");
70 }
71 clock_ = options_.clock;
72
73 deadline_ = options_.deadline;
74
75 // Set the sequence number
76 sequence_num_ = 0;
77
78 // Dump stage, first, we write the hader
79 IOStatus io_s = WriteHeader();
80 if (!io_s.ok()) {
81 return io_s;
82 }
83
84 // Then, we iterate the block cache and dump out the blocks that are not
85 // filtered out.
86 std::string buf;
87 cache_->ApplyToAllEntries(DumpOneBlockCallBack(buf), {});
88
89 // Finally, write the footer
90 io_s = WriteFooter();
91 if (!io_s.ok()) {
92 return io_s;
93 }
94 io_s = writer_->Close();
95 return io_s;
96}

98// Check if we need to filter out the block based on its key
99bool CacheDumperImpl::ShouldFilterOut(const Slice& key) {
100 if (key.size() < OffsetableCacheKey::kCommonPrefixSize) {
101 return /*filter out*/ true;
102 }
103 Slice key_prefix(key.data(), OffsetableCacheKey::kCommonPrefixSize);
104 std::string prefix = key_prefix.ToString();
105 // Filter out if not found
106 return prefix_filter_.find(prefix) == prefix_filter_.end();
107}

109// This is the callback function which will be applied to
110// Cache::ApplyToAllEntries. In this callback function, we will get the block
111// type, decide if the block needs to be dumped based on the filter, and write
112// the block through the provided writer. `buf` is passed in for efficiennt
113// reuse.
114std::function<void(const Slice&, Cache::ObjectPtr, size_t,
115 const Cache::CacheItemHelper*)>
116CacheDumperImpl::DumpOneBlockCallBack(std::string& buf) {
117 return [&](const Slice& key, Cache::ObjectPtr value, size_t /*charge*/,
118 const Cache::CacheItemHelper* helper) {
119 if (helper == nullptr || helper->size_cb == nullptr ||
120 helper->saveto_cb == nullptr) {
121 // Not compatible with dumping. Skip this entry.
122 return;
123 }
124
125 if (options_.max_size_bytes > 0 &&
126 dumped_size_bytes_ > options_.max_size_bytes) {
127 return;
128 }
129
130 uint64_t timestamp = clock_->NowMicros();
131 if (deadline_.count()) {
132 std::chrono::microseconds now = std::chrono::microseconds(timestamp);
133 if (now >= deadline_) {
134 return;
135 }
136 }
137
138 CacheEntryRole role = helper->role;
139 CacheDumpUnitType type = CacheDumpUnitType::kBlockTypeMax;
140
141 switch (role) {
142 case CacheEntryRole::kDataBlock:
143 type = CacheDumpUnitType::kData;
144 break;
145 case CacheEntryRole::kFilterBlock:
146 type = CacheDumpUnitType::kFilter;
147 break;
148 case CacheEntryRole::kFilterMetaBlock:
149 type = CacheDumpUnitType::kFilterMetaBlock;
150 break;
151 case CacheEntryRole::kIndexBlock:
152 type = CacheDumpUnitType::kIndex;
153 break;
154 default:
155 // Filter out other entries
156 // FIXME? Do we need the CacheDumpUnitTypes? UncompressionDict?
157 return;
158 }
159
160 // based on the key prefix, check if the block should be filter out.
161 if (!dump_all_keys_ && ShouldFilterOut(key)) {
162 return;
163 }
164
165 assert(type != CacheDumpUnitType::kBlockTypeMax);
166
167 // Use cache item helper to get persistable data
168 // FIXME: reduce copying
169 size_t len = helper->size_cb(value);
170 buf.assign(len, '\0');
171 Status s = helper->saveto_cb(value, /*start*/ 0, len, buf.data());
172
173 if (s.ok()) {
174 // Write it out
175 WriteBlock(type, key, buf, timestamp).PermitUncheckedError();
176 dumped_size_bytes_ += len;
177 }
178 };
179}

181// Write the block to the writer. It takes the timestamp of the
182// block being copied from block cache, block type, key, block pointer,
183// block size and block checksum as the input. When writing the dumper raw
184// block, we first create the dump unit and encoude it to a string. Then,
185// we calculate the checksum of the whole dump unit string and store it in
186// the dump unit metadata.
187// First, we write the metadata first, which is a fixed size string. Then, we
188// Append the dump unit string to the writer.
189IOStatus CacheDumperImpl::WriteBlock(CacheDumpUnitType type, const Slice& key,
190 const Slice& value, uint64_t timestamp) {
191 uint32_t value_checksum = crc32c::Value(value.data(), value.size());
192
193 // First, serialize the block information in a string
194 DumpUnit dump_unit;
195 dump_unit.timestamp = timestamp;
196 dump_unit.key = key;
197 dump_unit.type = type;
198 dump_unit.value_len = value.size();
199 dump_unit.value = const_cast<char*>(value.data());
200 dump_unit.value_checksum = value_checksum;
201 std::string encoded_data;
202 CacheDumperHelper::EncodeDumpUnit(dump_unit, &encoded_data);
203
204 // Second, create the metadata, which contains a sequence number, the dump
205 // unit string checksum and the string size. The sequence number monotonically
206 // increases from 0.
207 DumpUnitMeta unit_meta;
208 unit_meta.sequence_num = sequence_num_;
209 sequence_num_++;
210 unit_meta.dump_unit_checksum =
211 crc32c::Value(encoded_data.data(), encoded_data.size());
212 unit_meta.dump_unit_size = encoded_data.size();
213 std::string encoded_meta;
214 CacheDumperHelper::EncodeDumpUnitMeta(unit_meta, &encoded_meta);
215
216 // We write the metadata first.
217 assert(writer_ != nullptr);
218 IOStatus io_s = writer_->WriteMetadata(encoded_meta);
219 if (!io_s.ok()) {
220 return io_s;
221 }
222 // followed by the dump unit.
223 return writer_->WritePacket(encoded_data);
224}

226// Before we write any block, we write the header first to store the cache dump
227// format version, rocksdb version, and brief intro.
228IOStatus CacheDumperImpl::WriteHeader() {
229 std::string header_key = "header";
230 std::ostringstream s;
231 s << kTraceMagic << "\t"
232 << "Cache dump format version: " << kCacheDumpMajorVersion << "."
233 << kCacheDumpMinorVersion << "\t"
234 << "RocksDB Version: " << kMajorVersion << "." << kMinorVersion << "\t"
235 << "Format: dump_unit_metadata <sequence_number, dump_unit_checksum, "
236 "dump_unit_size>, dump_unit <timestamp, key, block_type, "
237 "block_size, block_data, block_checksum> cache_value\n";
238 std::string header_value(s.str());
239 CacheDumpUnitType type = CacheDumpUnitType::kHeader;
240 uint64_t timestamp = clock_->NowMicros();
241 return WriteBlock(type, header_key, header_value, timestamp);
242}

244// Write the footer after all the blocks are stored to indicate the ending.
245IOStatus CacheDumperImpl::WriteFooter() {
246 std::string footer_key = "footer";
247 std::string footer_value("cache dump completed");
248 CacheDumpUnitType type = CacheDumpUnitType::kFooter;
249 uint64_t timestamp = clock_->NowMicros();
250 return WriteBlock(type, footer_key, footer_value, timestamp);
251}

253// This is the main function to restore the cache entries to secondary cache.
254// First, we check if all the arguments are valid. Then, we read the block
255// sequentially from the reader and insert them to the secondary cache.
256IOStatus CacheDumpedLoaderImpl::RestoreCacheEntriesToSecondaryCache() {
257 // TODO: remove this line when options are used in the loader
258 (void)options_;
259 // Step 1: we check if all the arguments are valid
260 if (secondary_cache_ == nullptr) {
261 return IOStatus::InvalidArgument("Secondary Cache is null");
262 }
263 if (reader_ == nullptr) {
264 return IOStatus::InvalidArgument("CacheDumpReader is null");
265 }
266
267 // Step 2: read the header
268 // TODO: we need to check the cache dump format version and RocksDB version
269 // after the header is read out.
270 IOStatus io_s;
271 DumpUnit dump_unit;
272 std::string data;
273 io_s = ReadHeader(&data, &dump_unit);
274 if (!io_s.ok()) {
275 return io_s;
276 }
277
278 // Step 3: read out the rest of the blocks from the reader. The loop will stop
279 // either I/O status is not ok or we reach to the the end.
280 while (io_s.ok()) {
281 dump_unit.reset();
282 data.clear();
283 // read the content and store in the dump_unit
284 io_s = ReadCacheBlock(&data, &dump_unit);
285 if (!io_s.ok()) {
286 break;
287 }
288 if (dump_unit.type == CacheDumpUnitType::kFooter) {
289 break;
290 }
291 // Create the uncompressed_block based on the information in the dump_unit
292 // (There is no block trailer here compatible with block-based SST file.)
293 Slice content =
294 Slice(static_cast<char*>(dump_unit.value), dump_unit.value_len);
295 Status s = secondary_cache_->InsertSaved(dump_unit.key, content);
296 if (!s.ok()) {
297 io_s = status_to_io_status(std::move(s));
298 }
299 }
300 if (dump_unit.type == CacheDumpUnitType::kFooter) {
301 return IOStatus::OK();
302 } else {
303 return io_s;
304 }
305}

341// Read the header
342IOStatus CacheDumpedLoaderImpl::ReadHeader(std::string* data,
343 DumpUnit* dump_unit) {
344 DumpUnitMeta header_meta;
345 header_meta.reset();
346 std::string meta_string;
347 IOStatus io_s = ReadDumpUnitMeta(&meta_string, &header_meta);
348 if (!io_s.ok()) {
349 return io_s;
350 }
351
352 io_s = ReadDumpUnit(header_meta.dump_unit_size, data, dump_unit);
353 if (!io_s.ok()) {
354 return io_s;
355 }
356 uint32_t unit_checksum = crc32c::Value(data->data(), data->size());
357 if (unit_checksum != header_meta.dump_unit_checksum) {
358 return IOStatus::Corruption("Read header unit corrupted!");
359 }
360 return io_s;
361}

363// Read the blocks after header is read out
364IOStatus CacheDumpedLoaderImpl::ReadCacheBlock(std::string* data,
365 DumpUnit* dump_unit) {
366 // According to the write process, we read the dump_unit_metadata first
367 DumpUnitMeta unit_meta;
368 unit_meta.reset();
369 std::string unit_string;
370 IOStatus io_s = ReadDumpUnitMeta(&unit_string, &unit_meta);
371 if (!io_s.ok()) {
372 return io_s;
373 }
374
375 // Based on the information in the dump_unit_metadata, we read the dump_unit
376 // and verify if its content is correct.
377 io_s = ReadDumpUnit(unit_meta.dump_unit_size, data, dump_unit);
378 if (!io_s.ok()) {
379 return io_s;
380 }
381 uint32_t unit_checksum = crc32c::Value(data->data(), data->size());
382 if (unit_checksum != unit_meta.dump_unit_checksum) {
383 return IOStatus::Corruption(
384 "Checksum does not match! Read dumped unit corrupted!");
385 }
386 return io_s;
387}
