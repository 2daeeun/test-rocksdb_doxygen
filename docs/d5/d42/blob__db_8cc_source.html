16Status BlobDB::Open(const Options& options, const BlobDBOptions& bdb_options,
17 const std::string& dbname, BlobDB** blob_db) {
18 *blob_db = nullptr;
19 DBOptions db_options(options);
20 ColumnFamilyOptions cf_options(options);
21 std::vector<ColumnFamilyDescriptor> column_families;
22 column_families.emplace_back(kDefaultColumnFamilyName, cf_options);
23 std::vector<ColumnFamilyHandle*> handles;
24 Status s = BlobDB::Open(db_options, bdb_options, dbname, column_families,
25 &handles, blob_db);
26 if (s.ok()) {
27 assert(handles.size() == 1);
28 // i can delete the handle since DBImpl is always holding a reference to
29 // default column family
30 delete handles[0];
31 }
32 return s;
33}

35Status BlobDB::Open(const DBOptions& db_options,
36 const BlobDBOptions& bdb_options, const std::string& dbname,
37 const std::vector<ColumnFamilyDescriptor>& column_families,
38 std::vector<ColumnFamilyHandle*>* handles,
39 BlobDB** blob_db) {
40 assert(handles);
41
42 if (column_families.size() != 1 ||
43 column_families[0].name != kDefaultColumnFamilyName) {
44 return Status::NotSupported(
45 "Blob DB doesn't support non-default column family.");
46 }
47
48 BlobDBImpl* blob_db_impl = new BlobDBImpl(dbname, bdb_options, db_options,
49 column_families[0].options);
50 Status s = blob_db_impl->Open(handles);
51 if (s.ok()) {
52 *blob_db = static_cast<BlobDB*>(blob_db_impl);
53 } else {
54 if (!handles->empty()) {
55 for (ColumnFamilyHandle* cfh : *handles) {
56 blob_db_impl->DestroyColumnFamilyHandle(cfh);
57 }
58
59 handles->clear();
60 }
61
62 delete blob_db_impl;
63 *blob_db = nullptr;
64 }
65 return s;
66}
