21// Supported trace record types.
22enum TraceType : char {
23 kTraceNone = 0,
24 kTraceBegin = 1,
25 kTraceEnd = 2,
26 // Query level tracing related trace types.
27 kTraceWrite = 3,
28 kTraceGet = 4,
29 kTraceIteratorSeek = 5,
30 kTraceIteratorSeekForPrev = 6,
31 // Block cache tracing related trace types.
32 kBlockTraceIndexBlock = 7,
33 // TODO: split out kinds of filter blocks?
34 kBlockTraceFilterBlock = 8,
35 kBlockTraceDataBlock = 9,
36 kBlockTraceUncompressionDictBlock = 10,
37 kBlockTraceRangeDeletionBlock = 11,
38 // IO tracing related trace type.
39 kIOTracer = 12,
40 // Query level tracing related trace type.
41 kTraceMultiGet = 13,
42 // All trace types should be added before kTraceMax
43 kTraceMax,
44};

52// Base class for all types of trace records.
53class TraceRecord {
54 public:
55 explicit TraceRecord(uint64_t timestamp);
56
57 virtual ~TraceRecord() = default;
58
59 // Type of the trace record.
60 virtual TraceType GetTraceType() const = 0;

62 // Timestamp (in microseconds) of this trace.
63 virtual uint64_t GetTimestamp() const;

65 class Handler {

69 virtual Status Handle(const WriteQueryTraceRecord& record,

72 virtual Status Handle(const GetQueryTraceRecord& record,
73 std::unique_ptr<TraceRecordResult>* result) = 0;

75 virtual Status Handle(const IteratorSeekQueryTraceRecord& record,

77
78 virtual Status Handle(const MultiGetQueryTraceRecord& record,
79 std::unique_ptr<TraceRecordResult>* result) = 0;
80 };
81
82 // Accept the handler and report the corresponding result in `result`.
83 virtual Status Accept(Handler* handler,

85

87 static Handler* NewExecutionHandler(

90 private:

92};
