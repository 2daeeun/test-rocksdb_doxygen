36 Status InsertSaved(const Slice& key, const Slice& saved,
37 CompressionType type = kNoCompression,
38 CacheTier source = CacheTier::kVolatileTier) override {
39 CheckCacheKeyCommonPrefix(key);
40 size_t size;
41 char* buf;
42 Status s;
43
44 num_insert_saved_++;
45 size = saved.size();
46 buf = new char[size + sizeof(uint64_t) + 2 * sizeof(uint16_t)];
47 EncodeFixed64(buf, size);
48 buf += sizeof(uint64_t);
49 EncodeFixed16(buf, type);
50 buf += sizeof(uint16_t);
51 EncodeFixed16(buf, (uint16_t)source);
52 buf += sizeof(uint16_t);
53 memcpy(buf, saved.data(), size);
54 buf -= sizeof(uint64_t) + 2 * sizeof(uint16_t);
55 if (!s.ok()) {
56 delete[] buf;
57 return s;
58 }
59 return cache_.Insert(key, buf, size);

62 std::unique_ptr<SecondaryCacheResultHandle> Lookup(
63 const Slice& key, const Cache::CacheItemHelper* helper,
64 Cache::CreateContext* create_context, bool wait, bool /*advise_erase*/,
65 Statistics* /*stats*/, bool& kept_in_sec_cache) override {
66 std::string key_str = key.ToString();
67 TEST_SYNC_POINT_CALLBACK("TestSecondaryCache::Lookup", &key_str);
68
69 std::unique_ptr<SecondaryCacheResultHandle> secondary_handle;
70 kept_in_sec_cache = false;
71
72 TypedHandle* handle = cache_.Lookup(key);
73 if (handle) {
74 num_hits_++;
75 Cache::ObjectPtr value = nullptr;
76 size_t charge = 0;
77 Status s;
78 char* ptr = cache_.Value(handle);
79 CompressionType type;
80 CacheTier source;
81 size_t size = DecodeFixed64(ptr);
82 ptr += sizeof(uint64_t);
83 type = static_cast<CompressionType>(DecodeFixed16(ptr));
84 ptr += sizeof(uint16_t);
85 source = static_cast<CacheTier>(DecodeFixed16(ptr));
86 assert(source == CacheTier::kVolatileTier);
87 ptr += sizeof(uint16_t);
88 s = helper->create_cb(Slice(ptr, size), type, source, create_context,
89 /*alloc*/ nullptr, &value, &charge);
90 if (s.ok()) {
91 secondary_handle.reset(new TestSecondaryCacheResultHandle(
92 cache_.get(), handle, value, charge,
93 /*ready=*/wait || ready_before_wait_));
94 kept_in_sec_cache = true;
95 } else {
96 cache_.Release(handle);
97 }
98 } else {
99 num_misses_++;
100 }
101 return secondary_handle;
102 }

233// In this test, the block size is set to 4096. Each value is 1007 bytes, so
234// each data block contains exactly 4 KV pairs. Metadata blocks are not
235// cached, so we can accurately estimate the cache usage.
236TEST_F(DBTieredSecondaryCacheTest, BasicTest) {
237 if (!LZ4_Supported()) {
238 ROCKSDB_GTEST_SKIP("This test requires LZ4 support.");
239 return;
240 }
241
242 BlockBasedTableOptions table_options;
243 // We want a block cache of size 5KB, and a compressed secondary cache of
244 // size 5KB. However, we specify a block cache size of 256KB here in order
245 // to take into account the cache reservation in the block cache on
246 // behalf of the compressed cache. The unit of cache reservation is 256KB.
247 // The effective block cache capacity will be calculated as 256 + 5 = 261KB,
248 // and 256KB will be reserved for the compressed cache, leaving 5KB for
249 // the primary block cache. We only have to worry about this here because
250 // the cache size is so small.
251 table_options.block_cache = NewCache(256 * 1024, 5 * 1024, 256 * 1024);
252 table_options.block_size = 4 * 1024;
253 table_options.cache_index_and_filter_blocks = false;
254 Options options = GetDefaultOptions();
255 options.create_if_missing = true;
256 options.compression = kLZ4Compression;
257 options.table_factory.reset(NewBlockBasedTableFactory(table_options));
258
259 // Disable paranoid_file_checks so that flush will not read back the newly
260 // written file
261 options.paranoid_file_checks = false;
262 DestroyAndReopen(options);
263 Random rnd(301);
264 const int N = 256;
265 for (int i = 0; i < N; i++) {
266 std::string p_v;
267 test::CompressibleString(&rnd, 0.5, 1007, &p_v);
268 ASSERT_OK(Put(Key(i), p_v));
269 }
270
271 ASSERT_OK(Flush());
272
273 // The first 2 Gets, for keys 0 and 5, will load the corresponding data
274 // blocks as they will be cache misses. The nvm secondary cache will be
275 // warmed up with the compressed blocks
276 std::string v = Get(Key(0));
277 ASSERT_EQ(1007, v.size());
278 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 1u);
279 ASSERT_EQ(nvm_sec_cache()->num_misses(), 1u);
280
281 v = Get(Key(5));
282 ASSERT_EQ(1007, v.size());
283 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
284 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
285
286 // At this point, the nvm cache is warmed up with the data blocks for 0
287 // and 5. The next Get will lookup the block in nvm and will be a hit.
288 // It will be created as a standalone entry in memory, and a placeholder
289 // will be inserted in the primary and compressed caches.
290 v = Get(Key(0));
291 ASSERT_EQ(1007, v.size());
292 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
293 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
294 ASSERT_EQ(nvm_sec_cache()->num_hits(), 1u);
295
296 // For this Get, the primary and compressed only have placeholders for
297 // the required data block. So we will lookup the nvm cache and find the
298 // block there. This time, the block will be promoted to the primary
299 // block cache. No promotion to the compressed secondary cache happens,
300 // and it will retain the placeholder.
301 v = Get(Key(0));
302 ASSERT_EQ(1007, v.size());
303 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
304 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
305 ASSERT_EQ(nvm_sec_cache()->num_hits(), 2u);
306
307 // This Get will find the data block in the primary cache.
308 v = Get(Key(0));
309 ASSERT_EQ(1007, v.size());
310 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
311 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
312 ASSERT_EQ(nvm_sec_cache()->num_hits(), 2u);
313
314 // We repeat the sequence for key 5. This will end up evicting the block
315 // for 0 from the in-memory cache.
316 v = Get(Key(5));
317 ASSERT_EQ(1007, v.size());
318 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
319 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
320 ASSERT_EQ(nvm_sec_cache()->num_hits(), 3u);
321
322 v = Get(Key(5));
323 ASSERT_EQ(1007, v.size());
324 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
325 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
326 ASSERT_EQ(nvm_sec_cache()->num_hits(), 4u);
327
328 v = Get(Key(5));
329 ASSERT_EQ(1007, v.size());
330 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
331 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
332 ASSERT_EQ(nvm_sec_cache()->num_hits(), 4u);
333
334 // This Get for key 0 will find the data block in nvm. Since the compressed
335 // cache still has the placeholder, the block (compressed) will be
336 // admitted. It is theh inserted into the primary as a standalone entry.
337 v = Get(Key(0));
338 ASSERT_EQ(1007, v.size());
339 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
340 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
341 ASSERT_EQ(nvm_sec_cache()->num_hits(), 5u);
342
343 // This Get for key 0 will find the data block in the compressed secondary
344 // cache.
345 v = Get(Key(0));
346 ASSERT_EQ(1007, v.size());
347 ASSERT_EQ(nvm_sec_cache()->num_insert_saved(), 2u);
348 ASSERT_EQ(nvm_sec_cache()->num_misses(), 2u);
349 ASSERT_EQ(nvm_sec_cache()->num_hits(), 5u);
350
351 Destroy(options);
352}
