15struct CustomDeleter {
16 CustomDeleter(MemoryAllocator* a = nullptr) : allocator(a) {}
17
18 void operator()(char* ptr) const {
19 if (allocator) {
20 allocator->Deallocate(ptr);
21 } else {
22 delete[] ptr;
23 }
24 }

26 MemoryAllocator* allocator;

29using CacheAllocationPtr = std::unique_ptr<char[], CustomDeleter>;
30
31inline CacheAllocationPtr AllocateBlock(size_t size,
32 MemoryAllocator* allocator) {
33 if (allocator) {
34 auto block = reinterpret_cast<char*>(allocator->Allocate(size));
35 return CacheAllocationPtr(block, allocator);
36 }
37 return CacheAllocationPtr(new char[size]);
38}
39
40inline CacheAllocationPtr AllocateAndCopyBlock(const Slice& data,
41 MemoryAllocator* allocator) {
42 CacheAllocationPtr cap = AllocateBlock(data.size(), allocator);
43 std::copy_n(data.data(), data.size(), cap.get());
44 return cap;
45}
