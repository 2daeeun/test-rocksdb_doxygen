37void TailPrefetchStats::RecordEffectiveSize(size_t len) {
38 MutexLock l(&mutex_);
39 if (num_records_ < kNumTracked) {
40 num_records_++;
41 }
42 records_[next_++] = len;
43 if (next_ == kNumTracked) {
44 next_ = 0;
45 }

48size_t TailPrefetchStats::GetSuggestedPrefetchSize() {
49 std::vector<size_t> sorted;
50 {
51 MutexLock l(&mutex_);
52
53 if (num_records_ == 0) {
54 return 0;
55 }
56 sorted.assign(records_, records_ + num_records_);
57 }
58
59 // Of the historic size, we find the maximum one that satisifis the condtiion
60 // that if prefetching all, less than 1/8 will be wasted.
61 std::sort(sorted.begin(), sorted.end());
62
63 // Assuming we have 5 data points, and after sorting it looks like this:
64 //
65 // +---+
66 // +---+ | |
67 // | | | |
68 // | | | |
69 // | | | |
70 // | | | |
71 // +---+ | | | |
72 // | | | | | |
73 // +---+ | | | | | |
74 // | | | | | | | |
75 // +---+ | | | | | | | |
76 // | | | | | | | | | |
77 // | | | | | | | | | |
78 // | | | | | | | | | |
79 // | | | | | | | | | |
80 // | | | | | | | | | |
81 // +---+ +---+ +---+ +---+ +---+
82 //
83 // and we use every of the value as a candidate, and estimate how much we
84 // wasted, compared to read. For example, when we use the 3rd record
85 // as candidate. This area is what we read:
86 // +---+
87 // +---+ | |
88 // | | | |
89 // | | | |
90 // | | | |
91 // | | | |
92 // *** *** *** ***+ *** *** *** *** **
93 // * | | | | | |
94 // +---+ | | | | | *
95 // * | | | | | | | |
96 // +---+ | | | | | | | *
97 // * | | | | X | | | | |
98 // | | | | | | | | | *
99 // * | | | | | | | | |
100 // | | | | | | | | | *
101 // * | | | | | | | | |
102 // *** *** ***-*** ***--*** ***--*** +****
103 // which is (size of the record) X (number of records).
104 //
105 // While wasted is this area:
106 // +---+
107 // +---+ | |
108 // | | | |
109 // | | | |
110 // | | | |
111 // | | | |
112 // *** *** *** ****---+ | | | |
113 // * * | | | | |
114 // * *-*** *** | | | | |
115 // * * | | | | | | |
116 // *--** *** | | | | | | |
117 // | | | | | X | | | | |
118 // | | | | | | | | | |
119 // | | | | | | | | | |
120 // | | | | | | | | | |
121 // | | | | | | | | | |
122 // +---+ +---+ +---+ +---+ +---+
123 //
124 // Which can be calculated iteratively.
125 // The difference between wasted using 4th and 3rd record, will
126 // be following area:
127 // +---+
128 // +--+ +-+ ++ +-+ +-+ +---+ | |
129 // + xxxxxxxxxxxxxxxxxxxxxxxx | | | |
130 // xxxxxxxxxxxxxxxxxxxxxxxx | | | |
131 // + xxxxxxxxxxxxxxxxxxxxxxxx | | | |
132 // | xxxxxxxxxxxxxxxxxxxxxxxx | | | |
133 // +-+ +-+ +-+ ++ +---+ +--+ | | |
134 // | | | | | | |
135 // +---+ ++ | | | | | |
136 // | | | | | | X | | |
137 // +---+ ++ | | | | | | | |
138 // | | | | | | | | | |
139 // | | | | | | | | | |
140 // | | | | | | | | | |
141 // | | | | | | | | | |
142 // | | | | | | | | | |
143 // +---+ +---+ +---+ +---+ +---+
144 //
145 // which will be the size difference between 4th and 3rd record,
146 // times 3, which is number of records before the 4th.
147 // Here we assume that all data within the prefetch range will be useful. In
148 // reality, it may not be the case when a partial block is inside the range,
149 // or there are data in the middle that is not read. We ignore those cases
150 // for simplicity.
151 assert(!sorted.empty());
152 size_t prev_size = sorted[0];
153 size_t max_qualified_size = sorted[0];
154 size_t wasted = 0;
155 for (size_t i = 1; i < sorted.size(); i++) {
156 size_t read = sorted[i] * sorted.size();
157 wasted += (sorted[i] - prev_size) * i;
158 if (wasted <= read / 8) {
159 max_qualified_size = sorted[i];
160 }
161 prev_size = sorted[i];
162 }
163 const size_t kMaxPrefetchSize = 512 * 1024; // Never exceed 512KB
164 return std::min(kMaxPrefetchSize, max_qualified_size);
165}

419 table_options_.block_cache));

423void BlockBasedTableFactory::InitializeOptions() {
424 if (table_options_.flush_block_policy_factory == nullptr) {
425 table_options_.flush_block_policy_factory.reset(
426 new FlushBlockBySizePolicyFactory());
427 }
428 if (table_options_.no_block_cache) {
429 table_options_.block_cache.reset();
430 } else if (table_options_.block_cache == nullptr) {
431 LRUCacheOptions co;
432 // 32MB, the recommended minimum size for 64 shards, to reduce contention
433 co.capacity = 32 << 20;
434 table_options_.block_cache = NewLRUCache(co);
435 }
436 if (table_options_.block_size_deviation < 0 ||
437 table_options_.block_size_deviation > 100) {
438 table_options_.block_size_deviation = 0;
439 }
440 if (table_options_.block_restart_interval < 1) {
441 table_options_.block_restart_interval = 1;
442 }
443 if (table_options_.index_block_restart_interval < 1) {
444 table_options_.index_block_restart_interval = 1;
445 }
446 if (table_options_.index_type == BlockBasedTableOptions::kHashSearch &&
447 table_options_.index_block_restart_interval != 1) {
448 // Currently kHashSearch is incompatible with
449 // index_block_restart_interval > 1
450 table_options_.index_block_restart_interval = 1;
451 }
452 if (table_options_.partition_filters &&
453 table_options_.index_type !=
454 BlockBasedTableOptions::kTwoLevelIndexSearch) {
455 // We do not support partitioned filters without partitioning indexes
456 table_options_.partition_filters = false;
457 }
458 auto& options_overrides =
459 table_options_.cache_usage_options.options_overrides;
460 const auto options = table_options_.cache_usage_options.options;
461 for (std::uint32_t i = 0; i < kNumCacheEntryRoles; ++i) {
462 CacheEntryRole role = static_cast<CacheEntryRole>(i);
463 auto options_overrides_iter = options_overrides.find(role);
464 if (options_overrides_iter == options_overrides.end()) {
465 options_overrides.insert({role, options});
466 } else if (options_overrides_iter->second.charged ==
467 CacheEntryRoleOptions::Decision::kFallback) {
468 options_overrides_iter->second.charged = options.charged;
469 }
470 }

479// Different cache kinds use the same keys for physically different values, so
480// they must not share an underlying key space with each other.
481Status CheckCacheOptionCompatibility(const BlockBasedTableOptions& bbto) {
482 int cache_count = (bbto.block_cache != nullptr) +
483 (bbto.persistent_cache != nullptr);
484 if (cache_count <= 1) {
485 // Nothing to share / overlap
486 return Status::OK();
487 }
488
489 // More complex test of shared key space, in case the instances are wrappers
490 // for some shared underlying cache.
491 static Cache::CacheItemHelper kHelper{CacheEntryRole::kMisc};
492 CacheKey sentinel_key = CacheKey::CreateUniqueForProcessLifetime();
493 struct SentinelValue {
494 explicit SentinelValue(char _c) : c(_c) {}
495 char c;
496 };
497 static SentinelValue kRegularBlockCacheMarker{'b'};
498 static char kPersistentCacheMarker{'p'};
499 if (bbto.block_cache) {
500 bbto.block_cache
501 ->Insert(sentinel_key.AsSlice(), &kRegularBlockCacheMarker, &kHelper, 1)
502 .PermitUncheckedError();
503 }
504 if (bbto.persistent_cache) {
505 // Note: persistent cache copies the data, not keeping the pointer
506 bbto.persistent_cache
507 ->Insert(sentinel_key.AsSlice(), &kPersistentCacheMarker, 1)
508 .PermitUncheckedError();
509 }
510 // If we get something different from what we inserted, that indicates
511 // dangerously overlapping key spaces.
512 if (bbto.block_cache) {
513 auto handle = bbto.block_cache->Lookup(sentinel_key.AsSlice());
514 if (handle) {
515 auto v = static_cast<SentinelValue*>(bbto.block_cache->Value(handle));
516 char c = v->c;
517 bbto.block_cache->Release(handle);
518 if (c == kPersistentCacheMarker) {
519 return Status::InvalidArgument(
520 "block_cache and persistent_cache share the same key space, "
521 "which is not supported");
522 } else if (v != &kRegularBlockCacheMarker) {
523 return Status::Corruption("Unexpected mutation to block_cache");
524 }
525 }
526 }
527
528 if (bbto.persistent_cache) {
529 std::unique_ptr<char[]> data;
530 size_t size = 0;
531 bbto.persistent_cache->Lookup(sentinel_key.AsSlice(), &data, &size)
532 .PermitUncheckedError();
533 if (data && size > 0) {
534 if (data[0] == kRegularBlockCacheMarker.c) {
535 return Status::InvalidArgument(
536 "persistent_cache and block_cache share the same key space, "
537 "which is not supported");
538 } else if (data[0] != kPersistentCacheMarker) {
539 return Status::Corruption("Unexpected mutation to persistent_cache");
540 }
541 }
542 }
543 return Status::OK();
544}

578Status BlockBasedTableFactory::ValidateOptions(
579 const DBOptions& db_opts, const ColumnFamilyOptions& cf_opts) const {
580 if (table_options_.index_type == BlockBasedTableOptions::kHashSearch &&
581 cf_opts.prefix_extractor == nullptr) {
582 return Status::InvalidArgument(
583 "Hash index is specified for block-based "

594 return Status::InvalidArgument(

596 ", but block cache is disabled");
597 }
598 if (!IsSupportedFormatVersion(table_options_.format_version)) {
599 return Status::InvalidArgument(
600 "Unsupported BlockBasedTable format_version. Please check "
601 "include/rocksdb/table.h for more info");
602 }
603 if (table_options_.block_align && (cf_opts.compression != kNoCompression)) {
604 return Status::InvalidArgument(

622 }

630 "block size exceeds maximum number (4GiB) allowed");

634 table_options_.data_block_hash_table_util_ratio <= 0) {
635 return Status::InvalidArgument(

637 "data_block_index_type is set to kDataBlockBinaryAndHash");
638 }
639 if (db_opts.unordered_write && cf_opts.max_successive_merges > 0) {
640 // TODO(myabandeh): support it
641 return Status::InvalidArgument(
642 "max_successive_merges larger than 0 is currently inconsistent with "
643 "unordered_write");
644 }

646 table_options_.cache_usage_options.options_overrides;
647 for (auto options_overrides_iter = options_overrides.cbegin();
648 options_overrides_iter != options_overrides.cend();
649 ++options_overrides_iter) {
650 const CacheEntryRole role = options_overrides_iter->first;
651 const CacheEntryRoleOptions options = options_overrides_iter->second;
652 static const std::set<CacheEntryRole> kMemoryChargingSupported = {
653 CacheEntryRole::kCompressionDictionaryBuildingBuffer,
654 CacheEntryRole::kFilterConstruction,
655 CacheEntryRole::kBlockBasedTableReader, CacheEntryRole::kFileMetadata,
656 CacheEntryRole::kBlobCache};

660 "Enable/Disable CacheEntryRoleOptions::charged"
661 " for CacheEntryRole " +
662 kCacheEntryRoleToCamelString[static_cast<uint32_t>(role)] +
663 " is not supported");

669 " for CacheEntryRole " +
670 kCacheEntryRoleToCamelString[static_cast<uint32_t>(role)] +
671 " but block cache is disabled");
672 }

674 options.charged == CacheEntryRoleOptions::Decision::kEnabled) {
675 if (cf_opts.blob_cache == nullptr) {
676 return Status::InvalidArgument(
677 "Enable CacheEntryRoleOptions::charged"
678 " for CacheEntryRole " +
679 kCacheEntryRoleToCamelString[static_cast<uint32_t>(role)] +
680 " but blob cache is not configured");
681 }
682 if (table_options_.no_block_cache) {
683 return Status::InvalidArgument(
684 "Enable CacheEntryRoleOptions::charged"
685 " for CacheEntryRole " +
686 kCacheEntryRoleToCamelString[static_cast<uint32_t>(role)] +
687 " but block cache is disabled");
688 }
689 if (table_options_.block_cache == cf_opts.blob_cache) {
690 return Status::InvalidArgument(
691 "Enable CacheEntryRoleOptions::charged"
692 " for CacheEntryRole " +
693 kCacheEntryRoleToCamelString[static_cast<uint32_t>(role)] +
694 " but blob cache is the same as block cache");
695 }
696 if (cf_opts.blob_cache->GetCapacity() >
697 table_options_.block_cache->GetCapacity()) {
698 return Status::InvalidArgument(
699 "Enable CacheEntryRoleOptions::charged"
700 " for CacheEntryRole " +
701 kCacheEntryRoleToCamelString[static_cast<uint32_t>(role)] +
702 " but blob cache capacity is larger than block cache capacity");
703 }
704 }
705 }
706 {
707 Status s = CheckCacheOptionCompatibility(table_options_);
708 if (!s.ok()) {
709 return s;
710 }
711 }
712 std::string garbage;
713 if (!SerializeEnum<ChecksumType>(checksum_type_string_map,
714 table_options_.checksum, &garbage)) {
715 return Status::InvalidArgument(
716 "Unrecognized ChecksumType for checksum: " +
717 std::to_string(static_cast<uint32_t>(table_options_.checksum)));
718 }
719 return TableFactory::ValidateOptions(db_opts, cf_opts);

859// Take a default BlockBasedTableOptions "table_options" in addition to a
860// map "opts_map" of option name to option value to construct the new
861// BlockBasedTableOptions "new_table_options".
862//
863// Below are the instructions of how to config some non-primitive-typed
864// options in BlockBasedTableOptions:
865//
866// * filter_policy:
867// We currently only support the following FilterPolicy in the convenience
868// functions:
869// - BloomFilter: use "bloomfilter:[bits_per_key]:[use_block_based_builder]"
870// to specify BloomFilter. The above string is equivalent to calling
871// NewBloomFilterPolicy(bits_per_key, use_block_based_builder).
872// [Example]:
873// - Pass {"filter_policy", "bloomfilter:4:true"} in
874// GetBlockBasedTableOptionsFromMap to use a BloomFilter with 4-bits
875// per key and use_block_based_builder enabled.
876//
877// * block_cache / block_cache_compressed:
878// We currently only support LRU cache in the GetOptions API. The LRU
879// cache can be set by directly specifying its size.
880// [Example]:
881// - Passing {"block_cache", "1M"} in GetBlockBasedTableOptionsFromMap is
882// equivalent to setting block_cache using NewLRUCache(1024 * 1024).
883//
884// @param table_options the default options of the output "new_table_options".
885// @param opts_map an option name to value map for specifying how
886// "new_table_options" should be set.
887// @param new_table_options the resulting options based on "table_options"
888// with the change specified in "opts_map".
889// @param input_strings_escaped when set to true, each escaped characters
890// prefixed by '\' in the values of the opts_map will be further converted
891// back to the raw string before assigning to the associated options.
892// @param ignore_unknown_options when set to true, unknown options are ignored
893// instead of resulting in an unknown-option error.
894// @return Status::OK() on success. Otherwise, a non-ok status indicating
895// error will be returned, and "new_table_options" will be set to
896// "table_options".
897Status BlockBasedTableFactory::ParseOption(const ConfigOptions& config_options,
898 const OptionTypeInfo& opt_info,
899 const std::string& opt_name,
900 const std::string& opt_value,
901 void* opt_ptr) {
902 Status status = TableFactory::ParseOption(config_options, opt_info, opt_name,
903 opt_value, opt_ptr);
904 if (config_options.input_strings_escaped && !status.ok()) { // Got an error
905 // !input_strings_escaped indicates the old API, where everything is
906 // parsable.
907 if (opt_info.IsByName()) {
908 status = Status::OK();
909 }
910 }
911 return status;
912}
