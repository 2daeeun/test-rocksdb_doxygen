111std::shared_ptr<Tombstone> ExpiringColumn::ToTombstone() const {
112 auto expired_at = (TimePoint() + Ttl()).time_since_epoch();
113 int32_t local_deletion_time = static_cast<int32_t>(
114 std::chrono::duration_cast<std::chrono::seconds>(expired_at).count());
115 int64_t marked_for_delete_at =
116 std::chrono::duration_cast<std::chrono::microseconds>(expired_at).count();
117 return std::make_shared<Tombstone>(
118 static_cast<int8_t>(ColumnTypeMask::DELETION_MASK), Index(),
119 local_deletion_time, marked_for_delete_at);
120}

228 std::shared_ptr<ExpiringColumn> expiring_column =

281RowValue RowValue::Deserialize(const char* src, std::size_t size) {
282 std::size_t offset = 0;
283 assert(size >= sizeof(local_deletion_time_) + sizeof(marked_for_delete_at_));
284 int32_t local_deletion_time =
285 ROCKSDB_NAMESPACE::cassandra::Deserialize<int32_t>(src, offset);
286 offset += sizeof(int32_t);
287 int64_t marked_for_delete_at =
288 ROCKSDB_NAMESPACE::cassandra::Deserialize<int64_t>(src, offset);
289 offset += sizeof(int64_t);
290 if (offset == size) {
291 return RowValue(local_deletion_time, marked_for_delete_at);
292 }
293
294 assert(local_deletion_time == kDefaultLocalDeletionTime);
295 assert(marked_for_delete_at == kDefaultMarkedForDeleteAt);
296 Columns columns;
297 int64_t last_modified_time = 0;
298 while (offset < size) {
299 auto c = ColumnBase::Deserialize(src, offset);
300 offset += c->Size();
301 assert(offset <= size);
302 last_modified_time = std::max(last_modified_time, c->Timestamp());
303 columns.push_back(std::move(c));
304 }
305
306 return RowValue(std::move(columns), last_modified_time);
307}

309// Merge multiple row values into one.
310// For each column in rows with same index, we pick the one with latest
311// timestamp. And we also take row tombstone into consideration, by iterating
312// each row from reverse timestamp order, and stop once we hit the first
313// row tombstone.
314RowValue RowValue::Merge(std::vector<RowValue>&& values) {
315 assert(values.size() > 0);
316 if (values.size() == 1) {
317 return std::move(values[0]);
318 }
319
320 // Merge columns by their last modified time, and skip once we hit
321 // a row tombstone.
322 std::sort(values.begin(), values.end(),
323 [](const RowValue& r1, const RowValue& r2) {
324 return r1.LastModifiedTime() > r2.LastModifiedTime();
325 });
326
327 std::map<int8_t, std::shared_ptr<ColumnBase>> merged_columns;
328 int64_t tombstone_timestamp = 0;
329
330 for (auto& value : values) {
331 if (value.IsTombstone()) {
332 if (merged_columns.size() == 0) {
333 return std::move(value);
334 }
335 tombstone_timestamp = value.LastModifiedTime();
336 break;
337 }
338 for (auto& column : value.columns_) {
339 int8_t index = column->Index();
340 if (merged_columns.find(index) == merged_columns.end()) {
341 merged_columns[index] = column;
342 } else {
343 if (column->Timestamp() > merged_columns[index]->Timestamp()) {
344 merged_columns[index] = column;
345 }
346 }
347 }
348 }
349
350 int64_t last_modified_time = 0;
351 Columns columns;
352 for (auto& pair : merged_columns) {
353 // For some row, its last_modified_time > row tombstone_timestamp, but
354 // it might have rows whose timestamp is ealier than tombstone, so we
355 // ned to filter these rows.
356 if (pair.second->Timestamp() <= tombstone_timestamp) {
357 continue;
358 }
359 last_modified_time = std::max(last_modified_time, pair.second->Timestamp());
360 columns.push_back(std::move(pair.second));
361 }
362 return RowValue(std::move(columns), last_modified_time);
363}
