22bool GhostCache::Admit(const Slice& lookup_key) {
23 auto handle = sim_cache_->Lookup(lookup_key);
24 if (handle != nullptr) {
25 sim_cache_->Release(handle);
26 return true;
27 }
28 // TODO: Should we check for errors here?
29 auto s = sim_cache_->Insert(lookup_key, /*obj=*/nullptr,
30 &kNoopCacheItemHelper, lookup_key.size());
31 s.PermitUncheckedError();
32 return false;
33}

39void CacheSimulator::Access(const BlockCacheTraceRecord& access) {
40 bool admit = true;
41 const bool is_user_access =
42 BlockCacheTraceHelper::IsUserAccess(access.caller);
43 bool is_cache_miss = true;
44 if (ghost_cache_ && !access.no_insert) {
45 admit = ghost_cache_->Admit(access.block_key);
46 }
47 auto handle = sim_cache_->Lookup(access.block_key);
48 if (handle != nullptr) {
49 sim_cache_->Release(handle);
50 is_cache_miss = false;
51 } else {
52 if (!access.no_insert && admit && access.block_size > 0) {
53 // Ignore errors on insert
54 auto s = sim_cache_->Insert(access.block_key, /*obj=*/nullptr,
55 &kNoopCacheItemHelper, access.block_size);
56 s.PermitUncheckedError();
57 }
58 }
59 miss_ratio_stats_.UpdateMetrics(access.access_timestamp, is_user_access,
60 is_cache_miss);
61}

63void MissRatioStats::UpdateMetrics(uint64_t timestamp_in_ms,
64 bool is_user_access, bool is_cache_miss) {
65 uint64_t timestamp_in_seconds = timestamp_in_ms / kMicrosInSecond;
66 num_accesses_timeline_[timestamp_in_seconds] += 1;
67 num_accesses_ += 1;
68 if (num_misses_timeline_.find(timestamp_in_seconds) ==
69 num_misses_timeline_.end()) {
70 num_misses_timeline_[timestamp_in_seconds] = 0;
71 }
72 if (is_cache_miss) {
73 num_misses_ += 1;
74 num_misses_timeline_[timestamp_in_seconds] += 1;
75 }
76 if (is_user_access) {
77 user_accesses_ += 1;
78 if (is_cache_miss) {
79 user_misses_ += 1;
80 }
81 }
82}

84Cache::Priority PrioritizedCacheSimulator::ComputeBlockPriority(
85 const BlockCacheTraceRecord& access) const {
86 if (access.block_type == TraceType::kBlockTraceFilterBlock ||
87 access.block_type == TraceType::kBlockTraceIndexBlock ||
88 access.block_type == TraceType::kBlockTraceUncompressionDictBlock) {
89 return Cache::Priority::HIGH;
90 }
91 return Cache::Priority::LOW;
92}

94void PrioritizedCacheSimulator::AccessKVPair(
95 const Slice& key, uint64_t value_size, Cache::Priority priority,
96 const BlockCacheTraceRecord& access, bool no_insert, bool is_user_access,
97 bool* is_cache_miss, bool* admitted, bool update_metrics) {
98 assert(is_cache_miss);
99 assert(admitted);
100 *is_cache_miss = true;
101 *admitted = true;
102 if (ghost_cache_ && !no_insert) {
103 *admitted = ghost_cache_->Admit(key);
104 }
105 auto handle = sim_cache_->Lookup(key);
106 if (handle != nullptr) {
107 sim_cache_->Release(handle);
108 *is_cache_miss = false;
109 } else if (!no_insert && *admitted && value_size > 0) {
110 // TODO: Should we check for an error here?
111 auto s = sim_cache_->Insert(key, /*obj=*/nullptr, &kNoopCacheItemHelper,
112 value_size,
113 /*handle=*/nullptr, priority);
114 s.PermitUncheckedError();
115 }
116 if (update_metrics) {
117 miss_ratio_stats_.UpdateMetrics(access.access_timestamp, is_user_access,
118 *is_cache_miss);
119 }
120}

131void HybridRowBlockCacheSimulator::Access(const BlockCacheTraceRecord& access) {
132 // TODO (haoyu): We only support Get for now. We need to extend the tracing
133 // for MultiGet, i.e., non-data block accesses must log all keys in a
134 // MultiGet.
135 bool is_cache_miss = true;
136 bool admitted = false;
137 if (access.caller == TableReaderCaller::kUserGet &&
138 access.get_id != BlockCacheTraceHelper::kReservedGetId) {
139 // This is a Get request.
140 const std::string& row_key = BlockCacheTraceHelper::ComputeRowKey(access);
141 GetRequestStatus& status = getid_status_map_[access.get_id];
142 if (status.is_complete) {
143 // This Get request completes.
144 // Skip future accesses to its index/filter/data
145 // blocks. These block lookups are unnecessary if we observe a hit for the
146 // referenced key-value pair already. Thus, we treat these lookups as
147 // hits. This is also to ensure the total number of accesses are the same
148 // when comparing to other policies.
149 miss_ratio_stats_.UpdateMetrics(access.access_timestamp,
150 /*is_user_access=*/true,
151 /*is_cache_miss=*/false);
152 return;
153 }
154 if (status.row_key_status.find(row_key) == status.row_key_status.end()) {
155 // This is the first time that this key is accessed. Look up the key-value
156 // pair first. Do not update the miss/accesses metrics here since it will
157 // be updated later.
158 AccessKVPair(row_key, access.referenced_data_size, Cache::Priority::HIGH,
159 access,
160 /*no_insert=*/false,
161 /*is_user_access=*/true, &is_cache_miss, &admitted,
162 /*update_metrics=*/false);
163 InsertResult result = InsertResult::NO_INSERT;
164 if (admitted && access.referenced_data_size > 0) {
165 result = InsertResult::INSERTED;
166 } else if (admitted) {
167 result = InsertResult::ADMITTED;
168 }
169 status.row_key_status[row_key] = result;
170 }
171 if (!is_cache_miss) {
172 // A cache hit.
173 status.is_complete = true;
174 miss_ratio_stats_.UpdateMetrics(access.access_timestamp,
175 /*is_user_access=*/true,
176 /*is_cache_miss=*/false);
177 return;
178 }
179 // The row key-value pair observes a cache miss. We need to access its
180 // index/filter/data blocks.
181 InsertResult inserted = status.row_key_status[row_key];
182 AccessKVPair(
183 access.block_key, access.block_size, ComputeBlockPriority(access),
184 access,
185 /*no_insert=*/!insert_blocks_upon_row_kvpair_miss_ || access.no_insert,
186 /*is_user_access=*/true, &is_cache_miss, &admitted,
187 /*update_metrics=*/true);
188 if (access.referenced_data_size > 0 && inserted == InsertResult::ADMITTED) {
189 // TODO: Should we check for an error here?
190 auto s =
191 sim_cache_->Insert(row_key, /*obj=*/nullptr, &kNoopCacheItemHelper,
192 access.referenced_data_size,
193 /*handle=*/nullptr, Cache::Priority::HIGH);
194 s.PermitUncheckedError();
195 status.row_key_status[row_key] = InsertResult::INSERTED;
196 }
197 return;
198 }
199 AccessKVPair(access.block_key, access.block_size,
200 ComputeBlockPriority(access), access, access.no_insert,
201 BlockCacheTraceHelper::IsUserAccess(access.caller),
202 &is_cache_miss, &admitted, /*update_metrics=*/true);
203}

212Status BlockCacheTraceSimulator::InitializeCaches() {
213 for (auto const& config : cache_configurations_) {
214 for (auto cache_capacity : config.cache_capacities) {
215 // Scale down the cache capacity since the trace contains accesses on
216 // 1/'downsample_ratio' blocks.
217 uint64_t simulate_cache_capacity = cache_capacity / downsample_ratio_;
218 std::shared_ptr<CacheSimulator> sim_cache;
219 std::unique_ptr<GhostCache> ghost_cache;
220 std::string cache_name = config.cache_name;
221 if (cache_name.find(kGhostCachePrefix) != std::string::npos) {
222 ghost_cache.reset(new GhostCache(
223 NewLRUCache(config.ghost_cache_capacity, /*num_shard_bits=*/1,
224 /*strict_capacity_limit=*/false,
225 /*high_pri_pool_ratio=*/0)));
226 cache_name = cache_name.substr(kGhostCachePrefix.size());
227 }
228 if (cache_name == "lru") {
229 sim_cache = std::make_shared<CacheSimulator>(
230 std::move(ghost_cache),
231 NewLRUCache(simulate_cache_capacity, config.num_shard_bits,
232 /*strict_capacity_limit=*/false,
233 /*high_pri_pool_ratio=*/0));
234 } else if (cache_name == "lru_priority") {
235 sim_cache = std::make_shared<PrioritizedCacheSimulator>(
236 std::move(ghost_cache),
237 NewLRUCache(simulate_cache_capacity, config.num_shard_bits,
238 /*strict_capacity_limit=*/false,
239 /*high_pri_pool_ratio=*/0.5));
240 } else if (cache_name == "lru_hybrid") {
241 sim_cache = std::make_shared<HybridRowBlockCacheSimulator>(
242 std::move(ghost_cache),
243 NewLRUCache(simulate_cache_capacity, config.num_shard_bits,
244 /*strict_capacity_limit=*/false,
245 /*high_pri_pool_ratio=*/0.5),
246 /*insert_blocks_upon_row_kvpair_miss=*/true);
247 } else if (cache_name == "lru_hybrid_no_insert_on_row_miss") {
248 sim_cache = std::make_shared<HybridRowBlockCacheSimulator>(
249 std::move(ghost_cache),
250 NewLRUCache(simulate_cache_capacity, config.num_shard_bits,
251 /*strict_capacity_limit=*/false,
252 /*high_pri_pool_ratio=*/0.5),
253 /*insert_blocks_upon_row_kvpair_miss=*/false);
254 } else {
255 // Not supported.
256 return Status::InvalidArgument("Unknown cache name " +
257 config.cache_name);
258 }
259 sim_caches_[config].push_back(sim_cache);
260 }
261 }
262 return Status::OK();
263}

265void BlockCacheTraceSimulator::Access(const BlockCacheTraceRecord& access) {
266 if (trace_start_time_ == 0) {
267 trace_start_time_ = access.access_timestamp;
268 }
269 // access.access_timestamp is in microseconds.
270 if (!warmup_complete_ &&
271 trace_start_time_ + warmup_seconds_ * kMicrosInSecond <=
272 access.access_timestamp) {
273 for (auto& config_caches : sim_caches_) {
274 for (auto& sim_cache : config_caches.second) {
275 sim_cache->reset_counter();
276 }
277 }
278 warmup_complete_ = true;
279 }
280 for (auto& config_caches : sim_caches_) {
281 for (auto& sim_cache : config_caches.second) {
282 sim_cache->Access(access);
283 }
284 }
