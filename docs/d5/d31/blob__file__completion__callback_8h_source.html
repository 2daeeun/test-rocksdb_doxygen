41 Status OnBlobFileCompleted(const std::string& file_name,
42 const std::string& column_family_name, int job_id,
43 uint64_t file_number,
44 BlobFileCreationReason creation_reason,
45 const Status& report_status,
46 const std::string& checksum_value,
47 const std::string& checksum_method,
48 uint64_t blob_count, uint64_t blob_bytes) {
49 Status s;
50
51 auto sfm = static_cast<SstFileManagerImpl*>(sst_file_manager_);
52 if (sfm) {
53 // Report new blob files to SstFileManagerImpl
54 s = sfm->OnAddFile(file_name);
55 if (sfm->IsMaxAllowedSpaceReached()) {
56 s = Status::SpaceLimit("Max allowed space was reached");
57 TEST_SYNC_POINT(
58 "BlobFileCompletionCallback::CallBack::MaxAllowedSpaceReached");
59 InstrumentedMutexLock l(mutex_);
60 error_handler_->SetBGError(s, BackgroundErrorReason::kFlush);
61 }
62 }
63
64 // Notify the listeners.
65 EventHelpers::LogAndNotifyBlobFileCreationFinished(
66 event_logger_, listeners_, dbname_, column_family_name, file_name,
67 job_id, file_number, creation_reason,
68 (!report_status.ok() ? report_status : s),
69 (checksum_value.empty() ? kUnknownFileChecksum : checksum_value),
70 (checksum_method.empty() ? kUnknownFileChecksumFuncName
71 : checksum_method),
72 blob_count, blob_bytes);
73 return s;
74 }
75
76 private:
77 SstFileManager* sst_file_manager_;
78 InstrumentedMutex* mutex_;
79 ErrorHandler* error_handler_;
80 EventLogger* event_logger_;
81 std::vector<std::shared_ptr<EventListener>> listeners_;
82 std::string dbname_;
83};
