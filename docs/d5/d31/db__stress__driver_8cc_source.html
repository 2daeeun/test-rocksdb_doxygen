17void ThreadBody(void* v) {
18 ThreadStatusUtil::RegisterThread(db_stress_env, ThreadStatus::USER);
19 ThreadState* thread = static_cast<ThreadState*>(v);
20 SharedState* shared = thread->shared;
21
22 if (!FLAGS_skip_verifydb && shared->ShouldVerifyAtBeginning()) {
23 thread->shared->GetStressTest()->VerifyDb(thread);
24 }
25 {
26 MutexLock l(shared->GetMutex());
27 shared->IncInitialized();
28 if (shared->AllInitialized()) {
29 shared->GetCondVar()->SignalAll();
30 }
31 }
32 if (!FLAGS_verification_only) {
33 {
34 MutexLock l(shared->GetMutex());
35 while (!shared->Started()) {
36 shared->GetCondVar()->Wait();
37 }
38 }
39 thread->shared->GetStressTest()->OperateDb(thread);
40 {
41 MutexLock l(shared->GetMutex());
42 shared->IncOperated();
43 if (shared->AllOperated()) {
44 shared->GetCondVar()->SignalAll();
45 }
46 while (!shared->VerifyStarted()) {
47 shared->GetCondVar()->Wait();
48 }
49 }
50
51 if (!FLAGS_skip_verifydb) {
52 thread->shared->GetStressTest()->VerifyDb(thread);
53 }
54
55 {
56 MutexLock l(shared->GetMutex());
57 shared->IncDone();
58 if (shared->AllDone()) {
59 shared->GetCondVar()->SignalAll();
60 }
61 }
62 }
63
64 ThreadStatusUtil::UnregisterThread();
65}
66bool RunStressTestImpl(SharedState* shared) {
67 SystemClock* clock = db_stress_env->GetSystemClock().get();
68 StressTest* stress = shared->GetStressTest();
69
70 if (shared->ShouldVerifyAtBeginning() && FLAGS_preserve_unverified_changes) {
71 Status s = InitUnverifiedSubdir(FLAGS_db);
72 if (s.ok() && !FLAGS_expected_values_dir.empty()) {
73 s = InitUnverifiedSubdir(FLAGS_expected_values_dir);
74 }
75 if (!s.ok()) {
76 fprintf(stderr, "Failed to setup unverified state dir: %s\n",
77 s.ToString().c_str());
78 exit(1);
79 }
80 }
81
82 stress->InitDb(shared);
83 stress->FinishInitDb(shared);
84
85 uint32_t n = FLAGS_threads;
86 uint64_t now = clock->NowMicros();
87 fprintf(stdout, "%s Initializing worker threads\n",
88 clock->TimeToString(now / 1000000).c_str());
89
90 shared->SetThreads(n);
91
92 if (FLAGS_compaction_thread_pool_adjust_interval > 0) {
93 shared->IncBgThreads();
94 }
95
96 if (FLAGS_continuous_verification_interval > 0) {
97 shared->IncBgThreads();
98 }
99
100 if (FLAGS_compressed_secondary_cache_size > 0 ||
101 FLAGS_compressed_secondary_cache_ratio > 0.0) {
102 shared->IncBgThreads();
103 }
104
105 std::vector<ThreadState*> threads(n);
106 for (uint32_t i = 0; i < n; i++) {
107 threads[i] = new ThreadState(i, shared);
108 db_stress_env->StartThread(ThreadBody, threads[i]);
109 }
110
111 ThreadState bg_thread(0, shared);
112 if (FLAGS_compaction_thread_pool_adjust_interval > 0) {
113 db_stress_env->StartThread(PoolSizeChangeThread, &bg_thread);
114 }
115
116 ThreadState continuous_verification_thread(0, shared);
117 if (FLAGS_continuous_verification_interval > 0) {
118 db_stress_env->StartThread(DbVerificationThread,
119 &continuous_verification_thread);
120 }
121
122 ThreadState compressed_cache_set_capacity_thread(0, shared);
123 if (FLAGS_compressed_secondary_cache_size > 0 ||
124 FLAGS_compressed_secondary_cache_ratio > 0.0) {
125 db_stress_env->StartThread(CompressedCacheSetCapacityThread,
126 &compressed_cache_set_capacity_thread);
127 }
128
129 // Each thread goes through the following states:
130 // initializing -> wait for others to init -> read/populate/depopulate
131 // wait for others to operate -> verify -> done
132
133 {
134 MutexLock l(shared->GetMutex());
135 while (!shared->AllInitialized()) {
136 shared->GetCondVar()->Wait();
137 }
138 if (shared->ShouldVerifyAtBeginning()) {
139 if (shared->HasVerificationFailedYet()) {
140 fprintf(stderr, "Crash-recovery verification failed :(\n");
141 } else {
142 fprintf(stdout, "Crash-recovery verification passed :)\n");
143 Status s = DestroyUnverifiedSubdir(FLAGS_db);
144 if (s.ok() && !FLAGS_expected_values_dir.empty()) {
145 s = DestroyUnverifiedSubdir(FLAGS_expected_values_dir);
146 }
147 if (!s.ok()) {
148 fprintf(stderr, "Failed to cleanup unverified state dir: %s\n",
149 s.ToString().c_str());
150 exit(1);
151 }
152 }
153 }
154
155 if (!FLAGS_verification_only) {
156 // This is after the verification step to avoid making all those `Get()`s
157 // and `MultiGet()`s contend on the DB-wide trace mutex.
158 if (!FLAGS_expected_values_dir.empty()) {
159 stress->TrackExpectedState(shared);
160 }
161
162 if (FLAGS_sync_fault_injection || FLAGS_write_fault_one_in > 0) {
163 fault_fs_guard->SetFilesystemDirectWritable(false);
164 fault_fs_guard->SetInjectUnsyncedDataLoss(FLAGS_sync_fault_injection);
165 if (FLAGS_exclude_wal_from_write_fault_injection) {
166 fault_fs_guard->SetFileTypesExcludedFromWriteFaultInjection(
167 {FileType::kWalFile});
168 }
169 }
170 if (ShouldDisableAutoCompactionsBeforeVerifyDb()) {
171 Status s = stress->EnableAutoCompaction();
172 assert(s.ok());
173 }
174 fprintf(stdout, "%s Starting database operations\n",
175 clock->TimeToString(now / 1000000).c_str());
176
177 shared->SetStart();
178 shared->GetCondVar()->SignalAll();
179 while (!shared->AllOperated()) {
180 shared->GetCondVar()->Wait();
181 }
182
183 now = clock->NowMicros();
184 if (FLAGS_test_batches_snapshots) {
185 fprintf(stdout, "%s Limited verification already done during gets\n",
186 clock->TimeToString((uint64_t)now / 1000000).c_str());
187 } else if (FLAGS_skip_verifydb) {
188 fprintf(stdout, "%s Verification skipped\n",
189 clock->TimeToString((uint64_t)now / 1000000).c_str());
190 } else {
191 fprintf(stdout, "%s Starting verification\n",
192 clock->TimeToString((uint64_t)now / 1000000).c_str());
193 }
194
195 shared->SetStartVerify();
196 shared->GetCondVar()->SignalAll();
197 while (!shared->AllDone()) {
198 shared->GetCondVar()->Wait();
199 }
200 }
201 }
202
203 // If we are running verification_only
204 // stats will be empty and trying to report them will
205 // emit no ops or writes error. To avoid this, merging and reporting stats
206 // are not executed when running with verification_only
207 // TODO: We need to create verification stats (e.g. how many keys
208 // are verified by which method) and report them here instead of operation
209 // stats.
210 if (!FLAGS_verification_only) {
211 for (unsigned int i = 1; i < n; i++) {
212 threads[0]->stats.Merge(threads[i]->stats);
213 }
214 threads[0]->stats.Report("Stress Test");
215 }
216
217 for (unsigned int i = 0; i < n; i++) {
218 delete threads[i];
219 threads[i] = nullptr;
220 }
221 now = clock->NowMicros();
222 if (!FLAGS_skip_verifydb && !FLAGS_test_batches_snapshots &&
223 !shared->HasVerificationFailedYet()) {
224 fprintf(stdout, "%s Verification successful\n",
225 clock->TimeToString(now / 1000000).c_str());
226 }
227
228 if (!FLAGS_verification_only) {
229 stress->PrintStatistics();
230 }
231
232 if (FLAGS_compaction_thread_pool_adjust_interval > 0 ||
233 FLAGS_continuous_verification_interval > 0 ||
234 FLAGS_compressed_secondary_cache_size > 0 ||
235 FLAGS_compressed_secondary_cache_ratio > 0.0) {
236 MutexLock l(shared->GetMutex());
237 shared->SetShouldStopBgThread();
238 while (!shared->BgThreadsFinished()) {
239 shared->GetCondVar()->Wait();
240 }
241 }
242
243 if (shared->HasVerificationFailedYet()) {
244 fprintf(stderr, "Verification failed :(\n");
245 return false;
246 }
247 return true;
248}
249bool RunStressTest(SharedState* shared) {
250 ThreadStatusUtil::RegisterThread(db_stress_env, ThreadStatus::USER);
251 bool result = RunStressTestImpl(shared);
252 ThreadStatusUtil::UnregisterThread();
253 return result;
254}
