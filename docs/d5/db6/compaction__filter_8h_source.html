28// CompactionFilter allows an application to modify/delete a key-value during
29// table file creation.
30//
31// Some general notes:
32//
33// * RocksDB snapshots do not guarantee to preserve the state of the DB in the
34// presence of CompactionFilter. Data seen from a snapshot might disappear after
35// a table file created with a `CompactionFilter` is installed. If you use
36// snapshots, think twice about whether you want to use `CompactionFilter` and
37// whether you are using it in a safe way.
38//
39// * If multithreaded compaction is being used *and* a single CompactionFilter
40// instance was supplied via Options::compaction_filter, CompactionFilter
41// methods may be called from different threads concurrently. The application
42// must ensure that such calls are thread-safe. If the CompactionFilter was
43// created by a factory, then it will only ever be used by a single thread that
44// is doing the table file creation, and this call does not need to be
45// thread-safe. However, multiple filters may be in existence and operating
46// concurrently.
47//
48// * The key passed to the filtering methods includes the timestamp if
49// user-defined timestamps are enabled.
50//
51// * Exceptions MUST NOT propagate out of overridden functions into RocksDB,
52// because RocksDB is not exception-safe. This could cause undefined behavior
53// including data loss, unreported corruption, deadlocks, and more.
54class CompactionFilter : public Customizable {
55 public:
56 // Value type of the key-value passed to the compaction filter's FilterV2/V3
57 // methods.
58 enum ValueType {
59 // Plain key-value
60 kValue,
61 // Merge operand
62 kMergeOperand,
63 // Used internally by the old stacked BlobDB implementation; this value type
64 // is never passed to application code. Note that when using the new
65 // integrated BlobDB, values stored separately as blobs are retrieved and
66 // presented to FilterV2/V3 with the type kValue above.
67 kBlobIndex,
68 // Wide-column entity
69 kWideColumnEntity,
70 };
71
72 // Potential decisions that can be returned by the compaction filter's
73 // FilterV2/V3 and FilterBlobByKey methods. See decision-specific caveats and
74 // constraints below.
75 enum class Decision {
76 // Keep the current key-value as-is.
77 kKeep,
78
79 // Remove the current key-value. Note that the semantics of removal are
80 // dependent on the value type. If the current key-value is a plain
81 // key-value or a wide-column entity, it is converted to a tombstone
82 // (Delete), resulting in the deletion of any earlier versions of the key.
83 // If it is a merge operand, it is simply dropped. Note: if you are using
84 // a TransactionDB, it is not recommended to filter out merge operands.
85 // If a Merge operation is filtered out, TransactionDB may not realize there
86 // is a write conflict and may allow a Transaction that should have failed
87 // to Commit. Instead, it is better to implement any Merge filtering inside
88 // the MergeOperator.
89 kRemove,
90
91 // Change the value of the current key-value. If the current key-value is a
92 // plain key-value or a merge operand, its value is updated but its value
93 // type remains the same. If the current key-value is a wide-column entity,
94 // it is converted to a plain key-value with the new value specified.
95 kChangeValue,
96
97 // Remove all key-values with key in [key, *skip_until). This range of keys
98 // will be skipped in a way that potentially avoids some IO operations
99 // compared to removing the keys one by one. Note that removal in this case
100 // means dropping the key-value regardless of value type; in other words, in
101 // contrast with kRemove, plain values and entities are not converted to
102 // tombstones.
103 //
104 // *skip_until <= key is treated the same as Decision::kKeep (since the
105 // range [key, *skip_until) is empty).
106 //
107 // Caveats:
108 // * The keys are skipped even if there are snapshots containing them,
109 // i.e. values removed by kRemoveAndSkipUntil can disappear from a
110 // snapshot - beware if you're using TransactionDB or DB::GetSnapshot().
111 // * If value for a key was overwritten or merged into (multiple Put()s
112 // or Merge()s), and `CompactionFilter` skips this key with
113 // kRemoveAndSkipUntil, it's possible that it will remove only
114 // the new value, exposing the old value that was supposed to be
115 // overwritten.
116 // * Doesn't work with PlainTableFactory in prefix mode.
117 // * If you use kRemoveAndSkipUntil for table files created by compaction,
118 // consider also reducing compaction_readahead_size option.
119 kRemoveAndSkipUntil,
120
121 // Used internally by the old stacked BlobDB implementation. Returning this
122 // decision from application code is not supported.
123 kChangeBlobIndex,
124
125 // Used internally by the old stacked BlobDB implementation. Returning this
126 // decision from application code is not supported.
127 kIOError,
128
129 // Remove the current key-value by converting it to a SingleDelete-type
130 // tombstone. Only supported for plain-key values and wide-column entities;
131 // not supported for merge operands. All the caveats related to
132 // SingleDeletes apply.
133 kPurge,
134
135 // Change the current key-value to the wide-column entity specified. If the
136 // current key-value is already a wide-column entity, only its columns are
137 // updated; if it is a plain key-value, it is converted to a wide-column
138 // entity with the specified columns. Not supported for merge operands.
139 // Only applicable to FilterV3.
140 kChangeWideColumnEntity,
141
142 // When using the integrated BlobDB implementation, it may be possible for
143 // applications to make a filtering decision for a given blob based on
144 // the key only without actually reading the blob value, which saves some
145 // I/O; see the FilterBlobByKey method below. Returning kUndetermined from
146 // FilterBlobByKey signals that making a decision solely based on the
147 // key is not possible; in this case, RocksDB reads the blob value and
148 // passes the key-value to the regular filtering method. Only applicable to
149 // FilterBlobByKey; returning this value from FilterV2/V3 is not supported.
150 kUndetermined,
151 };
152
153 // Used internally by the old stacked BlobDB implementation.
154 enum class BlobDecision { kKeep, kChangeValue, kCorruption, kIOError };

186 // The table file creation process invokes this method before adding a kv to
187 // the table file. A return value of false indicates that the kv should be
188 // preserved in the new table file and a return value of true indicates
189 // that this key-value should be removed (that is, converted to a tombstone).
190 // The application can inspect the existing value of the key and make decision
191 // based on it.
192 //
193 // Key-Values that are results of merge operation during table file creation
194 // are not passed into this function. Currently, when you have a mix of Put()s
195 // and Merge()s on a same key, we only guarantee to process the merge operands
196 // through the `CompactionFilter`s. Put()s might be processed, or might not.
197 //
198 // When the value is to be preserved, the application has the option
199 // to modify the existing_value and pass it back through new_value.
200 // value_changed needs to be set to true in this case.
201 virtual bool Filter(int /*level*/, const Slice& /*key*/,
202 const Slice& /*existing_value*/,
203 std::string* /*new_value*/,
204 bool* /*value_changed*/) const {
205 return false;
206 }
207
208 // The table file creation process invokes this method on every merge operand.
209 // If this method returns true, the merge operand will be ignored and not
210 // written out in the new table file.
211 //
212 // Note: If you are using a TransactionDB, it is not recommended to implement
213 // FilterMergeOperand(). If a Merge operation is filtered out, TransactionDB
214 // may not realize there is a write conflict and may allow a Transaction to
215 // Commit that should have failed. Instead, it is better to implement any
216 // Merge filtering inside the MergeOperator.
217 virtual bool FilterMergeOperand(int /*level*/, const Slice& /*key*/,
218 const Slice& /*operand*/) const {
219 return false;
220 }
221
222 // A unified API for plain values and merge operands that may
223 // return a variety of decisions (see Decision above). The `value_type`
224 // parameter indicates the type of the key-value and the `existing_value`
225 // contains the current value or merge operand. The `new_value` output
226 // parameter can be used to set the updated value or merge operand when the
227 // kChangeValue decision is made by the filter. See the description of
228 // kRemoveAndSkipUntil above for the semantics of the `skip_until` output
229 // parameter, and see Decision above for more information on the semantics of
230 // the potential return values.
231 //
232 // The default implementation uses Filter() and FilterMergeOperand().
233 // If you're overriding this method, no need to override the other two.
234 virtual Decision FilterV2(int level, const Slice& key, ValueType value_type,
235 const Slice& existing_value, std::string* new_value,
236 std::string* /*skip_until*/) const {
237 switch (value_type) {
238 case ValueType::kValue: {
239 bool value_changed = false;
240 bool rv = Filter(level, key, existing_value, new_value, &value_changed);
241 if (rv) {
242 return Decision::kRemove;
243 }
244 return value_changed ? Decision::kChangeValue : Decision::kKeep;
245 }
246
247 case ValueType::kMergeOperand: {
248 bool rv = FilterMergeOperand(level, key, existing_value);
249 return rv ? Decision::kRemove : Decision::kKeep;
250 }
251
252 case ValueType::kBlobIndex:
253 return Decision::kKeep;
254
255 default:
256 assert(false);
257 return Decision::kKeep;
258 }
259 }
260
261 // Wide column aware unified API. Called for plain values, merge operands, and
262 // wide-column entities; the `value_type` parameter indicates the type of the
263 // key-value. When the key-value is a plain value or a merge operand, the
264 // `existing_value` parameter contains the existing value and the
265 // `existing_columns` parameter is invalid (nullptr). When the key-value is a
266 // wide-column entity, the `existing_columns` parameter contains the wide
267 // columns of the existing entity and the `existing_value` parameter is
268 // invalid (nullptr). The `new_value` output parameter can be used to set the
269 // updated value or merge operand when the kChangeValue decision is made by
270 // the filter. The `new_columns` output parameter can be used to specify
271 // the pairs of column names and column values when the
272 // kChangeWideColumnEntity decision is returned. See the description of
273 // kRemoveAndSkipUntil above for the semantics of the `skip_until` output
274 // parameter, and see Decision above for more information on the semantics of
275 // the potential return values.
276 //
277 // For compatibility, the default implementation keeps all wide-column
278 // entities, and falls back to FilterV2 for plain values and merge operands.
279 // If you override this method, there is no need to override FilterV2 (or
280 // Filter/FilterMergeOperand).
281 virtual Decision FilterV3(
282 int level, const Slice& key, ValueType value_type,
283 const Slice* existing_value, const WideColumns* existing_columns,
284 std::string* new_value,
285 std::vector<std::pair<std::string, std::string>>* /* new_columns */,
286 std::string* skip_until) const {
287#ifdef NDEBUG
288 (void)existing_columns;
289#endif
290
291 assert(!existing_value || !existing_columns);
292 assert(value_type == ValueType::kWideColumnEntity || existing_value);
293 assert(value_type != ValueType::kWideColumnEntity || existing_columns);
294
295 if (value_type == ValueType::kWideColumnEntity) {
296 return Decision::kKeep;
297 }
298
299 return FilterV2(level, key, value_type, *existing_value, new_value,
300 skip_until);
301 }

324 // In the case of BlobDB, it may be possible to reach a decision with only
325 // the key without reading the actual value, saving some I/O operations.
326 // Keys where the value is stored separately in a blob file will be
327 // passed to this method. If the method returns a supported decision other
328 // than kUndetermined, it will be considered final and performed without
329 // reading the existing value. Returning kUndetermined will cause FilterV3()
330 // to be called to make a decision as usual. The output parameters
331 // `new_value` and `skip_until` are applicable to the decisions kChangeValue
332 // and kRemoveAndSkipUntil respectively, and have the same semantics as
333 // the corresponding parameters of FilterV2/V3.
334 virtual Decision FilterBlobByKey(int /*level*/, const Slice& /*key*/,
335 std::string* /*new_value*/,
336 std::string* /*skip_until*/) const {
337 return Decision::kUndetermined;

341// Each thread of work involving creating table files will create a new
342// `CompactionFilter` according to `ShouldFilterTableFileCreation()`. This
343// allows the application to know about the different ongoing threads of work
344// and makes it unnecessary for `CompactionFilter` to provide thread-safety.
345//
346// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
347// because RocksDB is not exception-safe. This could cause undefined behavior
348// including data loss, unreported corruption, deadlocks, and more.
349class CompactionFilterFactory : public Customizable {
350 public:
351 virtual ~CompactionFilterFactory() {}
352 static const char* Type() { return "CompactionFilterFactory"; }
353 static Status CreateFromString(
354 const ConfigOptions& config_options, const std::string& name,
355 std::shared_ptr<CompactionFilterFactory>* result);
356
357 // Returns whether a thread creating table files for the specified `reason`
358 // should invoke `CreateCompactionFilter()` and pass KVs through the returned
359 // filter.
360 virtual bool ShouldFilterTableFileCreation(
361 TableFileCreationReason reason) const {
362 // For backward compatibility, default implementation only applies
363 // `CompactionFilter` to files generated by compaction.
364 return reason == TableFileCreationReason::kCompaction;
365 }
366
367 virtual std::unique_ptr<CompactionFilter> CreateCompactionFilter(
368 const CompactionFilter::Context& context) = 0;
369
370 // Returns a name that identifies this `CompactionFilter` factory.
371 const char* Name() const override = 0;
372};
