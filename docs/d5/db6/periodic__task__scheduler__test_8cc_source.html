38TEST_F(PeriodicTaskSchedulerTest, Basic) {
39 constexpr unsigned int kPeriodSec = 10;
40 Close();
41 Options options;
42 options.stats_dump_period_sec = kPeriodSec;
43 options.stats_persist_period_sec = kPeriodSec;
44 options.create_if_missing = true;
45 options.env = mock_env_.get();
46
47 int dump_st_counter = 0;
48 SyncPoint::GetInstance()->SetCallBack("DBImpl::DumpStats:StartRunning",
49 [&](void*) { dump_st_counter++; });
50
51 int pst_st_counter = 0;
52 SyncPoint::GetInstance()->SetCallBack("DBImpl::PersistStats:StartRunning",
53 [&](void*) { pst_st_counter++; });
54
55 int flush_info_log_counter = 0;
56 SyncPoint::GetInstance()->SetCallBack(
57 "DBImpl::FlushInfoLog:StartRunning",
58 [&](void*) { flush_info_log_counter++; });
59 SyncPoint::GetInstance()->EnableProcessing();
60
61 Reopen(options);
62
63 ASSERT_EQ(kPeriodSec, dbfull()->GetDBOptions().stats_dump_period_sec);
64 ASSERT_EQ(kPeriodSec, dbfull()->GetDBOptions().stats_persist_period_sec);
65
66 ASSERT_GT(kPeriodSec, 1u);
67 dbfull()->TEST_WaitForPeriodicTaskRun([&] {
68 mock_clock_->MockSleepForSeconds(static_cast<int>(kPeriodSec) - 1);
69 });
70
71 const PeriodicTaskScheduler& scheduler =
72 dbfull()->TEST_GetPeriodicTaskScheduler();
73 ASSERT_EQ(3, scheduler.TEST_GetValidTaskNum());
74
75 ASSERT_EQ(1, dump_st_counter);
76 ASSERT_EQ(1, pst_st_counter);
77 ASSERT_EQ(1, flush_info_log_counter);
78
79 dbfull()->TEST_WaitForPeriodicTaskRun(
80 [&] { mock_clock_->MockSleepForSeconds(static_cast<int>(kPeriodSec)); });
81
82 ASSERT_EQ(2, dump_st_counter);
83 ASSERT_EQ(2, pst_st_counter);
84 ASSERT_EQ(2, flush_info_log_counter);
85
86 dbfull()->TEST_WaitForPeriodicTaskRun(
87 [&] { mock_clock_->MockSleepForSeconds(static_cast<int>(kPeriodSec)); });
88
89 ASSERT_EQ(3, dump_st_counter);
90 ASSERT_EQ(3, pst_st_counter);
91 ASSERT_EQ(3, flush_info_log_counter);
92
93 // Disable scheduler with SetOption
94 ASSERT_OK(dbfull()->SetDBOptions(
95 {{"stats_dump_period_sec", "0"}, {"stats_persist_period_sec", "0"}}));
96 ASSERT_EQ(0u, dbfull()->GetDBOptions().stats_dump_period_sec);
97 ASSERT_EQ(0u, dbfull()->GetDBOptions().stats_persist_period_sec);
98
99 // Info log flush should still run.
100 dbfull()->TEST_WaitForPeriodicTaskRun(
101 [&] { mock_clock_->MockSleepForSeconds(static_cast<int>(kPeriodSec)); });
102 ASSERT_EQ(3, dump_st_counter);
103 ASSERT_EQ(3, pst_st_counter);
104 ASSERT_EQ(4, flush_info_log_counter);
105
106 ASSERT_EQ(1u, scheduler.TEST_GetValidTaskNum());
107
108 // Re-enable one task
109 ASSERT_OK(dbfull()->SetDBOptions({{"stats_dump_period_sec", "5"}}));
110 ASSERT_EQ(5u, dbfull()->GetDBOptions().stats_dump_period_sec);
111 ASSERT_EQ(0u, dbfull()->GetDBOptions().stats_persist_period_sec);
112
113 ASSERT_EQ(2, scheduler.TEST_GetValidTaskNum());
114
115 dbfull()->TEST_WaitForPeriodicTaskRun(
116 [&] { mock_clock_->MockSleepForSeconds(static_cast<int>(kPeriodSec)); });
117 ASSERT_EQ(4, dump_st_counter);
118 ASSERT_EQ(3, pst_st_counter);
119 ASSERT_EQ(5, flush_info_log_counter);
120
121 Close();
122}

124TEST_F(PeriodicTaskSchedulerTest, MultiInstances) {
125 constexpr int kPeriodSec = 5;
126 const int kInstanceNum = 10;
127
128 Close();
129 Options options;
130 options.stats_dump_period_sec = kPeriodSec;
131 options.stats_persist_period_sec = kPeriodSec;
132 options.create_if_missing = true;
133 options.env = mock_env_.get();
134
135 int dump_st_counter = 0;
136 SyncPoint::GetInstance()->SetCallBack("DBImpl::DumpStats:2",
137 [&](void*) { dump_st_counter++; });
138
139 int pst_st_counter = 0;
140 SyncPoint::GetInstance()->SetCallBack("DBImpl::PersistStats:StartRunning",
141 [&](void*) { pst_st_counter++; });
142 SyncPoint::GetInstance()->EnableProcessing();
143
144 auto dbs = std::vector<DB*>(kInstanceNum);
145 for (int i = 0; i < kInstanceNum; i++) {
146 ASSERT_OK(
147 DB::Open(options, test::PerThreadDBPath(std::to_string(i)), &(dbs[i])));
148 }
149
150 auto dbi = static_cast_with_check<DBImpl>(dbs[kInstanceNum - 1]);
151
152 const PeriodicTaskScheduler& scheduler = dbi->TEST_GetPeriodicTaskScheduler();
153 ASSERT_EQ(kInstanceNum * 3, scheduler.TEST_GetValidTaskNum());
154
155 int expected_run = kInstanceNum;
156 dbi->TEST_WaitForPeriodicTaskRun(
157 [&] { mock_clock_->MockSleepForSeconds(kPeriodSec - 1); });
158 ASSERT_EQ(expected_run, dump_st_counter);
159 ASSERT_EQ(expected_run, pst_st_counter);
160
161 expected_run += kInstanceNum;
162 dbi->TEST_WaitForPeriodicTaskRun(
163 [&] { mock_clock_->MockSleepForSeconds(kPeriodSec); });
164 ASSERT_EQ(expected_run, dump_st_counter);
165 ASSERT_EQ(expected_run, pst_st_counter);
166
167 expected_run += kInstanceNum;
168 dbi->TEST_WaitForPeriodicTaskRun(
169 [&] { mock_clock_->MockSleepForSeconds(kPeriodSec); });
170 ASSERT_EQ(expected_run, dump_st_counter);
171 ASSERT_EQ(expected_run, pst_st_counter);
172
173 int half = kInstanceNum / 2;
174 for (int i = 0; i < half; i++) {
175 delete dbs[i];
176 }
177
178 expected_run += (kInstanceNum - half) * 2;
179
180 dbi->TEST_WaitForPeriodicTaskRun(
181 [&] { mock_clock_->MockSleepForSeconds(kPeriodSec); });
182 dbi->TEST_WaitForPeriodicTaskRun(
183 [&] { mock_clock_->MockSleepForSeconds(kPeriodSec); });
184 ASSERT_EQ(expected_run, dump_st_counter);
185 ASSERT_EQ(expected_run, pst_st_counter);
186
187 for (int i = half; i < kInstanceNum; i++) {
188 ASSERT_OK(dbs[i]->Close());
189 delete dbs[i];
190 }
191}
