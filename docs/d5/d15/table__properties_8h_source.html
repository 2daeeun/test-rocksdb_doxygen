81// `TablePropertiesCollector` provides the mechanism for users to collect
82// their own properties that they are interested in. This class is essentially
83// a collection of callback functions that will be invoked during table
84// building. It is constructed with TablePropertiesCollectorFactory. The methods
85// don't need to be thread-safe, as we will create exactly one
86// TablePropertiesCollector object per table and then call it sequentially.
87//
88// Statuses from these callbacks are currently logged when not OK, but
89// otherwise ignored by RocksDB.
90//
91// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
92// because RocksDB is not exception-safe. This could cause undefined behavior
93// including data loss, unreported corruption, deadlocks, and more.
94class TablePropertiesCollector {
95 public:
96 virtual ~TablePropertiesCollector() {}
97
98 // DEPRECATE User defined collector should implement AddUserKey(), though
99 // this old function still works for backward compatible reason.
100 // Add() will be called when a new key/value pair is inserted into the table.
101 // @params key the user key that is inserted into the table.
102 // @params value the value that is inserted into the table.
103 virtual Status Add(const Slice& /*key*/, const Slice& /*value*/) {
104 return Status::InvalidArgument(
105 "TablePropertiesCollector::Add() deprecated.");
106 }
107
108 // AddUserKey() will be called when a new key/value pair is inserted into the
109 // table.
110 // @params key the user key that is inserted into the table.
111 // @params value the value that is inserted into the table.
112 virtual Status AddUserKey(const Slice& key, const Slice& value,
113 EntryType /*type*/, SequenceNumber /*seq*/,
114 uint64_t /*file_size*/) {
115 // For backwards-compatibility.
116 return Add(key, value);
117 }
118
119 // Called after each new block is cut
120 virtual void BlockAdd(uint64_t /* block_uncomp_bytes */,
121 uint64_t /* block_compressed_bytes_fast */,
122 uint64_t /* block_compressed_bytes_slow */) {
123 // Nothing to do here. Callback registers can override.
124 return;
125 }
126
127 // Finish() will be called when a table has already been built and is ready
128 // for writing the properties block.
129 // It will be called only once by RocksDB internal.
130 // When the returned Status is not OK, the collected properties will not be
131 // written to the file's property block.
132 //
133 // @params properties User will add their collected statistics to
134 // `properties`.
135 virtual Status Finish(UserCollectedProperties* properties) = 0;
136
137 // Return the human-readable properties, where the key is property name and
138 // the value is the human-readable form of value.
139 // Returned properties are used for logging.
140 // It will only be called after Finish() has been called by RocksDB internal.
141 virtual UserCollectedProperties GetReadableProperties() const = 0;
142
143 // The name of the properties collector can be used for debugging purpose.
144 virtual const char* Name() const = 0;
145
146 // EXPERIMENTAL Return whether the output file should be further compacted
147 virtual bool NeedCompact() const { return false; }
148
149 // For internal use only.
150 virtual InternalTblPropColl* AsInternal() { return nullptr; }
151};

153// Constructs TablePropertiesCollector instances for each table file creation.
154//
155// Exceptions MUST NOT propagate out of overridden functions into RocksDB,
156// because RocksDB is not exception-safe. This could cause undefined behavior
157// including data loss, unreported corruption, deadlocks, and more.
158class TablePropertiesCollectorFactory : public Customizable {
159 public:
160 struct Context {
161 uint32_t column_family_id;
162 // The level at creating the SST file (i.e, table), of which the
163 // properties are being collected.
164 int level_at_creation = kUnknownLevelAtCreation;
165 int num_levels = kUnknownNumLevels;
166 // In the tiering case, data with seqnos smaller than or equal to this
167 // cutoff sequence number will be considered by a compaction job as eligible
168 // to be placed on the last level. When this is the maximum sequence number,
169 // it indicates tiering is disabled.
170 SequenceNumber last_level_inclusive_max_seqno_threshold;
171 static const uint32_t kUnknownColumnFamily;
172 static const int kUnknownLevelAtCreation = -1;
173 static const int kUnknownNumLevels = -1;
174
175 Context() {}
176
177 Context(uint32_t _column_family_id, int _level_at_creation, int _num_levels,
178 SequenceNumber _last_level_inclusive_max_seqno_threshold)
179 : column_family_id(_column_family_id),
180 level_at_creation(_level_at_creation),
181 num_levels(_num_levels),
182 last_level_inclusive_max_seqno_threshold(
183 _last_level_inclusive_max_seqno_threshold) {}
184 };
185
186 ~TablePropertiesCollectorFactory() override {}
187 static const char* Type() { return "TablePropertiesCollectorFactory"; }
188 static Status CreateFromString(
189 const ConfigOptions& options, const std::string& value,
190 std::shared_ptr<TablePropertiesCollectorFactory>* result);
191
192 // To collect properties of a table with the given context, returns
193 // a new object inheriting from TablePropertiesCollector. The caller
194 // is responsible for deleting the object returned. Alternatively,
195 // nullptr may be returned to decline collecting properties for the
196 // file (and reduce callback overheads).
197 // MUST be thread-safe.
198 virtual TablePropertiesCollector* CreateTablePropertiesCollector(
199 TablePropertiesCollectorFactory::Context context) = 0;
200
201 // The name of the properties collector can be used for debugging purpose.
202 const char* Name() const override = 0;
203
204 // Can be overridden by sub-classes to return the Name, followed by
205 // configuration info that will // be logged to the info log when the
206 // DB is opened
207 virtual std::string ToString() const { return Name(); }
208};

212struct TableProperties {
213 public:
214 // the file number at creation time, or 0 for unknown. When known,

225 // Whether the index key is user key. Otherwise it includes 8 byte of sequence
226 // number added by internal key format.
227 uint64_t index_key_is_user_key = 0;
228 // Whether delta encoding is used to encode the index values.
229 uint64_t index_value_is_delta_encoded = 0;
230 // the size of filter block.
231 uint64_t filter_size = 0;
232 // total raw (uncompressed, undelineated) key size
233 uint64_t raw_key_size = 0;
234 // total raw (uncompressed, undelineated) value size
235 uint64_t raw_value_size = 0;
236 // the number of blocks in this table
237 uint64_t num_data_blocks = 0;
238 // the number of entries in this table
239 uint64_t num_entries = 0;
240 // the number of unique entries (keys or prefixes) added to filters
241 uint64_t num_filter_entries = 0;
242 // the number of deletions in the table
243 uint64_t num_deletions = 0;
244 // the number of merge operands in the table
245 uint64_t num_merge_operands = 0;
246 // the number of range deletions in this table
247 uint64_t num_range_deletions = 0;
248 // format version, reserved for backward compatibility
249 uint64_t format_version = 0;
250 // If 0, key is variable length. Otherwise number of bytes for each key.
251 uint64_t fixed_key_len = 0;
252 // ID of column family for this SST file, corresponding to the CF identified
253 // by column_family_name.
254 uint64_t column_family_id = ROCKSDB_NAMESPACE::
255 TablePropertiesCollectorFactory::Context::kUnknownColumnFamily;
256
257 // Oldest ancester time. 0 means unknown.
258 //
259 // For flush output file, oldest ancestor time is the oldest key time in the
260 // file. If the oldest key time is not available, flush time is used.
261 //
262 // For compaction output file, oldest ancestor time is the oldest
263 // among all the oldest key time of its input files, since the file could be
264 // the compaction output from other SST files, which could in turn be outputs
265 // for compact older SST files. If that's not available, creation time of this
266 // compaction output file is used.
267 //
268 // TODO(sagar0): Should be changed to oldest_ancester_time ... but don't know
269 // the full implications of backward compatibility. Hence retaining for now.
270 uint64_t creation_time = 0;
271
272 // Timestamp of the earliest key. 0 means unknown.
273 uint64_t oldest_key_time = 0;
274 // Timestamp of the newest key. 0 means unknown.
275 uint64_t newest_key_time = 0;
276 // Actual SST file creation time. 0 means unknown.
277 uint64_t file_creation_time = 0;
278 // Estimated size of data blocks if compressed using a relatively slower
279 // compression algorithm (see `ColumnFamilyOptions::sample_for_compression`).
280 // 0 means unknown.
281 uint64_t slow_compression_estimated_data_size = 0;
282 // Estimated size of data blocks if compressed using a relatively faster
283 // compression algorithm (see `ColumnFamilyOptions::sample_for_compression`).
284 // 0 means unknown.
285 uint64_t fast_compression_estimated_data_size = 0;
286 // Offset of the value of the property "external sst file global seqno" in the
287 // file if the property exists.
288 // 0 means not exists.
289 uint64_t external_sst_file_global_seqno_offset = 0;
290
291 // Offset where the "tail" part of SST file starts
292 // "Tail" refers to all blocks after data blocks till the end of the SST file
293 uint64_t tail_start_offset = 0;
294
295 // Value of the `AdvancedColumnFamilyOptions.persist_user_defined_timestamps`
296 // when the file is created. Default to be true, only when this flag is false,
297 // it's explicitly written to meta properties block.
298 uint64_t user_defined_timestamps_persisted = 1;
299
300 // The largest sequence number of keys in this file.
301 // UINT64_MAX means unknown.
302 // Only written to properties block if known (should be known unless the
303 // table is empty).
304 uint64_t key_largest_seqno = UINT64_MAX;
305
306 // DB identity
307 // db_id is an identifier generated the first time the DB is created
308 // If DB identity is unset or unassigned, `db_id` will be an empty string.
309 std::string db_id;
310
311 // DB session identity
312 // db_session_id is an identifier that gets reset every time the DB is opened
313 // If DB session identity is unset or unassigned, `db_session_id` will be an
314 // empty string.
315 std::string db_session_id;
316
317 // Location of the machine hosting the DB instance
318 // db_host_id identifies the location of the host in some form
319 // (hostname by default, but can also be any string of the user's choosing).
320 // It can potentially change whenever the DB is opened
321 std::string db_host_id;
322
323 // Name of the column family with which this SST file is associated.
324 // If column family is unknown, `column_family_name` will be an empty string.
325 std::string column_family_name;
326
327 // The name of the filter policy used in this table.
328 // If no filter policy is used, `filter_policy_name` will be an empty string.
329 std::string filter_policy_name;
330
331 // The name of the comparator used in this table.
332 std::string comparator_name;
333
334 // The name of the merge operator used in this table.
335 // If no merge operator is used, `merge_operator_name` will be "nullptr".
336 std::string merge_operator_name;
337
338 // The name of the prefix extractor used in this table
339 // If no prefix extractor is used, `prefix_extractor_name` will be "nullptr".
340 std::string prefix_extractor_name;
341
342 // The names of the property collectors factories used in this table
343 // separated by commas
344 // {collector_name[1]},{collector_name[2]},{collector_name[3]} ..
345 std::string property_collectors_names;
346
347 // The compression algo used to compress the SST files.
348 std::string compression_name;
349
350 // Compression options used to compress the SST files.
351 std::string compression_options;
352
353 // Sequence number to time mapping, delta encoded.
354 std::string seqno_to_time_mapping;
355
356 // user collected properties
357 UserCollectedProperties user_collected_properties;
358 UserCollectedProperties readable_properties;
359
360 // convert this object to a human readable form
361 // @prop_delim: delimiter for each property.
362 std::string ToString(const std::string& prop_delim = "; ",
363 const std::string& kv_delim = "=") const;
364
365 // Aggregate the numerical member variables of the specified
366 // TableProperties.
367 void Add(const TableProperties& tp);
368
369 // Subset of properties that make sense when added together
370 // between tables. Keys match field names in this class instead
371 // of using full property names.
372 std::map<std::string, uint64_t> GetAggregatablePropertiesAsMap() const;
373
374 // Return the approximated memory usage of this TableProperties object,
375 // including memory used by the string properties and UserCollectedProperties
376 std::size_t ApproximateMemoryUsage() const;
377
378 // Serialize and deserialize Table Properties
379 Status Serialize(const ConfigOptions& opts, std::string* output) const;
380 static Status Parse(const ConfigOptions& opts, const std::string& serialized,
381 TableProperties* table_properties);
382 bool AreEqual(const ConfigOptions& opts,
383 const TableProperties* other_table_properties,
384 std::string* mismatch) const;
385};
