127class FileChecksumTestHelper {
128 private:
129 Options options_;
130 DB* db_;
131 std::string dbname_;
132
133 Status VerifyChecksum(LiveFileMetaData& file_meta) {
134 std::string cur_checksum;
135 std::string checksum_func_name;
136
137 Status s;
138 EnvOptions soptions;
139 std::unique_ptr<SequentialFile> file_reader;
140 std::string file_path = dbname_ + "/" + file_meta.name;
141 s = options_.env->NewSequentialFile(file_path, &file_reader, soptions);
142 if (!s.ok()) {
143 return s;
144 }
145 std::unique_ptr<char[]> scratch(new char[2048]);
146 Slice result;
147 FileChecksumGenFactory* file_checksum_gen_factory =
148 options_.file_checksum_gen_factory.get();
149 if (file_checksum_gen_factory == nullptr) {
150 cur_checksum = kUnknownFileChecksum;
151 checksum_func_name = kUnknownFileChecksumFuncName;
152 } else {
153 FileChecksumGenContext gen_context;
154 gen_context.file_name = file_meta.name;
155 std::unique_ptr<FileChecksumGenerator> file_checksum_gen =
156 file_checksum_gen_factory->CreateFileChecksumGenerator(gen_context);
157 checksum_func_name = file_checksum_gen->Name();
158 s = file_reader->Read(2048, &result, scratch.get());
159 if (!s.ok()) {
160 return s;
161 }
162 while (result.size() != 0) {
163 file_checksum_gen->Update(scratch.get(), result.size());
164 s = file_reader->Read(2048, &result, scratch.get());
165 if (!s.ok()) {
166 return s;
167 }
168 }
169 file_checksum_gen->Finalize();
170 cur_checksum = file_checksum_gen->GetChecksum();
171 }
172
173 std::string stored_checksum = file_meta.file_checksum;
174 std::string stored_checksum_func_name = file_meta.file_checksum_func_name;
175 if ((cur_checksum != stored_checksum) ||
176 (checksum_func_name != stored_checksum_func_name)) {
177 return Status::Corruption(
178 "Checksum does not match! The file: " + file_meta.name +
179 ", checksum name: " + stored_checksum_func_name + " and checksum " +
180 stored_checksum + ". However, expected checksum name: " +
181 checksum_func_name + " and checksum " + cur_checksum);
182 }
183 return Status::OK();
184 }
185
186 public:
187 FileChecksumTestHelper(Options& options, DB* db, std::string db_name)
188 : options_(options), db_(db), dbname_(db_name) {}
189 ~FileChecksumTestHelper() = default;
190
191 // Verify the checksum information in Manifest.
192 Status VerifyChecksumInManifest(
193 const std::vector<LiveFileMetaData>& live_files) {
194 // Step 1: verify if the dbname_ is correct
195 if (dbname_.back() != '/') {
196 dbname_.append("/");
197 }
198
199 // Step 2, get the the checksum information by recovering the VersionSet
200 // from Manifest.
201 std::unique_ptr<FileChecksumList> checksum_list(NewFileChecksumList());
202 EnvOptions sopt;
203 std::shared_ptr<Cache> tc(NewLRUCache(options_.max_open_files - 10,
204 options_.table_cache_numshardbits));
205 options_.db_paths.emplace_back(dbname_, 0);
206 options_.num_levels = 64;
207 WriteController wc(options_.delayed_write_rate);
208 WriteBufferManager wb(options_.db_write_buffer_size);
209 ImmutableDBOptions immutable_db_options(options_);
210 VersionSet versions(dbname_, &immutable_db_options, sopt, tc.get(), &wb,
211 &wc, nullptr, nullptr, "", "",
212 options_.daily_offpeak_time_utc, nullptr,
213 /*read_only=*/false);
214 std::vector<std::string> cf_name_list;
215 Status s;
216 s = versions.ListColumnFamilies(&cf_name_list, dbname_,
217 immutable_db_options.fs.get());
218 if (s.ok()) {
219 std::vector<ColumnFamilyDescriptor> cf_list;
220 for (const auto& name : cf_name_list) {
221 fprintf(stdout, "cf_name: %s", name.c_str());
222 cf_list.emplace_back(name, ColumnFamilyOptions(options_));
223 }
224 s = versions.Recover(cf_list, true);
225 }
226 if (s.ok()) {
227 s = versions.GetLiveFilesChecksumInfo(checksum_list.get());
228 }
229 if (!s.ok()) {
230 return s;
231 }
232
233 // Step 3 verify the checksum
234 if (live_files.size() != checksum_list->size()) {
235 return Status::Corruption("The number of files does not match!");
236 }
237 for (size_t i = 0; i < live_files.size(); i++) {
238 std::string stored_checksum;
239 std::string stored_func_name;
240 s = checksum_list->SearchOneFileChecksum(
241 live_files[i].file_number, &stored_checksum, &stored_func_name);
242 if (s.IsNotFound()) {
243 return s;
244 }
245 if (live_files[i].file_checksum != stored_checksum ||
246 live_files[i].file_checksum_func_name != stored_func_name) {
247 return Status::Corruption(
248 "Checksum does not match! The file: " +
249 std::to_string(live_files[i].file_number) +
250 ". In Manifest, checksum name: " + stored_func_name +
251 " and checksum " + stored_checksum +
252 ". However, expected checksum name: " +
253 live_files[i].file_checksum_func_name + " and checksum " +
254 live_files[i].file_checksum);
255 }
256 }
257 return Status::OK();
258 }
259
260 // Verify the checksum of each file by recalculting the checksum and
261 // comparing it with the one being generated when a SST file is created.
262 Status VerifyEachFileChecksum() {
263 assert(db_ != nullptr);
264 EXPECT_OK(db_->DisableFileDeletions());
265 std::vector<LiveFileMetaData> live_files;
266 db_->GetLiveFilesMetaData(&live_files);
267 Status cs;
268 for (auto a_file : live_files) {
269 cs = VerifyChecksum(a_file);
270 if (!cs.ok()) {
271 break;
272 }
273 }
274 EXPECT_OK(db_->EnableFileDeletions());
275 return cs;
276 }
277};
