16class TestLogStatsParser(unittest.TestCase):

18 this_path = os.path.abspath(os.path.dirname(__file__))
19 stats_file = os.path.join(this_path, "input_files/log_stats_parser_keys_ts")

22 with open(stats_file) as fp:

30 self.stats_dict[NO_ENTITY][stat_name][timestamp] = value
31 self.log_stats_parser = LogStatsParser("dummy_log_file", 20)
32 self.log_stats_parser.keys_ts = self.stats_dict
