37// An abstract class to pick compactions from an existing LSM-tree.
38//
39// Each compaction style inherits the class and implement the
40// interface to form automatic compactions. If NeedCompaction() is true,
41// then call PickCompaction() to find what files need to be compacted
42// and where to put the output files.
43//
44// Non-virtual functions CompactRange() and CompactFiles() are used to
45// pick files to compact based on users' DB::CompactRange() and
46// DB::CompactFiles() requests, respectively. There is little
47// compaction style specific logic for them.
48class CompactionPicker {
49 public:
50 CompactionPicker(const ImmutableOptions& ioptions,
51 const InternalKeyComparator* icmp);
52 virtual ~CompactionPicker();
53
54 // Pick level and inputs for a new compaction.
55 //
56 // Returns nullptr if there is no compaction to be done.
57 // Otherwise returns a pointer to a heap-allocated object that
58 // describes the compaction. Caller should delete the result.
59 // Currently, only universal compaction will query existing snapshots and
60 // pass it to aid compaction picking. And it's only passed when user-defined
61 // timestamps is not enabled. The other compaction styles do not pass or use
62 // `existing_snapshots` or `snapshot_checker`.
63 virtual Compaction* PickCompaction(
64 const std::string& cf_name, const MutableCFOptions& mutable_cf_options,
65 const MutableDBOptions& mutable_db_options,
66 const std::vector<SequenceNumber>& existing_snapshots,
67 const SnapshotChecker* snapshot_checker, VersionStorageInfo* vstorage,
68 LogBuffer* log_buffer) = 0;
69
70 // The returned Compaction might not include the whole requested range.
71 // In that case, compaction_end will be set to the next key that needs
72 // compacting. In case the compaction will compact the whole range,
73 // compaction_end will be set to nullptr.
74 // Client is responsible for compaction_end storage -- when called,
75 // *compaction_end should point to valid InternalKey!
76 virtual Compaction* CompactRange(
77 const std::string& cf_name, const MutableCFOptions& mutable_cf_options,
78 const MutableDBOptions& mutable_db_options, VersionStorageInfo* vstorage,
79 int input_level, int output_level,
80 const CompactRangeOptions& compact_range_options,
81 const InternalKey* begin, const InternalKey* end,
82 InternalKey** compaction_end, bool* manual_conflict,
83 uint64_t max_file_num_to_ignore, const std::string& trim_ts);
84
85 // The maximum allowed output level. Default value is NumberLevels() - 1.
86 virtual int MaxOutputLevel() const { return NumberLevels() - 1; }
87
88 virtual bool NeedsCompaction(const VersionStorageInfo* vstorage) const = 0;
89
90 // Sanitize the input set of compaction input files and convert it to
91 // `std::vector<CompactionInputFiles>` in the output parameter
92 // `converted_input_files`.
93 // When the input parameters do not describe a valid
94 // compaction, the function will try to fix the input_files by adding
95 // necessary files. If it's not possible to convert an invalid input_files
96 // into a valid one by adding more files, the function will return a
97 // non-ok status with specific reason.
98 //
99 Status SanitizeAndConvertCompactionInputFiles(
100 std::unordered_set<uint64_t>* input_files, const int output_level,
101 Version* version,
102 std::vector<CompactionInputFiles>* converted_input_files) const;
103
104 // Free up the files that participated in a compaction
105 //
106 // Requirement: DB mutex held
107 void ReleaseCompactionFiles(Compaction* c, const Status& status);
108
109 // Returns true if any one of the specified files are being compacted
110 bool AreFilesInCompaction(const std::vector<FileMetaData*>& files);

112 // Takes a list of CompactionInputFiles and returns a (manual) Compaction
113 // object.
114 //
115 // Caller must provide a set of input files that has been passed through
116 // `SanitizeAndConvertCompactionInputFiles` earlier. The lock should not be
117 // released between that call and this one.

161 void GetRange(const std::vector<CompactionInputFiles>& inputs,
162 InternalKey* smallest, InternalKey* largest,
163 int exclude_level) const;
164
165 int NumberLevels() const { return ioptions_.num_levels; }
166
167 // Add more files to the inputs on "level" to make sure that
168 // no newer version of a key is compacted to "level+1" while leaving an older
169 // version in a "level". Otherwise, any Get() will search "level" first,
170 // and will likely return an old/stale value for the key, since it always
171 // searches in increasing order of level to find the value. This could
172 // also scramble the order of merge operands. This function should be
173 // called any time a new Compaction is created, and its inputs_[0] are
174 // populated.
175 //
176 // Will return false if it is impossible to apply this compaction.
177 bool ExpandInputsToCleanCut(const std::string& cf_name,
178 VersionStorageInfo* vstorage,
179 CompactionInputFiles* inputs,
180 InternalKey** next_smallest = nullptr);
181
182 // Returns true if any one of the parent files are being compacted
183 bool IsRangeInCompaction(VersionStorageInfo* vstorage,
184 const InternalKey* smallest,
185 const InternalKey* largest, int level, int* index);
186
187 // Returns true if the key range that `inputs` files cover overlap with the
188 // key range of a currently running compaction.
189 bool FilesRangeOverlapWithCompaction(
190 const std::vector<CompactionInputFiles>& inputs, int level,
191 int penultimate_level) const;
192
193 bool SetupOtherInputs(const std::string& cf_name,
194 const MutableCFOptions& mutable_cf_options,
195 VersionStorageInfo* vstorage,
196 CompactionInputFiles* inputs,
197 CompactionInputFiles* output_level_inputs,
198 int* parent_index, int base_index,

200
201 void GetGrandparents(VersionStorageInfo* vstorage,

206 void PickFilesMarkedForCompaction(

211
212 bool GetOverlappingL0Files(VersionStorageInfo* vstorage,
213 CompactionInputFiles* start_level_inputs,
214 int output_level, int* parent_index);
215
216 // Register this compaction in the set of running compactions
217 void RegisterCompaction(Compaction* c);
218
219 // Remove this compaction from the set of running compactions
220 void UnregisterCompaction(Compaction* c);
221
222 std::set<Compaction*>* level0_compactions_in_progress() {
223 return &level0_compactions_in_progress_;
224 }
225 std::unordered_set<Compaction*>* compactions_in_progress() {
226 return &compactions_in_progress_;
227 }
228
229 const InternalKeyComparator* icmp() const { return icmp_; }

231 protected:
232 const ImmutableOptions& ioptions_;
233
234 // A helper function to SanitizeAndConvertCompactionInputFiles() that
235 // sanitizes "input_files" by adding necessary files.
236 virtual Status SanitizeCompactionInputFilesForAllLevels(
237 std::unordered_set<uint64_t>* input_files,
238 const ColumnFamilyMetaData& cf_meta, const int output_level) const;
239
240 // Keeps track of all compactions that are running on Level0.
241 // Protected by DB mutex
242 std::set<Compaction*> level0_compactions_in_progress_;
243
244 // Keeps track of all compactions that are running.
245 // Protected by DB mutex
246 std::unordered_set<Compaction*> compactions_in_progress_;
247
248 const InternalKeyComparator* const icmp_;
249};
