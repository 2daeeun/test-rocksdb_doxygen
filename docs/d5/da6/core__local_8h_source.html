21// An array of core-local values. Ideally the value type, T, is cache aligned to
22// prevent false sharing.
23template <typename T>
24class CoreLocalArray {
25 public:
26 CoreLocalArray();
27
28 size_t Size() const;
29 // returns pointer to the element corresponding to the core that the thread
30 // currently runs on.
31 T* Access() const;
32 // same as above, but also returns the core index, which the client can cache
33 // to reduce how often core ID needs to be retrieved. Only do this if some
34 // inaccuracy is tolerable, as the thread may migrate to a different core.
35 std::pair<T*, size_t> AccessElementAndIndex() const;
36 // returns pointer to element for the specified core index. This can be used,
37 // e.g., for aggregation, or if the client caches core index.
38 T* AccessAtCore(size_t core_idx) const;
39
40 private:
41 std::unique_ptr<T[]> data_;
42 int size_shift_;
43};
44
45template <typename T>
46CoreLocalArray<T>::CoreLocalArray() {
47 int num_cpus = static_cast<int>(std::thread::hardware_concurrency());
48 // find a power of two >= num_cpus and >= 8
49 size_shift_ = 3;
50 while (1 << size_shift_ < num_cpus) {
51 ++size_shift_;
52 }
53 data_.reset(new T[static_cast<size_t>(1) << size_shift_]);
54}
55
56template <typename T>
57size_t CoreLocalArray<T>::Size() const {
58 return static_cast<size_t>(1) << size_shift_;
59}
60
61template <typename T>
62T* CoreLocalArray<T>::Access() const {
63 return AccessElementAndIndex().first;
64}
65
66template <typename T>
67std::pair<T*, size_t> CoreLocalArray<T>::AccessElementAndIndex() const {
68 int cpuid = port::PhysicalCoreID();
69 size_t core_idx;
70 if (UNLIKELY(cpuid < 0)) {
71 // cpu id unavailable, just pick randomly
72 core_idx = Random::GetTLSInstance()->Uniform(1 << size_shift_);
73 } else {
74 core_idx = static_cast<size_t>(BottomNBits(cpuid, size_shift_));
75 }
76 return {AccessAtCore(core_idx), core_idx};
77}
78
79template <typename T>
80T* CoreLocalArray<T>::AccessAtCore(size_t core_idx) const {
81 assert(core_idx < static_cast<size_t>(1) << size_shift_);
82 return &data_[core_idx];
83}
