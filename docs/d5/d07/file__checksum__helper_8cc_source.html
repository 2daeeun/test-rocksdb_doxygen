20Status FileChecksumListImpl::GetAllFileChecksums(
21 std::vector<uint64_t>* file_numbers, std::vector<std::string>* checksums,
22 std::vector<std::string>* checksum_func_names) {
23 if (file_numbers == nullptr || checksums == nullptr ||
24 checksum_func_names == nullptr) {
25 return Status::InvalidArgument("Pointer has not been initiated");
26 }
27
28 for (const auto& i : checksum_map_) {
29 file_numbers->push_back(i.first);
30 checksums->push_back(i.second.first);
31 checksum_func_names->push_back(i.second.second);
32 }
33 return Status::OK();
34}

36Status FileChecksumListImpl::SearchOneFileChecksum(
37 uint64_t file_number, std::string* checksum,
38 std::string* checksum_func_name) {
39 if (checksum == nullptr || checksum_func_name == nullptr) {
40 return Status::InvalidArgument("Pointer has not been initiated");
41 }
42
43 auto it = checksum_map_.find(file_number);
44 if (it == checksum_map_.end()) {
45 return Status::NotFound();
46 } else {
47 *checksum = it->second.first;
48 *checksum_func_name = it->second.second;
49 }
50 return Status::OK();
51}

53Status FileChecksumListImpl::InsertOneFileChecksum(
54 uint64_t file_number, const std::string& checksum,
55 const std::string& checksum_func_name) {
56 auto it = checksum_map_.find(file_number);
57 if (it == checksum_map_.end()) {
58 checksum_map_.insert(std::make_pair(
59 file_number, std::make_pair(checksum, checksum_func_name)));
60 } else {
61 it->second.first = checksum;
62 it->second.second = checksum_func_name;
63 }
64 return Status::OK();
65}
66
67Status FileChecksumListImpl::RemoveOneFileChecksum(uint64_t file_number) {
68 auto it = checksum_map_.find(file_number);
69 if (it == checksum_map_.end()) {
70 return Status::NotFound();
71 } else {
72 checksum_map_.erase(it);
73 }
74 return Status::OK();
75}

103Status FileChecksumGenFactory::CreateFromString(
104 const ConfigOptions& options, const std::string& value,
105 std::shared_ptr<FileChecksumGenFactory>* result) {
106 static std::once_flag once;
107 std::call_once(once, [&]() {
