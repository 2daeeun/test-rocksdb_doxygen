99Status BlobFileBuilder::Add(const Slice& key, const Slice& value,
100 std::string* blob_index) {
101 assert(blob_index);
102 assert(blob_index->empty());
103
104 if (value.size() < min_blob_size_) {
105 return Status::OK();
106 }
107
108 {
109 const Status s = OpenBlobFileIfNeeded();
110 if (!s.ok()) {
111 return s;
112 }
113 }
114
115 Slice blob = value;
116 std::string compressed_blob;
117
118 {
119 const Status s = CompressBlobIfNeeded(&blob, &compressed_blob);
120 if (!s.ok()) {
121 return s;
122 }
123 }
124
125 uint64_t blob_file_number = 0;
126 uint64_t blob_offset = 0;
127
128 {
129 const Status s =
130 WriteBlobToFile(key, blob, &blob_file_number, &blob_offset);
131 if (!s.ok()) {
132 return s;
133 }
134 }
135
136 {
137 const Status s = CloseBlobFileIfNeeded();
138 if (!s.ok()) {
139 return s;
140 }
141 }
142
143 {
144 const Status s =
145 PutBlobIntoCacheIfNeeded(value, blob_file_number, blob_offset);
146 if (!s.ok()) {
147 ROCKS_LOG_WARN(immutable_options_->info_log,
148 "Failed to pre-populate the blob into blob cache: %s",
149 s.ToString().c_str());
150 }
151 }
152
153 BlobIndex::EncodeBlob(blob_index, blob_file_number, blob_offset, blob.size(),
154 blob_compression_type_);
155
156 return Status::OK();
157}

159Status BlobFileBuilder::Finish() {
160 if (!IsBlobFileOpen()) {
161 return Status::OK();
162 }
163
164 return CloseBlobFile();
165}

169Status BlobFileBuilder::OpenBlobFileIfNeeded() {
170 if (IsBlobFileOpen()) {
171 return Status::OK();
172 }
173
174 assert(!blob_count_);
175 assert(!blob_bytes_);
176
177 assert(file_number_generator_);
178 const uint64_t blob_file_number = file_number_generator_();
179
180 assert(immutable_options_);
181 assert(!immutable_options_->cf_paths.empty());
182 std::string blob_file_path =
183 BlobFileName(immutable_options_->cf_paths.front().path, blob_file_number);
184
185 if (blob_callback_) {
186 blob_callback_->OnBlobFileCreationStarted(
187 blob_file_path, column_family_name_, job_id_, creation_reason_);
188 }
189
190 std::unique_ptr<FSWritableFile> file;
191
192 {
193 assert(file_options_);
194 Status s = NewWritableFile(fs_, blob_file_path, &file, *file_options_);
195
196 TEST_SYNC_POINT_CALLBACK(
197 "BlobFileBuilder::OpenBlobFileIfNeeded:NewWritableFile", &s);
198
199 if (!s.ok()) {
200 return s;
201 }
202 }
203
204 // Note: files get added to blob_file_paths_ right after the open, so they
205 // can be cleaned up upon failure. Contrast this with blob_file_additions_,
206 // which only contains successfully written files.
207 assert(blob_file_paths_);
208 blob_file_paths_->emplace_back(std::move(blob_file_path));
209
210 assert(file);
211 file->SetIOPriority(write_options_->rate_limiter_priority);
212 file->SetWriteLifeTimeHint(write_hint_);
213 FileTypeSet tmp_set = immutable_options_->checksum_handoff_file_types;
214 Statistics* const statistics = immutable_options_->stats;
215 std::unique_ptr<WritableFileWriter> file_writer(new WritableFileWriter(
216 std::move(file), blob_file_paths_->back(), *file_options_,
217 immutable_options_->clock, io_tracer_, statistics,
218 Histograms::BLOB_DB_BLOB_FILE_WRITE_MICROS, immutable_options_->listeners,
219 immutable_options_->file_checksum_gen_factory.get(),
220 tmp_set.Contains(FileType::kBlobFile), false));
221
222 constexpr bool do_flush = false;
223
224 std::unique_ptr<BlobLogWriter> blob_log_writer(new BlobLogWriter(
225 std::move(file_writer), immutable_options_->clock, statistics,
226 blob_file_number, immutable_options_->use_fsync, do_flush));
227
228 constexpr bool has_ttl = false;
229 constexpr ExpirationRange expiration_range;
230
231 BlobLogHeader header(column_family_id_, blob_compression_type_, has_ttl,
232 expiration_range);
233
234 {
235 Status s = blob_log_writer->WriteHeader(*write_options_, header);
236
237 TEST_SYNC_POINT_CALLBACK(
238 "BlobFileBuilder::OpenBlobFileIfNeeded:WriteHeader", &s);
239
240 if (!s.ok()) {
241 return s;
242 }
243 }
244
245 writer_ = std::move(blob_log_writer);
246
247 assert(IsBlobFileOpen());
248
249 return Status::OK();
250}

252Status BlobFileBuilder::CompressBlobIfNeeded(
253 Slice* blob, std::string* compressed_blob) const {
254 assert(blob);
255 assert(compressed_blob);
256 assert(compressed_blob->empty());
257 assert(immutable_options_);
258
259 if (blob_compression_type_ == kNoCompression) {
260 return Status::OK();
261 }
262
263 // TODO: allow user CompressionOptions, including max_compressed_bytes_per_kb
264 CompressionOptions opts;
265 CompressionContext context(blob_compression_type_, opts);
266 constexpr uint64_t sample_for_compression = 0;
267
268 CompressionInfo info(opts, context, CompressionDict::GetEmptyDict(),
269 blob_compression_type_, sample_for_compression);
270
271 constexpr uint32_t compression_format_version = 2;
272
273 bool success = false;
274
275 {
276 StopWatch stop_watch(immutable_options_->clock, immutable_options_->stats,
277 BLOB_DB_COMPRESSION_MICROS);
278 success =
279 CompressData(*blob, info, compression_format_version, compressed_blob);
280 }
281
282 if (!success) {
283 return Status::Corruption("Error compressing blob");
284 }
285
286 *blob = Slice(*compressed_blob);
287
288 return Status::OK();
289}

291Status BlobFileBuilder::WriteBlobToFile(const Slice& key, const Slice& blob,
292 uint64_t* blob_file_number,
293 uint64_t* blob_offset) {
294 assert(IsBlobFileOpen());
295 assert(blob_file_number);
296 assert(blob_offset);
297
298 uint64_t key_offset = 0;
299
300 Status s =
301 writer_->AddRecord(*write_options_, key, blob, &key_offset, blob_offset);
302
303 TEST_SYNC_POINT_CALLBACK("BlobFileBuilder::WriteBlobToFile:AddRecord", &s);
304
305 if (!s.ok()) {
306 return s;
307 }
308
309 *blob_file_number = writer_->get_log_number();
310
311 ++blob_count_;
312 blob_bytes_ += BlobLogRecord::kHeaderSize + key.size() + blob.size();
313
314 return Status::OK();
315}

317Status BlobFileBuilder::CloseBlobFile() {
318 assert(IsBlobFileOpen());
319
320 BlobLogFooter footer;
321 footer.blob_count = blob_count_;
322
323 std::string checksum_method;
324 std::string checksum_value;
325
326 Status s = writer_->AppendFooter(*write_options_, footer, &checksum_method,
327 &checksum_value);
328
329 TEST_SYNC_POINT_CALLBACK("BlobFileBuilder::WriteBlobToFile:AppendFooter", &s);
330
331 if (!s.ok()) {
332 return s;
333 }
334
335 const uint64_t blob_file_number = writer_->get_log_number();
336
337 if (blob_callback_) {
338 s = blob_callback_->OnBlobFileCompleted(
339 blob_file_paths_->back(), column_family_name_, job_id_,
340 blob_file_number, creation_reason_, s, checksum_value, checksum_method,
341 blob_count_, blob_bytes_);
342 }
343
344 assert(blob_file_additions_);
345 blob_file_additions_->emplace_back(blob_file_number, blob_count_, blob_bytes_,
346 std::move(checksum_method),
347 std::move(checksum_value));
348
349 assert(immutable_options_);
350 ROCKS_LOG_INFO(immutable_options_->logger,
351 "[%s] [JOB %d] Generated blob file #%" PRIu64 ": %" PRIu64
352 " total blobs, %" PRIu64 " total bytes",
353 column_family_name_.c_str(), job_id_, blob_file_number,
354 blob_count_, blob_bytes_);
355
356 writer_.reset();
357 blob_count_ = 0;
358 blob_bytes_ = 0;
359
360 return s;
361}

363Status BlobFileBuilder::CloseBlobFileIfNeeded() {
364 assert(IsBlobFileOpen());
365
366 const WritableFileWriter* const file_writer = writer_->file();
367 assert(file_writer);
368
369 if (file_writer->GetFileSize() < blob_file_size_) {
370 return Status::OK();
371 }
372
373 return CloseBlobFile();
374}

376void BlobFileBuilder::Abandon(const Status& s) {
377 if (!IsBlobFileOpen()) {
378 return;
379 }
380 if (blob_callback_) {
381 // BlobFileBuilder::Abandon() is called because of error while writing to
382 // Blob files. So we can ignore the below error.
383 blob_callback_
384 ->OnBlobFileCompleted(blob_file_paths_->back(), column_family_name_,
385 job_id_, writer_->get_log_number(),
386 creation_reason_, s, "", "", blob_count_,
387 blob_bytes_)
388 .PermitUncheckedError();
389 }
390
391 writer_.reset();
392 blob_count_ = 0;
393 blob_bytes_ = 0;
394}

396Status BlobFileBuilder::PutBlobIntoCacheIfNeeded(const Slice& blob,
397 uint64_t blob_file_number,
398 uint64_t blob_offset) const {
399 Status s = Status::OK();
400
401 BlobSource::SharedCacheInterface blob_cache{immutable_options_->blob_cache};
402 auto statistics = immutable_options_->statistics.get();
403 bool warm_cache =
404 prepopulate_blob_cache_ == PrepopulateBlobCache::kFlushOnly &&
405 creation_reason_ == BlobFileCreationReason::kFlush;
406
407 if (blob_cache && warm_cache) {
408 const OffsetableCacheKey base_cache_key(db_id_, db_session_id_,
409 blob_file_number);
410 const CacheKey cache_key = base_cache_key.WithOffset(blob_offset);
411 const Slice key = cache_key.AsSlice();
412
413 const Cache::Priority priority = Cache::Priority::BOTTOM;
414
415 s = blob_cache.InsertSaved(key, blob, nullptr /*context*/, priority,
416 immutable_options_->lowest_used_cache_tier);
417
418 if (s.ok()) {
419 RecordTick(statistics, BLOB_DB_CACHE_ADD);
420 RecordTick(statistics, BLOB_DB_CACHE_BYTES_WRITE, blob.size());
421 } else {
422 RecordTick(statistics, BLOB_DB_CACHE_ADD_FAILURES);
423 }
424 }
425
426 return s;
427}
