369 // Close the DB by releasing resources, closing files etc. This should be
370 // called before calling the destructor so that the caller can get back a
371 // status in case there are any errors. This will not fsync the WAL files.
372 // If syncing is required, the caller must first call SyncWAL(), or Write()
373 // using an empty write batch with WriteOptions.sync=true.
374 // Regardless of the return status, the DB must be freed.
375 //
376 // If the return status is Aborted(), closing fails because there is
377 // unreleased snapshot in the system. In this case, users can release
378 // the unreleased snapshots and try again and expect it to succeed. For
379 // other status, re-calling Close() will be no-op and return the original
380 // close status. If the return status is NotSupported(), then the DB
381 // implementation does cleanup in the destructor
382 //
383 // WaitForCompact() with WaitForCompactOptions.close_db=true will be a good
384 // choice for users who want to wait for background work before closing
385 // (rather than aborting and potentially redoing some work on re-open)

721 // Consistent Get of many keys across column families without the need
722 // for an explicit snapshot. The main difference between this set of
723 // MultiGet APis and the batched MultiGet APIs that follow are -
724 // 1. The APIs take std::vector instead of C style array pointers
725 // 2. Values are returned as std::string rather than PinnableSlice
726 //
727 // If keys[i] does not exist in the database, then the i'th returned
728 // status will be one for which Status::IsNotFound() is true, and
729 // (*values)[i] will be set to some arbitrary value (often ""). Otherwise,
730 // the i'th returned status will have Status::ok() true, and (*values)[i]
731 // will store the value associated with keys[i].
732 //
733 // (*values) will always be resized to be the same size as (keys).
734 // Similarly, the number of returned statuses will be the number of keys.
735 // If timestamps is non-null, the vector pointed to by it will be resized to
736 // number of keys and filled with timestamps of the keys on return.
737 // Note: keys will not be "de-duplicated". Duplicate keys will return
738 // duplicate values in order, and may return different status values
739 // in case there are errors.
740 // NOTE: virtual final => disallow override (was previously allowed)
741 virtual std::vector<Status> MultiGet(
742 const ReadOptions& options,
743 const std::vector<ColumnFamilyHandle*>& column_families,
744 const std::vector<Slice>& keys, std::vector<std::string>* values,
745 std::vector<std::string>* timestamps) final {
746 size_t num_keys = keys.size();
747 std::vector<Status> statuses(num_keys);
748 std::vector<PinnableSlice> pin_values(num_keys);
749
750 values->resize(num_keys);
751 if (timestamps) {
752 timestamps->resize(num_keys);
753 }
754 MultiGet(options, num_keys,
755 const_cast<ColumnFamilyHandle**>(column_families.data()),
756 keys.data(), pin_values.data(),
757 timestamps ? timestamps->data() : nullptr, statuses.data(),
758 /*sorted_input=*/false);
759 for (size_t i = 0; i < num_keys; ++i) {
760 if (statuses[i].ok()) {
761 (*values)[i].assign(pin_values[i].data(), pin_values[i].size());
762 }
763 }
764 return statuses;
765 }

1468 // Compact the underlying storage for the key range [*begin,*end].
