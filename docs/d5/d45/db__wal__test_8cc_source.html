68// A SpecialEnv enriched to give more insight about deleted files
69class EnrichedSpecialEnv : public SpecialEnv {
70 public:
71 explicit EnrichedSpecialEnv(Env* base) : SpecialEnv(base) {}
72 Status NewSequentialFile(const std::string& f,
73 std::unique_ptr<SequentialFile>* r,
74 const EnvOptions& soptions) override {
75 InstrumentedMutexLock l(&env_mutex_);
76 if (f == skipped_wal) {
77 deleted_wal_reopened = true;
78 if (IsWAL(f) && largest_deleted_wal.size() != 0 &&
79 f.compare(largest_deleted_wal) <= 0) {
80 gap_in_wals = true;
81 }
82 }
83 return SpecialEnv::NewSequentialFile(f, r, soptions);
84 }
85 Status DeleteFile(const std::string& fname) override {
86 if (IsWAL(fname)) {
87 deleted_wal_cnt++;
88 InstrumentedMutexLock l(&env_mutex_);
89 // If this is the first WAL, remember its name and skip deleting it. We
90 // remember its name partly because the application might attempt to
91 // delete the file again.
92 if (skipped_wal.size() != 0 && skipped_wal != fname) {
93 if (largest_deleted_wal.size() == 0 ||
94 largest_deleted_wal.compare(fname) < 0) {
95 largest_deleted_wal = fname;
96 }
97 } else {
98 skipped_wal = fname;
99 return Status::OK();
100 }
101 }
102 return SpecialEnv::DeleteFile(fname);
103 }
104 bool IsWAL(const std::string& fname) {
105 // printf("iswal %s\n", fname.c_str());
106 return fname.compare(fname.size() - 3, 3, "log") == 0;
107 }
108
109 InstrumentedMutex env_mutex_;
110 // the wal whose actual delete was skipped by the env
111 std::string skipped_wal;
112 // the largest WAL that was requested to be deleted
113 std::string largest_deleted_wal;
114 // number of WALs that were successfully deleted
115 std::atomic<size_t> deleted_wal_cnt = {0};
116 // the WAL whose delete from fs was skipped is reopened during recovery
117 std::atomic<bool> deleted_wal_reopened = {false};
118 // whether a gap in the WALs was detected during recovery
119 std::atomic<bool> gap_in_wals = {false};
120};

141// One known scenario that could cause this is that the application issue the

374 // Set up the option that enables user defined timestmp size.
375 std::string ts1;
376 PutFixed64(&ts1, 1);
377 Options ts_options;
378 ts_options.create_if_missing = true;
379 ts_options.comparator = test::BytewiseComparatorWithU64TsWrapper();

381 // the value of this flag because UDTs are saved in WAL nonetheless.

383 // `avoid_flush_during_recovery=true` so that we can avoid timestamps getting
384 // stripped when the `persist_user_defined_timestamps` flag is false, so that
385 // all written timestamps are available for testing user-defined time travel
386 // read.
387 bool persist_udt = test::ShouldPersistUDT(GetParam());

389 bool avoid_flush_during_recovery = true;

391 std::string full_history_ts_low;
392 ReadOptions read_opts;
393 do {
394 Slice ts_slice = ts1;
395 read_opts.timestamp = &ts_slice;
396 ASSERT_OK(CreateAndReopenWithTs({"pikachu"}, ts_options, persist_udt,
397 avoid_flush_during_recovery));
398 ASSERT_EQ(GetNumberOfSstFilesForColumnFamily(db_, "pikachu"), 0U);
399 ASSERT_OK(Put(1, "foo", ts1, "v1"));
400 ASSERT_OK(Put(1, "baz", ts1, "v5"));
401
402 ASSERT_OK(ReopenColumnFamiliesWithTs({"pikachu"}, ts_options, persist_udt,
403 avoid_flush_during_recovery));
404 ASSERT_EQ(GetNumberOfSstFilesForColumnFamily(db_, "pikachu"), 0U);
405 // Do a timestamped read with ts1 after second reopen.
406 CheckGet(read_opts, 1, "foo", "v1", ts1);
407 CheckGet(read_opts, 1, "baz", "v5", ts1);
408
409 // Write more value versions for key "foo" and "bar" before and after second
410 // reopen.
411 std::string ts2;
412 PutFixed64(&ts2, 2);
413 ASSERT_OK(Put(1, "bar", ts2, "v2"));
414 ASSERT_OK(Put(1, "foo", ts2, "v3"));
415
416 ASSERT_OK(ReopenColumnFamiliesWithTs({"pikachu"}, ts_options, persist_udt,
417 avoid_flush_during_recovery));
418 ASSERT_EQ(GetNumberOfSstFilesForColumnFamily(db_, "pikachu"), 0U);

420 PutFixed64(&ts3, 3);
421 ASSERT_OK(Put(1, "foo", ts3, "v4"));
422
423 // All the key value pairs available for read:
424 // "foo" -> [(ts1, "v1"), (ts2, "v3"), (ts3, "v4")]

426 // "baz" -> [(ts1, "v5")]

433
434 // Do a timestamped read with ts2 after third reopen.

448 ASSERT_TRUE(full_history_ts_low.empty());
449 } while (ChangeWalOptions());
