17Status BlobDBImpl::DisableFileDeletions() {
18 // Disable base DB file deletions.
19 Status s = db_impl_->DisableFileDeletions();
20 if (!s.ok()) {
21 return s;
22 }
23
24 int count = 0;
25 {
26 // Hold delete_file_mutex_ to make sure no DeleteObsoleteFiles job
27 // is running.
28 MutexLock l(&delete_file_mutex_);
29 count = ++disable_file_deletions_;
30 }
31
32 ROCKS_LOG_INFO(db_options_.info_log,
33 "Disabled blob file deletions. count: %d", count);
34 return Status::OK();
35}

37Status BlobDBImpl::EnableFileDeletions() {
38 // Enable base DB file deletions.
39 Status s = db_impl_->EnableFileDeletions();
40 if (!s.ok()) {
41 return s;
42 }
43
44 int count = 0;
45 {
46 MutexLock l(&delete_file_mutex_);
47 if (disable_file_deletions_ > 0) {
48 count = --disable_file_deletions_;
49 }
50 assert(count >= 0);
51 }
52
53 ROCKS_LOG_INFO(db_options_.info_log, "Enabled blob file deletions. count: %d",
54 count);
55 // Consider trigger DeleteobsoleteFiles once after re-enabled, if we are to
56 // make DeleteobsoleteFiles re-run interval configuration.
57 return Status::OK();
58}

60Status BlobDBImpl::GetLiveFiles(std::vector<std::string>& ret,
61 uint64_t* manifest_file_size,
62 bool flush_memtable) {
63 if (!bdb_options_.path_relative) {
64 return Status::NotSupported(
65 "Not able to get relative blob file path from absolute blob_dir.");
66 }
67 // Hold a lock in the beginning to avoid updates to base DB during the call
68 ReadLock rl(&mutex_);
69 Status s = db_->GetLiveFiles(ret, manifest_file_size, flush_memtable);
70 if (!s.ok()) {
71 return s;
72 }
73 ret.reserve(ret.size() + blob_files_.size());
74 for (const auto& bfile_pair : blob_files_) {
75 auto blob_file = bfile_pair.second;
76 // Path should be relative to db_name, but begin with slash.
77 ret.emplace_back(
78 BlobFileName("", bdb_options_.blob_dir, blob_file->BlobFileNumber()));
79 }
80 return Status::OK();
81}

83void BlobDBImpl::GetLiveFilesMetaData(std::vector<LiveFileMetaData>* metadata) {
84 // Path should be relative to db_name.
85 assert(bdb_options_.path_relative);
86 // Hold a lock in the beginning to avoid updates to base DB during the call
87 ReadLock rl(&mutex_);
88 db_->GetLiveFilesMetaData(metadata);
89 for (const auto& bfile_pair : blob_files_) {
90 auto blob_file = bfile_pair.second;
91 LiveFileMetaData filemetadata;
92 filemetadata.size = blob_file->GetFileSize();
93 const uint64_t file_number = blob_file->BlobFileNumber();
94 // Path should be relative to db_name, but begin with slash.
95 filemetadata.name = BlobFileName("", bdb_options_.blob_dir, file_number);
96 filemetadata.file_number = file_number;
97 if (blob_file->HasTTL()) {
98 filemetadata.oldest_ancester_time = blob_file->GetExpirationRange().first;
99 }
100 auto cfh =
101 static_cast_with_check<ColumnFamilyHandleImpl>(DefaultColumnFamily());
102 filemetadata.column_family_name = cfh->GetName();
103 metadata->emplace_back(filemetadata);
104 }
105}

107Status BlobDBImpl::GetLiveFilesStorageInfo(
108 const LiveFilesStorageInfoOptions& opts,
109 std::vector<LiveFileStorageInfo>* files) {
110 ReadLock rl(&mutex_);
111 Status s = db_->GetLiveFilesStorageInfo(opts, files);
112 if (s.ok()) {
113 files->reserve(files->size() + blob_files_.size());
114 for (const auto& [blob_number, blob_file] : blob_files_) {
115 LiveFileStorageInfo file;
116 file.size = blob_file->GetFileSize();
117 file.directory = blob_dir_;
118 file.relative_filename = BlobFileName(blob_number);
119 file.file_type = kBlobFile;
120 file.trim_to_size = true;
121 files->push_back(std::move(file));
122 }
123 }
124 return s;
125}
