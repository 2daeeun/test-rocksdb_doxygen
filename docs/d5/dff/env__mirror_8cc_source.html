15// An implementation of Env that mirrors all work over two backend
16// Env's. This is useful for debugging purposes.
17class SequentialFileMirror : public SequentialFile {
18 public:
19 std::unique_ptr<SequentialFile> a_, b_;
20 std::string fname;
21 explicit SequentialFileMirror(std::string f) : fname(f) {}
22
23 Status Read(size_t n, Slice* result, char* scratch) override {
24 Slice aslice;
25 Status as = a_->Read(n, &aslice, scratch);
26 if (as == Status::OK()) {
27 char* bscratch = new char[n];
28 Slice bslice;
29#ifndef NDEBUG
30 size_t off = 0;
31#endif
32 size_t left = aslice.size();
33 while (left) {
34 Status bs = b_->Read(left, &bslice, bscratch);
35#ifndef NDEBUG
36 assert(as == bs);
37 assert(memcmp(bscratch, scratch + off, bslice.size()) == 0);
38 off += bslice.size();
39#endif
40 left -= bslice.size();
41 }
42 delete[] bscratch;
43 *result = aslice;
44 } else {
45 Status bs = b_->Read(n, result, scratch);
46 assert(as == bs);
47 }
48 return as;
49 }
50
51 Status Skip(uint64_t n) override {
52 Status as = a_->Skip(n);
53 Status bs = b_->Skip(n);
54 assert(as == bs);
55 return as;
56 }
57 Status InvalidateCache(size_t offset, size_t length) override {
58 Status as = a_->InvalidateCache(offset, length);
59 Status bs = b_->InvalidateCache(offset, length);
60 assert(as == bs);
61 return as;
62 }
63};

65class RandomAccessFileMirror : public RandomAccessFile {
66 public:
67 std::unique_ptr<RandomAccessFile> a_, b_;
68 std::string fname;
69 explicit RandomAccessFileMirror(std::string f) : fname(f) {}
70
71 Status Read(uint64_t offset, size_t n, Slice* result,
72 char* scratch) const override {
73 Status as = a_->Read(offset, n, result, scratch);
74 if (as == Status::OK()) {
75 char* bscratch = new char[n];
76 Slice bslice;
77 size_t off = 0;
78 size_t left = result->size();
79 while (left) {
80 Status bs = b_->Read(offset + off, left, &bslice, bscratch);
81 assert(as == bs);
82 assert(memcmp(bscratch, scratch + off, bslice.size()) == 0);
83 off += bslice.size();
84 left -= bslice.size();
85 }
86 delete[] bscratch;
87 } else {
88 Status bs = b_->Read(offset, n, result, scratch);
89 assert(as == bs);
90 }
91 return as;
92 }
93
94 size_t GetUniqueId(char* id, size_t max_size) const override {
95 // NOTE: not verified
96 return a_->GetUniqueId(id, max_size);
97 }
98};

207Status EnvMirror::NewSequentialFile(const std::string& f,
208 std::unique_ptr<SequentialFile>* r,
209 const EnvOptions& options) {
210 if (f.find("/proc/") == 0) {
211 return a_->NewSequentialFile(f, r, options);
212 }
213 SequentialFileMirror* mf = new SequentialFileMirror(f);
214 Status as = a_->NewSequentialFile(f, &mf->a_, options);
215 Status bs = b_->NewSequentialFile(f, &mf->b_, options);
216 assert(as == bs);
217 if (as.ok()) {
218 r->reset(mf);
219 } else {
220 delete mf;
221 }
222 return as;
223}
