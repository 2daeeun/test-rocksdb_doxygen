29int main() {
30 // open DB
31 Options options;
32 options.create_if_missing = true;
33 DB* db;
34 OptimisticTransactionDB* txn_db;
35
36 Status s = OptimisticTransactionDB::Open(options, kDBPath, &txn_db);
37 assert(s.ok());
38 db = txn_db->GetBaseDB();
39
40 WriteOptions write_options;
41 ReadOptions read_options;
42 OptimisticTransactionOptions txn_options;
43 std::string value;
44
46 //
47 // Simple OptimisticTransaction Example ("Read Committed")
48 //
50
51 // Start a transaction
52 Transaction* txn = txn_db->BeginTransaction(write_options);
53 assert(txn);
54
55 // Read a key in this transaction
56 s = txn->Get(read_options, "abc", &value);
57 assert(s.IsNotFound());
58
59 // Write a key in this transaction
60 s = txn->Put("abc", "xyz");
61 assert(s.ok());
62
63 // Read a key OUTSIDE this transaction. Does not affect txn.
64 s = db->Get(read_options, "abc", &value);
65 assert(s.IsNotFound());
66
67 // Write a key OUTSIDE of this transaction.
68 // Does not affect txn since this is an unrelated key. If we wrote key 'abc'
69 // here, the transaction would fail to commit.
70 s = db->Put(write_options, "xyz", "zzz");
71 assert(s.ok());
72 s = db->Put(write_options, "abc", "def");
73 assert(s.ok());
74
75 // Commit transaction
76 s = txn->Commit();
77 assert(s.IsBusy());
78 delete txn;
79
80 s = db->Get(read_options, "xyz", &value);
81 assert(s.ok());
82 assert(value == "zzz");
83
84 s = db->Get(read_options, "abc", &value);
85 assert(s.ok());
86 assert(value == "def");
87
89 //
90 // "Repeatable Read" (Snapshot Isolation) Example
91 // -- Using a single Snapshot
92 //
94
95 // Set a snapshot at start of transaction by setting set_snapshot=true
96 txn_options.set_snapshot = true;
97 txn = txn_db->BeginTransaction(write_options, txn_options);
98
99 const Snapshot* snapshot = txn->GetSnapshot();
100
101 // Write a key OUTSIDE of transaction
102 s = db->Put(write_options, "abc", "xyz");
103 assert(s.ok());
104
105 // Read a key using the snapshot
106 read_options.snapshot = snapshot;
107 s = txn->GetForUpdate(read_options, "abc", &value);
108 assert(s.ok());
109 assert(value == "def");
110
111 // Attempt to commit transaction
112 s = txn->Commit();
113
114 // Transaction could not commit since the write outside of the txn conflicted
115 // with the read!
116 assert(s.IsBusy());
117
118 delete txn;
119 // Clear snapshot from read options since it is no longer valid
120 read_options.snapshot = nullptr;
121 snapshot = nullptr;
122
123 s = db->Get(read_options, "abc", &value);
124 assert(s.ok());
125 assert(value == "xyz");
126
128 //
129 // "Read Committed" (Monotonic Atomic Views) Example
130 // --Using multiple Snapshots
131 //
133
134 // In this example, we set the snapshot multiple times. This is probably
135 // only necessary if you have very strict isolation requirements to
136 // implement.
137
138 // Set a snapshot at start of transaction
139 txn_options.set_snapshot = true;
140 txn = txn_db->BeginTransaction(write_options, txn_options);
141
142 // Do some reads and writes to key "x"
143 read_options.snapshot = db->GetSnapshot();
144 s = txn->Get(read_options, "x", &value);
145 assert(s.IsNotFound());
146 s = txn->Put("x", "x");
147 assert(s.ok());
148
149 // The transaction hasn't committed, so the write is not visible
150 // outside of txn.
151 s = db->Get(read_options, "x", &value);
152 assert(s.IsNotFound());
153
154 // Do a write outside of the transaction to key "y"
155 s = db->Put(write_options, "y", "z");
156 assert(s.ok());
157
158 // Set a new snapshot in the transaction
159 txn->SetSnapshot();
160 read_options.snapshot = db->GetSnapshot();
161
162 // Do some reads and writes to key "y"
163 s = txn->GetForUpdate(read_options, "y", &value);
164 assert(s.ok());
165 assert(value == "z");
166 txn->Put("y", "y");
167
168 // Commit. Since the snapshot was advanced, the write done outside of the
169 // transaction does not prevent this transaction from Committing.
170 s = txn->Commit();
171 assert(s.ok());
172 delete txn;
173 // Clear snapshot from read options since it is no longer valid
174 read_options.snapshot = nullptr;
175
176 // txn is committed, read the latest values.
177 s = db->Get(read_options, "x", &value);
178 assert(s.ok());
179 assert(value == "x");
180
181 s = db->Get(read_options, "y", &value);
182 assert(s.ok());
183 assert(value == "y");
184
185 // Cleanup
186 delete txn_db;
187 DestroyDB(kDBPath, options);
188 return 0;
189}
